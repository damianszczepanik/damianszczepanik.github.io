21  import java.util.ArrayList;
22  import java.util.HashSet;
23  import java.util.List;
24  import java.util.Set;

26  import com.github.sevntu.checkstyle.Utils;
27  import com.puppycrawl.tools.checkstyle.api.Check;
28  import com.puppycrawl.tools.checkstyle.api.DetailAST;
29  import com.puppycrawl.tools.checkstyle.api.TokenTypes;

31  /**
32   * <p>
33   * This Check warns on propagation of inner private types to outer classes:<br>
34   * - Externally accessible method if it returns private inner type.<br>
35   * - Externally accessible field if it's type is a private inner type.<br>
36   * These types could be <a href=
37   * 'http://docs.oracle.com/javase/tutorial/java/javaOO/nested.html'>private
38   * inner classes</a>, interfaces or enumerations.<br>
39   * <br>
40   * Examples: <code>
41   * <pre> 
42   * class OuterClass {
43   *  public InnerClass innerFromMain = new InnerClass(); //WARNING
44   *  private class InnerClass { ... } 
45   *  public InnerClass  getValue() { //WARNING
46   *      return new InnerClass(); 
47   *  }
48   *  <br> 
49   *  private interface InnerInterface { ... }
50   *  public Set&ltInnerInterface&gt getValue() { //WARNING
51   *      return new TreeSet&ltInnerInterface&gt;;
52   *  }
53   *  <br>
54   *  private Enum Fruit {Apple, Pear}
55   *  public Fruit getValue() { //WARNING
56   *      return Fruit.Apple;
57   *  }
58   *  <br>
59   *  public someMethod(InnerClass innerClass) { ... }  //WARNING
60   *  <br>
61   * }
62   * </pre>
63   * </code>
64   * <b>Rationale:</b> it is possible to return<br>
65   * private inner type or use it as the parameter of non-private method, but it
66   * is impossible<br>
67   * to use it in other classes (besides inner classes)<br>
68   * unless it extends or implements at least one <u>non-private</u> class or
69   * interface.<br>
70   * Such situation usually happens after bulk refactoring and usually means
71   * dead/useless code<br>
72   * </p>
73   * <br>
74   * @author <a href="mailto:nesterenko-aleksey@list.ru">Aleksey Nesterenko</a>
75   */
76  public class PublicReferenceToPrivateTypeCheck extends Check
77  {
78      /**
79       * Check message key for private types
80       */
81      public final static String MSG_KEY = "public.reference.to.private.type";
82  
83      /**
84       * List containing names of private types (classes, interfaces or enums).
85       */
86      private final Set<DetailAST> privateTypes = new HashSet<DetailAST>();
87  
88      /**
89       * List containing the names of types returned by public methods or fields.
90       */
91      private final Set<DetailAST> externallyReferencedTypes = new HashSet<DetailAST>();
92  
93      @Override
94      public int[] getDefaultTokens()
95      {
96          return new int[] { TokenTypes.CLASS_DEF, TokenTypes.METHOD_DEF,
97                  TokenTypes.INTERFACE_DEF, TokenTypes.ENUM_DEF,
98                  TokenTypes.VARIABLE_DEF };
99      }
100 
101     @Override
102     public void beginTree(DetailAST rootAST)
103     {
104         privateTypes.clear();
105         externallyReferencedTypes.clear();
106     }
107 
108     @Override
109     public void visitToken(DetailAST defAst)
110     {
111         switch (defAst.getType()) {
112             case TokenTypes.CLASS_DEF:
113             case TokenTypes.INTERFACE_DEF:
114             case TokenTypes.ENUM_DEF:
115                 if (hasModifier(TokenTypes.LITERAL_PRIVATE, defAst)) {
116                     addPrivateTypes(defAst);
117                 }
118                 break;
119             case TokenTypes.METHOD_DEF:
120                 if (isDefinedInTopLevelClass(defAst)
121                     && !hasModifier(TokenTypes.LITERAL_PRIVATE, defAst)) {
122                     addExternallyAccessibleMethodTypes(defAst);
123                 }
124                 break;
125             case TokenTypes.VARIABLE_DEF:
126                 if (isDefinedInTopLevelClass(defAst)
127                     && !hasModifier(TokenTypes.LITERAL_PRIVATE, defAst)) {
128                     addExternallyAccessibleFieldTypes(defAst);
129                 }
130                 break;
131             default:
132                 Utils.reportInvalidToken(defAst.getType());
133                 break;
134         }
135     }
136 
137     @Override
138     public void finishTree(DetailAST rootAst)
139     {
140         for (DetailAST privateType : privateTypes) {
141             for (DetailAST outReturnedType : externallyReferencedTypes) {
142                 if (privateType.getText().equals(
143                         outReturnedType.getText())
144                         && !isExtendsOrImplementsSmth(privateType
145                                 .getParent())) {
146                     log(outReturnedType.getLineNo(), MSG_KEY,
147                             outReturnedType.getText());
148                 }
149             }
150         }
151     }
152 
153     /**
154      * Adds type to the list of private types.
155      * @param classOrInterfaceOrEnumDefAst
156      *        AST subtree that represent inner private type definition.
157      */
158     private void addPrivateTypes(DetailAST classOrInterfaceOrEnumDefAst)
159     {
160         final DetailAST definitionAst = classOrInterfaceOrEnumDefAst
161                 .findFirstToken(TokenTypes.IDENT);
162         privateTypes.add(definitionAst);
163     }
164 
165     /**
166      * Appends non-private, defined in top-level class method's returned or
167      * parameter type, or field's type to general list of out referenced types.
168      * @param methodDefAst
169      *        AST subtree that represent method definition.
170      */
171     private void addExternallyAccessibleMethodTypes(DetailAST methodDefAst)
172     {
173         DetailAST typeDefAst = methodDefAst
174                 .findFirstToken(TokenTypes.TYPE);
175         DetailAST parametersDefAst = methodDefAst
176                 .findFirstToken(TokenTypes.PARAMETERS);
177         externallyReferencedTypes.addAll(getMethodOrFieldReferencedTypes(typeDefAst));
178         externallyReferencedTypes.addAll(getMethodParameterTypes(parametersDefAst));
179     }
180 
181     /**
182      * Appends non-private, defined in top-level class field's type to general
183      * list of out referenced types.
184      * @param fieldDefAst
185      *        AST subtree that represent field definition.
186      */
187     private void addExternallyAccessibleFieldTypes(DetailAST fieldDefAst)
188     {
189         DetailAST typeDefAst = fieldDefAst.findFirstToken(TokenTypes.TYPE);
190         externallyReferencedTypes.addAll(getMethodOrFieldReferencedTypes(typeDefAst));
191     }
192 
193     /**
194      * Gets the return type of method or field type.
195      * @param typeAst
196      *        AST subtree to process.
197      */
198     private static List<DetailAST>
199             getMethodOrFieldReferencedTypes(DetailAST typeAst)
200     {
201 
202         DetailAST returnedType = null;
203         List<DetailAST> returnedTypes = new ArrayList<DetailAST>();
204         DetailAST currentNode = typeAst;
205         while (currentNode != null) {
206             if (currentNode.getType() == TokenTypes.IDENT) {
207                 returnedType = currentNode;
208                 returnedTypes.add(returnedType);
209             }
210             currentNode = Utils.getNextSubTreeNode(currentNode, typeAst);
211         }
212         return returnedTypes;
213     }
214 
215     /**
216      * Gets method's parameters types
217      * @param parametersDefAst
218      */
219     private static List<DetailAST>
220             getMethodParameterTypes(DetailAST parametersDefAst)
221     {
222         DetailAST parameterType = null;
223         List<DetailAST> parameterTypes = new ArrayList<DetailAST>();
224 
225         if (parametersDefAst.getFirstChild() != null) {
226             DetailAST currentNode = parametersDefAst;
227 
228             while (currentNode != null) {
229                 if (currentNode.getType() == TokenTypes.PARAMETER_DEF) {
230                     parameterType = currentNode;
231 
232                     while (parameterType != null) {
233                         parameterType = Utils.getNextSubTreeNode(parameterType,
234                                 currentNode);
235                         if (parameterType != null
236                                 && parameterType.getType() == TokenTypes.IDENT) {
237                             parameterTypes.add(parameterType);
238                         }
239                     }
240 
241                 }
242                 currentNode = Utils.getNextSubTreeNode(currentNode, parametersDefAst);
243             }
244 
245         }
246 
247         return parameterTypes;
248     }
249 
250     /**
251      * Checks if defined type or interface extends or implements any
252      * <u>non-private type</u>.
253      * @param classOrInterfaceDefAst
254      * @return Method returns true if class extends or implements something.
255      */
256     private boolean isExtendsOrImplementsSmth(DetailAST classOrInterfaceDefAst)
257     {
258         return (classOrInterfaceDefAst
259                 .branchContains(TokenTypes.EXTENDS_CLAUSE)
260                 || classOrInterfaceDefAst
261                 .branchContains(TokenTypes.IMPLEMENTS_CLAUSE))
262                 && !isExtendsOrImplementsPrivate(classOrInterfaceDefAst);
263     }
264 
265     /**
266      * Checks if inner class or interface extends or implements <u>inner private
267      * type</u>
268      * @param classOrInterfaceDefAst
269      * @return true if extending or implementing type is in collection of inner
270      *         private types
271      */
272     private boolean
273             isExtendsOrImplementsPrivate(DetailAST classOrInterfaceDefAst)
274     {
275         boolean result = false;
276 
277         Set<String> inheritedTypesNamesSet = new HashSet<String>();
278         DetailAST currentNode = classOrInterfaceDefAst;
279 
280         while (currentNode != null) {
281             if (currentNode.getType() == TokenTypes.EXTENDS_CLAUSE
282                     || currentNode.getType() == TokenTypes.IMPLEMENTS_CLAUSE) {
283                 DetailAST implementingOrExtendingAst = currentNode;
284 
285                 while (implementingOrExtendingAst != null) {
286                     implementingOrExtendingAst = Utils.getNextSubTreeNode(
287                             implementingOrExtendingAst, currentNode);
288                     if (implementingOrExtendingAst != null
289                             && implementingOrExtendingAst.getType() == TokenTypes.IDENT) {
290                         inheritedTypesNamesSet
291                                 .add(implementingOrExtendingAst.getText());
292                     }
293                 }
294 
295             }
296             currentNode = Utils.getNextSubTreeNode(currentNode, classOrInterfaceDefAst);
297         }
298 
299         Set<String> existingPrivateTypes = new HashSet<String>();
300         for (DetailAST privateType : privateTypes) {
301             existingPrivateTypes.add(privateType.getText());
302         }
303         if (existingPrivateTypes.containsAll(inheritedTypesNamesSet)) {
304             result = true;
305         }
306 
307         return result;
308     }
309 
310     /**
311      * Checks, class, interface, enumeration, method or field has "private"
312      * access modifier.
313      * @param modifierType
314      * @param defAst
315      */
316     public static boolean
317             hasModifier(int modifierType, DetailAST defAst)
318     {
319         DetailAST modifiersToken = defAst.getFirstChild();
320         return modifiersToken.findFirstToken(modifierType) != null;
321     }
322 
323     /**
324      * Checks if method or field is defined in top-level class
325      * @param methodOrFieldDefAst
326      * @return true if method is defined in top-level class
327      */
328     private static boolean
329             isDefinedInTopLevelClass(DetailAST methodOrFieldDefAst)
330     {
331         return methodOrFieldDefAst.getParent().getParent().getParent() == null;
332     }
