19  package com.github.sevntu.checkstyle.checks.design;
20  
21  import com.puppycrawl.tools.checkstyle.api.Check;
22  import com.puppycrawl.tools.checkstyle.api.TokenTypes;
23  import com.puppycrawl.tools.checkstyle.api.DetailAST;
24  
25  /**
26   * Make sure that utility classes (classes that contain only static methods)
27   * do not have a public constructor.
28   * <p>
29   * Rationale: Instantiating utility classes does not make sense.
30   * A common mistake is forgetting to hide the default constructor.
31   * </p>
32   *
33   * @author lkuehne
34   * @version $Revision: 1.12 $
35   */
36  public class HideUtilityClassConstructorCheck extends Check
37  {
38      /**
39       * Warning message key.
40       */
41      public final static String MSG_KEY = "hide.utility.class";
42  
43      @Override
44      public int[] getDefaultTokens()
45      {
46          return new int[] {TokenTypes.CLASS_DEF};
47      }
48  
49      @Override
50      public void visitToken(DetailAST ast)
51      {
52          if (isAbstract(ast)) {
53              // abstract class could not have private constructor
54              return;
55          }
56  
57          final DetailAST objBlock = ast.findFirstToken(TokenTypes.OBJBLOCK);
58          DetailAST child = objBlock.getFirstChild();
59          final boolean hasStaticModifier = isStatic(ast);
60          boolean hasMethodOrField = false;
61          boolean hasNonStaticMethodOrField = false;
62          boolean hasNonPrivateStaticMethodOrField = false;
63          boolean hasDefaultCtor = true;
64          boolean hasPublicCtor = false;
65  
66          while (child != null) {
67              final int type = child.getType();
68              if (type == TokenTypes.METHOD_DEF
69                      || type == TokenTypes.VARIABLE_DEF)
70              {
71                  hasMethodOrField = true;
72                  final DetailAST modifiers =
73                      child.findFirstToken(TokenTypes.MODIFIERS);
74                  final boolean isStatic =
75                      modifiers.branchContains(TokenTypes.LITERAL_STATIC);
76                  final boolean isPrivate =
77                      modifiers.branchContains(TokenTypes.LITERAL_PRIVATE);
78  
79                  if (!isStatic && !isPrivate) {
80                      hasNonStaticMethodOrField = true;
81                  }
82                  if (isStatic && !isPrivate) {
83                      hasNonPrivateStaticMethodOrField = true;
84                  }
85              }
86              if (type == TokenTypes.CTOR_DEF) {
87                  hasDefaultCtor = false;
88                  final DetailAST modifiers =
89                      child.findFirstToken(TokenTypes.MODIFIERS);
90                  if (!modifiers.branchContains(TokenTypes.LITERAL_PRIVATE)
91                      && !modifiers.branchContains(TokenTypes.LITERAL_PROTECTED))
92                  {
93                      // treat package visible as public
94                      // for the purpose of this Check
95                      hasPublicCtor = true;
96                  }
97  
98              }
99              child = child.getNextSibling();
100         }
101 
102         final boolean hasAccessibleCtor = hasDefaultCtor || hasPublicCtor;
103 
104         // figure out if class extends java.lang.object directly
105         // keep it simple for now and get a 99% solution
106         // TODO: check for "extends java.lang.Object" and "extends Object"
107         // consider "import org.omg.CORBA.*"
108         final boolean extendsJLO = // J.Lo even made it into in our sources :-)
109             ast.findFirstToken(TokenTypes.EXTENDS_CLAUSE) == null;
110 
111         final boolean isUtilClass = extendsJLO && hasMethodOrField
112             && !hasNonStaticMethodOrField && hasNonPrivateStaticMethodOrField;
113 
114         if (isUtilClass && (hasAccessibleCtor && !hasStaticModifier)) {
115             log(ast.getLineNo(), ast.getColumnNo(), "hide.utility.class");
116         }
117     }
118 
119     /**
120      * @param ast class definition for check.
121      * @return true if a given class declared as abstract.
122      */
123     private static boolean isAbstract(DetailAST ast)
124     {
125         final DetailAST abstractAST = ast.findFirstToken(TokenTypes.MODIFIERS)
126             .findFirstToken(TokenTypes.ABSTRACT);
127 
128         return abstractAST != null;
129     }
130 
131     /**
132      * @param ast class definition for check.
133      * @return true if a given class declared as static.
134      */
135     private static boolean isStatic(DetailAST ast)
136     {
137         final DetailAST staticAST = ast.findFirstToken(TokenTypes.MODIFIERS)
138             .findFirstToken(TokenTypes.LITERAL_STATIC);
139 
140         return staticAST != null;
141     }
142 }
