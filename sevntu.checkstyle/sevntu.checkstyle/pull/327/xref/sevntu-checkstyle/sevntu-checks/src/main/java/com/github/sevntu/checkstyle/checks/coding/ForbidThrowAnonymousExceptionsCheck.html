20  package com.github.sevntu.checkstyle.checks.coding;
21  
22  import java.util.ArrayList;
23  import java.util.List;
24  import java.util.regex.Pattern;
25  
26  import com.github.sevntu.checkstyle.Utils;
27  import com.puppycrawl.tools.checkstyle.api.Check;
28  import com.puppycrawl.tools.checkstyle.api.DetailAST;
29  import com.puppycrawl.tools.checkstyle.api.TokenTypes;
30  
31  /**
32   * <p>
33   * This Check warns on throwing anonymous exception.<br>
34   * Examples:
35   * <p>
36   * <code><pre>
37   * catch (Exception e) {
38   *        throw new RuntimeException()  { //WARNING 
39   *          //some code
40   *     };
41   * }
42   * <br>
43   * catch (Exception e) {
44   *     RuntimeException run = new RuntimeException()  {
45   *          //some code
46   *     };
47   *     throw run;  //WARNING
48   * }
49   * </code></pre> The distinguishing of <b>exception</b> types occurs by
50   * analyzing variable's class's name.<br>
51   * Check has an option which contains the regular expression for exception class name matching<br>
52   * Default value is "^.*Exception" because usually exception type ends with suffix "Exception".<br>
53   * Then, if we have an ObjBlock (distinguished by curly braces), it's anonymous<br>
54   * exception definition. It could be defined in <b>throw</b> statement
55   * immediately.<br>
56   * In that case, after literal new, there would be an expression type finishing
57   * with and ObjBlock.<br>
58   * <br>
59   * @author <a href="mailto:nesterenko-aleksey@list.ru">Aleksey Nesterenko</a>
60   * @author <a href="mailto:maxvetrenko2241@gmail.com">Max Vetrenko</a>
61   */
62  public class ForbidThrowAnonymousExceptionsCheck extends Check
63  {
64      /**
65       * Warning message key.
66       */
67      public static final String MSG_KEY = "forbid.throw.anonymous.exception";
68      
69      private static final String DEFAULT_EXCEPTION_CLASS_NAME_REGEX = "^.*Exception";
70      
71      private Pattern pattern = Pattern.compile(DEFAULT_EXCEPTION_CLASS_NAME_REGEX);
72  
73      private List<String> anonymousExceptions = new ArrayList<String>();
74      
75      public void setExceptionClassNameRegex(String exceptionClassNameRegex) {
76          this.pattern = Pattern.compile(exceptionClassNameRegex);
77      }
78  
79      @Override
80      public int[] getDefaultTokens()
81      {
82          return new int[] { TokenTypes.LITERAL_THROW, TokenTypes.VARIABLE_DEF };
83      }
84  
85      @Override
86      public void visitToken(DetailAST literalThrowOrVariableDefAst)
87      {
88          switch (literalThrowOrVariableDefAst.getType()) {
89              case TokenTypes.LITERAL_THROW:
90                  identifyThrowingAnonymousException(literalThrowOrVariableDefAst);
91                  break;
92              case TokenTypes.VARIABLE_DEF:
93                  lookForAnonymousExceptionDefinition(literalThrowOrVariableDefAst);
94                  break;
95              default:
96                  Utils.reportInvalidToken(literalThrowOrVariableDefAst.getType());
97                  break;
98          }
99      }
100 
101     /**
102      * Warns on throwing anonymous exception
103      */
104     private void
105             identifyThrowingAnonymousException(DetailAST throwDefAst)
106     {
107         DetailAST throwingLiteralNewAst = getLiteralNew(throwDefAst);
108 
109         if (throwingLiteralNewAst != null
110                 && hasObjectBlock(throwingLiteralNewAst)) {
111             log(throwDefAst.getLineNo(), MSG_KEY);
112         }
113 
114         else if (throwingLiteralNewAst == null) {
115             DetailAST throwingExceptionNameAst = getThrowingExceptionNameAst(throwDefAst
116                     .getFirstChild());
117             if (throwingExceptionNameAst != null
118                     && anonymousExceptions.contains(throwingExceptionNameAst
119                             .getText())) {
120                 log(throwDefAst.getLineNo(), MSG_KEY);
121             }
122         }
123     }
124 
125     /**
126      * Analyzes variable definition for anonymous exception definition, if found
127      * - adds it to list of anonymous exceptions
128      */
129     private void
130             lookForAnonymousExceptionDefinition(DetailAST variableDefAst)
131     {
132         DetailAST variableLiteralNewAst = null;
133         DetailAST variableAssignment = variableDefAst.findFirstToken(TokenTypes.ASSIGN);
134         if (variableAssignment != null && variableAssignment.getFirstChild() != null) {
135             variableLiteralNewAst = getLiteralNew(variableAssignment);
136         }
137 
138         DetailAST variableNameAst = variableDefAst
139                 .findFirstToken(TokenTypes.TYPE).getNextSibling();
140         if (isExceptionName(variableNameAst)) {
141             String exceptionName = variableNameAst.getText();
142 
143             if (anonymousExceptions.contains(exceptionName)) {
144                 anonymousExceptions.remove(exceptionName);
145             }
146 
147             if (variableLiteralNewAst != null
148                     && hasObjectBlock(variableLiteralNewAst)) {
149                 anonymousExceptions.add(exceptionName);
150             }
151         }
152     }
153 
154     /**
155      * Gets the literal new node from variable definition node or throw node
156      */
157     private static DetailAST
158             getLiteralNew(DetailAST literalThrowOrVariableDefAst)
159     {
160         return literalThrowOrVariableDefAst.getFirstChild().findFirstToken(
161                 TokenTypes.LITERAL_NEW);
162     }
163 
164     /**
165      * Retrieves the AST node which contains the name of throwing exception
166      */
167     private static DetailAST
168             getThrowingExceptionNameAst(DetailAST expressionAst)
169     {
170         return expressionAst.findFirstToken(TokenTypes.IDENT);
171     }
172 
173     /**
174      * Checks if definition with a literal new has an ObjBlock
175      */
176     private static boolean hasObjectBlock(DetailAST literalNewAst)
177     {
178         return literalNewAst.getLastChild().getType() == TokenTypes.OBJBLOCK;
179     }
180 
181     /**
182      * Checks if variable name is definitely an exception name. It is so if
183      * variable type ends with "Exception" suffix
184      */
185     private boolean isExceptionName(DetailAST variableNameAst)
186     {
187         DetailAST typeAst = variableNameAst.getPreviousSibling();
188         String typeName = typeAst.getFirstChild().getText();
189         return pattern.matcher(typeName).matches();
190     }
191 
192 }
