19  package com.github.sevntu.checkstyle.checks.coding;
20  
21  import java.util.LinkedList;
22  import java.util.List;
23  
24  import com.puppycrawl.tools.checkstyle.api.Check;
25  import com.puppycrawl.tools.checkstyle.api.DetailAST;
26  import com.puppycrawl.tools.checkstyle.api.TokenTypes;
27  
28  /**
29   * Restricts the number of ternary operators in expression to a specific limit.<br><br>
30   * <b>Rationale:</b> This Check helps to improve code readability by pointing developer on<br>
31   * expressions which contain more than user-defined count of ternary operators.<br><br>
32   * It points to complicated ternary 
33   * <a href="http://docs.oracle.com/javase/tutorial/java/nutsandbolts/expressions.html">expressions</a>. 
34   * Reason:<br>
35   * - Complicated ternary expressions are not easy to read.<br>
36   * - Complicated ternary expressions could lead to ambiguous result if user<br>
37   * does not know Java's operators priority well, e.g.:<br>
38   * <p>
39   * <code>
40   * <pre>
41   * String str = null;
42   * String x = str != null ? "A" : "B" + str == null ? "C" : "D";
43   * System.out.println(x);
44   * </pre>
45   * </code>
46   * </p>
47   * Output for code above is "D", but more obvious would be "BC".<br>
48   * </p>
49   * Check has following properties:<br>
50   * <ul>
51   * <li><b>maxTernaryPerExpressionCount</b> - limit of ternary operators per
52   * expression<br>
53   * </li>
54   * <li><b>ignoreTernaryOperatorsInBraces</b> - if true Check will ignore ternary operators<br>
55   * in braces (braces explicitly set priority level)<br>
56   * </li>
57   * <li><b>ignoreIsolatedTernaryOnLine</b> - if true Check will ignore one line ternary operators,<br>
58   * if only it is places in line alone.<br> 
59   * </li>
60   * </ul>
61   * Options <b>ignoreTernaryOperatorsInBraces</b> and <b>ignoreIsolatedTernaryOnLine</b> can<br>
62   * make Check less strict, e.g.:<br>
63   * Using <b>ignoreTernaryOperatorsInBraces</b> option (value = <b>true</b>)<br>
64   * does not put violation on code below:<br>
65   * <p>
66   * <code>
67   * <pre>
68   * callString = "{? = call " +
69   *   (StringUtils.hasLength(catalogNameToUse) 
70   *   ? catalogNameToUse + "." : "") +
71   *   (StringUtils.hasLength(schemaNameToUse) 
72   *   ? schemaNameToUse + "." : "") +
73   *   procedureNameToUse + "(";
74   * </pre>
75   * </code>
76   * </p>
77   * When using <b>ignoreIsolatedTernaryOnLine</b> (value = <b>true</b>), even without<br>
78   * <b>ignoreTernaryOperatorsInBraces</b> option Check won't warn on code below:<br> 
79   * <p>
80   * <code>
81   * <pre>
82   * int a = (d == 5) ? d : f
83   *   +
84   *   ((d == 6) ? g : k);
85   * </pre>
86   * </code>
87   * </p>
88   * @author <a href="mailto:nesterenko-aleksey@list.ru">Aleksey Nesterenko</a>
89   */
90  
91  public class TernaryPerExpressionCountCheck extends Check {
92  
93      public static final String MSG_KEY = "ternary.per.expression.count";
94  
95      private static final int DEFAULT_MAX_TERNARY_PER_EXPRESSION_COUNT = 1;
96  
97      private int maxTernaryPerExpressionCount = DEFAULT_MAX_TERNARY_PER_EXPRESSION_COUNT;
98  
99      private boolean ignoreTernaryOperatorsInBraces = true;
100     
101     private boolean ignoreIsolatedTernaryOnLine = true;
102 
103     @Override
104     public int[] getDefaultTokens() {
105         return new int[] { TokenTypes.EXPR };
106     }
107 
108     /**
109      * Sets the maximum number of ternary operators, default value = 1
110      * 
111      * @param maxTernaryPerExpressionCount
112      *            Number of ternary operators per expression
113      */
114     public void setMaxTernaryPerExpressionCount(int maxTernaryPerExpressionCount) {
115         if (maxTernaryPerExpressionCount < 0) {
116             throw new IllegalArgumentException();
117         }
118         this.maxTernaryPerExpressionCount = maxTernaryPerExpressionCount;
119     }
120 
121     /**
122      * Sets parameter to ignore ternary operators in braces, default value =
123      * true
124      * 
125      * @param ignoreTernaryOperatorsInBraces
126      */
127     public void setIgnoreTernaryOperatorsInBraces(boolean ignoreTernaryOperatorsInBraces) {
128         this.ignoreTernaryOperatorsInBraces = ignoreTernaryOperatorsInBraces;
129     }
130 
131     /**
132      * Sets parameter to ignore expressions in case if ternary operator is isolated in line
133      * 
134      * @param ignoreIsolatedTernaryOnLine
135      */
136     public void setIgnoreIsolatedTernaryOnLine(boolean ignoreIsolatedTernaryOnLine) {
137         this.ignoreIsolatedTernaryOnLine = ignoreIsolatedTernaryOnLine;
138     }
139 
140     @Override
141     public void visitToken(DetailAST expressionNode) {
142 
143         List<DetailAST> questionNodes = getQuestionNodes(expressionNode);
144 
145         if (questionNodes.size() > maxTernaryPerExpressionCount) {
146             DetailAST firstQuestionNode = questionNodes.get(0);
147             log(firstQuestionNode, MSG_KEY, maxTernaryPerExpressionCount);
148         }
149     }
150 
151     /**
152      * Puts question nodes from current expression node into the list
153      * @param expressionNode
154      *          Globally considering expression node
155      * @return
156      *          List of question nodes
157      */
158     private List<DetailAST> getQuestionNodes(DetailAST expressionNode) {
159 
160         List<DetailAST> questionNodes = new LinkedList<DetailAST>();
161 
162         DetailAST currentNode = expressionNode;
163 
164         do {
165             currentNode = getNextNode(expressionNode, currentNode);
166             if (currentNode != null
167                     && currentNode.getType() == TokenTypes.QUESTION
168                     && !isSkipTernaryOperator(currentNode)) {
169                 questionNodes.add(currentNode);
170             }
171 
172         } while (currentNode != null);
173         
174         return questionNodes;
175     }
176 
177     /**
178      * Checks if options <b>ignoreTernaryInBraces</b> or
179      * <b>ignoreOneTernaryPerLine</b> were set, hence, count ternary
180      * operators in current expression or not
181      * @param questionAST
182      */
183     private boolean isSkipTernaryOperator(DetailAST questionAST) {
184         return (ignoreTernaryOperatorsInBraces && isTernaryOperatorInBraces(questionAST))
185                 || (ignoreIsolatedTernaryOnLine && isIsolatedTernaryOnLine(questionAST));
186     }
187 
188     /**
189      * Checks ternary operator if it is in braces, which are explicitly setting
190      * the priority level
191      * @param questionAST
192      */
193     private static boolean isTernaryOperatorInBraces(DetailAST questionAST) {
194         return questionAST.getPreviousSibling() != null
195                 && questionAST.getPreviousSibling().getType() == TokenTypes.LPAREN;
196     }
197 
198 
199     /**
200      * Checks if there's one ternary operator per line
201      * @param questionAST
202      */
203     private boolean isIsolatedTernaryOnLine(DetailAST questionAST) {
204         int lineNo = questionAST.getLineNo() - 1;
205         String line = getFileContents().getText().get(lineNo);
206         
207         return isSingleTernaryLine(line, lineNo); 
208     }
209 
210     /**
211      * Checks line parameter on containing more than 1 ternary operator
212      * @param line
213      */
214     private boolean isSingleTernaryLine(String line, int lineNo) {
215         int questionsPerLine = 0;
216         char[] charArrayFromLine = line.toCharArray();
217         for (int i = 0; i < line.length(); i++) {
218             char currentSymbol = charArrayFromLine[i];
219             if (currentSymbol == '?' && !getFileContents().hasIntersectionWithComment(lineNo + 1,
220                     i, lineNo + 1, i)) {
221                 questionsPerLine++;
222             }
223             if (questionsPerLine > 1) {
224                 break;
225             }
226         }
227         
228         return questionsPerLine == 1;
229     }
230 
231     /**
232      * Gets the next node of a syntactical tree (child of a current node or
233      * sibling of a current node, or sibling of a parent of a current node)
234      * 
235      * @param expressionNode
236      *            Globally considering expression node
237      * @param currentNode
238      *            Current node of syntactical tree
239      * @return Next node after bypassing
240      */
241     private static DetailAST getNextNode(DetailAST expressionNode,
242             DetailAST currentNode) {
243         DetailAST toVisit = currentNode.getFirstChild();
244         while (toVisit == null && currentNode != expressionNode) {
245             toVisit = currentNode.getNextSibling();
246             if (toVisit == null) {
247                 currentNode = currentNode.getParent();
248             }
249         }
250         
251         return toVisit;
252     }
253 }
