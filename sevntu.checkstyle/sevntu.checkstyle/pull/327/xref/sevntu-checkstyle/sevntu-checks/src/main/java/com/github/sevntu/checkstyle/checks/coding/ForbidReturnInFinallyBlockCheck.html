19  package com.github.sevntu.checkstyle.checks.coding;
20  
21  import java.util.ArrayList;
22  import java.util.Collections;
23  import java.util.List;
24  
25  import com.puppycrawl.tools.checkstyle.api.Check;
26  import com.puppycrawl.tools.checkstyle.api.DetailAST;
27  import com.puppycrawl.tools.checkstyle.api.TokenTypes;
28  
29  /**
30   * <p>
31   * The finally block is always executed unless there is abnormal program termination, either
32   * resulting from a JVM crash or from a call to System.exit(0). On top of that, any value returned
33   * from within the finnally block will override the value returned prior to execution of the finally
34   * block. This check reports if the finally block contains a return statement.
35   * </p>
36   * 
37   * @author <a href="mailto:andrew.uljanenko@gmail.com">Andrew Uljanenko</a>
38   */
39  
40  public class ForbidReturnInFinallyBlockCheck extends Check
41  {
42  
43      public static final String MSG_KEY = "forbid.return.in.finally.block";
44  
45      @Override
46      public final int[] getDefaultTokens()
47      {
48          return new int[] { TokenTypes.LITERAL_FINALLY };
49      }
50  
51      @Override
52      public void visitToken(DetailAST finallyNode)
53      {
54          final DetailAST firstSlistNode = finallyNode.findFirstToken(TokenTypes.SLIST);
55  
56          final List<DetailAST> listOfReturnNodes = getReturnNodes(firstSlistNode);
57  
58          for (DetailAST returnNode : listOfReturnNodes) {
59              if (!isReturnInMethodDefinition(returnNode)) {
60                  log(finallyNode.getLineNo(), MSG_KEY);
61              }
62          }
63      }
64  
65      private List<DetailAST> getReturnNodes(DetailAST node)
66      {
67          final List<DetailAST> result = new ArrayList<DetailAST>();
68          DetailAST child = node.getFirstChild();
69          while (child != null) {
70              if (child.getType() == TokenTypes.LITERAL_RETURN) {
71                  return Collections.singletonList(child);
72              }
73              result.addAll(getReturnNodes(child));
74              child = child.getNextSibling();
75          }
76          return result;
77      }
78  
79      private boolean isReturnInMethodDefinition(DetailAST returnNode)
80      {
81          Boolean result = false;
82          DetailAST node = returnNode;
83          while (node.getType() != TokenTypes.LITERAL_FINALLY) {
84              if (node.getType() == TokenTypes.METHOD_DEF) {
85                  result = true;
86                  break;
87              }
88              node = node.getParent();
89          }
90          return result;
91      }
92  }
