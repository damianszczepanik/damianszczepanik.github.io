19  package com.github.sevntu.checkstyle.checks.coding;
20  
21  import java.util.regex.Pattern;
22  
23  import com.puppycrawl.tools.checkstyle.api.Check;
24  import com.puppycrawl.tools.checkstyle.api.DetailAST;
25  import com.puppycrawl.tools.checkstyle.api.TokenTypes;
26  
27  /**
28   * <p>
29   * This check verifies the name of JUnit4 test class for compliance with user
30   * defined naming convention(by default Check expects test classes names
31   * matching
32   * ".+Test\\d*|.+Tests\\d*|Test.+|Tests.+|.+IT|.+ITs|.+TestCase\\d*|.+TestCases\\d*"
33   * regex).
34   * </p>
35   * <p>
36   * Class is considered to be a test if its definition or one of its method
37   * definitions annotated with user defined annotations. By default Check looks
38   * for classes which contain methods annotated with "Test" or "org.junit.Test".
39   * </p>
40   * <p>
41   * Check has following options:
42   * </p>
43   * <p>
44   * "expectedClassNameRegex" - regular expression which matches expected test
45   * class names. If test class name does not matches this regex then Check will
46   * log violation. This option defaults to
47   * ".+Test\\d*|.+Tests\\d*|Test.+|Tests.+|.+IT|.+ITs|.+TestCase\\d*|.+TestCases\\d*".
48   * </p>
49   * <p>
50   * "classAnnotationNameRegex" - regular expression which matches test annotation
51   * names on classes. If class annotated with matching annotation, it is
52   * considered to be a test. This option defaults to empty regex(one that matches
53   * nothing). If for example this option set to "RunWith", then class "SomeClass"
54   * is considered to be a test:
55   * 
56   * <pre>
57   * <code>
58   * {@literal @}RunWith(Parameterized.class)
59   * class SomeClass
60   * {
61   * }
62   * </code>
63   * </pre>
64   * 
65   * </p>
66   * <p>
67   * "methodAnnotationNameRegex" - regular expression which matches test
68   * annotation names on methods. If class contains a method annotated with
69   * matching annotation, it is considered to be a test. This option defaults to
70   * "Test|org.junit.Test". For example, if this option set to "Test", then class
71   * "SomeClass" is considered to be a test.
72   * 
73   * <pre>
74   * <code>
75   * class SomeClass
76   * {
77   *      {@literal @}Test
78   *      void method() {
79   *      
80   *      }
81   * }
82   * </code>
83   * </pre>
84   * 
85   * </p>
86   * <p>
87   * Annotation names must be specified exactly the same way it specified in code,
88   * thus if Check must match annotation with fully qualified name, corresponding
89   * options must contain qualified annotation name and vice versa. For example,
90   * if annotation regex is "org.junit.Test" Check will recognize "{@literal @}
91   * org.junit.Test" annotation and will skip "{@literal @}Test" annotation and
92   * vice versa if annotation regex is "Test" Check will recognize "{@literal @}
93   * Test" annotation and skip "{@literal @}org.junit.Test" annotation.
94   * </p>
95   * <p>
96   * Following configuration will adjust Check to look for classes annotated with
97   * annotation "RunWith" or classes with methods annotated with "Test" and verify
98   * that classes names end with "Test" or "Tests".
99   * 
100  * <pre>
101  *     &ltmodule name="NameConventionForJUnit4TestClassesCheck"&gt
102  *       &ltproperty name="expectedClassNameRegex" value=".+Tests|.+Test"/&gt
103  *       &ltproperty name="classAnnotationNameRegex" value="RunWith"/&gt
104  *       &ltproperty name="methodAnnotationNameRegex" value="Test"/&gt
105  *     &lt/module&gt
106  * </pre>
107  * </p>
108  * 
109  * @author <a href="mailto:zuy_alexey@mail.ru">Zuy Alexey</a>
110  */
111 public class NameConventionForJunit4TestClassesCheck extends Check
112 {
113     /**
114      * Violation message key.
115      */
116     public static final String MSG_KEY = "name.convention.for.test.classes";
117 
118     /**
119      * <p>
120      * Regular expression which matches expected names of JUnit test classes.
121      * </p>
122      * <p>
123      * Default value is
124      * ".+Test\\d*|.+Tests\\d*|Test.+|Tests.+|.+IT|.+ITs|.+TestCase\\d*|.+TestCases\\d*".
125      * </p>
126      */
127     private Pattern expectedClassNameRegex =
128             Pattern.compile(".+Test\\d*|.+Tests\\d*|Test.+|Tests.+|.+IT|.+ITs|.+TestCase\\d*|.+TestCases\\d*");
129 
130     /**
131      * <p>
132      * Regular expression which matches JUnit class test annotation names.
133      * </p>
134      * <p>
135      * By default this regex is empty.
136      * </p>
137      */
138     private Pattern classAnnotationNameRegex;
139 
140     /**
141      * <p>
142      * Regular expression which matches JUnit method test annotation names.
143      * </p>
144      * <p>
145      * Default value is "Test|org.junit.Test".
146      * </p>
147      */
148     private Pattern methodAnnotationNameRegex =
149             Pattern.compile("Test|org.junit.Test");
150 
151     /**
152      * Sets regexp to match 'expected' class names for JUnit tests.
153      * @param expectedClassNameRegex
154      *        regexp to match 'correct' JUnit test class names.
155      */
156     public void setExpectedClassNameRegex(String expectedClassNameRegex)
157     {
158         if (expectedClassNameRegex != null && !expectedClassNameRegex.isEmpty()) {
159             this.expectedClassNameRegex = Pattern.compile(expectedClassNameRegex);
160         }
161         else {
162             this.expectedClassNameRegex = null;
163         }
164     }
165 
166     /**
167      * Sets class test annotation name regexp for JUnit tests.
168      * @param annotationNameRegex
169      *        regexp to match annotations for unit test classes.
170      */
171     public void setClassAnnotationNameRegex(String annotationNameRegex)
172     {
173         if (annotationNameRegex != null && !annotationNameRegex.isEmpty()) {
174             classAnnotationNameRegex = Pattern.compile(annotationNameRegex);
175         }
176         else {
177             classAnnotationNameRegex = null;
178         }
179     }
180 
181     /**
182      * Sets method test annotation name regexp for JUnit tests.
183      * @param annotationNameRegex
184      *        regexp to match annotations for unit test classes.
185      */
186     public void setMethodAnnotationNameRegex(String annotationNameRegex)
187     {
188         if (annotationNameRegex != null && !annotationNameRegex.isEmpty()) {
189             methodAnnotationNameRegex = Pattern.compile(annotationNameRegex);
190         }
191         else {
192             methodAnnotationNameRegex = null;
193         }
194     }
195 
196     @Override
197     public int[] getDefaultTokens()
198     {
199         return new int[] { TokenTypes.CLASS_DEF };
200     }
201 
202     @Override
203     public void visitToken(DetailAST classDefNode)
204     {
205         if ((isClassDefinitionAnnotated(classDefNode) || isAtleastOneMethodAnnotated(classDefNode))
206                 && hasUnexpectedName(classDefNode)) {
207             logUnexpectedClassName(classDefNode);
208         }
209     }
210 
211     /**
212      * Checks whether class definition annotated with user defined annotation.
213      * @param classDefNode
214      *        a class definition node
215      * @return true, if class definition annotated with user defined annotation
216      */
217     private boolean isClassDefinitionAnnotated(DetailAST classDefNode)
218     {
219         return hasAnnotation(classDefNode, classAnnotationNameRegex);
220     }
221 
222     /**
223      * Checks whether class contains at least one method annotated with user
224      * defined annotation.
225      * @param classDefNode
226      *        a class definition node
227      * @return true, if class contains at least one method annotated with user
228      *         defined annotation
229      */
230     private boolean isAtleastOneMethodAnnotated(DetailAST classDefNode)
231     {
232         DetailAST classMemberNode =
233                 classDefNode.findFirstToken(TokenTypes.OBJBLOCK).getFirstChild();
234 
235         while (classMemberNode != null) {
236             if (classMemberNode.getType() == TokenTypes.METHOD_DEF &&
237                     hasAnnotation(classMemberNode, methodAnnotationNameRegex))
238             {
239                 return true;
240             }
241 
242             classMemberNode = classMemberNode.getNextSibling();
243         }
244 
245         return false;
246     }
247 
248     /**
249      * Returns true, if class has unexpected name.
250      * @param classDefNode
251      *        a class definition node
252      * @return true, if class has unexpected name
253      */
254     private boolean hasUnexpectedName(final DetailAST classDefNode)
255     {
256         final String className = getIdentifierName(classDefNode);
257         return !isMatchesRegex(expectedClassNameRegex, className);
258     }
259 
260     /**
261      * Returns true, if class or method has annotation with name specified in
262      * regexp.
263      * @param methodOrClassDefNode
264      *        the node of type TokenTypes.METHOD_DEF or TokenTypes.CLASS_DEF
265      * @param annotationNamesRegexp
266      *        regexp contains annotation names
267      * @return true, if the class or method contains one of the annotations,
268      *         specified in the regexp
269      */
270     private boolean hasAnnotation(DetailAST methodOrClassDefNode, Pattern annotationNamesRegexp)
271     {
272         DetailAST modifierNode =
273                 methodOrClassDefNode.findFirstToken(TokenTypes.MODIFIERS).getFirstChild();
274 
275         boolean result = false;
276 
277         while (modifierNode != null) {
278             if (modifierNode.getType() == TokenTypes.ANNOTATION) {
279                 String annotationName = getIdentifierName(modifierNode);
280 
281                 if (isMatchesRegex(annotationNamesRegexp, annotationName)) {
282                     result = true;
283                     break;
284                 }
285             }
286 
287             modifierNode = modifierNode.getNextSibling();
288         }
289 
290         return result;
291     }
292 
293     /**
294      * Logs unexpected class name.
295      * @param classDef
296      *        the node of type TokenTypes.CLASS_DEF
297      */
298     private void logUnexpectedClassName(DetailAST classDef)
299     {
300         log(classDef.findFirstToken(TokenTypes.IDENT), MSG_KEY, expectedClassNameRegex);
301     }
302 
303     /**
304      * Returns name of identifier contained in specified node.
305      * @param aNodeWithIdent
306      *        a node containing identifier or qualified identifier.
307      * @return identifier name for specified node. If node contains qualified
308      *         name then method returns its text representation.
309      */
310     private static String getIdentifierName(DetailAST identifierNode)
311     {
312         DetailAST identNode = identifierNode.findFirstToken(TokenTypes.IDENT);
313         String result;
314 
315         if (identNode != null) {
316             result = identNode.getText();
317         }
318         else {
319             result = "";
320 
321             DetailAST node = identifierNode.findFirstToken(TokenTypes.DOT);
322 
323             while (node.getType() == TokenTypes.DOT) {
324                 result = "." + node.getLastChild().getText() + result;
325 
326                 node = node.getFirstChild();
327             }
328 
329             result = node.getText() + result;
330         }
331 
332         return result;
333     }
334 
335     /**
336      * Matches string against regexp.
337      * @param regexPattern
338      *        regex to match string with. May be null.
339      * @param str
340      *        a string to match against regex.
341      * @return false if regex is null, otherwise result of matching string
342      *         against regex.
343      */
344     private static boolean isMatchesRegex(Pattern regexPattern, String str)
345     {
346         if (regexPattern != null) {
347             return regexPattern.matcher(str).matches();
348         }
349         else {
350             return false;
351         }
352     }
