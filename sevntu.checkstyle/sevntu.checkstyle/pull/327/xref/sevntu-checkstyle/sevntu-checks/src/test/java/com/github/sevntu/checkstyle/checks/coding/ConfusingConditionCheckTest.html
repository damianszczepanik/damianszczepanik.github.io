31  public class ConfusingConditionCheckTest extends BaseCheckTestSupport
32  {
33  
34  	private final String warningMessage = getCheckMessage(MSG_KEY);
35  	
36  	private final DefaultConfiguration checkConfig = createCheckConfig(ConfusingConditionCheck.class);
37  
38      @Test
39  	public void testDefault() throws Exception
40      {
41  		checkConfig.addAttribute("ignoreInnerIf", "true");
42  		checkConfig.addAttribute("ignoreNullCaseInIf", "true");
43  		checkConfig.addAttribute("ignoreSequentialIf", "true");
44  		checkConfig.addAttribute("ignoreThrowInElse", "true");
45  		checkConfig.addAttribute("multiplyFactorForElseBlocks", "4");
46          
47          final String[] expected = {
48                  "10: " + warningMessage,
49                  "13: " + warningMessage,
50                  "16: " + warningMessage,
51                  "19: " + warningMessage,
52                  "22: " + warningMessage,
53                  "105: " + warningMessage,
54                  "108: " + warningMessage,
55                  "111: " + warningMessage,
56                  "149: " + warningMessage,
57                  "166: " + warningMessage,
58                  "177: " + warningMessage, //!!!
59                  "181: " + warningMessage,
60                  "200: " + warningMessage,
61                  "215: " + warningMessage,
62                  "231: " + warningMessage,};
63          
64          verify(checkConfig, getPath("InputConfusingConditionCheck.java"),
65                  expected);
66      }
67  }
