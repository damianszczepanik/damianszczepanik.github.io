12      @Test
13      public void testCustomDecrationOrder() throws Exception
14      {
15          final DefaultConfiguration checkConfig =
16              createCheckConfig(CustomDeclarationOrderCheck.class);
17          final String[] expected = {
18          		"10:5: " + getCheckMessage(MSG_KEY_FIELD, "Field(private static final long serialVersionUID)", "Field(.*final public .*|.*public final .*)"),
19          		"20:5: " + getCheckMessage(MSG_KEY_FIELD, "Field(protected .*)", "Field(private .*)"),
20          		"22:5: " + getCheckMessage(MSG_KEY_FIELD, "Field(@SuppressWarnings(.*serial.*).*)", "Field(private .*)"),
21          		"45:5: " + getCheckMessage(MSG_KEY_CONSTRUCTOR, "Ctor()", "Method(.*abstract .*public .*|.*public .*abstract .*|protected .*)"),
22          		"52:5: " + getCheckMessage(MSG_KEY_METHOD, "Method(public static .*(new|edit|create).*)", "Method(.*abstract .*public .*|.*public .*abstract .*|protected .*)"),
23                  "60:9: " + getCheckMessage(MSG_KEY_FIELD, "Field(.*final public .*|.*public final .*)", "Field(private .*)"),
24                  "80:9: " + getCheckMessage(MSG_KEY_CLASS, "InnerClass (public .*abstract .*)", "InnerClass (private .*)"),
25                  "84:5: " + getCheckMessage(MSG_KEY_METHOD, "Method(@Deprecated .*)", "InnerClass (private .*)"),
26                  "95:5: " + getCheckMessage(MSG_KEY_METHOD, "Method(.*abstract .*public .*|.*public .*abstract .*|protected .*)", "InnerClass (private .*)"),
27                  "105:5: " + getCheckMessage(MSG_KEY_FIELD, "Field(private .*)", "Ctor()"), 
28                  };
29          checkConfig.addAttribute("customDeclarationOrder",
30                                    "Field(private static final long serialVersionUID) ###"
31                                  + "Field(.*final public .*|.*public final .*) ###"
32                                  + "Field(@SuppressWarnings(.*serial.*).*) ###"
33                                  + "Field(protected .*) ###"
34                                  + "Field(private .*) ###"
35                                  + "Ctor()###"
36                                  + "Method(@Deprecated .*) ###"
37                                  + "Method(public static .*(new|edit|create).*) ###"
38                                  + "Method() ###"
39                                  + "Method(.*abstract .*public .*|.*public .*abstract .*|protected .*) ###"
40                                  + "InnerClass (public .*abstract .*) ###"
41                                  + "InnerClass (private .*)");
42          checkConfig.addAttribute("caseSensitive", "true");
43          verify(checkConfig, getPath("InputCustomDeclarationOrder.java"), expected);
44          checkConfig.addAttribute("customDeclarationOrder", "Field .*final.*public|.*public.*final)");
45      }
