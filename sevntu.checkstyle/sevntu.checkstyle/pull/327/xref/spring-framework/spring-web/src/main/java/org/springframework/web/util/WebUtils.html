141 	 * Set a system property to the web application root directory.
142 	 * The key of the system property can be defined with the "webAppRootKey"
143 	 * context-param in {@code web.xml}. Default is "webapp.root".
144 	 * <p>Can be used for tools that support substition with {@code System.getProperty}
145 	 * values, like log4j's "${key}" syntax within log file locations.
146 	 * @param servletContext the servlet context of the web application
147 	 * @throws IllegalStateException if the system property is already set,
148 	 * or if the WAR file is not expanded
149 	 * @see #WEB_APP_ROOT_KEY_PARAM
150 	 * @see #DEFAULT_WEB_APP_ROOT_KEY
151 	 * @see WebAppRootListener
152 	 * @see Log4jWebConfigurer
153 	 */
154 	public static void setWebAppRootSystemProperty(ServletContext servletContext) throws IllegalStateException {
155 		Assert.notNull(servletContext, "ServletContext must not be null");
156 		String root = servletContext.getRealPath("/");
157 		if (root == null) {
158 			throw new IllegalStateException(
159 				"Cannot set web app root system property when WAR file is not expanded");
160 		}
161 		String param = servletContext.getInitParameter(WEB_APP_ROOT_KEY_PARAM);
162 		String key = (param != null ? param : DEFAULT_WEB_APP_ROOT_KEY);
163 		String oldValue = System.getProperty(key);
164 		if (oldValue != null && !StringUtils.pathEquals(oldValue, root)) {
165 			throw new IllegalStateException(
166 				"Web app root system property already set to different value: '" +
167 				key + "' = [" + oldValue + "] instead of [" + root + "] - " +
168 				"Choose unique values for the 'webAppRootKey' context-param in your web.xml files!");
169 		}
170 		System.setProperty(key, root);
171 		servletContext.log("Set web app root system property: '" + key + "' = [" + root + "]");
172 	}

255 	 * Return the real path of the given path within the web application,
256 	 * as provided by the servlet container.
257 	 * <p>Prepends a slash if the path does not already start with a slash,
258 	 * and throws a FileNotFoundException if the path cannot be resolved to
259 	 * a resource (in contrast to ServletContext's {@code getRealPath},
260 	 * which returns null).
261 	 * @param servletContext the servlet context of the web application
262 	 * @param path the path within the web application
263 	 * @return the corresponding real path
264 	 * @throws FileNotFoundException if the path cannot be resolved to a resource
265 	 * @see javax.servlet.ServletContext#getRealPath
266 	 */
267 	public static String getRealPath(ServletContext servletContext, String path) throws FileNotFoundException {
268 		Assert.notNull(servletContext, "ServletContext must not be null");
269 		// Interpret location as relative to the web application root directory.
270 		if (!path.startsWith("/")) {
271 			path = "/" + path;
272 		}
273 		String realPath = servletContext.getRealPath(path);
274 		if (realPath == null) {
275 			throw new FileNotFoundException(
276 					"ServletContext resource [" + path + "] cannot be resolved to absolute file path - " +
277 					"web application archive not expanded?");
278 		}
279 		return realPath;
280 	}

348 	/**
349 	 * Get the specified session attribute, creating and setting a new attribute if
350 	 * no existing found. The given class needs to have a public no-arg constructor.
351 	 * Useful for on-demand state objects in a web tier, like shopping carts.
352 	 * @param session current HTTP session
353 	 * @param name the name of the session attribute
354 	 * @param clazz the class to instantiate for a new attribute
355 	 * @return the value of the session attribute, newly created if not found
356 	 * @throws IllegalArgumentException if the session attribute could not be instantiated
357 	 */
358 	public static Object getOrCreateSessionAttribute(HttpSession session, String name, Class<?> clazz)
359 			throws IllegalArgumentException {
360 
361 		Assert.notNull(session, "Session must not be null");
362 		Object sessionObject = session.getAttribute(name);
363 		if (sessionObject == null) {
364 			try {
365 				sessionObject = clazz.newInstance();
366 			}
367 			catch (InstantiationException ex) {
368 				throw new IllegalArgumentException(
369 					"Could not instantiate class [" + clazz.getName() +
370 					"] for session attribute '" + name + "': " + ex.getMessage());
371 			}
372 			catch (IllegalAccessException ex) {
373 				throw new IllegalArgumentException(
374 					"Could not access default constructor of class [" + clazz.getName() +
375 					"] for session attribute '" + name + "': " + ex.getMessage());
376 			}
377 			session.setAttribute(name, sessionObject);
378 		}
379 		return sessionObject;
380 	}

382 	/**
383 	 * Return the best available mutex for the given session:
384 	 * that is, an object to synchronize on for the given session.
385 	 * <p>Returns the session mutex attribute if available; usually,
386 	 * this means that the HttpSessionMutexListener needs to be defined
387 	 * in {@code web.xml}. Falls back to the HttpSession itself
388 	 * if no mutex attribute found.
389 	 * <p>The session mutex is guaranteed to be the same object during
390 	 * the entire lifetime of the session, available under the key defined
391 	 * by the {@code SESSION_MUTEX_ATTRIBUTE} constant. It serves as a
392 	 * safe reference to synchronize on for locking on the current session.
393 	 * <p>In many cases, the HttpSession reference itself is a safe mutex
394 	 * as well, since it will always be the same object reference for the
395 	 * same active logical session. However, this is not guaranteed across
396 	 * different servlet containers; the only 100% safe way is a session mutex.
397 	 * @param session the HttpSession to find a mutex for
398 	 * @return the mutex object (never {@code null})
399 	 * @see #SESSION_MUTEX_ATTRIBUTE
400 	 * @see HttpSessionMutexListener
401 	 */
402 	public static Object getSessionMutex(HttpSession session) {
403 		Assert.notNull(session, "Session must not be null");
404 		Object mutex = session.getAttribute(SESSION_MUTEX_ATTRIBUTE);
405 		if (mutex == null) {
406 			mutex = session;
407 		}
408 		return mutex;
409 	}

413 	 * Return an appropriate request object of the specified type, if available,
414 	 * unwrapping the given request as far as necessary.
415 	 * @param request the servlet request to introspect
416 	 * @param requiredType the desired type of request object
417 	 * @return the matching request object, or {@code null} if none
418 	 * of that type is available
419 	 */
420 	@SuppressWarnings("unchecked")
421 	public static <T> T getNativeRequest(ServletRequest request, Class<T> requiredType) {
422 		if (requiredType != null) {
423 			if (requiredType.isInstance(request)) {
424 				return (T) request;
425 			}
426 			else if (request instanceof ServletRequestWrapper) {
427 				return getNativeRequest(((ServletRequestWrapper) request).getRequest(), requiredType);
428 			}
429 		}
430 		return null;
431 	}

542 	 * cookies can have the same name but different paths or domains.

544 	 * @param name cookie name

548 		Assert.notNull(request, "Request must not be null");
549 		Cookie cookies[] = request.getCookies();

565 	 * @param name name of the parameter
566 	 * @return if the parameter was sent
567 	 * @see #SUBMIT_IMAGE_SUFFIXES
568 	 */
569 	public static boolean hasSubmitParameter(ServletRequest request, String name) {
570 		Assert.notNull(request, "Request must not be null");
571 		if (request.getParameter(name) != null) {
572 			return true;
573 		}
574 		for (String suffix : SUBMIT_IMAGE_SUFFIXES) {
575 			if (request.getParameter(name + suffix) != null) {
576 				return true;
577 			}
578 		}
579 		return false;
580 	}

595 	/**
596 	 * Obtain a named parameter from the given request parameters.
597 	 * <p>This method will try to obtain a parameter value using the
598 	 * following algorithm:
599 	 * <ol>
600 	 * <li>Try to get the parameter value using just the given <i>logical</i> name.
601 	 * This handles parameters of the form <tt>logicalName = value</tt>. For normal
602 	 * parameters, e.g. submitted using a hidden HTML form field, this will return
603 	 * the requested value.</li>
604 	 * <li>Try to obtain the parameter value from the parameter name, where the
605 	 * parameter name in the request is of the form <tt>logicalName_value = xyz</tt>
606 	 * with "_" being the configured delimiter. This deals with parameter values
607 	 * submitted using an HTML form submit button.</li>
608 	 * <li>If the value obtained in the previous step has a ".x" or ".y" suffix,
609 	 * remove that. This handles cases where the value was submitted using an
610 	 * HTML form image button. In this case the parameter in the request would
611 	 * actually be of the form <tt>logicalName_value.x = 123</tt>. </li>
612 	 * </ol>
613 	 * @param parameters the available parameter map
614 	 * @param name the <i>logical</i> name of the request parameter
615 	 * @return the value of the parameter, or {@code null}
616 	 * if the parameter does not exist in given request
617 	 */
618 	public static String findParameterValue(Map<String, ?> parameters, String name) {
619 		// First try to get it as a normal name=value parameter
620 		Object value = parameters.get(name);
621 		if (value instanceof String[]) {
622 			String[] values = (String[]) value;
623 			return (values.length > 0 ? values[0] : null);
624 		}
625 		else if (value != null) {
626 			return value.toString();
627 		}
628 		// If no value yet, try to get it as a name_value=xyz parameter
629 		String prefix = name + "_";
630 		for (String paramName : parameters.keySet()) {
631 			if (paramName.startsWith(prefix)) {
632 				// Support images buttons, which would submit parameters as name_value.x=123
633 				for (String suffix : SUBMIT_IMAGE_SUFFIXES) {
634 					if (paramName.endsWith(suffix)) {
635 						return paramName.substring(prefix.length(), paramName.length() - suffix.length());
636 					}
637 				}
638 				return paramName.substring(prefix.length());
639 			}
640 		}
641 		// We couldn't find the parameter value...
642 		return null;
643 	}

645 	/**
646 	 * Return a map containing all parameters with the given prefix.
647 	 * Maps single values to String and multiple values to String array.
648 	 * <p>For example, with a prefix of "spring_", "spring_param1" and
649 	 * "spring_param2" result in a Map with "param1" and "param2" as keys.
650 	 * @param request HTTP request in which to look for parameters
651 	 * @param prefix the beginning of parameter names
652 	 * (if this is null or the empty string, all parameters will match)
653 	 * @return map containing request parameters <b>without the prefix</b>,
654 	 * containing either a String or a String array as values
655 	 * @see javax.servlet.ServletRequest#getParameterNames
656 	 * @see javax.servlet.ServletRequest#getParameterValues
657 	 * @see javax.servlet.ServletRequest#getParameterMap
658 	 */
659 	public static Map<String, Object> getParametersStartingWith(ServletRequest request, String prefix) {
660 		Assert.notNull(request, "Request must not be null");
661 		Enumeration<String> paramNames = request.getParameterNames();
662 		Map<String, Object> params = new TreeMap<String, Object>();
663 		if (prefix == null) {
664 			prefix = "";
665 		}
666 		while (paramNames != null && paramNames.hasMoreElements()) {
667 			String paramName = paramNames.nextElement();
668 			if ("".equals(prefix) || paramName.startsWith(prefix)) {
669 				String unprefixed = paramName.substring(prefix.length());
670 				String[] values = request.getParameterValues(paramName);
671 				if (values == null || values.length == 0) {
672 					// Do nothing, no values found at all.
673 				}
674 				else if (values.length > 1) {
675 					params.put(unprefixed, values);
676 				}
677 				else {
678 					params.put(unprefixed, values[0]);
679 				}
680 			}
681 		}
682 		return params;
683 	}

686 	 * Return the target page specified in the request.
687 	 * @param request current servlet request
688 	 * @param paramPrefix the parameter prefix to check for
689 	 * (e.g. "_target" for parameters like "_target1" or "_target2")
690 	 * @param currentPage the current page, to be returned as fallback
691 	 * if no target page specified
692 	 * @return the page specified in the request, or current page if not found
693 	 */
694 	public static int getTargetPage(ServletRequest request, String paramPrefix, int currentPage) {
695 		Enumeration<String> paramNames = request.getParameterNames();
696 		while (paramNames.hasMoreElements()) {
697 			String paramName = paramNames.nextElement();
698 			if (paramName.startsWith(paramPrefix)) {
699 				for (int i = 0; i < WebUtils.SUBMIT_IMAGE_SUFFIXES.length; i++) {
700 					String suffix = WebUtils.SUBMIT_IMAGE_SUFFIXES[i];
701 					if (paramName.endsWith(suffix)) {
702 						paramName = paramName.substring(0, paramName.length() - suffix.length());
703 					}
704 				}
705 				return Integer.parseInt(paramName.substring(paramPrefix.length()));
706 			}
707 		}
708 		return currentPage;
709 	}

714 	 * Correctly resolves nested paths such as "/products/view.html" as well.

720 		int dotIndex = filename.lastIndexOf('.');

728 	 * Extract the full URL filename (including file extension) from the given request URL path.
729 	 * Correctly resolves nested paths such as "/products/view.html" as well.
730 	 * @param urlPath the request URL path (e.g. "/products/index.html")
731 	 * @return the extracted URI filename (e.g. "index.html")
732 	 */
733 	public static String extractFullFilenameFromUrlPath(String urlPath) {
734 		int end = urlPath.indexOf(';');
735 		if (end == -1) {
736 			end = urlPath.indexOf('?');
737 			if (end == -1) {
738 				end = urlPath.length();
739 			}
740 		}
741 		int begin = urlPath.lastIndexOf('/', end) + 1;
742 		return urlPath.substring(begin, end);
743 	}

745 	/**
746 	 * Parse the given string with matrix variables. An example string would look
747 	 * like this {@code "q1=a;q1=b;q2=a,b,c"}. The resulting map would contain
748 	 * keys {@code "q1"} and {@code "q2"} with values {@code ["a","b"]} and
749 	 * {@code ["a","b","c"]} respectively.
750 	 * @param matrixVariables the unparsed matrix variables string
751 	 * @return a map with matrix variable names and values, never {@code null}
752 	 * @since 3.2
753 	 */
754 	public static MultiValueMap<String, String> parseMatrixVariables(String matrixVariables) {
755 		MultiValueMap<String, String> result = new LinkedMultiValueMap<String, String>();
756 		if (!StringUtils.hasText(matrixVariables)) {
757 			return result;
758 		}
759 		StringTokenizer pairs = new StringTokenizer(matrixVariables, ";");
760 		while (pairs.hasMoreTokens()) {
761 			String pair = pairs.nextToken();
762 			int index = pair.indexOf('=');
763 			if (index != -1) {
764 				String name = pair.substring(0, index);
765 				String rawValue = pair.substring(index + 1);
766 				for (String value : StringUtils.commaDelimitedListToStringArray(rawValue)) {
767 					result.add(name, value);
768 				}
769 			}
770 			else {
771 				result.add(pair, "");
772 			}
773 		}
774 		return result;
775 	}

777 	/**
778 	 * Check the given request origin against a list of allowed origins.
779 	 * A list containing "*" means that all origins are allowed.
780 	 * An empty list means only same origin is allowed.
781 	 * @return true if the request origin is valid, false otherwise
782 	 * @since 4.1.5
783 	 * @see <a href="https://tools.ietf.org/html/rfc6454">RFC 6454: The Web Origin Concept</a>
784 	 */
785 	public static boolean isValidOrigin(HttpRequest request, Collection<String> allowedOrigins) {
786 		Assert.notNull(request, "Request must not be null");
787 		Assert.notNull(allowedOrigins, "Allowed origins must not be null");
788 
789 		String origin = request.getHeaders().getOrigin();
790 		if (origin == null || allowedOrigins.contains("*")) {
791 			return true;
792 		}
793 		else if (allowedOrigins.isEmpty()) {
794 			UriComponents originComponents;
795 			try {
796 				originComponents = UriComponentsBuilder.fromHttpUrl(origin).build();
797 			}
798 			catch (IllegalArgumentException ex) {
799 				if (logger.isWarnEnabled()) {
800 					logger.warn("Failed to parse Origin header value [" + origin + "]");
801 				}
802 				return false;
803 			}
804 			UriComponents requestComponents = UriComponentsBuilder.fromHttpRequest(request).build();
805 			int originPort = getPort(originComponents);
806 			int requestPort = getPort(requestComponents);
807 			return (originComponents.getHost().equals(requestComponents.getHost()) && originPort == requestPort);
808 		}
809 		else {
810 			return allowedOrigins.contains(origin);
811 		}
812 	}
813 
814 	private static int getPort(UriComponents component) {
815 		int port = component.getPort();
816 		if (port == -1) {
817 			if ("http".equals(component.getScheme())) {
818 				port = 80;
819 			}
820 			else if ("https".equals(component.getScheme())) {
821 				port = 443;
822 			}
823 		}
824 		return port;
825 	}
