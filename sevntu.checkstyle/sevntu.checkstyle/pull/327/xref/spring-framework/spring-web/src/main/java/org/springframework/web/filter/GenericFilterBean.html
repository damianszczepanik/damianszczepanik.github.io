51  /**
52   * Simple base implementation of {@link javax.servlet.Filter} which treats
53   * its config parameters ({@code init-param} entries within the
54   * {@code filter} tag in {@code web.xml}) as bean properties.
55   *
56   * <p>A handy superclass for any type of filter. Type conversion of config
57   * parameters is automatic, with the corresponding setter method getting
58   * invoked with the converted value. It is also possible for subclasses to
59   * specify required properties. Parameters without matching bean property
60   * setter will simply be ignored.
61   *
62   * <p>This filter leaves actual filtering to subclasses, which have to
63   * implement the {@link javax.servlet.Filter#doFilter} method.
64   *
65   * <p>This generic filter base class has no dependency on the Spring
66   * {@link org.springframework.context.ApplicationContext} concept.
67   * Filters usually don't load their own context but rather access service
68   * beans from the Spring root application context, accessible via the
69   * filter's {@link #getServletContext() ServletContext} (see
70   * {@link org.springframework.web.context.support.WebApplicationContextUtils}).
71   *
72   * @author Juergen Hoeller
73   * @since 06.12.2003
74   * @see #addRequiredProperty
75   * @see #initFilterBean
76   * @see #doFilter
77   */
78  public abstract class GenericFilterBean implements
79  		Filter, BeanNameAware, EnvironmentAware, ServletContextAware, InitializingBean, DisposableBean {
80  
81  	/** Logger available to subclasses */
82  	protected final Log logger = LogFactory.getLog(getClass());
83  
84  	/**
85  	 * Set of required properties (Strings) that must be supplied as
86  	 * config parameters to this filter.
87  	 */
88  	private final Set<String> requiredProperties = new HashSet<String>();
89  
90  	private FilterConfig filterConfig;
91  
92  	private String beanName;
93  
94  	private Environment environment = new StandardServletEnvironment();
95  
96  	private ServletContext servletContext;

99  	/**
100 	 * Stores the bean name as defined in the Spring bean factory.
101 	 * <p>Only relevant in case of initialization as bean, to have a name as
102 	 * fallback to the filter name usually provided by a FilterConfig instance.
103 	 * @see org.springframework.beans.factory.BeanNameAware
104 	 * @see #getFilterName()
105 	 */
106 	@Override
107 	public final void setBeanName(String beanName) {
108 		this.beanName = beanName;
109 	}
110 
111 	/**
112 	 * {@inheritDoc}
113 	 * <p>Any environment set here overrides the {@link StandardServletEnvironment}
114 	 * provided by default.
115 	 * <p>This {@code Environment} object is used only for resolving placeholders in
116 	 * resource paths passed into init-parameters for this filter. If no init-params are
117 	 * used, this {@code Environment} can be essentially ignored.
118 	 * @see #init(FilterConfig)
119 	 */
120 	@Override
121 	public void setEnvironment(Environment environment) {
122 		this.environment = environment;
123 	}
124 
125 	/**
126 	 * Stores the ServletContext that the bean factory runs in.
127 	 * <p>Only relevant in case of initialization as bean, to have a ServletContext
128 	 * as fallback to the context usually provided by a FilterConfig instance.
129 	 * @see org.springframework.web.context.ServletContextAware
130 	 * @see #getServletContext()
131 	 */
132 	@Override
133 	public final void setServletContext(ServletContext servletContext) {
134 		this.servletContext = servletContext;
135 	}
136 
137 	/**
138 	 * Calls the {@code initFilterBean()} method that might
139 	 * contain custom initialization of a subclass.
140 	 * <p>Only relevant in case of initialization as bean, where the
141 	 * standard {@code init(FilterConfig)} method won't be called.
142 	 * @see #initFilterBean()
143 	 * @see #init(javax.servlet.FilterConfig)
144 	 */
145 	@Override
146 	public void afterPropertiesSet() throws ServletException {
147 		initFilterBean();
148 	}
149 
150 
151 	/**
152 	 * Subclasses can invoke this method to specify that this property
153 	 * (which must match a JavaBean property they expose) is mandatory,
154 	 * and must be supplied as a config parameter. This should be called
155 	 * from the constructor of a subclass.
156 	 * <p>This method is only relevant in case of traditional initialization
157 	 * driven by a FilterConfig instance.
158 	 * @param property name of the required property
159 	 */
160 	protected final void addRequiredProperty(String property) {
161 		this.requiredProperties.add(property);
162 	}
163 
164 	/**
165 	 * Standard way of initializing this filter.
166 	 * Map config parameters onto bean properties of this filter, and
167 	 * invoke subclass initialization.
168 	 * @param filterConfig the configuration for this filter
169 	 * @throws ServletException if bean properties are invalid (or required
170 	 * properties are missing), or if subclass initialization fails.
171 	 * @see #initFilterBean
172 	 */
173 	@Override
174 	public final void init(FilterConfig filterConfig) throws ServletException {
175 		Assert.notNull(filterConfig, "FilterConfig must not be null");
176 		if (logger.isDebugEnabled()) {
177 			logger.debug("Initializing filter '" + filterConfig.getFilterName() + "'");
178 		}
179 
180 		this.filterConfig = filterConfig;
181 
182 		// Set bean properties from init parameters.
183 		try {
184 			PropertyValues pvs = new FilterConfigPropertyValues(filterConfig, this.requiredProperties);
185 			BeanWrapper bw = PropertyAccessorFactory.forBeanPropertyAccess(this);
186 			ResourceLoader resourceLoader = new ServletContextResourceLoader(filterConfig.getServletContext());
187 			bw.registerCustomEditor(Resource.class, new ResourceEditor(resourceLoader, this.environment));
188 			initBeanWrapper(bw);
189 			bw.setPropertyValues(pvs, true);
190 		}
191 		catch (BeansException ex) {
192 			String msg = "Failed to set bean properties on filter '" +
193 				filterConfig.getFilterName() + "': " + ex.getMessage();
194 			logger.error(msg, ex);
195 			throw new NestedServletException(msg, ex);
196 		}
197 
198 		// Let subclasses do whatever initialization they like.
199 		initFilterBean();
200 
201 		if (logger.isDebugEnabled()) {
202 			logger.debug("Filter '" + filterConfig.getFilterName() + "' configured successfully");
203 		}
204 	}
205 
206 	/**
207 	 * Initialize the BeanWrapper for this GenericFilterBean,
208 	 * possibly with custom editors.
209 	 * <p>This default implementation is empty.
210 	 * @param bw the BeanWrapper to initialize
211 	 * @throws BeansException if thrown by BeanWrapper methods
212 	 * @see org.springframework.beans.BeanWrapper#registerCustomEditor
213 	 */
214 	protected void initBeanWrapper(BeanWrapper bw) throws BeansException {
215 	}
216 
217 
218 	/**
219 	 * Make the FilterConfig of this filter available, if any.
220 	 * Analogous to GenericServlet's {@code getServletConfig()}.
221 	 * <p>Public to resemble the {@code getFilterConfig()} method
222 	 * of the Servlet Filter version that shipped with WebLogic 6.1.
223 	 * @return the FilterConfig instance, or {@code null} if none available
224 	 * @see javax.servlet.GenericServlet#getServletConfig()
225 	 */
226 	public final FilterConfig getFilterConfig() {
227 		return this.filterConfig;
228 	}
229 
230 	/**
231 	 * Make the name of this filter available to subclasses.
232 	 * Analogous to GenericServlet's {@code getServletName()}.
233 	 * <p>Takes the FilterConfig's filter name by default.
234 	 * If initialized as bean in a Spring application context,
235 	 * it falls back to the bean name as defined in the bean factory.
236 	 * @return the filter name, or {@code null} if none available
237 	 * @see javax.servlet.GenericServlet#getServletName()
238 	 * @see javax.servlet.FilterConfig#getFilterName()
239 	 * @see #setBeanName
240 	 */
241 	protected final String getFilterName() {
242 		return (this.filterConfig != null ? this.filterConfig.getFilterName() : this.beanName);
243 	}
244 
245 	/**
246 	 * Make the ServletContext of this filter available to subclasses.
247 	 * Analogous to GenericServlet's {@code getServletContext()}.
248 	 * <p>Takes the FilterConfig's ServletContext by default.
249 	 * If initialized as bean in a Spring application context,
250 	 * it falls back to the ServletContext that the bean factory runs in.
251 	 * @return the ServletContext instance, or {@code null} if none available
252 	 * @see javax.servlet.GenericServlet#getServletContext()
253 	 * @see javax.servlet.FilterConfig#getServletContext()
254 	 * @see #setServletContext
255 	 */
256 	protected final ServletContext getServletContext() {
257 		return (this.filterConfig != null ? this.filterConfig.getServletContext() : this.servletContext);
258 	}

261 	/**
262 	 * Subclasses may override this to perform custom initialization.
263 	 * All bean properties of this filter will have been set before this
264 	 * method is invoked.
265 	 * <p>Note: This method will be called from standard filter initialization
266 	 * as well as filter bean initialization in a Spring application context.
267 	 * Filter name and ServletContext will be available in both cases.
268 	 * <p>This default implementation is empty.
269 	 * @throws ServletException if subclass initialization fails
270 	 * @see #getFilterName()
271 	 * @see #getServletContext()
272 	 */
273 	protected void initFilterBean() throws ServletException {
274 	}

277 	 * Subclasses may override this to perform custom filter shutdown.
278 	 * <p>Note: This method will be called from standard filter destruction
279 	 * as well as filter bean destruction in a Spring application context.
280 	 * <p>This default implementation is empty.
281 	 */
282 	@Override
283 	public void destroy() {
284 	}

287 	/**
288 	 * PropertyValues implementation created from FilterConfig init parameters.
289 	 */
290 	@SuppressWarnings("serial")
291 	private static class FilterConfigPropertyValues extends MutablePropertyValues {
292 
293 		/**
294 		 * Create new FilterConfigPropertyValues.
295 		 * @param config FilterConfig we'll use to take PropertyValues from
296 		 * @param requiredProperties set of property names we need, where
297 		 * we can't accept default values
298 		 * @throws ServletException if any required properties are missing
299 		 */
300 		public FilterConfigPropertyValues(FilterConfig config, Set<String> requiredProperties)
301 			throws ServletException {
302 
303 			Set<String> missingProps = (requiredProperties != null && !requiredProperties.isEmpty()) ?
304 					new HashSet<String>(requiredProperties) : null;
305 
306 			Enumeration<?> en = config.getInitParameterNames();
307 			while (en.hasMoreElements()) {
308 				String property = (String) en.nextElement();
309 				Object value = config.getInitParameter(property);
310 				addPropertyValue(new PropertyValue(property, value));
311 				if (missingProps != null) {
312 					missingProps.remove(property);
313 				}
314 			}
315 
316 			// Fail if we are still missing properties.
317 			if (missingProps != null && missingProps.size() > 0) {
318 				throw new ServletException(
319 					"Initialization from FilterConfig for filter '" + config.getFilterName() +
320 					"' failed; the following required properties were missing: " +
321 					StringUtils.collectionToDelimitedString(missingProps, ", "));
322 			}
323 		}
324 	}
