34  /**

39  public class HttpPutFormContentFilterTests {
40  
41  	private HttpPutFormContentFilter filter;
42  
43  	private MockHttpServletRequest request;
44  
45  	private MockHttpServletResponse response;
46  
47  	private MockFilterChain filterChain;
48  
49  	@Before
50  	public void setup() {
51  		filter = new HttpPutFormContentFilter();
52  		request = new MockHttpServletRequest("PUT", "/");
53  		request.addHeader("Content-Type", "application/x-www-form-urlencoded; charset=ISO-8859-1");
54  		request.setContentType("application/x-www-form-urlencoded; charset=ISO-8859-1");
55  		response = new MockHttpServletResponse();
56  		filterChain = new MockFilterChain();
57  	}
58  
59  	@Test
60  	public void wrapPutAndPatchOnly() throws Exception {
61  		request.setContent("".getBytes("ISO-8859-1"));
62  		for (HttpMethod method : HttpMethod.values()) {
63  			request.setMethod(method.name());
64  			filterChain = new MockFilterChain();
65  			filter.doFilter(request, response, filterChain);
66  			if (method.equals(HttpMethod.PUT) || method.equals(HttpMethod.PATCH)) {
67  				assertNotSame("Should wrap HTTP method " + method, request, filterChain.getRequest());
68  			}
69  			else {
70  				assertSame("Should not wrap for HTTP method " + method, request, filterChain.getRequest());
71  			}
72  		}
73  	}
74  
75  	@Test
76  	public void wrapFormEncodedOnly() throws Exception {
77  		request.setContent("".getBytes("ISO-8859-1"));
78  		String[] contentTypes = new String[] {"text/plain", "multipart/form-data"};
79  		for (String contentType : contentTypes) {
80  			request.setContentType(contentType);
81  			filterChain = new MockFilterChain();
82  			filter.doFilter(request, response, filterChain);
83  			assertSame("Should not wrap for content type " + contentType, request, filterChain.getRequest());
84  		}
85  	}
86  
87  	@Test
88  	public void invalidMediaType() throws Exception {
89  		request.setContent("".getBytes("ISO-8859-1"));
90  		request.setContentType("foo");
91  		filterChain = new MockFilterChain();
92  		filter.doFilter(request, response, filterChain);
93  		assertSame(request, filterChain.getRequest());
94  	}

97  	public void getParameter() throws Exception {
98  		request.setContent("name=value".getBytes("ISO-8859-1"));
99  		filter.doFilter(request, response, filterChain);
100 
101 		assertEquals("value", filterChain.getRequest().getParameter("name"));
102 	}

105 	public void getParameterFromQueryString() throws Exception {
106 		request.addParameter("name", "value1");
107 		request.setContent("name=value2".getBytes("ISO-8859-1"));
108 		filter.doFilter(request, response, filterChain);

191 		request.addParameter("name", "value1");
192 		request.addParameter("name", "value2");
193 		request.setContent("name=value3&name4=value4".getBytes("ISO-8859-1"));
194 
195 		filter.doFilter(request, response, filterChain);
196 		Map<String, String[]> parameters = filterChain.getRequest().getParameterMap();
