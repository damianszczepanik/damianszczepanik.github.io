140 	 * Return the mapping lookup path for the given request, within the current
141 	 * servlet mapping if applicable, else within the web application.
142 	 * <p>Detects include request URL if called within a RequestDispatcher include.
143 	 * @param request current HTTP request
144 	 * @return the lookup path
145 	 * @see #getPathWithinApplication
146 	 * @see #getPathWithinServletMapping
147 	 */
148 	public String getLookupPathForRequest(HttpServletRequest request) {
149 		// Always use full path within current servlet context?
150 		if (this.alwaysUseFullPath) {
151 			return getPathWithinApplication(request);
152 		}
153 		// Else, use path within current servlet mapping if applicable
154 		String rest = getPathWithinServletMapping(request);
155 		if (!"".equals(rest)) {
156 			return rest;
157 		}
158 		else {
159 			return getPathWithinApplication(request);
160 		}
161 	}

163 	/**
164 	 * Return the path within the servlet mapping for the given request,
165 	 * i.e. the part of the request's URL beyond the part that called the servlet,
166 	 * or "" if the whole URL has been used to identify the servlet.
167 	 * <p>Detects include request URL if called within a RequestDispatcher include.
168 	 * <p>E.g.: servlet mapping = "/*"; request URI = "/test/a" -> "/test/a".
169 	 * <p>E.g.: servlet mapping = "/"; request URI = "/test/a" -> "/test/a".
170 	 * <p>E.g.: servlet mapping = "/test/*"; request URI = "/test/a" -> "/a".
171 	 * <p>E.g.: servlet mapping = "/test"; request URI = "/test" -> "".
172 	 * <p>E.g.: servlet mapping = "/*.test"; request URI = "/a.test" -> "".
173 	 * @param request current HTTP request
174 	 * @return the path within the servlet mapping, or ""
175 	 */
176 	public String getPathWithinServletMapping(HttpServletRequest request) {
177 		String pathWithinApp = getPathWithinApplication(request);
178 		String servletPath = getServletPath(request);
179 		String sanitizedPathWithinApp = getSanitizedPath(pathWithinApp);
180 		String path;
181 
182 		// if the app container sanitized the servletPath, check against the sanitized version
183 		if(servletPath.indexOf(sanitizedPathWithinApp) != -1) {
184 			path = getRemainingPath(sanitizedPathWithinApp, servletPath, false);
185 		}
186 		else {
187 			path = getRemainingPath(pathWithinApp, servletPath, false);
188 		}
189 
190 		if (path != null) {
191 			// Normal case: URI contains servlet path.
192 			return path;
193 		}
194 		else {
195 			// Special case: URI is different from servlet path.
196 			String pathInfo = request.getPathInfo();
197 			if (pathInfo != null) {
198 				// Use path info if available. Indicates index page within a servlet mapping?
199 				// e.g. with index page: URI="/", servletPath="/index.html"
200 				return pathInfo;
201 			}
202 			if (!this.urlDecode) {
203 				// No path info... (not mapped by prefix, nor by extension, nor "/*")
204 				// For the default servlet mapping (i.e. "/"), urlDecode=false can
205 				// cause issues since getServletPath() returns a decoded path.
206 				// If decoding pathWithinApp yields a match just use pathWithinApp.
207 				path = getRemainingPath(decodeInternal(request, pathWithinApp), servletPath, false);
208 				if (path != null) {
209 					return pathWithinApp;
210 				}
211 			}
212 			// Otherwise, use the full servlet path.
213 			return servletPath;
214 		}
215 	}

217 	/**
218 	 * Return the path within the web application for the given request.
219 	 * <p>Detects include request URL if called within a RequestDispatcher include.
220 	 * @param request current HTTP request
221 	 * @return the path within the web application
222 	 */
223 	public String getPathWithinApplication(HttpServletRequest request) {
224 		String contextPath = getContextPath(request);
225 		String requestUri = getRequestUri(request);
226 		String path = getRemainingPath(requestUri, contextPath, true);
227 		if (path != null) {
228 			// Normal case: URI contains context path.
229 			return (StringUtils.hasText(path) ? path : "/");
230 		}
231 		else {
232 			return requestUri;
233 		}
234 	}

236 	/**
237 	 * Match the given "mapping" to the start of the "requestUri" and if there
238 	 * is a match return the extra part. This method is needed because the
239 	 * context path and the servlet path returned by the HttpServletRequest are
240 	 * stripped of semicolon content unlike the requesUri.
241 	 */
242 	private String getRemainingPath(String requestUri, String mapping, boolean ignoreCase) {
243 		int index1 = 0;
244 		int index2 = 0;
245 		for (; (index1 < requestUri.length()) && (index2 < mapping.length()); index1++, index2++) {
246 			char c1 = requestUri.charAt(index1);
247 			char c2 = mapping.charAt(index2);
248 			if (c1 == ';') {
249 				index1 = requestUri.indexOf('/', index1);
250 				if (index1 == -1) {
251 					return null;
252 				}
253 				c1 = requestUri.charAt(index1);
254 			}
255 			if (c1 == c2) {
256 				continue;
257 			}
258 			else if (ignoreCase && (Character.toLowerCase(c1) == Character.toLowerCase(c2))) {
259 				continue;
260 			}
261 			return null;
262 		}
263 		if (index2 != mapping.length()) {
264 			return null;
265 		}
266 		else if (index1 == requestUri.length()) {
267 			return "";
268 		}
269 		else if (requestUri.charAt(index1) == ';') {
270 			index1 = requestUri.indexOf('/', index1);
271 		}
272 		return (index1 != -1 ? requestUri.substring(index1) : "");
273 	}

275 	/**
276 	 * Sanitize the given path with the following rules:
277 	 * <ul>
278 	 *     <li>replace all "//" by "/"</li>
279 	 * </ul>
280 	 */
281 	private String getSanitizedPath(final String path) {
282 		String sanitized = path;
283 		while (true) {
284 			int index = sanitized.indexOf("//");
285 			if (index < 0) {
286 				break;
287 			}
288 			else {
289 				sanitized = sanitized.substring(0, index) + sanitized.substring(index + 1);
290 			}
291 		}
292 		return sanitized;
293 	}

296 	 * Return the request URI for the given request, detecting an include request
297 	 * URL if called within a RequestDispatcher include.
298 	 * <p>As the value returned by {@code request.getRequestURI()} is <i>not</i>
299 	 * decoded by the servlet container, this method will decode it.
300 	 * <p>The URI that the web container resolves <i>should</i> be correct, but some
301 	 * containers like JBoss/Jetty incorrectly include ";" strings like ";jsessionid"
302 	 * in the URI. This method cuts off such incorrect appendices.
303 	 * @param request current HTTP request
304 	 * @return the request URI
305 	 */
306 	public String getRequestUri(HttpServletRequest request) {
307 		String uri = (String) request.getAttribute(WebUtils.INCLUDE_REQUEST_URI_ATTRIBUTE);
308 		if (uri == null) {
309 			uri = request.getRequestURI();
310 		}
311 		return decodeAndCleanUriString(request, uri);
312 	}

315 	 * Return the context path for the given request, detecting an include request
316 	 * URL if called within a RequestDispatcher include.
317 	 * <p>As the value returned by {@code request.getContextPath()} is <i>not</i>
318 	 * decoded by the servlet container, this method will decode it.
319 	 * @param request current HTTP request
320 	 * @return the context path
321 	 */
322 	public String getContextPath(HttpServletRequest request) {
323 		String contextPath = (String) request.getAttribute(WebUtils.INCLUDE_CONTEXT_PATH_ATTRIBUTE);
324 		if (contextPath == null) {
325 			contextPath = request.getContextPath();
326 		}
327 		if ("/".equals(contextPath)) {
328 			// Invalid case, but happens for includes on Jetty: silently adapt it.
329 			contextPath = "";
330 		}
331 		return decodeRequestString(request, contextPath);
332 	}

334 	/**
335 	 * Return the servlet path for the given request, regarding an include request
336 	 * URL if called within a RequestDispatcher include.
337 	 * <p>As the value returned by {@code request.getServletPath()} is already
338 	 * decoded by the servlet container, this method will not attempt to decode it.
339 	 * @param request current HTTP request
340 	 * @return the servlet path
341 	 */
342 	public String getServletPath(HttpServletRequest request) {
343 		String servletPath = (String) request.getAttribute(WebUtils.INCLUDE_SERVLET_PATH_ATTRIBUTE);
344 		if (servletPath == null) {
345 			servletPath = request.getServletPath();
346 		}
347 		if (servletPath.length() > 1 && servletPath.endsWith("/") && shouldRemoveTrailingServletPathSlash(request)) {
348 			// On WebSphere, in non-compliant mode, for a "/foo/" case that would be "/foo"
349 			// on all other servlet containers: removing trailing slash, proceeding with
350 			// that remaining slash as final lookup path...
351 			servletPath = servletPath.substring(0, servletPath.length() - 1);
352 		}
353 		return servletPath;
354 	}

422 		uri = removeSemicolonContent(uri);

428 	 * Decode the given source string with a URLDecoder. The encoding will be taken
429 	 * from the request, falling back to the default "ISO-8859-1".
430 	 * <p>The default implementation uses {@code URLDecoder.decode(input, enc)}.
431 	 * @param request current HTTP request
432 	 * @param source the String to decode
433 	 * @return the decoded String
434 	 * @see WebUtils#DEFAULT_CHARACTER_ENCODING
435 	 * @see javax.servlet.ServletRequest#getCharacterEncoding
436 	 * @see java.net.URLDecoder#decode(String, String)
437 	 * @see java.net.URLDecoder#decode(String)
438 	 */
439 	public String decodeRequestString(HttpServletRequest request, String source) {
440 		if (this.urlDecode) {
441 			return decodeInternal(request, source);
442 		}
443 		return source;
444 	}

446 	@SuppressWarnings("deprecation")
447 	private String decodeInternal(HttpServletRequest request, String source) {
448 		String enc = determineEncoding(request);
449 		try {
450 			return UriUtils.decode(source, enc);
451 		}
452 		catch (UnsupportedEncodingException ex) {
453 			if (logger.isWarnEnabled()) {
454 				logger.warn("Could not decode request string [" + source + "] with encoding '" + enc +
455 						"': falling back to platform default encoding; exception message: " + ex.getMessage());
456 			}
457 			return URLDecoder.decode(source);
458 		}
459 	}

479 	/**
480 	 * Remove ";" (semicolon) content from the given request URI if the
481 	 * {@linkplain #setRemoveSemicolonContent(boolean) removeSemicolonContent}
482 	 * property is set to "true". Note that "jssessionid" is always removed.
483 	 * @param requestUri the request URI string to remove ";" content from
484 	 * @return the updated URI string
485 	 */
486 	public String removeSemicolonContent(String requestUri) {
487 		return this.removeSemicolonContent ?
488 				removeSemicolonContentInternal(requestUri) : removeJsessionid(requestUri);
489 	}
490 
491 	private String removeSemicolonContentInternal(String requestUri) {
492 		int semicolonIndex = requestUri.indexOf(';');
493 		while (semicolonIndex != -1) {
494 			int slashIndex = requestUri.indexOf('/', semicolonIndex);
495 			String start = requestUri.substring(0, semicolonIndex);
496 			requestUri = (slashIndex != -1) ? start + requestUri.substring(slashIndex) : start;
497 			semicolonIndex = requestUri.indexOf(';', semicolonIndex);
498 		}
499 		return requestUri;
500 	}
501 
502 	private String removeJsessionid(String requestUri) {
503 		int startIndex = requestUri.toLowerCase().indexOf(";jsessionid=");
504 		if (startIndex != -1) {
505 			int endIndex = requestUri.indexOf(';', startIndex + 12);
506 			String start = requestUri.substring(0, startIndex);
507 			requestUri = (endIndex != -1) ? start + requestUri.substring(endIndex) : start;
508 		}
509 		return requestUri;
510 	}

512 	/**
513 	 * Decode the given URI path variables via
514 	 * {@link #decodeRequestString(HttpServletRequest, String)} unless
515 	 * {@link #setUrlDecode(boolean)} is set to {@code true} in which case it is
516 	 * assumed the URL path from which the variables were extracted is already
517 	 * decoded through a call to
518 	 * {@link #getLookupPathForRequest(HttpServletRequest)}.
519 	 * @param request current HTTP request
520 	 * @param vars URI variables extracted from the URL path
521 	 * @return the same Map or a new Map instance
522 	 */
523 	public Map<String, String> decodePathVariables(HttpServletRequest request, Map<String, String> vars) {
524 		if (this.urlDecode) {
525 			return vars;
526 		}
527 		else {
528 			Map<String, String> decodedVars = new LinkedHashMap<String, String>(vars.size());
529 			for (Entry<String, String> entry : vars.entrySet()) {
530 				decodedVars.put(entry.getKey(), decodeInternal(request, entry.getValue()));
531 			}
532 			return decodedVars;
533 		}
534 	}

537 	 * Decode the given matrix variables via
538 	 * {@link #decodeRequestString(HttpServletRequest, String)} unless
539 	 * {@link #setUrlDecode(boolean)} is set to {@code true} in which case it is
540 	 * assumed the URL path from which the variables were extracted is already
541 	 * decoded through a call to
542 	 * {@link #getLookupPathForRequest(HttpServletRequest)}.
543 	 * @param request current HTTP request
544 	 * @param vars URI variables extracted from the URL path
545 	 * @return the same Map or a new Map instance
546 	 */
547 	public MultiValueMap<String, String> decodeMatrixVariables(HttpServletRequest request, MultiValueMap<String, String> vars) {
548 		if (this.urlDecode) {
549 			return vars;
550 		}
551 		else {
552 			MultiValueMap<String, String> decodedVars = new LinkedMultiValueMap	<String, String>(vars.size());
553 			for (String key : vars.keySet()) {
554 				for (String value : vars.get(key)) {
555 					decodedVars.add(key, decodeInternal(request, value));
556 				}
557 			}
558 			return decodedVars;
559 		}

562 	private boolean shouldRemoveTrailingServletPathSlash(HttpServletRequest request) {
563 		if (request.getAttribute(WEBSPHERE_URI_ATTRIBUTE) == null) {
564 			// Regular servlet container: behaves as expected in any case,
565 			// so the trailing slash is the result of a "/" url-pattern mapping.
566 			// Don't remove that slash.
567 			return false;
568 		}
569 		if (websphereComplianceFlag == null) {
570 			ClassLoader classLoader = UrlPathHelper.class.getClassLoader();
571 			String className = "com.ibm.ws.webcontainer.WebContainer";
572 			String methodName = "getWebContainerProperties";
573 			String propName = "com.ibm.ws.webcontainer.removetrailingservletpathslash";
574 			boolean flag = false;
575 			try {
576 				Class<?> cl = classLoader.loadClass(className);
577 				Properties prop = (Properties) cl.getMethod(methodName).invoke(null);
578 				flag = Boolean.parseBoolean(prop.getProperty(propName));
579 			}
580 			catch (Throwable ex) {
581 				if (logger.isDebugEnabled()) {
582 					logger.debug("Could not introspect WebSphere web container properties: " + ex);
583 				}
584 			}
585 			websphereComplianceFlag = flag;
586 		}
587 		// Don't bother if WebSphere is configured to be fully Servlet compliant.
588 		// However, if it is not compliant, do remove the improper trailing slash!
589 		return !websphereComplianceFlag;
590 	}
