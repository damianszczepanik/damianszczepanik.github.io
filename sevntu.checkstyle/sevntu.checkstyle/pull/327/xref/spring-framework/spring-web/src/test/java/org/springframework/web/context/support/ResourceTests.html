7   import org.springframework.core.io.Resource;
8   import org.springframework.mock.web.test.MockServletContext;

10  import static org.junit.Assert.*;
11  
12  /**
13   * @author Chris Beams
14   * @see org.springframework.core.io.ResourceTests
15   */
16  public class ResourceTests {
17  
18  	@Test
19  	public void testServletContextResource() throws IOException {
20  		MockServletContext sc = new MockServletContext();
21  		Resource resource = new ServletContextResource(sc, "org/springframework/core/io/Resource.class");
22  		doTestResource(resource);
23  		assertEquals(resource, new ServletContextResource(sc, "org/springframework/core/../core/io/./Resource.class"));
24  	}
25  
26  	@Test
27  	public void testServletContextResourceWithRelativePath() throws IOException {
28  		MockServletContext sc = new MockServletContext();
29  		Resource resource = new ServletContextResource(sc, "dir/");
30  		Resource relative = resource.createRelative("subdir");
31  		assertEquals(new ServletContextResource(sc, "dir/subdir"), relative);
32  	}
33  
34  	private void doTestResource(Resource resource) throws IOException {
35  		assertEquals("Resource.class", resource.getFilename());
36  		assertTrue(resource.getURL().getFile().endsWith("Resource.class"));
37  
38  		Resource relative1 = resource.createRelative("ClassPathResource.class");
39  		assertEquals("ClassPathResource.class", relative1.getFilename());
40  		assertTrue(relative1.getURL().getFile().endsWith("ClassPathResource.class"));
41  		assertTrue(relative1.exists());
42  
43  		Resource relative2 = resource.createRelative("support/ResourcePatternResolver.class");
44  		assertEquals("ResourcePatternResolver.class", relative2.getFilename());
45  		assertTrue(relative2.getURL().getFile().endsWith("ResourcePatternResolver.class"));
46  		assertTrue(relative2.exists());
47  	}
48  }
