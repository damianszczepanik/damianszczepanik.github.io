17  package org.springframework.web.method.annotation;
18  
19  import java.util.Map;
20  
21  import org.springframework.beans.factory.config.ConfigurableBeanFactory;
22  import org.springframework.core.MethodParameter;
23  import org.springframework.web.bind.ServletRequestBindingException;
24  import org.springframework.web.bind.WebDataBinder;
25  import org.springframework.web.bind.annotation.RequestHeader;
26  import org.springframework.web.context.request.NativeWebRequest;
27  
28  /**
29   * Resolves method arguments annotated with {@code @RequestHeader} except for
30   * {@link Map} arguments. See {@link RequestHeaderMapMethodArgumentResolver} for
31   * details on {@link Map} arguments annotated with {@code @RequestHeader}.
32   *
33   * <p>An {@code @RequestHeader} is a named value resolved from a request header.
34   * It has a required flag and a default value to fall back on when the request
35   * header does not exist.
36   *
37   * <p>A {@link WebDataBinder} is invoked to apply type conversion to resolved
38   * request header values that don't yet match the method parameter type.
39   *
40   * @author Arjen Poutsma
41   * @author Rossen Stoyanchev
42   * @since 3.1
43   */
44  public class RequestHeaderMethodArgumentResolver extends AbstractNamedValueMethodArgumentResolver {
45  
46  	/**
47  	 * @param beanFactory a bean factory to use for resolving  ${...}
48  	 * placeholder and #{...} SpEL expressions in default values;
49  	 * or {@code null} if default values are not expected to have expressions
50  	 */
51  	public RequestHeaderMethodArgumentResolver(ConfigurableBeanFactory beanFactory) {
52  		super(beanFactory);
53  	}
54  
55  
56  	@Override
57  	public boolean supportsParameter(MethodParameter parameter) {
58  		return parameter.hasParameterAnnotation(RequestHeader.class) &&
59  				!Map.class.isAssignableFrom(parameter.getParameterType());
60  	}
61  
62  	@Override
63  	protected NamedValueInfo createNamedValueInfo(MethodParameter parameter) {
64  		RequestHeader annotation = parameter.getParameterAnnotation(RequestHeader.class);
65  		return new RequestHeaderNamedValueInfo(annotation);
66  	}
67  
68  	@Override
69  	protected Object resolveName(String name, MethodParameter parameter, NativeWebRequest request) throws Exception {
70  		String[] headerValues = request.getHeaderValues(name);
71  		if (headerValues != null) {
72  			return (headerValues.length == 1 ? headerValues[0] : headerValues);
73  		}
74  		else {
75  			return null;
76  		}
77  	}
78  
79  	@Override
80  	protected void handleMissingValue(String name, MethodParameter parameter) throws ServletRequestBindingException {
81  		throw new ServletRequestBindingException("Missing request header '" + name +
82  				"' for method parameter of type " + parameter.getParameterType().getSimpleName());
83  	}
84  
85  
86  	private static class RequestHeaderNamedValueInfo extends NamedValueInfo {
87  
88  		private RequestHeaderNamedValueInfo(RequestHeader annotation) {
89  			super(annotation.value(), annotation.required(), annotation.defaultValue());
90  		}
91  	}
92  
93  }
