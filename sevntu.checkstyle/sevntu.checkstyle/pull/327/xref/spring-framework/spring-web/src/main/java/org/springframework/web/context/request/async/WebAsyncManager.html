36  /**
37   * The central class for managing asynchronous request processing, mainly intended
38   * as an SPI and not typically used directly by application classes.
39   *
40   * <p>An async scenario starts with request processing as usual in a thread (T1).
41   * Concurrent request handling can be initiated by calling
42   * {@link #startCallableProcessing(Callable, Object...) startCallableProcessing} or
43   * {@link #startDeferredResultProcessing(DeferredResult, Object...) startDeferredResultProcessing},
44   * both of which produce a result in a separate thread (T2). The result is saved
45   * and the request dispatched to the container, to resume processing with the saved
46   * result in a third thread (T3). Within the dispatched thread (T3), the saved
47   * result can be accessed via {@link #getConcurrentResult()} or its presence
48   * detected via {@link #hasConcurrentResult()}.
49   *
50   * @author Rossen Stoyanchev
51   * @since 3.2
52   * @see org.springframework.web.context.request.AsyncWebRequestInterceptor
53   * @see org.springframework.web.servlet.AsyncHandlerInterceptor
54   * @see org.springframework.web.filter.OncePerRequestFilter#shouldNotFilterAsyncDispatch
55   * @see org.springframework.web.filter.OncePerRequestFilter#isAsyncDispatch
56   */
57  public final class WebAsyncManager {

96  	/**
97  	 * Configure the {@link AsyncWebRequest} to use. This property may be set
98  	 * more than once during a single request to accurately reflect the current
99  	 * state of the request (e.g. following a forward, request/response
100 	 * wrapping, etc). However, it should not be set while concurrent handling
101 	 * is in progress, i.e. while {@link #isConcurrentHandlingStarted()} is
102 	 * {@code true}.
103 	 * @param asyncWebRequest the web request to use
104 	 */
105 	public void setAsyncWebRequest(final AsyncWebRequest asyncWebRequest) {
106 		Assert.notNull(asyncWebRequest, "AsyncWebRequest must not be null");
107 		Assert.state(!isConcurrentHandlingStarted(), "Can't set AsyncWebRequest with concurrent handling in progress");
108 		this.asyncWebRequest = asyncWebRequest;
109 		this.asyncWebRequest.addCompletionHandler(new Runnable() {
110 			@Override
111 			public void run() {
112 				asyncWebRequest.removeAttribute(WebAsyncUtils.WEB_ASYNC_MANAGER_ATTRIBUTE, RequestAttributes.SCOPE_REQUEST);
113 			}
114 		});
115 	}

118 	 * Configure an AsyncTaskExecutor for use with concurrent processing via
119 	 * {@link #startCallableProcessing(Callable, Object...)}.
120 	 * <p>By default a {@link SimpleAsyncTaskExecutor} instance is used.
121 	 */

123 		this.taskExecutor = taskExecutor;
124 	}

126 	/**
127 	 * Whether the selected handler for the current request chose to handle the
128 	 * request asynchronously. A return value of "true" indicates concurrent
129 	 * handling is under way and the response will remain open. A return value
130 	 * of "false" means concurrent handling was either not started or possibly
131 	 * that it has completed and the request was dispatched for further
132 	 * processing of the concurrent result.
133 	 */
134 	public boolean isConcurrentHandlingStarted() {
135 		return ((this.asyncWebRequest != null) && this.asyncWebRequest.isAsyncStarted());
136 	}
137 
138 	/**
139 	 * Whether a result value exists as a result of concurrent handling.
140 	 */
141 	public boolean hasConcurrentResult() {
142 		return (this.concurrentResult != RESULT_NONE);
143 	}

193 	/**
194 	 * Register a {@link CallableProcessingInterceptor} without a key.
195 	 * The key is derived from the class name and hashcode.
196 	 * @param interceptors one or more interceptors to register
197 	 */
198 	public void registerCallableInterceptors(CallableProcessingInterceptor... interceptors) {
199 		Assert.notNull(interceptors, "A CallableProcessingInterceptor is required");
200 		for (CallableProcessingInterceptor interceptor : interceptors) {
201 			String key = interceptor.getClass().getName() + ":" + interceptor.hashCode();
202 			this.callableInterceptors.put(key, interceptor);
203 		}
204 	}

239 	/**
240 	 * Start concurrent request processing and execute the given task with an
241 	 * {@link #setTaskExecutor(AsyncTaskExecutor) AsyncTaskExecutor}. The result
242 	 * from the task execution is saved and the request dispatched in order to
243 	 * resume processing of that result. If the task raises an Exception then
244 	 * the saved result will be the raised Exception.
245 	 * @param callable a unit of work to be executed asynchronously
246 	 * @param processingContext additional context to save that can be accessed
247 	 * via {@link #getConcurrentResultContext()}
248 	 * @throws Exception if concurrent processing failed to start
249 	 * @see #getConcurrentResult()
250 	 * @see #getConcurrentResultContext()
251 	 */
252 	@SuppressWarnings({"unchecked", "rawtypes"})
253 	public void startCallableProcessing(Callable<?> callable, Object... processingContext) throws Exception {
254 		Assert.notNull(callable, "Callable must not be null");
255 		startCallableProcessing(new WebAsyncTask(callable), processingContext);
256 	}
257 
258 	/**
259 	 * Use the given {@link WebAsyncTask} to configure the task executor as well as
260 	 * the timeout value of the {@code AsyncWebRequest} before delegating to
261 	 * {@link #startCallableProcessing(Callable, Object...)}.
262 	 * @param webAsyncTask a WebAsyncTask containing the target {@code Callable}
263 	 * @param processingContext additional context to save that can be accessed
264 	 * via {@link #getConcurrentResultContext()}
265 	 * @throws Exception if concurrent processing failed to start
266 	 */
267 	public void startCallableProcessing(final WebAsyncTask<?> webAsyncTask, Object... processingContext) throws Exception {
268 		Assert.notNull(webAsyncTask, "WebAsyncTask must not be null");
269 		Assert.state(this.asyncWebRequest != null, "AsyncWebRequest must not be null");
270 
271 		Long timeout = webAsyncTask.getTimeout();
272 		if (timeout != null) {
273 			this.asyncWebRequest.setTimeout(timeout);
274 		}
275 
276 		AsyncTaskExecutor executor = webAsyncTask.getExecutor();
277 		if (executor != null) {
278 			this.taskExecutor = executor;
279 		}
280 
281 		List<CallableProcessingInterceptor> interceptors = new ArrayList<CallableProcessingInterceptor>();
282 		interceptors.add(webAsyncTask.getInterceptor());
283 		interceptors.addAll(this.callableInterceptors.values());
284 		interceptors.add(timeoutCallableInterceptor);
285 
286 		final Callable<?> callable = webAsyncTask.getCallable();
287 		final CallableInterceptorChain interceptorChain = new CallableInterceptorChain(interceptors);
288 
289 		this.asyncWebRequest.addTimeoutHandler(new Runnable() {
290 			@Override
291 			public void run() {
292 				logger.debug("Processing timeout");
293 				Object result = interceptorChain.triggerAfterTimeout(asyncWebRequest, callable);
294 				if (result != CallableProcessingInterceptor.RESULT_NONE) {
295 					setConcurrentResultAndDispatch(result);
296 				}
297 			}
298 		});
299 
300 		this.asyncWebRequest.addCompletionHandler(new Runnable() {
301 			@Override
302 			public void run() {
303 				interceptorChain.triggerAfterCompletion(asyncWebRequest, callable);
304 			}
305 		});
306 
307 		interceptorChain.applyBeforeConcurrentHandling(this.asyncWebRequest, callable);
308 		startAsyncProcessing(processingContext);
309 
310 		this.taskExecutor.submit(new Runnable() {
311 			@Override
312 			public void run() {
313 				Object result = null;
314 				try {
315 					interceptorChain.applyPreProcess(asyncWebRequest, callable);
316 					result = callable.call();
317 				}
318 				catch (Throwable ex) {
319 					result = ex;
320 				}
321 				finally {
322 					result = interceptorChain.applyPostProcess(asyncWebRequest, callable, result);
323 				}
324 				setConcurrentResultAndDispatch(result);
325 			}
326 		});
327 	}
328 
329 	private void setConcurrentResultAndDispatch(Object result) {
330 		synchronized (WebAsyncManager.this) {
331 			if (hasConcurrentResult()) {
332 				return;
333 			}
334 			this.concurrentResult = result;
335 		}
336 
337 		if (this.asyncWebRequest.isAsyncComplete()) {
338 			logger.error("Could not complete async processing due to timeout or network error");
339 			return;
340 		}
341 
342 		if (logger.isDebugEnabled()) {
343 			logger.debug("Concurrent result value [" + this.concurrentResult +
344 					"] - dispatching request to resume processing");
345 		}
346 
347 		this.asyncWebRequest.dispatch();
348 	}

350 	/**
351 	 * Start concurrent request processing and initialize the given
352 	 * {@link DeferredResult} with a {@link DeferredResultHandler} that saves
353 	 * the result and dispatches the request to resume processing of that
354 	 * result. The {@code AsyncWebRequest} is also updated with a completion
355 	 * handler that expires the {@code DeferredResult} and a timeout handler
356 	 * assuming the {@code DeferredResult} has a default timeout result.
357 	 * @param deferredResult the DeferredResult instance to initialize
358 	 * @param processingContext additional context to save that can be accessed
359 	 * via {@link #getConcurrentResultContext()}
360 	 * @throws Exception if concurrent processing failed to start
361 	 * @see #getConcurrentResult()
362 	 * @see #getConcurrentResultContext()
363 	 */
364 	public void startDeferredResultProcessing(
365 			final DeferredResult<?> deferredResult, Object... processingContext) throws Exception {
366 
367 		Assert.notNull(deferredResult, "DeferredResult must not be null");
368 		Assert.state(this.asyncWebRequest != null, "AsyncWebRequest must not be null");
369 
370 		Long timeout = deferredResult.getTimeoutValue();
371 		if (timeout != null) {
372 			this.asyncWebRequest.setTimeout(timeout);
373 		}
374 
375 		List<DeferredResultProcessingInterceptor> interceptors = new ArrayList<DeferredResultProcessingInterceptor>();
376 		interceptors.add(deferredResult.getInterceptor());
377 		interceptors.addAll(this.deferredResultInterceptors.values());
378 		interceptors.add(timeoutDeferredResultInterceptor);
379 
380 		final DeferredResultInterceptorChain interceptorChain = new DeferredResultInterceptorChain(interceptors);
381 
382 		this.asyncWebRequest.addTimeoutHandler(new Runnable() {
383 			@Override
384 			public void run() {
385 				try {
386 					interceptorChain.triggerAfterTimeout(asyncWebRequest, deferredResult);
387 				}
388 				catch (Throwable ex) {
389 					setConcurrentResultAndDispatch(ex);
390 				}
391 			}
392 		});
393 
394 		this.asyncWebRequest.addCompletionHandler(new Runnable() {
395 			@Override
396 			public void run() {
397 				interceptorChain.triggerAfterCompletion(asyncWebRequest, deferredResult);
398 			}
399 		});
400 
401 		interceptorChain.applyBeforeConcurrentHandling(this.asyncWebRequest, deferredResult);
402 		startAsyncProcessing(processingContext);
403 
404 		try {
405 			interceptorChain.applyPreProcess(this.asyncWebRequest, deferredResult);
406 			deferredResult.setResultHandler(new DeferredResultHandler() {
407 				@Override
408 				public void handleResult(Object result) {
409 					result = interceptorChain.applyPostProcess(asyncWebRequest, deferredResult, result);
410 					setConcurrentResultAndDispatch(result);
411 				}
412 			});
413 		}
414 		catch (Throwable ex) {
415 			setConcurrentResultAndDispatch(ex);
416 		}
417 	}
418 
419 	private void startAsyncProcessing(Object[] processingContext) {
420 		clearConcurrentResult();
421 		this.concurrentResultContext = processingContext;
422 		this.asyncWebRequest.startAsync();
423 
424 		if (logger.isDebugEnabled()) {
425 			HttpServletRequest request = this.asyncWebRequest.getNativeRequest(HttpServletRequest.class);
426 			String requestUri = urlPathHelper.getRequestUri(request);
427 			logger.debug("Concurrent handling starting for " + request.getMethod() + " [" + requestUri + "]");
428 		}
429 	}
