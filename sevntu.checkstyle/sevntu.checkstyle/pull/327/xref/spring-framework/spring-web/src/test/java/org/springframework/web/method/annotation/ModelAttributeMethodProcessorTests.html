150 	private void getAttributeFromModel(String expectedAttributeName, MethodParameter param) throws Exception {
151 		Object target = new TestBean();
152 		mavContainer.addAttribute(expectedAttributeName, target);
153 
154 		WebDataBinder dataBinder = new WebRequestDataBinder(target);
155 		WebDataBinderFactory factory = mock(WebDataBinderFactory.class);
156 		given(factory.createBinder(webRequest, target, expectedAttributeName)).willReturn(dataBinder);
157 
158 		processor.resolveArgument(param, mavContainer, webRequest, factory);
159 		verify(factory).createBinder(webRequest, target, expectedAttributeName);
160 	}

208 	@Test
209 	public void resolveArgumentOrdering() throws Exception {
210 		String name = "testBean";
211 		Object testBean = new TestBean(name);
212 		mavContainer.addAttribute(name, testBean);
213 		mavContainer.addAttribute(BindingResult.MODEL_KEY_PREFIX + name, testBean);
214 
215 		Object anotherTestBean = new TestBean();
216 		mavContainer.addAttribute("anotherTestBean", anotherTestBean);
217 
218 		StubRequestDataBinder dataBinder = new StubRequestDataBinder(testBean, name);
219 		WebDataBinderFactory binderFactory = mock(WebDataBinderFactory.class);
220 		given(binderFactory.createBinder(webRequest, testBean, name)).willReturn(dataBinder);
221 
222 		processor.resolveArgument(paramModelAttr, mavContainer, webRequest, binderFactory);
223 
224 		assertSame("Resolved attribute should be updated to be last in the order",
225 				testBean, mavContainer.getModel().values().toArray()[1]);
226 		assertSame("BindingResult of resolved attribute should be last in the order",
227 				dataBinder.getBindingResult(), mavContainer.getModel().values().toArray()[2]);
228 	}
