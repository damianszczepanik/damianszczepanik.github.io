17  package org.springframework.http.server;
18  
19  import java.io.IOException;
20  import java.util.concurrent.atomic.AtomicBoolean;
21  import javax.servlet.AsyncContext;
22  import javax.servlet.AsyncEvent;
23  import javax.servlet.AsyncListener;
24  import javax.servlet.http.HttpServletRequest;
25  import javax.servlet.http.HttpServletResponse;
26  
27  import org.springframework.util.Assert;
28  
29  /**
30   * A {@link ServerHttpAsyncRequestControl} to use on Servlet containers (Servlet 3.0+).
31   *
32   * @author Rossen Stoyanchev
33   * @since 4.0
34   */
35  public class ServletServerHttpAsyncRequestControl implements ServerHttpAsyncRequestControl, AsyncListener {
36  
37  	private static long NO_TIMEOUT_VALUE = Long.MIN_VALUE;
38  
39  
40  	private final ServletServerHttpRequest request;
41  
42  	private final ServletServerHttpResponse response;
43  
44  	private AsyncContext asyncContext;
45  
46  	private AtomicBoolean asyncCompleted = new AtomicBoolean(false);
47  
48  
49  	/**
50  	 * Constructor accepting a request and response pair that are expected to be of type
51  	 * {@link ServletServerHttpRequest} and {@link ServletServerHttpResponse}
52  	 * respectively.
53  	 */
54  	public ServletServerHttpAsyncRequestControl(ServletServerHttpRequest request, ServletServerHttpResponse response) {
55  		Assert.notNull(request, "request is required");
56  		Assert.notNull(response, "response is required");
57  
58  		Assert.isTrue(request.getServletRequest().isAsyncSupported(),
59  				"Async support must be enabled on a servlet and for all filters involved " +
60  				"in async request processing. This is done in Java code using the Servlet API " +
61  				"or by adding \"<async-supported>true</async-supported>\" to servlet and " +
62  				"filter declarations in web.xml. Also you must use a Servlet 3.0+ container");
63  
64  		this.request = request;
65  		this.response = response;
66  	}
67  
68  
69  	@Override
70  	public boolean isStarted() {
71  		return (this.asyncContext != null && this.request.getServletRequest().isAsyncStarted());
72  	}
73  
74  	@Override
75  	public boolean isCompleted() {
76  		return this.asyncCompleted.get();
77  	}
78  
79  	@Override
80  	public void start() {
81  		start(NO_TIMEOUT_VALUE);
82  	}
83  
84  	@Override
85  	public void start(long timeout) {
86  		Assert.state(!isCompleted(), "Async processing has already completed");
87  		if (isStarted()) {
88  			return;
89  		}
90  
91  		HttpServletRequest servletRequest = this.request.getServletRequest();
92  		HttpServletResponse servletResponse = this.response.getServletResponse();
93  
94  		this.asyncContext = servletRequest.startAsync(servletRequest, servletResponse);
95  		this.asyncContext.addListener(this);
96  
97  		if (timeout != NO_TIMEOUT_VALUE) {
98  			this.asyncContext.setTimeout(timeout);
99  		}
100 	}
101 
102 	@Override
103 	public void complete() {
104 		if (isStarted() && !isCompleted()) {
105 			this.asyncContext.complete();
106 		}
107 	}
108 
109 
110 	// ---------------------------------------------------------------------
111 	// Implementation of AsyncListener methods
112 	// ---------------------------------------------------------------------
113 
114 	@Override
115 	public void onComplete(AsyncEvent event) throws IOException {
116 		this.asyncContext = null;
117 		this.asyncCompleted.set(true);
118 	}
119 
120 	@Override
121 	public void onStartAsync(AsyncEvent event) throws IOException {
122 	}
123 
124 	@Override
125 	public void onError(AsyncEvent event) throws IOException {
126 	}
127 
128 	@Override
129 	public void onTimeout(AsyncEvent event) throws IOException {
130 	}
131 
132 }
