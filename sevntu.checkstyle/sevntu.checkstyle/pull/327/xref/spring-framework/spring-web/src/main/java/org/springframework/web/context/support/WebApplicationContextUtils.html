48  /**
49   * Convenience methods for retrieving the root {@link WebApplicationContext} for
50   * a given {@link ServletContext}. This is useful for programmatically accessing
51   * a Spring application context from within custom web views or MVC actions.
52   *
53   * <p>Note that there are more convenient ways of accessing the root context for
54   * many web frameworks, either part of Spring or available as an external library.
55   * This helper class is just the most generic way to access the root context.
56   *
57   * @author Juergen Hoeller
58   * @see org.springframework.web.context.ContextLoader
59   * @see org.springframework.web.servlet.FrameworkServlet
60   * @see org.springframework.web.servlet.DispatcherServlet
61   * @see org.springframework.web.jsf.FacesContextUtils
62   * @see org.springframework.web.jsf.el.SpringBeanFacesELResolver
63   */
64  public abstract class WebApplicationContextUtils {
65

67  			ClassUtils.isPresent("javax.faces.context.FacesContext", RequestContextHolder.class.getClassLoader());

70  	/**
71  	 * Find the root {@link WebApplicationContext} for this web app, typically
72  	 * loaded via {@link org.springframework.web.context.ContextLoaderListener}.
73  	 * <p>Will rethrow an exception that happened on root context startup,
74  	 * to differentiate between a failed context startup and no context at all.
75  	 * @param sc ServletContext to find the web application context for
76  	 * @return the root WebApplicationContext for this web app
77  	 * @throws IllegalStateException if the root WebApplicationContext could not be found
78  	 * @see org.springframework.web.context.WebApplicationContext#ROOT_WEB_APPLICATION_CONTEXT_ATTRIBUTE
79  	 */
80  	public static WebApplicationContext getRequiredWebApplicationContext(ServletContext sc) throws IllegalStateException {
81  		WebApplicationContext wac = getWebApplicationContext(sc);
82  		if (wac == null) {
83  			throw new IllegalStateException("No WebApplicationContext found: no ContextLoaderListener registered?");
84  		}
85  		return wac;
86  	}

89  	 * Find the root {@link WebApplicationContext} for this web app, typically
90  	 * loaded via {@link org.springframework.web.context.ContextLoaderListener}.
91  	 * <p>Will rethrow an exception that happened on root context startup,
92  	 * to differentiate between a failed context startup and no context at all.
93  	 * @param sc ServletContext to find the web application context for
94  	 * @return the root WebApplicationContext for this web app, or {@code null} if none
95  	 * @see org.springframework.web.context.WebApplicationContext#ROOT_WEB_APPLICATION_CONTEXT_ATTRIBUTE
96  	 */
97  	public static WebApplicationContext getWebApplicationContext(ServletContext sc) {
98  		return getWebApplicationContext(sc, WebApplicationContext.ROOT_WEB_APPLICATION_CONTEXT_ATTRIBUTE);
99  	}

101 	/**
102 	 * Find a custom {@link WebApplicationContext} for this web app.
103 	 * @param sc ServletContext to find the web application context for
104 	 * @param attrName the name of the ServletContext attribute to look for
105 	 * @return the desired WebApplicationContext for this web app, or {@code null} if none
106 	 */
107 	public static WebApplicationContext getWebApplicationContext(ServletContext sc, String attrName) {
108 		Assert.notNull(sc, "ServletContext must not be null");
109 		Object attr = sc.getAttribute(attrName);
110 		if (attr == null) {
111 			return null;
112 		}
113 		if (attr instanceof RuntimeException) {
114 			throw (RuntimeException) attr;
115 		}
116 		if (attr instanceof Error) {
117 			throw (Error) attr;
118 		}
119 		if (attr instanceof Exception) {
120 			throw new IllegalStateException((Exception) attr);
121 		}
122 		if (!(attr instanceof WebApplicationContext)) {
123 			throw new IllegalStateException("Context attribute is not of type WebApplicationContext: " + attr);
124 		}
125 		return (WebApplicationContext) attr;
126 	}

135 		registerWebApplicationScopes(beanFactory, null);
136 	}

138 	/**
139 	 * Register web-specific scopes ("request", "session", "globalSession", "application")
140 	 * with the given BeanFactory, as used by the WebApplicationContext.
141 	 * @param beanFactory the BeanFactory to configure
142 	 * @param sc the ServletContext that we're running within
143 	 */
144 	public static void registerWebApplicationScopes(ConfigurableListableBeanFactory beanFactory, ServletContext sc) {
145 		beanFactory.registerScope(WebApplicationContext.SCOPE_REQUEST, new RequestScope());
146 		beanFactory.registerScope(WebApplicationContext.SCOPE_SESSION, new SessionScope(false));
147 		beanFactory.registerScope(WebApplicationContext.SCOPE_GLOBAL_SESSION, new SessionScope(true));
148 		if (sc != null) {
149 			ServletContextScope appScope = new ServletContextScope(sc);
150 			beanFactory.registerScope(WebApplicationContext.SCOPE_APPLICATION, appScope);
151 			// Register as ServletContext attribute, for ContextCleanupListener to detect it.
152 			sc.setAttribute(ServletContextScope.class.getName(), appScope);
153 		}
154 
155 		beanFactory.registerResolvableDependency(ServletRequest.class, new RequestObjectFactory());
156 		beanFactory.registerResolvableDependency(ServletResponse.class, new ResponseObjectFactory());
157 		beanFactory.registerResolvableDependency(HttpSession.class, new SessionObjectFactory());
158 		beanFactory.registerResolvableDependency(WebRequest.class, new WebRequestObjectFactory());
159 		if (jsfPresent) {
160 			FacesDependencyRegistrar.registerFacesDependencies(beanFactory);
161 		}
162 	}

174 	/**
175 	 * Register web-specific environment beans ("contextParameters", "contextAttributes")
176 	 * with the given BeanFactory, as used by the WebApplicationContext.
177 	 * @param bf the BeanFactory to configure
178 	 * @param servletContext the ServletContext that we're running within
179 	 * @param servletConfig the ServletConfig of the containing Portlet
180 	 */
181 	public static void registerEnvironmentBeans(
182 			ConfigurableListableBeanFactory bf, ServletContext servletContext, ServletConfig servletConfig) {
183 
184 		if (servletContext != null && !bf.containsBean(WebApplicationContext.SERVLET_CONTEXT_BEAN_NAME)) {
185 			bf.registerSingleton(WebApplicationContext.SERVLET_CONTEXT_BEAN_NAME, servletContext);
186 		}
187 
188 		if (servletConfig != null && !bf.containsBean(ConfigurableWebApplicationContext.SERVLET_CONFIG_BEAN_NAME)) {
189 			bf.registerSingleton(ConfigurableWebApplicationContext.SERVLET_CONFIG_BEAN_NAME, servletConfig);
190 		}
191 
192 		if (!bf.containsBean(WebApplicationContext.CONTEXT_PARAMETERS_BEAN_NAME)) {
193 			Map<String, String> parameterMap = new HashMap<String, String>();
194 			if (servletContext != null) {
195 				Enumeration<?> paramNameEnum = servletContext.getInitParameterNames();
196 				while (paramNameEnum.hasMoreElements()) {
197 					String paramName = (String) paramNameEnum.nextElement();
198 					parameterMap.put(paramName, servletContext.getInitParameter(paramName));
199 				}
200 			}
201 			if (servletConfig != null) {
202 				Enumeration<?> paramNameEnum = servletConfig.getInitParameterNames();
203 				while (paramNameEnum.hasMoreElements()) {
204 					String paramName = (String) paramNameEnum.nextElement();
205 					parameterMap.put(paramName, servletConfig.getInitParameter(paramName));
206 				}
207 			}
208 			bf.registerSingleton(WebApplicationContext.CONTEXT_PARAMETERS_BEAN_NAME,
209 					Collections.unmodifiableMap(parameterMap));
210 		}
211 
212 		if (!bf.containsBean(WebApplicationContext.CONTEXT_ATTRIBUTES_BEAN_NAME)) {
213 			Map<String, Object> attributeMap = new HashMap<String, Object>();
214 			if (servletContext != null) {
215 				Enumeration<?> attrNameEnum = servletContext.getAttributeNames();
216 				while (attrNameEnum.hasMoreElements()) {
217 					String attrName = (String) attrNameEnum.nextElement();
218 					attributeMap.put(attrName, servletContext.getAttribute(attrName));
219 				}
220 			}
221 			bf.registerSingleton(WebApplicationContext.CONTEXT_ATTRIBUTES_BEAN_NAME,
222 					Collections.unmodifiableMap(attributeMap));
223 		}
224 	}

236 	/**
237 	 * Replace {@code Servlet}-based {@link StubPropertySource stub property sources} with
238 	 * actual instances populated with the given {@code servletContext} and
239 	 * {@code servletConfig} objects.
240 	 * <p>This method is idempotent with respect to the fact it may be called any number
241 	 * of times but will perform replacement of stub property sources with their
242 	 * corresponding actual property sources once and only once.
243 	 * @param propertySources the {@link MutablePropertySources} to initialize (must not
244 	 * be {@code null})
245 	 * @param servletContext the current {@link ServletContext} (ignored if {@code null}
246 	 * or if the {@link StandardServletEnvironment#SERVLET_CONTEXT_PROPERTY_SOURCE_NAME
247 	 * servlet context property source} has already been initialized)
248 	 * @param servletConfig the current {@link ServletConfig} (ignored if {@code null}
249 	 * or if the {@link StandardServletEnvironment#SERVLET_CONFIG_PROPERTY_SOURCE_NAME
250 	 * servlet config property source} has already been initialized)
251 	 * @see org.springframework.core.env.PropertySource.StubPropertySource
252 	 * @see org.springframework.core.env.ConfigurableEnvironment#getPropertySources()
253 	 */
254 	public static void initServletPropertySources(
255 			MutablePropertySources propertySources, ServletContext servletContext, ServletConfig servletConfig) {
256 
257 		Assert.notNull(propertySources, "propertySources must not be null");
258 		if (servletContext != null && propertySources.contains(StandardServletEnvironment.SERVLET_CONTEXT_PROPERTY_SOURCE_NAME) &&
259 				propertySources.get(StandardServletEnvironment.SERVLET_CONTEXT_PROPERTY_SOURCE_NAME) instanceof StubPropertySource) {
260 			propertySources.replace(StandardServletEnvironment.SERVLET_CONTEXT_PROPERTY_SOURCE_NAME,
261 					new ServletContextPropertySource(StandardServletEnvironment.SERVLET_CONTEXT_PROPERTY_SOURCE_NAME, servletContext));
262 		}
263 		if (servletConfig != null && propertySources.contains(StandardServletEnvironment.SERVLET_CONFIG_PROPERTY_SOURCE_NAME) &&
264 				propertySources.get(StandardServletEnvironment.SERVLET_CONFIG_PROPERTY_SOURCE_NAME) instanceof StubPropertySource) {
265 			propertySources.replace(StandardServletEnvironment.SERVLET_CONFIG_PROPERTY_SOURCE_NAME,
266 					new ServletConfigPropertySource(StandardServletEnvironment.SERVLET_CONFIG_PROPERTY_SOURCE_NAME, servletConfig));
267 		}
268 	}

270 	/**
271 	 * Return the current RequestAttributes instance as ServletRequestAttributes.
272 	 * @see RequestContextHolder#currentRequestAttributes()
273 	 */
274 	private static ServletRequestAttributes currentRequestAttributes() {
275 		RequestAttributes requestAttr = RequestContextHolder.currentRequestAttributes();
276 		if (!(requestAttr instanceof ServletRequestAttributes)) {
277 			throw new IllegalStateException("Current request is not a servlet request");
278 		}
279 		return (ServletRequestAttributes) requestAttr;
280 	}

284 	 * Factory that exposes the current request object on demand.
285 	 */
286 	@SuppressWarnings("serial")
287 	private static class RequestObjectFactory implements ObjectFactory<ServletRequest>, Serializable {
288 
289 		@Override
290 		public ServletRequest getObject() {
291 			return currentRequestAttributes().getRequest();
292 		}
293 
294 		@Override
295 		public String toString() {
296 			return "Current HttpServletRequest";
297 		}
298 	}

301 	/**
302 	 * Factory that exposes the current response object on demand.
303 	 */
304 	@SuppressWarnings("serial")
305 	private static class ResponseObjectFactory implements ObjectFactory<ServletResponse>, Serializable {
306 
307 		@Override
308 		public ServletResponse getObject() {
309 			ServletResponse response = currentRequestAttributes().getResponse();
310 			if (response == null) {
311 				throw new IllegalStateException("Current servlet response not available - " +
312 						"consider using RequestContextFilter instead of RequestContextListener");
313 			}
314 			return response;
315 		}
316 
317 		@Override
318 		public String toString() {
319 			return "Current HttpServletResponse";
320 		}
321 	}

361 	/**
362 	 * Inner class to avoid hard-coded JSF dependency.
363  	 */
364 	private static class FacesDependencyRegistrar {
365 
366 		public static void registerFacesDependencies(ConfigurableListableBeanFactory beanFactory) {
367 			beanFactory.registerResolvableDependency(FacesContext.class, new ObjectFactory<FacesContext>() {
368 				@Override
369 				public FacesContext getObject() {
370 					return FacesContext.getCurrentInstance();
371 				}
372 				@Override
373 				public String toString() {
374 					return "Current JSF FacesContext";
375 				}
376 			});
377 			beanFactory.registerResolvableDependency(ExternalContext.class, new ObjectFactory<ExternalContext>() {
378 				@Override
379 				public ExternalContext getObject() {
380 					return FacesContext.getCurrentInstance().getExternalContext();
381 				}
382 				@Override
383 				public String toString() {
384 					return "Current JSF ExternalContext";
385 				}
386 			});
387 		}
