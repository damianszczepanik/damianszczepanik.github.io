39   * Extension of {@link UriComponents} for hierarchical URIs.
40   *
41   * @author Arjen Poutsma

64  	/**
65  	 * Package-private constructor. All arguments are optional, and can be {@code null}.
66  	 * @param scheme the scheme
67  	 * @param userInfo the user info
68  	 * @param host the host
69  	 * @param port the port
70  	 * @param path the path
71  	 * @param queryParams the query parameters
72  	 * @param fragment the fragment
73  	 * @param encoded whether the components are already encoded
74  	 * @param verify whether the components need to be checked for illegal characters
75  	 */
76  	HierarchicalUriComponents(String scheme, String userInfo, String host, String port, PathComponent path,
77  			MultiValueMap<String, String> queryParams, String fragment, boolean encoded, boolean verify) {
78  
79  		super(scheme, fragment);
80  		this.userInfo = userInfo;
81  		this.host = host;
82  		this.port = port;
83  		this.path = path != null ? path : NULL_PATH_COMPONENT;
84  		this.queryParams = CollectionUtils.unmodifiableMultiValueMap(
85  				queryParams != null ? queryParams : new LinkedMultiValueMap<String, String>(0));
86  		this.encoded = encoded;
87  		if (verify) {
88  			verify();
89  		}
90  	}

132 	@Override
133 	public String getQuery() {
134 		if (!this.queryParams.isEmpty()) {
135 			StringBuilder queryBuilder = new StringBuilder();
136 			for (Map.Entry<String, List<String>> entry : this.queryParams.entrySet()) {
137 				String name = entry.getKey();
138 				List<String> values = entry.getValue();
139 				if (CollectionUtils.isEmpty(values)) {
140 					if (queryBuilder.length() != 0) {
141 						queryBuilder.append('&');
142 					}
143 					queryBuilder.append(name);
144 				}
145 				else {
146 					for (Object value : values) {
147 						if (queryBuilder.length() != 0) {
148 							queryBuilder.append('&');
149 						}
150 						queryBuilder.append(name);
151 
152 						if (value != null) {
153 							queryBuilder.append('=');
154 							queryBuilder.append(value.toString());
155 						}
156 					}
157 				}
158 			}
159 			return queryBuilder.toString();
160 		}
161 		else {
162 			return null;
163 		}
164 	}

177 	/**
178 	 * Encodes all URI components using their specific encoding rules, and returns the result as a new
179 	 * {@code UriComponents} instance.
180 	 * @param encoding the encoding of the values contained in this map
181 	 * @return the encoded uri components
182 	 * @throws UnsupportedEncodingException if the given encoding is not supported
183 	 */
184 	@Override
185 	public HierarchicalUriComponents encode(String encoding) throws UnsupportedEncodingException {
186 		Assert.hasLength(encoding, "Encoding must not be empty");
187 		if (this.encoded) {
188 			return this;
189 		}
190 		String encodedScheme = encodeUriComponent(getScheme(), encoding, Type.SCHEME);
191 		String encodedUserInfo = encodeUriComponent(this.userInfo, encoding, Type.USER_INFO);
192 		String encodedHost = encodeUriComponent(this.host, encoding, getHostType());
193 
194 		PathComponent encodedPath = this.path.encode(encoding);
195 		MultiValueMap<String, String> encodedQueryParams =
196 				new LinkedMultiValueMap<String, String>(this.queryParams.size());
197 		for (Map.Entry<String, List<String>> entry : this.queryParams.entrySet()) {
198 			String encodedName = encodeUriComponent(entry.getKey(), encoding, Type.QUERY_PARAM);
199 			List<String> encodedValues = new ArrayList<String>(entry.getValue().size());
200 			for (String value : entry.getValue()) {
201 				String encodedValue = encodeUriComponent(value, encoding, Type.QUERY_PARAM);
202 				encodedValues.add(encodedValue);
203 			}
204 			encodedQueryParams.put(encodedName, encodedValues);
205 		}
206 		String encodedFragment = encodeUriComponent(this.getFragment(), encoding, Type.FRAGMENT);
207 		return new HierarchicalUriComponents(encodedScheme, encodedUserInfo, encodedHost, this.port, encodedPath,
208 				encodedQueryParams, encodedFragment, true, false);
209 	}

211 	/**
212 	 * Encodes the given source into an encoded String using the rules specified
213 	 * by the given component and with the given options.
214 	 * @param source the source string
215 	 * @param encoding the encoding of the source string
216 	 * @param type the URI component for the source
217 	 * @return the encoded URI
218 	 * @throws IllegalArgumentException when the given uri parameter is not a valid URI
219 	 */
220 	static String encodeUriComponent(String source, String encoding, Type type) throws UnsupportedEncodingException {
221 		if (source == null) {
222 			return null;
223 		}
224 		Assert.hasLength(encoding, "Encoding must not be empty");
225 		byte[] bytes = encodeBytes(source.getBytes(encoding), type);
226 		return new String(bytes, "US-ASCII");
227 	}
228 
229 	private static byte[] encodeBytes(byte[] source, Type type) {
230 		Assert.notNull(source, "Source must not be null");
231 		Assert.notNull(type, "Type must not be null");
232 		ByteArrayOutputStream bos = new ByteArrayOutputStream(source.length);
233 		for (byte b : source) {
234 			if (b < 0) {
235 				b += 256;
236 			}
237 			if (type.isAllowed(b)) {
238 				bos.write(b);
239 			}
240 			else {
241 				bos.write('%');
242 				char hex1 = Character.toUpperCase(Character.forDigit((b >> 4) & 0xF, 16));
243 				char hex2 = Character.toUpperCase(Character.forDigit(b & 0xF, 16));
244 				bos.write(hex1);
245 				bos.write(hex2);
246 			}
247 		}
248 		return bos.toByteArray();
249 	}

252 		return (this.host != null && this.host.startsWith("[")) ? Type.HOST_IPV6 : Type.HOST_IPV4;

258 	/**
259 	 * Verifies all URI components to determine whether they contain any illegal
260 	 * characters, throwing an {@code IllegalArgumentException} if so.
261 	 * @throws IllegalArgumentException if any component has illegal characters
262 	 */
263 	private void verify() {
264 		if (!this.encoded) {
265 			return;
266 		}
267 		verifyUriComponent(getScheme(), Type.SCHEME);
268 		verifyUriComponent(this.userInfo, Type.USER_INFO);
269 		verifyUriComponent(this.host, getHostType());
270 		this.path.verify();
271 		for (Map.Entry<String, List<String>> entry : queryParams.entrySet()) {
272 			verifyUriComponent(entry.getKey(), Type.QUERY_PARAM);
273 			for (String value : entry.getValue()) {
274 				verifyUriComponent(value, Type.QUERY_PARAM);
275 			}
276 		}
277 		verifyUriComponent(getFragment(), Type.FRAGMENT);
278 	}
279 
280 	private static void verifyUriComponent(String source, Type type) {
281 		if (source == null) {
282 			return;
283 		}
284 		int length = source.length();
285 		for (int i=0; i < length; i++) {
286 			char ch = source.charAt(i);
287 			if (ch == '%') {
288 				if ((i + 2) < length) {
289 					char hex1 = source.charAt(i + 1);
290 					char hex2 = source.charAt(i + 2);
291 					int u = Character.digit(hex1, 16);
292 					int l = Character.digit(hex2, 16);
293 					if (u == -1 || l == -1) {
294 						throw new IllegalArgumentException("Invalid encoded sequence \"" + source.substring(i) + "\"");
295 					}
296 					i += 2;
297 				}
298 				else {
299 					throw new IllegalArgumentException("Invalid encoded sequence \"" + source.substring(i) + "\"");
300 				}
301 			}
302 			else if (!type.isAllowed(ch)) {
303 				throw new IllegalArgumentException(
304 						"Invalid character '" + ch + "' for " + type.name() + " in \"" + source + "\"");
305 			}
306 		}
307 	}

312 	@Override
313 	protected HierarchicalUriComponents expandInternal(UriTemplateVariables uriVariables) {
314 		Assert.state(!this.encoded, "Cannot expand an already encoded UriComponents object");
315 		String expandedScheme = expandUriComponent(getScheme(), uriVariables);
316 		String expandedUserInfo = expandUriComponent(this.userInfo, uriVariables);
317 		String expandedHost = expandUriComponent(this.host, uriVariables);
318 		String expandedPort = expandUriComponent(this.port, uriVariables);
319 		PathComponent expandedPath = this.path.expand(uriVariables);
320 		MultiValueMap<String, String> expandedQueryParams =
321 				new LinkedMultiValueMap<String, String>(this.queryParams.size());
322 		for (Map.Entry<String, List<String>> entry : this.queryParams.entrySet()) {
323 			String expandedName = expandUriComponent(entry.getKey(), uriVariables);
324 			List<String> expandedValues = new ArrayList<String>(entry.getValue().size());
325 			for (String value : entry.getValue()) {
326 				String expandedValue = expandUriComponent(value, uriVariables);
327 				expandedValues.add(expandedValue);
328 			}
329 			expandedQueryParams.put(expandedName, expandedValues);
330 		}
331 		String expandedFragment = expandUriComponent(this.getFragment(), uriVariables);
332 		return new HierarchicalUriComponents(expandedScheme, expandedUserInfo, expandedHost, expandedPort, expandedPath,
333 				expandedQueryParams, expandedFragment, false, false);
334 	}

351 	/**
352 	 * Returns a URI string from this {@code UriComponents} instance.
353 	 */
354 	@Override
355 	public String toUriString() {
356 		StringBuilder uriBuilder = new StringBuilder();
357 		if (getScheme() != null) {
358 			uriBuilder.append(getScheme());
359 			uriBuilder.append(':');
360 		}
361 		if (this.userInfo != null || this.host != null) {
362 			uriBuilder.append("//");
363 			if (this.userInfo != null) {
364 				uriBuilder.append(this.userInfo);
365 				uriBuilder.append('@');
366 			}
367 			if (this.host != null) {
368 				uriBuilder.append(host);
369 			}
370 			if (getPort() != -1) {
371 				uriBuilder.append(':');
372 				uriBuilder.append(port);
373 			}
374 		}
375 		String path = getPath();
376 		if (StringUtils.hasLength(path)) {
377 			if (uriBuilder.length() != 0 && path.charAt(0) != PATH_DELIMITER) {
378 				uriBuilder.append(PATH_DELIMITER);
379 			}
380 			uriBuilder.append(path);
381 		}
382 		String query = getQuery();
383 		if (query != null) {
384 			uriBuilder.append('?');
385 			uriBuilder.append(query);
386 		}
387 		if (getFragment() != null) {
388 			uriBuilder.append('#');
389 			uriBuilder.append(getFragment());
390 		}
391 		return uriBuilder.toString();
392 	}

394 	/**
395 	 * Returns a {@code URI} from this {@code UriComponents} instance.
396 	 */
397 	@Override
398 	public URI toUri() {
399 		try {
400 			if (this.encoded) {
401 				return new URI(toString());
402 			}
403 			else {
404 				String path = getPath();
405 				if (StringUtils.hasLength(path) && path.charAt(0) != PATH_DELIMITER) {
406 					// Only prefix the path delimiter if something exists before it
407 					if (getScheme() != null || getUserInfo() != null || getHost() != null || getPort() != -1) {
408 						path = PATH_DELIMITER + path;
409 					}
410 				}
411 				return new URI(getScheme(), getUserInfo(), getHost(), getPort(), path, getQuery(),
412 						getFragment());
413 			}
414 		}
415 		catch (URISyntaxException ex) {
416 			throw new IllegalStateException("Could not create URI object: " + ex.getMessage(), ex);
417 		}
418 	}

420 	@Override
421 	public boolean equals(Object obj) {
422 		if (this == obj) {
423 			return true;
424 		}
425 		if (!(obj instanceof HierarchicalUriComponents)) {
426 			return false;
427 		}
428 		HierarchicalUriComponents other = (HierarchicalUriComponents) obj;
429 		return ObjectUtils.nullSafeEquals(getScheme(), other.getScheme()) &&
430 				ObjectUtils.nullSafeEquals(getUserInfo(), other.getUserInfo()) &&
431 				ObjectUtils.nullSafeEquals(getHost(), other.getHost()) &&
432 				getPort() == other.getPort() &&
433 				this.path.equals(other.path) &&
434 				this.queryParams.equals(other.queryParams) &&
435 				ObjectUtils.nullSafeEquals(getFragment(), other.getFragment());
436 	}

438 	@Override
439 	public int hashCode() {
440 		int result = ObjectUtils.nullSafeHashCode(getScheme());
441 		result = 31 * result + ObjectUtils.nullSafeHashCode(this.userInfo);
442 		result = 31 * result + ObjectUtils.nullSafeHashCode(this.host);
443 		result = 31 * result + ObjectUtils.nullSafeHashCode(this.port);
444 		result = 31 * result + this.path.hashCode();
445 		result = 31 * result + this.queryParams.hashCode();
446 		result = 31 * result + ObjectUtils.nullSafeHashCode(getFragment());
447 		return result;
448 	}

453 	/**
454 	 * Enumeration used to identify the parts of a URI.
455 	 * <p>Contains methods to indicate whether a given character is valid in a specific URI component.
456 	 * @see <a href="http://www.ietf.org/rfc/rfc3986.txt">RFC 3986</a>
457 	 */
458 	static enum Type {
459 
460 		SCHEME {
461 			@Override
462 			public boolean isAllowed(int c) {
463 				return isAlpha(c) || isDigit(c) || '+' == c || '-' == c || '.' == c;
464 			}
465 		},
466 		AUTHORITY {
467 			@Override
468 			public boolean isAllowed(int c) {
469 				return isUnreserved(c) || isSubDelimiter(c) || ':' == c || '@' == c;
470 			}
471 		},
472 		USER_INFO {
473 			@Override
474 			public boolean isAllowed(int c) {
475 				return isUnreserved(c) || isSubDelimiter(c) || ':' == c;
476 			}
477 		},
478 		HOST_IPV4 {
479 			@Override
480 			public boolean isAllowed(int c) {
481 				return isUnreserved(c) || isSubDelimiter(c);
482 			}
483 		},
484 		HOST_IPV6 {
485 			@Override
486 			public boolean isAllowed(int c) {
487 				return isUnreserved(c) || isSubDelimiter(c) || '[' == c || ']' == c || ':' == c;
488 			}
489 		},
490 		PORT {
491 			@Override
492 			public boolean isAllowed(int c) {
493 				return isDigit(c);
494 			}
495 		},
496 		PATH {
497 			@Override
498 			public boolean isAllowed(int c) {
499 				return isPchar(c) || '/' == c;
500 			}
501 		},
502 		PATH_SEGMENT {
503 			@Override
504 			public boolean isAllowed(int c) {
505 				return isPchar(c);
506 			}
507 		},
508 		QUERY {
509 			@Override
510 			public boolean isAllowed(int c) {
511 				return isPchar(c) || '/' == c || '?' == c;
512 			}
513 		},
514 		QUERY_PARAM {
515 			@Override
516 			public boolean isAllowed(int c) {
517 				if ('=' == c || '+' == c || '&' == c) {
518 					return false;
519 				}
520 				else {
521 					return isPchar(c) || '/' == c || '?' == c;
522 				}
523 			}
524 		},
525 		FRAGMENT {
526 			@Override
527 			public boolean isAllowed(int c) {
528 				return isPchar(c) || '/' == c || '?' == c;
529 			}
530 		};
531 
532 		/**
533 		 * Indicates whether the given character is allowed in this URI component.
534 		 * @return {@code true} if the character is allowed; {@code false} otherwise
535 		 */
536 		public abstract boolean isAllowed(int c);
537 
538 		/**
539 		 * Indicates whether the given character is in the {@code ALPHA} set.
540 		 * @see <a href="http://www.ietf.org/rfc/rfc3986.txt">RFC 3986, appendix A</a>
541 		 */
542 		protected boolean isAlpha(int c) {
543 			return c >= 'a' && c <= 'z' || c >= 'A' && c <= 'Z';
544 		}
545 
546 		/**
547 		 * Indicates whether the given character is in the {@code DIGIT} set.
548 		 * @see <a href="http://www.ietf.org/rfc/rfc3986.txt">RFC 3986, appendix A</a>
549 		 */
550 		protected boolean isDigit(int c) {
551 			return c >= '0' && c <= '9';
552 		}
553 
554 		/**
555 		 * Indicates whether the given character is in the {@code gen-delims} set.
556 		 * @see <a href="http://www.ietf.org/rfc/rfc3986.txt">RFC 3986, appendix A</a>
557 		 */
558 		protected boolean isGenericDelimiter(int c) {
559 			return ':' == c || '/' == c || '?' == c || '#' == c || '[' == c || ']' == c || '@' == c;
560 		}
561 
562 		/**
563 		 * Indicates whether the given character is in the {@code sub-delims} set.
564 		 * @see <a href="http://www.ietf.org/rfc/rfc3986.txt">RFC 3986, appendix A</a>
565 		 */
566 		protected boolean isSubDelimiter(int c) {
567 			return '!' == c || '$' == c || '&' == c || '\'' == c || '(' == c || ')' == c || '*' == c || '+' == c ||
568 					',' == c || ';' == c || '=' == c;
569 		}
570 
571 		/**
572 		 * Indicates whether the given character is in the {@code reserved} set.
573 		 * @see <a href="http://www.ietf.org/rfc/rfc3986.txt">RFC 3986, appendix A</a>
574 		 */
575 		protected boolean isReserved(char c) {
576 			return isGenericDelimiter(c) || isSubDelimiter(c);
577 		}
578 
579 		/**
580 		 * Indicates whether the given character is in the {@code unreserved} set.
581 		 * @see <a href="http://www.ietf.org/rfc/rfc3986.txt">RFC 3986, appendix A</a>
582 		 */
583 		protected boolean isUnreserved(int c) {
584 			return isAlpha(c) || isDigit(c) || '-' == c || '.' == c || '_' == c || '~' == c;
585 		}
586 
587 		/**
588 		 * Indicates whether the given character is in the {@code pchar} set.
589 		 * @see <a href="http://www.ietf.org/rfc/rfc3986.txt">RFC 3986, appendix A</a>
590 		 */
591 		protected boolean isPchar(int c) {
592 			return isUnreserved(c) || isSubDelimiter(c) || ':' == c || '@' == c;
593 		}
594 	}

614 	/**
615 	 * Represents a path backed by a string.
616 	 */
617 	static final class FullPathComponent implements PathComponent {
618 
619 		private final String path;
620 
621 		public FullPathComponent(String path) {
622 			this.path = path;
623 		}
624 
625 		@Override
626 		public String getPath() {
627 			return this.path;
628 		}
629 
630 		@Override
631 		public List<String> getPathSegments() {
632 			String delimiter = new String(new char[]{PATH_DELIMITER});
633 			String[] pathSegments = StringUtils.tokenizeToStringArray(path, delimiter);
634 			return Collections.unmodifiableList(Arrays.asList(pathSegments));
635 		}
636 
637 		@Override
638 		public PathComponent encode(String encoding) throws UnsupportedEncodingException {
639 			String encodedPath = encodeUriComponent(getPath(),encoding, Type.PATH);
640 			return new FullPathComponent(encodedPath);
641 		}
642 
643 		@Override
644 		public void verify() {
645 			verifyUriComponent(this.path, Type.PATH);
646 		}
647 
648 		@Override
649 		public PathComponent expand(UriTemplateVariables uriVariables) {
650 			String expandedPath = expandUriComponent(getPath(), uriVariables);
651 			return new FullPathComponent(expandedPath);
652 		}
653

667 	/**
668 	 * Represents a path backed by a string list (i.e. path segments).
669 	 */
670 	static final class PathSegmentComponent implements PathComponent {
671 
672 		private final List<String> pathSegments;
673 
674 		public PathSegmentComponent(List<String> pathSegments) {
675 			this.pathSegments = Collections.unmodifiableList(new ArrayList<String>(pathSegments));
676 		}
677 
678 		@Override
679 		public String getPath() {
680 			StringBuilder pathBuilder = new StringBuilder();
681 			pathBuilder.append(PATH_DELIMITER);
682 			for (Iterator<String> iterator = this.pathSegments.iterator(); iterator.hasNext(); ) {
683 				String pathSegment = iterator.next();
684 				pathBuilder.append(pathSegment);
685 				if (iterator.hasNext()) {
686 					pathBuilder.append(PATH_DELIMITER);
687 				}
688 			}
689 			return pathBuilder.toString();
690 		}
691 
692 		@Override
693 		public List<String> getPathSegments() {
694 			return this.pathSegments;
695 		}
696 
697 		@Override
698 		public PathComponent encode(String encoding) throws UnsupportedEncodingException {
699 			List<String> pathSegments = getPathSegments();
700 			List<String> encodedPathSegments = new ArrayList<String>(pathSegments.size());
701 			for (String pathSegment : pathSegments) {
702 				String encodedPathSegment = encodeUriComponent(pathSegment, encoding, Type.PATH_SEGMENT);
703 				encodedPathSegments.add(encodedPathSegment);
704 			}
705 			return new PathSegmentComponent(encodedPathSegments);
706 		}
707 
708 		@Override
709 		public void verify() {
710 			for (String pathSegment : getPathSegments()) {
711 				verifyUriComponent(pathSegment, Type.PATH_SEGMENT);
712 			}
713 		}
714 
715 		@Override
716 		public PathComponent expand(UriTemplateVariables uriVariables) {
717 			List<String> pathSegments = getPathSegments();
718 			List<String> expandedPathSegments = new ArrayList<String>(pathSegments.size());
719 			for (String pathSegment : pathSegments) {
720 				String expandedPathSegment = expandUriComponent(pathSegment, uriVariables);
721 				expandedPathSegments.add(expandedPathSegment);
722 			}
723 			return new PathSegmentComponent(expandedPathSegments);
724 		}
725 
726 		@Override
727 		public boolean equals(Object obj) {
728 			return (this == obj || (obj instanceof PathSegmentComponent &&
729 					getPathSegments().equals(((PathSegmentComponent) obj).getPathSegments())));
730 		}
731 
732 		@Override
733 		public int hashCode() {
734 			return getPathSegments().hashCode();
735 		}
736 	}

739 	/**
740 	 * Represents a collection of PathComponents.
741 	 */
742 	static final class PathComponentComposite implements PathComponent {
743 
744 		private final List<PathComponent> pathComponents;
745 
746 		public PathComponentComposite(List<PathComponent> pathComponents) {
747 			this.pathComponents = pathComponents;
748 		}
749 
750 		@Override
751 		public String getPath() {
752 			StringBuilder pathBuilder = new StringBuilder();
753 			for (PathComponent pathComponent : this.pathComponents) {
754 				pathBuilder.append(pathComponent.getPath());
755 			}
756 			return pathBuilder.toString();
757 		}
758 
759 		@Override
760 		public List<String> getPathSegments() {
761 			List<String> result = new ArrayList<String>();
762 			for (PathComponent pathComponent : this.pathComponents) {
763 				result.addAll(pathComponent.getPathSegments());
764 			}
765 			return result;
766 		}
767 
768 		@Override
769 		public PathComponent encode(String encoding) throws UnsupportedEncodingException {
770 			List<PathComponent> encodedComponents = new ArrayList<PathComponent>(this.pathComponents.size());
771 			for (PathComponent pathComponent : this.pathComponents) {
772 				encodedComponents.add(pathComponent.encode(encoding));
773 			}
774 			return new PathComponentComposite(encodedComponents);
775 		}
776 
777 		@Override
778 		public void verify() {
779 			for (PathComponent pathComponent : this.pathComponents) {
780 				pathComponent.verify();
781 			}
782 		}
783 
784 		@Override
785 		public PathComponent expand(UriTemplateVariables uriVariables) {
786 			List<PathComponent> expandedComponents = new ArrayList<PathComponent>(this.pathComponents.size());
787 			for (PathComponent pathComponent : this.pathComponents) {
788 				expandedComponents.add(pathComponent.expand(uriVariables));
789 			}
790 			return new PathComponentComposite(expandedComponents);
791 		}
792 	}
