56  /**
57   * <strong>Spring's central class for synchronous client-side HTTP access.</strong>
58   * It simplifies communication with HTTP servers, and enforces RESTful principles.
59   * It handles HTTP connections, leaving application code to provide URLs
60   * (with possible template variables) and extract results.
61   *
62   * <p>The main entry points of this template are the methods named after the six main HTTP methods:
63   * <table>
64   * <tr><th>HTTP method</th><th>RestTemplate methods</th></tr>
65   * <tr><td>DELETE</td><td>{@link #delete}</td></tr>
66   * <tr><td>GET</td><td>{@link #getForObject}</td></tr>
67   * <tr><td></td><td>{@link #getForEntity}</td></tr>
68   * <tr><td>HEAD</td><td>{@link #headForHeaders}</td></tr>
69   * <tr><td>OPTIONS</td><td>{@link #optionsForAllow}</td></tr>
70   * <tr><td>POST</td><td>{@link #postForLocation}</td></tr>
71   * <tr><td></td><td>{@link #postForObject}</td></tr>
72   * <tr><td>PUT</td><td>{@link #put}</td></tr>
73   * <tr><td>any</td><td>{@link #exchange}</td></tr>
74   * <tr><td></td><td>{@link #execute}</td></tr> </table>
75   *
76   * <p>In addition the {@code exchange} and {@code execute} methods are generalized versions of
77   * the above methods and can be used to support additional, less frequent combinations (e.g.
78   * HTTP PATCH, HTTP PUT with response body, etc.). Note however that the underlying HTTP
79   * library used must also support the desired combination.
80   *
81   * <p>For each HTTP method there are 3 variants -- two accept a URI template string
82   * and URI variables (array or map) while a third accepts a {@link URI}.
83   * Note that for URI templates it is assumed encoding is necessary, e.g.
84   * {@code restTemplate.getForObject("http://example.com/hotel list")} becomes
85   * {@code "http://example.com/hotel%20list"}. This also means if the URI template
86   * or URI variables are already encoded, double encoding will occur, e.g.
87   * {@code http://example.com/hotel%20list} becomes
88   * {@code http://example.com/hotel%2520list}). To avoid that use a {@code URI} method
89   * variant to provide (or re-use) a previously encoded URI. To prepare such an URI
90   * with full control over encoding, consider using
91   * {@link org.springframework.web.util.UriComponentsBuilder}.
92   *
93   * <p>Internally the template uses {@link HttpMessageConverter} instances to
94   * convert HTTP messages to and from POJOs. Converters for the main mime types
95   * are registered by default but you can also register additional converters
96   * via {@link #setMessageConverters}.
97   *
98   * <p>This template uses a
99   * {@link org.springframework.http.client.SimpleClientHttpRequestFactory} and a
100  * {@link DefaultResponseErrorHandler} as default strategies for creating HTTP
101  * connections or handling HTTP errors, respectively. These defaults can be overridden
102  * through {@link #setRequestFactory} and {@link #setErrorHandler} respectively.
103  *
104  * @author Arjen Poutsma
105  * @author Brian Clozel
106  * @author Roy Clarkson
107  * @author Juergen Hoeller
108  * @since 3.0
109  * @see HttpMessageConverter
110  * @see RequestCallback
111  * @see ResponseExtractor
112  * @see ResponseErrorHandler
113  * @see AsyncRestTemplate
114  */
115 public class RestTemplate extends InterceptingHttpAccessor implements RestOperations {

141 	/**
142 	 * Create a new instance of the {@link RestTemplate} using default settings.
143 	 * Default {@link HttpMessageConverter}s are initialized.
144 	 */
145 	public RestTemplate() {
146 		this.messageConverters.add(new ByteArrayHttpMessageConverter());
147 		this.messageConverters.add(new StringHttpMessageConverter());
148 		this.messageConverters.add(new ResourceHttpMessageConverter());
149 		this.messageConverters.add(new SourceHttpMessageConverter<Source>());
150 		this.messageConverters.add(new AllEncompassingFormHttpMessageConverter());
151 
152 		if (romePresent) {
153 			this.messageConverters.add(new AtomFeedHttpMessageConverter());
154 			this.messageConverters.add(new RssChannelHttpMessageConverter());
155 		}
156 		if (jackson2XmlPresent) {
157 			messageConverters.add(new MappingJackson2XmlHttpMessageConverter());
158 		}
159 		else if (jaxb2Present) {
160 			this.messageConverters.add(new Jaxb2RootElementHttpMessageConverter());
161 		}
162 		if (jackson2Present) {
163 			this.messageConverters.add(new MappingJackson2HttpMessageConverter());
164 		}
165 		else if (gsonPresent) {
166 			this.messageConverters.add(new GsonHttpMessageConverter());
167 		}
168 	}

195 	 * <p>These converters are used to convert from and to HTTP requests and responses.
196 	 */

203 		}
204 	}

548 	/**
549 	 * Execute the given method on the provided URI.
550 	 * <p>The {@link ClientHttpRequest} is processed using the {@link RequestCallback};
551 	 * the response with the {@link ResponseExtractor}.
552 	 * @param url the fully-expanded URL to connect to
553 	 * @param method the HTTP method to execute (GET, POST, etc.)
554 	 * @param requestCallback object that prepares the request (can be {@code null})
555 	 * @param responseExtractor object that extracts the return value from the response (can be {@code null})
556 	 * @return an arbitrary object, as returned by the {@link ResponseExtractor}
557 	 */
558 	protected <T> T doExecute(URI url, HttpMethod method, RequestCallback requestCallback,
559 			ResponseExtractor<T> responseExtractor) throws RestClientException {
560 
561 		Assert.notNull(url, "'url' must not be null");
562 		Assert.notNull(method, "'method' must not be null");
563 		ClientHttpResponse response = null;
564 		try {
565 			ClientHttpRequest request = createRequest(url, method);
566 			if (requestCallback != null) {
567 				requestCallback.doWithRequest(request);
568 			}
569 			response = request.execute();
570 			handleResponse(url, method, response);
571 			if (responseExtractor != null) {
572 				return responseExtractor.extractData(response);
573 			}
574 			else {
575 				return null;
576 			}
577 		}
578 		catch (IOException ex) {
579 			throw new ResourceAccessException("I/O error on " + method.name() +
580 					" request for \"" + url + "\":" + ex.getMessage(), ex);
581 		}
582 		finally {
583 			if (response != null) {
584 				response.close();
585 			}
586 		}
587 	}

589 	/**
590 	 * Handle the given response, performing appropriate logging and
591 	 * invoking the {@link ResponseErrorHandler} if necessary.
592 	 * <p>Can be overridden in subclasses.
593 	 * @param url the fully-expanded URL to connect to
594 	 * @param method the HTTP method to execute (GET, POST, etc.)
595 	 * @param response the resulting {@link ClientHttpResponse}
596 	 * @throws IOException if propagated from {@link ResponseErrorHandler}
597 	 * @see #setErrorHandler
598 	 * @since 4.1.6
599 	 */
600 	protected void handleResponse(URI url, HttpMethod method, ClientHttpResponse response) throws IOException {
601 		ResponseErrorHandler errorHandler = getErrorHandler();
602 		boolean hasError = errorHandler.hasError(response);
603 		if (logger.isDebugEnabled()) {
604 			try {
605 				logger.debug(method.name() + " request for \"" + url + "\" resulted in " +
606 						response.getRawStatusCode() + " (" + response.getStatusText() + ")" +
607 						(hasError ? "; invoking error handler" : ""));
608 			}
609 			catch (IOException ex) {
610 				// ignore
611 			}
612 		}
613 		if (hasError) {
614 			errorHandler.handleError(response);
615 		}
616 	}

658 	/**
659 	 * Request callback implementation that prepares the request's accept headers.
660 	 */
661 	private class AcceptHeaderRequestCallback implements RequestCallback {
662 
663 		private final Type responseType;
664 
665 		private AcceptHeaderRequestCallback(Type responseType) {
666 			this.responseType = responseType;
667 		}
668 
669 		@Override
670 		public void doWithRequest(ClientHttpRequest request) throws IOException {
671 			if (this.responseType != null) {
672 				Class<?> responseClass = null;
673 				if (this.responseType instanceof Class) {
674 					responseClass = (Class<?>) this.responseType;
675 				}
676 				List<MediaType> allSupportedMediaTypes = new ArrayList<MediaType>();
677 				for (HttpMessageConverter<?> converter : getMessageConverters()) {
678 					if (responseClass != null) {
679 						if (converter.canRead(responseClass, null)) {
680 							allSupportedMediaTypes.addAll(getSupportedMediaTypes(converter));
681 						}
682 					}
683 					else if (converter instanceof GenericHttpMessageConverter) {
684 						GenericHttpMessageConverter<?> genericConverter = (GenericHttpMessageConverter<?>) converter;
685 						if (genericConverter.canRead(this.responseType, null, null)) {
686 							allSupportedMediaTypes.addAll(getSupportedMediaTypes(converter));
687 						}
688 					}
689 				}
690 				if (!allSupportedMediaTypes.isEmpty()) {
691 					MediaType.sortBySpecificity(allSupportedMediaTypes);
692 					if (logger.isDebugEnabled()) {
693 						logger.debug("Setting request Accept header to " +
694 								allSupportedMediaTypes);
695 					}
696 					request.getHeaders().setAccept(allSupportedMediaTypes);
697 				}
698 			}
699 		}
700 
701 		private List<MediaType> getSupportedMediaTypes(HttpMessageConverter<?> messageConverter) {
702 			List<MediaType> supportedMediaTypes = messageConverter.getSupportedMediaTypes();
703 			List<MediaType> result = new ArrayList<MediaType>(supportedMediaTypes.size());
704 			for (MediaType supportedMediaType : supportedMediaTypes) {
705 				if (supportedMediaType.getCharSet() != null) {
706 					supportedMediaType =
707 							new MediaType(supportedMediaType.getType(), supportedMediaType.getSubtype());
708 				}
709 				result.add(supportedMediaType);
710 			}
711 			return result;
712 		}
713 	}

716 	/**
717 	 * Request callback implementation that writes the given object to the request stream.
718 	 */
719 	private class HttpEntityRequestCallback extends AcceptHeaderRequestCallback {
720 
721 		private final HttpEntity<?> requestEntity;
722 
723 		private HttpEntityRequestCallback(Object requestBody) {
724 			this(requestBody, null);
725 		}
726 
727 		private HttpEntityRequestCallback(Object requestBody, Type responseType) {
728 			super(responseType);
729 			if (requestBody instanceof HttpEntity) {
730 				this.requestEntity = (HttpEntity<?>) requestBody;
731 			}
732 			else if (requestBody != null) {
733 				this.requestEntity = new HttpEntity<Object>(requestBody);
734 			}
735 			else {
736 				this.requestEntity = HttpEntity.EMPTY;
737 			}
738 		}
739 
740 		@Override
741 		@SuppressWarnings("unchecked")
742 		public void doWithRequest(ClientHttpRequest httpRequest) throws IOException {
743 			super.doWithRequest(httpRequest);
744 			if (!this.requestEntity.hasBody()) {
745 				HttpHeaders httpHeaders = httpRequest.getHeaders();
746 				HttpHeaders requestHeaders = this.requestEntity.getHeaders();
747 				if (!requestHeaders.isEmpty()) {
748 					httpHeaders.putAll(requestHeaders);
749 				}
750 				if (httpHeaders.getContentLength() == -1) {
751 					httpHeaders.setContentLength(0L);
752 				}
753 			}
754 			else {
755 				Object requestBody = this.requestEntity.getBody();
756 				Class<?> requestType = requestBody.getClass();
757 				HttpHeaders requestHeaders = this.requestEntity.getHeaders();
758 				MediaType requestContentType = requestHeaders.getContentType();
759 				for (HttpMessageConverter<?> messageConverter : getMessageConverters()) {
760 					if (messageConverter.canWrite(requestType, requestContentType)) {
761 						if (!requestHeaders.isEmpty()) {
762 							httpRequest.getHeaders().putAll(requestHeaders);
763 						}
764 						if (logger.isDebugEnabled()) {
765 							if (requestContentType != null) {
766 								logger.debug("Writing [" + requestBody + "] as \"" + requestContentType +
767 										"\" using [" + messageConverter + "]");
768 							}
769 							else {
770 								logger.debug("Writing [" + requestBody + "] using [" + messageConverter + "]");
771 							}
772 
773 						}
774 						((HttpMessageConverter<Object>) messageConverter).write(
775 								requestBody, requestContentType, httpRequest);
776 						return;
777 					}
778 				}
779 				String message = "Could not write request: no suitable HttpMessageConverter found for request type [" +
780 						requestType.getName() + "]";
781 				if (requestContentType != null) {
782 					message += " and content type [" + requestContentType + "]";
783 				}
784 				throw new RestClientException(message);
785 			}
786 		}
787 	}

790 	/**
791 	 * Response extractor for {@link HttpEntity}.
792 	 */
793 	private class ResponseEntityResponseExtractor<T> implements ResponseExtractor<ResponseEntity<T>> {
794 
795 		private final HttpMessageConverterExtractor<T> delegate;
796 
797 		public ResponseEntityResponseExtractor(Type responseType) {
798 			if (responseType != null && !Void.class.equals(responseType)) {
799 				this.delegate = new HttpMessageConverterExtractor<T>(responseType, getMessageConverters(), logger);
800 			}
801 			else {
802 				this.delegate = null;
803 			}
804 		}
805 
806 		@Override
807 		public ResponseEntity<T> extractData(ClientHttpResponse response) throws IOException {
808 			if (this.delegate != null) {
809 				T body = this.delegate.extractData(response);
810 				return new ResponseEntity<T>(body, response.getHeaders(), response.getStatusCode());
811 			}
812 			else {
813 				return new ResponseEntity<T>(response.getHeaders(), response.getStatusCode());
814 			}
815 		}
816 	}
