17  package org.springframework.web.method.support;
18  
19  import java.util.Map;
20  
21  import org.springframework.core.MethodParameter;
22  import org.springframework.core.convert.ConversionService;
23  import org.springframework.web.util.UriComponents;
24  import org.springframework.web.util.UriComponentsBuilder;
25  
26  /**
27   * Strategy for contributing to the building of a {@link UriComponents} by
28   * looking at a method parameter and an argument value and deciding what
29   * part of the target URL should be updated.
30   *
31   * @author Oliver Gierke
32   * @author Rossen Stoyanchev
33   * @since 4.0
34   */
35  public interface UriComponentsContributor {
36  
37  	/**
38  	 * Whether this contributor supports the given method parameter.
39  	 */
40  	boolean supportsParameter(MethodParameter parameter);
41  
42  	/**
43  	 * Process the given method argument and either update the
44  	 * {@link UriComponentsBuilder} or add to the map with URI variables to use to
45  	 * expand the URI after all arguments are processed.
46  	 * @param parameter the controller method parameter, never {@literal null}.
47  	 * @param value the argument value, possibly {@literal null}.
48  	 * @param builder the builder to update, never {@literal null}.
49  	 * @param uriVariables a map to add URI variables to, never {@literal null}.
50  	 * @param conversionService a ConversionService to format values as Strings
51  	 */
52  	void contributeMethodArgument(MethodParameter parameter, Object value, UriComponentsBuilder builder,
53  			Map<String, Object> uriVariables, ConversionService conversionService);
54  
55  }
