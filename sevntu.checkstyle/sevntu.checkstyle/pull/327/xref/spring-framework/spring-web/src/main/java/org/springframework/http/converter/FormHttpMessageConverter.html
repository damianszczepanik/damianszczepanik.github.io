46  /**
47   * Implementation of {@link HttpMessageConverter} to read and write 'normal' HTML
48   * forms and also to write (but not read) multipart data (e.g. file uploads).
49   *
50   * <p>In other words, this converter can read and write the
51   * {@code "application/x-www-form-urlencoded"} media type as
52   * {@link MultiValueMap MultiValueMap&lt;String, String&gt;} and it can also
53   * write (but not read) the {@code "multipart/form-data"} media type as
54   * {@link MultiValueMap MultiValueMap&lt;String, Object&gt;}.
55   *
56   * <p>When writing multipart data, this converter uses other
57   * {@link HttpMessageConverter HttpMessageConverters} to write the respective
58   * MIME parts. By default, basic converters are registered (for {@code Strings}
59   * and {@code Resources}). These can be overridden through the
60   * {@link #setPartConverters partConverters} property.
61   *
62   * <p>For example, the following snippet shows how to submit an HTML form:
63   * <pre class="code">
64   * RestTemplate template = new RestTemplate();  // FormHttpMessageConverter is configured by default
65   * MultiValueMap&lt;String, String&gt; form = new LinkedMultiValueMap&lt;String, String&gt;();
66   * form.add("field 1", "value 1");
67   * form.add("field 2", "value 2");
68   * form.add("field 2", "value 3");
69   * template.postForLocation("http://example.com/myForm", form);
70   * </pre>
71   *
72   * <p>The following snippet shows how to do a file upload:
73   * <pre class="code">
74   * MultiValueMap&lt;String, Object&gt; parts = new LinkedMultiValueMap&lt;String, Object&gt;();
75   * parts.add("field 1", "value 1");
76   * parts.add("file", new ClassPathResource("myFile.jpg"));
77   * template.postForLocation("http://example.com/myFileUpload", parts);
78   * </pre>
79   *
80   * <p>Some methods in this class were inspired by
81   * {@code org.apache.commons.httpclient.methods.multipart.MultipartRequestEntity}.
82   *
83   * @author Arjen Poutsma
84   * @author Rossen Stoyanchev
85   * @since 3.0
86   * @see MultiValueMap
87   */
88  public class FormHttpMessageConverter implements HttpMessageConverter<MultiValueMap<String, ?>> {

110 	public FormHttpMessageConverter() {
111 		this.supportedMediaTypes.add(MediaType.APPLICATION_FORM_URLENCODED);
112 		this.supportedMediaTypes.add(MediaType.MULTIPART_FORM_DATA);
113 
114 		this.partConverters.add(new ByteArrayHttpMessageConverter());
115 		StringHttpMessageConverter stringHttpMessageConverter = new StringHttpMessageConverter();
116 		stringHttpMessageConverter.setWriteAcceptCharset(false);
117 		this.partConverters.add(stringHttpMessageConverter);
118 		this.partConverters.add(new ResourceHttpMessageConverter());
119 	}

175 	@Override
176 	public boolean canRead(Class<?> clazz, MediaType mediaType) {
177 		if (!MultiValueMap.class.isAssignableFrom(clazz)) {
178 			return false;
179 		}
180 		if (mediaType == null) {
181 			return true;
182 		}
183 		for (MediaType supportedMediaType : getSupportedMediaTypes()) {
184 			// We can't read multipart....
185 			if (!supportedMediaType.equals(MediaType.MULTIPART_FORM_DATA) && supportedMediaType.includes(mediaType)) {
186 				return true;
187 			}
188 		}
189 		return false;
190 	}
191 
192 	@Override
193 	public boolean canWrite(Class<?> clazz, MediaType mediaType) {
194 		if (!MultiValueMap.class.isAssignableFrom(clazz)) {
195 			return false;
196 		}
197 		if (mediaType == null || MediaType.ALL.equals(mediaType)) {
198 			return true;
199 		}
200 		for (MediaType supportedMediaType : getSupportedMediaTypes()) {
201 			if (supportedMediaType.isCompatibleWith(mediaType)) {
202 				return true;
203 			}
204 		}
205 		return false;
206 	}

208 	@Override
209 	public MultiValueMap<String, String> read(Class<? extends MultiValueMap<String, ?>> clazz,
210 			HttpInputMessage inputMessage) throws IOException, HttpMessageNotReadableException {
211 
212 		MediaType contentType = inputMessage.getHeaders().getContentType();
213 		Charset charset = (contentType.getCharSet() != null ? contentType.getCharSet() : this.charset);
214 		String body = StreamUtils.copyToString(inputMessage.getBody(), charset);
215 
216 		String[] pairs = StringUtils.tokenizeToStringArray(body, "&");
217 		MultiValueMap<String, String> result = new LinkedMultiValueMap<String, String>(pairs.length);
218 		for (String pair : pairs) {
219 			int idx = pair.indexOf('=');
220 			if (idx == -1) {
221 				result.add(URLDecoder.decode(pair, charset.name()), null);
222 			}
223 			else {
224 				String name = URLDecoder.decode(pair.substring(0, idx), charset.name());
225 				String value = URLDecoder.decode(pair.substring(idx + 1), charset.name());
226 				result.add(name, value);
227 			}
228 		}
229 		return result;
230 	}

232 	@Override
233 	@SuppressWarnings("unchecked")
234 	public void write(MultiValueMap<String, ?> map, MediaType contentType, HttpOutputMessage outputMessage)
235 			throws IOException, HttpMessageNotWritableException {
236 
237 		if (!isMultipart(map, contentType)) {
238 			writeForm((MultiValueMap<String, String>) map, contentType, outputMessage);
239 		}
240 		else {
241 			writeMultipart((MultiValueMap<String, Object>) map, outputMessage);
242 		}
243 	}

246 	private boolean isMultipart(MultiValueMap<String, ?> map, MediaType contentType) {
247 		if (contentType != null) {
248 			return MediaType.MULTIPART_FORM_DATA.includes(contentType);
249 		}
250 		for (String name : map.keySet()) {
251 			for (Object value : map.get(name)) {
252 				if (value != null && !(value instanceof String)) {
253 					return true;
254 				}
255 			}
256 		}
257 		return false;
258 	}

260 	private void writeForm(MultiValueMap<String, String> form, MediaType contentType,
261 			HttpOutputMessage outputMessage) throws IOException {
262 
263 		Charset charset;
264 		if (contentType != null) {
265 			outputMessage.getHeaders().setContentType(contentType);
266 			charset = contentType.getCharSet() != null ? contentType.getCharSet() : this.charset;
267 		}
268 		else {
269 			outputMessage.getHeaders().setContentType(MediaType.APPLICATION_FORM_URLENCODED);
270 			charset = this.charset;
271 		}
272 		StringBuilder builder = new StringBuilder();
273 		for (Iterator<String> nameIterator = form.keySet().iterator(); nameIterator.hasNext();) {
274 			String name = nameIterator.next();
275 			for (Iterator<String> valueIterator = form.get(name).iterator(); valueIterator.hasNext();) {
276 				String value = valueIterator.next();
277 				builder.append(URLEncoder.encode(name, charset.name()));
278 				if (value != null) {
279 					builder.append('=');
280 					builder.append(URLEncoder.encode(value, charset.name()));
281 					if (valueIterator.hasNext()) {
282 						builder.append('&');
283 					}
284 				}
285 			}
286 			if (nameIterator.hasNext()) {
287 				builder.append('&');
288 			}
289 		}
290 		final byte[] bytes = builder.toString().getBytes(charset.name());
291 		outputMessage.getHeaders().setContentLength(bytes.length);
292 
293 		if (outputMessage instanceof StreamingHttpOutputMessage) {
294 			StreamingHttpOutputMessage streamingOutputMessage = (StreamingHttpOutputMessage) outputMessage;
295 			streamingOutputMessage.setBody(new StreamingHttpOutputMessage.Body() {
296 				@Override
297 				public void writeTo(OutputStream outputStream) throws IOException {
298 					StreamUtils.copy(bytes, outputStream);
299 				}
300 			});
301 		}
302 		else {
303 			StreamUtils.copy(bytes, outputMessage.getBody());
304 		}
305 	}

307 	private void writeMultipart(final MultiValueMap<String, Object> parts, HttpOutputMessage outputMessage) throws IOException {
308 		final byte[] boundary = generateMultipartBoundary();
309 		Map<String, String> parameters = Collections.singletonMap("boundary", new String(boundary, "US-ASCII"));
310 
311 		MediaType contentType = new MediaType(MediaType.MULTIPART_FORM_DATA, parameters);
312 		HttpHeaders headers = outputMessage.getHeaders();
313 		headers.setContentType(contentType);
314 
315 		if (outputMessage instanceof StreamingHttpOutputMessage) {
316 			StreamingHttpOutputMessage streamingOutputMessage = (StreamingHttpOutputMessage) outputMessage;
317 			streamingOutputMessage.setBody(new StreamingHttpOutputMessage.Body() {
318 				@Override
319 				public void writeTo(OutputStream outputStream) throws IOException {
320 					writeParts(outputStream, parts, boundary);
321 					writeEnd(outputStream, boundary);
322 				}
323 			});
324 		}
325 		else {
326 			writeParts(outputMessage.getBody(), parts, boundary);
327 			writeEnd(outputMessage.getBody(), boundary);
328 		}
329 	}

331 	private void writeParts(OutputStream os, MultiValueMap<String, Object> parts, byte[] boundary) throws IOException {
332 		for (Map.Entry<String, List<Object>> entry : parts.entrySet()) {
333 			String name = entry.getKey();
334 			for (Object part : entry.getValue()) {
335 				if (part != null) {
336 					writeBoundary(os, boundary);
337 					writePart(name, getHttpEntity(part), os);
338 					writeNewLine(os);
339 				}
340 			}
341 		}
342 	}

344 	@SuppressWarnings("unchecked")
345 	private void writePart(String name, HttpEntity<?> partEntity, OutputStream os) throws IOException {
346 		Object partBody = partEntity.getBody();
347 		Class<?> partType = partBody.getClass();
348 		HttpHeaders partHeaders = partEntity.getHeaders();
349 		MediaType partContentType = partHeaders.getContentType();
350 		for (HttpMessageConverter<?> messageConverter : this.partConverters) {
351 			if (messageConverter.canWrite(partType, partContentType)) {
352 				HttpOutputMessage multipartMessage = new MultipartHttpOutputMessage(os);
353 				multipartMessage.getHeaders().setContentDispositionFormData(name, getFilename(partBody));
354 				if (!partHeaders.isEmpty()) {
355 					multipartMessage.getHeaders().putAll(partHeaders);
356 				}
357 				((HttpMessageConverter<Object>) messageConverter).write(partBody, partContentType, multipartMessage);
358 				return;
359 			}
360 		}
361 		throw new HttpMessageNotWritableException("Could not write request: no suitable HttpMessageConverter " +
362 				"found for request type [" + partType.getName() + "]");
363 	}

366 	/**
367 	 * Generate a multipart boundary.
368 	 * <p>The default implementation returns a random boundary.
369 	 * Can be overridden in subclasses.
370 	 */
371 	protected byte[] generateMultipartBoundary() {
372 		byte[] boundary = new byte[this.random.nextInt(11) + 30];
373 		for (int i = 0; i < boundary.length; i++) {
374 			boundary[i] = BOUNDARY_CHARS[this.random.nextInt(BOUNDARY_CHARS.length)];
375 		}
376 		return boundary;
377 	}

379 	/**
380 	 * Return an {@link HttpEntity} for the given part Object.
381 	 * @param part the part to return an {@link HttpEntity} for
382 	 * @return the part Object itself it is an {@link HttpEntity},
383 	 * or a newly built {@link HttpEntity} wrapper for that part
384 	 */
385 	protected HttpEntity<?> getHttpEntity(Object part) {
386 		if (part instanceof HttpEntity) {
387 			return (HttpEntity<?>) part;
388 		}
389 		else {
390 			return new HttpEntity<Object>(part);
391 		}
392 	}

394 	/**
395 	 * Return the filename of the given multipart part. This value will be used for the
396 	 * {@code Content-Disposition} header.
397 	 * <p>The default implementation returns {@link Resource#getFilename()} if the part is a
398 	 * {@code Resource}, and {@code null} in other cases. Can be overridden in subclasses.
399 	 * @param part the part to determine the file name for
400 	 * @return the filename, or {@code null} if not known
401 	 */
402 	protected String getFilename(Object part) {
403 		if (part instanceof Resource) {
404 			Resource resource = (Resource) part;
405 			String filename = resource.getFilename();
406 			if (this.multipartCharset != null) {
407 				filename = MimeDelegate.encode(filename, this.multipartCharset.name());
408 			}
409 			return filename;
410 		}
411 		else {
412 			return null;
413 		}
414 	}

417 	private void writeBoundary(OutputStream os, byte[] boundary) throws IOException {
418 		os.write('-');
419 		os.write('-');
420 		os.write(boundary);
421 		writeNewLine(os);
422 	}
423 
424 	private static void writeEnd(OutputStream os, byte[] boundary) throws IOException {
425 		os.write('-');
426 		os.write('-');
427 		os.write(boundary);
428 		os.write('-');
429 		os.write('-');
430 		writeNewLine(os);
431 	}
432 
433 	private static void writeNewLine(OutputStream os) throws IOException {
434 		os.write('\r');
435 		os.write('\n');
436 	}

439 	/**
440 	 * Implementation of {@link org.springframework.http.HttpOutputMessage} used
441 	 * to write a MIME multipart.
442 	 */
443 	private static class MultipartHttpOutputMessage implements HttpOutputMessage {
444 
445 		private final OutputStream outputStream;
446 
447 		private final HttpHeaders headers = new HttpHeaders();
448 
449 		private boolean headersWritten = false;
450 
451 		public MultipartHttpOutputMessage(OutputStream outputStream) {
452 			this.outputStream = outputStream;
453 		}
454 
455 		@Override
456 		public HttpHeaders getHeaders() {
457 			return (this.headersWritten ? HttpHeaders.readOnlyHttpHeaders(this.headers) : this.headers);
458 		}
459 
460 		@Override
461 		public OutputStream getBody() throws IOException {
462 			writeHeaders();
463 			return this.outputStream;
464 		}
465 
466 		private void writeHeaders() throws IOException {
467 			if (!this.headersWritten) {
468 				for (Map.Entry<String, List<String>> entry : this.headers.entrySet()) {
469 					byte[] headerName = getAsciiBytes(entry.getKey());
470 					for (String headerValueString : entry.getValue()) {
471 						byte[] headerValue = getAsciiBytes(headerValueString);
472 						this.outputStream.write(headerName);
473 						this.outputStream.write(':');
474 						this.outputStream.write(' ');
475 						this.outputStream.write(headerValue);
476 						writeNewLine(this.outputStream);
477 					}
478 				}
479 				writeNewLine(this.outputStream);
480 				this.headersWritten = true;
481 			}
482 		}
483 
484 		private byte[] getAsciiBytes(String name) {
485 			try {
486 				return name.getBytes("US-ASCII");
487 			}
488 			catch (UnsupportedEncodingException ex) {
489 				// Should not happen - US-ASCII is always supported.
490 				throw new IllegalStateException(ex);
491 			}
492 		}
493 	}
