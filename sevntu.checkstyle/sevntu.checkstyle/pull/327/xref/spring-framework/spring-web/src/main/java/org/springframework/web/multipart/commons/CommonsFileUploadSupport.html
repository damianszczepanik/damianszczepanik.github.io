41  /**
42   * Base class for multipart resolvers that use Jakarta Commons FileUpload
43   * 1.2 or above.
44   *
45   * <p>Provides common configuration properties and parsing functionality
46   * for multipart requests, using a Map of Spring CommonsMultipartFile instances
47   * as representation of uploaded files and a String-based parameter Map as
48   * representation of uploaded form fields.
49   *
50   * <p>Subclasses implement concrete resolution strategies for Servlet or Portlet
51   * environments: see CommonsMultipartResolver and CommonsPortletMultipartResolver,
52   * respectively. This base class is not tied to either of those APIs, factoring
53   * out common functionality.
54   *
55   * @author Juergen Hoeller
56   * @since 2.0
57   * @see CommonsMultipartFile
58   * @see CommonsMultipartResolver
59   * @see org.springframework.web.portlet.multipart.CommonsPortletMultipartResolver
60   */
61  public abstract class CommonsFileUploadSupport {

123 	/**
124 	 * Set the default character encoding to use for parsing requests,
125 	 * to be applied to headers of individual parts and to form fields.
126 	 * Default is ISO-8859-1, according to the Servlet spec.
127 	 * <p>If the request specifies a character encoding itself, the request
128 	 * encoding will override this setting. This also allows for generically
129 	 * overriding the character encoding in a filter that invokes the
130 	 * {@code ServletRequest.setCharacterEncoding} method.
131 	 * @param defaultEncoding the character encoding to use
132 	 * @see javax.servlet.ServletRequest#getCharacterEncoding
133 	 * @see javax.servlet.ServletRequest#setCharacterEncoding
134 	 * @see WebUtils#DEFAULT_CHARACTER_ENCODING
135 	 * @see org.apache.commons.fileupload.FileUploadBase#setHeaderEncoding
136 	 */
137 	public void setDefaultEncoding(String defaultEncoding) {
138 		this.fileUpload.setHeaderEncoding(defaultEncoding);
139 	}

141 	protected String getDefaultEncoding() {
142 		String encoding = getFileUpload().getHeaderEncoding();
143 		if (encoding == null) {
144 			encoding = WebUtils.DEFAULT_CHARACTER_ENCODING;
145 		}
146 		return encoding;
147 	}

149 	/**
150 	 * Set the temporary directory where uploaded files get stored.
151 	 * Default is the servlet container's temporary directory for the web application.
152 	 * @see org.springframework.web.util.WebUtils#TEMP_DIR_CONTEXT_ATTRIBUTE
153 	 */
154 	public void setUploadTempDir(Resource uploadTempDir) throws IOException {
155 		if (!uploadTempDir.exists() && !uploadTempDir.getFile().mkdirs()) {
156 			throw new IllegalArgumentException("Given uploadTempDir [" + uploadTempDir + "] could not be created");
157 		}
158 		this.fileItemFactory.setRepository(uploadTempDir.getFile());
159 		this.uploadTempDirSpecified = true;
160 	}

163 		return this.uploadTempDirSpecified;
164 	}

186 	/**
187 	 * Determine an appropriate FileUpload instance for the given encoding.
188 	 * <p>Default implementation returns the shared FileUpload instance
189 	 * if the encoding matches, else creates a new FileUpload instance
190 	 * with the same configuration other than the desired encoding.
191 	 * @param encoding the character encoding to use
192 	 * @return an appropriate FileUpload instance.
193 	 */
194 	protected FileUpload prepareFileUpload(String encoding) {
195 		FileUpload fileUpload = getFileUpload();
196 		FileUpload actualFileUpload = fileUpload;
197 
198 		// Use new temporary FileUpload instance if the request specifies
199 		// its own encoding that does not match the default encoding.
200 		if (encoding != null && !encoding.equals(fileUpload.getHeaderEncoding())) {
201 			actualFileUpload = newFileUpload(getFileItemFactory());
202 			actualFileUpload.setSizeMax(fileUpload.getSizeMax());
203 			actualFileUpload.setHeaderEncoding(encoding);
204 		}
205 
206 		return actualFileUpload;
207 	}

209 	/**
210 	 * Parse the given List of Commons FileItems into a Spring MultipartParsingResult,
211 	 * containing Spring MultipartFile instances and a Map of multipart parameter.
212 	 * @param fileItems the Commons FileIterms to parse
213 	 * @param encoding the encoding to use for form fields
214 	 * @return the Spring MultipartParsingResult
215 	 * @see CommonsMultipartFile#CommonsMultipartFile(org.apache.commons.fileupload.FileItem)
216 	 */
217 	protected MultipartParsingResult parseFileItems(List<FileItem> fileItems, String encoding) {
218 		MultiValueMap<String, MultipartFile> multipartFiles = new LinkedMultiValueMap<String, MultipartFile>();
219 		Map<String, String[]> multipartParameters = new HashMap<String, String[]>();
220 		Map<String, String> multipartParameterContentTypes = new HashMap<String, String>();
221 
222 		// Extract multipart files and multipart parameters.
223 		for (FileItem fileItem : fileItems) {
224 			if (fileItem.isFormField()) {
225 				String value;
226 				String partEncoding = determineEncoding(fileItem.getContentType(), encoding);
227 				if (partEncoding != null) {
228 					try {
229 						value = fileItem.getString(partEncoding);
230 					}
231 					catch (UnsupportedEncodingException ex) {
232 						if (logger.isWarnEnabled()) {
233 							logger.warn("Could not decode multipart item '" + fileItem.getFieldName() +
234 									"' with encoding '" + partEncoding + "': using platform default");
235 						}
236 						value = fileItem.getString();
237 					}
238 				}
239 				else {
240 					value = fileItem.getString();
241 				}
242 				String[] curParam = multipartParameters.get(fileItem.getFieldName());
243 				if (curParam == null) {
244 					// simple form field
245 					multipartParameters.put(fileItem.getFieldName(), new String[] {value});
246 				}
247 				else {
248 					// array of simple form fields
249 					String[] newParam = StringUtils.addStringToArray(curParam, value);
250 					multipartParameters.put(fileItem.getFieldName(), newParam);
251 				}
252 				multipartParameterContentTypes.put(fileItem.getFieldName(), fileItem.getContentType());
253 			}
254 			else {
255 				// multipart file field
256 				CommonsMultipartFile file = new CommonsMultipartFile(fileItem);
257 				multipartFiles.add(file.getName(), file);
258 				if (logger.isDebugEnabled()) {
259 					logger.debug("Found multipart file [" + file.getName() + "] of size " + file.getSize() +
260 							" bytes with original filename [" + file.getOriginalFilename() + "], stored " +
261 							file.getStorageDescription());
262 				}
263 			}
264 		}
265 		return new MultipartParsingResult(multipartFiles, multipartParameters, multipartParameterContentTypes);
266 	}
267 
268 	/**
269 	 * Cleanup the Spring MultipartFiles created during multipart parsing,
270 	 * potentially holding temporary data on disk.
271 	 * <p>Deletes the underlying Commons FileItem instances.
272 	 * @param multipartFiles Collection of MultipartFile instances
273 	 * @see org.apache.commons.fileupload.FileItem#delete()
274 	 */
275 	protected void cleanupFileItems(MultiValueMap<String, MultipartFile> multipartFiles) {
276 		for (List<MultipartFile> files : multipartFiles.values()) {
277 			for (MultipartFile file : files) {
278 				if (file instanceof CommonsMultipartFile) {
279 					CommonsMultipartFile cmf = (CommonsMultipartFile) file;
280 					cmf.getFileItem().delete();
281 					if (logger.isDebugEnabled()) {
282 						logger.debug("Cleaning up multipart file [" + cmf.getName() + "] with original filename [" +
283 								cmf.getOriginalFilename() + "], stored " + cmf.getStorageDescription());
284 					}
285 				}
286 			}
287 		}
288 	}

290 	private String determineEncoding(String contentTypeHeader, String defaultEncoding) {
291 		if (!StringUtils.hasText(contentTypeHeader)) {
292 			return defaultEncoding;
293 		}
294 		MediaType contentType = MediaType.parseMediaType(contentTypeHeader);
295 		Charset charset = contentType.getCharSet();
296 		return (charset != null ? charset.name() : defaultEncoding);
297 	}

300 	/**
301 	 * Holder for a Map of Spring MultipartFiles and a Map of
302 	 * multipart parameters.
303 	 */
304 	protected static class MultipartParsingResult {
305 
306 		private final MultiValueMap<String, MultipartFile> multipartFiles;
307 
308 		private final Map<String, String[]> multipartParameters;
309 
310 		private final Map<String, String> multipartParameterContentTypes;
311 
312 		public MultipartParsingResult(MultiValueMap<String, MultipartFile> mpFiles,
313 				Map<String, String[]> mpParams, Map<String, String> mpParamContentTypes) {
314 			this.multipartFiles = mpFiles;
315 			this.multipartParameters = mpParams;
316 			this.multipartParameterContentTypes = mpParamContentTypes;
317 		}
318 
319 		public MultiValueMap<String, MultipartFile> getMultipartFiles() {
320 			return this.multipartFiles;
321 		}
322 
323 		public Map<String, String[]> getMultipartParameters() {
324 			return this.multipartParameters;
325 		}
326 
327 		public Map<String, String> getMultipartParameterContentTypes() {
328 			return this.multipartParameterContentTypes;
329 		}
330 	}
