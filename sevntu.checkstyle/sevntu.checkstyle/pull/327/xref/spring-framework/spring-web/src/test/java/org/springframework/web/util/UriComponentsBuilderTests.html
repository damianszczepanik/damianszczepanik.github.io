81  		assertEquals("baz", result2.getFragment());
82  		expected = new URI("http://example.com/foo/foo2?bar#baz");

142 		UriComponents result = UriComponentsBuilder.fromUriString("http://www.ietf.org/rfc/rfc3986.txt").build();
143 		assertEquals("http", result.getScheme());
144 		assertNull(result.getUserInfo());
145 		assertEquals("www.ietf.org", result.getHost());
146 		assertEquals(-1, result.getPort());
147 		assertEquals("/rfc/rfc3986.txt", result.getPath());
148 		assertEquals(Arrays.asList("rfc", "rfc3986.txt"), result.getPathSegments());
149 		assertNull(result.getQuery());
150 		assertNull(result.getFragment());
151 
152 		result = UriComponentsBuilder.fromUriString(
153 				"http://arjen:foobar@java.sun.com:80/javase/6/docs/api/java/util/BitSet.html?foo=bar#and(java.util.BitSet)")
154 				.build();
155 		assertEquals("http", result.getScheme());
156 		assertEquals("arjen:foobar", result.getUserInfo());
157 		assertEquals("java.sun.com", result.getHost());
158 		assertEquals(80, result.getPort());
159 		assertEquals("/javase/6/docs/api/java/util/BitSet.html", result.getPath());
160 		assertEquals("foo=bar", result.getQuery());
161 		MultiValueMap<String, String> expectedQueryParams = new LinkedMultiValueMap<String, String>(1);
162 		expectedQueryParams.add("foo", "bar");
163 		assertEquals(expectedQueryParams, result.getQueryParams());
164 		assertEquals("and(java.util.BitSet)", result.getFragment());
165 
166 		result = UriComponentsBuilder.fromUriString("mailto:java-net@java.sun.com#baz").build();
167 		assertEquals("mailto", result.getScheme());
168 		assertNull(result.getUserInfo());
169 		assertNull(result.getHost());
170 		assertEquals(-1, result.getPort());
171 		assertEquals("java-net@java.sun.com", result.getSchemeSpecificPart());
172 		assertNull(result.getPath());
173 		assertNull(result.getQuery());
174 		assertEquals("baz", result.getFragment());
175 
176 		result = UriComponentsBuilder.fromUriString("docs/guide/collections/designfaq.html#28").build();
177 		assertNull(result.getScheme());
178 		assertNull(result.getUserInfo());
179 		assertNull(result.getHost());
180 		assertEquals(-1, result.getPort());
181 		assertEquals("docs/guide/collections/designfaq.html", result.getPath());
182 		assertNull(result.getQuery());
183 		assertEquals("28", result.getFragment());
184 	}

488 	@Test
489 	public void testClone() throws URISyntaxException {
490 		UriComponentsBuilder builder1 = UriComponentsBuilder.newInstance();
491 		builder1.scheme("http").host("e1.com").path("/p1").pathSegment("ps1").queryParam("q1").fragment("f1");
492 
493 		UriComponentsBuilder builder2 = (UriComponentsBuilder) builder1.clone();
494 		builder2.scheme("https").host("e2.com").path("p2").pathSegment("ps2").queryParam("q2").fragment("f2");
495 
496 		UriComponents result1 = builder1.build();
497 		assertEquals("http", result1.getScheme());
498 		assertEquals("e1.com", result1.getHost());
499 		assertEquals("/p1/ps1", result1.getPath());
500 		assertEquals("q1", result1.getQuery());
501 		assertEquals("f1", result1.getFragment());
502 
503 		UriComponents result2 = builder2.build();
504 		assertEquals("https", result2.getScheme());
505 		assertEquals("e2.com", result2.getHost());
506 		assertEquals("/p1/ps1/p2/ps2", result2.getPath());
507 		assertEquals("q1&q2", result2.getQuery());
508 		assertEquals("f2", result2.getFragment());
509 	}
