492 	/**
493 	 * Java 5 enumeration of HTTP status series.
494 	 * <p>Retrievable via {@link HttpStatus#series()}.
495 	 */
496 	public static enum Series {
497 
498 		INFORMATIONAL(1),
499 		SUCCESSFUL(2),
500 		REDIRECTION(3),
501 		CLIENT_ERROR(4),
502 		SERVER_ERROR(5);
503 
504 		private final int value;
505 
506 		private Series(int value) {
507 			this.value = value;
508 		}
509 
510 		/**
511 		 * Return the integer value of this status series. Ranges from 1 to 5.
512 		 */
513 		public int value() {
514 			return this.value;
515 		}
516 
517 		public static Series valueOf(int status) {
518 			int seriesCode = status / 100;
519 			for (Series series : values()) {
520 				if (series.value == seriesCode) {
521 					return series;
522 				}
523 			}
524 			throw new IllegalArgumentException("No matching constant for [" + status + "]");
525 		}
526
