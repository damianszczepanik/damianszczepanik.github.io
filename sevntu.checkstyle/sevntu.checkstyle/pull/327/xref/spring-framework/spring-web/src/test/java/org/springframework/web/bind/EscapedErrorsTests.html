17  package org.springframework.web.bind;
18  
19  import org.junit.Test;
20  
21  import org.springframework.tests.sample.beans.TestBean;
22  import org.springframework.validation.BindException;
23  import org.springframework.validation.Errors;
24  import org.springframework.validation.FieldError;
25  import org.springframework.validation.ObjectError;
26  
27  import static org.junit.Assert.*;
28  
29  /**
30   * @author Juergen Hoeller
31   * @since 02.05.2003
32   */
33  public class EscapedErrorsTests {
34  
35  	@Test
36  	public void testEscapedErrors() {
37  		TestBean tb = new TestBean();
38  		tb.setName("empty &");
39  
40  		Errors errors = new EscapedErrors(new BindException(tb, "tb"));
41  		errors.rejectValue("name", "NAME_EMPTY &", null, "message: &");
42  		errors.rejectValue("age", "AGE_NOT_SET <tag>", null, "message: <tag>");
43  		errors.rejectValue("age", "AGE_NOT_32 <tag>", null, "message: <tag>");
44  		errors.reject("GENERAL_ERROR \" '", null, "message: \" '");
45  
46  		assertTrue("Correct errors flag", errors.hasErrors());
47  		assertTrue("Correct number of errors", errors.getErrorCount() == 4);
48  		assertTrue("Correct object name", "tb".equals(errors.getObjectName()));
49  
50  		assertTrue("Correct global errors flag", errors.hasGlobalErrors());
51  		assertTrue("Correct number of global errors", errors.getGlobalErrorCount() == 1);
52  		ObjectError globalError = errors.getGlobalError();
53  		assertTrue("Global error message escaped", "message: &quot; &#39;".equals(globalError.getDefaultMessage()));
54  		assertTrue("Global error code not escaped", "GENERAL_ERROR \" '".equals(globalError.getCode()));
55  		ObjectError globalErrorInList = errors.getGlobalErrors().get(0);
56  		assertTrue("Same global error in list", globalError.getDefaultMessage().equals(globalErrorInList.getDefaultMessage()));
57  		ObjectError globalErrorInAllList = errors.getAllErrors().get(3);
58  		assertTrue("Same global error in list", globalError.getDefaultMessage().equals(globalErrorInAllList.getDefaultMessage()));
59  
60  		assertTrue("Correct field errors flag", errors.hasFieldErrors());
61  		assertTrue("Correct number of field errors", errors.getFieldErrorCount() == 3);
62  		assertTrue("Correct number of field errors in list", errors.getFieldErrors().size() == 3);
63  		FieldError fieldError = errors.getFieldError();
64  		assertTrue("Field error code not escaped", "NAME_EMPTY &".equals(fieldError.getCode()));
65  		assertTrue("Field value escaped", "empty &amp;".equals(errors.getFieldValue("name")));
66  		FieldError fieldErrorInList = errors.getFieldErrors().get(0);
67  		assertTrue("Same field error in list", fieldError.getDefaultMessage().equals(fieldErrorInList.getDefaultMessage()));
68  
69  		assertTrue("Correct name errors flag", errors.hasFieldErrors("name"));
70  		assertTrue("Correct number of name errors", errors.getFieldErrorCount("name") == 1);
71  		assertTrue("Correct number of name errors in list", errors.getFieldErrors("name").size() == 1);
72  		FieldError nameError = errors.getFieldError("name");
73  		assertTrue("Name error message escaped", "message: &amp;".equals(nameError.getDefaultMessage()));
74  		assertTrue("Name error code not escaped", "NAME_EMPTY &".equals(nameError.getCode()));
75  		assertTrue("Name value escaped", "empty &amp;".equals(errors.getFieldValue("name")));
76  		FieldError nameErrorInList = errors.getFieldErrors("name").get(0);
77  		assertTrue("Same name error in list", nameError.getDefaultMessage().equals(nameErrorInList.getDefaultMessage()));
78  
79  		assertTrue("Correct age errors flag", errors.hasFieldErrors("age"));
80  		assertTrue("Correct number of age errors", errors.getFieldErrorCount("age") == 2);
81  		assertTrue("Correct number of age errors in list", errors.getFieldErrors("age").size() == 2);
82  		FieldError ageError = errors.getFieldError("age");
83  		assertTrue("Age error message escaped", "message: &lt;tag&gt;".equals(ageError.getDefaultMessage()));
84  		assertTrue("Age error code not escaped", "AGE_NOT_SET <tag>".equals(ageError.getCode()));
85  		assertTrue("Age value not escaped", (new Integer(0)).equals(errors.getFieldValue("age")));
86  		FieldError ageErrorInList = errors.getFieldErrors("age").get(0);
87  		assertTrue("Same name error in list", ageError.getDefaultMessage().equals(ageErrorInList.getDefaultMessage()));
88  		FieldError ageError2 = errors.getFieldErrors("age").get(1);
89  		assertTrue("Age error 2 message escaped", "message: &lt;tag&gt;".equals(ageError2.getDefaultMessage()));
90  		assertTrue("Age error 2 code not escaped", "AGE_NOT_32 <tag>".equals(ageError2.getCode()));
91  	}
92  
93  }
