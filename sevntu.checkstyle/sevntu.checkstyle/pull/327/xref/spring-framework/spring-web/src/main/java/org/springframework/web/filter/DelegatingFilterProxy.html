31  /**
32   * Proxy for a standard Servlet 2.3 Filter, delegating to a Spring-managed
33   * bean that implements the Filter interface. Supports a "targetBeanName"
34   * filter init-param in {@code web.xml}, specifying the name of the
35   * target bean in the Spring application context.
36   *
37   * <p>{@code web.xml} will usually contain a {@code DelegatingFilterProxy} definition,
38   * with the specified {@code filter-name} corresponding to a bean name in
39   * Spring's root application context. All calls to the filter proxy will then
40   * be delegated to that bean in the Spring context, which is required to implement
41   * the standard Servlet 2.3 Filter interface.
42   *
43   * <p>This approach is particularly useful for Filter implementation with complex
44   * setup needs, allowing to apply the full Spring bean definition machinery to
45   * Filter instances. Alternatively, consider standard Filter setup in combination
46   * with looking up service beans from the Spring root application context.
47   *
48   * <p><b>NOTE:</b> The lifecycle methods defined by the Servlet Filter interface
49   * will by default <i>not</i> be delegated to the target bean, relying on the
50   * Spring application context to manage the lifecycle of that bean. Specifying
51   * the "targetFilterLifecycle" filter init-param as "true" will enforce invocation
52   * of the {@code Filter.init} and {@code Filter.destroy} lifecycle methods
53   * on the target bean, letting the servlet container manage the filter lifecycle.
54   *
55   * <p>As of Spring 3.1, {@code DelegatingFilterProxy} has been updated to optionally accept
56   * constructor parameters when using Servlet 3.0's instance-based filter registration
57   * methods, usually in conjunction with Spring 3.1's
58   * {@link org.springframework.web.WebApplicationInitializer} SPI. These constructors allow
59   * for providing the delegate Filter bean directly, or providing the application context
60   * and bean name to fetch, avoiding the need to look up the application context from the
61   * ServletContext.
62   *
63   * <p>This class was originally inspired by Spring Security's {@code FilterToBeanProxy}
64   * class, written by Ben Alex.
65   *
66   * @author Juergen Hoeller
67   * @author Sam Brannen
68   * @author Chris Beams
69   * @since 1.2
70   * @see #setTargetBeanName
71   * @see #setTargetFilterLifecycle
72   * @see javax.servlet.Filter#doFilter
73   * @see javax.servlet.Filter#init
74   * @see javax.servlet.Filter#destroy
75   * @see #DelegatingFilterProxy(Filter)
76   * @see #DelegatingFilterProxy(String)
77   * @see #DelegatingFilterProxy(String, WebApplicationContext)
78   * @see javax.servlet.ServletContext#addFilter(String, Filter)
79   * @see org.springframework.web.WebApplicationInitializer
80   */
81  public class DelegatingFilterProxy extends GenericFilterBean {
82  
83  	private String contextAttribute;
84  
85  	private WebApplicationContext webApplicationContext;
86  
87  	private String targetBeanName;
88  
89  	private boolean targetFilterLifecycle = false;

92  
93  	private final Object delegateMonitor = new Object();

96  	/**
97  	 * Create a new {@code DelegatingFilterProxy}. For traditional (pre-Servlet 3.0) use
98  	 * in {@code web.xml}.
99  	 * @see #setTargetBeanName(String)
100 	 */
101 	public DelegatingFilterProxy() {
102 	}

104 	/**
105 	 * Create a new {@code DelegatingFilterProxy} with the given {@link Filter} delegate.
106 	 * Bypasses entirely the need for interacting with a Spring application context,
107 	 * specifying the {@linkplain #setTargetBeanName target bean name}, etc.
108 	 * <p>For use in Servlet 3.0+ environments where instance-based registration of
109 	 * filters is supported.
110 	 * @param delegate the {@code Filter} instance that this proxy will delegate to and
111 	 * manage the lifecycle for (must not be {@code null}).
112 	 * @see #doFilter(ServletRequest, ServletResponse, FilterChain)
113 	 * @see #invokeDelegate(Filter, ServletRequest, ServletResponse, FilterChain)
114 	 * @see #destroy()
115 	 * @see #setEnvironment(org.springframework.core.env.Environment)
116 	 */
117 	public DelegatingFilterProxy(Filter delegate) {
118 		Assert.notNull(delegate, "delegate Filter object must not be null");
119 		this.delegate = delegate;
120 	}
121 
122 	/**
123 	 * Create a new {@code DelegatingFilterProxy} that will retrieve the named target
124 	 * bean from the Spring {@code WebApplicationContext} found in the {@code ServletContext}
125 	 * (either the 'root' application context or the context named by
126 	 * {@link #setContextAttribute}).
127 	 * <p>For use in Servlet 3.0+ environments where instance-based registration of
128 	 * filters is supported.
129 	 * <p>The target bean must implement the standard Servlet Filter.
130 	 * @param targetBeanName name of the target filter bean to look up in the Spring
131 	 * application context (must not be {@code null}).
132 	 * @see #findWebApplicationContext()
133 	 * @see #setEnvironment(org.springframework.core.env.Environment)
134 	 */
135 	public DelegatingFilterProxy(String targetBeanName) {
136 		this(targetBeanName, null);
137 	}
138 
139 	/**
140 	 * Create a new {@code DelegatingFilterProxy} that will retrieve the named target
141 	 * bean from the given Spring {@code WebApplicationContext}.
142 	 * <p>For use in Servlet 3.0+ environments where instance-based registration of
143 	 * filters is supported.
144 	 * <p>The target bean must implement the standard Servlet Filter interface.
145 	 * <p>The given {@code WebApplicationContext} may or may not be refreshed when passed
146 	 * in. If it has not, and if the context implements {@link ConfigurableApplicationContext},
147 	 * a {@link ConfigurableApplicationContext#refresh() refresh()} will be attempted before
148 	 * retrieving the named target bean.
149 	 * <p>This proxy's {@code Environment} will be inherited from the given
150 	 * {@code WebApplicationContext}.
151 	 * @param targetBeanName name of the target filter bean in the Spring application
152 	 * context (must not be {@code null}).
153 	 * @param wac the application context from which the target filter will be retrieved;
154 	 * if {@code null}, an application context will be looked up from {@code ServletContext}
155 	 * as a fallback.
156 	 * @see #findWebApplicationContext()
157 	 * @see #setEnvironment(org.springframework.core.env.Environment)
158 	 */
159 	public DelegatingFilterProxy(String targetBeanName, WebApplicationContext wac) {
160 		Assert.hasText(targetBeanName, "target Filter bean name must not be null or empty");
161 		this.setTargetBeanName(targetBeanName);
162 		this.webApplicationContext = wac;
163 		if (wac != null) {
164 			this.setEnvironment(wac.getEnvironment());
165 		}
166 	}

169 	 * Set the name of the ServletContext attribute which should be used to retrieve the
170 	 * {@link WebApplicationContext} from which to load the delegate {@link Filter} bean.
171 	 */
172 	public void setContextAttribute(String contextAttribute) {
173 		this.contextAttribute = contextAttribute;
174 	}

177 	 * Return the name of the ServletContext attribute which should be used to retrieve the
178 	 * {@link WebApplicationContext} from which to load the delegate {@link Filter} bean.
179 	 */
180 	public String getContextAttribute() {
181 		return this.contextAttribute;
182 	}

184 	/**
185 	 * Set the name of the target bean in the Spring application context.
186 	 * The target bean must implement the standard Servlet 2.3 Filter interface.
187 	 * <p>By default, the {@code filter-name} as specified for the
188 	 * DelegatingFilterProxy in {@code web.xml} will be used.
189 	 */
190 	public void setTargetBeanName(String targetBeanName) {
191 		this.targetBeanName = targetBeanName;
192 	}
193 
194 	/**
195 	 * Return the name of the target bean in the Spring application context.
196 	 */
197 	protected String getTargetBeanName() {
198 		return this.targetBeanName;
199 	}
200 
201 	/**
202 	 * Set whether to invoke the {@code Filter.init} and
203 	 * {@code Filter.destroy} lifecycle methods on the target bean.
204 	 * <p>Default is "false"; target beans usually rely on the Spring application
205 	 * context for managing their lifecycle. Setting this flag to "true" means
206 	 * that the servlet container will control the lifecycle of the target
207 	 * Filter, with this proxy delegating the corresponding calls.
208 	 */
209 	public void setTargetFilterLifecycle(boolean targetFilterLifecycle) {
210 		this.targetFilterLifecycle = targetFilterLifecycle;
211 	}
212 
213 	/**
214 	 * Return whether to invoke the {@code Filter.init} and
215 	 * {@code Filter.destroy} lifecycle methods on the target bean.
216 	 */
217 	protected boolean isTargetFilterLifecycle() {
218 		return this.targetFilterLifecycle;
219 	}

221 
222 	@Override
223 	protected void initFilterBean() throws ServletException {
224 		synchronized (this.delegateMonitor) {
225 			if (this.delegate == null) {
226 				// If no target bean name specified, use filter name.
227 				if (this.targetBeanName == null) {
228 					this.targetBeanName = getFilterName();
229 				}
230 				// Fetch Spring root application context and initialize the delegate early,
231 				// if possible. If the root application context will be started after this
232 				// filter proxy, we'll have to resort to lazy initialization.
233 				WebApplicationContext wac = findWebApplicationContext();
234 				if (wac != null) {
235 					this.delegate = initDelegate(wac);
236 				}
237 			}
238 		}
239 	}
240 
241 	@Override
242 	public void doFilter(ServletRequest request, ServletResponse response, FilterChain filterChain)
243 			throws ServletException, IOException {
244 
245 		// Lazily initialize the delegate if necessary.
246 		Filter delegateToUse = this.delegate;
247 		if (delegateToUse == null) {
248 			synchronized (this.delegateMonitor) {
249 				if (this.delegate == null) {
250 					WebApplicationContext wac = findWebApplicationContext();
251 					if (wac == null) {
252 						throw new IllegalStateException("No WebApplicationContext found: no ContextLoaderListener registered?");
253 					}
254 					this.delegate = initDelegate(wac);
255 				}
256 				delegateToUse = this.delegate;
257 			}
258 		}
259 
260 		// Let the delegate perform the actual doFilter operation.
261 		invokeDelegate(delegateToUse, request, response, filterChain);
262 	}
263 
264 	@Override
265 	public void destroy() {
266 		Filter delegateToUse = this.delegate;
267 		if (delegateToUse != null) {
268 			destroyDelegate(delegateToUse);
269 		}
270 	}

273 	/**
274 	 * Return the {@code WebApplicationContext} passed in at construction time, if available.
275 	 * Otherwise, attempt to retrieve a {@code WebApplicationContext} from the
276 	 * {@code ServletContext} attribute with the {@linkplain #setContextAttribute
277 	 * configured name} if set. Otherwise look up a {@code WebApplicationContext} under
278 	 * the well-known "root" application context attribute. The
279 	 * {@code WebApplicationContext} must have already been loaded and stored in the
280 	 * {@code ServletContext} before this filter gets initialized (or invoked).
281 	 * <p>Subclasses may override this method to provide a different
282 	 * {@code WebApplicationContext} retrieval strategy.
283 	 * @return the {@code WebApplicationContext} for this proxy, or {@code null} if not found
284 	 * @see #DelegatingFilterProxy(String, WebApplicationContext)
285 	 * @see #getContextAttribute()
286 	 * @see WebApplicationContextUtils#getWebApplicationContext(javax.servlet.ServletContext)
287 	 * @see WebApplicationContext#ROOT_WEB_APPLICATION_CONTEXT_ATTRIBUTE
288 	 */
289 	protected WebApplicationContext findWebApplicationContext() {
290 		if (this.webApplicationContext != null) {
291 			// the user has injected a context at construction time -> use it
292 			if (this.webApplicationContext instanceof ConfigurableApplicationContext) {
293 				if (!((ConfigurableApplicationContext)this.webApplicationContext).isActive()) {
294 					// the context has not yet been refreshed -> do so before returning it
295 					((ConfigurableApplicationContext)this.webApplicationContext).refresh();
296 				}
297 			}
298 			return this.webApplicationContext;
299 		}
300 		String attrName = getContextAttribute();
301 		if (attrName != null) {
302 			return WebApplicationContextUtils.getWebApplicationContext(getServletContext(), attrName);
303 		}
304 		else {
305 			return WebApplicationContextUtils.getWebApplicationContext(getServletContext());
306 		}
307 	}
308 
309 	/**
310 	 * Initialize the Filter delegate, defined as bean the given Spring
311 	 * application context.
312 	 * <p>The default implementation fetches the bean from the application context
313 	 * and calls the standard {@code Filter.init} method on it, passing
314 	 * in the FilterConfig of this Filter proxy.
315 	 * @param wac the root application context
316 	 * @return the initialized delegate Filter
317 	 * @throws ServletException if thrown by the Filter
318 	 * @see #getTargetBeanName()
319 	 * @see #isTargetFilterLifecycle()
320 	 * @see #getFilterConfig()
321 	 * @see javax.servlet.Filter#init(javax.servlet.FilterConfig)
322 	 */
323 	protected Filter initDelegate(WebApplicationContext wac) throws ServletException {
324 		Filter delegate = wac.getBean(getTargetBeanName(), Filter.class);
325 		if (isTargetFilterLifecycle()) {
326 			delegate.init(getFilterConfig());
327 		}
328 		return delegate;
329 	}
330 
331 	/**
332 	 * Actually invoke the delegate Filter with the given request and response.
333 	 * @param delegate the delegate Filter
334 	 * @param request the current HTTP request
335 	 * @param response the current HTTP response
336 	 * @param filterChain the current FilterChain
337 	 * @throws ServletException if thrown by the Filter
338 	 * @throws IOException if thrown by the Filter
339 	 */
340 	protected void invokeDelegate(
341 			Filter delegate, ServletRequest request, ServletResponse response, FilterChain filterChain)
342 			throws ServletException, IOException {
343 
344 		delegate.doFilter(request, response, filterChain);
345 	}
346 
347 	/**
348 	 * Destroy the Filter delegate.
349 	 * Default implementation simply calls {@code Filter.destroy} on it.
350 	 * @param delegate the Filter delegate (never {@code null})
351 	 * @see #isTargetFilterLifecycle()
352 	 * @see javax.servlet.Filter#destroy()
353 	 */
354 	protected void destroyDelegate(Filter delegate) {
355 		if (isTargetFilterLifecycle()) {
356 			delegate.destroy();
357 		}
358 	}
