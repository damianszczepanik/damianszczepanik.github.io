71  	@BeforeClass
72  	public static void startJettyServer() throws Exception {
73  		int port = SocketUtils.findAvailableTcpPort();
74  		jettyServer = new Server(port);
75  		baseUrl = "http://localhost:" + port;
76  		ServletContextHandler handler = new ServletContextHandler();
77  
78  		partsServlet = new PartsServlet();
79  		partListServlet = new PartListServlet();
80  
81  		MultipartConfigElement multipartConfig = new MultipartConfigElement("");
82  
83  		ServletHolder holder = new ServletHolder(partsServlet);
84  		holder.getRegistration().setMultipartConfig(multipartConfig);
85  		handler.addServlet(holder, "/parts");
86  
87  		holder = new ServletHolder(partListServlet);
88  		holder.getRegistration().setMultipartConfig(multipartConfig);
89  		handler.addServlet(holder, "/partlist");
90  		jettyServer.setHandler(handler);
91  		jettyServer.start();
92  	}

94  	@AfterClass
95  	public static void stopJettyServer() throws Exception {
96  		if (jettyServer != null) {
97  			jettyServer.stop();
98  		}

102 	@Test
103 	public void testPartsBinding() {
104 
105 		PartsBean bean = new PartsBean();
106 		partsServlet.setBean(bean);
107 
108 		MultiValueMap<String, Object> parts = new LinkedMultiValueMap<String, Object>();
109 		Resource firstPart = new ClassPathResource("/org/springframework/http/converter/logo.jpg");
110 		parts.add("firstPart", firstPart);
111 		parts.add("secondPart", "secondValue");
112 
113 		template.postForLocation(baseUrl + "/parts", parts);
114 
115 		assertNotNull(bean.getFirstPart());
116 		assertNotNull(bean.getSecondPart());
117 	}

119 	@Test
120 	public void testPartListBinding() {
121 
122 		PartListBean bean = new PartListBean();
123 		partListServlet.setBean(bean);
124 
125 		MultiValueMap<String, Object> parts = new LinkedMultiValueMap<String, Object>();
126 		parts.add("partList", "first value");
127 		parts.add("partList", "second value");
128 		Resource logo = new ClassPathResource("/org/springframework/http/converter/logo.jpg");
129 		parts.add("partList", logo);

131 		template.postForLocation(baseUrl + "/partlist", parts);
132 
133 		assertNotNull(bean.getPartList());

138 	@SuppressWarnings("serial")
139 	private abstract static class AbstractStandardMultipartServlet<T> extends HttpServlet {
140 
141 		private T bean;
142 
143 		@Override
144 		public void service(HttpServletRequest request, HttpServletResponse response) throws
145 				ServletException, IOException {
146 
147 			WebRequestDataBinder binder = new WebRequestDataBinder(bean);
148 			ServletWebRequest webRequest = new ServletWebRequest(request, response);
149 
150 			binder.bind(webRequest);
151 
152 			response.setStatus(HttpServletResponse.SC_OK);
153 		}
154 
155 		public void setBean(T bean) {
156 			this.bean = bean;
157 		}
158 	}
