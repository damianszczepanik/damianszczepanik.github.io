17  package org.springframework.web.client;
18  
19  import java.io.IOException;
20  import java.io.InputStream;
21  import java.io.PushbackInputStream;
22  
23  import org.springframework.http.HttpHeaders;
24  import org.springframework.http.HttpStatus;
25  import org.springframework.http.client.ClientHttpResponse;
26  
27  /**
28   * Implementation of {@link ClientHttpResponse} that can not only check if
29   * the response has a message body, but also if its length is 0 (i.e. empty)
30   * by actually reading the input stream.
31   *
32   * @author Brian Clozel
33   * @since 4.1.5
34   * @see <a href="http://tools.ietf.org/html/rfc7230#section-3.3.3">rfc7230 Section 3.3.3</a>
35   */
36  class MessageBodyClientHttpResponseWrapper implements ClientHttpResponse {
37  
38  	private final ClientHttpResponse response;
39  
40  	private PushbackInputStream pushbackInputStream;
41  
42  
43  	public MessageBodyClientHttpResponseWrapper(ClientHttpResponse response) throws IOException {
44  		this.response = response;
45  	}
46  
47  
48  	/**
49  	 * Indicates whether the response has a message body.
50  	 * <p>Implementation returns {@code false} for:
51  	 * <ul>
52  	 * <li>a response status of {@code 1XX}, {@code 204} or {@code 304}</li>
53  	 * <li>a {@code Content-Length} header of {@code 0}</li>
54  	 * </ul>
55  	 * @return {@code true} if the response has a message body, {@code false} otherwise
56  	 * @throws IOException in case of I/O errors
57  	 */
58  	public boolean hasMessageBody() throws IOException {
59  		HttpStatus responseStatus = this.getStatusCode();
60  		if (responseStatus.is1xxInformational() || responseStatus == HttpStatus.NO_CONTENT ||
61  				responseStatus == HttpStatus.NOT_MODIFIED) {
62  			return false;
63  		}
64  		else if(this.getHeaders().getContentLength() == 0) {
65  			return false;
66  		}
67  		return true;
68  	}
69  
70  	/**
71  	 * Indicates whether the response has an empty message body.
72  	 * <p>Implementation tries to read the first bytes of the response stream:
73  	 * <ul>
74  	 * <li>if no bytes are available, the message body is empty</li>
75  	 * <li>otherwise it is not empty and the stream is reset to its start for further reading</li>
76  	 * </ul>
77  	 * @return {@code true} if the response has a zero-length message body, {@code false} otherwise
78  	 * @throws IOException in case of I/O errors
79  	 */
80  	public boolean hasEmptyMessageBody() throws IOException {
81  		InputStream body = this.response.getBody();
82  		if (body == null) {
83  			return true;
84  		}
85  		else if (body.markSupported()) {
86  			body.mark(1);
87  			if (body.read() == -1) {
88  				return true;
89  			}
90  			else {
91  				body.reset();
92  				return false;
93  			}
94  		}
95  		else {
96  			this.pushbackInputStream = new PushbackInputStream(body);
97  			int b = this.pushbackInputStream.read();
98  			if (b == -1) {
99  				return true;
100 			}
101 			else {
102 				this.pushbackInputStream.unread(b);
103 				return false;
104 			}
105 		}
106 	}
107 
108 
109 	@Override
110 	public HttpHeaders getHeaders() {
111 		return this.response.getHeaders();
112 	}
113 
114 	@Override
115 	public InputStream getBody() throws IOException {
116 		return (this.pushbackInputStream != null ? this.pushbackInputStream : this.response.getBody());
117 	}
118 
119 	@Override
120 	public HttpStatus getStatusCode() throws IOException {
121 		return this.response.getStatusCode();
122 	}
123 
124 	@Override
125 	public int getRawStatusCode() throws IOException {
126 		return this.response.getRawStatusCode();
127 	}
128 
129 	@Override
130 	public String getStatusText() throws IOException {
131 		return this.response.getStatusText();
132 	}
133 
134 	@Override
135 	public void close() {
136 		this.response.close();
137 	}
138 
139 }
