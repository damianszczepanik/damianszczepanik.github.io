17  package org.springframework.http.server;
18  
19  import java.io.IOException;
20  import java.io.OutputStream;
21  import java.util.ArrayList;
22  import java.util.Collection;
23  import java.util.List;
24  import java.util.Map;
25  import javax.servlet.http.HttpServletResponse;
26  
27  import org.springframework.http.HttpHeaders;
28  import org.springframework.http.HttpStatus;
29  import org.springframework.util.Assert;
30  import org.springframework.util.ClassUtils;
31  import org.springframework.util.CollectionUtils;
32  
33  /**
34   * {@link ServerHttpResponse} implementation that is based on a {@link HttpServletResponse}.
35   *
36   * @author Arjen Poutsma
37   * @author Rossen Stoyanchev
38   * @since 3.0
39   */
40  public class ServletServerHttpResponse implements ServerHttpResponse {
41  
42  	private static final boolean servlet3Present =
43  			ClassUtils.hasMethod(HttpServletResponse.class, "getHeader", String.class);
44  
45  
46  	private final HttpServletResponse servletResponse;
47  
48  	private final HttpHeaders headers;
49  
50  	private boolean headersWritten = false;
51  
52  
53  	/**
54  	 * Construct a new instance of the ServletServerHttpResponse based on the given {@link HttpServletResponse}.
55  	 * @param servletResponse the servlet response
56  	 */
57  	public ServletServerHttpResponse(HttpServletResponse servletResponse) {
58  		Assert.notNull(servletResponse, "'servletResponse' must not be null");
59  		this.servletResponse = servletResponse;
60  		this.headers = (servlet3Present ? new ServletResponseHttpHeaders() : new HttpHeaders());
61  	}
62  
63  
64  	/**
65  	 * Return the {@code HttpServletResponse} this object is based on.
66  	 */
67  	public HttpServletResponse getServletResponse() {
68  		return this.servletResponse;
69  	}
70  
71  	@Override
72  	public void setStatusCode(HttpStatus status) {
73  		this.servletResponse.setStatus(status.value());
74  	}
75  
76  	@Override
77  	public HttpHeaders getHeaders() {
78  		return (this.headersWritten ? HttpHeaders.readOnlyHttpHeaders(this.headers) : this.headers);
79  	}
80  
81  	@Override
82  	public OutputStream getBody() throws IOException {
83  		writeHeaders();
84  		return this.servletResponse.getOutputStream();
85  	}
86  
87  	@Override
88  	public void flush() throws IOException {
89  		writeHeaders();
90  		this.servletResponse.flushBuffer();
91  	}
92  
93  	@Override
94  	public void close() {
95  		writeHeaders();
96  	}
97  
98  	private void writeHeaders() {
99  		if (!this.headersWritten) {
100 			for (Map.Entry<String, List<String>> entry : this.headers.entrySet()) {
101 				String headerName = entry.getKey();
102 				for (String headerValue : entry.getValue()) {
103 					this.servletResponse.addHeader(headerName, headerValue);
104 				}
105 			}
106 			// HttpServletResponse exposes some headers as properties: we should include those if not already present
107 			if (this.servletResponse.getContentType() == null && this.headers.getContentType() != null) {
108 				this.servletResponse.setContentType(this.headers.getContentType().toString());
109 			}
110 			if (this.servletResponse.getCharacterEncoding() == null && this.headers.getContentType() != null &&
111 					this.headers.getContentType().getCharSet() != null) {
112 				this.servletResponse.setCharacterEncoding(this.headers.getContentType().getCharSet().name());
113 			}
114 			this.headersWritten = true;
115 		}
116 	}
117 
118 
119 	/**
120 	 * Extends HttpHeaders with the ability to look up headers already present in
121 	 * the underlying HttpServletResponse.
122 	 *
123 	 * <p>The intent is merely to expose what is available through the HttpServletResponse
124 	 * i.e. the ability to look up specific header values by name. All other
125 	 * map-related operations (e.g. iteration, removal, etc) apply only to values
126 	 * added directly through HttpHeaders methods.
127 	 *
128 	 * @since 4.0.3
129 	 */
130 	private class ServletResponseHttpHeaders extends HttpHeaders {
131 
132 		private static final long serialVersionUID = 3410708522401046302L;
133 
134 		@Override
135 		public String getFirst(String headerName) {
136 			String value = servletResponse.getHeader(headerName);
137 			if (value != null) {
138 				return value;
139 			}
140 			else {
141 				return super.getFirst(headerName);
142 			}
143 		}
144 
145 		@Override
146 		public List<String> get(Object key) {
147 			Assert.isInstanceOf(String.class, key, "Key must be a String-based header name");
148 
149 			Collection<String> values1 = servletResponse.getHeaders((String) key);
150 			boolean isEmpty1 = CollectionUtils.isEmpty(values1);
151 
152 			List<String> values2 = super.get(key);
153 			boolean isEmpty2 = CollectionUtils.isEmpty(values2);
154 
155 			if (isEmpty1 && isEmpty2) {
156 				return null;
157 			}
158 
159 			List<String> values = new ArrayList<String>();
160 			if (!isEmpty1) {
161 				values.addAll(values1);
162 			}
163 			if (!isEmpty2) {
164 				values.addAll(values2);
165 			}
166 			return values;
167 		}
168 	}
