613 	@Test
614 	public void exchange() throws Exception {
615 		given(converter.canRead(Integer.class, null)).willReturn(true);
616 		given(converter.getSupportedMediaTypes()).willReturn(Collections.singletonList(MediaType.TEXT_PLAIN));
617 		given(requestFactory.createRequest(new URI("http://example.com"), HttpMethod.POST)).willReturn(this.request);
618 		HttpHeaders requestHeaders = new HttpHeaders();
619 		given(this.request.getHeaders()).willReturn(requestHeaders);
620 		given(converter.canWrite(String.class, null)).willReturn(true);
621 		String body = "Hello World";
622 		converter.write(body, null, this.request);
623 		given(this.request.execute()).willReturn(response);
624 		given(errorHandler.hasError(response)).willReturn(false);
625 		Integer expected = 42;
626 		HttpHeaders responseHeaders = new HttpHeaders();
627 		responseHeaders.setContentType(MediaType.TEXT_PLAIN);
628 		responseHeaders.setContentLength(10);
629 		given(response.getStatusCode()).willReturn(HttpStatus.OK);
630 		given(response.getHeaders()).willReturn(responseHeaders);
631 		given(response.getBody()).willReturn(new ByteArrayInputStream(expected.toString().getBytes()));
632 		given(converter.canRead(Integer.class, MediaType.TEXT_PLAIN)).willReturn(true);
633 		given(converter.read(Integer.class, response)).willReturn(expected);
634 		given(converter.read(eq(Integer.class), any(HttpInputMessage.class))).willReturn(expected);
635 		given(response.getStatusCode()).willReturn(HttpStatus.OK);
636 		HttpStatus status = HttpStatus.OK;
637 		given(response.getStatusCode()).willReturn(status);
638 		given(response.getStatusText()).willReturn(status.getReasonPhrase());
639 
640 		HttpHeaders entityHeaders = new HttpHeaders();
641 		entityHeaders.set("MyHeader", "MyValue");
642 		HttpEntity<String> requestEntity = new HttpEntity<String>(body, entityHeaders);
643 		ResponseEntity<Integer> result = template.exchange("http://example.com", HttpMethod.POST, requestEntity, Integer.class);
644 		assertEquals("Invalid POST result", expected, result.getBody());
645 		assertEquals("Invalid Content-Type", MediaType.TEXT_PLAIN, result.getHeaders().getContentType());
646 		assertEquals("Invalid Accept header", MediaType.TEXT_PLAIN_VALUE, requestHeaders.getFirst("Accept"));
647 		assertEquals("Invalid custom header", "MyValue", requestHeaders.getFirst("MyHeader"));
648 		assertEquals("Invalid status code", HttpStatus.OK, result.getStatusCode());
649 
650 		verify(response).close();
651 	}
