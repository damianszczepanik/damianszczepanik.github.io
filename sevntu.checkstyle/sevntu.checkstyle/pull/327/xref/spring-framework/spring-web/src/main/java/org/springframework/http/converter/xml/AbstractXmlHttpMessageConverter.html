17  package org.springframework.http.converter.xml;
18  
19  import java.io.IOException;
20  import javax.xml.transform.Result;
21  import javax.xml.transform.Source;
22  import javax.xml.transform.TransformerException;
23  import javax.xml.transform.TransformerFactory;
24  import javax.xml.transform.stream.StreamResult;
25  import javax.xml.transform.stream.StreamSource;
26  
27  import org.springframework.http.HttpHeaders;
28  import org.springframework.http.HttpInputMessage;
29  import org.springframework.http.HttpOutputMessage;
30  import org.springframework.http.MediaType;
31  import org.springframework.http.converter.AbstractHttpMessageConverter;
32  import org.springframework.http.converter.HttpMessageConversionException;
33  
34  /**
35   * Abstract base class for {@link org.springframework.http.converter.HttpMessageConverter HttpMessageConverters}
36   * that convert from/to XML.
37   *
38   * <p>By default, subclasses of this converter support {@code text/xml}, {@code application/xml}, and {@code
39   * application/*-xml}. This can be overridden by setting the {@link #setSupportedMediaTypes(java.util.List)
40   * supportedMediaTypes} property.
41   *
42   * @author Arjen Poutsma
43   * @since 3.0
44   */
45  public abstract class AbstractXmlHttpMessageConverter<T> extends AbstractHttpMessageConverter<T> {
46  
47  	private final TransformerFactory transformerFactory = TransformerFactory.newInstance();
48  
49  
50  	/**
51  	 * Protected constructor that sets the {@link #setSupportedMediaTypes(java.util.List) supportedMediaTypes}
52  	 * to {@code text/xml} and {@code application/xml}, and {@code application/*-xml}.
53  	 */
54  	protected AbstractXmlHttpMessageConverter() {
55  		super(MediaType.APPLICATION_XML, MediaType.TEXT_XML, new MediaType("application", "*+xml"));
56  	}
57  
58  
59  	@Override
60  	public final T readInternal(Class<? extends T> clazz, HttpInputMessage inputMessage) throws IOException {
61  		return readFromSource(clazz, inputMessage.getHeaders(), new StreamSource(inputMessage.getBody()));
62  	}
63  
64  	@Override
65  	protected final void writeInternal(T t, HttpOutputMessage outputMessage) throws IOException {
66  		writeToResult(t, outputMessage.getHeaders(), new StreamResult(outputMessage.getBody()));
67  	}
68  
69  	/**
70  	 * Transforms the given {@code Source} to the {@code Result}.
71  	 * @param source the source to transform from
72  	 * @param result the result to transform to
73  	 * @throws TransformerException in case of transformation errors
74  	 */
75  	protected void transform(Source source, Result result) throws TransformerException {
76  		this.transformerFactory.newTransformer().transform(source, result);
77  	}
78  
79  
80  	/**
81  	 * Abstract template method called from {@link #read(Class, HttpInputMessage)}.
82  	 * @param clazz the type of object to return
83  	 * @param headers the HTTP input headers
84  	 * @param source the HTTP input body
85  	 * @return the converted object
86  	 * @throws IOException in case of I/O errors
87  	 * @throws org.springframework.http.converter.HttpMessageConversionException in case of conversion errors
88  	 */
89  	protected abstract T readFromSource(Class<? extends T> clazz, HttpHeaders headers, Source source)
90  			throws IOException;
91  
92  	/**
93  	 * Abstract template method called from {@link #writeInternal(Object, HttpOutputMessage)}.
94  	 * @param t the object to write to the output message
95  	 * @param headers the HTTP output headers
96  	 * @param result the HTTP output body
97  	 * @throws IOException in case of I/O errors
98  	 * @throws HttpMessageConversionException in case of conversion errors
99  	 */
100 	protected abstract void writeToResult(T t, HttpHeaders headers, Result result)
101 			throws IOException;
102 
103 }
