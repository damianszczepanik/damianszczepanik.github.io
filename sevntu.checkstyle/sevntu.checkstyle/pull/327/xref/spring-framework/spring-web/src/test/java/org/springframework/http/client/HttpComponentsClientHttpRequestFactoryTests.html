52  	@Test
53  	public void assertLegacyCustomConfig() {
54  		HttpClient httpClient = new org.apache.http.impl.client.DefaultHttpClient(); // Does not support RequestConfig
55  		HttpComponentsClientHttpRequestFactory hrf = new HttpComponentsClientHttpRequestFactory(httpClient);
56  		hrf.setConnectTimeout(1234);
57  		assertEquals(1234, httpClient.getParams().getIntParameter(
58  				org.apache.http.params.CoreConnectionPNames.CONNECTION_TIMEOUT, 0));
59  
60  		hrf.setReadTimeout(4567);
61  		assertEquals(4567, httpClient.getParams().getIntParameter(
62  				org.apache.http.params.CoreConnectionPNames.SO_TIMEOUT, 0));
63  	}

65  	@Test
66  	public void assertCustomConfig() throws Exception {
67  		HttpClient httpClient = HttpClientBuilder.create().build();
68  		HttpComponentsClientHttpRequestFactory hrf = new HttpComponentsClientHttpRequestFactory(httpClient);
69  		hrf.setConnectTimeout(1234);
70  		hrf.setConnectionRequestTimeout(4321);
71  		hrf.setReadTimeout(4567);
72  
73  		URI uri = new URI(baseUrl + "/status/ok");
74  		HttpComponentsClientHttpRequest request = (HttpComponentsClientHttpRequest)
75  				hrf.createRequest(uri, HttpMethod.GET);
76  
77  		Object config = request.getHttpContext().getAttribute(HttpClientContext.REQUEST_CONFIG);
78  		assertNotNull("Request config should be set", config);
79  		assertTrue("Wrong request config type" + config.getClass().getName(),
80  				RequestConfig.class.isInstance(config));
81  		RequestConfig requestConfig = (RequestConfig) config;
82  		assertEquals("Wrong custom connection timeout", 1234, requestConfig.getConnectTimeout());
83  		assertEquals("Wrong custom connection request timeout", 4321, requestConfig.getConnectionRequestTimeout());
84  		assertEquals("Wrong custom socket timeout", 4567, requestConfig.getSocketTimeout());
85  	}

87  	@Test
88  	public void defaultSettingsOfHttpClientMergedOnExecutorCustomization() throws Exception {
89  		RequestConfig defaultConfig = RequestConfig.custom().setConnectTimeout(1234).build();
90  		CloseableHttpClient client = mock(CloseableHttpClient.class,
91  				withSettings().extraInterfaces(Configurable.class));
92  		Configurable configurable = (Configurable) client;
93  		when(configurable.getConfig()).thenReturn(defaultConfig);
94  
95  		HttpComponentsClientHttpRequestFactory hrf = new HttpComponentsClientHttpRequestFactory(client);
96  		assertSame("Default client configuration is expected", defaultConfig, retrieveRequestConfig(hrf));
97  
98  		hrf.setConnectionRequestTimeout(4567);
99  		RequestConfig requestConfig = retrieveRequestConfig(hrf);
100 		assertNotNull(requestConfig);
101 		assertEquals(4567, requestConfig.getConnectionRequestTimeout());
102 		// Default connection timeout merged
103 		assertEquals(1234, requestConfig.getConnectTimeout());
104 	}

106 	@Test
107 	public void localSettingsOverrideClientDefaultSettings() throws Exception {
108 		RequestConfig defaultConfig = RequestConfig.custom()
109 				.setConnectTimeout(1234).setConnectionRequestTimeout(6789).build();
110 		CloseableHttpClient client = mock(CloseableHttpClient.class,
111 				withSettings().extraInterfaces(Configurable.class));
112 		Configurable configurable = (Configurable) client;
113 		when(configurable.getConfig()).thenReturn(defaultConfig);
114 
115 		HttpComponentsClientHttpRequestFactory hrf = new HttpComponentsClientHttpRequestFactory(client);
116 		hrf.setConnectTimeout(5000);
117 
118 		RequestConfig requestConfig = retrieveRequestConfig(hrf);
119 		assertEquals(5000, requestConfig.getConnectTimeout());
120 		assertEquals(6789, requestConfig.getConnectionRequestTimeout());
121 		assertEquals(-1, requestConfig.getSocketTimeout());
122 	}

124 	@Test
125 	public void mergeBasedOnCurrentHttpClient() throws Exception {
126 		RequestConfig defaultConfig = RequestConfig.custom()
127 				.setSocketTimeout(1234).build();
128 		final CloseableHttpClient client = mock(CloseableHttpClient.class,
129 				withSettings().extraInterfaces(Configurable.class));
130 		Configurable configurable = (Configurable) client;
131 		when(configurable.getConfig()).thenReturn(defaultConfig);
132 
133 		HttpComponentsClientHttpRequestFactory hrf = new HttpComponentsClientHttpRequestFactory() {
134 			@Override
135 			public HttpClient getHttpClient() {
136 				return client;
137 			}
138 		};
139 		hrf.setReadTimeout(5000);
140 
141 		RequestConfig requestConfig = retrieveRequestConfig(hrf);
142 		assertEquals(-1, requestConfig.getConnectTimeout());
143 		assertEquals(-1, requestConfig.getConnectionRequestTimeout());
144 		assertEquals(5000, requestConfig.getSocketTimeout());
145 
146 		// Update the Http client so that it returns an updated  config
147 		RequestConfig updatedDefaultConfig = RequestConfig.custom()
148 				.setConnectTimeout(1234).build();
149 		when(configurable.getConfig()).thenReturn(updatedDefaultConfig);
150 		hrf.setReadTimeout(7000);
151 		RequestConfig requestConfig2 = retrieveRequestConfig(hrf);
152 		assertEquals(1234, requestConfig2.getConnectTimeout());
153 		assertEquals(-1, requestConfig2.getConnectionRequestTimeout());
154 		assertEquals(7000, requestConfig2.getSocketTimeout());
155 	}

157 	private RequestConfig retrieveRequestConfig(HttpComponentsClientHttpRequestFactory factory) throws Exception {
158 		URI uri = new URI(baseUrl + "/status/ok");
159 		HttpComponentsClientHttpRequest request = (HttpComponentsClientHttpRequest)
160 				factory.createRequest(uri, HttpMethod.GET);
161 		return (RequestConfig) request.getHttpContext().getAttribute(HttpClientContext.REQUEST_CONFIG);
162 	}

164 	@Test
165 	public void createHttpUriRequest() throws Exception {
166 		URI uri = new URI("http://example.com");
167 		testRequestBodyAllowed(uri, HttpMethod.GET, false);
168 		testRequestBodyAllowed(uri, HttpMethod.HEAD, false);
169 		testRequestBodyAllowed(uri, HttpMethod.OPTIONS, false);
170 		testRequestBodyAllowed(uri, HttpMethod.TRACE, false);
171 		testRequestBodyAllowed(uri, HttpMethod.PUT, true);
172 		testRequestBodyAllowed(uri, HttpMethod.POST, true);
173 		testRequestBodyAllowed(uri, HttpMethod.PATCH, true);
174 		testRequestBodyAllowed(uri, HttpMethod.DELETE, true);
175 
176 	}

178 	private void testRequestBodyAllowed(URI uri, HttpMethod method, boolean allowed) {
179 		HttpUriRequest request = ((HttpComponentsClientHttpRequestFactory) this.factory).createHttpUriRequest(method, uri);
180 		assertEquals(allowed, request instanceof HttpEntityEnclosingRequest);
181 	}
