51  /**
52   * Resolves method arguments annotated with @{@link RequestParam}, arguments of
53   * type {@link MultipartFile} in conjunction with Spring's {@link MultipartResolver}
54   * abstraction, and arguments of type {@code javax.servlet.http.Part} in conjunction
55   * with Servlet 3.0 multipart requests. This resolver can also be created in default
56   * resolution mode in which simple types (int, long, etc.) not annotated
57   * with @{@link RequestParam} are also treated as request parameters with the
58   * parameter name derived from the argument name.
59   *
60   * <p>If the method parameter type is {@link Map}, the name specified in the
61   * annotation is used to resolve the request parameter String value. The value is
62   * then converted to a {@link Map} via type conversion assuming a suitable
63   * {@link Converter} or {@link PropertyEditor} has been registered.
64   * Or if a request parameter name is not specified the
65   * {@link RequestParamMapMethodArgumentResolver} is used instead to provide
66   * access to all request parameters in the form of a map.
67   *
68   * <p>A {@link WebDataBinder} is invoked to apply type conversion to resolved request
69   * header values that don't yet match the method parameter type.
70   *
71   * @author Arjen Poutsma
72   * @author Rossen Stoyanchev
73   * @author Brian Clozel
74   * @since 3.1
75   * @see RequestParamMapMethodArgumentResolver
76   */
77  public class RequestParamMethodArgumentResolver extends AbstractNamedValueMethodArgumentResolver
78  		implements UriComponentsContributor {
79  
80  	private static final TypeDescriptor STRING_TYPE_DESCRIPTOR = TypeDescriptor.valueOf(String.class);
81  
82  	private final boolean useDefaultResolution;
83  
84  
85  	/**
86  	 * @param useDefaultResolution in default resolution mode a method argument
87  	 * that is a simple type, as defined in {@link BeanUtils#isSimpleProperty},
88  	 * is treated as a request parameter even if it it isn't annotated, the
89  	 * request parameter name is derived from the method parameter name.
90  	 */
91  	public RequestParamMethodArgumentResolver(boolean useDefaultResolution) {
92  		this.useDefaultResolution = useDefaultResolution;
93  	}
94  
95  	/**
96  	 * @param beanFactory a bean factory used for resolving  ${...} placeholder
97  	 * and #{...} SpEL expressions in default values, or {@code null} if default
98  	 * values are not expected to contain expressions
99  	 * @param useDefaultResolution in default resolution mode a method argument
100 	 * that is a simple type, as defined in {@link BeanUtils#isSimpleProperty},
101 	 * is treated as a request parameter even if it it isn't annotated, the
102 	 * request parameter name is derived from the method parameter name.
103 	 */
104 	public RequestParamMethodArgumentResolver(ConfigurableBeanFactory beanFactory, boolean useDefaultResolution) {
105 		super(beanFactory);
106 		this.useDefaultResolution = useDefaultResolution;
107 	}

110 	/**
111 	 * Supports the following:
112 	 * <ul>
113 	 * <li>@RequestParam-annotated method arguments.
114 	 * This excludes {@link Map} params where the annotation doesn't
115 	 * specify a name.	See {@link RequestParamMapMethodArgumentResolver}
116 	 * instead for such params.
117 	 * <li>Arguments of type {@link MultipartFile}
118 	 * unless annotated with @{@link RequestPart}.
119 	 * <li>Arguments of type {@code javax.servlet.http.Part}
120 	 * unless annotated with @{@link RequestPart}.
121 	 * <li>In default resolution mode, simple type arguments
122 	 * even if not with @{@link RequestParam}.
123 	 * </ul>
124 	 */
125 	@Override
126 	public boolean supportsParameter(MethodParameter parameter) {
127 		Class<?> paramType = parameter.getParameterType();
128 		if (parameter.hasParameterAnnotation(RequestParam.class)) {
129 			if (Map.class.isAssignableFrom(paramType)) {
130 				String paramName = parameter.getParameterAnnotation(RequestParam.class).value();
131 				return StringUtils.hasText(paramName);
132 			}
133 			else {
134 				return true;
135 			}
136 		}
137 		else {
138 			if (parameter.hasParameterAnnotation(RequestPart.class)) {
139 				return false;
140 			}
141 			else if (MultipartFile.class.equals(paramType) || "javax.servlet.http.Part".equals(paramType.getName())) {
142 				return true;
143 			}
144 			else if (this.useDefaultResolution) {
145 				return BeanUtils.isSimpleProperty(paramType);
146 			}
147 			else {
148 				return false;
149 			}
150 		}
151 	}

153 	@Override
154 	protected NamedValueInfo createNamedValueInfo(MethodParameter parameter) {
155 		RequestParam ann = parameter.getParameterAnnotation(RequestParam.class);
156 		return (ann != null ? new RequestParamNamedValueInfo(ann) : new RequestParamNamedValueInfo());
157 	}

159 	@Override
160 	protected Object resolveName(String name, MethodParameter parameter, NativeWebRequest webRequest) throws Exception {
161 		HttpServletRequest servletRequest = webRequest.getNativeRequest(HttpServletRequest.class);
162 		MultipartHttpServletRequest multipartRequest =
163 				WebUtils.getNativeRequest(servletRequest, MultipartHttpServletRequest.class);
164 		Object arg;
165 
166 		if (MultipartFile.class.equals(parameter.getParameterType())) {
167 			assertIsMultipartRequest(servletRequest);
168 			Assert.notNull(multipartRequest, "Expected MultipartHttpServletRequest: is a MultipartResolver configured?");
169 			arg = multipartRequest.getFile(name);
170 		}
171 		else if (isMultipartFileCollection(parameter)) {
172 			assertIsMultipartRequest(servletRequest);
173 			Assert.notNull(multipartRequest, "Expected MultipartHttpServletRequest: is a MultipartResolver configured?");
174 			arg = multipartRequest.getFiles(name);
175 		}
176 		else if (isMultipartFileArray(parameter)) {
177 			assertIsMultipartRequest(servletRequest);
178 			Assert.notNull(multipartRequest, "Expected MultipartHttpServletRequest: is a MultipartResolver configured?");
179 			List<MultipartFile> multipartFiles = multipartRequest.getFiles(name);
180 			arg = multipartFiles.toArray(new MultipartFile[multipartFiles.size()]);
181 		}
182 		else if ("javax.servlet.http.Part".equals(parameter.getParameterType().getName())) {
183 			assertIsMultipartRequest(servletRequest);
184 			arg = servletRequest.getPart(name);
185 		}
186 		else if (isPartCollection(parameter)) {
187 			assertIsMultipartRequest(servletRequest);
188 			arg = new ArrayList<Object>(servletRequest.getParts());
189 		}
190 		else if (isPartArray(parameter)) {
191 			assertIsMultipartRequest(servletRequest);
192 			arg = RequestPartResolver.resolvePart(servletRequest);
193 		}
194 		else {
195 			arg = null;
196 			if (multipartRequest != null) {
197 				List<MultipartFile> files = multipartRequest.getFiles(name);
198 				if (!files.isEmpty()) {
199 					arg = (files.size() == 1 ? files.get(0) : files);
200 				}
201 			}
202 			if (arg == null) {
203 				String[] paramValues = webRequest.getParameterValues(name);
204 				if (paramValues != null) {
205 					arg = paramValues.length == 1 ? paramValues[0] : paramValues;
206 				}
207 			}
208 		}
209 
210 		return arg;
211 	}

213 	private void assertIsMultipartRequest(HttpServletRequest request) {
214 		String contentType = request.getContentType();
215 		if (contentType == null || !contentType.toLowerCase().startsWith("multipart/")) {
216 			throw new MultipartException("The current request is not a multipart request");
217 		}
218 	}

220 	private boolean isMultipartFileCollection(MethodParameter parameter) {
221 		Class<?> collectionType = getCollectionParameterType(parameter);
222 		return ((collectionType != null) && collectionType.equals(MultipartFile.class));
223 	}
224 
225 	private boolean isPartCollection(MethodParameter parameter) {
226 		Class<?> collectionType = getCollectionParameterType(parameter);
227 		return ((collectionType != null) && "javax.servlet.http.Part".equals(collectionType.getName()));
228 	}
229 
230 	private boolean isPartArray(MethodParameter parameter) {
231 		Class<?> paramType = parameter.getParameterType().getComponentType();
232 		return ((paramType != null) && "javax.servlet.http.Part".equals(paramType.getName()));
233 	}
234 
235 	private boolean isMultipartFileArray(MethodParameter parameter) {
236 		Class<?> paramType = parameter.getParameterType().getComponentType();
237 		return ((paramType != null) && MultipartFile.class.equals(paramType));
238 	}
239 
240 	private Class<?> getCollectionParameterType(MethodParameter parameter) {
241 		Class<?> paramType = parameter.getParameterType();
242 		if (Collection.class.equals(paramType) || List.class.isAssignableFrom(paramType)){
243 			Class<?> valueType = GenericCollectionTypeResolver.getCollectionParameterType(parameter);
244 			if (valueType != null) {
245 				return valueType;
246 			}
247 		}
248 		return null;
249 	}

251 	@Override
252 	protected void handleMissingValue(String name, MethodParameter parameter) throws ServletException {
253 		throw new MissingServletRequestParameterException(name, parameter.getParameterType().getSimpleName());
254 	}
255 
256 	@Override
257 	public void contributeMethodArgument(MethodParameter parameter, Object value,
258 			UriComponentsBuilder builder, Map<String, Object> uriVariables, ConversionService conversionService) {
259 
260 		Class<?> paramType = parameter.getParameterType();
261 		if (Map.class.isAssignableFrom(paramType) || MultipartFile.class.equals(paramType) ||
262 				"javax.servlet.http.Part".equals(paramType.getName())) {
263 			return;
264 		}
265 
266 		RequestParam ann = parameter.getParameterAnnotation(RequestParam.class);
267 		String name = (ann == null || StringUtils.isEmpty(ann.value()) ? parameter.getParameterName() : ann.value());
268 
269 		if (value == null) {
270 			builder.queryParam(name);
271 		}
272 		else if (value instanceof Collection) {
273 			for (Object element : (Collection<?>) value) {
274 				element = formatUriValue(conversionService, TypeDescriptor.nested(parameter, 1), element);
275 				builder.queryParam(name, element);
276 			}
277 		}
278 		else {
279 			builder.queryParam(name, formatUriValue(conversionService, new TypeDescriptor(parameter), value));
280 		}
281 	}
282 
283 	protected String formatUriValue(ConversionService cs, TypeDescriptor sourceType, Object value) {
284 		if (value == null) {
285 			return null;
286 		}
287 		else if (value instanceof String) {
288 			return (String) value;
289 		}
290 		else if (cs != null) {
291 			return (String) cs.convert(value, sourceType, STRING_TYPE_DESCRIPTOR);
292 		}
293 		else {
294 			return value.toString();
295 		}
296 	}

313 		public static Object resolvePart(HttpServletRequest servletRequest) throws Exception {
