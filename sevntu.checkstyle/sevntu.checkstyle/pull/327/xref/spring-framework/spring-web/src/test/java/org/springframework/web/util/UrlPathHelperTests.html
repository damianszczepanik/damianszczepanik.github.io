126 	@Test
127 	public void getRequestKeepSemicolonContent() throws UnsupportedEncodingException {
128 		helper.setRemoveSemicolonContent(false);
129 
130 		request.setRequestURI("/foo;a=b;c=d");
131 		assertEquals("/foo;a=b;c=d", helper.getRequestUri(request));
132 
133 		request.setRequestURI("/foo;jsessionid=c0o7fszeb1");
134 		assertEquals("jsessionid should always be removed", "/foo", helper.getRequestUri(request));
135 
136 		request.setRequestURI("/foo;a=b;jsessionid=c0o7fszeb1;c=d");
137 		assertEquals("jsessionid should always be removed", "/foo;a=b;c=d", helper.getRequestUri(request));
138 
139 		// SPR-10398
140 
141 		request.setRequestURI("/foo;a=b;JSESSIONID=c0o7fszeb1;c=d");
142 		assertEquals("JSESSIONID should always be removed", "/foo;a=b;c=d", helper.getRequestUri(request));
143 	}

208 	@Test
209 	public void defaultServletEndingWithDoubleSlash() throws Exception {
210 		request.setContextPath("/SPR-12372");
211 		request.setPathInfo(null);
212 		request.setServletPath("/foo/bar/");
213 		request.setRequestURI("/SPR-12372/foo//bar/");
214 
215 		assertEquals("/foo//bar/", helper.getLookupPathForRequest(request));
216 
217 		request.setServletPath("/foo/bar/");
218 		request.setRequestURI("/SPR-12372/foo/bar//");
219 
220 		assertEquals("/foo/bar//", helper.getLookupPathForRequest(request));
221 
222 		// "normal" case
223 		request.setServletPath("/foo/bar//");
224 		request.setRequestURI("/SPR-12372/foo/bar//");
225 
226 		assertEquals("/foo/bar//", helper.getLookupPathForRequest(request));
227 	}
