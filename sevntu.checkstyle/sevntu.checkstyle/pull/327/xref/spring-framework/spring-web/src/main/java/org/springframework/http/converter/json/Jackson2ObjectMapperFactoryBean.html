46  /**
47   * A {@link FactoryBean} for creating a Jackson 2.x {@link ObjectMapper} (default) or
48   * {@link XmlMapper} ({@code createXmlMapper} property set to true) with setters
49   * to enable or disable Jackson features from within XML configuration.
50   *
51   * <p>It customizes Jackson defaults properties with the following ones:
52   * <ul>
53   * <li>{@link MapperFeature#DEFAULT_VIEW_INCLUSION} is disabled</li>
54   * <li>{@link DeserializationFeature#FAIL_ON_UNKNOWN_PROPERTIES} is disabled</li>
55   * </ul>
56   *
57   * <p>Example usage with
58   * {@link MappingJackson2HttpMessageConverter}:
59   *
60   * <pre class="code">
61   * &lt;bean class="org.springframework.http.converter.json.MappingJackson2HttpMessageConverter">
62   *   &lt;property name="objectMapper">
63   *     &lt;bean class="org.springframework.http.converter.json.Jackson2ObjectMapperFactoryBean"
64   *       p:autoDetectFields="false"
65   *       p:autoDetectGettersSetters="false"
66   *       p:annotationIntrospector-ref="jaxbAnnotationIntrospector" />
67   *   &lt;/property>
68   * &lt;/bean>
69   * </pre>
70   *
71   * <p>Example usage with MappingJackson2JsonView:
72   *
73   * <pre class="code">
74   * &lt;bean class="org.springframework.web.servlet.view.json.MappingJackson2JsonView">
75   *   &lt;property name="objectMapper">
76   *     &lt;bean class="org.springframework.http.converter.json.Jackson2ObjectMapperFactoryBean"
77   *       p:failOnEmptyBeans="false"
78   *       p:indentOutput="true">
79   *       &lt;property name="serializers">
80   *         &lt;array>
81   *           &lt;bean class="org.mycompany.MyCustomSerializer" />
82   *         &lt;/array>
83   *       &lt;/property>
84   *     &lt;/bean>
85   *   &lt;/property>
86   * &lt;/bean>
87   * </pre>
88   *
89   * <p>In case there are no specific setters provided (for some rarely used options),
90   * you can still use the more general methods  {@link #setFeaturesToEnable} and
91   * {@link #setFeaturesToDisable}.
92   *
93   * <pre class="code">
94   * &lt;bean class="org.springframework.http.converter.json.Jackson2ObjectMapperFactoryBean">
95   *   &lt;property name="featuresToEnable">
96   *     &lt;array>
97   *       &lt;util:constant static-field="com.fasterxml.jackson.databind.SerializationFeature$WRAP_ROOT_VALUE"/>
98   *       &lt;util:constant static-field="com.fasterxml.jackson.databind.SerializationFeature$CLOSE_CLOSEABLE"/>
99   *     &lt;/array>
100  *   &lt;/property>
101  *   &lt;property name="featuresToDisable">
102  *     &lt;array>
103  *       &lt;util:constant static-field="com.fasterxml.jackson.databind.MapperFeature$USE_ANNOTATIONS"/>
104  *     &lt;/array>
105  *   &lt;/property>
106  * &lt;/bean>
107  * </pre>
108  *
109  * <p>In case you want to configure Jackson's {@link ObjectMapper} with a custom {@link Module},
110  * you can register one or more such Modules by class name via {@link #setModulesToInstall}:
111  *
112  * <pre class="code">
113  * &lt;bean class="org.springframework.http.converter.json.Jackson2ObjectMapperFactoryBean">
114  *   &lt;property name="modulesToInstall" value="myapp.jackson.MySampleModule,myapp.jackson.MyOtherModule"/>
115  * &lt;/bean
116  * </pre>
117  *
118  * Note that Jackson's JSR-310 and Joda-Time support modules will be registered automatically
119  * when available (and when Java 8 and Joda-Time themselves are available, respectively).
120  *
121  * <p>Tested against Jackson 2.2, 2.3 and 2.4; compatible with Jackson 2.0 and higher.
122  *
123  * @author <a href="mailto:dmitry.katsubo@gmail.com">Dmitry Katsubo</a>
124  * @author Rossen Stoyanchev
125  * @author Brian Clozel
126  * @author Juergen Hoeller
127  * @author Tadaya Tsuyukubo
128  * @author Sebastien Deleuze
129  * @since 3.2
130  */
131 public class Jackson2ObjectMapperFactoryBean implements FactoryBean<ObjectMapper>, BeanClassLoaderAware,
132 		ApplicationContextAware, InitializingBean {
133 
134 	private final Jackson2ObjectMapperBuilder builder = new Jackson2ObjectMapperBuilder();
135 
136 	private ObjectMapper objectMapper;

139 	/**
140 	 * Set the {@link ObjectMapper} instance to use. If not set, the {@link ObjectMapper}
141 	 * will be created using its default constructor.
142 	 */
143 	public void setObjectMapper(ObjectMapper objectMapper) {
144 		this.objectMapper = objectMapper;
145 	}
146 
147 	/**
148 	 * If set to true and no custom {@link ObjectMapper} has been set, a {@link XmlMapper}
149 	 * will be created using its default constructor.
150 	 * @since 4.1
151 	 */
152 	public void setCreateXmlMapper(boolean createXmlMapper) {
153 		this.builder.createXmlMapper(createXmlMapper);
154 	}

156 	/**
157 	 * Define the format for date/time with the given {@link DateFormat}.
158 	 * <p>Note: Setting this property makes the exposed {@link ObjectMapper}
159 	 * non-thread-safe, according to Jackson's thread safety rules.
160 	 * @see #setSimpleDateFormat(String)
161 	 */
162 	public void setDateFormat(DateFormat dateFormat) {
163 		this.builder.dateFormat(dateFormat);
164 	}
165 
166 	/**
167 	 * Define the date/time format with a {@link SimpleDateFormat}.
168 	 * <p>Note: Setting this property makes the exposed {@link ObjectMapper}
169 	 * non-thread-safe, according to Jackson's thread safety rules.
170 	 * @see #setDateFormat(DateFormat)
171 	 */
172 	public void setSimpleDateFormat(String format) {
173 		this.builder.simpleDateFormat(format);
174 	}
175 
176 	/**
177 	 * Override the default {@link Locale} to use for formatting.
178 	 * Default value used is {@link Locale#getDefault()}.
179 	 * @since 4.1.5
180 	 */
181 	public void setLocale(Locale locale) {
182 		this.builder.locale(locale);
183 	}
184 
185 	/**
186 	 * Override the default {@link TimeZone} to use for formatting.
187 	 * Default value used is UTC (NOT local timezone).
188 	 * @since 4.1.5

191 		this.builder.timeZone(timeZone);
192 	}

194 	/**
195 	 * Set an {@link AnnotationIntrospector} for both serialization and deserialization.

199 	}

203 	 * configure the {@link ObjectMapper} with.

207 		this.builder.propertyNamingStrategy(propertyNamingStrategy);
208 	}

211 	 * Set a custom inclusion strategy for serialization.
212 	 * @see com.fasterxml.jackson.annotation.JsonInclude.Include

215 		this.builder.serializationInclusion(serializationInclusion);
216 	}
217

223 	 */
224 	public void setSerializers(JsonSerializer<?>... serializers) {
225 		this.builder.serializers(serializers);
226 	}

230 	 * @see #setSerializers(JsonSerializer...)

232 	public void setSerializersByType(Map<Class<?>, JsonSerializer<?>> serializers) {
233 		this.builder.serializersByType(serializers);
234 	}
235 
236 	/**
237 	 * Configure custom deserializers for the given types.
238 	 */
239 	public void setDeserializersByType(Map<Class<?>, JsonDeserializer<?>> deserializers) {
240 		this.builder.deserializersByType(deserializers);
241 	}
242 
243 	/**
244 	 * Add mix-in annotations to use for augmenting specified class or interface.
245 	 * @param mixIns Map of entries with target classes (or interface) whose annotations
246 	 * to effectively override as key and mix-in classes (or interface) whose
247 	 * annotations are to be "added" to target's annotations as value.
248 	 * @since 4.1.2
249 	 * @see com.fasterxml.jackson.databind.ObjectMapper#addMixInAnnotations(Class, Class)
250 	 */
251 	public void setMixIns(Map<Class<?>, Class<?>> mixIns) {
252 		this.builder.mixIns(mixIns);
253 	}
254 
255 	/**
256 	 * Shortcut for {@link MapperFeature#AUTO_DETECT_FIELDS} option.

264 	 * {@link MapperFeature#AUTO_DETECT_GETTERS} option.

300 	/**
301 	 * Specify features to enable.
302 	 * @see com.fasterxml.jackson.core.JsonParser.Feature
303 	 * @see com.fasterxml.jackson.core.JsonGenerator.Feature
304 	 * @see com.fasterxml.jackson.databind.SerializationFeature
305 	 * @see com.fasterxml.jackson.databind.DeserializationFeature
306 	 * @see com.fasterxml.jackson.databind.MapperFeature
307 	 */
308 	public void setFeaturesToEnable(Object... featuresToEnable) {
309 		this.builder.featuresToEnable(featuresToEnable);
310 	}
311 
312 	/**
313 	 * Specify features to disable.
314 	 * @see com.fasterxml.jackson.core.JsonParser.Feature
315 	 * @see com.fasterxml.jackson.core.JsonGenerator.Feature
316 	 * @see com.fasterxml.jackson.databind.SerializationFeature
317 	 * @see com.fasterxml.jackson.databind.DeserializationFeature
318 	 * @see com.fasterxml.jackson.databind.MapperFeature
319 	 */
320 	public void setFeaturesToDisable(Object... featuresToDisable) {
321 		this.builder.featuresToDisable(featuresToDisable);
322 	}
323 
324 	/**
325 	 * Set a complete list of modules to be registered with the {@link ObjectMapper}.
326 	 * <p>Note: If this is set, no finding of modules is going to happen - not by
327 	 * Jackson, and not by Spring either (see {@link #setFindModulesViaServiceLoader}).
328 	 * As a consequence, specifying an empty list here will suppress any kind of
329 	 * module detection.
330 	 * <p>Specify either this or {@link #setModulesToInstall}, not both.
331 	 * @since 4.0
332 	 * @see com.fasterxml.jackson.databind.Module
333 	 */
334 	public void setModules(List<Module> modules) {
335 		this.builder.modules(modules);
336 	}
337 
338 	/**
339 	 * Specify one or more modules by class (or class name in XML)
340 	 * to be registered with the {@link ObjectMapper}.
341 	 * <p>Modules specified here will be registered after
342 	 * Spring's autodetection of JSR-310 and Joda-Time, or Jackson's
343 	 * finding of modules (see {@link #setFindModulesViaServiceLoader}),
344 	 * allowing to eventually override their configuration.
345 	 * <p>Specify either this or {@link #setModules}, not both.
346 	 * @since 4.0.1
347 	 * @see com.fasterxml.jackson.databind.Module
348 	 */
349 	public void setModulesToInstall(Class<? extends Module>... modules) {
350 		this.builder.modulesToInstall(modules);
351 	}
352 
353 	/**
354 	 * Set whether to let Jackson find available modules via the JDK ServiceLoader,
355 	 * based on META-INF metadata in the classpath. Requires Jackson 2.2 or higher.
356 	 * <p>If this mode is not set, Spring's Jackson2ObjectMapperFactoryBean itself
357 	 * will try to find the JSR-310 and Joda-Time support modules on the classpath -
358 	 * provided that Java 8 and Joda-Time themselves are available, respectively.
359 	 * @since 4.0.1
360 	 * @see com.fasterxml.jackson.databind.ObjectMapper#findModules()
361 	 */
362 	public void setFindModulesViaServiceLoader(boolean findModules) {
363 		this.builder.findModulesViaServiceLoader(findModules);
364 	}
365 
366 	/**
367 	 * Customize the construction of Jackson handlers ({@link JsonSerializer}, {@link JsonDeserializer},
368 	 * {@link KeyDeserializer}, {@code TypeResolverBuilder} and {@code TypeIdResolver}).
369 	 * @since 4.1.3
370 	 * @see Jackson2ObjectMapperFactoryBean#setApplicationContext(ApplicationContext)
371 	 */
372 	public void setHandlerInstantiator(HandlerInstantiator handlerInstantiator) {
373 		this.builder.handlerInstantiator(handlerInstantiator);
374 	}
375 
376 	@Override
377 	public void setBeanClassLoader(ClassLoader beanClassLoader) {
378 		this.builder.moduleClassLoader(beanClassLoader);
379 	}
380 
381 
382 	@Override
383 	@SuppressWarnings("unchecked")
384 	public void afterPropertiesSet() {
385 		if (this.objectMapper != null) {
386 			this.builder.configure(this.objectMapper);
387 		}
388 		else {
389 			this.objectMapper = this.builder.build();
390 		}
391 	}
392 
393 	/**
394 	 * Set the builder {@link ApplicationContext} in order to autowire Jackson handlers ({@link JsonSerializer},
395 	 * {@link JsonDeserializer}, {@link KeyDeserializer}, {@code TypeResolverBuilder} and {@code TypeIdResolver}).
396 	 * @since 4.1.3
397 	 * @see Jackson2ObjectMapperBuilder#applicationContext(ApplicationContext)
398 	 * @see SpringHandlerInstantiator
399 	 */
400 	@Override
401 	public void setApplicationContext(ApplicationContext applicationContext) {
402 		this.builder.applicationContext(applicationContext);
403 	}
404 
405 	/**
406 	 * Return the singleton ObjectMapper.
407 	 */
408 	@Override
409 	public ObjectMapper getObject() {
410 		return this.objectMapper;
411 	}
412 
413 	@Override
414 	public Class<?> getObjectType() {
415 		return (this.objectMapper != null ? this.objectMapper.getClass() : null);
416 	}
417 
418 	@Override
419 	public boolean isSingleton() {
420 		return true;
421 	}
