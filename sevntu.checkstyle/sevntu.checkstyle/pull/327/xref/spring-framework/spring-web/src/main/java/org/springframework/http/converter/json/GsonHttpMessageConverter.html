17  package org.springframework.http.converter.json;
18  
19  import java.io.IOException;
20  import java.io.InputStreamReader;
21  import java.io.OutputStreamWriter;
22  import java.io.Reader;
23  import java.lang.reflect.Type;
24  import java.nio.charset.Charset;
25  
26  import com.google.gson.Gson;
27  import com.google.gson.JsonIOException;
28  import com.google.gson.JsonParseException;
29  import com.google.gson.reflect.TypeToken;
30  
31  import org.springframework.http.HttpHeaders;
32  import org.springframework.http.HttpInputMessage;
33  import org.springframework.http.HttpOutputMessage;
34  import org.springframework.http.MediaType;
35  import org.springframework.http.converter.AbstractHttpMessageConverter;
36  import org.springframework.http.converter.GenericHttpMessageConverter;
37  import org.springframework.http.converter.HttpMessageNotReadableException;
38  import org.springframework.http.converter.HttpMessageNotWritableException;
39  import org.springframework.util.Assert;
40  
41  /**
42   * Implementation of {@link org.springframework.http.converter.HttpMessageConverter}
43   * that can read and write JSON using the
44   * <a href="https://code.google.com/p/google-gson/">Google Gson</a> library's
45   * {@link Gson} class.
46   *
47   * <p>This converter can be used to bind to typed beans or untyped {@code HashMap}s.
48   * By default, it supports {@code application/json} and {@code application/*+json}.
49   *
50   * <p>Tested against Gson 2.3; compatible with Gson 2.0 and higher.
51   *
52   * @author Roy Clarkson
53   * @since 4.1
54   * @see #setGson
55   * @see #setSupportedMediaTypes
56   */
57  public class GsonHttpMessageConverter extends AbstractHttpMessageConverter<Object>
58  		implements GenericHttpMessageConverter<Object> {
59  
60  	public static final Charset DEFAULT_CHARSET = Charset.forName("UTF-8");
61  
62  
63  	private Gson gson = new Gson();
64  
65  	private String jsonPrefix;
66  
67  
68  	/**
69  	 * Construct a new {@code GsonHttpMessageConverter}.
70  	 */
71  	public GsonHttpMessageConverter() {
72  		super(new MediaType("application", "json", DEFAULT_CHARSET),
73  				new MediaType("application", "*+json", DEFAULT_CHARSET));
74  	}
75  
76  
77  	/**
78  	 * Set the {@code Gson} instance to use.
79  	 * If not set, a default {@link Gson#Gson() Gson} instance is used.
80  	 * <p>Setting a custom-configured {@code Gson} is one way to take further
81  	 * control of the JSON serialization process.
82  	 */
83  	public void setGson(Gson gson) {
84  		Assert.notNull(gson, "'gson' is required");
85  		this.gson = gson;
86  	}
87  
88  	/**
89  	 * Return the configured {@code Gson} instance for this converter.
90  	 */
91  	public Gson getGson() {
92  		return this.gson;
93  	}
94  
95  	/**
96  	 * Specify a custom prefix to use for JSON output. Default is none.
97  	 * @see #setPrefixJson
98  	 */
99  	public void setJsonPrefix(String jsonPrefix) {
100 		this.jsonPrefix = jsonPrefix;
101 	}
102 
103 	/**
104 	 * Indicate whether the JSON output by this view should be prefixed with "{} &&".
105 	 * Default is {@code false}.
106 	 * <p>Prefixing the JSON string in this manner is used to help prevent JSON
107 	 * Hijacking. The prefix renders the string syntactically invalid as a script
108 	 * so that it cannot be hijacked. This prefix does not affect the evaluation
109 	 * of JSON, but if JSON validation is performed on the string, the prefix
110 	 * would need to be ignored.
111 	 * @see #setJsonPrefix
112 	 */
113 	public void setPrefixJson(boolean prefixJson) {
114 		this.jsonPrefix = (prefixJson ? "{} && " : null);
115 	}
116 
117 
118 	@Override
119 	public boolean canRead(Class<?> clazz, MediaType mediaType) {
120 		return canRead(mediaType);
121 	}
122 
123 	@Override
124 	public boolean canRead(Type type, Class<?> contextClass, MediaType mediaType) {
125 		return canRead(mediaType);
126 	}
127 
128 	@Override
129 	public boolean canWrite(Class<?> clazz, MediaType mediaType) {
130 		return canWrite(mediaType);
131 	}
132 
133 	@Override
134 	protected boolean supports(Class<?> clazz) {
135 		// should not be called, since we override canRead/Write instead
136 		throw new UnsupportedOperationException();
137 	}
138 
139 	@Override
140 	protected Object readInternal(Class<?> clazz, HttpInputMessage inputMessage)
141 			throws IOException, HttpMessageNotReadableException {
142 
143 		TypeToken<?> token = getTypeToken(clazz);
144 		return readTypeToken(token, inputMessage);
145 	}
146 
147 	@Override
148 	public Object read(Type type, Class<?> contextClass, HttpInputMessage inputMessage)
149 			throws IOException, HttpMessageNotReadableException {
150 
151 		TypeToken<?> token = getTypeToken(type);
152 		return readTypeToken(token, inputMessage);
153 	}
154 
155 	/**
156 	 * Return the Gson {@link TypeToken} for the specified type.
157 	 * <p>The default implementation returns {@code TypeToken.get(type)}, but
158 	 * this can be overridden in subclasses to allow for custom generic
159 	 * collection handling. For instance:
160 	 * <pre class="code">
161 	 * protected TypeToken<?> getTypeToken(Type type) {
162 	 *   if (type instanceof Class && List.class.isAssignableFrom((Class<?>) type)) {
163 	 *     return new TypeToken<ArrayList<MyBean>>() {};
164 	 *   }
165 	 *   else {
166 	 *     return super.getTypeToken(type);
167 	 *   }
168 	 * }
169 	 * </pre>
170 	 * @param type the type for which to return the TypeToken
171 	 * @return the type token
172 	 */
173 	protected TypeToken<?> getTypeToken(Type type) {
174 		return TypeToken.get(type);
175 	}
176 
177 	private Object readTypeToken(TypeToken<?> token, HttpInputMessage inputMessage) throws IOException {
178 		Reader json = new InputStreamReader(inputMessage.getBody(), getCharset(inputMessage.getHeaders()));
179 		try {
180 			return this.gson.fromJson(json, token.getType());
181 		}
182 		catch (JsonParseException ex) {
183 			throw new HttpMessageNotReadableException("Could not read JSON: " + ex.getMessage(), ex);
184 		}
185 	}
186 
187 	private Charset getCharset(HttpHeaders headers) {
188 		if (headers == null || headers.getContentType() == null || headers.getContentType().getCharSet() == null) {
189 			return DEFAULT_CHARSET;
190 		}
191 		return headers.getContentType().getCharSet();
192 	}
193 
194 	@Override
195 	protected void writeInternal(Object o, HttpOutputMessage outputMessage)
196 			throws IOException, HttpMessageNotWritableException {
197 
198 		Charset charset = getCharset(outputMessage.getHeaders());
199 		OutputStreamWriter writer = new OutputStreamWriter(outputMessage.getBody(), charset);
200 		try {
201 			if (this.jsonPrefix != null) {
202 				writer.append(this.jsonPrefix);
203 			}
204 			this.gson.toJson(o, writer);
205 			writer.close();
206 		}
207 		catch (JsonIOException ex) {
208 			throw new HttpMessageNotWritableException("Could not write JSON: " + ex.getMessage(), ex);
209 		}
210 	}
211
