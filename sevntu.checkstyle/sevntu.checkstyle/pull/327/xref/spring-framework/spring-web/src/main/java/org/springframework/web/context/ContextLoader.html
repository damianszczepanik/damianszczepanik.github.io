48  /**
49   * Performs the actual initialization work for the root application context.
50   * Called by {@link ContextLoaderListener}.
51   *
52   * <p>Looks for a {@link #CONTEXT_CLASS_PARAM "contextClass"} parameter
53   * at the {@code web.xml} context-param level to specify the context
54   * class type, falling back to the default of
55   * {@link org.springframework.web.context.support.XmlWebApplicationContext}
56   * if not found. With the default ContextLoader implementation, any context class
57   * specified needs to implement the ConfigurableWebApplicationContext interface.
58   *
59   * <p>Processes a {@link #CONFIG_LOCATION_PARAM "contextConfigLocation"}
60   * context-param and passes its value to the context instance, parsing it into
61   * potentially multiple file paths which can be separated by any number of
62   * commas and spaces, e.g. "WEB-INF/applicationContext1.xml,
63   * WEB-INF/applicationContext2.xml". Ant-style path patterns are supported as well,
64   * e.g. "WEB-INF/*Context.xml,WEB-INF/spring*.xml" or "WEB-INF/&#42;&#42;/*Context.xml".
65   * If not explicitly specified, the context implementation is supposed to use a
66   * default location (with XmlWebApplicationContext: "/WEB-INF/applicationContext.xml").
67   *
68   * <p>Note: In case of multiple config locations, later bean definitions will
69   * override ones defined in previously loaded files, at least when using one of
70   * Spring's default ApplicationContext implementations. This can be leveraged
71   * to deliberately override certain bean definitions via an extra XML file.
72   *
73   * <p>Above and beyond loading the root application context, this class
74   * can optionally load or obtain and hook up a shared parent context to
75   * the root application context. See the
76   * {@link #loadParentContext(ServletContext)} method for more information.
77   *
78   * <p>As of Spring 3.1, {@code ContextLoader} supports injecting the root web
79   * application context via the {@link #ContextLoader(WebApplicationContext)}
80   * constructor, allowing for programmatic configuration in Servlet 3.0+ environments.
81   * See {@link org.springframework.web.WebApplicationInitializer} for usage examples.
82   *
83   * @author Juergen Hoeller
84   * @author Colin Sampaleanu
85   * @author Sam Brannen
86   * @since 17.02.2003
87   * @see ContextLoaderListener
88   * @see ConfigurableWebApplicationContext
89   * @see org.springframework.web.context.support.XmlWebApplicationContext
90   */

227 	 * Create a new {@code ContextLoader} with the given application context. This
228 	 * constructor is useful in Servlet 3.0+ environments where instance-based
229 	 * registration of listeners is possible through the {@link ServletContext#addListener}
230 	 * API.
231 	 * <p>The context may or may not yet be {@linkplain
232 	 * ConfigurableApplicationContext#refresh() refreshed}. If it (a) is an implementation
233 	 * of {@link ConfigurableWebApplicationContext} and (b) has <strong>not</strong>
234 	 * already been refreshed (the recommended approach), then the following will occur:
235 	 * <ul>
236 	 * <li>If the given context has not already been assigned an {@linkplain
237 	 * ConfigurableApplicationContext#setId id}, one will be assigned to it</li>
238 	 * <li>{@code ServletContext} and {@code ServletConfig} objects will be delegated to
239 	 * the application context</li>
240 	 * <li>{@link #customizeContext} will be called</li>
241 	 * <li>Any {@link ApplicationContextInitializer}s specified through the
242 	 * "contextInitializerClasses" init-param will be applied.</li>
243 	 * <li>{@link ConfigurableApplicationContext#refresh refresh()} will be called</li>
244 	 * </ul>
245 	 * If the context has already been refreshed or does not implement
246 	 * {@code ConfigurableWebApplicationContext}, none of the above will occur under the
247 	 * assumption that the user has performed these actions (or not) per his or her
248 	 * specific needs.
249 	 * <p>See {@link org.springframework.web.WebApplicationInitializer} for usage examples.
250 	 * <p>In any case, the given application context will be registered into the
251 	 * ServletContext under the attribute name {@link
252 	 * WebApplicationContext#ROOT_WEB_APPLICATION_CONTEXT_ATTRIBUTE} and subclasses are
253 	 * free to call the {@link #closeWebApplicationContext} method on container shutdown
254 	 * to close the application context.
255 	 * @param context the application context to manage
256 	 * @see #initWebApplicationContext(ServletContext)
257 	 * @see #closeWebApplicationContext(ServletContext)
258 	 */
259 	public ContextLoader(WebApplicationContext context) {
260 		this.context = context;
261 	}

263 	/**
264 	 * Initialize Spring's web application context for the given servlet context,
265 	 * using the application context provided at construction time, or creating a new one
266 	 * according to the "{@link #CONTEXT_CLASS_PARAM contextClass}" and
267 	 * "{@link #CONFIG_LOCATION_PARAM contextConfigLocation}" context-params.
268 	 * @param servletContext current servlet context
269 	 * @return the new WebApplicationContext
270 	 * @see #ContextLoader(WebApplicationContext)
271 	 * @see #CONTEXT_CLASS_PARAM
272 	 * @see #CONFIG_LOCATION_PARAM
273 	 */
274 	public WebApplicationContext initWebApplicationContext(ServletContext servletContext) {
275 		if (servletContext.getAttribute(WebApplicationContext.ROOT_WEB_APPLICATION_CONTEXT_ATTRIBUTE) != null) {
276 			throw new IllegalStateException(
277 					"Cannot initialize context because there is already a root application context present - " +
278 					"check whether you have multiple ContextLoader* definitions in your web.xml!");
279 		}
280 
281 		Log logger = LogFactory.getLog(ContextLoader.class);
282 		servletContext.log("Initializing Spring root WebApplicationContext");
283 		if (logger.isInfoEnabled()) {
284 			logger.info("Root WebApplicationContext: initialization started");
285 		}
286 		long startTime = System.currentTimeMillis();
287 
288 		try {
289 			// Store context in local instance variable, to guarantee that
290 			// it is available on ServletContext shutdown.
291 			if (this.context == null) {
292 				this.context = createWebApplicationContext(servletContext);
293 			}
294 			if (this.context instanceof ConfigurableWebApplicationContext) {
295 				ConfigurableWebApplicationContext cwac = (ConfigurableWebApplicationContext) this.context;
296 				if (!cwac.isActive()) {
297 					// The context has not yet been refreshed -> provide services such as
298 					// setting the parent context, setting the application context id, etc
299 					if (cwac.getParent() == null) {
300 						// The context instance was injected without an explicit parent ->
301 						// determine parent for root web application context, if any.
302 						ApplicationContext parent = loadParentContext(servletContext);
303 						cwac.setParent(parent);
304 					}
305 					configureAndRefreshWebApplicationContext(cwac, servletContext);
306 				}
307 			}
308 			servletContext.setAttribute(WebApplicationContext.ROOT_WEB_APPLICATION_CONTEXT_ATTRIBUTE, this.context);
309 
310 			ClassLoader ccl = Thread.currentThread().getContextClassLoader();
311 			if (ccl == ContextLoader.class.getClassLoader()) {
312 				currentContext = this.context;
313 			}
314 			else if (ccl != null) {
315 				currentContextPerThread.put(ccl, this.context);
316 			}
317 
318 			if (logger.isDebugEnabled()) {
319 				logger.debug("Published root WebApplicationContext as ServletContext attribute with name [" +
320 						WebApplicationContext.ROOT_WEB_APPLICATION_CONTEXT_ATTRIBUTE + "]");
321 			}
322 			if (logger.isInfoEnabled()) {
323 				long elapsedTime = System.currentTimeMillis() - startTime;
324 				logger.info("Root WebApplicationContext: initialization completed in " + elapsedTime + " ms");
325 			}
326 
327 			return this.context;
328 		}
329 		catch (RuntimeException ex) {
330 			logger.error("Context initialization failed", ex);
331 			servletContext.setAttribute(WebApplicationContext.ROOT_WEB_APPLICATION_CONTEXT_ATTRIBUTE, ex);
332 			throw ex;
333 		}
334 		catch (Error err) {
335 			logger.error("Context initialization failed", err);
336 			servletContext.setAttribute(WebApplicationContext.ROOT_WEB_APPLICATION_CONTEXT_ATTRIBUTE, err);
337 			throw err;
338 		}
339 	}

341 	/**
342 	 * Instantiate the root WebApplicationContext for this loader, either the
343 	 * default context class or a custom context class if specified.
344 	 * <p>This implementation expects custom contexts to implement the
345 	 * {@link ConfigurableWebApplicationContext} interface.
346 	 * Can be overridden in subclasses.
347 	 * <p>In addition, {@link #customizeContext} gets called prior to refreshing the
348 	 * context, allowing subclasses to perform custom modifications to the context.
349 	 * @param sc current servlet context
350 	 * @return the root WebApplicationContext
351 	 * @see ConfigurableWebApplicationContext
352 	 */
353 	protected WebApplicationContext createWebApplicationContext(ServletContext sc) {
354 		Class<?> contextClass = determineContextClass(sc);
355 		if (!ConfigurableWebApplicationContext.class.isAssignableFrom(contextClass)) {
356 			throw new ApplicationContextException("Custom context class [" + contextClass.getName() +
357 					"] is not of type [" + ConfigurableWebApplicationContext.class.getName() + "]");
358 		}
359 		return (ConfigurableWebApplicationContext) BeanUtils.instantiateClass(contextClass);
360 	}
361 
362 	protected void configureAndRefreshWebApplicationContext(ConfigurableWebApplicationContext wac, ServletContext sc) {
363 		if (ObjectUtils.identityToString(wac).equals(wac.getId())) {
364 			// The application context id is still set to its original default value
365 			// -> assign a more useful id based on available information
366 			String idParam = sc.getInitParameter(CONTEXT_ID_PARAM);
367 			if (idParam != null) {
368 				wac.setId(idParam);
369 			}
370 			else {
371 				// Generate default id...
372 				wac.setId(ConfigurableWebApplicationContext.APPLICATION_CONTEXT_ID_PREFIX +
373 						ObjectUtils.getDisplayString(sc.getContextPath()));
374 			}
375 		}
376 
377 		wac.setServletContext(sc);
378 		String configLocationParam = sc.getInitParameter(CONFIG_LOCATION_PARAM);
379 		if (configLocationParam != null) {
380 			wac.setConfigLocation(configLocationParam);
381 		}
382 
383 		// The wac environment's #initPropertySources will be called in any case when the context
384 		// is refreshed; do it eagerly here to ensure servlet property sources are in place for
385 		// use in any post-processing or initialization that occurs below prior to #refresh
386 		ConfigurableEnvironment env = wac.getEnvironment();
387 		if (env instanceof ConfigurableWebEnvironment) {
388 			((ConfigurableWebEnvironment) env).initPropertySources(sc, null);
389 		}
390 
391 		customizeContext(sc, wac);
392 		wac.refresh();
393 	}

395 	/**
396 	 * Customize the {@link ConfigurableWebApplicationContext} created by this
397 	 * ContextLoader after config locations have been supplied to the context
398 	 * but before the context is <em>refreshed</em>.
399 	 * <p>The default implementation {@linkplain #determineContextInitializerClasses(ServletContext)
400 	 * determines} what (if any) context initializer classes have been specified through
401 	 * {@linkplain #CONTEXT_INITIALIZER_CLASSES_PARAM context init parameters} and
402 	 * {@linkplain ApplicationContextInitializer#initialize invokes each} with the
403 	 * given web application context.
404 	 * <p>Any {@code ApplicationContextInitializers} implementing
405 	 * {@link org.springframework.core.Ordered Ordered} or marked with @{@link
406 	 * org.springframework.core.annotation.Order Order} will be sorted appropriately.
407 	 * @param sc the current servlet context
408 	 * @param wac the newly created application context
409 	 * @see #CONTEXT_INITIALIZER_CLASSES_PARAM
410 	 * @see ApplicationContextInitializer#initialize(ConfigurableApplicationContext)
411 	 */
412 	protected void customizeContext(ServletContext sc, ConfigurableWebApplicationContext wac) {
413 		List<Class<ApplicationContextInitializer<ConfigurableApplicationContext>>> initializerClasses =
414 				determineContextInitializerClasses(sc);
415 		if (initializerClasses.isEmpty()) {
416 			// no ApplicationContextInitializers have been declared -> nothing to do
417 			return;
418 		}
419 
420 		ArrayList<ApplicationContextInitializer<ConfigurableApplicationContext>> initializerInstances =
421 				new ArrayList<ApplicationContextInitializer<ConfigurableApplicationContext>>();
422 
423 		for (Class<ApplicationContextInitializer<ConfigurableApplicationContext>> initializerClass : initializerClasses) {
424 			Class<?> initializerContextClass =
425 					GenericTypeResolver.resolveTypeArgument(initializerClass, ApplicationContextInitializer.class);
426 			if (initializerContextClass != null) {
427 				Assert.isAssignable(initializerContextClass, wac.getClass(), String.format(
428 						"Could not add context initializer [%s] since its generic parameter [%s] " +
429 						"is not assignable from the type of application context used by this " +
430 						"context loader [%s]: ", initializerClass.getName(), initializerContextClass.getName(),
431 						wac.getClass().getName()));
432 			}
433 			initializerInstances.add(BeanUtils.instantiateClass(initializerClass));
434 		}
435 
436 		AnnotationAwareOrderComparator.sort(initializerInstances);
437 		for (ApplicationContextInitializer<ConfigurableApplicationContext> initializer : initializerInstances) {
438 			initializer.initialize(wac);
439 		}
440 	}

442 	/**
443 	 * Return the WebApplicationContext implementation class to use, either the
444 	 * default XmlWebApplicationContext or a custom context class if specified.
445 	 * @param servletContext current servlet context
446 	 * @return the WebApplicationContext implementation class to use
447 	 * @see #CONTEXT_CLASS_PARAM
448 	 * @see org.springframework.web.context.support.XmlWebApplicationContext
449 	 */
450 	protected Class<?> determineContextClass(ServletContext servletContext) {
451 		String contextClassName = servletContext.getInitParameter(CONTEXT_CLASS_PARAM);
452 		if (contextClassName != null) {
453 			try {
454 				return ClassUtils.forName(contextClassName, ClassUtils.getDefaultClassLoader());
455 			}
456 			catch (ClassNotFoundException ex) {
457 				throw new ApplicationContextException(
458 						"Failed to load custom context class [" + contextClassName + "]", ex);
459 			}
460 		}
461 		else {
462 			contextClassName = defaultStrategies.getProperty(WebApplicationContext.class.getName());
463 			try {
464 				return ClassUtils.forName(contextClassName, ContextLoader.class.getClassLoader());
465 			}
466 			catch (ClassNotFoundException ex) {
467 				throw new ApplicationContextException(
468 						"Failed to load default context class [" + contextClassName + "]", ex);
469 			}
470 		}
471 	}

473 	/**
474 	 * Return the {@link ApplicationContextInitializer} implementation classes to use
475 	 * if any have been specified by {@link #CONTEXT_INITIALIZER_CLASSES_PARAM}.
476 	 * @param servletContext current servlet context
477 	 * @see #CONTEXT_INITIALIZER_CLASSES_PARAM
478 	 */
479 	protected List<Class<ApplicationContextInitializer<ConfigurableApplicationContext>>>
480 			determineContextInitializerClasses(ServletContext servletContext) {
481 
482 		List<Class<ApplicationContextInitializer<ConfigurableApplicationContext>>> classes =
483 				new ArrayList<Class<ApplicationContextInitializer<ConfigurableApplicationContext>>>();
484 
485 		String globalClassNames = servletContext.getInitParameter(GLOBAL_INITIALIZER_CLASSES_PARAM);
486 		if (globalClassNames != null) {
487 			for (String className : StringUtils.tokenizeToStringArray(globalClassNames, INIT_PARAM_DELIMITERS)) {
488 				classes.add(loadInitializerClass(className));
489 			}
490 		}
491 
492 		String localClassNames = servletContext.getInitParameter(CONTEXT_INITIALIZER_CLASSES_PARAM);
493 		if (localClassNames != null) {
494 			for (String className : StringUtils.tokenizeToStringArray(localClassNames, INIT_PARAM_DELIMITERS)) {
495 				classes.add(loadInitializerClass(className));
496 			}
497 		}
498 
499 		return classes;
500 	}

502 	@SuppressWarnings("unchecked")
503 	private Class<ApplicationContextInitializer<ConfigurableApplicationContext>> loadInitializerClass(String className) {
504 		try {
505 			Class<?> clazz = ClassUtils.forName(className, ClassUtils.getDefaultClassLoader());
506 			Assert.isAssignable(ApplicationContextInitializer.class, clazz);
507 			return (Class<ApplicationContextInitializer<ConfigurableApplicationContext>>) clazz;
508 		}
509 		catch (ClassNotFoundException ex) {
510 			throw new ApplicationContextException("Failed to load context initializer class [" + className + "]", ex);
511 		}
512 	}

514 	/**
515 	 * Template method with default implementation (which may be overridden by a
516 	 * subclass), to load or obtain an ApplicationContext instance which will be
517 	 * used as the parent context of the root WebApplicationContext. If the
518 	 * return value from the method is null, no parent context is set.
519 	 * <p>The main reason to load a parent context here is to allow multiple root
520 	 * web application contexts to all be children of a shared EAR context, or
521 	 * alternately to also share the same parent context that is visible to
522 	 * EJBs. For pure web applications, there is usually no need to worry about
523 	 * having a parent context to the root web application context.
524 	 * <p>The default implementation uses
525 	 * {@link org.springframework.context.access.ContextSingletonBeanFactoryLocator},
526 	 * configured via {@link #LOCATOR_FACTORY_SELECTOR_PARAM} and
527 	 * {@link #LOCATOR_FACTORY_KEY_PARAM}, to load a parent context
528 	 * which will be shared by all other users of ContextsingletonBeanFactoryLocator
529 	 * which also use the same configuration parameters.
530 	 * @param servletContext current servlet context
531 	 * @return the parent application context, or {@code null} if none
532 	 * @see org.springframework.context.access.ContextSingletonBeanFactoryLocator
533 	 */
534 	protected ApplicationContext loadParentContext(ServletContext servletContext) {
535 		ApplicationContext parentContext = null;
536 		String locatorFactorySelector = servletContext.getInitParameter(LOCATOR_FACTORY_SELECTOR_PARAM);
537 		String parentContextKey = servletContext.getInitParameter(LOCATOR_FACTORY_KEY_PARAM);
538 
539 		if (parentContextKey != null) {
540 			// locatorFactorySelector may be null, indicating the default "classpath*:beanRefContext.xml"
541 			BeanFactoryLocator locator = ContextSingletonBeanFactoryLocator.getInstance(locatorFactorySelector);
542 			Log logger = LogFactory.getLog(ContextLoader.class);
543 			if (logger.isDebugEnabled()) {
544 				logger.debug("Getting parent context definition: using parent context key of '" +
545 						parentContextKey + "' with BeanFactoryLocator");
546 			}
547 			this.parentContextRef = locator.useBeanFactory(parentContextKey);
548 			parentContext = (ApplicationContext) this.parentContextRef.getFactory();
549 		}
550 
551 		return parentContext;
552 	}

555 	 * Close Spring's web application context for the given servlet context. If
556 	 * the default {@link #loadParentContext(ServletContext)} implementation,
557 	 * which uses ContextSingletonBeanFactoryLocator, has loaded any shared
558 	 * parent context, release one reference to that shared parent context.
559 	 * <p>If overriding {@link #loadParentContext(ServletContext)}, you may have
560 	 * to override this method as well.
561 	 * @param servletContext the ServletContext that the WebApplicationContext runs in
562 	 */
563 	public void closeWebApplicationContext(ServletContext servletContext) {
564 		servletContext.log("Closing Spring root WebApplicationContext");
565 		try {
566 			if (this.context instanceof ConfigurableWebApplicationContext) {
567 				((ConfigurableWebApplicationContext) this.context).close();
568 			}
569 		}
570 		finally {
571 			ClassLoader ccl = Thread.currentThread().getContextClassLoader();
572 			if (ccl == ContextLoader.class.getClassLoader()) {
573 				currentContext = null;
574 			}
575 			else if (ccl != null) {
576 				currentContextPerThread.remove(ccl);
577 			}
578 			servletContext.removeAttribute(WebApplicationContext.ROOT_WEB_APPLICATION_CONTEXT_ATTRIBUTE);
579 			if (this.parentContextRef != null) {
580 				this.parentContextRef.release();
581 			}
582 		}
583 	}

586 	/**
587 	 * Obtain the Spring root web application context for the current thread
588 	 * (i.e. for the current thread's context ClassLoader, which needs to be
589 	 * the web application's ClassLoader).
590 	 * @return the current root web application context, or {@code null}
591 	 * if none found
592 	 * @see org.springframework.web.context.support.SpringBeanAutowiringSupport
593 	 */
594 	public static WebApplicationContext getCurrentWebApplicationContext() {
595 		ClassLoader ccl = Thread.currentThread().getContextClassLoader();
596 		if (ccl != null) {
597 			WebApplicationContext ccpt = currentContextPerThread.get(ccl);
598 			if (ccpt != null) {
599 				return ccpt;
600 			}
601 		}
602 		return currentContext;
603 	}
