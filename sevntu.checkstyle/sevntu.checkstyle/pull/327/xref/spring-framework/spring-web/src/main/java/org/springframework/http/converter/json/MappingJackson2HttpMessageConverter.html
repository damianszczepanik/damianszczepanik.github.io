17  package org.springframework.http.converter.json;
18  
19  import java.io.IOException;
20  
21  import com.fasterxml.jackson.core.JsonGenerator;
22  import com.fasterxml.jackson.databind.ObjectMapper;
23  
24  import org.springframework.http.MediaType;
25  
26  /**
27   * Implementation of {@link org.springframework.http.converter.HttpMessageConverter HttpMessageConverter} that
28   * can read and write JSON using <a href="http://jackson.codehaus.org/">Jackson 2.x's</a> {@link ObjectMapper}.
29   *
30   * <p>This converter can be used to bind to typed beans, or untyped {@link java.util.HashMap HashMap} instances.
31   *
32   * <p>By default, this converter supports {@code application/json} and {@code application/*+json}.
33   * This can be overridden by setting the {@link #setSupportedMediaTypes supportedMediaTypes} property.
34   *
35   * <p>The default constructor uses the default configuration provided by {@link Jackson2ObjectMapperBuilder}.
36   *
37   * <p>Compatible with Jackson 2.1 and higher.
38   *
39   * @author Arjen Poutsma
40   * @author Keith Donald
41   * @author Rossen Stoyanchev
42   * @author Juergen Hoeller
43   * @author Sebastien Deleuze
44   * @since 3.1.2
45   */
46  public class MappingJackson2HttpMessageConverter extends AbstractJackson2HttpMessageConverter {
47  
48  	private String jsonPrefix;
49  
50  
51  	/**
52  	 * Construct a new {@link MappingJackson2HttpMessageConverter} using default configuration
53  	 * provided by {@link Jackson2ObjectMapperBuilder}.
54  	 */
55  	public MappingJackson2HttpMessageConverter() {
56  		this(Jackson2ObjectMapperBuilder.json().build());
57  	}
58  
59  	/**
60  	 * Construct a new {@link MappingJackson2HttpMessageConverter} with a custom {@link ObjectMapper}.
61  	 * You can use {@link Jackson2ObjectMapperBuilder} to build it easily.
62  	 * @see Jackson2ObjectMapperBuilder#json()
63  	 */
64  	public MappingJackson2HttpMessageConverter(ObjectMapper objectMapper) {
65  		super(objectMapper, new MediaType("application", "json", DEFAULT_CHARSET),
66  				new MediaType("application", "*+json", DEFAULT_CHARSET));
67  	}
68  
69  	/**
70  	 * Specify a custom prefix to use for this view's JSON output.
71  	 * Default is none.
72  	 * @see #setPrefixJson
73  	 */
74  	public void setJsonPrefix(String jsonPrefix) {
75  		this.jsonPrefix = jsonPrefix;
76  	}
77  
78  	/**
79  	 * Indicate whether the JSON output by this view should be prefixed with "{} &&". Default is false.
80  	 * <p>Prefixing the JSON string in this manner is used to help prevent JSON Hijacking.
81  	 * The prefix renders the string syntactically invalid as a script so that it cannot be hijacked.
82  	 * This prefix does not affect the evaluation of JSON, but if JSON validation is performed on the
83  	 * string, the prefix would need to be ignored.
84  	 * @see #setJsonPrefix
85  	 */
86  	public void setPrefixJson(boolean prefixJson) {
87  		this.jsonPrefix = (prefixJson ? "{} && " : null);
88  	}
89  
90  
91  	@Override
92  	protected void writePrefix(JsonGenerator generator, Object object) throws IOException {
93  		if (this.jsonPrefix != null) {
94  			generator.writeRaw(this.jsonPrefix);
95  		}
96  		String jsonpFunction =
97  				(object instanceof MappingJacksonValue ? ((MappingJacksonValue) object).getJsonpFunction() : null);
98  		if (jsonpFunction != null) {
99  			generator.writeRaw(jsonpFunction + "(");
100 		}
101 	}
102 
103 	@Override
104 	protected void writeSuffix(JsonGenerator generator, Object object) throws IOException {
105 		String jsonpFunction =
106 				(object instanceof MappingJacksonValue ? ((MappingJacksonValue) object).getJsonpFunction() : null);
107 		if (jsonpFunction != null) {
108 			generator.writeRaw(");");
109 		}
110 	}
111 
112 }
