17  package org.springframework.web.bind.annotation;
18  
19  import java.lang.annotation.Documented;
20  import java.lang.annotation.ElementType;
21  import java.lang.annotation.Retention;
22  import java.lang.annotation.RetentionPolicy;
23  import java.lang.annotation.Target;
24  import java.util.concurrent.Callable;
25  
26  /**
27   * Annotation for mapping web requests onto specific handler classes and/or
28   * handler methods. Provides a consistent style between Servlet and Portlet
29   * environments, with the semantics adapting to the concrete environment.
30   *
31   * <p><b>NOTE:</b> The set of features supported for Servlets is a superset
32   * of the set of features supported for Portlets. The places where this applies
33   * are marked with the label "Servlet-only" in this source file. For Servlet
34   * environments there are some further distinctions depending on whether an
35   * application is configured with {@literal "@MVC 3.0"} or
36   * {@literal "@MVC 3.1"} support classes. The places where this applies are
37   * marked with {@literal "@MVC 3.1-only"} in this source file. For more
38   * details see the note on the new support classes added in Spring MVC 3.1
39   * further below.
40   *
41   * <p>Handler methods which are annotated with this annotation are allowed
42   * to have very flexible signatures. They may have arguments of the following
43   * types, in arbitrary order (except for validation results, which need to
44   * follow right after the corresponding command object, if desired):
45   * <ul>
46   * <li>Request and/or response objects (Servlet API or Portlet API).
47   * You may choose any specific request/response type, e.g.
48   * {@link javax.servlet.ServletRequest} / {@link javax.servlet.http.HttpServletRequest}
49   * or {@link javax.portlet.PortletRequest} / {@link javax.portlet.ActionRequest} /
50   * {@link javax.portlet.RenderRequest}. Note that in the Portlet case,
51   * an explicitly declared action/render argument is also used for mapping
52   * specific request types onto a handler method (in case of no other
53   * information given that differentiates between action and render requests).
54   * <li>Session object (Servlet API or Portlet API): either
55   * {@link javax.servlet.http.HttpSession} or {@link javax.portlet.PortletSession}.
56   * An argument of this type will enforce the presence of a corresponding session.
57   * As a consequence, such an argument will never be {@code null}.
58   * <i>Note that session access may not be thread-safe, in particular in a
59   * Servlet environment: Consider switching the
60   * {@link org.springframework.web.servlet.mvc.method.annotation.RequestMappingHandlerAdapter#setSynchronizeOnSession "synchronizeOnSession"}
61   * flag to "true" if multiple requests are allowed to access a session concurrently.</i>
62   * <li>{@link org.springframework.web.context.request.WebRequest} or
63   * {@link org.springframework.web.context.request.NativeWebRequest}.
64   * Allows for generic request parameter access as well as request/session
65   * attribute access, without ties to the native Servlet/Portlet API.
66   * <li>{@link java.util.Locale} for the current request locale
67   * (determined by the most specific locale resolver available,
68   * i.e. the configured {@link org.springframework.web.servlet.LocaleResolver}
69   * in a Servlet environment and the portal locale in a Portlet environment).
70   * <li>{@link java.io.InputStream} / {@link java.io.Reader} for access
71   * to the request's content. This will be the raw InputStream/Reader as
72   * exposed by the Servlet/Portlet API.
73   * <li>{@link java.io.OutputStream} / {@link java.io.Writer} for generating
74   * the response's content. This will be the raw OutputStream/Writer as
75   * exposed by the Servlet/Portlet API.
76   * <li>{@link org.springframework.http.HttpMethod} for the HTTP request method</li>
77   * <li>{@link PathVariable @PathVariable} annotated parameters (Servlet-only)
78   * for access to URI template values (i.e. /hotels/{hotel}). Variable values will be
79   * converted to the declared method argument type. By default, the URI template
80   * will match against the regular expression {@code [^\.]*} (i.e. any character
81   * other than period), but this can be changed by specifying another regular
82   * expression, like so: /hotels/{hotel:\d+}.
83   * Additionally, {@code @PathVariable} can be used on a
84   * {@link java.util.Map Map&lt;String, String&gt;} to gain access to all
85   * URI template variables.
86   * <li>{@link MatrixVariable @MatrixVariable} annotated parameters (Servlet-only)
87   * for access to name-value pairs located in URI path segments. Matrix variables
88   * must be represented with a URI template variable. For example /hotels/{hotel}
89   * where the incoming URL may be "/hotels/42;q=1".
90   * Additionally, {@code @MatrixVariable} can be used on a
91   * {@link java.util.Map Map&lt;String, String&gt;} to gain access to all
92   * matrix variables in the URL or to those in a specific path variable.
93   * <li>{@link RequestParam @RequestParam} annotated parameters for access to
94   * specific Servlet/Portlet request parameters. Parameter values will be
95   * converted to the declared method argument type. Additionally,
96   * {@code @RequestParam} can be used on a {@link java.util.Map Map&lt;String, String&gt;} or
97   * {@link org.springframework.util.MultiValueMap MultiValueMap&lt;String, String&gt;}
98   * method parameter to gain access to all request parameters.
99   * <li>{@link RequestHeader @RequestHeader} annotated parameters for access to
100  * specific Servlet/Portlet request HTTP headers. Parameter values will be
101  * converted to the declared method argument type. Additionally,
102  * {@code @RequestHeader} can be used on a {@link java.util.Map Map&lt;String, String&gt;},
103  * {@link org.springframework.util.MultiValueMap MultiValueMap&lt;String, String&gt;}, or
104  * {@link org.springframework.http.HttpHeaders HttpHeaders} method parameter to
105  * gain access to all request headers.
106  * <li>{@link RequestBody @RequestBody} annotated parameters (Servlet-only)
107  * for access to the Servlet request HTTP contents. The request stream will be
108  * converted to the declared method argument type using
109  * {@linkplain org.springframework.http.converter.HttpMessageConverter message
110  * converters}. Such parameters may optionally be annotated with {@code @Valid}
111  * and also support access to validation results through an
112  * {@link org.springframework.validation.Errors} argument.
113  * Instead a {@link org.springframework.web.bind.MethodArgumentNotValidException}
114  * exception is raised.
115  * <li>{@link RequestPart @RequestPart} annotated parameters
116  * (Servlet-only, {@literal @MVC 3.1-only})
117  * for access to the content
118  * of a part of "multipart/form-data" request. The request part stream will be
119  * converted to the declared method argument type using
120  * {@linkplain org.springframework.http.converter.HttpMessageConverter message
121  * converters}. Such parameters may optionally be annotated with {@code @Valid}
122  * and support access to validation results through a
123  * {@link org.springframework.validation.Errors} argument.
124  * Instead a {@link org.springframework.web.bind.MethodArgumentNotValidException}
125  * exception is raised.
126  * <li>{@link org.springframework.http.HttpEntity HttpEntity&lt;?&gt;} parameters
127  * (Servlet-only) for access to the Servlet request HTTP headers and contents.
128  * The request stream will be converted to the entity body using
129  * {@linkplain org.springframework.http.converter.HttpMessageConverter message
130  * converters}.
131  * <li>{@link java.util.Map} / {@link org.springframework.ui.Model} /
132  * {@link org.springframework.ui.ModelMap} for enriching the implicit model
133  * that will be exposed to the web view.
134  * <li>{@link org.springframework.web.servlet.mvc.support.RedirectAttributes}
135  * (Servlet-only, {@literal @MVC 3.1-only}) to specify the exact set of attributes
136  * to use in case of a redirect and also to add flash attributes (attributes
137  * stored temporarily on the server-side to make them available to the request
138  * after the redirect). {@code RedirectAttributes} is used instead of the
139  * implicit model if the method returns a "redirect:" prefixed view name or
140  * {@code RedirectView}.
141  * <li>Command/form objects to bind parameters to: as bean properties or fields,
142  * with customizable type conversion, depending on {@link InitBinder} methods
143  * and/or the HandlerAdapter configuration - see the "webBindingInitializer"
144  * property on RequestMappingHandlerMethodAdapter.
145  * Such command objects along with their validation results will be exposed
146  * as model attributes, by default using the non-qualified command class name
147  * in property notation (e.g. "orderAddress" for type "mypackage.OrderAddress").
148  * Specify a parameter-level {@link ModelAttribute @ModelAttribute} annotation for
149  * declaring a specific model attribute name.
150  * <li>{@link org.springframework.validation.Errors} /
151  * {@link org.springframework.validation.BindingResult} validation results
152  * for a preceding command/form object (the immediate preceding argument).
153  * <li>{@link org.springframework.web.bind.support.SessionStatus} status handle
154  * for marking form processing as complete (triggering the cleanup of session
155  * attributes that have been indicated by the {@link SessionAttributes @SessionAttributes}
156  * annotation at the handler type level).
157  * <li>{@link org.springframework.web.util.UriComponentsBuilder}
158  * (Servlet-only, {@literal @MVC 3.1-only})
159  * for preparing a URL relative to the current request's host, port, scheme,
160  * context path, and the literal part of the servlet mapping.
161  * </ul>
162  *
163  * <p><strong>Note:</strong> JDK 1.8's {@code java.util.Optional} is supported
164  * as a method parameter type with annotations that provide a {@code required}
165  * attribute (e.g. {@code @RequestParam}, {@code @RequestHeader}, etc.). The use
166  * of {@code java.util.Optional} in those cases is equivalent to having
167  * {@code required=false}.
168  *
169  * <p>The following return types are supported for handler methods:
170  * <ul>
171  * <li>A {@code ModelAndView} object (Servlet MVC or Portlet MVC),
172  * with the model implicitly enriched with command objects and the results
173  * of {@link ModelAttribute @ModelAttribute} annotated reference data accessor methods.
174  * <li>A {@link org.springframework.ui.Model Model} object, with the view name
175  * implicitly determined through a {@link org.springframework.web.servlet.RequestToViewNameTranslator}
176  * and the model implicitly enriched with command objects and the results
177  * of {@link ModelAttribute @ModelAttribute} annotated reference data accessor methods.
178  * <li>A {@link java.util.Map} object for exposing a model,
179  * with the view name implicitly determined through a
180  * {@link org.springframework.web.servlet.RequestToViewNameTranslator}
181  * and the model implicitly enriched with command objects and the results
182  * of {@link ModelAttribute @ModelAttribute} annotated reference data accessor methods.
183  * <li>A {@link org.springframework.web.servlet.View} object, with the
184  * model implicitly determined through command objects and
185  * {@link ModelAttribute @ModelAttribute} annotated reference data accessor methods.
186  * The handler method may also programmatically enrich the model by
187  * declaring a {@link org.springframework.ui.Model} argument (see above).
188  * <li>A {@link String} value which is interpreted as view name,
189  * with the model implicitly determined through command objects and
190  * {@link ModelAttribute @ModelAttribute} annotated reference data accessor methods.
191  * The handler method may also programmatically enrich the model by
192  * declaring a {@link org.springframework.ui.ModelMap} argument
193  * (see above).
194  * <li>{@link ResponseBody @ResponseBody} annotated methods (Servlet-only)
195  * for access to the Servlet response HTTP contents. The return value will
196  * be converted to the response stream using
197  * {@linkplain org.springframework.http.converter.HttpMessageConverter message
198  * converters}.
199  * <li>An {@link org.springframework.http.HttpEntity HttpEntity&lt;?&gt;} or
200  * {@link org.springframework.http.ResponseEntity ResponseEntity&lt;?&gt;} object
201  * (Servlet-only) to access to the Servlet response HTTP headers and contents.
202  * The entity body will be converted to the response stream using
203  * {@linkplain org.springframework.http.converter.HttpMessageConverter message
204  * converters}.
205  * <li>An {@link org.springframework.http.HttpHeaders HttpHeaders} object to
206  * return a response with no body.</li>
207  * <li>A {@link Callable} which is used by Spring MVC to obtain the return
208  * value asynchronously in a separate thread transparently managed by Spring MVC
209  * on behalf of the application.
210  * <li>A {@link org.springframework.web.context.request.async.DeferredResult}
211  * which the application uses to produce a return value in a separate
212  * thread of its own choosing, as an alternative to returning a Callable.
213  * <li>A {@link org.springframework.util.concurrent.ListenableFuture}
214  * which the application uses to produce a return value in a separate
215  * thread of its own choosing, as an alternative to returning a Callable.
216  * <li>A {@link org.springframework.web.servlet.mvc.method.annotation.ResponseBodyEmitter}
217  * can be used to write multiple objects to the response asynchronously;
218  * also supported as the body within {@code ResponseEntity}.</li>
219  * <li>An {@link org.springframework.web.servlet.mvc.method.annotation.SseEmitter}
220  * can be used to write Server-Sent Events to the response asynchronously;
221  * also supported as the body within {@code ResponseEntity}.</li>
222  * <li>{@code void} if the method handles the response itself (by
223  * writing the response content directly, declaring an argument of type
224  * {@link javax.servlet.ServletResponse} / {@link javax.servlet.http.HttpServletResponse}
225  * / {@link javax.portlet.RenderResponse} for that purpose)
226  * or if the view name is supposed to be implicitly determined through a
227  * {@link org.springframework.web.servlet.RequestToViewNameTranslator}
228  * (not declaring a response argument in the handler method signature;
229  * only applicable in a Servlet environment).
230  * <li>Any other return type will be considered as single model attribute
231  * to be exposed to the view, using the attribute name specified through
232  * {@link ModelAttribute @ModelAttribute} at the method level (or the default attribute
233  * name based on the return type's class name otherwise). The model will be
234  * implicitly enriched with command objects and the results of
235  * {@link ModelAttribute @ModelAttribute} annotated reference data accessor methods.
236  * </ul>
237  *
238  * <p><b>NOTE:</b> {@code @RequestMapping} will only be processed if an
239  * an appropriate {@code HandlerMapping}-{@code HandlerAdapter} pair
240  * is configured. This is the case by default in both the
241  * {@code DispatcherServlet} and the {@code DispatcherPortlet}.
242  * However, if you are defining custom {@code HandlerMappings} or
243  * {@code HandlerAdapters}, then you need to add
244  * {@code DefaultAnnotationHandlerMapping} and
245  * {@code AnnotationMethodHandlerAdapter} to your configuration.</code>.
246  *
247  * <p><b>NOTE:</b> Spring 3.1 introduced a new set of support classes for
248  * {@code @RequestMapping} methods in Servlet environments called
249  * {@code RequestMappingHandlerMapping} and
250  * {@code RequestMappingHandlerAdapter}. They are recommended for use and
251  * even required to take advantage of new features in Spring MVC 3.1 (search
252  * {@literal "@MVC 3.1-only"} in this source file) and going forward.
253  * The new support classes are enabled by default from the MVC namespace and
254  * with use of the MVC Java config ({@code @EnableWebMvc}) but must be
255  * configured explicitly if using neither.
256  *
257  * <p><b>NOTE:</b> When using controller interfaces (e.g. for AOP proxying),
258  * make sure to consistently put <i>all</i> your mapping annotations - such as
259  * {@code @RequestMapping} and {@code @SessionAttributes} - on
260  * the controller <i>interface</i> rather than on the implementation class.
261  *
262  * @author Juergen Hoeller
263  * @author Arjen Poutsma
264  * @author Sam Brannen
265  * @since 2.5
266  * @see RequestParam
267  * @see ModelAttribute
268  * @see SessionAttributes
269  * @see InitBinder
270  * @see org.springframework.web.context.request.WebRequest
271  * @see org.springframework.web.servlet.mvc.method.annotation.RequestMappingHandlerAdapter
272  * @see org.springframework.web.portlet.mvc.annotation.DefaultAnnotationHandlerMapping
273  * @see org.springframework.web.portlet.mvc.annotation.AnnotationMethodHandlerAdapter
274  */
275 @Target({ElementType.METHOD, ElementType.TYPE})
276 @Retention(RetentionPolicy.RUNTIME)
277 @Documented
278 @Mapping
279 public @interface RequestMapping {
280 
281 	/**
282 	 * Assign a name to this mapping.
283 	 * <p><b>Supported at the type level as well as at the method level!</b>
284 	 * When used on both levels, a combined name is derived by concatenation
285 	 * with "#" as separator.
286 	 * @see org.springframework.web.servlet.mvc.method.annotation.MvcUriComponentsBuilder
287 	 * @see org.springframework.web.servlet.handler.HandlerMethodMappingNamingStrategy
288 	 */
289 	String name() default "";
290 
291 	/**
292 	 * The primary mapping expressed by this annotation.
293 	 * <p>In a Servlet environment: the path mapping URIs (e.g. "/myPath.do").
294 	 * Ant-style path patterns are also supported (e.g. "/myPath/*.do").
295 	 * At the method level, relative paths (e.g. "edit.do") are supported
296 	 * within the primary mapping expressed at the type level.
297 	 * Path mapping URIs may contain placeholders (e.g. "/${connect}")
298 	 * <p>In a Portlet environment: the mapped portlet modes
299 	 * (i.e. "EDIT", "VIEW", "HELP" or any custom modes).
300 	 * <p><b>Supported at the type level as well as at the method level!</b>
301 	 * When used at the type level, all method-level mappings inherit
302 	 * this primary mapping, narrowing it for a specific handler method.
303 	 * @see org.springframework.web.bind.annotation.ValueConstants#DEFAULT_NONE
304 	 */
305 	String[] value() default {};
306 
307 	/**
308 	 * The HTTP request methods to map to, narrowing the primary mapping:
309 	 * GET, POST, HEAD, OPTIONS, PUT, PATCH, DELETE, TRACE.
310 	 * <p><b>Supported at the type level as well as at the method level!</b>
311 	 * When used at the type level, all method-level mappings inherit
312 	 * this HTTP method restriction (i.e. the type-level restriction
313 	 * gets checked before the handler method is even resolved).
314 	 * <p>Supported for Servlet environments as well as Portlet 2.0 environments.
315 	 */
316 	RequestMethod[] method() default {};
317 
318 	/**
319 	 * The parameters of the mapped request, narrowing the primary mapping.
320 	 * <p>Same format for any environment: a sequence of "myParam=myValue" style
321 	 * expressions, with a request only mapped if each such parameter is found
322 	 * to have the given value. Expressions can be negated by using the "!=" operator,
323 	 * as in "myParam!=myValue". "myParam" style expressions are also supported,
324 	 * with such parameters having to be present in the request (allowed to have
325 	 * any value). Finally, "!myParam" style expressions indicate that the
326 	 * specified parameter is <i>not</i> supposed to be present in the request.
327 	 * <p><b>Supported at the type level as well as at the method level!</b>
328 	 * When used at the type level, all method-level mappings inherit
329 	 * this parameter restriction (i.e. the type-level restriction
330 	 * gets checked before the handler method is even resolved).
331 	 * <p>In a Servlet environment, parameter mappings are considered as restrictions
332 	 * that are enforced at the type level. The primary path mapping (i.e. the
333 	 * specified URI value) still has to uniquely identify the target handler, with
334 	 * parameter mappings simply expressing preconditions for invoking the handler.
335 	 * <p>In a Portlet environment, parameters are taken into account as mapping
336 	 * differentiators, i.e. the primary portlet mode mapping plus the parameter
337 	 * conditions uniquely identify the target handler. Different handlers may be
338 	 * mapped onto the same portlet mode, as long as their parameter mappings differ.
339 	 */
340 	String[] params() default {};
341 
342 	/**
343 	 * The headers of the mapped request, narrowing the primary mapping.
344 	 * <p>Same format for any environment: a sequence of "My-Header=myValue" style
345 	 * expressions, with a request only mapped if each such header is found
346 	 * to have the given value. Expressions can be negated by using the "!=" operator,
347 	 * as in "My-Header!=myValue". "My-Header" style expressions are also supported,
348 	 * with such headers having to be present in the request (allowed to have
349 	 * any value). Finally, "!My-Header" style expressions indicate that the
350 	 * specified header is <i>not</i> supposed to be present in the request.
351 	 * <p>Also supports media type wildcards (*), for headers such as Accept
352 	 * and Content-Type. For instance,
353 	 * <pre class="code">
354 	 * &#064;RequestMapping(value = "/something", headers = "content-type=text/*")
355 	 * </pre>
356 	 * will match requests with a Content-Type of "text/html", "text/plain", etc.
357 	 * <p><b>Supported at the type level as well as at the method level!</b>
358 	 * When used at the type level, all method-level mappings inherit
359 	 * this header restriction (i.e. the type-level restriction
360 	 * gets checked before the handler method is even resolved).
361 	 * <p>Maps against HttpServletRequest headers in a Servlet environment,
362 	 * and against PortletRequest properties in a Portlet 2.0 environment.
363 	 * @see org.springframework.http.MediaType
364 	 */
365 	String[] headers() default {};
366 
367 	/**
368 	 * The consumable media types of the mapped request, narrowing the primary mapping.
369 	 * <p>The format is a single media type or a sequence of media types,
370 	 * with a request only mapped if the {@code Content-Type} matches one of these media types.
371 	 * Examples:
372 	 * <pre class="code">
373 	 * consumes = "text/plain"
374 	 * consumes = {"text/plain", "application/*"}
375 	 * </pre>
376 	 * Expressions can be negated by using the "!" operator, as in "!text/plain", which matches
377 	 * all requests with a {@code Content-Type} other than "text/plain".
378 	 * <p><b>Supported at the type level as well as at the method level!</b>
379 	 * When used at the type level, all method-level mappings override
380 	 * this consumes restriction.
381 	 * @see org.springframework.http.MediaType
382 	 * @see javax.servlet.http.HttpServletRequest#getContentType()
383 	 */
384 	String[] consumes() default {};
385 
386 	/**
387 	 * The producible media types of the mapped request, narrowing the primary mapping.
388 	 * <p>The format is a single media type or a sequence of media types,
389 	 * with a request only mapped if the {@code Accept} matches one of these media types.
390 	 * Examples:
391 	 * <pre class="code">
392 	 * produces = "text/plain"
393 	 * produces = {"text/plain", "application/*"}
394 	 * </pre>
395 	 * Expressions can be negated by using the "!" operator, as in "!text/plain", which matches
396 	 * all requests with a {@code Accept} other than "text/plain".
397 	 * <p><b>Supported at the type level as well as at the method level!</b>
398 	 * When used at the type level, all method-level mappings override
399 	 * this produces restriction.
400 	 * @see org.springframework.http.MediaType
401 	 */
402 	String[] produces() default {};
403 
404 }
