67  /**
68   * Helper class to configure Tiles 2.x for the Spring Framework. See
69   * <a href="http://tiles.apache.org">http://tiles.apache.org</a>
70   * for more information about Tiles, which basically is a templating mechanism
71   * for web applications using JSPs and other template engines.
72   *
73   * <b>Note: Spring 4.0 requires Tiles 2.2.2.</b> Tiles' EL support will
74   * be activated by default when the Tiles EL module is present in the classpath.
75   *
76   * <p>The TilesConfigurer simply configures a TilesContainer using a set of files
77   * containing definitions, to be accessed by {@link TilesView} instances. This is a
78   * Spring-based alternative (for usage in Spring configuration) to the Tiles-provided
79   * {@code ServletContextListener}
80   * (e.g. {@link org.apache.tiles.extras.complete.CompleteAutoloadTilesListener}
81   * for usage in {@code web.xml}.
82   *
83   * <p>TilesViews can be managed by any {@link org.springframework.web.servlet.ViewResolver}.
84   * For simple convention-based view resolution, consider using {@link TilesViewResolver}.
85   *
86   * <p>A typical TilesConfigurer bean definition looks as follows:
87   *
88   * <pre class="code">
89   * &lt;bean id="tilesConfigurer" class="org.springframework.web.servlet.view.tiles2.TilesConfigurer">
90   *   &lt;property name="definitions">
91   *     &lt;list>
92   *       &lt;value>/WEB-INF/defs/general.xml&lt;/value>
93   *       &lt;value>/WEB-INF/defs/widgets.xml&lt;/value>
94   *       &lt;value>/WEB-INF/defs/administrator.xml&lt;/value>
95   *       &lt;value>/WEB-INF/defs/customer.xml&lt;/value>
96   *       &lt;value>/WEB-INF/defs/templates.xml&lt;/value>
97   *     &lt;/list>
98   *   &lt;/property>
99   * &lt;/bean>
100  * </pre>
101  *
102  * The values in the list are the actual Tiles XML files containing the definitions.
103  * If the list is not specified, the default is {@code "/WEB-INF/tiles.xml"}.
104  *
105  * @author Juergen Hoeller
106  * @since 2.5
107  * @see TilesView
108  * @see TilesViewResolver
109  * @deprecated as of Spring 4.2, in favor of Tiles 3
110  */
111 @Deprecated

139 	 * <p>Default is a variant of {@link org.apache.tiles.startup.DefaultTilesInitializer},
140 	 * respecting the "definitions", "preparerFactoryClass" etc properties on this configurer.
141 	 * <p><b>NOTE: Specifying a custom TilesInitializer effectively disables all other bean
142 	 * properties on this configurer.</b> The entire initialization procedure is then left
143 	 * to the TilesInitializer as specified.
144 	 */

146 		this.tilesInitializer = tilesInitializer;
147 	}

149 	/**
150 	 * Specify whether to apply Tiles 2.2's "complete-autoload" configuration.
151 	 * <p>See {@link org.apache.tiles.extras.complete.CompleteAutoloadTilesContainerFactory}
152 	 * for details on the complete-autoload mode.
153 	 * <p><b>NOTE: Specifying the complete-autoload mode effectively disables all other bean
154 	 * properties on this configurer.</b> The entire initialization procedure is then left
155 	 * to {@link org.apache.tiles.extras.complete.CompleteAutoloadTilesInitializer}.
156 	 * @see org.apache.tiles.extras.complete.CompleteAutoloadTilesContainerFactory
157 	 * @see org.apache.tiles.extras.complete.CompleteAutoloadTilesInitializer
158 	 */
159 	public void setCompleteAutoload(boolean completeAutoload) {
160 		if (completeAutoload) {
161 			try {
162 				this.tilesInitializer = new SpringCompleteAutoloadTilesInitializer();
163 			}
164 			catch (Throwable ex) {
165 				throw new IllegalStateException("Tiles-Extras 2.2 not available", ex);
166 			}
167 		}
168 		else {
169 			this.tilesInitializer = null;
170 		}
171 	}

213 	 * <p>Specify {@link SimpleSpringPreparerFactory} to autowire

249 	 * delegating to the TilesInitializer.
250 	 * @throws TilesException in case of setup failure

252 	@Override
253 	public void afterPropertiesSet() throws TilesException {
254 		TilesApplicationContext preliminaryContext =
255 				new SpringWildcardServletTilesApplicationContext(this.servletContext);
256 		if (this.tilesInitializer == null) {
257 			this.tilesInitializer = createTilesInitializer();
258 		}
259 		this.tilesInitializer.initialize(preliminaryContext);
260 	}

290 	private class SpringTilesContainerFactory extends BasicTilesContainerFactory {
291 
292 		@Override
293 		protected BasicTilesContainer instantiateContainer(TilesApplicationContext context) {
294 			return (useMutableTilesContainer ? new CachingTilesContainer() : new BasicTilesContainer());
295 		}
296 
297 		@Override
298 		protected void registerRequestContextFactory(String className,
299 				List<TilesRequestContextFactory> factories, TilesRequestContextFactory parent) {
300 			// Avoid Tiles 2.2 warn logging when default RequestContextFactory impl class not found
301 			if (ClassUtils.isPresent(className, TilesConfigurer.class.getClassLoader())) {
302 				super.registerRequestContextFactory(className, factories, parent);
303 			}
304 		}
305 
306 		@Override
307 		protected List<URL> getSourceURLs(TilesApplicationContext applicationContext,
308 				TilesRequestContextFactory contextFactory) {
309 			if (definitions != null) {
310 				try {
311 					List<URL> result = new LinkedList<URL>();
312 					for (String definition : definitions) {
313 						Set<URL> resources = applicationContext.getResources(definition);
314 						if (resources != null) {
315 							result.addAll(resources);
316 						}
317 					}
318 					return result;
319 				}
320 				catch (IOException ex) {
321 					throw new DefinitionsFactoryException("Cannot load definition URLs", ex);
322 				}
323 			}
324 			else {
325 				return super.getSourceURLs(applicationContext, contextFactory);
326 			}
327 		}
328 
329 		@Override
330 		protected BaseLocaleUrlDefinitionDAO instantiateLocaleDefinitionDao(TilesApplicationContext applicationContext,
331 				TilesRequestContextFactory contextFactory, LocaleResolver resolver) {
332 			BaseLocaleUrlDefinitionDAO dao = super.instantiateLocaleDefinitionDao(
333 					applicationContext, contextFactory, resolver);
334 			if (checkRefresh && dao instanceof CachingLocaleUrlDefinitionDAO) {
335 				((CachingLocaleUrlDefinitionDAO) dao).setCheckRefresh(true);
336 			}
337 			return dao;
338 		}
339 
340 		@Override
341 		protected DefinitionsReader createDefinitionsReader(TilesApplicationContext applicationContext,
342 				TilesRequestContextFactory contextFactory) {
343 			DigesterDefinitionsReader reader = new DigesterDefinitionsReader();
344 			if (!validateDefinitions){
345 				Map<String,String> map = new HashMap<String,String>();
346 				map.put(DigesterDefinitionsReader.PARSER_VALIDATE_PARAMETER_NAME, Boolean.FALSE.toString());
347 				reader.init(map);
348 			}
349 			return reader;
350 		}
351 
352 		@Override
353 		protected DefinitionsFactory createDefinitionsFactory(TilesApplicationContext applicationContext,
354 				TilesRequestContextFactory contextFactory, LocaleResolver resolver) {
355 			if (definitionsFactoryClass != null) {
356 				DefinitionsFactory factory = BeanUtils.instantiate(definitionsFactoryClass);
357 				if (factory instanceof TilesApplicationContextAware) {
358 					((TilesApplicationContextAware) factory).setApplicationContext(applicationContext);
359 				}
360 				BeanWrapper bw = PropertyAccessorFactory.forBeanPropertyAccess(factory);
361 				if (bw.isWritableProperty("localeResolver")) {
362 					bw.setPropertyValue("localeResolver", resolver);
363 				}
364 				if (bw.isWritableProperty("definitionDAO")) {
365 					bw.setPropertyValue("definitionDAO",
366 							createLocaleDefinitionDao(applicationContext, contextFactory, resolver));
367 				}
368 				if (factory instanceof Refreshable) {
369 					((Refreshable) factory).refresh();
370 				}
371 				return factory;
372 			}
373 			else {
374 				return super.createDefinitionsFactory(applicationContext, contextFactory, resolver);
375 			}
376 		}
377 
378 		@Override
379 		protected PreparerFactory createPreparerFactory(TilesApplicationContext applicationContext,
380 				TilesRequestContextFactory contextFactory) {
381 			if (preparerFactoryClass != null) {
382 				return BeanUtils.instantiate(preparerFactoryClass);
383 			}
384 			else {
385 				return super.createPreparerFactory(applicationContext, contextFactory);
386 			}
387 		}
388 
389 		@Override
390 		protected LocaleResolver createLocaleResolver(TilesApplicationContext applicationContext,
391 				TilesRequestContextFactory contextFactory) {
392 			return new SpringLocaleResolver();
393 		}
394 
395 		@Override
396 		protected AttributeEvaluatorFactory createAttributeEvaluatorFactory(TilesApplicationContext applicationContext,
397 				TilesRequestContextFactory contextFactory, LocaleResolver resolver) {
398 			AttributeEvaluator evaluator;
399 			if (tilesElPresent && JspFactory.getDefaultFactory() != null) {
400 				evaluator = TilesElActivator.createEvaluator(applicationContext);
401 			}
402 			else {
403 				evaluator = new DirectAttributeEvaluator();
404 			}
405 			return new BasicAttributeEvaluatorFactory(evaluator);
406 		}
407 	}
