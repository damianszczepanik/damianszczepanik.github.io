102 	public void testGetsAreNotTransactionalWithProxyFactory3() throws NoSuchMethodException {
103 		ITestBean testBean = (ITestBean) factory.getBean("proxyFactory3");
104 		assertTrue("testBean is a full proxy", testBean instanceof DerivedTestBean);
105 		InvocationCounterPointcut txnCounter = (InvocationCounterPointcut) factory.getBean("txnInvocationCounterPointcut");
106 		InvocationCounterInterceptor preCounter = (InvocationCounterInterceptor) factory.getBean("preInvocationCounterInterceptor");
107 		InvocationCounterInterceptor postCounter = (InvocationCounterInterceptor) factory.getBean("postInvocationCounterInterceptor");
108 		txnCounter.counter = 0;
109 		preCounter.counter = 0;
110 		postCounter.counter = 0;
111 		doTestGetsAreNotTransactional(testBean);
112 		// Can't assert it's equal to 4 as the pointcut may be optimized and only invoked once
113 		assertTrue(0 < txnCounter.counter && txnCounter.counter <= 4);
114 		assertEquals(4, preCounter.counter);
115 		assertEquals(4, postCounter.counter);
116 	}

118 	private void doTestGetsAreNotTransactional(final ITestBean testBean) {
119 		// Install facade
120 		PlatformTransactionManager ptm = mock(PlatformTransactionManager.class);
121 		PlatformTransactionManagerFacade.delegate = ptm;
122 
123 		assertTrue("Age should not be " + testBean.getAge(), testBean.getAge() == 666);
124 
125 		// Expect no methods
126 		verifyZeroInteractions(ptm);
127 
128 		// Install facade expecting a call
129 		final TransactionStatus ts = mock(TransactionStatus.class);
130 		ptm = new PlatformTransactionManager() {
131 			private boolean invoked;
132 			@Override
133 			public TransactionStatus getTransaction(TransactionDefinition def) throws TransactionException {
134 				if (invoked) {
135 					throw new IllegalStateException("getTransaction should not get invoked more than once");
136 				}
137 				invoked = true;
138 				if (!(def.getName().contains(DerivedTestBean.class.getName()) && def.getName().contains("setAge"))) {
139 					throw new IllegalStateException(
140 							"transaction name should contain class and method name: " + def.getName());
141 				}
142 				return ts;
143 			}
144 			@Override
145 			public void commit(TransactionStatus status) throws TransactionException {
146 				assertTrue(status == ts);
147 			}
148 			@Override
149 			public void rollback(TransactionStatus status) throws TransactionException {
150 				throw new IllegalStateException("rollback should not get invoked");
151 			}
152 		};
153 		PlatformTransactionManagerFacade.delegate = ptm;
154 
155 		// TODO same as old age to avoid ordering effect for now
156 		int age = 666;
157 		testBean.setAge(age);
158 		assertTrue(testBean.getAge() == age);
159 	}

181 	/**
182 	 * Test that we can set the target to a dynamic TargetSource.
183 	 */
184 	public void testDynamicTargetSource() throws NoSuchMethodException {
185 		// Install facade
186 		CallCountingTransactionManager txMan = new CallCountingTransactionManager();
187 		PlatformTransactionManagerFacade.delegate = txMan;
188 
189 		TestBean tb = (TestBean) factory.getBean("hotSwapped");
190 		assertEquals(666, tb.getAge());
191 		int newAge = 557;
192 		tb.setAge(newAge);
193 		assertEquals(newAge, tb.getAge());
194 
195 		TestBean target2 = new TestBean();
196 		target2.setAge(65);
197 		HotSwappableTargetSource ts = (HotSwappableTargetSource) factory.getBean("swapper");
198 		ts.swap(target2);
199 		assertEquals(target2.getAge(), tb.getAge());
200 		tb.setAge(newAge);
201 		assertEquals(newAge, target2.getAge());
202 
203 		assertEquals(0, txMan.inflight);
204 		assertEquals(2, txMan.commits);
205 		assertEquals(0, txMan.rollbacks);
206 	}

209 	public static class InvocationCounterPointcut extends StaticMethodMatcherPointcut {
210 
211 		int counter = 0;
212 
213 		@Override
214 		public boolean matches(Method method, Class<?> clazz) {
215 			counter++;
216 			return true;
217 		}
218 	}

221 	public static class InvocationCounterInterceptor implements MethodInterceptor {
222 
223 		int counter = 0;
224 
225 		@Override
226 		public Object invoke(MethodInvocation methodInvocation) throws Throwable {
227 			counter++;
228 			return methodInvocation.proceed();
229 		}
