38  /**
39   * Abstract base class that implements Spring's standard transaction workflow,
40   * serving as basis for concrete platform transaction managers like
41   * {@link org.springframework.transaction.jta.JtaTransactionManager}.
42   *
43   * <p>This base class provides the following workflow handling:
44   * <ul>
45   * <li>determines if there is an existing transaction;
46   * <li>applies the appropriate propagation behavior;
47   * <li>suspends and resumes transactions if necessary;
48   * <li>checks the rollback-only flag on commit;
49   * <li>applies the appropriate modification on rollback
50   * (actual rollback or setting rollback-only);
51   * <li>triggers registered synchronization callbacks
52   * (if transaction synchronization is active).
53   * </ul>
54   *
55   * <p>Subclasses have to implement specific template methods for specific
56   * states of a transaction, e.g.: begin, suspend, resume, commit, rollback.
57   * The most important of them are abstract and must be provided by a concrete
58   * implementation; for the rest, defaults are provided, so overriding is optional.
59   *
60   * <p>Transaction synchronization is a generic mechanism for registering callbacks
61   * that get invoked at transaction completion time. This is mainly used internally
62   * by the data access support classes for JDBC, Hibernate, JPA, etc when running
63   * within a JTA transaction: They register resources that are opened within the
64   * transaction for closing at transaction completion time, allowing e.g. for reuse
65   * of the same Hibernate Session within the transaction. The same mechanism can
66   * also be leveraged for custom synchronization needs in an application.
67   *
68   * <p>The state of this class is serializable, to allow for serializing the
69   * transaction strategy along with proxies that carry a transaction interceptor.
70   * It is up to subclasses if they wish to make their state to be serializable too.
71   * They should implement the {@code java.io.Serializable} marker interface in
72   * that case, and potentially a private {@code readObject()} method (according
73   * to Java serialization rules) if they need to restore any transient state.
74   *
75   * @author Juergen Hoeller
76   * @since 28.03.2003
77   * @see #setTransactionSynchronization
78   * @see TransactionSynchronizationManager
79   * @see org.springframework.transaction.jta.JtaTransactionManager
80   */
81  @SuppressWarnings("serial")
82  public abstract class AbstractPlatformTransactionManager implements PlatformTransactionManager, Serializable {

230 	/**
231 	 * Set whether to globally mark an existing transaction as rollback-only
232 	 * after a participating transaction failed.
233 	 * <p>Default is "true": If a participating transaction (e.g. with
234 	 * PROPAGATION_REQUIRES or PROPAGATION_SUPPORTS encountering an existing
235 	 * transaction) fails, the transaction will be globally marked as rollback-only.
236 	 * The only possible outcome of such a transaction is a rollback: The
237 	 * transaction originator <i>cannot</i> make the transaction commit anymore.
238 	 * <p>Switch this to "false" to let the transaction originator make the rollback
239 	 * decision. If a participating transaction fails with an exception, the caller
240 	 * can still decide to continue with a different path within the transaction.
241 	 * However, note that this will only work as long as all participating resources
242 	 * are capable of continuing towards a transaction commit even after a data access
243 	 * failure: This is generally not the case for a Hibernate Session, for example;
244 	 * neither is it for a sequence of JDBC insert/update/delete operations.
245 	 * <p><b>Note:</b>This flag only applies to an explicit rollback attempt for a
246 	 * subtransaction, typically caused by an exception thrown by a data access operation
247 	 * (where TransactionInterceptor will trigger a {@code PlatformTransactionManager.rollback()}
248 	 * call according to a rollback rule). If the flag is off, the caller can handle the exception
249 	 * and decide on a rollback, independent of the rollback rules of the subtransaction.
250 	 * This flag does, however, <i>not</i> apply to explicit {@code setRollbackOnly}
251 	 * calls on a {@code TransactionStatus}, which will always cause an eventual
252 	 * global rollback (as it might not throw an exception after the rollback-only call).
253 	 * <p>The recommended solution for handling failure of a subtransaction
254 	 * is a "nested transaction", where the global transaction can be rolled
255 	 * back to a savepoint taken at the beginning of the subtransaction.
256 	 * PROPAGATION_NESTED provides exactly those semantics; however, it will
257 	 * only work when nested transaction support is available. This is the case
258 	 * with DataSourceTransactionManager, but not with JtaTransactionManager.
259 	 * @see #setNestedTransactionAllowed
260 	 * @see org.springframework.transaction.jta.JtaTransactionManager
261 	 */
262 	public final void setGlobalRollbackOnParticipationFailure(boolean globalRollbackOnParticipationFailure) {
263 		this.globalRollbackOnParticipationFailure = globalRollbackOnParticipationFailure;
264 	}

327 	/**
328 	 * This implementation handles propagation behavior. Delegates to
329 	 * {@code doGetTransaction}, {@code isExistingTransaction}
330 	 * and {@code doBegin}.
331 	 * @see #doGetTransaction
332 	 * @see #isExistingTransaction
333 	 * @see #doBegin
334 	 */
335 	@Override
336 	public final TransactionStatus getTransaction(TransactionDefinition definition) throws TransactionException {
337 		Object transaction = doGetTransaction();
338 
339 		// Cache debug flag to avoid repeated checks.
340 		boolean debugEnabled = logger.isDebugEnabled();
341 
342 		if (definition == null) {
343 			// Use defaults if no transaction definition given.
344 			definition = new DefaultTransactionDefinition();
345 		}
346 
347 		if (isExistingTransaction(transaction)) {
348 			// Existing transaction found -> check propagation behavior to find out how to behave.
349 			return handleExistingTransaction(definition, transaction, debugEnabled);
350 		}
351 
352 		// Check definition settings for new transaction.
353 		if (definition.getTimeout() < TransactionDefinition.TIMEOUT_DEFAULT) {
354 			throw new InvalidTimeoutException("Invalid transaction timeout", definition.getTimeout());
355 		}
356 
357 		// No existing transaction found -> check propagation behavior to find out how to proceed.
358 		if (definition.getPropagationBehavior() == TransactionDefinition.PROPAGATION_MANDATORY) {
359 			throw new IllegalTransactionStateException(
360 					"No existing transaction found for transaction marked with propagation 'mandatory'");
361 		}
362 		else if (definition.getPropagationBehavior() == TransactionDefinition.PROPAGATION_REQUIRED ||
363 				definition.getPropagationBehavior() == TransactionDefinition.PROPAGATION_REQUIRES_NEW ||
364 				definition.getPropagationBehavior() == TransactionDefinition.PROPAGATION_NESTED) {
365 			SuspendedResourcesHolder suspendedResources = suspend(null);
366 			if (debugEnabled) {
367 				logger.debug("Creating new transaction with name [" + definition.getName() + "]: " + definition);
368 			}
369 			try {
370 				boolean newSynchronization = (getTransactionSynchronization() != SYNCHRONIZATION_NEVER);
371 				DefaultTransactionStatus status = newTransactionStatus(
372 						definition, transaction, true, newSynchronization, debugEnabled, suspendedResources);
373 				doBegin(transaction, definition);
374 				prepareSynchronization(status, definition);
375 				return status;
376 			}
377 			catch (RuntimeException ex) {
378 				resume(null, suspendedResources);
379 				throw ex;
380 			}
381 			catch (Error err) {
382 				resume(null, suspendedResources);
383 				throw err;
384 			}
385 		}
386 		else {
387 			// Create "empty" transaction: no actual transaction, but potentially synchronization.
388 			if (definition.getIsolationLevel() != TransactionDefinition.ISOLATION_DEFAULT && logger.isWarnEnabled()) {
389 				logger.warn("Custom isolation level specified but no actual transaction initiated; " +
390 						"isolation level will effectively be ignored: " + definition);
391 			}
392 			boolean newSynchronization = (getTransactionSynchronization() == SYNCHRONIZATION_ALWAYS);
393 			return prepareTransactionStatus(definition, null, true, newSynchronization, debugEnabled, null);
394 		}
395 	}
396 
397 	/**
398 	 * Create a TransactionStatus for an existing transaction.
399 	 */
400 	private TransactionStatus handleExistingTransaction(
401 			TransactionDefinition definition, Object transaction, boolean debugEnabled)
402 			throws TransactionException {
403 
404 		if (definition.getPropagationBehavior() == TransactionDefinition.PROPAGATION_NEVER) {
405 			throw new IllegalTransactionStateException(
406 					"Existing transaction found for transaction marked with propagation 'never'");
407 		}
408 
409 		if (definition.getPropagationBehavior() == TransactionDefinition.PROPAGATION_NOT_SUPPORTED) {
410 			if (debugEnabled) {
411 				logger.debug("Suspending current transaction");
412 			}
413 			Object suspendedResources = suspend(transaction);
414 			boolean newSynchronization = (getTransactionSynchronization() == SYNCHRONIZATION_ALWAYS);
415 			return prepareTransactionStatus(
416 					definition, null, false, newSynchronization, debugEnabled, suspendedResources);
417 		}
418 
419 		if (definition.getPropagationBehavior() == TransactionDefinition.PROPAGATION_REQUIRES_NEW) {
420 			if (debugEnabled) {
421 				logger.debug("Suspending current transaction, creating new transaction with name [" +
422 						definition.getName() + "]");
423 			}
424 			SuspendedResourcesHolder suspendedResources = suspend(transaction);
425 			try {
426 				boolean newSynchronization = (getTransactionSynchronization() != SYNCHRONIZATION_NEVER);
427 				DefaultTransactionStatus status = newTransactionStatus(
428 						definition, transaction, true, newSynchronization, debugEnabled, suspendedResources);
429 				doBegin(transaction, definition);
430 				prepareSynchronization(status, definition);
431 				return status;
432 			}
433 			catch (RuntimeException beginEx) {
434 				resumeAfterBeginException(transaction, suspendedResources, beginEx);
435 				throw beginEx;
436 			}
437 			catch (Error beginErr) {
438 				resumeAfterBeginException(transaction, suspendedResources, beginErr);
439 				throw beginErr;
440 			}
441 		}
442 
443 		if (definition.getPropagationBehavior() == TransactionDefinition.PROPAGATION_NESTED) {
444 			if (!isNestedTransactionAllowed()) {
445 				throw new NestedTransactionNotSupportedException(
446 						"Transaction manager does not allow nested transactions by default - " +
447 						"specify 'nestedTransactionAllowed' property with value 'true'");
448 			}
449 			if (debugEnabled) {
450 				logger.debug("Creating nested transaction with name [" + definition.getName() + "]");
451 			}
452 			if (useSavepointForNestedTransaction()) {
453 				// Create savepoint within existing Spring-managed transaction,
454 				// through the SavepointManager API implemented by TransactionStatus.
455 				// Usually uses JDBC 3.0 savepoints. Never activates Spring synchronization.
456 				DefaultTransactionStatus status =
457 						prepareTransactionStatus(definition, transaction, false, false, debugEnabled, null);
458 				status.createAndHoldSavepoint();
459 				return status;
460 			}
461 			else {
462 				// Nested transaction through nested begin and commit/rollback calls.
463 				// Usually only for JTA: Spring synchronization might get activated here
464 				// in case of a pre-existing JTA transaction.
465 				boolean newSynchronization = (getTransactionSynchronization() != SYNCHRONIZATION_NEVER);
466 				DefaultTransactionStatus status = newTransactionStatus(
467 						definition, transaction, true, newSynchronization, debugEnabled, null);
468 				doBegin(transaction, definition);
469 				prepareSynchronization(status, definition);
470 				return status;
471 			}
472 		}
473 
474 		// Assumably PROPAGATION_SUPPORTS or PROPAGATION_REQUIRED.
475 		if (debugEnabled) {
476 			logger.debug("Participating in existing transaction");
477 		}
478 		if (isValidateExistingTransaction()) {
479 			if (definition.getIsolationLevel() != TransactionDefinition.ISOLATION_DEFAULT) {
480 				Integer currentIsolationLevel = TransactionSynchronizationManager.getCurrentTransactionIsolationLevel();
481 				if (currentIsolationLevel == null || currentIsolationLevel != definition.getIsolationLevel()) {
482 					Constants isoConstants = DefaultTransactionDefinition.constants;
483 					throw new IllegalTransactionStateException("Participating transaction with definition [" +
484 							definition + "] specifies isolation level which is incompatible with existing transaction: " +
485 							(currentIsolationLevel != null ?
486 									isoConstants.toCode(currentIsolationLevel, DefaultTransactionDefinition.PREFIX_ISOLATION) :
487 									"(unknown)"));
488 				}
489 			}
490 			if (!definition.isReadOnly()) {
491 				if (TransactionSynchronizationManager.isCurrentTransactionReadOnly()) {
492 					throw new IllegalTransactionStateException("Participating transaction with definition [" +
493 							definition + "] is not marked as read-only but existing transaction is");
494 				}
495 			}
496 		}
497 		boolean newSynchronization = (getTransactionSynchronization() != SYNCHRONIZATION_NEVER);
498 		return prepareTransactionStatus(definition, transaction, false, newSynchronization, debugEnabled, null);
499 	}

502 	 * Create a new TransactionStatus for the given arguments,
503 	 * also initializing transaction synchronization as appropriate.
504 	 * @see #newTransactionStatus
505 	 * @see #prepareTransactionStatus
506 	 */
507 	protected final DefaultTransactionStatus prepareTransactionStatus(
508 			TransactionDefinition definition, Object transaction, boolean newTransaction,
509 			boolean newSynchronization, boolean debug, Object suspendedResources) {
510 
511 		DefaultTransactionStatus status = newTransactionStatus(
512 				definition, transaction, newTransaction, newSynchronization, debug, suspendedResources);
513 		prepareSynchronization(status, definition);
514 		return status;
515 	}

517 	/**
518 	 * Create a rae TransactionStatus instance for the given arguments.
519 	 */
520 	protected DefaultTransactionStatus newTransactionStatus(
521 			TransactionDefinition definition, Object transaction, boolean newTransaction,
522 			boolean newSynchronization, boolean debug, Object suspendedResources) {
523 
524 		boolean actualNewSynchronization = newSynchronization &&
525 				!TransactionSynchronizationManager.isSynchronizationActive();
526 		return new DefaultTransactionStatus(
527 				transaction, newTransaction, actualNewSynchronization,
528 				definition.isReadOnly(), debug, suspendedResources);
529 	}

531 	/**
532 	 * Initialize transaction synchronization as appropriate.
533 	 */
534 	protected void prepareSynchronization(DefaultTransactionStatus status, TransactionDefinition definition) {
535 		if (status.isNewSynchronization()) {
536 			TransactionSynchronizationManager.setActualTransactionActive(status.hasTransaction());
537 			TransactionSynchronizationManager.setCurrentTransactionIsolationLevel(
538 					definition.getIsolationLevel() != TransactionDefinition.ISOLATION_DEFAULT ?
539 							definition.getIsolationLevel() : null);
540 			TransactionSynchronizationManager.setCurrentTransactionReadOnly(definition.isReadOnly());
541 			TransactionSynchronizationManager.setCurrentTransactionName(definition.getName());
542 			TransactionSynchronizationManager.initSynchronization();
543 		}
544 	}

547 	 * Determine the actual timeout to use for the given definition.
548 	 * Will fall back to this manager's default timeout if the
549 	 * transaction definition doesn't specify a non-default value.
550 	 * @param definition the transaction definition
551 	 * @return the actual timeout to use
552 	 * @see org.springframework.transaction.TransactionDefinition#getTimeout()
553 	 * @see #setDefaultTimeout
554 	 */
555 	protected int determineTimeout(TransactionDefinition definition) {
556 		if (definition.getTimeout() != TransactionDefinition.TIMEOUT_DEFAULT) {
557 			return definition.getTimeout();
558 		}
559 		return this.defaultTimeout;

563 	/**
564 	 * Suspend the given transaction. Suspends transaction synchronization first,
565 	 * then delegates to the {@code doSuspend} template method.
566 	 * @param transaction the current transaction object
567 	 * (or {@code null} to just suspend active synchronizations, if any)
568 	 * @return an object that holds suspended resources
569 	 * (or {@code null} if neither transaction nor synchronization active)
570 	 * @see #doSuspend
571 	 * @see #resume
572 	 */
573 	protected final SuspendedResourcesHolder suspend(Object transaction) throws TransactionException {
574 		if (TransactionSynchronizationManager.isSynchronizationActive()) {
575 			List<TransactionSynchronization> suspendedSynchronizations = doSuspendSynchronization();
576 			try {
577 				Object suspendedResources = null;
578 				if (transaction != null) {
579 					suspendedResources = doSuspend(transaction);
580 				}
581 				String name = TransactionSynchronizationManager.getCurrentTransactionName();
582 				TransactionSynchronizationManager.setCurrentTransactionName(null);
583 				boolean readOnly = TransactionSynchronizationManager.isCurrentTransactionReadOnly();
584 				TransactionSynchronizationManager.setCurrentTransactionReadOnly(false);
585 				Integer isolationLevel = TransactionSynchronizationManager.getCurrentTransactionIsolationLevel();
586 				TransactionSynchronizationManager.setCurrentTransactionIsolationLevel(null);
587 				boolean wasActive = TransactionSynchronizationManager.isActualTransactionActive();
588 				TransactionSynchronizationManager.setActualTransactionActive(false);
589 				return new SuspendedResourcesHolder(
590 						suspendedResources, suspendedSynchronizations, name, readOnly, isolationLevel, wasActive);
591 			}
592 			catch (RuntimeException ex) {
593 				// doSuspend failed - original transaction is still active...
594 				doResumeSynchronization(suspendedSynchronizations);
595 				throw ex;
596 			}
597 			catch (Error err) {
598 				// doSuspend failed - original transaction is still active...
599 				doResumeSynchronization(suspendedSynchronizations);
600 				throw err;
601 			}
602 		}
603 		else if (transaction != null) {
604 			// Transaction active but no synchronization active.
605 			Object suspendedResources = doSuspend(transaction);
606 			return new SuspendedResourcesHolder(suspendedResources);
607 		}
608 		else {
609 			// Neither transaction nor synchronization active.
610 			return null;
611 		}
612 	}

616 	 * template method first, then resuming transaction synchronization.
617 	 * @param transaction the current transaction object
618 	 * @param resourcesHolder the object that holds suspended resources,
619 	 * as returned by {@code suspend} (or {@code null} to just

621 	 * @see #doResume
622 	 * @see #suspend

626 
627 		if (resourcesHolder != null) {

629 			if (suspendedResources != null) {

637 				TransactionSynchronizationManager.setCurrentTransactionName(resourcesHolder.name);
638 				doResumeSynchronization(suspendedSynchronizations);
639 			}
640 		}

692 	/**
693 	 * This implementation of commit handles participating in existing
694 	 * transactions and programmatic rollback requests.
695 	 * Delegates to {@code isRollbackOnly}, {@code doCommit}
696 	 * and {@code rollback}.
697 	 * @see org.springframework.transaction.TransactionStatus#isRollbackOnly()
698 	 * @see #doCommit
699 	 * @see #rollback
700 	 */
701 	@Override
702 	public final void commit(TransactionStatus status) throws TransactionException {
703 		if (status.isCompleted()) {
704 			throw new IllegalTransactionStateException(
705 					"Transaction is already completed - do not call commit or rollback more than once per transaction");
706 		}
707 
708 		DefaultTransactionStatus defStatus = (DefaultTransactionStatus) status;
709 		if (defStatus.isLocalRollbackOnly()) {
710 			if (defStatus.isDebug()) {
711 				logger.debug("Transactional code has requested rollback");
712 			}
713 			processRollback(defStatus);
714 			return;
715 		}
716 		if (!shouldCommitOnGlobalRollbackOnly() && defStatus.isGlobalRollbackOnly()) {
717 			if (defStatus.isDebug()) {
718 				logger.debug("Global transaction is marked as rollback-only but transactional code requested commit");
719 			}
720 			processRollback(defStatus);
721 			// Throw UnexpectedRollbackException only at outermost transaction boundary
722 			// or if explicitly asked to.
723 			if (status.isNewTransaction() || isFailEarlyOnGlobalRollbackOnly()) {
724 				throw new UnexpectedRollbackException(
725 						"Transaction rolled back because it has been marked as rollback-only");
726 			}
727 			return;
728 		}
729 
730 		processCommit(defStatus);
731 	}
732 
733 	/**
734 	 * Process an actual commit.
735 	 * Rollback-only flags have already been checked and applied.
736 	 * @param status object representing the transaction
737 	 * @throws TransactionException in case of commit failure
738 	 */
739 	private void processCommit(DefaultTransactionStatus status) throws TransactionException {
740 		try {
741 			boolean beforeCompletionInvoked = false;
742 			try {
743 				prepareForCommit(status);
744 				triggerBeforeCommit(status);
745 				triggerBeforeCompletion(status);
746 				beforeCompletionInvoked = true;
747 				boolean globalRollbackOnly = false;
748 				if (status.isNewTransaction() || isFailEarlyOnGlobalRollbackOnly()) {
749 					globalRollbackOnly = status.isGlobalRollbackOnly();
750 				}
751 				if (status.hasSavepoint()) {
752 					if (status.isDebug()) {
753 						logger.debug("Releasing transaction savepoint");
754 					}
755 					status.releaseHeldSavepoint();
756 				}
757 				else if (status.isNewTransaction()) {
758 					if (status.isDebug()) {
759 						logger.debug("Initiating transaction commit");
760 					}
761 					doCommit(status);
762 				}
763 				// Throw UnexpectedRollbackException if we have a global rollback-only
764 				// marker but still didn't get a corresponding exception from commit.
765 				if (globalRollbackOnly) {
766 					throw new UnexpectedRollbackException(
767 							"Transaction silently rolled back because it has been marked as rollback-only");
768 				}
769 			}
770 			catch (UnexpectedRollbackException ex) {
771 				// can only be caused by doCommit
772 				triggerAfterCompletion(status, TransactionSynchronization.STATUS_ROLLED_BACK);
773 				throw ex;
774 			}
775 			catch (TransactionException ex) {
776 				// can only be caused by doCommit
777 				if (isRollbackOnCommitFailure()) {
778 					doRollbackOnCommitException(status, ex);
779 				}
780 				else {
781 					triggerAfterCompletion(status, TransactionSynchronization.STATUS_UNKNOWN);
782 				}
783 				throw ex;
784 			}
785 			catch (RuntimeException ex) {
786 				if (!beforeCompletionInvoked) {
787 					triggerBeforeCompletion(status);
788 				}
789 				doRollbackOnCommitException(status, ex);
790 				throw ex;
791 			}
792 			catch (Error err) {
793 				if (!beforeCompletionInvoked) {
794 					triggerBeforeCompletion(status);
795 				}
796 				doRollbackOnCommitException(status, err);
797 				throw err;
798 			}
799 
800 			// Trigger afterCommit callbacks, with an exception thrown there
801 			// propagated to callers but the transaction still considered as committed.
802 			try {
803 				triggerAfterCommit(status);
804 			}
805 			finally {
806 				triggerAfterCompletion(status, TransactionSynchronization.STATUS_COMMITTED);
807 			}
808 
809 		}
810 		finally {
811 			cleanupAfterCompletion(status);
812 		}
813 	}

815 	/**
816 	 * This implementation of rollback handles participating in existing
817 	 * transactions. Delegates to {@code doRollback} and
818 	 * {@code doSetRollbackOnly}.
819 	 * @see #doRollback
820 	 * @see #doSetRollbackOnly
821 	 */
822 	@Override
823 	public final void rollback(TransactionStatus status) throws TransactionException {
824 		if (status.isCompleted()) {
825 			throw new IllegalTransactionStateException(
826 					"Transaction is already completed - do not call commit or rollback more than once per transaction");
827 		}
828 
829 		DefaultTransactionStatus defStatus = (DefaultTransactionStatus) status;
830 		processRollback(defStatus);
831 	}
832 
833 	/**
834 	 * Process an actual rollback.
835 	 * The completed flag has already been checked.
836 	 * @param status object representing the transaction
837 	 * @throws TransactionException in case of rollback failure
838 	 */
839 	private void processRollback(DefaultTransactionStatus status) {
840 		try {
841 			try {
842 				triggerBeforeCompletion(status);
843 				if (status.hasSavepoint()) {
844 					if (status.isDebug()) {
845 						logger.debug("Rolling back transaction to savepoint");
846 					}
847 					status.rollbackToHeldSavepoint();
848 				}
849 				else if (status.isNewTransaction()) {
850 					if (status.isDebug()) {
851 						logger.debug("Initiating transaction rollback");
852 					}
853 					doRollback(status);
854 				}
855 				else if (status.hasTransaction()) {
856 					if (status.isLocalRollbackOnly() || isGlobalRollbackOnParticipationFailure()) {
857 						if (status.isDebug()) {
858 							logger.debug("Participating transaction failed - marking existing transaction as rollback-only");
859 						}
860 						doSetRollbackOnly(status);
861 					}
862 					else {
863 						if (status.isDebug()) {
864 							logger.debug("Participating transaction failed - letting transaction originator decide on rollback");
865 						}
866 					}
867 				}
868 				else {
869 					logger.debug("Should roll back transaction but cannot - no transaction available");
870 				}
871 			}
872 			catch (RuntimeException ex) {
873 				triggerAfterCompletion(status, TransactionSynchronization.STATUS_UNKNOWN);
874 				throw ex;
875 			}
876 			catch (Error err) {
877 				triggerAfterCompletion(status, TransactionSynchronization.STATUS_UNKNOWN);
878 				throw err;
879 			}
880 			triggerAfterCompletion(status, TransactionSynchronization.STATUS_ROLLED_BACK);
881 		}
882 		finally {
883 			cleanupAfterCompletion(status);
884 		}
885 	}

887 	/**
888 	 * Invoke {@code doRollback}, handling rollback exceptions properly.
889 	 * @param status object representing the transaction
890 	 * @param ex the thrown application exception or error
891 	 * @throws TransactionException in case of rollback failure
892 	 * @see #doRollback
893 	 */
894 	private void doRollbackOnCommitException(DefaultTransactionStatus status, Throwable ex) throws TransactionException {
895 		try {
896 			if (status.isNewTransaction()) {
897 				if (status.isDebug()) {
898 					logger.debug("Initiating transaction rollback after commit exception", ex);
899 				}
900 				doRollback(status);
901 			}
902 			else if (status.hasTransaction() && isGlobalRollbackOnParticipationFailure()) {
903 				if (status.isDebug()) {
904 					logger.debug("Marking existing transaction as rollback-only after commit exception", ex);
905 				}
906 				doSetRollbackOnly(status);
907 			}
908 		}
909 		catch (RuntimeException rbex) {
910 			logger.error("Commit exception overridden by rollback exception", ex);
911 			triggerAfterCompletion(status, TransactionSynchronization.STATUS_UNKNOWN);
912 			throw rbex;
913 		}
914 		catch (Error rberr) {
915 			logger.error("Commit exception overridden by rollback exception", ex);
916 			triggerAfterCompletion(status, TransactionSynchronization.STATUS_UNKNOWN);
917 			throw rberr;
918 		}
919 		triggerAfterCompletion(status, TransactionSynchronization.STATUS_ROLLED_BACK);
920 	}

963 	 * Trigger {@code afterCompletion} callbacks.
964 	 * @param status object representing the transaction
965 	 * @param completionStatus completion status according to TransactionSynchronization constants
966 	 */
967 	private void triggerAfterCompletion(DefaultTransactionStatus status, int completionStatus) {
968 		if (status.isNewSynchronization()) {
969 			List<TransactionSynchronization> synchronizations = TransactionSynchronizationManager.getSynchronizations();
970 			if (!status.hasTransaction() || status.isNewTransaction()) {
971 				if (status.isDebug()) {
972 					logger.trace("Triggering afterCompletion synchronization");
973 				}
974 				// No transaction or new transaction for the current scope ->
975 				// invoke the afterCompletion callbacks immediately
976 				invokeAfterCompletion(synchronizations, completionStatus);
977 			}
978 			else if (!synchronizations.isEmpty()) {
979 				// Existing transaction that we participate in, controlled outside
980 				// of the scope of this Spring transaction manager -> try to register
981 				// an afterCompletion callback with the existing (JTA) transaction.
982 				registerAfterCompletionWithExistingTransaction(status.getTransaction(), synchronizations);
983 			}
984 		}
