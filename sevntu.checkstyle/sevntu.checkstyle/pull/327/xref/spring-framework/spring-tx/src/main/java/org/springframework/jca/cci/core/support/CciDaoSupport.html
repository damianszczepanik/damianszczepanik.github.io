17  package org.springframework.jca.cci.core.support;
18  
19  import javax.resource.cci.Connection;
20  import javax.resource.cci.ConnectionFactory;
21  import javax.resource.cci.ConnectionSpec;
22  
23  import org.springframework.dao.support.DaoSupport;
24  import org.springframework.jca.cci.CannotGetCciConnectionException;
25  import org.springframework.jca.cci.connection.ConnectionFactoryUtils;
26  import org.springframework.jca.cci.core.CciTemplate;
27  
28  /**
29   * Convenient super class for CCI-based data access objects.
30   *
31   * <p>Requires a {@link javax.resource.cci.ConnectionFactory} to be set,
32   * providing a {@link org.springframework.jca.cci.core.CciTemplate} based
33   * on it to subclasses through the {@link #getCciTemplate()} method.
34   *
35   * <p>This base class is mainly intended for CciTemplate usage but can
36   * also be used when working with a Connection directly or when using
37   * {@code org.springframework.jca.cci.object} classes.
38   *
39   * @author Thierry Templier
40   * @author Juergen Hoeller
41   * @since 1.2
42   * @see #setConnectionFactory
43   * @see #getCciTemplate
44   * @see org.springframework.jca.cci.core.CciTemplate
45   */
46  public abstract class CciDaoSupport extends DaoSupport {
47  
48  	private CciTemplate cciTemplate;
49  
50  
51  	/**
52  	 * Set the ConnectionFactory to be used by this DAO.
53  	 */
54  	public final void setConnectionFactory(ConnectionFactory connectionFactory) {
55  		if (this.cciTemplate == null || connectionFactory != this.cciTemplate.getConnectionFactory()) {
56  		  this.cciTemplate = createCciTemplate(connectionFactory);
57  		}
58  	}
59  
60  	/**
61  	 * Create a CciTemplate for the given ConnectionFactory.
62  	 * Only invoked if populating the DAO with a ConnectionFactory reference!
63  	 * <p>Can be overridden in subclasses to provide a CciTemplate instance
64  	 * with different configuration, or a custom CciTemplate subclass.
65  	 * @param connectionFactory the CCI ConnectionFactory to create a CciTemplate for
66  	 * @return the new CciTemplate instance
67  	 * @see #setConnectionFactory(javax.resource.cci.ConnectionFactory)
68  	 */
69  	protected CciTemplate createCciTemplate(ConnectionFactory connectionFactory) {
70  		return new CciTemplate(connectionFactory);
71  	}
72  
73  	/**
74  	 * Return the ConnectionFactory used by this DAO.
75  	 */
76  	public final ConnectionFactory getConnectionFactory() {
77  		return this.cciTemplate.getConnectionFactory();
78  	}
79  
80  	/**
81  	 * Set the CciTemplate for this DAO explicitly,
82  	 * as an alternative to specifying a ConnectionFactory.
83  	 */
84  	public final void setCciTemplate(CciTemplate cciTemplate) {
85  		this.cciTemplate = cciTemplate;
86  	}
87  
88  	/**
89  	 * Return the CciTemplate for this DAO,
90  	 * pre-initialized with the ConnectionFactory or set explicitly.
91  	 */
92  	public final CciTemplate getCciTemplate() {
93  	  return this.cciTemplate;
94  	}
95  
96  	@Override
97  	protected final void checkDaoConfig() {
98  		if (this.cciTemplate == null) {
99  			throw new IllegalArgumentException("'connectionFactory' or 'cciTemplate' is required");
100 		}
101 	}
102 
103 
104 	/**
105 	 * Obtain a CciTemplate derived from the main template instance,
106 	 * inheriting the ConnectionFactory and other settings but
107 	 * overriding the ConnectionSpec used for obtaining Connections.
108 	 * @param connectionSpec the CCI ConnectionSpec that the returned
109 	 * template instance is supposed to obtain Connections for
110 	 * @return the derived template instance
111 	 * @see org.springframework.jca.cci.core.CciTemplate#getDerivedTemplate(javax.resource.cci.ConnectionSpec)
112 	 */
113 	protected final CciTemplate getCciTemplate(ConnectionSpec connectionSpec) {
114 		return getCciTemplate().getDerivedTemplate(connectionSpec);
115 	}
116 
117 	/**
118 	 * Get a CCI Connection, either from the current transaction or a new one.
119 	 * @return the CCI Connection
120 	 * @throws org.springframework.jca.cci.CannotGetCciConnectionException
121 	 * if the attempt to get a Connection failed
122 	 * @see org.springframework.jca.cci.connection.ConnectionFactoryUtils#getConnection(javax.resource.cci.ConnectionFactory)
123 	 */
124 	protected final Connection getConnection() throws CannotGetCciConnectionException {
125 		return ConnectionFactoryUtils.getConnection(getConnectionFactory());
126 	}
127 
128 	/**
129 	 * Close the given CCI Connection, created via this bean's ConnectionFactory,
130 	 * if it isn't bound to the thread.
131 	 * @param con Connection to close
132 	 * @see org.springframework.jca.cci.connection.ConnectionFactoryUtils#releaseConnection
133 	 */
134 	protected final void releaseConnection(Connection con) {
135 		ConnectionFactoryUtils.releaseConnection(con, getConnectionFactory());
136 	}
137 
138 }
