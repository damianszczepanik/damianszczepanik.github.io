17  package org.springframework.jca.work.glassfish;
18  
19  import java.lang.reflect.Method;
20  import javax.resource.spi.work.WorkManager;
21  
22  import org.springframework.jca.work.WorkManagerTaskExecutor;
23  import org.springframework.util.ReflectionUtils;
24  
25  /**
26   * Spring TaskExecutor adapter for the GlassFish JCA WorkManager.
27   * Can be defined in web applications to make a TaskExecutor reference
28   * available, talking to the GlassFish WorkManager (thread pool) underneath.
29   *
30   * <p>This is the GlassFish equivalent of the CommonJ
31   * {@link org.springframework.scheduling.commonj.WorkManagerTaskExecutor}
32   * adapter for WebLogic and WebSphere.
33   *
34   * <p>Note: On GlassFish 4 and higher, a
35   * {@link org.springframework.scheduling.concurrent.DefaultManagedTaskExecutor}
36   * should be preferred, following JSR-236 support in Java EE 7.
37   *
38   * @author Juergen Hoeller
39   * @since 2.5.2
40   */
41  public class GlassFishWorkManagerTaskExecutor extends WorkManagerTaskExecutor {
42  
43  	private static final String WORK_MANAGER_FACTORY_CLASS = "com.sun.enterprise.connectors.work.WorkManagerFactory";
44  
45  	private final Method getWorkManagerMethod;
46  
47  
48  	public GlassFishWorkManagerTaskExecutor() {
49  		try {
50  			Class<?> wmf = getClass().getClassLoader().loadClass(WORK_MANAGER_FACTORY_CLASS);
51  			this.getWorkManagerMethod = wmf.getMethod("getWorkManager", String.class);
52  		}
53  		catch (Exception ex) {
54  			throw new IllegalStateException(
55  					"Could not initialize GlassFishWorkManagerTaskExecutor because GlassFish API is not available", ex);
56  		}
57  	}
58  
59  	/**
60  	 * Identify a specific GlassFish thread pool to talk to.
61  	 * <p>The thread pool name matches the resource adapter name
62  	 * in default RAR deployment scenarios.
63  	 */
64  	public void setThreadPoolName(String threadPoolName) {
65  		WorkManager wm = (WorkManager) ReflectionUtils.invokeMethod(this.getWorkManagerMethod, null, threadPoolName);
66  		if (wm == null) {
67  			throw new IllegalArgumentException("Specified thread pool name '" + threadPoolName +
68  					"' does not correspond to an actual pool definition in GlassFish. Check your configuration!");
69  		}
70  		setWorkManager(wm);
71  	}
72  
73  	/**
74  	 * Obtains GlassFish's default thread pool.
75  	 */
76  	@Override
77  	protected WorkManager getDefaultWorkManager() {
78  		return (WorkManager) ReflectionUtils.invokeMethod(this.getWorkManagerMethod, null, new Object[] {null});
79  	}
80  
81  }
