17  package org.springframework.transaction.interceptor;
18  
19  import org.springframework.transaction.support.DefaultTransactionDefinition;
20  
21  /**
22   * Transaction attribute that takes the EJB approach to rolling
23   * back on runtime, but not checked, exceptions.
24   *
25   * @author Rod Johnson
26   * @since 16.03.2003
27   */
28  @SuppressWarnings("serial")
29  public class DefaultTransactionAttribute extends DefaultTransactionDefinition implements TransactionAttribute {
30  
31  	private String qualifier;
32  
33  
34  	/**
35  	 * Create a new DefaultTransactionAttribute, with default settings.
36  	 * Can be modified through bean property setters.
37  	 * @see #setPropagationBehavior
38  	 * @see #setIsolationLevel
39  	 * @see #setTimeout
40  	 * @see #setReadOnly
41  	 * @see #setName
42  	 */
43  	public DefaultTransactionAttribute() {
44  		super();
45  	}
46  
47  	/**
48  	 * Copy constructor. Definition can be modified through bean property setters.
49  	 * @see #setPropagationBehavior
50  	 * @see #setIsolationLevel
51  	 * @see #setTimeout
52  	 * @see #setReadOnly
53  	 * @see #setName
54  	 */
55  	public DefaultTransactionAttribute(TransactionAttribute other) {
56  		super(other);
57  	}
58  
59  	/**
60  	 * Create a new DefaultTransactionAttribute with the the given
61  	 * propagation behavior. Can be modified through bean property setters.
62  	 * @param propagationBehavior one of the propagation constants in the
63  	 * TransactionDefinition interface
64  	 * @see #setIsolationLevel
65  	 * @see #setTimeout
66  	 * @see #setReadOnly
67  	 */
68  	public DefaultTransactionAttribute(int propagationBehavior) {
69  		super(propagationBehavior);
70  	}
71  
72  
73  	/**
74  	 * Associate a qualifier value with this transaction attribute.
75  	 * <p>This may be used for choosing a corresponding transaction manager
76  	 * to process this specific transaction.
77  	 */
78  	public void setQualifier(String qualifier) {
79  		this.qualifier = qualifier;
80  	}
81  
82  	/**
83  	 * Return a qualifier value associated with this transaction attribute.
84  	 */
85  	@Override
86  	public String getQualifier() {
87  		return this.qualifier;
88  	}
89  
90  	/**
91  	 * The default behavior is as with EJB: rollback on unchecked exception.
92  	 * Additionally attempt to rollback on Error.
93  	 * <p>This is consistent with TransactionTemplate's default behavior.
94  	 */
95  	@Override
96  	public boolean rollbackOn(Throwable ex) {
97  		return (ex instanceof RuntimeException || ex instanceof Error);
98  	}
99  
100 
101 	/**
102 	 * Return an identifying description for this transaction attribute.
103 	 * <p>Available to subclasses, for inclusion in their {@code toString()} result.
104 	 */
105 	protected final StringBuilder getAttributeDescription() {
106 		StringBuilder result = getDefinitionDescription();
107 		if (this.qualifier != null) {
108 			result.append("; '").append(this.qualifier).append("'");
109 		}
110 		return result;
111 	}
112 
113 }
