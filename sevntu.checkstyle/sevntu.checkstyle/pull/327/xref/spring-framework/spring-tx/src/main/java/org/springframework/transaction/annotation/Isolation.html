17  package org.springframework.transaction.annotation;
18  
19  import org.springframework.transaction.TransactionDefinition;
20  
21  /**
22   * Enumeration that represents transaction isolation levels for use
23   * with the {@link Transactional} annotation, corresponding to the
24   * {@link TransactionDefinition} interface.
25   *
26   * @author Colin Sampaleanu
27   * @author Juergen Hoeller
28   * @since 1.2
29   */
30  public enum Isolation {
31  
32  	/**
33  	 * Use the default isolation level of the underlying datastore.
34  	 * All other levels correspond to the JDBC isolation levels.
35  	 * @see java.sql.Connection
36  	 */
37  	DEFAULT(TransactionDefinition.ISOLATION_DEFAULT),
38  
39  	/**
40  	 * A constant indicating that dirty reads, non-repeatable reads and phantom reads
41  	 * can occur. This level allows a row changed by one transaction to be read by
42  	 * another transaction before any changes in that row have been committed
43  	 * (a "dirty read"). If any of the changes are rolled back, the second
44  	 * transaction will have retrieved an invalid row.
45  	 * @see java.sql.Connection#TRANSACTION_READ_UNCOMMITTED
46  	 */
47  	READ_UNCOMMITTED(TransactionDefinition.ISOLATION_READ_UNCOMMITTED),
48  
49  	/**
50  	 * A constant indicating that dirty reads are prevented; non-repeatable reads
51  	 * and phantom reads can occur. This level only prohibits a transaction
52  	 * from reading a row with uncommitted changes in it.
53  	 * @see java.sql.Connection#TRANSACTION_READ_COMMITTED
54  	 */
55  	READ_COMMITTED(TransactionDefinition.ISOLATION_READ_COMMITTED),
56  
57  	/**
58  	 * A constant indicating that dirty reads and non-repeatable reads are
59  	 * prevented; phantom reads can occur. This level prohibits a transaction
60  	 * from reading a row with uncommitted changes in it, and it also prohibits
61  	 * the situation where one transaction reads a row, a second transaction
62  	 * alters the row, and the first transaction rereads the row, getting
63  	 * different values the second time (a "non-repeatable read").
64  	 * @see java.sql.Connection#TRANSACTION_REPEATABLE_READ
65  	 */
66  	REPEATABLE_READ(TransactionDefinition.ISOLATION_REPEATABLE_READ),
67  
68  	/**
69  	 * A constant indicating that dirty reads, non-repeatable reads and phantom
70  	 * reads are prevented. This level includes the prohibitions in
71  	 * {@code ISOLATION_REPEATABLE_READ} and further prohibits the situation
72  	 * where one transaction reads all rows that satisfy a {@code WHERE}
73  	 * condition, a second transaction inserts a row that satisfies that
74  	 * {@code WHERE} condition, and the first transaction rereads for the
75  	 * same condition, retrieving the additional "phantom" row in the second read.
76  	 * @see java.sql.Connection#TRANSACTION_SERIALIZABLE
77  	 */
78  	SERIALIZABLE(TransactionDefinition.ISOLATION_SERIALIZABLE);
79  
80  
81  	private final int value;
82  
83  
84  	Isolation(int value) { this.value = value; }
85  
86  	public int value() { return this.value; }
87  
88  }
