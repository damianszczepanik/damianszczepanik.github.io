17  package org.springframework.transaction;
18  
19  import java.lang.reflect.Method;
20  
21  import junit.framework.TestCase;
22  
23  import org.springframework.aop.support.AopUtils;
24  import org.springframework.context.ApplicationContext;
25  import org.springframework.context.support.ClassPathXmlApplicationContext;
26  import org.springframework.tests.sample.beans.ITestBean;
27  import org.springframework.tests.transaction.CallCountingTransactionManager;
28  import org.springframework.transaction.interceptor.TransactionAttribute;
29  import org.springframework.transaction.interceptor.TransactionAttributeSource;
30  import org.springframework.transaction.interceptor.TransactionInterceptor;
31  
32  /**
33   * @author Rob Harrop
34   * @author Adrian Colyer
35   */
36  public class TxNamespaceHandlerTests extends TestCase {
37  
38  	private ApplicationContext context;
39  
40  	private Method getAgeMethod;
41  
42  	private Method setAgeMethod;
43  
44  	@Override
45  	public void setUp() throws Exception {
46  		this.context = new ClassPathXmlApplicationContext("txNamespaceHandlerTests.xml", getClass());
47  		this.getAgeMethod = ITestBean.class.getMethod("getAge", new Class[0]);
48  		this.setAgeMethod = ITestBean.class.getMethod("setAge", new Class[] {int.class});
49  	}
50  
51  	public void testIsProxy() throws Exception {
52  		ITestBean bean = getTestBean();
53  		assertTrue("testBean is not a proxy", AopUtils.isAopProxy(bean));
54  	}
55  
56  	public void testInvokeTransactional() throws Exception {
57  		ITestBean testBean = getTestBean();
58  		CallCountingTransactionManager ptm = (CallCountingTransactionManager) context.getBean("transactionManager");
59  
60  		// try with transactional
61  		assertEquals("Should not have any started transactions", 0, ptm.begun);
62  		testBean.getName();
63  		assertTrue(ptm.lastDefinition.isReadOnly());
64  		assertEquals("Should have 1 started transaction", 1, ptm.begun);
65  		assertEquals("Should have 1 committed transaction", 1, ptm.commits);
66  
67  		// try with non-transaction
68  		testBean.haveBirthday();
69  		assertEquals("Should not have started another transaction", 1, ptm.begun);
70  
71  		// try with exceptional
72  		try {
73  			testBean.exceptional(new IllegalArgumentException("foo"));
74  			fail("Should NEVER get here");
75  		}
76  		catch (Throwable throwable) {
77  			assertEquals("Should have another started transaction", 2, ptm.begun);
78  			assertEquals("Should have 1 rolled back transaction", 1, ptm.rollbacks);
79  		}
80  	}
81  
82  	public void testRollbackRules() {
83  		TransactionInterceptor txInterceptor = (TransactionInterceptor) context.getBean("txRollbackAdvice");
84  		TransactionAttributeSource txAttrSource = txInterceptor.getTransactionAttributeSource();
85  		TransactionAttribute txAttr = txAttrSource.getTransactionAttribute(getAgeMethod,ITestBean.class);
86  		assertTrue("should be configured to rollback on Exception",txAttr.rollbackOn(new Exception()));
87  
88  		txAttr = txAttrSource.getTransactionAttribute(setAgeMethod, ITestBean.class);
89  		assertFalse("should not rollback on RuntimeException",txAttr.rollbackOn(new RuntimeException()));
90  	}
91  
92  	private ITestBean getTestBean() {
93  		return (ITestBean)context.getBean("testBean");
94  	}
95  
96  }
