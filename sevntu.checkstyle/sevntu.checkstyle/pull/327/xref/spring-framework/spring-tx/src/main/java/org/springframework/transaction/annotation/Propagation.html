17  package org.springframework.transaction.annotation;
18  
19  import org.springframework.transaction.TransactionDefinition;
20  
21  /**
22   * Enumeration that represents transaction propagation behaviors for use
23   * with the {@link Transactional} annotation, corresponding to the
24   * {@link TransactionDefinition} interface.
25   *
26   * @author Colin Sampaleanu
27   * @author Juergen Hoeller
28   * @since 1.2
29   */
30  public enum Propagation {
31  
32  	/**
33  	 * Support a current transaction, create a new one if none exists.
34  	 * Analogous to EJB transaction attribute of the same name.
35  	 * <p>This is the default setting of a transaction annotation.
36  	 */
37  	REQUIRED(TransactionDefinition.PROPAGATION_REQUIRED),
38  
39  	/**
40  	 * Support a current transaction, execute non-transactionally if none exists.
41  	 * Analogous to EJB transaction attribute of the same name.
42  	 * <p>Note: For transaction managers with transaction synchronization,
43  	 * PROPAGATION_SUPPORTS is slightly different from no transaction at all,
44  	 * as it defines a transaction scope that synchronization will apply for.
45  	 * As a consequence, the same resources (JDBC Connection, Hibernate Session, etc)
46  	 * will be shared for the entire specified scope. Note that this depends on
47  	 * the actual synchronization configuration of the transaction manager.
48  	 * @see org.springframework.transaction.support.AbstractPlatformTransactionManager#setTransactionSynchronization
49  	 */
50  	SUPPORTS(TransactionDefinition.PROPAGATION_SUPPORTS),
51  
52  	/**
53  	 * Support a current transaction, throw an exception if none exists.
54  	 * Analogous to EJB transaction attribute of the same name.
55  	 */
56  	MANDATORY(TransactionDefinition.PROPAGATION_MANDATORY),
57  
58  	/**
59  	 * Create a new transaction, and suspend the current transaction if one exists.
60  	 * Analogous to the EJB transaction attribute of the same name.
61  	 * <p>Note: Actual transaction suspension will not work out-of-the-box on
62  	 * all transaction managers. This in particular applies to JtaTransactionManager,
63  	 * which requires the {@code javax.transaction.TransactionManager} to be
64  	 * made available it to it (which is server-specific in standard J2EE).
65  	 * @see org.springframework.transaction.jta.JtaTransactionManager#setTransactionManager
66  	 */
67  	REQUIRES_NEW(TransactionDefinition.PROPAGATION_REQUIRES_NEW),
68  
69  	/**
70  	 * Execute non-transactionally, suspend the current transaction if one exists.
71  	 * Analogous to EJB transaction attribute of the same name.
72  	 * <p>Note: Actual transaction suspension will not work on out-of-the-box
73  	 * on all transaction managers. This in particular applies to JtaTransactionManager,
74  	 * which requires the {@code javax.transaction.TransactionManager} to be
75  	 * made available it to it (which is server-specific in standard J2EE).
76  	 * @see org.springframework.transaction.jta.JtaTransactionManager#setTransactionManager
77  	 */
78  	NOT_SUPPORTED(TransactionDefinition.PROPAGATION_NOT_SUPPORTED),
79  
80  	/**
81  	 * Execute non-transactionally, throw an exception if a transaction exists.
82  	 * Analogous to EJB transaction attribute of the same name.
83  	 */
84  	NEVER(TransactionDefinition.PROPAGATION_NEVER),
85  
86  	/**
87  	 * Execute within a nested transaction if a current transaction exists,
88  	 * behave like PROPAGATION_REQUIRED else. There is no analogous feature in EJB.
89  	 * <p>Note: Actual creation of a nested transaction will only work on specific
90  	 * transaction managers. Out of the box, this only applies to the JDBC
91  	 * DataSourceTransactionManager when working on a JDBC 3.0 driver.
92  	 * Some JTA providers might support nested transactions as well.
93  	 * @see org.springframework.jdbc.datasource.DataSourceTransactionManager
94  	 */
95  	NESTED(TransactionDefinition.PROPAGATION_NESTED);
96  
97  
98  	private final int value;
99  
100 
101 	Propagation(int value) { this.value = value; }
102 
103 	public int value() { return this.value; }
104 
105 }
