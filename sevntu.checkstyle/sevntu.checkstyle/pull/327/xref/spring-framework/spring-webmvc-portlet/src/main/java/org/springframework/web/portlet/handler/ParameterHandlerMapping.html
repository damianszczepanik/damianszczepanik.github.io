17  package org.springframework.web.portlet.handler;
18  
19  import java.util.Map;
20  import javax.portlet.PortletRequest;
21  
22  import org.springframework.beans.BeansException;
23  import org.springframework.util.Assert;
24  
25  /**
26   * Implementation of the {@link org.springframework.web.portlet.HandlerMapping}
27   * to map from a request parameter to request handler beans.
28   *
29   * <p>The default name of the parameter is "action", but can be changed using
30   * {@link #setParameterName setParameterName()}.
31   *
32   * <p>The bean configuration for this mapping will look somthing like this:
33   *
34   * <pre class="code">
35   * &lt;bean id="parameterHandlerMapping" class="org.springframework.web.portlet.handler.ParameterHandlerMapping"&gt;
36   *   &lt;property name="parameterMap"&gt;
37   *     &lt;map&gt;
38   * 	     &lt;entry key="add"&gt;&lt;ref bean="addItemHandler"/&gt;&lt;/entry&gt;
39   *       &lt;entry key="edit"&gt;&lt;ref bean="editItemHandler"/&gt;&lt;/entry&gt;
40   *       &lt;entry key="delete"&gt;&lt;ref bean="deleteItemHandler"/&gt;&lt;/entry&gt;
41   *     &lt;/map&gt;
42   *   &lt;/property&gt;
43   * &lt;/bean&gt;</pre>
44   *
45   * Thanks to Rainer Schmitz for suggesting this mapping strategy!
46   *
47   * @author John A. Lewis
48   * @author Juergen Hoeller
49   * @since 2.0
50   * @see ParameterMappingInterceptor
51   */
52  public class ParameterHandlerMapping extends AbstractMapBasedHandlerMapping<String> {
53  
54  	/**
55  	 * Default request parameter name to use for mapping to handlers: "action".
56  	 */
57  	public final static String DEFAULT_PARAMETER_NAME = "action";
58  
59  
60  	private String parameterName = DEFAULT_PARAMETER_NAME;
61  
62  	private Map<String, ?> parameterMap;
63  
64  
65  	/**
66  	 * Set the name of the parameter used for mapping to handlers.
67  	 * <p>Default is "action".
68  	 */
69  	public void setParameterName(String parameterName) {
70  		Assert.hasText(parameterName, "'parameterName' must not be empty");
71  		this.parameterName = parameterName;
72  	}
73  
74  	/**
75  	 * Set a Map with parameters as keys and handler beans or bean names as values.
76  	 * Convenient for population with bean references.
77  	 * @param parameterMap map with parameters as keys and beans as values
78  	 */
79  	public void setParameterMap(Map<String, ?> parameterMap) {
80  		this.parameterMap = parameterMap;
81  	}
82  
83  
84  	/**
85  	 * Calls the {@code registerHandlers} method in addition
86  	 * to the superclass's initialization.
87  	 * @see #registerHandlers
88  	 */
89  	@Override
90  	public void initApplicationContext() throws BeansException {
91  		super.initApplicationContext();
92  		registerHandlers(this.parameterMap);
93  	}
94  
95  	/**
96  	 * Uses the value of the specified parameter as lookup key.
97  	 * @see #setParameterName
98  	 */
99  	@Override
100 	protected String getLookupKey(PortletRequest request) throws Exception {
101 		return request.getParameter(this.parameterName);
102 	}
103 
104 }
