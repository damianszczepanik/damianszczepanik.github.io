500 	public static class MockMultipartResolver implements PortletMultipartResolver {
501 
502 		@Override
503 		public boolean isMultipart(ActionRequest request) {
504 			return true;
505 		}
506 
507 		@Override
508 		public MultipartActionRequest resolveMultipart(ActionRequest request) throws MultipartException {
509 			if (request.getAttribute("fail") != null) {
510 				throw new MaxUploadSizeExceededException(1000);
511 			}
512 			if (request instanceof MultipartActionRequest) {
513 				throw new IllegalStateException("Already a multipart request");
514 			}
515 			if (request.getAttribute("resolved") != null) {
516 				throw new IllegalStateException("Already resolved");
517 			}
518 			request.setAttribute("resolved", Boolean.TRUE);
519 			MultiValueMap<String, MultipartFile> files = new LinkedMultiValueMap<String, MultipartFile>();
520 			files.set("someFile", new MockMultipartFile("someFile", "someContent".getBytes()));
521 			Map<String, String[]> params = new HashMap<String, String[]>();
522 			params.put("someParam", new String[] {"someParam"});
523 			return new DefaultMultipartActionRequest(request, files, params, Collections.<String, String>emptyMap());
524 		}
525 
526 		@Override
527 		public void cleanupMultipart(MultipartActionRequest request) {
528 			if (request.getAttribute("cleanedUp") != null) {
529 				throw new IllegalStateException("Already cleaned up");
530 			}
531 			request.setAttribute("cleanedUp", Boolean.TRUE);
532 		}
