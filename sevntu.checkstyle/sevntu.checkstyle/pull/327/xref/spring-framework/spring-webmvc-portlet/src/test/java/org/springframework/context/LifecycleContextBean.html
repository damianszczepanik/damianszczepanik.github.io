4   import org.springframework.beans.BeansException;
5   import org.springframework.beans.factory.BeanFactory;
6   import org.springframework.tests.sample.beans.LifecycleBean;
7   
8   /**
9    * Simple bean to test ApplicationContext lifecycle methods for beans
10   *
11   * @author Colin Sampaleanu
12   * @since 03.07.2004
13   */
14  public class LifecycleContextBean extends LifecycleBean implements ApplicationContextAware {
15  
16  	protected ApplicationContext owningContext;
17  
18  	@Override
19  	public void setBeanFactory(BeanFactory beanFactory) {
20  		super.setBeanFactory(beanFactory);
21  		if (this.owningContext != null)
22  			throw new RuntimeException("Factory called setBeanFactory after setApplicationContext");
23  	}
24  
25  	@Override
26  	public void afterPropertiesSet() {
27  		super.afterPropertiesSet();
28  		if (this.owningContext == null)
29  			throw new RuntimeException("Factory didn't call setAppliationContext before afterPropertiesSet on lifecycle bean");
30  	}
31  
32  	@Override
33  	public void setApplicationContext(ApplicationContext applicationContext) throws BeansException {
34  		if (this.owningFactory == null)
35  			throw new RuntimeException("Factory called setApplicationContext before setBeanFactory");
36  
37  		this.owningContext = applicationContext;
38  	}
39  
40  }
