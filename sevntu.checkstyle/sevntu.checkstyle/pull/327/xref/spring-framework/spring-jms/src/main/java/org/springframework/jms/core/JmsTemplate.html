45  /**
46   * Helper class that simplifies synchronous JMS access code.
47   *
48   * <p>If you want to use dynamic destination creation, you must specify
49   * the type of JMS destination to create, using the "pubSubDomain" property.
50   * For other operations, this is not necessary. Point-to-Point (Queues) is the default
51   * domain.
52   *
53   * <p>Default settings for JMS Sessions are "not transacted" and "auto-acknowledge".
54   * As defined by the J2EE specification, the transaction and acknowledgement
55   * parameters are ignored when a JMS Session is created inside an active
56   * transaction, no matter if a JTA transaction or a Spring-managed transaction.
57   * To configure them for native JMS usage, specify appropriate values for
58   * the "sessionTransacted" and "sessionAcknowledgeMode" bean properties.
59   *
60   * <p>This template uses a
61   * {@link org.springframework.jms.support.destination.DynamicDestinationResolver}
62   * and a {@link org.springframework.jms.support.converter.SimpleMessageConverter}
63   * as default strategies for resolving a destination name or converting a message,
64   * respectively. These defaults can be overridden through the "destinationResolver"
65   * and "messageConverter" bean properties.
66   *
67   * <p><b>NOTE: The {@code ConnectionFactory} used with this template should
68   * return pooled Connections (or a single shared Connection) as well as pooled
69   * Sessions and MessageProducers. Otherwise, performance of ad-hoc JMS operations
70   * is going to suffer.</b> The simplest option is to use the Spring-provided
71   * {@link org.springframework.jms.connection.SingleConnectionFactory} as a
72   * decorator for your target {@code ConnectionFactory}, reusing a single
73   * JMS Connection in a thread-safe fashion; this is often good enough for the
74   * purpose of sending messages via this template. In a J2EE environment,
75   * make sure that the {@code ConnectionFactory} is obtained from the
76   * application's environment naming context via JNDI; application servers
77   * typically expose pooled, transaction-aware factories there.
78   *
79   * @author Mark Pollack
80   * @author Juergen Hoeller
81   * @author Stephane Nicoll
82   * @since 1.1
83   * @see #setConnectionFactory
84   * @see #setPubSubDomain
85   * @see #setDestinationResolver
86   * @see #setMessageConverter
87   * @see javax.jms.MessageProducer
88   * @see javax.jms.MessageConsumer
89   */

461 	/**
462 	 * Execute the action specified by the given action object within a
463 	 * JMS Session. Generalized version of {@code execute(SessionCallback)},
464 	 * allowing the JMS Connection to be started on the fly.
465 	 * <p>Use {@code execute(SessionCallback)} for the general case.
466 	 * Starting the JMS Connection is just necessary for receiving messages,
467 	 * which is preferably achieved through the {@code receive} methods.
468 	 * @param action callback object that exposes the Session
469 	 * @param startConnection whether to start the Connection
470 	 * @return the result object from working with the Session
471 	 * @throws JmsException if there is any problem
472 	 * @see #execute(SessionCallback)
473 	 * @see #receive
474 	 */
475 	public <T> T execute(SessionCallback<T> action, boolean startConnection) throws JmsException {
476 		Assert.notNull(action, "Callback object must not be null");
477 		Connection conToClose = null;
478 		Session sessionToClose = null;
479 		try {
480 			Session sessionToUse = ConnectionFactoryUtils.doGetTransactionalSession(
481 					getConnectionFactory(), this.transactionalResourceFactory, startConnection);
482 			if (sessionToUse == null) {
483 				conToClose = createConnection();
484 				sessionToClose = createSession(conToClose);
485 				if (startConnection) {
486 					conToClose.start();
487 				}
488 				sessionToUse = sessionToClose;
489 			}
490 			if (logger.isDebugEnabled()) {
491 				logger.debug("Executing callback on JMS Session: " + sessionToUse);
492 			}
493 			return action.doInJms(sessionToUse);
494 		}
495 		catch (JMSException ex) {
496 			throw convertJmsAccessException(ex);
497 		}
498 		finally {
499 			JmsUtils.closeSession(sessionToClose);
500 			ConnectionFactoryUtils.releaseConnection(conToClose, getConnectionFactory(), startConnection);
501 		}
502 	}

518 		return execute(new SessionCallback<T>() {
519 			@Override

521 				MessageProducer producer = createProducer(session, destination);

589 	/**
590 	 * Send the given JMS message.
591 	 * @param session the JMS Session to operate on
592 	 * @param destination the JMS Destination to send to
593 	 * @param messageCreator callback to create a JMS Message
594 	 * @throws JMSException if thrown by JMS API methods
595 	 */
596 	protected void doSend(Session session, Destination destination, MessageCreator messageCreator)
597 			throws JMSException {
598 
599 		Assert.notNull(messageCreator, "MessageCreator must not be null");
600 		MessageProducer producer = createProducer(session, destination);
601 		try {
602 			Message message = messageCreator.createMessage(session);
603 			if (logger.isDebugEnabled()) {
604 				logger.debug("Sending created message: " + message);
605 			}
606 			doSend(producer, message);
607 			// Check commit - avoid commit call within a JTA transaction.
608 			if (session.getTransacted() && isSessionLocallyTransacted(session)) {
609 				// Transacted session created by this template -> commit.
610 				JmsUtils.commitIfNecessary(session);
611 			}
612 		}
613 		finally {
614 			JmsUtils.closeMessageProducer(producer);
615 		}
616 	}
617 
618 	/**
619 	 * Actually send the given JMS message.
620 	 * @param producer the JMS MessageProducer to send with
621 	 * @param message the JMS Message to send
622 	 * @throws JMSException if thrown by JMS API methods
623 	 */
624 	protected void doSend(MessageProducer producer, Message message) throws JMSException {
625 		if (this.deliveryDelay >= 0) {
626 			if (setDeliveryDelayMethod == null) {
627 				throw new IllegalStateException("setDeliveryDelay requires JMS 2.0");
628 			}
629 			ReflectionUtils.invokeMethod(setDeliveryDelayMethod, producer, this.deliveryDelay);
630 		}
631 		if (isExplicitQosEnabled()) {
632 			producer.send(message, getDeliveryMode(), getPriority(), getTimeToLive());
633 		}
634 		else {
635 			producer.send(message);
636 		}
637 	}

786 	/**
787 	 * Actually receive a JMS message.
788 	 * @param session the JMS Session to operate on
789 	 * @param consumer the JMS MessageConsumer to receive with
790 	 * @return the JMS Message received, or {@code null} if none
791 	 * @throws JMSException if thrown by JMS API methods
792 	 */
793 	protected Message doReceive(Session session, MessageConsumer consumer) throws JMSException {
794 		try {
795 			// Use transaction timeout (if available).
796 			long timeout = getReceiveTimeout();
797 			JmsResourceHolder resourceHolder =
798 					(JmsResourceHolder) TransactionSynchronizationManager.getResource(getConnectionFactory());
799 			if (resourceHolder != null && resourceHolder.hasTimeout()) {
800 				timeout = Math.min(timeout, resourceHolder.getTimeToLiveInMillis());
801 			}
802 			Message message = doReceive(consumer, timeout);
803 			if (session.getTransacted()) {
804 				// Commit necessary - but avoid commit call within a JTA transaction.
805 				if (isSessionLocallyTransacted(session)) {
806 					// Transacted session created by this template -> commit.
807 					JmsUtils.commitIfNecessary(session);
808 				}
809 			}
810 			else if (isClientAcknowledge(session)) {
811 				// Manually acknowledge message, if any.
812 				if (message != null) {
813 					message.acknowledge();
814 				}
815 			}
816 			return message;
817 		}
818 		finally {
819 			JmsUtils.closeMessageConsumer(consumer);
820 		}
821 	}

823 	/**
824 	 * Actually receive a message from the given consumer.
825 	 * @param consumer the JMS MessageConsumer to receive with
826 	 * @param timeout the receive timeout
827 	 * @return the JMS Message received, or {@code null} if none
828 	 * @throws JMSException if thrown by JMS API methods
829 	 */
830 	private Message doReceive(MessageConsumer consumer, long timeout) throws JMSException {
831 		if (timeout == RECEIVE_TIMEOUT_NO_WAIT) {
832 			return consumer.receiveNoWait();
833 		}
834 		else if (timeout > 0) {
835 			return consumer.receive(timeout);
836 		}
837 		else {
838 			return consumer.receive();
839 		}
840 	}

931 	/**
932 	 * Send a request message to the given {@link Destination} and block until
933 	 * a reply has been received on a temporary queue created on-the-fly.
934 	 * <p>Return the response message or {@code null} if no message has
935 	 * @throws JMSException if thrown by JMS API methods
936 	 */
937 	protected Message doSendAndReceive(Session session, Destination destination, MessageCreator messageCreator)
938 			throws JMSException {
939 
940 		Assert.notNull(messageCreator, "MessageCreator must not be null");
941 		TemporaryQueue responseQueue = null;
942 		MessageProducer producer = null;
943 		MessageConsumer consumer = null;
944 		try {
945 			Message requestMessage = messageCreator.createMessage(session);
946 			responseQueue = session.createTemporaryQueue();
947 			producer = session.createProducer(destination);
948 			consumer = session.createConsumer(responseQueue);
949 			requestMessage.setJMSReplyTo(responseQueue);
950 			if (logger.isDebugEnabled()) {
951 				logger.debug("Sending created message: " + requestMessage);
952 			}
953 			doSend(producer, requestMessage);
954 			return doReceive(consumer, getReceiveTimeout());
955 		}
956 		finally {
957 			JmsUtils.closeMessageConsumer(consumer);
958 			JmsUtils.closeMessageProducer(producer);
959 			if (responseQueue != null) {
960 				responseQueue.delete();
961 			}
962 		}
963 	}

965 	/**
966 	 * A variant of {@link #execute(SessionCallback, boolean)} that explicitly
967 	 * creates a non-transactional {@link Session}. The given {@link SessionCallback}
968 	 * does not participate in an existing transaction.
969 	 */
970 	private <T> T executeLocal(SessionCallback<T> action, boolean startConnection) throws JmsException {
971 		Assert.notNull(action, "Callback object must not be null");
972 		Connection con = null;
973 		Session session = null;
974 		try {
975 			con = getConnectionFactory().createConnection();
976 			session = con.createSession(false, Session.AUTO_ACKNOWLEDGE);
977 			if (startConnection) {
978 				con.start();
979 			}
980 			if (logger.isDebugEnabled()) {
981 				logger.debug("Executing callback on JMS Session: " + session);
982 			}
983 			return action.doInJms(session);
984 		}
985 		catch (JMSException ex) {
986 			throw convertJmsAccessException(ex);
987 		}
988 		finally {
989 			JmsUtils.closeSession(session);
990 			ConnectionFactoryUtils.releaseConnection(con, getConnectionFactory(), startConnection);
991 		}
992 	}
