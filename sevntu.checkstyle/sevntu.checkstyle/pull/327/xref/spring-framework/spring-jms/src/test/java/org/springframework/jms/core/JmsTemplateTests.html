218 	@Test
219 	public void testSessionCallbackWithinSynchronizedTransaction() throws Exception {
220 		SingleConnectionFactory scf = new SingleConnectionFactory(connectionFactory);
221 		JmsTemplate template = createTemplate();
222 		template.setConnectionFactory(scf);
223 
224 		TransactionSynchronizationManager.initSynchronization();
225 		try {
226 			template.execute(new SessionCallback<Void>() {
227 				@Override
228 				public Void doInJms(Session session) throws JMSException {
229 					session.getTransacted();
230 					return null;
231 				}
232 			});
233 			template.execute(new SessionCallback<Void>() {
234 				@Override
235 				public Void doInJms(Session session) throws JMSException {
236 					session.getTransacted();
237 					return null;
238 				}
239 			});
240 
241 			assertSame(session, ConnectionFactoryUtils.getTransactionalSession(scf, null, false));
242 			assertSame(session, ConnectionFactoryUtils.getTransactionalSession(scf, scf.createConnection(), false));
243 
244 			TransactionAwareConnectionFactoryProxy tacf = new TransactionAwareConnectionFactoryProxy(scf);
245 			Connection tac = tacf.createConnection();
246 			Session tas = tac.createSession(false, Session.AUTO_ACKNOWLEDGE);
247 			tas.getTransacted();
248 			tas.close();
249 			tac.close();
250 
251 			List<TransactionSynchronization> synchs = TransactionSynchronizationManager.getSynchronizations();
252 			assertEquals(1, synchs.size());
253 			TransactionSynchronization synch = synchs.get(0);
254 			synch.beforeCommit(false);
255 			synch.beforeCompletion();
256 			synch.afterCommit();
257 			synch.afterCompletion(TransactionSynchronization.STATUS_UNKNOWN);
258 		}
259 		finally {
260 			TransactionSynchronizationManager.clearSynchronization();
261 			scf.destroy();
262 		}
263 		assertTrue(TransactionSynchronizationManager.getResourceMap().isEmpty());
264 
265 		verify(connection).start();
266 		if (useTransactedTemplate()) {
267 			verify(session).commit();
268 		}
269 		verify(session).close();
270 		verify(connection).stop();
271 		verify(connection).close();
272 	}

343 	 * Common method for testing a send method that uses the MessageCreator
344 	 * callback but with different QOS options.
345 	 * @param ignoreQOS test using default QOS options.
346 	 */
347 	private void doTestSendDestination(
348 			boolean explicitDestination, boolean useDefaultDestination,
349 			boolean ignoreQOS, boolean disableIdAndTimestamp) throws Exception {
350 
351 		JmsTemplate template = createTemplate();
352 		template.setConnectionFactory(connectionFactory);
353 
354 		String destinationName = "testDestination";
355 
356 		if (useDefaultDestination) {
357 			if (explicitDestination) {
358 				template.setDefaultDestination(queue);
359 			}
360 			else {
361 				template.setDefaultDestinationName(destinationName);
362 			}
363 		}
364 		if (disableIdAndTimestamp) {
365 			template.setMessageIdEnabled(false);
366 			template.setMessageTimestampEnabled(false);
367 		}
368 
369 		MessageProducer messageProducer = mock(MessageProducer.class);
370 		TextMessage textMessage = mock(TextMessage.class);
371 
372 		given(session.createProducer(queue)).willReturn(messageProducer);
373 		given(session.createTextMessage("just testing")).willReturn(textMessage);
374 
375 		if (!ignoreQOS) {
376 			template.setExplicitQosEnabled(true);
377 			template.setDeliveryMode(deliveryMode);
378 			template.setPriority(priority);
379 			template.setTimeToLive(timeToLive);
380 		}
381 
382 		if (useDefaultDestination) {
383 			template.send(new MessageCreator() {
384 				@Override
385 				public Message createMessage(Session session) throws JMSException {
386 					return session.createTextMessage("just testing");
387 				}
388 			});
389 		}
390 		else {
391 			if (explicitDestination) {
392 				template.send(queue, new MessageCreator() {
393 					@Override
394 					public Message createMessage(Session session) throws JMSException {
395 						return session.createTextMessage("just testing");
396 					}
397 				});
398 			}
399 			else {
400 				template.send(destinationName, new MessageCreator() {
401 					@Override
402 					public Message createMessage(Session session) throws JMSException {
403 						return session.createTextMessage("just testing");
404 					}
405 				});
406 			}
407 		}
408 
409 		if (useTransactedTemplate()) {
410 			verify(session).commit();
411 		}
412 
413 		if (disableIdAndTimestamp) {
414 			verify(messageProducer).setDisableMessageID(true);
415 			verify(messageProducer).setDisableMessageTimestamp(true);
416 		}
417 
418 		if (ignoreQOS) {
419 			verify(messageProducer).send(textMessage);
420 		}
421 		else {
422 			verify(messageProducer).send(textMessage, deliveryMode, priority, timeToLive);
423 		}
424 		verify(messageProducer).close();
425 		verify(session).close();
426 		verify(connection).close();
427 	}

538 	private void doTestReceive(
539 			boolean explicitDestination, boolean useDefaultDestination, boolean testConverter,
540 			boolean clientAcknowledge, boolean messageSelector, boolean noLocal, long timeout)
541 			throws Exception {
542 
543 		JmsTemplate template = createTemplate();
544 		template.setConnectionFactory(connectionFactory);
545 
546 		String destinationName = "testDestination";
547 
548 		if (useDefaultDestination) {
549 			if (explicitDestination) {
550 				template.setDefaultDestination(queue);
551 			}
552 			else {
553 				template.setDefaultDestinationName(destinationName);
554 			}
555 		}
556 		if (noLocal) {
557 			template.setPubSubNoLocal(true);
558 		}
559 		template.setReceiveTimeout(timeout);
560 
561 		MessageConsumer messageConsumer = mock(MessageConsumer.class);
562 
563 		String selectorString = "selector";
564 		given(session.createConsumer(queue,
565 				messageSelector ? selectorString : null)).willReturn(messageConsumer);
566 
567 		if (!useTransactedTemplate() && !useTransactedSession()) {
568 			given(session.getAcknowledgeMode()).willReturn(
569 					clientAcknowledge ? Session.CLIENT_ACKNOWLEDGE
570 							: Session.AUTO_ACKNOWLEDGE);
571 		}
572 
573 		TextMessage textMessage = mock(TextMessage.class);
574 
575 		if (testConverter) {
576 			given(textMessage.getText()).willReturn("Hello World!");
577 		}
578 
579 		if (timeout == JmsTemplate.RECEIVE_TIMEOUT_NO_WAIT) {
580 			given(messageConsumer.receiveNoWait()).willReturn(textMessage);
581 		}
582 		else if (timeout == JmsTemplate.RECEIVE_TIMEOUT_INDEFINITE_WAIT) {
583 			given(messageConsumer.receive()).willReturn(textMessage);
584 		}
585 		else {
586 			given(messageConsumer.receive(timeout)).willReturn(textMessage);
587 		}
588 
589 		Message message = null;
590 		String textFromMessage = null;
591 
592 		if (useDefaultDestination) {
593 			if (testConverter) {
594 				textFromMessage = (String)
595 						(messageSelector ? template.receiveSelectedAndConvert(selectorString) :
596 						template.receiveAndConvert());
597 			}
598 			else {
599 				message = (messageSelector ? template.receiveSelected(selectorString) : template.receive());
600 			}
601 		}
602 		else if (explicitDestination) {
603 			if (testConverter) {
604 				textFromMessage = (String)
605 						(messageSelector ? template.receiveSelectedAndConvert(queue, selectorString) :
606 						template.receiveAndConvert(queue));
607 			}
608 			else {
609 				message = (messageSelector ? template.receiveSelected(queue, selectorString) :
610 						template.receive(queue));
611 			}
612 		}
613 		else {
614 			if (testConverter) {
615 				textFromMessage = (String)
616 						(messageSelector ? template.receiveSelectedAndConvert(destinationName, selectorString) :
617 						template.receiveAndConvert(destinationName));
618 			}
619 			else {
620 				message = (messageSelector ? template.receiveSelected(destinationName, selectorString) :
621 						template.receive(destinationName));
622 			}
623 		}
624 
625 		if (testConverter) {
626 			assertEquals("Message text should be equal", "Hello World!", textFromMessage);
627 		}
628 		else {
629 			assertEquals("Messages should refer to the same object", message, textMessage);
630 		}
631 
632 		verify(connection).start();
633 		verify(connection).close();
634 		if (useTransactedTemplate()) {
635 			verify(session).commit();
636 		}
637 		verify(session).close();
638 		if (!useTransactedSession() && clientAcknowledge) {
639 			verify(textMessage).acknowledge();
640 		}
641 		verify(messageConsumer).close();
642 	}

682 		TemporaryQueue replyDestination = mock(TemporaryQueue.class);
