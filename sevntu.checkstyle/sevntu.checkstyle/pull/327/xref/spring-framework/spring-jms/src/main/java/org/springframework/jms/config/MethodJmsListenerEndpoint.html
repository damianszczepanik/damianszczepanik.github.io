17  package org.springframework.jms.config;
18  
19  import java.lang.reflect.Method;
20  import java.util.Arrays;
21  
22  import org.springframework.aop.framework.AopProxyUtils;
23  import org.springframework.aop.support.AopUtils;
24  import org.springframework.core.annotation.AnnotationUtils;
25  import org.springframework.jms.listener.MessageListenerContainer;
26  import org.springframework.jms.listener.adapter.MessagingMessageListenerAdapter;
27  import org.springframework.jms.support.converter.MessageConverter;
28  import org.springframework.messaging.handler.annotation.SendTo;
29  import org.springframework.messaging.handler.annotation.support.MessageHandlerMethodFactory;
30  import org.springframework.messaging.handler.invocation.InvocableHandlerMethod;
31  import org.springframework.util.Assert;
32  import org.springframework.util.StringUtils;
33  
34  /**
35   * A {@link JmsListenerEndpoint} providing the method to invoke to process
36   * an incoming message for this endpoint.
37   *
38   * @author Stephane Nicoll
39   * @since 4.1
40   */
41  public class MethodJmsListenerEndpoint extends AbstractJmsListenerEndpoint {
42  
43  	private Object bean;
44  
45  	private Method method;
46  
47  	private MessageHandlerMethodFactory messageHandlerMethodFactory;
48  
49  
50  	/**
51  	 * Set the object instance that should manage this endpoint.
52  	 */
53  	public void setBean(Object bean) {
54  		this.bean = bean;
55  	}
56  
57  	public Object getBean() {
58  		return this.bean;
59  	}
60  
61  	/**
62  	 * Set the method to invoke to process a message managed by this endpoint.
63  	 */
64  	public void setMethod(Method method) {
65  		this.method = method;
66  	}
67  
68  	public Method getMethod() {
69  		return this.method;
70  	}
71  
72  	/**
73  	 * Set the {@link MessageHandlerMethodFactory} to use to build the
74  	 * {@link InvocableHandlerMethod} responsible to manage the invocation
75  	 * of this endpoint.
76  	 */
77  	public void setMessageHandlerMethodFactory(MessageHandlerMethodFactory messageHandlerMethodFactory) {
78  		this.messageHandlerMethodFactory = messageHandlerMethodFactory;
79  	}
80  
81  
82  	@Override
83  	protected MessagingMessageListenerAdapter createMessageListener(MessageListenerContainer container) {
84  		Assert.state(this.messageHandlerMethodFactory != null,
85  				"Could not create message listener - MessageHandlerMethodFactory not set");
86  		MessagingMessageListenerAdapter messageListener = createMessageListenerInstance();
87  		InvocableHandlerMethod invocableHandlerMethod =
88  				this.messageHandlerMethodFactory.createInvocableHandlerMethod(getBean(), getMethod());
89  		messageListener.setHandlerMethod(invocableHandlerMethod);
90  		String responseDestination = getDefaultResponseDestination();
91  		if (StringUtils.hasText(responseDestination)) {
92  			if (container.isPubSubDomain()) {
93  				messageListener.setDefaultResponseTopicName(responseDestination);
94  			}
95  			else {
96  				messageListener.setDefaultResponseQueueName(responseDestination);
97  			}
98  		}
99  		MessageConverter messageConverter = container.getMessageConverter();
100 		if (messageConverter != null) {
101 			messageListener.setMessageConverter(messageConverter);
102 		}
103 		return messageListener;
104 	}
105 
106 	/**
107 	 * Create an empty {@link MessagingMessageListenerAdapter} instance.
108 	 */
109 	protected MessagingMessageListenerAdapter createMessageListenerInstance() {
110 		return new MessagingMessageListenerAdapter();
111 	}
112 
113 	private String getDefaultResponseDestination() {
114 		Method specificMethod = getMostSpecificMethod();
115 		SendTo ann = AnnotationUtils.getAnnotation(specificMethod, SendTo.class);
116 		if (ann != null) {
117 			Object[] destinations = ann.value();
118 			if (destinations.length != 1) {
119 				throw new IllegalStateException("Invalid @" + SendTo.class.getSimpleName() + " annotation on '"
120 						+ specificMethod + "' one destination must be set (got " + Arrays.toString(destinations) + ")");
121 			}
122 			return (String) destinations[0];
123 		}
124 		return null;
125 	}
126 
127 	private Method getMostSpecificMethod() {
128 		if (AopUtils.isAopProxy(this.bean)) {
129 			Class<?> target = AopProxyUtils.ultimateTargetClass(this.bean);
130 			return AopUtils.getMostSpecificMethod(getMethod(), target);
131 		}
132 		else {
133 			return getMethod();
134 		}
135 	}
136 
137 	@Override
138 	protected StringBuilder getEndpointDescription() {
139 		return super.getEndpointDescription()
140 				.append(" | bean='").append(this.bean).append("'")
141 				.append(" | method='").append(this.method).append("'");
142 	}
143 
144 }
