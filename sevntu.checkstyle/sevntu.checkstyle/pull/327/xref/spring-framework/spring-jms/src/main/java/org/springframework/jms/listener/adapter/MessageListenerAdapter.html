27  import org.springframework.jms.support.converter.MessageConverter;

29  import org.springframework.util.Assert;
30  import org.springframework.util.MethodInvoker;

33  /**
34   * Message listener adapter that delegates the handling of messages to target
35   * listener methods via reflection, with flexible message type conversion.
36   * Allows listener methods to operate on message content types, completely
37   * independent from the JMS API.
38   *
39   * <p>By default, the content of incoming JMS messages gets extracted before
40   * being passed into the target listener method, to let the target method
41   * operate on message content types such as String or byte array instead of
42   * the raw {@link Message}. Message type conversion is delegated to a Spring
43   * JMS {@link MessageConverter}. By default, a {@link SimpleMessageConverter}
44   * will be used. (If you do not want such automatic message conversion taking
45   * place, then be sure to set the {@link #setMessageConverter MessageConverter}
46   * to {@code null}.)
47   *
48   * <p>If a target listener method returns a non-null object (typically of a
49   * message content type such as {@code String} or byte array), it will get
50   * wrapped in a JMS {@code Message} and sent to the response destination
51   * (either the JMS "reply-to" destination or a
52   * {@link #setDefaultResponseDestination(javax.jms.Destination) specified default
53   * destination}).
54   *
55   * <p><b>Note:</b> The sending of response messages is only available when
56   * using the {@link SessionAwareMessageListener} entry point (typically through a
57   * Spring message listener container). Usage as standard JMS {@link MessageListener}
58   * does <i>not</i> support the generation of response messages.
59   *
60   * <p>Find below some examples of method signatures compliant with this
61   * adapter class. This first example handles all {@code Message} types
62   * and gets passed the contents of each {@code Message} type as an
63   * argument. No {@code Message} will be sent back as all of these
64   * methods return {@code void}.
65   *
66   * <pre class="code">public interface MessageContentsDelegate {
67   *    void handleMessage(String text);
68   *    void handleMessage(Map map);
69   *    void handleMessage(byte[] bytes);
70   *    void handleMessage(Serializable obj);
71   * }</pre>
72   *
73   * This next example handles all {@code Message} types and gets
74   * passed the actual (raw) {@code Message} as an argument. Again, no
75   * {@code Message} will be sent back as all of these methods return
76   * {@code void}.
77   *
78   * <pre class="code">public interface RawMessageDelegate {
79   *    void handleMessage(TextMessage message);
80   *    void handleMessage(MapMessage message);
81   *    void handleMessage(BytesMessage message);
82   *    void handleMessage(ObjectMessage message);
83   * }</pre>
84   *
85   * This next example illustrates a {@code Message} delegate
86   * that just consumes the {@code String} contents of
87   * {@link javax.jms.TextMessage TextMessages}. Notice also how the
88   * name of the {@code Message} handling method is different from the
89   * {@link #ORIGINAL_DEFAULT_LISTENER_METHOD original} (this will have to
90   * be configured in the attandant bean definition). Again, no {@code Message}
91   * will be sent back as the method returns {@code void}.
92   *
93   * <pre class="code">public interface TextMessageContentDelegate {
94   *    void onMessage(String text);
95   * }</pre>
96   *
97   * This final example illustrates a {@code Message} delegate
98   * that just consumes the {@code String} contents of
99   * {@link javax.jms.TextMessage TextMessages}. Notice how the return type
100  * of this method is {@code String}: This will result in the configured
101  * {@link MessageListenerAdapter} sending a {@link javax.jms.TextMessage} in response.
102  *
103  * <pre class="code">public interface ResponsiveTextMessageContentDelegate {
104  *    String handleMessage(String text);
105  * }</pre>
106  *
107  * For further examples and discussion please do refer to the Spring
108  * reference documentation which describes this class (and it's attendant
109  * XML configuration) in detail.
110  *
111  * @author Juergen Hoeller
112  * @since 2.0
113  * @see #setDelegate
114  * @see #setDefaultListenerMethod
115  * @see #setDefaultResponseDestination
116  * @see #setMessageConverter
117  * @see org.springframework.jms.support.converter.SimpleMessageConverter
118  * @see org.springframework.jms.listener.SessionAwareMessageListener
119  * @see org.springframework.jms.listener.AbstractMessageListenerContainer#setMessageListener
120  */
121 public class MessageListenerAdapter extends AbstractAdaptableMessageListener
122 		implements SubscriptionNameProvider {
123 
124 	/**
125 	 * Out-of-the-box value for the default listener method: "handleMessage".
126 	 */
127 	public static final String ORIGINAL_DEFAULT_LISTENER_METHOD = "handleMessage";
128 
129 
130 	private Object delegate;
131 
132 	private String defaultListenerMethod = ORIGINAL_DEFAULT_LISTENER_METHOD;
133 
134 
135 	/**
136 	 * Create a new {@link MessageListenerAdapter} with default settings.
137 	 */
138 	public MessageListenerAdapter() {
139 		this.delegate = this;
140 	}
141 
142 	/**
143 	 * Create a new {@link MessageListenerAdapter} for the given delegate.
144 	 * @param delegate the delegate object
145 	 */
146 	public MessageListenerAdapter(Object delegate) {
147 		setDelegate(delegate);
148 	}
149 
150 
151 	/**
152 	 * Set a target object to delegate message listening to.
153 	 * Specified listener methods have to be present on this target object.
154 	 * <p>If no explicit delegate object has been specified, listener
155 	 * methods are expected to present on this adapter instance, that is,
156 	 * on a custom subclass of this adapter, defining listener methods.
157 	 */
158 	public void setDelegate(Object delegate) {
159 		Assert.notNull(delegate, "Delegate must not be null");
160 		this.delegate = delegate;
161 	}
162 
163 	/**
164 	 * Return the target object to delegate message listening to.
165 	 */
166 	protected Object getDelegate() {
167 		return this.delegate;
168 	}
169 
170 	/**
171 	 * Specify the name of the default listener method to delegate to,
172 	 * for the case where no specific listener method has been determined.
173 	 * Out-of-the-box value is {@link #ORIGINAL_DEFAULT_LISTENER_METHOD "handleMessage"}.
174 	 * @see #getListenerMethodName
175 	 */
176 	public void setDefaultListenerMethod(String defaultListenerMethod) {
177 		this.defaultListenerMethod = defaultListenerMethod;
178 	}
179 
180 	/**
181 	 * Return the name of the default listener method to delegate to.
182 	 */
183 	protected String getDefaultListenerMethod() {
184 		return this.defaultListenerMethod;
185 	}
186 
187 	/**
188 	 * Spring {@link SessionAwareMessageListener} entry point.
189 	 * <p>Delegates the message to the target listener method, with appropriate
190 	 * conversion of the message argument. If the target method returns a
191 	 * non-null object, wrap in a JMS message and send it back.
192 	 * @param message the incoming JMS message
193 	 * @param session the JMS session to operate on
194 	 * @throws JMSException if thrown by JMS API methods
195 	 */
196 	@Override
197 	@SuppressWarnings("unchecked")
198 	public void onMessage(Message message, Session session) throws JMSException {
199 		// Check whether the delegate is a MessageListener impl itself.
200 		// In that case, the adapter will simply act as a pass-through.
201 		Object delegate = getDelegate();
202 		if (delegate != this) {
203 			if (delegate instanceof SessionAwareMessageListener) {
204 				if (session != null) {
205 					((SessionAwareMessageListener<Message>) delegate).onMessage(message, session);
206 					return;
207 				}
208 				else if (!(delegate instanceof MessageListener)) {
209 					throw new javax.jms.IllegalStateException("MessageListenerAdapter cannot handle a " +
210 							"SessionAwareMessageListener delegate if it hasn't been invoked with a Session itself");
211 				}
212 			}
213 			if (delegate instanceof MessageListener) {
214 				((MessageListener) delegate).onMessage(message);
215 				return;
216 			}
217 		}
218 
219 		// Regular case: find a handler method reflectively.
220 		Object convertedMessage = extractMessage(message);
221 		String methodName = getListenerMethodName(message, convertedMessage);
222 		if (methodName == null) {
223 			throw new javax.jms.IllegalStateException("No default listener method specified: " +
224 					"Either specify a non-null value for the 'defaultListenerMethod' property or " +
225 					"override the 'getListenerMethodName' method.");
226 		}
227 
228 		// Invoke the handler method with appropriate arguments.
229 		Object[] listenerArguments = buildListenerArguments(convertedMessage);
230 		Object result = invokeListenerMethod(methodName, listenerArguments);
231 		if (result != null) {
232 			handleResult(result, message, session);
233 		}
234 		else {
235 			logger.trace("No result object given - no result to handle");
236 		}
237 	}
238 
239 	@Override
240 	public String getSubscriptionName() {
241 		Object delegate = getDelegate();
242 		if (delegate != this && delegate instanceof SubscriptionNameProvider) {
243 			return ((SubscriptionNameProvider) delegate).getSubscriptionName();
244 		}
245 		else {
246 			return delegate.getClass().getName();
247 		}
248 	}
249 
250 	/**
251 	 * Determine the name of the listener method that is supposed to
252 	 * handle the given message.
253 	 * <p>The default implementation simply returns the configured
254 	 * default listener method, if any.
255 	 * @param originalMessage the JMS request message
256 	 * @param extractedMessage the converted JMS request message,
257 	 * to be passed into the listener method as argument
258 	 * @return the name of the listener method (never {@code null})
259 	 * @throws JMSException if thrown by JMS API methods
260 	 * @see #setDefaultListenerMethod
261 	 */
262 	protected String getListenerMethodName(Message originalMessage, Object extractedMessage) throws JMSException {
263 		return getDefaultListenerMethod();
264 	}
265 
266 	/**
267 	 * Build an array of arguments to be passed into the target listener method.
268 	 * Allows for multiple method arguments to be built from a single message object.
269 	 * <p>The default implementation builds an array with the given message object
270 	 * as sole element. This means that the extracted message will always be passed
271 	 * into a <i>single</i> method argument, even if it is an array, with the target
272 	 * method having a corresponding single argument of the array's type declared.
273 	 * <p>This can be overridden to treat special message content such as arrays
274 	 * differently, for example passing in each element of the message array
275 	 * as distinct method argument.
276 	 * @param extractedMessage the content of the message
277 	 * @return the array of arguments to be passed into the
278 	 * listener method (each element of the array corresponding
279 	 * to a distinct method argument)
280 	 */
281 	protected Object[] buildListenerArguments(Object extractedMessage) {
282 		return new Object[] {extractedMessage};
283 	}
284 
285 	/**
286 	 * Invoke the specified listener method.
287 	 * @param methodName the name of the listener method
288 	 * @param arguments the message arguments to be passed in
289 	 * @return the result returned from the listener method
290 	 * @throws JMSException if thrown by JMS API methods
291 	 * @see #getListenerMethodName
292 	 * @see #buildListenerArguments
293 	 */
294 	protected Object invokeListenerMethod(String methodName, Object[] arguments) throws JMSException {
295 		try {
296 			MethodInvoker methodInvoker = new MethodInvoker();
297 			methodInvoker.setTargetObject(getDelegate());
298 			methodInvoker.setTargetMethod(methodName);
299 			methodInvoker.setArguments(arguments);
300 			methodInvoker.prepare();
301 			return methodInvoker.invoke();
302 		}
303 		catch (InvocationTargetException ex) {
304 			Throwable targetEx = ex.getTargetException();
305 			if (targetEx instanceof JMSException) {
306 				throw (JMSException) targetEx;
307 			}
308 			else {
309 				throw new ListenerExecutionFailedException(
310 						"Listener method '" + methodName + "' threw exception", targetEx);
311 			}
312 		}
313 		catch (Throwable ex) {
314 			throw new ListenerExecutionFailedException("Failed to invoke target method '" + methodName +
315 					"' with arguments " + ObjectUtils.nullSafeToString(arguments), ex);
316 		}
317 	}
