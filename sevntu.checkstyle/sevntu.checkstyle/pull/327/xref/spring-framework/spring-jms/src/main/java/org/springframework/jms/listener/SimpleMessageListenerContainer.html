35  /**
36   * Message listener container that uses the plain JMS client API's
37   * {@code MessageConsumer.setMessageListener()} method to
38   * create concurrent MessageConsumers for the specified listeners.
39   *
40   * <p>This is the simplest form of a message listener container.
41   * It creates a fixed number of JMS Sessions to invoke the listener,
42   * not allowing for dynamic adaptation to runtime demands. Its main
43   * advantage is its low level of complexity and the minimum requirements
44   * on the JMS provider: Not even the ServerSessionPool facility is required.
45   *
46   * <p>See the {@link AbstractMessageListenerContainer} javadoc for details
47   * on acknowledge modes and transaction options. Note that this container
48   * exposes standard JMS behavior for the default "AUTO_ACKNOWLEDGE" mode:
49   * that is, automatic message acknowledgment after listener execution,
50   * with redelivery in case of an exception thrown.
51   *
52   * <p>For a different style of MessageListener handling, through looped
53   * {@code MessageConsumer.receive()} calls that also allow for
54   * transactional reception of messages (registering them with XA transactions),
55   * see {@link DefaultMessageListenerContainer}.
56   *
57   * @author Juergen Hoeller
58   * @since 2.0
59   * @see javax.jms.MessageConsumer#setMessageListener
60   * @see DefaultMessageListenerContainer
61   * @see org.springframework.jms.listener.endpoint.JmsMessageEndpointManager
62   */
63  public class SimpleMessageListenerContainer extends AbstractMessageListenerContainer implements ExceptionListener {
64  
65  	private boolean connectLazily = false;
66  
67  	private int concurrentConsumers = 1;
68  
69  	private Executor taskExecutor;
70  
71  	private Set<Session> sessions;
72  
73  	private Set<MessageConsumer> consumers;
74  
75  	private final Object consumersMonitor = new Object();
76  
77  
78  	/**
79  	 * Specify whether to connect lazily, i.e. whether to establish the JMS Connection
80  	 * and the corresponding Sessions and MessageConsumers as late as possible -
81  	 * in the start phase of this container.
82  	 * <p>Default is "false": connecting early, i.e. during the bean initialization phase.
83  	 * Set this flag to "true" in order to switch to lazy connecting if your target broker
84  	 * is likely to not have started up yet and you prefer to not even try a connection.
85  	 * @see #start()
86  	 * @see #initialize()
87  	 */
88  	public void setConnectLazily(boolean connectLazily) {
89  		this.connectLazily = connectLazily;
90  	}
91  
92  	/**
93  	 * Specify concurrency limits via a "lower-upper" String, e.g. "5-10", or a simple
94  	 * upper limit String, e.g. "10".
95  	 * <p>This listener container will always hold on to the maximum number of
96  	 * consumers {@link #setConcurrentConsumers} since it is unable to scale.
97  	 * <p>This property is primarily supported for configuration compatibility with
98  	 * {@link DefaultMessageListenerContainer}. For this local listener container,
99  	 * generally use {@link #setConcurrentConsumers} instead.
100 	 */
101 	@Override
102 	public void setConcurrency(String concurrency) {
103 		try {
104 			int separatorIndex = concurrency.indexOf('-');
105 			if (separatorIndex != -1) {
106 				setConcurrentConsumers(Integer.parseInt(concurrency.substring(separatorIndex + 1, concurrency.length())));
107 			}
108 			else {
109 				setConcurrentConsumers(Integer.parseInt(concurrency));
110 			}
111 		}
112 		catch (NumberFormatException ex) {
113 			throw new IllegalArgumentException("Invalid concurrency value [" + concurrency + "]: only " +
114 					"single maximum integer (e.g. \"5\") and minimum-maximum combo (e.g. \"3-5\") supported. " +
115 					"Note that SimpleMessageListenerContainer will effectively ignore the minimum value and " +
116 					"always keep a fixed number of consumers according to the maximum value.");
117 		}
118 	}
119 
120 	/**
121 	 * Specify the number of concurrent consumers to create. Default is 1.
122 	 * <p>Raising the number of concurrent consumers is recommendable in order
123 	 * to scale the consumption of messages coming in from a queue. However,
124 	 * note that any ordering guarantees are lost once multiple consumers are
125 	 * registered. In general, stick with 1 consumer for low-volume queues.
126 	 * <p><b>Do not raise the number of concurrent consumers for a topic.</b>
127 	 * This would lead to concurrent consumption of the same message,
128 	 * which is hardly ever desirable.
129 	 */
130 	public void setConcurrentConsumers(int concurrentConsumers) {
131 		Assert.isTrue(concurrentConsumers > 0, "'concurrentConsumers' value must be at least 1 (one)");
132 		this.concurrentConsumers = concurrentConsumers;
133 	}
134 
135 	/**
136 	 * Set the Spring TaskExecutor to use for executing the listener once
137 	 * a message has been received by the provider.
138 	 * <p>Default is none, that is, to run in the JMS provider's own receive thread,
139 	 * blocking the provider's receive endpoint while executing the listener.
140 	 * <p>Specify a TaskExecutor for executing the listener in a different thread,
141 	 * rather than blocking the JMS provider, usually integrating with an existing
142 	 * thread pool. This allows to keep the number of concurrent consumers low (1)
143 	 * while still processing messages concurrently (decoupled from receiving!).
144 	 * <p><b>NOTE: Specifying a TaskExecutor for listener execution affects
145 	 * acknowledgement semantics.</b> Messages will then always get acknowledged
146 	 * before listener execution, with the underlying Session immediately reused
147 	 * for receiving the next message. Using this in combination with a transacted
148 	 * session or with client acknowledgement will lead to unspecified results!
149 	 * <p><b>NOTE: Concurrent listener execution via a TaskExecutor will lead
150 	 * to concurrent processing of messages that have been received by the same
151 	 * underlying Session.</b> As a consequence, it is not recommended to use
152 	 * this setting with a {@link SessionAwareMessageListener}, at least not
153 	 * if the latter performs actual work on the given Session. A standard
154 	 * {@link javax.jms.MessageListener} will work fine, in general.
155 	 * @see #setConcurrentConsumers
156 	 * @see org.springframework.core.task.SimpleAsyncTaskExecutor
157 	 * @see org.springframework.scheduling.commonj.WorkManagerTaskExecutor
158 	 */
159 	public void setTaskExecutor(Executor taskExecutor) {
160 		this.taskExecutor = taskExecutor;
161 	}
162 
163 	@Override
164 	protected void validateConfiguration() {
165 		super.validateConfiguration();
166 		if (isSubscriptionDurable() && this.concurrentConsumers != 1) {
167 			throw new IllegalArgumentException("Only 1 concurrent consumer supported for durable subscription");
168 		}
169 	}

173 	// Implementation of AbstractMessageListenerContainer's template methods
174 	//-------------------------------------------------------------------------
175 
176 	/**
177 	 * Always use a shared JMS Connection.
178 	 */
179 	@Override
180 	protected final boolean sharedConnectionEnabled() {
181 		return true;
182 	}
183 
184 	/**
185 	 * Creates the specified number of concurrent consumers,
186 	 * in the form of a JMS Session plus associated MessageConsumer.
187 	 * @see #createListenerConsumer
188 	 */
189 	@Override
190 	protected void doInitialize() throws JMSException {
191 		if (!this.connectLazily) {
192 			try {
193 				establishSharedConnection();
194 			}
195 			catch (JMSException ex) {
196 				logger.debug("Could not connect on initialization - registering message consumers lazily", ex);
197 				return;
198 			}
199 			initializeConsumers();
200 		}
201 	}
202 
203 	/**
204 	 * Re-initializes this container's JMS message consumers,
205 	 * if not initialized already.
206 	 */
207 	@Override
208 	protected void doStart() throws JMSException {
209 		super.doStart();
210 		initializeConsumers();
211 	}
212 
213 	/**
214 	 * Registers this listener container as JMS ExceptionListener on the shared connection.
215 	 */
216 	@Override
217 	protected void prepareSharedConnection(Connection connection) throws JMSException {
218 		super.prepareSharedConnection(connection);
219 		connection.setExceptionListener(this);
220 	}
221 
222 	/**
223 	 * JMS ExceptionListener implementation, invoked by the JMS provider in
224 	 * case of connection failures. Re-initializes this listener container's
225 	 * shared connection and its sessions and consumers.
226 	 * @param ex the reported connection exception
227 	 */
228 	@Override
229 	public void onException(JMSException ex) {
230 		// First invoke the user-specific ExceptionListener, if any.
231 		invokeExceptionListener(ex);
232 
233 		// Now try to recover the shared Connection and all consumers...
234 		if (logger.isInfoEnabled()) {
235 			logger.info("Trying to recover from JMS Connection exception: " + ex);
236 		}
237 		try {
238 			synchronized (this.consumersMonitor) {
239 				this.sessions = null;
240 				this.consumers = null;
241 			}
242 			refreshSharedConnection();
243 			initializeConsumers();
244 			logger.info("Successfully refreshed JMS Connection");
245 		}
246 		catch (JMSException recoverEx) {
247 			logger.debug("Failed to recover JMS Connection", recoverEx);
248 			logger.error("Encountered non-recoverable JMSException", ex);
249 		}
250 	}
251 
252 	/**
253 	 * Initialize the JMS Sessions and MessageConsumers for this container.
254 	 * @throws JMSException in case of setup failure
255 	 */
256 	protected void initializeConsumers() throws JMSException {
257 		// Register Sessions and MessageConsumers.
258 		synchronized (this.consumersMonitor) {
259 			if (this.consumers == null) {
260 				this.sessions = new HashSet<Session>(this.concurrentConsumers);
261 				this.consumers = new HashSet<MessageConsumer>(this.concurrentConsumers);
262 				Connection con = getSharedConnection();
263 				for (int i = 0; i < this.concurrentConsumers; i++) {
264 					Session session = createSession(con);
265 					MessageConsumer consumer = createListenerConsumer(session);
266 					this.sessions.add(session);
267 					this.consumers.add(consumer);
268 				}
269 			}
270 		}
271 	}
272 
273 	/**
274 	 * Create a MessageConsumer for the given JMS Session,
275 	 * registering a MessageListener for the specified listener.
276 	 * @param session the JMS Session to work on
277 	 * @return the MessageConsumer
278 	 * @throws JMSException if thrown by JMS methods
279 	 * @see #executeListener
280 	 */
281 	protected MessageConsumer createListenerConsumer(final Session session) throws JMSException {
282 		Destination destination = getDestination();
283 		if (destination == null) {
284 			destination = resolveDestinationName(session, getDestinationName());
285 		}
286 		MessageConsumer consumer = createConsumer(session, destination);
287 
288 		if (this.taskExecutor != null) {
289 			consumer.setMessageListener(new MessageListener() {
290 				@Override
291 				public void onMessage(final Message message) {
292 					taskExecutor.execute(new Runnable() {
293 						@Override
294 						public void run() {
295 							processMessage(message, session);
296 						}
297 					});
298 				}
299 			});
300 		}
301 		else {
302 			consumer.setMessageListener(new MessageListener() {
303 				@Override
304 				public void onMessage(Message message) {
305 					processMessage(message, session);
306 				}
307 			});
308 		}
309 
310 		return consumer;
311 	}
312 
313 	/**
314 	 * Process a message received from the provider.
315 	 * <p>Executes the listener, exposing the current JMS Session as
316 	 * thread-bound resource (if "exposeListenerSession" is "true").
317 	 * @param message the received JMS Message
318 	 * @param session the JMS Session to operate on
319 	 * @see #executeListener
320 	 * @see #setExposeListenerSession
321 	 */
322 	protected void processMessage(Message message, Session session) {
323 		boolean exposeResource = isExposeListenerSession();
324 		if (exposeResource) {
325 			TransactionSynchronizationManager.bindResource(
326 					getConnectionFactory(), new LocallyExposedJmsResourceHolder(session));
327 		}
328 		try {
329 			executeListener(session, message);
330 		}
331 		finally {
332 			if (exposeResource) {
333 				TransactionSynchronizationManager.unbindResource(getConnectionFactory());
334 			}
335 		}
336 	}
337 
338 	/**
339 	 * Destroy the registered JMS Sessions and associated MessageConsumers.
340 	 */
341 	@Override
342 	protected void doShutdown() throws JMSException {
343 		synchronized (this.consumersMonitor) {
344 			if (this.consumers != null) {
345 				logger.debug("Closing JMS MessageConsumers");
346 				for (MessageConsumer consumer : this.consumers) {
347 					JmsUtils.closeMessageConsumer(consumer);
348 				}
349 				logger.debug("Closing JMS Sessions");
350 				for (Session session : this.sessions) {
351 					JmsUtils.closeSession(session);
352 				}
353 			}
354 		}
355 	}
