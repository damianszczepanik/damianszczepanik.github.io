37  /**
38   * Base class for listener container implementations which are based on polling.
39   * Provides support for listener handling based on {@link javax.jms.MessageConsumer},
40   * optionally participating in externally managed transactions.
41   *
42   * <p>This listener container variant is built for repeated polling attempts,
43   * each invoking the {@link #receiveAndExecute} method. The MessageConsumer used
44   * may be reobtained fo reach attempt or cached in between attempts; this is up
45   * to the concrete implementation. The receive timeout for each attempt can be
46   * configured through the {@link #setReceiveTimeout "receiveTimeout"} property.
47   *
48   * <p>The underlying mechanism is based on standard JMS MessageConsumer handling,
49   * which is perfectly compatible with both native JMS and JMS in a J2EE environment.
50   * Neither the JMS {@code MessageConsumer.setMessageListener} facility
51   * nor the JMS ServerSessionPool facility is required. A further advantage
52   * of this approach is full control over the listening process, allowing for
53   * custom scaling and throttling and of concurrent message processing
54   * (which is up to concrete subclasses).
55   *
56   * <p>Message reception and listener execution can automatically be wrapped
57   * in transactions through passing a Spring
58   * {@link org.springframework.transaction.PlatformTransactionManager} into the
59   * {@link #setTransactionManager "transactionManager"} property. This will usually
60   * be a {@link org.springframework.transaction.jta.JtaTransactionManager} in a
61   * J2EE enviroment, in combination with a JTA-aware JMS ConnectionFactory obtained
62   * from JNDI (check your J2EE server's documentation).
63   *
64   * <p>This base class does not assume any specific mechanism for asynchronous
65   * execution of polling invokers. Check out {@link DefaultMessageListenerContainer}
66   * for a concrete implementation which is based on Spring's
67   * {@link org.springframework.core.task.TaskExecutor} abstraction,
68   * including dynamic scaling of concurrent consumers and automatic self recovery.
69   *
70   * @author Juergen Hoeller
71   * @since 2.0.3
72   * @see #createListenerConsumer
73   * @see #receiveAndExecute
74   * @see #setTransactionManager
75   */
76  public abstract class AbstractPollingMessageListenerContainer extends AbstractMessageListenerContainer {

104 	/**
105 	 * Specify the Spring {@link org.springframework.transaction.PlatformTransactionManager}
106 	 * to use for transactional wrapping of message reception plus listener execution.
107 	 * <p>Default is none, not performing any transactional wrapping.
108 	 * If specified, this will usually be a Spring
109 	 * {@link org.springframework.transaction.jta.JtaTransactionManager} or one
110 	 * of its subclasses, in combination with a JTA-aware ConnectionFactory that
111 	 * this message listener container obtains its Connections from.
112 	 * <p><b>Note: Consider the use of local JMS transactions instead.</b>
113 	 * Simply switch the {@link #setSessionTransacted "sessionTransacted"} flag
114 	 * to "true" in order to use a locally transacted JMS Session for the entire
115 	 * receive processing, including any Session operations performed by a
116 	 * {@link SessionAwareMessageListener} (e.g. sending a response message). This
117 	 * allows for fully synchronized Spring transactions based on local JMS
118 	 * transactions, similar to what
119 	 * {@link org.springframework.jms.connection.JmsTransactionManager} provides. Check
120 	 * {@link AbstractMessageListenerContainer}'s javadoc for
121 	 * a discussion of transaction choices and message redelivery scenarios.
122 	 * @see #setSessionTransacted(boolean)
123 	 * @see org.springframework.transaction.jta.JtaTransactionManager
124 	 * @see org.springframework.jms.connection.JmsTransactionManager
125 	 */
126 	public void setTransactionManager(PlatformTransactionManager transactionManager) {
127 		this.transactionManager = transactionManager;
128 	}

173 	@Override
174 	public void initialize() {
175 		// Set sessionTransacted=true in case of a non-JTA transaction manager.
176 		if (!this.sessionTransactedCalled &&
177 				this.transactionManager instanceof ResourceTransactionManager &&
178 				!TransactionSynchronizationUtils.sameResourceFactory(
179 						(ResourceTransactionManager) this.transactionManager, getConnectionFactory())) {
180 			super.setSessionTransacted(true);
181 		}
182 
183 		// Use bean name as default transaction name.
184 		if (this.transactionDefinition.getName() == null) {
185 			this.transactionDefinition.setName(getBeanName());
186 		}
187 
188 		// Proceed with superclass initialization.
189 		super.initialize();
190 	}

194 	 * Create a MessageConsumer for the given JMS Session,
195 	 * registering a MessageListener for the specified listener.
196 	 * @param session the JMS Session to work on
197 	 * @return the MessageConsumer
198 	 * @throws javax.jms.JMSException if thrown by JMS methods
199 	 * @see #receiveAndExecute
200 	 */
201 	protected MessageConsumer createListenerConsumer(Session session) throws JMSException {
202 		Destination destination = getDestination();
203 		if (destination == null) {
204 			destination = resolveDestinationName(session, getDestinationName());
205 		}
206 		return createConsumer(session, destination);
207 	}

209 	/**
210 	 * Execute the listener for a message received from the given consumer,
211 	 * wrapping the entire operation in an external transaction if demanded.
212 	 * @param session the JMS Session to work on
213 	 * @param consumer the MessageConsumer to work on
214 	 * @return whether a message has been received
215 	 * @throws JMSException if thrown by JMS methods
216 	 * @see #doReceiveAndExecute
217 	 */
218 	protected boolean receiveAndExecute(Object invoker, Session session, MessageConsumer consumer)
219 			throws JMSException {
220 
221 		if (this.transactionManager != null) {
222 			// Execute receive within transaction.
223 			TransactionStatus status = this.transactionManager.getTransaction(this.transactionDefinition);
224 			boolean messageReceived;
225 			try {
226 				messageReceived = doReceiveAndExecute(invoker, session, consumer, status);
227 			}
228 			catch (JMSException ex) {
229 				rollbackOnException(status, ex);
230 				throw ex;
231 			}
232 			catch (RuntimeException ex) {
233 				rollbackOnException(status, ex);
234 				throw ex;
235 			}
236 			catch (Error err) {
237 				rollbackOnException(status, err);
238 				throw err;
239 			}
240 			this.transactionManager.commit(status);
241 			return messageReceived;
242 		}
243 
244 		else {
245 			// Execute receive outside of transaction.
246 			return doReceiveAndExecute(invoker, session, consumer, null);
247 		}
248 	}

250 	/**
251 	 * Actually execute the listener for a message received from the given consumer,
252 	 * fetching all requires resources and invoking the listener.
253 	 * @param session the JMS Session to work on
254 	 * @param consumer the MessageConsumer to work on
255 	 * @param status the TransactionStatus (may be {@code null})
256 	 * @return whether a message has been received
257 	 * @throws JMSException if thrown by JMS methods
258 	 * @see #doExecuteListener(javax.jms.Session, javax.jms.Message)
259 	 */
260 	protected boolean doReceiveAndExecute(
261 			Object invoker, Session session, MessageConsumer consumer, TransactionStatus status)
262 			throws JMSException {
263 
264 		Connection conToClose = null;
265 		Session sessionToClose = null;
266 		MessageConsumer consumerToClose = null;
267 		try {
268 			Session sessionToUse = session;
269 			boolean transactional = false;
270 			if (sessionToUse == null) {
271 				sessionToUse = ConnectionFactoryUtils.doGetTransactionalSession(
272 						getConnectionFactory(), this.transactionalResourceFactory, true);
273 				transactional = (sessionToUse != null);
274 			}
275 			if (sessionToUse == null) {
276 				Connection conToUse;
277 				if (sharedConnectionEnabled()) {
278 					conToUse = getSharedConnection();
279 				}
280 				else {
281 					conToUse = createConnection();
282 					conToClose = conToUse;
283 					conToUse.start();
284 				}
285 				sessionToUse = createSession(conToUse);
286 				sessionToClose = sessionToUse;
287 			}
288 			MessageConsumer consumerToUse = consumer;
289 			if (consumerToUse == null) {
290 				consumerToUse = createListenerConsumer(sessionToUse);
291 				consumerToClose = consumerToUse;
292 			}
293 			Message message = receiveMessage(consumerToUse);
294 			if (message != null) {
295 				if (logger.isDebugEnabled()) {
296 					logger.debug("Received message of type [" + message.getClass() + "] from consumer [" +
297 							consumerToUse + "] of " + (transactional ? "transactional " : "") + "session [" +
298 							sessionToUse + "]");
299 				}
300 				messageReceived(invoker, sessionToUse);
301 				boolean exposeResource = (!transactional && isExposeListenerSession() &&
302 						!TransactionSynchronizationManager.hasResource(getConnectionFactory()));
303 				if (exposeResource) {
304 					TransactionSynchronizationManager.bindResource(
305 							getConnectionFactory(), new LocallyExposedJmsResourceHolder(sessionToUse));
306 				}
307 				try {
308 					doExecuteListener(sessionToUse, message);
309 				}
310 				catch (Throwable ex) {
311 					if (status != null) {
312 						if (logger.isDebugEnabled()) {
313 							logger.debug("Rolling back transaction because of listener exception thrown: " + ex);
314 						}
315 						status.setRollbackOnly();
316 					}
317 					handleListenerException(ex);
318 					// Rethrow JMSException to indicate an infrastructure problem
319 					// that may have to trigger recovery...
320 					if (ex instanceof JMSException) {
321 						throw (JMSException) ex;
322 					}
323 				}
324 				finally {
325 					if (exposeResource) {
326 						TransactionSynchronizationManager.unbindResource(getConnectionFactory());
327 					}
328 				}
329 				// Indicate that a message has been received.
330 				return true;
331 			}
332 			else {
333 				if (logger.isTraceEnabled()) {
334 					logger.trace("Consumer [" + consumerToUse + "] of " + (transactional ? "transactional " : "") +
335 							"session [" + sessionToUse + "] did not receive a message");
336 				}
337 				noMessageReceived(invoker, sessionToUse);
338 				// Nevertheless call commit, in order to reset the transaction timeout (if any).
339 				// However, don't do this on Tibco since this may lead to a deadlock there.
340 				if (shouldCommitAfterNoMessageReceived(sessionToUse)) {
341 					commitIfNecessary(sessionToUse, message);
342 				}
343 				// Indicate that no message has been received.
344 				return false;
345 			}
346 		}
347 		finally {
348 			JmsUtils.closeMessageConsumer(consumerToClose);
349 			JmsUtils.closeSession(sessionToClose);
350 			ConnectionFactoryUtils.releaseConnection(conToClose, getConnectionFactory(), true);
351 		}
352 	}

354 	/**
355 	 * This implementation checks whether the Session is externally synchronized.
356 	 * In this case, the Session is not locally transacted, despite the listener
357 	 * container's "sessionTransacted" flag being set to "true".
358 	 * @see org.springframework.jms.connection.JmsResourceHolder
359 	 */
360 	@Override
361 	protected boolean isSessionLocallyTransacted(Session session) {
362 		if (!super.isSessionLocallyTransacted(session)) {
363 			return false;
364 		}
365 		JmsResourceHolder resourceHolder =
366 				(JmsResourceHolder) TransactionSynchronizationManager.getResource(getConnectionFactory());
367 		return (resourceHolder == null || resourceHolder instanceof LocallyExposedJmsResourceHolder ||
368 				!resourceHolder.containsSession(session));
369 	}

371 	/**
372 	 * Determine whether to trigger a commit after no message has been received.
373 	 * This is a good idea on any JMS provider other than Tibco, which is what
374 	 * this default implementation checks for.
375 	 * @param session the current JMS Session which received no message
376 	 * @return whether to call {@link #commitIfNecessary} on the given Session
377 	 */
378 	protected boolean shouldCommitAfterNoMessageReceived(Session session) {
379 		if (this.commitAfterNoMessageReceived == null) {
380 			Session target = ConnectionFactoryUtils.getTargetSession(session);
381 			this.commitAfterNoMessageReceived = !target.getClass().getName().startsWith("com.tibco.tibjms.");
382 		}
383 		return this.commitAfterNoMessageReceived;
384 	}

386 	/**
387 	 * Perform a rollback, handling rollback exceptions properly.
388 	 * @param status object representing the transaction
389 	 * @param ex the thrown listener exception or error
390 	 */
391 	private void rollbackOnException(TransactionStatus status, Throwable ex) {
392 		logger.debug("Initiating transaction rollback on listener exception", ex);
393 		try {
394 			this.transactionManager.rollback(status);
395 		}
396 		catch (RuntimeException ex2) {
397 			logger.error("Listener exception overridden by rollback exception", ex);
398 			throw ex2;
399 		}
400 		catch (Error err) {
401 			logger.error("Listener exception overridden by rollback error", ex);
402 			throw err;
403 		}
404 	}

407 	 * Receive a message from the given consumer.

409 	 * @return the Message, or {@code null} if none
410 	 * @throws JMSException if thrown by JMS methods
411 	 */

413 		return (this.receiveTimeout < 0 ? consumer.receive() : consumer.receive(this.receiveTimeout));

462 	private class MessageListenerContainerResourceFactory implements ConnectionFactoryUtils.ResourceFactory {
463 
464 		@Override
465 		public Connection getConnection(JmsResourceHolder holder) {
466 			return AbstractPollingMessageListenerContainer.this.getConnection(holder);
467 		}
468 
469 		@Override
470 		public Session getSession(JmsResourceHolder holder) {
471 			return AbstractPollingMessageListenerContainer.this.getSession(holder);

474 		@Override
475 		public Connection createConnection() throws JMSException {
476 			if (AbstractPollingMessageListenerContainer.this.sharedConnectionEnabled()) {
477 				Connection sharedCon = AbstractPollingMessageListenerContainer.this.getSharedConnection();
478 				return new SingleConnectionFactory(sharedCon).createConnection();
479 			}
480 			else {
481 				return AbstractPollingMessageListenerContainer.this.createConnection();
482 			}
483 		}
484 
485 		@Override
486 		public Session createSession(Connection con) throws JMSException {
487 			return AbstractPollingMessageListenerContainer.this.createSession(con);
488 		}
489 
490 		@Override
491 		public boolean isSynchedLocalTransactionAllowed() {
492 			return AbstractPollingMessageListenerContainer.this.isSessionTransacted();
493 		}
