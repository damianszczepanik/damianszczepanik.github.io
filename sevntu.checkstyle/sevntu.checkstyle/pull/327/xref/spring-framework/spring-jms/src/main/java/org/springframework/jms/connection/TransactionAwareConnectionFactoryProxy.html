39  /**
40   * Proxy for a target JMS {@link javax.jms.ConnectionFactory}, adding awareness of
41   * Spring-managed transactions. Similar to a transactional JNDI ConnectionFactory
42   * as provided by a J2EE server.
43   *
44   * <p>Messaging code which should remain unaware of Spring's JMS support can work with
45   * this proxy to seamlessly participate in Spring-managed transactions. Note that the
46   * transaction manager, for example {@link JmsTransactionManager}, still needs to work
47   * with the underlying ConnectionFactory, <i>not</i> with this proxy.
48   *
49   * <p><b>Make sure that TransactionAwareConnectionFactoryProxy is the outermost
50   * ConnectionFactory of a chain of ConnectionFactory proxies/adapters.</b>
51   * TransactionAwareConnectionFactoryProxy can delegate either directly to the
52   * target factory or to some intermediary adapter like
53   * {@link UserCredentialsConnectionFactoryAdapter}.
54   *
55   * <p>Delegates to {@link ConnectionFactoryUtils} for automatically participating
56   * in thread-bound transactions, for example managed by {@link JmsTransactionManager}.
57   * {@code createSession} calls and {@code close} calls on returned Sessions
58   * will behave properly within a transaction, that is, always work on the transactional
59   * Session. If not within a transaction, normal ConnectionFactory behavior applies.
60   *
61   * <p>Note that transactional JMS Sessions will be registered on a per-Connection
62   * basis. To share the same JMS Session across a transaction, make sure that you
63   * operate on the same JMS Connection handle - either through reusing the handle
64   * or through configuring a {@link SingleConnectionFactory} underneath.
65   *
66   * <p>Returned transactional Session proxies will implement the {@link SessionProxy}
67   * interface to allow for access to the underlying target Session. This is only
68   * intended for accessing vendor-specific Session API or for testing purposes
69   * (e.g. to perform manual transaction control). For typical application purposes,
70   * simply use the standard JMS Session interface.
71   *
72   * @author Juergen Hoeller
73   * @since 2.0
74   * @see UserCredentialsConnectionFactoryAdapter
75   * @see SingleConnectionFactory
76   */
77  public class TransactionAwareConnectionFactoryProxy
78  		implements ConnectionFactory, QueueConnectionFactory, TopicConnectionFactory {
79  
80  	private boolean synchedLocalTransactionAllowed = false;
81  
82  	private ConnectionFactory targetConnectionFactory;

100 	/**
101 	 * Set the target ConnectionFactory that this ConnectionFactory should delegate to.
102 	 */
103 	public final void setTargetConnectionFactory(ConnectionFactory targetConnectionFactory) {
104 		Assert.notNull(targetConnectionFactory, "targetConnectionFactory must not be nul");
105 		this.targetConnectionFactory = targetConnectionFactory;
106 	}
107 
108 	/**
109 	 * Return the target ConnectionFactory that this ConnectionFactory should delegate to.
110 	 */
111 	protected ConnectionFactory getTargetConnectionFactory() {
112 		return this.targetConnectionFactory;
113 	}
114 
115 	/**
116 	 * Set whether to allow for a local JMS transaction that is synchronized with a
117 	 * Spring-managed transaction (where the main transaction might be a JDBC-based
118 	 * one for a specific DataSource, for example), with the JMS transaction committing
119 	 * right after the main transaction. If not allowed, the given ConnectionFactory
120 	 * needs to handle transaction enlistment underneath the covers.
121 	 * <p>Default is "false": If not within a managed transaction that encompasses
122 	 * the underlying JMS ConnectionFactory, standard Sessions will be returned.
123 	 * Turn this flag on to allow participation in any Spring-managed transaction,
124 	 * with a local JMS transaction synchronized with the main transaction.
125 	 */
126 	public void setSynchedLocalTransactionAllowed(boolean synchedLocalTransactionAllowed) {
127 		this.synchedLocalTransactionAllowed = synchedLocalTransactionAllowed;
128 	}
129 
130 	/**
131 	 * Return whether to allow for a local JMS transaction that is synchronized
132 	 * with a Spring-managed transaction.
133 	 */
134 	protected boolean isSynchedLocalTransactionAllowed() {
135 		return this.synchedLocalTransactionAllowed;
136 	}

139 	@Override
140 	public Connection createConnection() throws JMSException {
141 		Connection targetConnection = this.targetConnectionFactory.createConnection();
142 		return getTransactionAwareConnectionProxy(targetConnection);
143 	}
144 
145 	@Override
146 	public Connection createConnection(String username, String password) throws JMSException {
147 		Connection targetConnection = this.targetConnectionFactory.createConnection(username, password);
148 		return getTransactionAwareConnectionProxy(targetConnection);
149 	}
150 
151 	@Override
152 	public QueueConnection createQueueConnection() throws JMSException {
153 		if (!(this.targetConnectionFactory instanceof QueueConnectionFactory)) {
154 			throw new javax.jms.IllegalStateException("'targetConnectionFactory' is no QueueConnectionFactory");
155 		}
156 		QueueConnection targetConnection =
157 				((QueueConnectionFactory) this.targetConnectionFactory).createQueueConnection();
158 		return (QueueConnection) getTransactionAwareConnectionProxy(targetConnection);
159 	}

192 	/**
193 	 * Wrap the given Connection with a proxy that delegates every method call to it
194 	 * but handles Session lookup in a transaction-aware fashion.
195 	 * @param target the original Connection to wrap
196 	 * @return the wrapped Connection
197 	 */
198 	protected Connection getTransactionAwareConnectionProxy(Connection target) {
199 		List<Class<?>> classes = new ArrayList<Class<?>>(3);
200 		classes.add(Connection.class);
201 		if (target instanceof QueueConnection) {
202 			classes.add(QueueConnection.class);
203 		}
204 		if (target instanceof TopicConnection) {
205 			classes.add(TopicConnection.class);
206 		}
207 		return (Connection) Proxy.newProxyInstance(
208 				Connection.class.getClassLoader(),
209 				classes.toArray(new Class<?>[classes.size()]),
210 				new TransactionAwareConnectionInvocationHandler(target));
211 	}

214 	/**
215 	 * Invocation handler that exposes transactional Sessions for the underlying Connection.
216 	 */
217 	private class TransactionAwareConnectionInvocationHandler implements InvocationHandler {
218 
219 		private final Connection target;
220 
221 		public TransactionAwareConnectionInvocationHandler(Connection target) {
222 			this.target = target;
223 		}
224 
225 		@Override
226 		public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
227 			// Invocation on ConnectionProxy interface coming in...
228 
229 			if (method.getName().equals("equals")) {
230 				// Only consider equal when proxies are identical.
231 				return (proxy == args[0]);
232 			}
233 			else if (method.getName().equals("hashCode")) {
234 				// Use hashCode of Connection proxy.
235 				return System.identityHashCode(proxy);
236 			}
237 			else if (Session.class.equals(method.getReturnType())) {
238 				Session session = ConnectionFactoryUtils.getTransactionalSession(
239 						getTargetConnectionFactory(), this.target, isSynchedLocalTransactionAllowed());
240 				if (session != null) {
241 					return getCloseSuppressingSessionProxy(session);
242 				}
243 			}
244 			else if (QueueSession.class.equals(method.getReturnType())) {
245 				QueueSession session = ConnectionFactoryUtils.getTransactionalQueueSession(
246 						(QueueConnectionFactory) getTargetConnectionFactory(), (QueueConnection) this.target,
247 						isSynchedLocalTransactionAllowed());
248 				if (session != null) {
249 					return getCloseSuppressingSessionProxy(session);
250 				}
251 			}
252 			else if (TopicSession.class.equals(method.getReturnType())) {
253 				TopicSession session = ConnectionFactoryUtils.getTransactionalTopicSession(
254 						(TopicConnectionFactory) getTargetConnectionFactory(), (TopicConnection) this.target,
255 						isSynchedLocalTransactionAllowed());
256 				if (session != null) {
257 					return getCloseSuppressingSessionProxy(session);
258 				}
259 			}
260 
261 			// Invoke method on target Connection.
262 			try {
263 				return method.invoke(this.target, args);
264 			}
265 			catch (InvocationTargetException ex) {
266 				throw ex.getTargetException();
267 			}
268 		}
269 
270 		private Session getCloseSuppressingSessionProxy(Session target) {
271 			List<Class<?>> classes = new ArrayList<Class<?>>(3);
272 			classes.add(SessionProxy.class);
273 			if (target instanceof QueueSession) {
274 				classes.add(QueueSession.class);
275 			}
276 			if (target instanceof TopicSession) {
277 				classes.add(TopicSession.class);
278 			}
279 			return (Session) Proxy.newProxyInstance(
280 					SessionProxy.class.getClassLoader(),
281 					classes.toArray(new Class<?>[classes.size()]),
282 					new CloseSuppressingSessionInvocationHandler(target));
283 		}
284 	}

287 	/**
288 	 * Invocation handler that suppresses close calls for a transactional JMS Session.
289 	 */
290 	private static class CloseSuppressingSessionInvocationHandler implements InvocationHandler {
291 
292 		private final Session target;
293 
294 		public CloseSuppressingSessionInvocationHandler(Session target) {
295 			this.target = target;
296 		}
297 
298 		@Override
299 		public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
300 			// Invocation on SessionProxy interface coming in...
301 
302 			if (method.getName().equals("equals")) {
303 				// Only consider equal when proxies are identical.
304 				return (proxy == args[0]);
305 			}
306 			else if (method.getName().equals("hashCode")) {
307 				// Use hashCode of Connection proxy.
308 				return System.identityHashCode(proxy);
309 			}
310 			else if (method.getName().equals("commit")) {
311 				throw new TransactionInProgressException("Commit call not allowed within a managed transaction");
312 			}
313 			else if (method.getName().equals("rollback")) {
314 				throw new TransactionInProgressException("Rollback call not allowed within a managed transaction");
315 			}
316 			else if (method.getName().equals("close")) {
317 				// Handle close method: not to be closed within a transaction.
318 				return null;
319 			}
320 			else if (method.getName().equals("getTargetSession")) {
321 				// Handle getTargetSession method: return underlying Session.
322 				return this.target;
323 			}
324 
325 			// Invoke method on target Session.
326 			try {
327 				return method.invoke(this.target, args);
328 			}
329 			catch (InvocationTargetException ex) {
330 				throw ex.getTargetException();
331 			}
332 		}
333 	}
