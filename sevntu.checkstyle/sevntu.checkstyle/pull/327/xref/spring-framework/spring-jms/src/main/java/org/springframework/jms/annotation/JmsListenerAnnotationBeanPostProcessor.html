54  /**
55   * Bean post-processor that registers methods annotated with {@link JmsListener}
56   * to be invoked by a JMS message listener container created under the cover
57   * by a {@link org.springframework.jms.config.JmsListenerContainerFactory} according
58   * to the parameters of the annotation.
59   *
60   * <p>Annotated methods can use flexible arguments as defined by {@link JmsListener}.
61   *
62   * <p>This post-processor is automatically registered by Spring's
63   * {@code <jms:annotation-driven>} XML element, and also by the {@link EnableJms}
64   * annotation.
65   *
66   * <p>Auto-detect any {@link JmsListenerConfigurer} instances in the container,
67   * allowing for customization of the registry to be used, the default container
68   * factory or for fine-grained control over endpoints registration. See
69   * {@link EnableJms} Javadoc for complete usage details.
70   *
71   * @author Stephane Nicoll
72   * @author Juergen Hoeller
73   * @since 4.1
74   * @see JmsListener
75   * @see EnableJms
76   * @see JmsListenerConfigurer
77   * @see JmsListenerEndpointRegistrar
78   * @see JmsListenerEndpointRegistry
79   * @see org.springframework.jms.config.JmsListenerEndpoint
80   * @see MethodJmsListenerEndpoint
81   */
82  public class JmsListenerAnnotationBeanPostProcessor
83  		implements BeanPostProcessor, Ordered, BeanFactoryAware, SmartInitializingSingleton {

153 	@Override
154 	public void afterSingletonsInstantiated() {
155 		this.registrar.setBeanFactory(this.beanFactory);
156 
157 		if (this.beanFactory instanceof ListableBeanFactory) {
158 			Map<String, JmsListenerConfigurer> instances =
159 					((ListableBeanFactory) this.beanFactory).getBeansOfType(JmsListenerConfigurer.class);
160 			for (JmsListenerConfigurer configurer : instances.values()) {
161 				configurer.configureJmsListeners(this.registrar);
162 			}
163 		}
164 
165 		if (this.registrar.getEndpointRegistry() == null) {
166 			if (this.endpointRegistry == null) {
167 				Assert.state(this.beanFactory != null, "BeanFactory must be set to find endpoint registry by bean name");
168 				this.endpointRegistry = this.beanFactory.getBean(
169 						JmsListenerConfigUtils.JMS_LISTENER_ENDPOINT_REGISTRY_BEAN_NAME, JmsListenerEndpointRegistry.class);
170 			}
171 			this.registrar.setEndpointRegistry(this.endpointRegistry);
172 		}
173 
174 		if (this.containerFactoryBeanName != null) {
175 			this.registrar.setContainerFactoryBeanName(this.containerFactoryBeanName);
176 		}
177 
178 		// Set the custom handler method factory once resolved by the configurer
179 		MessageHandlerMethodFactory handlerMethodFactory = this.registrar.getMessageHandlerMethodFactory();
180 		if (handlerMethodFactory != null) {
181 			this.messageHandlerMethodFactory.setMessageHandlerMethodFactory(handlerMethodFactory);
182 		}
183 
184 		// Actually register all listeners
185 		this.registrar.afterPropertiesSet();
186 	}

190 	public Object postProcessBeforeInitialization(Object bean, String beanName) throws BeansException {
191 		return bean;

194 	@Override
195 	public Object postProcessAfterInitialization(final Object bean, String beanName) throws BeansException {
196 		if (!this.nonAnnotatedClasses.contains(bean.getClass())) {
197 			final Set<Method> annotatedMethods = new LinkedHashSet<Method>(1);
198 			Class<?> targetClass = AopUtils.getTargetClass(bean);
199 			ReflectionUtils.doWithMethods(targetClass, new ReflectionUtils.MethodCallback() {
200 				@Override
201 				public void doWith(Method method) throws IllegalArgumentException, IllegalAccessException {
202 					JmsListener jmsListener = AnnotationUtils.getAnnotation(method, JmsListener.class);
203 					if (jmsListener != null) {
204 						processJmsListener(jmsListener, method, bean);
205 						annotatedMethods.add(method);
206 					}
207 				}
208 			});
209 			if (annotatedMethods.isEmpty()) {
210 				this.nonAnnotatedClasses.add(bean.getClass());
211 				if (logger.isTraceEnabled()) {
212 					logger.trace("No @JmsListener annotations found on bean class: " + bean.getClass());
213 				}
214 			}
215 			else {
216 				// Non-empty set of methods
217 				if (logger.isDebugEnabled()) {
218 					logger.debug(annotatedMethods.size() + " @JmsListener methods processed on bean '" + beanName +
219 							"': " + annotatedMethods);
220 				}
221 			}
222 		}
223 		return bean;
224 	}
225 
226 	protected void processJmsListener(JmsListener jmsListener, Method method, Object bean) {
227 		if (AopUtils.isJdkDynamicProxy(bean)) {
228 			try {
229 				// Found a @JmsListener method on the target class for this JDK proxy ->
230 				// is it also present on the proxy itself?
231 				method = bean.getClass().getMethod(method.getName(), method.getParameterTypes());
232 			}
233 			catch (SecurityException ex) {
234 				ReflectionUtils.handleReflectionException(ex);
235 			}
236 			catch (NoSuchMethodException ex) {
237 				throw new IllegalStateException(String.format(
238 						"@JmsListener method '%s' found on bean target class '%s', " +
239 						"but not found in any interface(s) for bean JDK proxy. Either " +
240 						"pull the method up to an interface or switch to subclass (CGLIB) " +
241 						"proxies by setting proxy-target-class/proxyTargetClass " +
242 						"attribute to 'true'", method.getName(), method.getDeclaringClass().getSimpleName()));
243 			}
244 		}
245 
246 		MethodJmsListenerEndpoint endpoint = new MethodJmsListenerEndpoint();
247 		endpoint.setBean(bean);
248 		endpoint.setMethod(method);
249 		endpoint.setMessageHandlerMethodFactory(this.messageHandlerMethodFactory);
250 		endpoint.setId(getEndpointId(jmsListener));
251 		endpoint.setDestination(resolve(jmsListener.destination()));
252 		if (StringUtils.hasText(jmsListener.selector())) {
253 			endpoint.setSelector(resolve(jmsListener.selector()));
254 		}
255 		if (StringUtils.hasText(jmsListener.subscription())) {
256 			endpoint.setSubscription(resolve(jmsListener.subscription()));
257 		}
258 		if (StringUtils.hasText(jmsListener.concurrency())) {
259 			endpoint.setConcurrency(resolve(jmsListener.concurrency()));
260 		}
261 
262 		JmsListenerContainerFactory<?> factory = null;
263 		String containerFactoryBeanName = resolve(jmsListener.containerFactory());
264 		if (StringUtils.hasText(containerFactoryBeanName)) {
265 			Assert.state(this.beanFactory != null, "BeanFactory must be set to obtain container factory by bean name");
266 			try {
267 				factory = this.beanFactory.getBean(containerFactoryBeanName, JmsListenerContainerFactory.class);
268 			}
269 			catch (NoSuchBeanDefinitionException ex) {
270 				throw new BeanInitializationException("Could not register jms listener endpoint on [" +
271 						method + "], no " + JmsListenerContainerFactory.class.getSimpleName() + " with id '" +
272 						containerFactoryBeanName + "' was found in the application context", ex);
273 			}
274 		}
275 
276 		this.registrar.registerEndpoint(endpoint, factory);
277 	}
278 
279 	private String getEndpointId(JmsListener jmsListener) {
280 		if (StringUtils.hasText(jmsListener.id())) {
281 			return resolve(jmsListener.id());
282 		}
283 		else {
284 			return "org.springframework.jms.JmsListenerEndpointContainer#" + counter.getAndIncrement();
285 		}
286 	}

288 	/**
289 	 * Resolve the specified value if possible.
290 	 * @see ConfigurableBeanFactory#resolveEmbeddedValue
291 	 */
292 	private String resolve(String value) {
293 		if (this.beanFactory instanceof ConfigurableBeanFactory) {
294 			return ((ConfigurableBeanFactory) this.beanFactory).resolveEmbeddedValue(value);
295 		}
296 		return value;
297 	}

300 	/**
301 	 * A {@link MessageHandlerMethodFactory} adapter that offers a configurable underlying
302 	 * instance to use. Useful if the factory to use is determined once the endpoints
303 	 * have been registered but not created yet.
304 	 * @see JmsListenerEndpointRegistrar#setMessageHandlerMethodFactory
305 	 */
306 	private class MessageHandlerMethodFactoryAdapter implements MessageHandlerMethodFactory {
307 
308 		private MessageHandlerMethodFactory messageHandlerMethodFactory;
309 
310 		public void setMessageHandlerMethodFactory(MessageHandlerMethodFactory messageHandlerMethodFactory) {
311 			this.messageHandlerMethodFactory = messageHandlerMethodFactory;
312 		}
313 
314 		@Override
315 		public InvocableHandlerMethod createInvocableHandlerMethod(Object bean, Method method) {
316 			return getMessageHandlerMethodFactory().createInvocableHandlerMethod(bean, method);
317 		}
318 
319 		private MessageHandlerMethodFactory getMessageHandlerMethodFactory() {
320 			if (this.messageHandlerMethodFactory == null) {
321 				this.messageHandlerMethodFactory = createDefaultJmsHandlerMethodFactory();
322 			}
323 			return this.messageHandlerMethodFactory;
324 		}
325 
326 		private MessageHandlerMethodFactory createDefaultJmsHandlerMethodFactory() {
327 			DefaultMessageHandlerMethodFactory defaultFactory = new DefaultMessageHandlerMethodFactory();
328 			defaultFactory.setBeanFactory(beanFactory);
329 			defaultFactory.afterPropertiesSet();
330 			return defaultFactory;
331 		}
332 	}
