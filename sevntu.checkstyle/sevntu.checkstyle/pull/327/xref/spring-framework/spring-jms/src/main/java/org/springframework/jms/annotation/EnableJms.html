17  package org.springframework.jms.annotation;
18  
19  import java.lang.annotation.Documented;
20  import java.lang.annotation.ElementType;
21  import java.lang.annotation.Retention;
22  import java.lang.annotation.RetentionPolicy;
23  import java.lang.annotation.Target;
24  
25  import org.springframework.context.annotation.Import;
26  
27  /**
28   * Enable JMS listener annotated endpoints that are created under the cover
29   * by a {@link org.springframework.jms.config.JmsListenerContainerFactory
30   * JmsListenerContainerFactory}. To be used on
31   * {@link org.springframework.context.annotation.Configuration Configuration}
32   * classes as follows:
33   *
34   * <pre class="code">
35   * &#064;Configuration
36   * &#064;EnableJms
37   * public class AppConfig {
38   *     &#064;Bean
39   *     public DefaultJmsListenerContainerFactory myJmsListenerContainerFactory() {
40   *       DefaultJmsListenerContainerFactory factory = new DefaultJmsListenerContainerFactory();
41   *       factory.setConnectionFactory(connectionFactory());
42   *       factory.setDestinationResolver(destinationResolver());
43   *       factory.setConcurrency("5");
44   *       return factory;
45   *     }
46   *     // other &#064;Bean definitions
47   * }</pre>
48   *
49   * The {@code JmsListenerContainerFactory} is responsible to create the listener container
50   * responsible for a particular endpoint. Typical implementations, as the
51   * {@link org.springframework.jms.config.DefaultJmsListenerContainerFactory DefaultJmsListenerContainerFactory}
52   * used in the sample above, provides the necessary configuration options that are supported by
53   * the underlying {@link org.springframework.jms.listener.MessageListenerContainer MessageListenerContainer}.
54   *
55   * <p>{@code @EnableJms} enables detection of {@link JmsListener} annotations on any
56   * Spring-managed bean in the container. For example, given a class {@code MyService}:
57   *
58   * <pre class="code">
59   * package com.acme.foo;
60   *
61   * public class MyService {
62   *     &#064;JmsListener(containerFactory = "myJmsListenerContainerFactory", destination="myQueue")
63   *     public void process(String msg) {
64   *         // process incoming message
65   *     }
66   * }</pre>
67   *
68   * The container factory to use is identified by the {@link JmsListener#containerFactory() containerFactory}
69   * attribute defining the name of the {@code JmsListenerContainerFactory} bean to use.  When none
70   * is set a {@code JmsListenerContainerFactory} bean with name {@code jmsListenerContainerFactory} is
71   * assumed to be present.
72   *
73   * <p>the following configuration would ensure that every time a {@link javax.jms.Message}
74   * is received on the {@link javax.jms.Destination} named "myQueue", {@code MyService.process()}
75   * is called with the content of the message:
76   *
77   * <pre class="code">
78   * &#064;Configuration
79   * &#064;EnableJms
80   * public class AppConfig {
81   *     &#064;Bean
82   *     public MyService myService() {
83   *         return new MyService();
84   *     }
85   *
86   *     // JMS infrastructure setup
87   * }</pre>
88   *
89   * Alternatively, if {@code MyService} were annotated with {@code @Component}, the
90   * following configuration would ensure that its {@code @JmsListener} annotated
91   * method is invoked with a matching incoming message:
92   *
93   * <pre class="code">
94   * &#064;Configuration
95   * &#064;EnableJms
96   * &#064;ComponentScan(basePackages="com.acme.foo")
97   * public class AppConfig {
98   * }</pre>
99   *
100  * Note that the created containers are not registered against the application context
101  * but can be easily located for management purposes using the
102  * {@link org.springframework.jms.config.JmsListenerEndpointRegistry JmsListenerEndpointRegistry}.
103  *
104  * <p>Annotated methods can use flexible signature; in particular, it is possible to use
105  * the {@link org.springframework.messaging.Message Message} abstraction and related annotations,
106  * see {@link JmsListener} Javadoc for more details. For instance, the following would
107  * inject the content of the message and a a custom "myCounter" JMS header:
108  *
109  * <pre class="code">
110  * &#064;JmsListener(containerFactory = "myJmsListenerContainerFactory", destination="myQueue")
111  * public void process(String msg, @Header("myCounter") int counter) {
112  *     // process incoming message
113  * }</pre>
114  *
115  * These features are abstracted by the {@link org.springframework.messaging.handler.annotation.support.MessageHandlerMethodFactory
116  * MessageHandlerMethodFactory} that is responsible to build the necessary invoker to process
117  * the annotated method. By default, {@link org.springframework.messaging.handler.annotation.support.DefaultMessageHandlerMethodFactory
118  * DefaultMessageHandlerMethodFactory} is used.
119  *
120  * <p>When more control is desired, a {@code @Configuration} class may implement
121  * {@link JmsListenerConfigurer}. This allows access to the underlying
122  * {@link org.springframework.jms.config.JmsListenerEndpointRegistrar JmsListenerEndpointRegistrar}
123  * instance. The following example demonstrates how to specify an explicit default
124  * {@code JmsListenerContainerFactory}
125  *
126  * <pre class="code">
127  * &#064;Configuration
128  * &#064;EnableJms
129  * public class AppConfig implements JmsListenerConfigurer {
130  *     &#064;Override
131  *     public void configureJmsListeners(JmsListenerEndpointRegistrar registrar) {
132  *         registrar.setContainerFactory(myJmsListenerContainerFactory());
133  *     }
134  *
135  *     &#064;Bean
136  *     public JmsListenerContainerFactory<?> myJmsListenerContainerFactory() {
137  *         // factory settings
138  *     }
139  *
140  *     &#064;Bean
141  *     public MyService myService() {
142  *         return new MyService();
143  *     }
144  * }</pre>
145  *
146  * For reference, the example above can be compared to the following Spring XML
147  * configuration:
148  * <pre class="code">
149  * {@code <beans>
150  *     <jms:annotation-driven container-factory="myJmsListenerContainerFactory"/>
151  *
152  *     <bean id="myJmsListenerContainerFactory"
153  *           class="org.springframework.jms.config.DefaultJmsListenerContainerFactory">
154  *           // factory settings
155  *     </bean>
156  *
157  *     <bean id="myService" class="com.acme.foo.MyService"/>
158  * </beans>
159  * }</pre>
160  *
161  * It is also possible to specify a custom {@link org.springframework.jms.config.JmsListenerEndpointRegistry
162  * JmsListenerEndpointRegistry} in case you need more control on the way the containers
163  * are created and managed. The example below also demonstrates how to customize the
164  * {@code JmsHandlerMethodFactory} to use with a custom {@link org.springframework.validation.Validator
165  * Validator} so that payloads annotated with {@link org.springframework.validation.annotation.Validated
166  * Validated} are first validated against a custom {@code Validator}.
167  *
168  * <pre class="code">
169  * &#064;Configuration
170  * &#064;EnableJms
171  * public class AppConfig implements JmsListenerConfigurer {
172  *     &#064;Override
173  *     public void configureJmsListeners(JmsListenerEndpointRegistrar registrar) {
174  *         registrar.setEndpointRegistry(myJmsListenerEndpointRegistry());
175  *         registrar.setMessageHandlerMethodFactory(myJmsHandlerMethodFactory);
176  *     }
177  *
178  *     &#064;Bean
179  *     public JmsListenerEndpointRegistry<?> myJmsListenerEndpointRegistry() {
180  *         // registry configuration
181  *     }
182  *
183  *     &#064;Bean
184  *     public JmsHandlerMethodFactory myJmsHandlerMethodFactory() {
185  *        DefaultJmsHandlerMethodFactory factory = new DefaultJmsHandlerMethodFactory();
186  *        factory.setValidator(new MyValidator());
187  *        return factory;
188  *     }
189  *
190  *     &#064;Bean
191  *     public MyService myService() {
192  *         return new MyService();
193  *     }
194  * }</pre>
195  *
196  * For reference, the example above can be compared to the following Spring XML
197  * configuration:
198  * <pre class="code">
199  * {@code <beans>
200  *     <jms:annotation-driven registry="myJmsListenerEndpointRegistry"
201  *         handler-method-factory="myJmsHandlerMethodFactory"/&gt;
202  *
203  *     <bean id="myJmsListenerEndpointRegistry"
204  *           class="org.springframework.jms.config.JmsListenerEndpointRegistry">
205  *           // registry configuration
206  *     </bean>
207  *
208  *     <bean id="myJmsHandlerMethodFactory"
209  *           class="org.springframework.messaging.handler.support.DefaultJmsHandlerMethodFactory">
210  *         <property name="validator" ref="myValidator"/>
211  *     </bean>
212  *
213  *     <bean id="myService" class="com.acme.foo.MyService"/>
214  * </beans>
215  * }</pre>
216  *
217  * Implementing {@code JmsListenerConfigurer} also allows for fine-grained
218  * control over endpoints registration via the {@code JmsListenerEndpointRegistrar}.
219  * For example, the following configures an extra endpoint:
220  *
221  * <pre class="code">
222  * &#064;Configuration
223  * &#064;EnableJms
224  * public class AppConfig implements JmsListenerConfigurer {
225  *     &#064;Override
226  *     public void configureJmsListeners(JmsListenerEndpointRegistrar registrar) {
227  *         SimpleJmsListenerEndpoint myEndpoint = new SimpleJmsListenerEndpoint();
228  *         // ... configure the endpoint
229  *         registrar.registerEndpoint(endpoint, anotherJmsListenerContainerFactory());
230  *     }
231  *
232  *     &#064;Bean
233  *     public MyService myService() {
234  *         return new MyService();
235  *     }
236  *
237  *     &#064;Bean
238  *     public JmsListenerContainerFactory<?> anotherJmsListenerContainerFactory() {
239  *         // ...
240  *     }
241  *
242  *     // JMS infrastructure setup
243  * }</pre>
244  *
245  * Note that all beans implementing {@code JmsListenerConfigurer} will be detected and
246  * invoked in a similar fashion. The example above can be translated in a regular bean
247  * definition registered in the context in case you use the XML configuration.
248  *
249  * @author Stephane Nicoll
250  * @since 4.1
251  * @see JmsListener
252  * @see JmsListenerAnnotationBeanPostProcessor
253  * @see org.springframework.jms.config.JmsListenerEndpointRegistrar
254  * @see org.springframework.jms.config.JmsListenerEndpointRegistry
255  */
256 @Target(ElementType.TYPE)
257 @Retention(RetentionPolicy.RUNTIME)
258 @Documented
259 @Import(JmsBootstrapConfiguration.class)
260 public @interface EnableJms {
261 }
