17  package org.springframework.jms.config;
18  
19  import java.util.ArrayList;
20  import java.util.List;
21  
22  import org.springframework.beans.factory.BeanFactory;
23  import org.springframework.beans.factory.BeanFactoryAware;
24  import org.springframework.beans.factory.InitializingBean;
25  import org.springframework.beans.factory.config.ConfigurableBeanFactory;
26  import org.springframework.messaging.handler.annotation.support.DefaultMessageHandlerMethodFactory;
27  import org.springframework.messaging.handler.annotation.support.MessageHandlerMethodFactory;
28  import org.springframework.util.Assert;
29  
30  /**
31   * Helper bean for registering {@link JmsListenerEndpoint} with
32   * a {@link JmsListenerEndpointRegistry}.
33   *
34   * @author Stephane Nicoll
35   * @author Juergen Hoeller
36   * @since 4.1
37   * @see org.springframework.jms.annotation.JmsListenerConfigurer
38   */
39  public class JmsListenerEndpointRegistrar implements BeanFactoryAware, InitializingBean {
40  
41  	private JmsListenerEndpointRegistry endpointRegistry;
42  
43  	private MessageHandlerMethodFactory messageHandlerMethodFactory;
44  
45  	private JmsListenerContainerFactory<?> containerFactory;
46  
47  	private String containerFactoryBeanName;
48  
49  	private BeanFactory beanFactory;
50  
51  	private final List<JmsListenerEndpointDescriptor> endpointDescriptors =
52  			new ArrayList<JmsListenerEndpointDescriptor>();
53  
54  	private boolean startImmediately;
55  
56  	private Object mutex = endpointDescriptors;
57  
58  
59  	/**
60  	 * Set the {@link JmsListenerEndpointRegistry} instance to use.
61  	 */
62  	public void setEndpointRegistry(JmsListenerEndpointRegistry endpointRegistry) {
63  		this.endpointRegistry = endpointRegistry;
64  	}
65  
66  	/**
67  	 * Return the {@link JmsListenerEndpointRegistry} instance for this
68  	 * registrar, may be {@code null}.
69  	 */
70  	public JmsListenerEndpointRegistry getEndpointRegistry() {
71  		return this.endpointRegistry;
72  	}
73  
74  	/**
75  	 * Set the {@link MessageHandlerMethodFactory} to use to configure the message
76  	 * listener responsible to serve an endpoint detected by this processor.
77  	 * <p>By default, {@link DefaultMessageHandlerMethodFactory} is used and it
78  	 * can be configured further to support additional method arguments
79  	 * or to customize conversion and validation support. See
80  	 * {@link DefaultMessageHandlerMethodFactory} javadoc for more details.
81  	 */
82  	public void setMessageHandlerMethodFactory(MessageHandlerMethodFactory messageHandlerMethodFactory) {
83  		this.messageHandlerMethodFactory = messageHandlerMethodFactory;
84  	}
85  
86  	/**
87  	 * Return the custom {@link MessageHandlerMethodFactory} to use, if any.
88  	 */
89  	public MessageHandlerMethodFactory getMessageHandlerMethodFactory() {
90  		return this.messageHandlerMethodFactory;
91  	}
92  
93  	/**
94  	 * Set the {@link JmsListenerContainerFactory} to use in case a {@link JmsListenerEndpoint}
95  	 * is registered with a {@code null} container factory.
96  	 * <p>Alternatively, the bean name of the {@link JmsListenerContainerFactory} to use
97  	 * can be specified for a lazy lookup, see {@link #setContainerFactoryBeanName}.
98  	 */
99  	public void setContainerFactory(JmsListenerContainerFactory<?> containerFactory) {
100 		this.containerFactory = containerFactory;
101 	}
102 
103 	/**
104 	 * Set the bean name of the {@link JmsListenerContainerFactory} to use in case
105 	 * a {@link JmsListenerEndpoint} is registered with a {@code null} container factory.
106 	 * Alternatively, the container factory instance can be registered directly:
107 	 * see {@link #setContainerFactory(JmsListenerContainerFactory)}.
108 	 * @see #setBeanFactory
109 	 */
110 	public void setContainerFactoryBeanName(String containerFactoryBeanName) {
111 		this.containerFactoryBeanName = containerFactoryBeanName;
112 	}
113 
114 	/**
115 	 * A {@link BeanFactory} only needs to be available in conjunction with
116 	 * {@link #setContainerFactoryBeanName}.
117 	 */
118 	@Override
119 	public void setBeanFactory(BeanFactory beanFactory) {
120 		this.beanFactory = beanFactory;
121 		if (beanFactory instanceof ConfigurableBeanFactory) {
122 			ConfigurableBeanFactory cbf = (ConfigurableBeanFactory) beanFactory;
123 			this.mutex = cbf.getSingletonMutex();
124 		}
125 	}
126 
127 
128 	@Override
129 	public void afterPropertiesSet() {
130 		registerAllEndpoints();
131 	}
132 
133 	protected void registerAllEndpoints() {
134 		synchronized (this.mutex) {
135 			for (JmsListenerEndpointDescriptor descriptor : this.endpointDescriptors) {
136 				this.endpointRegistry.registerListenerContainer(
137 						descriptor.endpoint, resolveContainerFactory(descriptor));
138 			}
139 			startImmediately = true;  // trigger immediate startup
140 		}
141 	}
142 
143 	private JmsListenerContainerFactory<?> resolveContainerFactory(JmsListenerEndpointDescriptor descriptor) {
144 		if (descriptor.containerFactory != null) {
145 			return descriptor.containerFactory;
146 		}
147 		else if (this.containerFactory != null) {
148 			return this.containerFactory;
149 		}
150 		else if (this.containerFactoryBeanName != null) {
151 			Assert.state(this.beanFactory != null, "BeanFactory must be set to obtain container factory by bean name");
152 			this.containerFactory = this.beanFactory.getBean(
153 					this.containerFactoryBeanName, JmsListenerContainerFactory.class);
154 			return this.containerFactory;  // Consider changing this if live change of the factory is required
155 		}
156 		else {
157 			throw new IllegalStateException("Could not resolve the " +
158 					JmsListenerContainerFactory.class.getSimpleName() + " to use for [" +
159 					descriptor.endpoint + "] no factory was given and no default is set.");
160 		}
161 	}
162 
163 	/**
164 	 * Register a new {@link JmsListenerEndpoint} alongside the
165 	 * {@link JmsListenerContainerFactory} to use to create the underlying container.
166 	 * <p>The {@code factory} may be {@code null} if the default factory has to be
167 	 * used for that endpoint.
168 	 */
169 	public void registerEndpoint(JmsListenerEndpoint endpoint, JmsListenerContainerFactory<?> factory) {
170 		Assert.notNull(endpoint, "Endpoint must be set");
171 		Assert.hasText(endpoint.getId(), "Endpoint id must be set");
172 		// Factory may be null, we defer the resolution right before actually creating the container
173 		JmsListenerEndpointDescriptor descriptor = new JmsListenerEndpointDescriptor(endpoint, factory);
174 		synchronized (this.mutex) {
175 			if (startImmediately) { // Register and start immediately
176 				this.endpointRegistry.registerListenerContainer(descriptor.endpoint,
177 						resolveContainerFactory(descriptor), true);
178 			}
179 			else {
180 				this.endpointDescriptors.add(descriptor);
181 			}
182 		}
183 	}
184 
185 	/**
186 	 * Register a new {@link JmsListenerEndpoint} using the default
187 	 * {@link JmsListenerContainerFactory} to create the underlying container.
188 	 * @see #setContainerFactory(JmsListenerContainerFactory)
189 	 * @see #registerEndpoint(JmsListenerEndpoint, JmsListenerContainerFactory)
190 	 */
191 	public void registerEndpoint(JmsListenerEndpoint endpoint) {
192 		registerEndpoint(endpoint, null);
193 	}
194 
195 
196 	private static class JmsListenerEndpointDescriptor {
197 
198 		public final JmsListenerEndpoint endpoint;
199 
200 		public final JmsListenerContainerFactory<?> containerFactory;
201 
202 		public JmsListenerEndpointDescriptor(JmsListenerEndpoint endpoint, JmsListenerContainerFactory<?> containerFactory) {
203 			this.endpoint = endpoint;
204 			this.containerFactory = containerFactory;
205 		}
206 	}
207 
208 }
