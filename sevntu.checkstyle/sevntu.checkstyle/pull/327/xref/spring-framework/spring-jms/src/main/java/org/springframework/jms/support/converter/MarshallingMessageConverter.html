34  import org.springframework.beans.factory.InitializingBean;
35  import org.springframework.oxm.Marshaller;
36  import org.springframework.oxm.Unmarshaller;

40  /**
41   * Spring JMS {@link MessageConverter} that uses a {@link Marshaller} and {@link Unmarshaller}.
42   * Marshals an object to a {@link BytesMessage}, or to a {@link TextMessage} if the
43   * {@link #setTargetType targetType} is set to {@link MessageType#TEXT}.
44   * Unmarshals from a {@link TextMessage} or {@link BytesMessage} to an object.
45   *
46   * @author Arjen Poutsma
47   * @author Juergen Hoeller
48   * @since 3.0
49   * @see org.springframework.jms.core.JmsTemplate#convertAndSend
50   * @see org.springframework.jms.core.JmsTemplate#receiveAndConvert
51   */
52  public class MarshallingMessageConverter implements MessageConverter, InitializingBean {
53  
54  	private Marshaller marshaller;
55  
56  	private Unmarshaller unmarshaller;
57  
58  	private MessageType targetType = MessageType.BYTES;
59  
60  
61  	/**
62  	 * Construct a new {@code MarshallingMessageConverter} with no {@link Marshaller}
63  	 * or {@link Unmarshaller} set. The marshaller must be set after construction by invoking
64  	 * {@link #setMarshaller(Marshaller)} and {@link #setUnmarshaller(Unmarshaller)} .
65  	 */
66  	public MarshallingMessageConverter() {
67  	}
68  
69  	/**
70  	 * Construct a new {@code MarshallingMessageConverter} with the given {@link Marshaller} set.
71  	 * <p>If the given {@link Marshaller} also implements the {@link Unmarshaller} interface,
72  	 * it is used for both marshalling and unmarshalling. Otherwise, an exception is thrown.
73  	 * <p>Note that all {@link Marshaller} implementations in Spring also implement the
74  	 * {@link Unmarshaller} interface, so that you can safely use this constructor.
75  	 * @param marshaller object used as marshaller and unmarshaller
76  	 * @throws IllegalArgumentException when {@code marshaller} does not implement the
77  	 * {@link Unmarshaller} interface as well
78  	 */
79  	public MarshallingMessageConverter(Marshaller marshaller) {
80  		Assert.notNull(marshaller, "Marshaller must not be null");
81  		if (!(marshaller instanceof Unmarshaller)) {
82  			throw new IllegalArgumentException(
83  					"Marshaller [" + marshaller + "] does not implement the Unmarshaller " +
84  					"interface. Please set an Unmarshaller explicitly by using the " +
85  					"MarshallingMessageConverter(Marshaller, Unmarshaller) constructor.");
86  		}
87  		else {
88  			this.marshaller = marshaller;
89  			this.unmarshaller = (Unmarshaller) marshaller;
90  		}
91  	}
92  
93  	/**
94  	 * Construct a new {@code MarshallingMessageConverter} with the
95  	 * given Marshaller and Unmarshaller.
96  	 * @param marshaller the Marshaller to use
97  	 * @param unmarshaller the Unmarshaller to use
98  	 */
99  	public MarshallingMessageConverter(Marshaller marshaller, Unmarshaller unmarshaller) {
100 		Assert.notNull(marshaller, "Marshaller must not be null");
101 		Assert.notNull(unmarshaller, "Unmarshaller must not be null");
102 		this.marshaller = marshaller;
103 		this.unmarshaller = unmarshaller;
104 	}
105 
106 
107 	/**
108 	 * Set the {@link Marshaller} to be used by this message converter.
109 	 */
110 	public void setMarshaller(Marshaller marshaller) {
111 		this.marshaller = marshaller;
112 	}
113 
114 	/**
115 	 * Set the {@link Unmarshaller} to be used by this message converter.
116 	 */
117 	public void setUnmarshaller(Unmarshaller unmarshaller) {
118 		this.unmarshaller = unmarshaller;
119 	}
120 
121 	/**
122 	 * Specify whether {@link #toMessage(Object, Session)} should marshal to
123 	 * a {@link BytesMessage} or a {@link TextMessage}.
124 	 * <p>The default is {@link MessageType#BYTES}, i.e. this converter marshals
125 	 * to a {@link BytesMessage}. Note that the default version of this converter
126 	 * supports {@link MessageType#BYTES} and {@link MessageType#TEXT} only.
127 	 * @see MessageType#BYTES
128 	 * @see MessageType#TEXT
129 	 */
130 	public void setTargetType(MessageType targetType) {
131 		Assert.notNull(targetType, "MessageType must not be null");
132 		this.targetType = targetType;
133 	}
134 
135 	@Override
136 	public void afterPropertiesSet() {
137 		Assert.notNull(this.marshaller, "Property 'marshaller' is required");
138 		Assert.notNull(this.unmarshaller, "Property 'unmarshaller' is required");
139 	}
140 
141 
142 	/**
143 	 * This implementation marshals the given object to a {@link javax.jms.TextMessage} or
144 	 * {@link javax.jms.BytesMessage}. The desired message type can be defined by setting
145 	 * the {@link #setTargetType "marshalTo"} property.
146 	 * @see #marshalToTextMessage
147 	 * @see #marshalToBytesMessage
148 	 */
149 	@Override
150 	public Message toMessage(Object object, Session session) throws JMSException, MessageConversionException {
151 		try {
152 			switch (this.targetType) {
153 				case TEXT:
154 					return marshalToTextMessage(object, session, this.marshaller);
155 				case BYTES:
156 					return marshalToBytesMessage(object, session, this.marshaller);
157 				default:
158 					return marshalToMessage(object, session, this.marshaller, this.targetType);
159 			}
160 		}
161 		catch (XmlMappingException ex) {
162 			throw new MessageConversionException("Could not marshal [" + object + "]", ex);
163 		}
164 		catch (IOException ex) {
165 			throw new MessageConversionException("Could not marshal [" + object + "]", ex);
166 		}
167 	}
168 
169 	/**
170 	 * This implementation unmarshals the given {@link Message} into an object.
171 	 * @see #unmarshalFromTextMessage
172 	 * @see #unmarshalFromBytesMessage
173 	 */
174 	@Override
175 	public Object fromMessage(Message message) throws JMSException, MessageConversionException {
176 		try {
177 			if (message instanceof TextMessage) {
178 				TextMessage textMessage = (TextMessage) message;
179 				return unmarshalFromTextMessage(textMessage, this.unmarshaller);
180 			}
181 			else if (message instanceof BytesMessage) {
182 				BytesMessage bytesMessage = (BytesMessage) message;
183 				return unmarshalFromBytesMessage(bytesMessage, this.unmarshaller);
184 			}
185 			else {
186 				return unmarshalFromMessage(message, this.unmarshaller);
187 			}
188 		}
189 		catch (IOException ex) {
190 			throw new MessageConversionException("Could not access message content: " + message, ex);
191 		}
192 		catch (XmlMappingException ex) {
193 			throw new MessageConversionException("Could not unmarshal message: " + message, ex);
194 		}
195 	}

198 	/**
199 	 * Marshal the given object to a {@link TextMessage}.
200 	 * @param object the object to be marshalled
201 	 * @param session current JMS session
202 	 * @param marshaller the marshaller to use
203 	 * @return the resulting message
204 	 * @throws JMSException if thrown by JMS methods
205 	 * @throws IOException in case of I/O errors
206 	 * @throws XmlMappingException in case of OXM mapping errors
207 	 * @see Session#createTextMessage
208 	 * @see Marshaller#marshal(Object, Result)
209 	 */
210 	protected TextMessage marshalToTextMessage(Object object, Session session, Marshaller marshaller)
211 			throws JMSException, IOException, XmlMappingException {
212 
213 		StringWriter writer = new StringWriter();
214 		Result result = new StreamResult(writer);
215 		marshaller.marshal(object, result);
216 		return session.createTextMessage(writer.toString());
217 	}
218 
219 	/**
220 	 * Marshal the given object to a {@link BytesMessage}.
221 	 * @param object the object to be marshalled
222 	 * @param session current JMS session
223 	 * @param marshaller the marshaller to use
224 	 * @return the resulting message
225 	 * @throws JMSException if thrown by JMS methods
226 	 * @throws IOException in case of I/O errors
227 	 * @throws XmlMappingException in case of OXM mapping errors
228 	 * @see Session#createBytesMessage
229 	 * @see Marshaller#marshal(Object, Result)
230 	 */
231 	protected BytesMessage marshalToBytesMessage(Object object, Session session, Marshaller marshaller)
232 			throws JMSException, IOException, XmlMappingException {
233 
234 		ByteArrayOutputStream bos = new ByteArrayOutputStream(1024);
235 		StreamResult streamResult = new StreamResult(bos);
236 		marshaller.marshal(object, streamResult);
237 		BytesMessage message = session.createBytesMessage();
238 		message.writeBytes(bos.toByteArray());
239 		return message;
240 	}
241 
242 	/**
243 	 * Template method that allows for custom message marshalling.
244 	 * Invoked when {@link #setTargetType} is not {@link MessageType#TEXT} or
245 	 * {@link MessageType#BYTES}.
246 	 * <p>The default implementation throws an {@link IllegalArgumentException}.
247 	 * @param object the object to marshal
248 	 * @param session the JMS session
249 	 * @param marshaller the marshaller to use
250 	 * @param targetType the target message type (other than TEXT or BYTES)
251 	 * @return the resulting message
252 	 * @throws JMSException if thrown by JMS methods
253 	 * @throws IOException in case of I/O errors
254 	 * @throws XmlMappingException in case of OXM mapping errors
255 	 */
256 	protected Message marshalToMessage(Object object, Session session, Marshaller marshaller, MessageType targetType)
257 			throws JMSException, IOException, XmlMappingException {
258 
259 		throw new IllegalArgumentException("Unsupported message type [" + targetType +
260 				"]. MarshallingMessageConverter by default only supports TextMessages and BytesMessages.");
261 	}
262 
263 
264 	/**
265 	 * Unmarshal the given {@link TextMessage} into an object.
266 	 * @param message the message
267 	 * @param unmarshaller the unmarshaller to use
268 	 * @return the unmarshalled object
269 	 * @throws JMSException if thrown by JMS methods
270 	 * @throws IOException in case of I/O errors
271 	 * @throws XmlMappingException in case of OXM mapping errors
272 	 * @see Unmarshaller#unmarshal(Source)
273 	 */
274 	protected Object unmarshalFromTextMessage(TextMessage message, Unmarshaller unmarshaller)
275 			throws JMSException, IOException, XmlMappingException {
276 
277 		Source source = new StreamSource(new StringReader(message.getText()));
278 		return unmarshaller.unmarshal(source);
279 	}
280 
281 	/**
282 	 * Unmarshal the given {@link BytesMessage} into an object.
283 	 * @param message the message
284 	 * @param unmarshaller the unmarshaller to use
285 	 * @return the unmarshalled object
286 	 * @throws JMSException if thrown by JMS methods
287 	 * @throws IOException in case of I/O errors
288 	 * @throws XmlMappingException in case of OXM mapping errors
289 	 * @see Unmarshaller#unmarshal(Source)
290 	 */
291 	protected Object unmarshalFromBytesMessage(BytesMessage message, Unmarshaller unmarshaller)
292 			throws JMSException, IOException, XmlMappingException {
293 
294 		byte[] bytes = new byte[(int) message.getBodyLength()];
295 		message.readBytes(bytes);
296 		ByteArrayInputStream bis = new ByteArrayInputStream(bytes);
297 		StreamSource source = new StreamSource(bis);
298 		return unmarshaller.unmarshal(source);
299 	}
300 
301 	/**
302 	 * Template method that allows for custom message unmarshalling.
303 	 * Invoked when {@link #fromMessage(Message)} is invoked with a message
304 	 * that is not a {@link TextMessage} or {@link BytesMessage}.
305 	 * <p>The default implementation throws an {@link IllegalArgumentException}.
306 	 * @param message the message
307 	 * @param unmarshaller the unmarshaller to use
308 	 * @return the unmarshalled object
309 	 * @throws JMSException if thrown by JMS methods
310 	 * @throws IOException in case of I/O errors
311 	 * @throws XmlMappingException in case of OXM mapping errors
312 	 */
313 	protected Object unmarshalFromMessage(Message message, Unmarshaller unmarshaller)
314 			throws JMSException, IOException, XmlMappingException {
315 
316 		throw new IllegalArgumentException("Unsupported message type [" + message.getClass() +
317 				"]. MarshallingMessageConverter by default only supports TextMessages and BytesMessages.");
318 	}
