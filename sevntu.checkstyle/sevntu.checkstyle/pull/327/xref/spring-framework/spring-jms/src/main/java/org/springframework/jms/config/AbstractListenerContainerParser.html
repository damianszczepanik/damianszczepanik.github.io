102 	@Override
103 	public BeanDefinition parse(Element element, ParserContext parserContext) {
104 		CompositeComponentDefinition compositeDef =
105 				new CompositeComponentDefinition(element.getTagName(), parserContext.extractSource(element));
106 		parserContext.pushContainingComponent(compositeDef);
107 
108 		PropertyValues commonProperties = parseCommonContainerProperties(element, parserContext);
109 		PropertyValues specificProperties = parseSpecificContainerProperties(element, parserContext);
110 
111 		String factoryId = element.getAttribute(FACTORY_ID_ATTRIBUTE);
112 		if (StringUtils.hasText(factoryId)) {
113 			RootBeanDefinition beanDefinition = createContainerFactory(
114 					factoryId, element, parserContext, commonProperties, specificProperties);
115 			if (beanDefinition != null) {
116 				beanDefinition.setSource(parserContext.extractSource(element));
117 				parserContext.registerBeanComponent(new BeanComponentDefinition(beanDefinition, factoryId));
118 			}
119 		}
120 
121 		NodeList childNodes = element.getChildNodes();
122 		for (int i = 0; i < childNodes.getLength(); i++) {
123 			Node child = childNodes.item(i);
124 			if (child.getNodeType() == Node.ELEMENT_NODE) {
125 				String localName = parserContext.getDelegate().getLocalName(child);
126 				if (LISTENER_ELEMENT.equals(localName)) {
127 					parseListener(element, (Element) child, parserContext, commonProperties, specificProperties);
128 				}
129 			}
130 		}
131 
132 		parserContext.popAndRegisterContainingComponent();
133 		return null;
134 	}

136 	private void parseListener(Element containerEle, Element listenerEle, ParserContext parserContext,
137 			PropertyValues commonContainerProperties, PropertyValues specificContainerProperties) {
138 
139 		RootBeanDefinition listenerDef = new RootBeanDefinition();
140 		listenerDef.setSource(parserContext.extractSource(listenerEle));
141 		listenerDef.setBeanClassName("org.springframework.jms.listener.adapter.MessageListenerAdapter");
142 
143 		String ref = listenerEle.getAttribute(REF_ATTRIBUTE);
144 		if (!StringUtils.hasText(ref)) {
145 			parserContext.getReaderContext().error(
146 					"Listener 'ref' attribute contains empty value.", listenerEle);
147 		}
148 		else {
149 			listenerDef.getPropertyValues().add("delegate", new RuntimeBeanReference(ref));
150 		}
151 
152 		String method = null;
153 		if (listenerEle.hasAttribute(METHOD_ATTRIBUTE)) {
154 			method = listenerEle.getAttribute(METHOD_ATTRIBUTE);
155 			if (!StringUtils.hasText(method)) {
156 				parserContext.getReaderContext().error(
157 						"Listener 'method' attribute contains empty value.", listenerEle);
158 			}
159 		}
160 		listenerDef.getPropertyValues().add("defaultListenerMethod", method);
161 
162 		PropertyValue messageConverterPv = commonContainerProperties.getPropertyValue("messageConverter");
163 		if (messageConverterPv != null) {
164 			listenerDef.getPropertyValues().addPropertyValue(messageConverterPv);
165 		}
166 
167 		BeanDefinition containerDef = createContainer(
168 				containerEle, listenerEle, parserContext, commonContainerProperties, specificContainerProperties);
169 		containerDef.getPropertyValues().add("messageListener", listenerDef);
170 
171 		if (listenerEle.hasAttribute(RESPONSE_DESTINATION_ATTRIBUTE)) {
172 			String responseDestination = listenerEle.getAttribute(RESPONSE_DESTINATION_ATTRIBUTE);
173 			Boolean pubSubDomain = (Boolean) commonContainerProperties.getPropertyValue("pubSubDomain").getValue();
174 			listenerDef.getPropertyValues().add(
175 					pubSubDomain ? "defaultResponseTopicName" : "defaultResponseQueueName", responseDestination);
176 			if (containerDef.getPropertyValues().contains("destinationResolver")) {
177 				listenerDef.getPropertyValues().add("destinationResolver",
178 						containerDef.getPropertyValues().getPropertyValue("destinationResolver").getValue());
179 			}
180 		}
181 
182 
183 		String containerBeanName = listenerEle.getAttribute(ID_ATTRIBUTE);
184 		// If no bean id is given auto generate one using the ReaderContext's BeanNameGenerator
185 		if (!StringUtils.hasText(containerBeanName)) {
186 			containerBeanName = parserContext.getReaderContext().generateBeanName(containerDef);
187 		}
188 
189 		// Register the listener and fire event
190 		parserContext.registerBeanComponent(new BeanComponentDefinition(containerDef, containerBeanName));
191 	}

193 	protected void parseListenerConfiguration(Element ele, ParserContext parserContext, MutablePropertyValues configValues) {
194 		String destination = ele.getAttribute(DESTINATION_ATTRIBUTE);
195 		if (!StringUtils.hasText(destination)) {
196 			parserContext.getReaderContext().error(
197 					"Listener 'destination' attribute contains empty value.", ele);
198 		}
199 		configValues.add("destinationName", destination);
200 
201 		if (ele.hasAttribute(SUBSCRIPTION_ATTRIBUTE)) {
202 			String subscription = ele.getAttribute(SUBSCRIPTION_ATTRIBUTE);
203 			if (!StringUtils.hasText(subscription)) {
204 				parserContext.getReaderContext().error(
205 						"Listener 'subscription' attribute contains empty value.", ele);
206 			}
207 			configValues.add("subscriptionName", subscription);
208 		}
209 
210 		if (ele.hasAttribute(SELECTOR_ATTRIBUTE)) {
211 			String selector = ele.getAttribute(SELECTOR_ATTRIBUTE);
212 			if (!StringUtils.hasText(selector)) {
213 				parserContext.getReaderContext().error(
214 						"Listener 'selector' attribute contains empty value.", ele);
215 			}
216 			configValues.add("messageSelector", selector);
217 		}
218 
219 		if (ele.hasAttribute(CONCURRENCY_ATTRIBUTE)) {
220 			String concurrency = ele.getAttribute(CONCURRENCY_ATTRIBUTE);
221 			if (!StringUtils.hasText(concurrency)) {
222 				parserContext.getReaderContext().error(
223 						"Listener 'concurrency' attribute contains empty value.", ele);
224 			}
225 			configValues.add("concurrency", concurrency);
226 		}
227 	}

229 	protected MutablePropertyValues parseCommonContainerProperties(Element containerEle, ParserContext parserContext) {
230 		MutablePropertyValues properties = new MutablePropertyValues();
231 
232 		String destinationType = containerEle.getAttribute(DESTINATION_TYPE_ATTRIBUTE);
233 		boolean pubSubDomain = false;
234 		boolean subscriptionDurable = false;
235 		boolean subscriptionShared = false;
236 		if (DESTINATION_TYPE_SHARED_DURABLE_TOPIC.equals(destinationType)) {
237 			pubSubDomain = true;
238 			subscriptionDurable = true;
239 			subscriptionShared = true;
240 		}
241 		else if (DESTINATION_TYPE_SHARED_TOPIC.equals(destinationType)) {
242 			pubSubDomain = true;
243 			subscriptionShared = true;
244 		}
245 		else if (DESTINATION_TYPE_DURABLE_TOPIC.equals(destinationType)) {
246 			pubSubDomain = true;
247 			subscriptionDurable = true;
248 		}
249 		else if (DESTINATION_TYPE_TOPIC.equals(destinationType)) {
250 			pubSubDomain = true;
251 		}
252 		else if ("".equals(destinationType) || DESTINATION_TYPE_QUEUE.equals(destinationType)) {
253 			// the default: queue
254 		}
255 		else {
256 			parserContext.getReaderContext().error("Invalid listener container 'destination-type': only " +
257 					"\"queue\", \"topic\", \"durableTopic\", \"sharedTopic\", \"sharedDurableTopic\" supported.", containerEle);
258 		}
259 		properties.add("pubSubDomain", pubSubDomain);
260 		properties.add("subscriptionDurable", subscriptionDurable);
261 		properties.add("subscriptionShared", subscriptionShared);
262 
263 		if (containerEle.hasAttribute(CLIENT_ID_ATTRIBUTE)) {
264 			String clientId = containerEle.getAttribute(CLIENT_ID_ATTRIBUTE);
265 			if (!StringUtils.hasText(clientId)) {
266 				parserContext.getReaderContext().error(
267 						"Listener 'client-id' attribute contains empty value.", containerEle);
268 			}
269 			properties.add("clientId", clientId);
270 		}
271 
272 		if (containerEle.hasAttribute(MESSAGE_CONVERTER_ATTRIBUTE)) {
273 			String messageConverter = containerEle.getAttribute(MESSAGE_CONVERTER_ATTRIBUTE);
274 			if (!StringUtils.hasText(messageConverter)) {
275 				parserContext.getReaderContext().error(
276 						"listener container 'message-converter' attribute contains empty value.", containerEle);
277 			}
278 			else {
279 				properties.add("messageConverter", new RuntimeBeanReference(messageConverter));
280 			}
281 		}
282 
283 		return properties;
284 	}

307 	protected Integer parseAcknowledgeMode(Element ele, ParserContext parserContext) {
308 		String acknowledge = ele.getAttribute(ACKNOWLEDGE_ATTRIBUTE);
309 		if (StringUtils.hasText(acknowledge)) {
310 			int acknowledgeMode = Session.AUTO_ACKNOWLEDGE;
311 			if (ACKNOWLEDGE_TRANSACTED.equals(acknowledge)) {
312 				acknowledgeMode = Session.SESSION_TRANSACTED;
313 			}
314 			else if (ACKNOWLEDGE_DUPS_OK.equals(acknowledge)) {
315 				acknowledgeMode = Session.DUPS_OK_ACKNOWLEDGE;
316 			}
317 			else if (ACKNOWLEDGE_CLIENT.equals(acknowledge)) {
318 				acknowledgeMode = Session.CLIENT_ACKNOWLEDGE;
319 			}
320 			else if (!ACKNOWLEDGE_AUTO.equals(acknowledge)) {
321 				parserContext.getReaderContext().error("Invalid listener container 'acknowledge' setting [" +
322 						acknowledge + "]: only \"auto\", \"client\", \"dups-ok\" and \"transacted\" supported.", ele);
323 			}
324 			return acknowledgeMode;
325 		}
326 		else {
327 			return null;
328 		}
329 	}
