97  	@Test
98  	public void testMethodThrowingException_SPR6760() {
99  		// Test method on inventor: throwException()
100 		// On 1 it will throw an IllegalArgumentException
101 		// On 2 it will throw a RuntimeException
102 		// On 3 it will exit normally
103 		// In each case it increments the Inventor field 'counter' when invoked
104 
105 		SpelExpressionParser parser = new SpelExpressionParser();
106 		Expression expr = parser.parseExpression("throwException(#bar)");
107 
108 		// Normal exit
109 		StandardEvaluationContext eContext = TestScenarioCreator.getTestEvaluationContext();
110 		eContext.setVariable("bar",3);
111 		Object o = expr.getValue(eContext);
112 		assertEquals(o, 3);
113 		assertEquals(1, parser.parseExpression("counter").getValue(eContext));
114 
115 		// Now the expression has cached that throwException(int) is the right thing to call
116 		// Let's change 'bar' to be a PlaceOfBirth which indicates the cached reference is
117 		// out of date.
118 		eContext.setVariable("bar",new PlaceOfBirth("London"));
119 		o = expr.getValue(eContext);
120 		assertEquals("London", o);
121 		// That confirms the logic to mark the cached reference stale and retry is working
122 
123 
124 		// Now let's cause the method to exit via exception and ensure it doesn't cause
125 		// a retry.
126 
127 		// First, switch back to throwException(int)
128 		eContext.setVariable("bar",3);
129 		o = expr.getValue(eContext);
130 		assertEquals(3, o);
131 		assertEquals(2, parser.parseExpression("counter").getValue(eContext));
132 
133 
134 		// Now cause it to throw an exception:
135 		eContext.setVariable("bar",1);
136 		try {
137 			o = expr.getValue(eContext);
138 			fail();
139 		} catch (Exception e) {
140 			if (e instanceof SpelEvaluationException) {
141 				e.printStackTrace();
142 				fail("Should not be a SpelEvaluationException");
143 			}
144 			// normal
145 		}
146 		// If counter is 4 then the method got called twice!
147 		assertEquals(3, parser.parseExpression("counter").getValue(eContext));
148 
149 		eContext.setVariable("bar",4);
150 		try {
151 			o = expr.getValue(eContext);
152 			fail();
153 		} catch (Exception e) {
154 			// 4 means it will throw a checked exception - this will be wrapped
155 			if (!(e instanceof ExpressionInvocationTargetException)) {
156 				e.printStackTrace();
157 				fail("Should have been wrapped");
158 			}
159 			// normal
160 		}
161 		// If counter is 5 then the method got called twice!
162 		assertEquals(4, parser.parseExpression("counter").getValue(eContext));
163 	}

217 	@Test
218 	public void testMethodFiltering_SPR6764() {
219 		SpelExpressionParser parser = new SpelExpressionParser();
220 		StandardEvaluationContext context = new StandardEvaluationContext();
221 		context.setRootObject(new TestObject());
222 		LocalFilter filter = new LocalFilter();
223 		context.registerMethodFilter(TestObject.class,filter);
224 
225 		// Filter will be called but not do anything, so first doit() will be invoked
226 		SpelExpression expr = (SpelExpression) parser.parseExpression("doit(1)");
227 		String result = expr.getValue(context,String.class);
228 		assertEquals("1", result);
229 		assertTrue(filter.filterCalled);
230 
231 		// Filter will now remove non @Anno annotated methods
232 		filter.removeIfNotAnnotated = true;
233 		filter.filterCalled = false;
234 		expr = (SpelExpression) parser.parseExpression("doit(1)");
235 		result = expr.getValue(context,String.class);
236 		assertEquals("double 1.0", result);
237 		assertTrue(filter.filterCalled);
238 
239 		// check not called for other types
240 		filter.filterCalled=false;
241 		context.setRootObject(new String("abc"));
242 		expr = (SpelExpression) parser.parseExpression("charAt(0)");
243 		result = expr.getValue(context,String.class);
244 		assertEquals("a", result);
245 		assertFalse(filter.filterCalled);
246 
247 		// check de-registration works
248 		filter.filterCalled = false;
249 		context.registerMethodFilter(TestObject.class,null);//clear filter
250 		context.setRootObject(new TestObject());
251 		expr = (SpelExpression) parser.parseExpression("doit(1)");
252 		result = expr.getValue(context,String.class);
253 		assertEquals("1", result);
254 		assertFalse(filter.filterCalled);
255 	}

257 	// Simple filter
258 	static class LocalFilter implements MethodFilter {
259 
260 		public boolean removeIfNotAnnotated = false;
261 
262 		public boolean filterCalled = false;
263 
264 		private boolean isAnnotated(Method m) {
265 			Annotation[] annos = m.getAnnotations();
266 			if (annos==null) {
267 				return false;
268 			}
269 			for (Annotation anno: annos) {
270 				String s = anno.annotationType().getName();
271 				if (s.endsWith("Anno")) {
272 					return true;
273 				}
274 			}
275 			return false;
276 		}
277 
278 		@Override
279 		public List<Method> filter(List<Method> methods) {
280 			filterCalled = true;
281 			List<Method> forRemoval = new ArrayList<Method>();
282 			for (Method m: methods) {
283 				if (removeIfNotAnnotated && !isAnnotated(m)) {
284 					forRemoval.add(m);
285 				}
286 			}
287 			for (Method m: forRemoval) {
288 				methods.remove(m);
289 			}
290 			return methods;
291 		}

311 	public void testAddingMethodResolvers() {
312 		StandardEvaluationContext ctx = new StandardEvaluationContext();
313 
314 		// reflective method accessor is the only one by default
315 		List<MethodResolver> methodResolvers = ctx.getMethodResolvers();
316 		assertEquals(1, methodResolvers.size());
317 
318 		MethodResolver dummy = new DummyMethodResolver();
319 		ctx.addMethodResolver(dummy);
320 		assertEquals(2, ctx.getMethodResolvers().size());
321 
322 		List<MethodResolver> copy = new ArrayList<MethodResolver>();
323 		copy.addAll(ctx.getMethodResolvers());
324 		assertTrue(ctx.removeMethodResolver(dummy));
325 		assertFalse(ctx.removeMethodResolver(dummy));
326 		assertEquals(1, ctx.getMethodResolvers().size());
327 
328 		ctx.setMethodResolvers(copy);
329 		assertEquals(2, ctx.getMethodResolvers().size());
330 	}

380 	public void invokeMethodWithoutConversion() throws Exception {
381 		final BytesService service = new BytesService();
382 		byte[] bytes = new byte[100];
383 		StandardEvaluationContext context = new StandardEvaluationContext(bytes);
384 		context.setBeanResolver(new BeanResolver() {
385 			@Override
386 			public Object resolve(EvaluationContext context, String beanName)
387 					throws AccessException {
388 				if ("service".equals(beanName)) {
389 					return service;
390 				}
391 				return null;
392 			}
393 		});
394 		Expression expression = parser.parseExpression("@service.handleBytes(#root)");
395 		byte[] outBytes = expression.getValue(context, byte[].class);
396 		assertSame(bytes, outBytes);
397 	}
