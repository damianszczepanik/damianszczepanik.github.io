142 	@Test
143 	public void testSelectionAST() throws Exception {
144 		SpelExpression expr = (SpelExpression) parser.parseExpression("'abc'.^[true]");
145 		assertEquals("'abc'.^[true]", expr.toStringAST());
146 		expr = (SpelExpression) parser.parseExpression("'abc'.?[true]");
147 		assertEquals("'abc'.?[true]", expr.toStringAST());
148 		expr = (SpelExpression) parser.parseExpression("'abc'.$[true]");
149 		assertEquals("'abc'.$[true]", expr.toStringAST());
150 	}
