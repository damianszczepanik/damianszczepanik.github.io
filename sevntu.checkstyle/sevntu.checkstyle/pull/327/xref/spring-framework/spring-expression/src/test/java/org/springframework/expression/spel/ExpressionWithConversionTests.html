72  	public void testConversionsAvailable() throws Exception {
73  		TypeConvertorUsingConversionService tcs = new TypeConvertorUsingConversionService();
74  
75  		// ArrayList containing List<Integer> to List<String>
76  		Class<?> clazz = typeDescriptorForListOfString.getElementTypeDescriptor().getType();
77  		assertEquals(String.class,clazz);
78  		List<?> l = (List<?>) tcs.convertValue(listOfInteger, TypeDescriptor.forObject(listOfInteger), typeDescriptorForListOfString);
79  		assertNotNull(l);
80  
81  		// ArrayList containing List<String> to List<Integer>
82  		clazz = typeDescriptorForListOfInteger.getElementTypeDescriptor().getType();
83  		assertEquals(Integer.class,clazz);
84  
85  		l = (List<?>) tcs.convertValue(listOfString, TypeDescriptor.forObject(listOfString), typeDescriptorForListOfString);
86  		assertNotNull(l);
87  	}

89  	@Test
90  	public void testSetParameterizedList() throws Exception {
91  		StandardEvaluationContext context = TestScenarioCreator.getTestEvaluationContext();
92  		Expression e = parser.parseExpression("listOfInteger.size()");
93  		assertEquals(0,e.getValue(context,Integer.class).intValue());
94  		context.setTypeConverter(new TypeConvertorUsingConversionService());
95  		// Assign a List<String> to the List<Integer> field - the component elements should be converted
96  		parser.parseExpression("listOfInteger").setValue(context,listOfString);
97  		assertEquals(3,e.getValue(context,Integer.class).intValue()); // size now 3
98  		Class<?> clazz = parser.parseExpression("listOfInteger[1].getClass()").getValue(context, Class.class); // element type correctly Integer
99  		assertEquals(Integer.class,clazz);
100 	}

102 	@Test
103 	public void testCoercionToCollectionOfPrimitive() throws Exception {
104 
105 		class TestTarget {
106 			@SuppressWarnings("unused")
107 			public int sum(Collection<Integer> numbers) {
108 				int total = 0;
109 				for (int i : numbers) {
110 					total += i;
111 				}
112 				return total;
113 			}
114 		}
115 
116 		StandardEvaluationContext evaluationContext = new StandardEvaluationContext();
117 
118 		TypeDescriptor collectionType = new TypeDescriptor(new MethodParameter(TestTarget.class.getDeclaredMethod(
119 				"sum", Collection.class), 0));
120 		// The type conversion is possible
121 		assertTrue(evaluationContext.getTypeConverter()
122 				.canConvert(TypeDescriptor.valueOf(String.class), collectionType));
123 		// ... and it can be done successfully
124 		assertEquals("[1, 2, 3, 4]", evaluationContext.getTypeConverter().convertValue("1,2,3,4", TypeDescriptor.valueOf(String.class), collectionType).toString());
125 
126 		evaluationContext.setVariable("target", new TestTarget());
127 
128 		// OK up to here, so the evaluation should be fine...
129 		// ... but this fails
130 		int result = (Integer) parser.parseExpression("#target.sum(#root)").getValue(evaluationContext, "1,2,3,4");
131 		assertEquals("Wrong result: " + result, 10, result);
132 
133 	}

135 	@Test
136 	public void testConvert() {
137 		Foo root = new Foo("bar");
138 		Collection<String> foos = Collections.singletonList("baz");
139 
140 		StandardEvaluationContext context = new StandardEvaluationContext(root);
141 
142 		// property access
143 		Expression expression = parser.parseExpression("foos");
144 		expression.setValue(context, foos);
145 		Foo baz = root.getFoos().iterator().next();
146 		assertEquals("baz", baz.value);
147 
148 		// method call
149 		expression = parser.parseExpression("setFoos(#foos)");
150 		context.setVariable("foos", foos);
151 		expression.getValue(context);
152 		baz = root.getFoos().iterator().next();
153 		assertEquals("baz", baz.value);
154 
155 		// method call with result from method call
156 		expression = parser.parseExpression("setFoos(getFoosAsStrings())");
157 		expression.getValue(context);
158 		baz = root.getFoos().iterator().next();
159 		assertEquals("baz", baz.value);
160 
161 		// method call with result from method call
162 		expression = parser.parseExpression("setFoos(getFoosAsObjects())");
163 		expression.getValue(context);
164 		baz = root.getFoos().iterator().next();
165 		assertEquals("baz", baz.value);
166 	}

169 	/**
170 	 * Type converter that uses the core conversion service.
171 	 */
172 	private static class TypeConvertorUsingConversionService implements TypeConverter {
173 
174 		private final ConversionService service = new DefaultConversionService();
175 
176 		@Override
177 		public boolean canConvert(TypeDescriptor sourceType, TypeDescriptor targetType) {
178 			return this.service.canConvert(sourceType, targetType);
179 		}
180 
181 		@Override
182 		public Object convertValue(Object value, TypeDescriptor sourceType, TypeDescriptor targetType) throws EvaluationException {
183 			return this.service.convert(value, sourceType, targetType);
184 		}
185 	}
186

188 	public static class Foo {
189 
190 		public final String value;
191 
192 		private Collection<Foo> foos;
193 
194 		public Foo(String value) {
195 			this.value = value;
196 		}
197 
198 		public void setFoos(Collection<Foo> foos) {
199 			this.foos = foos;
200 		}
201 
202 		public Collection<Foo> getFoos() {
203 			return this.foos;
204 		}
205 
206 		public Collection<String> getFoosAsStrings() {
207 			return Collections.singletonList("baz");
208 		}
209 
210 		public Collection<?> getFoosAsObjects() {
211 			return Collections.singletonList("baz");
212 		}
213 	}
