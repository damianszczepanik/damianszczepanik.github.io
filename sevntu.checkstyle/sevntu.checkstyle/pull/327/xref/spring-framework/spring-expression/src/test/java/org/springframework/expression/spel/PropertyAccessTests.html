70  	/**
71  	 * The standard reflection resolver cannot find properties on null objects but some
72  	 * supplied resolver might be able to - so null shouldn't crash the reflection resolver.
73  	 */
74  	@Test
75  	public void testAccessingOnNullObject() throws Exception {
76  		SpelExpression expr = (SpelExpression)parser.parseExpression("madeup");
77  		EvaluationContext context = new StandardEvaluationContext(null);
78  		try {
79  			expr.getValue(context);
80  			fail("Should have failed - default property resolver cannot resolve on null");
81  		} catch (Exception e) {
82  			checkException(e,SpelMessage.PROPERTY_OR_FIELD_NOT_READABLE_ON_NULL);
83  		}
84  		assertFalse(expr.isWritable(context));
85  		try {
86  			expr.setValue(context,"abc");
87  			fail("Should have failed - default property resolver cannot resolve on null");
88  		} catch (Exception e) {
89  			checkException(e,SpelMessage.PROPERTY_OR_FIELD_NOT_WRITABLE_ON_NULL);
90  		}
91  	}

93  	private void checkException(Exception e, SpelMessage expectedMessage) {
94  		if (e instanceof SpelEvaluationException) {
95  			SpelMessage sm = ((SpelEvaluationException)e).getMessageCode();
96  			assertEquals("Expected exception type did not occur",expectedMessage,sm);
97  		} else {
98  			fail("Should be a SpelException "+e);
99  		}
100 	}

102 	@Test
103 	// Adding a new property accessor just for a particular type
104 	public void testAddingSpecificPropertyAccessor() throws Exception {
105 		SpelExpressionParser parser = new SpelExpressionParser();
106 		StandardEvaluationContext ctx = new StandardEvaluationContext();
107 
108 		// Even though this property accessor is added after the reflection one, it specifically
109 		// names the String class as the type it is interested in so is chosen in preference to
110 		// any 'default' ones
111 		ctx.addPropertyAccessor(new StringyPropertyAccessor());
112 		Expression expr = parser.parseRaw("new String('hello').flibbles");
113 		Integer i = expr.getValue(ctx, Integer.class);
114 		assertEquals((int) i, 7);
115 
116 		// The reflection one will be used for other properties...
117 		expr = parser.parseRaw("new String('hello').CASE_INSENSITIVE_ORDER");
118 		Object o = expr.getValue(ctx);
119 		assertNotNull(o);
120 
121 		expr = parser.parseRaw("new String('hello').flibbles");
122 		expr.setValue(ctx, 99);
123 		i = expr.getValue(ctx, Integer.class);
124 		assertEquals((int) i, 99);
125 
126 		// Cannot set it to a string value
127 		try {
128 			expr.setValue(ctx, "not allowed");
129 			fail("Should not have been allowed");
130 		} catch (EvaluationException e) {
131 			// success - message will be: EL1063E:(pos 20): A problem occurred whilst attempting to set the property
132 			// 'flibbles': 'Cannot set flibbles to an object of type 'class java.lang.String''
133 			// System.out.println(e.getMessage());
134 		}
135 	}

137 	@Test
138 	public void testAddingRemovingAccessors() {
139 		StandardEvaluationContext ctx = new StandardEvaluationContext();
140 
141 		// reflective property accessor is the only one by default
142 		List<PropertyAccessor> propertyAccessors = ctx.getPropertyAccessors();
143 		assertEquals(1,propertyAccessors.size());
144 
145 		StringyPropertyAccessor spa = new StringyPropertyAccessor();
146 		ctx.addPropertyAccessor(spa);
147 		assertEquals(2,ctx.getPropertyAccessors().size());
148 
149 		List<PropertyAccessor> copy = new ArrayList<PropertyAccessor>();
150 		copy.addAll(ctx.getPropertyAccessors());
151 		assertTrue(ctx.removePropertyAccessor(spa));
152 		assertFalse(ctx.removePropertyAccessor(spa));
153 		assertEquals(1,ctx.getPropertyAccessors().size());
154 
155 		ctx.setPropertyAccessors(copy);
156 		assertEquals(2,ctx.getPropertyAccessors().size());
157 	}

167 	// This can resolve the property 'flibbles' on any String (very useful...)
168 	private static class StringyPropertyAccessor implements PropertyAccessor {
169 
170 		int flibbles = 7;
171 
172 		@Override
173 		public Class<?>[] getSpecificTargetClasses() {
174 			return new Class[] { String.class };
175 		}
176 
177 		@Override
178 		public boolean canRead(EvaluationContext context, Object target, String name) throws AccessException {
179 			if (!(target instanceof String))
180 				throw new RuntimeException("Assertion Failed! target should be String");
181 			return (name.equals("flibbles"));
182 		}
183 
184 		@Override
185 		public boolean canWrite(EvaluationContext context, Object target, String name) throws AccessException {
186 			if (!(target instanceof String))
187 				throw new RuntimeException("Assertion Failed! target should be String");
188 			return (name.equals("flibbles"));
189 		}
190 
191 		@Override
192 		public TypedValue read(EvaluationContext context, Object target, String name) throws AccessException {
193 			if (!name.equals("flibbles"))
194 				throw new RuntimeException("Assertion Failed! name should be flibbles");
195 			return new TypedValue(flibbles);
196 		}
197 
198 		@Override
199 		public void write(EvaluationContext context, Object target, String name, Object newValue)
200 				throws AccessException {
201 			if (!name.equals("flibbles"))
202 				throw new RuntimeException("Assertion Failed! name should be flibbles");
203 			try {
204 				flibbles = (Integer) context.getTypeConverter().convertValue(newValue, TypeDescriptor.forObject(newValue), TypeDescriptor.valueOf(Integer.class));
205 			}catch (EvaluationException e) {
206 				throw new AccessException("Cannot set flibbles to an object of type '" + newValue.getClass() + "'");
207 			}
208 		}
209 	}
