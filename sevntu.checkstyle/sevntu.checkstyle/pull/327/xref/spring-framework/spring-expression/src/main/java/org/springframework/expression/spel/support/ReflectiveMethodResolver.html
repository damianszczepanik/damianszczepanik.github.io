45  /**
46   * Reflection-based {@link MethodResolver} used by default in {@link StandardEvaluationContext}
47   * unless explicit method resolvers have been specified.
48   *
49   * @author Andy Clement
50   * @author Juergen Hoeller
51   * @author Chris Beams
52   * @since 3.0
53   * @see StandardEvaluationContext#addMethodResolver(MethodResolver)
54   */
55  public class ReflectiveMethodResolver implements MethodResolver {
56  
57  	// Using distance will ensure a more accurate match is discovered,
58  	// more closely following the Java rules.
59  	private final boolean useDistance;
60  
61  	private Map<Class<?>, MethodFilter> filters;
62  
63  
64  	public ReflectiveMethodResolver() {
65  		this.useDistance = false;
66  	}
67  
68  	/**
69  	 * This constructors allows the ReflectiveMethodResolver to be configured such that it will
70  	 * use a distance computation to check which is the better of two close matches (when there
71  	 * are multiple matches). Using the distance computation is intended to ensure matches
72  	 * are more closely representative of what a Java compiler would do when taking into
73  	 * account boxing/unboxing and whether the method candidates are declared to handle a
74  	 * supertype of the type (of the argument) being passed in.
75  	 * @param useDistance true if distance computation should be used when calculating matches
76  	 */
77  	public ReflectiveMethodResolver(boolean useDistance) {
78  		this.useDistance = useDistance;
79  	}
80  
81  
82  	public void registerMethodFilter(Class<?> type, MethodFilter filter) {
83  		if (this.filters == null) {
84  			this.filters = new HashMap<Class<?>, MethodFilter>();
85  		}
86  		if (filter != null) {
87  			this.filters.put(type, filter);
88  		}
89  		else {
90  			this.filters.remove(type);
91  		}
92  	}
93  
94  
95  	/**
96  	 * Locate a method on a type. There are three kinds of match that might occur:
97  	 * <ol>
98  	 * <li>an exact match where the types of the arguments match the types of the constructor
99  	 * <li>an in-exact match where the types we are looking for are subtypes of those defined on the constructor
100 	 * <li>a match where we are able to convert the arguments into those expected by the constructor,
101 	 * according to the registered type converter
102 	 * </ol>
103 	 */
104 	@Override
105 	public MethodExecutor resolve(EvaluationContext context, Object targetObject, String name,
106 			List<TypeDescriptor> argumentTypes) throws AccessException {
107 
108 		try {
109 			TypeConverter typeConverter = context.getTypeConverter();
110 			Class<?> type = (targetObject instanceof Class ? (Class<?>) targetObject : targetObject.getClass());
111 			List<Method> methods = new ArrayList<Method>(getMethods(type, targetObject));
112 
113 			// If a filter is registered for this type, call it
114 			MethodFilter filter = (this.filters != null ? this.filters.get(type) : null);
115 			if (filter != null) {
116 				List<Method> filtered = filter.filter(methods);
117 				methods = (filtered instanceof ArrayList ? filtered : new ArrayList<Method>(filtered));
118 			}
119 
120 			// Sort methods into a sensible order
121 			if (methods.size() > 1) {
122 				Collections.sort(methods, new Comparator<Method>() {
123 					@Override
124 					public int compare(Method m1, Method m2) {
125 						int m1pl = m1.getParameterTypes().length;
126 						int m2pl = m2.getParameterTypes().length;
127 						// varargs methods go last
128 						if (m1pl == m2pl) {
129 						    if (!m1.isVarArgs() && m2.isVarArgs()) {
130 						    	return -1;
131 						    }
132 						    else if (m1.isVarArgs() && !m2.isVarArgs()) {
133 						    	return 1;
134 						    }
135 						    else {
136 						    	return 0;
137 						    }
138 						}
139 						return (m1pl < m2pl ? -1 : (m1pl > m2pl ? 1 : 0));
140 					}
141 				});
142 			}
143 
144 			// Resolve any bridge methods
145 			for (int i = 0; i < methods.size(); i++) {
146 				methods.set(i, BridgeMethodResolver.findBridgedMethod(methods.get(i)));
147 			}
148 
149 			// Remove duplicate methods (possible due to resolved bridge methods)
150 			Set<Method> methodsToIterate = new LinkedHashSet<Method>(methods);
151 
152 			Method closeMatch = null;
153 			int closeMatchDistance = Integer.MAX_VALUE;
154 			Method matchRequiringConversion = null;
155 			boolean multipleOptions = false;
156 
157 			for (Method method : methodsToIterate) {
158 				if (method.getName().equals(name)) {
159 					Class<?>[] paramTypes = method.getParameterTypes();
160 					List<TypeDescriptor> paramDescriptors = new ArrayList<TypeDescriptor>(paramTypes.length);
161 					for (int i = 0; i < paramTypes.length; i++) {
162 						paramDescriptors.add(new TypeDescriptor(new MethodParameter(method, i)));
163 					}
164 					ReflectionHelper.ArgumentsMatchInfo matchInfo = null;
165 					if (method.isVarArgs() && argumentTypes.size() >= (paramTypes.length - 1)) {
166 						// *sigh* complicated
167 						matchInfo = ReflectionHelper.compareArgumentsVarargs(paramDescriptors, argumentTypes, typeConverter);
168 					}
169 					else if (paramTypes.length == argumentTypes.size()) {
170 						// Name and parameter number match, check the arguments
171 						matchInfo = ReflectionHelper.compareArguments(paramDescriptors, argumentTypes, typeConverter);
172 					}
173 					if (matchInfo != null) {
174 						if (matchInfo.isExactMatch()) {
175 							return new ReflectiveMethodExecutor(method);
176 						}
177 						else if (matchInfo.isCloseMatch()) {
178 							if (!this.useDistance) {
179 								// Take this as a close match if there isn't one already
180 								if (closeMatch == null) {
181 									closeMatch = method;
182 								}
183 							}
184 							else {
185 								int matchDistance = ReflectionHelper.getTypeDifferenceWeight(paramDescriptors, argumentTypes);
186 								if (matchDistance < closeMatchDistance) {
187 									// This is a better match...
188 									closeMatchDistance = matchDistance;
189 									closeMatch = method;
190 								}
191 							}
192 						}
193 						else if (matchInfo.isMatchRequiringConversion()) {
194 							if (matchRequiringConversion != null) {
195 								multipleOptions = true;
196 							}
197 							matchRequiringConversion = method;
198 						}
199 					}
200 				}
201 			}
202 			if (closeMatch != null) {
203 				return new ReflectiveMethodExecutor(closeMatch);
204 			}
205 			else if (matchRequiringConversion != null) {
206 				if (multipleOptions) {
207 					throw new SpelEvaluationException(SpelMessage.MULTIPLE_POSSIBLE_METHODS, name);
208 				}
209 				return new ReflectiveMethodExecutor(matchRequiringConversion);
210 			}
211 			else {
212 				return null;
213 			}
214 		}
215 		catch (EvaluationException ex) {
216 			throw new AccessException("Failed to resolve method", ex);
217 		}
218 	}

220 	private Collection<Method> getMethods(Class<?> type, Object targetObject) {
221 		if (targetObject instanceof Class) {
222 			Set<Method> result = new LinkedHashSet<Method>();
223 			result.addAll(Arrays.asList(getMethods(targetObject.getClass())));
224 			// Add these also so that static result are invocable on the type: e.g. Float.valueOf(..)
225 			Method[] methods = getMethods(type);
226 			for (Method method : methods) {
227 				if (Modifier.isStatic(method.getModifiers())) {
228 					result.add(method);
229 				}
230 			}
231 			return result;
232 		}
233 		else {
234 			return Arrays.asList(getMethods(type));
235 		}
236 	}
237 
238 	/**
239 	 * Return the set of methods for this type. The default implementation returns the
240 	 * result of {@link Class#getMethods()} for the given {@code type}, but subclasses
241 	 * may override in order to alter the results, e.g. specifying static methods
242 	 * declared elsewhere.
243 	 * @param type the class for which to return the methods
244 	 * @since 3.1.1
245 	 */
246 	protected Method[] getMethods(Class<?> type) {
247 		return type.getMethods();
248 	}
