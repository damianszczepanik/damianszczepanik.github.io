17  package org.springframework.expression.spel;
18  
19  import org.junit.Test;
20  
21  import org.springframework.expression.spel.standard.SpelExpressionParser;
22  import org.springframework.expression.spel.support.StandardEvaluationContext;
23  
24  import static org.junit.Assert.*;
25  
26  /**
27   * Tests the evaluation of expressions that access variables and functions (lambda/java).
28   *
29   * @author Andy Clement
30   */
31  public class VariableAndFunctionTests extends AbstractExpressionTests {
32  
33  	@Test
34  	public void testVariableAccess01() {
35  		evaluate("#answer", "42", Integer.class, SHOULD_BE_WRITABLE);
36  		evaluate("#answer / 2", 21, Integer.class, SHOULD_NOT_BE_WRITABLE);
37  	}
38  
39  	@Test
40  	public void testVariableAccess_WellKnownVariables() {
41  		evaluate("#this.getName()","Nikola Tesla",String.class);
42  		evaluate("#root.getName()","Nikola Tesla",String.class);
43  	}
44  
45  	@Test
46  	public void testFunctionAccess01() {
47  		evaluate("#reverseInt(1,2,3)", "int[3]{3,2,1}", int[].class);
48  		evaluate("#reverseInt('1',2,3)", "int[3]{3,2,1}", int[].class); // requires type conversion of '1' to 1
49  		evaluateAndCheckError("#reverseInt(1)", SpelMessage.INCORRECT_NUMBER_OF_ARGUMENTS_TO_FUNCTION, 0, 1, 3);
50  	}
51  
52  	@Test
53  	public void testFunctionAccess02() {
54  		evaluate("#reverseString('hello')", "olleh", String.class);
55  		evaluate("#reverseString(37)", "73", String.class); // requires type conversion of 37 to '37'
56  	}
57  
58  	@Test
59  	public void testCallVarargsFunction() {
60  		evaluate("#varargsFunctionReverseStringsAndMerge('a','b','c')", "cba", String.class);
61  		evaluate("#varargsFunctionReverseStringsAndMerge('a')", "a", String.class);
62  		evaluate("#varargsFunctionReverseStringsAndMerge()", "", String.class);
63  		evaluate("#varargsFunctionReverseStringsAndMerge('b',25)", "25b", String.class);
64  		evaluate("#varargsFunctionReverseStringsAndMerge(25)", "25", String.class);
65  		evaluate("#varargsFunctionReverseStringsAndMerge2(1,'a','b','c')", "1cba", String.class);
66  		evaluate("#varargsFunctionReverseStringsAndMerge2(2,'a')", "2a", String.class);
67  		evaluate("#varargsFunctionReverseStringsAndMerge2(3)", "3", String.class);
68  		evaluate("#varargsFunctionReverseStringsAndMerge2(4,'b',25)", "425b", String.class);
69  		evaluate("#varargsFunctionReverseStringsAndMerge2(5,25)", "525", String.class);
70  	}
71  
72  	@Test
73  	public void testCallingIllegalFunctions() throws Exception {
74  		SpelExpressionParser parser = new SpelExpressionParser();
75  		StandardEvaluationContext ctx = new StandardEvaluationContext();
76  		ctx.setVariable("notStatic", this.getClass().getMethod("nonStatic"));
77  		try {
78  			@SuppressWarnings("unused")
79  			Object v = parser.parseRaw("#notStatic()").getValue(ctx);
80  			fail("Should have failed with exception - cannot call non static method that way");
81  		} catch (SpelEvaluationException se) {
82  			if (se.getMessageCode() != SpelMessage.FUNCTION_MUST_BE_STATIC) {
83  				se.printStackTrace();
84  				fail("Should have failed a message about the function needing to be static, not: "
85  						+ se.getMessageCode());
86  			}
87  		}
88  	}
89  	// this method is used by the test above
90  	public void nonStatic() {
91  	}
92  
93  }
