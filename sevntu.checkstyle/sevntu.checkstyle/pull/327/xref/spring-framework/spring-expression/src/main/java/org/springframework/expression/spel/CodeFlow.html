32   * intermediate compilation state as the bytecode is generated. It also includes
33   * various bytecode generation helper functions.

102 	/**
103 	 * Record the descriptor for the most recently evaluated expression element.
104 	 * @param descriptor type descriptor for most recently evaluated element
105 	 */
106 	public void pushDescriptor(String descriptor) {
107 		Assert.notNull(descriptor, "Descriptor must not be null");
108 		this.compilationScopes.peek().add(descriptor);
109 	}

111 	/**
112 	 * Enter a new compilation scope, usually due to nested expression evaluation. For
113 	 * example when the arguments for a method invocation expression are being evaluated,
114 	 * each argument will be evaluated in a new scope.
115 	 */
116 	public void enterCompilationScope() {
117 		this.compilationScopes.push(new ArrayList<String>());
118 	}
119 
120 	/**
121 	 * Exit a compilation scope, usually after a nested expression has been evaluated. For
122 	 * example after an argument for a method invocation has been evaluated this method
123 	 * returns us to the previous (outer) scope.
124 	 */
125 	public void exitCompilationScope() {
126 		this.compilationScopes.pop();
127 	}

129 	/**
130 	 * Return the descriptor for the item currently on top of the stack (in the current scope).
131 	 */
132 	public String lastDescriptor() {
133 		if (this.compilationScopes.peek().isEmpty()) {
134 			return null;
135 		}
136 		return this.compilationScopes.peek().get(this.compilationScopes.peek().size() - 1);
137 	}

150 	/**
151 	 * Insert any necessary cast and value call to convert from a boxed type to a
152 	 * primitive value
153 	 * @param mv the method visitor into which instructions should be inserted
154 	 * @param ch the primitive type desired as output
155 	 * @param stackDescriptor the descriptor of the type on top of the stack
156 	 */
157 	public static void insertUnboxInsns(MethodVisitor mv, char ch, String stackDescriptor) {
158 		switch (ch) {
159 			case 'Z':
160 				if (!stackDescriptor.equals("Ljava/lang/Boolean")) {
161 					mv.visitTypeInsn(CHECKCAST, "java/lang/Boolean");
162 				}
163 				mv.visitMethodInsn(INVOKEVIRTUAL, "java/lang/Boolean", "booleanValue", "()Z", false);
164 				break;
165 			case 'B':
166 				if (!stackDescriptor.equals("Ljava/lang/Byte")) {
167 					mv.visitTypeInsn(CHECKCAST, "java/lang/Byte");
168 				}
169 				mv.visitMethodInsn(INVOKEVIRTUAL, "java/lang/Byte", "byteValue", "()B", false);
170 				break;
171 			case 'C':
172 				if (!stackDescriptor.equals("Ljava/lang/Character")) {
173 					mv.visitTypeInsn(CHECKCAST, "java/lang/Character");
174 				}
175 				mv.visitMethodInsn(INVOKEVIRTUAL, "java/lang/Character", "charValue", "()C", false);
176 				break;
177 			case 'D':
178 				if (!stackDescriptor.equals("Ljava/lang/Double")) {
179 					mv.visitTypeInsn(CHECKCAST, "java/lang/Double");
180 				}
181 				mv.visitMethodInsn(INVOKEVIRTUAL, "java/lang/Double", "doubleValue", "()D", false);
182 				break;
183 			case 'F':
184 				if (!stackDescriptor.equals("Ljava/lang/Float")) {
185 					mv.visitTypeInsn(CHECKCAST, "java/lang/Float");
186 				}
187 				mv.visitMethodInsn(INVOKEVIRTUAL, "java/lang/Float", "floatValue", "()F", false);
188 				break;
189 			case 'I':
190 				if (!stackDescriptor.equals("Ljava/lang/Integer")) {
191 					mv.visitTypeInsn(CHECKCAST, "java/lang/Integer");
192 				}
193 				mv.visitMethodInsn(INVOKEVIRTUAL, "java/lang/Integer", "intValue", "()I", false);
194 				break;
195 			case 'J':
196 				if (!stackDescriptor.equals("Ljava/lang/Long")) {
197 					mv.visitTypeInsn(CHECKCAST, "java/lang/Long");
198 				}
199 				mv.visitMethodInsn(INVOKEVIRTUAL, "java/lang/Long", "longValue", "()J", false);
200 				break;
201 			case 'S':
202 				if (!stackDescriptor.equals("Ljava/lang/Short")) {
203 					mv.visitTypeInsn(CHECKCAST, "java/lang/Short");
204 				}
205 				mv.visitMethodInsn(INVOKEVIRTUAL, "java/lang/Short", "shortValue", "()S", false);
206 				break;
207 			default:
208 				throw new IllegalArgumentException("Unboxing should not be attempted for descriptor '" + ch + "'");
209 		}
210 	}

212 	/**
213 	 * For numbers, use the appropriate method on the number to convert it to the primitive type requested.
214 	 * @param mv the method visitor into which instructions should be inserted
215 	 * @param targetDescriptor the primitive type desired as output
216 	 * @param stackDescriptor the descriptor of the type on top of the stack
217 	 */
218 	public static void insertUnboxNumberInsns(MethodVisitor mv, char targetDescriptor, String stackDescriptor) {
219 		switch (targetDescriptor) {
220 			case 'D':
221 				if (stackDescriptor.equals("Ljava/lang/Object")) {
222 					mv.visitTypeInsn(CHECKCAST, "java/lang/Number");
223 				}
224 				mv.visitMethodInsn(INVOKEVIRTUAL, "java/lang/Number", "doubleValue", "()D", false);
225 				break;		
226 			case 'F':
227 				if (stackDescriptor.equals("Ljava/lang/Object")) {
228 					mv.visitTypeInsn(CHECKCAST, "java/lang/Number");
229 				}
230 				mv.visitMethodInsn(INVOKEVIRTUAL, "java/lang/Number", "floatValue", "()F", false);
231 				break;
232 			case 'J':
233 				if (stackDescriptor.equals("Ljava/lang/Object")) {
234 					mv.visitTypeInsn(CHECKCAST, "java/lang/Number");
235 				}
236 				mv.visitMethodInsn(INVOKEVIRTUAL, "java/lang/Number", "longValue", "()J", false);
237 				break;
238 			case 'I':
239 				if (stackDescriptor.equals("Ljava/lang/Object")) {
240 					mv.visitTypeInsn(CHECKCAST, "java/lang/Number");
241 				}
242 				mv.visitMethodInsn(INVOKEVIRTUAL, "java/lang/Number", "intValue", "()I", false);
243 				break;
244 			// does not handle Z, B, C, S
245 			default:
246 				throw new IllegalArgumentException("Unboxing should not be attempted for descriptor '" + targetDescriptor + "'");
247 		}
248 	}

250 	/**
251 	 * Insert any necessary numeric conversion bytecodes based upon what is on the stack and the desired target type.
252 	 * @param mv the method visitor into which instructions should be placed
253 	 * @param targetDescriptor the (primitive) descriptor of the target type
254 	 * @param stackDescriptor the descriptor of the operand on top of the stack
255 	 */
256 	public static void insertAnyNecessaryTypeConversionBytecodes(MethodVisitor mv, char targetDescriptor, String stackDescriptor) {
257 		if (CodeFlow.isPrimitive(stackDescriptor)) {
258 			char stackTop = stackDescriptor.charAt(0);
259 			if (stackTop=='I' || stackTop=='B' || stackTop=='S' || stackTop=='C') {
260 				if (targetDescriptor=='D') {
261 					mv.visitInsn(I2D);
262 				}
263 				else if (targetDescriptor=='F') {
264 					mv.visitInsn(I2F);
265 				}
266 				else if (targetDescriptor=='J') {
267 					mv.visitInsn(I2L);
268 				}
269 				else if (targetDescriptor=='I') {
270 					// nop
271 				}
272 				else {
273 					throw new IllegalStateException("cannot get from "+stackTop+" to "+targetDescriptor);
274 				}
275 			}
276 			else if (stackTop=='J') {
277 				if (targetDescriptor=='D') {
278 					mv.visitInsn(L2D);
279 				}
280 				else if (targetDescriptor=='F') {
281 					mv.visitInsn(L2F);
282 				}
283 				else if (targetDescriptor=='J') {
284 					// nop
285 				}
286 				else if (targetDescriptor=='I') {
287 					mv.visitInsn(L2I);
288 				}
289 				else {
290 					throw new IllegalStateException("cannot get from "+stackTop+" to "+targetDescriptor);
291 				}
292 			}
293 			else if (stackTop=='F') {
294 				if (targetDescriptor=='D') {
295 					mv.visitInsn(F2D);
296 				}
297 				else if (targetDescriptor=='F') {
298 					// nop
299 				}
300 				else if (targetDescriptor=='J') {
301 					mv.visitInsn(F2L);
302 				}
303 				else if (targetDescriptor=='I') {
304 					mv.visitInsn(F2I);
305 				}
306 				else {
307 					throw new IllegalStateException("cannot get from "+stackTop+" to "+targetDescriptor);
308 				}
309 			}
310 			else if (stackTop=='D') {
311 				if (targetDescriptor=='D') {
312 					// nop
313 				}
314 				else if (targetDescriptor=='F') {
315 					mv.visitInsn(D2F);
316 				}
317 				else if (targetDescriptor=='J') {
318 					mv.visitInsn(D2L);
319 				}
320 				else if (targetDescriptor=='I') {
321 					mv.visitInsn(D2I);
322 				}
323 				else {
324 					throw new IllegalStateException("cannot get from "+stackDescriptor+" to "+targetDescriptor);
325 				}
326 			}
327 		}
328 	}

332 	 * Create the JVM signature descriptor for a method. This consists of the descriptors
333 	 * for the method parameters surrounded with parentheses, followed by the
334 	 * descriptor for the return type. Note the descriptors here are JVM descriptors,
335 	 * unlike the other descriptor forms the compiler is using which do not include the
336 	 * trailing semicolon.
337 	 * @param method the method
338 	 * @return a String signature descriptor (e.g. "(ILjava/lang/String;)V")
339 	 */
340 	public static String createSignatureDescriptor(Method method) {
341 		Class<?>[] params = method.getParameterTypes();
342 		StringBuilder sb = new StringBuilder();
343 		sb.append("(");
344 		for (Class<?> param : params) {
345 			sb.append(toJvmDescriptor(param));
346 		}
347 		sb.append(")");
348 		sb.append(toJvmDescriptor(method.getReturnType()));
349 		return sb.toString();
350 	}

352 	/**
353 	 * Create the JVM signature descriptor for a constructor. This consists of the
354 	 * descriptors for the constructor parameters surrounded with parentheses, followed by
355 	 * the descriptor for the return type, which is always "V". Note the
356 	 * descriptors here are JVM descriptors, unlike the other descriptor forms the
357 	 * compiler is using which do not include the trailing semicolon.
358 	 * @param ctor the constructor
359 	 * @return a String signature descriptor (e.g. "(ILjava/lang/String;)V")
360 	 */
361 	public static String createSignatureDescriptor(Constructor<?> ctor) {
362 		Class<?>[] params = ctor.getParameterTypes();
363 		StringBuilder sb = new StringBuilder();
364 		sb.append("(");
365 		for (Class<?> param : params) {
366 			sb.append(toJvmDescriptor(param));
367 		}
368 		sb.append(")V");
369 		return sb.toString();
370 	}

372 	/**
373 	 * Determine the JVM descriptor for a specified class. Unlike the other descriptors
374 	 * used in the compilation process, this is the one the JVM wants, so this one
375 	 * includes any necessary trailing semicolon (e.g. Ljava/lang/String; rather than
376 	 * Ljava/lang/String)
377 	 * @param clazz a class
378 	 * @return the JVM descriptor for the class
379 	 */
380 	public static String toJvmDescriptor(Class<?> clazz) {
381 		StringBuilder sb = new StringBuilder();
382 		if (clazz.isArray()) {
383 			while (clazz.isArray()) {
384 				sb.append("[");
385 				clazz = clazz.getComponentType();
386 			}
387 		}
388 		if (clazz.isPrimitive()) {
389 			if (clazz == Void.TYPE) {
390 				sb.append('V');
391 			}
392 			else if (clazz == Integer.TYPE) {
393 				sb.append('I');
394 			}
395 			else if (clazz == Boolean.TYPE) {
396 				sb.append('Z');
397 			}
398 			else if (clazz == Character.TYPE) {
399 				sb.append('C');
400 			}
401 			else if (clazz == Long.TYPE) {
402 				sb.append('J');
403 			}
404 			else if (clazz == Double.TYPE) {
405 				sb.append('D');
406 			}
407 			else if (clazz == Float.TYPE) {
408 				sb.append('F');
409 			}
410 			else if (clazz == Byte.TYPE) {
411 				sb.append('B');
412 			}
413 			else if (clazz == Short.TYPE) {
414 				sb.append('S');
415 			}
416 		}
417 		else {
418 			sb.append("L");
419 			sb.append(clazz.getName().replace('.', '/'));
420 			sb.append(";");
421 		}
422 		return sb.toString();
423 	}

456 	/**
457 	 * @param descriptor the descriptor for a possible primitive array
458 	 * @return {@code true} if the descriptor is for a primitive array (e.g. "[[I")
459 	 */
460 	public static boolean isPrimitiveArray(String descriptor) {
461 		boolean primitive = true;
462 		for (int i = 0, max = descriptor.length(); i < max; i++) {
463 			char ch = descriptor.charAt(i);
464 			if (ch == '[') {
465 				continue;
466 			}
467 			primitive = (ch != 'L');
468 			break;
469 		}
470 		return primitive;
471 	}

474 	 * Determine if boxing/unboxing can get from one type to the other. Assumes at least
475 	 * one of the types is in boxed form (i.e. single char descriptor).
476 	 * @return {@code true} if it is possible to get (via boxing) from one descriptor to the other
477 	 */
478 	public static boolean areBoxingCompatible(String desc1, String desc2) {
479 		if (desc1.equals(desc2)) {
480 			return true;
481 		}
482 		if (desc1.length() == 1) {
483 			if (desc1.equals("Z")) {
484 				return desc2.equals("Ljava/lang/Boolean");
485 			}
486 			else if (desc1.equals("D")) {
487 				return desc2.equals("Ljava/lang/Double");
488 			}
489 			else if (desc1.equals("F")) {
490 				return desc2.equals("Ljava/lang/Float");
491 			}
492 			else if (desc1.equals("I")) {
493 				return desc2.equals("Ljava/lang/Integer");
494 			}
495 			else if (desc1.equals("J")) {
496 				return desc2.equals("Ljava/lang/Long");
497 			}
498 		}
499 		else if (desc2.length() == 1) {
500 			if (desc2.equals("Z")) {
501 				return desc1.equals("Ljava/lang/Boolean");
502 			}
503 			else if (desc2.equals("D")) {
504 				return desc1.equals("Ljava/lang/Double");
505 			}
506 			else if (desc2.equals("F")) {
507 				return desc1.equals("Ljava/lang/Float");
508 			}
509 			else if (desc2.equals("I")) {
510 				return desc1.equals("Ljava/lang/Integer");
511 			}
512 			else if (desc2.equals("J")) {
513 				return desc1.equals("Ljava/lang/Long");
514 			}
515 		}
516 		return false;
517 	}

519 	/**
520 	 * Determine if the supplied descriptor is for a supported number type or boolean. The
521 	 * compilation process only (currently) supports certain number types. These are
522 	 * double, float, long and int.
523 	 * @param descriptor the descriptor for a type
524 	 * @return {@code true} if the descriptor is for a supported numeric type or boolean
525 	 */
526 	public static boolean isPrimitiveOrUnboxableSupportedNumberOrBoolean(String descriptor) {
527 		if (descriptor == null) {
528 			return false;
529 		}
530 		if (isPrimitiveOrUnboxableSupportedNumber(descriptor)) {
531 			return true;
532 		}
533 		return ("Z".equals(descriptor) || descriptor.equals("Ljava/lang/Boolean"));
534 	}

536 	/**
537 	 * Determine if the supplied descriptor is for a supported number. The compilation
538 	 * process only (currently) supports certain number types. These are double, float,
539 	 * long and int.
540 	 * @param descriptor the descriptor for a type
541 	 * @return {@code true} if the descriptor is for a supported numeric type
542 	 */
543 	public static boolean isPrimitiveOrUnboxableSupportedNumber(String descriptor) {
544 		if (descriptor == null) {
545 			return false;
546 		}
547 		if (descriptor.length() == 1) {
548 			return "DFIJ".contains(descriptor);
549 		}
550 		if (descriptor.startsWith("Ljava/lang/")) {
551 			String name = descriptor.substring("Ljava/lang/".length());
552 			if (name.equals("Double") || name.equals("Float") || name.equals("Integer") || name.equals("Long")) {
553 				return true;
554 			}
555 		}
556 		return false;
557 	}

570 	 * @param descriptor a descriptor for a type that should have a primitive representation
571 	 * @return the single character descriptor for a primitive input descriptor
572 	 */
573 	public static char toPrimitiveTargetDesc(String descriptor) {
574 		if (descriptor.length() == 1) {
575 			return descriptor.charAt(0);
576 		}
577 		else if (descriptor.equals("Ljava/lang/Boolean")) {
578 			return 'Z';
579 		}
580 		else if (descriptor.equals("Ljava/lang/Byte")) {
581 			return 'B';
582 		}
583 		else if (descriptor.equals("Ljava/lang/Character")) {
584 			return 'C';
585 		}
586 		else if (descriptor.equals("Ljava/lang/Double")) {
587 			return 'D';
588 		}
589 		else if (descriptor.equals("Ljava/lang/Float")) {
590 			return 'F';
591 		}
592 		else if (descriptor.equals("Ljava/lang/Integer")) {
593 			return 'I';
594 		}
595 		else if (descriptor.equals("Ljava/lang/Long")) {
596 			return 'J';
597 		}
598 		else if (descriptor.equals("Ljava/lang/Short")) {
599 			return 'S';
600 		}
601 		else {
602 			throw new IllegalStateException("No primitive for '" + descriptor + "'");
603 		}
604 	}

606 	/**
607 	 * Insert the appropriate CHECKCAST instruction for the supplied descriptor.
608 	 * @param mv the target visitor into which the instruction should be inserted
609 	 * @param descriptor the descriptor of the type to cast to
610 	 */
611 	public static void insertCheckCast(MethodVisitor mv, String descriptor) {
612 		if (descriptor.length() != 1) {
613 			if (descriptor.charAt(0) == '[') {
614 				if (isPrimitiveArray(descriptor)) {
615 					mv.visitTypeInsn(CHECKCAST, descriptor);
616 				}
617 				else {
618 					mv.visitTypeInsn(CHECKCAST, descriptor + ";");
619 				}
620 			}
621 			else {
622 				if (!descriptor.equals("Ljava/lang/Object")) {
623 					// This is chopping off the 'L' to leave us with "java/lang/String"
624 					mv.visitTypeInsn(CHECKCAST, descriptor.substring(1));
625 				}
626 			}
627 		}
628 	}

643 	 * Determine the appropriate boxing instruction for a specific type (if it needs
644 	 * boxing) and insert the instruction into the supplied visitor.
645 	 * @param mv the target visitor for the new instructions
646 	 * @param ch the descriptor of the type that might need boxing
647 	 */
648 	public static void insertBoxIfNecessary(MethodVisitor mv, char ch) {
649 		switch (ch) {
650 			case 'Z':
651 				mv.visitMethodInsn(INVOKESTATIC, "java/lang/Boolean", "valueOf", "(Z)Ljava/lang/Boolean;", false);
652 				break;
653 			case 'B':
654 				mv.visitMethodInsn(INVOKESTATIC, "java/lang/Byte", "valueOf", "(B)Ljava/lang/Byte;", false);
655 				break;
656 			case 'C':
657 				mv.visitMethodInsn(INVOKESTATIC, "java/lang/Character", "valueOf", "(C)Ljava/lang/Character;", false);
658 				break;
659 			case 'D':
660 				mv.visitMethodInsn(INVOKESTATIC, "java/lang/Double", "valueOf", "(D)Ljava/lang/Double;", false);
661 				break;
662 			case 'F':
663 				mv.visitMethodInsn(INVOKESTATIC, "java/lang/Float", "valueOf", "(F)Ljava/lang/Float;", false);
664 				break;
665 			case 'I':
666 				mv.visitMethodInsn(INVOKESTATIC, "java/lang/Integer", "valueOf", "(I)Ljava/lang/Integer;", false);
667 				break;
668 			case 'J':
669 				mv.visitMethodInsn(INVOKESTATIC, "java/lang/Long", "valueOf", "(J)Ljava/lang/Long;", false);
670 				break;
671 			case 'S':
672 				mv.visitMethodInsn(INVOKESTATIC, "java/lang/Short", "valueOf", "(S)Ljava/lang/Short;", false);
673 				break;
674 			case 'L':
675 			case 'V':
676 			case '[':
677 				// no box needed
678 				break;
679 			default:
680 				throw new IllegalArgumentException("Boxing should not be attempted for descriptor '" + ch + "'");
681 		}
682 	}

684 	/**
685 	 * Deduce the descriptor for a type. Descriptors are like JVM type names but missing
686 	 * the trailing ';' so for Object the descriptor is "Ljava/lang/Object" for int it is
687 	 * "I".
688 	 * @param type the type (may be primitive) for which to determine the descriptor
689 	 * @return the descriptor
690 	 */
691 	public static String toDescriptor(Class<?> type) {
692 		String name = type.getName();
693 		if (type.isPrimitive()) {
694 			switch (name.length()) {
695 				case 3:
696 					return "I";
697 				case 4:
698 					if (name.equals("byte")) {
699 						return "B";
700 					}
701 					else if (name.equals("char")) {
702 						return "C";
703 					}
704 					else if (name.equals("long")) {
705 						return "J";
706 					}
707 					else if (name.equals("void")) {
708 						return "V";
709 					}
710 					break;
711 				case 5:
712 					if (name.equals("float")) {
713 						return "F";
714 					}
715 					else if (name.equals("short")) {
716 						return "S";
717 					}
718 					break;
719 				case 6:
720 					if (name.equals("double")) {
721 						return "D";
722 					}
723 					break;
724 				case 7:
725 					if (name.equals("boolean")) {
726 						return "Z";
727 					}
728 					break;
729 			}
730 		}
731 		else {
732 			if (name.charAt(0) != '[') {
733 				return "L" + type.getName().replace('.', '/');
734 			}
735 			else {
736 				if (name.endsWith(";")) {
737 					return name.substring(0, name.length() - 1).replace('.', '/');
738 				}
739 				else {
740 					return name;  // array has primitive component type
741 				}
742 			}
743 		}
744 		return null;
745 	}

782 	 * Called after the main expression evaluation method has been generated, this
783 	 * method will callback any registered FieldAdders or ClinitAdders to add any
784 	 * extra information to the class representing the compiled expression.
785 	 */
786 	public void finish() {
787 		if (fieldAdders != null) {
788 			for (FieldAdder fieldAdder: fieldAdders) {
789 				fieldAdder.generateField(cw,this);
790 			}
791 		}
792 		if (clinitAdders != null) {
793 			MethodVisitor mv = cw.visitMethod(ACC_PUBLIC | ACC_STATIC, "<clinit>", "()V", null, null);
794 			mv.visitCode();
795 			nextFreeVariableId = 0; // To 0 because there is no 'this' in a clinit
796 			for (ClinitAdder clinitAdder: clinitAdders) {
797 				clinitAdder.generateCode(mv, this);
798 			}
799 			mv.visitInsn(RETURN);
800 			mv.visitMaxs(0,0); // not supplied due to COMPUTE_MAXS
801 			mv.visitEnd();
802 		}
803 	}

850 	 * Create the optimal instruction for loading a number on the stack.
851 	 * @param mv where to insert the bytecode
852 	 * @param value the value to be loaded
853 	 */
854 	public static void insertOptimalLoad(MethodVisitor mv, int value) {
855 		if (value < 6) {
856 			mv.visitInsn(ICONST_0+value);
857 		}
858 		else if (value < Byte.MAX_VALUE) {
859 			mv.visitIntInsn(BIPUSH, value);
860 		}
861 		else if (value < Short.MAX_VALUE) {
862 			mv.visitIntInsn(SIPUSH, value);
863 		}
864 		else {
865 			mv.visitLdcInsn(value);
866 		}
867 	}

870 	 * Produce appropriate bytecode to store a stack item in an array. The
871 	 * instruction to use varies depending on whether the type
872 	 * is a primitive or reference type.
873 	 * @param mv where to insert the bytecode
874 	 * @param arrayElementType the type of the array elements
875 	 */
876 	public static void insertArrayStore(MethodVisitor mv, String arrayElementType) {
877 		if (arrayElementType.length()==1) {
878 			switch (arrayElementType.charAt(0)) {
879 				case 'I': mv.visitInsn(IASTORE); break;
880 				case 'J': mv.visitInsn(LASTORE); break;
881 				case 'F': mv.visitInsn(FASTORE); break;
882 				case 'D': mv.visitInsn(DASTORE); break;
883 				case 'B': mv.visitInsn(BASTORE); break;
884 				case 'C': mv.visitInsn(CASTORE); break;
885 				case 'S': mv.visitInsn(SASTORE); break;
886 				case 'Z': mv.visitInsn(BASTORE); break;
887 				default:
888 					throw new IllegalArgumentException("Unexpected arraytype "+arrayElementType.charAt(0));
889 			}
890 		}
891 		else {
892 			mv.visitInsn(AASTORE);
893 		}
894 	}

899 	 * @return the T tag to use for NEWARRAY

901 	public static int arrayCodeFor(String arraytype) {
902 		switch (arraytype.charAt(0)) {

905 			case 'F': return T_FLOAT;

907 			case 'B': return T_BYTE;

909 			case 'S': return T_SHORT;

920 		int length = arraytype.length();
921 		for (int i=0;i<length;i++) {

929 	/**
930 	 * Produce the correct bytecode to build an array. The opcode to use and the
931 	 * signature to pass along with the opcode can vary depending on the signature
932 	 * of the array type.
933 	 * @param mv the methodvisitor into which code should be inserted
934 	 * @param size the size of the array
935 	 * @param arraytype the type of the array
936 	 */
937 	public static void insertNewArrayCode(MethodVisitor mv, int size, String arraytype) {
938 		insertOptimalLoad(mv, size);
939 		if (arraytype.length() == 1) {
940 			mv.visitIntInsn(NEWARRAY, CodeFlow.arrayCodeFor(arraytype));
941 		}
942 		else {
943 			if (arraytype.charAt(0) == '[') {
944 				// Handling the nested array case here. If vararg
945 				// is [[I then we want [I and not [I;
946 				if (CodeFlow.isReferenceTypeArray(arraytype)) {
947 					mv.visitTypeInsn(ANEWARRAY, arraytype+";");
948 				} else {
949 					mv.visitTypeInsn(ANEWARRAY, arraytype);
950 				}
951 			}
952 			else {
953 				mv.visitTypeInsn(ANEWARRAY, arraytype.substring(1));
954 			}
955 		}
956 	}
