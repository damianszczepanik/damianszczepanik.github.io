42  	/**
43  	 * Compare argument arrays and return information about whether they match. A supplied
44  	 * type converter and conversionAllowed flag allow for matches to take into account
45  	 * that a type may be transformed into a different type by the converter.
46  	 * @param expectedArgTypes the array of types the method/constructor is expecting
47  	 * @param suppliedArgTypes the array of types that are being supplied at the point of invocation
48  	 * @param typeConverter a registered type converter
49  	 * @return a MatchInfo object indicating what kind of match it was or null if it was not a match
50  	 */
51  	static ArgumentsMatchInfo compareArguments(
52  			List<TypeDescriptor> expectedArgTypes, List<TypeDescriptor> suppliedArgTypes, TypeConverter typeConverter) {
53  
54  		Assert.isTrue(expectedArgTypes.size() == suppliedArgTypes.size(),
55  				"Expected argument types and supplied argument types should be arrays of same length");
56  
57  		ArgumentsMatchKind match = ArgumentsMatchKind.EXACT;
58  		for (int i = 0; i < expectedArgTypes.size() && match != null; i++) {
59  			TypeDescriptor suppliedArg = suppliedArgTypes.get(i);
60  			TypeDescriptor expectedArg = expectedArgTypes.get(i);
61  			if (!expectedArg.equals(suppliedArg)) {
62  				// The user may supply null - and that will be ok unless a primitive is expected
63  				if (suppliedArg == null) {
64  					if (expectedArg.isPrimitive()) {
65  						match = null;
66  					}
67  				}
68  				else {
69  					if (suppliedArg.isAssignableTo(expectedArg)) {
70  						if (match != ArgumentsMatchKind.REQUIRES_CONVERSION) {
71  							match = ArgumentsMatchKind.CLOSE;
72  						}
73  					}
74  					else if (typeConverter.canConvert(suppliedArg, expectedArg)) {
75  						match = ArgumentsMatchKind.REQUIRES_CONVERSION;
76  					}
77  					else {
78  						match = null;
79  					}
80  				}
81  			}
82  		}
83  		return (match != null ? new ArgumentsMatchInfo(match) : null);
84  	}

86  	/**
87  	 * Based on {@link MethodInvoker#getTypeDifferenceWeight(Class[], Object[])} but operates on TypeDescriptors.
88  	 */
89  	public static int getTypeDifferenceWeight(List<TypeDescriptor> paramTypes, List<TypeDescriptor> argTypes) {
90  		int result = 0;
91  		for (int i = 0; i < paramTypes.size(); i++) {
92  			TypeDescriptor paramType = paramTypes.get(i);
93  			TypeDescriptor argType = argTypes.get(i);
94  			if (argType == null) {
95  				if (paramType.isPrimitive()) {
96  					return Integer.MAX_VALUE;
97  				}
98  			}
99  			else {
100 				Class<?> paramTypeClazz = paramType.getType();
101 				if (!ClassUtils.isAssignable(paramTypeClazz, argType.getType())) {
102 					return Integer.MAX_VALUE;
103 				}
104 				if (paramTypeClazz.isPrimitive()) {
105 					paramTypeClazz = Object.class;
106 				}
107 				Class<?> superClass = argType.getType().getSuperclass();
108 				while (superClass != null) {
109 					if (paramTypeClazz.equals(superClass)) {
110 						result = result + 2;
111 						superClass = null;
112 					}
113 					else if (ClassUtils.isAssignable(paramTypeClazz, superClass)) {
114 						result = result + 2;
115 						superClass = superClass.getSuperclass();
116 					}
117 					else {
118 						superClass = null;
119 					}
120 				}
121 				if (paramTypeClazz.isInterface()) {
122 					result = result + 1;
123 				}
124 			}
125 		}
126 		return result;
127 	}

129 	/**
130 	 * Compare argument arrays and return information about whether they match. A supplied type converter and
131 	 * conversionAllowed flag allow for matches to take into account that a type may be transformed into a different
132 	 * type by the converter. This variant of compareArguments also allows for a varargs match.
133 	 * @param expectedArgTypes the array of types the method/constructor is expecting
134 	 * @param suppliedArgTypes the array of types that are being supplied at the point of invocation
135 	 * @param typeConverter a registered type converter
136 	 * @return a MatchInfo object indicating what kind of match it was or null if it was not a match
137 	 */
138 	static ArgumentsMatchInfo compareArgumentsVarargs(
139 			List<TypeDescriptor> expectedArgTypes, List<TypeDescriptor> suppliedArgTypes, TypeConverter typeConverter) {
140 
141 		Assert.isTrue(expectedArgTypes != null && expectedArgTypes.size() > 0,
142 				"Expected arguments must at least include one array (the vargargs parameter)");
143 		Assert.isTrue(expectedArgTypes.get(expectedArgTypes.size() - 1).isArray(),
144 				"Final expected argument should be array type (the varargs parameter)");
145 
146 		ArgumentsMatchKind match = ArgumentsMatchKind.EXACT;
147 
148 		// Check up until the varargs argument:
149 
150 		// Deal with the arguments up to 'expected number' - 1 (that is everything but the varargs argument)
151 		int argCountUpToVarargs = expectedArgTypes.size() - 1;
152 		for (int i = 0; i < argCountUpToVarargs && match != null; i++) {
153 			TypeDescriptor suppliedArg = suppliedArgTypes.get(i);
154 			TypeDescriptor expectedArg = expectedArgTypes.get(i);
155 			if (suppliedArg == null) {
156 				if (expectedArg.isPrimitive()) {
157 					match = null;
158 				}
159 			}
160 			else {
161 				if (!expectedArg.equals(suppliedArg)) {
162 					if (suppliedArg.isAssignableTo(expectedArg)) {
163 						if (match != ArgumentsMatchKind.REQUIRES_CONVERSION) {
164 							match = ArgumentsMatchKind.CLOSE;
165 						}
166 					}
167 					else if (typeConverter.canConvert(suppliedArg, expectedArg)) {
168 						match = ArgumentsMatchKind.REQUIRES_CONVERSION;
169 					}
170 					else {
171 						match = null;
172 					}
173 				}
174 			}
175 		}
176 
177 		// If already confirmed it cannot be a match, then return
178 		if (match == null) {
179 			return null;
180 		}
181 
182 		if (suppliedArgTypes.size() == expectedArgTypes.size() &&
183 				expectedArgTypes.get(expectedArgTypes.size() - 1).equals(
184 						suppliedArgTypes.get(suppliedArgTypes.size() - 1))) {
185 			// Special case: there is one parameter left and it is an array and it matches the varargs
186 			// expected argument - that is a match, the caller has already built the array. Proceed with it.
187 		}
188 		else {
189 			// Now... we have the final argument in the method we are checking as a match and we have 0
190 			// or more other arguments left to pass to it.
191 			TypeDescriptor varargsDesc = expectedArgTypes.get(expectedArgTypes.size() - 1);
192 			Class<?> varargsParamType = varargsDesc.getElementTypeDescriptor().getType();
193 
194 			// All remaining parameters must be of this type or convertable to this type
195 			for (int i = expectedArgTypes.size() - 1; i < suppliedArgTypes.size(); i++) {
196 				TypeDescriptor suppliedArg = suppliedArgTypes.get(i);
197 				if (suppliedArg == null) {
198 					if (varargsParamType.isPrimitive()) {
199 						match = null;
200 					}
201 				}
202 				else {
203 					if (varargsParamType != suppliedArg.getType()) {
204 						if (ClassUtils.isAssignable(varargsParamType, suppliedArg.getType())) {
205 							if (match != ArgumentsMatchKind.REQUIRES_CONVERSION) {
206 								match = ArgumentsMatchKind.CLOSE;
207 							}
208 						}
209 						else if (typeConverter.canConvert(suppliedArg, TypeDescriptor.valueOf(varargsParamType))) {
210 							match = ArgumentsMatchKind.REQUIRES_CONVERSION;
211 						}
212 						else {
213 							match = null;
214 						}
215 					}
216 				}
217 			}
218 		}
219 
220 		return (match != null ? new ArgumentsMatchInfo(match) : null);
221 	}

225 	/**
226 	 * Convert a supplied set of arguments into the requested types. If the parameterTypes are related to
227 	 * a varargs method then the final entry in the parameterTypes array is going to be an array itself whose
228 	 * component type should be used as the conversion target for extraneous arguments. (For example, if the
229 	 * parameterTypes are {Integer, String[]} and the input arguments are {Integer, boolean, float} then both
230 	 * the boolean and float must be converted to strings). This method does *not* repackage the arguments
231 	 * into a form suitable for the varargs invocation - a subsequent call to setupArgumentsForVarargsInvocation handles that.
232 	 * @param converter the converter to use for type conversions
233 	 * @param arguments the arguments to convert to the requested parameter types
234 	 * @param method the target Method
235 	 * @return true if some kind of conversion occurred on the argument
236 	 * @throws SpelEvaluationException if there is a problem with conversion
237 	 */
238 	public static boolean convertAllArguments(TypeConverter converter, Object[] arguments, Method method) throws SpelEvaluationException {
239 		Integer varargsPosition = method.isVarArgs() ? method.getParameterTypes().length-1:null;
240 		return convertArguments(converter, arguments, method, varargsPosition);
241 	}

243 	/**
244 	 * Takes an input set of argument values and converts them to the types specified as the
245 	 * required parameter types. The arguments are converted 'in-place' in the input array.
246 	 * @param converter the type converter to use for attempting conversions
247 	 * @param arguments the actual arguments that need conversion
248 	 * @param methodOrCtor the target Method or Constructor
249 	 * @param varargsPosition the known position of the varargs argument, if any (null if not varargs)
250 	 * @return true if some kind of conversion occurred on an argument
251 	 * @throws EvaluationException if a problem occurs during conversion
252 	 */
253 	static boolean convertArguments(TypeConverter converter, Object[] arguments, Object methodOrCtor,
254 			Integer varargsPosition) throws EvaluationException {
255 		boolean conversionOccurred = false;
256 		if (varargsPosition == null) {
257 			for (int i = 0; i < arguments.length; i++) {
258 				TypeDescriptor targetType = new TypeDescriptor(MethodParameter.forMethodOrConstructor(methodOrCtor, i));
259 				Object argument = arguments[i];
260 				arguments[i] = converter.convertValue(argument, TypeDescriptor.forObject(argument), targetType);
261 				conversionOccurred |= (argument != arguments[i]);
262 			}
263 		}
264 		else {
265 			// Convert everything up to the varargs position
266 			for (int i = 0; i < varargsPosition; i++) {
267 				TypeDescriptor targetType = new TypeDescriptor(MethodParameter.forMethodOrConstructor(methodOrCtor, i));
268 				Object argument = arguments[i];
269 				arguments[i] = converter.convertValue(argument, TypeDescriptor.forObject(argument), targetType);
270 				conversionOccurred |= (argument != arguments[i]);
271 			}
272 			MethodParameter methodParam = MethodParameter.forMethodOrConstructor(methodOrCtor, varargsPosition);
273 			if (varargsPosition == arguments.length - 1) {
274 				// If the target is varargs and there is just one more argument
275 				// then convert it here
276 				TypeDescriptor targetType = new TypeDescriptor(methodParam);
277 				Object argument = arguments[varargsPosition];
278 				TypeDescriptor sourceType = TypeDescriptor.forObject(argument);
279 				arguments[varargsPosition] = converter.convertValue(argument, sourceType, targetType);
280 				// Three outcomes of that previous line:
281 				// 1) the input argument was already compatible (ie. array of valid type) and nothing was done
282 				// 2) the input argument was correct type but not in an array so it was made into an array
283 				// 3) the input argument was the wrong type and got converted and put into an array
284 				if (argument != arguments[varargsPosition] &&
285 						!isFirstEntryInArray(argument, arguments[varargsPosition])) {
286 					conversionOccurred = true; // case 3
287 				}
288 			}
289 			else {
290 				// Convert remaining arguments to the varargs element type
291 				TypeDescriptor targetType = new TypeDescriptor(methodParam).getElementTypeDescriptor();
292 				for (int i = varargsPosition; i < arguments.length; i++) {
293 					Object argument = arguments[i];
294 					arguments[i] = converter.convertValue(argument, TypeDescriptor.forObject(argument), targetType);
295 					conversionOccurred |= (argument != arguments[i]);
296 				}
297 			}
298 		}
299 		return conversionOccurred;
300 	}

302 	/**
303 	 * Check if the supplied value is the first entry in the array represented by the possibleArray value.
304 	 * @param value the value to check for in the array
305 	 * @param possibleArray an array object that may have the supplied value as the first element
306 	 * @return true if the supplied value is the first entry in the array
307 	 */
308 	private static boolean isFirstEntryInArray(Object value, Object possibleArray) {
309 		if (possibleArray == null) {
310 			return false;
311 		}
312 		Class<?> type = possibleArray.getClass();
313 		if (!type.isArray() || Array.getLength(possibleArray) == 0 ||
314 				!ClassUtils.isAssignableValue(type.getComponentType(), value)) {
315 			return false;
316 		}
317 		Object arrayValue = Array.get(possibleArray, 0);
318 		return (type.getComponentType().isPrimitive() ? arrayValue.equals(value) : arrayValue == value);
319 	}

321 	/**
322 	 * Package up the arguments so that they correctly match what is expected in parameterTypes.
323 	 * For example, if parameterTypes is (int, String[]) because the second parameter was declared String...
324 	 * then if arguments is [1,"a","b"] then it must be repackaged as [1,new String[]{"a","b"}] in order to
325 	 * match the expected parameterTypes.
326 	 * @param requiredParameterTypes the types of the parameters for the invocation
327 	 * @param args the arguments to be setup ready for the invocation
328 	 * @return a repackaged array of arguments where any varargs setup has been done
329 	 */
330 	public static Object[] setupArgumentsForVarargsInvocation(Class<?>[] requiredParameterTypes, Object... args) {
331 		// Check if array already built for final argument
332 		int parameterCount = requiredParameterTypes.length;
333 		int argumentCount = args.length;
334 
335 		// Check if repackaging is needed...
336 		if (parameterCount != args.length ||
337 				requiredParameterTypes[parameterCount - 1] !=
338 						(args[argumentCount - 1] != null ? args[argumentCount - 1].getClass() : null)) {
339 
340 			int arraySize = 0;  // zero size array if nothing to pass as the varargs parameter
341 			if (argumentCount >= parameterCount) {
342 				arraySize = argumentCount - (parameterCount - 1);
343 			}
344 
345 			// Create an array for the varargs arguments
346 			Object[] newArgs = new Object[parameterCount];
347 			System.arraycopy(args, 0, newArgs, 0, newArgs.length - 1);
348 
349 			// Now sort out the final argument, which is the varargs one. Before entering this method,
350 			// the arguments should have been converted to the box form of the required type.
351 			Class<?> componentType = requiredParameterTypes[parameterCount - 1].getComponentType();
352 			Object repackagedArgs = Array.newInstance(componentType, arraySize);
353 			for (int i = 0; i < arraySize; i++) {
354 				Array.set(repackagedArgs, i, args[parameterCount - 1 + i]);
355 			}
356 			newArgs[newArgs.length - 1] = repackagedArgs;
357 			return newArgs;
358 		}
359 		return args;
360 	}

363 	static enum ArgumentsMatchKind {
364 
365 		/** An exact match is where the parameter types exactly match what the method/constructor is expecting */
366 		EXACT,
367 
368 		/** A close match is where the parameter types either exactly match or are assignment-compatible */
369 		CLOSE,
370 
371 		/** A conversion match is where the type converter must be used to transform some of the parameter types */
372 		REQUIRES_CONVERSION
373 	}
374

376 	/**
377 	 * An instance of ArgumentsMatchInfo describes what kind of match was achieved between two sets of arguments -
378 	 * the set that a method/constructor is expecting and the set that are being supplied at the point of invocation.
379 	 * If the kind indicates that conversion is required for some of the arguments then the arguments that require
380 	 * conversion are listed in the argsRequiringConversion array.
381 	 */
382 	static class ArgumentsMatchInfo {
383 
384 		private final ArgumentsMatchKind kind;
385 
386 		ArgumentsMatchInfo(ArgumentsMatchKind kind) {
387 			this.kind = kind;
388 		}
389 
390 		public boolean isExactMatch() {
391 			return (this.kind == ArgumentsMatchKind.EXACT);
392 		}
393 
394 		public boolean isCloseMatch() {
395 			return (this.kind == ArgumentsMatchKind.CLOSE);
396 		}
397 
398 		public boolean isMatchRequiringConversion() {
399 			return (this.kind == ArgumentsMatchKind.REQUIRES_CONVERSION);
400 		}
401 
402 		@Override
403 		public String toString() {
404 			return "ArgumentMatchInfo: " + this.kind;
405 		}
406 	}
