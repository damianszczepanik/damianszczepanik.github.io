51  	public void valueType() throws Exception {
52  		SpelExpressionParser parser = new SpelExpressionParser();
53  		EvaluationContext ctx = new StandardEvaluationContext();
54  		Class<?> c = parser.parseRaw("2").getValueType();
55  		assertEquals(Integer.class, c);
56  		c = parser.parseRaw("12").getValueType(ctx);
57  		assertEquals(Integer.class, c);
58  		c = parser.parseRaw("null").getValueType();
59  		assertNull(c);
60  		c = parser.parseRaw("null").getValueType(ctx);
61  		assertNull(c);
62  		Object o = parser.parseRaw("null").getValue(ctx, Integer.class);
63  		assertNull(o);
64  	}

226 		expr = new SpelExpressionParser().parseRaw("!true");
227 		assertEquals(Boolean.FALSE, expr.getValue(Boolean.class));
228 		expr = new SpelExpressionParser().parseRaw("!(false or true)");
229 		assertEquals(Boolean.FALSE, expr.getValue(Boolean.class));
230 	}

283 	@Test
284 	public void positionalInformation() throws EvaluationException, ParseException {
285 		SpelExpression expr = new SpelExpressionParser().parseRaw("true and true or false");
286 		SpelNode rootAst = expr.getAST();
287 		OpOr operatorOr = (OpOr) rootAst;
288 		OpAnd operatorAnd = (OpAnd) operatorOr.getLeftOperand();
289 		SpelNode rightOrOperand = operatorOr.getRightOperand();
290 
291 		// check position for final 'false'
292 		assertEquals(17, rightOrOperand.getStartPosition());
293 		assertEquals(22, rightOrOperand.getEndPosition());
294 
295 		// check position for first 'true'
296 		assertEquals(0, operatorAnd.getLeftOperand().getStartPosition());
297 		assertEquals(4, operatorAnd.getLeftOperand().getEndPosition());
298 
299 		// check position for second 'true'
300 		assertEquals(9, operatorAnd.getRightOperand().getStartPosition());
301 		assertEquals(13, operatorAnd.getRightOperand().getEndPosition());
302 
303 		// check position for OperatorAnd
304 		assertEquals(5, operatorAnd.getStartPosition());
305 		assertEquals(8, operatorAnd.getEndPosition());
306 
307 		// check position for OperatorOr
308 		assertEquals(14, operatorOr.getStartPosition());
309 		assertEquals(16, operatorOr.getEndPosition());
310 	}

385 
386 		checkNumberError("3.4L", SpelMessage.REAL_CANNOT_BE_LONG);
387 
388 		checkNumber("3.5f", 3.5f, Float.class);
389 
390 		checkNumber("1.2e3", 1.2e3d, Double.class);
391 		checkNumber("1.2e+3", 1.2e3d, Double.class);
392 		checkNumber("1.2e-3", 1.2e-3d, Double.class);
393 		checkNumber("1.2e3", 1.2e3d, Double.class);
394 		checkNumber("1e+3", 1e3d, Double.class);

397 	private void checkNumber(String expression, Object value, Class<?> type) {
398 		try {
399 			SpelExpressionParser parser = new SpelExpressionParser();
400 			SpelExpression expr = parser.parseRaw(expression);
401 			Object o = expr.getValue();
402 			assertEquals(value, o);
403 			assertEquals(type, o.getClass());
404 		} catch (Exception e) {
405 			e.printStackTrace();
406 			fail(e.getMessage());
407 		}
408 	}
409 
410 	private void checkNumberError(String expression, SpelMessage expectedMessage) {
411 		try {
412 			SpelExpressionParser parser = new SpelExpressionParser();
413 			parser.parseRaw(expression);
414 			fail();
415 		} catch (ParseException e) {
416 			assertTrue(e instanceof SpelParseException);
417 			SpelParseException spe = (SpelParseException) e;
418 			assertEquals(expectedMessage, spe.getMessageCode());
419 		}
420 	}
