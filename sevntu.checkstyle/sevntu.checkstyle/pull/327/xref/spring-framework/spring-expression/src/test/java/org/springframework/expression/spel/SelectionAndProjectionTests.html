331 	static class IntegerTestBean {
332 
333 		private final IntegerWrapper wrapper;
334 
335 		IntegerTestBean(Number value) {
336 			this.wrapper = new IntegerWrapper(value);
337 		}
338 
339 		public IntegerWrapper getWrapper() {
340 			return this.wrapper;
341 		}
342 
343 		static List<IntegerTestBean> createList() {
344 			List<IntegerTestBean> list = new ArrayList<IntegerTestBean>();
345 			for (int i = 0; i < 3; i++) {
346 				list.add(new IntegerTestBean(i + 5));
347 			}
348 			return list;
349 		}
350 
351 		static Set<IntegerTestBean> createSet() {
352 			Set<IntegerTestBean> set = new LinkedHashSet<IntegerTestBean>();
353 			for (int i = 0; i < 3; i++) {
354 				set.add(new IntegerTestBean(i + 5));
355 			}
356 			return set;
357 		}
358 
359 		static IntegerTestBean[] createArray() {
360 			IntegerTestBean[] array = new IntegerTestBean[3];
361 			for (int i = 0; i < 3; i++) {
362 				if (i == 1) {
363 					array[i] = new IntegerTestBean(5.9f);
364 				} else {
365 					array[i] = new IntegerTestBean(i + 5);
366 				}
367 			}
368 			return array;
369 		}
370 	}
