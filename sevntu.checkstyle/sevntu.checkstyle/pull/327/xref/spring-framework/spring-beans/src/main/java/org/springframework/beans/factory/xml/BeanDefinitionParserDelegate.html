332 	/**
333 	 * Populate the given DocumentDefaultsDefinition instance with the default lazy-init,
334 	 * autowire, dependency check settings, init-method, destroy-method and merge settings.
335 	 * Support nested 'beans' element use cases by falling back to
336 	 * <literal>parentDefaults</literal> in case the defaults are not explicitly set
337 	 * locally.
338 	 * @param defaults the defaults to populate
339 	 * @param parentDefaults the parent BeanDefinitionParserDelegate (if any) defaults to fall back to
340 	 * @param root the root element of the current bean definition document (or nested beans element)
341 	 */
342 	protected void populateDefaults(DocumentDefaultsDefinition defaults, DocumentDefaultsDefinition parentDefaults, Element root) {
343 		String lazyInit = root.getAttribute(DEFAULT_LAZY_INIT_ATTRIBUTE);
344 		if (DEFAULT_VALUE.equals(lazyInit)) {
345 			lazyInit = parentDefaults != null ? parentDefaults.getLazyInit() : FALSE_VALUE;
346 		}
347 		defaults.setLazyInit(lazyInit);
348 
349 		String merge = root.getAttribute(DEFAULT_MERGE_ATTRIBUTE);
350 		if (DEFAULT_VALUE.equals(merge)) {
351 			merge = parentDefaults != null ? parentDefaults.getMerge() : FALSE_VALUE;
352 		}
353 		defaults.setMerge(merge);
354 
355 		String autowire = root.getAttribute(DEFAULT_AUTOWIRE_ATTRIBUTE);
356 		if (DEFAULT_VALUE.equals(autowire)) {
357 			autowire = parentDefaults != null ? parentDefaults.getAutowire() : AUTOWIRE_NO_VALUE;
358 		}
359 		defaults.setAutowire(autowire);
360 
361 		// don't fall back to parentDefaults for dependency-check as it's no
362 		// longer supported in <beans> as of 3.0. Therefore, no nested <beans>
363 		// would ever need to fall back to it.
364 		defaults.setDependencyCheck(root.getAttribute(DEFAULT_DEPENDENCY_CHECK_ATTRIBUTE));
365 
366 		if (root.hasAttribute(DEFAULT_AUTOWIRE_CANDIDATES_ATTRIBUTE)) {
367 			defaults.setAutowireCandidates(root.getAttribute(DEFAULT_AUTOWIRE_CANDIDATES_ATTRIBUTE));
368 		}
369 		else if (parentDefaults != null) {
370 			defaults.setAutowireCandidates(parentDefaults.getAutowireCandidates());
371 		}
372 
373 		if (root.hasAttribute(DEFAULT_INIT_METHOD_ATTRIBUTE)) {
374 			defaults.setInitMethod(root.getAttribute(DEFAULT_INIT_METHOD_ATTRIBUTE));
375 		}
376 		else if (parentDefaults != null) {
377 			defaults.setInitMethod(parentDefaults.getInitMethod());
378 		}
379 
380 		if (root.hasAttribute(DEFAULT_DESTROY_METHOD_ATTRIBUTE)) {
381 			defaults.setDestroyMethod(root.getAttribute(DEFAULT_DESTROY_METHOD_ATTRIBUTE));
382 		}
383 		else if (parentDefaults != null) {
384 			defaults.setDestroyMethod(parentDefaults.getDestroyMethod());
385 		}
386 
387 		defaults.setSource(this.readerContext.extractSource(root));
388 	}

431 	/**
432 	 * Parses the supplied {@code &lt;bean&gt;} element. May return {@code null}
433 	 * if there were errors during parse. Errors are reported to the
434 	 * {@link org.springframework.beans.factory.parsing.ProblemReporter}.
435 	 */
436 	public BeanDefinitionHolder parseBeanDefinitionElement(Element ele, BeanDefinition containingBean) {
437 		String id = ele.getAttribute(ID_ATTRIBUTE);
438 		String nameAttr = ele.getAttribute(NAME_ATTRIBUTE);
439 
440 		List<String> aliases = new ArrayList<String>();
441 		if (StringUtils.hasLength(nameAttr)) {
442 			String[] nameArr = StringUtils.tokenizeToStringArray(nameAttr, MULTI_VALUE_ATTRIBUTE_DELIMITERS);
443 			aliases.addAll(Arrays.asList(nameArr));
444 		}
445 
446 		String beanName = id;
447 		if (!StringUtils.hasText(beanName) && !aliases.isEmpty()) {
448 			beanName = aliases.remove(0);
449 			if (logger.isDebugEnabled()) {
450 				logger.debug("No XML 'id' specified - using '" + beanName +
451 						"' as bean name and " + aliases + " as aliases");
452 			}
453 		}
454 
455 		if (containingBean == null) {
456 			checkNameUniqueness(beanName, aliases, ele);
457 		}
458 
459 		AbstractBeanDefinition beanDefinition = parseBeanDefinitionElement(ele, beanName, containingBean);
460 		if (beanDefinition != null) {
461 			if (!StringUtils.hasText(beanName)) {
462 				try {
463 					if (containingBean != null) {
464 						beanName = BeanDefinitionReaderUtils.generateBeanName(
465 								beanDefinition, this.readerContext.getRegistry(), true);
466 					}
467 					else {
468 						beanName = this.readerContext.generateBeanName(beanDefinition);
469 						// Register an alias for the plain bean class name, if still possible,
470 						// if the generator returned the class name plus a suffix.
471 						// This is expected for Spring 1.2/2.0 backwards compatibility.
472 						String beanClassName = beanDefinition.getBeanClassName();
473 						if (beanClassName != null &&
474 								beanName.startsWith(beanClassName) && beanName.length() > beanClassName.length() &&
475 								!this.readerContext.getRegistry().isBeanNameInUse(beanClassName)) {
476 							aliases.add(beanClassName);
477 						}
478 					}
479 					if (logger.isDebugEnabled()) {
480 						logger.debug("Neither XML 'id' nor 'name' specified - " +
481 								"using generated bean name [" + beanName + "]");
482 					}
483 				}
484 				catch (Exception ex) {
485 					error(ex.getMessage(), ele);
486 					return null;
487 				}
488 			}
489 			String[] aliasesArray = StringUtils.toStringArray(aliases);
490 			return new BeanDefinitionHolder(beanDefinition, beanName, aliasesArray);
491 		}
492 
493 		return null;
494 	}

496 	/**
497 	 * Validate that the specified bean name and aliases have not been used already
498 	 * within the current level of beans element nesting.
499 	 */
500 	protected void checkNameUniqueness(String beanName, List<String> aliases, Element beanElement) {
501 		String foundName = null;
502 
503 		if (StringUtils.hasText(beanName) && this.usedNames.contains(beanName)) {
504 			foundName = beanName;
505 		}
506 		if (foundName == null) {
507 			foundName = CollectionUtils.findFirstMatch(this.usedNames, aliases);
508 		}
509 		if (foundName != null) {
510 			error("Bean name '" + foundName + "' is already used in this <beans> element", beanElement);
511 		}
512 
513 		this.usedNames.add(beanName);
514 		this.usedNames.addAll(aliases);
515 	}

517 	/**
518 	 * Parse the bean definition itself, without regard to name or aliases. May return
519 	 * {@code null} if problems occurred during the parsing of the bean definition.
520 	 */
521 	public AbstractBeanDefinition parseBeanDefinitionElement(
522 			Element ele, String beanName, BeanDefinition containingBean) {
523 
524 		this.parseState.push(new BeanEntry(beanName));
525 
526 		String className = null;
527 		if (ele.hasAttribute(CLASS_ATTRIBUTE)) {
528 			className = ele.getAttribute(CLASS_ATTRIBUTE).trim();
529 		}
530 
531 		try {
532 			String parent = null;
533 			if (ele.hasAttribute(PARENT_ATTRIBUTE)) {
534 				parent = ele.getAttribute(PARENT_ATTRIBUTE);
535 			}
536 			AbstractBeanDefinition bd = createBeanDefinition(className, parent);
537 
538 			parseBeanDefinitionAttributes(ele, beanName, containingBean, bd);
539 			bd.setDescription(DomUtils.getChildElementValueByTagName(ele, DESCRIPTION_ELEMENT));
540 
541 			parseMetaElements(ele, bd);
542 			parseLookupOverrideSubElements(ele, bd.getMethodOverrides());
543 			parseReplacedMethodSubElements(ele, bd.getMethodOverrides());
544 
545 			parseConstructorArgElements(ele, bd);
546 			parsePropertyElements(ele, bd);
547 			parseQualifierElements(ele, bd);
548 
549 			bd.setResource(this.readerContext.getResource());
550 			bd.setSource(extractSource(ele));
551 
552 			return bd;
553 		}
554 		catch (ClassNotFoundException ex) {
555 			error("Bean class [" + className + "] not found", ele, ex);
556 		}
557 		catch (NoClassDefFoundError err) {
558 			error("Class that bean class [" + className + "] depends on not found", ele, err);
559 		}
560 		catch (Throwable ex) {
561 			error("Unexpected failure during bean definition parsing", ele, ex);
562 		}
563 		finally {
564 			this.parseState.pop();
565 		}
566 
567 		return null;
568 	}

570 	/**
571 	 * Apply the attributes of the given bean element to the given bean * definition.
572 	 * @param ele bean declaration element
573 	 * @param beanName bean name
574 	 * @param containingBean containing bean definition
575 	 * @return a bean definition initialized according to the bean element attributes
576 	 */
577 	public AbstractBeanDefinition parseBeanDefinitionAttributes(Element ele, String beanName,
578 			BeanDefinition containingBean, AbstractBeanDefinition bd) {
579 
580 		if (ele.hasAttribute(SINGLETON_ATTRIBUTE)) {
581 			error("Old 1.x 'singleton' attribute in use - upgrade to 'scope' declaration", ele);
582 		}
583 		else if (ele.hasAttribute(SCOPE_ATTRIBUTE)) {
584 			bd.setScope(ele.getAttribute(SCOPE_ATTRIBUTE));
585 		}
586 		else if (containingBean != null) {
587 			// Take default from containing bean in case of an inner bean definition.
588 			bd.setScope(containingBean.getScope());
589 		}
590 
591 		if (ele.hasAttribute(ABSTRACT_ATTRIBUTE)) {
592 			bd.setAbstract(TRUE_VALUE.equals(ele.getAttribute(ABSTRACT_ATTRIBUTE)));
593 		}
594 
595 		String lazyInit = ele.getAttribute(LAZY_INIT_ATTRIBUTE);
596 		if (DEFAULT_VALUE.equals(lazyInit)) {
597 			lazyInit = this.defaults.getLazyInit();
598 		}
599 		bd.setLazyInit(TRUE_VALUE.equals(lazyInit));
600 
601 		String autowire = ele.getAttribute(AUTOWIRE_ATTRIBUTE);
602 		bd.setAutowireMode(getAutowireMode(autowire));
603 
604 		String dependencyCheck = ele.getAttribute(DEPENDENCY_CHECK_ATTRIBUTE);
605 		bd.setDependencyCheck(getDependencyCheck(dependencyCheck));
606 
607 		if (ele.hasAttribute(DEPENDS_ON_ATTRIBUTE)) {
608 			String dependsOn = ele.getAttribute(DEPENDS_ON_ATTRIBUTE);
609 			bd.setDependsOn(StringUtils.tokenizeToStringArray(dependsOn, MULTI_VALUE_ATTRIBUTE_DELIMITERS));
610 		}
611 
612 		String autowireCandidate = ele.getAttribute(AUTOWIRE_CANDIDATE_ATTRIBUTE);
613 		if ("".equals(autowireCandidate) || DEFAULT_VALUE.equals(autowireCandidate)) {
614 			String candidatePattern = this.defaults.getAutowireCandidates();
615 			if (candidatePattern != null) {
616 				String[] patterns = StringUtils.commaDelimitedListToStringArray(candidatePattern);

948 	/**
949 	 * Get the value of a property element. May be a list etc.
950 	 * Also used for constructor arguments, "propertyName" being null in this case.
951 	 */
952 	public Object parsePropertyValue(Element ele, BeanDefinition bd, String propertyName) {
953 		String elementName = (propertyName != null) ?
954 						"<property> element for property '" + propertyName + "'" :
955 						"<constructor-arg> element";
956 
957 		// Should only have one child element: ref, value, list, etc.
958 		NodeList nl = ele.getChildNodes();
959 		Element subElement = null;
960 		for (int i = 0; i < nl.getLength(); i++) {
961 			Node node = nl.item(i);
962 			if (node instanceof Element && !nodeNameEquals(node, DESCRIPTION_ELEMENT) &&
963 					!nodeNameEquals(node, META_ELEMENT)) {
964 				// Child element is what we're looking for.
965 				if (subElement != null) {
966 					error(elementName + " must not contain more than one sub-element", ele);
967 				}
968 				else {
969 					subElement = (Element) node;
970 				}
971 			}
972 		}
973 
974 		boolean hasRefAttribute = ele.hasAttribute(REF_ATTRIBUTE);
975 		boolean hasValueAttribute = ele.hasAttribute(VALUE_ATTRIBUTE);
976 		if ((hasRefAttribute && hasValueAttribute) ||
977 				((hasRefAttribute || hasValueAttribute) && subElement != null)) {
978 			error(elementName +
979 					" is only allowed to contain either 'ref' attribute OR 'value' attribute OR sub-element", ele);
980 		}
981 
982 		if (hasRefAttribute) {
983 			String refName = ele.getAttribute(REF_ATTRIBUTE);
984 			if (!StringUtils.hasText(refName)) {
985 				error(elementName + " contains empty 'ref' attribute", ele);
986 			}
987 			RuntimeBeanReference ref = new RuntimeBeanReference(refName);
988 			ref.setSource(extractSource(ele));
989 			return ref;
990 		}
991 		else if (hasValueAttribute) {
992 			TypedStringValue valueHolder = new TypedStringValue(ele.getAttribute(VALUE_ATTRIBUTE));
993 			valueHolder.setSource(extractSource(ele));
994 			return valueHolder;
995 		}
996 		else if (subElement != null) {
997 			return parsePropertySubElement(subElement, bd);
998 		}
999 		else {
1000 			// Neither child element nor "ref" or "value" attribute found.
1001 			error(elementName + " must specify a ref or value", ele);
1002 			return null;
1003 		}
1004 	}

1010 	/**
1011 	 * Parse a value, ref or collection sub-element of a property or
1012 	 * constructor-arg element.
1013 	 * @param ele subelement of property element; we don't know which yet
1014 	 * @param defaultValueType the default type (class name) for any
1015 	 * {@code &lt;value&gt;} tag that might be created
1016 	 */
1017 	public Object parsePropertySubElement(Element ele, BeanDefinition bd, String defaultValueType) {
1018 		if (!isDefaultNamespace(ele)) {
1019 			return parseNestedCustomElement(ele, bd);
1020 		}
1021 		else if (nodeNameEquals(ele, BEAN_ELEMENT)) {
1022 			BeanDefinitionHolder nestedBd = parseBeanDefinitionElement(ele, bd);
1023 			if (nestedBd != null) {
1024 				nestedBd = decorateBeanDefinitionIfRequired(ele, nestedBd, bd);
1025 			}
1026 			return nestedBd;
1027 		}
1028 		else if (nodeNameEquals(ele, REF_ELEMENT)) {
1029 			// A generic reference to any name of any bean.
1030 			String refName = ele.getAttribute(BEAN_REF_ATTRIBUTE);
1031 			boolean toParent = false;
1032 			if (!StringUtils.hasLength(refName)) {
1033 				// A reference to the id of another bean in the same XML file.
1034 				refName = ele.getAttribute(LOCAL_REF_ATTRIBUTE);
1035 				if (!StringUtils.hasLength(refName)) {
1036 					// A reference to the id of another bean in a parent context.
1037 					refName = ele.getAttribute(PARENT_REF_ATTRIBUTE);
1038 					toParent = true;
1039 					if (!StringUtils.hasLength(refName)) {
1040 						error("'bean', 'local' or 'parent' is required for <ref> element", ele);
1041 						return null;
1042 					}
1043 				}
1044 			}
1045 			if (!StringUtils.hasText(refName)) {
1046 				error("<ref> element contains empty target attribute", ele);
1047 				return null;
1048 			}
1049 			RuntimeBeanReference ref = new RuntimeBeanReference(refName, toParent);
1050 			ref.setSource(extractSource(ele));
1051 			return ref;
1052 		}
1053 		else if (nodeNameEquals(ele, IDREF_ELEMENT)) {
1054 			return parseIdRefElement(ele);
1055 		}
1056 		else if (nodeNameEquals(ele, VALUE_ELEMENT)) {
1057 			return parseValueElement(ele, defaultValueType);
1058 		}
1059 		else if (nodeNameEquals(ele, NULL_ELEMENT)) {
1060 			// It's a distinguished null value. Let's wrap it in a TypedStringValue
1061 			// object in order to preserve the source location.
1062 			TypedStringValue nullHolder = new TypedStringValue(null);
1063 			nullHolder.setSource(extractSource(ele));
1064 			return nullHolder;
1065 		}
1066 		else if (nodeNameEquals(ele, ARRAY_ELEMENT)) {
1067 			return parseArrayElement(ele, bd);
1068 		}
1069 		else if (nodeNameEquals(ele, LIST_ELEMENT)) {
1070 			return parseListElement(ele, bd);
1071 		}
1072 		else if (nodeNameEquals(ele, SET_ELEMENT)) {
1073 			return parseSetElement(ele, bd);
1074 		}
1075 		else if (nodeNameEquals(ele, MAP_ELEMENT)) {
1076 			return parseMapElement(ele, bd);
1077 		}
1078 		else if (nodeNameEquals(ele, PROPS_ELEMENT)) {
1079 			return parsePropsElement(ele);
1080 		}
1081 		else {
1082 			error("Unknown property sub-element: [" + ele.getNodeName() + "]", ele);
1083 			return null;
1084 		}

1212 		String defaultKeyType = mapEle.getAttribute(KEY_TYPE_ATTRIBUTE);
1213 		String defaultValueType = mapEle.getAttribute(VALUE_TYPE_ATTRIBUTE);
1214 
1215 		List<Element> entryEles = DomUtils.getChildElementsByTagName(mapEle, ENTRY_ELEMENT);
1216 		ManagedMap<Object, Object> map = new ManagedMap<Object, Object>(entryEles.size());
1217 		map.setSource(extractSource(mapEle));
1218 		map.setKeyTypeName(defaultKeyType);
1219 		map.setValueTypeName(defaultValueType);
1220 		map.setMergeEnabled(parseMergeAttribute(mapEle));
1221 
1222 		for (Element entryEle : entryEles) {
1223 			// Should only have one value child element: ref, value, list, etc.
1224 			// Optionally, there might be a key child element.
1225 			NodeList entrySubNodes = entryEle.getChildNodes();
1226 			Element keyEle = null;
1227 			Element valueEle = null;
1228 			for (int j = 0; j < entrySubNodes.getLength(); j++) {
1229 				Node node = entrySubNodes.item(j);
1230 				if (node instanceof Element) {
1231 					Element candidateEle = (Element) node;
1232 					if (nodeNameEquals(candidateEle, KEY_ELEMENT)) {
1233 						if (keyEle != null) {
1234 							error("<entry> element is only allowed to contain one <key> sub-element", entryEle);
1235 						}
1236 						else {
1237 							keyEle = candidateEle;
1238 						}
1239 					}
1240 					else {
1241 						// Child element is what we're looking for.
1242 						if (nodeNameEquals(candidateEle, DESCRIPTION_ELEMENT)) {
1243 							// the element is a <description> -> ignore it
1244 						}
1245 						else if (valueEle != null) {
1246 							error("<entry> element must not contain more than one value sub-element", entryEle);
1247 						}
1248 						else {
1249 							valueEle = candidateEle;
1250 						}
1251 					}
1252 				}
1253 			}
1254 
1255 			// Extract key from attribute or sub-element.
1256 			Object key = null;
1257 			boolean hasKeyAttribute = entryEle.hasAttribute(KEY_ATTRIBUTE);
1258 			boolean hasKeyRefAttribute = entryEle.hasAttribute(KEY_REF_ATTRIBUTE);
1259 			if ((hasKeyAttribute && hasKeyRefAttribute) ||
1260 					((hasKeyAttribute || hasKeyRefAttribute)) && keyEle != null) {
1261 				error("<entry> element is only allowed to contain either " +
1262 						"a 'key' attribute OR a 'key-ref' attribute OR a <key> sub-element", entryEle);
1263 			}
1264 			if (hasKeyAttribute) {
1265 				key = buildTypedStringValueForMap(entryEle.getAttribute(KEY_ATTRIBUTE), defaultKeyType, entryEle);
1266 			}
1267 			else if (hasKeyRefAttribute) {
1268 				String refName = entryEle.getAttribute(KEY_REF_ATTRIBUTE);
1269 				if (!StringUtils.hasText(refName)) {
1270 					error("<entry> element contains empty 'key-ref' attribute", entryEle);
1271 				}
1272 				RuntimeBeanReference ref = new RuntimeBeanReference(refName);
1273 				ref.setSource(extractSource(entryEle));
1274 				key = ref;
1275 			}
1276 			else if (keyEle != null) {
1277 				key = parseKeyElement(keyEle, bd, defaultKeyType);
1278 			}
1279 			else {
1280 				error("<entry> element must specify a key", entryEle);
1281 			}
1282 
1283 			// Extract value from attribute or sub-element.
1284 			Object value = null;
1285 			boolean hasValueAttribute = entryEle.hasAttribute(VALUE_ATTRIBUTE);
1286 			boolean hasValueRefAttribute = entryEle.hasAttribute(VALUE_REF_ATTRIBUTE);
1287 			boolean hasValueTypeAttribute = entryEle.hasAttribute(VALUE_TYPE_ATTRIBUTE);
1288 			if ((hasValueAttribute && hasValueRefAttribute) ||
1289 					((hasValueAttribute || hasValueRefAttribute)) && valueEle != null) {
1290 				error("<entry> element is only allowed to contain either " +
1291 						"'value' attribute OR 'value-ref' attribute OR <value> sub-element", entryEle);
1292 			}
1293 			if ((hasValueTypeAttribute && hasValueRefAttribute) ||
1294 				(hasValueTypeAttribute && !hasValueAttribute) ||
1295 					(hasValueTypeAttribute && valueEle != null)) {
1296 				error("<entry> element is only allowed to contain a 'value-type' " +
1297 						"attribute when it has a 'value' attribute", entryEle);
1298 			}
1299 			if (hasValueAttribute) {
1300 				String valueType = entryEle.getAttribute(VALUE_TYPE_ATTRIBUTE);
1301 				if (!StringUtils.hasText(valueType)) {
1302 					valueType = defaultValueType;
1303 				}
1304 				value = buildTypedStringValueForMap(entryEle.getAttribute(VALUE_ATTRIBUTE), valueType, entryEle);
1305 			}
1306 			else if (hasValueRefAttribute) {
1307 				String refName = entryEle.getAttribute(VALUE_REF_ATTRIBUTE);
1308 				if (!StringUtils.hasText(refName)) {
1309 					error("<entry> element contains empty 'value-ref' attribute", entryEle);
1310 				}
1311 				RuntimeBeanReference ref = new RuntimeBeanReference(refName);
1312 				ref.setSource(extractSource(entryEle));
1313 				value = ref;
1314 			}
1315 			else if (valueEle != null) {
1316 				value = parsePropertySubElement(valueEle, bd, defaultValueType);
1317 			}
1318 			else {
1319 				error("<entry> element must specify a value", entryEle);
1320 			}
1321 
1322 			// Add final key and value to the Map.
1323 			map.put(key, value);
1324 		}
1325 
1326 		return map;
1327 	}
