174 	private BeanDefinitionRegistry beanFactoryFor(String xmlName, String... activeProfiles) {
175 		DefaultListableBeanFactory beanFactory = new DefaultListableBeanFactory();
176 		XmlBeanDefinitionReader reader = new XmlBeanDefinitionReader(beanFactory);
177 		StandardEnvironment env = new StandardEnvironment();
178 		env.setActiveProfiles(activeProfiles);
179 		reader.setEnvironment(env);
180 		reader.loadBeanDefinitions(new ClassPathResource(xmlName, getClass()));
181 		return beanFactory;
182 	}

185 	private static Matcher<BeanDefinitionRegistry> containsBeanDefinition(final String beanName) {
186 		return new TypeSafeMatcher<BeanDefinitionRegistry>() {
187 
188 			@Override
189 			public void describeTo(Description desc) {
190 				desc.appendText("a BeanDefinitionRegistry containing bean named ")
191 					.appendValue(beanName);
192 			}
193 
194 			@Override
195 			public boolean matchesSafely(BeanDefinitionRegistry beanFactory) {
196 				return beanFactory.containsBeanDefinition(beanName);
197 			}
198 
199 		};
200 	}
201 
202 	private static Matcher<BeanDefinitionRegistry> containsTargetBean() {
203 		return containsBeanDefinition(TARGET_BEAN);
204 	}
