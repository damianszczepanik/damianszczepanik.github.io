47  	@Test
48  	public void testAutowireCandidateDefaultWithIrrelevantDescriptor() throws Exception {
49  		DefaultListableBeanFactory lbf = new DefaultListableBeanFactory();
50  		ConstructorArgumentValues cavs = new ConstructorArgumentValues();
51  		cavs.addGenericArgumentValue(JUERGEN);
52  		RootBeanDefinition rbd = new RootBeanDefinition(Person.class, cavs, null);
53  		lbf.registerBeanDefinition(JUERGEN, rbd);
54  		assertTrue(lbf.isAutowireCandidate(JUERGEN, null));
55  		assertTrue(lbf.isAutowireCandidate(JUERGEN,
56  				new DependencyDescriptor(Person.class.getDeclaredField("name"), false)));
57  		assertTrue(lbf.isAutowireCandidate(JUERGEN,
58  				new DependencyDescriptor(Person.class.getDeclaredField("name"), true)));
59  	}
