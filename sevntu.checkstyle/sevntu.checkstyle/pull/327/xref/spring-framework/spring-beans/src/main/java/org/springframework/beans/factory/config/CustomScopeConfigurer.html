17  package org.springframework.beans.factory.config;
18  
19  import java.util.LinkedHashMap;
20  import java.util.Map;
21  
22  import org.springframework.beans.BeanUtils;
23  import org.springframework.beans.BeansException;
24  import org.springframework.beans.factory.BeanClassLoaderAware;
25  import org.springframework.core.Ordered;
26  import org.springframework.util.Assert;
27  import org.springframework.util.ClassUtils;
28  
29  /**
30   * Simple {@link BeanFactoryPostProcessor} implementation that registers
31   * custom {@link Scope Scope(s)} with the containing {@link ConfigurableBeanFactory}.
32   *
33   * <p>Will register all of the supplied {@link #setScopes(java.util.Map) scopes}
34   * with the {@link ConfigurableListableBeanFactory} that is passed to the
35   * {@link #postProcessBeanFactory(ConfigurableListableBeanFactory)} method.
36   *
37   * <p>This class allows for <i>declarative</i> registration of custom scopes.
38   * Alternatively, consider implementing a custom {@link BeanFactoryPostProcessor}
39   * that calls {@link ConfigurableBeanFactory#registerScope} programmatically.
40   *
41   * @author Juergen Hoeller
42   * @author Rick Evans
43   * @since 2.0
44   * @see ConfigurableBeanFactory#registerScope
45   */
46  public class CustomScopeConfigurer implements BeanFactoryPostProcessor, BeanClassLoaderAware, Ordered {
47  
48  	private Map<String, Object> scopes;
49  
50  	private int order = Ordered.LOWEST_PRECEDENCE;
51  
52  	private ClassLoader beanClassLoader = ClassUtils.getDefaultClassLoader();
53  
54  
55  	/**
56  	 * Specify the custom scopes that are to be registered.
57  	 * <p>The keys indicate the scope names (of type String); each value
58  	 * is expected to be the corresponding custom {@link Scope} instance
59  	 * or class name.
60  	 */
61  	public void setScopes(Map<String, Object> scopes) {
62  		this.scopes = scopes;
63  	}
64  
65  	/**
66  	 * Add the given scope to this configurer's map of scopes.
67  	 * @param scopeName the name of the scope
68  	 * @param scope the scope implementation
69  	 * @since 4.1.1
70  	 */
71  	public void addScope(String scopeName, Scope scope) {
72  		if (this.scopes == null) {
73  			this.scopes = new LinkedHashMap<String, Object>(1);
74  		}
75  		this.scopes.put(scopeName, scope);
76  	}
77  
78  
79  	public void setOrder(int order) {
80  		this.order = order;
81  	}
82  
83  	@Override
84  	public int getOrder() {
85  		return this.order;
86  	}
87  
88  	@Override
89  	public void setBeanClassLoader(ClassLoader beanClassLoader) {
90  		this.beanClassLoader = beanClassLoader;
91  	}
92  
93  
94  	@Override
95  	public void postProcessBeanFactory(ConfigurableListableBeanFactory beanFactory) throws BeansException {
96  		if (this.scopes != null) {
97  			for (Map.Entry<String, Object> entry : this.scopes.entrySet()) {
98  				String scopeKey = entry.getKey();
99  				Object value = entry.getValue();
100 				if (value instanceof Scope) {
101 					beanFactory.registerScope(scopeKey, (Scope) value);
102 				}
103 				else if (value instanceof Class) {
104 					Class<?> scopeClass = (Class<?>) value;
105 					Assert.isAssignable(Scope.class, scopeClass);
106 					beanFactory.registerScope(scopeKey, (Scope) BeanUtils.instantiateClass(scopeClass));
107 				}
108 				else if (value instanceof String) {
109 					Class<?> scopeClass = ClassUtils.resolveClassName((String) value, this.beanClassLoader);
110 					Assert.isAssignable(Scope.class, scopeClass);
111 					beanFactory.registerScope(scopeKey, (Scope) BeanUtils.instantiateClass(scopeClass));
112 				}
113 				else {
114 					throw new IllegalArgumentException("Mapped value [" + value + "] for scope key [" +
115 							scopeKey + "] is not an instance of required type [" + Scope.class.getName() +
116 							"] or a corresponding Class or String value indicating a Scope implementation");
117 				}
118 			}
119 		}
120 	}
121 
122 }
