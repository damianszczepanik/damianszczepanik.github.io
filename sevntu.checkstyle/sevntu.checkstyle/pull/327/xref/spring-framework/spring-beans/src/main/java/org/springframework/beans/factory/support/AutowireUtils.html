52  	/**
53  	 * Sort the given constructors, preferring public constructors and "greedy" ones with
54  	 * a maximum number of arguments. The result will contain public constructors first,
55  	 * with decreasing number of arguments, then non-public constructors, again with
56  	 * decreasing number of arguments.
57  	 * @param constructors the constructor array to sort
58  	 */
59  	public static void sortConstructors(Constructor<?>[] constructors) {
60  		Arrays.sort(constructors, new Comparator<Constructor<?>>() {
61  			@Override
62  			public int compare(Constructor<?> c1, Constructor<?> c2) {
63  				boolean p1 = Modifier.isPublic(c1.getModifiers());
64  				boolean p2 = Modifier.isPublic(c2.getModifiers());
65  				if (p1 != p2) {
66  					return (p1 ? -1 : 1);
67  				}
68  				int c1pl = c1.getParameterTypes().length;
69  				int c2pl = c2.getParameterTypes().length;
70  				return (c1pl < c2pl ? 1 : (c1pl > c2pl ? -1 : 0));
71  			}
72  		});
73  	}
74  
75  	/**
76  	 * Sort the given factory methods, preferring public methods and "greedy" ones
77  	 * with a maximum of arguments. The result will contain public methods first,
78  	 * with decreasing number of arguments, then non-public methods, again with
79  	 * decreasing number of arguments.
80  	 * @param factoryMethods the factory method array to sort
81  	 */
82  	public static void sortFactoryMethods(Method[] factoryMethods) {
83  		Arrays.sort(factoryMethods, new Comparator<Method>() {
84  			@Override
85  			public int compare(Method fm1, Method fm2) {
86  				boolean p1 = Modifier.isPublic(fm1.getModifiers());
87  				boolean p2 = Modifier.isPublic(fm2.getModifiers());
88  				if (p1 != p2) {
89  					return (p1 ? -1 : 1);
90  				}
91  				int c1pl = fm1.getParameterTypes().length;
92  				int c2pl = fm2.getParameterTypes().length;
93  				return (c1pl < c2pl ? 1 : (c1pl > c2pl ? -1 : 0));
94  			}
95  		});
96  	}

98  	/**
99  	 * Determine whether the given bean property is excluded from dependency checks.
100 	 * <p>This implementation excludes properties defined by CGLIB.
101 	 * @param pd the PropertyDescriptor of the bean property
102 	 * @return whether the bean property is excluded
103 	 */
104 	public static boolean isExcludedFromDependencyCheck(PropertyDescriptor pd) {
105 		Method wm = pd.getWriteMethod();
106 		if (wm == null) {
107 			return false;
108 		}
109 		if (!wm.getDeclaringClass().getName().contains("$$")) {
110 			// Not a CGLIB method so it's OK.
111 			return false;
112 		}
113 		// It was declared by CGLIB, but we might still want to autowire it
114 		// if it was actually declared by the superclass.
115 		Class<?> superclass = wm.getDeclaringClass().getSuperclass();
116 		return !ClassUtils.hasMethod(superclass, wm.getName(), wm.getParameterTypes());
117 	}
118 
119 	/**
120 	 * Return whether the setter method of the given bean property is defined
121 	 * in any of the given interfaces.
122 	 * @param pd the PropertyDescriptor of the bean property
123 	 * @param interfaces the Set of interfaces (Class objects)
124 	 * @return whether the setter method is defined by an interface
125 	 */
126 	public static boolean isSetterDefinedInInterface(PropertyDescriptor pd, Set<Class<?>> interfaces) {
127 		Method setter = pd.getWriteMethod();
128 		if (setter != null) {
129 			Class<?> targetClass = setter.getDeclaringClass();
130 			for (Class<?> ifc : interfaces) {
131 				if (ifc.isAssignableFrom(targetClass) &&
132 						ClassUtils.hasMethod(ifc, setter.getName(), setter.getParameterTypes())) {
133 					return true;
134 				}
135 			}
136 		}
137 		return false;
138 	}

140 	/**
141 	 * Resolve the given autowiring value against the given required type,
142 	 * e.g. an {@link ObjectFactory} value to its actual object result.
143 	 * @param autowiringValue the value to resolve
144 	 * @param requiredType the type to assign the result to
145 	 * @return the resolved value
146 	 */
147 	public static Object resolveAutowiringValue(Object autowiringValue, Class<?> requiredType) {
148 		if (autowiringValue instanceof ObjectFactory && !requiredType.isInstance(autowiringValue)) {
149 			ObjectFactory<?> factory = (ObjectFactory<?>) autowiringValue;
150 			if (autowiringValue instanceof Serializable && requiredType.isInterface()) {
151 				autowiringValue = Proxy.newProxyInstance(requiredType.getClassLoader(),
152 						new Class<?>[] {requiredType}, new ObjectFactoryDelegatingInvocationHandler(factory));
153 			}
154 			else {
155 				return factory.getObject();
156 			}
157 		}
158 		return autowiringValue;
159 	}

161 	/**
162 	 * Determine the target type for the generic return type of the given
163 	 * <em>generic factory method</em>, where formal type variables are declared
164 	 * on the given method itself.
165 	 * <p>For example, given a factory method with the following signature,
166 	 * if {@code resolveReturnTypeForFactoryMethod()} is invoked with the reflected
167 	 * method for {@code creatProxy()} and an {@code Object[]} array containing
168 	 * {@code MyService.class}, {@code resolveReturnTypeForFactoryMethod()} will
169 	 * infer that the target return type is {@code MyService}.
170 	 * <pre class="code">{@code public static <T> T createProxy(Class<T> clazz)}</pre>
171 	 * <h4>Possible Return Values</h4>
172 	 * <ul>
173 	 * <li>the target return type, if it can be inferred</li>
174 	 * <li>the {@linkplain Method#getReturnType() standard return type}, if
175 	 * the given {@code method} does not declare any {@linkplain
176 	 * Method#getTypeParameters() formal type variables}</li>
177 	 * <li>the {@linkplain Method#getReturnType() standard return type}, if the
178 	 * target return type cannot be inferred (e.g., due to type erasure)</li>
179 	 * <li>{@code null}, if the length of the given arguments array is shorter
180 	 * than the length of the {@linkplain
181 	 * Method#getGenericParameterTypes() formal argument list} for the given
182 	 * method</li>
183 	 * </ul>
184 	 * @param method the method to introspect (never {@code null})
185 	 * @param args the arguments that will be supplied to the method when it is
186 	 * invoked (never {@code null})
187 	 * @param classLoader the ClassLoader to resolve class names against, if necessary
188 	 * (never {@code null})
189 	 * @return the resolved target return type, the standard return type, or {@code null}
190 	 * @since 3.2.5
191 	 */
192 	public static Class<?> resolveReturnTypeForFactoryMethod(Method method, Object[] args, ClassLoader classLoader) {
193 		Assert.notNull(method, "Method must not be null");
194 		Assert.notNull(args, "Argument array must not be null");
195 		Assert.notNull(classLoader, "ClassLoader must not be null");
196 
197 		TypeVariable<Method>[] declaredTypeVariables = method.getTypeParameters();
198 		Type genericReturnType = method.getGenericReturnType();
199 		Type[] methodParameterTypes = method.getGenericParameterTypes();
200 		Assert.isTrue(args.length == methodParameterTypes.length, "Argument array does not match parameter count");
201 
202 		// Ensure that the type variable (e.g., T) is declared directly on the method
203 		// itself (e.g., via <T>), not on the enclosing class or interface.
204 		boolean locallyDeclaredTypeVariableMatchesReturnType = false;
205 		for (TypeVariable<Method> currentTypeVariable : declaredTypeVariables) {
206 			if (currentTypeVariable.equals(genericReturnType)) {
207 				locallyDeclaredTypeVariableMatchesReturnType = true;
208 				break;
209 			}
210 		}
211 
212 		if (locallyDeclaredTypeVariableMatchesReturnType) {
213 			for (int i = 0; i < methodParameterTypes.length; i++) {
214 				Type methodParameterType = methodParameterTypes[i];
215 				Object arg = args[i];
216 				if (methodParameterType.equals(genericReturnType)) {
217 					if (arg instanceof TypedStringValue) {
218 						TypedStringValue typedValue = ((TypedStringValue) arg);
219 						if (typedValue.hasTargetType()) {
220 							return typedValue.getTargetType();
221 						}
222 						try {
223 							return typedValue.resolveTargetType(classLoader);
224 						}
225 						catch (ClassNotFoundException ex) {
226 							throw new IllegalStateException("Failed to resolve value type [" +
227 									typedValue.getTargetTypeName() + "] for factory method argument", ex);
228 						}
229 					}
230 					// Only consider argument type if it is a simple value...
231 					if (arg != null && !(arg instanceof BeanMetadataElement)) {
232 						return arg.getClass();
233 					}
234 					return method.getReturnType();
235 				}
236 				else if (methodParameterType instanceof ParameterizedType) {
237 					ParameterizedType parameterizedType = (ParameterizedType) methodParameterType;
238 					Type[] actualTypeArguments = parameterizedType.getActualTypeArguments();
239 					for (Type typeArg : actualTypeArguments) {
240 						if (typeArg.equals(genericReturnType)) {
241 							if (arg instanceof Class) {
242 								return (Class<?>) arg;
243 							}
244 							else {
245 								String className = null;
246 								if (arg instanceof String) {
247 									className = (String) arg;
248 								}
249 								else if (arg instanceof TypedStringValue) {
250 									TypedStringValue typedValue = ((TypedStringValue) arg);
251 									String targetTypeName = typedValue.getTargetTypeName();
252 									if (targetTypeName == null || Class.class.getName().equals(targetTypeName)) {
253 										className = typedValue.getValue();
254 									}
255 								}
256 								if (className != null) {
257 									try {
258 										return ClassUtils.forName(className, classLoader);
259 									}
260 									catch (ClassNotFoundException ex) {
261 										throw new IllegalStateException("Could not resolve class name [" + arg +
262 												"] for factory method argument", ex);
263 									}
264 								}
265 								// Consider adding logic to determine the class of the typeArg, if possible.
266 								// For now, just fall back...
267 								return method.getReturnType();
268 							}
269 						}
270 					}
271 				}
272 			}
273 		}
274 
275 		// Fall back...
276 		return method.getReturnType();
277 	}

280 	/**
281 	 * Reflective InvocationHandler for lazy access to the current target object.
282 	 */
283 	@SuppressWarnings("serial")
284 	private static class ObjectFactoryDelegatingInvocationHandler implements InvocationHandler, Serializable {
285 
286 		private final ObjectFactory<?> objectFactory;
287 
288 		public ObjectFactoryDelegatingInvocationHandler(ObjectFactory<?> objectFactory) {
289 			this.objectFactory = objectFactory;
290 		}
291 
292 		@Override
293 		public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
294 			String methodName = method.getName();
295 			if (methodName.equals("equals")) {
296 				// Only consider equal when proxies are identical.
297 				return (proxy == args[0]);
298 			}
299 			else if (methodName.equals("hashCode")) {
300 				// Use hashCode of proxy.
301 				return System.identityHashCode(proxy);
302 			}
303 			else if (methodName.equals("toString")) {
304 				return this.objectFactory.toString();
305 			}
306 			try {
307 				return method.invoke(this.objectFactory.getObject(), args);
308 			}
309 			catch (InvocationTargetException ex) {
310 				throw ex.getTargetException();
311 			}
312 		}
313 	}
