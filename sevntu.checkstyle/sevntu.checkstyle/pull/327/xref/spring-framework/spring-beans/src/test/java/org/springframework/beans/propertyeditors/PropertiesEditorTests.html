17  package org.springframework.beans.propertyeditors;
18  
19  import java.util.HashMap;
20  import java.util.Map;
21  import java.util.Properties;
22  
23  import junit.framework.TestCase;
24  
25  /**
26   * Test the conversion of Strings to {@link java.util.Properties} objects,
27   * and other property editors.
28   *
29   * @author Rod Johnson
30   * @author Juergen Hoeller
31   * @author Rick Evans
32   */
33  public class PropertiesEditorTests extends TestCase {
34  
35  	public void testOneProperty() {
36  		String s = "foo=bar";
37  		PropertiesEditor pe= new PropertiesEditor();
38  		pe.setAsText(s);
39  		Properties p = (Properties) pe.getValue();
40  		assertTrue("contains one entry", p.entrySet().size() == 1);
41  		assertTrue("foo=bar", p.get("foo").equals("bar"));
42  	}
43  
44  	public void testTwoProperties() {
45  		String s = "foo=bar with whitespace\n" +
46  			"me=mi";
47  		PropertiesEditor pe= new PropertiesEditor();
48  		pe.setAsText(s);
49  		Properties p = (Properties) pe.getValue();
50  		assertTrue("contains two entries", p.entrySet().size() == 2);
51  		assertTrue("foo=bar with whitespace", p.get("foo").equals("bar with whitespace"));
52  		assertTrue("me=mi", p.get("me").equals("mi"));
53  	}
54  
55  	public void testHandlesEqualsInValue() {
56  		String s = "foo=bar\n" +
57  			"me=mi\n" +
58  			"x=y=z";
59  		PropertiesEditor pe= new PropertiesEditor();
60  		pe.setAsText(s);
61  		Properties p = (Properties) pe.getValue();
62  		assertTrue("contains two entries", p.entrySet().size() == 3);
63  		assertTrue("foo=bar", p.get("foo").equals("bar"));
64  		assertTrue("me=mi", p.get("me").equals("mi"));
65  		assertTrue("x='y=z'", p.get("x").equals("y=z"));
66  	}
67  
68  	public void testHandlesEmptyProperty() {
69  		String s = "foo=bar\nme=mi\nx=";
70  		PropertiesEditor pe= new PropertiesEditor();
71  		pe.setAsText(s);
72  		Properties p = (Properties) pe.getValue();
73  		assertTrue("contains two entries", p.entrySet().size() == 3);
74  		assertTrue("foo=bar", p.get("foo").equals("bar"));
75  		assertTrue("me=mi", p.get("me").equals("mi"));
76  		assertTrue("x='y=z'", p.get("x").equals(""));
77  	}
78  
79  	public void testHandlesEmptyPropertyWithoutEquals() {
80  		String s = "foo\nme=mi\nx=x";
81  		PropertiesEditor pe= new PropertiesEditor();
82  		pe.setAsText(s);
83  		Properties p = (Properties) pe.getValue();
84  		assertTrue("contains three entries", p.entrySet().size() == 3);
85  		assertTrue("foo is empty", p.get("foo").equals(""));
86  		assertTrue("me=mi", p.get("me").equals("mi"));
87  	}
88  
89  	/**
90  	 * Comments begin with #
91  	 */
92  	public void testIgnoresCommentLinesAndEmptyLines() {
93  		String s = "#Ignore this comment\n" +
94  			"foo=bar\n" +
95  			"#Another=comment more junk /\n" +
96  			"me=mi\n" +
97  			"x=x\n" +
98  			"\n";
99  		PropertiesEditor pe= new PropertiesEditor();
100 		pe.setAsText(s);
101 		Properties p = (Properties) pe.getValue();
102 		assertTrue("contains three entries", p.entrySet().size() == 3);
103 		assertTrue("foo is bar", p.get("foo").equals("bar"));
104 		assertTrue("me=mi", p.get("me").equals("mi"));
105 	}
106 
107 	/**
108 	 * We'll typically align by indenting with tabs or spaces.
109 	 * These should be ignored if at the beginning of a line.
110 	 * We must ensure that comment lines beginning with whitespace are
111 	 * still ignored: The standard syntax doesn't allow this on JDK 1.3.
112 	 */
113 	public void testIgnoresLeadingSpacesAndTabs() {
114 		String s = "    #Ignore this comment\n" +
115 			"\t\tfoo=bar\n" +
116 			"\t#Another comment more junk \n" +
117 			" me=mi\n" +
118 			"x=x\n" +
119 			"\n";
120 		PropertiesEditor pe= new PropertiesEditor();
121 		pe.setAsText(s);
122 		Properties p = (Properties) pe.getValue();
123 		assertTrue("contains 3 entries, not " + p.size(), p.size() == 3);
124 		assertTrue("foo is bar", p.get("foo").equals("bar"));
125 		assertTrue("me=mi", p.get("me").equals("mi"));
126 	}
127 
128 	public void testNull() {
129 		PropertiesEditor pe= new PropertiesEditor();
130 		pe.setAsText(null);
131 		Properties p = (Properties) pe.getValue();
132 		assertEquals(0, p.size());
133 	}
134 
135 	public void testEmptyString() {
136 		PropertiesEditor pe = new PropertiesEditor();
137 		pe.setAsText("");
138 		Properties p = (Properties) pe.getValue();
139 		assertTrue("empty string means empty properties", p.isEmpty());
140 	}
141 
142 	public void testUsingMapAsValueSource() throws Exception {
143 		Map map = new HashMap();
144 		map.put("one", "1");
145 		map.put("two", "2");
146 		map.put("three", "3");
147 		PropertiesEditor pe = new PropertiesEditor();
148 		pe.setValue(map);
149 		Object value = pe.getValue();
150 		assertNotNull(value);
151 		assertTrue(value instanceof Properties);
152 		Properties props = (Properties) value;
153 		assertEquals(3, props.size());
154 		assertEquals("1", props.getProperty("one"));
155 		assertEquals("2", props.getProperty("two"));
156 		assertEquals("3", props.getProperty("three"));
157 	}
