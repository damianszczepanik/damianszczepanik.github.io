17  package org.springframework.beans.factory.annotation;
18  
19  import org.springframework.beans.factory.config.BeanDefinition;
20  import org.springframework.core.type.AnnotationMetadata;
21  import org.springframework.core.type.MethodMetadata;
22  
23  /**
24   * Extended {@link org.springframework.beans.factory.config.BeanDefinition}
25   * interface that exposes {@link org.springframework.core.type.AnnotationMetadata}
26   * about its bean class - without requiring the class to be loaded yet.
27   *
28   * @author Juergen Hoeller
29   * @since 2.5
30   * @see AnnotatedGenericBeanDefinition
31   * @see org.springframework.core.type.AnnotationMetadata
32   */
33  public interface AnnotatedBeanDefinition extends BeanDefinition {
34  
35  	/**
36  	 * Obtain the annotation metadata (as well as basic class metadata)
37  	 * for this bean definition's bean class.
38  	 * @return the annotation metadata object (never {@code null})
39  	 */
40  	AnnotationMetadata getMetadata();
41  
42  	/**
43  	 * Obtain metadata for this bean definition's factory method, if any.
44  	 * @return the factory method metadata, or {@code null} if none
45  	 * @since 4.1.1
46  	 */
47  	MethodMetadata getFactoryMethodMetadata();
48  
49  }
