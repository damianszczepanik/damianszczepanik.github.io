17  package org.springframework.beans;
18  
19  import org.springframework.core.convert.ConversionService;
20  
21  /**
22   * Interface that encapsulates configuration methods for a PropertyAccessor.
23   * Also extends the PropertyEditorRegistry interface, which defines methods
24   * for PropertyEditor management.
25   *
26   * <p>Serves as base interface for {@link BeanWrapper}.
27   *
28   * @author Juergen Hoeller
29   * @author Stephane Nicoll
30   * @since 2.0
31   * @see BeanWrapper
32   */
33  public interface ConfigurablePropertyAccessor extends PropertyAccessor, PropertyEditorRegistry, TypeConverter {
34  
35  	/**
36  	 * Specify a Spring 3.0 ConversionService to use for converting
37  	 * property values, as an alternative to JavaBeans PropertyEditors.
38  	 */
39  	void setConversionService(ConversionService conversionService);
40  
41  	/**
42  	 * Return the associated ConversionService, if any.
43  	 */
44  	ConversionService getConversionService();
45  
46  	/**
47  	 * Set whether to extract the old property value when applying a
48  	 * property editor to a new value for a property.
49  	 */
50  	void setExtractOldValueForEditor(boolean extractOldValueForEditor);
51  
52  	/**
53  	 * Return whether to extract the old property value when applying a
54  	 * property editor to a new value for a property.
55  	 */
56  	boolean isExtractOldValueForEditor();
57  
58  	/**
59  	 * Set whether this instance should attempt to "auto-grow" a
60  	 * nested path that contains a {@code null} value.
61  	 * <p>If {@code true}, a {@code null} path location will be populated
62  	 * with a default object value and traversed instead of resulting in a
63  	 * {@link NullValueInNestedPathException}.
64  	 * <p>Default is {@code false} on a plain PropertyAccessor instance.
65  	 */
66  	void setAutoGrowNestedPaths(boolean autoGrowNestedPaths);
67  
68  	/**
69  	 * Return whether "auto-growing" of nested paths has been activated.
70  	 */
71  	boolean isAutoGrowNestedPaths();
72  
73  }
