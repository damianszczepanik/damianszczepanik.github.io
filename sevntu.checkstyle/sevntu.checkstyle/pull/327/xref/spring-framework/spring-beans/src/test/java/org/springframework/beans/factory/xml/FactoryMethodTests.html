162 		DefaultListableBeanFactory xbf = new DefaultListableBeanFactory();
163 		XmlBeanDefinitionReader reader = new XmlBeanDefinitionReader(xbf);
164 		reader.loadBeanDefinitions(new ClassPathResource("factory-methods.xml", getClass()));
165 		FactoryMethods fm = (FactoryMethods) xbf.getBean("defaultPrototype");
166 		FactoryMethods fm2 = (FactoryMethods) xbf.getBean("defaultPrototype");
167 		assertEquals(0, fm.getNum());
168 		assertEquals("default", fm.getName());
169 		assertEquals("defaultInstance", fm.getTestBean().getName());
170 		assertEquals("setterString", fm.getStringValue());
171 		assertEquals(fm.getNum(), fm2.getNum());
172 		assertEquals(fm.getStringValue(), fm2.getStringValue());
173 		// The TestBean is created separately for each bean
174 		assertNotSame(fm.getTestBean(), fm2.getTestBean());
175 		assertNotSame(fm, fm2);
176 
177 		fm = (FactoryMethods) xbf.getBean("testBeanOnlyPrototype");
178 		fm2 = (FactoryMethods) xbf.getBean("testBeanOnlyPrototype");
179 		assertEquals(0, fm.getNum());
180 		assertEquals("default", fm.getName());
181 		// This comes from the test bean
182 		assertEquals("Juergen", fm.getTestBean().getName());
183 		assertEquals(fm.getNum(), fm2.getNum());
184 		assertEquals(fm.getStringValue(), fm2.getStringValue());
185 		// The TestBean reference is resolved to a prototype in the factory
186 		assertSame(fm.getTestBean(), fm2.getTestBean());
187 		assertNotSame(fm, fm2);
188 
189 		fm = (FactoryMethods) xbf.getBean("fullPrototype");
190 		fm2 = (FactoryMethods) xbf.getBean("fullPrototype");
191 		assertEquals(27, fm.getNum());
192 		assertEquals("gotcha", fm.getName());
193 		assertEquals("Juergen", fm.getTestBean().getName());
194 		assertEquals(fm.getNum(), fm2.getNum());
195 		assertEquals(fm.getStringValue(), fm2.getStringValue());
196 		// The TestBean reference is resolved to a prototype in the factory
197 		assertSame(fm.getTestBean(), fm2.getTestBean());
198 		assertNotSame(fm, fm2);

291 	public void testCanSpecifyFactoryMethodArgumentsOnFactoryMethodPrototype() {
292 		DefaultListableBeanFactory xbf = new DefaultListableBeanFactory();
293 		XmlBeanDefinitionReader reader = new XmlBeanDefinitionReader(xbf);
294 		reader.loadBeanDefinitions(new ClassPathResource("factory-methods.xml", getClass()));
295 		TestBean tbArg = new TestBean();
296 		tbArg.setName("arg1");
297 		TestBean tbArg2 = new TestBean();
298 		tbArg2.setName("arg2");
299 
300 		FactoryMethods fm1 = (FactoryMethods) xbf.getBean("testBeanOnlyPrototype", tbArg);
301 		assertEquals(0, fm1.getNum());
302 		assertEquals("default", fm1.getName());
303 		// This comes from the test bean
304 		assertEquals("arg1", fm1.getTestBean().getName());
305 
306 		FactoryMethods fm2 = (FactoryMethods) xbf.getBean("testBeanOnlyPrototype", tbArg2);
307 		assertEquals("arg2", fm2.getTestBean().getName());
308 		assertEquals(fm1.getNum(), fm2.getNum());
309 		assertEquals(fm2.getStringValue(), "testBeanOnlyPrototypeDISetterString");
310 		assertEquals(fm2.getStringValue(), fm2.getStringValue());
311 		// The TestBean reference is resolved to a prototype in the factory
312 		assertSame(fm2.getTestBean(), fm2.getTestBean());
313 		assertNotSame(fm1, fm2);
314 
315 		FactoryMethods fm3 = (FactoryMethods) xbf.getBean("testBeanOnlyPrototype", tbArg2, new Integer(1), "myName");
316 		assertEquals(1, fm3.getNum());
317 		assertEquals("myName", fm3.getName());
318 		assertEquals("arg2", fm3.getTestBean().getName());
319 
320 		FactoryMethods fm4 = (FactoryMethods) xbf.getBean("testBeanOnlyPrototype", tbArg);
321 		assertEquals(0, fm4.getNum());
322 		assertEquals("default", fm4.getName());
323 		assertEquals("arg1", fm4.getTestBean().getName());

355 	@Test
356 	public void testFactoryMethodWithDifferentReturnType() {
357 		DefaultListableBeanFactory xbf = new DefaultListableBeanFactory();
358 		XmlBeanDefinitionReader reader = new XmlBeanDefinitionReader(xbf);
359 		reader.loadBeanDefinitions(new ClassPathResource("factory-methods.xml", getClass()));
360 
361 		// Check that listInstance is not considered a bean of type FactoryMethods.
362 		assertTrue(List.class.isAssignableFrom(xbf.getType("listInstance")));
363 		String[] names = xbf.getBeanNamesForType(FactoryMethods.class);
364 		assertTrue(!Arrays.asList(names).contains("listInstance"));
365 		names = xbf.getBeanNamesForType(List.class);
366 		assertTrue(Arrays.asList(names).contains("listInstance"));
367 
368 		xbf.preInstantiateSingletons();
369 		assertTrue(List.class.isAssignableFrom(xbf.getType("listInstance")));
370 		names = xbf.getBeanNamesForType(FactoryMethods.class);
371 		assertTrue(!Arrays.asList(names).contains("listInstance"));
372 		names = xbf.getBeanNamesForType(List.class);
373 		assertTrue(Arrays.asList(names).contains("listInstance"));
374 		List<?> list = (List<?>) xbf.getBean("listInstance");
375 		assertEquals(Collections.EMPTY_LIST, list);
376 	}
