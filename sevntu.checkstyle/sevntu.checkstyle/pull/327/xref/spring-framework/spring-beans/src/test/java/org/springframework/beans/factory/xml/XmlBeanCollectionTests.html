63  	@Test
64  	public void testCollectionFactoryDefaults() throws Exception {
65  		ListFactoryBean listFactory = new ListFactoryBean();
66  		listFactory.setSourceList(new LinkedList());
67  		listFactory.afterPropertiesSet();
68  		assertTrue(listFactory.getObject() instanceof ArrayList);
69  
70  		SetFactoryBean setFactory = new SetFactoryBean();
71  		setFactory.setSourceSet(new TreeSet());
72  		setFactory.afterPropertiesSet();
73  		assertTrue(setFactory.getObject() instanceof LinkedHashSet);
74  
75  		MapFactoryBean mapFactory = new MapFactoryBean();
76  		mapFactory.setSourceMap(new TreeMap());
77  		mapFactory.afterPropertiesSet();
78  		assertTrue(mapFactory.getObject() instanceof LinkedHashMap);
79  	}

109 		TestBean jen = (TestBean) this.beanFactory.getBean("jenny");
110 		TestBean dave = (TestBean) this.beanFactory.getBean("david");
111 		TestBean rod = (TestBean) this.beanFactory.getBean("rod");
112 
113 		// Must be a list to support ordering
114 		// Our bean doesn't modify the collection:
115 		// of course it could be a different copy in a real object.
116 		Object[] friends = rod.getFriends().toArray();
117 		assertTrue(friends.length == 2);
118 
119 		assertTrue("First friend must be jen, not " + friends[0], friends[0] == jen);
120 		assertTrue(friends[1] == dave);
121 		// Should be ordered
122 	}

124 	@Test
125 	public void testRefSubelementsBuildCollectionWithPrototypes() throws Exception {
126 		TestBean jen = (TestBean) this.beanFactory.getBean("pJenny");
127 		TestBean dave = (TestBean) this.beanFactory.getBean("pDavid");
128 		TestBean rod = (TestBean) this.beanFactory.getBean("pRod");
129 
130 		Object[] friends = rod.getFriends().toArray();
131 		assertTrue(friends.length == 2);
132 		assertTrue("First friend must be jen, not " + friends[0],
133 				friends[0].toString().equals(jen.toString()));
134 		assertTrue("Jen not same instance", friends[0] != jen);
135 		assertTrue(friends[1].toString().equals(dave.toString()));
136 		assertTrue("Dave not same instance", friends[1] != dave);
137 		assertEquals("Jen", dave.getSpouse().getName());
138 
139 		TestBean rod2 = (TestBean) this.beanFactory.getBean("pRod");
140 		Object[] friends2 = rod2.getFriends().toArray();
141 		assertTrue(friends2.length == 2);
142 		assertTrue("First friend must be jen, not " + friends2[0],
143 				friends2[0].toString().equals(jen.toString()));
144 		assertTrue("Jen not same instance", friends2[0] != friends[0]);
145 		assertTrue(friends2[1].toString().equals(dave.toString()));
146 		assertTrue("Dave not same instance", friends2[1] != friends[1]);
147 	}

231 		HasMap hasMap = (HasMap) this.beanFactory.getBean("mixedMapWithList");
232 		assertTrue(hasMap.getMap().size() == 4);
233 		assertTrue(hasMap.getMap().get(null).equals("bar"));
234 		TestBean jenny = (TestBean) this.beanFactory.getBean("jenny");
235 		assertTrue(hasMap.getMap().get("jenny").equals(jenny));
236 
237 		// Check list
238 		List l = (List) hasMap.getMap().get("list");
239 		assertNotNull(l);
240 		assertTrue(l.size() == 4);
241 		assertTrue(l.get(0).equals("zero"));
242 		assertTrue(l.get(3) == null);
243 
244 		// Check nested map in list
245 		Map m = (Map) l.get(1);
246 		assertNotNull(m);
247 		assertTrue(m.size() == 2);
248 		assertTrue(m.get("fo").equals("bar"));
249 		assertTrue("Map element 'jenny' should be equal to jenny bean, not " + m.get("jen"),
250 				m.get("jen").equals(jenny));
251 
252 		// Check nested list in list
253 		l = (List) l.get(2);
254 		assertNotNull(l);
255 		assertTrue(l.size() == 2);
256 		assertTrue(l.get(0).equals(jenny));
257 		assertTrue(l.get(1).equals("ba"));
258 
259 		// Check nested map
260 		m = (Map) hasMap.getMap().get("map");
261 		assertNotNull(m);
262 		assertTrue(m.size() == 2);
263 		assertTrue(m.get("foo").equals("bar"));
264 		assertTrue("Map element 'jenny' should be equal to jenny bean, not " + m.get("jenny"),
