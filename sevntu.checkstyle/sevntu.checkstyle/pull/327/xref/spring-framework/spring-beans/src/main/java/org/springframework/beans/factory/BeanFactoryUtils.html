30  /**
31   * Convenience methods operating on bean factories, in particular
32   * on the {@link ListableBeanFactory} interface.
33   *
34   * <p>Returns bean counts, bean names or bean instances,
35   * taking into account the nesting hierarchy of a bean factory
36   * (which the methods defined on the ListableBeanFactory interface don't,
37   * in contrast to the methods defined on the BeanFactory interface).
38   *
39   * @author Rod Johnson
40   * @author Juergen Hoeller
41   * @author Chris Beams
42   * @since 04.07.2003
43   */
44  public abstract class BeanFactoryUtils {
45  
46  	/**
47  	 * Separator for generated bean names. If a class name or parent name is not
48  	 * unique, "#1", "#2" etc will be appended, until the name becomes unique.
49  	 */
50  	public static final String GENERATED_BEAN_NAME_SEPARATOR = "#";
51  
52  
53  	/**
54  	 * Return whether the given name is a factory dereference
55  	 * (beginning with the factory dereference prefix).
56  	 * @param name the name of the bean
57  	 * @return whether the given name is a factory dereference
58  	 * @see BeanFactory#FACTORY_BEAN_PREFIX
59  	 */
60  	public static boolean isFactoryDereference(String name) {
61  		return (name != null && name.startsWith(BeanFactory.FACTORY_BEAN_PREFIX));
62  	}
63  
64  	/**
65  	 * Return the actual bean name, stripping out the factory dereference
66  	 * prefix (if any, also stripping repeated factory prefixes if found).
67  	 * @param name the name of the bean
68  	 * @return the transformed name
69  	 * @see BeanFactory#FACTORY_BEAN_PREFIX
70  	 */
71  	public static String transformedBeanName(String name) {
72  		Assert.notNull(name, "'name' must not be null");
73  		String beanName = name;
74  		while (beanName.startsWith(BeanFactory.FACTORY_BEAN_PREFIX)) {
75  			beanName = beanName.substring(BeanFactory.FACTORY_BEAN_PREFIX.length());
76  		}
77  		return beanName;
78  	}
79  
80  	/**
81  	 * Return whether the given name is a bean name which has been generated
82  	 * by the default naming strategy (containing a "#..." part).
83  	 * @param name the name of the bean
84  	 * @return whether the given name is a generated bean name
85  	 * @see #GENERATED_BEAN_NAME_SEPARATOR
86  	 * @see org.springframework.beans.factory.support.BeanDefinitionReaderUtils#generateBeanName
87  	 * @see org.springframework.beans.factory.support.DefaultBeanNameGenerator
88  	 */
89  	public static boolean isGeneratedBeanName(String name) {
90  		return (name != null && name.contains(GENERATED_BEAN_NAME_SEPARATOR));
91  	}
92  
93  	/**
94  	 * Extract the "raw" bean name from the given (potentially generated) bean name,
95  	 * excluding any "#..." suffixes which might have been added for uniqueness.
96  	 * @param name the potentially generated bean name
97  	 * @return the raw bean name
98  	 * @see #GENERATED_BEAN_NAME_SEPARATOR
99  	 */
100 	public static String originalBeanName(String name) {
101 		Assert.notNull(name, "'name' must not be null");
102 		int separatorIndex = name.indexOf(GENERATED_BEAN_NAME_SEPARATOR);
103 		return (separatorIndex != -1 ? name.substring(0, separatorIndex) : name);
104 	}

108 	 * Count all beans in any hierarchy in which this factory participates.
109 	 * Includes counts of ancestor bean factories.
110 	 * <p>Beans that are "overridden" (specified in a descendant factory
111 	 * with the same name) are only counted once.
112 	 * @param lbf the bean factory
113 	 * @return count of beans including those defined in ancestor factories
114 	 */
115 	public static int countBeansIncludingAncestors(ListableBeanFactory lbf) {
116 		return beanNamesIncludingAncestors(lbf).length;
117 	}

120 	 * Return all bean names in the factory, including ancestor factories.
121 	 * @param lbf the bean factory
122 	 * @return the array of matching bean names, or an empty array if none
123 	 * @see #beanNamesForTypeIncludingAncestors
124 	 */
125 	public static String[] beanNamesIncludingAncestors(ListableBeanFactory lbf) {
126 		return beanNamesForTypeIncludingAncestors(lbf, Object.class);
127 	}

130 	/**
131 	 * Get all bean names for the given type, including those defined in ancestor
132 	 * factories. Will return unique names in case of overridden bean definitions.
133 	 * <p>Does consider objects created by FactoryBeans, which means that FactoryBeans
134 	 * will get initialized. If the object created by the FactoryBean doesn't match,
135 	 * the raw FactoryBean itself will be matched against the type.
136 	 * <p>This version of {@code beanNamesForTypeIncludingAncestors} automatically
137 	 * includes prototypes and FactoryBeans.
138 	 * @param lbf the bean factory
139 	 * @param type the type that beans must match
140 	 * @return the array of matching bean names, or an empty array if none
141 	 */
142 	public static String[] beanNamesForTypeIncludingAncestors(ListableBeanFactory lbf, Class<?> type) {
143 		Assert.notNull(lbf, "ListableBeanFactory must not be null");
144 		String[] result = lbf.getBeanNamesForType(type);
145 		if (lbf instanceof HierarchicalBeanFactory) {
146 			HierarchicalBeanFactory hbf = (HierarchicalBeanFactory) lbf;
147 			if (hbf.getParentBeanFactory() instanceof ListableBeanFactory) {
148 				String[] parentResult = beanNamesForTypeIncludingAncestors(
149 						(ListableBeanFactory) hbf.getParentBeanFactory(), type);
150 				List<String> resultList = new ArrayList<String>();
151 				resultList.addAll(Arrays.asList(result));
152 				for (String beanName : parentResult) {
153 					if (!resultList.contains(beanName) && !hbf.containsLocalBean(beanName)) {
154 						resultList.add(beanName);
155 					}
156 				}
157 				result = StringUtils.toStringArray(resultList);
158 			}
159 		}
160 		return result;
161 	}

163 	/**
164 	 * Get all bean names for the given type, including those defined in ancestor
165 	 * factories. Will return unique names in case of overridden bean definitions.
166 	 * <p>Does consider objects created by FactoryBeans if the "allowEagerInit"
167 	 * flag is set, which means that FactoryBeans will get initialized. If the
168 	 * object created by the FactoryBean doesn't match, the raw FactoryBean itself
169 	 * will be matched against the type. If "allowEagerInit" is not set,
170 	 * only raw FactoryBeans will be checked (which doesn't require initialization
171 	 * of each FactoryBean).
172 	 * @param lbf the bean factory
173 	 * @param includeNonSingletons whether to include prototype or scoped beans too
174 	 * or just singletons (also applies to FactoryBeans)
175 	 * @param allowEagerInit whether to initialize <i>lazy-init singletons</i> and
176 	 * <i>objects created by FactoryBeans</i> (or by factory methods with a
177 	 * "factory-bean" reference) for the type check. Note that FactoryBeans need to be
178 	 * eagerly initialized to determine their type: So be aware that passing in "true"
179 	 * for this flag will initialize FactoryBeans and "factory-bean" references.
180 	 * @param type the type that beans must match
181 	 * @return the array of matching bean names, or an empty array if none
182 	 */
183 	public static String[] beanNamesForTypeIncludingAncestors(
184 			ListableBeanFactory lbf, Class<?> type, boolean includeNonSingletons, boolean allowEagerInit) {
185 
186 		Assert.notNull(lbf, "ListableBeanFactory must not be null");
187 		String[] result = lbf.getBeanNamesForType(type, includeNonSingletons, allowEagerInit);
188 		if (lbf instanceof HierarchicalBeanFactory) {
189 			HierarchicalBeanFactory hbf = (HierarchicalBeanFactory) lbf;
190 			if (hbf.getParentBeanFactory() instanceof ListableBeanFactory) {
191 				String[] parentResult = beanNamesForTypeIncludingAncestors(
192 						(ListableBeanFactory) hbf.getParentBeanFactory(), type, includeNonSingletons, allowEagerInit);
193 				List<String> resultList = new ArrayList<String>();
194 				resultList.addAll(Arrays.asList(result));
195 				for (String beanName : parentResult) {
196 					if (!resultList.contains(beanName) && !hbf.containsLocalBean(beanName)) {
197 						resultList.add(beanName);
198 					}
199 				}
200 				result = StringUtils.toStringArray(resultList);
201 			}
202 		}
203 		return result;
204 	}

206 	/**
207 	 * Return all beans of the given type or subtypes, also picking up beans defined in
208 	 * ancestor bean factories if the current bean factory is a HierarchicalBeanFactory.
209 	 * The returned Map will only contain beans of this type.
210 	 * <p>Does consider objects created by FactoryBeans, which means that FactoryBeans
211 	 * will get initialized. If the object created by the FactoryBean doesn't match,
212 	 * the raw FactoryBean itself will be matched against the type.
213 	 * <p><b>Note: Beans of the same name will take precedence at the 'lowest' factory level,
214 	 * i.e. such beans will be returned from the lowest factory that they are being found in,
215 	 * hiding corresponding beans in ancestor factories.</b> This feature allows for
216 	 * 'replacing' beans by explicitly choosing the same bean name in a child factory;
217 	 * the bean in the ancestor factory won't be visible then, not even for by-type lookups.
218 	 * @param lbf the bean factory
219 	 * @param type type of bean to match
220 	 * @return the Map of matching bean instances, or an empty Map if none
221 	 * @throws BeansException if a bean could not be created
222 	 */
223 	public static <T> Map<String, T> beansOfTypeIncludingAncestors(ListableBeanFactory lbf, Class<T> type)
224 			throws BeansException {
225 
226 		Assert.notNull(lbf, "ListableBeanFactory must not be null");
227 		Map<String, T> result = new LinkedHashMap<String, T>(4);
228 		result.putAll(lbf.getBeansOfType(type));
229 		if (lbf instanceof HierarchicalBeanFactory) {
230 			HierarchicalBeanFactory hbf = (HierarchicalBeanFactory) lbf;
231 			if (hbf.getParentBeanFactory() instanceof ListableBeanFactory) {
232 				Map<String, T> parentResult = beansOfTypeIncludingAncestors(
233 						(ListableBeanFactory) hbf.getParentBeanFactory(), type);
234 				for (Map.Entry<String, T> entry : parentResult.entrySet()) {
235 					String beanName = entry.getKey();
236 					if (!result.containsKey(beanName) && !hbf.containsLocalBean(beanName)) {
237 						result.put(beanName, entry.getValue());
238 					}
239 				}
240 			}
241 		}
242 		return result;
243 	}

297 	 * defined in ancestor bean factories if the current bean factory is a
298 	 * HierarchicalBeanFactory. Useful convenience method when we expect a
299 	 * single bean and don't care about the bean name.
300 	 * <p>Does consider objects created by FactoryBeans, which means that FactoryBeans
301 	 * will get initialized. If the object created by the FactoryBean doesn't match,
302 	 * the raw FactoryBean itself will be matched against the type.

311 	 * @param type type of bean to match
312 	 * @return the matching bean instance
313 	 * @throws NoSuchBeanDefinitionException if no bean of the given type was found
314 	 * @throws NoUniqueBeanDefinitionException if more than one bean of the given type was found

416 	/**
417 	 * Extract a unique bean for the given type from the given Map of matching beans.
418 	 * @param type type of bean to match
419 	 * @param matchingBeans all matching beans found
420 	 * @return the unique bean instance
421 	 * @throws NoSuchBeanDefinitionException if no bean of the given type was found
422 	 * @throws NoUniqueBeanDefinitionException if more than one bean of the given type was found
423 	 */
424 	private static <T> T uniqueBean(Class<T> type, Map<String, T> matchingBeans) {
425 		int nrFound = matchingBeans.size();
426 		if (nrFound == 1) {
427 			return matchingBeans.values().iterator().next();
428 		}
429 		else if (nrFound > 1) {
430 			throw new NoUniqueBeanDefinitionException(type, matchingBeans.keySet());
431 		}
432 		else {
433 			throw new NoSuchBeanDefinitionException(type);
434 		}
435 	}
