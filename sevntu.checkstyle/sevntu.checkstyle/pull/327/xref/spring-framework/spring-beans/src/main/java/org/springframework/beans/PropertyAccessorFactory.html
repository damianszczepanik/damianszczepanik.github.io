17  package org.springframework.beans;
18  
19  /**
20   * Simple factory facade for obtaining {@link PropertyAccessor} instances,
21   * in particular for {@link BeanWrapper} instances. Conceals the actual
22   * target implementation classes and their extended public signature.
23   *
24   * @author Juergen Hoeller
25   * @since 2.5.2
26   */
27  public abstract class PropertyAccessorFactory {
28  
29  	/**
30  	 * Obtain a BeanWrapper for the given target object,
31  	 * accessing properties in JavaBeans style.
32  	 * @param target the target object to wrap
33  	 * @return the property accessor
34  	 * @see BeanWrapperImpl
35  	 */
36  	public static BeanWrapper forBeanPropertyAccess(Object target) {
37  		return new BeanWrapperImpl(target);
38  	}
39  
40  	/**
41  	 * Obtain a PropertyAccessor for the given target object,
42  	 * accessing properties in direct field style.
43  	 * @param target the target object to wrap
44  	 * @return the property accessor
45  	 * @see DirectFieldAccessor
46  	 */
47  	public static ConfigurablePropertyAccessor forDirectFieldAccess(Object target) {
48  		return new DirectFieldAccessor(target);
49  	}
50  
51  }
