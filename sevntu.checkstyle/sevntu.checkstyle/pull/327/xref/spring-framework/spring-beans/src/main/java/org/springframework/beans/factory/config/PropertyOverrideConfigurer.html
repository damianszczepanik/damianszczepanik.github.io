17  package org.springframework.beans.factory.config;
18  
19  import java.util.Collections;
20  import java.util.Enumeration;
21  import java.util.Properties;
22  import java.util.Set;
23  import java.util.concurrent.ConcurrentHashMap;
24  
25  import org.springframework.beans.BeansException;
26  import org.springframework.beans.PropertyValue;
27  import org.springframework.beans.factory.BeanInitializationException;
28  
29  /**
30   * Property resource configurer that overrides bean property values in an application
31   * context definition. It <i>pushes</i> values from a properties file into bean definitions.
32   *
33   * <p>Configuration lines are expected to be of the following form:
34   *
35   * <pre class="code">beanName.property=value</pre>
36   *
37   * Example properties file:
38   *
39   * <pre class="code">dataSource.driverClassName=com.mysql.jdbc.Driver
40   * dataSource.url=jdbc:mysql:mydb</pre>
41   *
42   * In contrast to PropertyPlaceholderConfigurer, the original definition can have default
43   * values or no values at all for such bean properties. If an overriding properties file does
44   * not have an entry for a certain bean property, the default context definition is used.
45   *
46   * <p>Note that the context definition <i>is not</i> aware of being overridden;
47   * so this is not immediately obvious when looking at the XML definition file.
48   * Furthermore, note that specified override values are always <i>literal</i> values;
49   * they are not translated into bean references. This also applies when the original
50   * value in the XML bean definition specifies a bean reference.
51   *
52   * <p>In case of multiple PropertyOverrideConfigurers that define different values for
53   * the same bean property, the <i>last</i> one will win (due to the overriding mechanism).
54   *
55   * <p>Property values can be converted after reading them in, through overriding
56   * the {@code convertPropertyValue} method. For example, encrypted values
57   * can be detected and decrypted accordingly before processing them.
58   *
59   * @author Juergen Hoeller
60   * @author Rod Johnson
61   * @since 12.03.2003
62   * @see #convertPropertyValue
63   * @see PropertyPlaceholderConfigurer
64   */
65  public class PropertyOverrideConfigurer extends PropertyResourceConfigurer {
66  
67  	public static final String DEFAULT_BEAN_NAME_SEPARATOR = ".";
68  
69  
70  	private String beanNameSeparator = DEFAULT_BEAN_NAME_SEPARATOR;
71  
72  	private boolean ignoreInvalidKeys = false;
73  
74  	/**
75  	 * Contains names of beans that have overrides
76  	 */
77  	private final Set<String> beanNames = Collections.newSetFromMap(new ConcurrentHashMap<String, Boolean>(16));
78  
79  
80  	/**
81  	 * Set the separator to expect between bean name and property path.
82  	 * Default is a dot (".").
83  	 */
84  	public void setBeanNameSeparator(String beanNameSeparator) {
85  		this.beanNameSeparator = beanNameSeparator;
86  	}
87  
88  	/**
89  	 * Set whether to ignore invalid keys. Default is "false".
90  	 * <p>If you ignore invalid keys, keys that do not follow the 'beanName.property' format
91  	 * (or refer to invalid bean names or properties) will just be logged at debug level.
92  	 * This allows one to have arbitrary other keys in a properties file.
93  	 */
94  	public void setIgnoreInvalidKeys(boolean ignoreInvalidKeys) {
95  		this.ignoreInvalidKeys = ignoreInvalidKeys;
96  	}
97  
98  
99  	@Override
100 	protected void processProperties(ConfigurableListableBeanFactory beanFactory, Properties props)
101 			throws BeansException {
102 
103 		for (Enumeration<?> names = props.propertyNames(); names.hasMoreElements();) {
104 			String key = (String) names.nextElement();
105 			try {
106 				processKey(beanFactory, key, props.getProperty(key));
107 			}
108 			catch (BeansException ex) {
109 				String msg = "Could not process key '" + key + "' in PropertyOverrideConfigurer";
110 				if (!this.ignoreInvalidKeys) {
111 					throw new BeanInitializationException(msg, ex);
112 				}
113 				if (logger.isDebugEnabled()) {
114 					logger.debug(msg, ex);
115 				}
116 			}
117 		}
118 	}
119 
120 	/**
121 	 * Process the given key as 'beanName.property' entry.
122 	 */
123 	protected void processKey(ConfigurableListableBeanFactory factory, String key, String value)
124 			throws BeansException {
125 
126 		int separatorIndex = key.indexOf(this.beanNameSeparator);
127 		if (separatorIndex == -1) {
128 			throw new BeanInitializationException("Invalid key '" + key +
129 					"': expected 'beanName" + this.beanNameSeparator + "property'");
130 		}
131 		String beanName = key.substring(0, separatorIndex);
132 		String beanProperty = key.substring(separatorIndex+1);
133 		this.beanNames.add(beanName);
134 		applyPropertyValue(factory, beanName, beanProperty, value);
135 		if (logger.isDebugEnabled()) {
136 			logger.debug("Property '" + key + "' set to value [" + value + "]");
137 		}
138 	}
139 
140 	/**
141 	 * Apply the given property value to the corresponding bean.
142 	 */
143 	protected void applyPropertyValue(
144 			ConfigurableListableBeanFactory factory, String beanName, String property, String value) {
145 
146 		BeanDefinition bd = factory.getBeanDefinition(beanName);
147 		while (bd.getOriginatingBeanDefinition() != null) {
148 			bd = bd.getOriginatingBeanDefinition();
149 		}
150 		PropertyValue pv = new PropertyValue(property, value);
151 		pv.setOptional(this.ignoreInvalidKeys);
152 		bd.getPropertyValues().addPropertyValue(pv);
153 	}
154 
155 
156 	/**
157 	 * Were there overrides for this bean?
158 	 * Only valid after processing has occurred at least once.
159 	 * @param beanName name of the bean to query status for
160 	 * @return whether there were property overrides for
161 	 * the named bean
162 	 */
163 	public boolean hasPropertyOverridesFor(String beanName) {
164 		return this.beanNames.contains(beanName);
165 	}
166 
167 }
