17  package org.springframework.beans.support;
18  
19  import java.io.Serializable;
20  
21  import org.springframework.util.StringUtils;
22  
23  /**
24   * Mutable implementation of the {@link SortDefinition} interface.
25   * Supports toggling the ascending value on setting the same property again.
26   *
27   * @author Juergen Hoeller
28   * @author Jean-Pierre Pawlak
29   * @since 26.05.2003
30   * @see #setToggleAscendingOnProperty
31   */
32  @SuppressWarnings("serial")
33  public class MutableSortDefinition implements SortDefinition, Serializable {
34  
35  	private String property = "";
36  
37  	private boolean ignoreCase = true;
38  
39  	private boolean ascending = true;
40  
41  	private boolean toggleAscendingOnProperty = false;
42  
43  
44  	/**
45  	 * Create an empty MutableSortDefinition,
46  	 * to be populated via its bean properties.
47  	 * @see #setProperty
48  	 * @see #setIgnoreCase
49  	 * @see #setAscending
50  	 */
51  	public MutableSortDefinition() {
52  	}
53  
54  	/**
55  	 * Copy constructor: create a new MutableSortDefinition
56  	 * that mirrors the given sort definition.
57  	 * @param source the original sort definition
58  	 */
59  	public MutableSortDefinition(SortDefinition source) {
60  		this.property = source.getProperty();
61  		this.ignoreCase = source.isIgnoreCase();
62  		this.ascending = source.isAscending();
63  	}
64  
65  	/**
66  	 * Create a MutableSortDefinition for the given settings.
67  	 * @param property the property to compare
68  	 * @param ignoreCase whether upper and lower case in String values should be ignored
69  	 * @param ascending whether to sort ascending (true) or descending (false)
70  	 */
71  	public MutableSortDefinition(String property, boolean ignoreCase, boolean ascending) {
72  		this.property = property;
73  		this.ignoreCase = ignoreCase;
74  		this.ascending = ascending;
75  	}
76  
77  	/**
78  	 * Create a new MutableSortDefinition.
79  	 * @param toggleAscendingOnSameProperty whether to toggle the ascending flag
80  	 * if the same property gets set again (that is, {@code setProperty} gets
81  	 * called with already set property name again).
82  	 */
83  	public MutableSortDefinition(boolean toggleAscendingOnSameProperty) {
84  		this.toggleAscendingOnProperty = toggleAscendingOnSameProperty;
85  	}
86  
87  
88  	/**
89  	 * Set the property to compare.
90  	 * <p>If the property was the same as the current, the sort is reversed if
91  	 * "toggleAscendingOnProperty" is activated, else simply ignored.
92  	 * @see #setToggleAscendingOnProperty
93  	 */
94  	public void setProperty(String property) {
95  		if (!StringUtils.hasLength(property)) {
96  			this.property = "";
97  		}
98  		else {
99  			// Implicit toggling of ascending?
100 			if (isToggleAscendingOnProperty()) {
101 				this.ascending = (!property.equals(this.property) || !this.ascending);
102 			}
103 			this.property = property;
104 		}
105 	}
106 
107 	@Override
108 	public String getProperty() {
109 		return this.property;
110 	}
111 
112 	/**
113 	 * Set whether upper and lower case in String values should be ignored.
114 	 */
115 	public void setIgnoreCase(boolean ignoreCase) {
116 		this.ignoreCase = ignoreCase;
117 	}
118 
119 	@Override
120 	public boolean isIgnoreCase() {
121 		return this.ignoreCase;
122 	}
123 
124 	/**
125 	 * Set whether to sort ascending (true) or descending (false).
126 	 */
127 	public void setAscending(boolean ascending) {
128 		this.ascending = ascending;
129 	}
130 
131 	@Override
132 	public boolean isAscending() {
133 		return this.ascending;
134 	}
135 
136 	/**
137 	 * Set whether to toggle the ascending flag if the same property gets set again
138 	 * (that is, {@link #setProperty} gets called with already set property name again).
139 	 * <p>This is particularly useful for parameter binding through a web request,
140 	 * where clicking on the field header again might be supposed to trigger a
141 	 * resort for the same field but opposite order.
142 	 */
143 	public void setToggleAscendingOnProperty(boolean toggleAscendingOnProperty) {
144 		this.toggleAscendingOnProperty = toggleAscendingOnProperty;
145 	}
146 
147 	/**
148 	 * Return whether to toggle the ascending flag if the same property gets set again
149 	 * (that is, {@code setProperty} gets called with already set property name again).
150 	 */
151 	public boolean isToggleAscendingOnProperty() {
152 		return this.toggleAscendingOnProperty;
153 	}
154 
155 
156 	@Override
157 	public boolean equals(Object other) {
158 		if (this == other) {
159 			return true;
160 		}
161 		if (!(other instanceof SortDefinition)) {
162 			return false;
163 		}
164 		SortDefinition otherSd = (SortDefinition) other;
165 		return (getProperty().equals(otherSd.getProperty()) &&
166 				isAscending() == otherSd.isAscending() &&
167 				isIgnoreCase() == otherSd.isIgnoreCase());
168 	}
169 
170 	@Override
171 	public int hashCode() {
172 		int hashCode = getProperty().hashCode();
173 		hashCode = 29 * hashCode + (isIgnoreCase() ? 1 : 0);
174 		hashCode = 29 * hashCode + (isAscending() ? 1 : 0);
175 		return hashCode;
176 	}
177 
178 }
