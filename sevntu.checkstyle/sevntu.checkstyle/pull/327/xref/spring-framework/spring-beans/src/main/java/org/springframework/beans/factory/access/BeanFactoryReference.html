17  package org.springframework.beans.factory.access;
18  
19  import org.springframework.beans.factory.BeanFactory;
20  
21  /**
22   * Used to track a reference to a {@link BeanFactory} obtained through
23   * a {@link BeanFactoryLocator}.
24   *
25   * <p>It is safe to call {@link #release()} multiple times, but
26   * {@link #getFactory()} must not be called after calling release.
27   *
28   * @author Colin Sampaleanu
29   * @see BeanFactoryLocator
30   * @see org.springframework.context.access.ContextBeanFactoryReference
31   */
32  public interface BeanFactoryReference {
33  
34  	/**
35  	 * Return the {@link BeanFactory} instance held by this reference.
36  	 * @throws IllegalStateException if invoked after {@code release()} has been called
37  	 */
38  	BeanFactory getFactory();
39  
40  	/**
41  	 * Indicate that the {@link BeanFactory} instance referred to by this object is not
42  	 * needed any longer by the client code which obtained the {@link BeanFactoryReference}.
43  	 * <p>Depending on the actual implementation of {@link BeanFactoryLocator}, and
44  	 * the actual type of {@code BeanFactory}, this may possibly not actually
45  	 * do anything; alternately in the case of a 'closeable' {@code BeanFactory}
46  	 * or derived class (such as {@link org.springframework.context.ApplicationContext})
47  	 * may 'close' it, or may 'close' it once no more references remain.
48  	 * <p>In an EJB usage scenario this would normally be called from
49  	 * {@code ejbRemove()} and {@code ejbPassivate()}.
50  	 * <p>This is safe to call multiple times.
51  	 * @see BeanFactoryLocator
52  	 * @see org.springframework.context.access.ContextBeanFactoryReference
53  	 * @see org.springframework.context.ConfigurableApplicationContext#close()
54  	 */
55  	void release();
56  
57  }
