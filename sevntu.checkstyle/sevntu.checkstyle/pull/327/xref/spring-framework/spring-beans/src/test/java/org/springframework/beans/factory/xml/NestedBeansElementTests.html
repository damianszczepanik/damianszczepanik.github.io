14  /**
15   * Tests for new nested beans element support in Spring XML
16   *
17   * @author Chris Beams
18   */
19  public class NestedBeansElementTests {
20  	private final Resource XML =
21  		new ClassPathResource("NestedBeansElementTests-context.xml", this.getClass());
22  
23  	@Test
24  	public void getBean_withoutActiveProfile() {
25  		DefaultListableBeanFactory bf = new DefaultListableBeanFactory();
26  		new XmlBeanDefinitionReader(bf).loadBeanDefinitions(XML);
27  
28  		Object foo = bf.getBean("foo");
29  		assertThat(foo, instanceOf(String.class));
30  	}
31  
32  	@Test
33  	public void getBean_withActiveProfile() {
34  		ConfigurableEnvironment env = new StandardEnvironment();
35  		env.setActiveProfiles("dev");
36  
37  		DefaultListableBeanFactory bf = new DefaultListableBeanFactory();
38  		XmlBeanDefinitionReader reader = new XmlBeanDefinitionReader(bf);
39  		reader.setEnvironment(env);
40  		reader.loadBeanDefinitions(XML);
41  
42  		bf.getBean("devOnlyBean"); // should not throw NSBDE
43  
44  		Object foo = bf.getBean("foo");
45  		assertThat(foo, instanceOf(Integer.class));
46  
47  		bf.getBean("devOnlyBean");
48  	}
