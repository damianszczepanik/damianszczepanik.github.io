17  package org.springframework.beans.factory.parsing;
18  
19  import org.springframework.core.io.Resource;
20  
21  /**
22   * Simple {@link SourceExtractor} implementation that just passes
23   * the candidate source metadata object through for attachment.
24   *
25   * <p>Using this implementation means that tools will get raw access to the
26   * underlying configuration source metadata provided by the tool.
27   *
28   * <p>This implementation <strong>should not</strong> be used in a production
29   * application since it is likely to keep too much metadata in memory
30   * (unnecessarily).
31   *
32   * @author Rob Harrop
33   * @since 2.0
34   */
35  public class PassThroughSourceExtractor implements SourceExtractor {
36  
37  	/**
38  	 * Simply returns the supplied {@code sourceCandidate} as-is.
39  	 * @param sourceCandidate the source metadata
40  	 * @return the supplied {@code sourceCandidate}
41  	 */
42  	@Override
43  	public Object extractSource(Object sourceCandidate, Resource definingResource) {
44  		return sourceCandidate;
45  	}
46  
47  }
