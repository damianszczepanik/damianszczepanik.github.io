55  /**
56   * CGLIB-based {@link AopProxy} implementation for the Spring AOP framework.
57   *
58   * <p>Formerly named {@code Cglib2AopProxy}, as of Spring 3.2, this class depends on
59   * Spring's own internally repackaged version of CGLIB 3.</i>.
60   *
61   * <p>Objects of this type should be obtained through proxy factories,
62   * configured by an {@link AdvisedSupport} object. This class is internal
63   * to Spring's AOP framework and need not be used directly by client code.
64   *
65   * <p>{@link DefaultAopProxyFactory} will automatically create CGLIB-based
66   * proxies if necessary, for example in case of proxying a target class
67   * (see the {@link DefaultAopProxyFactory attendant javadoc} for details).
68   *
69   * <p>Proxies created using this class are thread-safe if the underlying
70   * (target) class is thread-safe.
71   *
72   * @author Rod Johnson
73   * @author Rob Harrop
74   * @author Juergen Hoeller
75   * @author Ramnivas Laddad
76   * @author Chris Beams
77   * @author Dave Syer
78   * @see org.springframework.cglib.proxy.Enhancer
79   * @see AdvisedSupport#setProxyTargetClass
80   * @see DefaultAopProxyFactory
81   */
82  @SuppressWarnings("serial")
83  class CglibAopProxy implements AopProxy, Serializable {

117 	/**
118 	 * Create a new CglibAopProxy for the given AOP configuration.
119 	 * @param config the AOP configuration as AdvisedSupport object
120 	 * @throws AopConfigException if the config is invalid. We try to throw an informative
121 	 * exception in this case, rather than let a mysterious failure happen later.
122 	 */
123 	public CglibAopProxy(AdvisedSupport config) throws AopConfigException {
124 		Assert.notNull(config, "AdvisedSupport must not be null");
125 		if (config.getAdvisors().length == 0 && config.getTargetSource() == AdvisedSupport.EMPTY_TARGET_SOURCE) {
126 			throw new AopConfigException("No advisors and no TargetSource specified");
127 		}
128 		this.advised = config;
129 		this.advisedDispatcher = new AdvisedDispatcher(this.advised);
130 	}

133 	 * Set constructor arguments to use for creating the proxy.
134 	 * @param constructorArgs the constructor argument values
135 	 * @param constructorArgTypes the constructor argument types
136 	 */
137 	public void setConstructorArguments(Object[] constructorArgs, Class<?>[] constructorArgTypes) {
138 		if (constructorArgs == null || constructorArgTypes == null) {
139 			throw new IllegalArgumentException("Both 'constructorArgs' and 'constructorArgTypes' need to be specified");
140 		}
141 		if (constructorArgs.length != constructorArgTypes.length) {
142 			throw new IllegalArgumentException("Number of 'constructorArgs' (" + constructorArgs.length +
143 					") must match number of 'constructorArgTypes' (" + constructorArgTypes.length + ")");
144 		}
145 		this.constructorArgs = constructorArgs;
146 		this.constructorArgTypes = constructorArgTypes;
147 	}

155 	@Override
156 	public Object getProxy(ClassLoader classLoader) {
157 		if (logger.isDebugEnabled()) {
158 			logger.debug("Creating CGLIB proxy: target source is " + this.advised.getTargetSource());
159 		}
160 
161 		try {
162 			Class<?> rootClass = this.advised.getTargetClass();
163 			Assert.state(rootClass != null, "Target class must be available for creating a CGLIB proxy");
164 
165 			Class<?> proxySuperClass = rootClass;
166 			if (ClassUtils.isCglibProxyClass(rootClass)) {
167 				proxySuperClass = rootClass.getSuperclass();
168 				Class<?>[] additionalInterfaces = rootClass.getInterfaces();
169 				for (Class<?> additionalInterface : additionalInterfaces) {
170 					this.advised.addInterface(additionalInterface);
171 				}
172 			}
173 
174 			// Validate the class, writing log messages as necessary.
175 			validateClassIfNecessary(proxySuperClass, classLoader);
176 
177 			// Configure CGLIB Enhancer...
178 			Enhancer enhancer = createEnhancer();
179 			if (classLoader != null) {
180 				enhancer.setClassLoader(classLoader);
181 				if (classLoader instanceof SmartClassLoader &&
182 						((SmartClassLoader) classLoader).isClassReloadable(proxySuperClass)) {
183 					enhancer.setUseCache(false);
184 				}
185 			}
186 			enhancer.setSuperclass(proxySuperClass);
187 			enhancer.setInterfaces(AopProxyUtils.completeProxiedInterfaces(this.advised));
188 			enhancer.setNamingPolicy(SpringNamingPolicy.INSTANCE);
189 			enhancer.setStrategy(new UndeclaredThrowableStrategy(UndeclaredThrowableException.class));
190 
191 			Callback[] callbacks = getCallbacks(rootClass);
192 			Class<?>[] types = new Class<?>[callbacks.length];
193 			for (int x = 0; x < types.length; x++) {
194 				types[x] = callbacks[x].getClass();
195 			}
196 			// fixedInterceptorMap only populated at this point, after getCallbacks call above
197 			enhancer.setCallbackFilter(new ProxyCallbackFilter(
198 					this.advised.getConfigurationOnlyCopy(), this.fixedInterceptorMap, this.fixedInterceptorOffset));
199 			enhancer.setCallbackTypes(types);
200 
201 			// Generate the proxy class and create a proxy instance.
202 			return createProxyClassAndInstance(enhancer, callbacks);
203 		}
204 		catch (CodeGenerationException ex) {
205 			throw new AopConfigException("Could not generate CGLIB subclass of class [" +
206 					this.advised.getTargetClass() + "]: " +
207 					"Common causes of this problem include using a final class or a non-visible class",
208 					ex);
209 		}
210 		catch (IllegalArgumentException ex) {
211 			throw new AopConfigException("Could not generate CGLIB subclass of class [" +
212 					this.advised.getTargetClass() + "]: " +
213 					"Common causes of this problem include using a final class or a non-visible class",
214 					ex);
215 		}
216 		catch (Exception ex) {
217 			// TargetSource.getTarget() failed
218 			throw new AopConfigException("Unexpected AOP exception", ex);
219 		}
220 	}

222 	protected Object createProxyClassAndInstance(Enhancer enhancer, Callback[] callbacks) {
223 		enhancer.setInterceptDuringConstruction(false);
224 		enhancer.setCallbacks(callbacks);
225 		return (this.constructorArgs != null ?
226 				enhancer.create(this.constructorArgTypes, this.constructorArgs) :
227 				enhancer.create());
228 	}

238 	/**
239 	 * Checks to see whether the supplied {@code Class} has already been validated and
240 	 * validates it if not.
241 	 */
242 	private void validateClassIfNecessary(Class<?> proxySuperClass, ClassLoader proxyClassLoader) {
243 		if (logger.isInfoEnabled()) {
244 			synchronized (validatedClasses) {
245 				if (!validatedClasses.containsKey(proxySuperClass)) {
246 					doValidateClass(proxySuperClass, proxyClassLoader);
247 					validatedClasses.put(proxySuperClass, Boolean.TRUE);
248 				}
249 			}
250 		}
251 	}

253 	/**
254 	 * Checks for final methods on the given {@code Class}, as well as package-visible
255 	 * methods across ClassLoaders, and writes warnings to the log for each one found.
256 	 */
257 	private void doValidateClass(Class<?> proxySuperClass, ClassLoader proxyClassLoader) {
258 		if (!Object.class.equals(proxySuperClass)) {
259 			Method[] methods = proxySuperClass.getDeclaredMethods();
260 			for (Method method : methods) {
261 				int mod = method.getModifiers();
262 				if (!Modifier.isStatic(mod)) {
263 					if (Modifier.isFinal(mod)) {
264 						logger.info("Unable to proxy method [" + method + "] because it is final: " +
265 								"All calls to this method via a proxy will NOT be routed to the target instance.");
266 					}
267 					else if (!Modifier.isPublic(mod) && !Modifier.isProtected(mod) && !Modifier.isPrivate(mod) &&
268 							proxyClassLoader != null && proxySuperClass.getClassLoader() != proxyClassLoader) {
269 						logger.info("Unable to proxy method [" + method + "] because it is package-visible " +
270 								"across different ClassLoaders: All calls to this method via a proxy will " +
271 								"NOT be routed to the target instance.");
272 					}
273 				}
274 			}
275 			doValidateClass(proxySuperClass.getSuperclass(), proxyClassLoader);
276 		}
277 	}

279 	private Callback[] getCallbacks(Class<?> rootClass) throws Exception {
280 		// Parameters used for optimisation choices...
281 		boolean exposeProxy = this.advised.isExposeProxy();
282 		boolean isFrozen = this.advised.isFrozen();
283 		boolean isStatic = this.advised.getTargetSource().isStatic();
284 
285 		// Choose an "aop" interceptor (used for AOP calls).
286 		Callback aopInterceptor = new DynamicAdvisedInterceptor(this.advised);
287 
288 		// Choose a "straight to target" interceptor. (used for calls that are
289 		// unadvised but can return this). May be required to expose the proxy.
290 		Callback targetInterceptor;
291 		if (exposeProxy) {
292 			targetInterceptor = isStatic ?
293 					new StaticUnadvisedExposedInterceptor(this.advised.getTargetSource().getTarget()) :
294 					new DynamicUnadvisedExposedInterceptor(this.advised.getTargetSource());
295 		}
296 		else {
297 			targetInterceptor = isStatic ?
298 					new StaticUnadvisedInterceptor(this.advised.getTargetSource().getTarget()) :
299 					new DynamicUnadvisedInterceptor(this.advised.getTargetSource());
300 		}
301 
302 		// Choose a "direct to target" dispatcher (used for
303 		// unadvised calls to static targets that cannot return this).
304 		Callback targetDispatcher = isStatic ?
305 				new StaticDispatcher(this.advised.getTargetSource().getTarget()) : new SerializableNoOp();
306 
307 		Callback[] mainCallbacks = new Callback[]{
308 			aopInterceptor, // for normal advice
309 			targetInterceptor, // invoke target without considering advice, if optimized
310 			new SerializableNoOp(), // no override for methods mapped to this
311 			targetDispatcher, this.advisedDispatcher,
312 			new EqualsInterceptor(this.advised),
313 			new HashCodeInterceptor(this.advised)
314 		};
315 
316 		Callback[] callbacks;
317 
318 		// If the target is a static one and the advice chain is frozen,
319 		// then we can make some optimisations by sending the AOP calls
320 		// direct to the target using the fixed chain for that method.
321 		if (isStatic && isFrozen) {
322 			Method[] methods = rootClass.getMethods();
323 			Callback[] fixedCallbacks = new Callback[methods.length];
324 			this.fixedInterceptorMap = new HashMap<String, Integer>(methods.length);
325 
326 			// TODO: small memory optimisation here (can skip creation for methods with no advice)
327 			for (int x = 0; x < methods.length; x++) {
328 				List<Object> chain = this.advised.getInterceptorsAndDynamicInterceptionAdvice(methods[x], rootClass);
329 				fixedCallbacks[x] = new FixedChainStaticTargetInterceptor(
330 						chain, this.advised.getTargetSource().getTarget(), this.advised.getTargetClass());
331 				this.fixedInterceptorMap.put(methods[x].toString(), x);
332 			}
333 
334 			// Now copy both the callbacks from mainCallbacks
335 			// and fixedCallbacks into the callbacks array.
336 			callbacks = new Callback[mainCallbacks.length + fixedCallbacks.length];
337 			System.arraycopy(mainCallbacks, 0, callbacks, 0, mainCallbacks.length);
338 			System.arraycopy(fixedCallbacks, 0, callbacks, mainCallbacks.length, fixedCallbacks.length);
339 			this.fixedInterceptorOffset = mainCallbacks.length;
340 		}
341 		else {
342 			callbacks = mainCallbacks;
343 		}
344 		return callbacks;
345 	}

347 	/**
348 	 * Process a return value. Wraps a return of {@code this} if necessary to be the
349 	 * {@code proxy} and also verifies that {@code null} is not returned as a primitive.
350 	 */
351 	private static Object processReturnType(Object proxy, Object target, Method method, Object retVal) {
352 		// Massage return value if necessary
353 		if (retVal != null && retVal == target && !RawTargetAccess.class.isAssignableFrom(method.getDeclaringClass())) {
354 			// Special case: it returned "this". Note that we can't help
355 			// if the target sets a reference to itself in another returned object.
356 			retVal = proxy;
357 		}
358 		Class<?> returnType = method.getReturnType();
359 		if (retVal == null && returnType != Void.TYPE && returnType.isPrimitive()) {
360 			throw new AopInvocationException(
361 					"Null return value from advice does not match primitive return type for: " + method);
362 		}
363 		return retVal;
364 	}

529 	/**
530 	 * Dispatcher for the {@code equals} method.
531 	 * Ensures that the method call is always handled by this class.
532 	 */
533 	private static class EqualsInterceptor implements MethodInterceptor, Serializable {
534 
535 		private final AdvisedSupport advised;
536 
537 		public EqualsInterceptor(AdvisedSupport advised) {
538 			this.advised = advised;
539 		}
540 
541 		@Override
542 		public Object intercept(Object proxy, Method method, Object[] args, MethodProxy methodProxy) {
543 			Object other = args[0];
544 			if (proxy == other) {
545 				return true;
546 			}
547 			if (other instanceof Factory) {
548 				Callback callback = ((Factory) other).getCallback(INVOKE_EQUALS);
549 				if (!(callback instanceof EqualsInterceptor)) {
550 					return false;
551 				}
552 				AdvisedSupport otherAdvised = ((EqualsInterceptor) callback).advised;
553 				return AopProxyUtils.equalsInProxy(this.advised, otherAdvised);
554 			}
555 			else {
556 				return false;
557 			}
558 		}
559 	}

581 	/**
582 	 * Interceptor used specifically for advised methods on a frozen, static proxy.
583 	 */
584 	private static class FixedChainStaticTargetInterceptor implements MethodInterceptor, Serializable {
585 
586 		private final List<Object> adviceChain;
587 
588 		private final Object target;
589 
590 		private final Class<?> targetClass;
591 
592 		public FixedChainStaticTargetInterceptor(List<Object> adviceChain, Object target, Class<?> targetClass) {
593 			this.adviceChain = adviceChain;
594 			this.target = target;
595 			this.targetClass = targetClass;
596 		}
597 
598 		@Override
599 		public Object intercept(Object proxy, Method method, Object[] args, MethodProxy methodProxy) throws Throwable {
600 			MethodInvocation invocation = new CglibMethodInvocation(proxy, this.target, method, args,
601 					this.targetClass, this.adviceChain, methodProxy);
602 			// If we get here, we need to create a MethodInvocation.
603 			Object retVal = invocation.proceed();
604 			retVal = processReturnType(proxy, this.target, method, retVal);
605 			return retVal;
606 		}
607 	}

610 	/**
611 	 * General purpose AOP callback. Used when the target is dynamic or when the
612 	 * proxy is not frozen.
613 	 */
614 	private static class DynamicAdvisedInterceptor implements MethodInterceptor, Serializable {
615 
616 		private final AdvisedSupport advised;
617 
618 		public DynamicAdvisedInterceptor(AdvisedSupport advised) {
619 			this.advised = advised;
620 		}
621 
622 		@Override
623 		public Object intercept(Object proxy, Method method, Object[] args, MethodProxy methodProxy) throws Throwable {
624 			Object oldProxy = null;
625 			boolean setProxyContext = false;
626 			Class<?> targetClass = null;
627 			Object target = null;
628 			try {
629 				if (this.advised.exposeProxy) {
630 					// Make invocation available if necessary.
631 					oldProxy = AopContext.setCurrentProxy(proxy);
632 					setProxyContext = true;
633 				}
634 				// May be null. Get as late as possible to minimize the time we
635 				// "own" the target, in case it comes from a pool...
636 				target = getTarget();
637 				if (target != null) {
638 					targetClass = target.getClass();
639 				}
640 				List<Object> chain = this.advised.getInterceptorsAndDynamicInterceptionAdvice(method, targetClass);
641 				Object retVal;
642 				// Check whether we only have one InvokerInterceptor: that is,
643 				// no real advice, but just reflective invocation of the target.
644 				if (chain.isEmpty() && Modifier.isPublic(method.getModifiers())) {
645 					// We can skip creating a MethodInvocation: just invoke the target directly.
646 					// Note that the final invoker must be an InvokerInterceptor, so we know
647 					// it does nothing but a reflective operation on the target, and no hot
648 					// swapping or fancy proxying.
649 					retVal = methodProxy.invoke(target, args);
650 				}
651 				else {
652 					// We need to create a method invocation...
653 					retVal = new CglibMethodInvocation(proxy, target, method, args, targetClass, chain, methodProxy).proceed();
654 				}
655 				retVal = processReturnType(proxy, target, method, retVal);
656 				return retVal;
657 			}
658 			finally {
659 				if (target != null) {
660 					releaseTarget(target);
661 				}
662 				if (setProxyContext) {
663 					// Restore old proxy.
664 					AopContext.setCurrentProxy(oldProxy);
665 				}
666 			}
667 		}
668 
669 		@Override
670 		public boolean equals(Object other) {
671 			return (this == other ||
672 					(other instanceof DynamicAdvisedInterceptor &&
673 							this.advised.equals(((DynamicAdvisedInterceptor) other).advised)));
674 		}

677 		 * CGLIB uses this to drive proxy creation.

681 			return this.advised.hashCode();

695 	 * Implementation of AOP Alliance MethodInvocation used by this AOP proxy.
696 	 */
697 	private static class CglibMethodInvocation extends ReflectiveMethodInvocation {
698 
699 		private final MethodProxy methodProxy;
700 
701 		private final boolean publicMethod;
702 
703 		public CglibMethodInvocation(Object proxy, Object target, Method method, Object[] arguments,
704 				Class<?> targetClass, List<Object> interceptorsAndDynamicMethodMatchers, MethodProxy methodProxy) {
705 			super(proxy, target, method, arguments, targetClass, interceptorsAndDynamicMethodMatchers);
706 			this.methodProxy = methodProxy;
707 			this.publicMethod = Modifier.isPublic(method.getModifiers());
708 		}

710 		/**
711 		 * Gives a marginal performance improvement versus using reflection to
712 		 * invoke the target when invoking public methods.
713 		 */
714 		@Override
715 		protected Object invokeJoinpoint() throws Throwable {
716 			if (this.publicMethod) {
717 				return this.methodProxy.invoke(this.target, this.arguments);
718 			}
719 			else {
720 				return super.invokeJoinpoint();
721 			}

726 	/**
727 	 * CallbackFilter to assign Callbacks to methods.
728 	 */
729 	private static class ProxyCallbackFilter implements CallbackFilter {
730 
731 		private final AdvisedSupport advised;
732 
733 		private final Map<String, Integer> fixedInterceptorMap;
734 
735 		private final int fixedInterceptorOffset;
736 
737 		public ProxyCallbackFilter(AdvisedSupport advised, Map<String, Integer> fixedInterceptorMap, int fixedInterceptorOffset) {
738 			this.advised = advised;
739 			this.fixedInterceptorMap = fixedInterceptorMap;
740 			this.fixedInterceptorOffset = fixedInterceptorOffset;
741 		}
742 
743 		/**
744 		 * Implementation of CallbackFilter.accept() to return the index of the
745 		 * callback we need.
746 		 * <p>The callbacks for each proxy are built up of a set of fixed callbacks
747 		 * for general use and then a set of callbacks that are specific to a method
748 		 * for use on static targets with a fixed advice chain.
749 		 * <p>The callback used is determined thus:
750 		 * <dl>
751 		 * <dt>For exposed proxies</dt>
752 		 * <dd>Exposing the proxy requires code to execute before and after the
753 		 * method/chain invocation. This means we must use
754 		 * DynamicAdvisedInterceptor, since all other interceptors can avoid the
755 		 * need for a try/catch block</dd>
756 		 * <dt>For Object.finalize():</dt>
757 		 * <dd>No override for this method is used.</dd>
758 		 * <dt>For equals():</dt>
759 		 * <dd>The EqualsInterceptor is used to redirect equals() calls to a
760 		 * special handler to this proxy.</dd>
761 		 * <dt>For methods on the Advised class:</dt>
762 		 * <dd>the AdvisedDispatcher is used to dispatch the call directly to
763 		 * the target</dd>
764 		 * <dt>For advised methods:</dt>
765 		 * <dd>If the target is static and the advice chain is frozen then a
766 		 * FixedChainStaticTargetInterceptor specific to the method is used to
767 		 * invoke the advice chain. Otherwise a DyanmicAdvisedInterceptor is
768 		 * used.</dd>
769 		 * <dt>For non-advised methods:</dt>
770 		 * <dd>Where it can be determined that the method will not return {@code this}
771 		 * or when {@code ProxyFactory.getExposeProxy()} returns {@code false},
772 		 * then a Dispatcher is used. For static targets, the StaticDispatcher is used;
773 		 * and for dynamic targets, a DynamicUnadvisedInterceptor is used.
774 		 * If it possible for the method to return {@code this} then a
775 		 * StaticUnadvisedInterceptor is used for static targets - the
776 		 * DynamicUnadvisedInterceptor already considers this.</dd>
777 		 * </dl>
778 		 */
779 		@Override
780 		public int accept(Method method) {
781 			if (AopUtils.isFinalizeMethod(method)) {
782 				logger.debug("Found finalize() method - using NO_OVERRIDE");
783 				return NO_OVERRIDE;
784 			}
785 			if (!this.advised.isOpaque() && method.getDeclaringClass().isInterface() &&
786 					method.getDeclaringClass().isAssignableFrom(Advised.class)) {
787 				if (logger.isDebugEnabled()) {
788 					logger.debug("Method is declared on Advised interface: " + method);
789 				}
790 				return DISPATCH_ADVISED;
791 			}
792 			// We must always proxy equals, to direct calls to this.
793 			if (AopUtils.isEqualsMethod(method)) {
794 				logger.debug("Found 'equals' method: " + method);
795 				return INVOKE_EQUALS;
796 			}
797 			// We must always calculate hashCode based on the proxy.
798 			if (AopUtils.isHashCodeMethod(method)) {
799 				logger.debug("Found 'hashCode' method: " + method);
800 				return INVOKE_HASHCODE;
801 			}
802 			Class<?> targetClass = this.advised.getTargetClass();
803 			// Proxy is not yet available, but that shouldn't matter.
804 			List<?> chain = this.advised.getInterceptorsAndDynamicInterceptionAdvice(method, targetClass);
805 			boolean haveAdvice = !chain.isEmpty();
806 			boolean exposeProxy = this.advised.isExposeProxy();
807 			boolean isStatic = this.advised.getTargetSource().isStatic();
808 			boolean isFrozen = this.advised.isFrozen();
809 			if (haveAdvice || !isFrozen) {
810 				// If exposing the proxy, then AOP_PROXY must be used.
811 				if (exposeProxy) {
812 					if (logger.isDebugEnabled()) {
813 						logger.debug("Must expose proxy on advised method: " + method);
814 					}
815 					return AOP_PROXY;
816 				}
817 				String key = method.toString();
818 				// Check to see if we have fixed interceptor to serve this method.
819 				// Else use the AOP_PROXY.
820 				if (isStatic && isFrozen && this.fixedInterceptorMap.containsKey(key)) {
821 					if (logger.isDebugEnabled()) {
822 						logger.debug("Method has advice and optimisations are enabled: " + method);
823 					}
824 					// We know that we are optimising so we can use the
825 					// FixedStaticChainInterceptors.
826 					int index = this.fixedInterceptorMap.get(key);
827 					return (index + this.fixedInterceptorOffset);
828 				}
829 				else {
830 					if (logger.isDebugEnabled()) {
831 						logger.debug("Unable to apply any optimisations to advised method: " + method);
832 					}
833 					return AOP_PROXY;
834 				}
835 			}
836 			else {
837 				// See if the return type of the method is outside the class hierarchy
838 				// of the target type. If so we know it never needs to have return type
839 				// massage and can use a dispatcher.
840 				// If the proxy is being exposed, then must use the interceptor the
841 				// correct one is already configured. If the target is not static, then
842 				// cannot use a dispatcher because the target cannot be released.
843 				if (exposeProxy || !isStatic) {
844 					return INVOKE_TARGET;
845 				}
846 				Class<?> returnType = method.getReturnType();
847 				if (targetClass == returnType) {
848 					if (logger.isDebugEnabled()) {
849 						logger.debug("Method " + method +
850 								"has return type same as target type (may return this) - using INVOKE_TARGET");
851 					}
852 					return INVOKE_TARGET;
853 				}
854 				else if (returnType.isPrimitive() || !returnType.isAssignableFrom(targetClass)) {
855 					if (logger.isDebugEnabled()) {
856 						logger.debug("Method " + method +
857 								" has return type that ensures this cannot be returned- using DISPATCH_TARGET");
858 					}
859 					return DISPATCH_TARGET;
860 				}
861 				else {
862 					if (logger.isDebugEnabled()) {
863 						logger.debug("Method " + method +
864 								"has return type that is assignable from the target type (may return this) - " +
865 								"using INVOKE_TARGET");
866 					}
867 					return INVOKE_TARGET;
868 				}
869 			}
870 		}

872 		@Override
873 		public boolean equals(Object other) {
874 			if (this == other) {
875 				return true;
876 			}
877 			if (!(other instanceof ProxyCallbackFilter)) {
878 				return false;
879 			}
880 			ProxyCallbackFilter otherCallbackFilter = (ProxyCallbackFilter) other;
881 			AdvisedSupport otherAdvised = otherCallbackFilter.advised;
882 			if (this.advised == null || otherAdvised == null) {
883 				return false;
884 			}
885 			if (this.advised.isFrozen() != otherAdvised.isFrozen()) {
886 				return false;
887 			}
888 			if (this.advised.isExposeProxy() != otherAdvised.isExposeProxy()) {
889 				return false;
890 			}
891 			if (this.advised.getTargetSource().isStatic() != otherAdvised.getTargetSource().isStatic()) {
892 				return false;
893 			}
894 			if (!AopProxyUtils.equalsProxiedInterfaces(this.advised, otherAdvised)) {
895 				return false;
896 			}
897 			// Advice instance identity is unimportant to the proxy class:
898 			// All that matters is type and ordering.
899 			Advisor[] thisAdvisors = this.advised.getAdvisors();
900 			Advisor[] thatAdvisors = otherAdvised.getAdvisors();
901 			if (thisAdvisors.length != thatAdvisors.length) {
902 				return false;
903 			}
904 			for (int i = 0; i < thisAdvisors.length; i++) {
905 				Advisor thisAdvisor = thisAdvisors[i];
906 				Advisor thatAdvisor = thatAdvisors[i];
907 				if (!equalsAdviceClasses(thisAdvisor, thatAdvisor)) {
908 					return false;
909 				}
910 				if (!equalsPointcuts(thisAdvisor, thatAdvisor)) {
911 					return false;
912 				}
913 			}
914 			return true;
915 		}
916 
917 		private boolean equalsAdviceClasses(Advisor a, Advisor b) {
918 			Advice aa = a.getAdvice();
919 			Advice ba = b.getAdvice();
920 			if (aa == null || ba == null) {
921 				return (aa == ba);
922 			}
923 			return aa.getClass().equals(ba.getClass());
924 		}
925 
926 		private boolean equalsPointcuts(Advisor a, Advisor b) {
927 			// If only one of the advisor (but not both) is PointcutAdvisor, then it is a mismatch.
928 			// Takes care of the situations where an IntroductionAdvisor is used (see SPR-3959).
929 			return (!(a instanceof PointcutAdvisor) ||
930 					(b instanceof PointcutAdvisor &&
931 							ObjectUtils.nullSafeEquals(((PointcutAdvisor) a).getPointcut(), ((PointcutAdvisor) b).getPointcut())));
932 		}
933 
934 		@Override
935 		public int hashCode() {
936 			int hashCode = 0;
937 			Advisor[] advisors = this.advised.getAdvisors();
938 			for (Advisor advisor : advisors) {
939 				Advice advice = advisor.getAdvice();
940 				if (advice != null) {
941 					hashCode = 13 * hashCode + advice.getClass().hashCode();
942 				}
943 			}
944 			hashCode = 13 * hashCode + (this.advised.isFrozen() ? 1 : 0);
945 			hashCode = 13 * hashCode + (this.advised.isExposeProxy() ? 1 : 0);
946 			hashCode = 13 * hashCode + (this.advised.isOptimize() ? 1 : 0);
947 			hashCode = 13 * hashCode + (this.advised.isOpaque() ? 1 : 0);
948 			return hashCode;
949 		}
