17  package org.springframework.aop.support;
18  
19  import java.io.IOException;
20  
21  import org.junit.Before;
22  import org.junit.Test;
23  
24  import org.springframework.tests.sample.beans.TestBean;
25  import org.springframework.util.SerializationTestUtils;
26  
27  import static org.junit.Assert.*;
28  
29  /**
30   * @author Rod Johnson
31   * @author Dmitriy Kopylenko
32   * @author Chris Beams
33   */
34  public abstract class AbstractRegexpMethodPointcutTests {
35  
36  	private AbstractRegexpMethodPointcut rpc;
37  
38  	@Before
39  	public void setUp() {
40  		rpc = getRegexpMethodPointcut();
41  	}
42  
43  	protected abstract AbstractRegexpMethodPointcut getRegexpMethodPointcut();
44  
45  	@Test
46  	public void testNoPatternSupplied() throws Exception {
47  		noPatternSuppliedTests(rpc);
48  	}
49  
50  	@Test
51  	public void testSerializationWithNoPatternSupplied() throws Exception {
52  		rpc = (AbstractRegexpMethodPointcut) SerializationTestUtils.serializeAndDeserialize(rpc);
53  		noPatternSuppliedTests(rpc);
54  	}
55  
56  	protected void noPatternSuppliedTests(AbstractRegexpMethodPointcut rpc) throws Exception {
57  		assertFalse(rpc.matches(Object.class.getMethod("hashCode", (Class[]) null), String.class));
58  		assertFalse(rpc.matches(Object.class.getMethod("wait", (Class[]) null), Object.class));
59  		assertEquals(0, rpc.getPatterns().length);
60  	}
61  
62  	@Test
63  	public void testExactMatch() throws Exception {
64  		rpc.setPattern("java.lang.Object.hashCode");
65  		exactMatchTests(rpc);
66  		rpc = (AbstractRegexpMethodPointcut) SerializationTestUtils.serializeAndDeserialize(rpc);
67  		exactMatchTests(rpc);
68  	}
69  
70  	protected void exactMatchTests(AbstractRegexpMethodPointcut rpc) throws Exception {
71  		// assumes rpc.setPattern("java.lang.Object.hashCode");
72  		assertTrue(rpc.matches(Object.class.getMethod("hashCode", (Class[]) null), String.class));
73  		assertTrue(rpc.matches(Object.class.getMethod("hashCode", (Class[]) null), Object.class));
74  		assertFalse(rpc.matches(Object.class.getMethod("wait", (Class[]) null), Object.class));
75  	}
76  
77  	@Test
78  	public void testSpecificMatch() throws Exception {
79  		rpc.setPattern("java.lang.String.hashCode");
80  		assertTrue(rpc.matches(Object.class.getMethod("hashCode", (Class[]) null), String.class));
81  		assertFalse(rpc.matches(Object.class.getMethod("hashCode", (Class[]) null), Object.class));
82  	}
83  
84  	@Test
85  	public void testWildcard() throws Exception {
86  		rpc.setPattern(".*Object.hashCode");
87  		assertTrue(rpc.matches(Object.class.getMethod("hashCode", (Class[]) null), Object.class));
88  		assertFalse(rpc.matches(Object.class.getMethod("wait", (Class[]) null), Object.class));
89  	}
90  
91  	@Test
92  	public void testWildcardForOneClass() throws Exception {
93  		rpc.setPattern("java.lang.Object.*");
94  		assertTrue(rpc.matches(Object.class.getMethod("hashCode", (Class[]) null), String.class));
95  		assertTrue(rpc.matches(Object.class.getMethod("wait", (Class[]) null), String.class));
96  	}
97  
98  	@Test
99  	public void testMatchesObjectClass() throws Exception {
100 		rpc.setPattern("java.lang.Object.*");
101 		assertTrue(rpc.matches(Exception.class.getMethod("hashCode", (Class[]) null), IOException.class));
102 		// Doesn't match a method from Throwable
103 		assertFalse(rpc.matches(Exception.class.getMethod("getMessage", (Class[]) null), Exception.class));
104 	}
105 
106 	@Test
107 	public void testWithExclusion() throws Exception {
108 		this.rpc.setPattern(".*get.*");
109 		this.rpc.setExcludedPattern(".*Age.*");
110 		assertTrue(this.rpc.matches(TestBean.class.getMethod("getName"), TestBean.class));
111 		assertFalse(this.rpc.matches(TestBean.class.getMethod("getAge"), TestBean.class));
112 	}
113 
114 }
