82  	@Test
83  	public void testSunnyDayPathLogsCorrectly() throws Throwable {
84  
85  		MethodInvocation methodInvocation = mock(MethodInvocation.class);
86  		given(methodInvocation.getMethod()).willReturn(String.class.getMethod("toString", new Class[]{}));
87  		given(methodInvocation.getThis()).willReturn(this);
88  
89  		Log log = mock(Log.class);
90  		given(log.isTraceEnabled()).willReturn(true);
91  
92  		CustomizableTraceInterceptor interceptor = new StubCustomizableTraceInterceptor(log);
93  		interceptor.invoke(methodInvocation);
94  
95  		verify(log, times(2)).trace(anyString());
96  	}

98  	@Test
99  	public void testExceptionPathLogsCorrectly() throws Throwable {
100 
101 		MethodInvocation methodInvocation = mock(MethodInvocation.class);
102 
103 		IllegalArgumentException exception = new IllegalArgumentException();
104 		given(methodInvocation.getMethod()).willReturn(String.class.getMethod("toString", new Class[]{}));
105 		given(methodInvocation.getThis()).willReturn(this);
106 		given(methodInvocation.proceed()).willThrow(exception);
107 
108 		Log log = mock(Log.class);
109 		given(log.isTraceEnabled()).willReturn(true);
110 
111 		CustomizableTraceInterceptor interceptor = new StubCustomizableTraceInterceptor(log);
112 		try {
113 			interceptor.invoke(methodInvocation);
114 			fail("Must have propagated the IllegalArgumentException.");
115 		}
116 		catch (IllegalArgumentException expected) {
117 		}
118 
119 		verify(log).trace(anyString());
120 		verify(log).trace(anyString(), eq(exception));
121 	}

123 	@Test
124 	public void testSunnyDayPathLogsCorrectlyWithPrettyMuchAllPlaceholdersMatching() throws Throwable {
125 
126 		MethodInvocation methodInvocation = mock(MethodInvocation.class);
127 
128 		given(methodInvocation.getMethod()).willReturn(String.class.getMethod("toString", new Class[0]));
129 		given(methodInvocation.getThis()).willReturn(this);
130 		given(methodInvocation.getArguments()).willReturn(new Object[]{"$ One \\$", new Long(2)});
131 		given(methodInvocation.proceed()).willReturn("Hello!");
132 
133 		Log log = mock(Log.class);
134 		given(log.isTraceEnabled()).willReturn(true);
135 
136 		CustomizableTraceInterceptor interceptor = new StubCustomizableTraceInterceptor(log);
137 		interceptor.setEnterMessage(new StringBuffer()
138 			.append("Entering the '").append(CustomizableTraceInterceptor.PLACEHOLDER_METHOD_NAME)
139 			.append("' method of the [").append(CustomizableTraceInterceptor.PLACEHOLDER_TARGET_CLASS_NAME)
140 			.append("] class with the following args (").append(CustomizableTraceInterceptor.PLACEHOLDER_ARGUMENTS)
141 			.append(") and arg types (").append(CustomizableTraceInterceptor.PLACEHOLDER_ARGUMENT_TYPES)
142 			.append(").").toString());
143 		interceptor.setExitMessage(new StringBuffer()
144 			.append("Exiting the '").append(CustomizableTraceInterceptor.PLACEHOLDER_METHOD_NAME)
145 			.append("' method of the [").append(CustomizableTraceInterceptor.PLACEHOLDER_TARGET_CLASS_SHORT_NAME)
146 			.append("] class with the following args (").append(CustomizableTraceInterceptor.PLACEHOLDER_ARGUMENTS)
147 			.append(") and arg types (").append(CustomizableTraceInterceptor.PLACEHOLDER_ARGUMENT_TYPES)
148 			.append("), returning '").append(CustomizableTraceInterceptor.PLACEHOLDER_RETURN_VALUE)
149 			.append("' and taking '").append(CustomizableTraceInterceptor.PLACEHOLDER_INVOCATION_TIME)
150 			.append("' this long.").toString());
151 		interceptor.invoke(methodInvocation);
152 
153 		verify(log, times(2)).trace(anyString());
154 	}

157 	@SuppressWarnings("serial")
158 	private static class StubCustomizableTraceInterceptor extends CustomizableTraceInterceptor {
159 
160 		private final Log log;
161 
162 		public StubCustomizableTraceInterceptor(Log log) {
163 			super.setUseDynamicLogger(false);
164 			this.log = log;
165 		}
166 
167 		@Override
168 		protected Log getLoggerForInvocation(MethodInvocation invocation) {
169 			return this.log;
170 		}
171 	}
