17  package org.springframework.aop.framework;
18  
19  import org.springframework.beans.factory.Aware;
20  import org.springframework.beans.factory.BeanClassLoaderAware;
21  import org.springframework.beans.factory.DisposableBean;
22  import org.springframework.beans.factory.InitializingBean;
23  import org.springframework.core.Ordered;
24  import org.springframework.util.ClassUtils;
25  import org.springframework.util.ObjectUtils;
26  
27  /**
28   * Base class with common functionality for proxy processors, in particular
29   * ClassLoader management and the {@link #evaluateProxyInterfaces} algorithm.
30   *
31   * @author Juergen Hoeller
32   * @since 4.1
33   * @see AbstractAdvisingBeanPostProcessor
34   * @see org.springframework.aop.framework.autoproxy.AbstractAutoProxyCreator
35   */
36  @SuppressWarnings("serial")
37  public class ProxyProcessorSupport extends ProxyConfig implements Ordered, BeanClassLoaderAware, AopInfrastructureBean {
38  
39  	/**
40  	 * This should run after all other processors, so that it can just add
41  	 * an advisor to existing proxies rather than double-proxy.
42  	 */
43  	private int order = Ordered.LOWEST_PRECEDENCE;
44  
45  	private ClassLoader proxyClassLoader = ClassUtils.getDefaultClassLoader();
46  
47  	private boolean classLoaderConfigured = false;
48  
49  
50  	/**
51  	 * Set the ordering which will apply to this class's implementation
52  	 * of Ordered, used when applying multiple processors.
53  	 * <p>Default value is {@code Integer.MAX_VALUE}, meaning that it's non-ordered.
54  	 * @param order the ordering value
55  	 */
56  	public void setOrder(int order) {
57  		this.order = order;
58  	}
59  
60  	@Override
61  	public int getOrder() {
62  		return this.order;
63  	}
64  
65  	/**
66  	 * Set the ClassLoader to generate the proxy class in.
67  	 * <p>Default is the bean ClassLoader, i.e. the ClassLoader used by the containing
68  	 * {@link org.springframework.beans.factory.BeanFactory} for loading all bean classes.
69  	 * This can be overridden here for specific proxies.
70  	 */
71  	public void setProxyClassLoader(ClassLoader classLoader) {
72  		this.proxyClassLoader = classLoader;
73  		this.classLoaderConfigured = (classLoader != null);
74  	}
75  
76  	/**
77  	 * Return the configured proxy ClassLoader for this processor.
78  	 */
79  	protected ClassLoader getProxyClassLoader() {
80  		return this.proxyClassLoader;
81  	}
82  
83  	@Override
84  	public void setBeanClassLoader(ClassLoader classLoader) {
85  		if (!this.classLoaderConfigured) {
86  			this.proxyClassLoader = classLoader;
87  		}
88  	}
89  
90  
91  	/**
92  	 * Check the interfaces on the given bean class and apply them to the {@link ProxyFactory},
93  	 * if appropriate.
94  	 * <p>Calls {@link #isConfigurationCallbackInterface} and {@link #isInternalLanguageInterface}
95  	 * to filter for reasonable proxy interfaces, falling back to a target-class proxy otherwise.
96  	 * @param beanClass the class of the bean
97  	 * @param proxyFactory the ProxyFactory for the bean
98  	 */
99  	protected void evaluateProxyInterfaces(Class<?> beanClass, ProxyFactory proxyFactory) {
100 		Class<?>[] targetInterfaces = ClassUtils.getAllInterfacesForClass(beanClass, getProxyClassLoader());
101 		boolean hasReasonableProxyInterface = false;
102 		for (Class<?> ifc : targetInterfaces) {
103 			if (!isConfigurationCallbackInterface(ifc) && !isInternalLanguageInterface(ifc) &&
104 					ifc.getMethods().length > 0) {
105 				hasReasonableProxyInterface = true;
106 				break;
107 			}
108 		}
109 		if (hasReasonableProxyInterface) {
110 			// Must allow for introductions; can't just set interfaces to the target's interfaces only.
111 			for (Class<?> ifc : targetInterfaces) {
112 				proxyFactory.addInterface(ifc);
113 			}
114 		}
115 		else {
116 			proxyFactory.setProxyTargetClass(true);
117 		}
118 	}
119 
120 	/**
121 	 * Determine whether the given interface is just a container callback and
122 	 * therefore not to be considered as a reasonable proxy interface.
123 	 * <p>If no reasonable proxy interface is found for a given bean, it will get
124 	 * proxied with its full target class, assuming that as the user's intention.
125 	 * @param ifc the interface to check
126 	 * @return whether the given interface is just a container callback
127 	 */
128 	protected boolean isConfigurationCallbackInterface(Class<?> ifc) {
129 		return (ifc.equals(InitializingBean.class) || ifc.equals(DisposableBean.class) ||
130 				ObjectUtils.containsElement(ifc.getInterfaces(), Aware.class));
131 	}
132 
133 	/**
134 	 * Determine whether the given interface is a well-known internal language interface
135 	 * and therefore not to be considered as a reasonable proxy interface.
136 	 * <p>If no reasonable proxy interface is found for a given bean, it will get
137 	 * proxied with its full target class, assuming that as the user's intention.
138 	 * @param ifc the interface to check
139 	 * @return whether the given interface is an internal language interface
140 	 */
141 	protected boolean isInternalLanguageInterface(Class<?> ifc) {
142 		return ifc.getName().equals("groovy.lang.GroovyObject");
143 	}
144 
145 }
