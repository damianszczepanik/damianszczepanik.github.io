32  /**
33   * {@code MethodInterceptor} implementation that allows for highly customizable
34   * method-level tracing, using placeholders.
35   *
36   * <p>Trace messages are written on method entry, and if the method invocation succeeds
37   * on method exit. If an invocation results in an exception, then an exception message
38   * is written. The contents of these trace messages is fully customizable and special
39   * placeholders are available to allow you to include runtime information in your log
40   * messages. The placeholders available are:
41   *
42   * <p><ul>
43   * <li>{@code $[methodName]} - replaced with the name of the method being invoked</li>
44   * <li>{@code $[targetClassName]} - replaced with the name of the class that is
45   * the target of the invocation</li>
46   * <li>{@code $[targetClassShortName]} - replaced with the short name of the class
47   * that is the target of the invocation</li>
48   * <li>{@code $[returnValue]} - replaced with the value returned by the invocation</li>
49   * <li>{@code $[argumentTypes]} - replaced with a comma-separated list of the
50   * short class names of the method arguments</li>
51   * <li>{@code $[arguments]} - replaced with a comma-separated list of the
52   * {@code String} representation of the method arguments</li>
53   * <li>{@code $[exception]} - replaced with the {@code String} representation
54   * of any {@code Throwable} raised during the invocation</li>
55   * <li>{@code $[invocationTime]} - replaced with the time, in milliseconds,
56   * taken by the method invocation</li>
57   * </ul>
58   *
59   * <p>There are restrictions on which placeholders can be used in which messages:
60   * see the individual message properties for details on the valid placeholders.
61   *
62   * @author Rob Harrop
63   * @author Juergen Hoeller
64   * @since 1.2
65   * @see #setEnterMessage
66   * @see #setExitMessage
67   * @see #setExceptionMessage
68   * @see SimpleTraceInterceptor
69   */
70  @SuppressWarnings("serial")
71  public class CustomizableTraceInterceptor extends AbstractTraceInterceptor {
72  
73  	/**
74  	 * The {@code $[methodName]} placeholder.
75  	 * Replaced with the name of the method being invoked.
76  	 */
77  	public static final String PLACEHOLDER_METHOD_NAME = "$[methodName]";
78  
79  	/**
80  	 * The {@code $[targetClassName]} placeholder.
81  	 * Replaced with the fully-qualifed name of the {@code Class}
82  	 * of the method invocation target.
83  	 */
84  	public static final String PLACEHOLDER_TARGET_CLASS_NAME = "$[targetClassName]";
85  
86  	/**
87  	 * The {@code $[targetClassShortName]} placeholder.
88  	 * Replaced with the short name of the {@code Class} of the
89  	 * method invocation target.
90  	 */
91  	public static final String PLACEHOLDER_TARGET_CLASS_SHORT_NAME = "$[targetClassShortName]";
92  
93  	/**
94  	 * The {@code $[returnValue]} placeholder.
95  	 * Replaced with the {@code String} representation of the value
96  	 * returned by the method invocation.
97  	 */
98  	public static final String PLACEHOLDER_RETURN_VALUE = "$[returnValue]";
99  
100 	/**
101 	 * The {@code $[argumentTypes]} placeholder.
102 	 * Replaced with a comma-separated list of the argument types for the
103 	 * method invocation. Argument types are written as short class names.
104 	 */
105 	public static final String PLACEHOLDER_ARGUMENT_TYPES = "$[argumentTypes]";
106 
107 	/**
108 	 * The {@code $[arguments]} placeholder.
109 	 * Replaced with a comma separated list of the argument values for the
110 	 * method invocation. Relies on the {@code toString()} method of
111 	 * each argument type.
112 	 */
113 	public static final String PLACEHOLDER_ARGUMENTS = "$[arguments]";
114 
115 	/**
116 	 * The {@code $[exception]} placeholder.
117 	 * Replaced with the {@code String} representation of any
118 	 * {@code Throwable} raised during method invocation.
119 	 */
120 	public static final String PLACEHOLDER_EXCEPTION = "$[exception]";
121 
122 	/**
123 	 * The {@code $[invocationTime]} placeholder.
124 	 * Replaced with the time taken by the invocation (in milliseconds).
125 	 */
126 	public static final String PLACEHOLDER_INVOCATION_TIME = "$[invocationTime]";
127 
128 	/**
129 	 * The default message used for writing method entry messages.
130 	 */
131 	private static final String DEFAULT_ENTER_MESSAGE =
132 			"Entering method '" + PLACEHOLDER_METHOD_NAME + "' of class [" + PLACEHOLDER_TARGET_CLASS_NAME + "]";
133 
134 	/**
135 	 * The default message used for writing method exit messages.
136 	 */
137 	private static final String DEFAULT_EXIT_MESSAGE =
138 			"Exiting method '" + PLACEHOLDER_METHOD_NAME + "' of class [" + PLACEHOLDER_TARGET_CLASS_NAME + "]";
139 
140 	/**
141 	 * The default message used for writing exception messages.
142 	 */
143 	private static final String DEFAULT_EXCEPTION_MESSAGE =
144 			"Exception thrown in method '" + PLACEHOLDER_METHOD_NAME + "' of class [" + PLACEHOLDER_TARGET_CLASS_NAME + "]";
145 
146 	/**
147 	 * The {@code Pattern} used to match placeholders.
148 	 */
149 	private static final Pattern PATTERN = Pattern.compile("\\$\\[\\p{Alpha}+\\]");
150 
151 	/**
152 	 * The {@code Set} of allowed placeholders.
153 	 */
154 	private static final Set<Object> ALLOWED_PLACEHOLDERS =
155 			new Constants(CustomizableTraceInterceptor.class).getValues("PLACEHOLDER_");
156 
157 
158 	/**
159 	 * The message for method entry.
160 	 */
161 	private String enterMessage = DEFAULT_ENTER_MESSAGE;
162 
163 	/**
164 	 * The message for method exit.
165 	 */
166 	private String exitMessage = DEFAULT_EXIT_MESSAGE;
167 
168 	/**
169 	 * The message for exceptions during method execution.
170 	 */
171 	private String exceptionMessage = DEFAULT_EXCEPTION_MESSAGE;

174 	/**
175 	 * Set the template used for method entry log messages.
176 	 * This template can contain any of the following placeholders:
177 	 * <ul>
178 	 * <li>{@code $[targetClassName]}</li>
179 	 * <li>{@code $[targetClassShortName]}</li>
180 	 * <li>{@code $[argumentTypes]}</li>
181 	 * <li>{@code $[arguments]}</li>
182 	 * </ul>
183 	 */
184 	public void setEnterMessage(String enterMessage) throws IllegalArgumentException {
185 		Assert.hasText(enterMessage, "'enterMessage' must not be empty");
186 		checkForInvalidPlaceholders(enterMessage);
187 		Assert.doesNotContain(enterMessage, PLACEHOLDER_RETURN_VALUE,
188 				"enterMessage cannot contain placeholder [" + PLACEHOLDER_RETURN_VALUE + "]");
189 		Assert.doesNotContain(enterMessage, PLACEHOLDER_EXCEPTION,
190 				"enterMessage cannot contain placeholder [" + PLACEHOLDER_EXCEPTION + "]");
191 		Assert.doesNotContain(enterMessage, PLACEHOLDER_INVOCATION_TIME,
192 				"enterMessage cannot contain placeholder [" + PLACEHOLDER_INVOCATION_TIME + "]");
193 		this.enterMessage = enterMessage;
194 	}
195 
196 	/**
197 	 * Set the template used for method exit log messages.
198 	 * This template can contain any of the following placeholders:
199 	 * <ul>
200 	 * <li>{@code $[targetClassName]}</li>
201 	 * <li>{@code $[targetClassShortName]}</li>
202 	 * <li>{@code $[argumentTypes]}</li>
203 	 * <li>{@code $[arguments]}</li>
204 	 * <li>{@code $[returnValue]}</li>
205 	 * <li>{@code $[invocationTime]}</li>
206 	 * </ul>
207 	 */
208 	public void setExitMessage(String exitMessage) {
209 		Assert.hasText(exitMessage, "'exitMessage' must not be empty");
210 		checkForInvalidPlaceholders(exitMessage);
211 		Assert.doesNotContain(exitMessage, PLACEHOLDER_EXCEPTION,
212 				"exitMessage cannot contain placeholder [" + PLACEHOLDER_EXCEPTION + "]");
213 		this.exitMessage = exitMessage;
214 	}
215 
216 	/**
217 	 * Set the template used for method exception log messages.
218 	 * This template can contain any of the following placeholders:
219 	 * <ul>
220 	 * <li>{@code $[targetClassName]}</li>
221 	 * <li>{@code $[targetClassShortName]}</li>
222 	 * <li>{@code $[argumentTypes]}</li>
223 	 * <li>{@code $[arguments]}</li>
224 	 * <li>{@code $[exception]}</li>
225 	 * </ul>
226 	 */
227 	public void setExceptionMessage(String exceptionMessage) {
228 		Assert.hasText(exceptionMessage, "'exceptionMessage' must not be empty");
229 		checkForInvalidPlaceholders(exceptionMessage);
230 		Assert.doesNotContain(exceptionMessage, PLACEHOLDER_RETURN_VALUE,
231 				"exceptionMessage cannot contain placeholder [" + PLACEHOLDER_RETURN_VALUE + "]");
232 		Assert.doesNotContain(exceptionMessage, PLACEHOLDER_INVOCATION_TIME,
233 				"exceptionMessage cannot contain placeholder [" + PLACEHOLDER_INVOCATION_TIME + "]");
234 		this.exceptionMessage = exceptionMessage;
235 	}

238 	/**
239 	 * Writes a log message before the invocation based on the value of {@code enterMessage}.
240 	 * If the invocation succeeds, then a log message is written on exit based on the value
241 	 * {@code exitMessage}. If an exception occurs during invocation, then a message is
242 	 * written based on the value of {@code exceptionMessage}.
243 	 * @see #setEnterMessage
244 	 * @see #setExitMessage
245 	 * @see #setExceptionMessage
246 	 */
247 	@Override
248 	protected Object invokeUnderTrace(MethodInvocation invocation, Log logger) throws Throwable {
249 		String name = invocation.getMethod().getDeclaringClass().getName() + "." + invocation.getMethod().getName();
250 		StopWatch stopWatch = new StopWatch(name);
251 		Object returnValue = null;
252 		boolean exitThroughException = false;
253 		try {
254 			stopWatch.start(name);
255 			writeToLog(logger,
256 					replacePlaceholders(this.enterMessage, invocation, null, null, -1));
257 			returnValue = invocation.proceed();
258 			return returnValue;
259 		}
260 		catch (Throwable ex) {
261 			if (stopWatch.isRunning()) {
262 				stopWatch.stop();
263 			}
264 			exitThroughException = true;
265 			writeToLog(logger,
266 					replacePlaceholders(this.exceptionMessage, invocation, null, ex, stopWatch.getTotalTimeMillis()), ex);
267 			throw ex;
268 		}
269 		finally {
270 			if (!exitThroughException) {
271 				if (stopWatch.isRunning()) {
272 					stopWatch.stop();
273 				}
274 				writeToLog(logger,
275 						replacePlaceholders(this.exitMessage, invocation, returnValue, null, stopWatch.getTotalTimeMillis()));
276 			}
277 		}
278 	}

280 	/**
281 	 * Writes the supplied message to the supplied {@code Log} instance.
282 	 * @see #writeToLog(org.apache.commons.logging.Log, String, Throwable)
283 	 */
284 	protected void writeToLog(Log logger, String message) {
285 		writeToLog(logger, message, null);
286 	}
287 
288 	/**
289 	 * Writes the supplied message and {@link Throwable} to the
290 	 * supplied {@code Log} instance. By default messages are written
291 	 * at {@code TRACE} level. Sub-classes can override this method
292 	 * to control which level the message is written at.
293 	 */
294 	protected void writeToLog(Log logger, String message, Throwable ex) {
295 		if (ex != null) {
296 			logger.trace(message, ex);
297 		}
298 		else {
299 			logger.trace(message);
300 		}
301 	}

303 	/**
304 	 * Replace the placeholders in the given message with the supplied values,
305 	 * or values derived from those supplied.
306 	 * @param message the message template containing the placeholders to be replaced
307 	 * @param methodInvocation the {@code MethodInvocation} being logged.
308 	 * Used to derive values for all placeholders except {@code $[exception]}
309 	 * and {@code $[returnValue]}.
310 	 * @param returnValue any value returned by the invocation.
311 	 * Used to replace the {@code $[returnValue]} placeholder. May be {@code null}.
312 	 * @param throwable any {@code Throwable} raised during the invocation.
313 	 * The value of {@code Throwable.toString()} is replaced for the
314 	 * {@code $[exception]} placeholder. May be {@code null}.
315 	 * @param invocationTime the value to write in place of the
316 	 * {@code $[invocationTime]} placeholder
317 	 * @return the formatted output to write to the log
318 	 */
319 	protected String replacePlaceholders(String message, MethodInvocation methodInvocation,
320 			Object returnValue, Throwable throwable, long invocationTime) {
321 
322 		Matcher matcher = PATTERN.matcher(message);
323 
324 		StringBuffer output = new StringBuffer();
325 		while (matcher.find()) {
326 			String match = matcher.group();
327 			if (PLACEHOLDER_METHOD_NAME.equals(match)) {
328 				matcher.appendReplacement(output, Matcher.quoteReplacement(methodInvocation.getMethod().getName()));
329 			}
330 			else if (PLACEHOLDER_TARGET_CLASS_NAME.equals(match)) {
331 				String className = getClassForLogging(methodInvocation.getThis()).getName();
332 				matcher.appendReplacement(output, Matcher.quoteReplacement(className));
333 			}
334 			else if (PLACEHOLDER_TARGET_CLASS_SHORT_NAME.equals(match)) {
335 				String shortName = ClassUtils.getShortName(getClassForLogging(methodInvocation.getThis()));
336 				matcher.appendReplacement(output, Matcher.quoteReplacement(shortName));
337 			}
338 			else if (PLACEHOLDER_ARGUMENTS.equals(match)) {
339 				matcher.appendReplacement(output,
340 						Matcher.quoteReplacement(StringUtils.arrayToCommaDelimitedString(methodInvocation.getArguments())));
341 			}
342 			else if (PLACEHOLDER_ARGUMENT_TYPES.equals(match)) {
343 				appendArgumentTypes(methodInvocation, matcher, output);
344 			}
345 			else if (PLACEHOLDER_RETURN_VALUE.equals(match)) {
346 				appendReturnValue(methodInvocation, matcher, output, returnValue);
347 			}
348 			else if (throwable != null && PLACEHOLDER_EXCEPTION.equals(match)) {
349 				matcher.appendReplacement(output, Matcher.quoteReplacement(throwable.toString()));
350 			}
351 			else if (PLACEHOLDER_INVOCATION_TIME.equals(match)) {
352 				matcher.appendReplacement(output, Long.toString(invocationTime));
353 			}
354 			else {
355 				// Should not happen since placeholders are checked earlier.
356 				throw new IllegalArgumentException("Unknown placeholder [" + match + "]");
357 			}
358 		}
359 		matcher.appendTail(output);
360 
361 		return output.toString();
362 	}

364 	/**
365 	 * Adds the {@code String} representation of the method return value
366 	 * to the supplied {@code StringBuffer}. Correctly handles
367 	 * {@code null} and {@code void} results.
368 	 * @param methodInvocation the {@code MethodInvocation} that returned the value
369 	 * @param matcher the {@code Matcher} containing the matched placeholder
370 	 * @param output the {@code StringBuffer} to write output to
371 	 * @param returnValue the value returned by the method invocation.
372 	 */
373 	private void appendReturnValue(
374 			MethodInvocation methodInvocation, Matcher matcher, StringBuffer output, Object returnValue) {
375 
376 		if (methodInvocation.getMethod().getReturnType() == void.class) {
377 			matcher.appendReplacement(output, "void");
378 		}
379 		else if (returnValue == null) {
380 			matcher.appendReplacement(output, "null");
381 		}
382 		else {
383 			matcher.appendReplacement(output, Matcher.quoteReplacement(returnValue.toString()));
384 		}
385 	}
386 
387 	/**
388 	 * Adds a comma-separated list of the short {@code Class} names of the
389 	 * method argument types to the output. For example, if a method has signature
390 	 * {@code put(java.lang.String, java.lang.Object)} then the value returned
391 	 * will be {@code String, Object}.
392 	 * @param methodInvocation the {@code MethodInvocation} being logged.
393 	 * Arguments will be retrieved from the corresponding {@code Method}.
394 	 * @param matcher the {@code Matcher} containing the state of the output
395 	 * @param output the {@code StringBuffer} containing the output
396 	 */
397 	private void appendArgumentTypes(MethodInvocation methodInvocation, Matcher matcher, StringBuffer output) {
398 		Class<?>[] argumentTypes = methodInvocation.getMethod().getParameterTypes();
399 		String[] argumentTypeShortNames = new String[argumentTypes.length];
400 		for (int i = 0; i < argumentTypeShortNames.length; i++) {
401 			argumentTypeShortNames[i] = ClassUtils.getShortName(argumentTypes[i]);
402 		}
403 		matcher.appendReplacement(output,
404 				Matcher.quoteReplacement(StringUtils.arrayToCommaDelimitedString(argumentTypeShortNames)));
405 	}

407 	/**
408 	 * Checks to see if the supplied {@code String} has any placeholders
409 	 * that are not specified as constants on this class and throws an
410 	 * {@code IllegalArgumentException} if so.
411 	 */
412 	private void checkForInvalidPlaceholders(String message) throws IllegalArgumentException {
413 		Matcher matcher = PATTERN.matcher(message);
414 		while (matcher.find()) {
415 			String match = matcher.group();
416 			if (!ALLOWED_PLACEHOLDERS.contains(match)) {
417 				throw new IllegalArgumentException("Placeholder [" + match + "] is not valid");
418 			}
419 		}
420 	}
