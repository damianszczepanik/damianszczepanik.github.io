49  /**
50   * {@link org.springframework.beans.factory.config.BeanPostProcessor} implementation
51   * that wraps each eligible bean with an AOP proxy, delegating to specified interceptors
52   * before invoking the bean itself.
53   *
54   * <p>This class distinguishes between "common" interceptors: shared for all proxies it
55   * creates, and "specific" interceptors: unique per bean instance. There need not
56   * be any common interceptors. If there are, they are set using the interceptorNames
57   * property. As with ProxyFactoryBean, interceptors names in the current factory
58   * are used rather than bean references to allow correct handling of prototype
59   * advisors and interceptors: for example, to support stateful mixins.
60   * Any advice type is supported for "interceptorNames" entries.
61   *
62   * <p>Such auto-proxying is particularly useful if there's a large number of beans that
63   * need to be wrapped with similar proxies, i.e. delegating to the same interceptors.
64   * Instead of x repetitive proxy definitions for x target beans, you can register
65   * one single such post processor with the bean factory to achieve the same effect.
66   *
67   * <p>Subclasses can apply any strategy to decide if a bean is to be proxied,
68   * e.g. by type, by name, by definition details, etc. They can also return
69   * additional interceptors that should just be applied to the specific bean
70   * instance. The default concrete implementation is BeanNameAutoProxyCreator,
71   * identifying the beans to be proxied via a list of bean names.
72   *
73   * <p>Any number of {@link TargetSourceCreator} implementations can be used to create
74   * a custom target source - for example, to pool prototype objects. Auto-proxying will
75   * occur even if there is no advice, as long as a TargetSourceCreator specifies a custom
76   * {@link org.springframework.aop.TargetSource}. If there are no TargetSourceCreators set,
77   * or if none matches, a {@link org.springframework.aop.target.SingletonTargetSource}
78   * will be used by default to wrap the target bean instance.
79   *
80   * @author Juergen Hoeller
81   * @author Rod Johnson
82   * @author Rob Harrop
83   * @since 13.10.2003
84   * @see #setInterceptorNames
85   * @see #getAdvicesAndAdvisorsForBean
86   * @see BeanNameAutoProxyCreator
87   * @see DefaultAdvisorAutoProxyCreator
88   */
89  @SuppressWarnings("serial")
90  public abstract class AbstractAutoProxyCreator extends ProxyProcessorSupport
91  		implements SmartInstantiationAwareBeanPostProcessor, BeanFactoryAware {

227 	public Object getEarlyBeanReference(Object bean, String beanName) throws BeansException {
228 		Object cacheKey = getCacheKey(bean.getClass(), beanName);
229 		if (!this.earlyProxyReferences.contains(cacheKey)) {
230 			this.earlyProxyReferences.add(cacheKey);
231 		}
232 		return wrapIfNecessary(bean, beanName, cacheKey);
233 	}

235 	@Override
236 	public Object postProcessBeforeInstantiation(Class<?> beanClass, String beanName) throws BeansException {
237 		Object cacheKey = getCacheKey(beanClass, beanName);
238 
239 		if (beanName == null || !this.targetSourcedBeans.contains(beanName)) {
240 			if (this.advisedBeans.containsKey(cacheKey)) {
241 				return null;
242 			}
243 			if (isInfrastructureClass(beanClass) || shouldSkip(beanClass, beanName)) {
244 				this.advisedBeans.put(cacheKey, Boolean.FALSE);
245 				return null;
246 			}
247 		}
248 
249 		// Create proxy here if we have a custom TargetSource.
250 		// Suppresses unnecessary default instantiation of the target bean:
251 		// The TargetSource will handle target instances in a custom fashion.
252 		if (beanName != null) {
253 			TargetSource targetSource = getCustomTargetSource(beanClass, beanName);
254 			if (targetSource != null) {
255 				this.targetSourcedBeans.add(beanName);
256 				Object[] specificInterceptors = getAdvicesAndAdvisorsForBean(beanClass, beanName, targetSource);
257 				Object proxy = createProxy(beanClass, beanName, specificInterceptors, targetSource);
258 				this.proxyTypes.put(cacheKey, proxy.getClass());
259 				return proxy;
260 			}
261 		}
262 
263 		return null;
264 	}

310 	/**
311 	 * Wrap the given bean if necessary, i.e. if it is eligible for being proxied.
312 	 * @param bean the raw bean instance
313 	 * @param beanName the name of the bean
314 	 * @param cacheKey the cache key for metadata access
315 	 * @return a proxy wrapping the bean, or the raw bean instance as-is
316 	 */
317 	protected Object wrapIfNecessary(Object bean, String beanName, Object cacheKey) {
318 		if (beanName != null && this.targetSourcedBeans.contains(beanName)) {
319 			return bean;
320 		}
321 		if (Boolean.FALSE.equals(this.advisedBeans.get(cacheKey))) {
322 			return bean;
323 		}
324 		if (isInfrastructureClass(bean.getClass()) || shouldSkip(bean.getClass(), beanName)) {
325 			this.advisedBeans.put(cacheKey, Boolean.FALSE);
326 			return bean;
327 		}
328 
329 		// Create proxy if we have advice.
330 		Object[] specificInterceptors = getAdvicesAndAdvisorsForBean(bean.getClass(), beanName, null);
331 		if (specificInterceptors != DO_NOT_PROXY) {
332 			this.advisedBeans.put(cacheKey, Boolean.TRUE);
333 			Object proxy = createProxy(bean.getClass(), beanName, specificInterceptors, new SingletonTargetSource(bean));
334 			this.proxyTypes.put(cacheKey, proxy.getClass());
335 			return proxy;
336 		}
337 
338 		this.advisedBeans.put(cacheKey, Boolean.FALSE);
339 		return bean;
340 	}

343 	 * Return whether the given bean class represents an infrastructure class
344 	 * that should never be proxied.
345 	 * <p>The default implementation considers Advices, Advisors and
346 	 * AopInfrastructureBeans as infrastructure classes.
347 	 * @param beanClass the class of the bean
348 	 * @return whether the bean represents an infrastructure class
349 	 * @see org.aopalliance.aop.Advice
350 	 * @see org.springframework.aop.Advisor
351 	 * @see org.springframework.aop.framework.AopInfrastructureBean
352 	 * @see #shouldSkip
353 	 */
354 	protected boolean isInfrastructureClass(Class<?> beanClass) {
355 		boolean retVal = Advice.class.isAssignableFrom(beanClass) ||
356 				Advisor.class.isAssignableFrom(beanClass) ||
357 				AopInfrastructureBean.class.isAssignableFrom(beanClass);
358 		if (retVal && logger.isTraceEnabled()) {
359 			logger.trace("Did not attempt to auto-proxy infrastructure class [" + beanClass.getName() + "]");
360 		}
361 		return retVal;
362 	}

377 	/**
378 	 * Create a target source for bean instances. Uses any TargetSourceCreators if set.
379 	 * Returns {@code null} if no custom TargetSource should be used.
380 	 * <p>This implementation uses the "customTargetSourceCreators" property.
381 	 * Subclasses can override this method to use a different mechanism.
382 	 * @param beanClass the class of the bean to create a TargetSource for
383 	 * @param beanName the name of the bean
384 	 * @return a TargetSource for this bean
385 	 * @see #setCustomTargetSourceCreators
386 	 */
387 	protected TargetSource getCustomTargetSource(Class<?> beanClass, String beanName) {
388 		// We can't create fancy target sources for directly registered singletons.
389 		if (this.customTargetSourceCreators != null &&
390 				this.beanFactory != null && this.beanFactory.containsBean(beanName)) {
391 			for (TargetSourceCreator tsc : this.customTargetSourceCreators) {
392 				TargetSource ts = tsc.getTargetSource(beanClass, beanName);
393 				if (ts != null) {
394 					// Found a matching TargetSource.
395 					if (logger.isDebugEnabled()) {
396 						logger.debug("TargetSourceCreator [" + tsc +
397 								" found custom TargetSource for bean with name '" + beanName + "'");
398 					}
399 					return ts;
400 				}
401 			}
402 		}
403 
404 		// No custom TargetSource found.
405 		return null;
406 	}

408 	/**
409 	 * Create an AOP proxy for the given bean.
410 	 * @param beanClass the class of the bean
411 	 * @param beanName the name of the bean
412 	 * @param specificInterceptors the set of interceptors that is
413 	 * specific to this bean (may be empty, but not null)
414 	 * @param targetSource the TargetSource for the proxy,
415 	 * already pre-configured to access the bean
416 	 * @return the AOP proxy for the bean
417 	 * @see #buildAdvisors
418 	 */
419 	protected Object createProxy(
420 			Class<?> beanClass, String beanName, Object[] specificInterceptors, TargetSource targetSource) {
421 
422 		ProxyFactory proxyFactory = new ProxyFactory();
423 		proxyFactory.copyFrom(this);
424 
425 		if (!proxyFactory.isProxyTargetClass()) {
426 			if (shouldProxyTargetClass(beanClass, beanName)) {
427 				proxyFactory.setProxyTargetClass(true);
428 			}
429 			else {
430 				evaluateProxyInterfaces(beanClass, proxyFactory);
431 			}
432 		}
433 
434 		Advisor[] advisors = buildAdvisors(beanName, specificInterceptors);
435 		for (Advisor advisor : advisors) {
436 			proxyFactory.addAdvisor(advisor);
437 		}
438 
439 		proxyFactory.setTargetSource(targetSource);
440 		customizeProxyFactory(proxyFactory);
441 
442 		proxyFactory.setFrozen(this.freezeProxy);
443 		if (advisorsPreFiltered()) {
444 			proxyFactory.setPreFiltered(true);
445 		}
446 
447 		return proxyFactory.getProxy(getProxyClassLoader());
448 	}

478 	/**
479 	 * Determine the advisors for the given bean, including the specific interceptors
480 	 * as well as the common interceptor, all adapted to the Advisor interface.
481 	 * @param beanName the name of the bean
482 	 * @param specificInterceptors the set of interceptors that is
483 	 * specific to this bean (may be empty, but not null)
484 	 * @return the list of Advisors for the given bean
485 	 */
486 	protected Advisor[] buildAdvisors(String beanName, Object[] specificInterceptors) {
487 		// Handle prototypes correctly...
488 		Advisor[] commonInterceptors = resolveInterceptorNames();
489 
490 		List<Object> allInterceptors = new ArrayList<Object>();
491 		if (specificInterceptors != null) {
492 			allInterceptors.addAll(Arrays.asList(specificInterceptors));
493 			if (commonInterceptors != null) {
494 				if (this.applyCommonInterceptorsFirst) {
495 					allInterceptors.addAll(0, Arrays.asList(commonInterceptors));
496 				}
497 				else {
498 					allInterceptors.addAll(Arrays.asList(commonInterceptors));
499 				}
500 			}
501 		}
502 		if (logger.isDebugEnabled()) {
503 			int nrOfCommonInterceptors = (commonInterceptors != null ? commonInterceptors.length : 0);
504 			int nrOfSpecificInterceptors = (specificInterceptors != null ? specificInterceptors.length : 0);
505 			logger.debug("Creating implicit proxy for bean '" + beanName + "' with " + nrOfCommonInterceptors +
506 					" common interceptors and " + nrOfSpecificInterceptors + " specific interceptors");
507 		}
508 
509 		Advisor[] advisors = new Advisor[allInterceptors.size()];
510 		for (int i = 0; i < allInterceptors.size(); i++) {
511 			advisors[i] = this.advisorAdapterRegistry.wrap(allInterceptors.get(i));
512 		}
513 		return advisors;
514 	}
515 
516 	/**
517 	 * Resolves the specified interceptor names to Advisor objects.
518 	 * @see #setInterceptorNames
519 	 */
520 	private Advisor[] resolveInterceptorNames() {
521 		ConfigurableBeanFactory cbf = (this.beanFactory instanceof ConfigurableBeanFactory) ?
522 				(ConfigurableBeanFactory) this.beanFactory : null;
523 		List<Advisor> advisors = new ArrayList<Advisor>();
524 		for (String beanName : this.interceptorNames) {
525 			if (cbf == null || !cbf.isCurrentlyInCreation(beanName)) {
526 				Object next = this.beanFactory.getBean(beanName);
527 				advisors.add(this.advisorAdapterRegistry.wrap(next));
528 			}
529 		}
530 		return advisors.toArray(new Advisor[advisors.size()]);
531 	}
