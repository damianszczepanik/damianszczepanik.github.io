71  	@Test
72  	public void testCanGetMethodSignatureFromJoinPoint() {
73  		final Object raw = new TestBean();
74  		// Will be set by advice during a method call
75  		final int newAge = 23;
76  
77  		ProxyFactory pf = new ProxyFactory(raw);
78  		pf.setExposeProxy(true);
79  		pf.addAdvisor(ExposeInvocationInterceptor.ADVISOR);
80  		pf.addAdvice(new MethodBeforeAdvice() {
81  			private int depth;
82  
83  			@Override
84  			public void before(Method method, Object[] args, Object target) throws Throwable {
85  				JoinPoint jp = AbstractAspectJAdvice.currentJoinPoint();
86  				assertTrue("Method named in toString", jp.toString().contains(method.getName()));
87  				// Ensure that these don't cause problems
88  				jp.toShortString();
89  				jp.toLongString();
90  
91  				assertSame(target, AbstractAspectJAdvice.currentJoinPoint().getTarget());
92  				assertFalse(AopUtils.isAopProxy(AbstractAspectJAdvice.currentJoinPoint().getTarget()));
93  
94  				ITestBean thisProxy = (ITestBean) AbstractAspectJAdvice.currentJoinPoint().getThis();
95  				assertTrue(AopUtils.isAopProxy(AbstractAspectJAdvice.currentJoinPoint().getThis()));
96  
97  				assertNotSame(target, thisProxy);
98  
99  				// Check getting again doesn't cause a problem
100 				assertSame(thisProxy, AbstractAspectJAdvice.currentJoinPoint().getThis());
101 
102 				// Try reentrant call--will go through this advice.
103 				// Be sure to increment depth to avoid infinite recursion
104 				if (depth++ == 0) {
105 					// Check that toString doesn't cause a problem
106 					thisProxy.toString();
107 					// Change age, so this will be returned by invocation
108 					thisProxy.setAge(newAge);
109 					assertEquals(newAge, thisProxy.getAge());
110 				}
111 
112 				assertSame(AopContext.currentProxy(), thisProxy);
113 				assertSame(target, raw);
114 
115 				assertSame(method.getName(), AbstractAspectJAdvice.currentJoinPoint().getSignature().getName());
116 				assertEquals(method.getModifiers(), AbstractAspectJAdvice.currentJoinPoint().getSignature().getModifiers());
117 
118 				MethodSignature msig = (MethodSignature) AbstractAspectJAdvice.currentJoinPoint().getSignature();
119 				assertSame("Return same MethodSignature repeatedly", msig, AbstractAspectJAdvice.currentJoinPoint().getSignature());
120 				assertSame("Return same JoinPoint repeatedly", AbstractAspectJAdvice.currentJoinPoint(), AbstractAspectJAdvice.currentJoinPoint());
121 				assertEquals(method.getDeclaringClass(), msig.getDeclaringType());
122 				assertTrue(Arrays.equals(method.getParameterTypes(), msig.getParameterTypes()));
123 				assertEquals(method.getReturnType(), msig.getReturnType());
124 				assertTrue(Arrays.equals(method.getExceptionTypes(), msig.getExceptionTypes()));
125 				msig.toLongString();
126 				msig.toShortString();
127 			}
128 		});
129 		ITestBean itb = (ITestBean) pf.getProxy();
130 		// Any call will do
131 		assertEquals("Advice reentrantly set age", newAge, itb.getAge());
132 	}

187 	@Test
188 	public void toShortAndLongStringFormedCorrectly() throws Exception {
189 		final Object raw = new TestBean();
190 		ProxyFactory pf = new ProxyFactory(raw);
191 		pf.addAdvisor(ExposeInvocationInterceptor.ADVISOR);
192 		pf.addAdvice(new MethodBeforeAdvice() {
193 			@Override
194 			public void before(Method method, Object[] args, Object target) throws Throwable {
195 				// makeEncSJP, although meant for computing the enclosing join point,
196 				// it serves our purpose here
197 				JoinPoint.StaticPart aspectJVersionJp = Factory.makeEncSJP(method);
198 				JoinPoint jp = AbstractAspectJAdvice.currentJoinPoint();
199 
200 				assertEquals(aspectJVersionJp.getSignature().toLongString(), jp.getSignature().toLongString());
201 				assertEquals(aspectJVersionJp.getSignature().toShortString(), jp.getSignature().toShortString());
202 				assertEquals(aspectJVersionJp.getSignature().toString(), jp.getSignature().toString());
203 
204 				assertEquals(aspectJVersionJp.toLongString(), jp.toLongString());
205 				assertEquals(aspectJVersionJp.toShortString(), jp.toShortString());
206 				assertEquals(aspectJVersionJp.toString(), jp.toString());
207 			}
208 		});
209 		ITestBean itb = (ITestBean) pf.getProxy();
210 		itb.getAge();
211 		itb.setName("foo");
212 		itb.getDoctor();
213 		itb.getStringArray();
214 		itb.getSpouses();
215 		itb.setSpouse(new TestBean());
216 		try {
217 			itb.unreliableFileOperation();
218 		} catch (IOException ex) {
219 			// we don't realy care...
220 		}
221 	}
