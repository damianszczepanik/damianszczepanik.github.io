139 	public void testReplaceAdvisor() {
140 		TestBean target = new TestBean();
141 		ProxyFactory pf = new ProxyFactory(target);
142 		NopInterceptor nop = new NopInterceptor();
143 		CountingBeforeAdvice cba1 = new CountingBeforeAdvice();
144 		CountingBeforeAdvice cba2 = new CountingBeforeAdvice();
145 		Advisor advisor1 = new DefaultPointcutAdvisor(cba1);
146 		Advisor advisor2 = new DefaultPointcutAdvisor(cba2);
147 		pf.addAdvisor(advisor1);
148 		pf.addAdvice(nop);
149 		ITestBean proxied = (ITestBean) pf.getProxy();
150 		// Use the type cast feature
151 		// Replace etc methods on advised should be same as on ProxyFactory
152 		Advised advised = (Advised) proxied;
153 		proxied.setAge(5);
154 		assertEquals(1, cba1.getCalls());
155 		assertEquals(0, cba2.getCalls());
156 		assertEquals(1, nop.getCount());
157 		assertFalse(advised.replaceAdvisor(new DefaultPointcutAdvisor(new NopInterceptor()), advisor2));
158 		assertTrue(advised.replaceAdvisor(advisor1, advisor2));
159 		assertEquals(advisor2, pf.getAdvisors()[0]);
160 		assertEquals(5, proxied.getAge());
161 		assertEquals(1, cba1.getCalls());
162 		assertEquals(2, nop.getCount());
163 		assertEquals(1, cba2.getCalls());
164 		assertFalse(pf.replaceAdvisor(new DefaultPointcutAdvisor(null), advisor1));
165 	}

218 	@Test
219 	public void testInterceptorInclusionMethods() {
220 		class MyInterceptor implements MethodInterceptor {
221 			@Override
222 			public Object invoke(MethodInvocation invocation) throws Throwable {
223 				throw new UnsupportedOperationException();
224 			}
225 		}
226 
227 		NopInterceptor di = new NopInterceptor();
228 		NopInterceptor diUnused = new NopInterceptor();
229 		ProxyFactory factory = new ProxyFactory(new TestBean());
230 		factory.addAdvice(0, di);
231 		assertThat(factory.getProxy(), instanceOf(ITestBean.class));
232 		assertTrue(factory.adviceIncluded(di));
233 		assertTrue(!factory.adviceIncluded(diUnused));
234 		assertTrue(factory.countAdvicesOfType(NopInterceptor.class) == 1);
235 		assertTrue(factory.countAdvicesOfType(MyInterceptor.class) == 0);
236 
237 		factory.addAdvice(0, diUnused);
238 		assertTrue(factory.adviceIncluded(diUnused));
239 		assertTrue(factory.countAdvicesOfType(NopInterceptor.class) == 2);
240 	}

245 	@Test
246 	public void testCanAddAndRemoveAspectInterfacesOnSingleton() {
247 		ProxyFactory config = new ProxyFactory(new TestBean());
248 
249 		assertFalse("Shouldn't implement TimeStamped before manipulation",
250 				config.getProxy() instanceof TimeStamped);
251 
252 		long time = 666L;
253 		TimestampIntroductionInterceptor ti = new TimestampIntroductionInterceptor();
254 		ti.setTime(time);
255 
256 		// Add to front of interceptor chain
257 		int oldCount = config.getAdvisors().length;
258 		config.addAdvisor(0, new DefaultIntroductionAdvisor(ti, TimeStamped.class));
259 
260 		assertTrue(config.getAdvisors().length == oldCount + 1);
261 
262 		TimeStamped ts = (TimeStamped) config.getProxy();
263 		assertTrue(ts.getTimeStamp() == time);
264 
265 		// Can remove
266 		config.removeAdvice(ti);
267 
268 		assertTrue(config.getAdvisors().length == oldCount);
269 
270 		try {
271 			// Existing reference will fail
272 			ts.getTimeStamp();
273 			fail("Existing object won't implement this interface any more");
274 		}
275 		catch (RuntimeException ex) {
276 		}
277 
278 		assertFalse("Should no longer implement TimeStamped",
279 				config.getProxy() instanceof TimeStamped);
280 
281 		// Now check non-effect of removing interceptor that isn't there
282 		config.removeAdvice(new DebugInterceptor());
283 
284 		assertTrue(config.getAdvisors().length == oldCount);
285 
286 		ITestBean it = (ITestBean) ts;
287 		DebugInterceptor debugInterceptor = new DebugInterceptor();
288 		config.addAdvice(0, debugInterceptor);
289 		it.getSpouse();
290 		assertEquals(1, debugInterceptor.getCount());
291 		config.removeAdvice(debugInterceptor);
292 		it.getSpouse();
293 		// not invoked again
294 		assertTrue(debugInterceptor.getCount() == 1);
295 	}

343 			implements TimeStamped {
