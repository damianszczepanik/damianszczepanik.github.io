17  package org.springframework.aop.aspectj;
18  
19  import org.junit.Test;
20  
21  import org.springframework.beans.factory.BeanFactory;
22  import org.springframework.beans.factory.support.DefaultListableBeanFactory;
23  import org.springframework.tests.sample.beans.CountingTestBean;
24  import org.springframework.tests.sample.beans.IOther;
25  import org.springframework.tests.sample.beans.ITestBean;
26  import org.springframework.tests.sample.beans.TestBean;
27  import org.springframework.tests.sample.beans.subpkg.DeepBean;
28  
29  import static org.junit.Assert.*;
30  
31  /**
32   * Unit tests for the {@link TypePatternClassFilter} class.
33   *
34   * @author Rod Johnson
35   * @author Rick Evans
36   * @author Chris Beams
37   */
38  public final class TypePatternClassFilterTests {
39  
40  	@Test(expected=IllegalArgumentException.class)
41  	public void testInvalidPattern() {
42  		// should throw - pattern must be recognized as invalid
43  		new TypePatternClassFilter("-");
44  	}
45  
46  	@Test
47  	public void testValidPatternMatching() {
48  		TypePatternClassFilter tpcf = new TypePatternClassFilter("org.springframework.tests.sample.beans.*");
49  		assertTrue("Must match: in package", tpcf.matches(TestBean.class));
50  		assertTrue("Must match: in package", tpcf.matches(ITestBean.class));
51  		assertTrue("Must match: in package", tpcf.matches(IOther.class));
52  		assertFalse("Must be excluded: in wrong package", tpcf.matches(DeepBean.class));
53  		assertFalse("Must be excluded: in wrong package", tpcf.matches(BeanFactory.class));
54  		assertFalse("Must be excluded: in wrong package", tpcf.matches(DefaultListableBeanFactory.class));
55  	}
56  
57  	@Test
58  	public void testSubclassMatching() {
59  		TypePatternClassFilter tpcf = new TypePatternClassFilter("org.springframework.tests.sample.beans.ITestBean+");
60  		assertTrue("Must match: in package", tpcf.matches(TestBean.class));
61  		assertTrue("Must match: in package", tpcf.matches(ITestBean.class));
62  		assertTrue("Must match: in package", tpcf.matches(CountingTestBean.class));
63  		assertFalse("Must be excluded: not subclass", tpcf.matches(IOther.class));
64  		assertFalse("Must be excluded: not subclass", tpcf.matches(DefaultListableBeanFactory.class));
65  	}
66  
67  	@Test
68  	public void testAndOrNotReplacement() {
69  		TypePatternClassFilter tpcf = new TypePatternClassFilter("java.lang.Object or java.lang.String");
70  		assertFalse("matches Number",tpcf.matches(Number.class));
71  		assertTrue("matches Object",tpcf.matches(Object.class));
72  		assertTrue("matchesString",tpcf.matches(String.class));
73  		tpcf = new TypePatternClassFilter("java.lang.Number+ and java.lang.Float");
74  		assertTrue("matches Float",tpcf.matches(Float.class));
75  		assertFalse("matches Double",tpcf.matches(Double.class));
76  		tpcf = new TypePatternClassFilter("java.lang.Number+ and not java.lang.Float");
77  		assertFalse("matches Float",tpcf.matches(Float.class));
78  		assertTrue("matches Double",tpcf.matches(Double.class));
79  	}
80  
81  	@Test(expected=IllegalArgumentException.class)
82  	public void testSetTypePatternWithNullArgument() throws Exception {
83  		new TypePatternClassFilter(null);
84  	}
85  
86  	@Test(expected=IllegalStateException.class)
87  	public void testInvocationOfMatchesMethodBlowsUpWhenNoTypePatternHasBeenSet() throws Exception {
88  		new TypePatternClassFilter().matches(String.class);
89  	}
90  
91  }
