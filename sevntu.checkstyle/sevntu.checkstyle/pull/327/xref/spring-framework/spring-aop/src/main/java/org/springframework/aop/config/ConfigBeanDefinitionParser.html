98  	@Override
99  	public BeanDefinition parse(Element element, ParserContext parserContext) {
100 		CompositeComponentDefinition compositeDef =
101 				new CompositeComponentDefinition(element.getTagName(), parserContext.extractSource(element));
102 		parserContext.pushContainingComponent(compositeDef);
103 
104 		configureAutoProxyCreator(parserContext, element);
105 
106 		List<Element> childElts = DomUtils.getChildElements(element);
107 		for (Element elt: childElts) {
108 			String localName = parserContext.getDelegate().getLocalName(elt);
109 			if (POINTCUT.equals(localName)) {
110 				parsePointcut(elt, parserContext);
111 			}
112 			else if (ADVISOR.equals(localName)) {
113 				parseAdvisor(elt, parserContext);
114 			}
115 			else if (ASPECT.equals(localName)) {
116 				parseAspect(elt, parserContext);
117 			}
118 		}
119 
120 		parserContext.popAndRegisterContainingComponent();
121 		return null;
122 	}

124 	/**
125 	 * Configures the auto proxy creator needed to support the {@link BeanDefinition BeanDefinitions}
126 	 * created by the '{@code &lt;aop:config/&gt;}' tag. Will force class proxying if the
127 	 * '{@code proxy-target-class}' attribute is set to '{@code true}'.
128 	 * @see AopNamespaceUtils
129 	 */
130 	private void configureAutoProxyCreator(ParserContext parserContext, Element element) {
131 		AopNamespaceUtils.registerAspectJAutoProxyCreatorIfNecessary(parserContext, element);
132 	}

134 	/**
135 	 * Parses the supplied {@code &lt;advisor&gt;} element and registers the resulting
136 	 * {@link org.springframework.aop.Advisor} and any resulting {@link org.springframework.aop.Pointcut}
137 	 * with the supplied {@link BeanDefinitionRegistry}.
138 	 */
139 	private void parseAdvisor(Element advisorElement, ParserContext parserContext) {
140 		AbstractBeanDefinition advisorDef = createAdvisorBeanDefinition(advisorElement, parserContext);
141 		String id = advisorElement.getAttribute(ID);
142 
143 		try {
144 			this.parseState.push(new AdvisorEntry(id));
145 			String advisorBeanName = id;
146 			if (StringUtils.hasText(advisorBeanName)) {
147 				parserContext.getRegistry().registerBeanDefinition(advisorBeanName, advisorDef);
148 			}
149 			else {
150 				advisorBeanName = parserContext.getReaderContext().registerWithGeneratedName(advisorDef);
151 			}
152 
153 			Object pointcut = parsePointcutProperty(advisorElement, parserContext);
154 			if (pointcut instanceof BeanDefinition) {
155 				advisorDef.getPropertyValues().add(POINTCUT, pointcut);
156 				parserContext.registerComponent(
157 						new AdvisorComponentDefinition(advisorBeanName, advisorDef, (BeanDefinition) pointcut));
158 			}
159 			else if (pointcut instanceof String) {
160 				advisorDef.getPropertyValues().add(POINTCUT, new RuntimeBeanReference((String) pointcut));
161 				parserContext.registerComponent(
162 						new AdvisorComponentDefinition(advisorBeanName, advisorDef));
163 			}
164 		}
165 		finally {
166 			this.parseState.pop();
167 		}
168 	}

170 	/**
171 	 * Create a {@link RootBeanDefinition} for the advisor described in the supplied. Does <strong>not</strong>
172 	 * parse any associated '{@code pointcut}' or '{@code pointcut-ref}' attributes.
173 	 */
174 	private AbstractBeanDefinition createAdvisorBeanDefinition(Element advisorElement, ParserContext parserContext) {
175 		RootBeanDefinition advisorDefinition = new RootBeanDefinition(DefaultBeanFactoryPointcutAdvisor.class);
176 		advisorDefinition.setSource(parserContext.extractSource(advisorElement));
177 
178 		String adviceRef = advisorElement.getAttribute(ADVICE_REF);
179 		if (!StringUtils.hasText(adviceRef)) {
180 			parserContext.getReaderContext().error(
181 					"'advice-ref' attribute contains empty value.", advisorElement, this.parseState.snapshot());
182 		}
183 		else {
184 			advisorDefinition.getPropertyValues().add(
185 					ADVICE_BEAN_NAME, new RuntimeBeanNameReference(adviceRef));
186 		}
187 
188 		if (advisorElement.hasAttribute(ORDER_PROPERTY)) {
189 			advisorDefinition.getPropertyValues().add(
190 					ORDER_PROPERTY, advisorElement.getAttribute(ORDER_PROPERTY));
191 		}
192 
193 		return advisorDefinition;
194 	}

196 	private void parseAspect(Element aspectElement, ParserContext parserContext) {
197 		String aspectId = aspectElement.getAttribute(ID);
198 		String aspectName = aspectElement.getAttribute(REF);
199 
200 		try {
201 			this.parseState.push(new AspectEntry(aspectId, aspectName));
202 			List<BeanDefinition> beanDefinitions = new ArrayList<BeanDefinition>();
203 			List<BeanReference> beanReferences = new ArrayList<BeanReference>();
204 
205 			List<Element> declareParents = DomUtils.getChildElementsByTagName(aspectElement, DECLARE_PARENTS);
206 			for (int i = METHOD_INDEX; i < declareParents.size(); i++) {
207 				Element declareParentsElement = declareParents.get(i);
208 				beanDefinitions.add(parseDeclareParents(declareParentsElement, parserContext));
209 			}
210 
211 			// We have to parse "advice" and all the advice kinds in one loop, to get the
212 			// ordering semantics right.
213 			NodeList nodeList = aspectElement.getChildNodes();
214 			boolean adviceFoundAlready = false;
215 			for (int i = 0; i < nodeList.getLength(); i++) {
216 				Node node = nodeList.item(i);
217 				if (isAdviceNode(node, parserContext)) {
218 					if (!adviceFoundAlready) {
219 						adviceFoundAlready = true;
220 						if (!StringUtils.hasText(aspectName)) {
221 							parserContext.getReaderContext().error(
222 									"<aspect> tag needs aspect bean reference via 'ref' attribute when declaring advices.",
223 									aspectElement, this.parseState.snapshot());
224 							return;
225 						}
226 						beanReferences.add(new RuntimeBeanReference(aspectName));
227 					}
228 					AbstractBeanDefinition advisorDefinition = parseAdvice(
229 							aspectName, i, aspectElement, (Element) node, parserContext, beanDefinitions, beanReferences);
230 					beanDefinitions.add(advisorDefinition);
231 				}
232 			}
233 
234 			AspectComponentDefinition aspectComponentDefinition = createAspectComponentDefinition(
235 					aspectElement, aspectId, beanDefinitions, beanReferences, parserContext);
236 			parserContext.pushContainingComponent(aspectComponentDefinition);
237 
238 			List<Element> pointcuts = DomUtils.getChildElementsByTagName(aspectElement, POINTCUT);
239 			for (Element pointcutElement : pointcuts) {
240 				parsePointcut(pointcutElement, parserContext);
241 			}
242 
243 			parserContext.popAndRegisterContainingComponent();
244 		}
245 		finally {
246 			this.parseState.pop();
247 		}
248 	}
249 
250 	private AspectComponentDefinition createAspectComponentDefinition(
251 			Element aspectElement, String aspectId, List<BeanDefinition> beanDefs,
252 			List<BeanReference> beanRefs, ParserContext parserContext) {
253 
254 		BeanDefinition[] beanDefArray = beanDefs.toArray(new BeanDefinition[beanDefs.size()]);
255 		BeanReference[] beanRefArray = beanRefs.toArray(new BeanReference[beanRefs.size()]);
256 		Object source = parserContext.extractSource(aspectElement);
257 		return new AspectComponentDefinition(aspectId, beanDefArray, beanRefArray, source);
258 	}

260 	/**
261 	 * Return {@code true} if the supplied node describes an advice type. May be one of:
262 	 * '{@code before}', '{@code after}', '{@code after-returning}',
263 	 * '{@code after-throwing}' or '{@code around}'.
264 	 */
265 	private boolean isAdviceNode(Node aNode, ParserContext parserContext) {
266 		if (!(aNode instanceof Element)) {
267 			return false;
268 		}
269 		else {
270 			String name = parserContext.getDelegate().getLocalName(aNode);
271 			return (BEFORE.equals(name) || AFTER.equals(name) || AFTER_RETURNING_ELEMENT.equals(name) ||
272 					AFTER_THROWING_ELEMENT.equals(name) || AROUND.equals(name));
273 		}
274 	}

276 	/**
277 	 * Parse a '{@code declare-parents}' element and register the appropriate
278 	 * DeclareParentsAdvisor with the BeanDefinitionRegistry encapsulated in the
279 	 * supplied ParserContext.
280 	 */
281 	private AbstractBeanDefinition parseDeclareParents(Element declareParentsElement, ParserContext parserContext) {
282 		BeanDefinitionBuilder builder = BeanDefinitionBuilder.rootBeanDefinition(DeclareParentsAdvisor.class);
283 		builder.addConstructorArgValue(declareParentsElement.getAttribute(IMPLEMENT_INTERFACE));
284 		builder.addConstructorArgValue(declareParentsElement.getAttribute(TYPE_PATTERN));
285 
286 		String defaultImpl = declareParentsElement.getAttribute(DEFAULT_IMPL);
287 		String delegateRef = declareParentsElement.getAttribute(DELEGATE_REF);
288 
289 		if (StringUtils.hasText(defaultImpl) && !StringUtils.hasText(delegateRef)) {
290 			builder.addConstructorArgValue(defaultImpl);
291 		}
292 		else if (StringUtils.hasText(delegateRef) && !StringUtils.hasText(defaultImpl)) {
293 			builder.addConstructorArgReference(delegateRef);
294 		}
295 		else {
296 			parserContext.getReaderContext().error(
297 					"Exactly one of the " + DEFAULT_IMPL + " or " + DELEGATE_REF + " attributes must be specified",
298 					declareParentsElement, this.parseState.snapshot());
299 		}
300 
301 		AbstractBeanDefinition definition = builder.getBeanDefinition();
302 		definition.setSource(parserContext.extractSource(declareParentsElement));
303 		parserContext.getReaderContext().registerWithGeneratedName(definition);
304 		return definition;
305 	}

307 	/**
308 	 * Parses one of '{@code before}', '{@code after}', '{@code after-returning}',
309 	 * '{@code after-throwing}' or '{@code around}' and registers the resulting
310 	 * BeanDefinition with the supplied BeanDefinitionRegistry.
311 	 * @return the generated advice RootBeanDefinition
312 	 */
313 	private AbstractBeanDefinition parseAdvice(
314 			String aspectName, int order, Element aspectElement, Element adviceElement, ParserContext parserContext,
315 			List<BeanDefinition> beanDefinitions, List<BeanReference> beanReferences) {
316 
317 		try {
318 			this.parseState.push(new AdviceEntry(parserContext.getDelegate().getLocalName(adviceElement)));
319 
320 			// create the method factory bean
321 			RootBeanDefinition methodDefinition = new RootBeanDefinition(MethodLocatingFactoryBean.class);
322 			methodDefinition.getPropertyValues().add("targetBeanName", aspectName);
323 			methodDefinition.getPropertyValues().add("methodName", adviceElement.getAttribute("method"));
324 			methodDefinition.setSynthetic(true);
325 
326 			// create instance factory definition
327 			RootBeanDefinition aspectFactoryDef =
328 					new RootBeanDefinition(SimpleBeanFactoryAwareAspectInstanceFactory.class);
329 			aspectFactoryDef.getPropertyValues().add("aspectBeanName", aspectName);
330 			aspectFactoryDef.setSynthetic(true);
331 
332 			// register the pointcut
333 			AbstractBeanDefinition adviceDef = createAdviceDefinition(
334 					adviceElement, parserContext, aspectName, order, methodDefinition, aspectFactoryDef,
335 					beanDefinitions, beanReferences);
336 
337 			// configure the advisor
338 			RootBeanDefinition advisorDefinition = new RootBeanDefinition(AspectJPointcutAdvisor.class);
339 			advisorDefinition.setSource(parserContext.extractSource(adviceElement));
340 			advisorDefinition.getConstructorArgumentValues().addGenericArgumentValue(adviceDef);
341 			if (aspectElement.hasAttribute(ORDER_PROPERTY)) {
342 				advisorDefinition.getPropertyValues().add(
343 						ORDER_PROPERTY, aspectElement.getAttribute(ORDER_PROPERTY));
344 			}
345 
346 			// register the final advisor
347 			parserContext.getReaderContext().registerWithGeneratedName(advisorDefinition);
348 
349 			return advisorDefinition;
350 		}
351 		finally {
352 			this.parseState.pop();
353 		}
354 	}

356 	/**
357 	 * Creates the RootBeanDefinition for a POJO advice bean. Also causes pointcut
358 	 * parsing to occur so that the pointcut may be associate with the advice bean.
359 	 * This same pointcut is also configured as the pointcut for the enclosing
360 	 * Advisor definition using the supplied MutablePropertyValues.
361 	 */
362 	private AbstractBeanDefinition createAdviceDefinition(
363 			Element adviceElement, ParserContext parserContext, String aspectName, int order,
364 			RootBeanDefinition methodDef, RootBeanDefinition aspectFactoryDef,
365 			List<BeanDefinition> beanDefinitions, List<BeanReference> beanReferences) {
366 
367 		RootBeanDefinition adviceDefinition = new RootBeanDefinition(getAdviceClass(adviceElement, parserContext));
368 		adviceDefinition.setSource(parserContext.extractSource(adviceElement));
369 
370 		adviceDefinition.getPropertyValues().add(ASPECT_NAME_PROPERTY, aspectName);
371 		adviceDefinition.getPropertyValues().add(DECLARATION_ORDER_PROPERTY, order);
372 
373 		if (adviceElement.hasAttribute(RETURNING)) {
374 			adviceDefinition.getPropertyValues().add(
375 					RETURNING_PROPERTY, adviceElement.getAttribute(RETURNING));
376 		}
377 		if (adviceElement.hasAttribute(THROWING)) {
378 			adviceDefinition.getPropertyValues().add(
379 					THROWING_PROPERTY, adviceElement.getAttribute(THROWING));
380 		}
381 		if (adviceElement.hasAttribute(ARG_NAMES)) {
382 			adviceDefinition.getPropertyValues().add(
383 					ARG_NAMES_PROPERTY, adviceElement.getAttribute(ARG_NAMES));
384 		}
385 
386 		ConstructorArgumentValues cav = adviceDefinition.getConstructorArgumentValues();
387 		cav.addIndexedArgumentValue(METHOD_INDEX, methodDef);
388 
389 		Object pointcut = parsePointcutProperty(adviceElement, parserContext);
390 		if (pointcut instanceof BeanDefinition) {
391 			cav.addIndexedArgumentValue(POINTCUT_INDEX, pointcut);
392 			beanDefinitions.add((BeanDefinition) pointcut);
393 		}
394 		else if (pointcut instanceof String) {
395 			RuntimeBeanReference pointcutRef = new RuntimeBeanReference((String) pointcut);
396 			cav.addIndexedArgumentValue(POINTCUT_INDEX, pointcutRef);
397 			beanReferences.add(pointcutRef);
398 		}
399 
400 		cav.addIndexedArgumentValue(ASPECT_INSTANCE_FACTORY_INDEX, aspectFactoryDef);
401 
402 		return adviceDefinition;
403 	}

405 	/**
406 	 * Gets the advice implementation class corresponding to the supplied {@link Element}.
407 	 */
408 	private Class<?> getAdviceClass(Element adviceElement, ParserContext parserContext) {
409 		String elementName = parserContext.getDelegate().getLocalName(adviceElement);
410 		if (BEFORE.equals(elementName)) {
411 			return AspectJMethodBeforeAdvice.class;
412 		}
413 		else if (AFTER.equals(elementName)) {
414 			return AspectJAfterAdvice.class;
415 		}
416 		else if (AFTER_RETURNING_ELEMENT.equals(elementName)) {
417 			return AspectJAfterReturningAdvice.class;
418 		}
419 		else if (AFTER_THROWING_ELEMENT.equals(elementName)) {
420 			return AspectJAfterThrowingAdvice.class;
421 		}
422 		else if (AROUND.equals(elementName)) {
423 			return AspectJAroundAdvice.class;
424 		}
425 		else {
426 			throw new IllegalArgumentException("Unknown advice kind [" + elementName + "].");
427 		}
428 	}

430 	/**
431 	 * Parses the supplied {@code &lt;pointcut&gt;} and registers the resulting
432 	 * Pointcut with the BeanDefinitionRegistry.
433 	 */
434 	private AbstractBeanDefinition parsePointcut(Element pointcutElement, ParserContext parserContext) {
435 		String id = pointcutElement.getAttribute(ID);
436 		String expression = pointcutElement.getAttribute(EXPRESSION);
437 
438 		AbstractBeanDefinition pointcutDefinition = null;
439 
440 		try {
441 			this.parseState.push(new PointcutEntry(id));
442 			pointcutDefinition = createPointcutDefinition(expression);
443 			pointcutDefinition.setSource(parserContext.extractSource(pointcutElement));
444 
445 			String pointcutBeanName = id;
446 			if (StringUtils.hasText(pointcutBeanName)) {
447 				parserContext.getRegistry().registerBeanDefinition(pointcutBeanName, pointcutDefinition);
448 			}
449 			else {
450 				pointcutBeanName = parserContext.getReaderContext().registerWithGeneratedName(pointcutDefinition);
451 			}
452 
453 			parserContext.registerComponent(
454 					new PointcutComponentDefinition(pointcutBeanName, pointcutDefinition, expression));
455 		}
456 		finally {
457 			this.parseState.pop();
458 		}
459 
460 		return pointcutDefinition;
461 	}

463 	/**
464 	 * Parses the {@code pointcut} or {@code pointcut-ref} attributes of the supplied
465 	 * {@link Element} and add a {@code pointcut} property as appropriate. Generates a
466 	 * {@link org.springframework.beans.factory.config.BeanDefinition} for the pointcut if  necessary
467 	 * and returns its bean name, otherwise returns the bean name of the referred pointcut.
468 	 */
469 	private Object parsePointcutProperty(Element element, ParserContext parserContext) {
470 		if (element.hasAttribute(POINTCUT) && element.hasAttribute(POINTCUT_REF)) {
471 			parserContext.getReaderContext().error(
472 					"Cannot define both 'pointcut' and 'pointcut-ref' on <advisor> tag.",
473 					element, this.parseState.snapshot());
474 			return null;
475 		}
476 		else if (element.hasAttribute(POINTCUT)) {
477 			// Create a pointcut for the anonymous pc and register it.
478 			String expression = element.getAttribute(POINTCUT);
479 			AbstractBeanDefinition pointcutDefinition = createPointcutDefinition(expression);
480 			pointcutDefinition.setSource(parserContext.extractSource(element));
481 			return pointcutDefinition;
482 		}
483 		else if (element.hasAttribute(POINTCUT_REF)) {
484 			String pointcutRef = element.getAttribute(POINTCUT_REF);
485 			if (!StringUtils.hasText(pointcutRef)) {
486 				parserContext.getReaderContext().error(
487 						"'pointcut-ref' attribute contains empty value.", element, this.parseState.snapshot());
488 				return null;
489 			}
490 			return pointcutRef;
491 		}
492 		else {
493 			parserContext.getReaderContext().error(
494 					"Must define one of 'pointcut' or 'pointcut-ref' on <advisor> tag.",
495 					element, this.parseState.snapshot());
496 			return null;
497 		}
498 	}
499 
500 	/**
501 	 * Creates a {@link BeanDefinition} for the {@link AspectJExpressionPointcut} class using
502 	 * the supplied pointcut expression.
503 	 */
504 	protected AbstractBeanDefinition createPointcutDefinition(String expression) {
505 		RootBeanDefinition beanDefinition = new RootBeanDefinition(AspectJExpressionPointcut.class);
506 		beanDefinition.setScope(BeanDefinition.SCOPE_PROTOTYPE);
507 		beanDefinition.setSynthetic(true);
508 		beanDefinition.getPropertyValues().add(EXPRESSION, expression);
509 		return beanDefinition;
510 	}
