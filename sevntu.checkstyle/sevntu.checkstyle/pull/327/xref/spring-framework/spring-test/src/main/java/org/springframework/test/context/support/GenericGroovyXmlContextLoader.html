17  package org.springframework.test.context.support;
18  
19  import org.springframework.beans.factory.groovy.GroovyBeanDefinitionReader;
20  import org.springframework.context.support.GenericApplicationContext;
21  import org.springframework.test.context.MergedContextConfiguration;
22  
23  /**
24   * Concrete implementation of {@link AbstractGenericContextLoader} that reads
25   * bean definitions from Groovy scripts <em>and</em> XML configuration files.
26   *
27   * <p>Default resource locations are detected using the suffixes
28   * {@code "-context.xml"} and {@code "Context.groovy"}.
29   *
30   * @author Sam Brannen
31   * @since 4.1
32   * @see GroovyBeanDefinitionReader
33   * @see GenericXmlContextLoader
34   * @see AnnotationConfigContextLoader
35   */
36  public class GenericGroovyXmlContextLoader extends GenericXmlContextLoader {
37  
38  	/**
39  	 * Load bean definitions into the supplied {@link GenericApplicationContext context}
40  	 * from the locations in the supplied {@code MergedContextConfiguration} using a
41  	 * {@link GroovyBeanDefinitionReader}.
42  	 *
43  	 * @param context the context into which the bean definitions should be loaded
44  	 * @param mergedConfig the merged context configuration
45  	 * @see org.springframework.test.context.support.AbstractGenericContextLoader#loadBeanDefinitions
46  	 */
47  	@Override
48  	protected void loadBeanDefinitions(GenericApplicationContext context, MergedContextConfiguration mergedConfig) {
49  		new GroovyBeanDefinitionReader(context).loadBeanDefinitions(mergedConfig.getLocations());
50  	}
51  
52  	/**
53  	 * Returns {@code "-context.xml" and "Context.groovy"} in order to
54  	 * support detection of a default XML config file or Groovy script.
55  	 */
56  	@Override
57  	protected String[] getResourceSuffixes() {
58  		return new String[] { super.getResourceSuffix(), "Context.groovy" };
59  	}
60  
61  	/**
62  	 * {@code GenericGroovyXmlContextLoader} supports both Groovy and XML
63  	 * resource types for detection of defaults. Consequently, this method
64  	 * is not supported.
65  	 * @see #getResourceSuffixes()
66  	 * @throws UnsupportedOperationException
67  	 */
68  	@Override
69  	protected String getResourceSuffix() {
70  		throw new UnsupportedOperationException(
71  			"GenericGroovyXmlContextLoader does not support the getResourceSuffix() method");
72  	}
73  
74  }
