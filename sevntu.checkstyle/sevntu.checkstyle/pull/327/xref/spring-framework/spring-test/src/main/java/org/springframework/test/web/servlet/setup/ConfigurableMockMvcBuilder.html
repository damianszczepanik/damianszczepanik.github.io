17  package org.springframework.test.web.servlet.setup;
18  
19  import javax.servlet.Filter;
20  
21  import org.springframework.test.web.servlet.MockMvcBuilder;
22  import org.springframework.test.web.servlet.RequestBuilder;
23  import org.springframework.test.web.servlet.ResultHandler;
24  import org.springframework.test.web.servlet.ResultMatcher;
25  
26  /**
27   * Defines common methods for building a {@code MockMvc}.
28   *
29   * @author Rossen Stoyanchev
30   * @since 4.1
31   */
32  public interface ConfigurableMockMvcBuilder<B extends ConfigurableMockMvcBuilder<B>> extends MockMvcBuilder {
33  
34  	/**
35  	 * Add filters mapped to any request (i.e. "/*"). For example:
36  	 *
37  	 * <pre class="code">
38  	 * mockMvcBuilder.addFilters(springSecurityFilterChain);
39  	 * </pre>
40  	 *
41  	 * <p>is the equivalent of the following web.xml configuration:
42  	 *
43  	 * <pre class="code">
44  	 * &lt;filter-mapping&gt;
45  	 *     &lt;filter-name&gt;springSecurityFilterChain&lt;/filter-name&gt;
46  	 *     &lt;url-pattern&gt;/*&lt;/url-pattern&gt;
47  	 * &lt;/filter-mapping&gt;
48  	 * </pre>
49  	 *
50  	 * <p>Filters will be invoked in the order in which they are provided.
51  	 *
52  	 * @param filters the filters to add
53  	 */
54  	<T extends B> T addFilters(Filter... filters);
55  
56  	/**
57  	 * Add a filter mapped to a specific set of patterns. For example:
58  	 *
59  	 * <pre class="code">
60  	 * mockMvcBuilder.addFilters(myResourceFilter, "/resources/*");
61  	 * </pre>
62  	 *
63  	 * <p>is the equivalent of:
64  	 *
65  	 * <pre class="code">
66  	 * &lt;filter-mapping&gt;
67  	 *     &lt;filter-name&gt;myResourceFilter&lt;/filter-name&gt;
68  	 *     &lt;url-pattern&gt;/resources/*&lt;/url-pattern&gt;
69  	 * &lt;/filter-mapping&gt;
70  	 * </pre>
71  	 *
72  	 * <p>Filters will be invoked in the order in which they are provided.
73  	 *
74  	 * @param filter the filter to add
75  	 * @param urlPatterns URL patterns to map to; if empty, "/*" is used by default
76  	 */
77  	<T extends B> T addFilter(Filter filter, String... urlPatterns);
78  
79  	/**
80  	 * Define default request properties that should be merged into all
81  	 * performed requests. In effect this provides a mechanism for defining
82  	 * common initialization for all requests such as the content type, request
83  	 * parameters, session attributes, and any other request property.
84  	 *
85  	 * <p>Properties specified at the time of performing a request override the
86  	 * default properties defined here.
87  	 *
88  	 * @param requestBuilder a RequestBuilder; see static factory methods in
89  	 * {@link org.springframework.test.web.servlet.request.MockMvcRequestBuilders}
90  	 * .
91  	 */
92  	<T extends B> T defaultRequest(RequestBuilder requestBuilder);
93  
94  	/**
95  	 * Define a global expectation that should <em>always</em> be applied to
96  	 * every response. For example, status code 200 (OK), content type
97  	 * {@code "application/json"}, etc.
98  	 *
99  	 * @param resultMatcher a ResultMatcher; see static factory methods in
100 	 * {@link org.springframework.test.web.servlet.result.MockMvcResultMatchers}
101 	 */
102 	<T extends B> T alwaysExpect(ResultMatcher resultMatcher);
103 
104 	/**
105 	 * Define a global action that should <em>always</em> be applied to every
106 	 * response. For example, writing detailed information about the performed
107 	 * request and resulting response to {@code System.out}.
108 	 *
109 	 * @param resultHandler a ResultHandler; see static factory methods in
110 	 * {@link org.springframework.test.web.servlet.result.MockMvcResultHandlers}
111 	 */
112 	<T extends B> T alwaysDo(ResultHandler resultHandler);
113 
114 	/**
115 	 * Whether to enable the DispatcherServlet property
116 	 * {@link org.springframework.web.servlet.DispatcherServlet#setDispatchOptionsRequest
117 	 * dispatchOptionsRequest} which allows processing of HTTP OPTIONS requests.
118 	 */
119 	<T extends B> T dispatchOptions(boolean dispatchOptions);
120 
121 	/**
122 	 * Add a {@code MockMvcConfigurer} that automates MockMvc setup and
123 	 * configures it for some specific purpose (e.g. security).
124 	 */
125 	<T extends B> T apply(MockMvcConfigurer configurer);
126 
127 }
