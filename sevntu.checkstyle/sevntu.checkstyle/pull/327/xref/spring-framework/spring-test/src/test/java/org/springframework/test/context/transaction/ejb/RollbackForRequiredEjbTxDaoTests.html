17  package org.springframework.test.context.transaction.ejb;
18  
19  import org.junit.FixMethodOrder;
20  import org.junit.Test;
21  import org.junit.runners.MethodSorters;
22  
23  import org.springframework.test.context.transaction.TransactionConfiguration;
24  import org.springframework.test.context.transaction.TransactionalTestExecutionListener;
25  
26  import static org.junit.Assert.*;
27  
28  /**
29   * Extension of {@link CommitForRequiredEjbTxDaoTests} which sets the default
30   * rollback semantics for the {@link TransactionalTestExecutionListener} to
31   * {@code true}. The transaction managed by the TestContext framework will be
32   * rolled back after each test method. Consequently, any work performed in
33   * transactional methods that participate in the test-managed transaction will
34   * be rolled back automatically.
35   *
36   * @author Sam Brannen
37   * @since 4.0.1
38   */
39  @FixMethodOrder(MethodSorters.NAME_ASCENDING)
40  @TransactionConfiguration(defaultRollback = true)
41  public class RollbackForRequiredEjbTxDaoTests extends CommitForRequiredEjbTxDaoTests {
42  
43  	/**
44  	 * Redeclared to ensure test method execution order. Simply delegates to super.
45  	 */
46  	@Test
47  	@Override
48  	public void test1InitialState() {
49  		super.test1InitialState();
50  	}
51  
52  	/**
53  	 * Redeclared to ensure test method execution order. Simply delegates to super.
54  	 */
55  	@Test
56  	@Override
57  	public void test2IncrementCount1() {
58  		super.test2IncrementCount1();
59  	}
60  
61  	/**
62  	 * Overrides parent implementation in order to change expectations to align with
63  	 * behavior associated with "required" transactions on repositories/DAOs and
64  	 * default rollback semantics for transactions managed by the TestContext
65  	 * framework.
66  	 */
67  	@Test
68  	@Override
69  	public void test3IncrementCount2() {
70  		int count = dao.getCount(TEST_NAME);
71  		// Expecting count=0 after test2IncrementCount1() since REQUIRED transactions
72  		// participate in the existing transaction (if present), which in this case is the
73  		// transaction managed by the TestContext framework which will be rolled back
74  		// after each test method.
75  		assertEquals("Expected count=0 after test2IncrementCount1().", 0, count);
76  
77  		count = dao.incrementCount(TEST_NAME);
78  		assertEquals("Expected count=1 now.", 1, count);
79  	}
80  
81  }
