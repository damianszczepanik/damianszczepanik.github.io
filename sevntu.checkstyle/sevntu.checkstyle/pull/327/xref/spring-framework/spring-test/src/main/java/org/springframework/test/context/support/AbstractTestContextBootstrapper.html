58  /**
59   * Abstract implementation of the {@link TestContextBootstrapper} interface which
60   * provides most of the behavior required by a bootstrapper.
61   *
62   * <p>Concrete subclasses typically will only need to provide implementations for
63   * the following methods:
64   * <ul>
65   * <li>{@link #getDefaultContextLoaderClass}
66   * <li>{@link #processMergedContextConfiguration}
67   * </ul>
68   *
69   * @author Sam Brannen
70   * @author Juergen Hoeller
71   * @since 4.1
72   */
73  public abstract class AbstractTestContextBootstrapper implements TestContextBootstrapper {

97  	 * {@inheritDoc}
98  	 */
99  	@SuppressWarnings("unchecked")
100 	@Override
101 	public final List<TestExecutionListener> getTestExecutionListeners() {
102 		Class<?> clazz = getBootstrapContext().getTestClass();
103 		Class<TestExecutionListeners> annotationType = TestExecutionListeners.class;
104 		List<Class<? extends TestExecutionListener>> classesList = new ArrayList<Class<? extends TestExecutionListener>>();
105 		boolean usingDefaults = false;
106 
107 		AnnotationDescriptor<TestExecutionListeners> descriptor = MetaAnnotationUtils.findAnnotationDescriptor(clazz,
108 			annotationType);
109 
110 		// Use defaults?
111 		if (descriptor == null) {
112 			if (logger.isDebugEnabled()) {
113 				logger.debug(String.format("@TestExecutionListeners is not present for class [%s]: using defaults.",
114 					clazz.getName()));
115 			}
116 			usingDefaults = true;
117 			classesList.addAll(getDefaultTestExecutionListenerClasses());
118 		}
119 		else {
120 			// Traverse the class hierarchy...
121 			while (descriptor != null) {
122 				Class<?> declaringClass = descriptor.getDeclaringClass();
123 				AnnotationAttributes annAttrs = descriptor.getAnnotationAttributes();
124 				if (logger.isTraceEnabled()) {
125 					logger.trace(String.format(
126 						"Retrieved @TestExecutionListeners attributes [%s] for declaring class [%s].", annAttrs,
127 						declaringClass.getName()));
128 				}
129 
130 				Class<? extends TestExecutionListener>[] valueListenerClasses = (Class<? extends TestExecutionListener>[]) annAttrs.getClassArray("value");
131 				Class<? extends TestExecutionListener>[] listenerClasses = (Class<? extends TestExecutionListener>[]) annAttrs.getClassArray("listeners");
132 				if (!ObjectUtils.isEmpty(valueListenerClasses) && !ObjectUtils.isEmpty(listenerClasses)) {
133 					throw new IllegalStateException(String.format(
134 						"Class [%s] configured with @TestExecutionListeners' "
135 								+ "'value' [%s] and 'listeners' [%s] attributes. Use one or the other, but not both.",
136 						declaringClass.getName(), ObjectUtils.nullSafeToString(valueListenerClasses),
137 						ObjectUtils.nullSafeToString(listenerClasses)));
138 				}
139 				else if (!ObjectUtils.isEmpty(valueListenerClasses)) {
140 					listenerClasses = valueListenerClasses;
141 				}
142 
143 				boolean inheritListeners = annAttrs.getBoolean("inheritListeners");
144 				AnnotationDescriptor<TestExecutionListeners> superDescriptor = MetaAnnotationUtils.findAnnotationDescriptor(
145 					descriptor.getRootDeclaringClass().getSuperclass(), annotationType);
146 
147 				// If there are no listeners to inherit, we might need to merge the
148 				// locally declared listeners with the defaults.
149 				if ((!inheritListeners || superDescriptor == null)
150 						&& (annAttrs.getEnum("mergeMode") == MergeMode.MERGE_WITH_DEFAULTS)) {
151 					if (logger.isDebugEnabled()) {
152 						logger.debug(String.format(
153 							"Merging default listeners with listeners configured via @TestExecutionListeners for class [%s].",
154 							descriptor.getRootDeclaringClass().getName()));
155 					}
156 					usingDefaults = true;
157 					classesList.addAll(getDefaultTestExecutionListenerClasses());
158 				}
159 
160 				classesList.addAll(0, Arrays.<Class<? extends TestExecutionListener>> asList(listenerClasses));
161 
162 				descriptor = (inheritListeners ? superDescriptor : null);
163 			}
164 		}
165 
166 		// Remove possible duplicates if we loaded default listeners.
167 		if (usingDefaults) {
168 			Set<Class<? extends TestExecutionListener>> classesSet = new HashSet<Class<? extends TestExecutionListener>>();
169 			classesSet.addAll(classesList);
170 			classesList.clear();
171 			classesList.addAll(classesSet);
172 		}
173 
174 		List<TestExecutionListener> listeners = instantiateListeners(classesList);
175 
176 		// Sort by Ordered/@Order if we loaded default listeners.
177 		if (usingDefaults) {
178 			AnnotationAwareOrderComparator.sort(listeners);
179 		}
180 
181 		if (logger.isInfoEnabled()) {
182 			logger.info(String.format("Using TestExecutionListeners: %s", listeners));
183 		}
184 		return listeners;
185 	}
186 
187 	private List<TestExecutionListener> instantiateListeners(List<Class<? extends TestExecutionListener>> classesList) {
188 		List<TestExecutionListener> listeners = new ArrayList<TestExecutionListener>(classesList.size());
189 		for (Class<? extends TestExecutionListener> listenerClass : classesList) {
190 			NoClassDefFoundError ncdfe = null;
191 			try {
192 				listeners.add(BeanUtils.instantiateClass(listenerClass));
193 			}
194 			catch (NoClassDefFoundError err) {
195 				ncdfe = err;
196 			}
197 			catch (BeanInstantiationException ex) {
198 				if (ex.getCause() instanceof NoClassDefFoundError) {
199 					ncdfe = (NoClassDefFoundError) ex.getCause();
200 				}
201 			}
202 			if (ncdfe != null) {
203 				if (logger.isInfoEnabled()) {
204 					logger.info(String.format("Could not instantiate TestExecutionListener [%s]. "
205 							+ "Specify custom listener classes or make the default listener classes "
206 							+ "(and their required dependencies) available. Offending class: [%s]",
207 						listenerClass.getName(), ncdfe.getMessage()));
208 				}
209 			}
210 		}
211 		return listeners;
212 	}

214 	/**
215 	 * Get the default {@link TestExecutionListener} classes for this bootstrapper.
216 	 * <p>This method is invoked by {@link #getTestExecutionListeners()} and
217 	 * delegates to {@link #getDefaultTestExecutionListenerClassNames()} to
218 	 * retrieve the class names.
219 	 * <p>If a particular class cannot be loaded, a {@code DEBUG} message will
220 	 * be logged, but the associated exception will not be rethrown.
221 	 */
222 	@SuppressWarnings("unchecked")
223 	protected Set<Class<? extends TestExecutionListener>> getDefaultTestExecutionListenerClasses() {
224 		Set<Class<? extends TestExecutionListener>> defaultListenerClasses = new LinkedHashSet<Class<? extends TestExecutionListener>>();
225 		ClassLoader cl = getClass().getClassLoader();
226 		for (String className : getDefaultTestExecutionListenerClassNames()) {
227 			try {
228 				defaultListenerClasses.add((Class<? extends TestExecutionListener>) ClassUtils.forName(className, cl));
229 			}
230 			catch (Throwable ex) {
231 				if (logger.isDebugEnabled()) {
232 					logger.debug("Could not load default TestExecutionListener class [" + className
233 							+ "]. Specify custom listener classes or make the default listener classes available.", ex);
234 				}
235 			}
236 		}
237 		return defaultListenerClasses;
238 	}

240 	/**
241 	 * Get the names of the default {@link TestExecutionListener} classes for
242 	 * this bootstrapper.
243 	 * <p>The default implementation looks up all
244 	 * {@code org.springframework.test.context.TestExecutionListener} entries
245 	 * configured in all {@code META-INF/spring.factories} files on the classpath.
246 	 * <p>This method is invoked by {@link #getDefaultTestExecutionListenerClasses()}.
247 	 * @return an <em>unmodifiable</em> list of names of default {@code TestExecutionListener}
248 	 * classes
249 	 * @see SpringFactoriesLoader#loadFactoryNames
250 	 */
251 	protected List<String> getDefaultTestExecutionListenerClassNames() {
252 		final List<String> classNames = SpringFactoriesLoader.loadFactoryNames(TestExecutionListener.class,
253 			getClass().getClassLoader());
254 
255 		if (logger.isInfoEnabled()) {
256 			logger.info(String.format("Loaded default TestExecutionListener class names from location [%s]: %s",
257 				SpringFactoriesLoader.FACTORIES_RESOURCE_LOCATION, classNames));
258 		}
259 		return Collections.unmodifiableList(classNames);
260 	}

262 	/**
263 	 * {@inheritDoc}
264 	 */
265 	@SuppressWarnings("unchecked")
266 	@Override
267 	public final MergedContextConfiguration buildMergedContextConfiguration() {
268 		Class<?> testClass = getBootstrapContext().getTestClass();
269 		CacheAwareContextLoaderDelegate cacheAwareContextLoaderDelegate = getBootstrapContext().getCacheAwareContextLoaderDelegate();
270 
271 		if (MetaAnnotationUtils.findAnnotationDescriptorForTypes(testClass, ContextConfiguration.class,
272 			ContextHierarchy.class) == null) {
273 			if (logger.isInfoEnabled()) {
274 				logger.info(String.format(
275 					"Neither @ContextConfiguration nor @ContextHierarchy found for test class [%s]",
276 					testClass.getName()));
277 			}
278 			return new MergedContextConfiguration(testClass, null, null, null, null);
279 		}
280 
281 		if (AnnotationUtils.findAnnotation(testClass, ContextHierarchy.class) != null) {
282 			Map<String, List<ContextConfigurationAttributes>> hierarchyMap = ContextLoaderUtils.buildContextHierarchyMap(testClass);
283 			MergedContextConfiguration parentConfig = null;
284 			MergedContextConfiguration mergedConfig = null;
285 
286 			for (List<ContextConfigurationAttributes> list : hierarchyMap.values()) {
287 				List<ContextConfigurationAttributes> reversedList = new ArrayList<ContextConfigurationAttributes>(list);
288 				Collections.reverse(reversedList);
289 
290 				// Don't use the supplied testClass; instead ensure that we are
291 				// building the MCC for the actual test class that declared the
292 				// configuration for the current level in the context hierarchy.
293 				Assert.notEmpty(reversedList, "ContextConfigurationAttributes list must not be empty");
294 				Class<?> declaringClass = reversedList.get(0).getDeclaringClass();
295 
296 				mergedConfig = buildMergedContextConfiguration(declaringClass, reversedList, parentConfig,
297 					cacheAwareContextLoaderDelegate);
298 				parentConfig = mergedConfig;
299 			}
300 
301 			// Return the last level in the context hierarchy
302 			return mergedConfig;
303 		}
304 		else {
305 			return buildMergedContextConfiguration(testClass,
306 				ContextLoaderUtils.resolveContextConfigurationAttributes(testClass), null,
307 				cacheAwareContextLoaderDelegate);
308 		}
309 	}

311 	/**
312 	 * Build the {@link MergedContextConfiguration merged context configuration}
313 	 * for the supplied {@link Class testClass}, context configuration attributes,
314 	 * and parent context configuration.
315 	 * @param testClass the test class for which the {@code MergedContextConfiguration}
316 	 * should be built (must not be {@code null})
317 	 * @param configAttributesList the list of context configuration attributes for the
318 	 * specified test class, ordered <em>bottom-up</em> (i.e., as if we were
319 	 * traversing up the class hierarchy); never {@code null} or empty
320 	 * @param parentConfig the merged context configuration for the parent application
321 	 * context in a context hierarchy, or {@code null} if there is no parent
322 	 * @param cacheAwareContextLoaderDelegate the cache-aware context loader delegate to
323 	 * be passed to the {@code MergedContextConfiguration} constructor
324 	 * @return the merged context configuration
325 	 * @see #resolveContextLoader
326 	 * @see ContextLoaderUtils#resolveContextConfigurationAttributes
327 	 * @see SmartContextLoader#processContextConfiguration
328 	 * @see ContextLoader#processLocations
329 	 * @see ActiveProfilesUtils#resolveActiveProfiles
330 	 * @see ApplicationContextInitializerUtils#resolveInitializerClasses
331 	 * @see MergedContextConfiguration
332 	 */
333 	private MergedContextConfiguration buildMergedContextConfiguration(Class<?> testClass,
334 			List<ContextConfigurationAttributes> configAttributesList, MergedContextConfiguration parentConfig,
335 			CacheAwareContextLoaderDelegate cacheAwareContextLoaderDelegate) {
336 
337 		ContextLoader contextLoader = resolveContextLoader(testClass, configAttributesList);
338 		List<String> locationsList = new ArrayList<String>();
339 		List<Class<?>> classesList = new ArrayList<Class<?>>();
340 
341 		for (ContextConfigurationAttributes configAttributes : configAttributesList) {
342 			if (logger.isTraceEnabled()) {
343 				logger.trace(String.format("Processing locations and classes for context configuration attributes %s",
344 					configAttributes));
345 			}
346 			if (contextLoader instanceof SmartContextLoader) {
347 				SmartContextLoader smartContextLoader = (SmartContextLoader) contextLoader;
348 				smartContextLoader.processContextConfiguration(configAttributes);
349 				locationsList.addAll(0, Arrays.asList(configAttributes.getLocations()));
350 				classesList.addAll(0, Arrays.asList(configAttributes.getClasses()));
351 			}
352 			else {
353 				String[] processedLocations = contextLoader.processLocations(configAttributes.getDeclaringClass(),
354 					configAttributes.getLocations());
355 				locationsList.addAll(0, Arrays.asList(processedLocations));
356 				// Legacy ContextLoaders don't know how to process classes
357 			}
358 			if (!configAttributes.isInheritLocations()) {
359 				break;
360 			}
361 		}
362 
363 		String[] locations = StringUtils.toStringArray(locationsList);
364 		Class<?>[] classes = ClassUtils.toClassArray(classesList);
365 		Set<Class<? extends ApplicationContextInitializer<? extends ConfigurableApplicationContext>>> initializerClasses = //
366 		ApplicationContextInitializerUtils.resolveInitializerClasses(configAttributesList);
367 		String[] activeProfiles = ActiveProfilesUtils.resolveActiveProfiles(testClass);
368 		MergedTestPropertySources mergedTestPropertySources = TestPropertySourceUtils.buildMergedTestPropertySources(testClass);
369 
370 		MergedContextConfiguration mergedConfig = new MergedContextConfiguration(testClass, locations, classes,
371 			initializerClasses, activeProfiles, mergedTestPropertySources.getLocations(),
372 			mergedTestPropertySources.getProperties(), contextLoader, cacheAwareContextLoaderDelegate, parentConfig);
373 
374 		return processMergedContextConfiguration(mergedConfig);
375 	}

377 	/**
378 	 * Resolve the {@link ContextLoader} {@linkplain Class class} to use for the
379 	 * supplied list of {@link ContextConfigurationAttributes} and then instantiate
380 	 * and return that {@code ContextLoader}.
381 	 * <p>If the user has not explicitly declared which loader to use, the value
382 	 * returned from {@link #getDefaultContextLoaderClass} will be used as the
383 	 * default context loader class. For details on the class resolution process,
384 	 * see {@link #resolveExplicitContextLoaderClass} and
385 	 * {@link #getDefaultContextLoaderClass}.
386 	 * @param testClass the test class for which the {@code ContextLoader} should be
387 	 * resolved; must not be {@code null}
388 	 * @param configAttributesList the list of configuration attributes to process; must
389 	 * not be {@code null} or <em>empty</em>; must be ordered <em>bottom-up</em>
390 	 * (i.e., as if we were traversing up the class hierarchy)
391 	 * @return the resolved {@code ContextLoader} for the supplied {@code testClass}
392 	 * (never {@code null})
393 	 */
394 	private ContextLoader resolveContextLoader(Class<?> testClass,
395 			List<ContextConfigurationAttributes> configAttributesList) {
396 
397 		Assert.notNull(testClass, "Class must not be null");
398 		Assert.notEmpty(configAttributesList, "ContextConfigurationAttributes list must not be empty");
399 
400 		Class<? extends ContextLoader> contextLoaderClass = resolveExplicitContextLoaderClass(configAttributesList);
401 		if (contextLoaderClass == null) {
402 			contextLoaderClass = getDefaultContextLoaderClass(testClass);
403 		}
404 		if (logger.isTraceEnabled()) {
405 			logger.trace(String.format("Using ContextLoader class [%s] for test class [%s]",
406 				contextLoaderClass.getName(), testClass.getName()));
407 		}
408 		return BeanUtils.instantiateClass(contextLoaderClass, ContextLoader.class);
409 	}

411 	/**
412 	 * Resolve the {@link ContextLoader} {@linkplain Class class} to use for the supplied
413 	 * list of {@link ContextConfigurationAttributes}.
414 	 * <p>Beginning with the first level in the context configuration attributes hierarchy:
415 	 * <ol>
416 	 * <li>If the {@link ContextConfigurationAttributes#getContextLoaderClass()
417 	 * contextLoaderClass} property of {@link ContextConfigurationAttributes} is
418 	 * configured with an explicit class, that class will be returned.</li>
419 	 * <li>If an explicit {@code ContextLoader} class is not specified at the current
420 	 * level in the hierarchy, traverse to the next level in the hierarchy and return to
421 	 * step #1.</li>
422 	 * </ol>
423 	 * @param configAttributesList the list of configuration attributes to process;
424 	 * must not be {@code null} or <em>empty</em>; must be ordered <em>bottom-up</em>
425 	 * (i.e., as if we were traversing up the class hierarchy)
426 	 * @return the {@code ContextLoader} class to use for the supplied configuration
427 	 * attributes, or {@code null} if no explicit loader is found
428 	 * @throws IllegalArgumentException if supplied configuration attributes are
429 	 * {@code null} or <em>empty</em>
430 	 */
431 	private Class<? extends ContextLoader> resolveExplicitContextLoaderClass(
432 			List<ContextConfigurationAttributes> configAttributesList) {
433 
434 		Assert.notEmpty(configAttributesList, "ContextConfigurationAttributes list must not be empty");
435 		for (ContextConfigurationAttributes configAttributes : configAttributesList) {
436 			if (logger.isTraceEnabled()) {
437 				logger.trace(String.format("Resolving ContextLoader for context configuration attributes %s",
438 					configAttributes));
439 			}
440 			Class<? extends ContextLoader> contextLoaderClass = configAttributes.getContextLoaderClass();
441 			if (!ContextLoader.class.equals(contextLoaderClass)) {
442 				if (logger.isDebugEnabled()) {
443 					logger.debug(String.format(
444 						"Found explicit ContextLoader class [%s] for context configuration attributes %s",
445 						contextLoaderClass.getName(), configAttributes));
446 				}
447 				return contextLoaderClass;
448 			}
449 		}
450 		return null;
451 	}
