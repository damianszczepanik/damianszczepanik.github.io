16  package org.springframework.test.web.client.samples.matchers;
17  
18  import java.net.URI;
19  import java.util.ArrayList;
20  import java.util.Arrays;
21  import java.util.List;
22  
23  import org.junit.Before;
24  import org.junit.Test;
25  
26  import org.springframework.http.converter.HttpMessageConverter;
27  import org.springframework.http.converter.json.MappingJackson2HttpMessageConverter;
28  import org.springframework.test.web.Person;
29  import org.springframework.test.web.client.MockRestServiceServer;
30  import org.springframework.util.LinkedMultiValueMap;
31  import org.springframework.util.MultiValueMap;
32  import org.springframework.web.client.RestTemplate;
33  
34  import static org.hamcrest.Matchers.*;
35  import static org.springframework.test.web.client.match.MockRestRequestMatchers.*;
36  import static org.springframework.test.web.client.response.MockRestResponseCreators.*;

38  /**
39   * Examples of defining expectations on JSON request content with
40   * <a href="http://goessner.net/articles/JsonPath/">JSONPath</a> expressions.
41   *
42   * @author Rossen Stoyanchev
43   */
44  public class JsonPathRequestMatcherTests {
45  
46  	private MockRestServiceServer mockServer;
47  
48  	private RestTemplate restTemplate;
49  
50  	private MultiValueMap<String, Person> people;
51  
52  
53  	@Before
54  	public void setup() {
55  		this.people = new LinkedMultiValueMap<String, Person>();
56  		this.people.add("composers", new Person("Johann Sebastian Bach"));
57  		this.people.add("composers", new Person("Johannes Brahms"));
58  		this.people.add("composers", new Person("Edvard Grieg"));
59  		this.people.add("composers", new Person("Robert Schumann"));
60  		this.people.add("performers", new Person("Vladimir Ashkenazy"));
61  		this.people.add("performers", new Person("Yehudi Menuhin"));
62  
63  		List<HttpMessageConverter<?>> converters = new ArrayList<HttpMessageConverter<?>>();
64  		converters.add(new MappingJackson2HttpMessageConverter());
65  
66  		this.restTemplate = new RestTemplate();
67  		this.restTemplate.setMessageConverters(converters);
68  
69  		this.mockServer = MockRestServiceServer.createServer(this.restTemplate);
70  	}
71  
72  	@Test
73  	public void testExists() throws Exception {
74  		this.mockServer.expect(requestTo("/composers"))
75  			.andExpect(content().contentType("application/json;charset=UTF-8"))
76  			.andExpect(jsonPath("$.composers[0]").exists())
77  			.andExpect(jsonPath("$.composers[1]").exists())
78  			.andExpect(jsonPath("$.composers[2]").exists())
79  			.andExpect(jsonPath("$.composers[3]").exists())
80  			.andRespond(withSuccess());
81  
82  		this.restTemplate.put(new URI("/composers"), this.people);
83  		this.mockServer.verify();
84  	}
85  
86  	@Test
87  	public void testDoesNotExist() throws Exception {
88  		this.mockServer.expect(requestTo("/composers"))
89  			.andExpect(content().contentType("application/json;charset=UTF-8"))
90  			.andExpect(jsonPath("$.composers[?(@.name == 'Edvard Grieeeeeeg')]").doesNotExist())
91  			.andExpect(jsonPath("$.composers[?(@.name == 'Robert Schuuuuuuman')]").doesNotExist())
92  			.andExpect(jsonPath("$.composers[-1]").doesNotExist())
93  			.andExpect(jsonPath("$.composers[4]").doesNotExist())
94  			.andRespond(withSuccess());
95  
96  		this.restTemplate.put(new URI("/composers"), this.people);
97  		this.mockServer.verify();
98  	}
99  
100 	@Test
101 	public void testEqualTo() throws Exception {
102 		this.mockServer.expect(requestTo("/composers"))
103 			.andExpect(content().contentType("application/json;charset=UTF-8"))
104 			.andExpect(jsonPath("$.composers[0].name").value("Johann Sebastian Bach"))
105 			.andExpect(jsonPath("$.performers[1].name").value("Yehudi Menuhin"))
106 			.andExpect(jsonPath("$.composers[0].name").value(equalTo("Johann Sebastian Bach"))) // Hamcrest
107 			.andExpect(jsonPath("$.performers[1].name").value(equalTo("Yehudi Menuhin"))) // Hamcrest
108 			.andRespond(withSuccess());
109 
110 		this.restTemplate.put(new URI("/composers"), this.people);
111 		this.mockServer.verify();
112 	}
113 
114 	@Test
115 	public void testHamcrestMatcher() throws Exception {
116 		this.mockServer.expect(requestTo("/composers"))
117 			.andExpect(content().contentType("application/json;charset=UTF-8"))
118 			.andExpect(jsonPath("$.composers[0].name", startsWith("Johann")))
119 			.andExpect(jsonPath("$.performers[0].name", endsWith("Ashkenazy")))
120 			.andExpect(jsonPath("$.performers[1].name", containsString("di Me")))
121 			.andExpect(jsonPath("$.composers[1].name", isIn(Arrays.asList("Johann Sebastian Bach", "Johannes Brahms"))))
122 			.andExpect(jsonPath("$.composers[:3].name", hasItem("Johannes Brahms")))
123 			.andRespond(withSuccess());
124 
125 		this.restTemplate.put(new URI("/composers"), this.people);
126 		this.mockServer.verify();
127 	}
128 
129 	@Test
130 	public void testHamcrestMatcherWithParameterizedJsonPath() throws Exception {
131 		String composerName = "$.composers[%s].name";
132 		String performerName = "$.performers[%s].name";
133 
134 		this.mockServer.expect(requestTo("/composers"))
135 			.andExpect(content().contentType("application/json;charset=UTF-8"))
136 			.andExpect(jsonPath(composerName, 0).value(startsWith("Johann")))
137 			.andExpect(jsonPath(performerName, 0).value(endsWith("Ashkenazy")))
138 			.andExpect(jsonPath(performerName, 1).value(containsString("di Me")))
139 			.andExpect(jsonPath(composerName, 1).value(isIn(Arrays.asList("Johann Sebastian Bach", "Johannes Brahms"))))
140 			.andRespond(withSuccess());
141 
142 		this.restTemplate.put(new URI("/composers"), this.people);
143 		this.mockServer.verify();
144 	}
145 
146 }
