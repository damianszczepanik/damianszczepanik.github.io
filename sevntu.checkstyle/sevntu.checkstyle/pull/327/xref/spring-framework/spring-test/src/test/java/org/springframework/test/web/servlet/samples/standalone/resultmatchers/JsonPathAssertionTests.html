33  import static org.hamcrest.Matchers.*;
34  import static org.springframework.test.web.servlet.request.MockMvcRequestBuilders.*;

36  import static org.springframework.test.web.servlet.setup.MockMvcBuilders.*;

38  /**
39   * Examples of defining expectations on JSON response content with
40   * <a href="http://goessner.net/articles/JsonPath/">JSONPath</a> expressions.
41   *
42   * @author Rossen Stoyanchev
43   *
44   * @see ContentAssertionTests
45   */
46  public class JsonPathAssertionTests {
47  
48  	private MockMvc mockMvc;
49  
50  	@Before
51  	public void setup() {
52  		this.mockMvc = standaloneSetup(new MusicController())
53  				.defaultRequest(get("/").accept(MediaType.APPLICATION_JSON))
54  				.alwaysExpect(status().isOk())
55  				.alwaysExpect(content().contentType("application/json;charset=UTF-8"))
56  				.build();
57  	}
58  
59  	@Test
60  	public void testExists() throws Exception {
61  
62  		String composerByName = "$.composers[?(@.name == '%s')]";
63  		String performerByName = "$.performers[?(@.name == '%s')]";
64  
65  		this.mockMvc.perform(get("/music/people"))
66  			.andExpect(jsonPath(composerByName, "Johann Sebastian Bach").exists())
67  			.andExpect(jsonPath(composerByName, "Johannes Brahms").exists())
68  			.andExpect(jsonPath(composerByName, "Edvard Grieg").exists())
69  			.andExpect(jsonPath(composerByName, "Robert Schumann").exists())
70  			.andExpect(jsonPath(performerByName, "Vladimir Ashkenazy").exists())
71  			.andExpect(jsonPath(performerByName, "Yehudi Menuhin").exists())
72  			.andExpect(jsonPath("$.composers[0]").exists())
73  			.andExpect(jsonPath("$.composers[1]").exists())
74  			.andExpect(jsonPath("$.composers[2]").exists())
75  			.andExpect(jsonPath("$.composers[3]").exists());
76  
77  	}
78  
79  	@Test
80  	public void testDoesNotExist() throws Exception {
81  		this.mockMvc.perform(get("/music/people"))
82  			.andExpect(jsonPath("$.composers[?(@.name == 'Edvard Grieeeeeeg')]").doesNotExist())
83  			.andExpect(jsonPath("$.composers[?(@.name == 'Robert Schuuuuuuman')]").doesNotExist())
84  			.andExpect(jsonPath("$.composers[-1]").doesNotExist())
85  			.andExpect(jsonPath("$.composers[4]").doesNotExist());
86  	}
87  
88  	@Test
89  	public void testEqualTo() throws Exception {
90  		this.mockMvc.perform(get("/music/people"))
91  			.andExpect(jsonPath("$.composers[0].name").value("Johann Sebastian Bach"))
92  			.andExpect(jsonPath("$.performers[1].name").value("Yehudi Menuhin"));
93  
94  		// Hamcrest matchers...
95  		this.mockMvc.perform(get("/music/people"))
96  			.andExpect(jsonPath("$.composers[0].name").value(equalTo("Johann Sebastian Bach")))
97  			.andExpect(jsonPath("$.performers[1].name").value(equalTo("Yehudi Menuhin")));
98  	}
99  
100 	@Test
101 	public void testHamcrestMatcher() throws Exception {
102 		this.mockMvc.perform(get("/music/people"))
103 			.andExpect(jsonPath("$.composers[0].name", startsWith("Johann")))
104 			.andExpect(jsonPath("$.performers[0].name", endsWith("Ashkenazy")))
105 			.andExpect(jsonPath("$.performers[1].name", containsString("di Me")))
106 			.andExpect(jsonPath("$.composers[1].name", isIn(Arrays.asList("Johann Sebastian Bach", "Johannes Brahms"))));
107 	}
108 
109 	@Test
110 	public void testHamcrestMatcherWithParameterizedJsonPath() throws Exception {
111 
112 		String composerName = "$.composers[%s].name";
113 		String performerName = "$.performers[%s].name";
114 
115 		this.mockMvc.perform(get("/music/people"))
116 			.andExpect(jsonPath(composerName, 0).value(startsWith("Johann")))
117 			.andExpect(jsonPath(performerName, 0).value(endsWith("Ashkenazy")))
118 			.andExpect(jsonPath(performerName, 1).value(containsString("di Me")))
119 			.andExpect(jsonPath(composerName, 1).value(isIn(Arrays.asList("Johann Sebastian Bach", "Johannes Brahms"))));
120 	}
121 
122 
123 	@Controller
124 	private class MusicController {
125 
126 		@RequestMapping(value="/music/people")
127 		public @ResponseBody MultiValueMap<String, Person> get() {
128 			MultiValueMap<String, Person> map = new LinkedMultiValueMap<String, Person>();
129 
130 			map.add("composers", new Person("Johann Sebastian Bach"));
131 			map.add("composers", new Person("Johannes Brahms"));
132 			map.add("composers", new Person("Edvard Grieg"));
133 			map.add("composers", new Person("Robert Schumann"));
134 
135 			map.add("performers", new Person("Vladimir Ashkenazy"));
136 			map.add("performers", new Person("Yehudi Menuhin"));
137 
138 			return map;
139 		}
140 	}
