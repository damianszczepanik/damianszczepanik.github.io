17  package org.springframework.test.context;
18  
19  import org.springframework.context.ApplicationContext;
20  import org.springframework.context.ConfigurableApplicationContext;
21  import org.springframework.test.annotation.DirtiesContext.HierarchyMode;
22  
23  /**
24   * A {@code CacheAwareContextLoaderDelegate} is responsible for {@linkplain
25   * #loadContext loading} and {@linkplain #closeContext closing} application
26   * contexts, interacting transparently with a <em>context cache</em> behind
27   * the scenes.
28   *
29   * <p>Note: {@code CacheAwareContextLoaderDelegate} does not extend the
30   * {@link ContextLoader} or {@link SmartContextLoader} interface.
31   *
32   * @author Sam Brannen
33   * @since 3.2.2
34   */
35  public interface CacheAwareContextLoaderDelegate {
36  
37  	/**
38  	 * Load the {@linkplain ApplicationContext application context} for the supplied
39  	 * {@link MergedContextConfiguration} by delegating to the {@link ContextLoader}
40  	 * configured in the given {@code MergedContextConfiguration}.
41  	 * <p>If the context is present in the <em>context cache</em> it will simply
42  	 * be returned; otherwise, it will be loaded, stored in the cache, and returned.
43  	 * @param mergedContextConfiguration the merged context configuration to use
44  	 * to load the application context; never {@code null}
45  	 * @return the application context
46  	 * @throws IllegalStateException if an error occurs while retrieving or loading
47  	 * the application context
48  	 */
49  	ApplicationContext loadContext(MergedContextConfiguration mergedContextConfiguration);
50  
51  	/**
52  	 * Remove the {@linkplain ApplicationContext application context} for the
53  	 * supplied {@link MergedContextConfiguration} from the <em>context cache</em>
54  	 * and {@linkplain ConfigurableApplicationContext#close() close} it if it is
55  	 * an instance of {@link ConfigurableApplicationContext}.
56  	 * <p>The semantics of the supplied {@code HierarchyMode} must be honored when
57  	 * removing the context from the cache. See the Javadoc for {@link HierarchyMode}
58  	 * for details.
59  	 * <p>Generally speaking, this method should only be called if the state of
60  	 * a singleton bean has been changed (potentially affecting future interaction
61  	 * with the context) or if the context needs to be prematurely removed from
62  	 * the cache.
63  	 * @param mergedContextConfiguration the merged context configuration for the
64  	 * application context to close; never {@code null}
65  	 * @param hierarchyMode the hierarchy mode; may be {@code null} if the context
66  	 * is not part of a hierarchy
67  	 * @since 4.1
68  	 */
69  	void closeContext(MergedContextConfiguration mergedContextConfiguration, HierarchyMode hierarchyMode);
70  
71  }
