40  /**
41   * <p>
42   * JUnit 4 based integration test which verifies support of Spring's
43   * {@link Transactional &#64;Transactional}, {@link TestExecutionListeners
44   * &#64;TestExecutionListeners}, and {@link ContextConfiguration
45   * &#64;ContextConfiguration} annotations in conjunction with the
46   * {@link SpringJUnit4ClassRunner} and the following
47   * {@link TestExecutionListener TestExecutionListeners}:
48   * </p>
49   * <ul>
50   * <li>{@link DependencyInjectionTestExecutionListener}</li>
51   * <li>{@link DirtiesContextTestExecutionListener}</li>
52   * <li>{@link TransactionalTestExecutionListener}</li>
53   * </ul>
54   * <p>
55   * This class specifically tests usage of {@code @Transactional} defined at the
56   * <strong>class level</strong>.
57   * </p>
58   *
59   * @author Sam Brannen
60   * @since 2.5
61   * @see MethodLevelTransactionalSpringRunnerTests
62   */
63  @RunWith(SpringJUnit4ClassRunner.class)
64  @ContextConfiguration
65  @Transactional
66  public class ClassLevelTransactionalSpringRunnerTests extends AbstractTransactionalSpringRunnerTests {
67  
68  	protected static JdbcTemplate jdbcTemplate;
69  
70  
71  	@AfterClass
72  	public static void verifyFinalTestData() {
73  		assertEquals("Verifying the final number of rows in the person table after all tests.", 4,
74  			countRowsInPersonTable(jdbcTemplate));
75  	}
76  
77  	@Before
78  	public void verifyInitialTestData() {
79  		clearPersonTable(jdbcTemplate);
80  		assertEquals("Adding bob", 1, addPerson(jdbcTemplate, BOB));
81  		assertEquals("Verifying the initial number of rows in the person table.", 1,
82  			countRowsInPersonTable(jdbcTemplate));
83  	}
84  
85  	@Test
86  	public void modifyTestDataWithinTransaction() {
87  		assertInTransaction(true);
88  		assertEquals("Deleting bob", 1, deletePerson(jdbcTemplate, BOB));
89  		assertEquals("Adding jane", 1, addPerson(jdbcTemplate, JANE));
90  		assertEquals("Adding sue", 1, addPerson(jdbcTemplate, SUE));
91  		assertEquals("Verifying the number of rows in the person table within a transaction.", 2,
92  			countRowsInPersonTable(jdbcTemplate));
93  	}
94  
95  	@Test
96  	@Transactional(propagation = Propagation.NOT_SUPPORTED)
97  	public void modifyTestDataWithoutTransaction() {
98  		assertInTransaction(false);
99  		assertEquals("Adding luke", 1, addPerson(jdbcTemplate, LUKE));
100 		assertEquals("Adding leia", 1, addPerson(jdbcTemplate, LEIA));
101 		assertEquals("Adding yoda", 1, addPerson(jdbcTemplate, YODA));
102 		assertEquals("Verifying the number of rows in the person table without a transaction.", 4,
103 			countRowsInPersonTable(jdbcTemplate));
104 	}
105 
106 
107 	public static class DatabaseSetup {
108 
109 		@Resource
110 		public void setDataSource(DataSource dataSource) {
111 			jdbcTemplate = new JdbcTemplate(dataSource);
112 			createPersonTable(jdbcTemplate);
113 		}
114 	}
