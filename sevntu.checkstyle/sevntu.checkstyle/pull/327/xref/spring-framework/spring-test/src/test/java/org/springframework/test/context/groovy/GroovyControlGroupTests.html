17  package org.springframework.test.context.groovy;
18  
19  import org.junit.Test;
20  
21  import org.springframework.context.ApplicationContext;
22  import org.springframework.context.support.GenericGroovyApplicationContext;
23  import org.springframework.tests.sample.beans.Employee;
24  import org.springframework.tests.sample.beans.Pet;
25  
26  import static org.junit.Assert.*;
27  
28  /**
29   * Simple integration test to verify the expected functionality of
30   * {@link GenericGroovyApplicationContext}, thereby validating the proper
31   * syntax and configuration of {@code "context.groovy"} without using the
32   * Spring TestContext Framework.
33   *
34   * <p>In other words, this test class serves merely as a <em>control group</em>
35   * to ensure that there is nothing wrong with the Groovy script used by
36   * other tests in this package.
37   *
38   * @author Sam Brannen
39   * @since 4.1
40   */
41  public class GroovyControlGroupTests {
42  
43  	@Test
44  	@SuppressWarnings("resource")
45  	public void verifyScriptUsingGenericGroovyApplicationContext() {
46  		ApplicationContext ctx = new GenericGroovyApplicationContext(getClass(), "context.groovy");
47  
48  		String foo = ctx.getBean("foo", String.class);
49  		assertEquals("Foo", foo);
50  
51  		String bar = ctx.getBean("bar", String.class);
52  		assertEquals("Bar", bar);
53  
54  		Pet pet = ctx.getBean(Pet.class);
55  		assertNotNull("pet", pet);
56  		assertEquals("Dogbert", pet.getName());
57  
58  		Employee employee = ctx.getBean(Employee.class);
59  		assertNotNull("employee", employee);
60  		assertEquals("Dilbert", employee.getName());
61  		assertEquals("???", employee.getCompany());
62  	}
63  
64  }
