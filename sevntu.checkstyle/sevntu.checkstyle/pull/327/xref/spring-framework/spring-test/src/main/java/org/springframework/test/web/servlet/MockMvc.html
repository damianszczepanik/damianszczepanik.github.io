17  package org.springframework.test.web.servlet;
18  
19  import java.util.ArrayList;
20  import java.util.List;
21  import javax.servlet.Filter;
22  import javax.servlet.ServletContext;
23  
24  import org.springframework.beans.Mergeable;
25  import org.springframework.mock.web.MockFilterChain;
26  import org.springframework.mock.web.MockHttpServletRequest;
27  import org.springframework.mock.web.MockHttpServletResponse;
28  import org.springframework.util.Assert;
29  
30  /**
31   * <strong>Main entry point for server-side Spring MVC test support.</strong>
32   *
33   * <p>Below is an example:
34   *
35   * <pre class="code">
36   * static imports:
37   * MockMvcBuilders.*, MockMvcRequestBuilders.*, MockMvcResultMatchers.*
38   *
39   * WebApplicationContext wac = ...;
40   *
41   * MockMvc mockMvc = webAppContextSetup(wac).build();
42   *
43   * mockMvc.perform(get("/form"))
44   *     .andExpect(status().isOk())
45   *     .andExpect(content().mimeType("text/html"))
46   *     .andExpect(forwardedUrl("/WEB-INF/layouts/main.jsp"));
47   * </pre>
48   *
49   * @author Rossen Stoyanchev
50   * @author Rob Winch
51   * @since 3.2
52   */
53  public final class MockMvc {
54  
55  	static String MVC_RESULT_ATTRIBUTE = MockMvc.class.getName().concat(".MVC_RESULT_ATTRIBUTE");
56  
57  	private final TestDispatcherServlet servlet;
58  
59  	private final Filter[] filters;
60  
61  	private final ServletContext servletContext;
62  
63  	private RequestBuilder defaultRequestBuilder;
64  
65  	private List<ResultMatcher> defaultResultMatchers = new ArrayList<ResultMatcher>();
66  
67  	private List<ResultHandler> defaultResultHandlers = new ArrayList<ResultHandler>();
68  
69  
70  	/**
71  	 * Private constructor, not for direct instantiation.
72  	 * @see org.springframework.test.web.servlet.setup.MockMvcBuilders
73  	 */
74  	MockMvc(TestDispatcherServlet servlet, Filter[] filters, ServletContext servletContext) {
75  
76  		Assert.notNull(servlet, "DispatcherServlet is required");
77  		Assert.notNull(filters, "filters cannot be null");
78  		Assert.noNullElements(filters, "filters cannot contain null values");
79  		Assert.notNull(servletContext, "A ServletContext is required");
80  
81  		this.servlet = servlet;
82  		this.filters = filters;
83  		this.servletContext = servletContext;
84  	}
85  
86  	/**
87  	 * A default request builder merged into every performed request.
88  	 * @see org.springframework.test.web.servlet.setup.DefaultMockMvcBuilder#defaultRequest(RequestBuilder)
89  	 */
90  	void setDefaultRequest(RequestBuilder requestBuilder) {
91  		this.defaultRequestBuilder = requestBuilder;
92  	}
93  
94  	/**
95  	 * Expectations to assert after every performed request.
96  	 * @see org.springframework.test.web.servlet.setup.DefaultMockMvcBuilder#alwaysExpect(ResultMatcher)
97  	 */
98  	void setGlobalResultMatchers(List<ResultMatcher> resultMatchers) {
99  		Assert.notNull(resultMatchers, "resultMatchers is required");
100 		this.defaultResultMatchers = resultMatchers;
101 	}
102 
103 	/**
104 	 * General actions to apply after every performed request.
105 	 * @see org.springframework.test.web.servlet.setup.DefaultMockMvcBuilder#alwaysDo(ResultHandler)
106 	 */
107 	void setGlobalResultHandlers(List<ResultHandler> resultHandlers) {
108 		Assert.notNull(resultHandlers, "resultHandlers is required");
109 		this.defaultResultHandlers = resultHandlers;
110 	}
111 
112 	/**
113 	 * Perform a request and return a type that allows chaining further
114 	 * actions, such as asserting expectations, on the result.
115 	 *
116 	 * @param requestBuilder used to prepare the request to execute;
117 	 * see static factory methods in
118 	 * {@link org.springframework.test.web.servlet.request.MockMvcRequestBuilders}
119 	 *
120 	 * @return an instance of {@link ResultActions}; never {@code null}
121 	 *
122 	 * @see org.springframework.test.web.servlet.request.MockMvcRequestBuilders
123 	 * @see org.springframework.test.web.servlet.result.MockMvcResultMatchers
124 	 */
125 	public ResultActions perform(RequestBuilder requestBuilder) throws Exception {
126 
127 		if (this.defaultRequestBuilder != null) {
128 			if (requestBuilder instanceof Mergeable) {
129 				requestBuilder = (RequestBuilder) ((Mergeable) requestBuilder).merge(this.defaultRequestBuilder);
130 			}
131 		}
132 
133 		MockHttpServletRequest request = requestBuilder.buildRequest(this.servletContext);
134 		MockHttpServletResponse response = new MockHttpServletResponse();
135 
136 		if (requestBuilder instanceof SmartRequestBuilder) {
137 			request = ((SmartRequestBuilder) requestBuilder).postProcessRequest(request);
138 		}
139 
140 		final MvcResult mvcResult = new DefaultMvcResult(request, response);
141 		request.setAttribute(MVC_RESULT_ATTRIBUTE, mvcResult);
142 
143 		MockFilterChain filterChain = new MockFilterChain(this.servlet, this.filters);
144 		filterChain.doFilter(request, response);
145 
146 		applyDefaultResultActions(mvcResult);
147 
148 		return new ResultActions() {
149 
150 			@Override
151 			public ResultActions andExpect(ResultMatcher matcher) throws Exception {
152 				matcher.match(mvcResult);
153 				return this;
154 			}
155 
156 			@Override
157 			public ResultActions andDo(ResultHandler printer) throws Exception {
158 				printer.handle(mvcResult);
159 				return this;
160 			}
161 
162 			@Override
163 			public MvcResult andReturn() {
164 				return mvcResult;
165 			}
166 		};
167 	}
168 
169 	private void applyDefaultResultActions(MvcResult mvcResult) throws Exception {
170 
171 		for (ResultMatcher matcher : this.defaultResultMatchers) {
172 			matcher.match(mvcResult);
173 		}
174 
175 		for (ResultHandler handler : this.defaultResultHandlers) {
176 			handler.handle(mvcResult);
177 		}
178 	}
179 
180 }
