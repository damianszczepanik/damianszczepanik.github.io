49  /**
50   * {@code TestExecutionListener} that provides support for executing tests
51   * within <em>test-managed transactions</em> by honoring Spring's
52   * {@link org.springframework.transaction.annotation.Transactional @Transactional}
53   * annotation.
54   *
55   * <h3>Test-managed Transactions</h3>
56   * <p><em>Test-managed transactions</em> are transactions that are managed
57   * declaratively via this listener or programmatically via
58   * {@link TestTransaction}. Such transactions should not be confused with
59   * <em>Spring-managed transactions</em> (i.e., those managed directly
60   * by Spring within the {@code ApplicationContext} loaded for tests) or
61   * <em>application-managed transactions</em> (i.e., those managed
62   * programmatically within application code that is invoked via tests).
63   * Spring-managed and application-managed transactions will typically
64   * participate in test-managed transactions; however, caution should be
65   * taken if Spring-managed or application-managed transactions are
66   * configured with any propagation type other than
67   * {@link org.springframework.transaction.annotation.Propagation#REQUIRED REQUIRED}
68   * or {@link org.springframework.transaction.annotation.Propagation#SUPPORTS SUPPORTS}.
69   *
70   * <h3>Enabling and Disabling Transactions</h3>
71   * <p>Annotating a test method with {@code @Transactional} causes the test
72   * to be run within a transaction that will, by default, be automatically
73   * <em>rolled back</em> after completion of the test. If a test class is
74   * annotated with {@code @Transactional}, each test method within that class
75   * hierarchy will be run within a transaction. Test methods that are
76   * <em>not</em> annotated with {@code @Transactional} (at the class or method
77   * level) will not be run within a transaction. Furthermore, tests that
78   * <em>are</em> annotated with {@code @Transactional} but have the
79   * {@link org.springframework.transaction.annotation.Transactional#propagation propagation}
80   * type set to
81   * {@link org.springframework.transaction.annotation.Propagation#NOT_SUPPORTED NOT_SUPPORTED}
82   * will not be run within a transaction.
83   *
84   * <h3>Declarative Rollback and Commit Behavior</h3>
85   * <p>By default, test transactions will be automatically <em>rolled back</em>
86   * after completion of the test; however, transactional commit and rollback
87   * behavior can be configured declaratively via the class-level
88   * {@link TransactionConfiguration @TransactionConfiguration} and method-level
89   * {@link Rollback @Rollback} annotations.
90   *
91   * <h3>Programmatic Transaction Management</h3>
92   * <p>As of Spring Framework 4.1, it is possible to interact with test-managed
93   * transactions programmatically via the static methods in {@link TestTransaction}.
94   * {@code TestTransaction} may be used within <em>test</em> methods,
95   * <em>before</em> methods, and <em>after</em> methods.
96   *
97   * <h3>Executing Code outside of a Transaction</h3>
98   * <p>When executing transactional tests, it is sometimes useful to be able to
99   * execute certain <em>set up</em> or <em>tear down</em> code outside of a
100  * transaction. {@code TransactionalTestExecutionListener} provides such
101  * support for methods annotated with
102  * {@link BeforeTransaction @BeforeTransaction} or
103  * {@link AfterTransaction @AfterTransaction}.
104  *
105  * <h3>Configuring a Transaction Manager</h3>
106  * <p>{@code TransactionalTestExecutionListener} expects a
107  * {@link PlatformTransactionManager} bean to be defined in the Spring
108  * {@code ApplicationContext} for the test. In case there are multiple
109  * instances of {@code PlatformTransactionManager} within the test's
110  * {@code ApplicationContext}, {@code @TransactionConfiguration} supports
111  * configuring the bean name of the {@code PlatformTransactionManager} that
112  * should be used to drive transactions. Alternatively, a <em>qualifier</em>
113  * may be declared via
114  * {@link org.springframework.transaction.annotation.Transactional#value @Transactional("myQualifier")}, or
115  * {@link org.springframework.transaction.annotation.TransactionManagementConfigurer TransactionManagementConfigurer}
116  * can be implemented by an
117  * {@link org.springframework.context.annotation.Configuration @Configuration}
118  * class. See {@link TestContextTransactionUtils#retrieveTransactionManager}
119  * for details on the algorithm used to look up a transaction manager in
120  * the test's {@code ApplicationContext}.
121  *
122  * @author Sam Brannen
123  * @author Juergen Hoeller
124  * @since 2.5
125  * @see TransactionConfiguration
126  * @see org.springframework.transaction.annotation.TransactionManagementConfigurer
127  * @see org.springframework.transaction.annotation.Transactional
128  * @see org.springframework.test.annotation.Rollback
129  * @see BeforeTransaction
130  * @see AfterTransaction
131  * @see TestTransaction
132  */
133 public class TransactionalTestExecutionListener extends AbstractTestExecutionListener {

156 	/**
157 	 * If the test method of the supplied {@linkplain TestContext test context}
158 	 * is configured to run within a transaction, this method will run
159 	 * {@link BeforeTransaction @BeforeTransaction} methods and start a new
160 	 * transaction.
161 	 * <p>Note that if a {@code @BeforeTransaction} method fails, any remaining
162 	 * {@code @BeforeTransaction} methods will not be invoked, and a transaction
163 	 * will not be started.
164 	 * @see org.springframework.transaction.annotation.Transactional
165 	 * @see #getTransactionManager(TestContext, String)
166 	 */
167 	@Override
168 	public void beforeTestMethod(final TestContext testContext) throws Exception {
169 		final Method testMethod = testContext.getTestMethod();
170 		final Class<?> testClass = testContext.getTestClass();
171 		Assert.notNull(testMethod, "The test method of the supplied TestContext must not be null");
172 
173 		TransactionContext txContext = TransactionContextHolder.removeCurrentTransactionContext();
174 		if (txContext != null) {
175 			throw new IllegalStateException("Cannot start a new transaction without ending the existing transaction.");
176 		}
177 
178 		PlatformTransactionManager tm = null;
179 		TransactionAttribute transactionAttribute = this.attributeSource.getTransactionAttribute(testMethod, testClass);
180 
181 		if (transactionAttribute != null) {
182 			transactionAttribute = TestContextTransactionUtils.createDelegatingTransactionAttribute(testContext,
183 				transactionAttribute);
184 
185 			if (logger.isDebugEnabled()) {
186 				logger.debug("Explicit transaction definition [" + transactionAttribute + "] found for test context "
187 						+ testContext);
188 			}
189 
190 			if (transactionAttribute.getPropagationBehavior() == TransactionDefinition.PROPAGATION_NOT_SUPPORTED) {
191 				return;
192 			}
193 
194 			tm = getTransactionManager(testContext, transactionAttribute.getQualifier());
195 		}
196 
197 		if (tm != null) {
198 			txContext = new TransactionContext(testContext, tm, transactionAttribute, isRollback(testContext));
199 			runBeforeTransactionMethods(testContext);
200 			txContext.startTransaction();
201 			TransactionContextHolder.setCurrentTransactionContext(txContext);
202 		}
203 	}

205 	/**
206 	 * If a transaction is currently active for the supplied
207 	 * {@linkplain TestContext test context}, this method will end the transaction
208 	 * and run {@link AfterTransaction @AfterTransaction} methods.
209 	 * <p>{@code @AfterTransaction} methods are guaranteed to be invoked even if
210 	 * an error occurs while ending the transaction.
211 	 */
212 	@Override
213 	public void afterTestMethod(TestContext testContext) throws Exception {
214 		Method testMethod = testContext.getTestMethod();
215 		Assert.notNull(testMethod, "The test method of the supplied TestContext must not be null");
216 
217 		TransactionContext txContext = TransactionContextHolder.removeCurrentTransactionContext();
218 		// If there was (or perhaps still is) a transaction...
219 		if (txContext != null) {
220 			TransactionStatus transactionStatus = txContext.getTransactionStatus();
221 			try {
222 				// If the transaction is still active...
223 				if ((transactionStatus != null) && !transactionStatus.isCompleted()) {
224 					txContext.endTransaction();
225 				}
226 			}
227 			finally {
228 				runAfterTransactionMethods(testContext);
229 			}
230 		}
231 	}
232 
233 	/**
234 	 * Run all {@link BeforeTransaction @BeforeTransaction} methods for the
235 	 * specified {@link TestContext test context}. If one of the methods fails,
236 	 * however, the caught exception will be rethrown in a wrapped
237 	 * {@link RuntimeException}, and the remaining methods will <strong>not</strong>
238 	 * be given a chance to execute.
239 	 * @param testContext the current test context
240 	 */
241 	protected void runBeforeTransactionMethods(TestContext testContext) throws Exception {
242 		try {
243 			List<Method> methods = getAnnotatedMethods(testContext.getTestClass(), BeforeTransaction.class);
244 			Collections.reverse(methods);
245 			for (Method method : methods) {
246 				if (logger.isDebugEnabled()) {
247 					logger.debug("Executing @BeforeTransaction method [" + method + "] for test context " + testContext);
248 				}
249 				method.invoke(testContext.getTestInstance());
250 			}
251 		}
252 		catch (InvocationTargetException ex) {
253 			logger.error("Exception encountered while executing @BeforeTransaction methods for test context "
254 					+ testContext + ".", ex.getTargetException());
255 			ReflectionUtils.rethrowException(ex.getTargetException());
256 		}
257 	}
258 
259 	/**
260 	 * Run all {@link AfterTransaction @AfterTransaction} methods for the
261 	 * specified {@link TestContext test context}. If one of the methods fails,
262 	 * the caught exception will be logged as an error, and the remaining
263 	 * methods will be given a chance to execute. After all methods have
264 	 * executed, the first caught exception, if any, will be rethrown.
265 	 * @param testContext the current test context
266 	 */
267 	protected void runAfterTransactionMethods(TestContext testContext) throws Exception {
268 		Throwable afterTransactionException = null;
269 
270 		List<Method> methods = getAnnotatedMethods(testContext.getTestClass(), AfterTransaction.class);
271 		for (Method method : methods) {
272 			try {
273 				if (logger.isDebugEnabled()) {
274 					logger.debug("Executing @AfterTransaction method [" + method + "] for test context " + testContext);
275 				}
276 				method.invoke(testContext.getTestInstance());
277 			}
278 			catch (InvocationTargetException ex) {
279 				Throwable targetException = ex.getTargetException();
280 				if (afterTransactionException == null) {
281 					afterTransactionException = targetException;
282 				}
283 				logger.error("Exception encountered while executing @AfterTransaction method [" + method
284 						+ "] for test context " + testContext, targetException);
285 			}
286 			catch (Exception ex) {
287 				if (afterTransactionException == null) {
288 					afterTransactionException = ex;
289 				}
290 				logger.error("Exception encountered while executing @AfterTransaction method [" + method
291 						+ "] for test context " + testContext, ex);
292 			}
293 		}
294 
295 		if (afterTransactionException != null) {
296 			ReflectionUtils.rethrowException(afterTransactionException);
297 		}
298 	}

300 	/**
301 	 * Get the {@link PlatformTransactionManager transaction manager} to use
302 	 * for the supplied {@linkplain TestContext test context} and {@code qualifier}.
303 	 * <p>Delegates to {@link #getTransactionManager(TestContext)} if the
304 	 * supplied {@code qualifier} is {@code null} or empty.
305 	 * @param testContext the test context for which the transaction manager
306 	 * should be retrieved
307 	 * @param qualifier the qualifier for selecting between multiple bean matches;
308 	 * may be {@code null} or empty
309 	 * @return the transaction manager to use, or {@code null} if not found
310 	 * @throws BeansException if an error occurs while retrieving the transaction manager
311 	 * @see #getTransactionManager(TestContext)
312 	 */
313 	protected PlatformTransactionManager getTransactionManager(TestContext testContext, String qualifier) {
314 		// look up by type and qualifier from @Transactional
315 		if (StringUtils.hasText(qualifier)) {
316 			try {
317 				// Use autowire-capable factory in order to support extended qualifier
318 				// matching (only exposed on the internal BeanFactory, not on the
319 				// ApplicationContext).
320 				BeanFactory bf = testContext.getApplicationContext().getAutowireCapableBeanFactory();
321 
322 				return BeanFactoryAnnotationUtils.qualifiedBeanOfType(bf, PlatformTransactionManager.class, qualifier);
323 			}
324 			catch (RuntimeException ex) {
325 				if (logger.isWarnEnabled()) {
326 					logger.warn(
327 						String.format(
328 							"Caught exception while retrieving transaction manager with qualifier '%s' for test context %s",
329 							qualifier, testContext), ex);
330 				}
331 				throw ex;
332 			}
333 		}
334 
335 		// else
336 		return getTransactionManager(testContext);
337 	}

368 	/**
369 	 * Determine whether or not to rollback transactions for the supplied
370 	 * {@link TestContext test context} by taking into consideration the
371 	 * {@link #isDefaultRollback(TestContext) default rollback} flag and a
372 	 * possible method-level override via the {@link Rollback} annotation.
373 	 * @param testContext the test context for which the rollback flag
374 	 * should be retrieved
375 	 * @return the <em>rollback</em> flag for the supplied test context
376 	 * @throws Exception if an error occurs while determining the rollback flag
377 	 */
378 	protected final boolean isRollback(TestContext testContext) throws Exception {
379 		boolean rollback = isDefaultRollback(testContext);
380 		Rollback rollbackAnnotation = findAnnotation(testContext.getTestMethod(), Rollback.class);
381 		if (rollbackAnnotation != null) {
382 			boolean rollbackOverride = rollbackAnnotation.value();
383 			if (logger.isDebugEnabled()) {
384 				logger.debug(String.format(
385 					"Method-level @Rollback(%s) overrides default rollback [%s] for test context %s.",
386 					rollbackOverride, rollback, testContext));
387 			}
388 			rollback = rollbackOverride;
389 		}
390 		else {
391 			if (logger.isDebugEnabled()) {
392 				logger.debug(String.format(
393 					"No method-level @Rollback override: using default rollback [%s] for test context %s.", rollback,
394 					testContext));
395 			}
396 		}
397 		return rollback;
398 	}

400 	/**
401 	 * Gets all superclasses of the supplied {@link Class class}, including the
402 	 * class itself. The ordering of the returned list will begin with the
403 	 * supplied class and continue up the class hierarchy, excluding {@link Object}.
404 	 * <p>Note: This code has been borrowed from
405 	 * {@link org.junit.internal.runners.TestClass#getSuperClasses(Class)} and
406 	 * adapted.
407 	 * @param clazz the class for which to retrieve the superclasses
408 	 * @return all superclasses of the supplied class, excluding {@code Object}
409 	 */
410 	private List<Class<?>> getSuperClasses(Class<?> clazz) {
411 		List<Class<?>> results = new ArrayList<Class<?>>();
412 		Class<?> current = clazz;
413 		while (current != null && !current.equals(Object.class)) {
414 			results.add(current);
415 			current = current.getSuperclass();
416 		}
417 		return results;
418 	}

420 	/**
421 	 * Gets all methods in the supplied {@link Class class} and its superclasses
422 	 * which are annotated with the supplied {@code annotationType} but
423 	 * which are not <em>shadowed</em> by methods overridden in subclasses.
424 	 * <p>Note: This code has been borrowed from
425 	 * {@link org.junit.internal.runners.TestClass#getAnnotatedMethods(Class)}
426 	 * and adapted.
427 	 * @param clazz the class for which to retrieve the annotated methods
428 	 * @param annotationType the annotation type for which to search
429 	 * @return all annotated methods in the supplied class and its superclasses
430 	 */
431 	private List<Method> getAnnotatedMethods(Class<?> clazz, Class<? extends Annotation> annotationType) {
432 		List<Method> results = new ArrayList<Method>();
433 		for (Class<?> current : getSuperClasses(clazz)) {
434 			for (Method method : current.getDeclaredMethods()) {
435 				Annotation annotation = getAnnotation(method, annotationType);
436 				if (annotation != null && !isShadowed(method, results)) {
437 					results.add(method);
438 				}
439 			}
440 		}
441 		return results;
442 	}

444 	/**
445 	 * Determine if the supplied {@link Method method} is <em>shadowed</em> by
446 	 * a method in the supplied {@link List list} of previous methods.
447 	 * <p>Note: This code has been borrowed from
448 	 * {@link org.junit.internal.runners.TestClass#isShadowed(Method, List)}.
449 	 * @param method the method to check for shadowing
450 	 * @param previousMethods the list of methods which have previously been processed
451 	 * @return {@code true} if the supplied method is shadowed by a
452 	 * method in the {@code previousMethods} list
453 	 */
454 	private boolean isShadowed(Method method, List<Method> previousMethods) {
455 		for (Method each : previousMethods) {
456 			if (isShadowed(method, each)) {
457 				return true;
458 			}
459 		}
460 		return false;
461 	}
462 
463 	/**
464 	 * Determine if the supplied {@link Method current method} is <em>shadowed</em>
465 	 * by a {@link Method previous method}.
466 	 * <p>Note: This code has been borrowed from
467 	 * {@link org.junit.internal.runners.TestClass#isShadowed(Method, Method)}.
468 	 * @param current the current method
469 	 * @param previous the previous method
470 	 * @return {@code true} if the previous method shadows the current one
471 	 */
472 	private boolean isShadowed(Method current, Method previous) {
473 		if (!previous.getName().equals(current.getName())) {
474 			return false;
475 		}
476 		if (previous.getParameterTypes().length != current.getParameterTypes().length) {
477 			return false;
478 		}
479 		for (int i = 0; i < previous.getParameterTypes().length; i++) {
480 			if (!previous.getParameterTypes()[i].equals(current.getParameterTypes()[i])) {
481 				return false;
482 			}
483 		}
484 		return true;
485 	}

487 	/**
488 	 * Retrieves the {@link TransactionConfigurationAttributes} for the
489 	 * specified {@link Class class} which may optionally declare or inherit
490 	 * {@link TransactionConfiguration @TransactionConfiguration}. If
491 	 * {@code @TransactionConfiguration} is not present for the supplied
492 	 * class, the <em>default values</em> for attributes defined in
493 	 * {@code @TransactionConfiguration} will be used instead.
494 	 * @param testContext the test context for which the configuration
495 	 * attributes should be retrieved
496 	 * @return the TransactionConfigurationAttributes instance for this listener,
497 	 * potentially cached
498 	 */
499 	TransactionConfigurationAttributes retrieveConfigurationAttributes(TestContext testContext) {
500 		if (this.configurationAttributes == null) {
501 			Class<?> clazz = testContext.getTestClass();
502 
503 			AnnotationAttributes annAttrs = AnnotatedElementUtils.getAnnotationAttributes(clazz,
504 				TransactionConfiguration.class.getName());
505 			if (logger.isDebugEnabled()) {
506 				logger.debug(String.format("Retrieved @TransactionConfiguration attributes [%s] for test class [%s].",
507 					annAttrs, clazz));
508 			}
509 
510 			String transactionManagerName;
511 			boolean defaultRollback;
512 			if (annAttrs != null) {
513 				transactionManagerName = annAttrs.getString("transactionManager");
514 				defaultRollback = annAttrs.getBoolean("defaultRollback");
515 			}
516 			else {
517 				transactionManagerName = DEFAULT_TRANSACTION_MANAGER_NAME;
518 				defaultRollback = DEFAULT_DEFAULT_ROLLBACK;
519 			}
520 
521 			TransactionConfigurationAttributes configAttributes = new TransactionConfigurationAttributes(
522 				transactionManagerName, defaultRollback);
523 			if (logger.isDebugEnabled()) {
524 				logger.debug(String.format("Retrieved TransactionConfigurationAttributes %s for class [%s].",
525 					configAttributes, clazz));
526 			}
527 			this.configurationAttributes = configAttributes;
528 		}
529 		return this.configurationAttributes;
530 	}
