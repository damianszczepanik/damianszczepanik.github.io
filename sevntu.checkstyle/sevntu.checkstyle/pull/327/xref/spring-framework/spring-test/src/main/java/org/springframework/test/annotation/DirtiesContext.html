17  package org.springframework.test.annotation;
18  
19  import java.lang.annotation.Documented;
20  import java.lang.annotation.ElementType;
21  import java.lang.annotation.Inherited;
22  import java.lang.annotation.Retention;
23  import java.lang.annotation.RetentionPolicy;
24  import java.lang.annotation.Target;
25  
26  /**
27   * Test annotation which indicates that the
28   * {@link org.springframework.context.ApplicationContext ApplicationContext}
29   * associated with a test is <em>dirty</em> and should be closed:
30   *
31   * <ul>
32   * <li>after the current test, when declared at the method level</li>
33   * <li>after each test method in the current test class, when declared at the
34   * class level with class mode set to {@link ClassMode#AFTER_EACH_TEST_METHOD
35   * AFTER_EACH_TEST_METHOD}</li>
36   * <li>after the current test class, when declared at the class level with class
37   * mode set to {@link ClassMode#AFTER_CLASS AFTER_CLASS}</li>
38   * </ul>
39   *
40   * <p>Use this annotation if a test has modified the context &mdash; for example,
41   * by replacing a bean definition or changing the state of a singleton bean.
42   * Subsequent tests will be supplied a new context.
43   *
44   * <p>{@code @DirtiesContext} may be used as a class-level and method-level
45   * annotation within the same class. In such scenarios, the
46   * {@code ApplicationContext} will be marked as <em>dirty</em> after any
47   * such annotated method as well as after the entire class. If the
48   * {@link ClassMode} is set to {@link ClassMode#AFTER_EACH_TEST_METHOD
49   * AFTER_EACH_TEST_METHOD}, the context will be marked dirty after each test
50   * method in the class.
51   *
52   * <p>As of Spring Framework 4.0, this annotation may be used as a
53   * <em>meta-annotation</em> to create custom <em>composed annotations</em>.
54   *
55   * @author Sam Brannen
56   * @author Rod Johnson
57   * @since 2.0
58   * @see org.springframework.test.context.ContextConfiguration
59   */
60  @Documented
61  @Inherited
62  @Retention(RetentionPolicy.RUNTIME)
63  @Target({ ElementType.TYPE, ElementType.METHOD })
64  public @interface DirtiesContext {
65  
66  	/**
67  	 * Defines <i>modes</i> which determine how {@code @DirtiesContext} is
68  	 * interpreted when used to annotate a test class.
69  	 *
70  	 * @since 3.0
71  	 */
72  	static enum ClassMode {
73  
74  		/**
75  		 * The associated {@code ApplicationContext} will be marked as
76  		 * <em>dirty</em> after the test class.
77  		 */
78  		AFTER_CLASS,
79  
80  		/**
81  		 * The associated {@code ApplicationContext} will be marked as
82  		 * <em>dirty</em> after each test method in the class.
83  		 */
84  		AFTER_EACH_TEST_METHOD;
85  	}
86  
87  	/**
88  	 * Defines <i>modes</i> which determine how the context cache is cleared
89  	 * when {@code @DirtiesContext} is used in a test whose context is
90  	 * configured as part of a hierarchy via
91  	 * {@link org.springframework.test.context.ContextHierarchy @ContextHierarchy}.
92  	 *
93  	 * @since 3.2.2
94  	 */
95  	static enum HierarchyMode {
96  
97  		/**
98  		 * The context cache will be cleared using an <em>exhaustive</em> algorithm
99  		 * that includes not only the {@linkplain HierarchyMode#CURRENT_LEVEL current level}
100 		 * but also all other context hierarchies that share an ancestor context
101 		 * common to the current test.
102 		 *
103 		 * <p>All {@code ApplicationContexts} that reside in a subhierarchy of
104 		 * the common ancestor context will be removed from the context cache and
105 		 * closed.
106 		 */
107 		EXHAUSTIVE,
108 
109 		/**
110 		 * The {@code ApplicationContext} for the <em>current level</em> in the
111 		 * context hierarchy and all contexts in subhierarchies of the current
112 		 * level will be removed from the context cache and closed.
113 		 *
114 		 * <p>The <em>current level</em> refers to the {@code ApplicationContext}
115 		 * at the lowest level in the context hierarchy that is visible from the
116 		 * current test.
117 		 */
118 		CURRENT_LEVEL;
119 	}
120 
121 
122 	/**
123 	 * The <i>mode</i> to use when a test class is annotated with
124 	 * {@code @DirtiesContext}.
125 	 * <p>Defaults to {@link ClassMode#AFTER_CLASS AFTER_CLASS}.
126 	 * <p>Note: Setting the class mode on an annotated test method has no meaning,
127 	 * since the mere presence of the {@code @DirtiesContext} annotation on a
128 	 * test method is sufficient.
129 	 *
130 	 * @since 3.0
131 	 */
132 	ClassMode classMode() default ClassMode.AFTER_CLASS;
133 
134 	/**
135 	 * The context cache clearing <em>mode</em> to use when a context is
136 	 * configured as part of a hierarchy via
137 	 * {@link org.springframework.test.context.ContextHierarchy @ContextHierarchy}.
138 	 * <p>Defaults to {@link HierarchyMode#EXHAUSTIVE EXHAUSTIVE}.
139 	 *
140 	 * @since 3.2.2
141 	 */
142 	HierarchyMode hierarchyMode() default HierarchyMode.EXHAUSTIVE;
143 
144 }
