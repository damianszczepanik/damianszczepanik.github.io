34  /**
35   * {@code AbstractDelegatingSmartContextLoader} serves as an abstract base class
36   * for implementations of the {@link SmartContextLoader} SPI that delegate to a
37   * set of <em>candidate</em> SmartContextLoaders (i.e., one that supports XML
38   * configuration files or Groovy scripts and one that supports annotated classes)
39   * to determine which context loader is appropriate for a given test class's
40   * configuration. Each candidate is given a chance to
41   * {@linkplain #processContextConfiguration process} the
42   * {@link ContextConfigurationAttributes} for each class in the test class hierarchy
43   * that is annotated with {@link ContextConfiguration @ContextConfiguration}, and
44   * the candidate that supports the merged, processed configuration will be used to
45   * actually {@linkplain #loadContext load} the context.
46   *
47   * <p>Any reference to an <em>XML-based loader</em> can be interpreted to mean
48   * a context loader that supports only XML configuration files or one that
49   * supports both XML configuration files and Groovy scripts simultaneously.
50   *
51   * <p>Placing an empty {@code @ContextConfiguration} annotation on a test class signals
52   * that default resource locations (e.g., XML configuration files or Groovy scripts)
53   * or default
54   * {@linkplain org.springframework.context.annotation.Configuration configuration classes}
55   * should be detected. Furthermore, if a specific {@link ContextLoader} or
56   * {@link SmartContextLoader} is not explicitly declared via
57   * {@code @ContextConfiguration}, a concrete subclass of
58   * {@code AbstractDelegatingSmartContextLoader} will be used as the default loader,
59   * thus providing automatic support for either path-based resource locations
60   * (e.g., XML configuration files and Groovy scripts) or annotated classes,
61   * but not both simultaneously.
62   *
63   * <p>As of Spring 3.2, a test class may optionally declare neither path-based
64   * resource locations nor annotated classes and instead declare only {@linkplain
65   * ContextConfiguration#initializers application context initializers}. In such
66   * cases, an attempt will still be made to detect defaults, but their absence will
67   * not result in an exception.
68   *
69   * @author Sam Brannen
70   * @since 3.2
71   * @see SmartContextLoader
72   */
73  public abstract class AbstractDelegatingSmartContextLoader implements SmartContextLoader {
74  
75  	private static final Log logger = LogFactory.getLog(AbstractDelegatingSmartContextLoader.class);
76  
77  
78  	/**
79  	 * Get the delegate {@code SmartContextLoader} that supports XML configuration
80  	 * files and/or Groovy scripts.
81  	 */
82  	protected abstract SmartContextLoader getXmlLoader();
83  
84  	/**
85  	 * Get the delegate {@code SmartContextLoader} that supports annotated classes.
86  	 */
87  	protected abstract SmartContextLoader getAnnotationConfigLoader();
88  
89  	// --- SmartContextLoader --------------------------------------------------
90  
91  	private static String name(SmartContextLoader loader) {
92  		return loader.getClass().getSimpleName();
93  	}
94  
95  	private static void delegateProcessing(SmartContextLoader loader, ContextConfigurationAttributes configAttributes) {
96  		if (logger.isDebugEnabled()) {
97  			logger.debug(String.format("Delegating to %s to process context configuration %s.", name(loader),
98  				configAttributes));
99  		}
100 		loader.processContextConfiguration(configAttributes);
101 	}
102 
103 	private static ApplicationContext delegateLoading(SmartContextLoader loader, MergedContextConfiguration mergedConfig)
104 			throws Exception {
105 
106 		if (logger.isDebugEnabled()) {
107 			logger.debug(String.format("Delegating to %s to load context from %s.", name(loader), mergedConfig));
108 		}
109 		return loader.loadContext(mergedConfig);
110 	}
111 
112 	private boolean supports(SmartContextLoader loader, MergedContextConfiguration mergedConfig) {
113 		if (loader == getAnnotationConfigLoader()) {
114 			return mergedConfig.hasClasses() && !mergedConfig.hasLocations();
115 		}
116 		else {
117 			return mergedConfig.hasLocations() && !mergedConfig.hasClasses();
118 		}
119 	}
120 
121 	/**
122 	 * Delegates to candidate {@code SmartContextLoaders} to process the supplied
123 	 * {@link ContextConfigurationAttributes}.
124 	 * <p>Delegation is based on explicit knowledge of the implementations of the
125 	 * default loaders for {@linkplain #getXmlLoader() XML configuration files and
126 	 * Groovy scripts} and {@linkplain #getAnnotationConfigLoader() annotated classes}.
127 	 * Specifically, the delegation algorithm is as follows:
128 	 * <ul>
129 	 * <li>If the resource locations or annotated classes in the supplied
130 	 * {@code ContextConfigurationAttributes} are not empty, the appropriate
131 	 * candidate loader will be allowed to process the configuration <em>as is</em>,
132 	 * without any checks for detection of defaults.</li>
133 	 * <li>Otherwise, the XML-based loader will be allowed to process
134 	 * the configuration in order to detect default resource locations. If
135 	 * the XML-based loader detects default resource locations,
136 	 * an {@code info} message will be logged.</li>
137 	 * <li>Subsequently, the annotation-based loader will be allowed to
138 	 * process the configuration in order to detect default configuration classes.
139 	 * If the annotation-based loader detects default configuration
140 	 * classes, an {@code info} message will be logged.</li>
141 	 * </ul>
142 	 * @param configAttributes the context configuration attributes to process
143 	 * @throws IllegalArgumentException if the supplied configuration attributes are
144 	 * {@code null}, or if the supplied configuration attributes include both
145 	 * resource locations and annotated classes
146 	 * @throws IllegalStateException if the XML-based loader detects default
147 	 * configuration classes; if the annotation-based loader detects default
148 	 * resource locations; if neither candidate loader detects defaults for the supplied
149 	 * context configuration; or if both candidate loaders detect defaults for the
150 	 * supplied context configuration
151 	 */
152 	@Override
153 	public void processContextConfiguration(final ContextConfigurationAttributes configAttributes) {
154 		Assert.notNull(configAttributes, "configAttributes must not be null");
155 		Assert.isTrue(!(configAttributes.hasLocations() && configAttributes.hasClasses()), String.format(
156 			"Cannot process locations AND classes for context configuration %s: "
157 					+ "configure one or the other, but not both.", configAttributes));
158 
159 		// If the original locations or classes were not empty, there's no
160 		// need to bother with default detection checks; just let the
161 		// appropriate loader process the configuration.
162 		if (configAttributes.hasLocations()) {
163 			delegateProcessing(getXmlLoader(), configAttributes);
164 		}
165 		else if (configAttributes.hasClasses()) {
166 			delegateProcessing(getAnnotationConfigLoader(), configAttributes);
167 		}
168 		else {
169 			// Else attempt to detect defaults...
170 
171 			// Let the XML loader process the configuration.
172 			delegateProcessing(getXmlLoader(), configAttributes);
173 			boolean xmlLoaderDetectedDefaults = configAttributes.hasLocations();
174 
175 			if (xmlLoaderDetectedDefaults) {
176 				if (logger.isInfoEnabled()) {
177 					logger.info(String.format("%s detected default locations for context configuration %s.",
178 						name(getXmlLoader()), configAttributes));
179 				}
180 			}
181 
182 			if (configAttributes.hasClasses()) {
183 				throw new IllegalStateException(String.format(
184 					"%s should NOT have detected default configuration classes for context configuration %s.",
185 					name(getXmlLoader()), configAttributes));
186 			}
187 
188 			// Now let the annotation config loader process the configuration.
189 			delegateProcessing(getAnnotationConfigLoader(), configAttributes);
190 
191 			if (configAttributes.hasClasses()) {
192 				if (logger.isInfoEnabled()) {
193 					logger.info(String.format(
194 						"%s detected default configuration classes for context configuration %s.",
195 						name(getAnnotationConfigLoader()), configAttributes));
196 				}
197 			}
198 
199 			if (!xmlLoaderDetectedDefaults && configAttributes.hasLocations()) {
200 				throw new IllegalStateException(String.format(
201 					"%s should NOT have detected default locations for context configuration %s.",
202 					name(getAnnotationConfigLoader()), configAttributes));
203 			}
204 
205 			// If neither loader detected defaults and no initializers were declared,
206 			// throw an exception.
207 			if (!configAttributes.hasResources() && ObjectUtils.isEmpty(configAttributes.getInitializers())) {
208 				throw new IllegalStateException(String.format(
209 					"Neither %s nor %s was able to detect defaults, and no ApplicationContextInitializers "
210 							+ "were declared for context configuration %s", name(getXmlLoader()),
211 					name(getAnnotationConfigLoader()), configAttributes));
212 			}
213 
214 			if (configAttributes.hasLocations() && configAttributes.hasClasses()) {
215 				String message = String.format(
216 					"Configuration error: both default locations AND default configuration classes "
217 							+ "were detected for context configuration %s; configure one or the other, but not both.",
218 					configAttributes);
219 				logger.error(message);
220 				throw new IllegalStateException(message);
221 			}
222 		}
223 	}
224 
225 	/**
226 	 * Delegates to an appropriate candidate {@code SmartContextLoader} to load
227 	 * an {@link ApplicationContext}.
228 	 * <p>Delegation is based on explicit knowledge of the implementations of the
229 	 * default loaders for {@linkplain #getXmlLoader() XML configuration files and
230 	 * Groovy scripts} and {@linkplain #getAnnotationConfigLoader() annotated classes}.
231 	 * Specifically, the delegation algorithm is as follows:
232 	 * <ul>
233 	 * <li>If the resource locations in the supplied {@code MergedContextConfiguration}
234 	 * are not empty and the annotated classes are empty,
235 	 * the XML-based loader will load the {@code ApplicationContext}.</li>
236 	 * <li>If the annotated classes in the supplied {@code MergedContextConfiguration}
237 	 * are not empty and the resource locations are empty,
238 	 * the annotation-based loader will load the {@code ApplicationContext}.</li>
239 	 * </ul>
240 	 * @param mergedConfig the merged context configuration to use to load the application context
241 	 * @throws IllegalArgumentException if the supplied merged configuration is {@code null}
242 	 * @throws IllegalStateException if neither candidate loader is capable of loading an
243 	 * {@code ApplicationContext} from the supplied merged context configuration
244 	 */
245 	@Override
246 	public ApplicationContext loadContext(MergedContextConfiguration mergedConfig) throws Exception {
247 		Assert.notNull(mergedConfig, "mergedConfig must not be null");
248 		List<SmartContextLoader> candidates = Arrays.asList(getXmlLoader(), getAnnotationConfigLoader());
249 
250 		if (mergedConfig.hasLocations() && mergedConfig.hasClasses()) {
251 			throw new IllegalStateException(String.format(
252 				"Neither %s nor %s supports loading an ApplicationContext from %s: "
253 						+ "declare either 'locations' or 'classes' but not both.", name(getXmlLoader()),
254 				name(getAnnotationConfigLoader()), mergedConfig));
255 		}
256 
257 		for (SmartContextLoader loader : candidates) {
258 			// Determine if each loader can load a context from the mergedConfig. If it
259 			// can, let it; otherwise, keep iterating.
260 			if (supports(loader, mergedConfig)) {
261 				return delegateLoading(loader, mergedConfig);
262 			}
263 		}
264 
265 		// If neither of the candidates supports the mergedConfig based on resources but
266 		// ACIs were declared, then delegate to the annotation config loader.
267 		if (!mergedConfig.getContextInitializerClasses().isEmpty()) {
268 			return delegateLoading(getAnnotationConfigLoader(), mergedConfig);
269 		}
270 
271 		// else...
272 		throw new IllegalStateException(String.format(
273 			"Neither %s nor %s was able to load an ApplicationContext from %s.", name(getXmlLoader()),
274 			name(getAnnotationConfigLoader()), mergedConfig));
275 	}

279 	/**
280 	 * {@code AbstractDelegatingSmartContextLoader} does not support the
281 	 * {@link ContextLoader#processLocations(Class, String...)} method. Call
282 	 * {@link #processContextConfiguration(ContextConfigurationAttributes)} instead.
283 	 * @throws UnsupportedOperationException
284 	 */
285 	@Override
286 	public final String[] processLocations(Class<?> clazz, String... locations) {
287 		throw new UnsupportedOperationException("DelegatingSmartContextLoaders do not support the ContextLoader SPI. "
288 				+ "Call processContextConfiguration(ContextConfigurationAttributes) instead.");
289 	}

291 	/**
292 	 * {@code AbstractDelegatingSmartContextLoader} does not support the
293 	 * {@link ContextLoader#loadContext(String...) } method. Call
294 	 * {@link #loadContext(MergedContextConfiguration)} instead.
295 	 * @throws UnsupportedOperationException
296 	 */
297 	@Override
298 	public final ApplicationContext loadContext(String... locations) throws Exception {
299 		throw new UnsupportedOperationException("DelegatingSmartContextLoaders do not support the ContextLoader SPI. "
300 				+ "Call loadContext(MergedContextConfiguration) instead.");
301 	}
