40  /**
41   * Utility methods for resolving {@link ContextConfigurationAttributes} from the
42   * {@link ContextConfiguration @ContextConfiguration} and
43   * {@link ContextHierarchy @ContextHierarchy} annotations for use with
44   * {@link SmartContextLoader SmartContextLoaders}.
45   *
46   * @author Sam Brannen
47   * @since 3.1
48   * @see SmartContextLoader
49   * @see ContextConfigurationAttributes
50   * @see ContextConfiguration
51   * @see ContextHierarchy
52   */
53  abstract class ContextLoaderUtils {
54  
55  	static final String GENERATED_CONTEXT_HIERARCHY_LEVEL_PREFIX = "ContextHierarchyLevel#";
56  
57  	private static final Log logger = LogFactory.getLog(ContextLoaderUtils.class);
58  
59  
60  	private ContextLoaderUtils() {
61  		/* no-op */
62  	}

64  	/**
65  	 * Resolve the list of lists of {@linkplain ContextConfigurationAttributes context
66  	 * configuration attributes} for the supplied {@linkplain Class test class} and its
67  	 * superclasses, taking into account context hierarchies declared via
68  	 * {@link ContextHierarchy @ContextHierarchy} and
69  	 * {@link ContextConfiguration @ContextConfiguration}.
70  	 *
71  	 * <p>The outer list represents a top-down ordering of context configuration
72  	 * attributes, where each element in the list represents the context configuration
73  	 * declared on a given test class in the class hierarchy. Each nested list
74  	 * contains the context configuration attributes declared either via a single
75  	 * instance of {@code @ContextConfiguration} on the particular class or via
76  	 * multiple instances of {@code @ContextConfiguration} declared within a
77  	 * single {@code @ContextHierarchy} instance on the particular class.
78  	 * Furthermore, each nested list maintains the order in which
79  	 * {@code @ContextConfiguration} instances are declared.
80  	 *
81  	 * <p>Note that the {@link ContextConfiguration#inheritLocations inheritLocations} and
82  	 * {@link ContextConfiguration#inheritInitializers() inheritInitializers} flags of
83  	 * {@link ContextConfiguration @ContextConfiguration} will <strong>not</strong>
84  	 * be taken into consideration. If these flags need to be honored, that must be
85  	 * handled manually when traversing the nested lists returned by this method.
86  	 *
87  	 * @param testClass the class for which to resolve the context hierarchy attributes
88  	 * (must not be {@code null})
89  	 * @return the list of lists of configuration attributes for the specified class;
90  	 * never {@code null}
91  	 * @throws IllegalArgumentException if the supplied class is {@code null}; if
92  	 * neither {@code @ContextConfiguration} nor {@code @ContextHierarchy} is
93  	 * <em>present</em> on the supplied class; or if a test class or composed annotation
94  	 * in the class hierarchy declares both {@code @ContextConfiguration} and
95  	 * {@code @ContextHierarchy} as top-level annotations.
96  	 * @throws IllegalStateException if no class in the class hierarchy declares
97  	 * {@code @ContextHierarchy}.
98  	 *
99  	 * @since 3.2.2
100 	 * @see #buildContextHierarchyMap(Class)
101 	 * @see #resolveContextConfigurationAttributes(Class)
102 	 */
103 	@SuppressWarnings("unchecked")
104 	static List<List<ContextConfigurationAttributes>> resolveContextHierarchyAttributes(Class<?> testClass) {
105 		Assert.notNull(testClass, "Class must not be null");
106 		Assert.state(findAnnotation(testClass, ContextHierarchy.class) != null, "@ContextHierarchy must be present");
107 
108 		final Class<ContextConfiguration> contextConfigType = ContextConfiguration.class;
109 		final Class<ContextHierarchy> contextHierarchyType = ContextHierarchy.class;
110 		final List<List<ContextConfigurationAttributes>> hierarchyAttributes = new ArrayList<List<ContextConfigurationAttributes>>();
111 
112 		UntypedAnnotationDescriptor descriptor = findAnnotationDescriptorForTypes(testClass, contextConfigType,
113 			contextHierarchyType);
114 		Assert.notNull(descriptor, String.format(
115 			"Could not find an 'annotation declaring class' for annotation type [%s] or [%s] and test class [%s]",
116 			contextConfigType.getName(), contextHierarchyType.getName(), testClass.getName()));
117 
118 		while (descriptor != null) {
119 			Class<?> rootDeclaringClass = descriptor.getRootDeclaringClass();
120 			Class<?> declaringClass = descriptor.getDeclaringClass();
121 
122 			boolean contextConfigDeclaredLocally = isAnnotationDeclaredLocally(contextConfigType, declaringClass);
123 			boolean contextHierarchyDeclaredLocally = isAnnotationDeclaredLocally(contextHierarchyType, declaringClass);
124 
125 			if (contextConfigDeclaredLocally && contextHierarchyDeclaredLocally) {
126 				String msg = String.format("Class [%s] has been configured with both @ContextConfiguration "
127 						+ "and @ContextHierarchy. Only one of these annotations may be declared on a test class "
128 						+ "or composed annotation.", declaringClass.getName());
129 				logger.error(msg);
130 				throw new IllegalStateException(msg);
131 			}
132 
133 			final List<ContextConfigurationAttributes> configAttributesList = new ArrayList<ContextConfigurationAttributes>();
134 
135 			if (contextConfigDeclaredLocally) {
136 				convertAnnotationAttributesToConfigAttributesAndAddToList(descriptor.getAnnotationAttributes(),
137 					rootDeclaringClass, configAttributesList);
138 			}
139 			else if (contextHierarchyDeclaredLocally) {
140 				ContextHierarchy contextHierarchy = getAnnotation(declaringClass, contextHierarchyType);
141 				for (ContextConfiguration contextConfiguration : contextHierarchy.value()) {
142 					convertContextConfigToConfigAttributesAndAddToList(contextConfiguration, rootDeclaringClass,
143 						configAttributesList);
144 				}
145 			}
146 			else {
147 				// This should theoretically never happen...
148 				String msg = String.format("Test class [%s] has been configured with neither @ContextConfiguration "
149 						+ "nor @ContextHierarchy as a class-level annotation.", rootDeclaringClass.getName());
150 				logger.error(msg);
151 				throw new IllegalStateException(msg);
152 			}
153 
154 			hierarchyAttributes.add(0, configAttributesList);
155 
156 			descriptor = findAnnotationDescriptorForTypes(rootDeclaringClass.getSuperclass(), contextConfigType,
157 				contextHierarchyType);
158 		}
159 
160 		return hierarchyAttributes;
161 	}
162 
163 	/**
164 	 * Build a <em>context hierarchy map</em> for the supplied {@linkplain Class
165 	 * test class} and its superclasses, taking into account context hierarchies
166 	 * declared via {@link ContextHierarchy @ContextHierarchy} and
167 	 * {@link ContextConfiguration @ContextConfiguration}.
168 	 *
169 	 * <p>Each value in the map represents the consolidated list of {@linkplain
170 	 * ContextConfigurationAttributes context configuration attributes} for a
171 	 * given level in the context hierarchy (potentially across the test class
172 	 * hierarchy), keyed by the {@link ContextConfiguration#name() name} of the
173 	 * context hierarchy level.
174 	 *
175 	 * <p>If a given level in the context hierarchy does not have an explicit
176 	 * name (i.e., configured via {@link ContextConfiguration#name}), a name will
177 	 * be generated for that hierarchy level by appending the numerical level to
178 	 * the {@link #GENERATED_CONTEXT_HIERARCHY_LEVEL_PREFIX}.
179 	 *
180 	 * @param testClass the class for which to resolve the context hierarchy map
181 	 * (must not be {@code null})
182 	 * @return a map of context configuration attributes for the context hierarchy,
183 	 * keyed by context hierarchy level name; never {@code null}
184 	 * @throws IllegalArgumentException if the lists of context configuration
185 	 * attributes for each level in the {@code @ContextHierarchy} do not define
186 	 * unique context configuration within the overall hierarchy.
187 	 *
188 	 * @since 3.2.2
189 	 * @see #resolveContextHierarchyAttributes(Class)
190 	 */
191 	static Map<String, List<ContextConfigurationAttributes>> buildContextHierarchyMap(Class<?> testClass) {
192 		final Map<String, List<ContextConfigurationAttributes>> map = new LinkedHashMap<String, List<ContextConfigurationAttributes>>();
193 		int hierarchyLevel = 1;
194 
195 		for (List<ContextConfigurationAttributes> configAttributesList : resolveContextHierarchyAttributes(testClass)) {
196 			for (ContextConfigurationAttributes configAttributes : configAttributesList) {
197 				String name = configAttributes.getName();
198 
199 				// Assign a generated name?
200 				if (!StringUtils.hasText(name)) {
201 					name = GENERATED_CONTEXT_HIERARCHY_LEVEL_PREFIX + hierarchyLevel;
202 				}
203 
204 				// Encountered a new context hierarchy level?
205 				if (!map.containsKey(name)) {
206 					hierarchyLevel++;
207 					map.put(name, new ArrayList<ContextConfigurationAttributes>());
208 				}
209 
210 				map.get(name).add(configAttributes);
211 			}
212 		}
213 
214 		// Check for uniqueness
215 		Set<List<ContextConfigurationAttributes>> set = new HashSet<List<ContextConfigurationAttributes>>(map.values());
216 		if (set.size() != map.size()) {
217 			String msg = String.format("The @ContextConfiguration elements configured via "
218 					+ "@ContextHierarchy in test class [%s] and its superclasses must "
219 					+ "define unique contexts per hierarchy level.", testClass.getName());
220 			logger.error(msg);
221 			throw new IllegalStateException(msg);
222 		}
223 
224 		return map;
225 	}

227 	/**
228 	 * Resolve the list of {@linkplain ContextConfigurationAttributes context
229 	 * configuration attributes} for the supplied {@linkplain Class test class} and its
230 	 * superclasses.
231 	 *
232 	 * <p>Note that the {@link ContextConfiguration#inheritLocations inheritLocations} and
233 	 * {@link ContextConfiguration#inheritInitializers() inheritInitializers} flags of
234 	 * {@link ContextConfiguration @ContextConfiguration} will <strong>not</strong>
235 	 * be taken into consideration. If these flags need to be honored, that must be
236 	 * handled manually when traversing the list returned by this method.
237 	 *
238 	 * @param testClass the class for which to resolve the configuration attributes (must
239 	 * not be {@code null})
240 	 * @return the list of configuration attributes for the specified class, ordered
241 	 * <em>bottom-up</em> (i.e., as if we were traversing up the class hierarchy);
242 	 * never {@code null}
243 	 * @throws IllegalArgumentException if the supplied class is {@code null} or if
244 	 * {@code @ContextConfiguration} is not <em>present</em> on the supplied class
245 	 */
246 	static List<ContextConfigurationAttributes> resolveContextConfigurationAttributes(Class<?> testClass) {
247 		Assert.notNull(testClass, "Class must not be null");
248 
249 		final List<ContextConfigurationAttributes> attributesList = new ArrayList<ContextConfigurationAttributes>();
250 
251 		Class<ContextConfiguration> annotationType = ContextConfiguration.class;
252 
253 		AnnotationDescriptor<ContextConfiguration> descriptor = findAnnotationDescriptor(testClass, annotationType);
254 		Assert.notNull(descriptor, String.format(
255 			"Could not find an 'annotation declaring class' for annotation type [%s] and class [%s]",
256 			annotationType.getName(), testClass.getName()));
257 
258 		while (descriptor != null) {
259 			convertAnnotationAttributesToConfigAttributesAndAddToList(descriptor.getAnnotationAttributes(),
260 				descriptor.getRootDeclaringClass(), attributesList);
261 			descriptor = findAnnotationDescriptor(descriptor.getRootDeclaringClass().getSuperclass(), annotationType);
262 		}
263 
264 		return attributesList;
265 	}

267 	/**
268 	 * Convenience method for creating a {@link ContextConfigurationAttributes}
269 	 * instance from the supplied {@link ContextConfiguration} annotation and
270 	 * declaring class and then adding the attributes to the supplied list.
271 	 */
272 	private static void convertContextConfigToConfigAttributesAndAddToList(ContextConfiguration contextConfiguration,
273 			Class<?> declaringClass, final List<ContextConfigurationAttributes> attributesList) {
274 		if (logger.isTraceEnabled()) {
275 			logger.trace(String.format("Retrieved @ContextConfiguration [%s] for declaring class [%s].",
276 				contextConfiguration, declaringClass.getName()));
277 		}
278 
279 		ContextConfigurationAttributes attributes = new ContextConfigurationAttributes(declaringClass,
280 			contextConfiguration);
281 		if (logger.isTraceEnabled()) {
282 			logger.trace("Resolved context configuration attributes: " + attributes);
283 		}
284 		attributesList.add(attributes);
285 	}
286 
287 	/**
288 	 * Convenience method for creating a {@link ContextConfigurationAttributes}
289 	 * instance from the supplied {@link AnnotationAttributes} and declaring
290 	 * class and then adding the attributes to the supplied list.
291 	 * @since 4.0
292 	 */
293 	private static void convertAnnotationAttributesToConfigAttributesAndAddToList(AnnotationAttributes annAttrs,
294 			Class<?> declaringClass, final List<ContextConfigurationAttributes> attributesList) {
295 		if (logger.isTraceEnabled()) {
296 			logger.trace(String.format("Retrieved @ContextConfiguration attributes [%s] for declaring class [%s].",
297 				annAttrs, declaringClass.getName()));
298 		}
299 
300 		ContextConfigurationAttributes attributes = new ContextConfigurationAttributes(declaringClass, annAttrs);
301 		if (logger.isTraceEnabled()) {
302 			logger.trace("Resolved context configuration attributes: " + attributes);
303 		}
304 		attributesList.add(attributes);
305 	}
