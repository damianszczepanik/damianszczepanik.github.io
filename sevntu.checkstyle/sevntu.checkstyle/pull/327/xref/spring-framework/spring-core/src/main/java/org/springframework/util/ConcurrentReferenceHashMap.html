36  /**
37   * A {@link ConcurrentHashMap} that uses {@link ReferenceType#SOFT soft} or
38   * {@linkplain ReferenceType#WEAK weak} references for both {@code keys} and {@code values}.
39   *
40   * <p>This class can be used as an alternative to
41   * {@code Collections.synchronizedMap(new WeakHashMap<K, Reference<V>>())} in order to
42   * support better performance when accessed concurrently. This implementation follows the
43   * same design constraints as {@link ConcurrentHashMap} with the exception that
44   * {@code null} values and {@code null} keys are supported.
45   *
46   * <p><b>NOTE:</b> The use of references means that there is no guarantee that items
47   * placed into the map will be subsequently available. The garbage collector may discard
48   * references at any time, so it may appear that an unknown thread is silently removing
49   * entries.
50   *
51   * <p>If not explicitly specified, this implementation will use
52   * {@linkplain SoftReference soft entry references}.
53   *
54   * @param <K> The key type
55   * @param <V> The value type
56   * @author Phillip Webb
57   * @since 3.2
58   */
59  public class ConcurrentReferenceHashMap<K, V> extends AbstractMap<K, V> implements ConcurrentMap<K, V> {

156 	/**
157 	 * Create a new {@code ConcurrentReferenceHashMap} instance.
158 	 * @param initialCapacity the initial capacity of the map
159 	 * @param loadFactor the load factor. When the average number of references per
160 	 * table exceeds this value, resize will be attempted.
161 	 * @param concurrencyLevel the expected number of threads that will concurrently
162 	 * write to the map
163 	 * @param referenceType the reference type used for entries (soft or weak)
164 	 */
165 	@SuppressWarnings("unchecked")
166 	public ConcurrentReferenceHashMap(int initialCapacity, float loadFactor, int concurrencyLevel,
167 			ReferenceType referenceType) {
168 
169 		Assert.isTrue(initialCapacity >= 0, "Initial capacity must not be negative");
170 		Assert.isTrue(loadFactor > 0f, "Load factor must be positive");
171 		Assert.isTrue(concurrencyLevel > 0, "Concurrency level must be positive");
172 		Assert.notNull(referenceType, "Reference type must not be null");
173 		this.loadFactor = loadFactor;
174 		this.shift = calculateShift(concurrencyLevel, MAXIMUM_CONCURRENCY_LEVEL);
175 		int size = 1 << this.shift;
176 		this.referenceType = referenceType;
177 		int roundedUpSegmentCapacity = (int) ((initialCapacity + size - 1L) / size);
178 		this.segments = (Segment[]) Array.newInstance(Segment.class, size);
179 		for (int i = 0; i < this.segments.length; i++) {
180 			this.segments[i] = new Segment(roundedUpSegmentCapacity);
181 		}
182 	}

206 	/**
207 	 * Get the hash for a given object, apply an additional hash function to reduce
208 	 * collisions. This implementation uses the same Wang/Jenkins algorithm as
209 	 * {@link ConcurrentHashMap}. Subclasses can override to provide alternative hashing.
210 	 * @param o the object to hash (may be null)
211 	 * @return the resulting hash code
212 	 */
213 	protected int getHash(Object o) {
214 		int hash = o == null ? 0 : o.hashCode();
215 		hash += (hash << 15) ^ 0xffffcd7d;
216 		hash ^= (hash >>> 10);
217 		hash += (hash << 3);
218 		hash ^= (hash >>> 6);
219 		hash += (hash << 2) + (hash << 14);
220 		hash ^= (hash >>> 16);
221 		return hash;
222 	}

260 	private V put(final K key, final V value, final boolean overwriteExisting) {
261 		return doTask(key, new Task<V>(TaskOption.RESTRUCTURE_BEFORE, TaskOption.RESIZE) {
262 			@Override
263 			protected V execute(Reference<K, V> reference, Entry<K, V> entry, Entries entries) {
264 				if (entry != null) {
265 					V previousValue = entry.getValue();
266 					if (overwriteExisting) {
267 						entry.setValue(value);
268 					}
269 					return previousValue;
270 				}
271 				entries.add(value);
272 				return null;
273 			}
274 		});
275 	}

372 		int hash = getHash(key);

376 	private Segment getSegmentForHash(int hash) {
377 		return this.segments[(hash >>> (32 - this.shift)) & (this.segments.length - 1)];
378 	}

380 	/**
381 	 * Calculate a shift value that can be used to create a power-of-two value between
382 	 * the specified maximum and minimum values.
383 	 * @param minimumValue the minimum value
384 	 * @param maximumValue the maximum value
385 	 * @return the calculated shift (use {@code 1 << shift} to obtain a value)
386 	 */
387 	protected static int calculateShift(int minimumValue, int maximumValue) {
388 		int shift = 0;
389 		int value = 1;
390 		while (value < minimumValue && value < maximumValue) {
391 			value <<= 1;
392 			shift++;
393 		}
394 		return shift;
395 	}

411 	/**
412 	 * A single segment used to divide the map to allow better concurrent performance.
413 	 */
414 	@SuppressWarnings("serial")
415 	protected final class Segment extends ReentrantLock {
416 
417 		private final ReferenceManager referenceManager;
418 
419 		private final int initialSize;
420 
421 		/**
422 		 * Array of references indexed using the low order bits from the hash. This
423 		 * property should only be set via {@link #setReferences} to ensure that the
424 		 * {@code resizeThreshold} is maintained.
425 		 */
426 		private volatile Reference<K, V>[] references;
427 
428 		/**
429 		 * The total number of references contained in this segment. This includes chained
430 		 * references and references that have been garbage collected but not purged.
431 		 */
432 		private volatile int count = 0;
433 
434 		/**
435 		 * The threshold when resizing of the references should occur. When {@code count}
436 		 * exceeds this value references will be resized.
437 		 */
438 		private int resizeThreshold;
439 
440 		public Segment(int initialCapacity) {
441 			this.referenceManager = createReferenceManager();
442 			this.initialSize = 1 << calculateShift(initialCapacity, MAXIMUM_SEGMENT_SIZE);
443 			setReferences(createReferenceArray(this.initialSize));
444 		}
445 
446 		public Reference<K, V> getReference(Object key, int hash, Restructure restructure) {
447 			if (restructure == Restructure.WHEN_NECESSARY) {
448 				restructureIfNecessary(false);
449 			}
450 			if (this.count == 0) {
451 				return null;
452 			}
453 			// Use a local copy to protect against other threads writing
454 			Reference<K, V>[] references = this.references;
455 			int index = getIndex(hash, references);
456 			Reference<K, V> head = references[index];
457 			return findInChain(head, key, hash);
458 		}
459 
460 		/**
461 		 * Apply an update operation to this segment.
462 		 * The segment will be locked during the update.
463 		 * @param hash the hash of the key
464 		 * @param key the key
465 		 * @param task the update operation
466 		 * @return the result of the operation
467 		 */
468 		public <T> T doTask(final int hash, final Object key, final Task<T> task) {
469 			boolean resize = task.hasOption(TaskOption.RESIZE);
470 			if (task.hasOption(TaskOption.RESTRUCTURE_BEFORE)) {
471 				restructureIfNecessary(resize);
472 			}
473 			if (task.hasOption(TaskOption.SKIP_IF_EMPTY) && this.count == 0) {
474 				return task.execute(null, null, null);
475 			}
476 			lock();
477 			try {
478 				final int index = getIndex(hash, this.references);
479 				final Reference<K, V> head = this.references[index];
480 				Reference<K, V> reference = findInChain(head, key, hash);
481 				Entry<K, V> entry = (reference != null ? reference.get() : null);
482 				Entries entries = new Entries() {
483 					@Override
484 					public void add(V value) {
485 						@SuppressWarnings("unchecked")
486 						Entry<K, V> newEntry = new Entry<K, V>((K) key, value);
487 						Reference<K, V> newReference = Segment.this.referenceManager.createReference(newEntry, hash, head);
488 						Segment.this.references[index] = newReference;
489 						Segment.this.count++;
490 					}
491 				};
492 				return task.execute(reference, entry, entries);
493 			}
494 			finally {
495 				unlock();
496 				if (task.hasOption(TaskOption.RESTRUCTURE_AFTER)) {
497 					restructureIfNecessary(resize);
498 				}
499 			}
500 		}
501 
502 		/**
503 		 * Clear all items from this segment.
504 		 */
505 		public void clear() {
506 			if (this.count == 0) {
507 				return;
508 			}
509 			lock();
510 			try {
511 				setReferences(createReferenceArray(this.initialSize));
512 				this.count = 0;
513 			}
514 			finally {
515 				unlock();
516 			}
517 		}
518 
519 		/**
520 		 * Restructure the underlying data structure when it becomes necessary. This
521 		 * method can increase the size of the references table as well as purge any
522 		 * references that have been garbage collected.
523 		 * @param allowResize if resizing is permitted
524 		 */
525 		protected final void restructureIfNecessary(boolean allowResize) {
526 			boolean needsResize = ((this.count > 0) && (this.count >= this.resizeThreshold));
527 			Reference<K, V> reference = this.referenceManager.pollForPurge();
528 			if ((reference != null) || (needsResize && allowResize)) {
529 				lock();
530 				try {
531 					int countAfterRestructure = this.count;
532 
533 					Set<Reference<K, V>> toPurge = Collections.emptySet();
534 					if (reference != null) {
535 						toPurge = new HashSet<Reference<K, V>>();
536 						while (reference != null) {
537 							toPurge.add(reference);
538 							reference = this.referenceManager.pollForPurge();
539 						}
540 					}
541 					countAfterRestructure -= toPurge.size();
542 
543 					// Recalculate taking into account count inside lock and items that
544 					// will be purged
545 					needsResize = (countAfterRestructure > 0 && countAfterRestructure >= this.resizeThreshold);
546 					boolean resizing = false;
547 					int restructureSize = this.references.length;
548 					if (allowResize && needsResize && restructureSize < MAXIMUM_SEGMENT_SIZE) {
549 						restructureSize <<= 1;
550 						resizing = true;
551 					}
552 
553 					// Either create a new table or reuse the existing one
554 					Reference<K, V>[] restructured = (resizing ? createReferenceArray(restructureSize) : this.references);
555 
556 					// Restructure
557 					for (int i = 0; i < this.references.length; i++) {
558 						reference = this.references[i];
559 						if (!resizing) {
560 							restructured[i] = null;
561 						}
562 						while (reference != null) {
563 							if (!toPurge.contains(reference) && (reference.get() != null)) {
564 								int index = getIndex(reference.getHash(), restructured);
565 								restructured[index] = this.referenceManager.createReference(
566 										reference.get(), reference.getHash(),
567 										restructured[index]);
568 							}
569 							reference = reference.getNext();
570 						}
571 					}
572 
573 					// Replace volatile members
574 					if (resizing) {
575 						setReferences(restructured);
576 					}
577 					this.count = Math.max(countAfterRestructure, 0);
578 				}
579 				finally {
580 					unlock();
581 				}
582 			}
583 		}
584 
585 		private Reference<K, V> findInChain(Reference<K, V> reference, Object key, int hash) {
586 			while (reference != null) {
587 				if (reference.getHash() == hash) {
588 					Entry<K, V> entry = reference.get();
589 					if (entry != null) {
590 						K entryKey = entry.getKey();
591 						if (entryKey == key || entryKey.equals(key)) {
592 							return reference;
593 						}
594 					}
595 				}
596 				reference = reference.getNext();
597 			}
598 			return null;
599 		}
600 
601 		@SuppressWarnings("unchecked")
602 		private Reference<K, V>[] createReferenceArray(int size) {
603 			return (Reference<K, V>[]) Array.newInstance(Reference.class, size);
604 		}
605 
606 		private int getIndex(int hash, Reference<K, V>[] references) {
607 			return (hash & (references.length - 1));
608 		}
609 
610 		/**
611 		 * Replace the references with a new value, recalculating the resizeThreshold.
612 		 * @param references the new references
613 		 */
614 		private void setReferences(Reference<K, V>[] references) {
615 			this.references = references;
616 			this.resizeThreshold = (int) (references.length * getLoadFactor());
617 		}

635 	/**
636 	 * A reference to an {@link Entry} contained in the map. Implementations are usually
637 	 * wrappers around specific Java reference implementations (e.g., {@link SoftReference}).
638 	 */
639 	protected static interface Reference<K, V> {
640 
641 		/**
642 		 * Returns the referenced entry or {@code null} if the entry is no longer
643 		 * available.
644 		 * @return the entry or {@code null}
645 		 */
646 		Entry<K, V> get();
647 
648 		/**
649 		 * Returns the hash for the reference.
650 		 * @return the hash
651 		 */
652 		int getHash();
653 
654 		/**
655 		 * Returns the next reference in the chain or {@code null}
656 		 * @return the next reference of {@code null}
657 		 */
658 		Reference<K, V> getNext();
659 
660 		/**
661 		 * Release this entry and ensure that it will be returned from
662 		 * {@code ReferenceManager#pollForPurge()}.

831 	/**
832 	 * Internal entry iterator implementation.
833 	 */
834 	private class EntryIterator implements Iterator<Map.Entry<K, V>> {
835 
836 		private int segmentIndex;
837 
838 		private int referenceIndex;
839 
840 		private Reference<K, V>[] references;
841 
842 		private Reference<K, V> reference;
843 
844 		private Entry<K, V> next;
845 
846 		private Entry<K, V> last;
847 
848 		public EntryIterator() {
849 			moveToNextSegment();
850 		}
851 
852 		@Override
853 		public boolean hasNext() {
854 			getNextIfNecessary();
855 			return (this.next != null);
856 		}
857 
858 		@Override
859 		public Entry<K, V> next() {
860 			getNextIfNecessary();
861 			if (this.next == null) {
862 				throw new NoSuchElementException();
863 			}
864 			this.last = this.next;
865 			this.next = null;
866 			return this.last;
867 		}
868 
869 		private void getNextIfNecessary() {
870 			while (this.next == null) {
871 				moveToNextReference();
872 				if (this.reference == null) {
873 					return;
874 				}
875 				this.next = this.reference.get();
876 			}
877 		}
878 
879 		private void moveToNextReference() {
880 			if (this.reference != null) {
881 				this.reference = this.reference.getNext();
882 			}
883 			while (this.reference == null && this.references != null) {
884 				if (this.referenceIndex >= this.references.length) {
885 					moveToNextSegment();
886 					this.referenceIndex = 0;
887 				}
888 				else {
889 					this.reference = this.references[this.referenceIndex];
890 					this.referenceIndex++;
891 				}
892 			}
893 		}
894 
895 		private void moveToNextSegment() {
896 			this.reference = null;
897 			this.references = null;
898 			if (this.segmentIndex < ConcurrentReferenceHashMap.this.segments.length) {
899 				this.references = ConcurrentReferenceHashMap.this.segments[this.segmentIndex].references;
900 				this.segmentIndex++;
901 			}
902 		}
903 
904 		@Override
905 		public void remove() {
906 			Assert.state(this.last != null);
907 			ConcurrentReferenceHashMap.this.remove(this.last.getKey());
908 		}
909 	}

921 	/**
922 	 * Strategy class used to manage {@link Reference}s. This class can be overridden if
923 	 * alternative reference types need to be supported.
924 	 */
925 	protected class ReferenceManager {
926 
927 		private final ReferenceQueue<Entry<K, V>> queue = new ReferenceQueue<Entry<K, V>>();
928 
929 		/**
930 		 * Factory method used to create a new {@link Reference}.
931 		 * @param entry the entry contained in the reference
932 		 * @param hash the hash
933 		 * @param next the next reference in the chain or {@code null}
934 		 * @return a new {@link Reference}
935 		 */
936 		public Reference<K, V> createReference(Entry<K, V> entry, int hash, Reference<K, V> next) {
937 			if (ConcurrentReferenceHashMap.this.referenceType == ReferenceType.WEAK) {
938 				return new WeakEntryReference<K, V>(entry, hash, next, this.queue);
939 			}
940 			return new SoftEntryReference<K, V>(entry, hash, next, this.queue);
941 		}
942 
943 		/**
944 		 * Return any reference that has been garbage collected and can be purged from the
945 		 * underlying structure or {@code null} if no references need purging. This
946 		 * method must be thread safe and ideally should not block when returning
947 		 * {@code null}. References should be returned once and only once.
948 		 * @return a reference to purge or {@code null}
949 		 */
950 		@SuppressWarnings("unchecked")
951 		public Reference<K, V> pollForPurge() {
952 			return (Reference<K, V>) this.queue.poll();
953 		}
954 	}
