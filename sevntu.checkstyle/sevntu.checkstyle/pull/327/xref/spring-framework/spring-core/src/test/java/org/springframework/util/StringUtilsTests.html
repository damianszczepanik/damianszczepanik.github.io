169 	public void testReplace() throws Exception {
170 		String inString = "a6AazAaa77abaa";
171 		String oldPattern = "aa";
172 		String newPattern = "foo";
173 
174 		// Simple replace
175 		String s = StringUtils.replace(inString, oldPattern, newPattern);
176 		assertTrue("Replace 1 worked", s.equals("a6AazAfoo77abfoo"));
177 
178 		// Non match: no change
179 		s = StringUtils.replace(inString, "qwoeiruqopwieurpoqwieur", newPattern);
180 		assertTrue("Replace non matched is equal", s.equals(inString));
181 
182 		// Null new pattern: should ignore
183 		s = StringUtils.replace(inString, oldPattern, null);
184 		assertTrue("Replace non matched is equal", s.equals(inString));
185 
186 		// Null old pattern: should ignore
187 		s = StringUtils.replace(inString, null, newPattern);
188 		assertTrue("Replace non matched is equal", s.equals(inString));
189 	}

220 		String inString = "Able was I ere I saw Elba";
221 
222 		String res = StringUtils.deleteAny(inString, "I");
223 		assertTrue("Result has no Is [" + res + "]", res.equals("Able was  ere  saw Elba"));
224 
225 		res = StringUtils.deleteAny(inString, "AeEba!");
226 		assertTrue("Result has no Is [" + res + "]", res.equals("l ws I r I sw l"));
227 
228 		String mismatch = StringUtils.deleteAny(inString, "#@$#$^");
229 		assertTrue("Result is unchanged", mismatch.equals(inString));
230 
231 		String whitespace = "This is\n\n\n    \t   a messagy string with whitespace\n";
232 		assertTrue("Has CR", whitespace.contains("\n"));
233 		assertTrue("Has tab", whitespace.contains("\t"));
234 		assertTrue("Has  sp", whitespace.contains(" "));
235 		String cleaned = StringUtils.deleteAny(whitespace, "\n\t ");
236 		assertTrue("Has no CR", !cleaned.contains("\n"));
237 		assertTrue("Has no tab", !cleaned.contains("\t"));
238 		assertTrue("Has no sp", !cleaned.contains(" "));
239 		assertTrue("Still has chars", cleaned.length() > 10);

331 	@Test
332 	public void testPathEquals() {
333 		assertTrue("Must be true for the same strings",
334 				StringUtils.pathEquals("/dummy1/dummy2/dummy3",
335 						"/dummy1/dummy2/dummy3"));
336 		assertTrue("Must be true for the same win strings",
337 				StringUtils.pathEquals("C:\\dummy1\\dummy2\\dummy3",
338 						"C:\\dummy1\\dummy2\\dummy3"));
339 		assertTrue("Must be true for one top path on 1",
340 				StringUtils.pathEquals("/dummy1/bin/../dummy2/dummy3",
341 						"/dummy1/dummy2/dummy3"));
342 		assertTrue("Must be true for one win top path on 2",
343 				StringUtils.pathEquals("C:\\dummy1\\dummy2\\dummy3",
344 						"C:\\dummy1\\bin\\..\\dummy2\\dummy3"));
345 		assertTrue("Must be true for two top paths on 1",
346 				StringUtils.pathEquals("/dummy1/bin/../dummy2/bin/../dummy3",
347 						"/dummy1/dummy2/dummy3"));
348 		assertTrue("Must be true for two win top paths on 2",
349 				StringUtils.pathEquals("C:\\dummy1\\dummy2\\dummy3",
350 						"C:\\dummy1\\bin\\..\\dummy2\\bin\\..\\dummy3"));
351 		assertTrue("Must be true for double top paths on 1",
352 				StringUtils.pathEquals("/dummy1/bin/tmp/../../dummy2/dummy3",
353 						"/dummy1/dummy2/dummy3"));
354 		assertTrue("Must be true for double top paths on 2 with similarity",
355 				StringUtils.pathEquals("/dummy1/dummy2/dummy3",
356 						"/dummy1/dum/dum/../../dummy2/dummy3"));
357 		assertTrue("Must be true for current paths",
358 				StringUtils.pathEquals("./dummy1/dummy2/dummy3",
359 						"dummy1/dum/./dum/../../dummy2/dummy3"));
360 		assertFalse("Must be false for relative/absolute paths",
361 				StringUtils.pathEquals("./dummy1/dummy2/dummy3",
362 						"/dummy1/dum/./dum/../../dummy2/dummy3"));
363 		assertFalse("Must be false for different strings",
364 				StringUtils.pathEquals("/dummy1/dummy2/dummy3",
365 						"/dummy1/dummy4/dummy3"));
366 		assertFalse("Must be false for one false path on 1",
367 				StringUtils.pathEquals("/dummy1/bin/tmp/../dummy2/dummy3",
368 						"/dummy1/dummy2/dummy3"));
369 		assertFalse("Must be false for one false win top path on 2",
370 				StringUtils.pathEquals("C:\\dummy1\\dummy2\\dummy3",
371 						"C:\\dummy1\\bin\\tmp\\..\\dummy2\\dummy3"));
372 		assertFalse("Must be false for top path on 1 + difference",
373 				StringUtils.pathEquals("/dummy1/bin/../dummy2/dummy3",
374 						"/dummy1/dummy2/dummy4"));
375 	}

569 	private void doTestCommaDelimitedListToStringArrayLegalMatch(String[] components) {
570 		StringBuilder sb = new StringBuilder();
571 		for (int i = 0; i < components.length; i++) {
572 			if (i != 0) {
573 				sb.append(",");
574 			}
575 			sb.append(components[i]);
576 		}
577 		String[] sa = StringUtils.commaDelimitedListToStringArray(sb.toString());
578 		assertTrue("String array isn't null with legal match", sa != null);
579 		assertEquals("String array length is correct with legal match", components.length, sa.length);
580 		assertTrue("Output equals input", Arrays.equals(sa, components));
581 	}
