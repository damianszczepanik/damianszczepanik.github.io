17  package org.springframework.core.io;
18  
19  import java.io.IOException;
20  import java.io.InputStream;
21  
22  /**
23   * {@link Resource} implementation for a given {@link InputStream}.
24   * <p>Should only be used if no other specific {@code Resource} implementation
25   * is applicable. In particular, prefer {@link ByteArrayResource} or any of the
26   * file-based {@code Resource} implementations where possible.
27   *
28   * <p>In contrast to other {@code Resource} implementations, this is a descriptor
29   * for an <i>already opened</i> resource - therefore returning {@code true} from
30   * {@link #isOpen()}. Do not use an {@code InputStreamResource} if you need to
31   * keep the resource descriptor somewhere, or if you need to read from a stream
32   * multiple times.
33   *
34   * @author Juergen Hoeller
35   * @author Sam Brannen
36   * @since 28.12.2003
37   * @see ByteArrayResource
38   * @see ClassPathResource
39   * @see FileSystemResource
40   * @see UrlResource
41   */
42  public class InputStreamResource extends AbstractResource {
43  
44  	private final InputStream inputStream;
45  
46  	private final String description;
47  
48  	private boolean read = false;
49  
50  
51  	/**
52  	 * Create a new InputStreamResource.
53  	 * @param inputStream the InputStream to use
54  	 */
55  	public InputStreamResource(InputStream inputStream) {
56  		this(inputStream, "resource loaded through InputStream");
57  	}
58  
59  	/**
60  	 * Create a new InputStreamResource.
61  	 * @param inputStream the InputStream to use
62  	 * @param description where the InputStream comes from
63  	 */
64  	public InputStreamResource(InputStream inputStream, String description) {
65  		if (inputStream == null) {
66  			throw new IllegalArgumentException("InputStream must not be null");
67  		}
68  		this.inputStream = inputStream;
69  		this.description = (description != null ? description : "");
70  	}
71  
72  
73  	/**
74  	 * This implementation always returns {@code true}.
75  	 */
76  	@Override
77  	public boolean exists() {
78  		return true;
79  	}
80  
81  	/**
82  	 * This implementation always returns {@code true}.
83  	 */
84  	@Override
85  	public boolean isOpen() {
86  		return true;
87  	}
88  
89  	/**
90  	 * This implementation throws IllegalStateException if attempting to
91  	 * read the underlying stream multiple times.
92  	 */
93  	@Override
94  	public InputStream getInputStream() throws IOException, IllegalStateException {
95  		if (this.read) {
96  			throw new IllegalStateException("InputStream has already been read - " +
97  					"do not use InputStreamResource if a stream needs to be read multiple times");
98  		}
99  		this.read = true;
100 		return this.inputStream;
101 	}
102 
103 	/**
104 	 * This implementation returns a description that includes the passed-in
105 	 * description, if any.
106 	 */
107 	@Override
108 	public String getDescription() {
109 		return "InputStream resource [" + this.description + "]";
110 	}
111 
112 
113 	/**
114 	 * This implementation compares the underlying InputStream.
115 	 */
116 	@Override
117 	public boolean equals(Object obj) {
118 		return (obj == this ||
119 			(obj instanceof InputStreamResource && ((InputStreamResource) obj).inputStream.equals(this.inputStream)));
120 	}
121 
122 	/**
123 	 * This implementation returns the hash code of the underlying InputStream.
124 	 */
125 	@Override
126 	public int hashCode() {
127 		return this.inputStream.hashCode();
128 	}
129 
130 }
