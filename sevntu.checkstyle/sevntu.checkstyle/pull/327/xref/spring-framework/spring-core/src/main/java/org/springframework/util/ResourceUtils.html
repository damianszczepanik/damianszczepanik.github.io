17  package org.springframework.util;
18  
19  import java.io.File;
20  import java.io.FileNotFoundException;
21  import java.net.MalformedURLException;
22  import java.net.URI;
23  import java.net.URISyntaxException;
24  import java.net.URL;
25  import java.net.URLConnection;

27  /**
28   * Utility methods for resolving resource locations to files in the
29   * file system. Mainly for internal use within the framework.
30   *
31   * <p>Consider using Spring's Resource abstraction in the core package
32   * for handling all kinds of file resources in a uniform manner.
33   * {@link org.springframework.core.io.ResourceLoader}'s {@code getResource()}
34   * method can resolve any location to a {@link org.springframework.core.io.Resource}
35   * object, which in turn allows one to obtain a {@code java.io.File} in the
36   * file system through its {@code getFile()} method.
37   *
38   * <p>The main reason for these utility methods for resource location handling
39   * is to support {@link Log4jConfigurer}, which must be able to resolve
40   * resource locations <i>before the logging system has been initialized</i>.
41   * Spring's {@code Resource} abstraction in the core package, on the other hand,
42   * already expects the logging system to be available.
43   *
44   * @author Juergen Hoeller
45   * @since 1.1.5
46   * @see org.springframework.core.io.Resource
47   * @see org.springframework.core.io.ClassPathResource
48   * @see org.springframework.core.io.FileSystemResource
49   * @see org.springframework.core.io.UrlResource
50   * @see org.springframework.core.io.ResourceLoader
51   */
52  public abstract class ResourceUtils {
53  
54  	/** Pseudo URL prefix for loading from the class path: "classpath:" */
55  	public static final String CLASSPATH_URL_PREFIX = "classpath:";
56  
57  	/** URL prefix for loading from the file system: "file:" */
58  	public static final String FILE_URL_PREFIX = "file:";
59  
60  	/** URL prefix for loading from the file system: "jar:" */
61  	public static final String JAR_URL_PREFIX = "jar:";
62  
63  	/** URL protocol for a file in the file system: "file" */
64  	public static final String URL_PROTOCOL_FILE = "file";
65  
66  	/** URL protocol for an entry from a jar file: "jar" */
67  	public static final String URL_PROTOCOL_JAR = "jar";
68  
69  	/** URL protocol for an entry from a zip file: "zip" */
70  	public static final String URL_PROTOCOL_ZIP = "zip";
71  
72  	/** URL protocol for an entry from a WebSphere jar file: "wsjar" */
73  	public static final String URL_PROTOCOL_WSJAR = "wsjar";
74  
75  	/** URL protocol for an entry from a JBoss jar file: "vfszip" */
76  	public static final String URL_PROTOCOL_VFSZIP = "vfszip";
77  
78  	/** URL protocol for a JBoss file system resource: "vfsfile" */
79  	public static final String URL_PROTOCOL_VFSFILE = "vfsfile";
80  
81  	/** URL protocol for a general JBoss VFS resource: "vfs" */
82  	public static final String URL_PROTOCOL_VFS = "vfs";
83  
84  	/** File extension for a regular jar file: ".jar" */
85  	public static final String JAR_FILE_EXTENSION = ".jar";
86  
87  	/** Separator between JAR URL and file path within the JAR: "!/" */
88  	public static final String JAR_URL_SEPARATOR = "!/";

91  	/**
92  	 * Return whether the given resource location is a URL:
93  	 * either a special "classpath" pseudo URL or a standard URL.
94  	 * @param resourceLocation the location String to check
95  	 * @return whether the location qualifies as a URL
96  	 * @see #CLASSPATH_URL_PREFIX
97  	 * @see java.net.URL
98  	 */
99  	public static boolean isUrl(String resourceLocation) {
100 		if (resourceLocation == null) {
101 			return false;
102 		}
103 		if (resourceLocation.startsWith(CLASSPATH_URL_PREFIX)) {
104 			return true;
105 		}
106 		try {
107 			new URL(resourceLocation);
108 			return true;
109 		}
110 		catch (MalformedURLException ex) {
111 			return false;
112 		}
113 	}
114 
115 	/**
116 	 * Resolve the given resource location to a {@code java.net.URL}.
117 	 * <p>Does not check whether the URL actually exists; simply returns
118 	 * the URL that the given location would correspond to.
119 	 * @param resourceLocation the resource location to resolve: either a
120 	 * "classpath:" pseudo URL, a "file:" URL, or a plain file path
121 	 * @return a corresponding URL object
122 	 * @throws FileNotFoundException if the resource cannot be resolved to a URL
123 	 */
124 	public static URL getURL(String resourceLocation) throws FileNotFoundException {
125 		Assert.notNull(resourceLocation, "Resource location must not be null");
126 		if (resourceLocation.startsWith(CLASSPATH_URL_PREFIX)) {
127 			String path = resourceLocation.substring(CLASSPATH_URL_PREFIX.length());
128 			ClassLoader cl = ClassUtils.getDefaultClassLoader();
129 			URL url = (cl != null ? cl.getResource(path) : ClassLoader.getSystemResource(path));
130 			if (url == null) {
131 				String description = "class path resource [" + path + "]";
132 				throw new FileNotFoundException(description +
133 						" cannot be resolved to URL because it does not exist");
134 			}
135 			return url;
136 		}
137 		try {
138 			// try URL
139 			return new URL(resourceLocation);
140 		}
141 		catch (MalformedURLException ex) {
142 			// no URL -> treat as file path
143 			try {
144 				return new File(resourceLocation).toURI().toURL();
145 			}
146 			catch (MalformedURLException ex2) {
147 				throw new FileNotFoundException("Resource location [" + resourceLocation +
148 						"] is neither a URL not a well-formed file path");
149 			}
150 		}
151 	}
152 
153 	/**
154 	 * Resolve the given resource location to a {@code java.io.File},
155 	 * i.e. to a file in the file system.
156 	 * <p>Does not check whether the file actually exists; simply returns
157 	 * the File that the given location would correspond to.
158 	 * @param resourceLocation the resource location to resolve: either a
159 	 * "classpath:" pseudo URL, a "file:" URL, or a plain file path
160 	 * @return a corresponding File object
161 	 * @throws FileNotFoundException if the resource cannot be resolved to
162 	 * a file in the file system
163 	 */
164 	public static File getFile(String resourceLocation) throws FileNotFoundException {
165 		Assert.notNull(resourceLocation, "Resource location must not be null");
166 		if (resourceLocation.startsWith(CLASSPATH_URL_PREFIX)) {
167 			String path = resourceLocation.substring(CLASSPATH_URL_PREFIX.length());
168 			String description = "class path resource [" + path + "]";
169 			ClassLoader cl = ClassUtils.getDefaultClassLoader();
170 			URL url = (cl != null ? cl.getResource(path) : ClassLoader.getSystemResource(path));
171 			if (url == null) {
172 				throw new FileNotFoundException(description +
173 						" cannot be resolved to absolute file path because it does not exist");
174 			}
175 			return getFile(url, description);
176 		}
177 		try {
178 			// try URL
179 			return getFile(new URL(resourceLocation));
180 		}
181 		catch (MalformedURLException ex) {
182 			// no URL -> treat as file path
183 			return new File(resourceLocation);
184 		}
185 	}
186 
187 	/**
188 	 * Resolve the given resource URL to a {@code java.io.File},
189 	 * i.e. to a file in the file system.
190 	 * @param resourceUrl the resource URL to resolve
191 	 * @return a corresponding File object
192 	 * @throws FileNotFoundException if the URL cannot be resolved to
193 	 * a file in the file system
194 	 */
195 	public static File getFile(URL resourceUrl) throws FileNotFoundException {
196 		return getFile(resourceUrl, "URL");
197 	}
198 
199 	/**
200 	 * Resolve the given resource URL to a {@code java.io.File},
201 	 * i.e. to a file in the file system.
202 	 * @param resourceUrl the resource URL to resolve
203 	 * @param description a description of the original resource that
204 	 * the URL was created for (for example, a class path location)
205 	 * @return a corresponding File object
206 	 * @throws FileNotFoundException if the URL cannot be resolved to
207 	 * a file in the file system
208 	 */
209 	public static File getFile(URL resourceUrl, String description) throws FileNotFoundException {
210 		Assert.notNull(resourceUrl, "Resource URL must not be null");
211 		if (!URL_PROTOCOL_FILE.equals(resourceUrl.getProtocol())) {
212 			throw new FileNotFoundException(
213 					description + " cannot be resolved to absolute file path " +
214 					"because it does not reside in the file system: " + resourceUrl);
215 		}
216 		try {
217 			return new File(toURI(resourceUrl).getSchemeSpecificPart());
218 		}
219 		catch (URISyntaxException ex) {
220 			// Fallback for URLs that are not valid URIs (should hardly ever happen).
221 			return new File(resourceUrl.getFile());
222 		}
223 	}

225 	/**
226 	 * Resolve the given resource URI to a {@code java.io.File},
227 	 * i.e. to a file in the file system.
228 	 * @param resourceUri the resource URI to resolve
229 	 * @return a corresponding File object
230 	 * @throws FileNotFoundException if the URL cannot be resolved to
231 	 * a file in the file system
232 	 */
233 	public static File getFile(URI resourceUri) throws FileNotFoundException {
234 		return getFile(resourceUri, "URI");
235 	}
236 
237 	/**
238 	 * Resolve the given resource URI to a {@code java.io.File},
239 	 * i.e. to a file in the file system.
240 	 * @param resourceUri the resource URI to resolve
241 	 * @param description a description of the original resource that
242 	 * the URI was created for (for example, a class path location)
243 	 * @return a corresponding File object
244 	 * @throws FileNotFoundException if the URL cannot be resolved to
245 	 * a file in the file system
246 	 */
247 	public static File getFile(URI resourceUri, String description) throws FileNotFoundException {
248 		Assert.notNull(resourceUri, "Resource URI must not be null");
249 		if (!URL_PROTOCOL_FILE.equals(resourceUri.getScheme())) {
250 			throw new FileNotFoundException(
251 					description + " cannot be resolved to absolute file path " +
252 					"because it does not reside in the file system: " + resourceUri);
253 		}
254 		return new File(resourceUri.getSchemeSpecificPart());
255 	}

257 	/**
258 	 * Determine whether the given URL points to a resource in the file system,
259 	 * that is, has protocol "file", "vfsfile" or "vfs".
260 	 * @param url the URL to check
261 	 * @return whether the URL has been identified as a file system URL
262 	 */
263 	public static boolean isFileURL(URL url) {
264 		String protocol = url.getProtocol();
265 		return (URL_PROTOCOL_FILE.equals(protocol) || URL_PROTOCOL_VFSFILE.equals(protocol) ||
266 				URL_PROTOCOL_VFS.equals(protocol));
267 	}
268 
269 	/**
270 	 * Determine whether the given URL points to a resource in a jar file,
271 	 * that is, has protocol "jar", "zip", "vfszip" or "wsjar".
272 	 * @param url the URL to check
273 	 * @return whether the URL has been identified as a JAR URL
274 	 */
275 	public static boolean isJarURL(URL url) {
276 		String protocol = url.getProtocol();
277 		return (URL_PROTOCOL_JAR.equals(protocol) || URL_PROTOCOL_ZIP.equals(protocol) ||
278 				URL_PROTOCOL_VFSZIP.equals(protocol) || URL_PROTOCOL_WSJAR.equals(protocol));
279 	}
280 
281 	/**
282 	 * Determine whether the given URL points to a jar file itself,
283 	 * that is, has protocol "file" and ends with the ".jar" extension.
284 	 * @param url the URL to check
285 	 * @return whether the URL has been identified as a JAR file URL
286 	 * @since 4.1
287 	 */
288 	public static boolean isJarFileURL(URL url) {
289 		return (URL_PROTOCOL_FILE.equals(url.getProtocol()) &&
290 				url.getPath().toLowerCase().endsWith(JAR_FILE_EXTENSION));
291 	}

293 	/**
294 	 * Extract the URL for the actual jar file from the given URL
295 	 * (which may point to a resource in a jar file or to a jar file itself).
296 	 * @param jarUrl the original URL
297 	 * @return the URL for the actual jar file
298 	 * @throws MalformedURLException if no valid jar file URL could be extracted
299 	 */
300 	public static URL extractJarFileURL(URL jarUrl) throws MalformedURLException {
301 		String urlFile = jarUrl.getFile();
302 		int separatorIndex = urlFile.indexOf(JAR_URL_SEPARATOR);
303 		if (separatorIndex != -1) {
304 			String jarFile = urlFile.substring(0, separatorIndex);
305 			try {
306 				return new URL(jarFile);
307 			}
308 			catch (MalformedURLException ex) {
309 				// Probably no protocol in original jar URL, like "jar:C:/mypath/myjar.jar".
310 				// This usually indicates that the jar file resides in the file system.
311 				if (!jarFile.startsWith("/")) {
312 					jarFile = "/" + jarFile;
313 				}
314 				return new URL(FILE_URL_PREFIX + jarFile);
315 			}
316 		}
317 		else {
318 			return jarUrl;
319 		}
320 	}

322 	/**
323 	 * Create a URI instance for the given URL,
324 	 * replacing spaces with "%20" URI encoding first.
325 	 * <p>Furthermore, this method works on JDK 1.4 as well,
326 	 * in contrast to the {@code URL.toURI()} method.
327 	 * @param url the URL to convert into a URI instance
328 	 * @return the URI instance
329 	 * @throws URISyntaxException if the URL wasn't a valid URI
330 	 * @see java.net.URL#toURI()
331 	 */
332 	public static URI toURI(URL url) throws URISyntaxException {
333 		return toURI(url.toString());
334 	}
335 
336 	/**
337 	 * Create a URI instance for the given location String,
338 	 * replacing spaces with "%20" URI encoding first.
339 	 * @param location the location String to convert into a URI instance
340 	 * @return the URI instance
341 	 * @throws URISyntaxException if the location wasn't a valid URI
342 	 */
343 	public static URI toURI(String location) throws URISyntaxException {
344 		return new URI(StringUtils.replace(location, " ", "%20"));
345 	}

348 	 * Set the {@link URLConnection#setUseCaches "useCaches"} flag on the
349 	 * given connection, preferring {@code false} but leaving the
350 	 * flag at {@code true} for JNLP based resources.
351 	 * @param con the URLConnection to set the flag on
352 	 */
353 	public static void useCachesIfNecessary(URLConnection con) {
354 		con.setUseCaches(con.getClass().getSimpleName().startsWith("JNLP"));
