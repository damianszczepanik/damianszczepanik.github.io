28  /**
29   * PathMatcher implementation for Ant-style path patterns. Examples are provided below.
30   *
31   * <p>Part of this mapping code has been kindly borrowed from <a href="http://ant.apache.org">Apache Ant</a>.
32   *
33   * <p>The mapping matches URLs using the following rules:<br> <ul> <li>? matches one character</li> <li>* matches zero
34   * or more characters</li> <li>** matches zero or more 'directories' in a path</li> </ul>
35   *
36   * <p>Some examples:<br> <ul> <li>{@code com/t?st.jsp} - matches {@code com/test.jsp} but also
37   * {@code com/tast.jsp} or {@code com/txst.jsp}</li> <li>{@code com/*.jsp} - matches all
38   * {@code .jsp} files in the {@code com} directory</li> <li>{@code com/&#42;&#42;/test.jsp} - matches all
39   * {@code test.jsp} files underneath the {@code com} path</li> <li>{@code org/springframework/&#42;&#42;/*.jsp}
40   * - matches all {@code .jsp} files underneath the {@code org/springframework} path</li>
41   * <li>{@code org/&#42;&#42;/servlet/bla.jsp} - matches {@code org/springframework/servlet/bla.jsp} but also
42   * {@code org/springframework/testing/servlet/bla.jsp} and {@code org/servlet/bla.jsp}</li> </ul>
43   *
44   * @author Alef Arendsen
45   * @author Juergen Hoeller
46   * @author Rob Harrop
47   * @author Arjen Poutsma
48   * @author Rossen Stoyanchev
49   * @since 16.07.2003
50   */
51  public class AntPathMatcher implements PathMatcher {

148 	/**
149 	 * Actually match the given {@code path} against the given {@code pattern}.
150 	 * @param pattern the pattern to match against
151 	 * @param path the path String to test
152 	 * @param fullMatch whether a full pattern match is required (else a pattern match
153 	 * as far as the given base path goes is sufficient)
154 	 * @return {@code true} if the supplied {@code path} matched, {@code false} if it didn't
155 	 */
156 	protected boolean doMatch(String pattern, String path, boolean fullMatch, Map<String, String> uriTemplateVariables) {
157 		if (path.startsWith(this.pathSeparator) != pattern.startsWith(this.pathSeparator)) {
158 			return false;
159 		}
160 
161 		String[] pattDirs = tokenizePattern(pattern);
162 		String[] pathDirs = tokenizePath(path);
163 
164 		int pattIdxStart = 0;
165 		int pattIdxEnd = pattDirs.length - 1;
166 		int pathIdxStart = 0;
167 		int pathIdxEnd = pathDirs.length - 1;
168 
169 		// Match all elements up to the first **
170 		while (pattIdxStart <= pattIdxEnd && pathIdxStart <= pathIdxEnd) {
171 			String pattDir = pattDirs[pattIdxStart];
172 			if ("**".equals(pattDir)) {
173 				break;
174 			}
175 			if (!matchStrings(pattDir, pathDirs[pathIdxStart], uriTemplateVariables)) {
176 				return false;
177 			}
178 			pattIdxStart++;
179 			pathIdxStart++;
180 		}
181 
182 		if (pathIdxStart > pathIdxEnd) {
183 			// Path is exhausted, only match if rest of pattern is * or **'s
184 			if (pattIdxStart > pattIdxEnd) {
185 				return (pattern.endsWith(this.pathSeparator) ? path.endsWith(this.pathSeparator) :
186 						!path.endsWith(this.pathSeparator));
187 			}
188 			if (!fullMatch) {
189 				return true;
190 			}
191 			if (pattIdxStart == pattIdxEnd && pattDirs[pattIdxStart].equals("*") && path.endsWith(this.pathSeparator)) {
192 				return true;
193 			}
194 			for (int i = pattIdxStart; i <= pattIdxEnd; i++) {
195 				if (!pattDirs[i].equals("**")) {
196 					return false;
197 				}
198 			}
199 			return true;
200 		}
201 		else if (pattIdxStart > pattIdxEnd) {
202 			// String not exhausted, but pattern is. Failure.
203 			return false;
204 		}
205 		else if (!fullMatch && "**".equals(pattDirs[pattIdxStart])) {
206 			// Path start definitely matches due to "**" part in pattern.
207 			return true;
208 		}
209 
210 		// up to last '**'
211 		while (pattIdxStart <= pattIdxEnd && pathIdxStart <= pathIdxEnd) {
212 			String pattDir = pattDirs[pattIdxEnd];
213 			if (pattDir.equals("**")) {
214 				break;
215 			}
216 			if (!matchStrings(pattDir, pathDirs[pathIdxEnd], uriTemplateVariables)) {
217 				return false;
218 			}
219 			pattIdxEnd--;
220 			pathIdxEnd--;
221 		}
222 		if (pathIdxStart > pathIdxEnd) {
223 			// String is exhausted
224 			for (int i = pattIdxStart; i <= pattIdxEnd; i++) {
225 				if (!pattDirs[i].equals("**")) {
226 					return false;
227 				}
228 			}
229 			return true;
230 		}
231 
232 		while (pattIdxStart != pattIdxEnd && pathIdxStart <= pathIdxEnd) {
233 			int patIdxTmp = -1;
234 			for (int i = pattIdxStart + 1; i <= pattIdxEnd; i++) {
235 				if (pattDirs[i].equals("**")) {
236 					patIdxTmp = i;
237 					break;
238 				}
239 			}
240 			if (patIdxTmp == pattIdxStart + 1) {
241 				// '**/**' situation, so skip one
242 				pattIdxStart++;
243 				continue;
244 			}
245 			// Find the pattern between padIdxStart & padIdxTmp in str between
246 			// strIdxStart & strIdxEnd
247 			int patLength = (patIdxTmp - pattIdxStart - 1);
248 			int strLength = (pathIdxEnd - pathIdxStart + 1);
249 			int foundIdx = -1;
250 
251 			strLoop:
252 			for (int i = 0; i <= strLength - patLength; i++) {
253 				for (int j = 0; j < patLength; j++) {
254 					String subPat = pattDirs[pattIdxStart + j + 1];
255 					String subStr = pathDirs[pathIdxStart + i + j];
256 					if (!matchStrings(subPat, subStr, uriTemplateVariables)) {
257 						continue strLoop;
258 					}
259 				}
260 				foundIdx = pathIdxStart + i;
261 				break;
262 			}
263 
264 			if (foundIdx == -1) {
265 				return false;
266 			}
267 
268 			pattIdxStart = patIdxTmp;
269 			pathIdxStart = foundIdx + patLength;
270 		}
271 
272 		for (int i = pattIdxStart; i <= pattIdxEnd; i++) {
273 			if (!pattDirs[i].equals("**")) {
274 				return false;
275 			}
276 		}
277 
278 		return true;
279 	}

281 	/**
282 	 * Tokenize the given path pattern into parts, based on this matcher's settings.
283 	 * <p>Performs caching based on {@link #setCachePatterns}, delegating to
284 	 * {@link #tokenizePath(String)} for the actual tokenization algorithm.
285 	 * @param pattern the pattern to tokenize
286 	 * @return the tokenized pattern parts
287 	 */
288 	protected String[] tokenizePattern(String pattern) {
289 		String[] tokenized = null;
290 		Boolean cachePatterns = this.cachePatterns;
291 		if (cachePatterns == null || cachePatterns.booleanValue()) {
292 			tokenized = this.tokenizedPatternCache.get(pattern);
293 		}
294 		if (tokenized == null) {
295 			tokenized = tokenizePath(pattern);
296 			if (cachePatterns == null && this.tokenizedPatternCache.size() >= CACHE_TURNOFF_THRESHOLD) {
297 				// Try to adapt to the runtime situation that we're encountering:
298 				// There are obviously too many different patterns coming in here...
299 				// So let's turn off the cache since the patterns are unlikely to be reoccurring.
300 				deactivatePatternCache();
301 				return tokenized;
302 			}
303 			if (cachePatterns == null || cachePatterns.booleanValue()) {
304 				this.tokenizedPatternCache.put(pattern, tokenized);
305 			}
306 		}
307 		return tokenized;
308 	}

329 	/**
330 	 * Build or retrieve an {@link AntPathStringMatcher} for the given pattern.
331 	 * <p>The default implementation checks this AntPathMatcher's internal cache
332 	 * (see {@link #setCachePatterns}), creating a new AntPathStringMatcher instance
333 	 * if no cached copy is found.
334 	 * When encountering too many patterns to cache at runtime (the threshold is 65536),
335 	 * it turns the default cache off, assuming that arbitrary permutations of patterns
336 	 * are coming in, with little chance for encountering a reoccurring pattern.
337 	 * <p>This method may get overridden to implement a custom cache strategy.
338 	 * @param pattern the pattern to match against (never {@code null})
339 	 * @return a corresponding AntPathStringMatcher (never {@code null})
340 	 * @see #setCachePatterns
341 	 */
342 	protected AntPathStringMatcher getStringMatcher(String pattern) {
343 		AntPathStringMatcher matcher = null;
344 		Boolean cachePatterns = this.cachePatterns;
345 		if (cachePatterns == null || cachePatterns.booleanValue()) {
346 			matcher = this.stringMatcherCache.get(pattern);
347 		}
348 		if (matcher == null) {
349 			matcher = new AntPathStringMatcher(pattern);
350 			if (cachePatterns == null && this.stringMatcherCache.size() >= CACHE_TURNOFF_THRESHOLD) {
351 				// Try to adapt to the runtime situation that we're encountering:
352 				// There are obviously too many different patterns coming in here...
353 				// So let's turn off the cache since the patterns are unlikely to be reoccurring.
354 				deactivatePatternCache();
355 				return matcher;
356 			}
357 			if (cachePatterns == null || cachePatterns.booleanValue()) {
358 				this.stringMatcherCache.put(pattern, matcher);
359 			}
360 		}
361 		return matcher;
362 	}

364 	/**
365 	 * Given a pattern and a full path, determine the pattern-mapped part. <p>For example: <ul>
366 	 * <li>'{@code /docs/cvs/commit.html}' and '{@code /docs/cvs/commit.html} -> ''</li>
367 	 * <li>'{@code /docs/*}' and '{@code /docs/cvs/commit} -> '{@code cvs/commit}'</li>
368 	 * <li>'{@code /docs/cvs/*.html}' and '{@code /docs/cvs/commit.html} -> '{@code commit.html}'</li>
369 	 * <li>'{@code /docs/**}' and '{@code /docs/cvs/commit} -> '{@code cvs/commit}'</li>
370 	 * <li>'{@code /docs/**\/*.html}' and '{@code /docs/cvs/commit.html} -> '{@code cvs/commit.html}'</li>
371 	 * <li>'{@code /*.html}' and '{@code /docs/cvs/commit.html} -> '{@code docs/cvs/commit.html}'</li>
372 	 * <li>'{@code *.html}' and '{@code /docs/cvs/commit.html} -> '{@code /docs/cvs/commit.html}'</li>
373 	 * <li>'{@code *}' and '{@code /docs/cvs/commit.html} -> '{@code /docs/cvs/commit.html}'</li> </ul>
374 	 * <p>Assumes that {@link #match} returns {@code true} for '{@code pattern}' and '{@code path}', but
375 	 * does <strong>not</strong> enforce this.
376 	 */
377 	@Override
378 	public String extractPathWithinPattern(String pattern, String path) {
379 		String[] patternParts = StringUtils.tokenizeToStringArray(pattern, this.pathSeparator, this.trimTokens, true);
380 		String[] pathParts = StringUtils.tokenizeToStringArray(path, this.pathSeparator, this.trimTokens, true);
381 		StringBuilder builder = new StringBuilder();
382 		boolean pathStarted = false;
383 
384 		for (int segment = 0; segment < patternParts.length; segment++) {
385 			String patternPart = patternParts[segment];
386 			if (patternPart.indexOf('*') > -1 || patternPart.indexOf('?') > -1) {
387 				for (; segment < pathParts.length; segment++) {
388 					if (pathStarted || (segment == 0 && !pattern.startsWith(this.pathSeparator))) {
389 						builder.append(this.pathSeparator);
390 					}
391 					builder.append(pathParts[segment]);
392 					pathStarted = true;
393 				}
394 			}
395 		}
396 
397 		return builder.toString();
398 	}

408 	/**
409 	 * Combines two patterns into a new pattern that is returned.
410 	 * <p>This implementation simply concatenates the two patterns, unless the first pattern
411 	 * contains a file extension match (such as {@code *.html}. In that case, the second pattern
412 	 * should be included in the first, or an {@code IllegalArgumentException} is thrown.
413 	 * <p>For example: <table>
414 	 * <tr><th>Pattern 1</th><th>Pattern 2</th><th>Result</th></tr> <tr><td>/hotels</td><td>{@code
415 	 * null}</td><td>/hotels</td></tr> <tr><td>{@code null}</td><td>/hotels</td><td>/hotels</td></tr>
416 	 * <tr><td>/hotels</td><td>/bookings</td><td>/hotels/bookings</td></tr> <tr><td>/hotels</td><td>bookings</td><td>/hotels/bookings</td></tr>
417 	 * <tr><td>/hotels/*</td><td>/bookings</td><td>/hotels/bookings</td></tr> <tr><td>/hotels/&#42;&#42;</td><td>/bookings</td><td>/hotels/&#42;&#42;/bookings</td></tr>
418 	 * <tr><td>/hotels</td><td>{hotel}</td><td>/hotels/{hotel}</td></tr> <tr><td>/hotels/*</td><td>{hotel}</td><td>/hotels/{hotel}</td></tr>
419 	 * <tr><td>/hotels/&#42;&#42;</td><td>{hotel}</td><td>/hotels/&#42;&#42;/{hotel}</td></tr>
420 	 * <tr><td>/*.html</td><td>/hotels.html</td><td>/hotels.html</td></tr> <tr><td>/*.html</td><td>/hotels</td><td>/hotels.html</td></tr>
421 	 * <tr><td>/*.html</td><td>/*.txt</td><td>IllegalArgumentException</td></tr> </table>
422 	 * @param pattern1 the first pattern
423 	 * @param pattern2 the second pattern
424 	 * @return the combination of the two patterns
425 	 * @throws IllegalArgumentException when the two patterns cannot be combined
426 	 */
427 	@Override
428 	public String combine(String pattern1, String pattern2) {
429 		if (!StringUtils.hasText(pattern1) && !StringUtils.hasText(pattern2)) {
430 			return "";
431 		}
432 		if (!StringUtils.hasText(pattern1)) {
433 			return pattern2;
434 		}
435 		if (!StringUtils.hasText(pattern2)) {
436 			return pattern1;
437 		}
438 
439 		boolean pattern1ContainsUriVar = pattern1.indexOf('{') != -1;
440 		if (!pattern1.equals(pattern2) && !pattern1ContainsUriVar && match(pattern1, pattern2)) {
441 			// /* + /hotel -> /hotel ; "/*.*" + "/*.html" -> /*.html
442 			// However /user + /user -> /usr/user ; /{foo} + /bar -> /{foo}/bar
443 			return pattern2;
444 		}
445 
446 		// /hotels/* + /booking -> /hotels/booking
447 		// /hotels/* + booking -> /hotels/booking
448 		if (pattern1.endsWith(this.pathSeparatorPatternCache.getEndsOnWildCard())) {
449 			return concat(pattern1.substring(0, pattern1.length() - 2), pattern2);
450 		}
451 
452 		// /hotels/** + /booking -> /hotels/**/booking
453 		// /hotels/** + booking -> /hotels/**/booking
454 		if (pattern1.endsWith(this.pathSeparatorPatternCache.getEndsOnDoubleWildCard())) {
455 			return concat(pattern1, pattern2);
456 		}
457 
458 		int starDotPos1 = pattern1.indexOf("*.");
459 		if (pattern1ContainsUriVar || starDotPos1 == -1 || this.pathSeparator.equals(".")) {
460 			// simply concatenate the two patterns
461 			return concat(pattern1, pattern2);
462 		}
463 		String extension1 = pattern1.substring(starDotPos1 + 1);
464 		int dotPos2 = pattern2.indexOf('.');
465 		String fileName2 = (dotPos2 == -1 ? pattern2 : pattern2.substring(0, dotPos2));
466 		String extension2 = (dotPos2 == -1 ? "" : pattern2.substring(dotPos2));
467 		String extension = extension1.startsWith("*") ? extension2 : extension1;
468 		return fileName2 + extension;
469 	}
470 
471 	private String concat(String path1, String path2) {
472 		if (path1.endsWith(this.pathSeparator) || path2.startsWith(this.pathSeparator)) {
473 			return path1 + path2;
474 		}
475 		return path1 + this.pathSeparator + path2;
476 	}

479 	 * Given a full path, returns a {@link Comparator} suitable for sorting patterns in order of explicitness.
480 	 * <p>The returned {@code Comparator} will {@linkplain java.util.Collections#sort(java.util.List,
481 	 * java.util.Comparator) sort} a list so that more specific patterns (without uri templates or wild cards) come before
482 	 * generic patterns. So given a list with the following patterns: <ol> <li>{@code /hotels/new}</li>
483 	 * <li>{@code /hotels/{hotel}}</li> <li>{@code /hotels/*}</li> </ol> the returned comparator will sort this
484 	 * list so that the order will be as indicated.
485 	 * <p>The full path given as parameter is used to test for exact matches. So when the given path is {@code /hotels/2},
486 	 * the pattern {@code /hotels/2} will be sorted before {@code /hotels/1}.
487 	 * @param path the full path to use for comparison
488 	 * @return a comparator capable of sorting patterns in order of explicitness
489 	 */
490 	@Override
491 	public Comparator<String> getPatternComparator(String path) {
492 		return new AntPatternComparator(path);
493 	}

496 	/**
497 	 * Tests whether or not a string matches against a pattern via a {@link Pattern}.
498 	 * <p>The pattern may contain special characters: '*' means zero or more characters; '?' means one and
499 	 * only one character; '{' and '}' indicate a URI template pattern. For example <tt>/users/{user}</tt>.
500 	 */
501 	protected static class AntPathStringMatcher {
502 
503 		private static final Pattern GLOB_PATTERN = Pattern.compile("\\?|\\*|\\{((?:\\{[^/]+?\\}|[^/{}]|\\\\[{}])+?)\\}");
504 
505 		private static final String DEFAULT_VARIABLE_PATTERN = "(.*)";
506 
507 		private final Pattern pattern;
508 
509 		private final List<String> variableNames = new LinkedList<String>();
510 
511 		public AntPathStringMatcher(String pattern) {
512 			StringBuilder patternBuilder = new StringBuilder();
513 			Matcher m = GLOB_PATTERN.matcher(pattern);
514 			int end = 0;
515 			while (m.find()) {
516 				patternBuilder.append(quote(pattern, end, m.start()));
517 				String match = m.group();
518 				if ("?".equals(match)) {
519 					patternBuilder.append('.');
520 				}
521 				else if ("*".equals(match)) {
522 					patternBuilder.append(".*");
523 				}
524 				else if (match.startsWith("{") && match.endsWith("}")) {
525 					int colonIdx = match.indexOf(':');
526 					if (colonIdx == -1) {
527 						patternBuilder.append(DEFAULT_VARIABLE_PATTERN);
528 						this.variableNames.add(m.group(1));
529 					}
530 					else {
531 						String variablePattern = match.substring(colonIdx + 1, match.length() - 1);
532 						patternBuilder.append('(');
533 						patternBuilder.append(variablePattern);
534 						patternBuilder.append(')');
535 						String variableName = match.substring(1, colonIdx);
536 						this.variableNames.add(variableName);
537 					}
538 				}
539 				end = m.end();
540 			}
541 			patternBuilder.append(quote(pattern, end, pattern.length()));
542 			this.pattern = Pattern.compile(patternBuilder.toString());
543 		}
544 
545 		private String quote(String s, int start, int end) {
546 			if (start == end) {
547 				return "";
548 			}
549 			return Pattern.quote(s.substring(start, end));
550 		}

552 		/**
553 		 * Main entry point.
554 		 * @return {@code true} if the string matches against the pattern, or {@code false} otherwise.
555 		 */
556 		public boolean matchStrings(String str, Map<String, String> uriTemplateVariables) {
557 			Matcher matcher = this.pattern.matcher(str);
558 			if (matcher.matches()) {
559 				if (uriTemplateVariables != null) {
560 					// SPR-8455
561 					Assert.isTrue(this.variableNames.size() == matcher.groupCount(),
562 							"The number of capturing groups in the pattern segment " + this.pattern +
563 							" does not match the number of URI template variables it defines, which can occur if " +
564 							" capturing groups are used in a URI template regex. Use non-capturing groups instead.");
565 					for (int i = 1; i <= matcher.groupCount(); i++) {
566 						String name = this.variableNames.get(i - 1);
567 						String value = matcher.group(i);
568 						uriTemplateVariables.put(name, value);
569 					}
570 				}
571 				return true;
572 			}
573 			else {
574 				return false;
575 			}
576 		}
577 	}

580 	/**
581 	 * The default {@link Comparator} implementation returned by
582 	 * {@link #getPatternComparator(String)}.
583 	 * <p>In order, the most "generic" pattern is determined by the following:
584 	 * <ul>
585 	 * <li>if it's null or a capture all pattern (i.e. it is equal to "/**")</li>
586 	 * <li>if the other pattern is an actual match</li>
587 	 * <li>if it's a catch-all pattern (i.e. it ends with "**"</li>
588 	 * <li>if it's got more "*" than the other pattern</li>
589 	 * <li>if it's got more "{foo}" than the other pattern</li>
590 	 * <li>if it's shorter than the other pattern</li>
591 	 * </ul>
592 	 */
593 	protected static class AntPatternComparator implements Comparator<String> {
594 
595 		private final String path;
596 
597 		public AntPatternComparator(String path) {
598 			this.path = path;
599 		}
600 
601 		/**
602 		 * Compare two patterns to determine which should match first, i.e. which
603 		 * is the most specific regarding the current path.
604 		 * @return a negative integer, zero, or a positive integer as pattern1 is
605 		 * more specific, equally specific, or less specific than pattern2.
606 		 */
607 		@Override
608 		public int compare(String pattern1, String pattern2) {
609 			PatternInfo info1 = new PatternInfo(pattern1);
610 			PatternInfo info2 = new PatternInfo(pattern2);
611 
612 			if (info1.isLeastSpecific() && info2.isLeastSpecific()) {
613 				return 0;
614 			}
615 			else if (info1.isLeastSpecific()) {
616 				return 1;
617 			}
618 			else if (info2.isLeastSpecific()) {
619 				return -1;
620 			}
621 
622 			boolean pattern1EqualsPath = pattern1.equals(path);
623 			boolean pattern2EqualsPath = pattern2.equals(path);
624 			if (pattern1EqualsPath && pattern2EqualsPath) {
625 				return 0;
626 			}
627 			else if (pattern1EqualsPath) {
628 				return -1;
629 			}
630 			else if (pattern2EqualsPath) {
631 				return 1;
632 			}
633 
634 			if (info1.isPrefixPattern() && info2.getDoubleWildcards() == 0) {
635 				return 1;
636 			}
637 			else if (info2.isPrefixPattern() && info1.getDoubleWildcards() == 0) {
638 				return -1;
639 			}
640 
641 			if (info1.getTotalCount() != info2.getTotalCount()) {
642 				return info1.getTotalCount() - info2.getTotalCount();
643 			}
644 
645 			if (info1.getLength() != info2.getLength()) {
646 				return info2.getLength() - info1.getLength();
647 			}
648 
649 			if (info1.getSingleWildcards() < info2.getSingleWildcards()) {
650 				return -1;
651 			}
652 			else if (info2.getSingleWildcards() < info1.getSingleWildcards()) {
653 				return 1;
654 			}
655 
656 			if (info1.getUriVars() < info2.getUriVars()) {
657 				return -1;
658 			}
659 			else if (info2.getUriVars() < info1.getUriVars()) {
660 				return 1;
661 			}
662 
663 			return 0;
664 		}

667 		/**
668 		 * Value class that holds information about the pattern, e.g. number of
669 		 * occurrences of "*", "**", and "{" pattern elements.
670 		 */
671 		private static class PatternInfo {
672 
673 			private final String pattern;
674 
675 			private int uriVars;
676 
677 			private int singleWildcards;
678 
679 			private int doubleWildcards;
680 
681 			private boolean catchAllPattern;
682 
683 			private boolean prefixPattern;
684 
685 			private Integer length;
686 
687 			public PatternInfo(String pattern) {
688 				this.pattern = pattern;
689 				if (this.pattern != null) {
690 					initCounters();
691 					this.catchAllPattern = this.pattern.equals("/**");
692 					this.prefixPattern = !this.catchAllPattern && this.pattern.endsWith("/**");
693 				}
694 				if (this.uriVars == 0) {
695 					this.length = (this.pattern != null ? this.pattern.length() : 0);
696 				}
697 			}
698 
699 			protected void initCounters() {
700 				int pos = 0;
701 				while (pos < this.pattern.length()) {
702 					if (this.pattern.charAt(pos) == '{') {
703 						this.uriVars++;
704 						pos++;
705 					}
706 					else if (this.pattern.charAt(pos) == '*') {
707 						if (pos + 1 < this.pattern.length() && this.pattern.charAt(pos + 1) == '*') {
708 							this.doubleWildcards++;
709 							pos += 2;
710 						}
711 						else if (!this.pattern.substring(pos - 1).equals(".*")) {
712 							this.singleWildcards++;
713 							pos++;
714 						}
715 						else {
716 							pos++;
717 						}
718 					}
719 					else {
720 						pos++;
721 					}
722 				}
723 			}
724 
725 			public int getUriVars() {
726 				return this.uriVars;
727 			}
728 
729 			public int getSingleWildcards() {
730 				return this.singleWildcards;
731 			}
732 
733 			public int getDoubleWildcards() {
734 				return this.doubleWildcards;
735 			}
736 
737 			public boolean isLeastSpecific() {
738 				return (this.pattern == null || this.catchAllPattern);
739 			}
740 
741 			public boolean isPrefixPattern() {
742 				return this.prefixPattern;
743 			}
744 
745 			public int getTotalCount() {
746 				return this.uriVars + this.singleWildcards + (2 * this.doubleWildcards);
747 			}
748 
749 			/**
750 			 * Returns the length of the given pattern, where template variables are considered to be 1 long.
751 			 */
752 			public int getLength() {
753 				if (this.length == null) {
754 					this.length = VARIABLE_PATTERN.matcher(this.pattern).replaceAll("#").length();
755 				}
756 				return this.length;
757 			}
758 		}
759 	}
