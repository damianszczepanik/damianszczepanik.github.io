42  	@Test
43  	public void testIsEmpty() {
44  		assertTrue(CollectionUtils.isEmpty((Set<Object>) null));
45  		assertTrue(CollectionUtils.isEmpty((Map<String, String>) null));
46  		assertTrue(CollectionUtils.isEmpty(new HashMap<String, String>()));
47  		assertTrue(CollectionUtils.isEmpty(new HashSet<Object>()));
48  
49  		List<Object> list = new LinkedList<Object>();
50  		list.add(new Object());
51  		assertFalse(CollectionUtils.isEmpty(list));
52  
53  		Map<String, String> map = new HashMap<String, String>();
54  		map.put("foo", "bar");
55  		assertFalse(CollectionUtils.isEmpty(map));
56  	}

83  	public void testMergePropertiesIntoMap() {
84  		Properties defaults = new Properties();
85  		defaults.setProperty("prop1", "value1");
86  		Properties props = new Properties(defaults);
87  		props.setProperty("prop2", "value2");
88  		props.put("prop3", new Integer(3));
89  
90  		Map<String, String> map = new HashMap<String, String>();
91  		map.put("prop4", "value4");
92  
93  		CollectionUtils.mergePropertiesIntoMap(props, map);
94  		assertEquals("value1", map.get("prop1"));
95  		assertEquals("value2", map.get("prop2"));
96  		assertEquals(new Integer(3), map.get("prop3"));
97  		assertEquals("value4", map.get("prop4"));
98  	}

100 	@Test
101 	public void testContains() {
102 		assertFalse(CollectionUtils.contains((Iterator<String>) null, "myElement"));
103 		assertFalse(CollectionUtils.contains((Enumeration<String>) null, "myElement"));
104 		assertFalse(CollectionUtils.contains(new LinkedList<String>().iterator(), "myElement"));
105 		assertFalse(CollectionUtils.contains(new Hashtable<String, Object>().keys(), "myElement"));
106 
107 		List<String> list = new LinkedList<String>();
108 		list.add("myElement");
109 		assertTrue(CollectionUtils.contains(list.iterator(), "myElement"));
110 
111 		Hashtable<String, String> ht = new Hashtable<String, String>();
112 		ht.put("myElement", "myValue");
113 		assertTrue(CollectionUtils.contains(ht.keys(), "myElement"));
114 	}

218 	private static final class Instance {
219 
220 		private final String name;
221 
222 		public Instance(String name) {
223 			this.name = name;
224 		}
225 
226 		@Override
227 		public boolean equals(Object rhs) {
228 			if (this == rhs) {
229 				return true;
230 			}
231 			if (rhs == null || this.getClass() != rhs.getClass()) {
232 				return false;
233 			}
234 			Instance instance = (Instance) rhs;
235 			return this.name.equals(instance.name);
236 		}
237 
238 		@Override
239 		public int hashCode() {
240 			return this.name.hashCode();
241 		}
242 	}
