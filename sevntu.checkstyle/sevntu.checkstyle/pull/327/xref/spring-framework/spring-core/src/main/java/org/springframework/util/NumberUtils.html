41  	private static final BigInteger LONG_MAX = BigInteger.valueOf(Long.MAX_VALUE);

43  	/**
44  	 * Standard number types (all immutable):
45  	 * Byte, Short, Integer, Long, BigInteger, Float, Double, BigDecimal.
46  	 */
47  	public static final Set<Class<?>> STANDARD_NUMBER_TYPES;
48  
49  	static {
50  		Set<Class<?>> numberTypes = new HashSet<Class<?>>(8);
51  		numberTypes.add(Byte.class);
52  		numberTypes.add(Short.class);
53  		numberTypes.add(Integer.class);
54  		numberTypes.add(Long.class);
55  		numberTypes.add(BigInteger.class);
56  		numberTypes.add(Float.class);
57  		numberTypes.add(Double.class);
58  		numberTypes.add(BigDecimal.class);
59  		STANDARD_NUMBER_TYPES = Collections.unmodifiableSet(numberTypes);
60  	}

63  	/**
64  	 * Convert the given number into an instance of the given target class.
65  	 * @param number the number to convert
66  	 * @param targetClass the target class to convert to
67  	 * @return the converted number
68  	 * @throws IllegalArgumentException if the target class is not supported
69  	 * (i.e. not a standard Number subclass as included in the JDK)
70  	 * @see java.lang.Byte
71  	 * @see java.lang.Short
72  	 * @see java.lang.Integer
73  	 * @see java.lang.Long
74  	 * @see java.math.BigInteger
75  	 * @see java.lang.Float
76  	 * @see java.lang.Double
77  	 * @see java.math.BigDecimal
78  	 */
79  	@SuppressWarnings("unchecked")
80  	public static <T extends Number> T convertNumberToTargetClass(Number number, Class<T> targetClass)
81  			throws IllegalArgumentException {
82  
83  		Assert.notNull(number, "Number must not be null");
84  		Assert.notNull(targetClass, "Target class must not be null");
85  
86  		if (targetClass.isInstance(number)) {
87  			return (T) number;
88  		}
89  		else if (targetClass.equals(Byte.class)) {
90  			long value = number.longValue();
91  			if (value < Byte.MIN_VALUE || value > Byte.MAX_VALUE) {
92  				raiseOverflowException(number, targetClass);
93  			}
94  			return (T) new Byte(number.byteValue());
95  		}
96  		else if (targetClass.equals(Short.class)) {
97  			long value = number.longValue();
98  			if (value < Short.MIN_VALUE || value > Short.MAX_VALUE) {
99  				raiseOverflowException(number, targetClass);
100 			}
101 			return (T) new Short(number.shortValue());
102 		}
103 		else if (targetClass.equals(Integer.class)) {
104 			long value = number.longValue();
105 			if (value < Integer.MIN_VALUE || value > Integer.MAX_VALUE) {
106 				raiseOverflowException(number, targetClass);
107 			}
108 			return (T) new Integer(number.intValue());
109 		}
110 		else if (targetClass.equals(Long.class)) {
111 			BigInteger bigInt = null;
112 			if (number instanceof BigInteger) {
113 				bigInt = (BigInteger) number;
114 			}
115 			else if (number instanceof BigDecimal) {
116 				bigInt = ((BigDecimal) number).toBigInteger();
117 			}
118 			// Effectively analogous to JDK 8's BigInteger.longValueExact()
119 			if (bigInt != null && (bigInt.compareTo(LONG_MIN) < 0 || bigInt.compareTo(LONG_MAX) > 0)) {
120 				raiseOverflowException(number, targetClass);
121 			}
122 			return (T) new Long(number.longValue());
123 		}
124 		else if (targetClass.equals(BigInteger.class)) {
125 			if (number instanceof BigDecimal) {
126 				// do not lose precision - use BigDecimal's own conversion
127 				return (T) ((BigDecimal) number).toBigInteger();
128 			}
129 			else {
130 				// original value is not a Big* number - use standard long conversion
131 				return (T) BigInteger.valueOf(number.longValue());
132 			}
133 		}
134 		else if (targetClass.equals(Float.class)) {
135 			return (T) new Float(number.floatValue());
136 		}
137 		else if (targetClass.equals(Double.class)) {
138 			return (T) new Double(number.doubleValue());
139 		}
140 		else if (targetClass.equals(BigDecimal.class)) {
141 			// always use BigDecimal(String) here to avoid unpredictability of BigDecimal(double)
142 			// (see BigDecimal javadoc for details)
143 			return (T) new BigDecimal(number.toString());
144 		}
145 		else {
146 			throw new IllegalArgumentException("Could not convert number [" + number + "] of type [" +
147 					number.getClass().getName() + "] to unknown target class [" + targetClass.getName() + "]");
148 		}
149 	}

151 	/**
152 	 * Raise an overflow exception for the given number and target class.
153 	 * @param number the number we tried to convert
154 	 * @param targetClass the target class we tried to convert to
155 	 */
156 	private static void raiseOverflowException(Number number, Class<?> targetClass) {
157 		throw new IllegalArgumentException("Could not convert number [" + number + "] of type [" +
158 				number.getClass().getName() + "] to target class [" + targetClass.getName() + "]: overflow");
159 	}

161 	/**
162 	 * Parse the given text into a number instance of the given target class,
163 	 * using the corresponding {@code decode} / {@code valueOf} methods.
164 	 * <p>Trims the input {@code String} before attempting to parse the number.
165 	 * Supports numbers in hex format (with leading "0x", "0X" or "#") as well.
166 	 * @param text the text to convert
167 	 * @param targetClass the target class to parse into
168 	 * @return the parsed number
169 	 * @throws IllegalArgumentException if the target class is not supported
170 	 * (i.e. not a standard Number subclass as included in the JDK)
171 	 * @see Byte#decode
172 	 * @see Short#decode
173 	 * @see Integer#decode
174 	 * @see Long#decode
175 	 * @see #decodeBigInteger(String)
176 	 * @see Float#valueOf
177 	 * @see Double#valueOf
178 	 * @see java.math.BigDecimal#BigDecimal(String)
179 	 */
180 	@SuppressWarnings("unchecked")
181 	public static <T extends Number> T parseNumber(String text, Class<T> targetClass) {
182 		Assert.notNull(text, "Text must not be null");
183 		Assert.notNull(targetClass, "Target class must not be null");
184 		String trimmed = StringUtils.trimAllWhitespace(text);
185 
186 		if (targetClass.equals(Byte.class)) {
187 			return (T) (isHexNumber(trimmed) ? Byte.decode(trimmed) : Byte.valueOf(trimmed));
188 		}
189 		else if (targetClass.equals(Short.class)) {
190 			return (T) (isHexNumber(trimmed) ? Short.decode(trimmed) : Short.valueOf(trimmed));
191 		}
192 		else if (targetClass.equals(Integer.class)) {
193 			return (T) (isHexNumber(trimmed) ? Integer.decode(trimmed) : Integer.valueOf(trimmed));
194 		}
195 		else if (targetClass.equals(Long.class)) {
196 			return (T) (isHexNumber(trimmed) ? Long.decode(trimmed) : Long.valueOf(trimmed));
197 		}
198 		else if (targetClass.equals(BigInteger.class)) {
199 			return (T) (isHexNumber(trimmed) ? decodeBigInteger(trimmed) : new BigInteger(trimmed));
200 		}
201 		else if (targetClass.equals(Float.class)) {
202 			return (T) Float.valueOf(trimmed);
203 		}
204 		else if (targetClass.equals(Double.class)) {
205 			return (T) Double.valueOf(trimmed);
206 		}
207 		else if (targetClass.equals(BigDecimal.class) || targetClass.equals(Number.class)) {
208 			return (T) new BigDecimal(trimmed);
209 		}
210 		else {
211 			throw new IllegalArgumentException(
212 					"Cannot convert String [" + text + "] to target class [" + targetClass.getName() + "]");
213 		}
214 	}

216 	/**
217 	 * Parse the given text into a number instance of the given target class,
218 	 * using the given NumberFormat. Trims the input {@code String}
219 	 * before attempting to parse the number.
220 	 * @param text the text to convert
221 	 * @param targetClass the target class to parse into
222 	 * @param numberFormat the NumberFormat to use for parsing (if {@code null},
223 	 * this method falls back to {@code parseNumber(String, Class)})
224 	 * @return the parsed number
225 	 * @throws IllegalArgumentException if the target class is not supported
226 	 * (i.e. not a standard Number subclass as included in the JDK)
227 	 * @see java.text.NumberFormat#parse
228 	 * @see #convertNumberToTargetClass
229 	 * @see #parseNumber(String, Class)
230 	 */
231 	public static <T extends Number> T parseNumber(String text, Class<T> targetClass, NumberFormat numberFormat) {
232 		if (numberFormat != null) {
233 			Assert.notNull(text, "Text must not be null");
234 			Assert.notNull(targetClass, "Target class must not be null");
235 			DecimalFormat decimalFormat = null;
236 			boolean resetBigDecimal = false;
237 			if (numberFormat instanceof DecimalFormat) {
238 				decimalFormat = (DecimalFormat) numberFormat;
239 				if (BigDecimal.class.equals(targetClass) && !decimalFormat.isParseBigDecimal()) {
240 					decimalFormat.setParseBigDecimal(true);
241 					resetBigDecimal = true;
242 				}
243 			}
244 			try {
245 				Number number = numberFormat.parse(StringUtils.trimAllWhitespace(text));
246 				return convertNumberToTargetClass(number, targetClass);
247 			}
248 			catch (ParseException ex) {
249 				throw new IllegalArgumentException("Could not parse number: " + ex.getMessage());
250 			}
251 			finally {
252 				if (resetBigDecimal) {
253 					decimalFormat.setParseBigDecimal(false);
254 				}
255 			}
256 		}
257 		else {
258 			return parseNumber(text, targetClass);
259 		}
260 	}

262 	/**
263 	 * Determine whether the given value String indicates a hex number, i.e. needs to be
264 	 * passed into {@code Integer.decode} instead of {@code Integer.valueOf} (etc).
265 	 */
266 	private static boolean isHexNumber(String value) {
267 		int index = (value.startsWith("-") ? 1 : 0);
268 		return (value.startsWith("0x", index) || value.startsWith("0X", index) || value.startsWith("#", index));
269 	}

271 	/**
272 	 * Decode a {@link java.math.BigInteger} from a {@link String} value.
273 	 * Supports decimal, hex and octal notation.
274 	 * @see BigInteger#BigInteger(String, int)
275 	 */
276 	private static BigInteger decodeBigInteger(String value) {
277 		int radix = 10;
278 		int index = 0;
279 		boolean negative = false;
280 
281 		// Handle minus sign, if present.
282 		if (value.startsWith("-")) {
283 			negative = true;
284 			index++;
285 		}
286 
287 		// Handle radix specifier, if present.
288 		if (value.startsWith("0x", index) || value.startsWith("0X", index)) {
289 			index += 2;
290 			radix = 16;
291 		}
292 		else if (value.startsWith("#", index)) {
293 			index++;
294 			radix = 16;
295 		}
296 		else if (value.startsWith("0", index) && value.length() > 1 + index) {
297 			index++;
298 			radix = 8;
299 		}
300 
301 		BigInteger result = new BigInteger(value.substring(index), radix);
302 		return (negative ? result.negate() : result);
303 	}
