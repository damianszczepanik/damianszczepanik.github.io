17  package org.springframework.core;
18  
19  import java.util.ArrayList;
20  import java.util.HashSet;
21  import java.util.List;
22  import java.util.Set;
23  
24  import junit.framework.TestCase;
25  
26  import org.springframework.tests.sample.objects.TestObject;
27  
28  /**
29   * @author Rob Harrop
30   */
31  public class ConventionsTests extends TestCase {
32  
33  	public void testSimpleObject() {
34  		TestObject testObject = new TestObject();
35  		assertEquals("Incorrect singular variable name", "testObject", Conventions.getVariableName(testObject));
36  	}
37  
38  	public void testArray() {
39  		TestObject[] testObjects = new TestObject[0];
40  		assertEquals("Incorrect plural array form", "testObjectList", Conventions.getVariableName(testObjects));
41  	}
42  
43  	public void testCollections() {
44  		List<TestObject> list = new ArrayList<TestObject>();
45  		list.add(new TestObject());
46  		assertEquals("Incorrect plural List form", "testObjectList", Conventions.getVariableName(list));
47  
48  		Set<TestObject> set = new HashSet<TestObject>();
49  		set.add(new TestObject());
50  		assertEquals("Incorrect plural Set form", "testObjectList", Conventions.getVariableName(set));
51  
52  		List<?> emptyList = new ArrayList<Object>();
53  		try {
54  			Conventions.getVariableName(emptyList);
55  			fail("Should not be able to generate name for empty collection");
56  		}
57  		catch(IllegalArgumentException ex) {
58  			// success
59  		}
60  	}
61  
62  	public void testAttributeNameToPropertyName() throws Exception {
63  		assertEquals("transactionManager", Conventions.attributeNameToPropertyName("transaction-manager"));
64  		assertEquals("pointcutRef", Conventions.attributeNameToPropertyName("pointcut-ref"));
65  		assertEquals("lookupOnStartup", Conventions.attributeNameToPropertyName("lookup-on-startup"));
66  	}
67  
68  	public void testGetQualifiedAttributeName() throws Exception {
69  		String baseName = "foo";
70  		Class<String> cls = String.class;
71  		String desiredResult = "java.lang.String.foo";
72  		assertEquals(desiredResult, Conventions.getQualifiedAttributeName(cls, baseName));
73  	}
74  }
