17  package org.springframework.util;
18  
19  import java.io.File;
20  import java.io.IOException;
21  
22  /**
23   * Utility methods for working with the file system.
24   *
25   * @author Rob Harrop
26   * @author Juergen Hoeller
27   * @since 2.5.3
28   */
29  public abstract class FileSystemUtils {
30  
31  	/**
32  	 * Delete the supplied {@link File} - for directories,
33  	 * recursively delete any nested directories or files as well.
34  	 * @param root the root {@code File} to delete
35  	 * @return {@code true} if the {@code File} was deleted,
36  	 * otherwise {@code false}
37  	 */
38  	public static boolean deleteRecursively(File root) {
39  		if (root != null && root.exists()) {
40  			if (root.isDirectory()) {
41  				File[] children = root.listFiles();
42  				if (children != null) {
43  					for (File child : children) {
44  						deleteRecursively(child);
45  					}
46  				}
47  			}
48  			return root.delete();
49  		}
50  		return false;
51  	}
52  
53  	/**
54  	 * Recursively copy the contents of the {@code src} file/directory
55  	 * to the {@code dest} file/directory.
56  	 * @param src the source directory
57  	 * @param dest the destination directory
58  	 * @throws IOException in the case of I/O errors
59  	 */
60  	public static void copyRecursively(File src, File dest) throws IOException {
61  		Assert.isTrue(src != null && (src.isDirectory() || src.isFile()), "Source File must denote a directory or file");
62  		Assert.notNull(dest, "Destination File must not be null");
63  		doCopyRecursively(src, dest);
64  	}
65  
66  	/**
67  	 * Actually copy the contents of the {@code src} file/directory
68  	 * to the {@code dest} file/directory.
69  	 * @param src the source directory
70  	 * @param dest the destination directory
71  	 * @throws IOException in the case of I/O errors
72  	 */
73  	private static void doCopyRecursively(File src, File dest) throws IOException {
74  		if (src.isDirectory()) {
75  			dest.mkdir();
76  			File[] entries = src.listFiles();
77  			if (entries == null) {
78  				throw new IOException("Could not list files in directory: " + src);
79  			}
80  			for (File entry : entries) {
81  				doCopyRecursively(entry, new File(dest, entry.getName()));
82  			}
83  		}
84  		else if (src.isFile()) {
85  			try {
86  				dest.createNewFile();
87  			}
88  			catch (IOException ex) {
89  				IOException ioex = new IOException("Failed to create file: " + dest);
90  				ioex.initCause(ex);
91  				throw ioex;
92  			}
93  			FileCopyUtils.copy(src, dest);
94  		}
95  		else {
96  			// Special File handle: neither a file not a directory.
97  			// Simply skip it when contained in nested directory...
98  		}
99  	}
100 
101 }
