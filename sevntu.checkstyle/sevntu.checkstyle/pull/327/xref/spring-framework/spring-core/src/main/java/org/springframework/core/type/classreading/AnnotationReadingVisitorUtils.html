31  /**
32   * Internal utility class used when reading annotations.
33   *
34   * @author Juergen Hoeller
35   * @author Mark Fisher
36   * @author Costin Leau
37   * @author Phillip Webb
38   * @author Sam Brannen
39   * @since 4.0
40   */
41  abstract class AnnotationReadingVisitorUtils {
42  
43  	public static AnnotationAttributes convertClassValues(ClassLoader classLoader, AnnotationAttributes original,
44  			boolean classValuesAsString) {
45  
46  		if (original == null) {
47  			return null;
48  		}
49  
50  		AnnotationAttributes result = new AnnotationAttributes(original.size());
51  		for (Map.Entry<String, Object> entry : original.entrySet()) {
52  			try {
53  				Object value = entry.getValue();
54  				if (value instanceof AnnotationAttributes) {
55  					value = convertClassValues(classLoader, (AnnotationAttributes) value, classValuesAsString);
56  				}
57  				else if (value instanceof AnnotationAttributes[]) {
58  					AnnotationAttributes[] values = (AnnotationAttributes[]) value;
59  					for (int i = 0; i < values.length; i++) {
60  						values[i] = convertClassValues(classLoader, values[i], classValuesAsString);
61  					}
62  				}
63  				else if (value instanceof Type) {
64  					value = (classValuesAsString ? ((Type) value).getClassName()
65  							: classLoader.loadClass(((Type) value).getClassName()));
66  				}
67  				else if (value instanceof Type[]) {
68  					Type[] array = (Type[]) value;
69  					Object[] convArray = (classValuesAsString ? new String[array.length] : new Class<?>[array.length]);
70  					for (int i = 0; i < array.length; i++) {
71  						convArray[i] = (classValuesAsString ? array[i].getClassName()
72  								: classLoader.loadClass(array[i].getClassName()));
73  					}
74  					value = convArray;
75  				}
76  				else if (classValuesAsString) {
77  					if (value instanceof Class) {
78  						value = ((Class<?>) value).getName();
79  					}
80  					else if (value instanceof Class[]) {
81  						Class<?>[] clazzArray = (Class[]) value;
82  						String[] newValue = new String[clazzArray.length];
83  						for (int i = 0; i < clazzArray.length; i++) {
84  							newValue[i] = clazzArray[i].getName();
85  						}
86  						value = newValue;
87  					}
88  				}
89  				result.put(entry.getKey(), value);
90  			}
91  			catch (Exception ex) {
92  				// Class not found - can't resolve class reference in annotation
93  				// attribute.
94  			}
95  		}
96  		return result;
97  	}

99  	/**
100 	 * Retrieve the merged attributes of the annotation of the given type,
101 	 * if any, from the supplied {@code attributesMap}.
102 	 * <p>Annotation attribute values appearing <em>lower</em> in the annotation
103 	 * hierarchy (i.e., closer to the declaring class) will override those
104 	 * defined <em>higher</em> in the annotation hierarchy.
105 	 * @param attributesMap the map of annotation attribute lists,
106 	 * keyed by annotation type name
107 	 * @param metaAnnotationMap the map of meta annotation relationships,
108 	 * keyed by annotation type name
109 	 * @param annotationType the name of the annotation type to look for
110 	 * @return the merged annotation attributes, or {@code null} if no
111 	 * matching annotation is present in the {@code attributesMap}
112 	 * @since 4.0.3
113 	 */
114 	public static AnnotationAttributes getMergedAnnotationAttributes(
115 			LinkedMultiValueMap<String, AnnotationAttributes> attributesMap,
116 			Map<String, Set<String>> metaAnnotationMap, String annotationType) {
117 
118 		// Get the unmerged list of attributes for the target annotation.
119 		List<AnnotationAttributes> attributesList = attributesMap.get(annotationType);
120 		if (attributesList == null || attributesList.isEmpty()) {
121 			return null;
122 		}
123 
124 		// To start with, we populate the results with a copy of all attribute
125 		// values from the target annotation. A copy is necessary so that we do
126 		// not inadvertently mutate the state of the metadata passed to this
127 		// method.
128 		AnnotationAttributes results = new AnnotationAttributes(attributesList.get(0));
129 
130 		Set<String> overridableAttributeNames = new HashSet<String>(results.keySet());
131 		overridableAttributeNames.remove(AnnotationUtils.VALUE);
132 
133 		// Since the map is a LinkedMultiValueMap, we depend on the ordering of
134 		// elements in the map and reverse the order of the keys in order to traverse
135 		// "down" the annotation hierarchy.
136 		List<String> annotationTypes = new ArrayList<String>(attributesMap.keySet());
137 		Collections.reverse(annotationTypes);
138 
139 		// No need to revisit the target annotation type:
140 		annotationTypes.remove(annotationType);
141 
142 		for (String currentAnnotationType : annotationTypes) {
143 			List<AnnotationAttributes> currentAttributesList = attributesMap.get(currentAnnotationType);
144 			if (currentAttributesList != null && !currentAttributesList.isEmpty()) {
145 				Set<String> metaAnns = metaAnnotationMap.get(currentAnnotationType);
146 				if (metaAnns != null && metaAnns.contains(annotationType)) {
147 					AnnotationAttributes currentAttributes = currentAttributesList.get(0);
148 					for (String overridableAttributeName : overridableAttributeNames) {
149 						Object value = currentAttributes.get(overridableAttributeName);
150 						if (value != null) {
151 							// Store the value, potentially overriding a value from an
152 							// attribute of the same name found higher in the annotation
153 							// hierarchy.
154 							results.put(overridableAttributeName, value);
155 						}
156 					}
157 				}
158 			}
159 		}
160 
161 		return results;
162 	}
