35  /**
36   * A Java class parser to make a {@link ClassVisitor} visit an existing class.
37   * This class parses a byte array conforming to the Java class file format and
38   * calls the appropriate visit methods of a given class visitor for each field,
39   * method and bytecode instruction encountered.
40   * 
41   * @author Eric Bruneton
42   * @author Eugene Kuleshov
43   */
44  public class ClassReader {

156     /**
157      * Constructs a new {@link ClassReader} object.
158      * 
159      * @param b
160      *            the bytecode of the class to be read.
161      * @param off
162      *            the start offset of the class data.
163      * @param len
164      *            the length of the class data.
165      */
166     public ClassReader(final byte[] b, final int off, final int len) {
167         this.b = b;
168         // checks the class version
169 		/* SPRING PATCH: REMOVED FOR FORWARD COMPATIBILITY WITH JDK 9
170         if (readShort(off + 6) > Opcodes.V1_8) {
171             throw new IllegalArgumentException();
172         }
173 		*/
174         // parses the constant pool
175         items = new int[readUnsignedShort(off + 8)];
176         int n = items.length;
177         strings = new String[n];
178         int max = 0;
179         int index = off + 10;
180         for (int i = 1; i < n; ++i) {
181             items[i] = index + 1;
182             int size;
183             switch (b[index]) {
184             case ClassWriter.FIELD:
185             case ClassWriter.METH:
186             case ClassWriter.IMETH:
187             case ClassWriter.INT:
188             case ClassWriter.FLOAT:
189             case ClassWriter.NAME_TYPE:
190             case ClassWriter.INDY:
191                 size = 5;
192                 break;
193             case ClassWriter.LONG:
194             case ClassWriter.DOUBLE:
195                 size = 9;
196                 ++i;
197                 break;
198             case ClassWriter.UTF8:
199                 size = 3 + readUnsignedShort(index + 1);
200                 if (size > max) {
201                     max = size;
202                 }
203                 break;
204             case ClassWriter.HANDLE:
205                 size = 4;
206                 break;
207             // case ClassWriter.CLASS:
208             // case ClassWriter.STR:
209             // case ClassWriter.MTYPE
210             default:
211                 size = 3;
212                 break;
213             }
214             index += size;
215         }
216         maxStringLength = max;
217         // the class header information starts just after the constant pool
218         header = index;
219     }

260     /**

262      * {@link Type#getInternalName() getInternalName}).
263      *

266      * 
267      * @see ClassVisitor#visit(int, int, String, String, String, String[])

269     public String[] getInterfaces() {
270         int index = header + 6;
271         int n = readUnsignedShort(index);
272         String[] interfaces = new String[n];
273         if (n > 0) {
274             char[] buf = new char[maxStringLength];
275             for (int i = 0; i < n; ++i) {
276                 index += 2;
277                 interfaces[i] = readClass(index, buf);
278             }
279         }
280         return interfaces;
281     }

283     /**
284      * Copies the constant pool data into the given {@link ClassWriter}. Should
285      * be called before the {@link #accept(ClassVisitor,int)} method.
286      * 
287      * @param classWriter
288      *            the {@link ClassWriter} to copy constant pool into.
289      */
290     void copyPool(final ClassWriter classWriter) {
291         char[] buf = new char[maxStringLength];
292         int ll = items.length;
293         Item[] items2 = new Item[ll];
294         for (int i = 1; i < ll; i++) {
295             int index = items[i];
296             int tag = b[index - 1];
297             Item item = new Item(i);
298             int nameType;
299             switch (tag) {
300             case ClassWriter.FIELD:
301             case ClassWriter.METH:
302             case ClassWriter.IMETH:
303                 nameType = items[readUnsignedShort(index + 2)];
304                 item.set(tag, readClass(index, buf), readUTF8(nameType, buf),
305                         readUTF8(nameType + 2, buf));
306                 break;
307             case ClassWriter.INT:
308                 item.set(readInt(index));
309                 break;
310             case ClassWriter.FLOAT:
311                 item.set(Float.intBitsToFloat(readInt(index)));
312                 break;
313             case ClassWriter.NAME_TYPE:
314                 item.set(tag, readUTF8(index, buf), readUTF8(index + 2, buf),
315                         null);
316                 break;
317             case ClassWriter.LONG:
318                 item.set(readLong(index));
319                 ++i;
320                 break;
321             case ClassWriter.DOUBLE:
322                 item.set(Double.longBitsToDouble(readLong(index)));
323                 ++i;
324                 break;
325             case ClassWriter.UTF8: {
326                 String s = strings[i];
327                 if (s == null) {
328                     index = items[i];
329                     s = strings[i] = readUTF(index + 2,
330                             readUnsignedShort(index), buf);
331                 }
332                 item.set(tag, s, null, null);
333                 break;
334             }
335             case ClassWriter.HANDLE: {
336                 int fieldOrMethodRef = items[readUnsignedShort(index + 1)];
337                 nameType = items[readUnsignedShort(fieldOrMethodRef + 2)];
338                 item.set(ClassWriter.HANDLE_BASE + readByte(index),
339                         readClass(fieldOrMethodRef, buf),
340                         readUTF8(nameType, buf), readUTF8(nameType + 2, buf));
341                 break;
342             }
343             case ClassWriter.INDY:
344                 if (classWriter.bootstrapMethods == null) {
345                     copyBootstrapMethods(classWriter, items2, buf);
346                 }
347                 nameType = items[readUnsignedShort(index + 2)];
348                 item.set(readUTF8(nameType, buf), readUTF8(nameType + 2, buf),
349                         readUnsignedShort(index));
350                 break;
351             // case ClassWriter.STR:
352             // case ClassWriter.CLASS:
353             // case ClassWriter.MTYPE
354             default:
355                 item.set(tag, readUTF8(index, buf), null, null);
356                 break;
357             }
358 
359             int index2 = item.hashCode % items2.length;
360             item.next = items2[index2];
361             items2[index2] = item;
362         }
363 
364         int off = items[1] - 1;
365         classWriter.pool.putByteArray(b, off, header - off);
366         classWriter.items = items2;
367         classWriter.threshold = (int) (0.75d * ll);
368         classWriter.index = ll;
369     }

371     /**
372      * Copies the bootstrap method data into the given {@link ClassWriter}.
373      * Should be called before the {@link #accept(ClassVisitor,int)} method.
374      * 
375      * @param classWriter
376      *            the {@link ClassWriter} to copy bootstrap methods into.
377      */
378     private void copyBootstrapMethods(final ClassWriter classWriter,
379             final Item[] items, final char[] c) {
380         // finds the "BootstrapMethods" attribute
381         int u = getAttributes();
382         boolean found = false;
383         for (int i = readUnsignedShort(u); i > 0; --i) {
384             String attrName = readUTF8(u + 2, c);
385             if ("BootstrapMethods".equals(attrName)) {
386                 found = true;
387                 break;
388             }
389             u += 6 + readInt(u + 4);
390         }
391         if (!found) {
392             return;
393         }
394         // copies the bootstrap methods in the class writer
395         int boostrapMethodCount = readUnsignedShort(u + 8);
396         for (int j = 0, v = u + 10; j < boostrapMethodCount; j++) {
397             int position = v - u - 10;
398             int hashCode = readConst(readUnsignedShort(v), c).hashCode();
399             for (int k = readUnsignedShort(v + 2); k > 0; --k) {
400                 hashCode ^= readConst(readUnsignedShort(v + 4), c).hashCode();
401                 v += 2;
402             }
403             v += 4;
404             Item item = new Item(j);
405             item.set(position, hashCode & 0x7FFFFFFF);
406             int index = item.hashCode % items.length;
407             item.next = items[index];
408             items[index] = item;
409         }
410         int attrSize = readInt(u + 4);
411         ByteVector bootstrapMethods = new ByteVector(attrSize + 62);
412         bootstrapMethods.putByteArray(b, u + 10, attrSize - 2);
413         classWriter.bootstrapMethodsCount = boostrapMethodCount;
414         classWriter.bootstrapMethods = bootstrapMethods;
415     }

443     /**
444      * Reads the bytecode of a class.
445      * 
446      * @param is
447      *            an input stream from which to read the class.
448      * @param close
449      *            true to close the input stream after reading.
450      * @return the bytecode read from the given input stream.
451      * @throws IOException
452      *             if a problem occurs during reading.
453      */
454     private static byte[] readClass(final InputStream is, boolean close)
455             throws IOException {
456         if (is == null) {
457             throw new IOException("Class not found");
458         }
459         try {
460             byte[] b = new byte[is.available()];
461             int len = 0;
462             while (true) {
463                 int n = is.read(b, len, b.length - len);
464                 if (n == -1) {
465                     if (len < b.length) {
466                         byte[] c = new byte[len];
467                         System.arraycopy(b, 0, c, 0, len);
468                         b = c;
469                     }
470                     return b;
471                 }
472                 len += n;
473                 if (len == b.length) {
474                     int last = is.read();
475                     if (last < 0) {
476                         return b;
477                     }
478                     byte[] c = new byte[b.length + 1000];
479                     System.arraycopy(b, 0, c, 0, len);
480                     c[len++] = (byte) last;
481                     b = c;
482                 }
483             }
484         } finally {
485             if (close) {
486                 is.close();
487             }
488         }
489     }

511     /**
512      * Makes the given visitor visit the Java class of this {@link ClassReader}.
513      * This class is the one specified in the constructor (see
514      * {@link #ClassReader(byte[]) ClassReader}).
515      * 
516      * @param classVisitor
517      *            the visitor that must visit this class.
518      * @param attrs
519      *            prototypes of the attributes that must be parsed during the
520      *            visit of the class. Any attribute whose type is not equal to
521      *            the type of one the prototypes will not be parsed: its byte
522      *            array value will be passed unchanged to the ClassWriter.
523      *            <i>This may corrupt it if this value contains references to
524      *            the constant pool, or has syntactic or semantic links with a
525      *            class element that has been transformed by a class adapter
526      *            between the reader and the writer</i>.
527      * @param flags
528      *            option flags that can be used to modify the default behavior
529      *            of this class. See {@link #SKIP_DEBUG}, {@link #EXPAND_FRAMES}
530      *            , {@link #SKIP_FRAMES}, {@link #SKIP_CODE}.
531      */
532     public void accept(final ClassVisitor classVisitor,
533             final Attribute[] attrs, final int flags) {
534         int u = header; // current offset in the class file
535         char[] c = new char[maxStringLength]; // buffer used to read strings
536 
537         Context context = new Context();
538         context.attrs = attrs;
539         context.flags = flags;
540         context.buffer = c;
541 
542         // reads the class declaration
543         int access = readUnsignedShort(u);
544         String name = readClass(u + 2, c);
545         String superClass = readClass(u + 4, c);
546         String[] interfaces = new String[readUnsignedShort(u + 6)];
547         u += 8;
548         for (int i = 0; i < interfaces.length; ++i) {
549             interfaces[i] = readClass(u, c);
550             u += 2;
551         }
552 
553         // reads the class attributes
554         String signature = null;
555         String sourceFile = null;
556         String sourceDebug = null;
557         String enclosingOwner = null;
558         String enclosingName = null;
559         String enclosingDesc = null;
560         int anns = 0;
561         int ianns = 0;
562         int tanns = 0;
563         int itanns = 0;
564         int innerClasses = 0;
565         Attribute attributes = null;
566 
567         u = getAttributes();
568         for (int i = readUnsignedShort(u); i > 0; --i) {
569             String attrName = readUTF8(u + 2, c);
570             // tests are sorted in decreasing frequency order
571             // (based on frequencies observed on typical classes)
572             if ("SourceFile".equals(attrName)) {
573                 sourceFile = readUTF8(u + 8, c);
574             } else if ("InnerClasses".equals(attrName)) {
575                 innerClasses = u + 8;
576             } else if ("EnclosingMethod".equals(attrName)) {
577                 enclosingOwner = readClass(u + 8, c);
578                 int item = readUnsignedShort(u + 10);
579                 if (item != 0) {
580                     enclosingName = readUTF8(items[item], c);
581                     enclosingDesc = readUTF8(items[item] + 2, c);
582                 }
583             } else if (SIGNATURES && "Signature".equals(attrName)) {
584                 signature = readUTF8(u + 8, c);
585             } else if (ANNOTATIONS
586                     && "RuntimeVisibleAnnotations".equals(attrName)) {
587                 anns = u + 8;
588             } else if (ANNOTATIONS
589                     && "RuntimeVisibleTypeAnnotations".equals(attrName)) {
590                 tanns = u + 8;
591             } else if ("Deprecated".equals(attrName)) {
592                 access |= Opcodes.ACC_DEPRECATED;
593             } else if ("Synthetic".equals(attrName)) {
594                 access |= Opcodes.ACC_SYNTHETIC
595                         | ClassWriter.ACC_SYNTHETIC_ATTRIBUTE;
596             } else if ("SourceDebugExtension".equals(attrName)) {
597                 int len = readInt(u + 4);
598                 sourceDebug = readUTF(u + 8, len, new char[len]);
599             } else if (ANNOTATIONS
600                     && "RuntimeInvisibleAnnotations".equals(attrName)) {
601                 ianns = u + 8;
602             } else if (ANNOTATIONS
603                     && "RuntimeInvisibleTypeAnnotations".equals(attrName)) {
604                 itanns = u + 8;
605             } else if ("BootstrapMethods".equals(attrName)) {
606                 int[] bootstrapMethods = new int[readUnsignedShort(u + 8)];
607                 for (int j = 0, v = u + 10; j < bootstrapMethods.length; j++) {
608                     bootstrapMethods[j] = v;
609                     v += 2 + readUnsignedShort(v + 2) << 1;
610                 }
611                 context.bootstrapMethods = bootstrapMethods;
612             } else {
613                 Attribute attr = readAttribute(attrs, attrName, u + 8,
614                         readInt(u + 4), c, -1, null);
615                 if (attr != null) {
616                     attr.next = attributes;
617                     attributes = attr;
618                 }
619             }
620             u += 6 + readInt(u + 4);
621         }
622 
623         // visits the class declaration
624         classVisitor.visit(readInt(items[1] - 7), access, name, signature,
625                 superClass, interfaces);
626 
627         // visits the source and debug info
628         if ((flags & SKIP_DEBUG) == 0
629                 && (sourceFile != null || sourceDebug != null)) {
630             classVisitor.visitSource(sourceFile, sourceDebug);
631         }
632 
633         // visits the outer class
634         if (enclosingOwner != null) {
635             classVisitor.visitOuterClass(enclosingOwner, enclosingName,
636                     enclosingDesc);
637         }
638 
639         // visits the class annotations and type annotations
640         if (ANNOTATIONS && anns != 0) {
641             for (int i = readUnsignedShort(anns), v = anns + 2; i > 0; --i) {
642                 v = readAnnotationValues(v + 2, c, true,
643                         classVisitor.visitAnnotation(readUTF8(v, c), true));
644             }
645         }
646         if (ANNOTATIONS && ianns != 0) {
647             for (int i = readUnsignedShort(ianns), v = ianns + 2; i > 0; --i) {
648                 v = readAnnotationValues(v + 2, c, true,
649                         classVisitor.visitAnnotation(readUTF8(v, c), false));
650             }
651         }
652         if (ANNOTATIONS && tanns != 0) {
653             for (int i = readUnsignedShort(tanns), v = tanns + 2; i > 0; --i) {
654                 v = readAnnotationTarget(context, v);
655                 v = readAnnotationValues(v + 2, c, true,
656                         classVisitor.visitTypeAnnotation(context.typeRef,
657                                 context.typePath, readUTF8(v, c), true));
658             }
659         }
660         if (ANNOTATIONS && itanns != 0) {
661             for (int i = readUnsignedShort(itanns), v = itanns + 2; i > 0; --i) {
662                 v = readAnnotationTarget(context, v);
663                 v = readAnnotationValues(v + 2, c, true,
664                         classVisitor.visitTypeAnnotation(context.typeRef,
665                                 context.typePath, readUTF8(v, c), false));
666             }
667         }
668 
669         // visits the attributes
670         while (attributes != null) {
671             Attribute attr = attributes.next;
672             attributes.next = null;
673             classVisitor.visitAttribute(attributes);
674             attributes = attr;
675         }
676 
677         // visits the inner classes
678         if (innerClasses != 0) {
679             int v = innerClasses + 2;
680             for (int i = readUnsignedShort(innerClasses); i > 0; --i) {
681                 classVisitor.visitInnerClass(readClass(v, c),
682                         readClass(v + 2, c), readUTF8(v + 4, c),
683                         readUnsignedShort(v + 6));
684                 v += 8;
685             }
686         }
687 
688         // visits the fields and methods
689         u = header + 10 + 2 * interfaces.length;
690         for (int i = readUnsignedShort(u - 2); i > 0; --i) {
691             u = readField(classVisitor, context, u);
692         }
693         u += 2;
694         for (int i = readUnsignedShort(u - 2); i > 0; --i) {
695             u = readMethod(classVisitor, context, u);
696         }
697 
698         // visits the end of the class
699         classVisitor.visitEnd();
700     }

702     /**
703      * Reads a field and makes the given visitor visit it.
704      * 
705      * @param classVisitor
706      *            the visitor that must visit the field.
707      * @param context
708      *            information about the class being parsed.
709      * @param u
710      *            the start offset of the field in the class file.
711      * @return the offset of the first byte following the field in the class.
712      */
713     private int readField(final ClassVisitor classVisitor,
714             final Context context, int u) {
715         // reads the field declaration
716         char[] c = context.buffer;
717         int access = readUnsignedShort(u);
718         String name = readUTF8(u + 2, c);
719         String desc = readUTF8(u + 4, c);
720         u += 6;
721 
722         // reads the field attributes
723         String signature = null;
724         int anns = 0;
725         int ianns = 0;
726         int tanns = 0;
727         int itanns = 0;
728         Object value = null;
729         Attribute attributes = null;
730 
731         for (int i = readUnsignedShort(u); i > 0; --i) {
732             String attrName = readUTF8(u + 2, c);
733             // tests are sorted in decreasing frequency order
734             // (based on frequencies observed on typical classes)
735             if ("ConstantValue".equals(attrName)) {
736                 int item = readUnsignedShort(u + 8);
737                 value = item == 0 ? null : readConst(item, c);
738             } else if (SIGNATURES && "Signature".equals(attrName)) {
739                 signature = readUTF8(u + 8, c);
740             } else if ("Deprecated".equals(attrName)) {
741                 access |= Opcodes.ACC_DEPRECATED;
742             } else if ("Synthetic".equals(attrName)) {
743                 access |= Opcodes.ACC_SYNTHETIC
744                         | ClassWriter.ACC_SYNTHETIC_ATTRIBUTE;
745             } else if (ANNOTATIONS
746                     && "RuntimeVisibleAnnotations".equals(attrName)) {
747                 anns = u + 8;
748             } else if (ANNOTATIONS
749                     && "RuntimeVisibleTypeAnnotations".equals(attrName)) {
750                 tanns = u + 8;
751             } else if (ANNOTATIONS
752                     && "RuntimeInvisibleAnnotations".equals(attrName)) {
753                 ianns = u + 8;
754             } else if (ANNOTATIONS
755                     && "RuntimeInvisibleTypeAnnotations".equals(attrName)) {
756                 itanns = u + 8;
757             } else {
758                 Attribute attr = readAttribute(context.attrs, attrName, u + 8,
759                         readInt(u + 4), c, -1, null);
760                 if (attr != null) {
761                     attr.next = attributes;
762                     attributes = attr;
763                 }
764             }
765             u += 6 + readInt(u + 4);
766         }
767         u += 2;
768 
769         // visits the field declaration
770         FieldVisitor fv = classVisitor.visitField(access, name, desc,
771                 signature, value);
772         if (fv == null) {
773             return u;
774         }
775 
776         // visits the field annotations and type annotations
777         if (ANNOTATIONS && anns != 0) {
778             for (int i = readUnsignedShort(anns), v = anns + 2; i > 0; --i) {
779                 v = readAnnotationValues(v + 2, c, true,
780                         fv.visitAnnotation(readUTF8(v, c), true));
781             }
782         }
783         if (ANNOTATIONS && ianns != 0) {
784             for (int i = readUnsignedShort(ianns), v = ianns + 2; i > 0; --i) {
785                 v = readAnnotationValues(v + 2, c, true,
786                         fv.visitAnnotation(readUTF8(v, c), false));
787             }
788         }
789         if (ANNOTATIONS && tanns != 0) {
790             for (int i = readUnsignedShort(tanns), v = tanns + 2; i > 0; --i) {
791                 v = readAnnotationTarget(context, v);
792                 v = readAnnotationValues(v + 2, c, true,
793                         fv.visitTypeAnnotation(context.typeRef,
794                                 context.typePath, readUTF8(v, c), true));
795             }
796         }
797         if (ANNOTATIONS && itanns != 0) {
798             for (int i = readUnsignedShort(itanns), v = itanns + 2; i > 0; --i) {
799                 v = readAnnotationTarget(context, v);
800                 v = readAnnotationValues(v + 2, c, true,
801                         fv.visitTypeAnnotation(context.typeRef,
802                                 context.typePath, readUTF8(v, c), false));
803             }
804         }
805 
806         // visits the field attributes
807         while (attributes != null) {
808             Attribute attr = attributes.next;
809             attributes.next = null;
810             fv.visitAttribute(attributes);
811             attributes = attr;
812         }
813 
814         // visits the end of the field
815         fv.visitEnd();
816 
817         return u;
818     }

820     /**
821      * Reads a method and makes the given visitor visit it.
822      * 
823      * @param classVisitor
824      *            the visitor that must visit the method.
825      * @param context
826      *            information about the class being parsed.
827      * @param u
828      *            the start offset of the method in the class file.
829      * @return the offset of the first byte following the method in the class.
830      */
831     private int readMethod(final ClassVisitor classVisitor,
832             final Context context, int u) {
833         // reads the method declaration
834         char[] c = context.buffer;
835         context.access = readUnsignedShort(u);
836         context.name = readUTF8(u + 2, c);
837         context.desc = readUTF8(u + 4, c);
838         u += 6;
839 
840         // reads the method attributes
841         int code = 0;
842         int exception = 0;
843         String[] exceptions = null;
844         String signature = null;
845         int methodParameters = 0;
846         int anns = 0;
847         int ianns = 0;
848         int tanns = 0;
849         int itanns = 0;
850         int dann = 0;
851         int mpanns = 0;
852         int impanns = 0;
853         int firstAttribute = u;
854         Attribute attributes = null;
855 
856         for (int i = readUnsignedShort(u); i > 0; --i) {
857             String attrName = readUTF8(u + 2, c);
858             // tests are sorted in decreasing frequency order
859             // (based on frequencies observed on typical classes)
860             if ("Code".equals(attrName)) {
861                 if ((context.flags & SKIP_CODE) == 0) {
862                     code = u + 8;
863                 }
864             } else if ("Exceptions".equals(attrName)) {
865                 exceptions = new String[readUnsignedShort(u + 8)];
866                 exception = u + 10;
867                 for (int j = 0; j < exceptions.length; ++j) {
868                     exceptions[j] = readClass(exception, c);
869                     exception += 2;
870                 }
871             } else if (SIGNATURES && "Signature".equals(attrName)) {
872                 signature = readUTF8(u + 8, c);
873             } else if ("Deprecated".equals(attrName)) {
874                 context.access |= Opcodes.ACC_DEPRECATED;
875             } else if (ANNOTATIONS
876                     && "RuntimeVisibleAnnotations".equals(attrName)) {
877                 anns = u + 8;
878             } else if (ANNOTATIONS
879                     && "RuntimeVisibleTypeAnnotations".equals(attrName)) {
880                 tanns = u + 8;
881             } else if (ANNOTATIONS && "AnnotationDefault".equals(attrName)) {
882                 dann = u + 8;
883             } else if ("Synthetic".equals(attrName)) {
884                 context.access |= Opcodes.ACC_SYNTHETIC
885                         | ClassWriter.ACC_SYNTHETIC_ATTRIBUTE;
886             } else if (ANNOTATIONS
887                     && "RuntimeInvisibleAnnotations".equals(attrName)) {
888                 ianns = u + 8;
889             } else if (ANNOTATIONS
890                     && "RuntimeInvisibleTypeAnnotations".equals(attrName)) {
891                 itanns = u + 8;
892             } else if (ANNOTATIONS
893                     && "RuntimeVisibleParameterAnnotations".equals(attrName)) {
894                 mpanns = u + 8;
895             } else if (ANNOTATIONS
896                     && "RuntimeInvisibleParameterAnnotations".equals(attrName)) {
897                 impanns = u + 8;
898             } else if ("MethodParameters".equals(attrName)) {
899                 methodParameters = u + 8;
900             } else {
901                 Attribute attr = readAttribute(context.attrs, attrName, u + 8,
902                         readInt(u + 4), c, -1, null);
903                 if (attr != null) {
904                     attr.next = attributes;
905                     attributes = attr;
906                 }
907             }
908             u += 6 + readInt(u + 4);
909         }
910         u += 2;
911 
912         // visits the method declaration
913         MethodVisitor mv = classVisitor.visitMethod(context.access,
914                 context.name, context.desc, signature, exceptions);
915         if (mv == null) {
916             return u;
917         }
918 
919         /*
920          * if the returned MethodVisitor is in fact a MethodWriter, it means
921          * there is no method adapter between the reader and the writer. If, in
922          * addition, the writer's constant pool was copied from this reader
923          * (mw.cw.cr == this), and the signature and exceptions of the method
924          * have not been changed, then it is possible to skip all visit events
925          * and just copy the original code of the method to the writer (the
926          * access, name and descriptor can have been changed, this is not
927          * important since they are not copied as is from the reader).
928          */
929         if (WRITER && mv instanceof MethodWriter) {
930             MethodWriter mw = (MethodWriter) mv;
931             if (mw.cw.cr == this &&
932 					(signature != null ? signature.equals(mw.signature) : mw.signature == null)) {
933                 boolean sameExceptions = false;
934                 if (exceptions == null) {
935                     sameExceptions = mw.exceptionCount == 0;
936                 } else if (exceptions.length == mw.exceptionCount) {
937                     sameExceptions = true;
938                     for (int j = exceptions.length - 1; j >= 0; --j) {
939                         exception -= 2;
940                         if (mw.exceptions[j] != readUnsignedShort(exception)) {
941                             sameExceptions = false;
942                             break;
943                         }
944                     }
945                 }
946                 if (sameExceptions) {
947                     /*
948                      * we do not copy directly the code into MethodWriter to
949                      * save a byte array copy operation. The real copy will be
950                      * done in ClassWriter.toByteArray().
951                      */
952                     mw.classReaderOffset = firstAttribute;
953                     mw.classReaderLength = u - firstAttribute;
954                     return u;
955                 }
956             }
957         }
958 
959         // visit the method parameters
960         if (methodParameters != 0) {
961             for (int i = b[methodParameters] & 0xFF, v = methodParameters + 1; i > 0; --i, v = v + 4) {
962                 mv.visitParameter(readUTF8(v, c), readUnsignedShort(v + 2));
963             }
964         }
965 
966         // visits the method annotations
967         if (ANNOTATIONS && dann != 0) {
968             AnnotationVisitor dv = mv.visitAnnotationDefault();
969             readAnnotationValue(dann, c, null, dv);
970             if (dv != null) {
971                 dv.visitEnd();
972             }
973         }
974         if (ANNOTATIONS && anns != 0) {
975             for (int i = readUnsignedShort(anns), v = anns + 2; i > 0; --i) {
976                 v = readAnnotationValues(v + 2, c, true,
977                         mv.visitAnnotation(readUTF8(v, c), true));
978             }
979         }
980         if (ANNOTATIONS && ianns != 0) {
981             for (int i = readUnsignedShort(ianns), v = ianns + 2; i > 0; --i) {
982                 v = readAnnotationValues(v + 2, c, true,
983                         mv.visitAnnotation(readUTF8(v, c), false));
984             }
985         }
986         if (ANNOTATIONS && tanns != 0) {
987             for (int i = readUnsignedShort(tanns), v = tanns + 2; i > 0; --i) {
988                 v = readAnnotationTarget(context, v);
989                 v = readAnnotationValues(v + 2, c, true,
990                         mv.visitTypeAnnotation(context.typeRef,
991                                 context.typePath, readUTF8(v, c), true));
992             }
993         }
994         if (ANNOTATIONS && itanns != 0) {
995             for (int i = readUnsignedShort(itanns), v = itanns + 2; i > 0; --i) {
996                 v = readAnnotationTarget(context, v);
997                 v = readAnnotationValues(v + 2, c, true,
998                         mv.visitTypeAnnotation(context.typeRef,
999                                 context.typePath, readUTF8(v, c), false));
1000             }
1001         }
1002         if (ANNOTATIONS && mpanns != 0) {
1003             readParameterAnnotations(mv, context, mpanns, true);
1004         }
1005         if (ANNOTATIONS && impanns != 0) {
1006             readParameterAnnotations(mv, context, impanns, false);
1007         }
1008 
1009         // visits the method attributes
1010         while (attributes != null) {
1011             Attribute attr = attributes.next;
1012             attributes.next = null;
1013             mv.visitAttribute(attributes);
1014             attributes = attr;
1015         }
1016 
1017         // visits the method code
1018         if (code != 0) {
1019             mv.visitCode();
1020             readCode(mv, context, code);
1021         }
1022 
1023         // visits the end of the method
1024         mv.visitEnd();
1025 
1026         return u;
1027     }

1029     /**
1030      * Reads the bytecode of a method and makes the given visitor visit it.
1031      * 
1032      * @param mv
1033      *            the visitor that must visit the method's code.
1034      * @param context
1035      *            information about the class being parsed.
1036      * @param u
1037      *            the start offset of the code attribute in the class file.
1038      */
1039     private void readCode(final MethodVisitor mv, final Context context, int u) {
1040         // reads the header
1041         byte[] b = this.b;
1042         char[] c = context.buffer;
1043         int maxStack = readUnsignedShort(u);
1044         int maxLocals = readUnsignedShort(u + 2);
1045         int codeLength = readInt(u + 4);
1046         u += 8;
1047 
1048         // reads the bytecode to find the labels
1049         int codeStart = u;
1050         int codeEnd = u + codeLength;
1051         Label[] labels = context.labels = new Label[codeLength + 2];
1052         readLabel(codeLength + 1, labels);
1053         while (u < codeEnd) {
1054             int offset = u - codeStart;
1055             int opcode = b[u] & 0xFF;
1056             switch (ClassWriter.TYPE[opcode]) {
1057             case ClassWriter.NOARG_INSN:
1058             case ClassWriter.IMPLVAR_INSN:
1059                 u += 1;
1060                 break;
1061             case ClassWriter.LABEL_INSN:
1062                 readLabel(offset + readShort(u + 1), labels);
1063                 u += 3;
1064                 break;
1065             case ClassWriter.LABELW_INSN:
1066                 readLabel(offset + readInt(u + 1), labels);
1067                 u += 5;
1068                 break;
1069             case ClassWriter.WIDE_INSN:
1070                 opcode = b[u + 1] & 0xFF;
1071                 if (opcode == Opcodes.IINC) {
1072                     u += 6;
1073                 } else {
1074                     u += 4;
1075                 }
1076                 break;
1077             case ClassWriter.TABL_INSN:
1078                 // skips 0 to 3 padding bytes
1079                 u = u + 4 - (offset & 3);
1080                 // reads instruction
1081                 readLabel(offset + readInt(u), labels);
1082                 for (int i = readInt(u + 8) - readInt(u + 4) + 1; i > 0; --i) {
1083                     readLabel(offset + readInt(u + 12), labels);
1084                     u += 4;
1085                 }
1086                 u += 12;
1087                 break;
1088             case ClassWriter.LOOK_INSN:
1089                 // skips 0 to 3 padding bytes
1090                 u = u + 4 - (offset & 3);
1091                 // reads instruction
1092                 readLabel(offset + readInt(u), labels);
1093                 for (int i = readInt(u + 4); i > 0; --i) {
1094                     readLabel(offset + readInt(u + 12), labels);
1095                     u += 8;
1096                 }
1097                 u += 8;
1098                 break;
1099             case ClassWriter.VAR_INSN:
1100             case ClassWriter.SBYTE_INSN:
1101             case ClassWriter.LDC_INSN:
1102                 u += 2;
1103                 break;
1104             case ClassWriter.SHORT_INSN:
1105             case ClassWriter.LDCW_INSN:
1106             case ClassWriter.FIELDORMETH_INSN:
1107             case ClassWriter.TYPE_INSN:
1108             case ClassWriter.IINC_INSN:
1109                 u += 3;
1110                 break;
1111             case ClassWriter.ITFMETH_INSN:
1112             case ClassWriter.INDYMETH_INSN:
1113                 u += 5;
1114                 break;
1115             // case MANA_INSN:
1116             default:
1117                 u += 4;
1118                 break;
1119             }
1120         }
1121 
1122         // reads the try catch entries to find the labels, and also visits them
1123         for (int i = readUnsignedShort(u); i > 0; --i) {
1124             Label start = readLabel(readUnsignedShort(u + 2), labels);
1125             Label end = readLabel(readUnsignedShort(u + 4), labels);
1126             Label handler = readLabel(readUnsignedShort(u + 6), labels);
1127             String type = readUTF8(items[readUnsignedShort(u + 8)], c);
1128             mv.visitTryCatchBlock(start, end, handler, type);
1129             u += 8;
1130         }
1131         u += 2;
1132 
1133         // reads the code attributes
1134         int[] tanns = null; // start index of each visible type annotation
1135         int[] itanns = null; // start index of each invisible type annotation
1136         int tann = 0; // current index in tanns array
1137         int itann = 0; // current index in itanns array
1138         int ntoff = -1; // next visible type annotation code offset
1139         int nitoff = -1; // next invisible type annotation code offset
1140         int varTable = 0;
1141         int varTypeTable = 0;
1142         boolean zip = true;
1143         boolean unzip = (context.flags & EXPAND_FRAMES) != 0;
1144         int stackMap = 0;
1145         int stackMapSize = 0;
1146         int frameCount = 0;
1147         Context frame = null;
1148         Attribute attributes = null;
1149 
1150         for (int i = readUnsignedShort(u); i > 0; --i) {
1151             String attrName = readUTF8(u + 2, c);
1152             if ("LocalVariableTable".equals(attrName)) {
1153                 if ((context.flags & SKIP_DEBUG) == 0) {
1154                     varTable = u + 8;
1155                     for (int j = readUnsignedShort(u + 8), v = u; j > 0; --j) {
1156                         int label = readUnsignedShort(v + 10);
1157                         if (labels[label] == null) {
1158                             readLabel(label, labels).status |= Label.DEBUG;
1159                         }
1160                         label += readUnsignedShort(v + 12);
1161                         if (labels[label] == null) {
1162                             readLabel(label, labels).status |= Label.DEBUG;
1163                         }
1164                         v += 10;
1165                     }
1166                 }
1167             } else if ("LocalVariableTypeTable".equals(attrName)) {
1168                 varTypeTable = u + 8;
1169             } else if ("LineNumberTable".equals(attrName)) {
1170                 if ((context.flags & SKIP_DEBUG) == 0) {
1171                     for (int j = readUnsignedShort(u + 8), v = u; j > 0; --j) {
1172                         int label = readUnsignedShort(v + 10);
1173                         if (labels[label] == null) {
1174                             readLabel(label, labels).status |= Label.DEBUG;
1175                         }
1176                         Label l = labels[label];
1177                         while (l.line > 0) {
1178                             if (l.next == null) {
1179                                 l.next = new Label();
1180                             }
1181                             l = l.next;
1182                         }
1183                         l.line = readUnsignedShort(v + 12);
1184                         v += 4;
1185                     }
1186                 }
1187             } else if (ANNOTATIONS
1188                     && "RuntimeVisibleTypeAnnotations".equals(attrName)) {
1189                 tanns = readTypeAnnotations(mv, context, u + 8, true);
1190                 ntoff = tanns.length == 0 || readByte(tanns[0]) < 0x43 ? -1
1191                         : readUnsignedShort(tanns[0] + 1);
1192             } else if (ANNOTATIONS
1193                     && "RuntimeInvisibleTypeAnnotations".equals(attrName)) {
1194                 itanns = readTypeAnnotations(mv, context, u + 8, false);
1195                 nitoff = itanns.length == 0 || readByte(itanns[0]) < 0x43 ? -1
1196                         : readUnsignedShort(itanns[0] + 1);
1197             } else if (FRAMES && "StackMapTable".equals(attrName)) {
1198                 if ((context.flags & SKIP_FRAMES) == 0) {
1199                     stackMap = u + 10;
1200                     stackMapSize = readInt(u + 4);
1201                     frameCount = readUnsignedShort(u + 8);
1202                 }
1203                 /*
1204                  * here we do not extract the labels corresponding to the
1205                  * attribute content. This would require a full parsing of the
1206                  * attribute, which would need to be repeated in the second
1207                  * phase (see below). Instead the content of the attribute is
1208                  * read one frame at a time (i.e. after a frame has been
1209                  * visited, the next frame is read), and the labels it contains
1210                  * are also extracted one frame at a time. Thanks to the
1211                  * ordering of frames, having only a "one frame lookahead" is
1212                  * not a problem, i.e. it is not possible to see an offset
1213                  * smaller than the offset of the current insn and for which no
1214                  * Label exist.
1215                  */
1216                 /*
1217                  * This is not true for UNINITIALIZED type offsets. We solve
1218                  * this by parsing the stack map table without a full decoding
1219                  * (see below).
1220                  */
1221             } else if (FRAMES && "StackMap".equals(attrName)) {
1222                 if ((context.flags & SKIP_FRAMES) == 0) {
1223                     zip = false;
1224                     stackMap = u + 10;
1225                     stackMapSize = readInt(u + 4);
1226                     frameCount = readUnsignedShort(u + 8);
1227                 }
1228                 /*
1229                  * IMPORTANT! here we assume that the frames are ordered, as in
1230                  * the StackMapTable attribute, although this is not guaranteed
1231                  * by the attribute format.
1232                  */
1233             } else {
1234                 for (int j = 0; j < context.attrs.length; ++j) {
1235                     if (context.attrs[j].type.equals(attrName)) {
1236                         Attribute attr = context.attrs[j].read(this, u + 8,
1237                                 readInt(u + 4), c, codeStart - 8, labels);
1238                         if (attr != null) {
1239                             attr.next = attributes;
1240                             attributes = attr;
1241                         }
1242                     }
1243                 }
1244             }
1245             u += 6 + readInt(u + 4);
1246         }
1247         u += 2;
1248 
1249         // generates the first (implicit) stack map frame
1250         if (FRAMES && stackMap != 0) {
1251             /*
1252              * for the first explicit frame the offset is not offset_delta + 1
1253              * but only offset_delta; setting the implicit frame offset to -1
1254              * allow the use of the "offset_delta + 1" rule in all cases
1255              */
1256             frame = context;
1257             frame.offset = -1;
1258             frame.mode = 0;
1259             frame.localCount = 0;
1260             frame.localDiff = 0;
1261             frame.stackCount = 0;
1262             frame.local = new Object[maxLocals];
1263             frame.stack = new Object[maxStack];
1264             if (unzip) {
1265                 getImplicitFrame(context);
1266             }
1267             /*
1268              * Finds labels for UNINITIALIZED frame types. Instead of decoding
1269              * each element of the stack map table, we look for 3 consecutive
1270              * bytes that "look like" an UNINITIALIZED type (tag 8, offset
1271              * within code bounds, NEW instruction at this offset). We may find
1272              * false positives (i.e. not real UNINITIALIZED types), but this
1273              * should be rare, and the only consequence will be the creation of
1274              * an unneeded label. This is better than creating a label for each
1275              * NEW instruction, and faster than fully decoding the whole stack
1276              * map table.
1277              */
1278             for (int i = stackMap; i < stackMap + stackMapSize - 2; ++i) {
1279                 if (b[i] == 8) { // UNINITIALIZED FRAME TYPE
1280                     int v = readUnsignedShort(i + 1);
1281                     if (v >= 0 && v < codeLength) {
1282                         if ((b[codeStart + v] & 0xFF) == Opcodes.NEW) {
1283                             readLabel(v, labels);
1284                         }
1285                     }
1286                 }
1287             }
1288         }
1289 
1290         // visits the instructions
1291         u = codeStart;
1292         while (u < codeEnd) {
1293             int offset = u - codeStart;
1294 
1295             // visits the label and line number for this offset, if any
1296             Label l = labels[offset];
1297             if (l != null) {
1298                 Label next = l.next;
1299                 l.next = null;
1300                 mv.visitLabel(l);
1301                 if ((context.flags & SKIP_DEBUG) == 0 && l.line > 0) {
1302                     mv.visitLineNumber(l.line, l);
1303                     while (next != null) {
1304                         mv.visitLineNumber(next.line, l);
1305                         next = next.next;
1306                     }
1307                 }
1308             }
1309 
1310             // visits the frame for this offset, if any
1311             while (FRAMES && frame != null
1312                     && (frame.offset == offset || frame.offset == -1)) {
1313                 // if there is a frame for this offset, makes the visitor visit
1314                 // it, and reads the next frame if there is one.
1315                 if (frame.offset != -1) {
1316                     if (!zip || unzip) {
1317                         mv.visitFrame(Opcodes.F_NEW, frame.localCount,
1318                                 frame.local, frame.stackCount, frame.stack);
1319                     } else {
1320                         mv.visitFrame(frame.mode, frame.localDiff, frame.local,
1321                                 frame.stackCount, frame.stack);
1322                     }
1323                 }
1324                 if (frameCount > 0) {
1325                     stackMap = readFrame(stackMap, zip, unzip, frame);
1326                     --frameCount;
1327                 } else {
1328                     frame = null;
1329                 }
1330             }
1331 
1332             // visits the instruction at this offset
1333             int opcode = b[u] & 0xFF;
1334             switch (ClassWriter.TYPE[opcode]) {
1335             case ClassWriter.NOARG_INSN:
1336                 mv.visitInsn(opcode);
1337                 u += 1;
1338                 break;
1339             case ClassWriter.IMPLVAR_INSN:
1340                 if (opcode > Opcodes.ISTORE) {
1341                     opcode -= 59; // ISTORE_0
1342                     mv.visitVarInsn(Opcodes.ISTORE + (opcode >> 2),
1343                             opcode & 0x3);
1344                 } else {
1345                     opcode -= 26; // ILOAD_0
1346                     mv.visitVarInsn(Opcodes.ILOAD + (opcode >> 2), opcode & 0x3);
1347                 }
1348                 u += 1;
1349                 break;
1350             case ClassWriter.LABEL_INSN:
1351                 mv.visitJumpInsn(opcode, labels[offset + readShort(u + 1)]);
1352                 u += 3;
1353                 break;
1354             case ClassWriter.LABELW_INSN:
1355                 mv.visitJumpInsn(opcode - 33, labels[offset + readInt(u + 1)]);
1356                 u += 5;
1357                 break;
1358             case ClassWriter.WIDE_INSN:
1359                 opcode = b[u + 1] & 0xFF;
1360                 if (opcode == Opcodes.IINC) {
1361                     mv.visitIincInsn(readUnsignedShort(u + 2), readShort(u + 4));
1362                     u += 6;
1363                 } else {
1364                     mv.visitVarInsn(opcode, readUnsignedShort(u + 2));
1365                     u += 4;
1366                 }
1367                 break;
1368             case ClassWriter.TABL_INSN: {
1369                 // skips 0 to 3 padding bytes
1370                 u = u + 4 - (offset & 3);
1371                 // reads instruction
1372                 int label = offset + readInt(u);
1373                 int min = readInt(u + 4);
1374                 int max = readInt(u + 8);
1375                 Label[] table = new Label[max - min + 1];
1376                 u += 12;
1377                 for (int i = 0; i < table.length; ++i) {
1378                     table[i] = labels[offset + readInt(u)];
1379                     u += 4;
1380                 }
1381                 mv.visitTableSwitchInsn(min, max, labels[label], table);
1382                 break;
1383             }
1384             case ClassWriter.LOOK_INSN: {
1385                 // skips 0 to 3 padding bytes
1386                 u = u + 4 - (offset & 3);
1387                 // reads instruction
1388                 int label = offset + readInt(u);
1389                 int len = readInt(u + 4);
1390                 int[] keys = new int[len];
1391                 Label[] values = new Label[len];
1392                 u += 8;
1393                 for (int i = 0; i < len; ++i) {
1394                     keys[i] = readInt(u);
1395                     values[i] = labels[offset + readInt(u + 4)];
1396                     u += 8;
1397                 }
1398                 mv.visitLookupSwitchInsn(labels[label], keys, values);
1399                 break;
1400             }
1401             case ClassWriter.VAR_INSN:
1402                 mv.visitVarInsn(opcode, b[u + 1] & 0xFF);
1403                 u += 2;
1404                 break;
1405             case ClassWriter.SBYTE_INSN:
1406                 mv.visitIntInsn(opcode, b[u + 1]);
1407                 u += 2;
1408                 break;
1409             case ClassWriter.SHORT_INSN:
1410                 mv.visitIntInsn(opcode, readShort(u + 1));
1411                 u += 3;
1412                 break;
1413             case ClassWriter.LDC_INSN:
1414                 mv.visitLdcInsn(readConst(b[u + 1] & 0xFF, c));
1415                 u += 2;
1416                 break;
1417             case ClassWriter.LDCW_INSN:
1418                 mv.visitLdcInsn(readConst(readUnsignedShort(u + 1), c));
1419                 u += 3;
1420                 break;
1421             case ClassWriter.FIELDORMETH_INSN:
1422             case ClassWriter.ITFMETH_INSN: {
1423                 int cpIndex = items[readUnsignedShort(u + 1)];
1424                 boolean itf = b[cpIndex - 1] == ClassWriter.IMETH;
1425                 String iowner = readClass(cpIndex, c);
1426                 cpIndex = items[readUnsignedShort(cpIndex + 2)];
1427                 String iname = readUTF8(cpIndex, c);
1428                 String idesc = readUTF8(cpIndex + 2, c);
1429                 if (opcode < Opcodes.INVOKEVIRTUAL) {
1430                     mv.visitFieldInsn(opcode, iowner, iname, idesc);
1431                 } else {
1432                     mv.visitMethodInsn(opcode, iowner, iname, idesc, itf);
1433                 }
1434                 if (opcode == Opcodes.INVOKEINTERFACE) {
1435                     u += 5;
1436                 } else {
1437                     u += 3;
1438                 }
1439                 break;
1440             }
1441             case ClassWriter.INDYMETH_INSN: {
1442                 int cpIndex = items[readUnsignedShort(u + 1)];
1443                 int bsmIndex = context.bootstrapMethods[readUnsignedShort(cpIndex)];
1444                 Handle bsm = (Handle) readConst(readUnsignedShort(bsmIndex), c);
1445                 int bsmArgCount = readUnsignedShort(bsmIndex + 2);
1446                 Object[] bsmArgs = new Object[bsmArgCount];
1447                 bsmIndex += 4;
1448                 for (int i = 0; i < bsmArgCount; i++) {
1449                     bsmArgs[i] = readConst(readUnsignedShort(bsmIndex), c);
1450                     bsmIndex += 2;
1451                 }
1452                 cpIndex = items[readUnsignedShort(cpIndex + 2)];
1453                 String iname = readUTF8(cpIndex, c);
1454                 String idesc = readUTF8(cpIndex + 2, c);
1455                 mv.visitInvokeDynamicInsn(iname, idesc, bsm, bsmArgs);
1456                 u += 5;
1457                 break;
1458             }
1459             case ClassWriter.TYPE_INSN:
1460                 mv.visitTypeInsn(opcode, readClass(u + 1, c));
1461                 u += 3;
1462                 break;
1463             case ClassWriter.IINC_INSN:
1464                 mv.visitIincInsn(b[u + 1] & 0xFF, b[u + 2]);
1465                 u += 3;
1466                 break;
1467             // case MANA_INSN:
1468             default:
1469                 mv.visitMultiANewArrayInsn(readClass(u + 1, c), b[u + 3] & 0xFF);
1470                 u += 4;
1471                 break;
1472             }
1473 
1474             // visit the instruction annotations, if any
1475             while (tanns != null && tann < tanns.length && ntoff <= offset) {
1476                 if (ntoff == offset) {
1477                     int v = readAnnotationTarget(context, tanns[tann]);
1478                     readAnnotationValues(v + 2, c, true,
1479                             mv.visitInsnAnnotation(context.typeRef,
1480                                     context.typePath, readUTF8(v, c), true));
1481                 }
1482                 ntoff = ++tann >= tanns.length || readByte(tanns[tann]) < 0x43 ? -1
1483                         : readUnsignedShort(tanns[tann] + 1);
1484             }
1485             while (itanns != null && itann < itanns.length && nitoff <= offset) {
1486                 if (nitoff == offset) {
1487                     int v = readAnnotationTarget(context, itanns[itann]);
1488                     readAnnotationValues(v + 2, c, true,
1489                             mv.visitInsnAnnotation(context.typeRef,
1490                                     context.typePath, readUTF8(v, c), false));
1491                 }
1492                 nitoff = ++itann >= itanns.length
1493                         || readByte(itanns[itann]) < 0x43 ? -1
1494                         : readUnsignedShort(itanns[itann] + 1);
1495             }
1496         }
1497         if (labels[codeLength] != null) {
1498             mv.visitLabel(labels[codeLength]);
1499         }
1500 
1501         // visits the local variable tables
1502         if ((context.flags & SKIP_DEBUG) == 0 && varTable != 0) {
1503             int[] typeTable = null;
1504             if (varTypeTable != 0) {
1505                 u = varTypeTable + 2;
1506                 typeTable = new int[readUnsignedShort(varTypeTable) * 3];
1507                 for (int i = typeTable.length; i > 0;) {
1508                     typeTable[--i] = u + 6; // signature
1509                     typeTable[--i] = readUnsignedShort(u + 8); // index
1510                     typeTable[--i] = readUnsignedShort(u); // start
1511                     u += 10;
1512                 }
1513             }
1514             u = varTable + 2;
1515             for (int i = readUnsignedShort(varTable); i > 0; --i) {
1516                 int start = readUnsignedShort(u);
1517                 int length = readUnsignedShort(u + 2);
1518                 int index = readUnsignedShort(u + 8);
1519                 String vsignature = null;
1520                 if (typeTable != null) {
1521                     for (int j = 0; j < typeTable.length; j += 3) {
1522                         if (typeTable[j] == start && typeTable[j + 1] == index) {
1523                             vsignature = readUTF8(typeTable[j + 2], c);
1524                             break;
1525                         }
1526                     }
1527                 }
1528                 mv.visitLocalVariable(readUTF8(u + 4, c), readUTF8(u + 6, c),
1529                         vsignature, labels[start], labels[start + length],
1530                         index);
1531                 u += 10;
1532             }
1533         }
1534 
1535         // visits the local variables type annotations
1536         if (tanns != null) {
1537             for (int i = 0; i < tanns.length; ++i) {
1538                 if ((readByte(tanns[i]) >> 1) == (0x40 >> 1)) {
1539                     int v = readAnnotationTarget(context, tanns[i]);
1540                     v = readAnnotationValues(v + 2, c, true,
1541                             mv.visitLocalVariableAnnotation(context.typeRef,
1542                                     context.typePath, context.start,
1543                                     context.end, context.index, readUTF8(v, c),
1544                                     true));
1545                 }
1546             }
1547         }
1548         if (itanns != null) {
1549             for (int i = 0; i < itanns.length; ++i) {
1550                 if ((readByte(itanns[i]) >> 1) == (0x40 >> 1)) {
1551                     int v = readAnnotationTarget(context, itanns[i]);
1552                     v = readAnnotationValues(v + 2, c, true,
1553                             mv.visitLocalVariableAnnotation(context.typeRef,
1554                                     context.typePath, context.start,
1555                                     context.end, context.index, readUTF8(v, c),
1556                                     false));
1557                 }
1558             }
1559         }
1560 
1561         // visits the code attributes
1562         while (attributes != null) {
1563             Attribute attr = attributes.next;
1564             attributes.next = null;
1565             mv.visitAttribute(attributes);
1566             attributes = attr;
1567         }
1568 
1569         // visits the max stack and max locals values
1570         mv.visitMaxs(maxStack, maxLocals);
1571     }

1573     /**
1574      * Parses a type annotation table to find the labels, and to visit the try
1575      * catch block annotations.
1576      * 
1577      * @param u
1578      *            the start offset of a type annotation table.
1579      * @param mv
1580      *            the method visitor to be used to visit the try catch block
1581      *            annotations.
1582      * @param context
1583      *            information about the class being parsed.
1584      * @param visible
1585      *            if the type annotation table to parse contains runtime visible
1586      *            annotations.
1587      * @return the start offset of each type annotation in the parsed table.
1588      */
1589     private int[] readTypeAnnotations(final MethodVisitor mv,
1590             final Context context, int u, boolean visible) {
1591         char[] c = context.buffer;
1592         int[] offsets = new int[readUnsignedShort(u)];
1593         u += 2;
1594         for (int i = 0; i < offsets.length; ++i) {
1595             offsets[i] = u;
1596             int target = readInt(u);
1597             switch (target >>> 24) {
1598             case 0x00: // CLASS_TYPE_PARAMETER
1599             case 0x01: // METHOD_TYPE_PARAMETER
1600             case 0x16: // METHOD_FORMAL_PARAMETER
1601                 u += 2;
1602                 break;
1603             case 0x13: // FIELD
1604             case 0x14: // METHOD_RETURN
1605             case 0x15: // METHOD_RECEIVER
1606                 u += 1;
1607                 break;
1608             case 0x40: // LOCAL_VARIABLE
1609             case 0x41: // RESOURCE_VARIABLE
1610                 for (int j = readUnsignedShort(u + 1); j > 0; --j) {
1611                     int start = readUnsignedShort(u + 3);
1612                     int length = readUnsignedShort(u + 5);
1613                     readLabel(start, context.labels);
1614                     readLabel(start + length, context.labels);
1615                     u += 6;
1616                 }
1617                 u += 3;
1618                 break;
1619             case 0x47: // CAST
1620             case 0x48: // CONSTRUCTOR_INVOCATION_TYPE_ARGUMENT
1621             case 0x49: // METHOD_INVOCATION_TYPE_ARGUMENT
1622             case 0x4A: // CONSTRUCTOR_REFERENCE_TYPE_ARGUMENT
1623             case 0x4B: // METHOD_REFERENCE_TYPE_ARGUMENT
1624                 u += 4;
1625                 break;
1626             // case 0x10: // CLASS_EXTENDS
1627             // case 0x11: // CLASS_TYPE_PARAMETER_BOUND
1628             // case 0x12: // METHOD_TYPE_PARAMETER_BOUND
1629             // case 0x17: // THROWS
1630             // case 0x42: // EXCEPTION_PARAMETER
1631             // case 0x43: // INSTANCEOF
1632             // case 0x44: // NEW
1633             // case 0x45: // CONSTRUCTOR_REFERENCE
1634             // case 0x46: // METHOD_REFERENCE
1635             default:
1636                 u += 3;
1637                 break;
1638             }
1639             int pathLength = readByte(u);
1640             if ((target >>> 24) == 0x42) {
1641                 TypePath path = pathLength == 0 ? null : new TypePath(b, u);
1642                 u += 1 + 2 * pathLength;
1643                 u = readAnnotationValues(u + 2, c, true,
1644                         mv.visitTryCatchAnnotation(target, path,
1645                                 readUTF8(u, c), visible));
1646             } else {
1647                 u = readAnnotationValues(u + 3 + 2 * pathLength, c, true, null);
1648             }
1649         }
1650         return offsets;
1651     }

1653     /**
1654      * Parses the header of a type annotation to extract its target_type and
1655      * target_path (the result is stored in the given context), and returns the
1656      * start offset of the rest of the type_annotation structure (i.e. the
1657      * offset to the type_index field, which is followed by
1658      * num_element_value_pairs and then the name,value pairs).
1659      * 
1660      * @param context
1661      *            information about the class being parsed. This is where the
1662      *            extracted target_type and target_path must be stored.
1663      * @param u
1664      *            the start offset of a type_annotation structure.
1665      * @return the start offset of the rest of the type_annotation structure.
1666      */
1667     private int readAnnotationTarget(final Context context, int u) {
1668         int target = readInt(u);
1669         switch (target >>> 24) {
1670         case 0x00: // CLASS_TYPE_PARAMETER
1671         case 0x01: // METHOD_TYPE_PARAMETER
1672         case 0x16: // METHOD_FORMAL_PARAMETER
1673             target &= 0xFFFF0000;
1674             u += 2;
1675             break;
1676         case 0x13: // FIELD
1677         case 0x14: // METHOD_RETURN
1678         case 0x15: // METHOD_RECEIVER
1679             target &= 0xFF000000;
1680             u += 1;
1681             break;
1682         case 0x40: // LOCAL_VARIABLE
1683         case 0x41: { // RESOURCE_VARIABLE
1684             target &= 0xFF000000;
1685             int n = readUnsignedShort(u + 1);
1686             context.start = new Label[n];
1687             context.end = new Label[n];
1688             context.index = new int[n];
1689             u += 3;
1690             for (int i = 0; i < n; ++i) {
1691                 int start = readUnsignedShort(u);
1692                 int length = readUnsignedShort(u + 2);
1693                 context.start[i] = readLabel(start, context.labels);
1694                 context.end[i] = readLabel(start + length, context.labels);
1695                 context.index[i] = readUnsignedShort(u + 4);
1696                 u += 6;
1697             }
1698             break;
1699         }
1700         case 0x47: // CAST
1701         case 0x48: // CONSTRUCTOR_INVOCATION_TYPE_ARGUMENT
1702         case 0x49: // METHOD_INVOCATION_TYPE_ARGUMENT
1703         case 0x4A: // CONSTRUCTOR_REFERENCE_TYPE_ARGUMENT
1704         case 0x4B: // METHOD_REFERENCE_TYPE_ARGUMENT
1705             target &= 0xFF0000FF;
1706             u += 4;
1707             break;
1708         // case 0x10: // CLASS_EXTENDS
1709         // case 0x11: // CLASS_TYPE_PARAMETER_BOUND
1710         // case 0x12: // METHOD_TYPE_PARAMETER_BOUND
1711         // case 0x17: // THROWS
1712         // case 0x42: // EXCEPTION_PARAMETER
1713         // case 0x43: // INSTANCEOF
1714         // case 0x44: // NEW
1715         // case 0x45: // CONSTRUCTOR_REFERENCE
1716         // case 0x46: // METHOD_REFERENCE
1717         default:
1718             target &= (target >>> 24) < 0x43 ? 0xFFFFFF00 : 0xFF000000;
1719             u += 3;
1720             break;
1721         }
1722         int pathLength = readByte(u);
1723         context.typeRef = target;
1724         context.typePath = pathLength == 0 ? null : new TypePath(b, u);
1725         return u + 1 + 2 * pathLength;
1726     }

1728     /**
1729      * Reads parameter annotations and makes the given visitor visit them.
1730      * 
1731      * @param mv
1732      *            the visitor that must visit the annotations.
1733      * @param context
1734      *            information about the class being parsed.
1735      * @param v
1736      *            start offset in {@link #b b} of the annotations to be read.
1737      * @param visible
1738      *            <tt>true</tt> if the annotations to be read are visible at
1739      *            runtime.
1740      */
1741     private void readParameterAnnotations(final MethodVisitor mv,
1742             final Context context, int v, final boolean visible) {
1743         int i;
1744         int n = b[v++] & 0xFF;
1745         // workaround for a bug in javac (javac compiler generates a parameter
1746         // annotation array whose size is equal to the number of parameters in
1747         // the Java source file, while it should generate an array whose size is
1748         // equal to the number of parameters in the method descriptor - which
1749         // includes the synthetic parameters added by the compiler). This work-
1750         // around supposes that the synthetic parameters are the first ones.
1751         int synthetics = Type.getArgumentTypes(context.desc).length - n;
1752         AnnotationVisitor av;
1753         for (i = 0; i < synthetics; ++i) {
1754             // virtual annotation to detect synthetic parameters in MethodWriter
1755             av = mv.visitParameterAnnotation(i, "Ljava/lang/Synthetic;", false);
1756             if (av != null) {
1757                 av.visitEnd();
1758             }
1759         }
1760         char[] c = context.buffer;
1761         for (; i < n + synthetics; ++i) {
1762             int j = readUnsignedShort(v);
1763             v += 2;
1764             for (; j > 0; --j) {
1765                 av = mv.visitParameterAnnotation(i, readUTF8(v, c), visible);
1766                 v = readAnnotationValues(v + 2, c, true, av);
1767             }
1768         }
1769     }

1776      *            (including the unsigned short that gives the number of
1777      *            values).

1779      *            buffer to be used to call {@link #readUTF8 readUTF8},

1783      *            if the annotation values are named or not.
1784      * @param av
1785      *            the visitor that must visit the values.

1787      */

1795             }
1796         } else {
1797             for (; i > 0; --i) {
1798                 v = readAnnotationValue(v, buf, null, av);
1799             }
1800         }
1801         if (av != null) {
1802             av.visitEnd();
1803         }
1804         return v;

1807     /**
1808      * Reads a value of an annotation and makes the given visitor visit it.
1809      * 
1810      * @param v
1811      *            the start offset in {@link #b b} of the value to be read
1812      *            (<i>not including the value name constant pool index</i>).
1813      * @param buf
1814      *            buffer to be used to call {@link #readUTF8 readUTF8},
1815      *            {@link #readClass(int,char[]) readClass} or {@link #readConst
1816      *            readConst}.
1817      * @param name
1818      *            the name of the value to be read.
1819      * @param av
1820      *            the visitor that must visit the value.
1821      * @return the end offset of the annotation value.
1822      */
1823     private int readAnnotationValue(int v, final char[] buf, final String name,
1824             final AnnotationVisitor av) {
1825         int i;
1826         if (av == null) {
1827             switch (b[v] & 0xFF) {
1828             case 'e': // enum_const_value
1829                 return v + 5;
1830             case '@': // annotation_value
1831                 return readAnnotationValues(v + 3, buf, true, null);
1832             case '[': // array_value
1833                 return readAnnotationValues(v + 1, buf, false, null);
1834             default:
1835                 return v + 3;
1836             }
1837         }
1838         switch (b[v++] & 0xFF) {
1839         case 'I': // pointer to CONSTANT_Integer
1840         case 'J': // pointer to CONSTANT_Long
1841         case 'F': // pointer to CONSTANT_Float
1842         case 'D': // pointer to CONSTANT_Double
1843             av.visit(name, readConst(readUnsignedShort(v), buf));
1844             v += 2;
1845             break;
1846         case 'B': // pointer to CONSTANT_Byte
1847             av.visit(name, (byte) readInt(items[readUnsignedShort(v)]));
1848             v += 2;
1849             break;
1850         case 'Z': // pointer to CONSTANT_Boolean
1851             av.visit(name, readInt(items[readUnsignedShort(v)]) == 0 ? Boolean.FALSE : Boolean.TRUE);
1852             v += 2;
1853             break;
1854         case 'S': // pointer to CONSTANT_Short
1855             av.visit(name, (short) readInt(items[readUnsignedShort(v)]));
1856             v += 2;
1857             break;
1858         case 'C': // pointer to CONSTANT_Char
1859             av.visit(name, (char) readInt(items[readUnsignedShort(v)]));
1860             v += 2;
1861             break;
1862         case 's': // pointer to CONSTANT_Utf8
1863             av.visit(name, readUTF8(v, buf));
1864             v += 2;
1865             break;
1866         case 'e': // enum_const_value
1867             av.visitEnum(name, readUTF8(v, buf), readUTF8(v + 2, buf));
1868             v += 4;
1869             break;
1870         case 'c': // class_info
1871             av.visit(name, Type.getType(readUTF8(v, buf)));
1872             v += 2;
1873             break;
1874         case '@': // annotation_value
1875             v = readAnnotationValues(v + 2, buf, true,
1876                     av.visitAnnotation(name, readUTF8(v, buf)));
1877             break;
1878         case '[': // array_value
1879             int size = readUnsignedShort(v);
1880             v += 2;
1881             if (size == 0) {
1882                 return readAnnotationValues(v - 2, buf, false,
1883                         av.visitArray(name));
1884             }
1885             switch (this.b[v++] & 0xFF) {
1886             case 'B':
1887                 byte[] bv = new byte[size];
1888                 for (i = 0; i < size; i++) {
1889                     bv[i] = (byte) readInt(items[readUnsignedShort(v)]);
1890                     v += 3;
1891                 }
1892                 av.visit(name, bv);
1893                 --v;
1894                 break;
1895             case 'Z':
1896                 boolean[] zv = new boolean[size];
1897                 for (i = 0; i < size; i++) {
1898                     zv[i] = readInt(items[readUnsignedShort(v)]) != 0;
1899                     v += 3;
1900                 }
1901                 av.visit(name, zv);
1902                 --v;
1903                 break;
1904             case 'S':
1905                 short[] sv = new short[size];
1906                 for (i = 0; i < size; i++) {
1907                     sv[i] = (short) readInt(items[readUnsignedShort(v)]);
1908                     v += 3;
1909                 }
1910                 av.visit(name, sv);
1911                 --v;
1912                 break;
1913             case 'C':
1914                 char[] cv = new char[size];
1915                 for (i = 0; i < size; i++) {
1916                     cv[i] = (char) readInt(items[readUnsignedShort(v)]);
1917                     v += 3;
1918                 }
1919                 av.visit(name, cv);
1920                 --v;
1921                 break;
1922             case 'I':
1923                 int[] iv = new int[size];
1924                 for (i = 0; i < size; i++) {
1925                     iv[i] = readInt(items[readUnsignedShort(v)]);
1926                     v += 3;
1927                 }
1928                 av.visit(name, iv);
1929                 --v;
1930                 break;
1931             case 'J':
1932                 long[] lv = new long[size];
1933                 for (i = 0; i < size; i++) {
1934                     lv[i] = readLong(items[readUnsignedShort(v)]);
1935                     v += 3;
1936                 }
1937                 av.visit(name, lv);
1938                 --v;
1939                 break;
1940             case 'F':
1941                 float[] fv = new float[size];
1942                 for (i = 0; i < size; i++) {
1943                     fv[i] = Float
1944                             .intBitsToFloat(readInt(items[readUnsignedShort(v)]));
1945                     v += 3;
1946                 }
1947                 av.visit(name, fv);
1948                 --v;
1949                 break;
1950             case 'D':
1951                 double[] dv = new double[size];
1952                 for (i = 0; i < size; i++) {
1953                     dv[i] = Double
1954                             .longBitsToDouble(readLong(items[readUnsignedShort(v)]));
1955                     v += 3;
1956                 }
1957                 av.visit(name, dv);
1958                 --v;
1959                 break;
1960             default:
1961                 v = readAnnotationValues(v - 3, buf, false, av.visitArray(name));
1962             }
1963         }
1964         return v;

1967     /**
1968      * Computes the implicit frame of the method currently being parsed (as
1969      * defined in the given {@link Context}) and stores it in the given context.
1970      * 
1971      * @param frame
1972      *            information about the class being parsed.
1973      */
1974     private void getImplicitFrame(final Context frame) {
1975         String desc = frame.desc;
1976         Object[] locals = frame.local;
1977         int local = 0;
1978         if ((frame.access & Opcodes.ACC_STATIC) == 0) {
1979             if ("<init>".equals(frame.name)) {
1980                 locals[local++] = Opcodes.UNINITIALIZED_THIS;
1981             } else {
1982                 locals[local++] = readClass(header + 2, frame.buffer);
1983             }
1984         }
1985         int i = 1;
1986         loop: while (true) {
1987             int j = i;
1988             switch (desc.charAt(i++)) {
1989             case 'Z':
1990             case 'C':
1991             case 'B':
1992             case 'S':
1993             case 'I':
1994                 locals[local++] = Opcodes.INTEGER;
1995                 break;
1996             case 'F':
1997                 locals[local++] = Opcodes.FLOAT;
1998                 break;
1999             case 'J':
2000                 locals[local++] = Opcodes.LONG;
2001                 break;
2002             case 'D':
2003                 locals[local++] = Opcodes.DOUBLE;
2004                 break;
2005             case '[':
2006                 while (desc.charAt(i) == '[') {
2007                     ++i;
2008                 }
2009                 if (desc.charAt(i) == 'L') {
2010                     ++i;
2011                     while (desc.charAt(i) != ';') {
2012                         ++i;
2013                     }
2014                 }
2015                 locals[local++] = desc.substring(j, ++i);
2016                 break;
2017             case 'L':
2018                 while (desc.charAt(i) != ';') {
2019                     ++i;
2020                 }
2021                 locals[local++] = desc.substring(j + 1, i++);
2022                 break;
2023             default:
2024                 break loop;
2025             }
2026         }
2027         frame.localCount = local;
2028     }

2030     /**
2031      * Reads a stack map frame and stores the result in the given
2032      * {@link Context} object.
2033      * 
2034      * @param stackMap
2035      *            the start offset of a stack map frame in the class file.
2036      * @param zip
2037      *            if the stack map frame at stackMap is compressed or not.
2038      * @param unzip
2039      *            if the stack map frame must be uncompressed.
2040      * @param frame
2041      *            where the parsed stack map frame must be stored.
2042      * @return the offset of the first byte following the parsed frame.
2043      */
2044     private int readFrame(int stackMap, boolean zip, boolean unzip,
2045             Context frame) {
2046         char[] c = frame.buffer;
2047         Label[] labels = frame.labels;
2048         int tag;
2049         int delta;
2050         if (zip) {
2051             tag = b[stackMap++] & 0xFF;
2052         } else {
2053             tag = MethodWriter.FULL_FRAME;
2054             frame.offset = -1;
2055         }
2056         frame.localDiff = 0;
2057         if (tag < MethodWriter.SAME_LOCALS_1_STACK_ITEM_FRAME) {
2058             delta = tag;
2059             frame.mode = Opcodes.F_SAME;
2060             frame.stackCount = 0;
2061         } else if (tag < MethodWriter.RESERVED) {
2062             delta = tag - MethodWriter.SAME_LOCALS_1_STACK_ITEM_FRAME;
2063             stackMap = readFrameType(frame.stack, 0, stackMap, c, labels);
2064             frame.mode = Opcodes.F_SAME1;
2065             frame.stackCount = 1;
2066         } else {
2067             delta = readUnsignedShort(stackMap);
2068             stackMap += 2;
2069             if (tag == MethodWriter.SAME_LOCALS_1_STACK_ITEM_FRAME_EXTENDED) {
2070                 stackMap = readFrameType(frame.stack, 0, stackMap, c, labels);
2071                 frame.mode = Opcodes.F_SAME1;
2072                 frame.stackCount = 1;
2073             } else if (tag >= MethodWriter.CHOP_FRAME
2074                     && tag < MethodWriter.SAME_FRAME_EXTENDED) {
2075                 frame.mode = Opcodes.F_CHOP;
2076                 frame.localDiff = MethodWriter.SAME_FRAME_EXTENDED - tag;
2077                 frame.localCount -= frame.localDiff;
2078                 frame.stackCount = 0;
2079             } else if (tag == MethodWriter.SAME_FRAME_EXTENDED) {
2080                 frame.mode = Opcodes.F_SAME;
2081                 frame.stackCount = 0;
2082             } else if (tag < MethodWriter.FULL_FRAME) {
2083                 int local = unzip ? frame.localCount : 0;
2084                 for (int i = tag - MethodWriter.SAME_FRAME_EXTENDED; i > 0; i--) {
2085                     stackMap = readFrameType(frame.local, local++, stackMap, c,
2086                             labels);
2087                 }
2088                 frame.mode = Opcodes.F_APPEND;
2089                 frame.localDiff = tag - MethodWriter.SAME_FRAME_EXTENDED;
2090                 frame.localCount += frame.localDiff;
2091                 frame.stackCount = 0;
2092             } else { // if (tag == FULL_FRAME) {
2093                 frame.mode = Opcodes.F_FULL;
2094                 int n = readUnsignedShort(stackMap);
2095                 stackMap += 2;
2096                 frame.localDiff = n;
2097                 frame.localCount = n;
2098                 for (int local = 0; n > 0; n--) {
2099                     stackMap = readFrameType(frame.local, local++, stackMap, c,
2100                             labels);
2101                 }
2102                 n = readUnsignedShort(stackMap);
2103                 stackMap += 2;
2104                 frame.stackCount = n;
2105                 for (int stack = 0; n > 0; n--) {
2106                     stackMap = readFrameType(frame.stack, stack++, stackMap, c,
2107                             labels);
2108                 }
2109             }
2110         }
2111         frame.offset += delta + 1;
2112         readLabel(frame.offset, labels);
2113         return stackMap;
2114     }

2117      * Reads a stack map frame type and stores it at the given index in the
2118      * given array.
2119      * 
2120      * @param frame
2121      *            the array where the parsed type must be stored.
2122      * @param index
2123      *            the index in 'frame' where the parsed type must be stored.
2124      * @param v
2125      *            the start offset of the stack map frame type to read.
2126      * @param buf
2127      *            a buffer to read strings.
2128      * @param labels
2129      *            the labels of the method currently being parsed, indexed by
2130      *            their offset. If the parsed type is an Uninitialized type, a
2131      *            new label for the corresponding NEW instruction is stored in
2132      *            this array if it does not already exist.
2133      * @return the offset of the first byte after the parsed type.
2134      */
2135     private int readFrameType(final Object[] frame, final int index, int v,
2136             final char[] buf, final Label[] labels) {
2137         int type = b[v++] & 0xFF;
2138         switch (type) {
2139         case 0:
2140             frame[index] = Opcodes.TOP;
2141             break;
2142         case 1:
2143             frame[index] = Opcodes.INTEGER;
2144             break;
2145         case 2:
2146             frame[index] = Opcodes.FLOAT;
2147             break;
2148         case 3:
2149             frame[index] = Opcodes.DOUBLE;
2150             break;
2151         case 4:
2152             frame[index] = Opcodes.LONG;
2153             break;
2154         case 5:
2155             frame[index] = Opcodes.NULL;
2156             break;
2157         case 6:
2158             frame[index] = Opcodes.UNINITIALIZED_THIS;
2159             break;
2160         case 7: // Object
2161             frame[index] = readClass(v, buf);
2162             v += 2;
2163             break;
2164         default: // Uninitialized
2165             frame[index] = readLabel(readUnsignedShort(v), labels);
2166             v += 2;
2167         }

2196     private int getAttributes() {
2197         // skips the header
2198         int u = header + 8 + readUnsignedShort(header + 6) * 2;
2199         // skips fields and methods
2200         for (int i = readUnsignedShort(u); i > 0; --i) {
2201             for (int j = readUnsignedShort(u + 8); j > 0; --j) {
2202                 u += 6 + readInt(u + 12);
2203             }
2204             u += 8;

2206         u += 2;
2207         for (int i = readUnsignedShort(u); i > 0; --i) {

2209                 u += 6 + readInt(u + 12);
2210             }
2211             u += 8;
2212         }
2213         // the attribute_info structure starts just after the methods
2214         return u + 2;

2256             }
2257         }

2396     /**
2397      * Reads UTF8 string in {@link #b b}.
2398      * 
2399      * @param index
2400      *            start offset of the UTF8 string to be read.
2401      * @param utfLen
2402      *            length of the UTF8 string to be read.
2403      * @param buf
2404      *            buffer to be used to read the string. This buffer must be
2405      *            sufficiently large. It is not automatically resized.
2406      * @return the String corresponding to the specified UTF8 string.
2407      */
2408     private String readUTF(int index, final int utfLen, final char[] buf) {
2409         int endIndex = index + utfLen;
2410         byte[] b = this.b;
2411         int strLen = 0;
2412         int c;
2413         int st = 0;
2414         char cc = 0;
2415         while (index < endIndex) {
2416             c = b[index++];
2417             switch (st) {
2418             case 0:
2419                 c = c & 0xFF;
2420                 if (c < 0x80) { // 0xxxxxxx
2421                     buf[strLen++] = (char) c;
2422                 } else if (c < 0xE0 && c > 0xBF) { // 110x xxxx 10xx xxxx
2423                     cc = (char) (c & 0x1F);
2424                     st = 1;
2425                 } else { // 1110 xxxx 10xx xxxx 10xx xxxx
2426                     cc = (char) (c & 0x0F);
2427                     st = 2;
2428                 }
2429                 break;
2430 
2431             case 1: // byte 2 of 2-byte char or byte 3 of 3-byte char
2432                 buf[strLen++] = (char) ((cc << 6) | (c & 0x3F));
2433                 st = 0;
2434                 break;
2435 
2436             case 2: // byte 2 of 3-byte char
2437                 cc = (char) ((cc << 6) | (c & 0x3F));
2438                 st = 1;
2439                 break;
2440             }
2441         }
2442         return new String(buf, 0, strLen);
2443     }

2467      * method is intended for {@link Attribute} sub classes, and is normally not
2468      * needed by class generators or adapters.</i>
2469      * 
2470      * @param item
2471      *            the index of a constant pool item.
2472      * @param buf
2473      *            buffer to be used to read the item. This buffer must be
2474      *            sufficiently large. It is not automatically resized.
