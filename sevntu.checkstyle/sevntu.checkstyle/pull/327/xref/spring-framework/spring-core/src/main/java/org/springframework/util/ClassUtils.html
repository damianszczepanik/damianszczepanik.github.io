142 	/**
143 	 * Return the default ClassLoader to use: typically the thread context
144 	 * ClassLoader, if available; the ClassLoader that loaded the ClassUtils
145 	 * class will be used as fallback.
146 	 * <p>Call this method if you intend to use the thread context ClassLoader
147 	 * in a scenario where you clearly prefer a non-null ClassLoader reference:
148 	 * for example, for class path resource loading (but not necessarily for
149 	 * {@code Class.forName}, which accepts a {@code null} ClassLoader
150 	 * reference as well).
151 	 * @return the default ClassLoader (only {@code null} if even the system
152 	 * ClassLoader isn't accessible)
153 	 * @see Thread#getContextClassLoader()
154 	 * @see ClassLoader#getSystemClassLoader()
155 	 */
156 	public static ClassLoader getDefaultClassLoader() {
157 		ClassLoader cl = null;
158 		try {
159 			cl = Thread.currentThread().getContextClassLoader();
160 		}
161 		catch (Throwable ex) {
162 			// Cannot access thread context ClassLoader - falling back...
163 		}
164 		if (cl == null) {
165 			// No thread context class loader -> use class loader of this class.
166 			cl = ClassUtils.class.getClassLoader();
167 			if (cl == null) {
168 				// getClassLoader() returning null indicates the bootstrap ClassLoader
169 				try {
170 					cl = ClassLoader.getSystemClassLoader();
171 				}
172 				catch (Throwable ex) {
173 					// Cannot access system ClassLoader - oh well, maybe the caller can live with null...
174 				}
175 			}
176 		}
177 		return cl;
178 	}

180 	/**
181 	 * Override the thread context ClassLoader with the environment's bean ClassLoader
182 	 * if necessary, i.e. if the bean ClassLoader is not equivalent to the thread
183 	 * context ClassLoader already.
184 	 * @param classLoaderToUse the actual ClassLoader to use for the thread context
185 	 * @return the original thread context ClassLoader, or {@code null} if not overridden
186 	 */
187 	public static ClassLoader overrideThreadContextClassLoader(ClassLoader classLoaderToUse) {
188 		Thread currentThread = Thread.currentThread();
189 		ClassLoader threadContextClassLoader = currentThread.getContextClassLoader();
190 		if (classLoaderToUse != null && !classLoaderToUse.equals(threadContextClassLoader)) {
191 			currentThread.setContextClassLoader(classLoaderToUse);
192 			return threadContextClassLoader;
193 		}
194 		else {
195 			return null;
196 		}
197 	}

199 	/**
200 	 * Replacement for {@code Class.forName()} that also returns Class instances
201 	 * for primitives (e.g. "int") and array class names (e.g. "String[]").
202 	 * Furthermore, it is also capable of resolving inner class names in Java source
203 	 * style (e.g. "java.lang.Thread.State" instead of "java.lang.Thread$State").
204 	 * @param name the name of the Class
205 	 * @param classLoader the class loader to use
206 	 * (may be {@code null}, which indicates the default class loader)
207 	 * @return Class instance for the supplied name
208 	 * @throws ClassNotFoundException if the class was not found
209 	 * @throws LinkageError if the class file could not be loaded
210 	 * @see Class#forName(String, boolean, ClassLoader)
211 	 */
212 	public static Class<?> forName(String name, ClassLoader classLoader) throws ClassNotFoundException, LinkageError {
213 		Assert.notNull(name, "Name must not be null");
214 
215 		Class<?> clazz = resolvePrimitiveClassName(name);
216 		if (clazz == null) {
217 			clazz = commonClassCache.get(name);
218 		}
219 		if (clazz != null) {
220 			return clazz;
221 		}
222 
223 		// "java.lang.String[]" style arrays
224 		if (name.endsWith(ARRAY_SUFFIX)) {
225 			String elementClassName = name.substring(0, name.length() - ARRAY_SUFFIX.length());
226 			Class<?> elementClass = forName(elementClassName, classLoader);
227 			return Array.newInstance(elementClass, 0).getClass();
228 		}
229 
230 		// "[Ljava.lang.String;" style arrays
231 		if (name.startsWith(NON_PRIMITIVE_ARRAY_PREFIX) && name.endsWith(";")) {
232 			String elementName = name.substring(NON_PRIMITIVE_ARRAY_PREFIX.length(), name.length() - 1);
233 			Class<?> elementClass = forName(elementName, classLoader);
234 			return Array.newInstance(elementClass, 0).getClass();
235 		}
236 
237 		// "[[I" or "[[Ljava.lang.String;" style arrays
238 		if (name.startsWith(INTERNAL_ARRAY_PREFIX)) {
239 			String elementName = name.substring(INTERNAL_ARRAY_PREFIX.length());
240 			Class<?> elementClass = forName(elementName, classLoader);
241 			return Array.newInstance(elementClass, 0).getClass();
242 		}
243 
244 		ClassLoader clToUse = classLoader;
245 		if (clToUse == null) {
246 			clToUse = getDefaultClassLoader();
247 		}
248 		try {
249 			return (clToUse != null ? clToUse.loadClass(name) : Class.forName(name));
250 		}
251 		catch (ClassNotFoundException ex) {
252 			int lastDotIndex = name.lastIndexOf(PACKAGE_SEPARATOR);
253 			if (lastDotIndex != -1) {
254 				String innerClassName =
255 						name.substring(0, lastDotIndex) + INNER_CLASS_SEPARATOR + name.substring(lastDotIndex + 1);
256 				try {
257 					return (clToUse != null ? clToUse.loadClass(innerClassName) : Class.forName(innerClassName));
258 				}
259 				catch (ClassNotFoundException ex2) {
260 					// Swallow - let original exception get through
261 				}
262 			}
263 			throw ex;
264 		}
265 	}

267 	/**
268 	 * Resolve the given class name into a Class instance. Supports
269 	 * primitives (like "int") and array class names (like "String[]").
270 	 * <p>This is effectively equivalent to the {@code forName}
271 	 * method with the same arguments, with the only difference being
272 	 * the exceptions thrown in case of class loading failure.
273 	 * @param className the name of the Class
274 	 * @param classLoader the class loader to use
275 	 * (may be {@code null}, which indicates the default class loader)
276 	 * @return Class instance for the supplied name
277 	 * @throws IllegalArgumentException if the class name was not resolvable
278 	 * (that is, the class could not be found or the class file could not be loaded)
279 	 * @see #forName(String, ClassLoader)
280 	 */
281 	public static Class<?> resolveClassName(String className, ClassLoader classLoader) throws IllegalArgumentException {
282 		try {
283 			return forName(className, classLoader);
284 		}
285 		catch (ClassNotFoundException ex) {
286 			throw new IllegalArgumentException("Cannot find class [" + className + "]", ex);
287 		}
288 		catch (LinkageError ex) {
289 			throw new IllegalArgumentException(
290 					"Error loading class [" + className + "]: problem with class file or dependent class.", ex);
291 		}
292 	}

295 	 * Resolve the given class name as primitive class, if appropriate,
296 	 * according to the JVM's naming rules for primitive classes.
297 	 * <p>Also supports the JVM's internal class names for primitive arrays.
298 	 * Does <i>not</i> support the "[]" suffix notation for primitive arrays;
299 	 * this is only supported by {@link #forName(String, ClassLoader)}.
300 	 * @param name the name of the potentially primitive class
301 	 * @return the primitive class, or {@code null} if the name does not denote
302 	 * a primitive class or primitive array class
303 	 */
304 	public static Class<?> resolvePrimitiveClassName(String name) {
305 		Class<?> result = null;
306 		// Most class names will be quite long, considering that they
307 		// SHOULD sit in a package, so a length check is worthwhile.
308 		if (name != null && name.length() <= 8) {
309 			// Could be a primitive - likely.
310 			result = primitiveTypeNameMap.get(name);
311 		}
312 		return result;
313 	}

348 	 * Return the user-defined class for the given class: usually simply the given
349 	 * class, but the original class in case of a CGLIB-generated subclass.
350 	 * @param clazz the class to check
351 	 * @return the user-defined class
352 	 */
353 	public static Class<?> getUserClass(Class<?> clazz) {
354 		if (clazz != null && clazz.getName().contains(CGLIB_CLASS_SEPARATOR)) {
355 			Class<?> superClass = clazz.getSuperclass();
356 			if (superClass != null && !Object.class.equals(superClass)) {
357 				return superClass;
358 			}
359 		}
360 		return clazz;
361 	}

363 	/**
364 	 * Check whether the given class is cache-safe in the given context,
365 	 * i.e. whether it is loaded by the given ClassLoader or a parent of it.
366 	 * @param clazz the class to analyze
367 	 * @param classLoader the ClassLoader to potentially cache metadata in
368 	 */
369 	public static boolean isCacheSafe(Class<?> clazz, ClassLoader classLoader) {
370 		Assert.notNull(clazz, "Class must not be null");
371 		try {
372 			ClassLoader target = clazz.getClassLoader();
373 			if (target == null) {
374 				return true;
375 			}
376 			ClassLoader cur = classLoader;
377 			if (cur == target) {
378 				return true;
379 			}
380 			while (cur != null) {
381 				cur = cur.getParent();
382 				if (cur == target) {
383 					return true;
384 				}
385 			}
386 			return false;
387 		}
388 		catch (SecurityException ex) {
389 			// Probably from the system ClassLoader - let's consider it safe.
390 			return true;
391 		}
392 	}

396 	 * Get the class name without the qualified package name.
397 	 * @param className the className to get the short name for
398 	 * @return the class name of the class without the package name
399 	 * @throws IllegalArgumentException if the className is empty
400 	 */
401 	public static String getShortName(String className) {
402 		Assert.hasLength(className, "Class name must not be empty");
403 		int lastDotIndex = className.lastIndexOf(PACKAGE_SEPARATOR);
404 		int nameEndIndex = className.indexOf(CGLIB_CLASS_SEPARATOR);
405 		if (nameEndIndex == -1) {
406 			nameEndIndex = className.length();
407 		}
408 		String shortName = className.substring(lastDotIndex + 1, nameEndIndex);
409 		shortName = shortName.replace(INNER_CLASS_SEPARATOR, PACKAGE_SEPARATOR);
410 		return shortName;
411 	}

491 	 * Build a nice qualified name for an array:
492 	 * component type class name + "[]".
493 	 * @param clazz the array class
494 	 * @return a qualified name for the array class
495 	 */
496 	private static String getQualifiedNameForArray(Class<?> clazz) {
497 		StringBuilder result = new StringBuilder();
498 		while (clazz.isArray()) {
499 			clazz = clazz.getComponentType();
500 			result.append(ClassUtils.ARRAY_SUFFIX);
501 		}
502 		result.insert(0, clazz.getName());
503 		return result.toString();
504 	}

517 	/**
518 	 * Return a descriptive name for the given object's type: usually simply
519 	 * the class name, but component type class name + "[]" for arrays,
520 	 * and an appended list of implemented interfaces for JDK proxies.
521 	 * @param value the value to introspect
522 	 * @return the qualified name of the class
523 	 */
524 	public static String getDescriptiveType(Object value) {
525 		if (value == null) {
526 			return null;
527 		}
528 		Class<?> clazz = value.getClass();
529 		if (Proxy.isProxyClass(clazz)) {
530 			StringBuilder result = new StringBuilder(clazz.getName());
531 			result.append(" implementing ");
532 			Class<?>[] ifcs = clazz.getInterfaces();
533 			for (int i = 0; i < ifcs.length; i++) {
534 				result.append(ifcs[i].getName());
535 				if (i < ifcs.length - 1) {
536 					result.append(',');
537 				}
538 			}
539 			return result.toString();
540 		}
541 		else if (clazz.isArray()) {
542 			return getQualifiedNameForArray(clazz);
543 		}
544 		else {
545 			return clazz.getName();
546 		}
547 	}

607 	 * and return it if available (else throws an {@code IllegalStateException}).
608 	 * <p>In case of any signature specified, only returns the method if there is a
609 	 * unique candidate, i.e. a single public method with the specified name.
610 	 * <p>Essentially translates {@code NoSuchMethodException} to {@code IllegalStateException}.
611 	 * @param clazz the clazz to analyze
612 	 * @param methodName the name of the method
613 	 * @param paramTypes the parameter types of the method
614 	 * (may be {@code null} to indicate any signature)
615 	 * @return the method (never {@code null})
616 	 * @throws IllegalStateException if the method has not been found
617 	 * @see Class#getMethod
618 	 */
619 	public static Method getMethod(Class<?> clazz, String methodName, Class<?>... paramTypes) {
620 		Assert.notNull(clazz, "Class must not be null");
621 		Assert.notNull(methodName, "Method name must not be null");
622 		if (paramTypes != null) {
623 			try {
624 				return clazz.getMethod(methodName, paramTypes);
625 			}
626 			catch (NoSuchMethodException ex) {
627 				throw new IllegalStateException("Expected method not found: " + ex);
628 			}
629 		}
630 		else {
631 			Set<Method> candidates = new HashSet<Method>(1);
632 			Method[] methods = clazz.getMethods();
633 			for (Method method : methods) {
634 				if (methodName.equals(method.getName())) {
635 					candidates.add(method);
636 				}
637 			}
638 			if (candidates.size() == 1) {
639 				return candidates.iterator().next();
640 			}
641 			else if (candidates.isEmpty()) {
642 				throw new IllegalStateException("Expected method not found: " + clazz + "." + methodName);
643 			}
644 			else {
645 				throw new IllegalStateException("No unique method found: " + clazz + "." + methodName);
646 			}
647 		}
648 	}

743 	 * Given a method, which may come from an interface, and a target class used
744 	 * in the current reflective invocation, find the corresponding target method
745 	 * if there is one. E.g. the method may be {@code IFoo.bar()} and the
746 	 * target class may be {@code DefaultFoo}. In this case, the method may be
747 	 * {@code DefaultFoo.bar()}. This enables attributes on that method to be found.
748 	 * <p><b>NOTE:</b> In contrast to {@link org.springframework.aop.support.AopUtils#getMostSpecificMethod},
749 	 * this method does <i>not</i> resolve Java 5 bridge methods automatically.
750 	 * Call {@link org.springframework.core.BridgeMethodResolver#findBridgedMethod}
751 	 * if bridge method resolution is desirable (e.g. for obtaining metadata from
752 	 * the original method definition).
753 	 * <p><b>NOTE:</b> Since Spring 3.1.1, if Java security settings disallow reflective
754 	 * access (e.g. calls to {@code Class#getDeclaredMethods} etc, this implementation
755 	 * will fall back to returning the originally provided method.
756 	 * @param method the method to be invoked, which may come from an interface
757 	 * @param targetClass the target class for the current invocation.
758 	 * May be {@code null} or may not even implement the method.
759 	 * @return the specific target method, or the original method if the
760 	 * {@code targetClass} doesn't implement it or is {@code null}
761 	 */
762 	public static Method getMostSpecificMethod(Method method, Class<?> targetClass) {
763 		if (method != null && isOverridable(method, targetClass) &&
764 				targetClass != null && !targetClass.equals(method.getDeclaringClass())) {
765 			try {
766 				if (Modifier.isPublic(method.getModifiers())) {
767 					try {
768 						return targetClass.getMethod(method.getName(), method.getParameterTypes());
769 					}
770 					catch (NoSuchMethodException ex) {
771 						return method;
772 					}
773 				}
774 				else {
775 					Method specificMethod =
776 							ReflectionUtils.findMethod(targetClass, method.getName(), method.getParameterTypes());
777 					return (specificMethod != null ? specificMethod : method);
778 				}
779 			}
780 			catch (SecurityException ex) {
781 				// Security settings are disallowing reflective access; fall back to 'method' below.
782 			}
783 		}
784 		return method;
785 	}

1164 	/**
1165 	 * Determine the common ancestor of the given classes, if any.
1166 	 * @param clazz1 the class to introspect
1167 	 * @param clazz2 the other class to introspect
1168 	 * @return the common ancestor (i.e. common superclass, one interface
1169 	 * extending the other), or {@code null} if none found. If any of the
1170 	 * given classes is {@code null}, the other class will be returned.
1171 	 * @since 3.2.6
1172 	 */
1173 	public static Class<?> determineCommonAncestor(Class<?> clazz1, Class<?> clazz2) {
1174 		if (clazz1 == null) {
1175 			return clazz2;
1176 		}
1177 		if (clazz2 == null) {
1178 			return clazz1;
1179 		}
1180 		if (clazz1.isAssignableFrom(clazz2)) {
1181 			return clazz1;
1182 		}
1183 		if (clazz2.isAssignableFrom(clazz1)) {
1184 			return clazz2;
1185 		}
1186 		Class<?> ancestor = clazz1;
1187 		do {
1188 			ancestor = ancestor.getSuperclass();
1189 			if (ancestor == null || Object.class.equals(ancestor)) {
1190 				return null;
1191 			}
1192 		}
1193 		while (!ancestor.isAssignableFrom(clazz2));
1194 		return ancestor;
1195 	}
