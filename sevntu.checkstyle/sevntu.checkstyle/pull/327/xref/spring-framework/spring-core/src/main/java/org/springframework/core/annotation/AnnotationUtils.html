40  /**
41   * General utility methods for working with annotations, handling bridge methods
42   * (which the compiler generates for generic declarations) as well as super methods
43   * (for optional &quot;annotation inheritance&quot;). Note that none of this is
44   * provided by the JDK's introspection facilities themselves.
45   *
46   * <p>As a general rule for runtime-retained annotations (e.g. for transaction
47   * control, authorization, or service exposure), always use the lookup methods
48   * on this class (e.g., {@link #findAnnotation(Method, Class)},
49   * {@link #getAnnotation(Method, Class)}, and {@link #getAnnotations(Method)})
50   * instead of the plain annotation lookup methods in the JDK. You can still
51   * explicitly choose between a <em>get</em> lookup on the given class level only
52   * ({@link #getAnnotation(Method, Class)}) and a <em>find</em> lookup in the entire
53   * inheritance hierarchy of the given method ({@link #findAnnotation(Method, Class)}).
54   *
55   * @author Rob Harrop
56   * @author Juergen Hoeller
57   * @author Sam Brannen
58   * @author Mark Fisher
59   * @author Chris Beams
60   * @author Phillip Webb
61   * @since 2.0
62   * @see java.lang.reflect.Method#getAnnotations()
63   * @see java.lang.reflect.Method#getAnnotation(Class)
64   */
65  public abstract class AnnotationUtils {

81  	 * Get a single {@link Annotation} of {@code annotationType} from the supplied
82  	 * annotation: either the given annotation itself or a meta-annotation thereof.
83  	 * @param ann the Annotation to check
84  	 * @param annotationType the annotation type to look for, both locally and as a meta-annotation
85  	 * @return the matching annotation, or {@code null} if none found
86  	 * @since 4.0
87  	 */
88  	@SuppressWarnings("unchecked")
89  	public static <T extends Annotation> T getAnnotation(Annotation ann, Class<T> annotationType) {
90  		if (annotationType.isInstance(ann)) {
91  			return (T) ann;
92  		}
93  		try {
94  			return ann.annotationType().getAnnotation(annotationType);
95  		}
96  		catch (Exception ex) {
97  			// Assuming nested Class values not resolvable within annotation attributes...
98  			logIntrospectionFailure(ann.annotationType(), ex);
99  			return null;
100 		}
101 	}

104 	 * Get a single {@link Annotation} of {@code annotationType} from the supplied
105 	 * Method, Constructor or Field. Meta-annotations will be searched if the annotation
106 	 * is not declared locally on the supplied element.
107 	 * @param annotatedElement the Method, Constructor or Field from which to get the annotation
108 	 * @param annotationType the annotation type to look for, both locally and as a meta-annotation
109 	 * @return the matching annotation, or {@code null} if none found
110 	 * @since 3.1
111 	 */
112 	public static <T extends Annotation> T getAnnotation(AnnotatedElement annotatedElement, Class<T> annotationType) {
113 		try {
114 			T ann = annotatedElement.getAnnotation(annotationType);
115 			if (ann == null) {
116 				for (Annotation metaAnn : annotatedElement.getAnnotations()) {
117 					ann = metaAnn.annotationType().getAnnotation(annotationType);
118 					if (ann != null) {
119 						break;
120 					}
121 				}
122 			}
123 			return ann;
124 		}
125 		catch (Exception ex) {
126 			// Assuming nested Class values not resolvable within annotation attributes...
127 			logIntrospectionFailure(annotatedElement, ex);
128 			return null;
129 		}
130 	}

204 	 * <p>Correctly handles bridge {@link Method Methods} generated by the compiler.

206 	 * @param containerAnnotationType the class of the container that holds the annotations
207 	 * @param annotationType the annotation type to look for

213 			Class<? extends Annotation> containerAnnotationType, Class<A> annotationType) {
214 
215 		try {
216 			if (annotatedElement.getAnnotations().length > 0) {
217 				return new AnnotationCollector<A>(containerAnnotationType, annotationType).getResult(annotatedElement);
218 			}
219 		}
220 		catch (Exception ex) {
221 			// Assuming nested Class values not resolvable within annotation attributes...
222 			logIntrospectionFailure(annotatedElement, ex);
223 		}
224 		return Collections.emptySet();
225 	}

227 	/**
228 	 * Find a single {@link Annotation} of {@code annotationType} from the supplied
229 	 * {@link Method}, traversing its super methods (i.e., from superclasses and
230 	 * interfaces) if no annotation can be found on the given method itself.
231 	 * <p>Annotations on methods are not inherited by default, so we need to handle
232 	 * this explicitly.
233 	 * @param method the method to look for annotations on
234 	 * @param annotationType the annotation type to look for
235 	 * @return the annotation found, or {@code null} if none
236 	 */
237 	@SuppressWarnings("unchecked")
238 	public static <A extends Annotation> A findAnnotation(Method method, Class<A> annotationType) {
239 		AnnotationCacheKey cacheKey = new AnnotationCacheKey(method, annotationType);
240 		A result = (A) findAnnotationCache.get(cacheKey);
241 		if (result == null) {
242 			result = getAnnotation(method, annotationType);
243 			Class<?> clazz = method.getDeclaringClass();
244 			if (result == null) {
245 				result = searchOnInterfaces(method, annotationType, clazz.getInterfaces());
246 			}
247 			while (result == null) {
248 				clazz = clazz.getSuperclass();
249 				if (clazz == null || clazz.equals(Object.class)) {
250 					break;
251 				}
252 				try {
253 					Method equivalentMethod = clazz.getDeclaredMethod(method.getName(), method.getParameterTypes());
254 					result = getAnnotation(equivalentMethod, annotationType);
255 				}
256 				catch (NoSuchMethodException ex) {
257 					// No equivalent method found
258 				}
259 				if (result == null) {
260 					result = searchOnInterfaces(method, annotationType, clazz.getInterfaces());
261 				}
262 			}
263 			if (result != null) {
264 				findAnnotationCache.put(cacheKey, result);
265 			}
266 		}
267 		return result;
268 	}
269 
270 	private static <A extends Annotation> A searchOnInterfaces(Method method, Class<A> annotationType, Class<?>... ifcs) {
271 		A annotation = null;
272 		for (Class<?> iface : ifcs) {
273 			if (isInterfaceWithAnnotatedMethods(iface)) {
274 				try {
275 					Method equivalentMethod = iface.getMethod(method.getName(), method.getParameterTypes());
276 					annotation = getAnnotation(equivalentMethod, annotationType);
277 				}
278 				catch (NoSuchMethodException ex) {
279 					// Skip this interface - it doesn't have the method...
280 				}
281 				if (annotation != null) {
282 					break;
283 				}
284 			}
285 		}
286 		return annotation;
287 	}

289 	private static boolean isInterfaceWithAnnotatedMethods(Class<?> iface) {
290 		Boolean flag = annotatedInterfaceCache.get(iface);
291 		if (flag != null) {
292 			return flag;
293 		}
294 		boolean found = false;
295 		for (Method ifcMethod : iface.getMethods()) {
296 			try {
297 				if (ifcMethod.getAnnotations().length > 0) {
298 					found = true;
299 					break;
300 				}
301 			}
302 			catch (Exception ex) {
303 				// Assuming nested Class values not resolvable within annotation attributes...
304 				logIntrospectionFailure(ifcMethod, ex);
305 			}
306 		}
307 		annotatedInterfaceCache.put(iface, found);
308 		return found;
309 	}

311 	/**
312 	 * Find a single {@link Annotation} of {@code annotationType} on the
313 	 * supplied {@link Class}, traversing its interfaces, annotations, and
314 	 * superclasses if the annotation is not <em>present</em> on the given class
315 	 * itself.
316 	 * <p>This method explicitly handles class-level annotations which are not
317 	 * declared as {@link java.lang.annotation.Inherited inherited} <em>as well
318 	 * as meta-annotations and annotations on interfaces</em>.
319 	 * <p>The algorithm operates as follows:
320 	 * <ol>
321 	 * <li>Search for the annotation on the given class and return it if found.
322 	 * <li>Recursively search through all annotations that the given class declares.
323 	 * <li>Recursively search through all interfaces that the given class declares.
324 	 * <li>Recursively search through the superclass hierarchy of the given class.
325 	 * </ol>
326 	 * <p>Note: in this context, the term <em>recursively</em> means that the search
327 	 * process continues by returning to step #1 with the current interface,
328 	 * annotation, or superclass as the class to look for annotations on.
329 	 * @param clazz the class to look for annotations on
330 	 * @param annotationType the type of annotation to look for
331 	 * @return the annotation if found, or {@code null} if not found
332 	 */
333 	@SuppressWarnings("unchecked")
334 	public static <A extends Annotation> A findAnnotation(Class<?> clazz, Class<A> annotationType) {
335 		AnnotationCacheKey cacheKey = new AnnotationCacheKey(clazz, annotationType);
336 		A result = (A) findAnnotationCache.get(cacheKey);
337 		if (result == null) {
338 			result = findAnnotation(clazz, annotationType, new HashSet<Annotation>());
339 			if (result != null) {
340 				findAnnotationCache.put(cacheKey, result);
341 			}
342 		}
343 		return result;
344 	}

346 	/**
347 	 * Perform the search algorithm for {@link #findAnnotation(Class, Class)},
348 	 * avoiding endless recursion by tracking which annotations have already
349 	 * been <em>visited</em>.
350 	 * @param clazz the class to look for annotations on
351 	 * @param annotationType the type of annotation to look for
352 	 * @param visited the set of annotations that have already been visited
353 	 * @return the annotation if found, or {@code null} if not found
354 	 */
355 	@SuppressWarnings("unchecked")
356 	private static <A extends Annotation> A findAnnotation(Class<?> clazz, Class<A> annotationType, Set<Annotation> visited) {
357 		Assert.notNull(clazz, "Class must not be null");
358 
359 		try {
360 			Annotation[] anns = clazz.getDeclaredAnnotations();
361 			for (Annotation ann : anns) {
362 				if (ann.annotationType().equals(annotationType)) {
363 					return (A) ann;
364 				}
365 			}
366 			for (Annotation ann : anns) {
367 				if (!isInJavaLangAnnotationPackage(ann) && visited.add(ann)) {
368 					A annotation = findAnnotation(ann.annotationType(), annotationType, visited);
369 					if (annotation != null) {
370 						return annotation;
371 					}
372 				}
373 			}
374 		}
375 		catch (Exception ex) {
376 			// Assuming nested Class values not resolvable within annotation attributes...
377 			logIntrospectionFailure(clazz, ex);
378 			return null;
379 		}
380 
381 		for (Class<?> ifc : clazz.getInterfaces()) {
382 			A annotation = findAnnotation(ifc, annotationType, visited);
383 			if (annotation != null) {
384 				return annotation;
385 			}
386 		}
387 
388 		Class<?> superclass = clazz.getSuperclass();
389 		if (superclass == null || superclass.equals(Object.class)) {
390 			return null;
391 		}
392 		return findAnnotation(superclass, annotationType, visited);
393 	}

396 	 * Find the first {@link Class} in the inheritance hierarchy of the specified {@code clazz}
397 	 * (including the specified {@code clazz} itself) which declares an annotation for the
398 	 * specified {@code annotationType}, or {@code null} if not found. If the supplied
399 	 * {@code clazz} is {@code null}, {@code null} will be returned.
400 	 * <p>If the supplied {@code clazz} is an interface, only the interface itself will be checked;
401 	 * the inheritance hierarchy for interfaces will not be traversed.
402 	 * <p>The standard {@link Class} API does not provide a mechanism for determining which class
403 	 * in an inheritance hierarchy actually declares an {@link Annotation}, so we need to handle
404 	 * this explicitly.
405 	 * @param annotationType the annotation type to look for, both locally and as a meta-annotation
406 	 * @param clazz the class on which to check for the annotation (may be {@code null})
407 	 * @return the first {@link Class} in the inheritance hierarchy of the specified {@code clazz}
408 	 * which declares an annotation for the specified {@code annotationType}, or {@code null}
409 	 * if not found
410 	 * @see Class#isAnnotationPresent(Class)
411 	 * @see Class#getDeclaredAnnotations()
412 	 * @see #findAnnotationDeclaringClassForTypes(List, Class)
413 	 * @see #isAnnotationDeclaredLocally(Class, Class)
414 	 */
415 	public static Class<?> findAnnotationDeclaringClass(Class<? extends Annotation> annotationType, Class<?> clazz) {
416 		Assert.notNull(annotationType, "Annotation type must not be null");
417 		if (clazz == null || clazz.equals(Object.class)) {
418 			return null;
419 		}
420 		if (isAnnotationDeclaredLocally(annotationType, clazz)) {
421 			return clazz;
422 		}
423 		return findAnnotationDeclaringClass(annotationType, clazz.getSuperclass());
424 	}

427 	 * Find the first {@link Class} in the inheritance hierarchy of the specified
428 	 * {@code clazz} (including the specified {@code clazz} itself) which declares
429 	 * at least one of the specified {@code annotationTypes}, or {@code null} if
430 	 * none of the specified annotation types could be found.
431 	 * <p>If the supplied {@code clazz} is {@code null}, {@code null} will be
432 	 * returned.
433 	 * <p>If the supplied {@code clazz} is an interface, only the interface itself
434 	 * will be checked; the inheritance hierarchy for interfaces will not be traversed.
435 	 * <p>The standard {@link Class} API does not provide a mechanism for determining
436 	 * which class in an inheritance hierarchy actually declares one of several
437 	 * candidate {@linkplain Annotation annotations}, so we need to handle this
438 	 * explicitly.
439 	 * @param annotationTypes the list of Class objects corresponding to the
440 	 * annotation types
441 	 * @param clazz the Class object corresponding to the class on which to check
442 	 * for the annotations, or {@code null}
443 	 * @return the first {@link Class} in the inheritance hierarchy of the specified
444 	 * {@code clazz} which declares an annotation of at least one of the specified
445 	 * {@code annotationTypes}, or {@code null} if not found
446 	 * @since 3.2.2
447 	 * @see Class#isAnnotationPresent(Class)
448 	 * @see Class#getDeclaredAnnotations()
449 	 * @see #findAnnotationDeclaringClass(Class, Class)
450 	 * @see #isAnnotationDeclaredLocally(Class, Class)
451 	 */
452 	public static Class<?> findAnnotationDeclaringClassForTypes(List<Class<? extends Annotation>> annotationTypes, Class<?> clazz) {
453 		Assert.notEmpty(annotationTypes, "The list of annotation types must not be empty");
454 		if (clazz == null || clazz.equals(Object.class)) {
455 			return null;
456 		}
457 		for (Class<? extends Annotation> annotationType : annotationTypes) {
458 			if (isAnnotationDeclaredLocally(annotationType, clazz)) {
459 				return clazz;
460 			}
461 		}
462 		return findAnnotationDeclaringClassForTypes(annotationTypes, clazz.getSuperclass());
463 	}

466 	 * Determine whether an annotation for the specified {@code annotationType} is
467 	 * declared locally on the supplied {@code clazz}. The supplied {@link Class}
468 	 * may represent any type.
469 	 * <p>Note: This method does <strong>not</strong> determine if the annotation is
470 	 * {@linkplain java.lang.annotation.Inherited inherited}. For greater clarity
471 	 * regarding inherited annotations, consider using
472 	 * {@link #isAnnotationInherited(Class, Class)} instead.
473 	 * @param annotationType the Class object corresponding to the annotation type
474 	 * @param clazz the Class object corresponding to the class on which to check for the annotation
475 	 * @return {@code true} if an annotation for the specified {@code annotationType}
476 	 * is declared locally on the supplied {@code clazz}
477 	 * @see Class#getDeclaredAnnotations()
478 	 * @see #isAnnotationInherited(Class, Class)
479 	 */
480 	public static boolean isAnnotationDeclaredLocally(Class<? extends Annotation> annotationType, Class<?> clazz) {
481 		Assert.notNull(annotationType, "Annotation type must not be null");
482 		Assert.notNull(clazz, "Class must not be null");
483 		boolean declaredLocally = false;
484 		try {
485 			for (Annotation ann : clazz.getDeclaredAnnotations()) {
486 				if (ann.annotationType().equals(annotationType)) {
487 					declaredLocally = true;
488 					break;
489 				}
490 			}
491 		}
492 		catch (Exception ex) {
493 			// Assuming nested Class values not resolvable within annotation attributes...
494 			logIntrospectionFailure(clazz, ex);
495 		}
496 		return declaredLocally;
497 	}

500 	 * Determine whether an annotation for the specified {@code annotationType} is present
501 	 * on the supplied {@code clazz} and is {@linkplain java.lang.annotation.Inherited inherited}
502 	 * (i.e., not declared locally for the class).
503 	 * <p>If the supplied {@code clazz} is an interface, only the interface itself will be checked.
504 	 * In accordance with standard meta-annotation semantics, the inheritance hierarchy for interfaces
505 	 * will not be traversed. See the {@linkplain java.lang.annotation.Inherited Javadoc} for the
506 	 * {@code @Inherited} meta-annotation for further details regarding annotation inheritance.
507 	 * @param annotationType the Class object corresponding to the annotation type
508 	 * @param clazz the Class object corresponding to the class on which to check for the annotation
509 	 * @return {@code true} if an annotation for the specified {@code annotationType} is present
510 	 * on the supplied {@code clazz} and is <em>inherited</em>
511 	 * @see Class#isAnnotationPresent(Class)
512 	 * @see #isAnnotationDeclaredLocally(Class, Class)
513 	 */
514 	public static boolean isAnnotationInherited(Class<? extends Annotation> annotationType, Class<?> clazz) {
515 		Assert.notNull(annotationType, "Annotation type must not be null");
516 		Assert.notNull(clazz, "Class must not be null");
517 		return (clazz.isAnnotationPresent(annotationType) && !isAnnotationDeclaredLocally(annotationType, clazz));
518 	}

570 	 * @param nestedAnnotationsAsMap whether to turn nested Annotation instances into
571 	 * {@link AnnotationAttributes} maps (for compatibility with
572 	 * {@link org.springframework.core.type.AnnotationMetadata} or to preserve them as
573 	 * Annotation instances
574 	 * @return the annotation attributes (a specialized Map) with attribute names as keys
575 	 * and corresponding attribute values as values
576 	 * @since 3.1.1
577 	 */
578 	public static AnnotationAttributes getAnnotationAttributes(Annotation annotation, boolean classValuesAsString,
579 			boolean nestedAnnotationsAsMap) {
580 
581 		AnnotationAttributes attrs = new AnnotationAttributes();
582 		Method[] methods = annotation.annotationType().getDeclaredMethods();
583 		for (Method method : methods) {
584 			if (method.getParameterTypes().length == 0 && method.getReturnType() != void.class) {
585 				try {
586 					Object value = method.invoke(annotation);
587 					attrs.put(method.getName(), adaptValue(value, classValuesAsString, nestedAnnotationsAsMap));
588 				}
589 				catch (Exception ex) {
590 					throw new IllegalStateException("Could not obtain annotation attribute values", ex);
591 				}
592 			}
593 		}
594 		return attrs;
595 	}

597 	/**
598 	 * Adapt the given value according to the given class and nested annotation settings.
599 	 * @param value the annotation attribute value
600 	 * @param classValuesAsString whether to turn Class references into Strings (for
601 	 * compatibility with {@link org.springframework.core.type.AnnotationMetadata}
602 	 * or to preserve them as Class references
603 	 * @param nestedAnnotationsAsMap whether to turn nested Annotation instances into
604 	 * {@link AnnotationAttributes} maps (for compatibility with
605 	 * {@link org.springframework.core.type.AnnotationMetadata} or to preserve them as
606 	 * Annotation instances
607 	 * @return the adapted value, or the original value if no adaptation is needed
608 	 */
609 	static Object adaptValue(Object value, boolean classValuesAsString, boolean nestedAnnotationsAsMap) {
610 		if (classValuesAsString) {
611 			if (value instanceof Class) {
612 				value = ((Class<?>) value).getName();
613 			}
614 			else if (value instanceof Class[]) {
615 				Class<?>[] clazzArray = (Class[]) value;
616 				String[] newValue = new String[clazzArray.length];
617 				for (int i = 0; i < clazzArray.length; i++) {
618 					newValue[i] = clazzArray[i].getName();
619 				}
620 				value = newValue;
621 			}
622 		}
623 		if (nestedAnnotationsAsMap && value instanceof Annotation) {
624 			return getAnnotationAttributes((Annotation) value, classValuesAsString, true);
625 		}
626 		else if (nestedAnnotationsAsMap && value instanceof Annotation[]) {
627 			Annotation[] realAnnotations = (Annotation[]) value;
628 			AnnotationAttributes[] mappedAnnotations = new AnnotationAttributes[realAnnotations.length];
629 			for (int i = 0; i < realAnnotations.length; i++) {
630 				mappedAnnotations[i] = getAnnotationAttributes(realAnnotations[i], classValuesAsString, true);
631 			}
632 			return mappedAnnotations;
633 		}
634 		else {
635 			return value;
636 		}
637 	}

653 	 * @param attributeName the name of the attribute value to retrieve
654 	 * @return the attribute value, or {@code null} if not found
655 	 * @see #getValue(Annotation)

657 	public static Object getValue(Annotation annotation, String attributeName) {
658 		if (annotation == null || !StringUtils.hasLength(attributeName)) {
659 			return null;

666 		catch (Exception ex) {

744

746 
747 		private final Class<? extends Annotation> annotationType;

749 		public AnnotationCacheKey(AnnotatedElement element, Class<? extends Annotation> annotationType) {

774 	private static class AnnotationCollector<A extends Annotation> {
775 
776 		private final Class<? extends Annotation> containerAnnotationType;
777 
778 		private final Class<A> annotationType;
779 
780 		private final Set<AnnotatedElement> visited = new HashSet<AnnotatedElement>();
781 
782 		private final Set<A> result = new LinkedHashSet<A>();
783 
784 		public AnnotationCollector(Class<? extends Annotation> containerAnnotationType, Class<A> annotationType) {
785 			this.containerAnnotationType = containerAnnotationType;
786 			this.annotationType = annotationType;
787 		}
788 
789 		public Set<A> getResult(AnnotatedElement element) {
790 			process(element);
791 			return Collections.unmodifiableSet(this.result);
792 		}
793 
794 		@SuppressWarnings("unchecked")
795 		private void process(AnnotatedElement annotatedElement) {
796 			if (this.visited.add(annotatedElement)) {
797 				for (Annotation ann : annotatedElement.getAnnotations()) {
798 					if (ObjectUtils.nullSafeEquals(this.annotationType, ann.annotationType())) {
799 						this.result.add((A) ann);
800 					}
801 					else if (ObjectUtils.nullSafeEquals(this.containerAnnotationType, ann.annotationType())) {
802 						this.result.addAll(getValue(ann));
803 					}
804 					else if (!isInJavaLangAnnotationPackage(ann)) {
805 						process(ann.annotationType());
806 					}
807 				}
808 			}
809 		}
810 
811 		@SuppressWarnings("unchecked")
812 		private List<A> getValue(Annotation annotation) {
813 			try {
814 				Method method = annotation.annotationType().getDeclaredMethod("value");
815 				ReflectionUtils.makeAccessible(method);
816 				return Arrays.asList((A[]) method.invoke(annotation));
817 			}
818 			catch (Exception ex) {
819 				// Unable to read value from repeating annotation container -> ignore it.
820 				return Collections.emptyList();
821 			}
822 		}
823 	}
