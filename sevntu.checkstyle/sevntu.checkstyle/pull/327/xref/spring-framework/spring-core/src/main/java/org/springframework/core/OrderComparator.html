17  package org.springframework.core;
18  
19  import java.util.Arrays;
20  import java.util.Collections;
21  import java.util.Comparator;
22  import java.util.List;
23  
24  import org.springframework.util.ObjectUtils;
25  
26  /**
27   * {@link Comparator} implementation for {@link Ordered} objects,
28   * sorting by order value ascending (resp. by priority descending).
29   *
30   * <p>Non-{@code Ordered} objects are treated as greatest order
31   * values, thus ending up at the end of the list, in arbitrary order
32   * (just like same order values of {@code Ordered} objects).
33   *
34   * @author Juergen Hoeller
35   * @since 07.04.2003
36   * @see Ordered
37   * @see java.util.Collections#sort(java.util.List, java.util.Comparator)
38   * @see java.util.Arrays#sort(Object[], java.util.Comparator)
39   */
40  public class OrderComparator implements Comparator<Object> {
41  
42  	/**
43  	 * Shared default instance of OrderComparator.
44  	 */
45  	public static final OrderComparator INSTANCE = new OrderComparator();
46  
47  
48  	/**
49  	 * Build an adapted order comparator with the given source provider.
50  	 * @param sourceProvider the order source provider to use
51  	 * @return the adapted comparator
52  	 * @since 4.1
53  	 */
54  	public Comparator<Object> withSourceProvider(final OrderSourceProvider sourceProvider) {
55  		return new Comparator<Object>() {
56  			@Override
57  			public int compare(Object o1, Object o2) {
58  				return doCompare(o1, o2, sourceProvider);
59  			}
60  		};
61  	}
62  
63  	@Override
64  	public int compare(Object o1, Object o2) {
65  		return doCompare(o1, o2, null);
66  	}
67  
68  	private int doCompare(Object o1, Object o2, OrderSourceProvider sourceProvider) {
69  		boolean p1 = (o1 instanceof PriorityOrdered);
70  		boolean p2 = (o2 instanceof PriorityOrdered);
71  		if (p1 && !p2) {
72  			return -1;
73  		}
74  		else if (p2 && !p1) {
75  			return 1;
76  		}
77  
78  		// Direct evaluation instead of Integer.compareTo to avoid unnecessary object creation.
79  		int i1 = getOrder(o1, sourceProvider);
80  		int i2 = getOrder(o2, sourceProvider);
81  		return (i1 < i2) ? -1 : (i1 > i2) ? 1 : 0;
82  	}
83  
84  	/**
85  	 * Determine the order value for the given object.
86  	 * <p>The default implementation checks against the given {@link OrderSourceProvider}
87  	 * using {@link #findOrder} and falls back to a regular {@link #getOrder(Object)} call.
88  	 * @param obj the object to check
89  	 * @return the order value, or {@code Ordered.LOWEST_PRECEDENCE} as fallback
90  	 */
91  	private int getOrder(Object obj, OrderSourceProvider sourceProvider) {
92  		Integer order = null;
93  		if (sourceProvider != null) {
94  			Object orderSource = sourceProvider.getOrderSource(obj);
95  			if (orderSource != null && orderSource.getClass().isArray()) {
96  				Object[] sources = ObjectUtils.toObjectArray(orderSource);
97  				for (Object source : sources) {
98  					order = findOrder(source);
99  					if (order != null) {
100 						break;
101 					}
102 				}
103 			}
104 			else {
105 				order = findOrder(orderSource);
106 			}
107 		}
108 		return (order != null ? order : getOrder(obj));
109 	}
110 
111 	/**
112 	 * Determine the order value for the given object.
113 	 * <p>The default implementation checks against the {@link Ordered} interface
114 	 * through delegating to {@link #findOrder}. Can be overridden in subclasses.
115 	 * @param obj the object to check
116 	 * @return the order value, or {@code Ordered.LOWEST_PRECEDENCE} as fallback
117 	 */
118 	protected int getOrder(Object obj) {
119 		Integer order = findOrder(obj);
120 		return (order != null ? order : Ordered.LOWEST_PRECEDENCE);
121 	}
122 
123 	/**
124 	 * Find an order value indicated by the given object.
125 	 * <p>The default implementation checks against the {@link Ordered} interface.
126 	 * Can be overridden in subclasses.
127 	 * @param obj the object to check
128 	 * @return the order value, or {@code null} if none found
129 	 */
130 	protected Integer findOrder(Object obj) {
131 		return (obj instanceof Ordered ? ((Ordered) obj).getOrder() : null);
132 	}
133 
134 	/**
135 	 * Determine a priority value for the given object, if any.
136 	 * <p>The default implementation always returns {@code null}.
137 	 * Subclasses may override this to give specific kinds of values a
138 	 * 'priority' characteristic, in addition to their 'order' semantics.
139 	 * A priority indicates that it may be used for selecting one object over
140 	 * another, in addition to serving for ordering purposes in a list/array.
141 	 * @param obj the object to check
142 	 * @return the priority value, or {@code null} if none
143 	 * @since 4.1
144 	 */
145 	public Integer getPriority(Object obj) {
146 		return null;
147 	}
148 
149 
150 	/**
151 	 * Sort the given List with a default OrderComparator.
152 	 * <p>Optimized to skip sorting for lists with size 0 or 1,
153 	 * in order to avoid unnecessary array extraction.
154 	 * @param list the List to sort
155 	 * @see java.util.Collections#sort(java.util.List, java.util.Comparator)
156 	 */
157 	public static void sort(List<?> list) {
158 		if (list.size() > 1) {
159 			Collections.sort(list, INSTANCE);
160 		}
161 	}
162 
163 	/**
164 	 * Sort the given array with a default OrderComparator.
165 	 * <p>Optimized to skip sorting for lists with size 0 or 1,
166 	 * in order to avoid unnecessary array extraction.
167 	 * @param array the array to sort
168 	 * @see java.util.Arrays#sort(Object[], java.util.Comparator)
169 	 */
170 	public static void sort(Object[] array) {
171 		if (array.length > 1) {
172 			Arrays.sort(array, INSTANCE);
173 		}
174 	}
175 
176 	/**
177 	 * Sort the given array or List with a default OrderComparator,
178 	 * if necessary. Simply skips sorting when given any other value.
179 	 * <p>Optimized to skip sorting for lists with size 0 or 1,
180 	 * in order to avoid unnecessary array extraction.
181 	 * @param value the array or List to sort
182 	 * @see java.util.Arrays#sort(Object[], java.util.Comparator)
183 	 */
184 	public static void sortIfNecessary(Object value) {
185 		if (value instanceof Object[]) {
186 			sort((Object[]) value);
187 		}
188 		else if (value instanceof List) {
189 			sort((List<?>) value);
190 		}
191 	}
192 
193 
194 	/**
195 	 * Strategy interface to provide an order source for a given object.
196 	 * @since 4.1
197 	 */
198 	public static interface OrderSourceProvider {
199 
200 		/**
201 		 * Return an order source for the specified object, i.e. an object that
202 		 * should be checked for an order value as a replacement to the given object.
203 		 * <p>Can also be an array of order source objects.
204 		 * <p>If the returned object does not indicate any order, the comparator
205 		 * will fall back to checking the original object.
206 		 * @param obj the object to find an order source for
207 		 * @return the order source for that object, or {@code null} if none found
208 		 */
209 		Object getOrderSource(Object obj);
210 	}
211 
212 }
