17  package org.springframework.core.convert.support;
18  
19  import java.util.Locale;
20  import java.util.UUID;
21  
22  import org.springframework.core.convert.ConversionService;
23  import org.springframework.core.convert.converter.ConverterRegistry;
24  import org.springframework.util.ClassUtils;
25  
26  /**
27   * A specialization of {@link GenericConversionService} configured by default with
28   * converters appropriate for most environments.
29   *
30   * <p>Designed for direct instantiation but also exposes the static
31   * {@link #addDefaultConverters(ConverterRegistry)} utility method for ad hoc use against any
32   * {@code ConverterRegistry} instance.
33   *
34   * @author Chris Beams
35   * @author Juergen Hoeller
36   * @author Stephane Nicoll
37   * @since 3.1
38   */
39  public class DefaultConversionService extends GenericConversionService {
40  
41  	/** Java 8's java.util.Optional class available? */
42  	private static final boolean javaUtilOptionalClassAvailable =
43  			ClassUtils.isPresent("java.util.Optional", DefaultConversionService.class.getClassLoader());
44  
45  	/** Java 8's java.time package available? */
46  	private static final boolean jsr310Available =
47  			ClassUtils.isPresent("java.time.ZoneId", DefaultConversionService.class.getClassLoader());
48  
49  	/** Java 8's java.util.stream.Stream class available? */
50  	private static final boolean streamAvailable = ClassUtils.isPresent(
51  			"java.util.stream.Stream", DefaultConversionService.class.getClassLoader());
52  
53  
54  
55  	/**
56  	 * Create a new {@code DefaultConversionService} with the set of
57  	 * {@linkplain DefaultConversionService#addDefaultConverters(ConverterRegistry) default converters}.
58  	 */
59  	public DefaultConversionService() {
60  		addDefaultConverters(this);
61  	}
62  
63  
64  	// static utility methods
65  
66  	/**
67  	 * Add converters appropriate for most environments.
68  	 * @param converterRegistry the registry of converters to add to (must also be castable to ConversionService,
69  	 * e.g. being a {@link ConfigurableConversionService})
70  	 * @throws ClassCastException if the given ConverterRegistry could not be cast to a ConversionService
71  	 */
72  	public static void addDefaultConverters(ConverterRegistry converterRegistry) {
73  		addScalarConverters(converterRegistry);
74  		addCollectionConverters(converterRegistry);
75  
76  		converterRegistry.addConverter(new ByteBufferConverter((ConversionService) converterRegistry));
77  		if (jsr310Available) {
78  			Jsr310ConverterRegistrar.registerZoneIdConverters(converterRegistry);
79  		}
80  
81  		converterRegistry.addConverter(new ObjectToObjectConverter());
82  		converterRegistry.addConverter(new IdToEntityConverter((ConversionService) converterRegistry));
83  		converterRegistry.addConverter(new FallbackObjectToStringConverter());
84  		if (javaUtilOptionalClassAvailable) {
85  			converterRegistry.addConverter(new ObjectToOptionalConverter((ConversionService) converterRegistry));
86  		}
87  	}
88  
89  	// internal helpers
90  
91  	private static void addScalarConverters(ConverterRegistry converterRegistry) {
92  		converterRegistry.addConverterFactory(new NumberToNumberConverterFactory());
93  
94  		converterRegistry.addConverterFactory(new StringToNumberConverterFactory());
95  		converterRegistry.addConverter(Number.class, String.class, new ObjectToStringConverter());
96  
97  		converterRegistry.addConverter(new StringToCharacterConverter());
98  		converterRegistry.addConverter(Character.class, String.class, new ObjectToStringConverter());
99  
100 		converterRegistry.addConverter(new NumberToCharacterConverter());
101 		converterRegistry.addConverterFactory(new CharacterToNumberFactory());
102 
103 		converterRegistry.addConverter(new StringToBooleanConverter());
104 		converterRegistry.addConverter(Boolean.class, String.class, new ObjectToStringConverter());
105 
106 		converterRegistry.addConverterFactory(new StringToEnumConverterFactory());
107 		converterRegistry.addConverter(Enum.class, String.class,
108 				new EnumToStringConverter((ConversionService) converterRegistry));
109 
110 		converterRegistry.addConverter(new StringToLocaleConverter());
111 		converterRegistry.addConverter(Locale.class, String.class, new ObjectToStringConverter());
112 
113 		converterRegistry.addConverter(new StringToPropertiesConverter());
114 		converterRegistry.addConverter(new PropertiesToStringConverter());
115 
116 		converterRegistry.addConverter(new StringToUUIDConverter());
117 		converterRegistry.addConverter(UUID.class, String.class, new ObjectToStringConverter());
118 	}
119 
120 	private static void addCollectionConverters(ConverterRegistry converterRegistry) {
121 		ConversionService conversionService = (ConversionService) converterRegistry;
122 
123 		converterRegistry.addConverter(new ArrayToCollectionConverter(conversionService));
124 		converterRegistry.addConverter(new CollectionToArrayConverter(conversionService));
125 
126 		converterRegistry.addConverter(new ArrayToArrayConverter(conversionService));
127 		converterRegistry.addConverter(new CollectionToCollectionConverter(conversionService));
128 		converterRegistry.addConverter(new MapToMapConverter(conversionService));
129 
130 		converterRegistry.addConverter(new ArrayToStringConverter(conversionService));
131 		converterRegistry.addConverter(new StringToArrayConverter(conversionService));
132 
133 		converterRegistry.addConverter(new ArrayToObjectConverter(conversionService));
134 		converterRegistry.addConverter(new ObjectToArrayConverter(conversionService));
135 
136 		converterRegistry.addConverter(new CollectionToStringConverter(conversionService));
137 		converterRegistry.addConverter(new StringToCollectionConverter(conversionService));
138 
139 		converterRegistry.addConverter(new CollectionToObjectConverter(conversionService));
140 		converterRegistry.addConverter(new ObjectToCollectionConverter(conversionService));
141 
142 		if (streamAvailable) {
143 			converterRegistry.addConverter(new StreamConverter(conversionService));
144 		}
145 	}
146 
147 
148 	/**
149 	 * Inner class to avoid a hard-coded dependency on Java 8's {@code java.time} package.
150 	 */
151 	private static final class Jsr310ConverterRegistrar {
152 
153 		public static void registerZoneIdConverters(ConverterRegistry converterRegistry) {
154 			converterRegistry.addConverter(new ZoneIdToTimeZoneConverter());
155 			converterRegistry.addConverter(new ZonedDateTimeToCalendarConverter());
156 		}
157 	}
158
