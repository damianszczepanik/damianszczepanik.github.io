17  package org.springframework.tests;
18  
19  import java.util.Collections;
20  import java.util.EnumSet;
21  import java.util.HashSet;
22  import java.util.Set;
23  
24  import org.junit.Rule;
25  import org.junit.Test;
26  import org.junit.rules.ExpectedException;
27  
28  import static org.hamcrest.Matchers.*;
29  import static org.junit.Assert.*;
30  
31  /**
32   * Tests for {@link TestGroup}.
33   *
34   * @author Phillip Webb
35   */
36  public class TestGroupTests {
37  
38  	@Rule
39  	public ExpectedException thrown = ExpectedException.none();
40  
41  	@Test
42  	public void parseNull() throws Exception {
43  		assertThat(TestGroup.parse(null), equalTo(Collections.<TestGroup> emptySet()));
44  	}
45  
46  	@Test
47  	public void parseEmptyString() throws Exception {
48  		assertThat(TestGroup.parse(""), equalTo(Collections.<TestGroup> emptySet()));
49  	}
50  
51  	@Test
52  	public void parseWithSpaces() throws Exception {
53  		assertThat(TestGroup.parse("PERFORMANCE,  PERFORMANCE"),
54  				equalTo((Set<TestGroup>) EnumSet.of(TestGroup.PERFORMANCE)));
55  	}
56  
57  	@Test
58  	public void parseInMixedCase() throws Exception {
59  		assertThat(TestGroup.parse("performance,  PERFormaNCE"),
60  				equalTo((Set<TestGroup>) EnumSet.of(TestGroup.PERFORMANCE)));
61  	}
62  
63  	@Test
64  	public void parseMissing() throws Exception {
65  		thrown.expect(IllegalArgumentException.class);
66  		thrown.expectMessage("Unable to find test group 'missing' when parsing " +
67  				"testGroups value: 'performance, missing'. Available groups include: " +
68  				"[LONG_RUNNING,PERFORMANCE,JMXMP,CI,CUSTOM_COMPILATION]");
69  		TestGroup.parse("performance, missing");
70  	}
71  
72  	@Test
73  	public void parseAll() throws Exception {
74  		assertThat(TestGroup.parse("all"), equalTo((Set<TestGroup>)EnumSet.allOf(TestGroup.class)));
75  	}
76  
77  	@Test
78  	public void parseAllExcept() throws Exception {
79  		Set<TestGroup> expected = new HashSet<TestGroup>(EnumSet.allOf(TestGroup.class));
80  		expected.remove(TestGroup.CUSTOM_COMPILATION);
81  		expected.remove(TestGroup.PERFORMANCE);
82  		assertThat(TestGroup.parse("all-custom_compilation,performance"), equalTo(expected));
83  	}
84  
85  }
