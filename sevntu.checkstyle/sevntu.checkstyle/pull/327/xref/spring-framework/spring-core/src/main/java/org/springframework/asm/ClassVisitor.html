30  package org.springframework.asm;
31  
32  /**
33   * A visitor to visit a Java class. The methods of this class must be called in
34   * the following order: <tt>visit</tt> [ <tt>visitSource</tt> ] [
35   * <tt>visitOuterClass</tt> ] ( <tt>visitAnnotation</tt> |
36   * <tt>visitTypeAnnotation</tt> | <tt>visitAttribute</tt> )* (
37   * <tt>visitInnerClass</tt> | <tt>visitField</tt> | <tt>visitMethod</tt> )*
38   * <tt>visitEnd</tt>.
39   * 
40   * @author Eric Bruneton
41   */
42  public abstract class ClassVisitor {
43  
44      /**
45       * The ASM API version implemented by this visitor. The value of this field
46       * must be one of {@link Opcodes#ASM4} or {@link Opcodes#ASM5}.
47       */
48      protected final int api;
49  
50      /**
51       * The class visitor to which this visitor must delegate method calls. May
52       * be null.
53       */
54      protected ClassVisitor cv;
55  
56      /**
57       * Constructs a new {@link ClassVisitor}.
58       * 
59       * @param api
60       *            the ASM API version implemented by this visitor. Must be one
61       *            of {@link Opcodes#ASM4} or {@link Opcodes#ASM5}.
62       */
63      public ClassVisitor(final int api) {
64          this(api, null);
65      }
66  
67      /**
68       * Constructs a new {@link ClassVisitor}.
69       * 
70       * @param api
71       *            the ASM API version implemented by this visitor. Must be one
72       *            of {@link Opcodes#ASM4} or {@link Opcodes#ASM5}.
73       * @param cv
74       *            the class visitor to which this visitor must delegate method
75       *            calls. May be null.
76       */
77      public ClassVisitor(final int api, final ClassVisitor cv) {
78          if (api != Opcodes.ASM4 && api != Opcodes.ASM5) {
79              throw new IllegalArgumentException();
80          }
81          this.api = api;
82          this.cv = cv;
83      }
84  
85      /**
86       * Visits the header of the class.
87       * 
88       * @param version
89       *            the class version.
90       * @param access
91       *            the class's access flags (see {@link Opcodes}). This parameter
92       *            also indicates if the class is deprecated.
93       * @param name
94       *            the internal name of the class (see
95       *            {@link Type#getInternalName() getInternalName}).
96       * @param signature
97       *            the signature of this class. May be <tt>null</tt> if the class
98       *            is not a generic one, and does not extend or implement generic
99       *            classes or interfaces.
100      * @param superName
101      *            the internal of name of the super class (see
102      *            {@link Type#getInternalName() getInternalName}). For
103      *            interfaces, the super class is {@link Object}. May be
104      *            <tt>null</tt>, but only for the {@link Object} class.
105      * @param interfaces
106      *            the internal names of the class's interfaces (see
107      *            {@link Type#getInternalName() getInternalName}). May be
108      *            <tt>null</tt>.
109      */
110     public void visit(int version, int access, String name, String signature,
111             String superName, String[] interfaces) {
112         if (cv != null) {
113             cv.visit(version, access, name, signature, superName, interfaces);
114         }
115     }
116 
117     /**
118      * Visits the source of the class.
119      * 
120      * @param source
121      *            the name of the source file from which the class was compiled.
122      *            May be <tt>null</tt>.
123      * @param debug
124      *            additional debug information to compute the correspondance
125      *            between source and compiled elements of the class. May be
126      *            <tt>null</tt>.
127      */
128     public void visitSource(String source, String debug) {
129         if (cv != null) {
130             cv.visitSource(source, debug);
131         }
132     }
133 
134     /**
135      * Visits the enclosing class of the class. This method must be called only
136      * if the class has an enclosing class.
137      * 
138      * @param owner
139      *            internal name of the enclosing class of the class.
140      * @param name
141      *            the name of the method that contains the class, or
142      *            <tt>null</tt> if the class is not enclosed in a method of its
143      *            enclosing class.
144      * @param desc
145      *            the descriptor of the method that contains the class, or
146      *            <tt>null</tt> if the class is not enclosed in a method of its
147      *            enclosing class.
148      */
149     public void visitOuterClass(String owner, String name, String desc) {
150         if (cv != null) {
151             cv.visitOuterClass(owner, name, desc);
152         }
153     }
154 
155     /**
156      * Visits an annotation of the class.
157      * 
158      * @param desc
159      *            the class descriptor of the annotation class.
160      * @param visible
161      *            <tt>true</tt> if the annotation is visible at runtime.
162      * @return a visitor to visit the annotation values, or <tt>null</tt> if
163      *         this visitor is not interested in visiting this annotation.
164      */
165     public AnnotationVisitor visitAnnotation(String desc, boolean visible) {
166         if (cv != null) {
167             return cv.visitAnnotation(desc, visible);
168         }
169         return null;
170     }
171 
172     /**
173      * Visits an annotation on a type in the class signature.
174      * 
175      * @param typeRef
176      *            a reference to the annotated type. The sort of this type
177      *            reference must be {@link TypeReference#CLASS_TYPE_PARAMETER
178      *            CLASS_TYPE_PARAMETER},
179      *            {@link TypeReference#CLASS_TYPE_PARAMETER_BOUND
180      *            CLASS_TYPE_PARAMETER_BOUND} or
181      *            {@link TypeReference#CLASS_EXTENDS CLASS_EXTENDS}. See
182      *            {@link TypeReference}.
183      * @param typePath
184      *            the path to the annotated type argument, wildcard bound, array
185      *            element type, or static inner type within 'typeRef'. May be
186      *            <tt>null</tt> if the annotation targets 'typeRef' as a whole.
187      * @param desc
188      *            the class descriptor of the annotation class.
189      * @param visible
190      *            <tt>true</tt> if the annotation is visible at runtime.
191      * @return a visitor to visit the annotation values, or <tt>null</tt> if
192      *         this visitor is not interested in visiting this annotation.
193      */
194     public AnnotationVisitor visitTypeAnnotation(int typeRef,
195             TypePath typePath, String desc, boolean visible) {
196 		/* SPRING PATCH: REMOVED FOR COMPATIBILITY WITH CGLIB 3.1
197         if (api < Opcodes.ASM5) {
198             throw new RuntimeException();
199         }
200         */
201         if (cv != null) {
202             return cv.visitTypeAnnotation(typeRef, typePath, desc, visible);
203         }
204         return null;
205     }
206 
207     /**
208      * Visits a non standard attribute of the class.
209      * 
210      * @param attr
211      *            an attribute.
212      */
213     public void visitAttribute(Attribute attr) {
214         if (cv != null) {
215             cv.visitAttribute(attr);
216         }
217     }
218 
219     /**
220      * Visits information about an inner class. This inner class is not
221      * necessarily a member of the class being visited.
222      * 
223      * @param name
224      *            the internal name of an inner class (see
225      *            {@link Type#getInternalName() getInternalName}).
226      * @param outerName
227      *            the internal name of the class to which the inner class
228      *            belongs (see {@link Type#getInternalName() getInternalName}).
229      *            May be <tt>null</tt> for not member classes.
230      * @param innerName
231      *            the (simple) name of the inner class inside its enclosing
232      *            class. May be <tt>null</tt> for anonymous inner classes.
233      * @param access
234      *            the access flags of the inner class as originally declared in
235      *            the enclosing class.
236      */
237     public void visitInnerClass(String name, String outerName,
238             String innerName, int access) {
239         if (cv != null) {
240             cv.visitInnerClass(name, outerName, innerName, access);
241         }
242     }
243 
244     /**
245      * Visits a field of the class.
246      * 
247      * @param access
248      *            the field's access flags (see {@link Opcodes}). This parameter
249      *            also indicates if the field is synthetic and/or deprecated.
250      * @param name
251      *            the field's name.
252      * @param desc
253      *            the field's descriptor (see {@link Type Type}).
254      * @param signature
255      *            the field's signature. May be <tt>null</tt> if the field's
256      *            type does not use generic types.
257      * @param value
258      *            the field's initial value. This parameter, which may be
259      *            <tt>null</tt> if the field does not have an initial value,
260      *            must be an {@link Integer}, a {@link Float}, a {@link Long}, a
261      *            {@link Double} or a {@link String} (for <tt>int</tt>,
262      *            <tt>float</tt>, <tt>long</tt> or <tt>String</tt> fields
263      *            respectively). <i>This parameter is only used for static
264      *            fields</i>. Its value is ignored for non static fields, which
265      *            must be initialized through bytecode instructions in
266      *            constructors or methods.
267      * @return a visitor to visit field annotations and attributes, or
268      *         <tt>null</tt> if this class visitor is not interested in visiting
269      *         these annotations and attributes.
270      */
271     public FieldVisitor visitField(int access, String name, String desc,
272             String signature, Object value) {
273         if (cv != null) {
274             return cv.visitField(access, name, desc, signature, value);
275         }
276         return null;
277     }
278 
279     /**
280      * Visits a method of the class. This method <i>must</i> return a new
281      * {@link MethodVisitor} instance (or <tt>null</tt>) each time it is called,
282      * i.e., it should not return a previously returned visitor.
283      * 
284      * @param access
285      *            the method's access flags (see {@link Opcodes}). This
286      *            parameter also indicates if the method is synthetic and/or
287      *            deprecated.
288      * @param name
289      *            the method's name.
290      * @param desc
291      *            the method's descriptor (see {@link Type Type}).
292      * @param signature
293      *            the method's signature. May be <tt>null</tt> if the method
294      *            parameters, return type and exceptions do not use generic
295      *            types.
296      * @param exceptions
297      *            the internal names of the method's exception classes (see
298      *            {@link Type#getInternalName() getInternalName}). May be
299      *            <tt>null</tt>.
300      * @return an object to visit the byte code of the method, or <tt>null</tt>
301      *         if this class visitor is not interested in visiting the code of
302      *         this method.
303      */
304     public MethodVisitor visitMethod(int access, String name, String desc,
305             String signature, String[] exceptions) {
306         if (cv != null) {
307             return cv.visitMethod(access, name, desc, signature, exceptions);
308         }
309         return null;
310     }
311 
312     /**
313      * Visits the end of the class. This method, which is the last one to be
314      * called, is used to inform the visitor that all the fields and methods of
315      * the class have been visited.
316      */
317     public void visitEnd() {
318         if (cv != null) {
319             cv.visitEnd();
320         }
321     }
