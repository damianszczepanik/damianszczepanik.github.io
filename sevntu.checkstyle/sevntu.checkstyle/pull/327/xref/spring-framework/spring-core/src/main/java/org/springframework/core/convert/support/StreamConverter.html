17  package org.springframework.core.convert.support;
18  
19  import java.util.Collection;
20  import java.util.HashSet;
21  import java.util.List;
22  import java.util.Set;
23  import java.util.stream.Collectors;
24  import java.util.stream.Stream;
25  
26  import org.springframework.core.convert.ConversionService;
27  import org.springframework.core.convert.TypeDescriptor;
28  import org.springframework.core.convert.converter.ConditionalGenericConverter;
29  import org.springframework.lang.UsesJava8;
30  
31  /**
32   * Converts a {@link Stream} to and from a collection or array, converting the
33   * element type if necessary.
34   *
35   * @author Stephane Nicoll
36   * @since 4.2
37   */
38  @UsesJava8
39  public class StreamConverter implements ConditionalGenericConverter {
40  
41  	private static final TypeDescriptor STREAM_TYPE = TypeDescriptor.valueOf(Stream.class);
42  
43  	private static final Set<ConvertiblePair> CONVERTIBLE_TYPES = createConvertibleTypes();
44  
45  	private final ConversionService conversionService;
46  
47  	public StreamConverter(ConversionService conversionService) {
48  		this.conversionService = conversionService;
49  	}
50  
51  	@Override
52  	public Set<ConvertiblePair> getConvertibleTypes() {
53  		return CONVERTIBLE_TYPES;
54  	}
55  
56  	@Override
57  	public boolean matches(TypeDescriptor sourceType, TypeDescriptor targetType) {
58  		if (sourceType.isAssignableTo(STREAM_TYPE)) {
59  			return matchesFromStream(sourceType.getElementTypeDescriptor(), targetType);
60  		}
61  		if (targetType.isAssignableTo(STREAM_TYPE)) {
62  			return matchesToStream(targetType.getElementTypeDescriptor(), sourceType);
63  		}
64  		return false;
65  	}
66  
67  	/**
68  	 * Validate that a {@link Collection} of the elements held within the stream can be
69  	 * converted to the specified {@code targetType}.
70  	 * @param elementType the type of the stream elements
71  	 * @param targetType the type to convert to
72  	 */
73  	public boolean matchesFromStream(TypeDescriptor elementType, TypeDescriptor targetType) {
74  		TypeDescriptor collectionOfElement = TypeDescriptor.collection(Collection.class, elementType);
75  		return this.conversionService.canConvert(collectionOfElement, targetType);
76  	}
77  
78  	/**
79  	 * Validate that the specified {@code sourceType} can be converted to a {@link Collection} of
80  	 * the type of the stream elements.
81  	 * @param elementType the type of the stream elements
82  	 * @param sourceType the type to convert from
83  	 */
84  	public boolean matchesToStream(TypeDescriptor elementType, TypeDescriptor sourceType) {
85  		TypeDescriptor collectionOfElement = TypeDescriptor.collection(Collection.class, elementType);
86  		return this.conversionService.canConvert(sourceType, collectionOfElement);
87  	}
88  
89  	@Override
90  	public Object convert(Object source, TypeDescriptor sourceType, TypeDescriptor targetType) {
91  		if (sourceType.isAssignableTo(STREAM_TYPE)) {
92  			return convertFromStream((Stream<?>) source, sourceType, targetType);
93  		}
94  		if (targetType.isAssignableTo(STREAM_TYPE)) {
95  			return convertToStream(source, sourceType, targetType);
96  		}
97  		// Should not happen
98  		throw new IllegalStateException("Unexpected source/target types");
99  	}
100 
101 	private Object convertFromStream(Stream<?> source, TypeDescriptor streamType, TypeDescriptor targetType) {
102 		List<Object> content = source.collect(Collectors.toList());
103 		TypeDescriptor listType = TypeDescriptor.collection(List.class, streamType.getElementTypeDescriptor());
104 		return this.conversionService.convert(content, listType, targetType);
105 	}
106 
107 	private Object convertToStream(Object source, TypeDescriptor sourceType, TypeDescriptor streamType) {
108 		TypeDescriptor targetCollection =
109 				TypeDescriptor.collection(List.class, streamType.getElementTypeDescriptor());
110 		List<?> target = (List<?>) this.conversionService.convert(source, sourceType, targetCollection);
111 		return target.stream();
112 	}
113 
114 
115 	private static Set<ConvertiblePair> createConvertibleTypes() {
116 		Set<ConvertiblePair> convertiblePairs = new HashSet<ConvertiblePair>();
117 		convertiblePairs.add(new ConvertiblePair(Stream.class, Collection.class));
118 		convertiblePairs.add(new ConvertiblePair(Stream.class, Object[].class));
119 		convertiblePairs.add(new ConvertiblePair(Collection.class, Stream.class));
120 		convertiblePairs.add(new ConvertiblePair(Object[].class, Stream.class));
121 		return convertiblePairs;
122 	}
123 
124 }
