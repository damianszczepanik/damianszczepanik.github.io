950 
951 		// Generic assignment can be a little confusing, given:
952 		//
953 		// List<CharSequence> c1, List<? extends CharSequence> c2, List<String> s;
954 		//
955 		// c2 = s; is allowed and is often used for argument input, for example
956 		// see List.addAll(). You can get items from c2 but you cannot add items without
957 		// getting a generic type 'is not applicable for the arguments' error. This makes
958 		// sense since if you added a StringBuffer to c2 it would break the rules on s.
959 		//
960 		// c1 = s; not allowed. Since there is no '? extends' to cause the generic
961 		// 'is not applicable for the arguments' error when adding (which would pollute
962 		// s).
963 
964 		ResolvableType objectList = ResolvableType.forField(AssignmentBase.class.getField("listo"), Assignment.class);
965 		ResolvableType charSequenceList = ResolvableType.forField(AssignmentBase.class.getField("listc"), Assignment.class);

969 		ResolvableType extendsStringList = ResolvableType.forField(AssignmentBase.class.getField("listxs"), Assignment.class);
970 
971 		assertAssignable(objectList, extendsObjectList, extendsCharSequenceList, extendsStringList).equalTo(false, false, false);
972 		assertAssignable(charSequenceList, extendsObjectList, extendsCharSequenceList, extendsStringList).equalTo(false, false, false);
973 		assertAssignable(stringList, extendsObjectList, extendsCharSequenceList, extendsStringList).equalTo(false, false, false);
974 		assertAssignable(extendsObjectList, objectList, charSequenceList, stringList).equalTo(true, true, true);
