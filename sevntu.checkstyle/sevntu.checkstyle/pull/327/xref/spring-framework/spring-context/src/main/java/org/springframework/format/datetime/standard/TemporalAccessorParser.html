17  package org.springframework.format.datetime.standard;
18  
19  import java.text.ParseException;
20  import java.time.LocalDate;
21  import java.time.LocalDateTime;
22  import java.time.LocalTime;
23  import java.time.OffsetDateTime;
24  import java.time.OffsetTime;
25  import java.time.ZonedDateTime;
26  import java.time.format.DateTimeFormatter;
27  import java.time.temporal.TemporalAccessor;
28  import java.util.Locale;
29  
30  import org.springframework.format.Parser;
31  import org.springframework.lang.UsesJava8;
32  
33  /**
34   * {@link Parser} implementation for a JSR-310 {@link java.time.temporal.TemporalAccessor},
35   * using a {@link java.time.format.DateTimeFormatter}) (the contextual one, if available).
36   *
37   * @author Juergen Hoeller
38   * @since 4.0
39   * @see DateTimeContextHolder#getFormatter
40   * @see java.time.LocalDate#parse(CharSequence, java.time.format.DateTimeFormatter)
41   * @see java.time.LocalTime#parse(CharSequence, java.time.format.DateTimeFormatter)
42   * @see java.time.LocalDateTime#parse(CharSequence, java.time.format.DateTimeFormatter)
43   * @see java.time.ZonedDateTime#parse(CharSequence, java.time.format.DateTimeFormatter)
44   * @see java.time.OffsetDateTime#parse(CharSequence, java.time.format.DateTimeFormatter)
45   * @see java.time.OffsetTime#parse(CharSequence, java.time.format.DateTimeFormatter)
46   */
47  @UsesJava8
48  public final class TemporalAccessorParser implements Parser<TemporalAccessor> {
49  
50  	private final Class<? extends TemporalAccessor> temporalAccessorType;
51  
52  	private final DateTimeFormatter formatter;
53  
54  
55  	/**
56  	 * Create a new TemporalAccessorParser for the given TemporalAccessor type.
57  	 * @param temporalAccessorType the specific TemporalAccessor class
58  	 * (LocalDate, LocalTime, LocalDateTime, ZonedDateTime, OffsetDateTime, OffsetTime)
59  	 * @param formatter the base DateTimeFormatter instance
60  	 */
61  	public TemporalAccessorParser(Class<? extends TemporalAccessor> temporalAccessorType, DateTimeFormatter formatter) {
62  		this.temporalAccessorType = temporalAccessorType;
63  		this.formatter = formatter;
64  	}
65  
66  
67  	@Override
68  	public TemporalAccessor parse(String text, Locale locale) throws ParseException {
69  		DateTimeFormatter formatterToUse = DateTimeContextHolder.getFormatter(this.formatter, locale);
70  		if (LocalDate.class.equals(this.temporalAccessorType)) {
71  			return LocalDate.parse(text, formatterToUse);
72  		}
73  		else if (LocalTime.class.equals(this.temporalAccessorType)) {
74  			return LocalTime.parse(text, formatterToUse);
75  		}
76  		else if (LocalDateTime.class.equals(this.temporalAccessorType)) {
77  			return LocalDateTime.parse(text, formatterToUse);
78  		}
79  		else if (ZonedDateTime.class.equals(this.temporalAccessorType)) {
80  			return ZonedDateTime.parse(text, formatterToUse);
81  		}
82  		else if (OffsetDateTime.class.equals(this.temporalAccessorType)) {
83  			return OffsetDateTime.parse(text, formatterToUse);
84  		}
85  		else if (OffsetTime.class.equals(this.temporalAccessorType)) {
86  			return OffsetTime.parse(text, formatterToUse);
87  		}
88  		else {
89  			throw new IllegalStateException("Unsupported TemporalAccessor type: " + this.temporalAccessorType);
90  		}
91  	}
92  
93  }
