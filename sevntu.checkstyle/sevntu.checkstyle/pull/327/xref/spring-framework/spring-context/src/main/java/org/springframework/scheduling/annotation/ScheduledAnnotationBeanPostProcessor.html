61  /**
62   * Bean post-processor that registers methods annotated with @{@link Scheduled}
63   * to be invoked by a {@link org.springframework.scheduling.TaskScheduler} according
64   * to the "fixedRate", "fixedDelay", or "cron" expression provided via the annotation.
65   *
66   * <p>This post-processor is automatically registered by Spring's
67   * {@code <task:annotation-driven>} XML element, and also by the
68   * @{@link EnableScheduling} annotation.
69   *
70   * <p>Autodetects any {@link SchedulingConfigurer} instances in the container,
71   * allowing for customization of the scheduler to be used or for fine-grained
72   * control over task registration (e.g. registration of {@link Trigger} tasks.
73   * See the @{@link EnableScheduling} javadocs for complete usage details.
74   *
75   * @author Mark Fisher
76   * @author Juergen Hoeller
77   * @author Chris Beams
78   * @since 3.0
79   * @see Scheduled
80   * @see EnableScheduling
81   * @see SchedulingConfigurer
82   * @see org.springframework.scheduling.TaskScheduler
83   * @see org.springframework.scheduling.config.ScheduledTaskRegistrar
84   */
85  public class ScheduledAnnotationBeanPostProcessor implements BeanPostProcessor, Ordered,
86  		EmbeddedValueResolverAware, BeanFactoryAware, ApplicationContextAware,
87  		SmartInitializingSingleton, ApplicationListener<ContextRefreshedEvent>, DisposableBean {
88  
89  	protected final Log logger = LogFactory.getLog(getClass());

134 	/**
135 	 * Setting an {@link ApplicationContext} is optional: If set, registered
136 	 * tasks will be activated in the {@link ContextRefreshedEvent} phase;
137 	 * if not set, it will happen at {@link #afterSingletonsInstantiated} time.
138 	 */
139 	@Override
140 	public void setApplicationContext(ApplicationContext applicationContext) {
141 		this.applicationContext = applicationContext;
142 		if (this.beanFactory == null) {
143 			this.beanFactory = applicationContext;
144 		}
145 	}

148 	@Override
149 	public void afterSingletonsInstantiated() {
150 		if (this.applicationContext == null) {
151 			// Not running in an ApplicationContext -> register tasks early...
152 			finishRegistration();
153 		}
154 	}
155 
156 	@Override
157 	public void onApplicationEvent(ContextRefreshedEvent event) {
158 		if (event.getApplicationContext() == this.applicationContext) {
159 			// Running in an ApplicationContext -> register tasks this late...
160 			// giving other ContextRefreshedEvent listeners a chance to perform
161 			// their work at the same time (e.g. Spring Batch's job registration).
162 			finishRegistration();
163 		}
164 	}
165 
166 	private void finishRegistration() {
167 		if (this.scheduler != null) {
168 			this.registrar.setScheduler(this.scheduler);
169 		}
170 
171 		if (this.beanFactory instanceof ListableBeanFactory) {
172 			Map<String, SchedulingConfigurer> configurers =
173 					((ListableBeanFactory) this.beanFactory).getBeansOfType(SchedulingConfigurer.class);
174 			for (SchedulingConfigurer configurer : configurers.values()) {
175 				configurer.configureTasks(this.registrar);
176 			}
177 		}
178 
179 		if (this.registrar.hasTasks() && this.registrar.getScheduler() == null) {
180 			Assert.state(this.beanFactory != null, "BeanFactory must be set to find scheduler by type");
181 			try {
182 				// Search for TaskScheduler bean...
183 				this.registrar.setScheduler(this.beanFactory.getBean(TaskScheduler.class));
184 			}
185 			catch (NoUniqueBeanDefinitionException ex) {
186 				throw new IllegalStateException("More than one TaskScheduler exists within the context. " +
187 						"Remove all but one of the beans; or implement the SchedulingConfigurer interface and call " +
188 						"ScheduledTaskRegistrar#setScheduler explicitly within the configureTasks() callback.", ex);
189 			}
190 			catch (NoSuchBeanDefinitionException ex) {
191 				logger.debug("Could not find default TaskScheduler bean", ex);
192 				// Search for ScheduledExecutorService bean next...
193 				try {
194 					this.registrar.setScheduler(this.beanFactory.getBean(ScheduledExecutorService.class));
195 				}
196 				catch (NoUniqueBeanDefinitionException ex2) {
197 					throw new IllegalStateException("More than one ScheduledExecutorService exists within the context. " +
198 							"Remove all but one of the beans; or implement the SchedulingConfigurer interface and call " +
199 							"ScheduledTaskRegistrar#setScheduler explicitly within the configureTasks() callback.", ex);
200 				}
201 				catch (NoSuchBeanDefinitionException ex2) {
202 					logger.debug("Could not find default ScheduledExecutorService bean", ex);
203 					// Giving up -> falling back to default scheduler within the registrar...
204 				}
205 			}
206 		}
207 
208 		this.registrar.afterPropertiesSet();
209 	}

217 	@Override
218 	public Object postProcessAfterInitialization(final Object bean, String beanName) {
219 		Class<?> targetClass = AopUtils.getTargetClass(bean);
220 		if (!this.nonAnnotatedClasses.contains(targetClass)) {
221 			final Set<Method> annotatedMethods = new LinkedHashSet<Method>(1);
222 			ReflectionUtils.doWithMethods(targetClass, new MethodCallback() {
223 				@Override
224 				public void doWith(Method method) throws IllegalArgumentException, IllegalAccessException {
225 					for (Scheduled scheduled :
226 							AnnotationUtils.getRepeatableAnnotation(method, Schedules.class, Scheduled.class)) {
227 						processScheduled(scheduled, method, bean);
228 						annotatedMethods.add(method);
229 					}
230 				}
231 			});
232 			if (annotatedMethods.isEmpty()) {
233 				this.nonAnnotatedClasses.add(targetClass);
234 				if (logger.isTraceEnabled()) {
235 					logger.trace("No @Scheduled annotations found on bean class: " + bean.getClass());
236 				}
237 			}
238 			else {
239 				// Non-empty set of methods
240 				if (logger.isDebugEnabled()) {
241 					logger.debug(annotatedMethods.size() + " @Scheduled methods processed on bean '" + beanName +
242 							"': " + annotatedMethods);
243 				}
244 			}
245 		}
246 		return bean;
247 	}
248 
249 	protected void processScheduled(Scheduled scheduled, Method method, Object bean) {
250 		try {
251 			Assert.isTrue(void.class.equals(method.getReturnType()),
252 					"Only void-returning methods may be annotated with @Scheduled");
253 			Assert.isTrue(method.getParameterTypes().length == 0,
254 					"Only no-arg methods may be annotated with @Scheduled");
255 
256 			if (AopUtils.isJdkDynamicProxy(bean)) {
257 				try {
258 					// Found a @Scheduled method on the target class for this JDK proxy ->
259 					// is it also present on the proxy itself?
260 					method = bean.getClass().getMethod(method.getName(), method.getParameterTypes());
261 				}
262 				catch (SecurityException ex) {
263 					ReflectionUtils.handleReflectionException(ex);
264 				}
265 				catch (NoSuchMethodException ex) {
266 					throw new IllegalStateException(String.format(
267 							"@Scheduled method '%s' found on bean target class '%s' but not " +
268 							"found in any interface(s) for a dynamic proxy. Either pull the " +
269 							"method up to a declared interface or switch to subclass (CGLIB) " +
270 							"proxies by setting proxy-target-class/proxyTargetClass to 'true'",
271 							method.getName(), method.getDeclaringClass().getSimpleName()));
272 				}
273 			}
274 			else if (AopUtils.isCglibProxy(bean)) {
275 				// Common problem: private methods end up in the proxy instance, not getting delegated.
276 				if (Modifier.isPrivate(method.getModifiers())) {
277 					throw new IllegalStateException(String.format(
278 							"@Scheduled method '%s' found on CGLIB proxy for target class '%s' but cannot " +
279 							"be delegated to target bean. Switch its visibility to package or protected.",
280 							method.getName(), method.getDeclaringClass().getSimpleName()));
281 				}
282 			}
283 
284 			Runnable runnable = new ScheduledMethodRunnable(bean, method);
285 			boolean processedSchedule = false;
286 			String errorMessage =
287 					"Exactly one of the 'cron', 'fixedDelay(String)', or 'fixedRate(String)' attributes is required";
288 
289 			// Determine initial delay
290 			long initialDelay = scheduled.initialDelay();
291 			String initialDelayString = scheduled.initialDelayString();
292 			if (StringUtils.hasText(initialDelayString)) {
293 				Assert.isTrue(initialDelay < 0, "Specify 'initialDelay' or 'initialDelayString', not both");
294 				if (this.embeddedValueResolver != null) {
295 					initialDelayString = this.embeddedValueResolver.resolveStringValue(initialDelayString);
296 				}
297 				try {
298 					initialDelay = Integer.parseInt(initialDelayString);
299 				}
300 				catch (NumberFormatException ex) {
301 					throw new IllegalArgumentException(
302 							"Invalid initialDelayString value \"" + initialDelayString + "\" - cannot parse into integer");
303 				}
304 			}
305 
306 			// Check cron expression
307 			String cron = scheduled.cron();
308 			if (StringUtils.hasText(cron)) {
309 				Assert.isTrue(initialDelay == -1, "'initialDelay' not supported for cron triggers");
310 				processedSchedule = true;
311 				String zone = scheduled.zone();
312 				if (this.embeddedValueResolver != null) {
313 					cron = this.embeddedValueResolver.resolveStringValue(cron);
314 					zone = this.embeddedValueResolver.resolveStringValue(zone);
315 				}
316 				TimeZone timeZone;
317 				if (StringUtils.hasText(zone)) {
318 					timeZone = StringUtils.parseTimeZoneString(zone);
319 				}
320 				else {
321 					timeZone = TimeZone.getDefault();
322 				}
323 				this.registrar.addCronTask(new CronTask(runnable, new CronTrigger(cron, timeZone)));
324 			}
325 
326 			// At this point we don't need to differentiate between initial delay set or not anymore
327 			if (initialDelay < 0) {
328 				initialDelay = 0;
329 			}
330 
331 			// Check fixed delay
332 			long fixedDelay = scheduled.fixedDelay();
333 			if (fixedDelay >= 0) {
334 				Assert.isTrue(!processedSchedule, errorMessage);
335 				processedSchedule = true;
336 				this.registrar.addFixedDelayTask(new IntervalTask(runnable, fixedDelay, initialDelay));
337 			}
338 			String fixedDelayString = scheduled.fixedDelayString();
339 			if (StringUtils.hasText(fixedDelayString)) {
340 				Assert.isTrue(!processedSchedule, errorMessage);
341 				processedSchedule = true;
342 				if (this.embeddedValueResolver != null) {
343 					fixedDelayString = this.embeddedValueResolver.resolveStringValue(fixedDelayString);
344 				}
345 				try {
346 					fixedDelay = Integer.parseInt(fixedDelayString);
347 				}
348 				catch (NumberFormatException ex) {
349 					throw new IllegalArgumentException(
350 							"Invalid fixedDelayString value \"" + fixedDelayString + "\" - cannot parse into integer");
351 				}
352 				this.registrar.addFixedDelayTask(new IntervalTask(runnable, fixedDelay, initialDelay));
353 			}
354 
355 			// Check fixed rate
356 			long fixedRate = scheduled.fixedRate();
357 			if (fixedRate >= 0) {
358 				Assert.isTrue(!processedSchedule, errorMessage);
359 				processedSchedule = true;
360 				this.registrar.addFixedRateTask(new IntervalTask(runnable, fixedRate, initialDelay));
361 			}
362 			String fixedRateString = scheduled.fixedRateString();
363 			if (StringUtils.hasText(fixedRateString)) {
364 				Assert.isTrue(!processedSchedule, errorMessage);
365 				processedSchedule = true;
366 				if (this.embeddedValueResolver != null) {
367 					fixedRateString = this.embeddedValueResolver.resolveStringValue(fixedRateString);
368 				}
369 				try {
370 					fixedRate = Integer.parseInt(fixedRateString);
371 				}
372 				catch (NumberFormatException ex) {
373 					throw new IllegalArgumentException(
374 							"Invalid fixedRateString value \"" + fixedRateString + "\" - cannot parse into integer");
375 				}
376 				this.registrar.addFixedRateTask(new IntervalTask(runnable, fixedRate, initialDelay));
377 			}
378 
379 			// Check whether we had any attribute set
380 			Assert.isTrue(processedSchedule, errorMessage);
381 		}
382 		catch (IllegalArgumentException ex) {
383 			throw new IllegalStateException(
384 					"Encountered invalid @Scheduled method '" + method.getName() + "': " + ex.getMessage());
385 		}
386 	}
