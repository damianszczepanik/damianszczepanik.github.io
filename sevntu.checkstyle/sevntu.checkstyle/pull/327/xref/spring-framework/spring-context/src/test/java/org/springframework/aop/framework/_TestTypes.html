3   /**
4    * Definitions of testing types for use in within this package.
5    * Wherever possible, test types should be defined local to the java
6    * file that makes use of them.  In some cases however, a test type may
7    * need to be shared across tests.  Such types reside here, with the
8    * intention of reducing the surface area of java files within this
9    * package.  This allows developers to think about tests first, and deal
10   * with these second class testing artifacts on an as-needed basis.
11   *
12   * Types here should be defined as package-private top level classes in
13   * order to avoid needing to fully qualify, e.g.: _TestTypes$Foo.
14   *
15   * @author Chris Beams
16   */

20  interface IEcho {
21  	int echoException(int i, Throwable t) throws Throwable;
22  	int getA();

24  }

27  class Echo implements IEcho {
28  	private int a;
29  
30  	@Override
31  	public int echoException(int i, Throwable t) throws Throwable {
32  		if (t != null)
33  			throw t;
34  		return i;
35  	}
36  	@Override
37  	public void setA(int a) {
38  		this.a = a;
39  	}
40  	@Override
41  	public int getA() {
42  		return a;
43  	}
