71  /**
72   * {@link org.springframework.beans.factory.config.BeanPostProcessor} implementation
73   * that supports common Java annotations out of the box, in particular the JSR-250
74   * annotations in the {@code javax.annotation} package. These common Java
75   * annotations are supported in many Java EE 5 technologies (e.g. JSF 1.2),
76   * as well as in Java 6's JAX-WS.
77   *
78   * <p>This post-processor includes support for the {@link javax.annotation.PostConstruct}
79   * and {@link javax.annotation.PreDestroy} annotations - as init annotation
80   * and destroy annotation, respectively - through inheriting from
81   * {@link InitDestroyAnnotationBeanPostProcessor} with pre-configured annotation types.
82   *
83   * <p>The central element is the {@link javax.annotation.Resource} annotation
84   * for annotation-driven injection of named beans, by default from the containing
85   * Spring BeanFactory, with only {@code mappedName} references resolved in JNDI.
86   * The {@link #setAlwaysUseJndiLookup "alwaysUseJndiLookup" flag} enforces JNDI lookups
87   * equivalent to standard Java EE 5 resource injection for {@code name} references
88   * and default names as well. The target beans can be simple POJOs, with no special
89   * requirements other than the type having to match.
90   *
91   * <p>The JAX-WS {@link javax.xml.ws.WebServiceRef} annotation is supported too,
92   * analogous to {@link javax.annotation.Resource} but with the capability of creating
93   * specific JAX-WS service endpoints. This may either point to an explicitly defined
94   * resource by name or operate on a locally specified JAX-WS service class. Finally,
95   * this post-processor also supports the EJB 3 {@link javax.ejb.EJB} annotation,
96   * analogous to {@link javax.annotation.Resource} as well, with the capability to
97   * specify both a local bean name and a global JNDI name for fallback retrieval.
98   * The target beans can be plain POJOs as well as EJB 3 Session Beans in this case.
99   *
100  * <p>The common annotations supported by this post-processor are available in
101  * Java 6 (JDK 1.6) as well as in Java EE 5/6 (which provides a standalone jar for
102  * its common annotations as well, allowing for use in any Java 5 based application).
103  *
104  * <p>For default usage, resolving resource names as Spring bean names,
105  * simply define the following in your application context:
106  *
107  * <pre class="code">
108  * &lt;bean class="org.springframework.context.annotation.CommonAnnotationBeanPostProcessor"/&gt;</pre>
109  *
110  * For direct JNDI access, resolving resource names as JNDI resource references
111  * within the Java EE application's "java:comp/env/" namespace, use the following:
112  *
113  * <pre class="code">
114  * &lt;bean class="org.springframework.context.annotation.CommonAnnotationBeanPostProcessor"&gt;
115  *   &lt;property name="alwaysUseJndiLookup" value="true"/&gt;
116  * &lt;/bean&gt;</pre>
117  *
118  * {@code mappedName} references will always be resolved in JNDI,
119  * allowing for global JNDI names (including "java:" prefix) as well. The
120  * "alwaysUseJndiLookup" flag just affects {@code name} references and
121  * default names (inferred from the field name / property name).
122  *
123  * <p><b>NOTE:</b> A default CommonAnnotationBeanPostProcessor will be registered
124  * by the "context:annotation-config" and "context:component-scan" XML tags.
125  * Remove or turn off the default annotation configuration there if you intend
126  * to specify a custom CommonAnnotationBeanPostProcessor bean definition!
127  * <p><b>NOTE:</b> Annotation injection will be performed <i>before</i> XML injection; thus
128  * the latter configuration will override the former for properties wired through
129  * both approaches.
130  *
131  * @author Juergen Hoeller
132  * @since 2.5
133  * @see #setAlwaysUseJndiLookup
134  * @see #setResourceFactory
135  * @see org.springframework.beans.factory.annotation.InitDestroyAnnotationBeanPostProcessor
136  * @see org.springframework.beans.factory.annotation.AutowiredAnnotationBeanPostProcessor
137  */
138 @SuppressWarnings("serial")
139 public class CommonAnnotationBeanPostProcessor extends InitDestroyAnnotationBeanPostProcessor
140 		implements InstantiationAwareBeanPostProcessor, BeanFactoryAware, Serializable {

316 	private InjectionMetadata findResourceMetadata(String beanName, final Class<?> clazz, PropertyValues pvs) {
317 		// Fall back to class name as cache key, for backwards compatibility with custom callers.
318 		String cacheKey = (StringUtils.hasLength(beanName) ? beanName : clazz.getName());
319 		// Quick check on the concurrent map first, with minimal locking.
320 		InjectionMetadata metadata = this.injectionMetadataCache.get(cacheKey);
321 		if (InjectionMetadata.needsRefresh(metadata, clazz)) {
322 			synchronized (this.injectionMetadataCache) {
323 				metadata = this.injectionMetadataCache.get(cacheKey);
324 				if (InjectionMetadata.needsRefresh(metadata, clazz)) {
325 					if (metadata != null) {
326 						metadata.clear(pvs);
327 					}
328 					try {
329 						metadata = buildResourceMetadata(clazz);
330 						this.injectionMetadataCache.put(cacheKey, metadata);
331 					}
332 					catch (NoClassDefFoundError err) {
333 						throw new IllegalStateException("Failed to introspect bean class [" + clazz.getName() +
334 								"] for resource metadata: could not find class that it depends on", err);
335 					}
336 				}
337 			}
338 		}
339 		return metadata;
340 	}

342 	private InjectionMetadata buildResourceMetadata(Class<?> clazz) {
343 		LinkedList<InjectionMetadata.InjectedElement> elements = new LinkedList<InjectionMetadata.InjectedElement>();
344 		Class<?> targetClass = clazz;
345 
346 		do {
347 			LinkedList<InjectionMetadata.InjectedElement> currElements = new LinkedList<InjectionMetadata.InjectedElement>();
348 			for (Field field : targetClass.getDeclaredFields()) {
349 				if (webServiceRefClass != null && field.isAnnotationPresent(webServiceRefClass)) {
350 					if (Modifier.isStatic(field.getModifiers())) {
351 						throw new IllegalStateException("@WebServiceRef annotation is not supported on static fields");
352 					}
353 					currElements.add(new WebServiceRefElement(field, field, null));
354 				}
355 				else if (ejbRefClass != null && field.isAnnotationPresent(ejbRefClass)) {
356 					if (Modifier.isStatic(field.getModifiers())) {
357 						throw new IllegalStateException("@EJB annotation is not supported on static fields");
358 					}
359 					currElements.add(new EjbRefElement(field, field, null));
360 				}
361 				else if (field.isAnnotationPresent(Resource.class)) {
362 					if (Modifier.isStatic(field.getModifiers())) {
363 						throw new IllegalStateException("@Resource annotation is not supported on static fields");
364 					}
365 					if (!ignoredResourceTypes.contains(field.getType().getName())) {
366 						currElements.add(new ResourceElement(field, field, null));
367 					}
368 				}
369 			}
370 			for (Method method : targetClass.getDeclaredMethods()) {
371 				Method bridgedMethod = BridgeMethodResolver.findBridgedMethod(method);
372 				if (!BridgeMethodResolver.isVisibilityBridgeMethodPair(method, bridgedMethod)) {
373 					continue;
374 				}
375 				if (method.equals(ClassUtils.getMostSpecificMethod(method, clazz))) {
376 					if (webServiceRefClass != null && bridgedMethod.isAnnotationPresent(webServiceRefClass)) {
377 						if (Modifier.isStatic(method.getModifiers())) {
378 							throw new IllegalStateException("@WebServiceRef annotation is not supported on static methods");
379 						}
380 						if (method.getParameterTypes().length != 1) {
381 							throw new IllegalStateException("@WebServiceRef annotation requires a single-arg method: " + method);
382 						}
383 						PropertyDescriptor pd = BeanUtils.findPropertyForMethod(bridgedMethod, clazz);
384 						currElements.add(new WebServiceRefElement(method, bridgedMethod, pd));
385 					}
386 					else if (ejbRefClass != null && bridgedMethod.isAnnotationPresent(ejbRefClass)) {
387 						if (Modifier.isStatic(method.getModifiers())) {
388 							throw new IllegalStateException("@EJB annotation is not supported on static methods");
389 						}
390 						if (method.getParameterTypes().length != 1) {
391 							throw new IllegalStateException("@EJB annotation requires a single-arg method: " + method);
392 						}
393 						PropertyDescriptor pd = BeanUtils.findPropertyForMethod(bridgedMethod, clazz);
394 						currElements.add(new EjbRefElement(method, bridgedMethod, pd));
395 					}
396 					else if (bridgedMethod.isAnnotationPresent(Resource.class)) {
397 						if (Modifier.isStatic(method.getModifiers())) {
398 							throw new IllegalStateException("@Resource annotation is not supported on static methods");
399 						}
400 						Class<?>[] paramTypes = method.getParameterTypes();
401 						if (paramTypes.length != 1) {
402 							throw new IllegalStateException("@Resource annotation requires a single-arg method: " + method);
403 						}
404 						if (!ignoredResourceTypes.contains(paramTypes[0].getName())) {
405 							PropertyDescriptor pd = BeanUtils.findPropertyForMethod(bridgedMethod, clazz);
406 							currElements.add(new ResourceElement(method, bridgedMethod, pd));
407 						}
408 					}
409 				}
410 			}
411 			elements.addAll(0, currElements);
412 			targetClass = targetClass.getSuperclass();
413 		}
414 		while (targetClass != null && targetClass != Object.class);
415 
416 		return new InjectionMetadata(clazz, elements);
417 	}

419 	/**
420 	 * Obtain a lazily resolving resource proxy for the given name and type,
421 	 * delegating to {@link #getResource} on demand once a method call comes in.
422 	 * @param element the descriptor for the annotated field/method
423 	 * @param requestingBeanName the name of the requesting bean
424 	 * @return the resource object (never {@code null})
425 	 * @since 4.2
426 	 * @see #getResource
427 	 * @see Lazy
428 	 */
429 	protected Object buildLazyResourceProxy(final LookupElement element, final String requestingBeanName) {
430 		TargetSource ts = new TargetSource() {
431 			@Override
432 			public Class<?> getTargetClass() {
433 				return element.lookupType;
434 			}
435 			@Override
436 			public boolean isStatic() {
437 				return false;
438 			}
439 			@Override
440 			public Object getTarget() {
441 				return getResource(element, requestingBeanName);
442 			}
443 			@Override
444 			public void releaseTarget(Object target) {
445 			}
446 		};
447 		ProxyFactory pf = new ProxyFactory();
448 		pf.setTargetSource(ts);
449 		if (element.lookupType.isInterface()) {
450 			pf.addInterface(element.lookupType);
451 		}
452 		ClassLoader classLoader = (this.beanFactory instanceof ConfigurableBeanFactory ?
453 				((ConfigurableBeanFactory) this.beanFactory).getBeanClassLoader() : null);
454 		return pf.getProxy(classLoader);
455 	}

457 	/**
458 	 * Obtain the resource object for the given name and type.
459 	 * @param element the descriptor for the annotated field/method
460 	 * @param requestingBeanName the name of the requesting bean
461 	 * @return the resource object (never {@code null})
462 	 * @throws BeansException if we failed to obtain the target resource
463 	 */
464 	protected Object getResource(LookupElement element, String requestingBeanName) throws BeansException {
465 		if (StringUtils.hasLength(element.mappedName)) {
466 			return this.jndiFactory.getBean(element.mappedName, element.lookupType);
467 		}
468 		if (this.alwaysUseJndiLookup) {
469 			return this.jndiFactory.getBean(element.name, element.lookupType);
470 		}
471 		if (this.resourceFactory == null) {
472 			throw new NoSuchBeanDefinitionException(element.lookupType,
473 					"No resource factory configured - specify the 'resourceFactory' property");
474 		}
475 		return autowireResource(this.resourceFactory, element, requestingBeanName);
476 	}
477 
478 	/**
479 	 * Obtain a resource object for the given name and type through autowiring
480 	 * based on the given factory.
481 	 * @param factory the factory to autowire against
482 	 * @param element the descriptor for the annotated field/method
483 	 * @param requestingBeanName the name of the requesting bean
484 	 * @return the resource object (never {@code null})
485 	 * @throws BeansException if we failed to obtain the target resource
486 	 */
487 	protected Object autowireResource(BeanFactory factory, LookupElement element, String requestingBeanName)
488 			throws BeansException {
489 
490 		Object resource;
491 		Set<String> autowiredBeanNames;
492 		String name = element.name;
493 
494 		if (this.fallbackToDefaultTypeMatch && element.isDefaultName &&
495 				factory instanceof AutowireCapableBeanFactory && !factory.containsBean(name)) {
496 			autowiredBeanNames = new LinkedHashSet<String>();
497 			resource = ((AutowireCapableBeanFactory) factory).resolveDependency(
498 					element.getDependencyDescriptor(), requestingBeanName, autowiredBeanNames, null);
499 		}
500 		else {
501 			resource = factory.getBean(name, element.lookupType);
502 			autowiredBeanNames = Collections.singleton(name);
503 		}
504 
505 		if (factory instanceof ConfigurableBeanFactory) {
506 			ConfigurableBeanFactory beanFactory = (ConfigurableBeanFactory) factory;
507 			for (String autowiredBeanName : autowiredBeanNames) {
508 				if (beanFactory.containsBean(autowiredBeanName)) {
509 					beanFactory.registerDependentBean(autowiredBeanName, requestingBeanName);
510 				}
511 			}
512 		}
513 
514 		return resource;
515 	}

520 	 * or setter method, supporting @Resource and related annotations.

522 	protected abstract class LookupElement extends InjectionMetadata.InjectedElement {
523 
524 		protected String name;
525 
526 		protected boolean isDefaultName = false;
527 
528 		protected Class<?> lookupType;
529 
530 		protected String mappedName;
531 
532 		public LookupElement(Member member, PropertyDescriptor pd) {
533 			super(member, pd);
534 		}
535 
536 		/**
537 		 * Return the resource name for the lookup.
538 		 */
539 		public final String getName() {
540 			return this.name;
541 		}
542 
543 		/**
544 		 * Return the desired type for the lookup.
545 		 */
546 		public final Class<?> getLookupType() {
547 			return this.lookupType;
548 		}
549 
550 		/**
551 		 * Build a DependencyDescriptor for the underlying field/method.
552 		 */
553 		public final DependencyDescriptor getDependencyDescriptor() {
554 			if (this.isField) {
555 				return new LookupDependencyDescriptor((Field) this.member, this.lookupType);
556 			}
557 			else {
558 				return new LookupDependencyDescriptor((Method) this.member, this.lookupType);
559 			}
560 		}

564 	/**
565 	 * Class representing injection information about an annotated field
566 	 * or setter method, supporting the @Resource annotation.
567 	 */
568 	private class ResourceElement extends LookupElement {
569 
570 		private final boolean lazyLookup;
571 
572 		public ResourceElement(Member member, AnnotatedElement ae, PropertyDescriptor pd) {
573 			super(member, pd);
574 			Resource resource = ae.getAnnotation(Resource.class);
575 			String resourceName = resource.name();
576 			Class<?> resourceType = resource.type();
577 			this.isDefaultName = !StringUtils.hasLength(resourceName);
578 			if (this.isDefaultName) {
579 				resourceName = this.member.getName();
580 				if (this.member instanceof Method && resourceName.startsWith("set") && resourceName.length() > 3) {
581 					resourceName = Introspector.decapitalize(resourceName.substring(3));
582 				}
583 			}
584 			else if (beanFactory instanceof ConfigurableBeanFactory){
585 				resourceName = ((ConfigurableBeanFactory) beanFactory).resolveEmbeddedValue(resourceName);
586 			}
587 			if (resourceType != null && !Object.class.equals(resourceType)) {
588 				checkResourceType(resourceType);
589 			}
590 			else {
591 				// No resource type specified... check field/method.
592 				resourceType = getResourceType();
593 			}
594 			this.name = resourceName;
595 			this.lookupType = resourceType;
596 			this.mappedName = resource.mappedName();
597 			Lazy lazy = ae.getAnnotation(Lazy.class);
598 			this.lazyLookup = (lazy != null && lazy.value());
599 		}
600 
601 		@Override
602 		protected Object getResourceToInject(Object target, String requestingBeanName) {
603 			return (this.lazyLookup ? buildLazyResourceProxy(this, requestingBeanName) :
604 					getResource(this, requestingBeanName));
605 		}
606 	}

609 	/**
610 	 * Class representing injection information about an annotated field
611 	 * or setter method, supporting the @WebServiceRef annotation.
612 	 */
613 	private class WebServiceRefElement extends LookupElement {
614 
615 		private final Class<?> elementType;
616 
617 		private final String wsdlLocation;
618 
619 		public WebServiceRefElement(Member member, AnnotatedElement ae, PropertyDescriptor pd) {
620 			super(member, pd);
621 			WebServiceRef resource = ae.getAnnotation(WebServiceRef.class);
622 			String resourceName = resource.name();
623 			Class<?> resourceType = resource.type();
624 			this.isDefaultName = !StringUtils.hasLength(resourceName);
625 			if (this.isDefaultName) {
626 				resourceName = this.member.getName();
627 				if (this.member instanceof Method && resourceName.startsWith("set") && resourceName.length() > 3) {
628 					resourceName = Introspector.decapitalize(resourceName.substring(3));
629 				}
630 			}
631 			if (resourceType != null && !Object.class.equals(resourceType)) {
632 				checkResourceType(resourceType);
633 			}
634 			else {
635 				// No resource type specified... check field/method.
636 				resourceType = getResourceType();
637 			}
638 			this.name = resourceName;
639 			this.elementType = resourceType;
640 			if (Service.class.isAssignableFrom(resourceType)) {
641 				this.lookupType = resourceType;
642 			}
643 			else {
644 				this.lookupType = (!Object.class.equals(resource.value()) ? resource.value() : Service.class);
645 			}
646 			this.mappedName = resource.mappedName();
647 			this.wsdlLocation = resource.wsdlLocation();
648 		}
649 
650 		@Override
651 		protected Object getResourceToInject(Object target, String requestingBeanName) {
652 			Service service;
653 			try {
654 				service = (Service) getResource(this, requestingBeanName);
655 			}
656 			catch (NoSuchBeanDefinitionException notFound) {
657 				// Service to be created through generated class.
658 				if (Service.class.equals(this.lookupType)) {
659 					throw new IllegalStateException("No resource with name '" + this.name + "' found in context, " +
660 							"and no specific JAX-WS Service subclass specified. The typical solution is to either specify " +
661 							"a LocalJaxWsServiceFactoryBean with the given name or to specify the (generated) Service " +
662 							"subclass as @WebServiceRef(...) value.");
663 				}
664 				if (StringUtils.hasLength(this.wsdlLocation)) {
665 					try {
666 						Constructor<?> ctor = this.lookupType.getConstructor(URL.class, QName.class);
667 						WebServiceClient clientAnn = this.lookupType.getAnnotation(WebServiceClient.class);
668 						if (clientAnn == null) {
669 							throw new IllegalStateException("JAX-WS Service class [" + this.lookupType.getName() +
670 									"] does not carry a WebServiceClient annotation");
671 						}
672 						service = (Service) BeanUtils.instantiateClass(ctor,
673 								new URL(this.wsdlLocation), new QName(clientAnn.targetNamespace(), clientAnn.name()));
674 					}
675 					catch (NoSuchMethodException ex) {
676 						throw new IllegalStateException("JAX-WS Service class [" + this.lookupType.getName() +
677 								"] does not have a (URL, QName) constructor. Cannot apply specified WSDL location [" +
678 								this.wsdlLocation + "].");
679 					}
680 					catch (MalformedURLException ex) {
681 						throw new IllegalArgumentException(
682 								"Specified WSDL location [" + this.wsdlLocation + "] isn't a valid URL");
683 					}
684 				}
685 				else {
686 					service = (Service) BeanUtils.instantiateClass(this.lookupType);
687 				}
688 			}
689 			return service.getPort(this.elementType);
690 		}
691 	}

695 	 * Class representing injection information about an annotated field
696 	 * or setter method, supporting the @EJB annotation.
697 	 */
698 	private class EjbRefElement extends LookupElement {
699 
700 		private final String beanName;
701 
702 		public EjbRefElement(Member member, AnnotatedElement ae, PropertyDescriptor pd) {
703 			super(member, pd);
704 			EJB resource = ae.getAnnotation(EJB.class);
705 			String resourceBeanName = resource.beanName();
706 			String resourceName = resource.name();
707 			this.isDefaultName = !StringUtils.hasLength(resourceName);
708 			if (this.isDefaultName) {
709 				resourceName = this.member.getName();
710 				if (this.member instanceof Method && resourceName.startsWith("set") && resourceName.length() > 3) {
711 					resourceName = Introspector.decapitalize(resourceName.substring(3));
712 				}
713 			}
714 			Class<?> resourceType = resource.beanInterface();
715 			if (resourceType != null && !Object.class.equals(resourceType)) {
716 				checkResourceType(resourceType);
717 			}
718 			else {
719 				// No resource type specified... check field/method.
720 				resourceType = getResourceType();
721 			}
722 			this.beanName = resourceBeanName;
723 			this.name = resourceName;
724 			this.lookupType = resourceType;
725 			this.mappedName = resource.mappedName();
726 		}
727 
728 		@Override
729 		protected Object getResourceToInject(Object target, String requestingBeanName) {
730 			if (StringUtils.hasLength(this.beanName)) {
731 				if (beanFactory != null && beanFactory.containsBean(this.beanName)) {
732 					// Local match found for explicitly specified local bean name.
733 					Object bean = beanFactory.getBean(this.beanName, this.lookupType);
734 					if (beanFactory instanceof ConfigurableBeanFactory) {
735 						((ConfigurableBeanFactory) beanFactory).registerDependentBean(this.beanName, requestingBeanName);
736 					}
737 					return bean;
738 				}
739 				else if (this.isDefaultName && !StringUtils.hasLength(this.mappedName)) {
740 					throw new NoSuchBeanDefinitionException(this.beanName,
741 							"Cannot resolve 'beanName' in local BeanFactory. Consider specifying a general 'name' value instead.");
742 				}
743 			}
744 			// JNDI name lookup - may still go to a local BeanFactory.
745 			return getResource(this, requestingBeanName);
746 		}
747 	}
