31  /**
32   * Date sequence generator for a <a href="http://www.manpagez.com/man/5/crontab/">Crontab pattern</a>,
33   * allowing clients to specify a pattern that the sequence matches.
34   *
35   * <p>The pattern is a list of six single space-separated fields: representing
36   * second, minute, hour, day, month, weekday. Month and weekday names can be
37   * given as the first three letters of the English names.
38   *
39   * <p>Example patterns:
40   * <ul>
41   * <li>"0 0 * * * *" = the top of every hour of every day.</li>
42   * <li>"*&#47;10 * * * * *" = every ten seconds.</li>
43   * <li>"0 0 8-10 * * *" = 8, 9 and 10 o'clock of every day.</li>
44   * <li>"0 0/30 8-10 * * *" = 8:00, 8:30, 9:00, 9:30 and 10 o'clock every day.</li>
45   * <li>"0 0 9-17 * * MON-FRI" = on the hour nine-to-five weekdays</li>
46   * <li>"0 0 0 25 12 ?" = every Christmas Day at midnight</li>
47   * </ul>
48   *
49   * @author Dave Syer
50   * @author Juergen Hoeller
51   * @since 3.0
52   * @see CronTrigger
53   */
54  public class CronSequenceGenerator {

75  	 * using the default {@link TimeZone}.

77  	 * @throws IllegalArgumentException if the pattern cannot be parsed
78  	 * @see java.util.TimeZone#getDefault()

84  	/**
85  	 * Construct a {@link CronSequenceGenerator} from the pattern provided,
86  	 * using the specified {@link TimeZone}.
87  	 * @param expression a space-separated list of time fields
88  	 * @param timeZone the TimeZone to use for generated trigger times
89  	 * @throws IllegalArgumentException if the pattern cannot be parsed

91  	public CronSequenceGenerator(String expression, TimeZone timeZone) {
92  		this.expression = expression;
93  		this.timeZone = timeZone;
94  		parse(expression);
95  	}

98  	/**
99  	 * Get the next {@link Date} in the sequence matching the Cron pattern and
100 	 * after the value provided. The return value will have a whole number of
101 	 * seconds, and will be after the input value.
102 	 * @param date a seed value
103 	 * @return the next value matching the pattern
104 	 */
105 	public Date next(Date date) {
106 		/*
107 		The plan:
108 
109 		1 Round up to the next whole second
110 
111 		2 If seconds match move on, otherwise find the next match:
112 		2.1 If next match is in the next minute then roll forwards
113 
114 		3 If minute matches move on, otherwise find the next match
115 		3.1 If next match is in the next hour then roll forwards
116 		3.2 Reset the seconds and go to 2
117 
118 		4 If hour matches move on, otherwise find the next match
119 		4.1 If next match is in the next day then roll forwards,
120 		4.2 Reset the minutes and seconds and go to 2
121 
122 		...
123 		*/
124 
125 		Calendar calendar = new GregorianCalendar();
126 		calendar.setTimeZone(this.timeZone);
127 		calendar.setTime(date);
128 
129 		// First, just reset the milliseconds and try to calculate from there...
130 		calendar.set(Calendar.MILLISECOND, 0);
131 		long originalTimestamp = calendar.getTimeInMillis();
132 		doNext(calendar, calendar.get(Calendar.YEAR));
133 
134 		if (calendar.getTimeInMillis() == originalTimestamp) {
135 			// We arrived at the original timestamp - round up to the next whole second and try again...
136 			calendar.add(Calendar.SECOND, 1);
137 			doNext(calendar, calendar.get(Calendar.YEAR));
138 		}
139 
140 		return calendar.getTime();
141 	}
142 
143 	private void doNext(Calendar calendar, int dot) {
144 		List<Integer> resets = new ArrayList<Integer>();
145 
146 		int second = calendar.get(Calendar.SECOND);
147 		List<Integer> emptyList = Collections.emptyList();
148 		int updateSecond = findNext(this.seconds, second, calendar, Calendar.SECOND, Calendar.MINUTE, emptyList);
149 		if (second == updateSecond) {
150 			resets.add(Calendar.SECOND);
151 		}
152 
153 		int minute = calendar.get(Calendar.MINUTE);
154 		int updateMinute = findNext(this.minutes, minute, calendar, Calendar.MINUTE, Calendar.HOUR_OF_DAY, resets);
155 		if (minute == updateMinute) {
156 			resets.add(Calendar.MINUTE);
157 		}
158 		else {
159 			doNext(calendar, dot);
160 		}
161 
162 		int hour = calendar.get(Calendar.HOUR_OF_DAY);
163 		int updateHour = findNext(this.hours, hour, calendar, Calendar.HOUR_OF_DAY, Calendar.DAY_OF_WEEK, resets);
164 		if (hour == updateHour) {
165 			resets.add(Calendar.HOUR_OF_DAY);
166 		}
167 		else {
168 			doNext(calendar, dot);
169 		}
170 
171 		int dayOfWeek = calendar.get(Calendar.DAY_OF_WEEK);
172 		int dayOfMonth = calendar.get(Calendar.DAY_OF_MONTH);
173 		int updateDayOfMonth = findNextDay(calendar, this.daysOfMonth, dayOfMonth, daysOfWeek, dayOfWeek, resets);
174 		if (dayOfMonth == updateDayOfMonth) {
175 			resets.add(Calendar.DAY_OF_MONTH);
176 		}
177 		else {
178 			doNext(calendar, dot);
179 		}
180 
181 		int month = calendar.get(Calendar.MONTH);
182 		int updateMonth = findNext(this.months, month, calendar, Calendar.MONTH, Calendar.YEAR, resets);
183 		if (month != updateMonth) {
184 			if (calendar.get(Calendar.YEAR) - dot > 4) {
185 				throw new IllegalArgumentException("Invalid cron expression \"" + this.expression +
186 						"\" led to runaway search for next trigger");
187 			}
188 			doNext(calendar, dot);
189 		}
190 
191 	}
192 
193 	private int findNextDay(Calendar calendar, BitSet daysOfMonth, int dayOfMonth, BitSet daysOfWeek, int dayOfWeek,
194 			List<Integer> resets) {
195 
196 		int count = 0;
197 		int max = 366;
198 		// the DAY_OF_WEEK values in java.util.Calendar start with 1 (Sunday),
199 		// but in the cron pattern, they start with 0, so we subtract 1 here
200 		while ((!daysOfMonth.get(dayOfMonth) || !daysOfWeek.get(dayOfWeek - 1)) && count++ < max) {
201 			calendar.add(Calendar.DAY_OF_MONTH, 1);
202 			dayOfMonth = calendar.get(Calendar.DAY_OF_MONTH);
203 			dayOfWeek = calendar.get(Calendar.DAY_OF_WEEK);
204 			reset(calendar, resets);
205 		}
206 		if (count >= max) {
207 			throw new IllegalArgumentException("Overflow in day for expression \"" + this.expression + "\"");
208 		}
209 		return dayOfMonth;
210 	}

212 	/**
213 	 * Search the bits provided for the next set bit after the value provided,
214 	 * and reset the calendar.
215 	 * @param bits a {@link BitSet} representing the allowed values of the field
216 	 * @param value the current value of the field
217 	 * @param calendar the calendar to increment as we move through the bits
218 	 * @param field the field to increment in the calendar (@see
219 	 * {@link Calendar} for the static constants defining valid fields)
220 	 * @param lowerOrders the Calendar field ids that should be reset (i.e. the
221 	 * ones of lower significance than the field of interest)
222 	 * @return the value of the calendar field that is next in the sequence
223 	 */
224 	private int findNext(BitSet bits, int value, Calendar calendar, int field, int nextField, List<Integer> lowerOrders) {
225 		int nextValue = bits.nextSetBit(value);
226 		// roll over if needed
227 		if (nextValue == -1) {
228 			calendar.add(nextField, 1);
229 			reset(calendar, Arrays.asList(field));
230 			nextValue = bits.nextSetBit(0);
231 		}
232 		if (nextValue != value) {
233 			calendar.set(field, nextValue);
234 			reset(calendar, lowerOrders);
235 		}
236 		return nextValue;
237 	}
238 
239 	/**
240 	 * Reset the calendar setting all the fields provided to zero.
241 	 */
242 	private void reset(Calendar calendar, List<Integer> fields) {
243 		for (int field : fields) {
244 			calendar.set(field, field == Calendar.DAY_OF_MONTH ? 1 : 0);
245 		}
246 	}

251 	/**
252 	 * Parse the given pattern expression.
253 	 */
254 	private void parse(String expression) throws IllegalArgumentException {
255 		String[] fields = StringUtils.tokenizeToStringArray(expression, " ");
256 		if (fields.length != 6) {
257 			throw new IllegalArgumentException(String.format(
258 					"Cron expression must consist of 6 fields (found %d in \"%s\")", fields.length, expression));
259 		}
260 		setNumberHits(this.seconds, fields[0], 0, 60);
261 		setNumberHits(this.minutes, fields[1], 0, 60);
262 		setNumberHits(this.hours, fields[2], 0, 24);
263 		setDaysOfMonth(this.daysOfMonth, fields[3]);
264 		setMonths(this.months, fields[4]);
265 		setDays(this.daysOfWeek, replaceOrdinals(fields[5], "SUN,MON,TUE,WED,THU,FRI,SAT"), 8);
266 		if (this.daysOfWeek.get(7)) {
267 			// Sunday can be represented as 0 or 7
268 			this.daysOfWeek.set(0);
269 			this.daysOfWeek.clear(7);
270 		}
271 	}

273 	/**
274 	 * Replace the values in the commaSeparatedList (case insensitive) with
275 	 * their index in the list.
276 	 * @return a new string with the values from the list replaced
277 	 */
278 	private String replaceOrdinals(String value, String commaSeparatedList) {
279 		String[] list = StringUtils.commaDelimitedListToStringArray(commaSeparatedList);
280 		for (int i = 0; i < list.length; i++) {
281 			String item = list[i].toUpperCase();
282 			value = StringUtils.replace(value.toUpperCase(), item, "" + i);
283 		}
284 		return value;
285 	}

287 	private void setDaysOfMonth(BitSet bits, String field) {
288 		int max = 31;
289 		// Days of month start with 1 (in Cron and Calendar) so add one
290 		setDays(bits, field, max + 1);
291 		// ... and remove it from the front
292 		bits.clear(0);
293 	}
294 
295 	private void setDays(BitSet bits, String field, int max) {
296 		if (field.contains("?")) {
297 			field = "*";
298 		}
299 		setNumberHits(bits, field, 0, max);
300 	}
301 
302 	private void setMonths(BitSet bits, String value) {
303 		int max = 12;
304 		value = replaceOrdinals(value, "FOO,JAN,FEB,MAR,APR,MAY,JUN,JUL,AUG,SEP,OCT,NOV,DEC");
305 		BitSet months = new BitSet(13);
306 		// Months start with 1 in Cron and 0 in Calendar, so push the values first into a longer bit set
307 		setNumberHits(months, value, 1, max + 1);
308 		// ... and then rotate it to the front of the months
309 		for (int i = 1; i <= max; i++) {
310 			if (months.get(i)) {
311 				bits.set(i - 1);
312 			}
313 		}
314 	}
315 
316 	private void setNumberHits(BitSet bits, String value, int min, int max) {
317 		String[] fields = StringUtils.delimitedListToStringArray(value, ",");
318 		for (String field : fields) {
319 			if (!field.contains("/")) {
320 				// Not an incrementer so it must be a range (possibly empty)
321 				int[] range = getRange(field, min, max);
322 				bits.set(range[0], range[1] + 1);
323 			}
324 			else {
325 				String[] split = StringUtils.delimitedListToStringArray(field, "/");
326 				if (split.length > 2) {
327 					throw new IllegalArgumentException("Incrementer has more than two fields: '" +
328 							field + "' in expression \"" + this.expression + "\"");
329 				}
330 				int[] range = getRange(split[0], min, max);
331 				if (!split[0].contains("-")) {
332 					range[1] = max - 1;
333 				}
334 				int delta = Integer.valueOf(split[1]);
335 				for (int i = range[0]; i <= range[1]; i += delta) {
336 					bits.set(i);
337 				}
338 			}
339 		}
340 	}
341 
342 	private int[] getRange(String field, int min, int max) {
343 		int[] result = new int[2];
344 		if (field.contains("*")) {
345 			result[0] = min;
346 			result[1] = max - 1;
347 			return result;
348 		}
349 		if (!field.contains("-")) {
350 			result[0] = result[1] = Integer.valueOf(field);
351 		}
352 		else {
353 			String[] split = StringUtils.delimitedListToStringArray(field, "-");
354 			if (split.length > 2) {
355 				throw new IllegalArgumentException("Range has more than two fields: '" +
356 						field + "' in expression \"" + this.expression + "\"");
357 			}
358 			result[0] = Integer.valueOf(split[0]);
359 			result[1] = Integer.valueOf(split[1]);
360 		}
361 		if (result[0] >= max || result[1] >= max) {
362 			throw new IllegalArgumentException("Range exceeds maximum (" + max + "): '" +
363 					field + "' in expression \"" + this.expression + "\"");
364 		}
365 		if (result[0] < min || result[1] < min) {
366 			throw new IllegalArgumentException("Range less than minimum (" + min + "): '" +
367 					field + "' in expression \"" + this.expression + "\"");
368 		}
369 		return result;
370 	}

376 	@Override
377 	public boolean equals(Object obj) {
378 		if (!(obj instanceof CronSequenceGenerator)) {
379 			return false;
380 		}
381 		CronSequenceGenerator cron = (CronSequenceGenerator) obj;
382 		return cron.months.equals(this.months) && cron.daysOfMonth.equals(this.daysOfMonth)
383 				&& cron.daysOfWeek.equals(this.daysOfWeek) && cron.hours.equals(this.hours)
384 				&& cron.minutes.equals(this.minutes) && cron.seconds.equals(this.seconds);
385 	}

388 	public int hashCode() {
389 		return 37 + 17 * this.months.hashCode() + 29 * this.daysOfMonth.hashCode() + 37 * this.daysOfWeek.hashCode()
390 				+ 41 * this.hours.hashCode() + 53 * this.minutes.hashCode() + 61 * this.seconds.hashCode();
391 	}
