42  /**
43   * Implementation of the {@link MBeanInfoAssembler}
44   * interface that reads the management interface information from source level metadata.
45   *
46   * <p>Uses the {@link JmxAttributeSource} strategy interface, so that
47   * metadata can be read using any supported implementation. Out of the box,
48   * Spring provides an implementation based on JDK 1.5+ annotations,
49   * {@code AnnotationJmxAttributeSource}.
50   *
51   * @author Rob Harrop
52   * @author Juergen Hoeller
53   * @author Jennifer Hickey
54   * @since 1.2
55   * @see #setAttributeSource
56   * @see org.springframework.jmx.export.annotation.AnnotationJmxAttributeSource
57   */
58  public class MetadataMBeanInfoAssembler extends AbstractReflectiveMBeanInfoAssembler
59  		implements AutodetectCapableMBeanInfoAssembler, InitializingBean {
60  
61  	private JmxAttributeSource attributeSource;
62  
63  
64  	/**
65  	 * Create a new {@code MetadataMBeanInfoAssembler} which needs to be
66  	 * configured through the {@link #setAttributeSource} method.
67  	 */
68  	public MetadataMBeanInfoAssembler() {
69  	}
70  
71  	/**
72  	 * Create a new {@code MetadataMBeanInfoAssembler} for the given
73  	 * {@code JmxAttributeSource}.
74  	 * @param attributeSource the JmxAttributeSource to use
75  	 */
76  	public MetadataMBeanInfoAssembler(JmxAttributeSource attributeSource) {
77  		Assert.notNull(attributeSource, "JmxAttributeSource must not be null");
78  		this.attributeSource = attributeSource;
79  	}
80  
81  
82  	/**
83  	 * Set the {@code JmxAttributeSource} implementation to use for
84  	 * reading the metadata from the bean class.
85  	 * @see org.springframework.jmx.export.annotation.AnnotationJmxAttributeSource
86  	 */
87  	public void setAttributeSource(JmxAttributeSource attributeSource) {
88  		Assert.notNull(attributeSource, "JmxAttributeSource must not be null");
89  		this.attributeSource = attributeSource;
90  	}

92  	@Override
93  	public void afterPropertiesSet() {
94  		if (this.attributeSource == null) {
95  			throw new IllegalArgumentException("Property 'attributeSource' is required");
96  		}
97  	}

100 	/**
101 	 * Throws an IllegalArgumentException if it encounters a JDK dynamic proxy.
102 	 * Metadata can only be read from target classes and CGLIB proxies!
103 	 */
104 	@Override
105 	protected void checkManagedBean(Object managedBean) throws IllegalArgumentException {
106 		if (AopUtils.isJdkDynamicProxy(managedBean)) {
107 			throw new IllegalArgumentException(
108 					"MetadataMBeanInfoAssembler does not support JDK dynamic proxies - " +
109 					"export the target beans directly or use CGLIB proxies instead");
110 		}
111 	}

113 	/**
114 	 * Used for autodetection of beans. Checks to see if the bean's class has a
115 	 * {@code ManagedResource} attribute. If so it will add it list of included beans.
116 	 * @param beanClass the class of the bean
117 	 * @param beanName the name of the bean in the bean factory
118 	 */
119 	@Override
120 	public boolean includeBean(Class<?> beanClass, String beanName) {
121 		return (this.attributeSource.getManagedResource(getClassToExpose(beanClass)) != null);
122 	}

124 	/**
125 	 * Vote on the inclusion of an attribute accessor.
126 	 * @param method the accessor method
127 	 * @param beanKey the key associated with the MBean in the beans map
128 	 * @return whether the method has the appropriate metadata
129 	 */
130 	@Override
131 	protected boolean includeReadAttribute(Method method, String beanKey) {
132 		return hasManagedAttribute(method) || hasManagedMetric(method);
133 	}
134 
135 	/**
136 	 * Votes on the inclusion of an attribute mutator.
137 	 * @param method the mutator method
138 	 * @param beanKey the key associated with the MBean in the beans map
139 	 * @return whether the method has the appropriate metadata
140 	 */
141 	@Override
142 	protected boolean includeWriteAttribute(Method method, String beanKey) {
143 		return hasManagedAttribute(method);
144 	}
145 
146 	/**
147 	 * Votes on the inclusion of an operation.
148 	 * @param method the operation method
149 	 * @param beanKey the key associated with the MBean in the beans map
150 	 * @return whether the method has the appropriate metadata
151 	 */
152 	@Override
153 	protected boolean includeOperation(Method method, String beanKey) {
154 		PropertyDescriptor pd = BeanUtils.findPropertyForMethod(method);
155 		if (pd != null) {
156 			if (hasManagedAttribute(method)) {
157 				return true;
158 			}
159 		}
160 		return hasManagedOperation(method);
161 	}

186 	/**
187 	 * Reads managed resource description from the source level metadata.
188 	 * Returns an empty {@code String} if no description can be found.
189 	 */
190 	@Override
191 	protected String getDescription(Object managedBean, String beanKey) {
192 		ManagedResource mr = this.attributeSource.getManagedResource(getClassToExpose(managedBean));
193 		return (mr != null ? mr.getDescription() : "");
194 	}
195 
196 	/**
197 	 * Creates a description for the attribute corresponding to this property
198 	 * descriptor. Attempts to create the description using metadata from either
199 	 * the getter or setter attributes, otherwise uses the property name.
200 	 */
201 	@Override
202 	protected String getAttributeDescription(PropertyDescriptor propertyDescriptor, String beanKey) {
203 		Method readMethod = propertyDescriptor.getReadMethod();
204 		Method writeMethod = propertyDescriptor.getWriteMethod();
205 
206 		ManagedAttribute getter =
207 				(readMethod != null ? this.attributeSource.getManagedAttribute(readMethod) : null);
208 		ManagedAttribute setter =
209 				(writeMethod != null ? this.attributeSource.getManagedAttribute(writeMethod) : null);
210 
211 		if (getter != null && StringUtils.hasText(getter.getDescription())) {
212 			return getter.getDescription();
213 		}
214 		else if (setter != null && StringUtils.hasText(setter.getDescription())) {
215 			return setter.getDescription();
216 		}
217 
218 		ManagedMetric metric = (readMethod != null ? this.attributeSource.getManagedMetric(readMethod) : null);
219 		if (metric != null && StringUtils.hasText(metric.getDescription())) {
220 			return metric.getDescription();
221 		}
222 
223 		return propertyDescriptor.getDisplayName();
224 	}
225 
226 	/**
227 	 * Retrieves the description for the supplied {@code Method} from the
228 	 * metadata. Uses the method name is no description is present in the metadata.
229 	 */
230 	@Override
231 	protected String getOperationDescription(Method method, String beanKey) {
232 		PropertyDescriptor pd = BeanUtils.findPropertyForMethod(method);
233 		if (pd != null) {
234 			ManagedAttribute ma = this.attributeSource.getManagedAttribute(method);
235 			if (ma != null && StringUtils.hasText(ma.getDescription())) {
236 				return ma.getDescription();
237 			}
238 			ManagedMetric metric = this.attributeSource.getManagedMetric(method);
239 			if (metric != null && StringUtils.hasText(metric.getDescription())) {
240 				return metric.getDescription();
241 			}
242 			return method.getName();
243 		}
244 		else {
245 			ManagedOperation mo = this.attributeSource.getManagedOperation(method);
246 			if (mo != null && StringUtils.hasText(mo.getDescription())) {
247 				return mo.getDescription();
248 			}
249 			return method.getName();
250 		}
251 	}

253 	/**
254 	 * Reads {@code MBeanParameterInfo} from the {@code ManagedOperationParameter}
255 	 * attributes attached to a method. Returns an empty array of {@code MBeanParameterInfo}
256 	 * if no attributes are found.
257 	 */
258 	@Override
259 	protected MBeanParameterInfo[] getOperationParameters(Method method, String beanKey) {
260 		ManagedOperationParameter[] params = this.attributeSource.getManagedOperationParameters(method);
261 		if (ObjectUtils.isEmpty(params)) {
262 			return super.getOperationParameters(method, beanKey);
263 		}
264 
265 		MBeanParameterInfo[] parameterInfo = new MBeanParameterInfo[params.length];
266 		Class<?>[] methodParameters = method.getParameterTypes();
267 		for (int i = 0; i < params.length; i++) {
268 			ManagedOperationParameter param = params[i];
269 			parameterInfo[i] =
270 					new MBeanParameterInfo(param.getName(), methodParameters[i].getName(), param.getDescription());
271 		}
272 		return parameterInfo;
273 	}

275 	/**
276 	 * Reads the {@link ManagedNotification} metadata from the {@code Class} of the managed resource
277 	 * and generates and returns the corresponding {@link ModelMBeanNotificationInfo} metadata.
278 	 */
279 	@Override
280 	protected ModelMBeanNotificationInfo[] getNotificationInfo(Object managedBean, String beanKey) {
281 		ManagedNotification[] notificationAttributes =
282 				this.attributeSource.getManagedNotifications(getClassToExpose(managedBean));
283 		ModelMBeanNotificationInfo[] notificationInfos =
284 				new ModelMBeanNotificationInfo[notificationAttributes.length];
285 
286 		for (int i = 0; i < notificationAttributes.length; i++) {
287 			ManagedNotification attribute = notificationAttributes[i];
288 			notificationInfos[i] = JmxMetadataUtils.convertToModelMBeanNotificationInfo(attribute);
289 		}
290 
291 		return notificationInfos;
292 	}

294 	/**
295 	 * Adds descriptor fields from the {@code ManagedResource} attribute
296 	 * to the MBean descriptor. Specifically, adds the {@code currencyTimeLimit},
297 	 * {@code persistPolicy}, {@code persistPeriod}, {@code persistLocation}
298 	 * and {@code persistName} descriptor fields if they are present in the metadata.
299 	 */
300 	@Override
301 	protected void populateMBeanDescriptor(Descriptor desc, Object managedBean, String beanKey) {
302 		ManagedResource mr = this.attributeSource.getManagedResource(getClassToExpose(managedBean));
303 		if (mr == null) {
304 			throw new InvalidMetadataException(
305 					"No ManagedResource attribute found for class: " + getClassToExpose(managedBean));
306 		}
307 
308 		applyCurrencyTimeLimit(desc, mr.getCurrencyTimeLimit());
309 
310 		if (mr.isLog()) {
311 			desc.setField(FIELD_LOG, "true");
312 		}
313 		if (StringUtils.hasLength(mr.getLogFile())) {
314 			desc.setField(FIELD_LOG_FILE, mr.getLogFile());
315 		}
316 
317 		if (StringUtils.hasLength(mr.getPersistPolicy())) {
318 			desc.setField(FIELD_PERSIST_POLICY, mr.getPersistPolicy());
319 		}
320 		if (mr.getPersistPeriod() >= 0) {
321 			desc.setField(FIELD_PERSIST_PERIOD, Integer.toString(mr.getPersistPeriod()));
322 		}
323 		if (StringUtils.hasLength(mr.getPersistName())) {
324 			desc.setField(FIELD_PERSIST_NAME, mr.getPersistName());
325 		}
326 		if (StringUtils.hasLength(mr.getPersistLocation())) {
327 			desc.setField(FIELD_PERSIST_LOCATION, mr.getPersistLocation());
328 		}
329 	}

331 	/**
332 	 * Adds descriptor fields from the {@code ManagedAttribute} attribute or the {@code ManagedMetric} attribute
333 	 * to the attribute descriptor.
334 	 */
335 	@Override
336 	protected void populateAttributeDescriptor(Descriptor desc, Method getter, Method setter, String beanKey) {
337 		if (getter != null && hasManagedMetric(getter)) {
338 			populateMetricDescriptor(desc, this.attributeSource.getManagedMetric(getter));
339 		}
340 		else {
341 			ManagedAttribute gma =
342 				(getter == null) ? ManagedAttribute.EMPTY : this.attributeSource.getManagedAttribute(getter);
343 			ManagedAttribute sma =
344 				(setter == null) ? ManagedAttribute.EMPTY : this.attributeSource.getManagedAttribute(setter);
345 			populateAttributeDescriptor(desc,gma,sma);
346 		}
347 	}
348 
349 	private void populateAttributeDescriptor(Descriptor desc, ManagedAttribute gma, ManagedAttribute sma) {
350 		applyCurrencyTimeLimit(desc, resolveIntDescriptor(gma.getCurrencyTimeLimit(), sma.getCurrencyTimeLimit()));
351 
352 		Object defaultValue = resolveObjectDescriptor(gma.getDefaultValue(), sma.getDefaultValue());
353 		desc.setField(FIELD_DEFAULT, defaultValue);
354 
355 		String persistPolicy = resolveStringDescriptor(gma.getPersistPolicy(), sma.getPersistPolicy());
356 		if (StringUtils.hasLength(persistPolicy)) {
357 			desc.setField(FIELD_PERSIST_POLICY, persistPolicy);
358 		}
359 		int persistPeriod = resolveIntDescriptor(gma.getPersistPeriod(), sma.getPersistPeriod());
360 		if (persistPeriod >= 0) {
361 			desc.setField(FIELD_PERSIST_PERIOD, Integer.toString(persistPeriod));
362 		}
363 	}
364 
365 	private void populateMetricDescriptor(Descriptor desc, ManagedMetric metric) {
366 		applyCurrencyTimeLimit(desc, metric.getCurrencyTimeLimit());
367 
368 		if (StringUtils.hasLength(metric.getPersistPolicy())) {
369 			desc.setField(FIELD_PERSIST_POLICY, metric.getPersistPolicy());
370 		}
371 		if (metric.getPersistPeriod() >= 0) {
372 			desc.setField(FIELD_PERSIST_PERIOD, Integer.toString(metric.getPersistPeriod()));
373 		}
374 
375 		if (StringUtils.hasLength(metric.getDisplayName())) {
376 			desc.setField(FIELD_DISPLAY_NAME, metric.getDisplayName());
377 		}
378 
379 		if (StringUtils.hasLength(metric.getUnit())) {
380 			desc.setField(FIELD_UNITS, metric.getUnit());
381 		}
382 
383 		if (StringUtils.hasLength(metric.getCategory())) {
384 			desc.setField(FIELD_METRIC_CATEGORY, metric.getCategory());
385 		}
386 
387 		String metricType = (metric.getMetricType() == null) ? MetricType.GAUGE.toString() : metric.getMetricType().toString();
388 		desc.setField(FIELD_METRIC_TYPE, metricType);
389 	}

392 	 * Adds descriptor fields from the {@code ManagedAttribute} attribute
393 	 * to the attribute descriptor. Specifically, adds the {@code currencyTimeLimit}
394 	 * descriptor field if it is present in the metadata.
395 	 */
396 	@Override
397 	protected void populateOperationDescriptor(Descriptor desc, Method method, String beanKey) {
398 		ManagedOperation mo = this.attributeSource.getManagedOperation(method);
399 		if (mo != null) {
400 			applyCurrencyTimeLimit(desc, mo.getCurrencyTimeLimit());
401 		}

404 	/**
405 	 * Determines which of two {@code int} values should be used as the value
406 	 * for an attribute descriptor. In general, only the getter or the setter will
407 	 * be have a non-negative value so we use that value. In the event that both values
408 	 * are non-negative, we use the greater of the two. This method can be used to
409 	 * resolve any {@code int} valued descriptor where there are two possible values.
410 	 * @param getter the int value associated with the getter for this attribute
411 	 * @param setter the int associated with the setter for this attribute
412 	 */
413 	private int resolveIntDescriptor(int getter, int setter) {
414 		return (getter >= setter ? getter : setter);
415 	}
416 
417 	/**
418 	 * Locates the value of a descriptor based on values attached
419 	 * to both the getter and setter methods. If both have values
420 	 * supplied then the value attached to the getter is preferred.
421 	 * @param getter the Object value associated with the get method
422 	 * @param setter the Object value associated with the set method
423 	 * @return the appropriate Object to use as the value for the descriptor
424 	 */
425 	private Object resolveObjectDescriptor(Object getter, Object setter) {
426 		return (getter != null ? getter : setter);
427 	}
428 
429 	/**
430 	 * Locates the value of a descriptor based on values attached
431 	 * to both the getter and setter methods. If both have values
432 	 * supplied then the value attached to the getter is preferred.
433 	 * The supplied default value is used to check to see if the value
434 	 * associated with the getter has changed from the default.
435 	 * @param getter the String value associated with the get method
436 	 * @param setter the String value associated with the set method
437 	 * @return the appropriate String to use as the value for the descriptor
438 	 */
439 	private String resolveStringDescriptor(String getter, String setter) {
440 		return (StringUtils.hasLength(getter) ? getter : setter);
441 	}
