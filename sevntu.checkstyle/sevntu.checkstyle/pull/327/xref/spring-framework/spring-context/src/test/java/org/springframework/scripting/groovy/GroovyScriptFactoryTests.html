65  	public void testStaticScript() throws Exception {
66  		ApplicationContext ctx = new ClassPathXmlApplicationContext("groovyContext.xml", getClass());
67  
68  		assertTrue(Arrays.asList(ctx.getBeanNamesForType(Calculator.class)).contains("calculator"));
69  		assertTrue(Arrays.asList(ctx.getBeanNamesForType(Messenger.class)).contains("messenger"));
70  
71  		Calculator calc = (Calculator) ctx.getBean("calculator");
72  		Messenger messenger = (Messenger) ctx.getBean("messenger");
73  
74  		assertFalse("Shouldn't get proxy when refresh is disabled", AopUtils.isAopProxy(calc));
75  		assertFalse("Shouldn't get proxy when refresh is disabled", AopUtils.isAopProxy(messenger));
76  
77  		assertFalse("Scripted object should not be instance of Refreshable", calc instanceof Refreshable);
78  		assertFalse("Scripted object should not be instance of Refreshable", messenger instanceof Refreshable);
79  
80  		assertEquals(calc, calc);
81  		assertEquals(messenger, messenger);
82  		assertTrue(!messenger.equals(calc));
83  		assertTrue(messenger.hashCode() != calc.hashCode());
84  		assertTrue(!messenger.toString().equals(calc.toString()));
85  
86  		String desiredMessage = "Hello World!";
87  		assertEquals("Message is incorrect", desiredMessage, messenger.getMessage());
88  
89  		assertTrue(ctx.getBeansOfType(Calculator.class).values().contains(calc));
90  		assertTrue(ctx.getBeansOfType(Messenger.class).values().contains(messenger));
91  	}

487 		@Override
488 		public void customize(GroovyObject goo) {
489 			DelegatingMetaClass dmc = new DelegatingMetaClass(goo.getMetaClass()) {
490 				@Override
491 				public Object invokeMethod(Object arg0, String mName, Object[] arg2) {
492 					if (mName.contains("Missing")) {
493 						throw new IllegalStateException("Gotcha");
494 					}
495 					else {
496 						return super.invokeMethod(arg0, mName, arg2);
497 					}
498 				}
499 			};
500 			dmc.initialize();
501 			goo.setMetaClass(dmc);
502 		}
