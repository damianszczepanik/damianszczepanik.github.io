17  package org.springframework.context.support;
18  
19  import java.util.HashMap;
20  import java.util.Map;
21  
22  import org.apache.commons.logging.Log;
23  import org.apache.commons.logging.LogFactory;
24  
25  import org.springframework.beans.factory.ObjectFactory;
26  import org.springframework.beans.factory.config.Scope;
27  import org.springframework.core.NamedThreadLocal;
28  
29  /**
30   * A simple thread-backed {@link Scope} implementation.
31   *
32   * <p><strong>Note:</strong> {@code SimpleThreadScope} <em>does not clean up
33   * any objects</em> associated with it. As such, it is typically preferable to
34   * use {@link org.springframework.web.context.request.RequestScope RequestScope}
35   * in web environments.
36   *
37   * <p>For an implementation of a thread-based {@code Scope} with support for
38   * destruction callbacks, refer to the
39   * <a href="http://www.springbyexample.org/examples/custom-thread-scope-module.html">
40  *  Spring by Example Custom Thread Scope Module</a>.
41   *
42   * <p>Thanks to Eugene Kuleshov for submitting the original prototype for a thread scope!
43   *
44   * @author Arjen Poutsma
45   * @author Juergen Hoeller
46   * @since 3.0
47   * @see org.springframework.web.context.request.RequestScope
48   */
49  public class SimpleThreadScope implements Scope {
50  
51  	private static final Log logger = LogFactory.getLog(SimpleThreadScope.class);
52  
53  	private final ThreadLocal<Map<String, Object>> threadScope =
54  			new NamedThreadLocal<Map<String, Object>>("SimpleThreadScope") {
55  				@Override
56  				protected Map<String, Object> initialValue() {
57  					return new HashMap<String, Object>();
58  				}
59  			};
60  
61  
62  	@Override
63  	public Object get(String name, ObjectFactory<?> objectFactory) {
64  		Map<String, Object> scope = this.threadScope.get();
65  		Object object = scope.get(name);
66  		if (object == null) {
67  			object = objectFactory.getObject();
68  			scope.put(name, object);
69  		}
70  		return object;
71  	}
72  
73  	@Override
74  	public Object remove(String name) {
75  		Map<String, Object> scope = this.threadScope.get();
76  		return scope.remove(name);
77  	}
78  
79  	@Override
80  	public void registerDestructionCallback(String name, Runnable callback) {
81  		logger.warn("SimpleThreadScope does not support destruction callbacks. " +
82  				"Consider using RequestScope in a web environment.");
83  	}
84  
85  	@Override
86  	public Object resolveContextualObject(String key) {
87  		return null;
88  	}
89  
90  	@Override
91  	public String getConversationId() {
92  		return Thread.currentThread().getName();
93  	}
94  
95  }
