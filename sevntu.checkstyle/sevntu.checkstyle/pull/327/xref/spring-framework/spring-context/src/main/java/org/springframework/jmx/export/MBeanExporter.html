71  /**
72   * JMX exporter that allows for exposing any <i>Spring-managed bean</i> to a
73   * JMX {@link javax.management.MBeanServer}, without the need to define any
74   * JMX-specific information in the bean classes.
75   *
76   * <p>If a bean implements one of the JMX management interfaces, MBeanExporter can
77   * simply register the MBean with the server through its autodetection process.
78   *
79   * <p>If a bean does not implement one of the JMX management interfaces, MBeanExporter
80   * will create the management information using the supplied {@link MBeanInfoAssembler}.
81   *
82   * <p>A list of {@link MBeanExporterListener MBeanExporterListeners} can be registered
83   * via the {@link #setListeners(MBeanExporterListener[]) listeners} property, allowing
84   * application code to be notified of MBean registration and unregistration events.
85   *
86   * <p>This exporter is compatible with MBeans and MXBeans on Java 6 and above.
87   *
88   * @author Rob Harrop
89   * @author Juergen Hoeller
90   * @author Rick Evans
91   * @author Mark Fisher
92   * @author Stephane Nicoll
93   * @since 1.2
94   * @see #setBeans
95   * @see #setAutodetect
96   * @see #setAssembler
97   * @see #setListeners
98   * @see org.springframework.jmx.export.assembler.MBeanInfoAssembler
99   * @see MBeanExporterListener
100  */
101 public class MBeanExporter extends MBeanRegistrationSupport implements MBeanExportOperations,
102 		BeanClassLoaderAware, BeanFactoryAware, InitializingBean, SmartInitializingSingleton, DisposableBean {

352 	/**
353 	 * Set the {@link NotificationListener NotificationListeners} to register
354 	 * with the {@link javax.management.MBeanServer}.
355 	 * <P>The key of each entry in the {@code Map} is a {@link String}
356 	 * representation of the {@link javax.management.ObjectName} or the bean
357 	 * name of the MBean the listener should be registered for. Specifying an
358 	 * asterisk ({@code *}) for a key will cause the listener to be
359 	 * associated with all MBeans registered by this class at startup time.
360 	 * <p>The value of each entry is the
361 	 * {@link javax.management.NotificationListener} to register. For more
362 	 * advanced options such as registering
363 	 * {@link javax.management.NotificationFilter NotificationFilters} and
364 	 * handback objects see {@link #setNotificationListeners(NotificationListenerBean[])}.
365 	 */
366 	public void setNotificationListenerMappings(Map<?, ? extends NotificationListener> listeners) {
367 		Assert.notNull(listeners, "'listeners' must not be null");
368 		List<NotificationListenerBean> notificationListeners =
369 				new ArrayList<NotificationListenerBean>(listeners.size());
370 
371 		for (Map.Entry<?, ? extends NotificationListener> entry : listeners.entrySet()) {
372 			// Get the listener from the map value.
373 			NotificationListenerBean bean = new NotificationListenerBean(entry.getValue());
374 			// Get the ObjectName from the map key.
375 			Object key = entry.getKey();
376 			if (key != null && !WILDCARD.equals(key)) {
377 				// This listener is mapped to a specific ObjectName.
378 				bean.setMappedObjectName(entry.getKey());
379 			}
380 			notificationListeners.add(bean);
381 		}
382 
383 		this.notificationListeners =
384 				notificationListeners.toArray(new NotificationListenerBean[notificationListeners.size()]);
385 	}

459 	@Override
460 	public ObjectName registerManagedResource(Object managedResource) throws MBeanExportException {
461 		Assert.notNull(managedResource, "Managed resource must not be null");
462 		ObjectName objectName;
463 		try {
464 			objectName = getObjectName(managedResource, null);
465 			if (this.ensureUniqueRuntimeObjectNames) {
466 				objectName = JmxUtils.appendIdentityToObjectName(objectName, managedResource);
467 			}
468 		}
469 		catch (Exception ex) {
470 			throw new MBeanExportException("Unable to generate ObjectName for MBean [" + managedResource + "]", ex);
471 		}
472 		registerManagedResource(managedResource, objectName);
473 		return objectName;
474 	}
475 
476 	@Override
477 	public void registerManagedResource(Object managedResource, ObjectName objectName) throws MBeanExportException {
478 		Assert.notNull(managedResource, "Managed resource must not be null");
479 		Assert.notNull(objectName, "ObjectName must not be null");
480 		try {
481 			if (isMBean(managedResource.getClass())) {
482 				doRegister(managedResource, objectName);
483 			}
484 			else {
485 				ModelMBean mbean = createAndConfigureMBean(managedResource, managedResource.getClass().getName());
486 				doRegister(mbean, objectName);
487 				injectNotificationPublisherIfNecessary(managedResource, mbean, objectName);
488 			}
489 		}
490 		catch (JMException ex) {
491 			throw new UnableToRegisterMBeanException(
492 					"Unable to register MBean [" + managedResource + "] with object name [" + objectName + "]", ex);
493 		}
494 	}
495 
496 	@Override
497 	public void unregisterManagedResource(ObjectName objectName) {
498 		Assert.notNull(objectName, "ObjectName must not be null");
499 		doUnregister(objectName);
500 	}

507 	/**
508 	 * Registers the defined beans with the {@link MBeanServer}.
509 	 * <p>Each bean is exposed to the {@code MBeanServer} via a
510 	 * {@code ModelMBean}. The actual implemetation of the
511 	 * {@code ModelMBean} interface used depends on the implementation of
512 	 * the {@code ModelMBeanProvider} interface that is configured. By
513 	 * default the {@code RequiredModelMBean} class that is supplied with
514 	 * all JMX implementations is used.
515 	 * <p>The management interface produced for each bean is dependent on the
516 	 * {@code MBeanInfoAssembler} implementation being used. The
517 	 * {@code ObjectName} given to each bean is dependent on the
518 	 * implementation of the {@code ObjectNamingStrategy} interface being used.
519 	 */
520 	protected void registerBeans() {
521 		// The beans property may be null, for example if we are relying solely on autodetection.
522 		if (this.beans == null) {
523 			this.beans = new HashMap<String, Object>();
524 			// Use AUTODETECT_ALL as default in no beans specified explicitly.
525 			if (this.autodetectMode == null) {
526 				this.autodetectMode = AUTODETECT_ALL;
527 			}
528 		}
529 
530 		// Perform autodetection, if desired.
531 		int mode = (this.autodetectMode != null ? this.autodetectMode : AUTODETECT_NONE);
532 		if (mode != AUTODETECT_NONE) {
533 			if (this.beanFactory == null) {
534 				throw new MBeanExportException("Cannot autodetect MBeans if not running in a BeanFactory");
535 			}
536 			if (mode == AUTODETECT_MBEAN || mode == AUTODETECT_ALL) {
537 				// Autodetect any beans that are already MBeans.
538 				logger.debug("Autodetecting user-defined JMX MBeans");
539 				autodetectMBeans();
540 			}
541 			// Allow the assembler a chance to vote for bean inclusion.
542 			if ((mode == AUTODETECT_ASSEMBLER || mode == AUTODETECT_ALL) &&
543 					this.assembler instanceof AutodetectCapableMBeanInfoAssembler) {
544 				autodetectBeans((AutodetectCapableMBeanInfoAssembler) this.assembler);
545 			}
546 		}
547 
548 		if (!this.beans.isEmpty()) {
549 			for (Map.Entry<String, Object> entry : this.beans.entrySet()) {
550 				registerBeanNameOrInstance(entry.getValue(), entry.getKey());
551 			}
552 		}
553 	}

556 	 * Return whether the specified bean definition should be considered as lazy-init.
557 	 * @param beanFactory the bean factory that is supposed to contain the bean definition
558 	 * @param beanName the name of the bean to check
559 	 * @see org.springframework.beans.factory.config.ConfigurableListableBeanFactory#getBeanDefinition
560 	 * @see org.springframework.beans.factory.config.BeanDefinition#isLazyInit
561 	 */
562 	protected boolean isBeanDefinitionLazyInit(ListableBeanFactory beanFactory, String beanName) {
563 		return (beanFactory instanceof ConfigurableListableBeanFactory && beanFactory.containsBeanDefinition(beanName) &&
564 				((ConfigurableListableBeanFactory) beanFactory).getBeanDefinition(beanName).isLazyInit());
565 	}

567 	/**
568 	 * Registers an individual bean with the {@link #setServer MBeanServer}.
569 	 * <p>This method is responsible for deciding <strong>how</strong> a bean
570 	 * should be exposed to the {@code MBeanServer}. Specifically, if the
571 	 * supplied {@code mapValue} is the name of a bean that is configured
572 	 * for lazy initialization, then a proxy to the resource is registered with
573 	 * the {@code MBeanServer} so that the the lazy load behavior is
574 	 * honored. If the bean is already an MBean then it will be registered
575 	 * directly with the {@code MBeanServer} without any intervention. For
576 	 * all other beans or bean names, the resource itself is registered with
577 	 * the {@code MBeanServer} directly.
578 	 * @param mapValue the value configured for this bean in the beans map;
579 	 * may be either the {@code String} name of a bean, or the bean itself
580 	 * @param beanKey the key associated with this bean in the beans map
581 	 * @return the {@code ObjectName} under which the resource was registered
582 	 * @throws MBeanExportException if the export failed
583 	 * @see #setBeans
584 	 * @see #registerBeanInstance
585 	 * @see #registerLazyInit
586 	 */
587 	protected ObjectName registerBeanNameOrInstance(Object mapValue, String beanKey) throws MBeanExportException {
588 		try {
589 			if (mapValue instanceof String) {
590 				// Bean name pointing to a potentially lazy-init bean in the factory.
591 				if (this.beanFactory == null) {
592 					throw new MBeanExportException("Cannot resolve bean names if not running in a BeanFactory");
593 				}
594 				String beanName = (String) mapValue;
595 				if (isBeanDefinitionLazyInit(this.beanFactory, beanName)) {
596 					ObjectName objectName = registerLazyInit(beanName, beanKey);
597 					replaceNotificationListenerBeanNameKeysIfNecessary(beanName, objectName);
598 					return objectName;
599 				}
600 				else {
601 					Object bean = this.beanFactory.getBean(beanName);
602 					ObjectName objectName = registerBeanInstance(bean, beanKey);
603 					replaceNotificationListenerBeanNameKeysIfNecessary(beanName, objectName);
604 					return objectName;
605 				}
606 			}
607 			else {
608 				// Plain bean instance -> register it directly.
609 				if (this.beanFactory != null) {
610 					Map<String, ?> beansOfSameType =
611 							this.beanFactory.getBeansOfType(mapValue.getClass(), false, this.allowEagerInit);
612 					for (Map.Entry<String, ?> entry : beansOfSameType.entrySet()) {
613 						if (entry.getValue() == mapValue) {
614 							String beanName = entry.getKey();
615 							ObjectName objectName = registerBeanInstance(mapValue, beanKey);
616 							replaceNotificationListenerBeanNameKeysIfNecessary(beanName, objectName);
617 							return objectName;
618 						}
619 					}
620 				}
621 				return registerBeanInstance(mapValue, beanKey);
622 			}
623 		}
624 		catch (Exception ex) {
625 			throw new UnableToRegisterMBeanException(
626 					"Unable to register MBean [" + mapValue + "] with key '" + beanKey + "'", ex);
627 		}
628 	}

631 	 * Replaces any bean names used as keys in the {@code NotificationListener}

645 	/**
646 	 * Registers an existing MBean or an MBean adapter for a plain bean
647 	 * with the {@code MBeanServer}.
648 	 * @param bean the bean to register, either an MBean or a plain bean
649 	 * @param beanKey the key associated with this bean in the beans map
650 	 * @return the {@code ObjectName} under which the bean was registered
651 	 * with the {@code MBeanServer}
652 	 */
653 	private ObjectName registerBeanInstance(Object bean, String beanKey) throws JMException {
654 		ObjectName objectName = getObjectName(bean, beanKey);
655 		Object mbeanToExpose = null;
656 		if (isMBean(bean.getClass())) {
657 			mbeanToExpose = bean;
658 		}
659 		else {
660 			DynamicMBean adaptedBean = adaptMBeanIfPossible(bean);
661 			if (adaptedBean != null) {
662 				mbeanToExpose = adaptedBean;
663 			}
664 		}
665 		if (mbeanToExpose != null) {
666 			if (logger.isInfoEnabled()) {
667 				logger.info("Located MBean '" + beanKey + "': registering with JMX server as MBean [" +
668 						objectName + "]");
669 			}
670 			doRegister(mbeanToExpose, objectName);
671 		}
672 		else {
673 			if (logger.isInfoEnabled()) {
674 				logger.info("Located managed bean '" + beanKey + "': registering with JMX server as MBean [" +
675 						objectName + "]");
676 			}
677 			ModelMBean mbean = createAndConfigureMBean(bean, beanKey);
678 			doRegister(mbean, objectName);
679 			injectNotificationPublisherIfNecessary(bean, mbean, objectName);
680 		}
681 		return objectName;
682 	}

684 	/**
685 	 * Registers beans that are configured for lazy initialization with the
686 	 * {@code MBeanServer} indirectly through a proxy.
687 	 * @param beanName the name of the bean in the {@code BeanFactory}
688 	 * @param beanKey the key associated with this bean in the beans map
689 	 * @return the {@code ObjectName} under which the bean was registered
690 	 * with the {@code MBeanServer}
691 	 */
692 	private ObjectName registerLazyInit(String beanName, String beanKey) throws JMException {
693 		ProxyFactory proxyFactory = new ProxyFactory();
694 		proxyFactory.setProxyTargetClass(true);
695 		proxyFactory.setFrozen(true);
696 
697 		if (isMBean(this.beanFactory.getType(beanName))) {
698 			// A straight MBean... Let's create a simple lazy-init CGLIB proxy for it.
699 			LazyInitTargetSource targetSource = new LazyInitTargetSource();
700 			targetSource.setTargetBeanName(beanName);
701 			targetSource.setBeanFactory(this.beanFactory);
702 			proxyFactory.setTargetSource(targetSource);
703 
704 			Object proxy = proxyFactory.getProxy(this.beanClassLoader);
705 			ObjectName objectName = getObjectName(proxy, beanKey);
706 			if (logger.isDebugEnabled()) {
707 				logger.debug("Located MBean '" + beanKey + "': registering with JMX server as lazy-init MBean [" +
708 						objectName + "]");
709 			}
710 			doRegister(proxy, objectName);
711 			return objectName;
712 		}
713 
714 		else {
715 			// A simple bean... Let's create a lazy-init ModelMBean proxy with notification support.
716 			NotificationPublisherAwareLazyTargetSource targetSource = new NotificationPublisherAwareLazyTargetSource();
717 			targetSource.setTargetBeanName(beanName);
718 			targetSource.setBeanFactory(this.beanFactory);
719 			proxyFactory.setTargetSource(targetSource);
720 
721 			Object proxy = proxyFactory.getProxy(this.beanClassLoader);
722 			ObjectName objectName = getObjectName(proxy, beanKey);
723 			if (logger.isDebugEnabled()) {
724 				logger.debug("Located simple bean '" + beanKey + "': registering with JMX server as lazy-init MBean [" +
725 						objectName + "]");
726 			}
727 			ModelMBean mbean = createAndConfigureMBean(proxy, beanKey);
728 			targetSource.setModelMBean(mbean);
729 			targetSource.setObjectName(objectName);
730 			doRegister(mbean, objectName);
731 			return objectName;
732 		}
733 	}

769 	/**
770 	 * Build an adapted MBean for the given bean instance, if possible.
771 	 * <p>The default implementation builds a JMX 1.2 StandardMBean
772 	 * for the target's MBean/MXBean interface in case of an AOP proxy,
773 	 * delegating the interface's management operations to the proxy.
774 	 * @param bean the original bean instance
775 	 * @return the adapted MBean, or {@code null} if not possible
776 	 */
777 	@SuppressWarnings("unchecked")
778 	protected DynamicMBean adaptMBeanIfPossible(Object bean) throws JMException {
779 		Class<?> targetClass = AopUtils.getTargetClass(bean);
780 		if (targetClass != bean.getClass()) {
781 			Class<?> ifc = JmxUtils.getMXBeanInterface(targetClass);
782 			if (ifc != null) {
783 				if (!ifc.isInstance(bean)) {
784 					throw new NotCompliantMBeanException("Managed bean [" + bean +
785 							"] has a target class with an MXBean interface but does not expose it in the proxy");
786 				}
787 				return new StandardMBean(bean, ((Class<Object>) ifc), true);
788 			}
789 			else {
790 				ifc = JmxUtils.getMBeanInterface(targetClass);
791 				if (ifc != null) {
792 					if (!ifc.isInstance(bean)) {
793 						throw new NotCompliantMBeanException("Managed bean [" + bean +
794 								"] has a target class with an MBean interface but does not expose it in the proxy");
795 					}
796 					return new StandardMBean(bean, ((Class<Object>) ifc));
797 				}
798 			}
799 		}
800 		return null;
801 	}

886 	/**
887 	 * Performs the actual autodetection process, delegating to an
888 	 * {@code AutodetectCallback} instance to vote on the inclusion of a
889 	 * given bean.
890 	 * @param callback the {@code AutodetectCallback} to use when deciding
891 	 * whether to include a bean or not
892 	 */
893 	private void autodetect(AutodetectCallback callback) {
894 		Set<String> beanNames = new LinkedHashSet<String>(this.beanFactory.getBeanDefinitionCount());
895 		beanNames.addAll(Arrays.asList(this.beanFactory.getBeanDefinitionNames()));
896 		if (this.beanFactory instanceof ConfigurableBeanFactory) {
897 			beanNames.addAll(Arrays.asList(((ConfigurableBeanFactory) this.beanFactory).getSingletonNames()));
898 		}
899 		for (String beanName : beanNames) {
900 			if (!isExcluded(beanName) && !isBeanDefinitionAbstract(this.beanFactory, beanName)) {
901 				try {
902 					Class<?> beanClass = this.beanFactory.getType(beanName);
903 					if (beanClass != null && callback.include(beanClass, beanName)) {
904 						boolean lazyInit = isBeanDefinitionLazyInit(this.beanFactory, beanName);
905 						Object beanInstance = (!lazyInit ? this.beanFactory.getBean(beanName) : null);
906 						if (!ScopedProxyUtils.isScopedTarget(beanName) && !this.beans.containsValue(beanName) &&
907 								(beanInstance == null ||
908 										!CollectionUtils.containsInstance(this.beans.values(), beanInstance))) {
909 							// Not already registered for JMX exposure.
910 							this.beans.put(beanName, (beanInstance != null ? beanInstance : beanName));
911 							if (logger.isInfoEnabled()) {
912 								logger.info("Bean with name '" + beanName + "' has been autodetected for JMX exposure");
913 							}
914 						}
915 						else {
916 							if (logger.isDebugEnabled()) {
917 								logger.debug("Bean with name '" + beanName + "' is already registered for JMX exposure");
918 							}
919 						}
920 					}
921 				}
922 				catch (CannotLoadBeanClassException ex) {
923 					if (this.allowEagerInit) {
924 						throw ex;
925 					}
926 					// otherwise ignore beans where the class is not resolvable
927 				}
928 			}
929 		}
930 	}

968 	 * Register the configured {@link NotificationListener NotificationListeners}
969 	 * with the {@link MBeanServer}.
970 	 */
971 	private void registerNotificationListeners() throws MBeanExportException {
972 		if (this.notificationListeners != null) {
973 			for (NotificationListenerBean bean : this.notificationListeners) {
974 				try {
975 					ObjectName[] mappedObjectNames = bean.getResolvedObjectNames();
976 					if (mappedObjectNames == null) {
977 						// Mapped to all MBeans registered by the MBeanExporter.
978 						mappedObjectNames = getRegisteredObjectNames();
979 					}
980 					if (this.registeredNotificationListeners.put(bean, mappedObjectNames) == null) {
981 						for (ObjectName mappedObjectName : mappedObjectNames) {
982 							this.server.addNotificationListener(mappedObjectName, bean.getNotificationListener(),
983 									bean.getNotificationFilter(), bean.getHandback());
984 						}
985 					}
986 				}
987 				catch (Exception ex) {
988 					throw new MBeanExportException("Unable to register NotificationListener", ex);
989 				}
990 			}
991 		}
992 	}

995 	 * Unregister the configured {@link NotificationListener NotificationListeners}
996 	 * from the {@link MBeanServer}.
997 	 */
998 	private void unregisterNotificationListeners() {
999 		for (Map.Entry<NotificationListenerBean, ObjectName[]> entry : this.registeredNotificationListeners.entrySet()) {
1000 			NotificationListenerBean bean = entry.getKey();
1001 			ObjectName[] mappedObjectNames = entry.getValue();
1002 			for (ObjectName mappedObjectName : mappedObjectNames) {
1003 				try {
1004 					this.server.removeNotificationListener(mappedObjectName, bean.getNotificationListener(),
1005 							bean.getNotificationFilter(), bean.getHandback());
1006 				}
1007 				catch (Exception ex) {
1008 					if (logger.isDebugEnabled()) {
1009 						logger.debug("Unable to unregister NotificationListener", ex);
1010 					}
1011 				}
1012 			}
1013 		}
1014 		this.registeredNotificationListeners.clear();
1015 	}
