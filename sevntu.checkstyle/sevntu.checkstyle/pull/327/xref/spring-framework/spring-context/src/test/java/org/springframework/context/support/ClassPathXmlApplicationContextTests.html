126 	@Test
127 	public void testContextWithInvalidValueType() throws IOException {
128 		ClassPathXmlApplicationContext context = new ClassPathXmlApplicationContext(
129 				new String[] {INVALID_VALUE_TYPE_CONTEXT}, false);
130 		try {
131 			context.refresh();
132 			fail("Should have thrown BeanCreationException");
133 		}
134 		catch (BeanCreationException ex) {
135 			assertTrue(ex.contains(TypeMismatchException.class));
136 			assertTrue(ex.toString().contains("someMessageSource"));
137 			assertTrue(ex.toString().contains("useCodeAsDefaultMessage"));
138 			checkExceptionFromInvalidValueType(ex);
139 			checkExceptionFromInvalidValueType(new ExceptionInInitializerError(ex));
140 			assertFalse(context.isActive());
141 		}
142 	}
143 
144 	private void checkExceptionFromInvalidValueType(Throwable ex) throws IOException {
145 		ByteArrayOutputStream baos = new ByteArrayOutputStream();
146 		ex.printStackTrace(new PrintStream(baos));
147 		String dump = FileCopyUtils.copyToString(
148 				new InputStreamReader(new ByteArrayInputStream(baos.toByteArray())));
149 		assertTrue(dump.contains("someMessageSource"));
150 		assertTrue(dump.contains("useCodeAsDefaultMessage"));
151 	}

232 	@Test
233 	public void testAliasForParentContext() {
234 		ClassPathXmlApplicationContext ctx = new ClassPathXmlApplicationContext(FQ_SIMPLE_CONTEXT);
235 		assertTrue(ctx.containsBean("someMessageSource"));
236 
237 		ClassPathXmlApplicationContext child = new ClassPathXmlApplicationContext(
238 				new String[] {ALIAS_FOR_PARENT_CONTEXT}, ctx);
239 		assertTrue(child.containsBean("someMessageSource"));
240 		assertTrue(child.containsBean("yourMessageSource"));
241 		assertTrue(child.containsBean("myMessageSource"));
242 		assertTrue(child.isSingleton("someMessageSource"));
243 		assertTrue(child.isSingleton("yourMessageSource"));
244 		assertTrue(child.isSingleton("myMessageSource"));
245 		assertEquals(StaticMessageSource.class, child.getType("someMessageSource"));
246 		assertEquals(StaticMessageSource.class, child.getType("yourMessageSource"));
247 		assertEquals(StaticMessageSource.class, child.getType("myMessageSource"));
248 
249 		Object someMs = child.getBean("someMessageSource");
250 		Object yourMs = child.getBean("yourMessageSource");
251 		Object myMs = child.getBean("myMessageSource");
252 		assertSame(someMs, yourMs);
253 		assertSame(someMs, myMs);
254 
255 		String[] aliases = child.getAliases("someMessageSource");
256 		assertEquals(2, aliases.length);
257 		assertEquals("myMessageSource", aliases[0]);
258 		assertEquals("yourMessageSource", aliases[1]);
259 		aliases = child.getAliases("myMessageSource");
260 		assertEquals(2, aliases.length);
261 		assertEquals("someMessageSource", aliases[0]);
262 		assertEquals("yourMessageSource", aliases[1]);
263 
264 		child.close();
265 		ctx.close();
266 	}

292 	private void assertOneMessageSourceOnly(ClassPathXmlApplicationContext ctx, Object myMessageSource) {
293 		String[] beanNamesForType = ctx.getBeanNamesForType(StaticMessageSource.class);
294 		assertEquals(1, beanNamesForType.length);
295 		assertEquals("myMessageSource", beanNamesForType[0]);
296 		beanNamesForType = ctx.getBeanNamesForType(StaticMessageSource.class, true, true);
297 		assertEquals(1, beanNamesForType.length);
298 		assertEquals("myMessageSource", beanNamesForType[0]);
299 		beanNamesForType = BeanFactoryUtils.beanNamesForTypeIncludingAncestors(ctx, StaticMessageSource.class);
300 		assertEquals(1, beanNamesForType.length);
301 		assertEquals("myMessageSource", beanNamesForType[0]);
302 		beanNamesForType = BeanFactoryUtils.beanNamesForTypeIncludingAncestors(ctx, StaticMessageSource.class, true, true);
303 		assertEquals(1, beanNamesForType.length);
304 		assertEquals("myMessageSource", beanNamesForType[0]);
305 
306 		Map<?, StaticMessageSource> beansOfType = ctx.getBeansOfType(StaticMessageSource.class);
307 		assertEquals(1, beansOfType.size());
308 		assertSame(myMessageSource, beansOfType.values().iterator().next());
309 		beansOfType = ctx.getBeansOfType(StaticMessageSource.class, true, true);
310 		assertEquals(1, beansOfType.size());
311 		assertSame(myMessageSource, beansOfType.values().iterator().next());
312 		beansOfType = BeanFactoryUtils.beansOfTypeIncludingAncestors(ctx, StaticMessageSource.class);
313 		assertEquals(1, beansOfType.size());
314 		assertSame(myMessageSource, beansOfType.values().iterator().next());
315 		beansOfType = BeanFactoryUtils.beansOfTypeIncludingAncestors(ctx, StaticMessageSource.class, true, true);
316 		assertEquals(1, beansOfType.size());
317 		assertSame(myMessageSource, beansOfType.values().iterator().next());
318 	}

320 	@Test
321 	public void testResourceAndInputStream() throws IOException {
322 		ClassPathXmlApplicationContext ctx = new ClassPathXmlApplicationContext(RESOURCE_CONTEXT) {
323 			@Override
324 			public Resource getResource(String location) {
325 				if (TEST_PROPERTIES.equals(location)) {
326 					return new ClassPathResource(TEST_PROPERTIES, ClassPathXmlApplicationContextTests.class);
327 				}
328 				return super.getResource(location);
329 			}
330 		};
331 		ResourceTestBean resource1 = (ResourceTestBean) ctx.getBean("resource1");
332 		ResourceTestBean resource2 = (ResourceTestBean) ctx.getBean("resource2");
333 		assertTrue(resource1.getResource() instanceof ClassPathResource);
334 		StringWriter writer = new StringWriter();
335 		FileCopyUtils.copy(new InputStreamReader(resource1.getResource().getInputStream()), writer);
336 		assertEquals("contexttest", writer.toString());
337 		writer = new StringWriter();
338 		FileCopyUtils.copy(new InputStreamReader(resource1.getInputStream()), writer);
339 		assertEquals("test", writer.toString());
340 		writer = new StringWriter();
341 		FileCopyUtils.copy(new InputStreamReader(resource2.getResource().getInputStream()), writer);
342 		assertEquals("contexttest", writer.toString());
343 		writer = new StringWriter();
344 		FileCopyUtils.copy(new InputStreamReader(resource2.getInputStream()), writer);
345 		assertEquals("test", writer.toString());
346 		ctx.close();
347 	}
