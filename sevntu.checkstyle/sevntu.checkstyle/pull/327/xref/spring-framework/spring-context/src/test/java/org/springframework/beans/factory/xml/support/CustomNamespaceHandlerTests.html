85  	@Before
86  	public void setUp() throws Exception {
87  		NamespaceHandlerResolver resolver = new DefaultNamespaceHandlerResolver(CLASS.getClassLoader(), NS_PROPS);
88  		this.beanFactory = new GenericApplicationContext();
89  		XmlBeanDefinitionReader reader = new XmlBeanDefinitionReader(this.beanFactory);
90  		reader.setNamespaceHandlerResolver(resolver);
91  		reader.setValidationMode(XmlBeanDefinitionReader.VALIDATION_XSD);
92  		reader.setEntityResolver(new DummySchemaResolver());
93  		reader.loadBeanDefinitions(getResource());
94  		this.beanFactory.refresh();

192 	private final class DummySchemaResolver extends PluggableSchemaResolver {
193 
194 		public DummySchemaResolver() {
195 			super(CLASS.getClassLoader());
196 		}
197 
198 		@Override
199 		public InputSource resolveEntity(String publicId, String systemId) throws IOException {
200 			InputSource source = super.resolveEntity(publicId, systemId);
201 			if (source == null) {
202 				Resource resource = new ClassPathResource(TEST_XSD);
203 				source = new InputSource(resource.getInputStream());
204 				source.setPublicId(publicId);
205 				source.setSystemId(systemId);
206 			}
207 			return source;
208 		}
209 	}

214 /**
215  * Custom namespace handler implementation.
216  *
217  * @author Rob Harrop
218  */
219 final class TestNamespaceHandler extends NamespaceHandlerSupport {
220 
221 	@Override
222 	public void init() {
223 		registerBeanDefinitionParser("testBean", new TestBeanDefinitionParser());
224 		registerBeanDefinitionParser("person", new PersonDefinitionParser());
225 
226 		registerBeanDefinitionDecorator("set", new PropertyModifyingBeanDefinitionDecorator());
227 		registerBeanDefinitionDecorator("debug", new DebugBeanDefinitionDecorator());
228 		registerBeanDefinitionDecorator("nop", new NopInterceptorBeanDefinitionDecorator());
229 		registerBeanDefinitionDecoratorForAttribute("object-name", new ObjectNameBeanDefinitionDecorator());
230 	}
231 
232 	private static class TestBeanDefinitionParser implements BeanDefinitionParser {
233 
234 		@Override
235 		public BeanDefinition parse(Element element, ParserContext parserContext) {
236 			RootBeanDefinition definition = new RootBeanDefinition();
237 			definition.setBeanClass(TestBean.class);
238 
239 			MutablePropertyValues mpvs = new MutablePropertyValues();
240 			mpvs.add("name", element.getAttribute("name"));
241 			mpvs.add("age", element.getAttribute("age"));
242 			definition.setPropertyValues(mpvs);
243 
244 			parserContext.getRegistry().registerBeanDefinition(element.getAttribute("id"), definition);
245 
246 			return null;
247 		}
248 	}
249 
250 	private static final class PersonDefinitionParser extends AbstractSingleBeanDefinitionParser {
251 
252 		@Override
253 		protected Class<?> getBeanClass(Element element) {
254 			return TestBean.class;
255 		}
256 
257 		@Override
258 		protected void doParse(Element element, BeanDefinitionBuilder builder) {
259 			builder.addPropertyValue("name", element.getAttribute("name"));
260 			builder.addPropertyValue("age", element.getAttribute("age"));
261 		}
262 	}
263 
264 	private static class PropertyModifyingBeanDefinitionDecorator implements BeanDefinitionDecorator {
265 
266 		@Override
267 		public BeanDefinitionHolder decorate(Node node, BeanDefinitionHolder definition, ParserContext parserContext) {
268 			Element element = (Element) node;
269 			BeanDefinition def = definition.getBeanDefinition();
270 
271 			MutablePropertyValues mpvs = (def.getPropertyValues() == null) ? new MutablePropertyValues() : def.getPropertyValues();
272 			mpvs.add("name", element.getAttribute("name"));
273 			mpvs.add("age", element.getAttribute("age"));
274 
275 			((AbstractBeanDefinition) def).setPropertyValues(mpvs);
276 			return definition;
277 		}
278 	}

285 		}

293 		}

296 	private static class ObjectNameBeanDefinitionDecorator implements BeanDefinitionDecorator {
297 
298 		@Override
299 		public BeanDefinitionHolder decorate(Node node, BeanDefinitionHolder definition, ParserContext parserContext) {
300 			Attr objectNameAttribute = (Attr) node;
301 			definition.getBeanDefinition().setAttribute("objectName", objectNameAttribute.getValue());
302 			return definition;
303 		}
