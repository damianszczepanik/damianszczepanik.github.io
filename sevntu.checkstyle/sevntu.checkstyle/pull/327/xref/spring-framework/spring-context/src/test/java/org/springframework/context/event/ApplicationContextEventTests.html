88  	private void multicastEvent(boolean match, Class<?> listenerType,
89  			ApplicationEvent event, ResolvableType eventType) {
90  		@SuppressWarnings("unchecked")
91  		ApplicationListener<ApplicationEvent> listener =
92  				(ApplicationListener<ApplicationEvent>) mock(listenerType);
93  		SimpleApplicationEventMulticaster smc = new SimpleApplicationEventMulticaster();
94  		smc.addApplicationListener(listener);
95  
96  		if (eventType != null) {
97  			smc.multicastEvent(event, eventType);
98  		} else {
99  			smc.multicastEvent(event);
100 		}
101 		int invocation = match ? 1 : 0;
102 		verify(listener, times(invocation)).onApplicationEvent(event);
103 	}

217 	@Test
218 	public void listenersInApplicationContext() {
219 		StaticApplicationContext context = new StaticApplicationContext();
220 		context.registerBeanDefinition("listener1", new RootBeanDefinition(MyOrderedListener1.class));
221 		RootBeanDefinition listener2 = new RootBeanDefinition(MyOrderedListener2.class);
222 		listener2.getConstructorArgumentValues().addGenericArgumentValue(new RuntimeBeanReference("listener1"));
223 		listener2.setLazyInit(true);
224 		context.registerBeanDefinition("listener2", listener2);
225 		context.refresh();
226 		assertFalse(context.getDefaultListableBeanFactory().containsSingleton("listener2"));
227 
228 		MyOrderedListener1 listener1 = context.getBean("listener1", MyOrderedListener1.class);
229 		MyOtherEvent event1 = new MyOtherEvent(context);
230 		context.publishEvent(event1);
231 		assertFalse(context.getDefaultListableBeanFactory().containsSingleton("listener2"));
232 		MyEvent event2 = new MyEvent(context);
233 		context.publishEvent(event2);
234 		assertTrue(context.getDefaultListableBeanFactory().containsSingleton("listener2"));
235 		MyEvent event3 = new MyEvent(context);
236 		context.publishEvent(event3);
237 		MyOtherEvent event4 = new MyOtherEvent(context);
238 		context.publishEvent(event4);
239 		assertTrue(listener1.seenEvents.contains(event1));
240 		assertTrue(listener1.seenEvents.contains(event2));
241 		assertTrue(listener1.seenEvents.contains(event3));
242 		assertTrue(listener1.seenEvents.contains(event4));
243 
244 		listener1.seenEvents.clear();
245 		context.publishEvent(event1);
246 		context.publishEvent(event2);
247 		context.publishEvent(event3);
248 		context.publishEvent(event4);
249 		assertTrue(listener1.seenEvents.contains(event1));
250 		assertTrue(listener1.seenEvents.contains(event2));
251 		assertTrue(listener1.seenEvents.contains(event3));
252 		assertTrue(listener1.seenEvents.contains(event4));
253 
254 		context.close();
255 	}

272 		assertTrue(listener1.seenEvents.contains(event1));
