17  package org.springframework.validation;
18  
19  import org.springframework.util.Assert;
20  import org.springframework.util.ObjectUtils;
21  
22  /**
23   * Encapsulates a field error, that is, a reason for rejecting a specific
24   * field value.
25   *
26   * <p>See the {@link DefaultMessageCodesResolver} javadoc for details on
27   * how a message code list is built for a {@code FieldError}.
28   *
29   * @author Rod Johnson
30   * @author Juergen Hoeller
31   * @since 10.03.2003
32   * @see DefaultMessageCodesResolver
33   */
34  @SuppressWarnings("serial")
35  public class FieldError extends ObjectError {
36  
37  	private final String field;
38  
39  	private final Object rejectedValue;
40  
41  	private final boolean bindingFailure;
42  
43  
44  	/**
45  	 * Create a new FieldError instance.
46  	 * @param objectName the name of the affected object
47  	 * @param field the affected field of the object
48  	 * @param defaultMessage the default message to be used to resolve this message
49  	 */
50  	public FieldError(String objectName, String field, String defaultMessage) {
51  		this(objectName, field, null, false, null, null, defaultMessage);
52  	}
53  
54  	/**
55  	 * Create a new FieldError instance.
56  	 * @param objectName the name of the affected object
57  	 * @param field the affected field of the object
58  	 * @param rejectedValue the rejected field value
59  	 * @param bindingFailure whether this error represents a binding failure
60  	 * (like a type mismatch); else, it is a validation failure
61  	 * @param codes the codes to be used to resolve this message
62  	 * @param arguments the array of arguments to be used to resolve this message
63  	 * @param defaultMessage the default message to be used to resolve this message
64  	 */
65  	public FieldError(
66  			String objectName, String field, Object rejectedValue, boolean bindingFailure,
67  			String[] codes, Object[] arguments, String defaultMessage) {
68  
69  		super(objectName, codes, arguments, defaultMessage);
70  		Assert.notNull(field, "Field must not be null");
71  		this.field = field;
72  		this.rejectedValue = rejectedValue;
73  		this.bindingFailure = bindingFailure;
74  	}
75  
76  
77  	/**
78  	 * Return the affected field of the object.
79  	 */
80  	public String getField() {
81  		return this.field;
82  	}
83  
84  	/**
85  	 * Return the rejected field value.
86  	 */
87  	public Object getRejectedValue() {
88  		return this.rejectedValue;
89  	}
90  
91  	/**
92  	 * Return whether this error represents a binding failure
93  	 * (like a type mismatch); otherwise it is a validation failure.
94  	 */
95  	public boolean isBindingFailure() {
96  		return this.bindingFailure;
97  	}
98  
99  
100 	@Override
101 	public String toString() {
102 		return "Field error in object '" + getObjectName() + "' on field '" + this.field +
103 				"': rejected value [" + this.rejectedValue + "]; " + resolvableToString();
104 	}
105 
106 	@Override
107 	public boolean equals(Object other) {
108 		if (this == other) {
109 			return true;
110 		}
111 		if (!super.equals(other)) {
112 			return false;
113 		}
114 		FieldError otherError = (FieldError) other;
115 		return getField().equals(otherError.getField()) &&
116 				ObjectUtils.nullSafeEquals(getRejectedValue(), otherError.getRejectedValue()) &&
117 				isBindingFailure() == otherError.isBindingFailure();
118 	}
119 
120 	@Override
121 	public int hashCode() {
122 		int hashCode = super.hashCode();
123 		hashCode = 29 * hashCode + getField().hashCode();
124 		hashCode = 29 * hashCode + ObjectUtils.nullSafeHashCode(getRejectedValue());
125 		hashCode = 29 * hashCode + (isBindingFailure() ? 1 : 0);
126 		return hashCode;
127 	}
128 
129 }
