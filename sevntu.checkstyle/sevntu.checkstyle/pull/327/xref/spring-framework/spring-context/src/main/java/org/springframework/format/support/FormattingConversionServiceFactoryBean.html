17  package org.springframework.format.support;
18  
19  import java.util.Set;
20  
21  import org.springframework.beans.factory.FactoryBean;
22  import org.springframework.beans.factory.InitializingBean;
23  import org.springframework.context.EmbeddedValueResolverAware;
24  import org.springframework.core.convert.support.ConversionServiceFactory;
25  import org.springframework.format.AnnotationFormatterFactory;
26  import org.springframework.format.Formatter;
27  import org.springframework.format.FormatterRegistrar;
28  import org.springframework.format.FormatterRegistry;
29  import org.springframework.format.Parser;
30  import org.springframework.format.Printer;
31  import org.springframework.util.StringValueResolver;
32  
33  /**
34   * A factory providing convenient access to a {@code FormattingConversionService}
35   * configured with converters and formatters for common types such as numbers and
36   * datetimes.
37   *
38   * <p>Additional converters and formatters can be registered declaratively through
39   * {@link #setConverters(Set)} and {@link #setFormatters(Set)}. Another option
40   * is to register converters and formatters in code by implementing the
41   * {@link FormatterRegistrar} interface. You can then configure provide the set
42   * of registrars to use through {@link #setFormatterRegistrars(Set)}.
43   *
44   * <p>A good example for registering converters and formatters in code is
45   * {@code JodaTimeFormatterRegistrar}, which registers a number of
46   * date-related formatters and converters. For a more detailed list of cases
47   * see {@link #setFormatterRegistrars(Set)}
48   *
49   * <p>Like all {@code FactoryBean} implementations, this class is suitable for
50   * use when configuring a Spring application context using Spring {@code <beans>}
51   * XML. When configuring the container with
52   * {@link org.springframework.context.annotation.Configuration @Configuration}
53   * classes, simply instantiate, configure and return the appropriate
54   * {@code FormattingConversionService} object from a
55   * {@link org.springframework.context.annotation.Bean @Bean} method.
56   *
57   * @author Keith Donald
58   * @author Juergen Hoeller
59   * @author Rossen Stoyanchev
60   * @author Chris Beams
61   * @since 3.0
62   */
63  public class FormattingConversionServiceFactoryBean
64  		implements FactoryBean<FormattingConversionService>, EmbeddedValueResolverAware, InitializingBean {
65  
66  	private Set<?> converters;
67  
68  	private Set<?> formatters;
69  
70  	private Set<FormatterRegistrar> formatterRegistrars;
71  
72  	private boolean registerDefaultFormatters = true;
73  
74  	private StringValueResolver embeddedValueResolver;
75  
76  	private FormattingConversionService conversionService;
77  
78  
79  	/**
80  	 * Configure the set of custom converter objects that should be added.
81  	 * @param converters instances of any of the following:
82  	 * {@link org.springframework.core.convert.converter.Converter},
83  	 * {@link org.springframework.core.convert.converter.ConverterFactory},
84  	 * {@link org.springframework.core.convert.converter.GenericConverter}
85  	 */
86  	public void setConverters(Set<?> converters) {
87  		this.converters = converters;
88  	}
89  
90  	/**
91  	 * Configure the set of custom formatter objects that should be added.
92  	 * @param formatters instances of {@link Formatter} or {@link AnnotationFormatterFactory}
93  	 */
94  	public void setFormatters(Set<?> formatters) {
95  		this.formatters = formatters;
96  	}
97  
98  	/**
99  	 * <p>Configure the set of FormatterRegistrars to invoke to register
100 	 * Converters and Formatters in addition to those added declaratively
101 	 * via {@link #setConverters(Set)} and {@link #setFormatters(Set)}.
102 	 * <p>FormatterRegistrars are useful when registering multiple related
103 	 * converters and formatters for a formatting category, such as Date
104 	 * formatting. All types related needed to support the formatting
105 	 * category can be registered from one place.
106 	 * <p>FormatterRegistrars can also be used to register Formatters
107 	 * indexed under a specific field type different from its own &lt;T&gt;,
108 	 * or when registering a Formatter from a Printer/Parser pair.
109 	 * @see FormatterRegistry#addFormatterForFieldType(Class, Formatter)
110 	 * @see FormatterRegistry#addFormatterForFieldType(Class, Printer, Parser)
111 	 */
112 	public void setFormatterRegistrars(Set<FormatterRegistrar> formatterRegistrars) {
113 		this.formatterRegistrars = formatterRegistrars;
114 	}
115 
116 	/**
117 	 * Indicate whether default formatters should be registered or not.
118 	 * <p>By default, built-in formatters are registered. This flag can be used
119 	 * to turn that off and rely on explicitly registered formatters only.
120 	 * @see #setFormatters(Set)
121 	 * @see #setFormatterRegistrars(Set)
122 	 */
123 	public void setRegisterDefaultFormatters(boolean registerDefaultFormatters) {
124 		this.registerDefaultFormatters = registerDefaultFormatters;
125 	}
126 
127 	@Override
128 	public void setEmbeddedValueResolver(StringValueResolver embeddedValueResolver) {
129 		this.embeddedValueResolver = embeddedValueResolver;
130 	}
131 
132 
133 	@Override
134 	public void afterPropertiesSet() {
135 		this.conversionService = new DefaultFormattingConversionService(this.embeddedValueResolver, this.registerDefaultFormatters);
136 		ConversionServiceFactory.registerConverters(this.converters, this.conversionService);
137 		registerFormatters();
138 	}
139 
140 	private void registerFormatters() {
141 		if (this.formatters != null) {
142 			for (Object formatter : this.formatters) {
143 				if (formatter instanceof Formatter<?>) {
144 					this.conversionService.addFormatter((Formatter<?>) formatter);
145 				}
146 				else if (formatter instanceof AnnotationFormatterFactory<?>) {
147 					this.conversionService.addFormatterForFieldAnnotation((AnnotationFormatterFactory<?>) formatter);
148 				}
149 				else {
150 					throw new IllegalArgumentException(
151 							"Custom formatters must be implementations of Formatter or AnnotationFormatterFactory");
152 				}
153 			}
154 		}
155 		if (this.formatterRegistrars != null) {
156 			for (FormatterRegistrar registrar : this.formatterRegistrars) {
157 				registrar.registerFormatters(this.conversionService);
158 			}
159 		}
160 	}
161 
162 
163 	@Override
164 	public FormattingConversionService getObject() {
165 		return this.conversionService;
166 	}
167 
168 	@Override
169 	public Class<? extends FormattingConversionService> getObjectType() {
170 		return FormattingConversionService.class;
171 	}
172 
173 	@Override
174 	public boolean isSingleton() {
175 		return true;
176 	}
177 
178 }
