17  package org.springframework.context.annotation;
18  
19  import java.io.IOException;
20  import java.util.HashSet;
21  import java.util.Set;
22  
23  import org.apache.commons.logging.Log;
24  import org.apache.commons.logging.LogFactory;
25  
26  import org.springframework.beans.factory.annotation.AnnotatedBeanDefinition;
27  import org.springframework.beans.factory.config.BeanDefinition;
28  import org.springframework.beans.factory.support.AbstractBeanDefinition;
29  import org.springframework.core.Conventions;
30  import org.springframework.core.type.AnnotationMetadata;
31  import org.springframework.core.type.StandardAnnotationMetadata;
32  import org.springframework.core.type.classreading.MetadataReader;
33  import org.springframework.core.type.classreading.MetadataReaderFactory;
34  import org.springframework.stereotype.Component;
35  
36  /**
37   * Utilities for processing @{@link Configuration} classes.
38   *
39   * @author Chris Beams
40   * @author Juergen Hoeller
41   * @since 3.1
42   */
43  abstract class ConfigurationClassUtils {
44  
45  	private static final String CONFIGURATION_CLASS_FULL = "full";
46  
47  	private static final String CONFIGURATION_CLASS_LITE = "lite";
48  
49  	private static final String CONFIGURATION_CLASS_ATTRIBUTE =
50  			Conventions.getQualifiedAttributeName(ConfigurationClassPostProcessor.class, "configurationClass");
51  
52  
53  	private static final Log logger = LogFactory.getLog(ConfigurationClassUtils.class);
54  
55  	private static final Set<String> candidateIndicators = new HashSet<String>(4);
56  
57  	static {
58  		candidateIndicators.add(Component.class.getName());
59  		candidateIndicators.add(ComponentScan.class.getName());
60  		candidateIndicators.add(Import.class.getName());
61  		candidateIndicators.add(ImportResource.class.getName());
62  	}
63  
64  
65  	/**
66  	 * Check whether the given bean definition is a candidate for a configuration class
67  	 * (or a nested component class declared within a configuration/component class,
68  	 * to be auto-registered as well), and mark it accordingly.
69  	 * @param beanDef the bean definition to check
70  	 * @param metadataReaderFactory the current factory in use by the caller
71  	 * @return whether the candidate qualifies as (any kind of) configuration class
72  	 */
73  	public static boolean checkConfigurationClassCandidate(BeanDefinition beanDef, MetadataReaderFactory metadataReaderFactory) {
74  		String className = beanDef.getBeanClassName();
75  		if (className == null) {
76  			return false;
77  		}
78  
79  		AnnotationMetadata metadata;
80  		if (beanDef instanceof AnnotatedBeanDefinition &&
81  				className.equals(((AnnotatedBeanDefinition) beanDef).getMetadata().getClassName())) {
82  			// Can reuse the pre-parsed metadata from the given BeanDefinition...
83  			metadata = ((AnnotatedBeanDefinition) beanDef).getMetadata();
84  		}
85  		else if (beanDef instanceof AbstractBeanDefinition && ((AbstractBeanDefinition) beanDef).hasBeanClass()) {
86  			// Check already loaded Class if present...
87  			// since we possibly can't even load the class file for this Class.
88  			Class<?> beanClass = ((AbstractBeanDefinition) beanDef).getBeanClass();
89  			metadata = new StandardAnnotationMetadata(beanClass, true);
90  		}
91  		else {
92  			try {
93  				MetadataReader metadataReader = metadataReaderFactory.getMetadataReader(className);
94  				metadata = metadataReader.getAnnotationMetadata();
95  			}
96  			catch (IOException ex) {
97  				if (logger.isDebugEnabled()) {
98  					logger.debug("Could not find class file for introspecting configuration annotations: " + className, ex);
99  				}
100 				return false;
101 			}
102 		}
103 
104 		if (isFullConfigurationCandidate(metadata)) {
105 			beanDef.setAttribute(CONFIGURATION_CLASS_ATTRIBUTE, CONFIGURATION_CLASS_FULL);
106 			return true;
107 		}
108 		else if (isLiteConfigurationCandidate(metadata)) {
109 			beanDef.setAttribute(CONFIGURATION_CLASS_ATTRIBUTE, CONFIGURATION_CLASS_LITE);
110 			return true;
111 		}
112 		else {
113 			return false;
114 		}
115 	}
116 
117 	/**
118 	 * Check the given metadata for a configuration class candidate
119 	 * (or nested component class declared within a configuration/component class).
120 	 * @param metadata the metadata of the annotated class
121 	 * @return {@code true} if the given class is to be registered as a
122 	 * reflection-detected bean definition; {@code false} otherwise
123 	 */
124 	public static boolean isConfigurationCandidate(AnnotationMetadata metadata) {
125 		return (isFullConfigurationCandidate(metadata) || isLiteConfigurationCandidate(metadata));
126 	}
127 
128 	/**
129 	 * Check the given metadata for a full configuration class candidate
130 	 * (i.e. a class annotated with {@code @Configuration}).
131 	 * @param metadata the metadata of the annotated class
132 	 * @return {@code true} if the given class is to be processed as a full
133 	 * configuration class, including cross-method call interception
134 	 */
135 	public static boolean isFullConfigurationCandidate(AnnotationMetadata metadata) {
136 		return metadata.isAnnotated(Configuration.class.getName());
137 	}
138 
139 	/**
140 	 * Check the given metadata for a lite configuration class candidate
141 	 * (e.g. a class annotated with {@code @Component} or just having
142 	 * {@code @Import} declarations or {@code @Bean methods}).
143 	 * @param metadata the metadata of the annotated class
144 	 * @return {@code true} if the given class is to be processed as a lite
145 	 * configuration class, just registering it and scanning it for {@code @Bean} methods
146 	 */
147 	public static boolean isLiteConfigurationCandidate(AnnotationMetadata metadata) {
148 		// Do not consider an interface or an annotation...
149 		if (metadata.isInterface()) {
150 			return false;
151 		}
152 		for (String indicator : candidateIndicators) {
153 			if (metadata.isAnnotated(indicator)) {
154 				return true;
155 			}
156 		}
157 		return metadata.hasAnnotatedMethods(Bean.class.getName());
158 	}
159 
160 	/**
161 	 * Determine whether the given bean definition indicates a full {@code @Configuration}
162 	 * class, through checking {@link #checkConfigurationClassCandidate}'s metadata marker.
163 	 */
164 	public static boolean isFullConfigurationClass(BeanDefinition beanDef) {
165 		return CONFIGURATION_CLASS_FULL.equals(beanDef.getAttribute(CONFIGURATION_CLASS_ATTRIBUTE));
166 	}
167 
168 	/**
169 	 * Determine whether the given bean definition indicates a lite {@code @Configuration}
170 	 * class, through checking {@link #checkConfigurationClassCandidate}'s metadata marker.
171 	 */
172 	public static boolean isLiteConfigurationClass(BeanDefinition beanDef) {
173 		return CONFIGURATION_CLASS_LITE.equals(beanDef.getAttribute(CONFIGURATION_CLASS_ATTRIBUTE));
174 	}
175 
176 }
