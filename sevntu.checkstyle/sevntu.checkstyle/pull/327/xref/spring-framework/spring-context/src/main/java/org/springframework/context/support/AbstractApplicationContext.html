80  /**
81   * Abstract implementation of the {@link org.springframework.context.ApplicationContext}
82   * interface. Doesn't mandate the type of storage used for configuration; simply
83   * implements common context functionality. Uses the Template Method design pattern,
84   * requiring concrete subclasses to implement abstract methods.
85   *
86   * <p>In contrast to a plain BeanFactory, an ApplicationContext is supposed
87   * to detect special beans defined in its internal bean factory:
88   * Therefore, this class automatically registers
89   * {@link org.springframework.beans.factory.config.BeanFactoryPostProcessor BeanFactoryPostProcessors},
90   * {@link org.springframework.beans.factory.config.BeanPostProcessor BeanPostProcessors}
91   * and {@link org.springframework.context.ApplicationListener ApplicationListeners}
92   * which are defined as beans in the context.
93   *
94   * <p>A {@link org.springframework.context.MessageSource} may also be supplied
95   * as a bean in the context, with the name "messageSource"; otherwise, message
96   * resolution is delegated to the parent context. Furthermore, a multicaster
97   * for application events can be supplied as "applicationEventMulticaster" bean
98   * of type {@link org.springframework.context.event.ApplicationEventMulticaster}
99   * in the context; otherwise, a default multicaster of type
100  * {@link org.springframework.context.event.SimpleApplicationEventMulticaster} will be used.
101  *
102  * <p>Implements resource loading through extending
103  * {@link org.springframework.core.io.DefaultResourceLoader}.
104  * Consequently treats non-URL resource paths as class path resources
105  * (supporting full class path resource names that include the package path,
106  * e.g. "mypackage/myresource.dat"), unless the {@link #getResourceByPath}
107  * method is overwritten in a subclass.
108  *
109  * @author Rod Johnson
110  * @author Juergen Hoeller
111  * @author Mark Fisher
112  * @author Stephane Nicoll
113  * @since January 21, 2001
114  * @see #refreshBeanFactory
115  * @see #getBeanFactory
116  * @see org.springframework.beans.factory.config.BeanFactoryPostProcessor
117  * @see org.springframework.beans.factory.config.BeanPostProcessor
118  * @see org.springframework.context.event.ApplicationEventMulticaster
119  * @see org.springframework.context.ApplicationListener
120  * @see org.springframework.context.MessageSource
121  */
122 public abstract class AbstractApplicationContext extends DefaultResourceLoader
123 		implements ConfigurableApplicationContext, DisposableBean {

338 	protected void publishEvent(Object event, ResolvableType eventType) {
339 		Assert.notNull(event, "Event must not be null");
340 		if (logger.isTraceEnabled()) {
341 			logger.trace("Publishing event in " + getDisplayName() + ": " + event);
342 		}
343 		final ApplicationEvent applicationEvent;
344 		if (event instanceof ApplicationEvent) {
345 			applicationEvent = (ApplicationEvent) event;
346 		}
347 		else {
348 			applicationEvent = new PayloadApplicationEvent<Object>(this, event);
349 			if (eventType == null) {
350 				eventType = ResolvableType.forClassWithGenerics(PayloadApplicationEvent.class, event.getClass());
351 			}
352 		}
353 		getApplicationEventMulticaster().multicastEvent(applicationEvent, eventType);
354 		if (this.parent != null) {
355 			if (this.parent instanceof AbstractApplicationContext) {
356 				((AbstractApplicationContext) this.parent).publishEvent(event, eventType);
357 			}
358 			else {
359 				this.parent.publishEvent(event);
360 			}
361 		}
362 	}

472 	@Override
473 	public void refresh() throws BeansException, IllegalStateException {
474 		synchronized (this.startupShutdownMonitor) {
475 			// Prepare this context for refreshing.
476 			prepareRefresh();
477 
478 			// Tell the subclass to refresh the internal bean factory.
479 			ConfigurableListableBeanFactory beanFactory = obtainFreshBeanFactory();
480 
481 			// Prepare the bean factory for use in this context.
482 			prepareBeanFactory(beanFactory);
483 
484 			try {
485 				// Allows post-processing of the bean factory in context subclasses.
486 				postProcessBeanFactory(beanFactory);
487 
488 				// Invoke factory processors registered as beans in the context.
489 				invokeBeanFactoryPostProcessors(beanFactory);
490 
491 				// Register bean processors that intercept bean creation.
492 				registerBeanPostProcessors(beanFactory);
493 
494 				// Initialize message source for this context.
495 				initMessageSource();
496 
497 				// Initialize event multicaster for this context.
498 				initApplicationEventMulticaster();
499 
500 				// Initialize other special beans in specific context subclasses.
501 				onRefresh();
502 
503 				// Check for listener beans and register them.
504 				registerListeners();
505 
506 				// Instantiate all remaining (non-lazy-init) singletons.
507 				finishBeanFactoryInitialization(beanFactory);
508 
509 				// Last step: publish corresponding event.
510 				finishRefresh();
511 			}
512 
513 			catch (BeansException ex) {
514 				logger.warn("Exception encountered during context initialization - cancelling refresh attempt", ex);
515 
516 				// Destroy already created singletons to avoid dangling resources.
517 				destroyBeans();
518 
519 				// Reset 'active' flag.
520 				cancelRefresh(ex);
521 
522 				// Propagate exception to caller.
523 				throw ex;
524 			}
525 		}
526 	}

572 	/**
573 	 * Configure the factory's standard context characteristics,
574 	 * such as the context's ClassLoader and post-processors.
575 	 * @param beanFactory the BeanFactory to configure
576 	 */
577 	protected void prepareBeanFactory(ConfigurableListableBeanFactory beanFactory) {
578 		// Tell the internal bean factory to use the context's class loader etc.
579 		beanFactory.setBeanClassLoader(getClassLoader());
580 		beanFactory.setBeanExpressionResolver(new StandardBeanExpressionResolver(beanFactory.getBeanClassLoader()));
581 		beanFactory.addPropertyEditorRegistrar(new ResourceEditorRegistrar(this, getEnvironment()));
582 
583 		// Configure the bean factory with context callbacks.
584 		beanFactory.addBeanPostProcessor(new ApplicationContextAwareProcessor(this));
585 		beanFactory.ignoreDependencyInterface(ResourceLoaderAware.class);
586 		beanFactory.ignoreDependencyInterface(ApplicationEventPublisherAware.class);
587 		beanFactory.ignoreDependencyInterface(MessageSourceAware.class);
588 		beanFactory.ignoreDependencyInterface(ApplicationContextAware.class);
589 		beanFactory.ignoreDependencyInterface(EnvironmentAware.class);
590 
591 		// BeanFactory interface not registered as resolvable type in a plain factory.
592 		// MessageSource registered (and found for autowiring) as a bean.
593 		beanFactory.registerResolvableDependency(BeanFactory.class, beanFactory);
594 		beanFactory.registerResolvableDependency(ResourceLoader.class, this);
595 		beanFactory.registerResolvableDependency(ApplicationEventPublisher.class, this);
596 		beanFactory.registerResolvableDependency(ApplicationContext.class, this);
597 
598 		// Detect a LoadTimeWeaver and prepare for weaving, if found.
599 		if (beanFactory.containsBean(LOAD_TIME_WEAVER_BEAN_NAME)) {
600 			beanFactory.addBeanPostProcessor(new LoadTimeWeaverAwareProcessor(beanFactory));
601 			// Set a temporary ClassLoader for type matching.
602 			beanFactory.setTempClassLoader(new ContextTypeMatchClassLoader(beanFactory.getBeanClassLoader()));
603 		}
604 
605 		// Register default environment beans.
606 		if (!beanFactory.containsLocalBean(ENVIRONMENT_BEAN_NAME)) {
607 			beanFactory.registerSingleton(ENVIRONMENT_BEAN_NAME, getEnvironment());
608 		}
609 		if (!beanFactory.containsLocalBean(SYSTEM_PROPERTIES_BEAN_NAME)) {
610 			beanFactory.registerSingleton(SYSTEM_PROPERTIES_BEAN_NAME, getEnvironment().getSystemProperties());
611 		}
612 		if (!beanFactory.containsLocalBean(SYSTEM_ENVIRONMENT_BEAN_NAME)) {
613 			beanFactory.registerSingleton(SYSTEM_ENVIRONMENT_BEAN_NAME, getEnvironment().getSystemEnvironment());
614 		}
615 	}

645 	/**
646 	 * Initialize the MessageSource.
647 	 * Use parent's if none defined in this context.
648 	 */
649 	protected void initMessageSource() {
650 		ConfigurableListableBeanFactory beanFactory = getBeanFactory();
651 		if (beanFactory.containsLocalBean(MESSAGE_SOURCE_BEAN_NAME)) {
652 			this.messageSource = beanFactory.getBean(MESSAGE_SOURCE_BEAN_NAME, MessageSource.class);
653 			// Make MessageSource aware of parent MessageSource.
654 			if (this.parent != null && this.messageSource instanceof HierarchicalMessageSource) {
655 				HierarchicalMessageSource hms = (HierarchicalMessageSource) this.messageSource;
656 				if (hms.getParentMessageSource() == null) {
657 					// Only set parent context as parent MessageSource if no parent MessageSource

761 	 * initializing all remaining singleton beans.
762 	 */
763 	protected void finishBeanFactoryInitialization(ConfigurableListableBeanFactory beanFactory) {
764 		// Initialize conversion service for this context.
765 		if (beanFactory.containsBean(CONVERSION_SERVICE_BEAN_NAME) &&
766 				beanFactory.isTypeMatch(CONVERSION_SERVICE_BEAN_NAME, ConversionService.class)) {
767 			beanFactory.setConversionService(
768 					beanFactory.getBean(CONVERSION_SERVICE_BEAN_NAME, ConversionService.class));
769 		}
770 
771 		// Initialize LoadTimeWeaverAware beans early to allow for registering their transformers early.
772 		String[] weaverAwareNames = beanFactory.getBeanNamesForType(LoadTimeWeaverAware.class, false, false);
773 		for (String weaverAwareName : weaverAwareNames) {
774 			getBean(weaverAwareName);
775 		}
776 
777 		// Stop using the temporary ClassLoader for type matching.
778 		beanFactory.setTempClassLoader(null);
779 
780 		// Allow for caching all bean definition metadata, not expecting further changes.
781 		beanFactory.freezeConfiguration();
782 
783 		// Instantiate all remaining (non-lazy-init) singletons.
784 		beanFactory.preInstantiateSingletons();
785 	}

853 	/**
854 	 * Close this application context, destroying all beans in its bean factory.
855 	 * <p>Delegates to {@code doClose()} for the actual closing procedure.
856 	 * Also removes a JVM shutdown hook, if registered, as it's not needed anymore.
857 	 * @see #doClose()
858 	 * @see #registerShutdownHook()
859 	 */
860 	@Override
861 	public void close() {
862 		synchronized (this.startupShutdownMonitor) {
863 			doClose();
864 			// If we registered a JVM shutdown hook, we don't need it anymore now:
865 			// We've already explicitly closed the context.
866 			if (this.shutdownHook != null) {
867 				try {
868 					Runtime.getRuntime().removeShutdownHook(this.shutdownHook);
869 				}
870 				catch (IllegalStateException ex) {
871 					// ignore - VM is already shutting down
872 				}
873 			}
874 		}
875 	}

877 	/**
878 	 * Actually performs context closing: publishes a ContextClosedEvent and
879 	 * destroys the singletons in the bean factory of this application context.
880 	 * <p>Called by both {@code close()} and a JVM shutdown hook, if any.
881 	 * @see org.springframework.context.event.ContextClosedEvent
882 	 * @see #destroyBeans()
883 	 * @see #close()
884 	 * @see #registerShutdownHook()
885 	 */
886 	protected void doClose() {
887 		if (this.active.get() && this.closed.compareAndSet(false, true)) {
888 			if (logger.isInfoEnabled()) {
889 				logger.info("Closing " + this);
890 			}
891 
892 			LiveBeansView.unregisterApplicationContext(this);
893 
894 			try {
895 				// Publish shutdown event.
896 				publishEvent(new ContextClosedEvent(this));
897 			}
898 			catch (Throwable ex) {
899 				logger.warn("Exception thrown from ApplicationListener handling ContextClosedEvent", ex);
900 			}
901 
902 			// Stop all Lifecycle beans, to avoid delays during individual destruction.
903 			try {
904 				getLifecycleProcessor().onClose();
905 			}
906 			catch (Throwable ex) {
907 				logger.warn("Exception thrown from LifecycleProcessor on context close", ex);
908 			}
909 
910 			// Destroy all cached singletons in the context's BeanFactory.
911 			destroyBeans();
912 
913 			// Close the state of this context itself.
914 			closeBeanFactory();
915 
916 			// Let subclasses do some final clean-up if they wish...
917 			onClose();
918 
919 			this.active.set(false);
920 		}
921 	}
