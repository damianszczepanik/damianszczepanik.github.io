17  package org.springframework.context.support;
18  
19  import java.text.MessageFormat;
20  import java.util.HashMap;
21  import java.util.Locale;
22  import java.util.Map;
23  
24  import org.springframework.util.Assert;
25  
26  /**
27   * Simple implementation of {@link org.springframework.context.MessageSource}
28   * which allows messages to be registered programmatically.
29   * This MessageSource supports basic internationalization.
30   *
31   * <p>Intended for testing rather than for use in production systems.
32   *
33   * @author Rod Johnson
34   * @author Juergen Hoeller
35   */
36  public class StaticMessageSource extends AbstractMessageSource {
37  
38  	/** Map from 'code + locale' keys to message Strings */
39  	private final Map<String, String> messages = new HashMap<String, String>();
40  
41  	private final Map<String, MessageFormat> cachedMessageFormats = new HashMap<String, MessageFormat>();
42  
43  
44  	@Override
45  	protected String resolveCodeWithoutArguments(String code, Locale locale) {
46  		return this.messages.get(code + "_" + locale.toString());
47  	}
48  
49  	@Override
50  	protected MessageFormat resolveCode(String code, Locale locale) {
51  		String key = code + "_" + locale.toString();
52  		String msg = this.messages.get(key);
53  		if (msg == null) {
54  			return null;
55  		}
56  		synchronized (this.cachedMessageFormats) {
57  			MessageFormat messageFormat = this.cachedMessageFormats.get(key);
58  			if (messageFormat == null) {
59  				messageFormat = createMessageFormat(msg, locale);
60  				this.cachedMessageFormats.put(key, messageFormat);
61  			}
62  			return messageFormat;
63  		}
64  	}
65  
66  	/**
67  	 * Associate the given message with the given code.
68  	 * @param code the lookup code
69  	 * @param locale the locale that the message should be found within
70  	 * @param msg the message associated with this lookup code
71  	 */
72  	public void addMessage(String code, Locale locale, String msg) {
73  		Assert.notNull(code, "Code must not be null");
74  		Assert.notNull(locale, "Locale must not be null");
75  		Assert.notNull(msg, "Message must not be null");
76  		this.messages.put(code + "_" + locale.toString(), msg);
77  		if (logger.isDebugEnabled()) {
78  			logger.debug("Added message [" + msg + "] for code [" + code + "] and Locale [" + locale + "]");
79  		}
80  	}
81  
82  	/**
83  	 * Associate the given message values with the given keys as codes.
84  	 * @param messages the messages to register, with messages codes
85  	 * as keys and message texts as values
86  	 * @param locale the locale that the messages should be found within
87  	 */
88  	public void addMessages(Map<String, String> messages, Locale locale) {
89  		Assert.notNull(messages, "Messages Map must not be null");
90  		for (Map.Entry<String, String> entry : messages.entrySet()) {
91  			addMessage(entry.getKey(), locale, entry.getValue());
92  		}
93  	}
94  
95  
96  	@Override
97  	public String toString() {
98  		return getClass().getName() + ": " + this.messages;
99  	}
100 
101 }
