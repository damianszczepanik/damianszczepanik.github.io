31  /**
32   * Abstract implementation of the {@link HierarchicalMessageSource} interface,
33   * implementing common handling of message variants, making it easy
34   * to implement a specific strategy for a concrete MessageSource.
35   *
36   * <p>Subclasses must implement the abstract {@link #resolveCode}
37   * method. For efficient resolution of messages without arguments, the
38   * {@link #resolveCodeWithoutArguments} method should be overridden
39   * as well, resolving messages without a MessageFormat being involved.
40   *
41   * <p><b>Note:</b> By default, message texts are only parsed through
42   * MessageFormat if arguments have been passed in for the message. In case
43   * of no arguments, message texts will be returned as-is. As a consequence,
44   * you should only use MessageFormat escaping for messages with actual
45   * arguments, and keep all other messages unescaped. If you prefer to
46   * escape all messages, set the "alwaysUseMessageFormat" flag to "true".
47   *
48   * <p>Supports not only MessageSourceResolvables as primary messages
49   * but also resolution of message arguments that are in turn
50   * MessageSourceResolvables themselves.
51   *
52   * <p>This class does not implement caching of messages per code, thus
53   * subclasses can dynamically change messages over time. Subclasses are
54   * encouraged to cache their messages in a modification-aware fashion,
55   * allowing for hot deployment of updated messages.
56   *
57   * @author Juergen Hoeller
58   * @author Rod Johnson
59   * @see #resolveCode(String, java.util.Locale)
60   * @see #resolveCodeWithoutArguments(String, java.util.Locale)
61   * @see #setAlwaysUseMessageFormat
62   * @see java.text.MessageFormat
63   */
64  public abstract class AbstractMessageSource extends MessageSourceSupport implements HierarchicalMessageSource {

100 	/**
101 	 * Set whether to use the message code as default message instead of
102 	 * throwing a NoSuchMessageException. Useful for development and debugging.
103 	 * Default is "false".
104 	 * <p>Note: In case of a MessageSourceResolvable with multiple codes
105 	 * (like a FieldError) and a MessageSource that has a parent MessageSource,
106 	 * do <i>not</i> activate "useCodeAsDefaultMessage" in the <i>parent</i>:
107 	 * Else, you'll get the first code returned as message by the parent,
108 	 * without attempts to check further codes.
109 	 * <p>To be able to work with "useCodeAsDefaultMessage" turned on in the parent,
110 	 * AbstractMessageSource and AbstractApplicationContext contain special checks
111 	 * to delegate to the internal {@link #getMessageInternal} method if available.
112 	 * In general, it is recommended to just use "useCodeAsDefaultMessage" during
113 	 * development and not rely on it in production in the first place, though.
114 	 * @see #getMessage(String, Object[], Locale)
115 	 * @see org.springframework.validation.FieldError
116 	 */
117 	public void setUseCodeAsDefaultMessage(boolean useCodeAsDefaultMessage) {
118 		this.useCodeAsDefaultMessage = useCodeAsDefaultMessage;
119 	}

121 	/**
122 	 * Return whether to use the message code as default message instead of
123 	 * throwing a NoSuchMessageException. Useful for development and debugging.
124 	 * Default is "false".
125 	 * <p>Alternatively, consider overriding the {@link #getDefaultMessage}
126 	 * method to return a custom fallback message for an unresolvable code.
127 	 * @see #getDefaultMessage(String)
128 	 */
129 	protected boolean isUseCodeAsDefaultMessage() {
130 		return this.useCodeAsDefaultMessage;
131 	}

134 	@Override
135 	public final String getMessage(String code, Object[] args, String defaultMessage, Locale locale) {
136 		String msg = getMessageInternal(code, args, locale);
137 		if (msg != null) {
138 			return msg;
139 		}
140 		if (defaultMessage == null) {
141 			String fallback = getDefaultMessage(code);
142 			if (fallback != null) {
143 				return fallback;
144 			}
145 		}
146 		return renderDefaultMessage(defaultMessage, args, locale);
147 	}
148 
149 	@Override
150 	public final String getMessage(String code, Object[] args, Locale locale) throws NoSuchMessageException {
151 		String msg = getMessageInternal(code, args, locale);
152 		if (msg != null) {
153 			return msg;
154 		}
155 		String fallback = getDefaultMessage(code);
156 		if (fallback != null) {
157 			return fallback;
158 		}
159 		throw new NoSuchMessageException(code, locale);
160 	}
161 
162 	@Override
163 	public final String getMessage(MessageSourceResolvable resolvable, Locale locale)
164 			throws NoSuchMessageException {
165 
166 		String[] codes = resolvable.getCodes();
167 		if (codes == null) {
168 			codes = new String[0];
169 		}
170 		for (String code : codes) {
171 			String msg = getMessageInternal(code, resolvable.getArguments(), locale);
172 			if (msg != null) {
173 				return msg;
174 			}
175 		}
176 		String defaultMessage = resolvable.getDefaultMessage();
177 		if (defaultMessage != null) {
178 			return renderDefaultMessage(defaultMessage, resolvable.getArguments(), locale);
179 		}
180 		if (codes.length > 0) {
181 			String fallback = getDefaultMessage(codes[0]);
182 			if (fallback != null) {
183 				return fallback;
184 			}
185 		}
186 		throw new NoSuchMessageException(codes.length > 0 ? codes[codes.length - 1] : null, locale);
187 	}

190 	/**
191 	 * Resolve the given code and arguments as message in the given Locale,
192 	 * returning {@code null} if not found. Does <i>not</i> fall back to
193 	 * the code as default message. Invoked by {@code getMessage} methods.
194 	 * @param code the code to lookup up, such as 'calculator.noRateSet'
195 	 * @param args array of arguments that will be filled in for params
196 	 * within the message
197 	 * @param locale the Locale in which to do the lookup
198 	 * @return the resolved message, or {@code null} if not found
199 	 * @see #getMessage(String, Object[], String, Locale)
200 	 * @see #getMessage(String, Object[], Locale)
201 	 * @see #getMessage(MessageSourceResolvable, Locale)
202 	 * @see #setUseCodeAsDefaultMessage
203 	 */
204 	protected String getMessageInternal(String code, Object[] args, Locale locale) {
205 		if (code == null) {
206 			return null;
207 		}
208 		if (locale == null) {
209 			locale = Locale.getDefault();
210 		}
211 		Object[] argsToUse = args;
212 
213 		if (!isAlwaysUseMessageFormat() && ObjectUtils.isEmpty(args)) {
214 			// Optimized resolution: no arguments to apply,
215 			// therefore no MessageFormat needs to be involved.
216 			// Note that the default implementation still uses MessageFormat;
217 			// this can be overridden in specific subclasses.
218 			String message = resolveCodeWithoutArguments(code, locale);
219 			if (message != null) {
220 				return message;
221 			}
222 		}
223 
224 		else {
225 			// Resolve arguments eagerly, for the case where the message
226 			// is defined in a parent MessageSource but resolvable arguments
227 			// are defined in the child MessageSource.
228 			argsToUse = resolveArguments(args, locale);
229 
230 			MessageFormat messageFormat = resolveCode(code, locale);
231 			if (messageFormat != null) {
232 				synchronized (messageFormat) {
233 					return messageFormat.format(argsToUse);
234 				}
235 			}
236 		}
237 
238 		// Check locale-independent common messages for the given message code.
239 		Properties commonMessages = getCommonMessages();
240 		if (commonMessages != null) {
241 			String commonMessage = commonMessages.getProperty(code);
242 			if (commonMessage != null) {
243 				return formatMessage(commonMessage, args, locale);
244 			}
245 		}
246 
247 		// Not found -> check parent, if any.
248 		return getMessageFromParent(code, argsToUse, locale);
249 	}
250 
251 	/**
252 	 * Try to retrieve the given message from the parent MessageSource, if any.
253 	 * @param code the code to lookup up, such as 'calculator.noRateSet'
254 	 * @param args array of arguments that will be filled in for params
255 	 * within the message
256 	 * @param locale the Locale in which to do the lookup
257 	 * @return the resolved message, or {@code null} if not found
258 	 * @see #getParentMessageSource()
259 	 */
260 	protected String getMessageFromParent(String code, Object[] args, Locale locale) {
261 		MessageSource parent = getParentMessageSource();
262 		if (parent != null) {
263 			if (parent instanceof AbstractMessageSource) {
264 				// Call internal method to avoid getting the default code back
265 				// in case of "useCodeAsDefaultMessage" being activated.
266 				return ((AbstractMessageSource) parent).getMessageInternal(code, args, locale);
267 			}
268 			else {
269 				// Check parent MessageSource, returning null if not found there.
270 				return parent.getMessage(code, args, null, locale);
271 			}
272 		}
273 		// Not found in parent either.
274 		return null;
275 	}
276 
277 	/**
278 	 * Return a fallback default message for the given code, if any.
279 	 * <p>Default is to return the code itself if "useCodeAsDefaultMessage" is activated,
280 	 * or return no fallback else. In case of no fallback, the caller will usually
281 	 * receive a NoSuchMessageException from {@code getMessage}.
282 	 * @param code the message code that we couldn't resolve
283 	 * and that we didn't receive an explicit default message for
284 	 * @return the default message to use, or {@code null} if none
285 	 * @see #setUseCodeAsDefaultMessage
286 	 */
287 	protected String getDefaultMessage(String code) {
288 		if (isUseCodeAsDefaultMessage()) {
289 			return code;
290 		}
291 		return null;
292 	}

295 	/**
296 	 * Searches through the given array of objects, finds any MessageSourceResolvable
297 	 * objects and resolves them.
298 	 * <p>Allows for messages to have MessageSourceResolvables as arguments.
299 	 * @param args array of arguments for a message
300 	 * @param locale the locale to resolve through
301 	 * @return an array of arguments with any MessageSourceResolvables resolved
302 	 */
303 	@Override
304 	protected Object[] resolveArguments(Object[] args, Locale locale) {
305 		if (args == null) {
306 			return new Object[0];
307 		}
308 		List<Object> resolvedArgs = new ArrayList<Object>(args.length);
309 		for (Object arg : args) {
310 			if (arg instanceof MessageSourceResolvable) {
311 				resolvedArgs.add(getMessage((MessageSourceResolvable) arg, locale));
312 			}
313 			else {
314 				resolvedArgs.add(arg);
315 			}
316 		}
317 		return resolvedArgs.toArray(new Object[resolvedArgs.size()]);
318 	}

320 	/**
321 	 * Subclasses can override this method to resolve a message without arguments
322 	 * in an optimized fashion, i.e. to resolve without involving a MessageFormat.
323 	 * <p>The default implementation <i>does</i> use MessageFormat, through
324 	 * delegating to the {@link #resolveCode} method. Subclasses are encouraged
325 	 * to replace this with optimized resolution.
326 	 * <p>Unfortunately, {@code java.text.MessageFormat} is not implemented
327 	 * in an efficient fashion. In particular, it does not detect that a message
328 	 * pattern doesn't contain argument placeholders in the first place. Therefore,
329 	 * it is advisable to circumvent MessageFormat for messages without arguments.
330 	 * @param code the code of the message to resolve
331 	 * @param locale the Locale to resolve the code for
332 	 * (subclasses are encouraged to support internationalization)
333 	 * @return the message String, or {@code null} if not found
334 	 * @see #resolveCode
335 	 * @see java.text.MessageFormat
336 	 */
337 	protected String resolveCodeWithoutArguments(String code, Locale locale) {
338 		MessageFormat messageFormat = resolveCode(code, locale);
339 		if (messageFormat != null) {
340 			synchronized (messageFormat) {
341 				return messageFormat.format(new Object[0]);
342 			}
343 		}
344 		return null;
345 	}

347 	/**
348 	 * Subclasses must implement this method to resolve a message.
349 	 * <p>Returns a MessageFormat instance rather than a message String,
350 	 * to allow for appropriate caching of MessageFormats in subclasses.
351 	 * <p><b>Subclasses are encouraged to provide optimized resolution
352 	 * for messages without arguments, not involving MessageFormat.</b>
353 	 * See the {@link #resolveCodeWithoutArguments} javadoc for details.
354 	 * @param code the code of the message to resolve
355 	 * @param locale the Locale to resolve the code for
356 	 * (subclasses are encouraged to support internationalization)
357 	 * @return the MessageFormat for the message, or {@code null} if not found
358 	 * @see #resolveCodeWithoutArguments(String, java.util.Locale)
359 	 */
360 	protected abstract MessageFormat resolveCode(String code, Locale locale);
