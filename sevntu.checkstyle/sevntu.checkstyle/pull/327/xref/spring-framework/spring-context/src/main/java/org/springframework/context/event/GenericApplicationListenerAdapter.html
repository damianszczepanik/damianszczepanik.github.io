17  package org.springframework.context.event;
18  
19  import org.springframework.aop.support.AopUtils;
20  import org.springframework.context.ApplicationEvent;
21  import org.springframework.context.ApplicationListener;
22  import org.springframework.core.Ordered;
23  import org.springframework.core.ResolvableType;
24  import org.springframework.util.Assert;
25  
26  /**
27   * {@link GenericApplicationListener} adapter that determines supported event types
28   * through introspecting the generically declared type of the target listener.
29   *
30   * @author Juergen Hoeller
31   * @author Stephane Nicoll
32   * @since 3.0
33   * @see org.springframework.context.ApplicationListener#onApplicationEvent
34   */
35  public class GenericApplicationListenerAdapter implements GenericApplicationListener {
36  
37  	private final ApplicationListener<ApplicationEvent> delegate;
38  
39  	private final ResolvableType declaredEventType;
40  
41  	/**
42  	 * Create a new GenericApplicationListener for the given delegate.
43  	 * @param delegate the delegate listener to be invoked
44  	 */
45  	@SuppressWarnings("unchecked")
46  	public GenericApplicationListenerAdapter(ApplicationListener<?> delegate) {
47  		Assert.notNull(delegate, "Delegate listener must not be null");
48  		this.delegate = (ApplicationListener<ApplicationEvent>) delegate;
49  		this.declaredEventType = resolveDeclaredEventType(this.delegate);
50  	}
51  
52  
53  	@Override
54  	public void onApplicationEvent(ApplicationEvent event) {
55  		this.delegate.onApplicationEvent(event);
56  	}
57  
58  	@Override
59  	@SuppressWarnings("unchecked")
60  	public boolean supportsEventType(ResolvableType eventType) {
61  		if (this.delegate instanceof SmartApplicationListener) {
62  			Class<? extends ApplicationEvent> eventClass = (Class<? extends ApplicationEvent>) eventType.getRawClass();
63  			return ((SmartApplicationListener) this.delegate).supportsEventType(eventClass);
64  		}
65  		else {
66  			return (this.declaredEventType == null || this.declaredEventType.isAssignableFrom(eventType));
67  		}
68  	}
69  
70  	@Override
71  	public boolean supportsSourceType(Class<?> sourceType) {
72  		if (this.delegate instanceof SmartApplicationListener) {
73  			return ((SmartApplicationListener) this.delegate).supportsSourceType(sourceType);
74  		}
75  		else {
76  			return true;
77  		}
78  	}
79  
80  	@Override
81  	public int getOrder() {
82  		return (this.delegate instanceof Ordered ? ((Ordered) this.delegate).getOrder() : Ordered.LOWEST_PRECEDENCE);
83  	}
84  
85  	static ResolvableType resolveDeclaredEventType(Class<?> listenerType) {
86  		ResolvableType resolvableType = ResolvableType.forClass(listenerType).as(ApplicationListener.class);
87  		if (resolvableType == null || !resolvableType.hasGenerics()) {
88  			return null;
89  		}
90  		return resolvableType.getGeneric();
91  	}
92  
93  	private static ResolvableType resolveDeclaredEventType(ApplicationListener<ApplicationEvent> listener) {
94  		ResolvableType declaredEventType = resolveDeclaredEventType(listener.getClass());
95  		if (declaredEventType == null || declaredEventType.isAssignableFrom(
96  				ResolvableType.forClass(ApplicationEvent.class))) {
97  
98  			Class<?> targetClass = AopUtils.getTargetClass(listener);
99  			if (targetClass != listener.getClass()) {
100 				declaredEventType = resolveDeclaredEventType(targetClass);
101 			}
102 		}
103 		return declaredEventType;
104 	}
105 
106 }
