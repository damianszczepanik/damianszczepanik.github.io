17  package org.springframework.context.annotation;
18  
19  import java.lang.annotation.Annotation;
20  
21  import org.springframework.beans.factory.annotation.AnnotatedGenericBeanDefinition;
22  import org.springframework.beans.factory.config.BeanDefinitionHolder;
23  import org.springframework.beans.factory.support.AutowireCandidateQualifier;
24  import org.springframework.beans.factory.support.BeanDefinitionReaderUtils;
25  import org.springframework.beans.factory.support.BeanDefinitionRegistry;
26  import org.springframework.beans.factory.support.BeanNameGenerator;
27  import org.springframework.core.env.Environment;
28  import org.springframework.core.env.EnvironmentCapable;
29  import org.springframework.core.env.StandardEnvironment;
30  import org.springframework.util.Assert;
31  
32  /**
33   * Convenient adapter for programmatic registration of annotated bean classes.
34   * This is an alternative to {@link ClassPathBeanDefinitionScanner}, applying
35   * the same resolution of annotations but for explicitly registered classes only.
36   *
37   * @author Juergen Hoeller
38   * @author Chris Beams
39   * @author Sam Brannen
40   * @author Phillip Webb
41   * @since 3.0
42   * @see AnnotationConfigApplicationContext#register
43   */
44  public class AnnotatedBeanDefinitionReader {
45  
46  	private final BeanDefinitionRegistry registry;
47  
48  	private BeanNameGenerator beanNameGenerator = new AnnotationBeanNameGenerator();
49  
50  	private ScopeMetadataResolver scopeMetadataResolver = new AnnotationScopeMetadataResolver();
51  
52  	private ConditionEvaluator conditionEvaluator;
53  
54  
55  	/**
56  	 * Create a new {@code AnnotatedBeanDefinitionReader} for the given registry.
57  	 * If the registry is {@link EnvironmentCapable}, e.g. is an {@code ApplicationContext},
58  	 * the {@link Environment} will be inherited, otherwise a new
59  	 * {@link StandardEnvironment} will be created and used.
60  	 * @param registry the {@code BeanFactory} to load bean definitions into,
61  	 * in the form of a {@code BeanDefinitionRegistry}
62  	 * @see #AnnotatedBeanDefinitionReader(BeanDefinitionRegistry, Environment)
63  	 * @see #setEnvironment(Environment)
64  	 */
65  	public AnnotatedBeanDefinitionReader(BeanDefinitionRegistry registry) {
66  		this(registry, getOrCreateEnvironment(registry));
67  	}
68  
69  	/**
70  	 * Create a new {@code AnnotatedBeanDefinitionReader} for the given registry and using
71  	 * the given {@link Environment}.
72  	 * @param registry the {@code BeanFactory} to load bean definitions into,
73  	 * in the form of a {@code BeanDefinitionRegistry}
74  	 * @param environment the {@code Environment} to use when evaluating bean definition
75  	 * profiles.
76  	 * @since 3.1
77  	 */
78  	public AnnotatedBeanDefinitionReader(BeanDefinitionRegistry registry, Environment environment) {
79  		Assert.notNull(registry, "BeanDefinitionRegistry must not be null");
80  		Assert.notNull(environment, "Environment must not be null");
81  		this.registry = registry;
82  		this.conditionEvaluator = new ConditionEvaluator(registry, environment, null);
83  		AnnotationConfigUtils.registerAnnotationConfigProcessors(this.registry);
84  	}
85  
86  	/**
87  	 * Return the BeanDefinitionRegistry that this scanner operates on.
88  	 */
89  	public final BeanDefinitionRegistry getRegistry() {
90  		return this.registry;
91  	}
92  
93  	/**
94  	 * Set the Environment to use when evaluating whether
95  	 * {@link Conditional @Conditional}-annotated component classes should be registered.
96  	 * <p>The default is a {@link StandardEnvironment}.
97  	 * @see #registerBean(Class, String, Class...)
98  	 */
99  	public void setEnvironment(Environment environment) {
100 		this.conditionEvaluator = new ConditionEvaluator(this.registry, environment, null);
101 	}
102 
103 	/**
104 	 * Set the BeanNameGenerator to use for detected bean classes.
105 	 * <p>The default is a {@link AnnotationBeanNameGenerator}.
106 	 */
107 	public void setBeanNameGenerator(BeanNameGenerator beanNameGenerator) {
108 		this.beanNameGenerator = (beanNameGenerator != null ? beanNameGenerator : new AnnotationBeanNameGenerator());
109 	}
110 
111 	/**
112 	 * Set the ScopeMetadataResolver to use for detected bean classes.
113 	 * <p>The default is an {@link AnnotationScopeMetadataResolver}.
114 	 */
115 	public void setScopeMetadataResolver(ScopeMetadataResolver scopeMetadataResolver) {
116 		this.scopeMetadataResolver =
117 				(scopeMetadataResolver != null ? scopeMetadataResolver : new AnnotationScopeMetadataResolver());
118 	}
119 
120 	public void register(Class<?>... annotatedClasses) {
121 		for (Class<?> annotatedClass : annotatedClasses) {
122 			registerBean(annotatedClass);
123 		}
124 	}
125 
126 	public void registerBean(Class<?> annotatedClass) {
127 		registerBean(annotatedClass, null, (Class<? extends Annotation>[]) null);
128 	}
129 
130 	public void registerBean(Class<?> annotatedClass,
131 			@SuppressWarnings("unchecked") Class<? extends Annotation>... qualifiers) {
132 
133 		registerBean(annotatedClass, null, qualifiers);
134 	}
135 
136 	public void registerBean(Class<?> annotatedClass, String name,
137 			@SuppressWarnings("unchecked") Class<? extends Annotation>... qualifiers) {
138 
139 		AnnotatedGenericBeanDefinition abd = new AnnotatedGenericBeanDefinition(annotatedClass);
140 		if (this.conditionEvaluator.shouldSkip(abd.getMetadata())) {
141 			return;
142 		}
143 
144 		ScopeMetadata scopeMetadata = this.scopeMetadataResolver.resolveScopeMetadata(abd);
145 		abd.setScope(scopeMetadata.getScopeName());
146 		String beanName = (name != null ? name : this.beanNameGenerator.generateBeanName(abd, this.registry));
147 		AnnotationConfigUtils.processCommonDefinitionAnnotations(abd);
148 		if (qualifiers != null) {
149 			for (Class<? extends Annotation> qualifier : qualifiers) {
150 				if (Primary.class.equals(qualifier)) {
151 					abd.setPrimary(true);
152 				}
153 				else if (Lazy.class.equals(qualifier)) {
154 					abd.setLazyInit(true);
155 				}
156 				else {
157 					abd.addQualifier(new AutowireCandidateQualifier(qualifier));
158 				}
159 			}
160 		}
161 
162 		BeanDefinitionHolder definitionHolder = new BeanDefinitionHolder(abd, beanName);
163 		definitionHolder = AnnotationConfigUtils.applyScopedProxyMode(scopeMetadata, definitionHolder, this.registry);
164 		BeanDefinitionReaderUtils.registerBeanDefinition(definitionHolder, this.registry);
165 	}
166 
167 
168 	/**
169 	 * Get the Environment from the given registry if possible, otherwise return a new
170 	 * StandardEnvironment.
171 	 */
172 	private static Environment getOrCreateEnvironment(BeanDefinitionRegistry registry) {
173 		Assert.notNull(registry, "BeanDefinitionRegistry must not be null");
174 		if (registry instanceof EnvironmentCapable) {
175 			return ((EnvironmentCapable) registry).getEnvironment();
176 		}
177 		return new StandardEnvironment();
178 	}
