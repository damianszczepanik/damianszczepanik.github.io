17  package org.springframework.scheduling.annotation;
18  
19  import org.springframework.scheduling.config.ScheduledTaskRegistrar;
20  
21  /**
22   * Optional interface to be implemented by @{@link
23   * org.springframework.context.annotation.Configuration Configuration} classes annotated
24   * with @{@link EnableScheduling}. Typically used for setting a specific
25   * {@link org.springframework.scheduling.TaskScheduler TaskScheduler} bean to be used when
26   * executing scheduled tasks or for registering scheduled tasks in a <em>programmatic</em>
27   * fashion as opposed to the <em>declarative</em> approach of using the @{@link Scheduled}
28   * annotation. For example, this may be necessary when implementing {@link
29   * org.springframework.scheduling.Trigger Trigger}-based tasks, which are not supported by
30   * the {@code @Scheduled} annotation.
31   *
32   * <p>See @{@link EnableScheduling} for detailed usage examples.
33   *
34   * @author Chris Beams
35   * @since 3.1
36   * @see EnableScheduling
37   * @see ScheduledTaskRegistrar
38   */
39  public interface SchedulingConfigurer {
40  
41  	/**
42  	 * Callback allowing a {@link org.springframework.scheduling.TaskScheduler
43  	 * TaskScheduler} and specific {@link org.springframework.scheduling.config.Task Task}
44  	 * instances to be registered against the given the {@link ScheduledTaskRegistrar}
45  	 * @param taskRegistrar the registrar to be configured.
46  	 */
47  	void configureTasks(ScheduledTaskRegistrar taskRegistrar);
48  
49  }
