44  	@Test
45  	public void testDefaultLazyInit() {
46  		GenericApplicationContext context = new GenericApplicationContext();
47  		XmlBeanDefinitionReader reader = new XmlBeanDefinitionReader(context);
48  		reader.loadBeanDefinitions(LOCATION_PREFIX + "defaultWithNoOverridesTests.xml");
49  		assertFalse("lazy-init should be false", context.getBeanDefinition(TEST_BEAN_NAME).isLazyInit());
50  		assertEquals("initCount should be 0", 0, DefaultsTestBean.INIT_COUNT);
51  		context.refresh();
52  		assertEquals("bean should have been instantiated", 1, DefaultsTestBean.INIT_COUNT);
53  	}
