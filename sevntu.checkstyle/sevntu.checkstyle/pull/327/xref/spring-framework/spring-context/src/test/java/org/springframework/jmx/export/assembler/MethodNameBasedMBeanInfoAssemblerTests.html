17  package org.springframework.jmx.export.assembler;
18  
19  import javax.management.MBeanOperationInfo;
20  import javax.management.modelmbean.ModelMBeanAttributeInfo;
21  import javax.management.modelmbean.ModelMBeanInfo;
22  
23  import org.junit.Test;
24  
25  import static org.junit.Assert.*;
26  
27  /**
28   * @author Rob Harrop
29   * @author David Boden
30   * @author Chris Beams
31   */
32  public class MethodNameBasedMBeanInfoAssemblerTests extends AbstractJmxAssemblerTests {
33  
34  	protected static final String OBJECT_NAME = "bean:name=testBean5";
35  
36  	@Override
37  	protected String getObjectName() {
38  		return OBJECT_NAME;
39  	}
40  
41  	@Override
42  	protected int getExpectedOperationCount() {
43  		return 5;
44  	}
45  
46  	@Override
47  	protected int getExpectedAttributeCount() {
48  		return 2;
49  	}
50  
51  	@Override
52  	protected MBeanInfoAssembler getAssembler() {
53  		MethodNameBasedMBeanInfoAssembler assembler = new MethodNameBasedMBeanInfoAssembler();
54  		assembler.setManagedMethods(new String[] {"add", "myOperation", "getName", "setName", "getAge"});
55  		return assembler;
56  	}
57  
58  	@Test
59  	public void testGetAgeIsReadOnly() throws Exception {
60  		ModelMBeanInfo info = getMBeanInfoFromAssembler();
61  		ModelMBeanAttributeInfo attr = info.getAttribute(AGE_ATTRIBUTE);
62  
63  		assertTrue(attr.isReadable());
64  		assertFalse(attr.isWritable());
65  	}
66  
67  	@Test
68  	public void testSetNameParameterIsNamed() throws Exception {
69  		ModelMBeanInfo info = getMBeanInfoFromAssembler();
70  
71  		MBeanOperationInfo operationSetAge = info.getOperation("setName");
72  		assertEquals("name", operationSetAge.getSignature()[0].getName());
73  	}
74  
75  	@Override
76  	protected String getApplicationContextPath() {
77  		return "org/springframework/jmx/export/assembler/methodNameAssembler.xml";
78  	}
79  
80  }
