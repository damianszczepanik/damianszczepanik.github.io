38  /**
39   * Abstract implementation of the {@link ApplicationEventMulticaster} interface,
40   * providing the basic listener registration facility.
41   *
42   * <p>Doesn't permit multiple instances of the same listener by default,
43   * as it keeps listeners in a linked Set. The collection class used to hold
44   * ApplicationListener objects can be overridden through the "collectionClass"
45   * bean property.
46   *
47   * <p>Implementing ApplicationEventMulticaster's actual {@link #multicastEvent} method
48   * is left to subclasses. {@link SimpleApplicationEventMulticaster} simply multicasts
49   * all events to all registered listeners, invoking them in the calling thread.
50   * Alternative implementations could be more sophisticated in those respects.
51   *
52   * @author Juergen Hoeller
53   * @author Stephane Nicoll
54   * @since 1.2.3
55   * @see #getApplicationListeners(ApplicationEvent, ResolvableType)
56   * @see SimpleApplicationEventMulticaster
57   */
58  public abstract class AbstractApplicationEventMulticaster
59  		implements ApplicationEventMulticaster, BeanClassLoaderAware, BeanFactoryAware {
60  
61  	private final ListenerRetriever defaultRetriever = new ListenerRetriever(false);
62  
63  	private final Map<ListenerCacheKey, ListenerRetriever> retrieverCache =
64  			new ConcurrentHashMap<ListenerCacheKey, ListenerRetriever>(64);
65  
66  	private ClassLoader beanClassLoader;
67  
68  	private BeanFactory beanFactory;
69  
70  	private Object retrievalMutex = this.defaultRetriever;
71  
72  
73  	@Override
74  	public void setBeanClassLoader(ClassLoader classLoader) {
75  		this.beanClassLoader = classLoader;
76  	}
77  
78  	@Override
79  	public void setBeanFactory(BeanFactory beanFactory) {
80  		this.beanFactory = beanFactory;
81  		if (beanFactory instanceof ConfigurableBeanFactory) {
82  			ConfigurableBeanFactory cbf = (ConfigurableBeanFactory) beanFactory;
83  			if (this.beanClassLoader == null) {
84  				this.beanClassLoader = cbf.getBeanClassLoader();
85  			}
86  			this.retrievalMutex = cbf.getSingletonMutex();
87  		}
88  	}
89  
90  	private BeanFactory getBeanFactory() {
91  		if (this.beanFactory == null) {
92  			throw new IllegalStateException("ApplicationEventMulticaster cannot retrieve listener beans " +
93  					"because it is not associated with a BeanFactory");
94  		}
95  		return this.beanFactory;
96  	}

99  	@Override
100 	public void addApplicationListener(ApplicationListener<?> listener) {
101 		synchronized (this.retrievalMutex) {
102 			this.defaultRetriever.applicationListeners.add(listener);
103 			this.retrieverCache.clear();
104 		}
105 	}

107 	@Override
108 	public void addApplicationListenerBean(String listenerBeanName) {
109 		synchronized (this.retrievalMutex) {
110 			this.defaultRetriever.applicationListenerBeans.add(listenerBeanName);
111 			this.retrieverCache.clear();
112 		}
113 	}

115 	@Override
116 	public void removeApplicationListener(ApplicationListener<?> listener) {
117 		synchronized (this.retrievalMutex) {
118 			this.defaultRetriever.applicationListeners.remove(listener);
119 			this.retrieverCache.clear();
120 		}
121 	}

123 	@Override
124 	public void removeApplicationListenerBean(String listenerBeanName) {
125 		synchronized (this.retrievalMutex) {
126 			this.defaultRetriever.applicationListenerBeans.remove(listenerBeanName);
127 			this.retrieverCache.clear();
128 		}
129 	}

131 	@Override
132 	public void removeAllListeners() {
133 		synchronized (this.retrievalMutex) {
134 			this.defaultRetriever.applicationListeners.clear();
135 			this.defaultRetriever.applicationListenerBeans.clear();
136 			this.retrieverCache.clear();
137 		}
138 	}

141 	/**
142 	 * Return a Collection containing all ApplicationListeners.
143 	 * @return a Collection of ApplicationListeners
144 	 * @see org.springframework.context.ApplicationListener
145 	 */
146 	protected Collection<ApplicationListener<?>> getApplicationListeners() {
147 		synchronized (this.retrievalMutex) {
148 			return this.defaultRetriever.getApplicationListeners();
149 		}
150 	}
151 
152 	/**
153 	 * Return a Collection of ApplicationListeners matching the given
154 	 * event type. Non-matching listeners get excluded early.
155 	 * @param event the event to be propagated. Allows for excluding
156 	 * non-matching listeners early, based on cached matching information.
157 	 * @param eventType the event type
158 	 * @return a Collection of ApplicationListeners
159 	 * @see org.springframework.context.ApplicationListener
160 	 */
161 	protected Collection<ApplicationListener<?>> getApplicationListeners(
162 			ApplicationEvent event, ResolvableType eventType) {
163 
164 		Object source = event.getSource();
165 		Class<?> sourceType = (source != null ? source.getClass() : null);
166 		ListenerCacheKey cacheKey = new ListenerCacheKey(eventType, sourceType);
167 
168 		// Quick check for existing entry on ConcurrentHashMap...
169 		ListenerRetriever retriever = this.retrieverCache.get(cacheKey);
170 		if (retriever != null) {
171 			return retriever.getApplicationListeners();
172 		}
173 
174 		if (this.beanClassLoader == null ||
175 				(ClassUtils.isCacheSafe(event.getClass(), this.beanClassLoader) &&
176 						(sourceType == null || ClassUtils.isCacheSafe(sourceType, this.beanClassLoader)))) {
177 			// Fully synchronized building and caching of a ListenerRetriever
178 			synchronized (this.retrievalMutex) {
179 				retriever = this.retrieverCache.get(cacheKey);
180 				if (retriever != null) {
181 					return retriever.getApplicationListeners();
182 				}
183 				retriever = new ListenerRetriever(true);
184 				Collection<ApplicationListener<?>> listeners =
185 						retrieveApplicationListeners(eventType, sourceType, retriever);
186 				this.retrieverCache.put(cacheKey, retriever);
187 				return listeners;
188 			}
189 		}
190 		else {
191 			// No ListenerRetriever caching -> no synchronization necessary
192 			return retrieveApplicationListeners(eventType, sourceType, null);
193 		}
194 	}
195 
196 	/**
197 	 * Actually retrieve the application listeners for the given event and source type.
198 	 * @param eventType the event type
199 	 * @param sourceType the event source type
200 	 * @param retriever the ListenerRetriever, if supposed to populate one (for caching purposes)
201 	 * @return the pre-filtered list of application listeners for the given event and source type
202 	 */
203 	private Collection<ApplicationListener<?>> retrieveApplicationListeners(
204 			ResolvableType eventType, Class<?> sourceType, ListenerRetriever retriever) {
205 
206 		LinkedList<ApplicationListener<?>> allListeners = new LinkedList<ApplicationListener<?>>();
207 		Set<ApplicationListener<?>> listeners;
208 		Set<String> listenerBeans;
209 		synchronized (this.retrievalMutex) {
210 			listeners = new LinkedHashSet<ApplicationListener<?>>(this.defaultRetriever.applicationListeners);
211 			listenerBeans = new LinkedHashSet<String>(this.defaultRetriever.applicationListenerBeans);
212 		}
213 		for (ApplicationListener<?> listener : listeners) {
214 			if (supportsEvent(listener, eventType, sourceType)) {
215 				if (retriever != null) {
216 					retriever.applicationListeners.add(listener);
217 				}
218 				allListeners.add(listener);
219 			}
220 		}
221 		if (!listenerBeans.isEmpty()) {
222 			BeanFactory beanFactory = getBeanFactory();
223 			for (String listenerBeanName : listenerBeans) {
224 				try {
225 					Class<?> listenerType = beanFactory.getType(listenerBeanName);
226 					if (listenerType == null || supportsEvent(listenerType, eventType)) {
227 						ApplicationListener<?> listener =
228 								beanFactory.getBean(listenerBeanName, ApplicationListener.class);
229 						if (!allListeners.contains(listener) && supportsEvent(listener, eventType, sourceType)) {
230 							if (retriever != null) {
231 								retriever.applicationListenerBeans.add(listenerBeanName);
232 							}
233 							allListeners.add(listener);
234 						}
235 					}
236 				}
237 				catch (NoSuchBeanDefinitionException ex) {
238 					// Singleton listener instance (without backing bean definition) disappeared -
239 					// probably in the middle of the destruction phase
240 				}
241 			}
242 		}
243 		AnnotationAwareOrderComparator.sort(allListeners);
244 		return allListeners;
245 	}

247 	/**
248 	 * Filter a listener early through checking its generically declared event
249 	 * type before trying to instantiate it.
250 	 * <p>If this method returns {@code true} for a given listener as a first pass,
251 	 * the listener instance will get retrieved and fully evaluated through a
252 	 * {@link #supportsEvent(ApplicationListener,ResolvableType, Class)}  call afterwards.
253 	 * @param listenerType the listener's type as determined by the BeanFactory
254 	 * @param eventType the event type to check
255 	 * @return whether the given listener should be included in the candidates
256 	 * for the given event type
257 	 */
258 	protected boolean supportsEvent(Class<?> listenerType, ResolvableType eventType) {
259 		if (GenericApplicationListener.class.isAssignableFrom(listenerType)
260 				|| SmartApplicationListener.class.isAssignableFrom(listenerType)) {
261 			return true;
262 		}
263 		ResolvableType declaredEventType = GenericApplicationListenerAdapter.resolveDeclaredEventType(listenerType);
264 		return (declaredEventType == null || declaredEventType.isAssignableFrom(eventType));
265 	}
266 
267 	/**
268 	 * Determine whether the given listener supports the given event.
269 	 * <p>The default implementation detects the {@link SmartApplicationListener}
270 	 * and {@link GenericApplicationListener} interfaces. In case of a standard
271 	 * {@link ApplicationListener}, a {@link GenericApplicationListenerAdapter}
272 	 * will be used to introspect the generically declared type of the target listener.
273 	 * @param listener the target listener to check
274 	 * @param eventType the event type to check against
275 	 * @param sourceType the source type to check against
276 	 * @return whether the given listener should be included in the candidates
277 	 * for the given event type
278 	 */
279 	protected boolean supportsEvent(ApplicationListener<?> listener, ResolvableType eventType, Class<?> sourceType) {
280 		GenericApplicationListener smartListener = (listener instanceof GenericApplicationListener ?
281 				(GenericApplicationListener) listener : new GenericApplicationListenerAdapter(listener));
282 		return (smartListener.supportsEventType(eventType) && smartListener.supportsSourceType(sourceType));
283 	}

286 	/**
287 	 * Cache key for ListenerRetrievers, based on event type and source type.
288 	 */
289 	private static class ListenerCacheKey {
290 
291 		private final ResolvableType eventType;
292 
293 		private final Class<?> sourceType;
294 
295 		public ListenerCacheKey(ResolvableType eventType, Class<?> sourceType) {
296 			this.eventType = eventType;
297 			this.sourceType = sourceType;
298 		}
299 
300 		@Override
301 		public boolean equals(Object other) {
302 			if (this == other) {
303 				return true;
304 			}
305 			ListenerCacheKey otherKey = (ListenerCacheKey) other;
306 			return ObjectUtils.nullSafeEquals(this.eventType, otherKey.eventType) &&
307 					ObjectUtils.nullSafeEquals(this.sourceType, otherKey.sourceType);
308 		}
309 
310 		@Override
311 		public int hashCode() {
312 			return ObjectUtils.nullSafeHashCode(this.eventType) * 29 + ObjectUtils.nullSafeHashCode(this.sourceType);
313 		}
314 	}

317 	/**
318 	 * Helper class that encapsulates a specific set of target listeners,
319 	 * allowing for efficient retrieval of pre-filtered listeners.
320 	 * <p>An instance of this helper gets cached per event type and source type.
321 	 */
322 	private class ListenerRetriever {
323 
324 		public final Set<ApplicationListener<?>> applicationListeners;
325 
326 		public final Set<String> applicationListenerBeans;
327 
328 		private final boolean preFiltered;
329 
330 		public ListenerRetriever(boolean preFiltered) {
331 			this.applicationListeners = new LinkedHashSet<ApplicationListener<?>>();
332 			this.applicationListenerBeans = new LinkedHashSet<String>();
333 			this.preFiltered = preFiltered;
334 		}
335 
336 		public Collection<ApplicationListener<?>> getApplicationListeners() {
337 			LinkedList<ApplicationListener<?>> allListeners = new LinkedList<ApplicationListener<?>>();
338 			for (ApplicationListener<?> listener : this.applicationListeners) {
339 				allListeners.add(listener);
340 			}
341 			if (!this.applicationListenerBeans.isEmpty()) {
342 				BeanFactory beanFactory = getBeanFactory();
343 				for (String listenerBeanName : this.applicationListenerBeans) {
344 					try {
345 						ApplicationListener<?> listener = beanFactory.getBean(listenerBeanName, ApplicationListener.class);
346 						if (this.preFiltered || !allListeners.contains(listener)) {
347 							allListeners.add(listener);
348 						}
349 					}
350 					catch (NoSuchBeanDefinitionException ex) {
351 						// Singleton listener instance (without backing bean definition) disappeared -
352 						// probably in the middle of the destruction phase
353 					}
354 				}
355 			}
356 			AnnotationAwareOrderComparator.sort(allListeners);
357 			return allListeners;
358 		}
359 	}
