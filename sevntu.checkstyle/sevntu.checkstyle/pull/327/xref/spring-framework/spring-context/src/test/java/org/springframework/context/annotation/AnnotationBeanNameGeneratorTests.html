47  public class AnnotationBeanNameGeneratorTests {

49  	private AnnotationBeanNameGenerator beanNameGenerator = new AnnotationBeanNameGenerator();
50  
51  
52  	@Test
53  	public void generateBeanNameWithNamedComponent() {
54  		BeanDefinitionRegistry registry = new SimpleBeanDefinitionRegistry();
55  		AnnotatedBeanDefinition bd = new AnnotatedGenericBeanDefinition(ComponentWithName.class);
56  		String beanName = this.beanNameGenerator.generateBeanName(bd, registry);
57  		assertNotNull("The generated beanName must *never* be null.", beanName);
58  		assertTrue("The generated beanName must *never* be blank.", StringUtils.hasText(beanName));
59  		assertEquals("walden", beanName);
60  	}
61  
62  	@Test
63  	public void generateBeanNameWithDefaultNamedComponent() {
64  		BeanDefinitionRegistry registry = new SimpleBeanDefinitionRegistry();
65  		AnnotatedBeanDefinition bd = new AnnotatedGenericBeanDefinition(DefaultNamedComponent.class);
66  		String beanName = this.beanNameGenerator.generateBeanName(bd, registry);
67  		assertNotNull("The generated beanName must *never* be null.", beanName);
68  		assertTrue("The generated beanName must *never* be blank.", StringUtils.hasText(beanName));
69  		assertEquals("thoreau", beanName);
70  	}
71  
72  	@Test
73  	public void generateBeanNameWithNamedComponentWhereTheNameIsBlank() {
74  		BeanDefinitionRegistry registry = new SimpleBeanDefinitionRegistry();
75  		AnnotatedBeanDefinition bd = new AnnotatedGenericBeanDefinition(ComponentWithBlankName.class);
76  		String beanName = this.beanNameGenerator.generateBeanName(bd, registry);
77  		assertNotNull("The generated beanName must *never* be null.", beanName);
78  		assertTrue("The generated beanName must *never* be blank.", StringUtils.hasText(beanName));
79  		String expectedGeneratedBeanName = this.beanNameGenerator.buildDefaultBeanName(bd);
80  		assertEquals(expectedGeneratedBeanName, beanName);
81  	}
