43  public abstract class AbstractJmxAssemblerTests extends AbstractJmxTests {
44  
45  	protected static final String AGE_ATTRIBUTE = "Age";

47  	protected static final String NAME_ATTRIBUTE = "Name";

49  	protected abstract String getObjectName();

51  	@Test
52  	public void testMBeanRegistration() throws Exception {
53  		// beans are registered at this point - just grab them from the server
54  		ObjectInstance instance = getObjectInstance();
55  		assertNotNull("Bean should not be null", instance);
56  	}

58  	@Test
59  	public void testRegisterOperations() throws Exception {
60  		IJmxTestBean bean = getBean();
61  		assertNotNull(bean);
62  		MBeanInfo inf = getMBeanInfo();
63  		assertEquals("Incorrect number of operations registered",
64  				getExpectedOperationCount(), inf.getOperations().length);
65  	}

67  	@Test
68  	public void testRegisterAttributes() throws Exception {
69  		IJmxTestBean bean = getBean();
70  		assertNotNull(bean);
71  		MBeanInfo inf = getMBeanInfo();
72  		assertEquals("Incorrect number of attributes registered",
73  				getExpectedAttributeCount(), inf.getAttributes().length);
74  	}
75  
76  	@Test
77  	public void testGetMBeanInfo() throws Exception {
78  		ModelMBeanInfo info = getMBeanInfoFromAssembler();
79  		assertNotNull("MBeanInfo should not be null", info);
80  	}
81  
82  	@Test
83  	public void testGetMBeanAttributeInfo() throws Exception {
84  		ModelMBeanInfo info = getMBeanInfoFromAssembler();
85  		MBeanAttributeInfo[] inf = info.getAttributes();
86  		assertEquals("Invalid number of Attributes returned",
87  				getExpectedAttributeCount(), inf.length);
88  
89  		for (int x = 0; x < inf.length; x++) {
90  			assertNotNull("MBeanAttributeInfo should not be null", inf[x]);
91  			assertNotNull(
92  					"Description for MBeanAttributeInfo should not be null",
93  					inf[x].getDescription());
94  		}
95  	}
96  
97  	@Test
98  	public void testGetMBeanOperationInfo() throws Exception {
99  		ModelMBeanInfo info = getMBeanInfoFromAssembler();
100 		MBeanOperationInfo[] inf = info.getOperations();
101 		assertEquals("Invalid number of Operations returned",
102 				getExpectedOperationCount(), inf.length);
103 
104 		for (int x = 0; x < inf.length; x++) {
105 			assertNotNull("MBeanOperationInfo should not be null", inf[x]);
106 			assertNotNull(
107 					"Description for MBeanOperationInfo should not be null",
108 					inf[x].getDescription());
109 		}
110 	}

112 	@Test
113 	public void testDescriptionNotNull() throws Exception {
114 		ModelMBeanInfo info = getMBeanInfoFromAssembler();
115 
116 		assertNotNull("The MBean description should not be null",
117 				info.getDescription());

120 	@Test
121 	public void testSetAttribute() throws Exception {
122 		ObjectName objectName = ObjectNameManager.getInstance(getObjectName());
123 		getServer().setAttribute(objectName, new Attribute(NAME_ATTRIBUTE, "Rob Harrop"));
124 		IJmxTestBean bean = (IJmxTestBean) getContext().getBean("testBean");
125 		assertEquals("Rob Harrop", bean.getName());
126 	}
127 
128 	@Test
129 	public void testGetAttribute() throws Exception {
130 		ObjectName objectName = ObjectNameManager.getInstance(getObjectName());
131 		getBean().setName("John Smith");
132 		Object val = getServer().getAttribute(objectName, NAME_ATTRIBUTE);
133 		assertEquals("Incorrect result", "John Smith", val);
134 	}
135 
136 	@Test
137 	public void testOperationInvocation() throws Exception{
138 		ObjectName objectName = ObjectNameManager.getInstance(getObjectName());
139 		Object result = getServer().invoke(objectName, "add",
140 				new Object[] {new Integer(20), new Integer(30)}, new String[] {"int", "int"});
141 	assertEquals("Incorrect result", new Integer(50), result);
142 	}
143 
144 	@Test
145 	public void testAttributeInfoHasDescriptors() throws Exception {
146 		ModelMBeanInfo info = getMBeanInfoFromAssembler();
147 
148 		ModelMBeanAttributeInfo attr = info.getAttribute(NAME_ATTRIBUTE);
149 		Descriptor desc = attr.getDescriptor();
150 		assertNotNull("getMethod field should not be null",
151 				desc.getFieldValue("getMethod"));
152 		assertNotNull("setMethod field should not be null",
153 				desc.getFieldValue("setMethod"));
154 		assertEquals("getMethod field has incorrect value", "getName",
155 				desc.getFieldValue("getMethod"));
156 		assertEquals("setMethod field has incorrect value", "setName",
157 				desc.getFieldValue("setMethod"));
158 	}
159 
160 	@Test
161 	public void testAttributeHasCorrespondingOperations() throws Exception {
162 		ModelMBeanInfo info = getMBeanInfoFromAssembler();
163 
164 		ModelMBeanOperationInfo get = info.getOperation("getName");
165 		assertNotNull("get operation should not be null", get);
166 		assertEquals("get operation should have visibility of four",
167 				get.getDescriptor().getFieldValue("visibility"),
168 				new Integer(4));
169 		assertEquals("get operation should have role \"getter\"", "getter", get.getDescriptor().getFieldValue("role"));
170 
171 		ModelMBeanOperationInfo set = info.getOperation("setName");
172 		assertNotNull("set operation should not be null", set);
173 		assertEquals("set operation should have visibility of four",
174 				set.getDescriptor().getFieldValue("visibility"),
175 				new Integer(4));
176 		assertEquals("set operation should have role \"setter\"", "setter", set.getDescriptor().getFieldValue("role"));
177 	}
178 
179 	@Test
180 	public void testNotificationMetadata() throws Exception {
181 		ModelMBeanInfo info = (ModelMBeanInfo) getMBeanInfo();
182 		MBeanNotificationInfo[] notifications = info.getNotifications();
183 		assertEquals("Incorrect number of notifications", 1, notifications.length);
184 		assertEquals("Incorrect notification name", "My Notification", notifications[0].getName());
185 
186 		String[] notifTypes = notifications[0].getNotifTypes();
187 
188 		assertEquals("Incorrect number of notification types", 2, notifTypes.length);
189 		assertEquals("Notification type.foo not found", "type.foo", notifTypes[0]);
190 		assertEquals("Notification type.bar not found", "type.bar", notifTypes[1]);
191 	}
192 
193 	protected ModelMBeanInfo getMBeanInfoFromAssembler() throws Exception {
194 		IJmxTestBean bean = getBean();
195 		ModelMBeanInfo info = getAssembler().getMBeanInfo(bean, getObjectName());
196 		return info;
197 	}
198 
199 	protected IJmxTestBean getBean() {
200 		Object bean = getContext().getBean("testBean");
201 		return (IJmxTestBean) bean;
202 	}
203 
204 	protected MBeanInfo getMBeanInfo() throws Exception {
205 		return getServer().getMBeanInfo(ObjectNameManager.getInstance(getObjectName()));
206 	}
207 
208 	protected ObjectInstance getObjectInstance() throws Exception {
209 		return getServer().getObjectInstance(ObjectNameManager.getInstance(getObjectName()));
210 	}

212 	protected abstract int getExpectedOperationCount();
213 
214 	protected abstract int getExpectedAttributeCount();
215 
216 	protected abstract MBeanInfoAssembler getAssembler() throws Exception;
