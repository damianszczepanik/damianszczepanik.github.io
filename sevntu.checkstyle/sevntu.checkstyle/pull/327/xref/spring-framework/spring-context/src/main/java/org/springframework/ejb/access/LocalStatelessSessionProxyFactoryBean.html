17  package org.springframework.ejb.access;
18  
19  import javax.naming.NamingException;
20  
21  import org.springframework.aop.framework.ProxyFactory;
22  import org.springframework.beans.factory.BeanClassLoaderAware;
23  import org.springframework.beans.factory.FactoryBean;
24  import org.springframework.util.ClassUtils;
25  
26  /**
27   * Convenient {@link FactoryBean} for local Stateless Session Bean (SLSB) proxies.
28   * Designed for EJB 2.x, but works for EJB 3 Session Beans as well.
29   *
30   * <p>See {@link org.springframework.jndi.JndiObjectLocator} for info on
31   * how to specify the JNDI location of the target EJB.
32   *
33   * <p>If you want control over interceptor chaining, use an AOP ProxyFactoryBean
34   * with LocalSlsbInvokerInterceptor rather than rely on this class.
35   *
36   * <p>In a bean container, this class is normally best used as a singleton. However,
37   * if that bean container pre-instantiates singletons (as do the XML ApplicationContext
38   * variants) you may have a problem if the bean container is loaded before the EJB
39   * container loads the target EJB. That is because by default the JNDI lookup will be
40   * performed in the init method of this class and cached, but the EJB will not have been
41   * bound at the target location yet. The best solution is to set the "lookupHomeOnStartup"
42   * property to "false", in which case the home will be fetched on first access to the EJB.
43   * (This flag is only true by default for backwards compatibility reasons).
44   *
45   * @author Rod Johnson
46   * @author Colin Sampaleanu
47   * @since 09.05.2003
48   * @see AbstractSlsbInvokerInterceptor#setLookupHomeOnStartup
49   * @see AbstractSlsbInvokerInterceptor#setCacheHome
50   */
51  public class LocalStatelessSessionProxyFactoryBean extends LocalSlsbInvokerInterceptor
52  		implements FactoryBean<Object>, BeanClassLoaderAware {
53  
54  	/** The business interface of the EJB we're proxying */
55  	private Class<?> businessInterface;
56  
57  	private ClassLoader beanClassLoader = ClassUtils.getDefaultClassLoader();
58  
59  	/** EJBLocalObject */
60  	private Object proxy;
61  
62  
63  	/**
64  	 * Set the business interface of the EJB we're proxying.
65  	 * This will normally be a super-interface of the EJB local component interface.
66  	 * Using a business methods interface is a best practice when implementing EJBs.
67  	 * @param businessInterface set the business interface of the EJB
68  	 */
69  	public void setBusinessInterface(Class<?> businessInterface) {
70  		this.businessInterface = businessInterface;
71  	}
72  
73  	/**
74  	 * Return the business interface of the EJB we're proxying.
75  	 */
76  	public Class<?> getBusinessInterface() {
77  		return this.businessInterface;
78  	}
79  
80  	@Override
81  	public void setBeanClassLoader(ClassLoader classLoader) {
82  		this.beanClassLoader = classLoader;
83  	}
84  
85  	@Override
86  	public void afterPropertiesSet() throws NamingException {
87  		super.afterPropertiesSet();
88  		if (this.businessInterface == null) {
89  			throw new IllegalArgumentException("businessInterface is required");
90  		}
91  		this.proxy = new ProxyFactory(this.businessInterface, this).getProxy(this.beanClassLoader);
92  	}
93  
94  
95  	@Override
96  	public Object getObject() {
97  		return this.proxy;
98  	}
99  
100 	@Override
101 	public Class<?> getObjectType() {
102 		return this.businessInterface;
103 	}
104 
105 	@Override
106 	public boolean isSingleton() {
107 		return true;
108 	}
109 
110 }
