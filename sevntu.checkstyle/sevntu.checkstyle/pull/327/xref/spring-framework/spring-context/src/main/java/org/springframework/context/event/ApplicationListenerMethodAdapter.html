44  /**
45   * {@link GenericApplicationListener} adapter that delegates the processing of
46   * an event to an {@link EventListener} annotated method.
47   *
48   * <p>Delegates to {@link #processEvent(ApplicationEvent)} to give a chance to
49   * sub-classes to deviate from the default. Unwrap the content of a
50   * {@link PayloadApplicationEvent} if necessary to allow method declaration
51   * to define any arbitrary event type. If a condition is defined, it is
52   * evaluated prior to invoking the underlying method.
53   *
54   * @author Stephane Nicoll
55   * @since 4.2
56   */
57  public class ApplicationListenerMethodAdapter implements GenericApplicationListener {
58  
59  	protected final Log logger = LogFactory.getLog(getClass());
60  
61  	private final String beanName;
62  
63  	private final Method method;
64  
65  	private final Class<?> targetClass;
66  
67  	private final Method bridgedMethod;
68  
69  	private final ResolvableType declaredEventType;
70  
71  	private final AnnotatedElementKey methodKey;
72  
73  	private ApplicationContext applicationContext;
74  
75  	private EventExpressionEvaluator evaluator;
76  
77  	private String condition;
78  
79  	public ApplicationListenerMethodAdapter(String beanName, Class<?> targetClass, Method method) {
80  		this.beanName = beanName;
81  		this.method = method;
82  		this.targetClass = targetClass;
83  		this.bridgedMethod = BridgeMethodResolver.findBridgedMethod(method);
84  		this.declaredEventType = resolveDeclaredEventType();
85  		this.methodKey = new AnnotatedElementKey(this.method, this.targetClass);
86  	}
87  
88  	/**
89  	 * Initialize this instance.
90  	 */
91  	void init(ApplicationContext applicationContext, EventExpressionEvaluator evaluator) {
92  		this.applicationContext = applicationContext;
93  		this.evaluator = evaluator;
94  	}
95  
96  	@Override
97  	public void onApplicationEvent(ApplicationEvent event) {
98  		processEvent(event);
99  	}
100 
101 	/**
102 	 * Process the specified {@link ApplicationEvent}, checking if the condition
103 	 * match and handling non-null result, if any.
104 	 */
105 	public void processEvent(ApplicationEvent event) {
106 		Object[] args = resolveArguments(event);
107 		if (shouldHandle(event, args)) {
108 			Object result = doInvoke(args);
109 			if (result != null) {
110 				handleResult(result);
111 			}
112 			else {
113 				logger.trace("No result object given - no result to handle");
114 			}
115 		}
116 	}
117 
118 	/**
119 	 * Resolve the method arguments to use for the specified {@link ApplicationEvent}.
120 	 * <p>These arguments will be used to invoke the method handled by this instance. Can
121 	 * return {@code null} to indicate that no suitable arguments could be resolved and
122 	 * therefore the method should not be invoked at all for the specified event.
123 	 */
124 	protected Object[] resolveArguments(ApplicationEvent event) {
125 		if (!ApplicationEvent.class.isAssignableFrom(this.declaredEventType.getRawClass())
126 				&& event instanceof PayloadApplicationEvent) {
127 			Object payload = ((PayloadApplicationEvent) event).getPayload();
128 			if (this.declaredEventType.isAssignableFrom(ResolvableType.forClass(payload.getClass()))) {
129 				return new Object[] {payload};
130 			}
131 		}
132 		else {
133 			return new Object[] {event};
134 		}
135 		return null;
136 	}
137 
138 	protected void handleResult(Object result) {
139 		Assert.notNull(this.applicationContext, "ApplicationContext must no be null.");
140 		if (result.getClass().isArray()) {
141 			Object[] events = ObjectUtils.toObjectArray(result);
142 			for (Object event : events) {
143 				publishEvent(event);
144 			}
145 		}
146 		else if (result instanceof Collection<?>) {
147 			Collection<?> events = (Collection<?>) result;
148 			for (Object event : events) {
149 				publishEvent(event);
150 			}
151 		}
152 		else {
153 			publishEvent(result);
154 		}
155 	}
156 
157 	private void publishEvent(Object event) {
158 		if (event != null) {
159 			this.applicationContext.publishEvent(event);
160 		}
161 	}
162 
163 
164 	private boolean shouldHandle(ApplicationEvent event, Object[] args) {
165 		if (args == null) {
166 			return false;
167 		}
168 		String condition = getCondition();
169 		if (StringUtils.hasText(condition)) {
170 			Assert.notNull(this.evaluator, "Evaluator must no be null.");
171 			EvaluationContext evaluationContext = this.evaluator.createEvaluationContext(event,
172 					this.targetClass, this.method, args);
173 			return this.evaluator.condition(condition, this.methodKey, evaluationContext);
174 		}
175 		return true;
176 	}
177 
178 	@Override
179 	public boolean supportsEventType(ResolvableType eventType) {
180 		if (this.declaredEventType.isAssignableFrom(eventType)) {
181 			return true;
182 		}
183 		else if (PayloadApplicationEvent.class.isAssignableFrom(eventType.getRawClass())) {
184 			ResolvableType payloadType = eventType.as(PayloadApplicationEvent.class).getGeneric();
185 			return eventType.hasUnresolvableGenerics() || this.declaredEventType.isAssignableFrom(payloadType);
186 		}
187 		return false;
188 	}
189 
190 	@Override
191 	public boolean supportsSourceType(Class<?> sourceType) {
192 		return true;
193 	}
194 
195 	@Override
196 	public int getOrder() {
197 		Order order = getMethodAnnotation(Order.class);
198 		return (order != null ? order.value() : 0);
199 	}
200 
201 	protected <A extends Annotation> A getMethodAnnotation(Class<A> annotationType) {
202 		return AnnotationUtils.findAnnotation(this.method, annotationType);
203 	}
204 
205 	/**
206 	 * Invoke the event listener method with the given argument values.
207 	 */
208 	protected Object doInvoke(Object... args) {
209 		Object bean = getTargetBean();
210 		ReflectionUtils.makeAccessible(this.bridgedMethod);
211 		try {
212 			return this.bridgedMethod.invoke(bean, args);
213 		}
214 		catch (IllegalArgumentException ex) {
215 			assertTargetBean(this.bridgedMethod, bean, args);
216 			throw new IllegalStateException(getInvocationErrorMessage(bean, ex.getMessage(), args), ex);
217 		}
218 		catch (IllegalAccessException ex) {
219 			throw new IllegalStateException(getInvocationErrorMessage(bean, ex.getMessage(), args), ex);
220 		}
221 		catch (InvocationTargetException ex) {
222 			// Throw underlying exception
223 			Throwable targetException = ex.getTargetException();
224 			if (targetException instanceof RuntimeException) {
225 				throw (RuntimeException) targetException;
226 			}
227 			else {
228 				String msg = getInvocationErrorMessage(bean, "Failed to invoke event listener method", args);
229 				throw new UndeclaredThrowableException(targetException, msg);
230 			}
231 		}
232 	}
233 
234 	/**
235 	 * Return the target bean instance to use.
236 	 */
237 	protected Object getTargetBean() {
238 		Assert.notNull(this.applicationContext, "ApplicationContext must no be null.");
239 		return this.applicationContext.getBean(this.beanName);
240 	}
241 
242 	/**
243 	 * Return the condition to use. Matches the {@code condition} attribute of the
244 	 * {@link EventListener} annotation or any matching attribute on a meta-annotation.
245 	 */
246 	protected String getCondition() {
247 		if (this.condition == null) {
248 			AnnotationAttributes annotationAttributes = AnnotatedElementUtils
249 					.getAnnotationAttributes(this.method, EventListener.class.getName());
250 			if (annotationAttributes != null) {
251 				String value = annotationAttributes.getString("condition");
252 				this.condition = (value != null ? value : "");
253 			}
254 			else { // TODO annotationAttributes null with proxy
255 				EventListener eventListener = getMethodAnnotation(EventListener.class);
256 				this.condition = (eventListener != null ? eventListener.condition() : null);
257 			}
258 		}
259 		return this.condition;
260 	}
261 
262 	/**
263 	 * Add additional details such as the bean type and method signature to
264 	 * the given error message.
265 	 * @param message error message to append the HandlerMethod details to
266 	 */
267 	protected String getDetailedErrorMessage(Object bean, String message) {
268 		StringBuilder sb = new StringBuilder(message).append("\n");
269 		sb.append("HandlerMethod details: \n");
270 		sb.append("Bean [").append(bean.getClass().getName()).append("]\n");
271 		sb.append("Method [").append(this.bridgedMethod.toGenericString()).append("]\n");
272 		return sb.toString();
273 	}
274 
275 	/**
276 	 * Assert that the target bean class is an instance of the class where the given
277 	 * method is declared. In some cases the actual bean instance at event-
278 	 * processing time may be a JDK dynamic proxy (lazy initialization, prototype
279 	 * beans, and others). Event listener beans that require proxying should prefer
280 	 * class-based proxy mechanisms.
281 	 */
282 	private void assertTargetBean(Method method, Object targetBean, Object[] args) {
283 		Class<?> methodDeclaringClass = method.getDeclaringClass();
284 		Class<?> targetBeanClass = targetBean.getClass();
285 		if (!methodDeclaringClass.isAssignableFrom(targetBeanClass)) {
286 			String msg = "The event listener method class '" + methodDeclaringClass.getName() +
287 					"' is not an instance of the actual bean instance '" +
288 					targetBeanClass.getName() + "'. If the bean requires proxying " +
289 					"(e.g. due to @Transactional), please use class-based proxying.";
290 			throw new IllegalStateException(getInvocationErrorMessage(targetBean, msg, args));
291 		}
292 	}
293 
294 	private String getInvocationErrorMessage(Object bean, String message, Object[] resolvedArgs) {
295 		StringBuilder sb = new StringBuilder(getDetailedErrorMessage(bean, message));
296 		sb.append("Resolved arguments: \n");
297 		for (int i = 0; i < resolvedArgs.length; i++) {
298 			sb.append("[").append(i).append("] ");
299 			if (resolvedArgs[i] == null) {
300 				sb.append("[null] \n");
301 			}
302 			else {
303 				sb.append("[type=").append(resolvedArgs[i].getClass().getName()).append("] ");
304 				sb.append("[value=").append(resolvedArgs[i]).append("]\n");
305 			}
306 		}
307 		return sb.toString();
308 	}

311 	private ResolvableType resolveDeclaredEventType() {
312 		int count = this.method.getParameterTypes().length;
313 		if (count != 1) {
314 			throw new IllegalStateException("Only one parameter is allowed " +
315 					"for event listener method: " + method);
316 		}
317 		return ResolvableType.forMethodParameter(this.method, 0);
318 	}

320 	@Override
321 	public String toString() {
322 		return this.method.toGenericString();
323 	}
