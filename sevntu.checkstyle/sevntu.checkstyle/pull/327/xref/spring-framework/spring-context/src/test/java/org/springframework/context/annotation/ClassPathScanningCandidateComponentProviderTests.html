99  	public void testWithComponentAnnotationOnly() {
100 		ClassPathScanningCandidateComponentProvider provider = new ClassPathScanningCandidateComponentProvider(false);
101 		provider.addIncludeFilter(new AnnotationTypeFilter(Component.class));
102 		provider.addExcludeFilter(new AnnotationTypeFilter(Repository.class));
103 		provider.addExcludeFilter(new AnnotationTypeFilter(Service.class));
104 		provider.addExcludeFilter(new AnnotationTypeFilter(Controller.class));
105 		Set<BeanDefinition> candidates = provider.findCandidateComponents(TEST_BASE_PACKAGE);
106 		assertEquals(2, candidates.size());
107 		assertTrue(containsBeanClass(candidates, NamedComponent.class));
108 		assertTrue(containsBeanClass(candidates, ServiceInvocationCounter.class));
109 		assertFalse(containsBeanClass(candidates, FooServiceImpl.class));
110 		assertFalse(containsBeanClass(candidates, StubFooDao.class));

307 	private boolean containsBeanClass(Set<BeanDefinition> candidates, Class<?> beanClass) {
308 		for (BeanDefinition candidate : candidates) {
309 			ScannedGenericBeanDefinition definition = (ScannedGenericBeanDefinition) candidate;
310 			if (beanClass.getName().equals(definition.getBeanClassName())) {
311 				return true;
312 			}
313 		}
314 		return false;
315 	}
