70  	public void testContextAwareSingletonWasCalledBack() throws Exception {
71  		ACATester aca = (ACATester) applicationContext.getBean("aca");
72  		assertTrue("has had context set", aca.getApplicationContext() == applicationContext);
73  		Object aca2 = applicationContext.getBean("aca");
74  		assertTrue("Same instance", aca == aca2);
75  		assertTrue("Says is singleton", applicationContext.isSingleton("aca"));
76  	}

78  	public void testContextAwarePrototypeWasCalledBack() throws Exception {
79  		ACATester aca = (ACATester) applicationContext.getBean("aca-prototype");
80  		assertTrue("has had context set", aca.getApplicationContext() == applicationContext);
81  		Object aca2 = applicationContext.getBean("aca-prototype");
82  		assertTrue("NOT Same instance", aca != aca2);
83  		assertTrue("Says is prototype", !applicationContext.isSingleton("aca-prototype"));
84  	}

109 	public void testCloseTriggersDestroy() {
110 		LifecycleBean lb = (LifecycleBean) applicationContext.getBean("lifecycle");
111 		assertTrue("Not destroyed", !lb.isDestroyed());
112 		applicationContext.close();
113 		if (applicationContext.getParent() != null) {
114 			((ConfigurableApplicationContext) applicationContext.getParent()).close();
115 		}
116 		assertTrue("Destroyed", lb.isDestroyed());
117 		applicationContext.close();
118 		if (applicationContext.getParent() != null) {
119 			((ConfigurableApplicationContext) applicationContext.getParent()).close();
120 		}
121 		assertTrue("Destroyed", lb.isDestroyed());
122 	}

141 	@Test
142 	public void testEventsWithNoSource() throws Exception {
143 		// See SPR-10945 Serialized events result in a null source
144 		MyEvent event = new MyEvent(this);
145 		ByteArrayOutputStream bos = new ByteArrayOutputStream();
146 		ObjectOutputStream oos = new ObjectOutputStream(bos);
147 		oos.writeObject(event);
148 		oos.close();
149 		event = (MyEvent) new ObjectInputStream(new ByteArrayInputStream(
150 				bos.toByteArray())).readObject();
151 		doTestEvents(this.listener, this.parentListener, event);
152 	}
153 
154 	protected void doTestEvents(TestListener listener, TestListener parentListener,
155 			MyEvent event) {
156 		listener.zeroCounter();
157 		parentListener.zeroCounter();
158 		assertTrue("0 events before publication", listener.getEventCount() == 0);
159 		assertTrue("0 parent events before publication", parentListener.getEventCount() == 0);
160 		this.applicationContext.publishEvent(event);
161 		assertTrue("1 events after publication, not " + listener.getEventCount(), listener.getEventCount() == 1);
162 		assertTrue("1 parent events after publication", parentListener.getEventCount() == 1);
163 	}
