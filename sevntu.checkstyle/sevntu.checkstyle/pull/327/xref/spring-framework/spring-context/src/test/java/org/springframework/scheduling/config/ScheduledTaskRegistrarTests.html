27  /**

33  public class ScheduledTaskRegistrarTests {
34  
35  	private final ScheduledTaskRegistrar taskRegistrar = new ScheduledTaskRegistrar();
36  
37  	@Test
38  	public void getTriggerTasks() {
39  		TriggerTask mockTriggerTask = mock(TriggerTask.class);
40  		List<TriggerTask> triggerTaskList = Collections.singletonList(mockTriggerTask);
41  		this.taskRegistrar.setTriggerTasksList(triggerTaskList);
42  		List<TriggerTask> retrievedList = this.taskRegistrar.getTriggerTaskList();
43  		assertEquals(1, retrievedList.size());
44  		assertEquals(mockTriggerTask, retrievedList.get(0));
45  	}
46
