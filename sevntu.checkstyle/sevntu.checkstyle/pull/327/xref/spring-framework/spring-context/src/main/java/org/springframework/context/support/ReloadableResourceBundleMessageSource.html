41  /**
42   * Spring-specific {@link org.springframework.context.MessageSource} implementation
43   * that accesses resource bundles using specified basenames, participating in the
44   * Spring {@link org.springframework.context.ApplicationContext}'s resource loading.
45   *
46   * <p>In contrast to the JDK-based {@link ResourceBundleMessageSource}, this class uses
47   * {@link java.util.Properties} instances as its custom data structure for messages,
48   * loading them via a {@link org.springframework.util.PropertiesPersister} strategy
49   * from Spring {@link Resource} handles. This strategy is not only capable of
50   * reloading files based on timestamp changes, but also of loading properties files
51   * with a specific character encoding. It will detect XML property files as well.
52   *
53   * <p>In contrast to {@link ResourceBundleMessageSource}, this class supports
54   * reloading of properties files through the {@link #setCacheSeconds "cacheSeconds"}
55   * setting, and also through programmatically clearing the properties cache.
56   * Since application servers typically cache all files loaded from the classpath,
57   * it is necessary to store resources somewhere else (for example, in the
58   * "WEB-INF" directory of a web app). Otherwise changes of files in the
59   * classpath will <i>not</i> be reflected in the application.
60   *
61   * <p>Note that the base names set as {@link #setBasenames "basenames"} property
62   * are treated in a slightly different fashion than the "basenames" property of
63   * {@link ResourceBundleMessageSource}. It follows the basic ResourceBundle rule of not
64   * specifying file extension or language codes, but can refer to any Spring resource
65   * location (instead of being restricted to classpath resources). With a "classpath:"
66   * prefix, resources can still be loaded from the classpath, but "cacheSeconds" values
67   * other than "-1" (caching forever) will not work in this case.
68   *
69   * <p>This MessageSource implementation is usually slightly faster than
70   * {@link ResourceBundleMessageSource}, which builds on {@link java.util.ResourceBundle}
71   * - in the default mode, i.e. when caching forever. With "cacheSeconds" set to 1,
72   * message lookup takes about twice as long - with the benefit that changes in
73   * individual properties files are detected with a maximum delay of 1 second.
74   * Higher "cacheSeconds" values usually <i>do not</i> make a significant difference.
75   *
76   * <p>This MessageSource can easily be used outside of an
77   * {@link org.springframework.context.ApplicationContext}: It will use a
78   * {@link org.springframework.core.io.DefaultResourceLoader} as default,
79   * simply getting overridden with the ApplicationContext's resource loader
80   * if running in a context. It does not have any other specific dependencies.
81   *
82   * <p>Thanks to Thomas Achleitner for providing the initial implementation of
83   * this message source!
84   *
85   * @author Juergen Hoeller
86   * @see #setCacheSeconds
87   * @see #setBasenames
88   * @see #setDefaultEncoding
89   * @see #setFileEncodings
90   * @see #setPropertiesPersister
91   * @see #setResourceLoader
92   * @see org.springframework.util.DefaultPropertiesPersister
93   * @see org.springframework.core.io.DefaultResourceLoader
94   * @see ResourceBundleMessageSource
95   * @see java.util.ResourceBundle
96   */
97  public class ReloadableResourceBundleMessageSource extends AbstractMessageSource implements ResourceLoaderAware {

150 	/**
151 	 * Set an array of basenames, each following the basic ResourceBundle convention
152 	 * of not specifying file extension or language codes, but in contrast to
153 	 * {@link ResourceBundleMessageSource} referring to a Spring resource location:
154 	 * e.g. "WEB-INF/messages" for "WEB-INF/messages.properties",
155 	 * "WEB-INF/messages_en.properties", etc.
156 	 * <p>XML properties files are also supported: .g. "WEB-INF/messages" will find
157 	 * and load "WEB-INF/messages.xml", "WEB-INF/messages_en.xml", etc as well.
158 	 * <p>The associated resource bundles will be checked sequentially when resolving
159 	 * a message code. Note that message definitions in a <i>previous</i> resource
160 	 * bundle will override ones in a later bundle, due to the sequential lookup.
161 	 * @param basenames an array of basenames
162 	 * @see #setBasename
163 	 * @see java.util.ResourceBundle
164 	 */
165 	public void setBasenames(String... basenames) {
166 		if (basenames != null) {
167 			this.basenames = new String[basenames.length];
168 			for (int i = 0; i < basenames.length; i++) {
169 				String basename = basenames[i];
170 				Assert.hasText(basename, "Basename must not be empty");
171 				this.basenames[i] = basename.trim();
172 			}
173 		}
174 		else {
175 			this.basenames = new String[0];
176 		}
177 	}

222 	 * Set the number of seconds to cache loaded properties files.
223 	 * <ul>
224 	 * <li>Default is "-1", indicating to cache forever (just like
225 	 * {@code java.util.ResourceBundle}).
226 	 * <li>A positive number will cache loaded properties files for the given
227 	 * number of seconds. This is essentially the interval between refresh checks.
228 	 * Note that a refresh attempt will first check the last-modified timestamp
229 	 * of the file before actually reloading it; so if files don't change, this
230 	 * interval can be set rather low, as refresh attempts will not actually reload.
231 	 * <li>A value of "0" will check the last-modified timestamp of the file on
232 	 * every message access. <b>Do not use this in a production environment!</b>
233 	 * </ul>
234 	 */
235 	public void setCacheSeconds(int cacheSeconds) {
236 		this.cacheMillis = (cacheSeconds * 1000);
237 	}

278 	/**
279 	 * Resolves the given message code as key in the retrieved bundle files,
280 	 * returning the value found in the bundle as-is (without MessageFormat parsing).
281 	 */
282 	@Override
283 	protected String resolveCodeWithoutArguments(String code, Locale locale) {
284 		if (this.cacheMillis < 0) {
285 			PropertiesHolder propHolder = getMergedProperties(locale);
286 			String result = propHolder.getProperty(code);
287 			if (result != null) {
288 				return result;
289 			}
290 		}
291 		else {
292 			for (String basename : this.basenames) {
293 				List<String> filenames = calculateAllFilenames(basename, locale);
294 				for (String filename : filenames) {
295 					PropertiesHolder propHolder = getProperties(filename);
296 					String result = propHolder.getProperty(code);
297 					if (result != null) {
298 						return result;
299 					}
300 				}
301 			}
302 		}
303 		return null;
304 	}

306 	/**
307 	 * Resolves the given message code as key in the retrieved bundle files,
308 	 * using a cached MessageFormat instance per message code.
309 	 */
310 	@Override
311 	protected MessageFormat resolveCode(String code, Locale locale) {
312 		if (this.cacheMillis < 0) {
313 			PropertiesHolder propHolder = getMergedProperties(locale);
314 			MessageFormat result = propHolder.getMessageFormat(code, locale);
315 			if (result != null) {
316 				return result;
317 			}
318 		}
319 		else {
320 			for (String basename : this.basenames) {
321 				List<String> filenames = calculateAllFilenames(basename, locale);
322 				for (String filename : filenames) {
323 					PropertiesHolder propHolder = getProperties(filename);
324 					MessageFormat result = propHolder.getMessageFormat(code, locale);
325 					if (result != null) {
326 						return result;
327 					}
328 				}
329 			}
330 		}
331 		return null;
332 	}

335 	/**
336 	 * Get a PropertiesHolder that contains the actually visible properties
337 	 * for a Locale, after merging all specified resource bundles.
338 	 * Either fetches the holder from the cache or freshly loads it.
339 	 * <p>Only used when caching resource bundle contents forever, i.e.
340 	 * with cacheSeconds < 0. Therefore, merged properties are always
341 	 * cached forever.
342 	 */
343 	protected PropertiesHolder getMergedProperties(Locale locale) {
344 		PropertiesHolder mergedHolder = this.cachedMergedProperties.get(locale);
345 		if (mergedHolder != null) {
346 			return mergedHolder;
347 		}
348 		Properties mergedProps = newProperties();
349 		mergedHolder = new PropertiesHolder(mergedProps, -1);
350 		for (int i = this.basenames.length - 1; i >= 0; i--) {
351 			List<String> filenames = calculateAllFilenames(this.basenames[i], locale);
352 			for (int j = filenames.size() - 1; j >= 0; j--) {
353 				String filename = filenames.get(j);
354 				PropertiesHolder propHolder = getProperties(filename);
355 				if (propHolder.getProperties() != null) {
356 					mergedProps.putAll(propHolder.getProperties());
357 				}
358 			}
359 		}
360 		PropertiesHolder existing = this.cachedMergedProperties.putIfAbsent(locale, mergedHolder);
361 		if (existing != null) {
362 			mergedHolder = existing;
363 		}
364 		return mergedHolder;
365 	}

367 	/**
368 	 * Calculate all filenames for the given bundle basename and Locale.
369 	 * Will calculate filenames for the given Locale, the system Locale
370 	 * (if applicable), and the default file.
371 	 * @param basename the basename of the bundle
372 	 * @param locale the locale
373 	 * @return the List of filenames to check
374 	 * @see #setFallbackToSystemLocale
375 	 * @see #calculateFilenamesForLocale
376 	 */
377 	protected List<String> calculateAllFilenames(String basename, Locale locale) {
378 		Map<Locale, List<String>> localeMap = this.cachedFilenames.get(basename);
379 		if (localeMap != null) {
380 			List<String> filenames = localeMap.get(locale);
381 			if (filenames != null) {
382 				return filenames;
383 			}
384 		}
385 		List<String> filenames = new ArrayList<String>(7);
386 		filenames.addAll(calculateFilenamesForLocale(basename, locale));
387 		if (this.fallbackToSystemLocale && !locale.equals(Locale.getDefault())) {
388 			List<String> fallbackFilenames = calculateFilenamesForLocale(basename, Locale.getDefault());
389 			for (String fallbackFilename : fallbackFilenames) {
390 				if (!filenames.contains(fallbackFilename)) {
391 					// Entry for fallback locale that isn't already in filenames list.
392 					filenames.add(fallbackFilename);
393 				}
394 			}
395 		}
396 		filenames.add(basename);
397 		if (localeMap == null) {
398 			localeMap = new ConcurrentHashMap<Locale, List<String>>();
399 			Map<Locale, List<String>> existing = this.cachedFilenames.putIfAbsent(basename, localeMap);
400 			if (existing != null) {
401 				localeMap = existing;
402 			}
403 		}
404 		localeMap.put(locale, filenames);
405 		return filenames;
406 	}

408 	/**
409 	 * Calculate the filenames for the given bundle basename and Locale,
410 	 * appending language code, country code, and variant code.
411 	 * E.g.: basename "messages", Locale "de_AT_oo" -> "messages_de_AT_OO",
412 	 * "messages_de_AT", "messages_de".
413 	 * <p>Follows the rules defined by {@link java.util.Locale#toString()}.
414 	 * @param basename the basename of the bundle
415 	 * @param locale the locale
416 	 * @return the List of filenames to check
417 	 */
418 	protected List<String> calculateFilenamesForLocale(String basename, Locale locale) {
419 		List<String> result = new ArrayList<String>(3);
420 		String language = locale.getLanguage();
421 		String country = locale.getCountry();
422 		String variant = locale.getVariant();
423 		StringBuilder temp = new StringBuilder(basename);
424 
425 		temp.append('_');
426 		if (language.length() > 0) {
427 			temp.append(language);
428 			result.add(0, temp.toString());
429 		}
430 
431 		temp.append('_');
432 		if (country.length() > 0) {
433 			temp.append(country);
434 			result.add(0, temp.toString());
435 		}
436 
437 		if (variant.length() > 0 && (language.length() > 0 || country.length() > 0)) {
438 			temp.append('_').append(variant);
439 			result.add(0, temp.toString());
440 		}
441 
442 		return result;
443 	}

446 	/**
447 	 * Get a PropertiesHolder for the given filename, either from the
448 	 * cache or freshly loaded.
449 	 * @param filename the bundle filename (basename + Locale)
450 	 * @return the current PropertiesHolder for the bundle
451 	 */
452 	protected PropertiesHolder getProperties(String filename) {
453 		PropertiesHolder propHolder = this.cachedProperties.get(filename);
454 		long originalTimestamp = -2;
455 
456 		if (propHolder != null) {
457 			originalTimestamp = propHolder.getRefreshTimestamp();
458 			if (originalTimestamp == -1 || originalTimestamp > System.currentTimeMillis() - this.cacheMillis) {
459 				// Up to date
460 				return propHolder;
461 			}
462 		}
463 		else {
464 			propHolder = new PropertiesHolder();
465 			PropertiesHolder existingHolder = this.cachedProperties.putIfAbsent(filename, propHolder);
466 			if (existingHolder != null) {
467 				propHolder = existingHolder;
468 			}
469 		}
470 
471 		// At this point, we need to refresh...
472 		if (this.concurrentRefresh && propHolder.getRefreshTimestamp() >= 0) {
473 			// A populated but stale holder -> could keep using it.
474 			if (!propHolder.refreshLock.tryLock()) {
475 				// Getting refreshed by another thread already ->
476 				// let's return the existing properties for the time being.
477 				return propHolder;
478 			}
479 		}
480 		else {
481 			propHolder.refreshLock.lock();
482 		}
483 		try {
484 			PropertiesHolder existingHolder = this.cachedProperties.get(filename);
485 			if (existingHolder != null && existingHolder.getRefreshTimestamp() > originalTimestamp) {
486 				return existingHolder;
487 			}
488 			return refreshProperties(filename, propHolder);
489 		}
490 		finally {
491 			propHolder.refreshLock.unlock();
492 		}
493 	}

495 	/**
496 	 * Refresh the PropertiesHolder for the given bundle filename.
497 	 * The holder can be {@code null} if not cached before, or a timed-out cache entry
498 	 * (potentially getting re-validated against the current last-modified timestamp).
499 	 * @param filename the bundle filename (basename + Locale)
500 	 * @param propHolder the current PropertiesHolder for the bundle
501 	 */
502 	protected PropertiesHolder refreshProperties(String filename, PropertiesHolder propHolder) {
503 		long refreshTimestamp = (this.cacheMillis < 0 ? -1 : System.currentTimeMillis());
504 
505 		Resource resource = this.resourceLoader.getResource(filename + PROPERTIES_SUFFIX);
506 		if (!resource.exists()) {
507 			resource = this.resourceLoader.getResource(filename + XML_SUFFIX);
508 		}
509 
510 		if (resource.exists()) {
511 			long fileTimestamp = -1;
512 			if (this.cacheMillis >= 0) {
513 				// Last-modified timestamp of file will just be read if caching with timeout.
514 				try {
515 					fileTimestamp = resource.lastModified();
516 					if (propHolder != null && propHolder.getFileTimestamp() == fileTimestamp) {
517 						if (logger.isDebugEnabled()) {
518 							logger.debug("Re-caching properties for filename [" + filename + "] - file hasn't been modified");
519 						}
520 						propHolder.setRefreshTimestamp(refreshTimestamp);
521 						return propHolder;
522 					}
523 				}
524 				catch (IOException ex) {
525 					// Probably a class path resource: cache it forever.
526 					if (logger.isDebugEnabled()) {
527 						logger.debug(resource + " could not be resolved in the file system - assuming that it hasn't changed", ex);
528 					}
529 					fileTimestamp = -1;
530 				}
531 			}
532 			try {
533 				Properties props = loadProperties(resource, filename);
534 				propHolder = new PropertiesHolder(props, fileTimestamp);
535 			}
536 			catch (IOException ex) {
537 				if (logger.isWarnEnabled()) {
538 					logger.warn("Could not parse properties file [" + resource.getFilename() + "]", ex);
539 				}
540 				// Empty holder representing "not valid".
541 				propHolder = new PropertiesHolder();
542 			}
543 		}
544 
545 		else {
546 			// Resource does not exist.
547 			if (logger.isDebugEnabled()) {
548 				logger.debug("No properties file found for [" + filename + "] - neither plain properties nor XML");
549 			}
550 			// Empty holder representing "not found".
551 			propHolder = new PropertiesHolder();
552 		}
553 
554 		propHolder.setRefreshTimestamp(refreshTimestamp);
555 		this.cachedProperties.put(filename, propHolder);
556 		return propHolder;
557 	}

559 	/**
560 	 * Load the properties from the given resource.
561 	 * @param resource the resource to load from
562 	 * @param filename the original bundle filename (basename + Locale)
563 	 * @return the populated Properties instance
564 	 * @throws IOException if properties loading failed
565 	 */
566 	protected Properties loadProperties(Resource resource, String filename) throws IOException {
567 		InputStream is = resource.getInputStream();
568 		Properties props = newProperties();
569 		try {
570 			if (resource.getFilename().endsWith(XML_SUFFIX)) {
571 				if (logger.isDebugEnabled()) {
572 					logger.debug("Loading properties [" + resource.getFilename() + "]");
573 				}
574 				this.propertiesPersister.loadFromXml(props, is);
575 			}
576 			else {
577 				String encoding = null;
578 				if (this.fileEncodings != null) {
579 					encoding = this.fileEncodings.getProperty(filename);
580 				}
581 				if (encoding == null) {
582 					encoding = this.defaultEncoding;
583 				}
584 				if (encoding != null) {
585 					if (logger.isDebugEnabled()) {
586 						logger.debug("Loading properties [" + resource.getFilename() + "] with encoding '" + encoding + "'");
587 					}
588 					this.propertiesPersister.load(props, new InputStreamReader(is, encoding));
589 				}
590 				else {
591 					if (logger.isDebugEnabled()) {
592 						logger.debug("Loading properties [" + resource.getFilename() + "]");
593 					}
594 					this.propertiesPersister.load(props, is);
595 				}
596 			}
597 			return props;
598 		}
599 		finally {
600 			is.close();
601 		}
602 	}

646 	/**
647 	 * PropertiesHolder for caching.
648 	 * Stores the last-modified timestamp of the source file for efficient
649 	 * change detection, and the timestamp of the last refresh attempt
650 	 * (updated every time the cache entry gets re-validated).
651 	 */
652 	protected class PropertiesHolder {
653 
654 		private final Properties properties;
655 
656 		private final long fileTimestamp;
657 
658 		private volatile long refreshTimestamp = -2;
659 
660 		private final ReentrantLock refreshLock = new ReentrantLock();
661 
662 		/** Cache to hold already generated MessageFormats per message code */
663 		private final ConcurrentMap<String, Map<Locale, MessageFormat>> cachedMessageFormats =
664 				new ConcurrentHashMap<String, Map<Locale, MessageFormat>>();
665 
666 		public PropertiesHolder() {
667 			this.properties = null;
668 			this.fileTimestamp = -1;
669 		}
670 
671 		public PropertiesHolder(Properties properties, long fileTimestamp) {
672 			this.properties = properties;
673 			this.fileTimestamp = fileTimestamp;
674 		}
675 
676 		public Properties getProperties() {
677 			return this.properties;
678 		}
679 
680 		public long getFileTimestamp() {
681 			return this.fileTimestamp;
682 		}
683 
684 		public void setRefreshTimestamp(long refreshTimestamp) {
685 			this.refreshTimestamp = refreshTimestamp;
686 		}
687 
688 		public long getRefreshTimestamp() {
689 			return this.refreshTimestamp;
690 		}
691 
692 		public String getProperty(String code) {
693 			if (this.properties == null) {
694 				return null;
695 			}
696 			return this.properties.getProperty(code);
697 		}
698 
699 		public MessageFormat getMessageFormat(String code, Locale locale) {
700 			if (this.properties == null) {
701 				return null;
702 			}
703 			Map<Locale, MessageFormat> localeMap = this.cachedMessageFormats.get(code);
704 			if (localeMap != null) {
705 				MessageFormat result = localeMap.get(locale);
706 				if (result != null) {
707 					return result;
708 				}
709 			}
710 			String msg = this.properties.getProperty(code);
711 			if (msg != null) {
712 				if (localeMap == null) {
713 					localeMap = new ConcurrentHashMap<Locale, MessageFormat>();
714 					Map<Locale, MessageFormat> existing = this.cachedMessageFormats.putIfAbsent(code, localeMap);
715 					if (existing != null) {
716 						localeMap = existing;
717 					}
718 				}
719 				MessageFormat result = createMessageFormat(msg, locale);
720 				localeMap.put(locale, result);
721 				return result;
722 			}
723 			return null;
724 		}
725 	}
