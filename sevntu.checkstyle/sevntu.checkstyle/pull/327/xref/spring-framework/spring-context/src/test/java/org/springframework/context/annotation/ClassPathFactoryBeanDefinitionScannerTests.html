36  /**
37   * @author Mark Pollack
38   * @author Juergen Hoeller
39   */
40  public class ClassPathFactoryBeanDefinitionScannerTests {
41  
42  	private static final String BASE_PACKAGE = FactoryMethodComponent.class.getPackage().getName();
43  
44  
45  	@Test
46  	public void testSingletonScopedFactoryMethod() {
47  		GenericApplicationContext context = new GenericApplicationContext();
48  		ClassPathBeanDefinitionScanner scanner = new ClassPathBeanDefinitionScanner(context);
49  
50  		context.getBeanFactory().registerScope("request", new SimpleMapScope());
51  
52  		scanner.scan(BASE_PACKAGE);
53  		context.registerBeanDefinition("clientBean", new RootBeanDefinition(QualifiedClientBean.class));
54  		context.refresh();
55  
56  		FactoryMethodComponent fmc = context.getBean("factoryMethodComponent", FactoryMethodComponent.class);
57  		assertFalse(fmc.getClass().getName().contains(ClassUtils.CGLIB_CLASS_SEPARATOR));
58  
59  		TestBean tb = (TestBean) context.getBean("publicInstance"); //2
60  		assertEquals("publicInstance", tb.getName());
61  		TestBean tb2 = (TestBean) context.getBean("publicInstance"); //2
62  		assertEquals("publicInstance", tb2.getName());
63  		assertSame(tb2, tb);
64  
65  		tb = (TestBean) context.getBean("protectedInstance"); //3
66  		assertEquals("protectedInstance", tb.getName());
67  		assertSame(tb, context.getBean("protectedInstance"));
68  		assertEquals("0", tb.getCountry());
69  		tb2 = context.getBean("protectedInstance", TestBean.class); //3
70  		assertEquals("protectedInstance", tb2.getName());
71  		assertSame(tb2, tb);
72  
73  		tb = context.getBean("privateInstance", TestBean.class); //4
74  		assertEquals("privateInstance", tb.getName());
75  		assertEquals(1, tb.getAge());
76  		tb2 = context.getBean("privateInstance", TestBean.class); //4
77  		assertEquals(2, tb2.getAge());
78  		assertNotSame(tb2, tb);
79  
80  		Object bean = context.getBean("requestScopedInstance"); //5
81  		assertTrue(AopUtils.isCglibProxy(bean));
82  		assertTrue(bean instanceof ScopedObject);
83  
84  		QualifiedClientBean clientBean = context.getBean("clientBean", QualifiedClientBean.class);
85  		assertSame(context.getBean("publicInstance"), clientBean.testBean);
86  		assertSame(context.getBean("dependencyBean"), clientBean.dependencyBean);
87  		assertSame(context, clientBean.applicationContext);
88  	}
89  
90  
91  	public static class QualifiedClientBean {
92  
93  		@Autowired @Qualifier("public")
94  		public TestBean testBean;
95  
96  		@Autowired
97  		public DependencyBean dependencyBean;
98  
99  		@Autowired
100 		AbstractApplicationContext applicationContext;
101 	}
102 
103 }
