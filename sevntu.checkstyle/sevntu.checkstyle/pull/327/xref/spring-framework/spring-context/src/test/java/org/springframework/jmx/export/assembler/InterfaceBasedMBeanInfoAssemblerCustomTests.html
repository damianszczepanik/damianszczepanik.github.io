17  package org.springframework.jmx.export.assembler;
18  
19  import javax.management.modelmbean.ModelMBeanAttributeInfo;
20  import javax.management.modelmbean.ModelMBeanInfo;
21  
22  import org.junit.Test;
23  
24  import static org.junit.Assert.*;
25  
26  /**
27   * @author Rob Harrop
28   * @author Chris Beams
29   */
30  public class InterfaceBasedMBeanInfoAssemblerCustomTests extends AbstractJmxAssemblerTests {
31  
32  	protected static final String OBJECT_NAME = "bean:name=testBean5";
33  
34  	@Override
35  	protected String getObjectName() {
36  		return OBJECT_NAME;
37  	}
38  
39  	@Override
40  	protected int getExpectedOperationCount() {
41  		return 5;
42  	}
43  
44  	@Override
45  	protected int getExpectedAttributeCount() {
46  		return 2;
47  	}
48  
49  	@Override
50  	protected MBeanInfoAssembler getAssembler() {
51  		InterfaceBasedMBeanInfoAssembler assembler = new InterfaceBasedMBeanInfoAssembler();
52  		assembler.setManagedInterfaces(new Class<?>[] {ICustomJmxBean.class});
53  		return assembler;
54  	}
55  
56  	@Test
57  	public void testGetAgeIsReadOnly() throws Exception {
58  		ModelMBeanInfo info = getMBeanInfoFromAssembler();
59  		ModelMBeanAttributeInfo attr = info.getAttribute(AGE_ATTRIBUTE);
60  
61  		assertTrue(attr.isReadable());
62  		assertFalse(attr.isWritable());
63  	}
64  
65  	@Override
66  	protected String getApplicationContextPath() {
67  		return "org/springframework/jmx/export/assembler/interfaceAssemblerCustom.xml";
68  	}
69  
70  }
