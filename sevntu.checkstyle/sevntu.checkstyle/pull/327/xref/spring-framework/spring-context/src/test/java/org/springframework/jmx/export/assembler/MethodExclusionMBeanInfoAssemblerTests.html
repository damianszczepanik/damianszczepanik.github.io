17  package org.springframework.jmx.export.assembler;
18  
19  import java.lang.reflect.Method;
20  import java.util.Properties;
21  import javax.management.modelmbean.ModelMBeanAttributeInfo;
22  import javax.management.modelmbean.ModelMBeanInfo;
23  
24  import org.junit.Test;
25  
26  import org.springframework.jmx.JmxTestBean;
27  
28  import static org.junit.Assert.*;
29  
30  /**
31   * @author Rob Harrop
32   * @author Rick Evans
33   * @author Chris Beams
34   */
35  public class MethodExclusionMBeanInfoAssemblerTests extends AbstractJmxAssemblerTests {
36  
37  	private static final String OBJECT_NAME = "bean:name=testBean5";
38  
39  
40  	@Override
41  	protected String getObjectName() {
42  		return OBJECT_NAME;
43  	}
44  
45  	@Override
46  	protected int getExpectedOperationCount() {
47  		return 9;
48  	}
49  
50  	@Override
51  	protected int getExpectedAttributeCount() {
52  		return 4;
53  	}
54  
55  	@Override
56  	protected String getApplicationContextPath() {
57  		return "org/springframework/jmx/export/assembler/methodExclusionAssembler.xml";
58  	}
59  
60  	@Override
61  	protected MBeanInfoAssembler getAssembler() {
62  		MethodExclusionMBeanInfoAssembler assembler = new MethodExclusionMBeanInfoAssembler();
63  		assembler.setIgnoredMethods(new String[] {"dontExposeMe", "setSuperman"});
64  		return assembler;
65  	}
66  
67  	@Test
68  	public void testSupermanIsReadOnly() throws Exception {
69  		ModelMBeanInfo info = getMBeanInfoFromAssembler();
70  		ModelMBeanAttributeInfo attr = info.getAttribute("Superman");
71  
72  		assertTrue(attr.isReadable());
73  		assertFalse(attr.isWritable());
74  	}
75  
76  	/*
77  	 * http://opensource.atlassian.com/projects/spring/browse/SPR-2754
78  	 */
79  	@Test
80  	public void testIsNotIgnoredDoesntIgnoreUnspecifiedBeanMethods() throws Exception {
81  		final String beanKey = "myTestBean";
82  		MethodExclusionMBeanInfoAssembler assembler = new MethodExclusionMBeanInfoAssembler();
83  		Properties ignored = new Properties();
84  		ignored.setProperty(beanKey, "dontExposeMe,setSuperman");
85  		assembler.setIgnoredMethodMappings(ignored);
86  		Method method = JmxTestBean.class.getMethod("dontExposeMe");
87  		assertFalse(assembler.isNotIgnored(method, beanKey));
88  		// this bean does not have any ignored methods on it, so must obviously not be ignored...
89  		assertTrue(assembler.isNotIgnored(method, "someOtherBeanKey"));
90  	}

92  }
