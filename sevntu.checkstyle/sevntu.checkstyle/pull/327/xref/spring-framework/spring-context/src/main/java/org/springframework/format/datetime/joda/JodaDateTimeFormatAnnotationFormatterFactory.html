17  package org.springframework.format.datetime.joda;
18  
19  import java.util.Calendar;
20  import java.util.Collections;
21  import java.util.Date;
22  import java.util.HashSet;
23  import java.util.Set;
24  
25  import org.joda.time.LocalDate;
26  import org.joda.time.LocalDateTime;
27  import org.joda.time.LocalTime;
28  import org.joda.time.ReadableInstant;
29  import org.joda.time.ReadablePartial;
30  import org.joda.time.format.DateTimeFormatter;
31  
32  import org.springframework.context.support.EmbeddedValueResolutionSupport;
33  import org.springframework.format.AnnotationFormatterFactory;
34  import org.springframework.format.Parser;
35  import org.springframework.format.Printer;
36  import org.springframework.format.annotation.DateTimeFormat;
37  
38  /**
39   * Formats fields annotated with the {@link DateTimeFormat} annotation using Joda-Time.
40   *
41   * <p><b>NOTE:</b> Spring's Joda-Time support requires Joda-Time 2.x, as of Spring 4.0.
42   *
43   * @author Keith Donald
44   * @author Juergen Hoeller
45   * @since 3.0
46   * @see DateTimeFormat
47   */
48  public class JodaDateTimeFormatAnnotationFormatterFactory extends EmbeddedValueResolutionSupport
49  		implements AnnotationFormatterFactory<DateTimeFormat> {
50  
51  	private static final Set<Class<?>> FIELD_TYPES;
52  
53  	static {
54  		// Create the set of field types that may be annotated with @DateTimeFormat.
55  		// Note: the 3 ReadablePartial concrete types are registered explicitly since
56  		// addFormatterForFieldType rules exist for each of these types
57  		// (if we did not do this, the default byType rules for LocalDate, LocalTime,
58  		// and LocalDateTime would take precedence over the annotation rule, which
59  		// is not what we want)
60  		Set<Class<?>> fieldTypes = new HashSet<Class<?>>(8);
61  		fieldTypes.add(ReadableInstant.class);
62  		fieldTypes.add(LocalDate.class);
63  		fieldTypes.add(LocalTime.class);
64  		fieldTypes.add(LocalDateTime.class);
65  		fieldTypes.add(Date.class);
66  		fieldTypes.add(Calendar.class);
67  		fieldTypes.add(Long.class);
68  		FIELD_TYPES = Collections.unmodifiableSet(fieldTypes);
69  	}
70  
71  
72  	@Override
73  	public final Set<Class<?>> getFieldTypes() {
74  		return FIELD_TYPES;
75  	}
76  
77  	@Override
78  	public Printer<?> getPrinter(DateTimeFormat annotation, Class<?> fieldType) {
79  		DateTimeFormatter formatter = getFormatter(annotation, fieldType);
80  		if (ReadablePartial.class.isAssignableFrom(fieldType)) {
81  			return new ReadablePartialPrinter(formatter);
82  		}
83  		else if (ReadableInstant.class.isAssignableFrom(fieldType) || Calendar.class.isAssignableFrom(fieldType)) {
84  			// assumes Calendar->ReadableInstant converter is registered
85  			return new ReadableInstantPrinter(formatter);
86  		}
87  		else {
88  			// assumes Date->Long converter is registered
89  			return new MillisecondInstantPrinter(formatter);
90  		}
91  	}
92  
93  	@Override
94  	public Parser<?> getParser(DateTimeFormat annotation, Class<?> fieldType) {
95  		if (LocalDate.class.equals(fieldType)) {
96  			return new LocalDateParser(getFormatter(annotation, fieldType));
97  		}
98  		else if (LocalTime.class.equals(fieldType)) {
99  			return new LocalTimeParser(getFormatter(annotation, fieldType));
100 		}
101 		else if (LocalDateTime.class.equals(fieldType)) {
102 			return new LocalDateTimeParser(getFormatter(annotation, fieldType));
103 		}
104 		else {
105 			return new DateTimeParser(getFormatter(annotation, fieldType));
106 		}
107 	}
108 
109 	/**
110 	 * Factory method used to create a {@link DateTimeFormatter}.
111 	 * @param annotation the format annotation for the field
112 	 * @param fieldType the type of field
113 	 * @return a {@link DateTimeFormatter} instance
114 	 * @since 3.2
115 	 */
116 	protected DateTimeFormatter getFormatter(DateTimeFormat annotation, Class<?> fieldType) {
117 		DateTimeFormatterFactory factory = new DateTimeFormatterFactory();
118 		factory.setStyle(resolveEmbeddedValue(annotation.style()));
119 		factory.setIso(annotation.iso());
120 		factory.setPattern(resolveEmbeddedValue(annotation.pattern()));
121 		return factory.createDateTimeFormatter();
122 	}
123 
124 }
