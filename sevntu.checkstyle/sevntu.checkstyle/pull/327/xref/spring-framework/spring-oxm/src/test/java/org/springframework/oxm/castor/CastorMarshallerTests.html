129 	@Test
130 	public void marshalSaxResult() throws Exception {
131 		ContentHandler contentHandler = mock(ContentHandler.class);
132 		SAXResult result = new SAXResult(contentHandler);
133 		marshaller.marshal(flights, result);
134 		InOrder ordered = inOrder(contentHandler);
135 		ordered.verify(contentHandler).startDocument();
136 		ordered.verify(contentHandler).startPrefixMapping("tns", "http://samples.springframework.org/flight");
137 		ordered.verify(contentHandler).startElement(eq("http://samples.springframework.org/flight"),
138 				eq("flights"), eq("tns:flights"), isA(Attributes.class));
139 		ordered.verify(contentHandler).startElement(eq("http://samples.springframework.org/flight"),
140 				eq("flight"), eq("tns:flight"), isA(Attributes.class));
141 		ordered.verify(contentHandler).startElement(eq("http://samples.springframework.org/flight"),
142 				eq("number"), eq("tns:number"), isA(Attributes.class));
143 		ordered.verify(contentHandler).characters(eq(new char[]{'4', '2'}), eq(0), eq(2));
144 		ordered.verify(contentHandler).endElement("http://samples.springframework.org/flight", "number", "tns:number");
145 		ordered.verify(contentHandler).endElement("http://samples.springframework.org/flight", "flight", "tns:flight");
146 		ordered.verify(contentHandler).endElement("http://samples.springframework.org/flight", "flights", "tns:flights");
147 		ordered.verify(contentHandler).endPrefixMapping("tns");
148 		ordered.verify(contentHandler).endDocument();
149 	}

259 		StringWriter writer = new StringWriter();

269 	/**
270 	 * Assert the values of xpath expression evaluation is exactly the same as expected value.
271 	 * <p>The xpath may contain the xml namespace prefixes, since namespaces from flight example
272 	 * are being registered.
273 	 * @param msg the error message that will be used in case of test failure
274 	 * @param expected the expected value
275 	 * @param xpath the xpath to evaluate
276 	 * @param xmlDoc the xml to use
277 	 * @throws Exception if any error occurs during xpath evaluation
278 	 */
279 	private void assertXpathEvaluatesTo(String msg, String expected, String xpath, String xmlDoc) throws Exception {
280 		Map<String, String> namespaces = new HashMap<String, String>();
281 		namespaces.put("tns", "http://samples.springframework.org/flight");
282 		namespaces.put("xsi", "http://www.w3.org/2001/XMLSchema-instance");
283 
284 		NamespaceContext ctx = new SimpleNamespaceContext(namespaces);
285 		XpathEngine engine = XMLUnit.newXpathEngine();
286 		engine.setNamespaceContext(ctx);
287 
288 		Document doc = XMLUnit.buildControlDocument(xmlDoc);
289 		NodeList node = engine.getMatchingNodes(xpath, doc);
290 		assertEquals(msg, expected, node.item(0).getNodeValue());
291 	}
