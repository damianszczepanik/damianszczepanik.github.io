17  package org.springframework.oxm.jibx;
18  
19  import java.io.ByteArrayInputStream;
20  import javax.xml.transform.stream.StreamSource;
21  
22  import org.junit.BeforeClass;
23  import org.junit.Ignore;
24  import org.junit.Test;
25  
26  import org.springframework.oxm.AbstractUnmarshallerTests;
27  import org.springframework.oxm.Unmarshaller;
28  import org.springframework.tests.Assume;
29  import org.springframework.tests.TestGroup;
30  
31  import static org.junit.Assert.*;
32  
33  
34  /**
35   * @author Arjen Poutsma
36   *
37   * NOTE: These tests fail under Eclipse/IDEA because JiBX binding does
38   * not occur by default. The Gradle build should succeed, however.
39   */
40  public class JibxUnmarshallerTests extends AbstractUnmarshallerTests {
41  
42  	protected static final String INPUT_STRING_WITH_SPECIAL_CHARACTERS =
43  			"<tns:flights xmlns:tns=\"http://samples.springframework.org/flight\">" +
44  					"<tns:flight><tns:airline>Air Libert\u00e9</tns:airline><tns:number>42</tns:number></tns:flight></tns:flights>";
45  
46  	@BeforeClass
47  	public static void compilerAssumptions() {
48  		Assume.group(TestGroup.CUSTOM_COMPILATION);
49  	}
50  
51  	@Override
52  	protected Unmarshaller createUnmarshaller() throws Exception {
53  		JibxMarshaller unmarshaller = new JibxMarshaller();
54  		unmarshaller.setTargetClass(Flights.class);
55  		unmarshaller.afterPropertiesSet();
56  		return unmarshaller;
57  	}
58  
59  	@Override
60  	protected void testFlights(Object o) {
61  		Flights flights = (Flights) o;
62  		assertNotNull("Flights is null", flights);
63  		assertEquals("Invalid amount of flight elements", 1, flights.sizeFlightList());
64  		testFlight(flights.getFlight(0));
65  	}
66  
67  	@Override
68  	protected void testFlight(Object o) {
69  		FlightType flight = (FlightType) o;
70  		assertNotNull("Flight is null", flight);
71  		assertEquals("Number is invalid", 42L, flight.getNumber());
72  	}
73  
74  	@Override
75  	@Ignore
76  	public void unmarshalPartialStaxSourceXmlStreamReader() throws Exception {
77  		// JiBX does not support reading XML fragments, hence the override here
78  	}
79  
80  	@Test
81  	public void unmarshalStreamSourceInputStreamUsingNonDefaultEncoding() throws Exception {
82  		String encoding = "ISO-8859-1";
83  		((JibxMarshaller)unmarshaller).setEncoding(encoding);
84  
85  		StreamSource source = new StreamSource(new ByteArrayInputStream(INPUT_STRING_WITH_SPECIAL_CHARACTERS.getBytes(encoding)));
86  		Object flights = unmarshaller.unmarshal(source);
87  		testFlights(flights);
88  
89  		FlightType flight = ((Flights)flights).getFlight(0);
90  		assertEquals("Airline is invalid", "Air Libert\u00e9", flight.getAirline());
91  	}
