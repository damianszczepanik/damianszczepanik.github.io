18  
19  import java.io.StringReader;
20  import javax.xml.namespace.QName;
21  import javax.xml.stream.XMLInputFactory;
22  import javax.xml.stream.XMLStreamReader;
23  import javax.xml.transform.Source;
24  import javax.xml.transform.stream.StreamSource;

26  import org.junit.Test;
27  
28  import org.springframework.oxm.AbstractUnmarshallerTests;
29  import org.springframework.oxm.Unmarshaller;
30  import org.springframework.oxm.ValidationFailureException;
31  import org.springframework.samples.flight.FlightDocument;
32  import org.springframework.samples.flight.FlightType;
33  import org.springframework.samples.flight.FlightsDocument;
34  import org.springframework.util.xml.StaxUtils;
35  
36  import static org.junit.Assert.*;
37  
38  /**
39   * @author Arjen Poutsma
40   */
41  public class XmlBeansUnmarshallerTests extends AbstractUnmarshallerTests {
42  
43  	@Override
44  	protected Unmarshaller createUnmarshaller() throws Exception {
45  		return new XmlBeansMarshaller();
46  	}
47  
48  	@Override
49  	protected void testFlights(Object o) {
50  		FlightsDocument flightsDocument = (FlightsDocument) o;
51  		assertNotNull("FlightsDocument is null", flightsDocument);
52  		FlightsDocument.Flights flights = flightsDocument.getFlights();
53  		assertEquals("Invalid amount of flight elements", 1, flights.sizeOfFlightArray());
54  		testFlight(flights.getFlightArray(0));
55  	}
56  
57  	@Override
58  	protected void testFlight(Object o) {
59  		FlightType flight = null;
60  		if (o instanceof FlightType) {
61  			flight = (FlightType) o;
62  		}
63  		else if (o instanceof FlightDocument) {
64  			FlightDocument flightDocument = (FlightDocument) o;
65  			flight = flightDocument.getFlight();
66  		}
67  		assertNotNull("Flight is null", flight);
68  		assertEquals("Number is invalid", 42L, flight.getNumber());
69  	}
70  
71  	@Override
72  	public void unmarshalPartialStaxSourceXmlStreamReader() throws Exception {
73  		XMLInputFactory inputFactory = XMLInputFactory.newInstance();
74  		XMLStreamReader streamReader = inputFactory.createXMLStreamReader(new StringReader(INPUT_STRING));
75  		streamReader.nextTag(); // skip to flights
76  		assertEquals("Invalid element", new QName("http://samples.springframework.org/flight", "flights"),
77  				streamReader.getName());
78  		streamReader.nextTag(); // skip to flight
79  		assertEquals("Invalid element", new QName("http://samples.springframework.org/flight", "flight"),
80  				streamReader.getName());
81  		Source source = StaxUtils.createStaxSource(streamReader);
82  		Object flight = unmarshaller.unmarshal(source);
83  		testFlight(flight);
84  	}
85  
86  	@Test(expected = ValidationFailureException.class)
87  	public void testValidate() throws Exception {
88  		((XmlBeansMarshaller) unmarshaller).setValidating(true);
89  		String invalidInput = "<tns:flights xmlns:tns=\"http://samples.springframework.org/flight\">" +
90  				"<tns:flight><tns:number>abc</tns:number></tns:flight></tns:flights>";
91  		unmarshaller.unmarshal(new StreamSource(new StringReader(invalidInput)));
92  	}
