419 	@Override
420 	public void afterPropertiesSet() throws Exception {
421 		boolean hasContextPath = StringUtils.hasLength(this.contextPath);
422 		boolean hasClassesToBeBound = !ObjectUtils.isEmpty(this.classesToBeBound);
423 		boolean hasPackagesToScan = !ObjectUtils.isEmpty(this.packagesToScan);
424 
425 		if (hasContextPath && (hasClassesToBeBound || hasPackagesToScan) ||
426 				(hasClassesToBeBound && hasPackagesToScan)) {
427 			throw new IllegalArgumentException("Specify either 'contextPath', 'classesToBeBound', " +
428 					"or 'packagesToScan'");
429 		}
430 		if (!hasContextPath && !hasClassesToBeBound && !hasPackagesToScan) {
431 			throw new IllegalArgumentException(
432 					"Setting either 'contextPath', 'classesToBeBound', " + "or 'packagesToScan' is required");
433 		}
434 		if (!this.lazyInit) {
435 			getJaxbContext();
436 		}
437 		if (!ObjectUtils.isEmpty(this.schemaResources)) {
438 			this.schema = loadSchema(this.schemaResources, this.schemaLanguage);
439 		}
440 	}

442 	/**
443 	 * Return the JAXBContext used by this marshaller, lazily building it if necessary.
444 	 */
445 	public JAXBContext getJaxbContext() {
446 		if (this.jaxbContext != null) {
447 			return this.jaxbContext;
448 		}
449 		synchronized (this.jaxbContextMonitor) {
450 			if (this.jaxbContext == null) {
451 				try {
452 					if (StringUtils.hasLength(this.contextPath)) {
453 						this.jaxbContext = createJaxbContextFromContextPath();
454 					}
455 					else if (!ObjectUtils.isEmpty(this.classesToBeBound)) {
456 						this.jaxbContext = createJaxbContextFromClasses();
457 					}
458 					else if (!ObjectUtils.isEmpty(this.packagesToScan)) {
459 						this.jaxbContext = createJaxbContextFromPackages();
460 					}
461 				}
462 				catch (JAXBException ex) {
463 					throw convertJaxbException(ex);
464 				}
465 			}
466 			return this.jaxbContext;
467 		}
468 	}

470 	private JAXBContext createJaxbContextFromContextPath() throws JAXBException {
471 		if (logger.isInfoEnabled()) {
472 			logger.info("Creating JAXBContext with context path [" + this.contextPath + "]");
473 		}
474 		if (this.jaxbContextProperties != null) {
475 			if (this.beanClassLoader != null) {
476 				return JAXBContext.newInstance(this.contextPath, this.beanClassLoader, this.jaxbContextProperties);
477 			}
478 			else {
479 				// analogous to the JAXBContext.newInstance(String) implementation
480 				return JAXBContext.newInstance(this.contextPath, Thread.currentThread().getContextClassLoader(),
481 						this.jaxbContextProperties);
482 			}
483 		}
484 		else {
485 			if (this.beanClassLoader != null) {
486 				return JAXBContext.newInstance(this.contextPath, this.beanClassLoader);
487 			}
488 			else {
489 				return JAXBContext.newInstance(this.contextPath);
490 			}
491 		}
492 	}

494 	private JAXBContext createJaxbContextFromClasses() throws JAXBException {
495 		if (logger.isInfoEnabled()) {
496 			logger.info("Creating JAXBContext with classes to be bound [" +
497 					StringUtils.arrayToCommaDelimitedString(this.classesToBeBound) + "]");
498 		}
499 		if (this.jaxbContextProperties != null) {
500 			return JAXBContext.newInstance(this.classesToBeBound, this.jaxbContextProperties);
501 		}
502 		else {
503 			return JAXBContext.newInstance(this.classesToBeBound);
504 		}
505 	}
506 
507 	private JAXBContext createJaxbContextFromPackages() throws JAXBException {
508 		if (logger.isInfoEnabled()) {
509 			logger.info("Creating JAXBContext by scanning packages [" +
510 					StringUtils.arrayToCommaDelimitedString(this.packagesToScan) + "]");
511 		}
512 		ClassPathJaxb2TypeScanner scanner = new ClassPathJaxb2TypeScanner(this.beanClassLoader, this.packagesToScan);
513 		Class<?>[] jaxb2Classes = scanner.scanPackages();
514 		if (logger.isDebugEnabled()) {
515 			logger.debug("Found JAXB2 classes: [" + StringUtils.arrayToCommaDelimitedString(jaxb2Classes) + "]");
516 		}
517 		this.classesToBeBound = jaxb2Classes;
518 		if (this.jaxbContextProperties != null) {
519 			return JAXBContext.newInstance(jaxb2Classes, this.jaxbContextProperties);
520 		}
521 		else {
522 			return JAXBContext.newInstance(jaxb2Classes);
523 		}
524 	}

526 	private Schema loadSchema(Resource[] resources, String schemaLanguage) throws IOException, SAXException {
527 		if (logger.isDebugEnabled()) {
528 			logger.debug("Setting validation schema to " +
529 					StringUtils.arrayToCommaDelimitedString(this.schemaResources));
530 		}
531 		Assert.notEmpty(resources, "No resources given");
532 		Assert.hasLength(schemaLanguage, "No schema language provided");
533 		Source[] schemaSources = new Source[resources.length];
534 		XMLReader xmlReader = XMLReaderFactory.createXMLReader();
535 		xmlReader.setFeature("http://xml.org/sax/features/namespace-prefixes", true);
536 		for (int i = 0; i < resources.length; i++) {
537 			Assert.notNull(resources[i], "Resource is null");
538 			Assert.isTrue(resources[i].exists(), "Resource " + resources[i] + " does not exist");
539 			InputSource inputSource = SaxResourceUtils.createInputSource(resources[i]);
540 			schemaSources[i] = new SAXSource(xmlReader, inputSource);
541 		}
542 		SchemaFactory schemaFactory = SchemaFactory.newInstance(schemaLanguage);
543 		if (this.schemaResourceResolver != null) {
544 			schemaFactory.setResourceResolver(this.schemaResourceResolver);
545 		}
546 		return schemaFactory.newSchema(schemaSources);
547 	}

551 	public boolean supports(Class<?> clazz) {
552 		return ((this.supportJaxbElementClass && JAXBElement.class.isAssignableFrom(clazz)) ||
553 				supportsInternal(clazz, this.checkForXmlRootElement));
554 	}

556 	@Override
557 	public boolean supports(Type genericType) {
558 		if (genericType instanceof ParameterizedType) {
559 			ParameterizedType parameterizedType = (ParameterizedType) genericType;
560 			if (JAXBElement.class.equals(parameterizedType.getRawType()) &&
561 					parameterizedType.getActualTypeArguments().length == 1) {
562 				Type typeArgument = parameterizedType.getActualTypeArguments()[0];
563 				if (typeArgument instanceof Class) {
564 					Class<?> classArgument = (Class<?>) typeArgument;
565 					if (JdkVersion.getMajorJavaVersion() >= JdkVersion.JAVA_17 && classArgument.isArray()) {
566 						return classArgument.getComponentType().equals(Byte.TYPE);
567 					}
568 					else {
569 						return (isPrimitiveWrapper(classArgument) || isStandardClass(classArgument) ||
570 								supportsInternal(classArgument, false));
571 					}
572 				}
573 				else if (JdkVersion.getMajorJavaVersion() <= JdkVersion.JAVA_16 &&
574 						typeArgument instanceof GenericArrayType) {
575 					// see http://bugs.sun.com/bugdatabase/view_bug.do?bug_id=5041784
576 					GenericArrayType arrayType = (GenericArrayType) typeArgument;
577 					return arrayType.getGenericComponentType().equals(Byte.TYPE);
578 				}
579 			}
580 		}
581 		else if (genericType instanceof Class) {
582 			Class<?> clazz = (Class<?>) genericType;
583 			return supportsInternal(clazz, this.checkForXmlRootElement);
584 		}
585 		return false;
586 	}
587 
588 	private boolean supportsInternal(Class<?> clazz, boolean checkForXmlRootElement) {
589 		if (checkForXmlRootElement && AnnotationUtils.findAnnotation(clazz, XmlRootElement.class) == null) {
590 			return false;
591 		}
592 		if (StringUtils.hasLength(this.contextPath)) {
593 			String packageName = ClassUtils.getPackageName(clazz);
594 			String[] contextPaths = StringUtils.tokenizeToStringArray(this.contextPath, ":");
595 			for (String contextPath : contextPaths) {
596 				if (contextPath.equals(packageName)) {
597 					return true;
598 				}
599 			}
600 			return false;
601 		}
602 		else if (!ObjectUtils.isEmpty(this.classesToBeBound)) {
603 			return Arrays.asList(this.classesToBeBound).contains(clazz);
604 		}
605 		return false;
606 	}

608 	/**
609 	 * Checks whether the given type is a primitive wrapper type.
610 	 * Compare section 8.5.1 of the JAXB2 spec.
611 	 */
612 	private boolean isPrimitiveWrapper(Class<?> clazz) {
613 		return Boolean.class.equals(clazz) ||
614 				Byte.class.equals(clazz) ||
615 				Short.class.equals(clazz) ||
616 				Integer.class.equals(clazz) ||
617 				Long.class.equals(clazz) ||
618 				Float.class.equals(clazz) ||
619 				Double.class.equals(clazz);
620 	}

622 	/**
623 	 * Checks whether the given type is a standard class.
624 	 * Compare section 8.5.2 of the JAXB2 spec.
625 	 */
626 	private boolean isStandardClass(Class<?> clazz) {
627 		return String.class.equals(clazz) ||
628 				BigInteger.class.isAssignableFrom(clazz) ||
629 				BigDecimal.class.isAssignableFrom(clazz) ||
630 				Calendar.class.isAssignableFrom(clazz) ||
631 				Date.class.isAssignableFrom(clazz) ||
632 				QName.class.isAssignableFrom(clazz) ||
633 				URI.class.equals(clazz) ||
634 				XMLGregorianCalendar.class.isAssignableFrom(clazz) ||
635 				Duration.class.isAssignableFrom(clazz) ||
636 				Image.class.equals(clazz) ||
637 				DataHandler.class.equals(clazz) ||
638 				// Source and subclasses should be supported according to the JAXB2 spec, but aren't in the RI
639 				// Source.class.isAssignableFrom(clazz) ||
640 				UUID.class.equals(clazz);
641

671 		XMLStreamWriter streamWriter = StaxUtils.getXMLStreamWriter(staxResult);

682 			}
683 		}

700 	/**
701 	 * Template method that can be overridden by concrete JAXB marshallers for custom initialization behavior.
702 	 * Gets called after creation of JAXB {@code Marshaller}, and after the respective properties have been set.
703 	 * <p>The default implementation sets the {@link #setMarshallerProperties(Map) defined properties}, the {@link
704 	 * #setValidationEventHandler(ValidationEventHandler) validation event handler}, the {@link #setSchemas(Resource[])
705 	 * schemas}, {@link #setMarshallerListener(javax.xml.bind.Marshaller.Listener) listener}, and
706 	 * {@link #setAdapters(XmlAdapter[]) adapters}.
707 	 */
708 	protected void initJaxbMarshaller(Marshaller marshaller) throws JAXBException {
709 		if (this.marshallerProperties != null) {
710 			for (String name : this.marshallerProperties.keySet()) {
711 				marshaller.setProperty(name, this.marshallerProperties.get(name));
712 			}
713 		}
714 		if (this.marshallerListener != null) {
715 			marshaller.setListener(this.marshallerListener);
716 		}
717 		if (this.validationEventHandler != null) {
718 			marshaller.setEventHandler(this.validationEventHandler);
719 		}
720 		if (this.adapters != null) {
721 			for (XmlAdapter<?, ?> adapter : this.adapters) {
722 				marshaller.setAdapter(adapter);
723 			}
724 		}
725 		if (this.schema != null) {
726 			marshaller.setSchema(this.schema);
727 		}
728 	}

734 	public Object unmarshal(Source source) throws XmlMappingException {
735 		return unmarshal(source, null);
736 	}

738 	@Override
739 	public Object unmarshal(Source source, MimeContainer mimeContainer) throws XmlMappingException {
740 		source = processSource(source);
741 
742 		try {
743 			Unmarshaller unmarshaller = createUnmarshaller();
744 			if (this.mtomEnabled && mimeContainer != null) {
745 				unmarshaller.setAttachmentUnmarshaller(new Jaxb2AttachmentUnmarshaller(mimeContainer));
746 			}
747 			if (StaxUtils.isStaxSource(source)) {
748 				return unmarshalStaxSource(unmarshaller, source);
749 			}
750 			else if (this.mappedClass != null) {
751 				return unmarshaller.unmarshal(source, this.mappedClass).getValue();
752 			}
753 			else {
754 				return unmarshaller.unmarshal(source);
755 			}
756 		}
757 		catch (JAXBException ex) {
758 			throw convertJaxbException(ex);
759 		}
760 	}
761 
762 	protected Object unmarshalStaxSource(Unmarshaller jaxbUnmarshaller, Source staxSource) throws JAXBException {
763 		XMLStreamReader streamReader = StaxUtils.getXMLStreamReader(staxSource);
764 		if (streamReader != null) {
765 			return (this.mappedClass != null ?
766 					jaxbUnmarshaller.unmarshal(streamReader, this.mappedClass).getValue() :
767 					jaxbUnmarshaller.unmarshal(streamReader));
768 		}
769 		else {
770 			XMLEventReader eventReader = StaxUtils.getXMLEventReader(staxSource);
771 			if (eventReader != null) {
772 				return (this.mappedClass != null ?
773 						jaxbUnmarshaller.unmarshal(eventReader, this.mappedClass).getValue() :
774 						jaxbUnmarshaller.unmarshal(eventReader));
775 			}
776 			else {
777 				throw new IllegalArgumentException("StaxSource contains neither XMLStreamReader nor XMLEventReader");
778 			}
779 		}
780 	}
781 
782 	private Source processSource(Source source) {
783 		if (StaxUtils.isStaxSource(source) || source instanceof DOMSource) {
784 			return source;
785 		}
786 
787 		XMLReader xmlReader = null;
788 		InputSource inputSource = null;
789 
790 		if (source instanceof SAXSource) {
791 			SAXSource saxSource = (SAXSource) source;
792 			xmlReader = saxSource.getXMLReader();
793 			inputSource = saxSource.getInputSource();
794 		}
795 		else if (source instanceof StreamSource) {
796 			StreamSource streamSource = (StreamSource) source;
797 			if (streamSource.getInputStream() != null) {
798 				inputSource = new InputSource(streamSource.getInputStream());
799 			}
800 			else if (streamSource.getReader() != null) {
801 				inputSource = new InputSource(streamSource.getReader());
802 			}
803 			else {
804 				inputSource = new InputSource(streamSource.getSystemId());
805 			}
806 		}
807 
808 		try {
809 			if (xmlReader == null) {
810 				xmlReader = XMLReaderFactory.createXMLReader();
811 			}
812 			String name = "http://xml.org/sax/features/external-general-entities";
813 			xmlReader.setFeature(name, isProcessExternalEntities());
814 			if (!isProcessExternalEntities()) {
815 				xmlReader.setEntityResolver(NO_OP_ENTITY_RESOLVER);
816 			}
817 			return new SAXSource(xmlReader, inputSource);
818 		}
819 		catch (SAXException ex) {
820 			logger.warn("Processing of external entities could not be disabled", ex);
821 			return source;
822 		}
823 	}

870 	/**
871 	 * Convert the given {@code JAXBException} to an appropriate exception from the
872 	 * {@code org.springframework.oxm} hierarchy.
873 	 * @param ex {@code JAXBException} that occured
874 	 * @return the corresponding {@code XmlMappingException}
875 	 */
876 	protected XmlMappingException convertJaxbException(JAXBException ex) {
877 		if (ex instanceof ValidationException) {
878 			return new ValidationFailureException("JAXB validation exception", ex);
879 		}
880 		else if (ex instanceof MarshalException) {
881 			return new MarshallingFailureException("JAXB marshalling exception", ex);
882 		}
883 		else if (ex instanceof UnmarshalException) {
884 			return new UnmarshallingFailureException("JAXB unmarshalling exception", ex);
885 		}
886 		else {
887 			// fallback
888 			return new UncategorizedMappingException("Unknown JAXB exception", ex);
889 		}
890 	}

903 				String elementNamespace, String elementLocalName) {

905 			return addMtomAttachment(new DataHandler(dataSource), elementNamespace, elementLocalName);

908 		@Override
909 		public String addMtomAttachment(DataHandler dataHandler, String elementNamespace, String elementLocalName) {
910 			String host = getHost(elementNamespace, dataHandler);
911 			String contentId = UUID.randomUUID() + "@" + host;
912 			this.mimeContainer.addAttachment("<" + contentId + ">", dataHandler);
913 			try {
914 				contentId = URLEncoder.encode(contentId, "UTF-8");
915 			}
916 			catch (UnsupportedEncodingException e) {
917 				// ignore
918 			}
919 			return CID + contentId;
920 		}

923 			try {
924 				URI uri = new URI(elementNamespace);
925 				return uri.getHost();
926 			}
927 			catch (URISyntaxException e) {
928 				// ignore
929 			}
930 			return dataHandler.getName();
931 		}
932 
933 		@Override
934 		public String addSwaRefAttachment(DataHandler dataHandler) {
935 			String contentId = UUID.randomUUID() + "@" + dataHandler.getName();
936 			this.mimeContainer.addAttachment(contentId, dataHandler);
937 			return contentId;
938 		}

941 		public boolean isXOPPackage() {
942 			return this.mimeContainer.convertToXopPackage();
