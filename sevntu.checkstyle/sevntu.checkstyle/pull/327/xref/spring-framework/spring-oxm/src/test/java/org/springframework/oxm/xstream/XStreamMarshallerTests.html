84  	@Test
85  	public void marshalDOMResult() throws Exception {
86  		DocumentBuilderFactory documentBuilderFactory = DocumentBuilderFactory.newInstance();
87  		DocumentBuilder builder = documentBuilderFactory.newDocumentBuilder();
88  		Document document = builder.newDocument();
89  		DOMResult domResult = new DOMResult(document);
90  		marshaller.marshal(flight, domResult);
91  		Document expected = builder.newDocument();
92  		Element flightElement = expected.createElement("flight");
93  		expected.appendChild(flightElement);
94  		Element numberElement = expected.createElement("flightNumber");
95  		flightElement.appendChild(numberElement);
96  		Text text = expected.createTextNode("42");
97  		numberElement.appendChild(text);
98  		assertXMLEqual("Marshaller writes invalid DOMResult", expected, document);
99  	}

107 		Element rootElement = existent.createElement("root");
108 		Element flightsElement = existent.createElement("flights");
109 		rootElement.appendChild(flightsElement);
110 		existent.appendChild(rootElement);

114 		marshaller.marshal(flight, domResult);

122 		expected.appendChild(eRootElement);
123 		Element eNumberElement = expected.createElement("flightNumber");
124 		eFlightElement.appendChild(eNumberElement);
125 		Text text = expected.createTextNode("42");
126 		eNumberElement.appendChild(text);
127 		assertXMLEqual("Marshaller writes invalid DOMResult", expected, existent);

149 		ContentHandler contentHandler = mock(ContentHandler.class);
150 		SAXResult result = new SAXResult(contentHandler);
151 		marshaller.marshal(flight, result);
152 		InOrder ordered = inOrder(contentHandler);
153 		ordered.verify(contentHandler).startDocument();
154 		ordered.verify(contentHandler).startElement(eq(""), eq("flight"), eq("flight"), isA(Attributes.class));
155 		ordered.verify(contentHandler).startElement(eq(""), eq("flightNumber"), eq("flightNumber"), isA(Attributes.class));
156 		ordered.verify(contentHandler).characters(isA(char[].class), eq(0), eq(2));
157 		ordered.verify(contentHandler).endElement("", "flightNumber", "flightNumber");
158 		ordered.verify(contentHandler).endElement("", "flight", "flight");
159 		ordered.verify(contentHandler).endDocument();

281 		Flights flights = new Flights();
282 		flights.getFlights().add(flight);
283 		flights.getStrings().add("42");
284 
285 		Map<String, Class<?>> aliases = new HashMap<String, Class<?>>();
286 		aliases.put("flight", Flight.class);
287 		aliases.put("flights", Flights.class);
288 		marshaller.setAliases(aliases);
289 
290 		Map implicitCollections = Collections.singletonMap(Flights.class, "flights,strings");
291 		marshaller.setImplicitCollections(implicitCollections);
292 
293 		Writer writer = new StringWriter();
294 		marshaller.marshal(flights, new StreamResult(writer));
295 		String result = writer.toString();
296 		assertXpathNotExists("/flights/flights", result);
297 		assertXpathExists("/flights/flight", result);
298 		assertXpathNotExists("/flights/strings", result);
299 		assertXpathExists("/flights/string", result);
300 	}

302 	@Test
303 	public void jettisonDriver() throws Exception {
304 		marshaller.setStreamDriver(new JettisonMappedXmlDriver());
305 		Writer writer = new StringWriter();
306 		marshaller.marshal(flight, new StreamResult(writer));
307 		assertEquals("Invalid result", "{\"flight\":{\"flightNumber\":42}}", writer.toString());
308 		Object o = marshaller.unmarshal(new StreamSource(new StringReader(writer.toString())));
309 		assertTrue("Unmarshalled object is not Flights", o instanceof Flight);
310 		Flight unflight = (Flight) o;
311 		assertNotNull("Flight is null", unflight);
312 		assertEquals("Number is invalid", 42L, unflight.getFlightNumber());
313 	}
