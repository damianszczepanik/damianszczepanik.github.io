206 	@Override
207 	public void afterPropertiesSet() throws JiBXException {
208 		if (this.targetClass != null) {
209 			if (StringUtils.hasLength(this.bindingName)) {
210 				if (logger.isInfoEnabled()) {
211 					logger.info("Configured for target class [" + this.targetClass + "] using binding [" + this.bindingName + "]");
212 				}
213 				this.bindingFactory = BindingDirectory.getFactory(this.bindingName, this.targetClass);
214 			}
215 			else {
216 				if (logger.isInfoEnabled()) {
217 					logger.info("Configured for target class [" + this.targetClass + "]");
218 				}
219 				this.bindingFactory = BindingDirectory.getFactory(this.targetClass);
220 			}
221 		}
222 		else if (this.targetPackage != null) {
223 			if (!StringUtils.hasLength(bindingName)) {
224 				bindingName = DEFAULT_BINDING_NAME;
225 			}
226 			if (logger.isInfoEnabled()) {
227 				logger.info("Configured for target package [" + this.targetPackage	+ "] using binding [" + this.bindingName + "]");
228 			}
229 			this.bindingFactory = BindingDirectory.getFactory(this.bindingName, this.targetPackage);
230 		}
231 		else {
232 			throw new IllegalArgumentException("Either 'targetClass' or 'targetPackage' is required");
233 		}
234 	}

237 	@Override
238 	public boolean supports(Class<?> clazz) {
239 		Assert.notNull(clazz, "'clazz' must not be null");
240 		if (this.targetClass != null) {
241 			return this.targetClass.equals(clazz);
242 		}
243 		String[] mappedClasses = this.bindingFactory.getMappedClasses();
244 		String className = clazz.getName();
245 		for (String mappedClass : mappedClasses) {
246 			if (className.equals(mappedClass)) {
247 				return true;
248 			}
249 		}
250 		return false;
251 	}

256 	@Override
257 	protected void marshalOutputStream(Object graph, OutputStream outputStream)
258 			throws XmlMappingException, IOException {
259 		try {
260 			IMarshallingContext marshallingContext = createMarshallingContext();
261 			marshallingContext.startDocument(this.encoding, this.standalone, outputStream);
262 			marshalDocument(marshallingContext, graph);
263 		}
264 		catch (JiBXException ex) {
265 			throw convertJibxException(ex, true);
266 		}
267 	}

270 	protected void marshalWriter(Object graph, Writer writer) throws XmlMappingException, IOException {
271 		try {
272 			IMarshallingContext marshallingContext = createMarshallingContext();
273 			marshallingContext.startDocument(this.encoding, this.standalone, writer);
274 			marshalDocument(marshallingContext, graph);
275 		}
276 		catch (JiBXException ex) {
277 			throw convertJibxException(ex, true);
278 		}
279 	}

281 	private void marshalDocument(IMarshallingContext marshallingContext, Object graph) throws IOException, JiBXException {
282 		if (StringUtils.hasLength(docTypeRootElementName)) {
283 			IXMLWriter xmlWriter = marshallingContext.getXmlWriter();
284 			xmlWriter.writeDocType(docTypeRootElementName, docTypeSystemId, docTypePublicId, docTypeInternalSubset);
285 		}
286 		marshallingContext.marshalDocument(graph);
287 	}

292 	@Override
293 	protected void marshalDomNode(Object graph, Node node) throws XmlMappingException {
294 		try {
295 			// JiBX does not support DOM natively, so we write to a buffer first, and transform that to the Node
296 			Result result = new DOMResult(node);
297 			transformAndMarshal(graph, result);
298 		}
299 		catch (IOException ex) {
300 			throw new MarshallingFailureException("JiBX marshalling exception", ex);
301 		}
302 	}

305 	protected void marshalXmlEventWriter(Object graph, XMLEventWriter eventWriter) {
306 		XMLStreamWriter streamWriter = StaxUtils.createEventStreamWriter(eventWriter);
307 		marshalXmlStreamWriter(graph, streamWriter);
308 	}

310 	@Override
311 	protected void marshalXmlStreamWriter(Object graph, XMLStreamWriter streamWriter) throws XmlMappingException {
312 		try {
313 			MarshallingContext marshallingContext = (MarshallingContext) createMarshallingContext();
314 			IXMLWriter xmlWriter = new StAXWriter(marshallingContext.getNamespaces(), streamWriter);
315 			marshallingContext.setXmlWriter(xmlWriter);
316 			marshallingContext.marshalDocument(graph);
317 		}
318 		catch (JiBXException ex) {
319 			throw convertJibxException(ex, false);
320 		}
321 	}
322 
323 	@Override
324 	protected void marshalSaxHandlers(Object graph, ContentHandler contentHandler, LexicalHandler lexicalHandler)
325 			throws XmlMappingException {
326 		try {
327 			// JiBX does not support SAX natively, so we write to a buffer first, and transform that to the handlers
328 			SAXResult saxResult = new SAXResult(contentHandler);
329 			saxResult.setLexicalHandler(lexicalHandler);
330 			transformAndMarshal(graph, saxResult);
331 		}
332 		catch (IOException ex) {
333 			throw new MarshallingFailureException("JiBX marshalling exception", ex);
334 		}
335 	}
336 
337 	private void transformAndMarshal(Object graph, Result result) throws IOException {
338 		try {
339 			ByteArrayOutputStream os = new ByteArrayOutputStream(1024);
340 			marshalOutputStream(graph, os);
341 			ByteArrayInputStream is = new ByteArrayInputStream(os.toByteArray());
342 			Transformer transformer = this.transformerFactory.newTransformer();
343 			transformer.transform(new StreamSource(is), result);
344 		}
345 		catch (TransformerException ex) {
346 			throw new MarshallingFailureException(
347 					"Could not transform to [" + ClassUtils.getShortName(result.getClass()) + "]", ex);
348 		}
349 
350 	}

414 	@Override
415 	protected Object unmarshalSaxReader(XMLReader xmlReader, InputSource inputSource)
416 			throws XmlMappingException, IOException {
417 
418 		return transformAndUnmarshal(new SAXSource(xmlReader, inputSource), inputSource.getEncoding());
419 	}
420 
421 	private Object transformAndUnmarshal(Source source, String encoding) throws IOException {
422 		try {
423 			Transformer transformer = this.transformerFactory.newTransformer();
424 			if (encoding != null) {
425 				transformer.setOutputProperty(OutputKeys.ENCODING, encoding);
426 			}
427 			ByteArrayOutputStream os = new ByteArrayOutputStream(1024);
428 			transformer.transform(source, new StreamResult(os));
429 			ByteArrayInputStream is = new ByteArrayInputStream(os.toByteArray());
430 			return unmarshalInputStream(is);
431 		}
432 		catch (TransformerException ex) {
433 			throw new MarshallingFailureException(
434 					"Could not transform from [" + ClassUtils.getShortName(source.getClass()) + "]", ex);
435 		}
436 	}

459 	/**
460 	 * Convert the given {@code JiBXException} to an appropriate exception from the
461 	 * {@code org.springframework.oxm} hierarchy.
462 	 * <p>A boolean flag is used to indicate whether this exception occurs during marshalling or
463 	 * unmarshalling, since JiBX itself does not make this distinction in its exception hierarchy.
464 	 * @param ex {@code JiBXException} that occured
465 	 * @param marshalling indicates whether the exception occurs during marshalling ({@code true}),
466 	 * or unmarshalling ({@code false})
467 	 * @return the corresponding {@code XmlMappingException}
468 	 */
469 	public XmlMappingException convertJibxException(JiBXException ex, boolean marshalling) {
470 		if (ex instanceof ValidationException) {
471 			return new ValidationFailureException("JiBX validation exception", ex);
472 		}
473 		else {
474 			if (marshalling) {
475 				return new MarshallingFailureException("JiBX marshalling exception", ex);
476 			}
477 			else {
478 				return new UnmarshallingFailureException("JiBX unmarshalling exception", ex);
479 			}
480 		}
