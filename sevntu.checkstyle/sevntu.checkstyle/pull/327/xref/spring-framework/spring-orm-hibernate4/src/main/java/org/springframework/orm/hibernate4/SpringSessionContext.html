17  package org.springframework.orm.hibernate4;
18  
19  import java.lang.reflect.Method;
20  import javax.transaction.Status;
21  import javax.transaction.SystemException;
22  import javax.transaction.TransactionManager;
23  
24  import org.apache.commons.logging.LogFactory;
25  import org.hibernate.FlushMode;
26  import org.hibernate.HibernateException;
27  import org.hibernate.Session;
28  import org.hibernate.context.spi.CurrentSessionContext;
29  import org.hibernate.engine.spi.SessionFactoryImplementor;
30  
31  import org.springframework.transaction.support.TransactionSynchronizationManager;
32  import org.springframework.util.ReflectionUtils;
33  
34  /**
35   * Implementation of Hibernate 3.1's CurrentSessionContext interface
36   * that delegates to Spring's SessionFactoryUtils for providing a
37   * Spring-managed current Session.
38   *
39   * <p>This CurrentSessionContext implementation can also be specified in custom
40   * SessionFactory setup through the "hibernate.current_session_context_class"
41   * property, with the fully qualified name of this class as value.
42   *
43   * @author Juergen Hoeller
44   * @since 3.1
45   */
46  @SuppressWarnings("serial")
47  public class SpringSessionContext implements CurrentSessionContext {
48  
49  	private final SessionFactoryImplementor sessionFactory;
50  
51  	private TransactionManager transactionManager;
52  
53  	private CurrentSessionContext jtaSessionContext;
54  
55  
56  	/**
57  	 * Create a new SpringSessionContext for the given Hibernate SessionFactory.
58  	 * @param sessionFactory the SessionFactory to provide current Sessions for
59  	 */
60  	public SpringSessionContext(SessionFactoryImplementor sessionFactory) {
61  		this.sessionFactory = sessionFactory;
62  		try {
63  			Object jtaPlatform = sessionFactory.getServiceRegistry().getService(ConfigurableJtaPlatform.jtaPlatformClass);
64  			Method rtmMethod = ConfigurableJtaPlatform.jtaPlatformClass.getMethod("retrieveTransactionManager");
65  			this.transactionManager = (TransactionManager) ReflectionUtils.invokeMethod(rtmMethod, jtaPlatform);
66  			if (this.transactionManager != null) {
67  				this.jtaSessionContext = new SpringJtaSessionContext(sessionFactory);
68  			}
69  		}
70  		catch (Exception ex) {
71  			LogFactory.getLog(SpringSessionContext.class).warn(
72  					"Could not introspect Hibernate JtaPlatform for SpringJtaSessionContext", ex);
73  		}
74  	}
75  
76  
77  	/**
78  	 * Retrieve the Spring-managed Session for the current thread, if any.
79  	 */
80  	@Override
81  	public Session currentSession() throws HibernateException {
82  		Object value = TransactionSynchronizationManager.getResource(this.sessionFactory);
83  		if (value instanceof Session) {
84  			return (Session) value;
85  		}
86  		else if (value instanceof SessionHolder) {
87  			SessionHolder sessionHolder = (SessionHolder) value;
88  			Session session = sessionHolder.getSession();
89  			if (!sessionHolder.isSynchronizedWithTransaction() &&
90  					TransactionSynchronizationManager.isSynchronizationActive()) {
91  				TransactionSynchronizationManager.registerSynchronization(
92  						new SpringSessionSynchronization(sessionHolder, this.sessionFactory, false));
93  				sessionHolder.setSynchronizedWithTransaction(true);
94  				// Switch to FlushMode.AUTO, as we have to assume a thread-bound Session
95  				// with FlushMode.MANUAL, which needs to allow flushing within the transaction.
96  				FlushMode flushMode = session.getFlushMode();
97  				if (flushMode.equals(FlushMode.MANUAL) &&
98  						!TransactionSynchronizationManager.isCurrentTransactionReadOnly()) {
99  					session.setFlushMode(FlushMode.AUTO);
100 					sessionHolder.setPreviousFlushMode(flushMode);
101 				}
102 			}
103 			return session;
104 		}
105 
106 		if (this.transactionManager != null) {
107 			try {
108 				if (this.transactionManager.getStatus() == Status.STATUS_ACTIVE) {
109 					Session session = this.jtaSessionContext.currentSession();
110 					if (TransactionSynchronizationManager.isSynchronizationActive()) {
111 						TransactionSynchronizationManager.registerSynchronization(new SpringFlushSynchronization(session));
112 					}
113 					return session;
114 				}
115 			}
116 			catch (SystemException ex) {
117 				throw new HibernateException("JTA TransactionManager found but status check failed", ex);
118 			}
119 		}
120 
121 		if (TransactionSynchronizationManager.isSynchronizationActive()) {
122 			Session session = this.sessionFactory.openSession();
123 			if (TransactionSynchronizationManager.isCurrentTransactionReadOnly()) {
124 				session.setFlushMode(FlushMode.MANUAL);
125 			}
126 			SessionHolder sessionHolder = new SessionHolder(session);
127 			TransactionSynchronizationManager.registerSynchronization(
128 					new SpringSessionSynchronization(sessionHolder, this.sessionFactory, true));
129 			TransactionSynchronizationManager.bindResource(this.sessionFactory, sessionHolder);
130 			sessionHolder.setSynchronizedWithTransaction(true);
131 			return session;
132 		}
133 		else {
134 			throw new HibernateException("Could not obtain transaction-synchronized Session for current thread");
135 		}
136 	}
137 
138 }
