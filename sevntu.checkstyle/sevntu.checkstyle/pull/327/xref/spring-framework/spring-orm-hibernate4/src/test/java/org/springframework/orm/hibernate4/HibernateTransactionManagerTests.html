85  		final DataSource ds = mock(DataSource.class);
86  		Connection con = mock(Connection.class);
87  		final SessionFactory sf = mock(SessionFactory.class);
88  		final ImplementingSession session = mock(ImplementingSession.class);
89  		Transaction tx = mock(Transaction.class);
90  		Query query = mock(Query.class);
91  
92  		final List list = new ArrayList();
93  		list.add("test");
94  		given(con.getTransactionIsolation()).willReturn(Connection.TRANSACTION_READ_COMMITTED);
95  		given(sf.openSession()).willReturn(session);
96  		given(session.getTransaction()).willReturn(tx);
97  		given(session.connection()).willReturn(con);
98  		given(session.isOpen()).willReturn(true);
99  		given(session.createQuery("some query string")).willReturn(query);
100 		given(query.list()).willReturn(list);
101 		given(session.isConnected()).willReturn(true);
102 
103 		LocalSessionFactoryBean lsfb = new LocalSessionFactoryBean() {

107 			}

109 		lsfb.afterPropertiesSet();
110 		final SessionFactory sfProxy = lsfb.getObject();
111 
112 		HibernateTransactionManager tm = new HibernateTransactionManager();

127 				assertFalse(TransactionSynchronizationManager.isCurrentTransactionReadOnly());
128 				assertTrue(TransactionSynchronizationManager.isActualTransactionActive());

134

349 		final SessionFactory sf = mock(SessionFactory.class);
350 		ImplementingSession session1 = mock(ImplementingSession.class);
351 		ImplementingSession session2 = mock(ImplementingSession.class);
352 		Connection con = mock(Connection.class);
353 		Transaction tx = mock(Transaction.class);
354 
355 		given(sf.openSession()).willReturn(session1, session2);
356 		given(session1.beginTransaction()).willReturn(tx);
357 		given(session1.isOpen()).willReturn(true);
358 		given(session2.beginTransaction()).willReturn(tx);
359 		given(session2.isOpen()).willReturn(true);
360 		given(session2.getFlushMode()).willReturn(FlushMode.AUTO);
361 		given(session1.isConnected()).willReturn(true);
362 		given(session1.connection()).willReturn(con);
363 		given(session2.isConnected()).willReturn(true);
364 		given(session2.connection()).willReturn(con);
365 
366 		PlatformTransactionManager tm = new HibernateTransactionManager(sf);
367 		final TransactionTemplate tt = new TransactionTemplate(tm);
368 		tt.setPropagationBehavior(TransactionDefinition.PROPAGATION_REQUIRES_NEW);
369 
370 		assertTrue("Hasn't thread session", !TransactionSynchronizationManager.hasResource(sf));
371 		tt.execute(new TransactionCallback() {
372 			@Override
373 			public Object doInTransaction(TransactionStatus status) {
374 				final SessionHolder holder = (SessionHolder) TransactionSynchronizationManager.getResource(sf);
375 				assertTrue("Has thread session", holder != null);
376 				assertFalse(TransactionSynchronizationManager.isCurrentTransactionReadOnly());
377 				assertTrue(TransactionSynchronizationManager.isActualTransactionActive());
378 				tt.execute(new TransactionCallback() {
379 					@Override
380 					public Object doInTransaction(TransactionStatus status) {
381 						Session session = ((SessionHolder) TransactionSynchronizationManager.getResource(sf)).getSession();
382 						assertTrue("Not enclosing session", session != holder.getSession());
383 						session.flush();
384 						assertFalse(TransactionSynchronizationManager.isCurrentTransactionReadOnly());
385 						assertTrue(TransactionSynchronizationManager.isActualTransactionActive());
386 						return null;
387 					}
388 				});
389 				assertTrue("Same thread session as before",
390 						holder.getSession() == ((SessionHolder) TransactionSynchronizationManager.getResource(sf)).getSession());
391 				assertFalse(TransactionSynchronizationManager.isCurrentTransactionReadOnly());
392 				assertTrue(TransactionSynchronizationManager.isActualTransactionActive());
393 				return null;
394 			}
395 		});
396 		assertTrue("Hasn't thread session", !TransactionSynchronizationManager.hasResource(sf));
397 
398 		verify(session2).flush();
399 		verify(session1).close();
400 		verify(session2).close();
401 		verify(tx, times(2)).commit();
402 	}
