17  package org.springframework.web.servlet.mvc;
18  
19  import javax.servlet.http.HttpServletRequest;
20  import javax.servlet.http.HttpServletResponse;
21  
22  import org.springframework.util.Assert;
23  import org.springframework.web.servlet.ModelAndView;
24  import org.springframework.web.servlet.support.RequestContextUtils;
25  import org.springframework.web.util.UrlPathHelper;
26  
27  /**
28   * Abstract base class for {@code Controllers} that return a view name
29   * based on the request URL.
30   *
31   * <p>Provides infrastructure for determining view names from URLs and configurable
32   * URL lookup. For information on the latter, see {@code alwaysUseFullPath}
33   * and {@code urlDecode} properties.
34   *
35   * @author Juergen Hoeller
36   * @since 1.2.6
37   * @see #setAlwaysUseFullPath
38   * @see #setUrlDecode
39   */
40  public abstract class AbstractUrlViewController extends AbstractController {
41  
42  	private UrlPathHelper urlPathHelper = new UrlPathHelper();
43  
44  
45  	/**
46  	 * Set if URL lookup should always use full path within current servlet
47  	 * context. Else, the path within the current servlet mapping is used
48  	 * if applicable (i.e. in the case of a ".../*" servlet mapping in web.xml).
49  	 * Default is "false".
50  	 * @see org.springframework.web.util.UrlPathHelper#setAlwaysUseFullPath
51  	 */
52  	public void setAlwaysUseFullPath(boolean alwaysUseFullPath) {
53  		this.urlPathHelper.setAlwaysUseFullPath(alwaysUseFullPath);
54  	}
55  
56  	/**
57  	 * Set if context path and request URI should be URL-decoded.
58  	 * Both are returned <i>undecoded</i> by the Servlet API,
59  	 * in contrast to the servlet path.
60  	 * <p>Uses either the request encoding or the default encoding according
61  	 * to the Servlet spec (ISO-8859-1).
62  	 * @see org.springframework.web.util.UrlPathHelper#setUrlDecode
63  	 */
64  	public void setUrlDecode(boolean urlDecode) {
65  		this.urlPathHelper.setUrlDecode(urlDecode);
66  	}
67  
68  	/**
69  	 * Set if ";" (semicolon) content should be stripped from the request URI.
70  	 * @see org.springframework.web.util.UrlPathHelper#setRemoveSemicolonContent(boolean)
71  	 */
72  	public void setRemoveSemicolonContent(boolean removeSemicolonContent) {
73  		this.urlPathHelper.setRemoveSemicolonContent(removeSemicolonContent);
74  	}
75  
76  	/**
77  	 * Set the UrlPathHelper to use for the resolution of lookup paths.
78  	 * <p>Use this to override the default UrlPathHelper with a custom subclass,
79  	 * or to share common UrlPathHelper settings across multiple MethodNameResolvers
80  	 * and HandlerMappings.
81  	 * @see org.springframework.web.servlet.handler.AbstractUrlHandlerMapping#setUrlPathHelper
82  	 */
83  	public void setUrlPathHelper(UrlPathHelper urlPathHelper) {
84  		Assert.notNull(urlPathHelper, "UrlPathHelper must not be null");
85  		this.urlPathHelper = urlPathHelper;
86  	}
87  
88  	/**
89  	 * Return the UrlPathHelper to use for the resolution of lookup paths.
90  	 */
91  	protected UrlPathHelper getUrlPathHelper() {
92  		return this.urlPathHelper;
93  	}
94  
95  
96  	/**
97  	 * Retrieves the URL path to use for lookup and delegates to
98  	 * {@link #getViewNameForRequest}. Also adds the content of
99  	 * {@link RequestContextUtils#getInputFlashMap} to the model.
100 	 */
101 	@Override
102 	protected ModelAndView handleRequestInternal(HttpServletRequest request, HttpServletResponse response) {
103 		String lookupPath = getUrlPathHelper().getLookupPathForRequest(request);
104 		String viewName = getViewNameForRequest(request);
105 		if (logger.isDebugEnabled()) {
106 			logger.debug("Returning view name '" + viewName + "' for lookup path [" + lookupPath + "]");
107 		}
108 		return new ModelAndView(viewName, RequestContextUtils.getInputFlashMap(request));
109 	}
110 
111 	/**
112 	 * Return the name of the view to render for this request, based on the
113 	 * given lookup path. Called by {@link #handleRequestInternal}.
114 	 * @param request current HTTP request
115 	 * @return a view name for this request (never {@code null})
116 	 * @see #handleRequestInternal
117 	 * @see #setAlwaysUseFullPath
118 	 * @see #setUrlDecode
119 	 */
120 	protected abstract String getViewNameForRequest(HttpServletRequest request);
121 
122 }
