97  /**
98   * This is the main class providing the configuration behind the MVC Java config.
99   * It is typically imported by adding {@link EnableWebMvc @EnableWebMvc} to an
100  * application {@link Configuration @Configuration} class. An alternative more
101  * advanced option is to extend directly from this class and override methods as
102  * necessary remembering to add {@link Configuration @Configuration} to the
103  * subclass and {@link Bean @Bean} to overridden {@link Bean @Bean} methods.
104  * For more details see the Javadoc of {@link EnableWebMvc @EnableWebMvc}.
105  *
106  * <p>This class registers the following {@link HandlerMapping}s:</p>
107  * <ul>
108  * <li>{@link RequestMappingHandlerMapping}
109  * ordered at 0 for mapping requests to annotated controller methods.
110  * <li>{@link HandlerMapping}
111  * ordered at 1 to map URL paths directly to view names.
112  * <li>{@link BeanNameUrlHandlerMapping}
113  * ordered at 2 to map URL paths to controller bean names.
114  * <li>{@link HandlerMapping}
115  * ordered at {@code Integer.MAX_VALUE-1} to serve static resource requests.
116  * <li>{@link HandlerMapping}
117  * ordered at {@code Integer.MAX_VALUE} to forward requests to the default servlet.
118  * </ul>
119  *
120  * <p>Registers these {@link HandlerAdapter}s:
121  * <ul>
122  * <li>{@link RequestMappingHandlerAdapter}
123  * for processing requests with annotated controller methods.
124  * <li>{@link HttpRequestHandlerAdapter}
125  * for processing requests with {@link HttpRequestHandler}s.
126  * <li>{@link SimpleControllerHandlerAdapter}
127  * for processing requests with interface-based {@link Controller}s.
128  * </ul>
129  *
130  * <p>Registers a {@link HandlerExceptionResolverComposite} with this chain of
131  * exception resolvers:
132  * <ul>
133  * <li>{@link ExceptionHandlerExceptionResolver} for handling exceptions
134  * through @{@link ExceptionHandler} methods.
135  * <li>{@link ResponseStatusExceptionResolver} for exceptions annotated
136  * with @{@link ResponseStatus}.
137  * <li>{@link DefaultHandlerExceptionResolver} for resolving known Spring
138  * exception types
139  * </ul>
140  *
141  * <p>Registers an {@link AntPathMatcher} and a {@link UrlPathHelper}
142  * to be used by:
143  * <ul>
144  * <li>the {@link RequestMappingHandlerMapping},
145  * <li>the {@link HandlerMapping} for ViewControllers
146  * <li>and the {@link HandlerMapping} for serving resources
147  * </ul>
148  * Note that those beans can be configured with a {@link PathMatchConfigurer}.
149  *
150  * <p>Both the {@link RequestMappingHandlerAdapter} and the
151  * {@link ExceptionHandlerExceptionResolver} are configured with default
152  * instances of the following by default:
153  * <ul>
154  * <li>a {@link ContentNegotiationManager}
155  * <li>a {@link DefaultFormattingConversionService}
156  * <li>a {@link org.springframework.validation.beanvalidation.OptionalValidatorFactoryBean}
157  * if a JSR-303 implementation is available on the classpath
158  * <li>a range of {@link HttpMessageConverter}s depending on the third-party
159  * libraries available on the classpath.
160  * </ul>
161  *
162  * @author Rossen Stoyanchev
163  * @author Brian Clozel
164  * @author Sebastien Deleuze
165  * @since 3.1
166  * @see EnableWebMvc
167  * @see WebMvcConfigurer
168  * @see WebMvcConfigurerAdapter
169  */

228 	 * Return a {@link RequestMappingHandlerMapping} ordered at 0 for mapping
229 	 * requests to annotated controllers.
230 	 */
231 	@Bean
232 	public RequestMappingHandlerMapping requestMappingHandlerMapping() {
233 		RequestMappingHandlerMapping handlerMapping = createRequestMappingHandlerMapping();
234 		handlerMapping.setOrder(0);
235 		handlerMapping.setInterceptors(getInterceptors());
236 		handlerMapping.setContentNegotiationManager(mvcContentNegotiationManager());
237 
238 		PathMatchConfigurer configurer = getPathMatchConfigurer();
239 		if (configurer.isUseSuffixPatternMatch() != null) {
240 			handlerMapping.setUseSuffixPatternMatch(configurer.isUseSuffixPatternMatch());
241 		}
242 		if (configurer.isUseRegisteredSuffixPatternMatch() != null) {
243 			handlerMapping.setUseRegisteredSuffixPatternMatch(configurer.isUseRegisteredSuffixPatternMatch());
244 		}
245 		if (configurer.isUseTrailingSlashMatch() != null) {
246 			handlerMapping.setUseTrailingSlashMatch(configurer.isUseTrailingSlashMatch());
247 		}
248 		if (configurer.getPathMatcher() != null) {
249 			handlerMapping.setPathMatcher(configurer.getPathMatcher());
250 		}
251 		if (configurer.getUrlPathHelper() != null) {
252 			handlerMapping.setUrlPathHelper(configurer.getUrlPathHelper());
253 		}
254 
255 		return handlerMapping;
256 	}

267 	 * Provide access to the shared handler interceptors used to configure
268 	 * {@link HandlerMapping} instances with. This method cannot be overridden,
269 	 * use {@link #addInterceptors(InterceptorRegistry)} instead.
270 	 */
271 	protected final Object[] getInterceptors() {
272 		if (this.interceptors == null) {
273 			InterceptorRegistry registry = new InterceptorRegistry();
274 			addInterceptors(registry);
275 			registry.addInterceptor(new ConversionServiceExposingInterceptor(mvcConversionService()));
276 			registry.addInterceptor(new ResourceUrlProviderExposingInterceptor(mvcResourceUrlProvider()));
277 			this.interceptors = registry.getInterceptors();
278 		}
279 		return this.interceptors.toArray();
280 	}

313 	 * requested {@linkplain MediaType media types} in a given request.
314 	 */
315 	@Bean
316 	public ContentNegotiationManager mvcContentNegotiationManager() {
317 		if (this.contentNegotiationManager == null) {
318 			ContentNegotiationConfigurer configurer = new ContentNegotiationConfigurer(this.servletContext);
319 			configurer.mediaTypes(getDefaultMediaTypes());
320 			configureContentNegotiation(configurer);
321 			try {
322 				this.contentNegotiationManager = configurer.getContentNegotiationManager();
323 			}
324 			catch (Exception ex) {
325 				throw new BeanInitializationException("Could not create ContentNegotiationManager", ex);
326 			}
327 		}
328 		return this.contentNegotiationManager;
329 	}

331 	protected Map<String, MediaType> getDefaultMediaTypes() {
332 		Map<String, MediaType> map = new HashMap<String, MediaType>();
333 		if (romePresent) {
334 			map.put("atom", MediaType.APPLICATION_ATOM_XML);
335 			map.put("rss", MediaType.valueOf("application/rss+xml"));
336 		}
337 		if (jaxb2Present || jackson2XmlPresent) {
338 			map.put("xml", MediaType.APPLICATION_XML);
339 		}
340 		if (jackson2Present || gsonPresent) {
341 			map.put("json", MediaType.APPLICATION_JSON);
342 		}
343 		return map;
344 	}

455 	/**
456 	 * Returns a {@link RequestMappingHandlerAdapter} for processing requests
457 	 * through annotated controller methods. Consider overriding one of these
458 	 * other more fine-grained methods:
459 	 * <ul>
460 	 * <li>{@link #addArgumentResolvers} for adding custom argument resolvers.
461 	 * <li>{@link #addReturnValueHandlers} for adding custom return value handlers.
462 	 * <li>{@link #configureMessageConverters} for adding custom message converters.
463 	 * </ul>
464 	 */
465 	@Bean
466 	public RequestMappingHandlerAdapter requestMappingHandlerAdapter() {
467 		List<HandlerMethodArgumentResolver> argumentResolvers = new ArrayList<HandlerMethodArgumentResolver>();
468 		addArgumentResolvers(argumentResolvers);
469 
470 		List<HandlerMethodReturnValueHandler> returnValueHandlers = new ArrayList<HandlerMethodReturnValueHandler>();
471 		addReturnValueHandlers(returnValueHandlers);
472 
473 		RequestMappingHandlerAdapter adapter = new RequestMappingHandlerAdapter();
474 		adapter.setContentNegotiationManager(mvcContentNegotiationManager());
475 		adapter.setMessageConverters(getMessageConverters());
476 		adapter.setWebBindingInitializer(getConfigurableWebBindingInitializer());
477 		adapter.setCustomArgumentResolvers(argumentResolvers);
478 		adapter.setCustomReturnValueHandlers(returnValueHandlers);
479 
480 		if (jackson2Present) {
481 			List<ResponseBodyAdvice<?>> interceptors = new ArrayList<ResponseBodyAdvice<?>>();
482 			interceptors.add(new JsonViewResponseBodyAdvice());
483 			adapter.setResponseBodyAdvice(interceptors);
484 		}
485 
486 		AsyncSupportConfigurer configurer = new AsyncSupportConfigurer();
487 		configureAsyncSupport(configurer);
488 
489 		if (configurer.getTaskExecutor() != null) {
490 			adapter.setTaskExecutor(configurer.getTaskExecutor());
491 		}
492 		if (configurer.getTimeout() != null) {
493 			adapter.setAsyncRequestTimeout(configurer.getTimeout());
494 		}
495 		adapter.setCallableInterceptors(configurer.getCallableInterceptors());
496 		adapter.setDeferredResultInterceptors(configurer.getDeferredResultInterceptors());
497 
498 		return adapter;
499 	}

525 	/**
526 	 * Return a global {@link Validator} instance for example for validating
527 	 * {@code @ModelAttribute} and {@code @RequestBody} method arguments.
528 	 * Delegates to {@link #getValidator()} first and if that returns {@code null}
529 	 * checks the classpath for the presence of a JSR-303 implementations
530 	 * before creating a {@code OptionalValidatorFactoryBean}.If a JSR-303
531 	 * implementation is not available, a no-op {@link Validator} is returned.
532 	 */
533 	@Bean
534 	public Validator mvcValidator() {
535 		Validator validator = getValidator();
536 		if (validator == null) {
537 			if (ClassUtils.isPresent("javax.validation.Validator", getClass().getClassLoader())) {
538 				Class<?> clazz;
539 				try {
540 					String className = "org.springframework.validation.beanvalidation.OptionalValidatorFactoryBean";
541 					clazz = ClassUtils.forName(className, WebMvcConfigurationSupport.class.getClassLoader());
542 				}
543 				catch (ClassNotFoundException ex) {
544 					throw new BeanInitializationException("Could not find default validator class", ex);
545 				}
546 				catch (LinkageError ex) {
547 					throw new BeanInitializationException("Could not load default validator class", ex);
548 				}
549 				validator = (Validator) BeanUtils.instantiate(clazz);
550 			}
551 			else {
552 				validator = new NoOpValidator();
553 			}
554 		}
555 		return validator;
556 	}

606 	/**
607 	 * Add custom {@link HandlerMethodArgumentResolver}s to use in addition to
608 	 * the ones registered by default.
609 	 * <p>Custom argument resolvers are invoked before built-in resolvers

673 	 * Subclasses can call this method from {@link #configureMessageConverters(List)}.
674 	 * @param messageConverters the list to add the default message converters to
675 	 */
676 	protected final void addDefaultHttpMessageConverters(List<HttpMessageConverter<?>> messageConverters) {
677 		StringHttpMessageConverter stringConverter = new StringHttpMessageConverter();
678 		stringConverter.setWriteAcceptCharset(false);
679 
680 		messageConverters.add(new ByteArrayHttpMessageConverter());
681 		messageConverters.add(stringConverter);
682 		messageConverters.add(new ResourceHttpMessageConverter());
683 		messageConverters.add(new SourceHttpMessageConverter<Source>());
684 		messageConverters.add(new AllEncompassingFormHttpMessageConverter());
685 
686 		if (romePresent) {
687 			messageConverters.add(new AtomFeedHttpMessageConverter());
688 			messageConverters.add(new RssChannelHttpMessageConverter());
689 		}
690 
691 		if (jackson2XmlPresent) {
692 			ObjectMapper objectMapper = Jackson2ObjectMapperBuilder.xml().applicationContext(this.applicationContext).build();
693 			messageConverters.add(new MappingJackson2XmlHttpMessageConverter(objectMapper));
694 		}
695 		else if (jaxb2Present) {
696 			messageConverters.add(new Jaxb2RootElementHttpMessageConverter());
697 		}
698 
699 		if (jackson2Present) {
700 			ObjectMapper objectMapper = Jackson2ObjectMapperBuilder.json().applicationContext(this.applicationContext).build();
701 			messageConverters.add(new MappingJackson2HttpMessageConverter(objectMapper));
702 		}
703 		else if (gsonPresent) {
704 			messageConverters.add(new GsonHttpMessageConverter());
705 		}

751 	 * of exception resolvers obtained either through
752 	 * {@link #configureHandlerExceptionResolvers(List)} or through
753 	 * {@link #addDefaultHandlerExceptionResolvers(List)}.
754 	 * <p><strong>Note:</strong> This method cannot be made final due to CGLib
755 	 * constraints. Rather than overriding it, consider overriding
756 	 * {@link #configureHandlerExceptionResolvers(List)}, which allows
757 	 * providing a list of resolvers.
758 	 */
759 	@Bean
760 	public HandlerExceptionResolver handlerExceptionResolver() {
761 		List<HandlerExceptionResolver> exceptionResolvers = new ArrayList<HandlerExceptionResolver>();
762 		configureHandlerExceptionResolvers(exceptionResolvers);

765 			addDefaultHandlerExceptionResolvers(exceptionResolvers);
766 		}
767 
768 		HandlerExceptionResolverComposite composite = new HandlerExceptionResolverComposite();
769 		composite.setOrder(0);
770 		composite.setExceptionResolvers(exceptionResolvers);
771 		return composite;

786 	/**
787 	 * A method available to subclasses for adding default {@link HandlerExceptionResolver}s.
788 	 * <p>Adds the following exception resolvers:
789 	 * <ul>
790 	 * <li>{@link ExceptionHandlerExceptionResolver}
791 	 * for handling exceptions through @{@link ExceptionHandler} methods.
792 	 * <li>{@link ResponseStatusExceptionResolver}
793 	 * for exceptions annotated with @{@link ResponseStatus}.
794 	 * <li>{@link DefaultHandlerExceptionResolver}
795 	 * for resolving known Spring exception types
796 	 * </ul>
797 	 */
798 	protected final void addDefaultHandlerExceptionResolvers(List<HandlerExceptionResolver> exceptionResolvers) {
799 		ExceptionHandlerExceptionResolver exceptionHandlerExceptionResolver = new ExceptionHandlerExceptionResolver();
800 		exceptionHandlerExceptionResolver.setContentNegotiationManager(mvcContentNegotiationManager());
801 		exceptionHandlerExceptionResolver.setMessageConverters(getMessageConverters());
802 		if (jackson2Present) {
803 			List<ResponseBodyAdvice<?>> interceptors = new ArrayList<ResponseBodyAdvice<?>>();
804 			interceptors.add(new JsonViewResponseBodyAdvice());
805 			exceptionHandlerExceptionResolver.setResponseBodyAdvice(interceptors);
806 		}
807 		exceptionHandlerExceptionResolver.setApplicationContext(this.applicationContext);
808 		exceptionHandlerExceptionResolver.afterPropertiesSet();
809 		exceptionResolvers.add(exceptionHandlerExceptionResolver);
810 
811 		ResponseStatusExceptionResolver responseStatusExceptionResolver = new ResponseStatusExceptionResolver();
812 		responseStatusExceptionResolver.setMessageSource(this.applicationContext);
813 		exceptionResolvers.add(responseStatusExceptionResolver);
814 
815 		exceptionResolvers.add(new DefaultHandlerExceptionResolver());
816 	}

818 	/**
819 	 * Register a {@link ViewResolverComposite} that contains a chain of view resolvers
820 	 * to use for view resolution.
821 	 * By default this resolver is ordered at 0 unless content negotiation view
822 	 * resolution is used in which case the order is raised to
823 	 * {@link org.springframework.core.Ordered#HIGHEST_PRECEDENCE
824 	 * Ordered.HIGHEST_PRECEDENCE}.
825 	 * <p>If no other resolvers are configured,
826 	 * {@link ViewResolverComposite#resolveViewName(String, Locale)} returns null in order
827 	 * to allow other potential {@link ViewResolver} beans to resolve views.
828 	 * @since 4.1
829 	 */
830 	@Bean
831 	public ViewResolver mvcViewResolver() {
832 		ViewResolverRegistry registry = new ViewResolverRegistry();
833 		registry.setContentNegotiationManager(mvcContentNegotiationManager());
834 		registry.setApplicationContext(this.applicationContext);
835 		configureViewResolvers(registry);
836 
837 		if (registry.getViewResolvers().isEmpty()) {
838 			String[] names = BeanFactoryUtils.beanNamesForTypeIncludingAncestors(
839 					this.applicationContext, ViewResolver.class, true, false);
840 			if (names.length == 1) {
841 				registry.getViewResolvers().add(new InternalResourceViewResolver());
842 			}
843 		}
844 
845 		ViewResolverComposite composite = new ViewResolverComposite();
846 		composite.setOrder(registry.getOrder());
847 		composite.setViewResolvers(registry.getViewResolvers());
848 		composite.setApplicationContext(this.applicationContext);
849 		composite.setServletContext(this.servletContext);
850 		return composite;
851 	}
