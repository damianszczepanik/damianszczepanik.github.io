17  package org.springframework.web.servlet.mvc;
18  
19  import javax.servlet.RequestDispatcher;
20  import javax.servlet.ServletException;
21  import javax.servlet.http.HttpServletRequest;
22  import javax.servlet.http.HttpServletResponse;
23  
24  import org.springframework.beans.factory.BeanNameAware;
25  import org.springframework.web.servlet.ModelAndView;
26  import org.springframework.web.util.WebUtils;
27  
28  /**
29   * Spring Controller implementation that forwards to a named servlet,
30   * i.e. the "servlet-name" in web.xml rather than a URL path mapping.
31   * A target servlet doesn't even need a "servlet-mapping" in web.xml
32   * in the first place: A "servlet" declaration is sufficient.
33   *
34   * <p>Useful to invoke an existing servlet via Spring's dispatching infrastructure,
35   * for example to apply Spring HandlerInterceptors to its requests. This will work
36   * even in a minimal Servlet container that does not support Servlet filters.
37   *
38   * <p><b>Example:</b> web.xml, mapping all "/myservlet" requests to a Spring dispatcher.
39   * Also defines a custom "myServlet", but <i>without</i> servlet mapping.
40   *
41   * <pre class="code">
42   * &lt;servlet&gt;
43   *   &lt;servlet-name&gt;myServlet&lt;/servlet-name&gt;
44   *   &lt;servlet-class&gt;mypackage.TestServlet&lt;/servlet-class&gt;
45   * &lt;/servlet&gt;
46   *
47   * &lt;servlet&gt;
48   *   &lt;servlet-name&gt;myDispatcher&lt;/servlet-name&gt;
49   *   &lt;servlet-class&gt;org.springframework.web.servlet.DispatcherServlet&lt;/servlet-class&gt;
50   * &lt;/servlet&gt;
51   *
52   * &lt;servlet-mapping&gt;
53   *   &lt;servlet-name&gt;myDispatcher&lt;/servlet-name&gt;
54   *   &lt;url-pattern&gt;/myservlet&lt;/url-pattern&gt;
55   * &lt;/servlet-mapping&gt;</pre>
56   *
57   * <b>Example:</b> myDispatcher-servlet.xml, in turn forwarding "/myservlet" to your
58   * servlet (identified by servlet name). All such requests will go through the
59   * configured HandlerInterceptor chain (e.g. an OpenSessionInViewInterceptor).
60   * From the servlet point of view, everything will work as usual.
61   *
62   * <pre class="code">
63   * &lt;bean id="urlMapping" class="org.springframework.web.servlet.handler.SimpleUrlHandlerMapping"&gt;
64   *   &lt;property name="interceptors"&gt;
65   *     &lt;list&gt;
66   *       &lt;ref bean="openSessionInViewInterceptor"/&gt;
67   *     &lt;/list&gt;
68   *   &lt;/property&gt;
69   *   &lt;property name="mappings"&gt;
70   *     &lt;props&gt;
71   *       &lt;prop key="/myservlet"&gt;myServletForwardingController&lt;/prop&gt;
72   *     &lt;/props&gt;
73   *   &lt;/property&gt;
74   * &lt;/bean&gt;
75   *
76   * &lt;bean id="myServletForwardingController" class="org.springframework.web.servlet.mvc.ServletForwardingController"&gt;
77   *   &lt;property name="servletName"&gt;&lt;value&gt;myServlet&lt;/value&gt;&lt;/property&gt;
78   * &lt;/bean&gt;</pre>
79   *
80   * @author Juergen Hoeller
81   * @since 1.1.1
82   * @see ServletWrappingController
83   * @see org.springframework.orm.jpa.support.OpenEntityManagerInViewInterceptor
84   * @see org.springframework.orm.jpa.support.OpenEntityManagerInViewFilter
85   */
86  public class ServletForwardingController extends AbstractController implements BeanNameAware {
87  
88  	private String servletName;
89  
90  	private String beanName;
91  
92  
93  	/**
94  	 * Set the name of the servlet to forward to,
95  	 * i.e. the "servlet-name" of the target servlet in web.xml.
96  	 * <p>Default is the bean name of this controller.
97  	 */
98  	public void setServletName(String servletName) {
99  		this.servletName = servletName;
100 	}
101 
102 	@Override
103 	public void setBeanName(String name) {
104 		this.beanName = name;
105 		if (this.servletName == null) {
106 			this.servletName = name;
107 		}
108 	}
109 
110 
111 	@Override
112 	protected ModelAndView handleRequestInternal(HttpServletRequest request, HttpServletResponse response)
113 			throws Exception {
114 
115 		RequestDispatcher rd = getServletContext().getNamedDispatcher(this.servletName);
116 		if (rd == null) {
117 			throw new ServletException("No servlet with name '" + this.servletName + "' defined in web.xml");
118 		}
119 		// If already included, include again, else forward.
120 		if (useInclude(request, response)) {
121 			rd.include(request, response);
122 			if (logger.isDebugEnabled()) {
123 				logger.debug("Included servlet [" + this.servletName +
124 						"] in ServletForwardingController '" + this.beanName + "'");
125 			}
126 		}
127 		else {
128 			rd.forward(request, response);
129 			if (logger.isDebugEnabled()) {
130 				logger.debug("Forwarded to servlet [" + this.servletName +
131 						"] in ServletForwardingController '" + this.beanName + "'");
132 			}
133 		}
134 		return null;
135 	}
136 
137 	/**
138 	 * Determine whether to use RequestDispatcher's {@code include} or
139 	 * {@code forward} method.
140 	 * <p>Performs a check whether an include URI attribute is found in the request,
141 	 * indicating an include request, and whether the response has already been committed.
142 	 * In both cases, an include will be performed, as a forward is not possible anymore.
143 	 * @param request current HTTP request
144 	 * @param response current HTTP response
145 	 * @return {@code true} for include, {@code false} for forward
146 	 * @see javax.servlet.RequestDispatcher#forward
147 	 * @see javax.servlet.RequestDispatcher#include
148 	 * @see javax.servlet.ServletResponse#isCommitted
149 	 * @see org.springframework.web.util.WebUtils#isIncludeRequest
150 	 */
151 	protected boolean useInclude(HttpServletRequest request, HttpServletResponse response) {
152 		return (WebUtils.isIncludeRequest(request) || response.isCommitted());
153 	}
154 
155 }
