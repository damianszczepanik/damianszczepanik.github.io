17  package org.springframework.web.servlet.mvc.condition;
18  
19  import javax.servlet.http.HttpServletRequest;
20  
21  import org.junit.Test;
22  
23  import org.springframework.mock.web.test.MockHttpServletRequest;
24  import org.springframework.web.bind.annotation.RequestMethod;
25  
26  import static org.junit.Assert.*;
27  
28  /**
29   * A test fixture for {@link RequestConditionHolder} tests.
30   *
31   * @author Rossen Stoyanchev
32   */
33  public class RequestConditionHolderTests {
34  
35  	@Test
36  	public void combine() {
37  		RequestConditionHolder params1 = new RequestConditionHolder(new ParamsRequestCondition("name1"));
38  		RequestConditionHolder params2 = new RequestConditionHolder(new ParamsRequestCondition("name2"));
39  		RequestConditionHolder expected = new RequestConditionHolder(new ParamsRequestCondition("name1", "name2"));
40  
41  		assertEquals(expected, params1.combine(params2));
42  	}
43  
44  	@Test
45  	public void combineEmpty() {
46  		RequestConditionHolder empty = new RequestConditionHolder(null);
47  		RequestConditionHolder notEmpty = new RequestConditionHolder(new ParamsRequestCondition("name"));
48  
49  		assertSame(empty, empty.combine(empty));
50  		assertSame(notEmpty, notEmpty.combine(empty));
51  		assertSame(notEmpty, empty.combine(notEmpty));
52  	}
53  
54  	@Test(expected=ClassCastException.class)
55  	public void combineIncompatible() {
56  		RequestConditionHolder params = new RequestConditionHolder(new ParamsRequestCondition("name"));
57  		RequestConditionHolder headers = new RequestConditionHolder(new HeadersRequestCondition("name"));
58  		params.combine(headers);
59  	}
60  
61  	@Test
62  	public void match() {
63  		MockHttpServletRequest request = new MockHttpServletRequest("GET", "/");
64  		request.setParameter("name1", "value1");
65  
66  		RequestMethodsRequestCondition rm = new RequestMethodsRequestCondition(RequestMethod.GET, RequestMethod.POST);
67  		RequestConditionHolder custom = new RequestConditionHolder(rm);
68  		RequestMethodsRequestCondition expected = new RequestMethodsRequestCondition(RequestMethod.GET);
69  
70  		assertEquals(expected, custom.getMatchingCondition(request).getCondition());
71  	}
72  
73  	@Test
74  	public void noMatch() {
75  		MockHttpServletRequest request = new MockHttpServletRequest("GET", "/");
76  
77  		RequestMethodsRequestCondition rm = new RequestMethodsRequestCondition(RequestMethod.POST);
78  		RequestConditionHolder custom = new RequestConditionHolder(rm);
79  
80  		assertNull(custom.getMatchingCondition(request));
81  	}
82  
83  	@Test
84  	public void matchEmpty() {
85  		RequestConditionHolder empty = new RequestConditionHolder(null);
86  		assertSame(empty, empty.getMatchingCondition(new MockHttpServletRequest()));
87  	}
88  
89  	@Test
90  	public void compare() {
91  		HttpServletRequest request = new MockHttpServletRequest();
92  
93  		RequestConditionHolder params11 = new RequestConditionHolder(new ParamsRequestCondition("1"));
94  		RequestConditionHolder params12 = new RequestConditionHolder(new ParamsRequestCondition("1", "2"));
95  
96  		assertEquals(1, params11.compareTo(params12, request));
97  		assertEquals(-1, params12.compareTo(params11, request));
98  	}
99  
100 	@Test
101 	public void compareEmpty() {
102 		HttpServletRequest request = new MockHttpServletRequest();
103 
104 		RequestConditionHolder empty = new RequestConditionHolder(null);
105 		RequestConditionHolder empty2 = new RequestConditionHolder(null);
106 		RequestConditionHolder notEmpty = new RequestConditionHolder(new ParamsRequestCondition("name"));
107 
108 		assertEquals(0, empty.compareTo(empty2, request));
109 		assertEquals(-1, notEmpty.compareTo(empty, request));
110 		assertEquals(1, empty.compareTo(notEmpty, request));
111 	}
112 
113 	@Test(expected=ClassCastException.class)
114 	public void compareIncompatible() {
115 		RequestConditionHolder params = new RequestConditionHolder(new ParamsRequestCondition("name"));
116 		RequestConditionHolder headers = new RequestConditionHolder(new HeadersRequestCondition("name"));
117 		params.compareTo(headers, new MockHttpServletRequest());
118 	}
119 
120 }
