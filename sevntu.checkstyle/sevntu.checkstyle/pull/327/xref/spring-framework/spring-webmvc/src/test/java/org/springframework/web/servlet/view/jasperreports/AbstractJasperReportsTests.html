88  	protected List<Object> getData() {
89  		List<Object> list = new ArrayList<Object>();
90  		for (int x = 0; x < 10; x++) {
91  			PersonBean bean = new PersonBean();
92  			bean.setId(x);
93  			bean.setName("Rob Harrop");
94  			bean.setStreet("foo");
95  			list.add(bean);
96  		}
97  		return list;
98  	}

100 	protected List<Object> getProductData() {
101 		List<Object> list = new ArrayList<Object>();
102 		for (int x = 0; x < 10; x++) {
103 			ProductBean bean = new ProductBean();

105 			bean.setName("Foo Bar");
106 			bean.setPrice(1.9f);
107 			bean.setQuantity(1.0f);
108 			list.add(bean);
109 		}
110 		return list;
