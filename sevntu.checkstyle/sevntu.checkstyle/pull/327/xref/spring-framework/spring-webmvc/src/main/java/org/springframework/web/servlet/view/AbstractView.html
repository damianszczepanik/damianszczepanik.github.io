41  /**
42   * Abstract base class for {@link org.springframework.web.servlet.View}
43   * implementations. Subclasses should be JavaBeans, to allow for
44   * convenient configuration as Spring-managed bean instances.
45   *
46   * <p>Provides support for static attributes, to be made available to the view,
47   * with a variety of ways to specify them. Static attributes will be merged
48   * with the given dynamic attributes (the model that the controller returned)
49   * for each render operation.
50   *
51   * <p>Extends {@link WebApplicationObjectSupport}, which will be helpful to
52   * some views. Subclasses just need to implement the actual rendering.
53   *
54   * @author Rod Johnson
55   * @author Juergen Hoeller
56   * @see #setAttributes
57   * @see #setAttributesMap
58   * @see #renderMergedOutputModel
59   */
60  public abstract class AbstractView extends WebApplicationObjectSupport implements View, BeanNameAware {

134 	/**
135 	 * Set static attributes as a CSV string.
136 	 * Format is: attname0={value1},attname1={value1}
137 	 * <p>"Static" attributes are fixed attributes that are specified in
138 	 * the View instance configuration. "Dynamic" attributes, on the other hand,
139 	 * are values passed in as part of the model.
140 	 */
141 	public void setAttributesCSV(String propString) throws IllegalArgumentException {
142 		if (propString != null) {
143 			StringTokenizer st = new StringTokenizer(propString, ",");
144 			while (st.hasMoreTokens()) {
145 				String tok = st.nextToken();
146 				int eqIdx = tok.indexOf("=");
147 				if (eqIdx == -1) {
148 					throw new IllegalArgumentException("Expected = in attributes CSV string '" + propString + "'");
149 				}
150 				if (eqIdx >= tok.length() - 2) {
151 					throw new IllegalArgumentException(
152 							"At least 2 characters ([]) required in attributes CSV string '" + propString + "'");
153 				}
154 				String name = tok.substring(0, eqIdx);
155 				String value = tok.substring(eqIdx + 1);
156 
157 				// Delete first and last characters of value: { and }
158 				value = value.substring(1);
159 				value = value.substring(0, value.length() - 1);
160 
161 				addStaticAttribute(name, value);
162 			}
163 		}
164 	}

166 	/**
167 	 * Set static attributes for this view from a
168 	 * {@code java.util.Properties} object.
169 	 * <p>"Static" attributes are fixed attributes that are specified in
170 	 * the View instance configuration. "Dynamic" attributes, on the other hand,
171 	 * are values passed in as part of the model.
172 	 * <p>This is the most convenient way to set static attributes. Note that
173 	 * static attributes can be overridden by dynamic attributes, if a value
174 	 * with the same name is included in the model.
175 	 * <p>Can be populated with a String "value" (parsed via PropertiesEditor)
176 	 * or a "props" element in XML bean definitions.
177 	 * @see org.springframework.beans.propertyeditors.PropertiesEditor
178 	 */
179 	public void setAttributes(Properties attributes) {
180 		CollectionUtils.mergePropertiesIntoMap(attributes, this.staticAttributes);
181 	}

184 	 * Set static attributes for this view from a Map. This allows to set
185 	 * any kind of attribute values, for example bean references.
186 	 * <p>"Static" attributes are fixed attributes that are specified in
187 	 * the View instance configuration. "Dynamic" attributes, on the other hand,
188 	 * are values passed in as part of the model.
189 	 * <p>Can be populated with a "map" or "props" element in XML bean definitions.
190 	 * @param attributes Map with name Strings as keys and attribute objects as values
191 	 */
192 	public void setAttributesMap(Map<String, ?> attributes) {
193 		if (attributes != null) {
194 			for (Map.Entry<String, ?> entry : attributes.entrySet()) {
195 				addStaticAttribute(entry.getKey(), entry.getValue());
196 			}
197 		}
198 	}

258 	 * Set whether to make all Spring beans in the application context accessible
259 	 * as request attributes, through lazy checking once an attribute gets accessed.
260 	 * <p>This will make all such beans accessible in plain {@code ${...}}
261 	 * expressions in a JSP 2.0 page, as well as in JSTL's {@code c:out}
262 	 * value expressions.
263 	 * <p>Default is "false". Switch this flag on to transparently expose all

265 	 * <p><b>NOTE:</b> Context beans will override any custom request or session
266 	 * attributes of the same name that have been manually added. However, model

269 	 * @see #getRequestToExpose

288 	/**
289 	 * Prepares the view given the specified model, merging it with static
290 	 * attributes and a RequestContext attribute, if necessary.
291 	 * Delegates to renderMergedOutputModel for the actual rendering.
292 	 * @see #renderMergedOutputModel
293 	 */
294 	@Override
295 	public void render(Map<String, ?> model, HttpServletRequest request, HttpServletResponse response) throws Exception {
296 		if (logger.isTraceEnabled()) {
297 			logger.trace("Rendering view with name '" + this.beanName + "' with model " + model +
298 				" and static attributes " + this.staticAttributes);
299 		}
300 
301 		Map<String, Object> mergedModel = createMergedOutputModel(model, request, response);
302 		prepareResponse(request, response);
303 		renderMergedOutputModel(mergedModel, getRequestToExpose(request), response);
304 	}
305 
306 	/**
307 	 * Creates a combined output Map (never {@code null}) that includes dynamic values and static attributes.
308 	 * Dynamic values take precedence over static attributes.
309 	 */
310 	protected Map<String, Object> createMergedOutputModel(Map<String, ?> model, HttpServletRequest request,
311 			HttpServletResponse response) {
312 
313 		@SuppressWarnings("unchecked")
314 		Map<String, Object> pathVars = (this.exposePathVariables ?
315 				(Map<String, Object>) request.getAttribute(View.PATH_VARIABLES) : null);
316 
317 		// Consolidate static and dynamic model attributes.
318 		int size = this.staticAttributes.size();
319 		size += (model != null ? model.size() : 0);
320 		size += (pathVars != null ? pathVars.size() : 0);
321 
322 		Map<String, Object> mergedModel = new LinkedHashMap<String, Object>(size);
323 		mergedModel.putAll(this.staticAttributes);
324 		if (pathVars != null) {
325 			mergedModel.putAll(pathVars);
326 		}
327 		if (model != null) {
328 			mergedModel.putAll(model);
329 		}
330 
331 		// Expose RequestContext?
332 		if (this.requestContextAttribute != null) {
333 			mergedModel.put(this.requestContextAttribute, createRequestContext(request, response, mergedModel));
334 		}
335 
336 		return mergedModel;
337 	}

340 	 * Create a RequestContext to expose under the specified attribute name.
341 	 * <p>Default implementation creates a standard RequestContext instance for the
342 	 * given request and model. Can be overridden in subclasses for custom instances.
343 	 * @param request current HTTP request
344 	 * @param model combined output Map (never {@code null}),
345 	 * with dynamic values taking precedence over static attributes
346 	 * @return the RequestContext instance
347 	 * @see #setRequestContextAttribute
348 	 * @see org.springframework.web.servlet.support.RequestContext
349 	 */

351 			HttpServletRequest request, HttpServletResponse response, Map<String, Object> model) {
352

356 	/**
357 	 * Prepare the given response for rendering.
358 	 * <p>The default implementation applies a workaround for an IE bug
359 	 * when sending download content via HTTPS.
360 	 * @param request current HTTP request
361 	 * @param response current HTTP response
362 	 */
363 	protected void prepareResponse(HttpServletRequest request, HttpServletResponse response) {
364 		if (generatesDownloadContent()) {
365 			response.setHeader("Pragma", "private");
366 			response.setHeader("Cache-Control", "private, must-revalidate");
367 		}
368 	}

372 	 * (typically binary content like PDF or Excel files).
373 	 * <p>The default implementation returns {@code false}. Subclasses are
374 	 * encouraged to return {@code true} here if they know that they are
375 	 * generating download content that requires temporary caching on the
376 	 * client side, typically via the response OutputStream.
377 	 * @see #prepareResponse
378 	 * @see javax.servlet.http.HttpServletResponse#getOutputStream()
379 	 */

381 		return false;

384 	/**
385 	 * Get the request handle to expose to {@link #renderMergedOutputModel}, i.e. to the view.
386 	 * <p>The default implementation wraps the original request for exposure of Spring beans
387 	 * as request attributes (if demanded).
388 	 * @param originalRequest the original servlet request as provided by the engine
389 	 * @return the wrapped request, or the original request if no wrapping is necessary
390 	 * @see #setExposeContextBeansAsAttributes
391 	 * @see #setExposedContextBeanNames
392 	 * @see org.springframework.web.context.support.ContextExposingHttpServletRequest
393 	 */
394 	protected HttpServletRequest getRequestToExpose(HttpServletRequest originalRequest) {
395 		if (this.exposeContextBeansAsAttributes || this.exposedContextBeanNames != null) {
396 			return new ContextExposingHttpServletRequest(
397 					originalRequest, getWebApplicationContext(), this.exposedContextBeanNames);
398 		}
399 		return originalRequest;
400 	}

403 	 * Subclasses must implement this method to actually render the view.
404 	 * <p>The first step will be preparing the request: In the JSP case,
405 	 * this would mean setting model objects as request attributes.
406 	 * The second step will be the actual rendering of the view,
407 	 * for example including the JSP via a RequestDispatcher.
408 	 * @param model combined output Map (never {@code null}),
409 	 * with dynamic values taking precedence over static attributes
410 	 * @param request current HTTP request
411 	 * @param response current HTTP response
412 	 * @throws Exception if rendering failed
413 	 */
414 	protected abstract void renderMergedOutputModel(
415 			Map<String, Object> model, HttpServletRequest request, HttpServletResponse response) throws Exception;

418 	/**
419 	 * Expose the model objects in the given map as request attributes.
420 	 * Names will be taken from the model Map.
421 	 * This method is suitable for all resources reachable by {@link javax.servlet.RequestDispatcher}.
422 	 * @param model Map of model objects to expose
423 	 * @param request current HTTP request
424 	 */
425 	protected void exposeModelAsRequestAttributes(Map<String, Object> model, HttpServletRequest request) throws Exception {
426 		for (Map.Entry<String, Object> entry : model.entrySet()) {
427 			String modelName = entry.getKey();
428 			Object modelValue = entry.getValue();
429 			if (modelValue != null) {
430 				request.setAttribute(modelName, modelValue);
431 				if (logger.isDebugEnabled()) {
432 					logger.debug("Added model object '" + modelName + "' of type [" + modelValue.getClass().getName() +
433 							"] to request in view with name '" + getBeanName() + "'");
434 				}
435 			}
436 			else {
437 				request.removeAttribute(modelName);
438 				if (logger.isDebugEnabled()) {
439 					logger.debug("Removed model object '" + modelName +
440 							"' from request in view with name '" + getBeanName() + "'");
441 				}
442 			}
443 		}
444 	}

447 	 * Create a temporary OutputStream for this view.
448 	 * <p>This is typically used as IE workaround, for setting the content length header
449 	 * from the temporary stream before actually writing the content to the HTTP response.

451 	protected ByteArrayOutputStream createTemporaryOutputStream() {
452 		return new ByteArrayOutputStream(OUTPUT_BYTE_ARRAY_INITIAL_SIZE);
453 	}

455 	/**
456 	 * Write the given temporary OutputStream to the HTTP response.
457 	 * @param response current HTTP response
458 	 * @param baos the temporary OutputStream to write
459 	 * @throws IOException if writing/flushing failed
460 	 */
461 	protected void writeToResponse(HttpServletResponse response, ByteArrayOutputStream baos) throws IOException {
462 		// Write content type and also length (determined via byte array).
463 		response.setContentType(getContentType());
464 		response.setContentLength(baos.size());
465 
466 		// Flush byte array to servlet output stream.
467 		ServletOutputStream out = response.getOutputStream();
468 		baos.writeTo(out);
469 		out.flush();
470 	}

472 	/**
473 	 * Set the content type of the response to the configured
474 	 * {@link #setContentType(String) content type} unless the
475 	 * {@link View#SELECTED_CONTENT_TYPE} request attribute is present and set
476 	 * to a concrete media type.
477 	 */
478 	protected void setResponseContentType(HttpServletRequest request, HttpServletResponse response) {
479 		MediaType mediaType = (MediaType) request.getAttribute(View.SELECTED_CONTENT_TYPE);
480 		if (mediaType != null && mediaType.isConcrete()) {
481 			response.setContentType(mediaType.toString());
482 		}
483 		else {
484 			response.setContentType(getContentType());
485 		}
486 	}

488 	@Override
489 	public String toString() {
490 		StringBuilder sb = new StringBuilder(getClass().getName());
491 		if (getBeanName() != null) {
492 			sb.append(": name '").append(getBeanName()).append("'");
493 		}
494 		else {
495 			sb.append(": unnamed");
496 		}
497 		return sb.toString();
498 	}
