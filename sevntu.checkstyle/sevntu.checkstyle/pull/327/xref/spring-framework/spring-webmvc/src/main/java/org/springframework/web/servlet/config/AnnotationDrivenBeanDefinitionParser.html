83  /**
84   * A {@link BeanDefinitionParser} that provides the configuration for the
85   * {@code <annotation-driven/>} MVC namespace element.
86   *
87   * <p>This class registers the following {@link HandlerMapping}s:</p>
88   * <ul>
89   * <li>{@link RequestMappingHandlerMapping}
90   * ordered at 0 for mapping requests to annotated controller methods.
91   * <li>{@link BeanNameUrlHandlerMapping}
92   * ordered at 2 to map URL paths to controller bean names.
93   * </ul>
94   *
95   * <p><strong>Note:</strong> Additional HandlerMappings may be registered
96   * as a result of using the {@code <view-controller>} or the
97   * {@code <resources>} MVC namespace elements.
98   *
99   * <p>This class registers the following {@link HandlerAdapter}s:
100  * <ul>
101  * <li>{@link RequestMappingHandlerAdapter}
102  * for processing requests with annotated controller methods.
103  * <li>{@link HttpRequestHandlerAdapter}
104  * for processing requests with {@link HttpRequestHandler}s.
105  * <li>{@link SimpleControllerHandlerAdapter}
106  * for processing requests with interface-based {@link Controller}s.
107  * </ul>
108  *
109  * <p>This class registers the following {@link HandlerExceptionResolver}s:
110  * <ul>
111  * <li>{@link ExceptionHandlerExceptionResolver} for handling exceptions
112  * through @{@link ExceptionHandler} methods.
113  * <li>{@link ResponseStatusExceptionResolver} for exceptions annotated
114  * with @{@link ResponseStatus}.
115  * <li>{@link DefaultHandlerExceptionResolver} for resolving known Spring
116  * exception types
117  * </ul>
118  *
119  * <p>This class registers an {@link org.springframework.util.AntPathMatcher}
120  * and a {@link org.springframework.web.util.UrlPathHelper} to be used by:
121  * <ul>
122  * <li>the {@link RequestMappingHandlerMapping},
123  * <li>the {@link HandlerMapping} for ViewControllers
124  * <li>and the {@link HandlerMapping} for serving resources
125  * </ul>
126  * Note that those beans can be configured by using the {@code path-matching} MVC namespace element.
127  *
128  * <p>Both the {@link RequestMappingHandlerAdapter} and the
129  * {@link ExceptionHandlerExceptionResolver} are configured with instances of
130  * the following by default:
131  * <ul>
132  * <li>A {@link ContentNegotiationManager}
133  * <li>A {@link DefaultFormattingConversionService}
134  * <li>A {@link org.springframework.validation.beanvalidation.LocalValidatorFactoryBean}
135  * if a JSR-303 implementation is available on the classpath
136  * <li>A range of {@link HttpMessageConverter}s depending on what 3rd party
137  * libraries are available on the classpath.
138  * </ul>
139  *
140  * @author Keith Donald
141  * @author Juergen Hoeller
142  * @author Arjen Poutsma
143  * @author Rossen Stoyanchev
144  * @author Brian Clozel
145  * @author Agim Emruli
146  * @since 3.0
147  */

172 	@Override
173 	public BeanDefinition parse(Element element, ParserContext parserContext) {
174 		Object source = parserContext.extractSource(element);
175 
176 		CompositeComponentDefinition compDefinition = new CompositeComponentDefinition(element.getTagName(), source);
177 		parserContext.pushContainingComponent(compDefinition);
178 
179 		RuntimeBeanReference contentNegotiationManager = getContentNegotiationManager(element, source, parserContext);
180 
181 		RootBeanDefinition handlerMappingDef = new RootBeanDefinition(RequestMappingHandlerMapping.class);
182 		handlerMappingDef.setSource(source);
183 		handlerMappingDef.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
184 		handlerMappingDef.getPropertyValues().add("order", 0);
185 		handlerMappingDef.getPropertyValues().add("contentNegotiationManager", contentNegotiationManager);
186 		String methodMappingName = parserContext.getReaderContext().registerWithGeneratedName(handlerMappingDef);
187 
188 		if (element.hasAttribute("enable-matrix-variables")) {
189 			Boolean enableMatrixVariables = Boolean.valueOf(element.getAttribute("enable-matrix-variables"));
190 			handlerMappingDef.getPropertyValues().add("removeSemicolonContent", !enableMatrixVariables);
191 		}
192 		else if (element.hasAttribute("enableMatrixVariables")) {
193 			Boolean enableMatrixVariables = Boolean.valueOf(element.getAttribute("enableMatrixVariables"));
194 			handlerMappingDef.getPropertyValues().add("removeSemicolonContent", !enableMatrixVariables);
195 		}
196 
197 		configurePathMatchingProperties(handlerMappingDef, element, parserContext);
198 
199 		RuntimeBeanReference conversionService = getConversionService(element, source, parserContext);
200 		RuntimeBeanReference validator = getValidator(element, source, parserContext);
201 		RuntimeBeanReference messageCodesResolver = getMessageCodesResolver(element);
202 
203 		RootBeanDefinition bindingDef = new RootBeanDefinition(ConfigurableWebBindingInitializer.class);
204 		bindingDef.setSource(source);
205 		bindingDef.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
206 		bindingDef.getPropertyValues().add("conversionService", conversionService);
207 		bindingDef.getPropertyValues().add("validator", validator);
208 		bindingDef.getPropertyValues().add("messageCodesResolver", messageCodesResolver);
209 
210 		ManagedList<?> messageConverters = getMessageConverters(element, source, parserContext);
211 		ManagedList<?> argumentResolvers = getArgumentResolvers(element, parserContext);
212 		ManagedList<?> returnValueHandlers = getReturnValueHandlers(element, parserContext);
213 		String asyncTimeout = getAsyncTimeout(element);
214 		RuntimeBeanReference asyncExecutor = getAsyncExecutor(element);
215 		ManagedList<?> callableInterceptors = getCallableInterceptors(element, source, parserContext);
216 		ManagedList<?> deferredResultInterceptors = getDeferredResultInterceptors(element, source, parserContext);
217 
218 		RootBeanDefinition handlerAdapterDef = new RootBeanDefinition(RequestMappingHandlerAdapter.class);
219 		handlerAdapterDef.setSource(source);
220 		handlerAdapterDef.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
221 		handlerAdapterDef.getPropertyValues().add("contentNegotiationManager", contentNegotiationManager);
222 		handlerAdapterDef.getPropertyValues().add("webBindingInitializer", bindingDef);
223 		handlerAdapterDef.getPropertyValues().add("messageConverters", messageConverters);
224 		addResponseBodyAdvice(handlerAdapterDef);
225 
226 		if (element.hasAttribute("ignore-default-model-on-redirect")) {
227 			Boolean ignoreDefaultModel = Boolean.valueOf(element.getAttribute("ignore-default-model-on-redirect"));
228 			handlerAdapterDef.getPropertyValues().add("ignoreDefaultModelOnRedirect", ignoreDefaultModel);
229 		}
230 		else if (element.hasAttribute("ignoreDefaultModelOnRedirect")) {
231 			// "ignoreDefaultModelOnRedirect" spelling is deprecated
232 			Boolean ignoreDefaultModel = Boolean.valueOf(element.getAttribute("ignoreDefaultModelOnRedirect"));
233 			handlerAdapterDef.getPropertyValues().add("ignoreDefaultModelOnRedirect", ignoreDefaultModel);
234 		}
235 
236 		if (argumentResolvers != null) {
237 			handlerAdapterDef.getPropertyValues().add("customArgumentResolvers", argumentResolvers);
238 		}
239 		if (returnValueHandlers != null) {
240 			handlerAdapterDef.getPropertyValues().add("customReturnValueHandlers", returnValueHandlers);
241 		}
242 		if (asyncTimeout != null) {
243 			handlerAdapterDef.getPropertyValues().add("asyncRequestTimeout", asyncTimeout);
244 		}
245 		if (asyncExecutor != null) {
246 			handlerAdapterDef.getPropertyValues().add("taskExecutor", asyncExecutor);
247 		}
248 
249 		handlerAdapterDef.getPropertyValues().add("callableInterceptors", callableInterceptors);
250 		handlerAdapterDef.getPropertyValues().add("deferredResultInterceptors", deferredResultInterceptors);
251 		String handlerAdapterName = parserContext.getReaderContext().registerWithGeneratedName(handlerAdapterDef);
252 
253 		String uriCompContribName = MvcUriComponentsBuilder.MVC_URI_COMPONENTS_CONTRIBUTOR_BEAN_NAME;
254 		RootBeanDefinition uriCompContribDef = new RootBeanDefinition(CompositeUriComponentsContributorFactoryBean.class);
255 		uriCompContribDef.setSource(source);
256 		uriCompContribDef.getPropertyValues().addPropertyValue("handlerAdapter", handlerAdapterDef);
257 		uriCompContribDef.getPropertyValues().addPropertyValue("conversionService", conversionService);
258 		parserContext.getReaderContext().getRegistry().registerBeanDefinition(uriCompContribName, uriCompContribDef);
259 
260 		RootBeanDefinition csInterceptorDef = new RootBeanDefinition(ConversionServiceExposingInterceptor.class);
261 		csInterceptorDef.setSource(source);
262 		csInterceptorDef.getConstructorArgumentValues().addIndexedArgumentValue(0, conversionService);
263 		RootBeanDefinition mappedCsInterceptorDef = new RootBeanDefinition(MappedInterceptor.class);
264 		mappedCsInterceptorDef.setSource(source);
265 		mappedCsInterceptorDef.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
266 		mappedCsInterceptorDef.getConstructorArgumentValues().addIndexedArgumentValue(0, (Object) null);
267 		mappedCsInterceptorDef.getConstructorArgumentValues().addIndexedArgumentValue(1, csInterceptorDef);
268 		String mappedInterceptorName = parserContext.getReaderContext().registerWithGeneratedName(mappedCsInterceptorDef);
269 
270 		RootBeanDefinition exceptionHandlerExceptionResolver = new RootBeanDefinition(ExceptionHandlerExceptionResolver.class);
271 		exceptionHandlerExceptionResolver.setSource(source);
272 		exceptionHandlerExceptionResolver.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
273 		exceptionHandlerExceptionResolver.getPropertyValues().add("contentNegotiationManager", contentNegotiationManager);
274 		exceptionHandlerExceptionResolver.getPropertyValues().add("messageConverters", messageConverters);
275 		exceptionHandlerExceptionResolver.getPropertyValues().add("order", 0);
276 		addResponseBodyAdvice(exceptionHandlerExceptionResolver);
277 
278 		String methodExceptionResolverName =
279 				parserContext.getReaderContext().registerWithGeneratedName(exceptionHandlerExceptionResolver);
280 
281 		RootBeanDefinition responseStatusExceptionResolver = new RootBeanDefinition(ResponseStatusExceptionResolver.class);
282 		responseStatusExceptionResolver.setSource(source);
283 		responseStatusExceptionResolver.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
284 		responseStatusExceptionResolver.getPropertyValues().add("order", 1);
285 		String responseStatusExceptionResolverName =
286 				parserContext.getReaderContext().registerWithGeneratedName(responseStatusExceptionResolver);
287 
288 		RootBeanDefinition defaultExceptionResolver = new RootBeanDefinition(DefaultHandlerExceptionResolver.class);
289 		defaultExceptionResolver.setSource(source);
290 		defaultExceptionResolver.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
291 		defaultExceptionResolver.getPropertyValues().add("order", 2);
292 		String defaultExceptionResolverName =
293 				parserContext.getReaderContext().registerWithGeneratedName(defaultExceptionResolver);
294 
295 		parserContext.registerComponent(new BeanComponentDefinition(handlerMappingDef, methodMappingName));
296 		parserContext.registerComponent(new BeanComponentDefinition(handlerAdapterDef, handlerAdapterName));
297 		parserContext.registerComponent(new BeanComponentDefinition(uriCompContribDef, uriCompContribName));
298 		parserContext.registerComponent(new BeanComponentDefinition(exceptionHandlerExceptionResolver, methodExceptionResolverName));
299 		parserContext.registerComponent(new BeanComponentDefinition(responseStatusExceptionResolver, responseStatusExceptionResolverName));
300 		parserContext.registerComponent(new BeanComponentDefinition(defaultExceptionResolver, defaultExceptionResolverName));
301 		parserContext.registerComponent(new BeanComponentDefinition(mappedCsInterceptorDef, mappedInterceptorName));
302 
303 		// Ensure BeanNameUrlHandlerMapping (SPR-8289) and default HandlerAdapters are not "turned off"
304 		MvcNamespaceUtils.registerDefaultComponents(parserContext, source);
305 
306 		parserContext.popAndRegisterContainingComponent();
307 
308 		return null;
309 	}

318 	private RuntimeBeanReference getConversionService(Element element, Object source, ParserContext parserContext) {
319 		RuntimeBeanReference conversionServiceRef;
320 		if (element.hasAttribute("conversion-service")) {
321 			conversionServiceRef = new RuntimeBeanReference(element.getAttribute("conversion-service"));
322 		}
323 		else {
324 			RootBeanDefinition conversionDef = new RootBeanDefinition(FormattingConversionServiceFactoryBean.class);
325 			conversionDef.setSource(source);
326 			conversionDef.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
327 			String conversionName = parserContext.getReaderContext().registerWithGeneratedName(conversionDef);
328 			parserContext.registerComponent(new BeanComponentDefinition(conversionDef, conversionName));
329 			conversionServiceRef = new RuntimeBeanReference(conversionName);
330 		}
331 		return conversionServiceRef;
332 	}

334 	private RuntimeBeanReference getValidator(Element element, Object source, ParserContext parserContext) {
335 		if (element.hasAttribute("validator")) {
336 			return new RuntimeBeanReference(element.getAttribute("validator"));

339 			RootBeanDefinition validatorDef = new RootBeanDefinition(

341 			validatorDef.setSource(source);
342 			validatorDef.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
343 			String validatorName = parserContext.getReaderContext().registerWithGeneratedName(validatorDef);
344 			parserContext.registerComponent(new BeanComponentDefinition(validatorDef, validatorName));
345 			return new RuntimeBeanReference(validatorName);
346 		}
347 		else {
348 			return null;
349 		}
350 	}

352 	private RuntimeBeanReference getContentNegotiationManager(Element element, Object source, ParserContext parserContext) {
353 		RuntimeBeanReference contentNegotiationManagerRef;
354 		if (element.hasAttribute("content-negotiation-manager")) {
355 			contentNegotiationManagerRef = new RuntimeBeanReference(element.getAttribute("content-negotiation-manager"));
356 		}
357 		else {
358 			RootBeanDefinition factoryBeanDef = new RootBeanDefinition(ContentNegotiationManagerFactoryBean.class);
359 			factoryBeanDef.setSource(source);
360 			factoryBeanDef.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
361 			factoryBeanDef.getPropertyValues().add("mediaTypes", getDefaultMediaTypes());
362 
363 			String beanName = CONTENT_NEGOTIATION_MANAGER_BEAN_NAME;
364 			parserContext.getReaderContext().getRegistry().registerBeanDefinition(beanName , factoryBeanDef);
365 			parserContext.registerComponent(new BeanComponentDefinition(factoryBeanDef, beanName));
366 			contentNegotiationManagerRef = new RuntimeBeanReference(beanName);
367 		}
368 		return contentNegotiationManagerRef;
369 	}

371 	private void configurePathMatchingProperties(RootBeanDefinition handlerMappingDef,
372 	                                             Element element, ParserContext parserContext) {
373 		Element pathMatchingElement = DomUtils.getChildElementByTagName(element, "path-matching");
374 		if (pathMatchingElement != null) {
375 			Object source = parserContext.extractSource(element);
376 			if (pathMatchingElement.hasAttribute("suffix-pattern")) {
377 				Boolean useSuffixPatternMatch = Boolean.valueOf(pathMatchingElement.getAttribute("suffix-pattern"));
378 				handlerMappingDef.getPropertyValues().add("useSuffixPatternMatch", useSuffixPatternMatch);
379 			}
380 			if (pathMatchingElement.hasAttribute("trailing-slash")) {
381 				Boolean useTrailingSlashMatch = Boolean.valueOf(pathMatchingElement.getAttribute("trailing-slash"));
382 				handlerMappingDef.getPropertyValues().add("useTrailingSlashMatch", useTrailingSlashMatch);
383 			}
384 			if (pathMatchingElement.hasAttribute("registered-suffixes-only")) {
385 				Boolean useRegisteredSuffixPatternMatch = Boolean.valueOf(pathMatchingElement.getAttribute("registered-suffixes-only"));
386 				handlerMappingDef.getPropertyValues().add("useRegisteredSuffixPatternMatch", useRegisteredSuffixPatternMatch);
387 			}
388 			RuntimeBeanReference pathHelperRef = null;
389 			if (pathMatchingElement.hasAttribute("path-helper")) {
390 				pathHelperRef = new RuntimeBeanReference(pathMatchingElement.getAttribute("path-helper"));
391 			}
392 			pathHelperRef = MvcNamespaceUtils.registerUrlPathHelper(pathHelperRef, parserContext, source);
393 			handlerMappingDef.getPropertyValues().add("urlPathHelper", pathHelperRef);
394 
395 			RuntimeBeanReference pathMatcherRef = null;
396 			if (pathMatchingElement.hasAttribute("path-matcher")) {
397 				pathMatcherRef = new RuntimeBeanReference(pathMatchingElement.getAttribute("path-matcher"));
398 			}
399 			pathMatcherRef = MvcNamespaceUtils.registerPathMatcher(pathMatcherRef, parserContext, source);
400 			handlerMappingDef.getPropertyValues().add("pathMatcher", pathMatcherRef);
401 		}
402 	}

404 	private Properties getDefaultMediaTypes() {
405 		Properties props = new Properties();
406 		if (romePresent) {
407 			props.put("atom", MediaType.APPLICATION_ATOM_XML_VALUE);
408 			props.put("rss", "application/rss+xml");
409 		}

412 		}
413 		if (jackson2Present || gsonPresent) {
414 			props.put("json", MediaType.APPLICATION_JSON_VALUE);
415 		}
416 		return props;
417 	}

443 	private ManagedList<?> getCallableInterceptors(Element element, Object source, ParserContext parserContext) {
444 		ManagedList<? super Object> interceptors = new ManagedList<Object>();
445 		Element asyncElement = DomUtils.getChildElementByTagName(element, "async-support");
446 		if (asyncElement != null) {
447 			Element interceptorsElement = DomUtils.getChildElementByTagName(asyncElement, "callable-interceptors");
448 			if (interceptorsElement != null) {
449 				interceptors.setSource(source);
450 				for (Element converter : DomUtils.getChildElementsByTagName(interceptorsElement, "bean")) {
451 					BeanDefinitionHolder beanDef = parserContext.getDelegate().parseBeanDefinitionElement(converter);
452 					beanDef = parserContext.getDelegate().decorateBeanDefinitionIfRequired(converter, beanDef);
453 					interceptors.add(beanDef);
454 				}
455 			}
456 		}
457 		return interceptors;
458 	}

477 	private ManagedList<?> getArgumentResolvers(Element element, ParserContext parserContext) {
478 		Element resolversElement = DomUtils.getChildElementByTagName(element, "argument-resolvers");
479 		if (resolversElement != null) {
480 			ManagedList<Object> resolvers = extractBeanSubElements(resolversElement, parserContext);
481 			return wrapLegacyResolvers(resolvers, parserContext);
482 		}
483 		return null;
484 	}

486 	private ManagedList<Object> wrapLegacyResolvers(List<Object> list, ParserContext context) {
487 		ManagedList<Object> result = new ManagedList<Object>();
488 		for (Object object : list) {
489 			if (object instanceof BeanDefinitionHolder) {
490 				BeanDefinitionHolder beanDef = (BeanDefinitionHolder) object;
491 				String className = beanDef.getBeanDefinition().getBeanClassName();
492 				Class<?> clazz = ClassUtils.resolveClassName(className, context.getReaderContext().getBeanClassLoader());
493 				if (WebArgumentResolver.class.isAssignableFrom(clazz)) {
494 					RootBeanDefinition adapter = new RootBeanDefinition(ServletWebArgumentResolverAdapter.class);
495 					adapter.getConstructorArgumentValues().addIndexedArgumentValue(0, beanDef);
496 					result.add(new BeanDefinitionHolder(adapter, beanDef.getBeanName() + "Adapter"));
497 					continue;
498 				}
499 			}
500 			result.add(object);
501 		}
502 		return result;
503 	}

510 	private ManagedList<?> getMessageConverters(Element element, Object source, ParserContext parserContext) {
511 		Element convertersElement = DomUtils.getChildElementByTagName(element, "message-converters");
512 		ManagedList<? super Object> messageConverters = new ManagedList<Object>();
513 		if (convertersElement != null) {
514 			messageConverters.setSource(source);
515 			for (Element beanElement : DomUtils.getChildElementsByTagName(convertersElement, "bean", "ref")) {
516 				Object object = parserContext.getDelegate().parsePropertySubElement(beanElement, null);
517 				messageConverters.add(object);
518 			}
519 		}
520 
521 		if (convertersElement == null || Boolean.valueOf(convertersElement.getAttribute("register-defaults"))) {
522 			messageConverters.setSource(source);
523 			messageConverters.add(createConverterDefinition(ByteArrayHttpMessageConverter.class, source));
524 
525 			RootBeanDefinition stringConverterDef = createConverterDefinition(StringHttpMessageConverter.class, source);
526 			stringConverterDef.getPropertyValues().add("writeAcceptCharset", false);
527 			messageConverters.add(stringConverterDef);
528 
529 			messageConverters.add(createConverterDefinition(ResourceHttpMessageConverter.class, source));
530 			messageConverters.add(createConverterDefinition(SourceHttpMessageConverter.class, source));
531 			messageConverters.add(createConverterDefinition(AllEncompassingFormHttpMessageConverter.class, source));
532 
533 			if (romePresent) {
534 				messageConverters.add(createConverterDefinition(AtomFeedHttpMessageConverter.class, source));
535 				messageConverters.add(createConverterDefinition(RssChannelHttpMessageConverter.class, source));
536 			}
537 
538 			if (jackson2XmlPresent) {
539 				RootBeanDefinition jacksonConverterDef = createConverterDefinition(MappingJackson2XmlHttpMessageConverter.class, source);
540 				GenericBeanDefinition jacksonFactoryDef = createObjectMapperFactoryDefinition(source);
541 				jacksonFactoryDef.getPropertyValues().add("createXmlMapper", true);
542 				jacksonConverterDef.getConstructorArgumentValues().addIndexedArgumentValue(0, jacksonFactoryDef);
543 				messageConverters.add(jacksonConverterDef);
544 			}
545 			else if (jaxb2Present) {
546 				messageConverters.add(createConverterDefinition(Jaxb2RootElementHttpMessageConverter.class, source));
547 			}
548 
549 			if (jackson2Present) {
550 				RootBeanDefinition jacksonConverterDef = createConverterDefinition(MappingJackson2HttpMessageConverter.class, source);
551 				GenericBeanDefinition jacksonFactoryDef = createObjectMapperFactoryDefinition(source);
552 				jacksonConverterDef.getConstructorArgumentValues().addIndexedArgumentValue(0, jacksonFactoryDef);
553 				messageConverters.add(jacksonConverterDef);
554 			}
555 			else if (gsonPresent) {
556 				messageConverters.add(createConverterDefinition(GsonHttpMessageConverter.class, source));
557 			}
558 		}
559 		return messageConverters;
560 	}

579 		ManagedList<Object> list = new ManagedList<Object>();
580 		list.setSource(parserContext.extractSource(parentElement));
581 		for (Element beanElement : DomUtils.getChildElementsByTagName(parentElement, "bean", "ref")) {

583 			list.add(object);

588     private ManagedList<BeanReference> extractBeanRefSubElements(Element parentElement, ParserContext parserContext){
589         ManagedList<BeanReference> list = new ManagedList<BeanReference>();
590         list.setSource(parserContext.extractSource(parentElement));
591         for (Element refElement : DomUtils.getChildElementsByTagName(parentElement, "ref")) {
592             BeanReference reference;
593             if (StringUtils.hasText("bean")) {
594                 reference = new RuntimeBeanReference(refElement.getAttribute("bean"),false);
595                 list.add(reference);
596             }
597 			else if (StringUtils.hasText("parent")){
598                 reference = new RuntimeBeanReference(refElement.getAttribute("parent"),true);
599                 list.add(reference);
600             }
601 			else {
602                 parserContext.getReaderContext().error("'bean' or 'parent' attribute is required for <ref> element",
603                         parserContext.extractSource(parentElement));
604             }
605         }
606         return list;
607     }
