17  package org.springframework.web.servlet.view.velocity;
18  
19  import org.springframework.web.servlet.view.AbstractTemplateViewResolver;
20  import org.springframework.web.servlet.view.AbstractUrlBasedView;
21  
22  /**
23   * Convenience subclass of {@link org.springframework.web.servlet.view.UrlBasedViewResolver}
24   * that supports {@link VelocityView} (i.e. Velocity templates) and custom subclasses of it.
25   *
26   * <p>The view class for all views generated by this resolver can be specified
27   * via the "viewClass" property. See UrlBasedViewResolver's javadoc for details.
28   *
29   * <p><b>Note:</b> When chaining ViewResolvers, a VelocityViewResolver will
30   * check for the existence of the specified template resources and only return
31   * a non-null View object if the template was actually found.
32   *
33   * @author Juergen Hoeller
34   * @since 13.12.2003
35   * @see #setViewClass
36   * @see #setPrefix
37   * @see #setSuffix
38   * @see #setRequestContextAttribute
39   * @see #setExposeSpringMacroHelpers
40   * @see #setDateToolAttribute
41   * @see #setNumberToolAttribute
42   * @see VelocityView
43   */
44  public class VelocityViewResolver extends AbstractTemplateViewResolver {
45  
46  	private String dateToolAttribute;
47  
48  	private String numberToolAttribute;
49  
50  	private String toolboxConfigLocation;
51  
52  
53  	public VelocityViewResolver() {
54  		setViewClass(requiredViewClass());
55  	}
56  
57  	/**
58  	 * Requires {@link VelocityView}.
59  	 */
60  	@Override
61  	protected Class<?> requiredViewClass() {
62  		return VelocityView.class;
63  	}
64  
65  
66  	/**
67  	 * Set the name of the DateTool helper object to expose in the Velocity context
68  	 * of this view, or {@code null} if not needed. DateTool is part of Velocity Tools 1.0.
69  	 * @see org.apache.velocity.tools.generic.DateTool
70  	 * @see VelocityView#setDateToolAttribute
71  	 */
72  	public void setDateToolAttribute(String dateToolAttribute) {
73  		this.dateToolAttribute = dateToolAttribute;
74  	}
75  
76  	/**
77  	 * Set the name of the NumberTool helper object to expose in the Velocity context
78  	 * of this view, or {@code null} if not needed. NumberTool is part of Velocity Tools 1.1.
79  	 * @see org.apache.velocity.tools.generic.NumberTool
80  	 * @see VelocityView#setNumberToolAttribute
81  	 */
82  	public void setNumberToolAttribute(String numberToolAttribute) {
83  		this.numberToolAttribute = numberToolAttribute;
84  	}
85  
86  	/**
87  	 * Set a Velocity Toolbox config location, for example "/WEB-INF/toolbox.xml",
88  	 * to automatically load a Velocity Tools toolbox definition file and expose
89  	 * all defined tools in the specified scopes. If no config location is
90  	 * specified, no toolbox will be loaded and exposed.
91  	 * <p>The specified location string needs to refer to a ServletContext
92  	 * resource, as expected by ServletToolboxManager which is part of
93  	 * the view package of Velocity Tools.
94  	 * <p><b>Note:</b> Specifying a toolbox config location will lead to
95  	 * VelocityToolboxView instances being created.
96  	 * @see org.apache.velocity.tools.view.servlet.ServletToolboxManager#getInstance
97  	 * @see VelocityToolboxView#setToolboxConfigLocation
98  	 */
99  	public void setToolboxConfigLocation(String toolboxConfigLocation) {
100 		this.toolboxConfigLocation = toolboxConfigLocation;
101 	}
102 
103 
104 	@Override
105 	protected void initApplicationContext() {
106 		super.initApplicationContext();
107 
108 		if (this.toolboxConfigLocation != null) {
109 			if (VelocityView.class.equals(getViewClass())) {
110 				logger.info("Using VelocityToolboxView instead of default VelocityView " +
111 						"due to specified toolboxConfigLocation");
112 				setViewClass(VelocityToolboxView.class);
113 			}
114 			else if (!VelocityToolboxView.class.isAssignableFrom(getViewClass())) {
115 				throw new IllegalArgumentException(
116 						"Given view class [" + getViewClass().getName() +
117 						"] is not of type [" + VelocityToolboxView.class.getName() +
118 						"], which it needs to be in case of a specified toolboxConfigLocation");
119 			}
120 		}
121 	}
122 
123 
124 	@Override
125 	protected AbstractUrlBasedView buildView(String viewName) throws Exception {
126 		VelocityView view = (VelocityView) super.buildView(viewName);
127 		view.setDateToolAttribute(this.dateToolAttribute);
128 		view.setNumberToolAttribute(this.numberToolAttribute);
129 		if (this.toolboxConfigLocation != null) {
130 			((VelocityToolboxView) view).setToolboxConfigLocation(this.toolboxConfigLocation);
131 		}
132 		return view;
133 	}
134 
135 }
