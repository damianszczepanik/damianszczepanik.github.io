17  package org.springframework.web.servlet.handler;
18  
19  import org.springframework.web.method.HandlerMethod;
20  
21  /**
22   * A strategy for assigning a name to a handler method's mapping.
23   *
24   * <p>The strategy can be configured on
25   * {@link org.springframework.web.servlet.handler.AbstractHandlerMethodMapping
26   * AbstractHandlerMethodMapping}. It is used to assign a name to the mapping of
27   * every registered handler method. The names can then be queried via
28   * {@link org.springframework.web.servlet.handler.AbstractHandlerMethodMapping#getHandlerMethodsForMappingName(String)
29   * AbstractHandlerMethodMapping#getHandlerMethodsForMappingName}.
30   *
31   * <p>Applications can build a URL to a controller method by name with the help
32   * of the static method
33   * {@link org.springframework.web.servlet.mvc.method.annotation.MvcUriComponentsBuilder#fromMappingName(String)
34   * MvcUriComponentsBuilder#fromMappingName} or in JSPs through the "mvcUrl"
35   * function registered by the Spring tag library.
36   *
37   * @author Rossen Stoyanchev
38   * @since 4.1
39   */
40  public interface HandlerMethodMappingNamingStrategy<T> {
41  
42  	/**
43  	 * Determine the name for the given HandlerMethod and mapping.
44  	 * @param handlerMethod the handler method
45  	 * @param mapping the mapping
46  	 * @return the name
47  	 */
48  	String getName(HandlerMethod handlerMethod, T mapping);
49  
50  }
