265 	@Test
266 	public void saveOutputFlashMapDecodeParameters() throws Exception {
267 		this.request.setCharacterEncoding("UTF-8");
268 
269 		FlashMap flashMap = new FlashMap();
270 		flashMap.put("anyKey", "anyValue");
271 
272 		flashMap.addTargetRequestParam("key", "%D0%90%D0%90");
273 		flashMap.addTargetRequestParam("key", "%D0%91%D0%91");
274 		flashMap.addTargetRequestParam("key", "%D0%92%D0%92");
275 		flashMap.addTargetRequestParam("%3A%2F%3F%23%5B%5D%40", "value");
276 		this.flashMapManager.saveOutputFlashMap(flashMap, this.request, this.response);
277 
278 		MultiValueMap<String,String> targetRequestParams = flashMap.getTargetRequestParams();
279 		assertEquals(Arrays.asList("\u0410\u0410", "\u0411\u0411", "\u0412\u0412"), targetRequestParams.get("key"));
280 		assertEquals(Arrays.asList("value"), targetRequestParams.get(":/?#[]@"));
281 	}

284 	private static class TestFlashMapManager extends AbstractFlashMapManager {
285 
286 		private List<FlashMap> flashMaps;
287 
288 		public List<FlashMap> getFlashMaps() {
289 			return this.flashMaps;
290 		}
291 
292 		public void setFlashMaps(FlashMap... flashMaps) {
293 			setFlashMaps(Arrays.asList(flashMaps));
294 		}
295 
296 		public void setFlashMaps(List<FlashMap> flashMaps) {
297 			this.flashMaps = new CopyOnWriteArrayList<FlashMap>(flashMaps);
298 		}
299 
300 		@Override
301 		protected List<FlashMap> retrieveFlashMaps(HttpServletRequest request) {
302 			return this.flashMaps;
303 		}
304 
305 		@Override
306 		protected void updateFlashMaps(List<FlashMap> flashMaps, HttpServletRequest request, HttpServletResponse response) {
307 			this.flashMaps = flashMaps;
308 		}
