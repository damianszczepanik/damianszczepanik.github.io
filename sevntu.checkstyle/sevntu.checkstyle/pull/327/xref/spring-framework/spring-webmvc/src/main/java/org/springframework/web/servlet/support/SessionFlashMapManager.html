17  package org.springframework.web.servlet.support;
18  
19  import java.util.List;
20  import javax.servlet.http.HttpServletRequest;
21  import javax.servlet.http.HttpServletResponse;
22  import javax.servlet.http.HttpSession;
23  
24  import org.springframework.web.servlet.FlashMap;
25  import org.springframework.web.util.WebUtils;
26  
27  /**
28   * Store and retrieve {@link FlashMap} instances to and from the HTTP session.
29   *
30   * @author Rossen Stoyanchev
31   * @author Juergen Hoeller
32   * @since 3.1.1
33   */
34  public class SessionFlashMapManager extends AbstractFlashMapManager {
35  
36  	private static final String FLASH_MAPS_SESSION_ATTRIBUTE = SessionFlashMapManager.class.getName() + ".FLASH_MAPS";
37  
38  
39  	/**
40  	 * Retrieves saved FlashMap instances from the HTTP session, if any.
41  	 */
42  	@Override
43  	@SuppressWarnings("unchecked")
44  	protected List<FlashMap> retrieveFlashMaps(HttpServletRequest request) {
45  		HttpSession session = request.getSession(false);
46  		return (session != null ? (List<FlashMap>) session.getAttribute(FLASH_MAPS_SESSION_ATTRIBUTE) : null);
47  	}
48  
49  	/**
50  	 * Saves the given FlashMap instances in the HTTP session.
51  	 */
52  	@Override
53  	protected void updateFlashMaps(List<FlashMap> flashMaps, HttpServletRequest request, HttpServletResponse response) {
54  		WebUtils.setSessionAttribute(request, FLASH_MAPS_SESSION_ATTRIBUTE, (!flashMaps.isEmpty() ? flashMaps : null));
55  	}
56  
57  	/**
58  	 * Exposes the best available session mutex.
59  	 * @see org.springframework.web.util.WebUtils#getSessionMutex
60  	 * @see org.springframework.web.util.HttpSessionMutexListener
61  	 */
62  	@Override
63  	protected Object getFlashMapsMutex(HttpServletRequest request) {
64  		return WebUtils.getSessionMutex(request.getSession());
65  	}
66  
67  }
