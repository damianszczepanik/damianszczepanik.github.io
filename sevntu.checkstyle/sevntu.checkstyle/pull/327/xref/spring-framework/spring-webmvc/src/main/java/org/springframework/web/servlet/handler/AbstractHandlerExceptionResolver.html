17  package org.springframework.web.servlet.handler;
18  
19  import java.util.Set;
20  import javax.servlet.http.HttpServletRequest;
21  import javax.servlet.http.HttpServletResponse;
22  
23  import org.apache.commons.logging.Log;
24  import org.apache.commons.logging.LogFactory;
25  
26  import org.springframework.core.Ordered;
27  import org.springframework.web.servlet.HandlerExceptionResolver;
28  import org.springframework.web.servlet.ModelAndView;
29  
30  /**
31   * Abstract base class for {@link HandlerExceptionResolver} implementations.
32   *
33   * <p>Provides a set of mapped handlers that the resolver should map to,
34   * and the {@link Ordered} implementation.
35   *
36   * @author Arjen Poutsma
37   * @author Juergen Hoeller
38   * @since 3.0
39   */
40  public abstract class AbstractHandlerExceptionResolver implements HandlerExceptionResolver, Ordered {
41  
42  	private static final String HEADER_PRAGMA = "Pragma";
43  
44  	private static final String HEADER_EXPIRES = "Expires";
45  
46  	private static final String HEADER_CACHE_CONTROL = "Cache-Control";
47  
48  
49  	/** Logger available to subclasses */
50  	protected final Log logger = LogFactory.getLog(getClass());
51  
52  	private int order = Ordered.LOWEST_PRECEDENCE;
53  
54  	private Set<?> mappedHandlers;
55  
56  	private Class<?>[] mappedHandlerClasses;
57  
58  	private Log warnLogger;
59  
60  	private boolean preventResponseCaching = false;
61  
62  
63  	public void setOrder(int order) {
64  		this.order = order;
65  	}
66  
67  	@Override
68  	public int getOrder() {
69  		return this.order;
70  	}
71  
72  	/**
73  	 * Specify the set of handlers that this exception resolver should apply to.
74  	 * The exception mappings and the default error view will only apply to the specified handlers.
75  	 * <p>If no handlers and handler classes are set, the exception mappings and the default error
76  	 * view will apply to all handlers. This means that a specified default error view will be used
77  	 * as fallback for all exceptions; any further HandlerExceptionResolvers in the chain will be
78  	 * ignored in this case.
79  	 */
80  	public void setMappedHandlers(Set<?> mappedHandlers) {
81  		this.mappedHandlers = mappedHandlers;
82  	}
83  
84  	/**
85  	 * Specify the set of classes that this exception resolver should apply to.
86  	 * The exception mappings and the default error view will only apply to handlers of the
87  	 * specified type; the specified types may be interfaces and superclasses of handlers as well.
88  	 * <p>If no handlers and handler classes are set, the exception mappings and the default error
89  	 * view will apply to all handlers. This means that a specified default error view will be used
90  	 * as fallback for all exceptions; any further HandlerExceptionResolvers in the chain will be
91  	 * ignored in this case.
92  	 */
93  	public void setMappedHandlerClasses(Class<?>[] mappedHandlerClasses) {
94  		this.mappedHandlerClasses = mappedHandlerClasses;
95  	}
96  
97  	/**
98  	 * Set the log category for warn logging. The name will be passed to the underlying logger
99  	 * implementation through Commons Logging, getting interpreted as log category according
100 	 * to the logger's configuration.
101 	 * <p>Default is no warn logging. Specify this setting to activate warn logging into a specific
102 	 * category. Alternatively, override the {@link #logException} method for custom logging.
103 	 * @see org.apache.commons.logging.LogFactory#getLog(String)
104 	 * @see org.apache.log4j.Logger#getLogger(String)
105 	 * @see java.util.logging.Logger#getLogger(String)
106 	 */
107 	public void setWarnLogCategory(String loggerName) {
108 		this.warnLogger = LogFactory.getLog(loggerName);
109 	}
110 
111 	/**
112 	 * Specify whether to prevent HTTP response caching for any view resolved
113 	 * by this HandlerExceptionResolver.
114 	 * <p>Default is "false". Switch this to "true" in order to automatically
115 	 * generate HTTP response headers that suppress response caching.
116 	 */
117 	public void setPreventResponseCaching(boolean preventResponseCaching) {
118 		this.preventResponseCaching = preventResponseCaching;
119 	}
120 
121 
122 	/**
123 	 * Checks whether this resolver is supposed to apply (i.e. the handler matches
124 	 * in case of "mappedHandlers" having been specified), then delegates to the
125 	 * {@link #doResolveException} template method.
126 	 */
127 	@Override
128 	public ModelAndView resolveException(HttpServletRequest request, HttpServletResponse response,
129 			Object handler, Exception ex) {
130 
131 		if (shouldApplyTo(request, handler)) {
132 			// Log exception, both at debug log level and at warn level, if desired.
133 			if (logger.isDebugEnabled()) {
134 				logger.debug("Resolving exception from handler [" + handler + "]: " + ex);
135 			}
136 			logException(ex, request);
137 			prepareResponse(ex, response);
138 			return doResolveException(request, response, handler, ex);
139 		}
140 		else {
141 			return null;
142 		}
143 	}
144 
145 	/**
146 	 * Check whether this resolver is supposed to apply to the given handler.
147 	 * <p>The default implementation checks against the specified mapped handlers
148 	 * and handler classes, if any.
149 	 * @param request current HTTP request
150 	 * @param handler the executed handler, or {@code null} if none chosen
151 	 * at the time of the exception (for example, if multipart resolution failed)
152 	 * @return whether this resolved should proceed with resolving the exception
153 	 * for the given request and handler
154 	 * @see #setMappedHandlers
155 	 * @see #setMappedHandlerClasses
156 	 */
157 	protected boolean shouldApplyTo(HttpServletRequest request, Object handler) {
158 		if (handler != null) {
159 			if (this.mappedHandlers != null && this.mappedHandlers.contains(handler)) {
160 				return true;
161 			}
162 			if (this.mappedHandlerClasses != null) {
163 				for (Class<?> handlerClass : this.mappedHandlerClasses) {
164 					if (handlerClass.isInstance(handler)) {
165 						return true;
166 					}
167 				}
168 			}
169 		}
170 		// Else only apply if there are no explicit handler mappings.
171 		return (this.mappedHandlers == null && this.mappedHandlerClasses == null);
172 	}
173 
174 	/**
175 	 * Log the given exception at warn level, provided that warn logging has been
176 	 * activated through the {@link #setWarnLogCategory "warnLogCategory"} property.
177 	 * <p>Calls {@link #buildLogMessage} in order to determine the concrete message to log.
178 	 * Always passes the full exception to the logger.
179 	 * @param ex the exception that got thrown during handler execution
180 	 * @param request current HTTP request (useful for obtaining metadata)
181 	 * @see #setWarnLogCategory
182 	 * @see #buildLogMessage
183 	 * @see org.apache.commons.logging.Log#warn(Object, Throwable)
184 	 */
185 	protected void logException(Exception ex, HttpServletRequest request) {
186 		if (this.warnLogger != null && this.warnLogger.isWarnEnabled()) {
187 			this.warnLogger.warn(buildLogMessage(ex, request), ex);
188 		}
189 	}
190 
191 	/**
192 	 * Build a log message for the given exception, occurred during processing the given request.
193 	 * @param ex the exception that got thrown during handler execution
194 	 * @param request current HTTP request (useful for obtaining metadata)
195 	 * @return the log message to use
196 	 */
197 	protected String buildLogMessage(Exception ex, HttpServletRequest request) {
198 		return "Handler execution resulted in exception";
199 	}
200 
201 	/**
202 	 * Prepare the response for the exceptional case.
203 	 * <p>The default implementation prevents the response from being cached,
204 	 * if the {@link #setPreventResponseCaching "preventResponseCaching"} property
205 	 * has been set to "true".
206 	 * @param ex the exception that got thrown during handler execution
207 	 * @param response current HTTP response
208 	 * @see #preventCaching
209 	 */
210 	protected void prepareResponse(Exception ex, HttpServletResponse response) {
211 		if (this.preventResponseCaching) {
212 			preventCaching(response);
213 		}
214 	}
215 
216 	/**
217 	 * Prevents the response from being cached, through setting corresponding
218 	 * HTTP headers. See {@code http://www.mnot.net/cache_docs}.
219 	 * @param response current HTTP response
220 	 */
221 	protected void preventCaching(HttpServletResponse response) {
222 		response.setHeader(HEADER_PRAGMA, "no-cache");
223 		response.setDateHeader(HEADER_EXPIRES, 1L);
224 		response.setHeader(HEADER_CACHE_CONTROL, "no-cache");
225 		response.addHeader(HEADER_CACHE_CONTROL, "no-store");
226 	}
227 
228 
229 	/**
230 	 * Actually resolve the given exception that got thrown during on handler execution,
231 	 * returning a ModelAndView that represents a specific error page if appropriate.
232 	 * <p>May be overridden in subclasses, in order to apply specific exception checks.
233 	 * Note that this template method will be invoked <i>after</i> checking whether this
234 	 * resolved applies ("mappedHandlers" etc), so an implementation may simply proceed
235 	 * with its actual exception handling.
236 	 * @param request current HTTP request
237 	 * @param response current HTTP response
238 	 * @param handler the executed handler, or {@code null} if none chosen at the time
239 	 * of the exception (for example, if multipart resolution failed)
240 	 * @param ex the exception that got thrown during handler execution
241 	 * @return a corresponding ModelAndView to forward to, or {@code null} for default processing
242 	 */
243 	protected abstract ModelAndView doResolveException(HttpServletRequest request,
244 			HttpServletResponse response, Object handler, Exception ex);
245 
246 }
