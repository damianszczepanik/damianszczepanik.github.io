49  /**
50   * {@link HttpRequestHandler} that serves static resources optimized for superior browser performance
51   * (according to the guidelines of Page Speed, YSlow, etc.) by allowing for flexible cache settings
52   * ({@linkplain #setCacheSeconds "cacheSeconds" property}, last-modified support).
53   *
54   * <p>The {@linkplain #setLocations "locations" property} takes a list of Spring {@link Resource}
55   * locations from which static resources are allowed  to be served by this handler. For a given request,
56   * the list of locations will be consulted in order for the presence of the requested resource, and the
57   * first found match will be written to the response, with {@code Expires} and {@code Cache-Control}
58   * headers set as configured. The handler also properly evaluates the {@code Last-Modified} header
59   * (if present) so that a {@code 304} status code will be returned as appropriate, avoiding unnecessary
60   * overhead for resources that are already cached by the client. The use of {@code Resource} locations
61   * allows resource requests to easily be mapped to locations other than the web application root.
62   * For example, resources could be served from a classpath location such as
63   * "classpath:/META-INF/public-web-resources/", allowing convenient packaging and serving of resources
64   * such as a JavaScript library from within jar files.
65   *
66   * <p>To ensure that users with a primed browser cache get the latest changes to application-specific
67   * resources upon deployment of new versions of the application, it is recommended that a version
68   * string is used in the URL  mapping pattern that selects this handler. Such patterns can be easily
69   * parameterized using Spring EL. See the reference manual for further examples of this approach.
70   *
71   * <p>For various front-end needs &mdash; such as ensuring that users with a primed browser cache
72   * get the latest changes, or serving variations of resources (e.g., minified versions) &mdash;
73   * {@link org.springframework.web.servlet.resource.ResourceResolver}s can be configured.
74   *
75   * <p>This handler can be configured through use of a
76   * {@link org.springframework.web.servlet.config.annotation.ResourceHandlerRegistry}
77   * or the {@code <mvc:resources/>} XML configuration element.
78   *
79   * @author Keith Donald
80   * @author Jeremy Grelle
81   * @author Juergen Hoeller
82   * @since 3.0.4
83   */
84  public class ResourceHttpRequestHandler extends WebContentGenerator implements HttpRequestHandler, InitializingBean {

159 	@Override
160 	public void afterPropertiesSet() throws Exception {
161 		if (logger.isWarnEnabled() && CollectionUtils.isEmpty(this.locations)) {
162 			logger.warn("Locations list is empty. No resources will be served unless a " +
163 					"custom ResourceResolver is configured as an alternative to PathResourceResolver.");
164 		}
165 		initAllowedLocations();
166 	}
167 
168 	/**
169 	 * Look for a {@link org.springframework.web.servlet.resource.PathResourceResolver}
170 	 * among the {@link #getResourceResolvers() resource resolvers} and configure
171 	 * its {@code "allowedLocations"} to match the value of the
172 	 * {@link #setLocations(java.util.List) locations} property unless the "allowed
173 	 * locations" of the {@code PathResourceResolver} is non-empty.
174 	 */
175 	protected void initAllowedLocations() {
176 		if (CollectionUtils.isEmpty(this.locations)) {
177 			return;
178 		}
179 		for (int i = getResourceResolvers().size()-1; i >= 0; i--) {
180 			if (getResourceResolvers().get(i) instanceof PathResourceResolver) {
181 				PathResourceResolver pathResolver = (PathResourceResolver) getResourceResolvers().get(i);
182 				if (ObjectUtils.isEmpty(pathResolver.getAllowedLocations())) {
183 					pathResolver.setAllowedLocations(getLocations().toArray(new Resource[getLocations().size()]));
184 				}
185 				break;
186 			}
187 		}
188 	}

190 	/**
191 	 * Processes a resource request.
192 	 * <p>Checks for the existence of the requested resource in the configured list of locations.
193 	 * If the resource does not exist, a {@code 404} response will be returned to the client.
194 	 * If the resource exists, the request will be checked for the presence of the
195 	 * {@code Last-Modified} header, and its value will be compared against the last-modified
196 	 * timestamp of the given resource, returning a {@code 304} status code if the
197 	 * {@code Last-Modified} value  is greater. If the resource is newer than the
198 	 * {@code Last-Modified} value, or the header is not present, the content resource
199 	 * of the resource will be written to the response with caching headers
200 	 * set to expire one year in the future.
201 	 */
202 	@Override
203 	public void handleRequest(HttpServletRequest request, HttpServletResponse response)
204 			throws ServletException, IOException {
205 
206 		checkAndPrepare(request, response, true);
207 
208 		// check whether a matching resource exists
209 		Resource resource = getResource(request);
210 		if (resource == null) {
211 			logger.trace("No matching resource found - returning 404");
212 			response.sendError(HttpServletResponse.SC_NOT_FOUND);
213 			return;
214 		}
215 
216 		// check the resource's media type
217 		MediaType mediaType = getMediaType(resource);
218 		if (mediaType != null) {
219 			if (logger.isTraceEnabled()) {
220 				logger.trace("Determined media type '" + mediaType + "' for " + resource);
221 			}
222 		}
223 		else {
224 			if (logger.isTraceEnabled()) {
225 				logger.trace("No media type found for " + resource + " - not sending a content-type header");
226 			}
227 		}
228 
229 		// header phase
230 		if (new ServletWebRequest(request, response).checkNotModified(resource.lastModified())) {
231 			logger.trace("Resource not modified - returning 304");
232 			return;
233 		}
234 		setHeaders(response, resource, mediaType);
235 
236 		// content phase
237 		if (METHOD_HEAD.equals(request.getMethod())) {
238 			logger.trace("HEAD request - skipping content");
239 			return;
240 		}
241 		writeContent(response, resource);
242 	}

244 	protected Resource getResource(HttpServletRequest request) throws IOException {
245 		String path = (String) request.getAttribute(HandlerMapping.PATH_WITHIN_HANDLER_MAPPING_ATTRIBUTE);
246 		if (path == null) {
247 			throw new IllegalStateException("Required request attribute '" +
248 					HandlerMapping.PATH_WITHIN_HANDLER_MAPPING_ATTRIBUTE + "' is not set");
249 		}
250 		path = processPath(path);
251 		if (!StringUtils.hasText(path) || isInvalidPath(path)) {
252 			if (logger.isTraceEnabled()) {
253 				logger.trace("Ignoring invalid resource path [" + path + "]");
254 			}
255 			return null;
256 		}
257 		if (path.contains("%")) {
258 			try {
259 				// Use URLDecoder (vs UriUtils) to preserve potentially decoded UTF-8 chars
260 				if (isInvalidPath(URLDecoder.decode(path, "UTF-8"))) {
261 					if (logger.isTraceEnabled()) {
262 						logger.trace("Ignoring invalid resource path with escape sequences [" + path + "].");
263 					}
264 					return null;
265 				}
266 			}
267 			catch (IllegalArgumentException ex) {
268 				// ignore
269 			}
270 		}
271 		ResourceResolverChain resolveChain = new DefaultResourceResolverChain(getResourceResolvers());
272 		Resource resource = resolveChain.resolveResource(request, path, getLocations());
273 		if (resource == null || getResourceTransformers().isEmpty()) {
274 			return resource;
275 		}
276 		ResourceTransformerChain transformChain =
277 				new DefaultResourceTransformerChain(resolveChain, getResourceTransformers());
278 		resource = transformChain.transform(request, resource);
279 		return resource;
280 	}

282 	/**
283 	 * Process the given resource path to be used.
284 	 * <p>The default implementation replaces any combination of leading '/' and
285 	 * control characters (00-1F and 7F) with a single "/" or "". For example
286 	 * {@code "  // /// ////  foo/bar"} becomes {@code "/foo/bar"}.
287 	 * @since 3.2.12
288 	 */
289 	protected String processPath(String path) {
290 		boolean slash = false;
291 		for (int i = 0; i < path.length(); i++) {
292 			if (path.charAt(i) == '/') {
293 				slash = true;
294 			}
295 			else if (path.charAt(i) > ' ' && path.charAt(i) != 127) {
296 				if (i == 0 || (i == 1 && slash)) {
297 					return path;
298 				}
299 				path = slash ? "/" + path.substring(i) : path.substring(i);
300 				if (logger.isTraceEnabled()) {
301 					logger.trace("Path after trimming leading '/' and control characters: " + path);
302 				}
303 				return path;
304 			}
305 		}
306 		return (slash ? "/" : "");
307 	}

309 	/**
310 	 * Identifies invalid resource paths. By default rejects:
311 	 * <ul>
312 	 * <li>Paths that contain "WEB-INF" or "META-INF"
313 	 * <li>Paths that contain "../" after a call to
314 	 * {@link org.springframework.util.StringUtils#cleanPath}.
315 	 * <li>Paths that represent a {@link org.springframework.util.ResourceUtils#isUrl
316 	 * valid URL} or would represent one after the leading slash is removed.
317 	 * </ul>
318 	 * <p><strong>Note:</strong> this method assumes that leading, duplicate '/'
319 	 * or control characters (e.g. white space) have been trimmed so that the
320 	 * path starts predictably with a single '/' or does not have one.
321 	 * @param path the path to validate
322 	 * @return {@code true} if the path is invalid, {@code false} otherwise
323 	 */
324 	protected boolean isInvalidPath(String path) {
325 		if (logger.isTraceEnabled()) {
326 			logger.trace("Applying \"invalid path\" checks to path: " + path);
327 		}
328 		if (path.contains("WEB-INF") || path.contains("META-INF")) {
329 			if (logger.isTraceEnabled()) {
330 				logger.trace("Path contains \"WEB-INF\" or \"META-INF\".");
331 			}
332 			return true;
333 		}
334 		if (path.contains(":/")) {
335 			String relativePath = (path.charAt(0) == '/' ? path.substring(1) : path);
336 			if (ResourceUtils.isUrl(relativePath) || relativePath.startsWith("url:")) {
337 				if (logger.isTraceEnabled()) {
338 					logger.trace("Path represents URL or has \"url:\" prefix.");
339 				}
340 				return true;
341 			}
342 		}
343 		if (path.contains("../")) {
344 			path = StringUtils.cleanPath(path);
345 			if (path.contains("../")) {
346 				if (logger.isTraceEnabled()) {
347 					logger.trace("Path contains \"../\" after call to StringUtils#cleanPath.");
348 				}
349 				return true;
350 			}
351 		}
352 		return false;
353 	}

355 	/**
356 	 * Determine an appropriate media type for the given resource.
357 	 * @param resource the resource to check
358 	 * @return the corresponding media type, or {@code null} if none found
359 	 */
360 	protected MediaType getMediaType(Resource resource) {
361 		MediaType mediaType = null;
362 		String mimeType = getServletContext().getMimeType(resource.getFilename());
363 		if (StringUtils.hasText(mimeType)) {
364 			mediaType = MediaType.parseMediaType(mimeType);
365 		}
366 		if (jafPresent && (mediaType == null || MediaType.APPLICATION_OCTET_STREAM.equals(mediaType))) {
367 			MediaType jafMediaType = ActivationMediaTypeFactory.getMediaType(resource.getFilename());
368 			if (jafMediaType != null && !MediaType.APPLICATION_OCTET_STREAM.equals(jafMediaType)) {
369 				mediaType = jafMediaType;
370 			}
371 		}
372 		return mediaType;
373 	}

375 	/**
376 	 * Set headers on the given servlet response.
377 	 * Called for GET requests as well as HEAD requests.
378 	 * @param response current servlet response
379 	 * @param resource the identified resource (never {@code null})
380 	 * @param mediaType the resource's media type (never {@code null})
381 	 * @throws IOException in case of errors while setting the headers
382 	 */
383 	protected void setHeaders(HttpServletResponse response, Resource resource, MediaType mediaType) throws IOException {
384 		long length = resource.contentLength();
385 		if (length > Integer.MAX_VALUE) {
386 			throw new IOException("Resource content too long (beyond Integer.MAX_VALUE): " + resource);
387 		}
388 		response.setContentLength((int) length);
389 
390 		if (mediaType != null) {
391 			response.setContentType(mediaType.toString());
392 		}
393 
394 		if (resource instanceof EncodedResource) {
395 			response.setHeader(CONTENT_ENCODING, ((EncodedResource) resource).getContentEncoding());
396 		}
397 	}
398 
399 	/**
400 	 * Write the actual content out to the given servlet response,
401 	 * streaming the resource's content.
402 	 * @param response current servlet response
403 	 * @param resource the identified resource (never {@code null})
404 	 * @throws IOException in case of errors while writing the content
405 	 */
406 	protected void writeContent(HttpServletResponse response, Resource resource) throws IOException {
407 		InputStream in = resource.getInputStream();
408 		try {
409 			StreamUtils.copy(in, response.getOutputStream());
410 		}
411 		finally {
412 			try {
413 				in.close();
414 			}
415 			catch (IOException ex) {
416 			}
417 		}
418 	}

427 	/**
428 	 * Inner class to avoid a hard-coded JAF dependency.
429 	 */
430 	private static class ActivationMediaTypeFactory {
431 
432 		private static final FileTypeMap fileTypeMap;
433 
434 		static {
435 			fileTypeMap = loadFileTypeMapFromContextSupportModule();
436 		}
437 
438 		private static FileTypeMap loadFileTypeMapFromContextSupportModule() {
439 			// See if we can find the extended mime.types from the context-support module...
440 			Resource mappingLocation = new ClassPathResource("org/springframework/mail/javamail/mime.types");
441 			if (mappingLocation.exists()) {
442 				InputStream inputStream = null;
443 				try {
444 					inputStream = mappingLocation.getInputStream();
445 					return new MimetypesFileTypeMap(inputStream);
446 				}
447 				catch (IOException ex) {
448 					// ignore
449 				}
450 				finally {
451 					if (inputStream != null) {
452 						try {
453 							inputStream.close();
454 						}
455 						catch (IOException ex) {
456 							// ignore
457 						}
458 					}
459 				}
460 			}
461 			return FileTypeMap.getDefaultFileTypeMap();
462 		}
463 
464 		public static MediaType getMediaType(String filename) {
465 			String mediaType = fileTypeMap.getContentType(filename);
466 			return (StringUtils.hasText(mediaType) ? MediaType.parseMediaType(mediaType) : null);
467 		}
