71  	@Test
72  	public void getResource() throws Exception {
73  		this.request.setAttribute(HandlerMapping.PATH_WITHIN_HANDLER_MAPPING_ATTRIBUTE, "foo.css");
74  		this.handler.handleRequest(this.request, this.response);
75  
76  		assertEquals("text/css", this.response.getContentType());
77  		assertEquals(17, this.response.getContentLength());
78  		assertTrue(headerAsLong("Expires") >= System.currentTimeMillis() - 1000 + (3600 * 1000));
79  		assertEquals("max-age=3600, must-revalidate", this.response.getHeader("Cache-Control"));
80  		assertTrue(this.response.containsHeader("Last-Modified"));
81  		assertEquals(headerAsLong("Last-Modified"), resourceLastModified("test/foo.css"));
82  		assertEquals("h1 { color:red; }", this.response.getContentAsString());
83  	}

130 	public void invalidPath() throws Exception {
131 
132 		Resource location = new ClassPathResource("test/", getClass());
133 		this.handler.setLocations(Arrays.asList(location));
134 
135 		testInvalidPath(location, "../testsecret/secret.txt");
136 		testInvalidPath(location, "test/../../testsecret/secret.txt");
137 		testInvalidPath(location, ":/../../testsecret/secret.txt");
138 
139 		location = new UrlResource(getClass().getResource("./test/"));
140 		this.handler.setLocations(Arrays.asList(location));
141 		Resource secretResource = new UrlResource(getClass().getResource("testsecret/secret.txt"));
142 		String secretPath = secretResource.getURL().getPath();
143 
144 		testInvalidPath(location, "file:" + secretPath);
145 		testInvalidPath(location, "/file:" + secretPath);
146 		testInvalidPath(location, "url:" + secretPath);
147 		testInvalidPath(location, "/url:" + secretPath);
148 		testInvalidPath(location, "/" + secretPath);
149 		testInvalidPath(location, "////../.." + secretPath);
150 		testInvalidPath(location, "/%2E%2E/testsecret/secret.txt");
151 		testInvalidPath(location, "/  " + secretPath);
152 		testInvalidPath(location, "url:" + secretPath);
153 	}

155 	private void testInvalidPath(Resource location, String requestPath) throws Exception {
156 		this.request.setAttribute(HandlerMapping.PATH_WITHIN_HANDLER_MAPPING_ATTRIBUTE, requestPath);
157 		this.response = new MockHttpServletResponse();
158 		this.handler.handleRequest(this.request, this.response);
159 		if (!location.createRelative(requestPath).exists() && !requestPath.contains(":")) {
160 			fail(requestPath + " doesn't actually exist as a relative path");
161 		}
162 		assertEquals(404, this.response.getStatus());
163 	}

173 	@Test
174 	public void processPath() throws Exception {
175 		assertSame("/foo/bar", this.handler.processPath("/foo/bar"));
176 		assertSame("foo/bar", this.handler.processPath("foo/bar"));
177 
178 		// leading whitespace control characters (00-1F)
179 		assertEquals("/foo/bar", this.handler.processPath("  /foo/bar"));
180 		assertEquals("/foo/bar", this.handler.processPath((char) 1 + "/foo/bar"));
181 		assertEquals("/foo/bar", this.handler.processPath((char) 31 + "/foo/bar"));
182 		assertEquals("foo/bar", this.handler.processPath("  foo/bar"));
183 		assertEquals("foo/bar", this.handler.processPath((char) 31 + "foo/bar"));
184 
185 		// leading control character 0x7F (DEL)
186 		assertEquals("/foo/bar", this.handler.processPath((char) 127 + "/foo/bar"));
187 		assertEquals("/foo/bar", this.handler.processPath((char) 127 + "/foo/bar"));
188 
189 		// leading control and '/' characters
190 		assertEquals("/foo/bar", this.handler.processPath("  /  foo/bar"));
191 		assertEquals("/foo/bar", this.handler.processPath("  /  /  foo/bar"));
192 		assertEquals("/foo/bar", this.handler.processPath("  // /// ////  foo/bar"));
193 		assertEquals("/foo/bar", this.handler.processPath((char) 1 + " / " + (char) 127 + " // foo/bar"));
194 
195 		// root or empty path
196 		assertEquals("", this.handler.processPath("   "));
197 		assertEquals("/", this.handler.processPath("/"));
198 		assertEquals("/", this.handler.processPath("///"));
199 		assertEquals("/", this.handler.processPath("/ /   / "));
200 	}
