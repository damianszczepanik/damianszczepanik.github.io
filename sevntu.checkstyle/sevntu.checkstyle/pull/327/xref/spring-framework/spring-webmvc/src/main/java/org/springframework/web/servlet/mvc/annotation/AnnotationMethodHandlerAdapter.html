120 /**
121  * Implementation of the {@link org.springframework.web.servlet.HandlerAdapter} interface
122  * that maps handler methods based on HTTP paths, HTTP methods and request parameters
123  * expressed through the {@link RequestMapping} annotation.
124  *
125  * <p>Supports request parameter binding through the {@link RequestParam} annotation.
126  * Also supports the {@link ModelAttribute} annotation for exposing model attribute
127  * values to the view, as well as {@link InitBinder} for binder initialization methods
128  * and {@link SessionAttributes} for automatic session management of specific attributes.
129  *
130  * <p>This adapter can be customized through various bean properties.
131  * A common use case is to apply shared binder initialization logic through
132  * a custom {@link #setWebBindingInitializer WebBindingInitializer}.
133  *
134  * @author Juergen Hoeller
135  * @author Arjen Poutsma
136  * @since 2.5
137  * @see #setPathMatcher
138  * @see #setMethodNameResolver
139  * @see #setWebBindingInitializer
140  * @see #setSessionAttributeStore
141  *
142  * @deprecated in Spring 3.2 in favor of
143  * {@link org.springframework.web.servlet.mvc.method.annotation.RequestMappingHandlerAdapter RequestMappingHandlerAdapter}
144  */
145 @Deprecated

402 	@Override
403 	public ModelAndView handle(HttpServletRequest request, HttpServletResponse response, Object handler)
404 			throws Exception {
405 
406 		Class<?> clazz = ClassUtils.getUserClass(handler);
407 		Boolean annotatedWithSessionAttributes = this.sessionAnnotatedClassesCache.get(clazz);
408 		if (annotatedWithSessionAttributes == null) {
409 			annotatedWithSessionAttributes = (AnnotationUtils.findAnnotation(clazz, SessionAttributes.class) != null);
410 			this.sessionAnnotatedClassesCache.put(clazz, annotatedWithSessionAttributes);
411 		}
412 
413 		if (annotatedWithSessionAttributes) {
414 			// Always prevent caching in case of session attribute management.
415 			checkAndPrepare(request, response, this.cacheSecondsForSessionAttributeHandlers, true);
416 			// Prepare cached set of session attributes names.
417 		}
418 		else {
419 			// Uses configured default cacheSeconds setting.
420 			checkAndPrepare(request, response, true);
421 		}
422 
423 		// Execute invokeHandlerMethod in synchronized block if required.
424 		if (this.synchronizeOnSession) {
425 			HttpSession session = request.getSession(false);
426 			if (session != null) {
427 				Object mutex = WebUtils.getSessionMutex(session);
428 				synchronized (mutex) {
429 					return invokeHandlerMethod(request, response, handler);
430 				}
431 			}
432 		}
433 
434 		return invokeHandlerMethod(request, response, handler);
435 	}

437 	protected ModelAndView invokeHandlerMethod(HttpServletRequest request, HttpServletResponse response, Object handler)
438 			throws Exception {
439 
440 		ServletHandlerMethodResolver methodResolver = getMethodResolver(handler);
441 		Method handlerMethod = methodResolver.resolveHandlerMethod(request);
442 		ServletHandlerMethodInvoker methodInvoker = new ServletHandlerMethodInvoker(methodResolver);
443 		ServletWebRequest webRequest = new ServletWebRequest(request, response);
444 		ExtendedModelMap implicitModel = new BindingAwareModelMap();
445 
446 		Object result = methodInvoker.invokeHandlerMethod(handlerMethod, handler, webRequest, implicitModel);
447 		ModelAndView mav =
448 				methodInvoker.getModelAndView(handlerMethod, handler.getClass(), result, implicitModel, webRequest);
449 		methodInvoker.updateModelAttributes(handler, (mav != null ? mav.getModel() : null), implicitModel, webRequest);
450 		return mav;
451 	}

468 	 * Build a HandlerMethodResolver for the given handler type.
469 	 */
470 	private ServletHandlerMethodResolver getMethodResolver(Object handler) {
471 		Class<?> handlerClass = ClassUtils.getUserClass(handler);
472 		ServletHandlerMethodResolver resolver = this.methodResolverCache.get(handlerClass);
473 		if (resolver == null) {
474 			synchronized (this.methodResolverCache) {
475 				resolver = this.methodResolverCache.get(handlerClass);
476 				if (resolver == null) {
477 					resolver = new ServletHandlerMethodResolver(handlerClass);
478 					this.methodResolverCache.put(handlerClass, resolver);
479 				}
480 			}
481 		}
482 		return resolver;
483 	}

529 	 * Servlet-specific subclass of {@link HandlerMethodResolver}.
530 	 */
531 	private class ServletHandlerMethodResolver extends HandlerMethodResolver {
532 
533 		private final Map<Method, RequestMappingInfo> mappings = new HashMap<Method, RequestMappingInfo>();
534 
535 		private ServletHandlerMethodResolver(Class<?> handlerType) {
536 			init(handlerType);
537 		}
538 
539 		@Override
540 		protected boolean isHandlerMethod(Method method) {
541 			if (this.mappings.containsKey(method)) {
542 				return true;
543 			}
544 			RequestMapping mapping = AnnotationUtils.findAnnotation(method, RequestMapping.class);
545 			if (mapping != null) {
546 				String[] patterns = mapping.value();
547 				RequestMethod[] methods = new RequestMethod[0];
548 				String[] params = new String[0];
549 				String[] headers = new String[0];
550 				if (!hasTypeLevelMapping() || !Arrays.equals(mapping.method(), getTypeLevelMapping().method())) {
551 					methods = mapping.method();
552 				}
553 				if (!hasTypeLevelMapping() || !Arrays.equals(mapping.params(), getTypeLevelMapping().params())) {
554 					params = mapping.params();
555 				}
556 				if (!hasTypeLevelMapping() || !Arrays.equals(mapping.headers(), getTypeLevelMapping().headers())) {
557 					headers = mapping.headers();
558 				}
559 				RequestMappingInfo mappingInfo = new RequestMappingInfo(patterns, methods, params, headers);
560 				this.mappings.put(method, mappingInfo);
561 				return true;
562 			}
563 			return false;
564 		}
565 
566 		public Method resolveHandlerMethod(HttpServletRequest request) throws ServletException {
567 			String lookupPath = urlPathHelper.getLookupPathForRequest(request);
568 			Comparator<String> pathComparator = pathMatcher.getPatternComparator(lookupPath);
569 			Map<RequestSpecificMappingInfo, Method> targetHandlerMethods = new LinkedHashMap<RequestSpecificMappingInfo, Method>();
570 			Set<String> allowedMethods = new LinkedHashSet<String>(7);
571 			String resolvedMethodName = null;
572 			for (Method handlerMethod : getHandlerMethods()) {
573 				RequestSpecificMappingInfo mappingInfo = new RequestSpecificMappingInfo(this.mappings.get(handlerMethod));
574 				boolean match = false;
575 				if (mappingInfo.hasPatterns()) {
576 					for (String pattern : mappingInfo.getPatterns()) {
577 						if (!hasTypeLevelMapping() && !pattern.startsWith("/")) {
578 							pattern = "/" + pattern;
579 						}
580 						String combinedPattern = getCombinedPattern(pattern, lookupPath, request);
581 						if (combinedPattern != null) {
582 							if (mappingInfo.matches(request)) {
583 								match = true;
584 								mappingInfo.addMatchedPattern(combinedPattern);
585 							}
586 							else {
587 								if (!mappingInfo.matchesRequestMethod(request)) {
588 									allowedMethods.addAll(mappingInfo.methodNames());
589 								}
590 								break;
591 							}
592 						}
593 					}
594 					mappingInfo.sortMatchedPatterns(pathComparator);
595 				}
596 				else if (useTypeLevelMapping(request)) {
597 					String[] typeLevelPatterns = getTypeLevelMapping().value();
598 					for (String typeLevelPattern : typeLevelPatterns) {
599 						if (!typeLevelPattern.startsWith("/")) {
600 							typeLevelPattern = "/" + typeLevelPattern;
601 						}
602 						boolean useSuffixPattern = useSuffixPattern(request);
603 						if (getMatchingPattern(typeLevelPattern, lookupPath, useSuffixPattern) != null) {
604 							if (mappingInfo.matches(request)) {
605 								match = true;
606 								mappingInfo.addMatchedPattern(typeLevelPattern);
607 							}
608 							else {
609 								if (!mappingInfo.matchesRequestMethod(request)) {
610 									allowedMethods.addAll(mappingInfo.methodNames());
611 								}
612 								break;
613 							}
614 						}
615 					}
616 					mappingInfo.sortMatchedPatterns(pathComparator);
617 				}
618 				else {
619 					// No paths specified: parameter match sufficient.
620 					match = mappingInfo.matches(request);
621 					if (match && mappingInfo.getMethodCount() == 0 && mappingInfo.getParamCount() == 0 &&
622 							resolvedMethodName != null && !resolvedMethodName.equals(handlerMethod.getName())) {
623 						match = false;
624 					}
625 					else {
626 						if (!mappingInfo.matchesRequestMethod(request)) {
627 							allowedMethods.addAll(mappingInfo.methodNames());
628 						}
629 					}
630 				}
631 				if (match) {
632 					Method oldMappedMethod = targetHandlerMethods.put(mappingInfo, handlerMethod);
633 					if (oldMappedMethod != null && oldMappedMethod != handlerMethod) {
634 						if (methodNameResolver != null && !mappingInfo.hasPatterns()) {
635 							if (!oldMappedMethod.getName().equals(handlerMethod.getName())) {
636 								if (resolvedMethodName == null) {
637 									resolvedMethodName = methodNameResolver.getHandlerMethodName(request);
638 								}
639 								if (!resolvedMethodName.equals(oldMappedMethod.getName())) {
640 									oldMappedMethod = null;
641 								}
642 								if (!resolvedMethodName.equals(handlerMethod.getName())) {
643 									if (oldMappedMethod != null) {
644 										targetHandlerMethods.put(mappingInfo, oldMappedMethod);
645 										oldMappedMethod = null;
646 									}
647 									else {
648 										targetHandlerMethods.remove(mappingInfo);
649 									}
650 								}
651 							}
652 						}
653 						if (oldMappedMethod != null) {
654 							throw new IllegalStateException(
655 									"Ambiguous handler methods mapped for HTTP path '" + lookupPath + "': {" +
656 									oldMappedMethod + ", " + handlerMethod +
657 									"}. If you intend to handle the same path in multiple methods, then factor " +
658 									"them out into a dedicated handler class with that path mapped at the type level!");
659 						}
660 					}
661 				}
662 			}
663 			if (!targetHandlerMethods.isEmpty()) {
664 				List<RequestSpecificMappingInfo> matches = new ArrayList<RequestSpecificMappingInfo>(targetHandlerMethods.keySet());
665 				RequestSpecificMappingInfoComparator requestMappingInfoComparator =
666 						new RequestSpecificMappingInfoComparator(pathComparator, request);
667 				Collections.sort(matches, requestMappingInfoComparator);
668 				RequestSpecificMappingInfo bestMappingMatch = matches.get(0);
669 				String bestMatchedPath = bestMappingMatch.bestMatchedPattern();
670 				if (bestMatchedPath != null) {
671 					extractHandlerMethodUriTemplates(bestMatchedPath, lookupPath, request);
672 				}
673 				return targetHandlerMethods.get(bestMappingMatch);
674 			}
675 			else {
676 				if (!allowedMethods.isEmpty()) {
677 					throw new HttpRequestMethodNotSupportedException(request.getMethod(), StringUtils.toStringArray(allowedMethods));
678 				}
679 				throw new NoSuchRequestHandlingMethodException(lookupPath, request.getMethod(), request.getParameterMap());
680 			}
681 		}

696 		/**
697 		 * Determines the combined pattern for the given methodLevelPattern and path.
698 		 * <p>Uses the following algorithm:
699 		 * <ol>
700 		 * <li>If there is a type-level mapping with path information, it is {@linkplain
701 		 * PathMatcher#combine(String, String) combined} with the method-level pattern.</li>
702 		 * <li>If there is a {@linkplain HandlerMapping#BEST_MATCHING_PATTERN_ATTRIBUTE best matching pattern}
703 		 * in the request, it is combined with the method-level pattern.</li>
704 		 * <li>Otherwise, the method-level pattern is returned.</li>
705 		 * </ol>
706 		 */
707 		private String getCombinedPattern(String methodLevelPattern, String lookupPath, HttpServletRequest request) {
708 			boolean useSuffixPattern = useSuffixPattern(request);
709 			if (useTypeLevelMapping(request)) {
710 				String[] typeLevelPatterns = getTypeLevelMapping().value();
711 				for (String typeLevelPattern : typeLevelPatterns) {
712 					if (!typeLevelPattern.startsWith("/")) {
713 						typeLevelPattern = "/" + typeLevelPattern;
714 					}
715 					String combinedPattern = pathMatcher.combine(typeLevelPattern, methodLevelPattern);
716 					String matchingPattern = getMatchingPattern(combinedPattern, lookupPath, useSuffixPattern);
717 					if (matchingPattern != null) {
718 						return matchingPattern;
719 					}
720 				}
721 				return null;
722 			}
723 			String bestMatchingPattern = (String) request.getAttribute(HandlerMapping.BEST_MATCHING_PATTERN_ATTRIBUTE);
724 			if (StringUtils.hasText(bestMatchingPattern) && bestMatchingPattern.endsWith("*")) {
725 				String combinedPattern = pathMatcher.combine(bestMatchingPattern, methodLevelPattern);
726 				String matchingPattern = getMatchingPattern(combinedPattern, lookupPath, useSuffixPattern);
727 				if (matchingPattern != null && !matchingPattern.equals(bestMatchingPattern)) {
728 					return matchingPattern;
729 				}
730 			}
731 			return getMatchingPattern(methodLevelPattern, lookupPath, useSuffixPattern);
732 		}
733 
734 		private String getMatchingPattern(String pattern, String lookupPath, boolean useSuffixPattern) {
735 			if (pattern.equals(lookupPath)) {
736 				return pattern;
737 			}
738 			boolean hasSuffix = pattern.indexOf('.') != -1;
739 			if (useSuffixPattern && !hasSuffix) {
740 				String patternWithSuffix = pattern + ".*";
741 				if (pathMatcher.match(patternWithSuffix, lookupPath)) {
742 					return patternWithSuffix;
743 				}
744 			}
745 			if (pathMatcher.match(pattern, lookupPath)) {
746 				return pattern;
747 			}
748 			boolean endsWithSlash = pattern.endsWith("/");
749 			if (useSuffixPattern && !endsWithSlash) {
750 				String patternWithSlash = pattern + "/";
751 				if (pathMatcher.match(patternWithSlash, lookupPath)) {
752 					return patternWithSlash;
753 				}
754 			}
755 			return null;
756 		}

758 		@SuppressWarnings("unchecked")
759 		private void extractHandlerMethodUriTemplates(String mappedPattern, String lookupPath, HttpServletRequest request) {
760 			Map<String, String> variables =
761 					(Map<String, String>) request.getAttribute(HandlerMapping.URI_TEMPLATE_VARIABLES_ATTRIBUTE);
762 			int patternVariableCount = StringUtils.countOccurrencesOf(mappedPattern, "{");
763 			if ((variables == null || patternVariableCount != variables.size()) && pathMatcher.match(mappedPattern, lookupPath)) {
764 				variables = pathMatcher.extractUriTemplateVariables(mappedPattern, lookupPath);
765 				Map<String, String> decodedVariables = urlPathHelper.decodePathVariables(request, variables);
766 				request.setAttribute(HandlerMapping.URI_TEMPLATE_VARIABLES_ATTRIBUTE, decodedVariables);
767 			}
768 		}

914 		@SuppressWarnings("unchecked")
915 		public ModelAndView getModelAndView(Method handlerMethod, Class<?> handlerType, Object returnValue,
916 				ExtendedModelMap implicitModel, ServletWebRequest webRequest) throws Exception {
917 
918 			ResponseStatus responseStatusAnn = AnnotationUtils.findAnnotation(handlerMethod, ResponseStatus.class);
919 			if (responseStatusAnn != null) {
920 				HttpStatus responseStatus = responseStatusAnn.value();
921 				String reason = responseStatusAnn.reason();
922 				if (!StringUtils.hasText(reason)) {
923 					webRequest.getResponse().setStatus(responseStatus.value());
924 				}
925 				else {
926 					webRequest.getResponse().sendError(responseStatus.value(), reason);
927 				}
928 
929 				// to be picked up by the RedirectView
930 				webRequest.getRequest().setAttribute(View.RESPONSE_STATUS_ATTRIBUTE, responseStatus);
931 
932 				responseArgumentUsed = true;
933 			}
934 
935 			// Invoke custom resolvers if present...
936 			if (customModelAndViewResolvers != null) {
937 				for (ModelAndViewResolver mavResolver : customModelAndViewResolvers) {
938 					ModelAndView mav = mavResolver.resolveModelAndView(
939 							handlerMethod, handlerType, returnValue, implicitModel, webRequest);
940 					if (mav != ModelAndViewResolver.UNRESOLVED) {
941 						return mav;
942 					}
943 				}
944 			}
945 
946 			if (returnValue instanceof HttpEntity) {
947 				handleHttpEntityResponse((HttpEntity<?>) returnValue, webRequest);
948 				return null;
949 			}
950 			else if (AnnotationUtils.findAnnotation(handlerMethod, ResponseBody.class) != null) {
951 				handleResponseBody(returnValue, webRequest);
952 				return null;
953 			}
954 			else if (returnValue instanceof ModelAndView) {
955 				ModelAndView mav = (ModelAndView) returnValue;
956 				mav.getModelMap().mergeAttributes(implicitModel);
957 				return mav;
958 			}
959 			else if (returnValue instanceof Model) {
960 				return new ModelAndView().addAllObjects(implicitModel).addAllObjects(((Model) returnValue).asMap());
961 			}
962 			else if (returnValue instanceof View) {
963 				return new ModelAndView((View) returnValue).addAllObjects(implicitModel);
964 			}
965 			else if (AnnotationUtils.findAnnotation(handlerMethod, ModelAttribute.class) != null) {
966 				addReturnValueAsModelAttribute(handlerMethod, handlerType, returnValue, implicitModel);
967 				return new ModelAndView().addAllObjects(implicitModel);
968 			}
969 			else if (returnValue instanceof Map) {
970 				return new ModelAndView().addAllObjects(implicitModel).addAllObjects((Map<String, ?>) returnValue);
971 			}
972 			else if (returnValue instanceof String) {
973 				return new ModelAndView((String) returnValue).addAllObjects(implicitModel);
974 			}
975 			else if (returnValue == null) {
976 				// Either returned null or was 'void' return.
977 				if (this.responseArgumentUsed || webRequest.isNotModified()) {
978 					return null;
979 				}
980 				else {
981 					// Assuming view name translation...
982 					return new ModelAndView().addAllObjects(implicitModel);
983 				}
984 			}
985 			else if (!BeanUtils.isSimpleProperty(returnValue.getClass())) {
986 				// Assume a single model attribute...
987 				addReturnValueAsModelAttribute(handlerMethod, handlerType, returnValue, implicitModel);
988 				return new ModelAndView().addAllObjects(implicitModel);
989 			}
990 			else {
991 				throw new IllegalArgumentException("Invalid handler method return value: " + returnValue);
992 			}
993 		}

1005 		private void handleHttpEntityResponse(HttpEntity<?> responseEntity, ServletWebRequest webRequest)
1006 				throws Exception {
1007 			if (responseEntity == null) {
1008 				return;
1009 			}
1010 			HttpInputMessage inputMessage = createHttpInputMessage(webRequest);
1011 			HttpOutputMessage outputMessage = createHttpOutputMessage(webRequest);
1012 			if (responseEntity instanceof ResponseEntity && outputMessage instanceof ServerHttpResponse) {
1013 				((ServerHttpResponse) outputMessage).setStatusCode(((ResponseEntity<?>) responseEntity).getStatusCode());
1014 			}

1016 			if (!entityHeaders.isEmpty()) {
1017 				outputMessage.getHeaders().putAll(entityHeaders);
1018 			}

1021 				writeWithMessageConverters(body, inputMessage, outputMessage);

1026 			}

1029 		@SuppressWarnings({ "unchecked", "rawtypes" })
1030 		private void writeWithMessageConverters(Object returnValue,
1031 				HttpInputMessage inputMessage, HttpOutputMessage outputMessage)
1032 				throws IOException, HttpMediaTypeNotAcceptableException {
1033 			List<MediaType> acceptedMediaTypes = inputMessage.getHeaders().getAccept();
1034 			if (acceptedMediaTypes.isEmpty()) {
1035 				acceptedMediaTypes = Collections.singletonList(MediaType.ALL);
1036 			}
1037 			MediaType.sortByQualityValue(acceptedMediaTypes);
1038 			Class<?> returnValueType = returnValue.getClass();
1039 			List<MediaType> allSupportedMediaTypes = new ArrayList<MediaType>();
1040 			if (getMessageConverters() != null) {
1041 				for (MediaType acceptedMediaType : acceptedMediaTypes) {
1042 					for (HttpMessageConverter messageConverter : getMessageConverters()) {
1043 						if (messageConverter.canWrite(returnValueType, acceptedMediaType)) {
1044 							messageConverter.write(returnValue, acceptedMediaType, outputMessage);
1045 							if (logger.isDebugEnabled()) {
1046 								MediaType contentType = outputMessage.getHeaders().getContentType();
1047 								if (contentType == null) {
1048 									contentType = acceptedMediaType;
1049 								}
1050 								logger.debug("Written [" + returnValue + "] as \"" + contentType +
1051 										"\" using [" + messageConverter + "]");
1052 							}
1053 							this.responseArgumentUsed = true;
1054 							return;
1055 						}
1056 					}
1057 				}
1058 				for (HttpMessageConverter messageConverter : messageConverters) {
1059 					allSupportedMediaTypes.addAll(messageConverter.getSupportedMediaTypes());
1060 				}
1061 			}
1062 			throw new HttpMediaTypeNotAcceptableException(allSupportedMediaTypes);
1063 		}

1195 	/**
1196 	 * Comparator capable of sorting {@link RequestSpecificMappingInfo}s (RHIs) so that
1197 	 * sorting a list with this comparator will result in:
1198 	 * <ul>
1199 	 * <li>RHIs with {@linkplain AnnotationMethodHandlerAdapter.RequestSpecificMappingInfo#matchedPatterns better matched paths}
1200 	 * take precedence over those with a weaker match (as expressed by the {@linkplain PathMatcher#getPatternComparator(String)
1201 	 * path pattern comparator}.) Typically, this means that patterns without wild cards and uri templates
1202 	 * will be ordered before those without.</li>
1203 	 * <li>RHIs with one single {@linkplain RequestMappingInfo#methods request method} will be
1204 	 * ordered before those without a method, or with more than one method.</li>
1205 	 * <li>RHIs with more {@linkplain RequestMappingInfo#params request parameters} will be ordered
1206 	 * before those with less parameters</li>
1207 	 * </ol>
1208 	 */
1209 	static class RequestSpecificMappingInfoComparator implements Comparator<RequestSpecificMappingInfo> {
1210 
1211 		private final Comparator<String> pathComparator;
1212 
1213 		private final ServerHttpRequest request;
1214 
1215 		RequestSpecificMappingInfoComparator(Comparator<String> pathComparator, HttpServletRequest request) {
1216 			this.pathComparator = pathComparator;
1217 			this.request = new ServletServerHttpRequest(request);
1218 		}
1219 
1220 		@Override
1221 		public int compare(RequestSpecificMappingInfo info1, RequestSpecificMappingInfo info2) {
1222 			int pathComparison = pathComparator.compare(info1.bestMatchedPattern(), info2.bestMatchedPattern());
1223 			if (pathComparison != 0) {
1224 				return pathComparison;
1225 			}
1226 			int info1ParamCount = info1.getParamCount();
1227 			int info2ParamCount = info2.getParamCount();
1228 			if (info1ParamCount != info2ParamCount) {
1229 				return info2ParamCount - info1ParamCount;
1230 			}
1231 			int info1HeaderCount = info1.getHeaderCount();
1232 			int info2HeaderCount = info2.getHeaderCount();
1233 			if (info1HeaderCount != info2HeaderCount) {
1234 				return info2HeaderCount - info1HeaderCount;
1235 			}
1236 			int acceptComparison = compareAcceptHeaders(info1, info2);
1237 			if (acceptComparison != 0) {
1238 				return acceptComparison;
1239 			}
1240 			int info1MethodCount = info1.getMethodCount();
1241 			int info2MethodCount = info2.getMethodCount();
1242 			if (info1MethodCount == 0 && info2MethodCount > 0) {
1243 				return 1;
1244 			}
1245 			else if (info2MethodCount == 0 && info1MethodCount > 0) {
1246 				return -1;
1247 			}
1248 			else if (info1MethodCount == 1 & info2MethodCount > 1) {
1249 				return -1;
1250 			}
1251 			else if (info2MethodCount == 1 & info1MethodCount > 1) {
1252 				return 1;
1253 			}
1254 			return 0;
1255 		}
1256 
1257 		private int compareAcceptHeaders(RequestMappingInfo info1, RequestMappingInfo info2) {
1258 			List<MediaType> requestAccepts = request.getHeaders().getAccept();
1259 			MediaType.sortByQualityValue(requestAccepts);
1260 
1261 			List<MediaType> info1Accepts = getAcceptHeaderValue(info1);
1262 			List<MediaType> info2Accepts = getAcceptHeaderValue(info2);
1263 
1264 			for (MediaType requestAccept : requestAccepts) {
1265 				int pos1 = indexOfIncluded(info1Accepts, requestAccept);
1266 				int pos2 = indexOfIncluded(info2Accepts, requestAccept);
1267 				if (pos1 != pos2) {
1268 					return pos2 - pos1;
1269 				}
1270 			}
1271 			return 0;
1272 		}
1273 
1274 		private int indexOfIncluded(List<MediaType> infoAccepts, MediaType requestAccept) {
1275 			for (int i = 0; i < infoAccepts.size(); i++) {
1276 				MediaType info1Accept = infoAccepts.get(i);
1277 				if (requestAccept.includes(info1Accept)) {
1278 					return i;
1279 				}
1280 			}
1281 			return -1;
1282 		}
1283 
1284 		private List<MediaType> getAcceptHeaderValue(RequestMappingInfo info) {
1285 			for (String header : info.headers) {
1286 				int separator = header.indexOf('=');
1287 				if (separator != -1) {
1288 					String key = header.substring(0, separator);
1289 					String value = header.substring(separator + 1);
1290 					if ("Accept".equalsIgnoreCase(key)) {
1291 						return MediaType.parseMediaTypes(value);
1292 					}
1293 				}
1294 			}
1295 			return Collections.emptyList();
1296 		}
1297 	}
