135 	@Test
136 	public void handle() throws Exception {
137 
138 		Class<?>[] parameterTypes = new Class<?>[] { int.class, String.class, String.class, String.class, Map.class,
139 				Date.class, Map.class, String.class, String.class, TestBean.class, Errors.class, TestBean.class,
140 				Color.class, HttpServletRequest.class, HttpServletResponse.class, User.class, OtherUser.class,
141 				Model.class, UriComponentsBuilder.class };
142 
143 		String datePattern = "yyyy.MM.dd";
144 		String formattedDate = "2011.03.16";
145 		Date date = new GregorianCalendar(2011, Calendar.MARCH, 16).getTime();
146 
147 		request.addHeader("Content-Type", "text/plain; charset=utf-8");
148 		request.addHeader("header", "headerValue");
149 		request.addHeader("anotherHeader", "anotherHeaderValue");
150 		request.addParameter("datePattern", datePattern);
151 		request.addParameter("dateParam", formattedDate);
152 		request.addParameter("paramByConvention", "paramByConventionValue");
153 		request.addParameter("age", "25");
154 		request.setCookies(new Cookie("cookie", "99"));
155 		request.setContent("Hello World".getBytes("UTF-8"));
156 		request.setUserPrincipal(new User());
157 		request.setContextPath("/contextPath");
158 		request.setServletPath("/main");
159 		System.setProperty("systemHeader", "systemHeaderValue");
160 		Map<String, String> uriTemplateVars = new HashMap<String, String>();
161 		uriTemplateVars.put("pathvar", "pathvarValue");
162 		request.setAttribute(HandlerMapping.URI_TEMPLATE_VARIABLES_ATTRIBUTE, uriTemplateVars);
163 
164 		HandlerMethod handlerMethod = handlerMethod("handle", parameterTypes);
165 		ModelAndView mav = handlerAdapter.handle(request, response, handlerMethod);
166 		ModelMap model = mav.getModelMap();
167 
168 		assertEquals("viewName", mav.getViewName());
169 		assertEquals(99, model.get("cookie"));
170 		assertEquals("pathvarValue", model.get("pathvar"));
171 		assertEquals("headerValue", model.get("header"));
172 		assertEquals(date, model.get("dateParam"));
173 
174 		Map<?,?> map = (Map<?,?>) model.get("headerMap");
175 		assertEquals("headerValue", map.get("header"));
176 		assertEquals("anotherHeaderValue", map.get("anotherHeader"));
177 		assertEquals("systemHeaderValue", model.get("systemHeader"));
178 
179 		map = (Map<?,?>) model.get("paramMap");
180 		assertEquals(formattedDate, map.get("dateParam"));
181 		assertEquals("paramByConventionValue", map.get("paramByConvention"));
182 
183 		assertEquals("/contextPath", model.get("value"));
184 
185 		TestBean modelAttr = (TestBean) model.get("modelAttr");
186 		assertEquals(25, modelAttr.getAge());
187 		assertEquals("Set by model method [modelAttr]", modelAttr.getName());
188 		assertSame(modelAttr, request.getSession().getAttribute("modelAttr"));
189 
190 		BindingResult bindingResult = (BindingResult) model.get(BindingResult.MODEL_KEY_PREFIX + "modelAttr");
191 		assertSame(modelAttr, bindingResult.getTarget());
192 		assertEquals(1, bindingResult.getErrorCount());
193 
194 		String conventionAttrName = "testBean";
195 		TestBean modelAttrByConvention = (TestBean) model.get(conventionAttrName);
196 		assertEquals(25, modelAttrByConvention.getAge());
197 		assertEquals("Set by model method [modelAttrByConvention]", modelAttrByConvention.getName());
198 		assertSame(modelAttrByConvention, request.getSession().getAttribute(conventionAttrName));
199 
200 		bindingResult = (BindingResult) model.get(BindingResult.MODEL_KEY_PREFIX + conventionAttrName);
201 		assertSame(modelAttrByConvention, bindingResult.getTarget());
202 
203 		assertTrue(model.get("customArg") instanceof Color);
204 		assertEquals(User.class, model.get("user").getClass());
205 		assertEquals(OtherUser.class, model.get("otherUser").getClass());
206 
207 		assertEquals(new URI("http://localhost/contextPath/main/path"), model.get("url"));
208 	}

294 	private HandlerMethod handlerMethod(String methodName, Class<?>... paramTypes) throws Exception {
295 		Method method = handler.getClass().getDeclaredMethod(methodName, paramTypes);
296 		return new InvocableHandlerMethod(handler, method);
297 	}

303 		@InitBinder("dateParam")
304 		public void initBinder(WebDataBinder dataBinder, @RequestParam("datePattern") String datePattern) {
305 			SimpleDateFormat dateFormat = new SimpleDateFormat(datePattern);
306 			dataBinder.registerCustomEditor(Date.class, new CustomDateEditor(dateFormat, false));
307 		}

309 		@ModelAttribute
310 		public void model(Model model) {
311 			TestBean modelAttr = new TestBean();
312 			modelAttr.setName("Set by model method [modelAttr]");
313 			model.addAttribute("modelAttr", modelAttr);
314 
315 			modelAttr = new TestBean();
316 			modelAttr.setName("Set by model method [modelAttrByConvention]");
317 			model.addAttribute(modelAttr);
318 
319 			model.addAttribute(new OtherUser());
320 		}
321 
322 		public String handle(
323 							@CookieValue("cookie") int cookie,
324 						 	@PathVariable("pathvar") String pathvar,
325 						 	@RequestHeader("header") String header,
326 						 	@RequestHeader(defaultValue="#{systemProperties.systemHeader}") String systemHeader,
327 						 	@RequestHeader Map<String, Object> headerMap,
328 						 	@RequestParam("dateParam") Date dateParam,
329 						 	@RequestParam Map<String, Object> paramMap,
330 						 	String paramByConvention,
331 						 	@Value("#{request.contextPath}") String value,
332 						 	@ModelAttribute("modelAttr") @Valid TestBean modelAttr,
333 						 	Errors errors,
334 						 	TestBean modelAttrByConvention,
335 						 	Color customArg,
336 						 	HttpServletRequest request,
337 						 	HttpServletResponse response,
338 						 	User user,
339 						 	@ModelAttribute OtherUser otherUser,
340 						 	Model model,
341 						 	UriComponentsBuilder builder) throws Exception {
342 
343 			model.addAttribute("cookie", cookie).addAttribute("pathvar", pathvar).addAttribute("header", header)
344 					.addAttribute("systemHeader", systemHeader).addAttribute("headerMap", headerMap)
345 					.addAttribute("dateParam", dateParam).addAttribute("paramMap", paramMap)
346 					.addAttribute("paramByConvention", paramByConvention).addAttribute("value", value)
347 					.addAttribute("customArg", customArg).addAttribute(user)
348 					.addAttribute("url", builder.path("/path").build().toUri());
349 
350 			assertNotNull(request);
351 			assertNotNull(response);
352 
353 			return "viewName";
354 		}
355 
356 		@ResponseStatus(value=HttpStatus.ACCEPTED)
357 		@ResponseBody
358 		public String handleRequestBody(@RequestBody byte[] bytes) throws Exception {
359 			String requestBody = new String(bytes, "UTF-8");
360 			return "Handled requestBody=[" + requestBody + "]";
361 		}
362 
363 		@ResponseStatus(value=HttpStatus.ACCEPTED)
364 		@ResponseBody
365 		public String handleAndValidateRequestBody(@Valid TestBean modelAttr, Errors errors) throws Exception {
366 			return "Error count [" + errors.getErrorCount() + "]";
367 		}
368 
369 		public ResponseEntity<String> handleHttpEntity(HttpEntity<byte[]> httpEntity) throws Exception {
370 			HttpHeaders responseHeaders = new HttpHeaders();
371 			responseHeaders.set("header", "headerValue");
372 			String responseBody = "Handled requestBody=[" + new String(httpEntity.getBody(), "UTF-8") + "]";
373 			return new ResponseEntity<String>(responseBody, responseHeaders, HttpStatus.ACCEPTED);
374 		}
