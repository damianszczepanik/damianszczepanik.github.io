16  package org.springframework.web.servlet.view.tiles3;
17  
18  import java.util.Locale;
19  import javax.servlet.http.HttpServletRequest;
20  
21  import org.apache.tiles.locale.impl.DefaultLocaleResolver;
22  import org.apache.tiles.request.Request;
23  import org.apache.tiles.request.servlet.NotAServletEnvironmentException;
24  import org.apache.tiles.request.servlet.ServletUtil;
25  
26  import org.springframework.web.servlet.support.RequestContextUtils;
27  
28  /**
29   * Tiles LocaleResolver adapter that delegates to a Spring
30   * {@link org.springframework.web.servlet.LocaleResolver}, exposing the
31   * DispatcherServlet-managed locale.
32   *
33   * <p>This adapter gets automatically registered by {@link TilesConfigurer}.
34   *
35   * @author Nicolas Le Bas
36   * @since 3.2
37   * @see org.apache.tiles.definition.UrlDefinitionsFactory#LOCALE_RESOLVER_IMPL_PROPERTY
38   */
39  public class SpringLocaleResolver extends DefaultLocaleResolver {
40  
41  	@Override
42  	public Locale resolveLocale(Request request) {
43  		try {
44  			HttpServletRequest servletRequest = ServletUtil.getServletRequest(request).getRequest();
45  			if (servletRequest != null) {
46  				return RequestContextUtils.getLocale(servletRequest);
47  			}
48  		}
49  		catch (NotAServletEnvironmentException e) {
50  			// Ignore
51  		}
52  		return super.resolveLocale(request);
53  	}
54  
55  }
