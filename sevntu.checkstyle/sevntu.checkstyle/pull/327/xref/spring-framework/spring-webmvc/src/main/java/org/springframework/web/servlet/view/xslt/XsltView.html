55  /**
56   * XSLT-driven View that allows for response context to be rendered as the
57   * result of an XSLT transformation.
58   *
59   * <p>The XSLT Source object is supplied as a parameter in the model and then
60   * {@link #locateSource detected} during response rendering. Users can either specify
61   * a specific entry in the model via the {@link #setSourceKey sourceKey} property or
62   * have Spring locate the Source object. This class also provides basic conversion
63   * of objects into Source implementations. See {@link #getSourceTypes() here}
64   * for more details.
65   *
66   * <p>All model parameters are passed to the XSLT Transformer as parameters.
67   * In addition the user can configure {@link #setOutputProperties output properties}
68   * to be passed to the Transformer.
69   *
70   * @author Rob Harrop
71   * @author Juergen Hoeller
72   * @since 2.0
73   */
74  public class XsltView extends AbstractUrlBasedView {

170 	/**
171 	 * Initialize this XsltView's TransformerFactory.
172 	 */
173 	@Override
174 	protected void initApplicationContext() throws BeansException {
175 		this.transformerFactory = newTransformerFactory(this.transformerFactoryClass);
176 		this.transformerFactory.setErrorListener(this.errorListener);
177 		if (this.uriResolver != null) {
178 			this.transformerFactory.setURIResolver(this.uriResolver);
179 		}
180 		if (this.cacheTemplates) {
181 			this.cachedTemplates = loadTemplates();
182 		}
183 	}

185 	/**
186 	 * Instantiate a new TransformerFactory for this view.
187 	 * <p>The default implementation simply calls
188 	 * {@link javax.xml.transform.TransformerFactory#newInstance()}.
189 	 * If a {@link #setTransformerFactoryClass "transformerFactoryClass"}
190 	 * has been specified explicitly, the default constructor of the
191 	 * specified class will be called instead.
192 	 * <p>Can be overridden in subclasses.
193 	 * @param transformerFactoryClass the specified factory class (if any)
194 	 * @return the new TransactionFactory instance
195 	 * @see #setTransformerFactoryClass
196 	 * @see #getTransformerFactory()
197 	 */
198 	protected TransformerFactory newTransformerFactory(Class<?> transformerFactoryClass) {
199 		if (transformerFactoryClass != null) {
200 			try {
201 				return (TransformerFactory) transformerFactoryClass.newInstance();
202 			}
203 			catch (Exception ex) {
204 				throw new TransformerFactoryConfigurationError(ex, "Could not instantiate TransformerFactory");
205 			}
206 		}
207 		else {
208 			return TransformerFactory.newInstance();
209 		}
210 	}

213 	 * Return the TransformerFactory that this XsltView uses.
214 	 * @return the TransformerFactory (never {@code null})
215 	 */
216 	protected final TransformerFactory getTransformerFactory() {
217 		return this.transformerFactory;
218 	}

221 	@Override
222 	protected void renderMergedOutputModel(
223 			Map<String, Object> model, HttpServletRequest request, HttpServletResponse response)
224 			throws Exception {
225 
226 		Templates templates = this.cachedTemplates;
227 		if (templates == null) {
228 			templates = loadTemplates();
229 		}
230 
231 		Transformer transformer = createTransformer(templates);
232 		configureTransformer(model, response, transformer);
233 		configureResponse(model, response, transformer);
234 		Source source = null;
235 		try {
236 			source = locateSource(model);
237 			if (source == null) {
238 				throw new IllegalArgumentException("Unable to locate Source object in model: " + model);
239 			}
240 			transformer.transform(source, createResult(response));
241 		}
242 		finally {
243 			closeSourceIfNecessary(source);
244 		}
245 	}

247 	/**
248 	 * Create the XSLT {@link Result} used to render the result of the transformation.
249 	 * <p>The default implementation creates a {@link StreamResult} wrapping the supplied
250 	 * HttpServletResponse's {@link HttpServletResponse#getOutputStream() OutputStream}.
251 	 * @param response current HTTP response
252 	 * @return the XSLT Result to use
253 	 * @throws Exception if the Result cannot be built
254 	 */
255 	protected Result createResult(HttpServletResponse response) throws Exception {
256 		return new StreamResult(response.getOutputStream());
257 	}
258 
259 	/**
260 	 * <p>Locate the {@link Source} object in the supplied model,
261 	 * converting objects as required.
262 	 * The default implementation first attempts to look under the configured
263 	 * {@link #setSourceKey source key}, if any, before attempting to locate
264 	 * an object of {@link #getSourceTypes() supported type}.
265 	 * @param model the merged model Map
266 	 * @return the XSLT Source object (or {@code null} if none found)
267 	 * @throws Exception if an error occurred during locating the source
268 	 * @see #setSourceKey
269 	 * @see #convertSource
270 	 */
271 	protected Source locateSource(Map<String, Object> model) throws Exception {
272 		if (this.sourceKey != null) {
273 			return convertSource(model.get(this.sourceKey));
274 		}
275 		Object source = CollectionUtils.findValueOfType(model.values(), getSourceTypes());
276 		return (source != null ? convertSource(source) : null);
277 	}
278 
279 	/**
280 	 * Return the array of {@link Class Classes} that are supported when converting to an
281 	 * XSLT {@link Source}.
282 	 * <p>Currently supports {@link Source}, {@link Document}, {@link Node},
283 	 * {@link Reader}, {@link InputStream} and {@link Resource}.
284 	 * @return the supported source types
285 	 */
286 	protected Class<?>[] getSourceTypes() {
287 		return new Class<?>[] {Source.class, Document.class, Node.class, Reader.class, InputStream.class, Resource.class};
288 	}
289 
290 	/**
291 	 * Convert the supplied {@link Object} into an XSLT {@link Source} if the
292 	 * {@link Object} type is {@link #getSourceTypes() supported}.
293 	 * @param source the original source object
294 	 * @return the adapted XSLT Source
295 	 * @throws IllegalArgumentException if the given Object is not of a supported type
296 	 */
297 	protected Source convertSource(Object source) throws Exception {
298 		if (source instanceof Source) {
299 			return (Source) source;
300 		}
301 		else if (source instanceof Document) {
302 			return new DOMSource(((Document) source).getDocumentElement());
303 		}
304 		else if (source instanceof Node) {
305 			return new DOMSource((Node) source);
306 		}
307 		else if (source instanceof Reader) {
308 			return new StreamSource((Reader) source);
309 		}
310 		else if (source instanceof InputStream) {
311 			return new StreamSource((InputStream) source);
312 		}
313 		else if (source instanceof Resource) {
314 			Resource resource = (Resource) source;
315 			return new StreamSource(resource.getInputStream(), resource.getURI().toASCIIString());
316 		}
317 		else {
318 			throw new IllegalArgumentException("Value '" + source + "' cannot be converted to XSLT Source");
319 		}
320 	}

322 	/**
323 	 * Configure the supplied {@link Transformer} instance.
324 	 * <p>The default implementation copies parameters from the model into the
325 	 * Transformer's {@link Transformer#setParameter parameter set}.
326 	 * This implementation also copies the {@link #setOutputProperties output properties}
327 	 * into the {@link Transformer} {@link Transformer#setOutputProperty output properties}.
328 	 * Indentation properties are set as well.
329 	 * @param model merged output Map (never {@code null})
330 	 * @param response current HTTP response
331 	 * @param transformer the target transformer
332 	 * @see #copyModelParameters(Map, Transformer)
333 	 * @see #copyOutputProperties(Transformer)
334 	 * @see #configureIndentation(Transformer)
335 	 */
336 	protected void configureTransformer(Map<String, Object> model, HttpServletResponse response, Transformer transformer) {
337 		copyModelParameters(model, transformer);
338 		copyOutputProperties(transformer);
339 		configureIndentation(transformer);
340 	}

342 	/**
343 	 * Configure the indentation settings for the supplied {@link Transformer}.
344 	 * @param transformer the target transformer
345 	 * @see org.springframework.util.xml.TransformerUtils#enableIndenting(javax.xml.transform.Transformer)
346 	 * @see org.springframework.util.xml.TransformerUtils#disableIndenting(javax.xml.transform.Transformer)
347 	 */
348 	protected final void configureIndentation(Transformer transformer) {
349 		if (this.indent) {
350 			TransformerUtils.enableIndenting(transformer);
351 		}
352 		else {
353 			TransformerUtils.disableIndenting(transformer);
354 		}
355 	}

357 	/**
358 	 * Copy the configured output {@link Properties}, if any, into the
359 	 * {@link Transformer#setOutputProperty output property set} of the supplied
360 	 * {@link Transformer}.
361 	 * @param transformer the target transformer
362 	 */
363 	protected final void copyOutputProperties(Transformer transformer) {
364 		if (this.outputProperties != null) {
365 			Enumeration<?> en = this.outputProperties.propertyNames();
366 			while (en.hasMoreElements()) {
367 				String name = (String) en.nextElement();
368 				transformer.setOutputProperty(name, this.outputProperties.getProperty(name));
369 			}
370 		}
371 	}
372 
373 	/**
374 	 * Copy all entries from the supplied Map into the
375 	 * {@link Transformer#setParameter(String, Object) parameter set}
376 	 * of the supplied {@link Transformer}.
377 	 * @param model merged output Map (never {@code null})
378 	 * @param transformer the target transformer
379 	 */
380 	protected final void copyModelParameters(Map<String, Object> model, Transformer transformer) {
381 		for (Map.Entry<String, Object> entry : model.entrySet()) {
382 			transformer.setParameter(entry.getKey(), entry.getValue());
383 		}
384 	}

386 	/**
387 	 * Configure the supplied {@link HttpServletResponse}.
388 	 * <p>The default implementation of this method sets the
389 	 * {@link HttpServletResponse#setContentType content type} and
390 	 * {@link HttpServletResponse#setCharacterEncoding encoding}
391 	 * from the "media-type" and "encoding" output properties
392 	 * specified in the {@link Transformer}.
393 	 * @param model merged output Map (never {@code null})
394 	 * @param response current HTTP response
395 	 * @param transformer the target transformer
396 	 */
397 	protected void configureResponse(Map<String, Object> model, HttpServletResponse response, Transformer transformer) {
398 		String contentType = getContentType();
399 		String mediaType = transformer.getOutputProperty(OutputKeys.MEDIA_TYPE);
400 		String encoding = transformer.getOutputProperty(OutputKeys.ENCODING);
401 		if (StringUtils.hasText(mediaType)) {
402 			contentType = mediaType;
403 		}
404 		if (StringUtils.hasText(encoding)) {
405 			// Only apply encoding if content type is specified but does not contain charset clause already.
406 			if (contentType != null && !contentType.toLowerCase().contains(WebUtils.CONTENT_TYPE_CHARSET_PREFIX)) {
407 				contentType = contentType + WebUtils.CONTENT_TYPE_CHARSET_PREFIX + encoding;
408 			}
409 		}
410 		response.setContentType(contentType);
411 	}

413 	/**
414 	 * Load the {@link Templates} instance for the stylesheet at the configured location.
415 	 */
416 	private Templates loadTemplates() throws ApplicationContextException {
417 		Source stylesheetSource = getStylesheetSource();
418 		try {
419 			Templates templates = this.transformerFactory.newTemplates(stylesheetSource);
420 			if (logger.isDebugEnabled()) {
421 				logger.debug("Loading templates '" + templates + "'");
422 			}
423 			return templates;
424 		}
425 		catch (TransformerConfigurationException ex) {
426 			throw new ApplicationContextException("Can't load stylesheet from '" + getUrl() + "'", ex);
427 		}
428 		finally {
429 			closeSourceIfNecessary(stylesheetSource);
430 		}
431 	}

433 	/**
434 	 * Create the {@link Transformer} instance used to prefer the XSLT transformation.
435 	 * <p>The default implementation simply calls {@link Templates#newTransformer()}, and
436 	 * configures the {@link Transformer} with the custom {@link URIResolver} if specified.
437 	 * @param templates the XSLT Templates instance to create a Transformer for
438 	 * @return the Transformer object
439 	 * @throws TransformerConfigurationException in case of creation failure
440 	 */
441 	protected Transformer createTransformer(Templates templates) throws TransformerConfigurationException {
442 		Transformer transformer = templates.newTransformer();
443 		if (this.uriResolver != null) {
444 			transformer.setURIResolver(this.uriResolver);
445 		}
446 		return transformer;
447 	}

449 	/**
450 	 * Get the XSLT {@link Source} for the XSLT template under the {@link #setUrl configured URL}.
451 	 * @return the Source object
452 	 */
453 	protected Source getStylesheetSource() {
454 		String url = getUrl();
455 		if (logger.isDebugEnabled()) {
456 			logger.debug("Loading XSLT stylesheet from '" + url + "'");
457 		}
458 		try {
459 			Resource resource = getApplicationContext().getResource(url);
460 			return new StreamSource(resource.getInputStream(), resource.getURI().toASCIIString());
461 		}
462 		catch (IOException ex) {
463 			throw new ApplicationContextException("Can't load XSLT stylesheet from '" + url + "'", ex);
464 		}
465 	}

467 	/**
468 	 * Close the underlying resource managed by the supplied {@link Source} if applicable.
469 	 * <p>Only works for {@link StreamSource StreamSources}.
470 	 * @param source the XSLT Source to close (may be {@code null})
471 	 */
472 	private void closeSourceIfNecessary(Source source) {
473 		if (source instanceof StreamSource) {
474 			StreamSource streamSource = (StreamSource) source;
475 			if (streamSource.getReader() != null) {
476 				try {
477 					streamSource.getReader().close();
478 				}
479 				catch (IOException ex) {
480 					// ignore
481 				}
482 			}
483 			if (streamSource.getInputStream() != null) {
484 				try {
485 					streamSource.getInputStream().close();
486 				}
487 				catch (IOException ex) {
488 					// ignore
489 				}
490 			}
491 		}
492 	}
