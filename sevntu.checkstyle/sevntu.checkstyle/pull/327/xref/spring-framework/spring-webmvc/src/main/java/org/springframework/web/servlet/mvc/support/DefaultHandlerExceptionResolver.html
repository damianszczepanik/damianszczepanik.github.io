53  /**
54   * Default implementation of the {@link org.springframework.web.servlet.HandlerExceptionResolver
55   * HandlerExceptionResolver} interface that resolves standard Spring exceptions and translates
56   * them to corresponding HTTP status codes.
57   *
58   * <p>This exception resolver is enabled by default in the {@link org.springframework.web.servlet.DispatcherServlet}.
59   *
60   * @author Arjen Poutsma
61   * @author Rossen Stoyanchev
62   * @since 3.0
63   *
64   * @see org.springframework.web.servlet.mvc.method.annotation.ResponseEntityExceptionHandler
65   * @see #handleNoSuchRequestHandlingMethod
66   * @see #handleHttpRequestMethodNotSupported
67   * @see #handleHttpMediaTypeNotSupported
68   * @see #handleMissingServletRequestParameter
69   * @see #handleServletRequestBindingException
70   * @see #handleTypeMismatch
71   * @see #handleHttpMessageNotReadable
72   * @see #handleHttpMessageNotWritable
73   * @see #handleMethodArgumentNotValidException
74   * @see #handleMissingServletRequestParameter
75   * @see #handleMissingServletRequestPartException
76   * @see #handleBindException
77   */
78  public class DefaultHandlerExceptionResolver extends AbstractHandlerExceptionResolver {

81  	 * Log category to use when no mapped handler is found for a request.
82  	 * @see #pageNotFoundLogger

84  	public static final String PAGE_NOT_FOUND_LOG_CATEGORY = "org.springframework.web.servlet.PageNotFound";
85  
86  	/**
87  	 * Additional logger to use when no mapped handler is found for a request.
88  	 * @see #PAGE_NOT_FOUND_LOG_CATEGORY

90  	protected static final Log pageNotFoundLogger = LogFactory.getLog(PAGE_NOT_FOUND_LOG_CATEGORY);

94  	 * Sets the {@linkplain #setOrder(int) order} to {@link #LOWEST_PRECEDENCE}.

101 	@Override
102 	protected ModelAndView doResolveException(HttpServletRequest request, HttpServletResponse response,
103 			Object handler, Exception ex) {
104 
105 		try {
106 			if (ex instanceof NoSuchRequestHandlingMethodException) {
107 				return handleNoSuchRequestHandlingMethod((NoSuchRequestHandlingMethodException) ex, request, response,
108 						handler);
109 			}
110 			else if (ex instanceof HttpRequestMethodNotSupportedException) {
111 				return handleHttpRequestMethodNotSupported((HttpRequestMethodNotSupportedException) ex, request,
112 						response, handler);
113 			}
114 			else if (ex instanceof HttpMediaTypeNotSupportedException) {
115 				return handleHttpMediaTypeNotSupported((HttpMediaTypeNotSupportedException) ex, request, response,
116 						handler);
117 			}
118 			else if (ex instanceof HttpMediaTypeNotAcceptableException) {
119 				return handleHttpMediaTypeNotAcceptable((HttpMediaTypeNotAcceptableException) ex, request, response,
120 						handler);
121 			}
122 			else if (ex instanceof MissingServletRequestParameterException) {
123 				return handleMissingServletRequestParameter((MissingServletRequestParameterException) ex, request,
124 						response, handler);
125 			}
126 			else if (ex instanceof ServletRequestBindingException) {
127 				return handleServletRequestBindingException((ServletRequestBindingException) ex, request, response,
128 						handler);
129 			}
130 			else if (ex instanceof ConversionNotSupportedException) {
131 				return handleConversionNotSupported((ConversionNotSupportedException) ex, request, response, handler);
132 			}
133 			else if (ex instanceof TypeMismatchException) {
134 				return handleTypeMismatch((TypeMismatchException) ex, request, response, handler);
135 			}
136 			else if (ex instanceof HttpMessageNotReadableException) {
137 				return handleHttpMessageNotReadable((HttpMessageNotReadableException) ex, request, response, handler);
138 			}
139 			else if (ex instanceof HttpMessageNotWritableException) {
140 				return handleHttpMessageNotWritable((HttpMessageNotWritableException) ex, request, response, handler);
141 			}
142 			else if (ex instanceof MethodArgumentNotValidException) {
143 				return handleMethodArgumentNotValidException((MethodArgumentNotValidException) ex, request, response, handler);
144 			}
145 			else if (ex instanceof MissingServletRequestPartException) {
146 				return handleMissingServletRequestPartException((MissingServletRequestPartException) ex, request, response, handler);
147 			}
148 			else if (ex instanceof BindException) {
149 				return handleBindException((BindException) ex, request, response, handler);
150 			}
151 			else if (ex instanceof NoHandlerFoundException) {
152 				return handleNoHandlerFoundException((NoHandlerFoundException) ex, request, response, handler);
153 			}
154 		}
155 		catch (Exception handlerException) {
156 			logger.warn("Handling of [" + ex.getClass().getName() + "] resulted in Exception", handlerException);
157 		}
158 		return null;
159 	}

161 	/**
162 	 * Handle the case where no request handler method was found.
163 	 * <p>The default implementation logs a warning, sends an HTTP 404 error, and returns
164 	 * an empty {@code ModelAndView}. Alternatively, a fallback view could be chosen,
165 	 * or the NoSuchRequestHandlingMethodException could be rethrown as-is.
166 	 * @param ex the NoSuchRequestHandlingMethodException to be handled
167 	 * @param request current HTTP request
168 	 * @param response current HTTP response
169 	 * @param handler the executed handler, or {@code null} if none chosen
170 	 * at the time of the exception (for example, if multipart resolution failed)
171 	 * @return an empty ModelAndView indicating the exception was handled
172 	 * @throws IOException potentially thrown from response.sendError()
173 	 */
174 	protected ModelAndView handleNoSuchRequestHandlingMethod(NoSuchRequestHandlingMethodException ex,
175 			HttpServletRequest request, HttpServletResponse response, Object handler) throws IOException {
176 
177 		pageNotFoundLogger.warn(ex.getMessage());
178 		response.sendError(HttpServletResponse.SC_NOT_FOUND);
179 		return new ModelAndView();
180 	}

182 	/**
183 	 * Handle the case where no request handler method was found for the particular HTTP request method.
184 	 * <p>The default implementation logs a warning, sends an HTTP 405 error, sets the "Allow" header,
185 	 * and returns an empty {@code ModelAndView}. Alternatively, a fallback view could be chosen,
186 	 * or the HttpRequestMethodNotSupportedException could be rethrown as-is.
187 	 * @param ex the HttpRequestMethodNotSupportedException to be handled
188 	 * @param request current HTTP request
189 	 * @param response current HTTP response
190 	 * @param handler the executed handler, or {@code null} if none chosen
191 	 * at the time of the exception (for example, if multipart resolution failed)
192 	 * @return an empty ModelAndView indicating the exception was handled
193 	 * @throws IOException potentially thrown from response.sendError()
194 	 */
195 	protected ModelAndView handleHttpRequestMethodNotSupported(HttpRequestMethodNotSupportedException ex,
196 			HttpServletRequest request, HttpServletResponse response, Object handler) throws IOException {
197 
198 		pageNotFoundLogger.warn(ex.getMessage());
199 		String[] supportedMethods = ex.getSupportedMethods();
200 		if (supportedMethods != null) {
201 			response.setHeader("Allow", StringUtils.arrayToDelimitedString(supportedMethods, ", "));
202 		}
203 		response.sendError(HttpServletResponse.SC_METHOD_NOT_ALLOWED, ex.getMessage());
204 		return new ModelAndView();
205 	}

207 	/**
208 	 * Handle the case where no {@linkplain org.springframework.http.converter.HttpMessageConverter message converters}
209 	 * were found for the PUT or POSTed content. <p>The default implementation sends an HTTP 415 error,
210 	 * sets the "Accept" header, and returns an empty {@code ModelAndView}. Alternatively, a fallback
211 	 * view could be chosen, or the HttpMediaTypeNotSupportedException could be rethrown as-is.
212 	 * @param ex the HttpMediaTypeNotSupportedException to be handled
213 	 * @param request current HTTP request
214 	 * @param response current HTTP response
215 	 * @param handler the executed handler
216 	 * @return an empty ModelAndView indicating the exception was handled
217 	 * @throws IOException potentially thrown from response.sendError()
218 	 */
219 	protected ModelAndView handleHttpMediaTypeNotSupported(HttpMediaTypeNotSupportedException ex,
220 			HttpServletRequest request, HttpServletResponse response, Object handler) throws IOException {
221 
222 		response.sendError(HttpServletResponse.SC_UNSUPPORTED_MEDIA_TYPE);
223 		List<MediaType> mediaTypes = ex.getSupportedMediaTypes();
224 		if (!CollectionUtils.isEmpty(mediaTypes)) {
225 			response.setHeader("Accept", MediaType.toString(mediaTypes));
226 		}
227 		return new ModelAndView();
228 	}

305 	/**
306 	 * Invoked to send a server error. Sets the status to 500 and also sets the
307 	 * request attribute "javax.servlet.error.exception" to the Exception.
308 	 */
309 	protected void sendServerError(Exception ex,
310 			HttpServletRequest request, HttpServletResponse response) throws IOException {
311 
312 		request.setAttribute("javax.servlet.error.exception", ex);
313 		response.sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR);
314 	}
