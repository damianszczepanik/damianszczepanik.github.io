17  package org.springframework.web.servlet.view.xml;
18  
19  import java.util.Map;
20  
21  import com.fasterxml.jackson.annotation.JsonView;
22  import com.fasterxml.jackson.dataformat.xml.XmlMapper;
23  
24  import org.springframework.http.converter.json.Jackson2ObjectMapperBuilder;
25  import org.springframework.validation.BindingResult;
26  import org.springframework.web.servlet.View;
27  import org.springframework.web.servlet.view.json.AbstractJackson2View;
28  
29  /**
30   * Spring MVC {@link View} that renders XML content by serializing the model for the current request
31   * using <a href="http://jackson.codehaus.org/">Jackson 2's</a> {@link XmlMapper}.
32   *
33   * <p>The Object to be serialized is supplied as a parameter in the model. The first serializable
34   * entry is used. Users can either specify a specific entry in the model via the
35   * {@link #setModelKey(String) sourceKey} property.
36   *
37   * <p>The default constructor uses the default configuration provided by {@link Jackson2ObjectMapperBuilder}.
38   *
39   * <p>Compatible with Jackson 2.1 and higher.
40   *
41   * @author Sebastien Deleuze
42   * @since 4.1
43   */
44  public class MappingJackson2XmlView extends AbstractJackson2View {
45  
46  	public static final String DEFAULT_CONTENT_TYPE = "application/xml";
47  
48  
49  	private String modelKey;
50  
51  
52  	/**
53  	 * Construct a new {@code MappingJackson2XmlView} using default configuration
54  	 * provided by {@link Jackson2ObjectMapperBuilder} and setting the content type
55  	 * to {@code application/xml}.
56  	 */
57  	public MappingJackson2XmlView() {
58  		super(Jackson2ObjectMapperBuilder.xml().build(), DEFAULT_CONTENT_TYPE);
59  	}
60  
61  
62  	/**
63  	 * {@inheritDoc}
64  	 */
65  	@Override
66  	public void setModelKey(String modelKey) {
67  		this.modelKey = modelKey;
68  	}
69  
70  	/**
71  	 * Filter out undesired attributes from the given model.
72  	 * The return value can be either another {@link Map} or a single value object.
73  	 * @param model the model, as passed on to {@link #renderMergedOutputModel}
74  	 * @return the value to be rendered
75  	 */
76  	@Override
77  	protected Object filterModel(Map<String, Object> model) {
78  		Object value = null;
79  		if (this.modelKey != null) {
80  			value = model.get(this.modelKey);
81  			if (value == null) {
82  				throw new IllegalStateException(
83  						"Model contains no object with key [" + this.modelKey + "]");
84  			}
85  		}
86  		else {
87  			for (Map.Entry<String, Object> entry : model.entrySet()) {
88  				if (!(entry.getValue() instanceof BindingResult) && !entry.getKey().equals(JsonView.class.getName())) {
89  					if (value != null) {
90  						throw new IllegalStateException("Model contains more than one object to render, only one is supported");
91  					}
92  					value = entry.getValue();
93  				}
94  			}
95  		}
96  		return value;
97  	}
98  
99  }
