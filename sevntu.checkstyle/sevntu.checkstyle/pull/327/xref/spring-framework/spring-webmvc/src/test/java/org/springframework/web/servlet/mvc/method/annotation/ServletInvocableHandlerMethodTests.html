168 	private void wrapConcurrentResult_ResponseBody(Object handler) throws Exception {
169 		List<HttpMessageConverter<?>> converters = new ArrayList<HttpMessageConverter<?>>();
170 		converters.add(new StringHttpMessageConverter());
171 		this.returnValueHandlers.addHandler(new RequestResponseBodyMethodProcessor(converters));
172 		ServletInvocableHandlerMethod handlerMethod = getHandlerMethod(handler, "handle");
173 		handlerMethod = handlerMethod.wrapConcurrentResult("bar");
174 		handlerMethod.invokeAndHandle(this.webRequest, this.mavContainer);
175 
176 		assertEquals("bar", this.response.getContentAsString());
177 	}

238 	private ServletInvocableHandlerMethod getHandlerMethod(Object controller,
239 			String methodName, Class<?>... argTypes) throws NoSuchMethodException {
240 
241 		Method method = controller.getClass().getDeclaredMethod(methodName, argTypes);
242 		ServletInvocableHandlerMethod handlerMethod = new ServletInvocableHandlerMethod(controller, method);
243 		handlerMethod.setHandlerMethodArgumentResolvers(this.argumentResolvers);
244 		handlerMethod.setHandlerMethodReturnValueHandlers(this.returnValueHandlers);
245 		return handlerMethod;
246 	}
