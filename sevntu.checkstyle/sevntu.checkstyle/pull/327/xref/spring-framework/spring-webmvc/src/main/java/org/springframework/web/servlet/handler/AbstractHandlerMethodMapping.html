42  /**
43   * Abstract base class for {@link HandlerMapping} implementations that define a
44   * mapping between a request and a {@link HandlerMethod}.
45   *
46   * <p>For each registered handler method, a unique mapping is maintained with
47   * subclasses defining the details of the mapping type {@code <T>}.
48   *
49   * @param <T> The mapping for a {@link HandlerMethod} containing the conditions
50   * needed to match the handler method to incoming request.
51   *
52   * @author Arjen Poutsma
53   * @author Rossen Stoyanchev
54   * @author Juergen Hoeller
55   * @since 3.1
56   */
57  public abstract class AbstractHandlerMethodMapping<T> extends AbstractHandlerMapping implements InitializingBean {
58  
59  	/**
60  	 * Bean name prefix for target beans behind scoped proxies. Used to exclude those
61  	 * targets from handler method detection, in favor of the corresponding proxies.
62  	 * <p>We're not checking the autowire-candidate status here, which is how the
63  	 * proxy target filtering problem is being handled at the autowiring level,
64  	 * since autowire-candidate may have been turned to {@code false} for other
65  	 * reasons, while still expecting the bean to be eligible for handler methods.
66  	 * <p>Originally defined in {@link org.springframework.aop.scope.ScopedProxyUtils}
67  	 * but duplicated here to avoid a hard dependency on the spring-aop module.
68  	 */
69  	private static final String SCOPED_TARGET_NAME_PREFIX = "scopedTarget.";

85  	 * Whether to detect handler methods in beans in ancestor ApplicationContexts.
86  	 * <p>Default is "false": Only beans in the current ApplicationContext are
87  	 * considered, i.e. only in the context that this HandlerMapping itself
88  	 * is defined in (typically the current DispatcherServlet's context).
89  	 * <p>Switch this flag on to detect handler beans in ancestor contexts
90  	 * (typically the Spring root WebApplicationContext) as well.
91  	 */
92  	public void setDetectHandlerMethodsInAncestorContexts(boolean detectHandlerMethodsInAncestorContexts) {
93  		this.detectHandlerMethodsInAncestorContexts = detectHandlerMethodsInAncestorContexts;
94  	}

129 	/**
130 	 * Scan beans in the ApplicationContext, detect and register handler methods.
131 	 * @see #isHandler(Class)
132 	 * @see #getMappingForMethod(Method, Class)
133 	 * @see #handlerMethodsInitialized(Map)
134 	 */
135 	protected void initHandlerMethods() {
136 		if (logger.isDebugEnabled()) {
137 			logger.debug("Looking for request mappings in application context: " + getApplicationContext());
138 		}
139 
140 		String[] beanNames = (this.detectHandlerMethodsInAncestorContexts ?
141 				BeanFactoryUtils.beanNamesForTypeIncludingAncestors(getApplicationContext(), Object.class) :
142 				getApplicationContext().getBeanNamesForType(Object.class));
143 
144 		for (String beanName : beanNames) {
145 			if (!beanName.startsWith(SCOPED_TARGET_NAME_PREFIX) &&
146 					isHandler(getApplicationContext().getType(beanName))){
147 				detectHandlerMethods(beanName);
148 			}
149 		}
150 		handlerMethodsInitialized(getHandlerMethods());
151 	}

153 	/**
154 	 * Whether the given type is a handler with handler methods.
155 	 * @param beanType the type of the bean being checked
156 	 * @return "true" if this a handler type, "false" otherwise.
157 	 */
158 	protected abstract boolean isHandler(Class<?> beanType);

160 	/**
161 	 * Look for handler methods in a handler.
162 	 * @param handler the bean name of a handler or a handler instance
163 	 */
164 	protected void detectHandlerMethods(final Object handler) {
165 		Class<?> handlerType =
166 				(handler instanceof String ? getApplicationContext().getType((String) handler) : handler.getClass());
167 
168 		// Avoid repeated calls to getMappingForMethod which would rebuild RequestMappingInfo instances
169 		final Map<Method, T> mappings = new IdentityHashMap<Method, T>();
170 		final Class<?> userType = ClassUtils.getUserClass(handlerType);
171 
172 		Set<Method> methods = HandlerMethodSelector.selectMethods(userType, new MethodFilter() {
173 			@Override
174 			public boolean matches(Method method) {
175 				T mapping = getMappingForMethod(method, userType);
176 				if (mapping != null) {
177 					mappings.put(method, mapping);
178 					return true;
179 				}
180 				else {
181 					return false;
182 				}
183 			}
184 		});
185 
186 		for (Method method : methods) {
187 			registerHandlerMethod(handler, method, mappings.get(method));
188 		}
189 	}
190 
191 	/**
192 	 * Provide the mapping for a handler method. A method for which no
193 	 * mapping can be provided is not a handler method.
194 	 * @param method the method to provide a mapping for
195 	 * @param handlerType the handler type, possibly a sub-type of the method's
196 	 * declaring class
197 	 * @return the mapping, or {@code null} if the method is not mapped
198 	 */
199 	protected abstract T getMappingForMethod(Method method, Class<?> handlerType);
200 
201 	/**
202 	 * Register a handler method and its unique mapping.
203 	 * @param handler the bean name of the handler or the handler instance
204 	 * @param method the method to register
205 	 * @param mapping the mapping conditions associated with the handler method
206 	 * @throws IllegalStateException if another method was already registered
207 	 * under the same mapping
208 	 */
209 	protected void registerHandlerMethod(Object handler, Method method, T mapping) {
210 		HandlerMethod newHandlerMethod = createHandlerMethod(handler, method);
211 		HandlerMethod oldHandlerMethod = this.handlerMethods.get(mapping);
212 		if (oldHandlerMethod != null && !oldHandlerMethod.equals(newHandlerMethod)) {
213 			throw new IllegalStateException("Ambiguous mapping found. Cannot map '" + newHandlerMethod.getBean() +
214 					"' bean method \n" + newHandlerMethod + "\nto " + mapping + ": There is already '" +
215 					oldHandlerMethod.getBean() + "' bean method\n" + oldHandlerMethod + " mapped.");
216 		}
217 
218 		this.handlerMethods.put(mapping, newHandlerMethod);
219 		if (logger.isInfoEnabled()) {
220 			logger.info("Mapped \"" + mapping + "\" onto " + newHandlerMethod);
221 		}
222 
223 		Set<String> patterns = getMappingPathPatterns(mapping);
224 		for (String pattern : patterns) {
225 			if (!getPathMatcher().isPattern(pattern)) {
226 				this.urlMap.add(pattern, mapping);
227 			}
228 		}
229 
230 		if (this.namingStrategy != null) {
231 			String name = this.namingStrategy.getName(newHandlerMethod, mapping);
232 			updateNameMap(name, newHandlerMethod);
233 		}
234 	}
235 
236 	private void updateNameMap(String name, HandlerMethod newHandlerMethod) {
237 
238 		List<HandlerMethod> handlerMethods = this.nameMap.get(name);
239 		if (handlerMethods != null) {
240 			for (HandlerMethod handlerMethod : handlerMethods) {
241 				if (handlerMethod.getMethod().equals(newHandlerMethod.getMethod())) {
242 					logger.trace("Mapping name already registered. Multiple controller instances perhaps?");
243 					return;
244 				}
245 			}
246 		}
247 
248 		logger.trace("Mapping name=" + name);
249 		this.nameMap.add(name, newHandlerMethod);
250 
251 		if (this.nameMap.get(name).size() > 1) {
252 			if (logger.isDebugEnabled()) {
253 				logger.debug("Mapping name clash for handlerMethods=" + this.nameMap.get(name) +
254 						". Consider assigning explicit names.");
255 			}
256 		}
257 	}

259 	/**
260 	 * Create the HandlerMethod instance.
261 	 * @param handler either a bean name or an actual handler instance
262 	 * @param method the target method
263 	 * @return the created HandlerMethod
264 	 */
265 	protected HandlerMethod createHandlerMethod(Object handler, Method method) {
266 		HandlerMethod handlerMethod;
267 		if (handler instanceof String) {
268 			String beanName = (String) handler;
269 			handlerMethod = new HandlerMethod(beanName, getApplicationContext(), method);
270 		}
271 		else {
272 			handlerMethod = new HandlerMethod(handler, method);
273 		}
274 		return handlerMethod;
275 	}

290 	/**
291 	 * Look up a handler method for the given request.
292 	 */
293 	@Override
294 	protected HandlerMethod getHandlerInternal(HttpServletRequest request) throws Exception {
295 		String lookupPath = getUrlPathHelper().getLookupPathForRequest(request);
296 		if (logger.isDebugEnabled()) {
297 			logger.debug("Looking up handler method for path " + lookupPath);
298 		}
299 		HandlerMethod handlerMethod = lookupHandlerMethod(lookupPath, request);
300 		if (logger.isDebugEnabled()) {
301 			if (handlerMethod != null) {
302 				logger.debug("Returning handler method [" + handlerMethod + "]");
303 			}
304 			else {
305 				logger.debug("Did not find handler method for [" + lookupPath + "]");
306 			}
307 		}
308 		return (handlerMethod != null ? handlerMethod.createWithResolvedBean() : null);
309 	}

311 	/**
312 	 * Look up the best-matching handler method for the current request.
313 	 * If multiple matches are found, the best match is selected.
314 	 * @param lookupPath mapping lookup path within the current servlet mapping
315 	 * @param request the current request
316 	 * @return the best-matching handler method, or {@code null} if no match
317 	 * @see #handleMatch(Object, String, HttpServletRequest)
318 	 * @see #handleNoMatch(Set, String, HttpServletRequest)
319 	 */
320 	protected HandlerMethod lookupHandlerMethod(String lookupPath, HttpServletRequest request) throws Exception {
321 		List<Match> matches = new ArrayList<Match>();
322 		List<T> directPathMatches = this.urlMap.get(lookupPath);
323 		if (directPathMatches != null) {
324 			addMatchingMappings(directPathMatches, matches, request);
325 		}
326 		if (matches.isEmpty()) {
327 			// No choice but to go through all mappings...
328 			addMatchingMappings(this.handlerMethods.keySet(), matches, request);
329 		}
330 
331 		if (!matches.isEmpty()) {
332 			Comparator<Match> comparator = new MatchComparator(getMappingComparator(request));
333 			Collections.sort(matches, comparator);
334 			if (logger.isTraceEnabled()) {
335 				logger.trace("Found " + matches.size() + " matching mapping(s) for [" + lookupPath + "] : " + matches);
336 			}
337 			Match bestMatch = matches.get(0);
338 			if (matches.size() > 1) {
339 				Match secondBestMatch = matches.get(1);
340 				if (comparator.compare(bestMatch, secondBestMatch) == 0) {
341 					Method m1 = bestMatch.handlerMethod.getMethod();
342 					Method m2 = secondBestMatch.handlerMethod.getMethod();
343 					throw new IllegalStateException(
344 							"Ambiguous handler methods mapped for HTTP path '" + request.getRequestURL() + "': {" +
345 							m1 + ", " + m2 + "}");
346 				}
347 			}
348 			handleMatch(bestMatch.mapping, lookupPath, request);
349 			return bestMatch.handlerMethod;
350 		}
351 		else {
352 			return handleNoMatch(handlerMethods.keySet(), lookupPath, request);
353 		}
354 	}
355 
356 	private void addMatchingMappings(Collection<T> mappings, List<Match> matches, HttpServletRequest request) {
357 		for (T mapping : mappings) {
358 			T match = getMatchingMapping(mapping, request);
359 			if (match != null) {
360 				matches.add(new Match(match, this.handlerMethods.get(mapping)));
361 			}
362 		}
363 	}

366 	 * Check if a mapping matches the current request and return a (potentially
367 	 * new) mapping with conditions relevant to the current request.
368 	 * @param mapping the mapping to get a match for
369 	 * @param request the current HTTP servlet request
370 	 * @return the match, or {@code null} if the mapping doesn't match
371 	 */
372 	protected abstract T getMatchingMapping(T mapping, HttpServletRequest request);

374 	/**
375 	 * Return a comparator for sorting matching mappings.
376 	 * The returned comparator should sort 'better' matches higher.
377 	 * @param request the current request
378 	 * @return the comparator, never {@code null}
379 	 */
380 	protected abstract Comparator<T> getMappingComparator(HttpServletRequest request);

382 	/**
383 	 * Invoked when a matching mapping is found.
384 	 * @param mapping the matching mapping
385 	 * @param lookupPath mapping lookup path within the current servlet mapping
386 	 * @param request the current request
387 	 */
388 	protected void handleMatch(T mapping, String lookupPath, HttpServletRequest request) {
389 		request.setAttribute(HandlerMapping.PATH_WITHIN_HANDLER_MAPPING_ATTRIBUTE, lookupPath);
390 	}

392 	/**
393 	 * Invoked when no matching mapping is not found.

397 	 * @throws ServletException in case of errors

399 	protected HandlerMethod handleNoMatch(Set<T> mappings, String lookupPath, HttpServletRequest request)

406 	/**
407 	 * A thin wrapper around a matched HandlerMethod and its mapping, for the purpose of
408 	 * comparing the best match with a comparator in the context of the current request.
409 	 */
410 	private class Match {
411 
412 		private final T mapping;
413 
414 		private final HandlerMethod handlerMethod;
415 
416 		public Match(T mapping, HandlerMethod handlerMethod) {
417 			this.mapping = mapping;
418 			this.handlerMethod = handlerMethod;
419 		}
420 
421 		@Override
422 		public String toString() {
423 			return this.mapping.toString();
424 		}
425 	}
426 
427 
428 	private class MatchComparator implements Comparator<Match> {
429 
430 		private final Comparator<T> comparator;
431 
432 		public MatchComparator(Comparator<T> comparator) {
433 			this.comparator = comparator;
434 		}
435 
436 		@Override
437 		public int compare(Match match1, Match match2) {
438 			return this.comparator.compare(match1.mapping, match2.mapping);
439 		}
440 	}
