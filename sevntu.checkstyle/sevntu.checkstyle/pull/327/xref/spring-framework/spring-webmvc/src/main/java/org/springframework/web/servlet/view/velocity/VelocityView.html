43  /**
44   * View using the Velocity template engine.
45   *
46   * <p>Exposes the following JavaBean properties:
47   * <ul>
48   * <li><b>url</b>: the location of the Velocity template to be wrapped,
49   * relative to the Velocity resource loader path (see VelocityConfigurer).
50   * <li><b>encoding</b> (optional, default is determined by Velocity configuration):
51   * the encoding of the Velocity template file
52   * <li><b>velocityFormatterAttribute</b> (optional, default=null): the name of
53   * the VelocityFormatter helper object to expose in the Velocity context of this
54   * view, or {@code null} if not needed. VelocityFormatter is part of standard Velocity.
55   * <li><b>dateToolAttribute</b> (optional, default=null): the name of the
56   * DateTool helper object to expose in the Velocity context of this view,
57   * or {@code null} if not needed. DateTool is part of Velocity Tools.
58   * <li><b>numberToolAttribute</b> (optional, default=null): the name of the
59   * NumberTool helper object to expose in the Velocity context of this view,
60   * or {@code null} if not needed. NumberTool is part of Velocity Tools.
61   * <li><b>cacheTemplate</b> (optional, default=false): whether or not the Velocity
62   * template should be cached. It should normally be true in production, but setting
63   * this to false enables us to modify Velocity templates without restarting the
64   * application (similar to JSPs). Note that this is a minor optimization only,
65   * as Velocity itself caches templates in a modification-aware fashion.
66   * </ul>
67   *
68   * <p>Depends on a VelocityConfig object such as VelocityConfigurer being
69   * accessible in the current web application context, with any bean name.
70   * Alternatively, you can set the VelocityEngine object as bean property.
71   *
72   * <p>Note: Spring 3.0's VelocityView requires Velocity 1.4 or higher, and optionally
73   * Velocity Tools 1.1 or higher (depending on the use of DateTool and/or NumberTool).
74   *
75   * @author Rod Johnson
76   * @author Juergen Hoeller
77   * @author Dave Syer
78   * @see VelocityConfig
79   * @see VelocityConfigurer
80   * @see #setUrl
81   * @see #setExposeSpringMacroHelpers
82   * @see #setEncoding
83   * @see #setVelocityEngine
84   * @see VelocityConfig
85   * @see VelocityConfigurer
86   */

105 	 * Set tool attributes to expose to the view, as attribute name / class name pairs.
106 	 * An instance of the given class will be added to the Velocity context for each
107 	 * rendering operation, under the given attribute name.
108 	 * <p>For example, an instance of MathTool, which is part of the generic package
109 	 * of Velocity Tools, can be bound under the attribute name "math", specifying the
110 	 * fully qualified class name "org.apache.velocity.tools.generic.MathTool" as value.
111 	 * <p>Note that VelocityView can only create simple generic tools or values, that is,
112 	 * classes with a public default constructor and no further initialization needs.
113 	 * This class does not do any further checks, to not introduce a required dependency
114 	 * on a specific tools package.
115 	 * <p>For tools that are part of the view package of Velocity Tools, a special
116 	 * Velocity context and a special init callback are needed. Use VelocityToolboxView
117 	 * in such a case, or override {@code createVelocityContext} and
118 	 * {@code initTool} accordingly.
119 	 * <p>For a simple VelocityFormatter instance or special locale-aware instances
120 	 * of DateTool/NumberTool, which are part of the generic package of Velocity Tools,
121 	 * specify the "velocityFormatterAttribute", "dateToolAttribute" or
122 	 * "numberToolAttribute" properties, respectively.
123 	 * @param toolAttributes attribute names as keys, and tool class names as values
124 	 * @see org.apache.velocity.tools.generic.MathTool
125 	 * @see VelocityToolboxView
126 	 * @see #createVelocityContext
127 	 * @see #initTool
128 	 * @see #setDateToolAttribute
129 	 * @see #setNumberToolAttribute
130 	 */

132 		this.toolAttributes = toolAttributes;

217  	 * Invoked on startup. Looks for a single VelocityConfig bean to
218  	 * find the relevant VelocityEngine for this factory.
219  	 */
220 	@Override
221 	protected void initApplicationContext() throws BeansException {
222 		super.initApplicationContext();
223 
224 		if (getVelocityEngine() == null) {
225 			// No explicit VelocityEngine: try to autodetect one.
226 			setVelocityEngine(autodetectVelocityEngine());
227 		}
228 	}

230 	/**
231 	 * Autodetect a VelocityEngine via the ApplicationContext.
232 	 * Called if no explicit VelocityEngine has been specified.
233 	 * @return the VelocityEngine to use for VelocityViews
234 	 * @throws BeansException if no VelocityEngine could be found
235 	 * @see #getApplicationContext
236 	 * @see #setVelocityEngine
237 	 */
238 	protected VelocityEngine autodetectVelocityEngine() throws BeansException {
239 		try {
240 			VelocityConfig velocityConfig = BeanFactoryUtils.beanOfTypeIncludingAncestors(
241 					getApplicationContext(), VelocityConfig.class, true, false);
242 			return velocityConfig.getVelocityEngine();
243 		}
244 		catch (NoSuchBeanDefinitionException ex) {
245 			throw new ApplicationContextException(
246 					"Must define a single VelocityConfig bean in this web application context " +
247 					"(may be inherited): VelocityConfigurer is the usual implementation. " +
248 					"This bean may be given any name.", ex);
249 		}
250 	}

252 	/**
253 	 * Check that the Velocity template used for this view exists and is valid.
254 	 * <p>Can be overridden to customize the behavior, for example in case of
255 	 * multiple templates to be rendered into a single view.
256 	 */
257 	@Override
258 	public boolean checkResource(Locale locale) throws Exception {
259 		try {
260 			// Check that we can get the template, even if we might subsequently get it again.
261 			this.template = getTemplate(getUrl());
262 			return true;
263 		}
264 		catch (ResourceNotFoundException ex) {
265 			if (logger.isDebugEnabled()) {
266 				logger.debug("No Velocity view found for URL: " + getUrl());
267 			}
268 			return false;
269 		}
270 		catch (Exception ex) {
271 			throw new NestedIOException(
272 					"Could not load Velocity template for URL [" + getUrl() + "]", ex);
273 		}
274 	}

278 	 * Process the model map by merging it with the Velocity template.
279 	 * Output is directed to the servlet response.
280 	 * <p>This method can be overridden if custom behavior is needed.
281 	 */
282 	@Override
283 	protected void renderMergedTemplateModel(
284 			Map<String, Object> model, HttpServletRequest request, HttpServletResponse response) throws Exception {
285 
286 		exposeHelpers(model, request);
287 
288 		Context velocityContext = createVelocityContext(model, request, response);
289 		exposeHelpers(velocityContext, request, response);
290 		exposeToolAttributes(velocityContext, request);
291 
292 		doRender(velocityContext, response);
293 	}

381 	/**
382 	 * Expose the tool attributes, according to corresponding bean property settings.
383 	 * <p>Do not override this method unless for further tools driven by bean properties.
384 	 * Override one of the {@code exposeHelpers} methods to add custom helpers.
385 	 * @param velocityContext Velocity context that will be passed to the template
386 	 * @param request current HTTP request
387 	 * @throws Exception if there's a fatal error while we're adding model attributes
388 	 * @see #setDateToolAttribute
389 	 * @see #setNumberToolAttribute
390 	 * @see #exposeHelpers(Map, HttpServletRequest)
391 	 * @see #exposeHelpers(org.apache.velocity.context.Context, HttpServletRequest, HttpServletResponse)
392 	 */
393 	protected void exposeToolAttributes(Context velocityContext, HttpServletRequest request) throws Exception {
394 		// Expose generic attributes.
395 		if (this.toolAttributes != null) {
396 			for (Map.Entry<String, Class<?>> entry : this.toolAttributes.entrySet()) {
397 				String attributeName = entry.getKey();
398 				Class<?> toolClass = entry.getValue();
399 				try {
400 					Object tool = toolClass.newInstance();
401 					initTool(tool, velocityContext);
402 					velocityContext.put(attributeName, tool);
403 				}
404 				catch (Exception ex) {
405 					throw new NestedServletException("Could not instantiate Velocity tool '" + attributeName + "'", ex);
406 				}
407 			}
408 		}
409 
410 		// Expose locale-aware DateTool/NumberTool attributes.
411 		if (this.dateToolAttribute != null || this.numberToolAttribute != null) {
412 			if (this.dateToolAttribute != null) {
413 				velocityContext.put(this.dateToolAttribute, new LocaleAwareDateTool(request));
414 			}
415 			if (this.numberToolAttribute != null) {
416 				velocityContext.put(this.numberToolAttribute, new LocaleAwareNumberTool(request));
417 			}
418 		}
419 	}

442 	/**
443 	 * Render the Velocity view to the given response, using the given Velocity
444 	 * context which contains the complete template model to use.
445 	 * <p>The default implementation renders the template specified by the "url"
446 	 * bean property, retrieved via {@code getTemplate}. It delegates to the
447 	 * {@code mergeTemplate} method to merge the template instance with the
448 	 * given Velocity context.
449 	 * <p>Can be overridden to customize the behavior, for example to render
450 	 * multiple templates into a single view.
451 	 * @param context the Velocity context to use for rendering
452 	 * @param response servlet response (use this to get the OutputStream or Writer)
453 	 * @throws Exception if thrown by Velocity
454 	 * @see #setUrl
455 	 * @see #getTemplate()
456 	 * @see #mergeTemplate
457 	 */
458 	protected void doRender(Context context, HttpServletResponse response) throws Exception {
459 		if (logger.isDebugEnabled()) {
460 			logger.debug("Rendering Velocity template [" + getUrl() + "] in VelocityView '" + getBeanName() + "'");
461 		}
462 		mergeTemplate(getTemplate(), context, response);
463 	}

465 	/**
466 	 * Retrieve the Velocity template to be rendered by this view.
467 	 * <p>By default, the template specified by the "url" bean property will be
468 	 * retrieved: either returning a cached template instance or loading a fresh
469 	 * instance (according to the "cacheTemplate" bean property)
470 	 * @return the Velocity template to render
471 	 * @throws Exception if thrown by Velocity
472 	 * @see #setUrl
473 	 * @see #setCacheTemplate
474 	 * @see #getTemplate(String)
475 	 */
476 	protected Template getTemplate() throws Exception {
477 		// We already hold a reference to the template, but we might want to load it
478 		// if not caching. Velocity itself caches templates, so our ability to
479 		// cache templates in this class is a minor optimization only.
480 		if (isCacheTemplate() && this.template != null) {
481 			return this.template;
482 		}
483 		else {
484 			return getTemplate(getUrl());
485 		}
486 	}
487 
488 	/**
489 	 * Retrieve the Velocity template specified by the given name,
490 	 * using the encoding specified by the "encoding" bean property.
491 	 * <p>Can be called by subclasses to retrieve a specific template,
492 	 * for example to render multiple templates into a single view.
493 	 * @param name the file name of the desired template
494 	 * @return the Velocity template
495 	 * @throws Exception if thrown by Velocity
496 	 * @see org.apache.velocity.app.VelocityEngine#getTemplate
497 	 */
498 	protected Template getTemplate(String name) throws Exception {
499 		return (getEncoding() != null ?
500 				getVelocityEngine().getTemplate(name, getEncoding()) :
501 				getVelocityEngine().getTemplate(name));
502 	}
503 
504 	/**
505 	 * Merge the template with the context.
506 	 * Can be overridden to customize the behavior.
507 	 * @param template the template to merge
508 	 * @param context the Velocity context to use for rendering
509 	 * @param response servlet response (use this to get the OutputStream or Writer)
510 	 * @throws Exception if thrown by Velocity
511 	 * @see org.apache.velocity.Template#merge
512 	 */
513 	protected void mergeTemplate(
514 			Template template, Context context, HttpServletResponse response) throws Exception {
515 
516 		try {
517 			template.merge(context, response.getWriter());
518 		}
519 		catch (MethodInvocationException ex) {
520 			Throwable cause = ex.getWrappedThrowable();
521 			throw new NestedServletException(
522 					"Method invocation failed during rendering of Velocity view with name '" +
523 					getBeanName() + "': " + ex.getMessage() + "; reference [" + ex.getReferenceName() +
524 					"], method '" + ex.getMethodName() + "'",
525 					cause==null ? ex : cause);
526 		}
527 	}
