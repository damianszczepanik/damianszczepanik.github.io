90  	@Test
91  	public void testExposeSpringMacroHelpers() throws Exception {
92  		FreeMarkerView fv = new FreeMarkerView() {
93  			@Override
94  			protected void processTemplate(Template template, SimpleHash fmModel, HttpServletResponse response)
95  					throws TemplateException {
96  				Map model = fmModel.toMap();
97  				assertTrue(model.get(FreeMarkerView.SPRING_MACRO_REQUEST_CONTEXT_ATTRIBUTE) instanceof RequestContext);
98  				RequestContext rc = (RequestContext) model.get(FreeMarkerView.SPRING_MACRO_REQUEST_CONTEXT_ATTRIBUTE);
99  				BindStatus status = rc.getBindStatus("tb.name");
100 				assertEquals("name", status.getExpression());
101 				assertEquals("juergen", status.getValue());
102 			}
103 		};
104 		fv.setUrl(TEMPLATE_FILE);
105 		fv.setApplicationContext(wac);
106 		fv.setExposeSpringMacroHelpers(true);
107 
108 		Map<String, Object> model = new HashMap<String, Object>();
109 		model.put("tb", new TestBean("juergen", 99));
110 		fv.render(model, request, response);
111 	}

276 	private String getMacroOutput(String name) throws Exception {
277 		String macro = fetchMacro(name);
278 		assertNotNull(macro);
279 
280 		FileSystemResource resource = new FileSystemResource(System.getProperty("java.io.tmpdir") + "/tmp.ftl");
281 		FileCopyUtils.copy("<#import \"spring.ftl\" as spring />\n" + macro, new FileWriter(resource.getPath()));
282 
283 		DummyMacroRequestContext rc = new DummyMacroRequestContext(request);
284 		Map<String, String> msgMap = new HashMap<String, String>();
285 		msgMap.put("hello", "Howdy");
286 		msgMap.put("world", "Mundo");
287 		rc.setMessageMap(msgMap);
288 		Map<String, String> themeMsgMap = new HashMap<String, String>();
289 		themeMsgMap.put("hello", "Howdy!");
290 		themeMsgMap.put("world", "Mundo!");
291 		rc.setThemeMessageMap(themeMsgMap);
292 		rc.setContextPath("/springtest");
293 
294 		TestBean darren = new TestBean("Darren", 99);
295 		TestBean fred = new TestBean("Fred");
296 		fred.setJedi(true);
297 		darren.setSpouse(fred);
298 		darren.setJedi(true);
299 		darren.setStringArray(new String[] {"John", "Fred"});
300 		request.setAttribute("command", darren);
301 
302 		Map<String, String> names = new HashMap<String, String>();
303 		names.put("Darren", "Darren Davison");
304 		names.put("John", "John Doe");
305 		names.put("Fred", "Fred Bloggs");
306 		names.put("Rob&Harrop", "Rob Harrop");
307 
308 		Configuration config = fc.getConfiguration();
309 		Map<String, Object> model = new HashMap<String, Object>();
310 		model.put("command", darren);
311 		model.put("springMacroRequestContext", rc);
312 		model.put("msgArgs", new Object[] { "World" });
313 		model.put("nameOptionMap", names);
314 		model.put("options", names.values());
315 
316 		FreeMarkerView view = new FreeMarkerView();
317 		view.setBeanName("myView");
318 		view.setUrl("tmp.ftl");
319 		view.setExposeSpringMacroHelpers(false);
320 		view.setConfiguration(config);
321 		view.setServletContext(new MockServletContext());
322 
323 		view.render(model, request, response);
324 
325 		// tokenize output and ignore whitespace
326 		String output = response.getContentAsString();
327 		return output.trim();
328 	}
329 
330 	private String fetchMacro(String name) throws Exception {
331 		ClassPathResource resource = new ClassPathResource("test.ftl", getClass());
332 		assertTrue(resource.exists());
333 		String all = FileCopyUtils.copyToString(new InputStreamReader(resource.getInputStream()));
334 		all = all.replace("\r\n", "\n");
335 		String[] macros = StringUtils.delimitedListToStringArray(all, "\n\n");
336 		for (String macro : macros) {
337 			if (macro.startsWith(name)) {
338 				return macro.substring(macro.indexOf("\n")).trim();
339 			}
340 		}
341 		return null;
342 	}
