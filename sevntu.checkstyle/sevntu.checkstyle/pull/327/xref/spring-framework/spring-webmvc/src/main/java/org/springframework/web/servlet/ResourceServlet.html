22  import javax.servlet.http.HttpServletRequest;
23  import javax.servlet.http.HttpServletResponse;

25  import org.springframework.util.AntPathMatcher;
26  import org.springframework.util.PathMatcher;
27  import org.springframework.util.StringUtils;
28  import org.springframework.web.context.support.ServletContextResource;

30  /**
31   * Simple servlet that can expose an internal resource, including a
32   * default URL if the specified resource is not found. An alternative,
33   * for example, to trying and catching exceptions when using JSP include.
34   *
35   * <p>A further usage of this servlet is the ability to apply last-modified
36   * timestamps to quasi-static resources (typically JSPs). This can happen
37   * as bridge to parameter-specified resources, or as proxy for a specific
38   * target resource (or a list of specific target resources to combine).
39   *
40   * <p>A typical usage would map a URL like "/ResourceServlet" onto an instance
41   * of this servlet, and use the "JSP include" action to include this URL,
42   * with the "resource" parameter indicating the actual target path in the WAR.
43   *
44   * <p>The {@code defaultUrl} property can be set to the internal
45   * resource path of a default URL, to be rendered when the target resource
46   * is not found or not specified in the first place.
47   *
48   * <p>The "resource" parameter and the {@code defaultUrl} property can
49   * also specify a list of target resources to combine. Those resources will be
50   * included one by one to build the response. If last-modified determination
51   * is active, the newest timestamp among those files will be used.
52   *
53   * <p>The {@code allowedResources} property can be set to a URL
54   * pattern of resources that should be available via this servlet.
55   * If not set, any target resource can be requested, including resources
56   * in the WEB-INF directory!
57   *
58   * <p>If using this servlet for direct access rather than via includes,
59   * the {@code contentType} property should be specified to apply a
60   * proper content type. Note that a content type header in the target JSP will
61   * be ignored when including the resource via a RequestDispatcher include.
62   *
63   * <p>To apply last-modified timestamps for the target resource, set the
64   * {@code applyLastModified} property to true. This servlet will then
65   * return the file timestamp of the target resource as last-modified value,
66   * falling back to the startup time of this servlet if not retrievable.
67   *
68   * <p>Note that applying the last-modified timestamp in the above fashion
69   * just makes sense if the target resource does not generate content that
70   * depends on the HttpSession or cookies; it is just allowed to evaluate
71   * request parameters.
72   *
73   * <p>A typical case for such last-modified usage is a JSP that just makes
74   * minimal usage of basic means like includes or message resolution to
75   * build quasi-static content. Regenerating such content on every request
76   * is unnecessary; it can be cached as long as the file hasn't changed.
77   *
78   * <p>Note that this servlet will apply the last-modified timestamp if you
79   * tell it to do so: It's your decision whether the content of the target
80   * resource can be cached in such a fashion. Typical use cases are helper
81   * resources that are not fronted by a controller, like JavaScript files
82   * that are generated by a JSP (without depending on the HttpSession).
83   *
84   * @author Juergen Hoeller
85   * @author Rod Johnson
86   * @see #setDefaultUrl
87   * @see #setAllowedResources
88   * @see #setApplyLastModified
89   */
90  @SuppressWarnings("serial")
91  public class ResourceServlet extends HttpServletBean {
92  
93  	/**
94  	 * Any number of these characters are considered delimiters
95  	 * between multiple resource paths in a single String value.
96  	 */
97  	public static final String RESOURCE_URL_DELIMITERS = ",; \t\n";
98  
99  	/**
100 	 * Name of the parameter that must contain the actual resource path.
101 	 */
102 	public static final String RESOURCE_PARAM_NAME = "resource";
103 
104 
105 	private String defaultUrl;
106 
107 	private String allowedResources;
108 
109 	private String contentType;
110 
111 	private boolean applyLastModified = false;
112 
113 	private PathMatcher pathMatcher;
114 
115 	private long startupTime;
116 
117 
118 	/**
119 	 * Set the URL within the current web application from which to
120 	 * include content if the requested path isn't found, or if none
121 	 * is specified in the first place.
122 	 * <p>If specifying multiple URLs, they will be included one by one
123 	 * to build the response. If last-modified determination is active,
124 	 * the newest timestamp among those files will be used.
125 	 * @see #setApplyLastModified
126 	 */
127 	public void setDefaultUrl(String defaultUrl) {
128 		this.defaultUrl = defaultUrl;
129 	}
130 
131 	/**
132 	 * Set allowed resources as URL pattern, e.g. "/WEB-INF/res/*.jsp",
133 	 * The parameter can be any Ant-style pattern parsable by AntPathMatcher.
134 	 * @see org.springframework.util.AntPathMatcher
135 	 */
136 	public void setAllowedResources(String allowedResources) {
137 		this.allowedResources = allowedResources;
138 	}
139 
140 	/**
141 	 * Set the content type of the target resource (typically a JSP).
142 	 * Default is none, which is appropriate when including resources.
143 	 * <p>For directly accessing resources, for example to leverage this
144 	 * servlet's last-modified support, specify a content type here.
145 	 * Note that a content type header in the target JSP will be ignored
146 	 * when including the resource via a RequestDispatcher include.
147 	 */
148 	public void setContentType(String contentType) {
149 		this.contentType = contentType;
150 	}
151 
152 	/**
153 	 * Set whether to apply the file timestamp of the target resource
154 	 * as last-modified value. Default is "false".
155 	 * <p>This is mainly intended for JSP targets that don't generate
156 	 * session-specific or database-driven content: Such files can be
157 	 * cached by the browser as long as the last-modified timestamp
158 	 * of the JSP file doesn't change.
159 	 * <p>This will only work correctly with expanded WAR files that
160 	 * allow access to the file timestamps. Else, the startup time
161 	 * of this servlet is returned.
162 	 */
163 	public void setApplyLastModified(boolean applyLastModified) {
164 		this.applyLastModified = applyLastModified;
165 	}
166 
167 
168 	/**
169 	 * Remember the startup time, using no last-modified time before it.
170 	 */
171 	@Override
172 	protected void initServletBean() {
173 		this.pathMatcher = getPathMatcher();
174 		this.startupTime = System.currentTimeMillis();
175 	}
176 
177 	/**
178 	 * Return a PathMatcher to use for matching the "allowedResources" URL pattern.
179 	 * Default is AntPathMatcher.
180 	 * @see #setAllowedResources
181 	 * @see org.springframework.util.AntPathMatcher
182 	 */
183 	protected PathMatcher getPathMatcher() {
184 		return new AntPathMatcher();
185 	}
186 
187 
188 	/**
189 	 * Determine the URL of the target resource and include it.
190 	 * @see #determineResourceUrl
191 	 */
192 	@Override
193 	protected final void doGet(HttpServletRequest request, HttpServletResponse response)
194 		throws ServletException, IOException {
195 
196 		// determine URL of resource to include
197 		String resourceUrl = determineResourceUrl(request);
198 
199 		if (resourceUrl != null) {
200 			try {
201 				doInclude(request, response, resourceUrl);
202 			}
203 			catch (ServletException ex) {
204 				if (logger.isWarnEnabled()) {
205 					logger.warn("Failed to include content of resource [" + resourceUrl + "]", ex);
206 				}
207 				// Try including default URL if appropriate.
208 				if (!includeDefaultUrl(request, response)) {
209 					throw ex;
210 				}
211 			}
212 			catch (IOException ex) {
213 				if (logger.isWarnEnabled()) {
214 					logger.warn("Failed to include content of resource [" + resourceUrl + "]", ex);
215 				}
216 				// Try including default URL if appropriate.
217 				if (!includeDefaultUrl(request, response)) {
218 					throw ex;
219 				}
220 			}
221 		}
222 
223 		// no resource URL specified -> try to include default URL.
224 		else if (!includeDefaultUrl(request, response)) {
225 			throw new ServletException("No target resource URL found for request");
226 		}
227 	}
228 
229 	/**
230 	 * Determine the URL of the target resource of this request.
231 	 * <p>Default implementation returns the value of the "resource" parameter.
232 	 * Can be overridden in subclasses.
233 	 * @param request current HTTP request
234 	 * @return the URL of the target resource, or {@code null} if none found
235 	 * @see #RESOURCE_PARAM_NAME
236 	 */
237 	protected String determineResourceUrl(HttpServletRequest request) {
238 		return request.getParameter(RESOURCE_PARAM_NAME);
239 	}
240 
241 	/**
242 	 * Include the specified default URL, if appropriate.
243 	 * @param request current HTTP request
244 	 * @param response current HTTP response
245 	 * @return whether a default URL was included
246 	 * @throws ServletException if thrown by the RequestDispatcher
247 	 * @throws IOException if thrown by the RequestDispatcher
248 	 */
249 	private boolean includeDefaultUrl(HttpServletRequest request, HttpServletResponse response)
250 		throws ServletException, IOException {
251 
252 		if (this.defaultUrl == null) {
253 			return false;
254 		}
255 		doInclude(request, response, this.defaultUrl);
256 		return true;
257 	}
258 
259 	/**
260 	 * Include the specified resource via the RequestDispatcher.
261 	 * @param request current HTTP request
262 	 * @param response current HTTP response
263 	 * @param resourceUrl the URL of the target resource
264 	 * @throws ServletException if thrown by the RequestDispatcher
265 	 * @throws IOException if thrown by the RequestDispatcher
266 	 */
267 	private void doInclude(HttpServletRequest request, HttpServletResponse response, String resourceUrl)
268 		throws ServletException, IOException {
269 
270 		if (this.contentType != null) {
271 			response.setContentType(this.contentType);
272 		}
273 		String[] resourceUrls =
274 			StringUtils.tokenizeToStringArray(resourceUrl, RESOURCE_URL_DELIMITERS);
275 		for (int i = 0; i < resourceUrls.length; i++) {
276 			// check whether URL matches allowed resources
277 			if (this.allowedResources != null && !this.pathMatcher.match(this.allowedResources, resourceUrls[i])) {
278 				throw new ServletException("Resource [" + resourceUrls[i] +
279 						"] does not match allowed pattern [" + this.allowedResources + "]");
280 			}
281 			if (logger.isDebugEnabled()) {
282 				logger.debug("Including resource [" + resourceUrls[i] + "]");
283 			}
284 			RequestDispatcher rd = request.getRequestDispatcher(resourceUrls[i]);
285 			rd.include(request, response);
286 		}
287 	}
288 
289 	/**
290 	 * Return the last-modified timestamp of the file that corresponds
291 	 * to the target resource URL (i.e. typically the request ".jsp" file).
292 	 * Will simply return -1 if "applyLastModified" is false (the default).
293 	 * <p>Returns no last-modified date before the startup time of this servlet,
294 	 * to allow for message resolution etc that influences JSP contents,
295 	 * assuming that those background resources might have changed on restart.
296 	 * <p>Returns the startup time of this servlet if the file that corresponds
297 	 * to the target resource URL couldn't be resolved (for example, because
298 	 * the WAR is not expanded).
299 	 * @see #determineResourceUrl
300 	 * @see #getFileTimestamp
301 	 */
302 	@Override
303 	protected final long getLastModified(HttpServletRequest request) {
304 		if (this.applyLastModified) {
305 			String resourceUrl = determineResourceUrl(request);
306 			if (resourceUrl == null) {
307 				resourceUrl = this.defaultUrl;
308 			}
309 			if (resourceUrl != null) {
310 				String[] resourceUrls = StringUtils.tokenizeToStringArray(resourceUrl, RESOURCE_URL_DELIMITERS);
311 				long latestTimestamp = -1;
312 				for (int i = 0; i < resourceUrls.length; i++) {
313 					long timestamp = getFileTimestamp(resourceUrls[i]);
314 					if (timestamp > latestTimestamp) {
315 						latestTimestamp = timestamp;
316 					}
317 				}
318 				return (latestTimestamp > this.startupTime ? latestTimestamp : this.startupTime);
319 			}
320 		}
321 		return -1;
322 	}
323 
324 	/**
325 	 * Return the file timestamp for the given resource.
326 	 * @param resourceUrl the URL of the resource
327 	 * @return the file timestamp in milliseconds, or -1 if not determinable
328 	 */
329 	protected long getFileTimestamp(String resourceUrl) {
330 		ServletContextResource resource = new ServletContextResource(getServletContext(), resourceUrl);
331 		try {
332 			long lastModifiedTime = resource.lastModified();
333 			if (logger.isDebugEnabled()) {
334 				logger.debug("Last-modified timestamp of " + resource + " is " + lastModifiedTime);
335 			}
336 			return lastModifiedTime;
337 		}
338 		catch (IOException ex) {
339 			logger.warn("Couldn't retrieve last-modified timestamp of [" + resource +
340 					"] - using ResourceServlet startup time");
341 			return -1;
342 		}
343 	}
