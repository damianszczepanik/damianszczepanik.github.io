45  	public void testParameterizableViewController() throws Exception {
46  		String viewName = "viewName";
47  		ParameterizableViewController pvc = new ParameterizableViewController();
48  		pvc.setViewName(viewName);
49  		// We don't care about the params.
50  		ModelAndView mv = pvc.handleRequest(new MockHttpServletRequest("GET", "foo.html"), null);
51  		assertTrue("model has no data", mv.getModel().size() == 0);
52  		assertTrue("model has correct viewname", mv.getViewName().equals(viewName));
53  		assertTrue("getViewName matches", pvc.getViewName().equals(viewName));
54  	}

74  	private void doTestServletForwardingController(ServletForwardingController sfc, boolean include)
75  			throws Exception {
76  
77  		HttpServletRequest request = mock(HttpServletRequest.class);
78  		HttpServletResponse response = mock(HttpServletResponse.class);
79  		ServletContext context = mock(ServletContext.class);
80  		RequestDispatcher dispatcher = mock(RequestDispatcher.class);
81  
82  		given(request.getMethod()).willReturn("GET");
83  		given(context.getNamedDispatcher("action")).willReturn(dispatcher);
84  		if (include) {
85  			given(request.getAttribute(WebUtils.INCLUDE_REQUEST_URI_ATTRIBUTE)).willReturn("somePath");
86  		}
87  		else {
88  			given(request.getAttribute(WebUtils.INCLUDE_REQUEST_URI_ATTRIBUTE)).willReturn(null);
89  		}
90  
91  		StaticWebApplicationContext sac = new StaticWebApplicationContext();
92  		sac.setServletContext(context);
93  		sfc.setApplicationContext(sac);
94  		assertNull(sfc.handleRequest(request, response));
95  
96  		if (include) {
97  			verify(dispatcher).include(request, response);
98  		}
99  		else {
100 			verify(dispatcher).forward(request, response);
101 		}
102 	}

104 	public void testServletWrappingController() throws Exception {
105 		HttpServletRequest request = new MockHttpServletRequest("GET", "/somePath");
106 		HttpServletResponse response = new MockHttpServletResponse();
107 
108 		ServletWrappingController swc = new ServletWrappingController();
109 		swc.setServletClass(TestServlet.class);
110 		swc.setServletName("action");
111 		Properties props = new Properties();
112 		props.setProperty("config", "myValue");
113 		swc.setInitParameters(props);
114 
115 		swc.afterPropertiesSet();
116 		assertNotNull(TestServlet.config);
117 		assertEquals("action", TestServlet.config.getServletName());
118 		assertEquals("myValue", TestServlet.config.getInitParameter("config"));
119 		assertNull(TestServlet.request);
120 		assertFalse(TestServlet.destroyed);
121 
122 		assertNull(swc.handleRequest(request, response));
123 		assertEquals(request, TestServlet.request);
124 		assertEquals(response, TestServlet.response);
125 		assertFalse(TestServlet.destroyed);
126 
127 		swc.destroy();
128 		assertTrue(TestServlet.destroyed);
129 	}

155 	public static class TestServlet implements Servlet {
156 
157 		private static ServletConfig config;
158 		private static ServletRequest request;
159 		private static ServletResponse response;
160 		private static boolean destroyed;
161 
162 		public TestServlet() {
163 			config = null;
164 			request = null;
165 			response = null;
166 			destroyed = false;
167 		}
168

170 		public void init(ServletConfig servletConfig) {
171 			config = servletConfig;
172 		}
173 
174 		@Override

176 			return config;
177 		}
178 
179 		@Override
180 		public void service(ServletRequest servletRequest, ServletResponse servletResponse) {
181 			request = servletRequest;
182 			response = servletResponse;
183 		}
184 
185 		@Override
186 		public String getServletInfo() {
187 			return "TestServlet";
188 		}
189

193 		}
