494 	@Override
495 	public void afterPropertiesSet() {
496 		// Do this first, it may add ResponseBody advice beans
497 		initControllerAdviceCache();
498 
499 		if (this.argumentResolvers == null) {
500 			List<HandlerMethodArgumentResolver> resolvers = getDefaultArgumentResolvers();
501 			this.argumentResolvers = new HandlerMethodArgumentResolverComposite().addResolvers(resolvers);
502 		}
503 		if (this.initBinderArgumentResolvers == null) {
504 			List<HandlerMethodArgumentResolver> resolvers = getDefaultInitBinderArgumentResolvers();
505 			this.initBinderArgumentResolvers = new HandlerMethodArgumentResolverComposite().addResolvers(resolvers);
506 		}
507 		if (this.returnValueHandlers == null) {
508 			List<HandlerMethodReturnValueHandler> handlers = getDefaultReturnValueHandlers();
509 			this.returnValueHandlers = new HandlerMethodReturnValueHandlerComposite().addHandlers(handlers);
510 		}
511 	}

513 	private void initControllerAdviceCache() {
514 		if (getApplicationContext() == null) {
515 			return;
516 		}
517 		if (logger.isInfoEnabled()) {
518 			logger.info("Looking for @ControllerAdvice: " + getApplicationContext());
519 		}
520 
521 		List<ControllerAdviceBean> beans = ControllerAdviceBean.findAnnotatedBeans(getApplicationContext());
522 		Collections.sort(beans, new OrderComparator());
523 
524 		List<Object> responseBodyAdviceBeans = new ArrayList<Object>();
525 
526 		for (ControllerAdviceBean bean : beans) {
527 			Set<Method> attrMethods = HandlerMethodSelector.selectMethods(bean.getBeanType(), MODEL_ATTRIBUTE_METHODS);
528 			if (!attrMethods.isEmpty()) {
529 				this.modelAttributeAdviceCache.put(bean, attrMethods);
530 				logger.info("Detected @ModelAttribute methods in " + bean);
531 			}
532 			Set<Method> binderMethods = HandlerMethodSelector.selectMethods(bean.getBeanType(), INIT_BINDER_METHODS);
533 			if (!binderMethods.isEmpty()) {
534 				this.initBinderAdviceCache.put(bean, binderMethods);
535 				logger.info("Detected @InitBinder methods in " + bean);
536 			}
537 			if (ResponseBodyAdvice.class.isAssignableFrom(bean.getBeanType())) {
538 				responseBodyAdviceBeans.add(bean);
539 				logger.info("Detected ResponseBodyAdvice bean in " + bean);
540 			}
541 		}
542 
543 		if (!responseBodyAdviceBeans.isEmpty()) {
544 			this.responseBodyAdvice.addAll(0, responseBodyAdviceBeans);
545 		}
546 	}

548 	/**
549 	 * Return the list of argument resolvers to use including built-in resolvers
550 	 * and custom resolvers provided via {@link #setCustomArgumentResolvers}.
551 	 */
552 	private List<HandlerMethodArgumentResolver> getDefaultArgumentResolvers() {
553 		List<HandlerMethodArgumentResolver> resolvers = new ArrayList<HandlerMethodArgumentResolver>();
554 
555 		// Annotation-based argument resolution
556 		resolvers.add(new RequestParamMethodArgumentResolver(getBeanFactory(), false));
557 		resolvers.add(new RequestParamMapMethodArgumentResolver());
558 		resolvers.add(new PathVariableMethodArgumentResolver());
559 		resolvers.add(new PathVariableMapMethodArgumentResolver());
560 		resolvers.add(new MatrixVariableMethodArgumentResolver());
561 		resolvers.add(new MatrixVariableMapMethodArgumentResolver());
562 		resolvers.add(new ServletModelAttributeMethodProcessor(false));
563 		resolvers.add(new RequestResponseBodyMethodProcessor(getMessageConverters()));
564 		resolvers.add(new RequestPartMethodArgumentResolver(getMessageConverters()));
565 		resolvers.add(new RequestHeaderMethodArgumentResolver(getBeanFactory()));
566 		resolvers.add(new RequestHeaderMapMethodArgumentResolver());
567 		resolvers.add(new ServletCookieValueMethodArgumentResolver(getBeanFactory()));
568 		resolvers.add(new ExpressionValueMethodArgumentResolver(getBeanFactory()));
569 
570 		// Type-based argument resolution
571 		resolvers.add(new ServletRequestMethodArgumentResolver());
572 		resolvers.add(new ServletResponseMethodArgumentResolver());
573 		resolvers.add(new HttpEntityMethodProcessor(getMessageConverters()));
574 		resolvers.add(new RedirectAttributesMethodArgumentResolver());
575 		resolvers.add(new ModelMethodProcessor());
576 		resolvers.add(new MapMethodProcessor());
577 		resolvers.add(new ErrorsMethodArgumentResolver());

683 	@Override
684 	protected ModelAndView handleInternal(HttpServletRequest request,
685 			HttpServletResponse response, HandlerMethod handlerMethod) throws Exception {
686 
687 		if (getSessionAttributesHandler(handlerMethod).hasSessionAttributes()) {
688 			// Always prevent caching in case of session attribute management.
689 			checkAndPrepare(request, response, this.cacheSecondsForSessionAttributeHandlers, true);
690 		}
691 		else {
692 			// Uses configured default cacheSeconds setting.
693 			checkAndPrepare(request, response, true);
694 		}
695 
696 		// Execute invokeHandlerMethod in synchronized block if required.
697 		if (this.synchronizeOnSession) {
698 			HttpSession session = request.getSession(false);
699 			if (session != null) {
700 				Object mutex = WebUtils.getSessionMutex(session);
701 				synchronized (mutex) {
702 					return invokeHandleMethod(request, response, handlerMethod);
703 				}
704 			}
705 		}
706 
707 		return invokeHandleMethod(request, response, handlerMethod);
708 	}

722 	 * Return the {@link SessionAttributesHandler} instance for the given handler type
723 	 * (never {@code null}).
724 	 */
725 	private SessionAttributesHandler getSessionAttributesHandler(HandlerMethod handlerMethod) {
726 		Class<?> handlerType = handlerMethod.getBeanType();
727 		SessionAttributesHandler sessionAttrHandler = this.sessionAttributesHandlerCache.get(handlerType);
728 		if (sessionAttrHandler == null) {
729 			synchronized (this.sessionAttributesHandlerCache) {
730 				sessionAttrHandler = this.sessionAttributesHandlerCache.get(handlerType);
731 				if (sessionAttrHandler == null) {
732 					sessionAttrHandler = new SessionAttributesHandler(handlerType, sessionAttributeStore);
733 					this.sessionAttributesHandlerCache.put(handlerType, sessionAttrHandler);
734 				}
735 			}
736 		}
737 		return sessionAttrHandler;
738 	}

740 	/**
741 	 * Invoke the {@link RequestMapping} handler method preparing a {@link ModelAndView}
742 	 * if view resolution is required.
743 	 */
744 	private ModelAndView invokeHandleMethod(HttpServletRequest request,
745 			HttpServletResponse response, HandlerMethod handlerMethod) throws Exception {
746 
747 		ServletWebRequest webRequest = new ServletWebRequest(request, response);
748 
749 		WebDataBinderFactory binderFactory = getDataBinderFactory(handlerMethod);
750 		ModelFactory modelFactory = getModelFactory(handlerMethod, binderFactory);
751 		ServletInvocableHandlerMethod requestMappingMethod = createRequestMappingMethod(handlerMethod, binderFactory);
752 
753 		ModelAndViewContainer mavContainer = new ModelAndViewContainer();
754 		mavContainer.addAllAttributes(RequestContextUtils.getInputFlashMap(request));
755 		modelFactory.initModel(webRequest, mavContainer, requestMappingMethod);
756 		mavContainer.setIgnoreDefaultModelOnRedirect(this.ignoreDefaultModelOnRedirect);
757 
758 		AsyncWebRequest asyncWebRequest = WebAsyncUtils.createAsyncWebRequest(request, response);
759 		asyncWebRequest.setTimeout(this.asyncRequestTimeout);
760 
761 		final WebAsyncManager asyncManager = WebAsyncUtils.getAsyncManager(request);
762 		asyncManager.setTaskExecutor(this.taskExecutor);
763 		asyncManager.setAsyncWebRequest(asyncWebRequest);
764 		asyncManager.registerCallableInterceptors(this.callableInterceptors);
765 		asyncManager.registerDeferredResultInterceptors(this.deferredResultInterceptors);
766 
767 		if (asyncManager.hasConcurrentResult()) {
768 			Object result = asyncManager.getConcurrentResult();
769 			mavContainer = (ModelAndViewContainer) asyncManager.getConcurrentResultContext()[0];
770 			asyncManager.clearConcurrentResult();
771 
772 			if (logger.isDebugEnabled()) {
773 				logger.debug("Found concurrent result value [" + result + "]");
774 			}
775 			requestMappingMethod = requestMappingMethod.wrapConcurrentResult(result);
776 		}
777 
778 		requestMappingMethod.invokeAndHandle(webRequest, mavContainer);
779 
780 		if (asyncManager.isConcurrentHandlingStarted()) {
781 			return null;
782 		}
783 
784 		return getModelAndView(mavContainer, modelFactory, webRequest);
785 	}

799 	private ModelFactory getModelFactory(HandlerMethod handlerMethod, WebDataBinderFactory binderFactory) {
800 		SessionAttributesHandler sessionAttrHandler = getSessionAttributesHandler(handlerMethod);
801 		Class<?> handlerType = handlerMethod.getBeanType();
802 		Set<Method> methods = this.modelAttributeCache.get(handlerType);
803 		if (methods == null) {
804 			methods = HandlerMethodSelector.selectMethods(handlerType, MODEL_ATTRIBUTE_METHODS);
805 			this.modelAttributeCache.put(handlerType, methods);
806 		}
807 		List<InvocableHandlerMethod> attrMethods = new ArrayList<InvocableHandlerMethod>();
808 		// Global methods first
809 		for (Entry<ControllerAdviceBean, Set<Method>> entry : this.modelAttributeAdviceCache.entrySet()) {
810 			if (entry.getKey().isApplicableToBeanType(handlerType)) {
811 				Object bean = entry.getKey().resolveBean();
812 				for (Method method : entry.getValue()) {
813 					attrMethods.add(createModelAttributeMethod(binderFactory, bean, method));
814 				}
815 			}
816 		}
817 		for (Method method : methods) {
818 			Object bean = handlerMethod.getBean();
819 			attrMethods.add(createModelAttributeMethod(binderFactory, bean, method));
820 		}
821 		return new ModelFactory(attrMethods, binderFactory, sessionAttrHandler);
822 	}

824 	private InvocableHandlerMethod createModelAttributeMethod(WebDataBinderFactory factory, Object bean, Method method) {
825 		InvocableHandlerMethod attrMethod = new InvocableHandlerMethod(bean, method);
826 		attrMethod.setHandlerMethodArgumentResolvers(this.argumentResolvers);
827 		attrMethod.setParameterNameDiscoverer(this.parameterNameDiscoverer);
828 		attrMethod.setDataBinderFactory(factory);
829 		return attrMethod;

832 	private WebDataBinderFactory getDataBinderFactory(HandlerMethod handlerMethod) throws Exception {
833 		Class<?> handlerType = handlerMethod.getBeanType();
834 		Set<Method> methods = this.initBinderCache.get(handlerType);
835 		if (methods == null) {
836 			methods = HandlerMethodSelector.selectMethods(handlerType, INIT_BINDER_METHODS);
837 			this.initBinderCache.put(handlerType, methods);
838 		}
839 		List<InvocableHandlerMethod> initBinderMethods = new ArrayList<InvocableHandlerMethod>();
840 		// Global methods first
841 		for (Entry<ControllerAdviceBean, Set<Method>> entry : this.initBinderAdviceCache .entrySet()) {
842 			if (entry.getKey().isApplicableToBeanType(handlerType)) {
843 				Object bean = entry.getKey().resolveBean();
844 				for (Method method : entry.getValue()) {
845 					initBinderMethods.add(createInitBinderMethod(bean, method));
846 				}
847 			}
848 		}
849 		for (Method method : methods) {
850 			Object bean = handlerMethod.getBean();
851 			initBinderMethods.add(createInitBinderMethod(bean, method));
852 		}
853 		return createDataBinderFactory(initBinderMethods);
854 	}

856 	private InvocableHandlerMethod createInitBinderMethod(Object bean, Method method) {
857 		InvocableHandlerMethod binderMethod = new InvocableHandlerMethod(bean, method);
858 		binderMethod.setHandlerMethodArgumentResolvers(this.initBinderArgumentResolvers);
859 		binderMethod.setDataBinderFactory(new DefaultDataBinderFactory(this.webBindingInitializer));
860 		binderMethod.setParameterNameDiscoverer(this.parameterNameDiscoverer);
861 		return binderMethod;
862 	}

878 	private ModelAndView getModelAndView(ModelAndViewContainer mavContainer,
879 			ModelFactory modelFactory, NativeWebRequest webRequest) throws Exception {
880 
881 		modelFactory.updateModel(webRequest, mavContainer);
882 		if (mavContainer.isRequestHandled()) {
883 			return null;
884 		}
885 		ModelMap model = mavContainer.getModel();
886 		ModelAndView mav = new ModelAndView(mavContainer.getViewName(), model);
887 		if (!mavContainer.isViewReference()) {
888 			mav.setView((View) mavContainer.getView());
889 		}
890 		if (model instanceof RedirectAttributes) {
891 			Map<String, ?> flashAttributes = ((RedirectAttributes) model).getFlashAttributes();
892 			HttpServletRequest request = webRequest.getNativeRequest(HttpServletRequest.class);
893 			RequestContextUtils.getOutputFlashMap(request).putAll(flashAttributes);
894 		}
895 		return mav;
896 	}
