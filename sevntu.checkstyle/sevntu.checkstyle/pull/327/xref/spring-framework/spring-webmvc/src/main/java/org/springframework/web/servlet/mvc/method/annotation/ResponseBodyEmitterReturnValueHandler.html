45  /**
46   * Supports return values of type {@link ResponseBodyEmitter} and also
47   * {@code ResponseEntity<ResponseBodyEmitter>}.
48   *
49   * @author Rossen Stoyanchev
50   * @since 4.2
51   */
52  public class ResponseBodyEmitterReturnValueHandler implements HandlerMethodReturnValueHandler {
53  
54  	private static final Log logger = LogFactory.getLog(ResponseBodyEmitterReturnValueHandler.class);
55  
56  	private final List<HttpMessageConverter<?>> messageConverters;
57  
58  
59  	public ResponseBodyEmitterReturnValueHandler(List<HttpMessageConverter<?>> messageConverters) {
60  		Assert.notEmpty(messageConverters, "'messageConverters' must not be empty");
61  		this.messageConverters = messageConverters;
62  	}
63  
64  	@Override
65  	public boolean supportsReturnType(MethodParameter returnType) {
66  		if (ResponseBodyEmitter.class.isAssignableFrom(returnType.getParameterType())) {
67  			return true;
68  		}
69  		else if (ResponseEntity.class.isAssignableFrom(returnType.getParameterType())) {
70  			Class<?> bodyType = ResolvableType.forMethodParameter(returnType).getGeneric(0).resolve();
71  			return (bodyType != null && ResponseBodyEmitter.class.isAssignableFrom(bodyType));
72  		}
73  		return false;
74  	}
75  
76  	@Override
77  	public void handleReturnValue(Object returnValue, MethodParameter returnType,
78  			ModelAndViewContainer mavContainer, NativeWebRequest webRequest) throws Exception {
79  
80  		if (returnValue == null) {
81  			mavContainer.setRequestHandled(true);
82  			return;
83  		}
84  
85  		HttpServletResponse response = webRequest.getNativeResponse(HttpServletResponse.class);
86  		ServerHttpResponse outputMessage = new ServletServerHttpResponse(response);
87  
88  		if (ResponseEntity.class.isAssignableFrom(returnValue.getClass())) {
89  			ResponseEntity<?> responseEntity = (ResponseEntity<?>) returnValue;
90  			outputMessage.setStatusCode(responseEntity.getStatusCode());
91  			outputMessage.getHeaders().putAll(responseEntity.getHeaders());
92  			returnValue = responseEntity.getBody();
93  			if (returnValue == null) {
94  				mavContainer.setRequestHandled(true);
95  				return;
96  			}
97  		}
98  
99  		Assert.isInstanceOf(ResponseBodyEmitter.class, returnValue);
100 		ResponseBodyEmitter emitter = (ResponseBodyEmitter) returnValue;
101 		emitter.extendResponse(outputMessage);
102 
103 		// Commit the response and wrap to ignore further header changes
104 		outputMessage.getBody();
105 		outputMessage = new StreamingServletServerHttpResponse(outputMessage);
106 
107 		DeferredResult<?> deferredResult = new DeferredResult<Object>();
108 		WebAsyncUtils.getAsyncManager(webRequest).startDeferredResultProcessing(deferredResult, mavContainer);
109 
110 		HttpMessageConvertingHandler handler = new HttpMessageConvertingHandler(outputMessage, deferredResult);
111 		emitter.initialize(handler);
112 	}

115 	/**
116 	 * ResponseBodyEmitter.Handler that writes with HttpMessageConverter's.
117 	 */
118 	private class HttpMessageConvertingHandler implements ResponseBodyEmitter.Handler {
119 
120 		private final ServerHttpResponse outputMessage;
121 
122 		private final DeferredResult<?> deferredResult;
123 
124 
125 		public HttpMessageConvertingHandler(ServerHttpResponse outputMessage, DeferredResult<?> deferredResult) {
126 			this.outputMessage = outputMessage;
127 			this.deferredResult = deferredResult;
128 		}
129 
130 
131 		@Override
132 		public void send(Object data, MediaType mediaType) throws IOException {
133 			sendInternal(data, mediaType);
134 		}
135 
136 		@SuppressWarnings("unchecked")
137 		private <T> void sendInternal(T data, MediaType mediaType) throws IOException {
138 			for (HttpMessageConverter<?> converter : ResponseBodyEmitterReturnValueHandler.this.messageConverters) {
139 				if (converter.canWrite(data.getClass(), mediaType)) {
140 					((HttpMessageConverter<T>) converter).write(data, mediaType, this.outputMessage);
141 					this.outputMessage.flush();
142 					if (logger.isDebugEnabled()) {
143 						logger.debug("Written [" + data + "] using [" + converter + "]");
144 					}
145 					return;
146 				}
147 			}
148 			throw new IllegalArgumentException("No suitable converter for " + data);
149 		}
150 
151 		@Override
152 		public void complete() {
153 			this.deferredResult.setResult(null);
154 		}
155 
156 		@Override
157 		public void completeWithError(Throwable failure) {
158 			this.deferredResult.setErrorResult(failure);
159 		}
160 	}
161

163 	/**
164 	 * Wrap to silently ignore header changes HttpMessageConverter's that would
165 	 * otherwise cause HttpHeaders to raise exceptions.
166 	 */
167 	private static class StreamingServletServerHttpResponse implements ServerHttpResponse {
168 
169 		private final ServerHttpResponse delegate;
170 
171 		private final HttpHeaders mutableHeaders = new HttpHeaders();
172 
173 
174 		public StreamingServletServerHttpResponse(ServerHttpResponse delegate) {
175 			this.delegate = delegate;
176 			this.mutableHeaders.putAll(delegate.getHeaders());
177 		}
178 
179 
180 		@Override
181 		public void setStatusCode(HttpStatus status) {
182 			this.delegate.setStatusCode(status);
183 		}
184 
185 		@Override
186 		public HttpHeaders getHeaders() {
187 			return this.mutableHeaders;
188 		}
189 
190 		@Override
191 		public OutputStream getBody() throws IOException {
192 			return this.delegate.getBody();
193 		}
194 
195 		@Override
196 		public void flush() throws IOException {
197 			this.delegate.flush();
198 		}
199 
200 		@Override
201 		public void close() {
202 			this.delegate.close();
203 		}
204 	}
