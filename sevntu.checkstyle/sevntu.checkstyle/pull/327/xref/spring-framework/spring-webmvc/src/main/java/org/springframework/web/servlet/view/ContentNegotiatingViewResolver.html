55  /**
56   * Implementation of {@link ViewResolver} that resolves a view based on the request file name
57   * or {@code Accept} header.
58   *
59   * <p>The {@code ContentNegotiatingViewResolver} does not resolve views itself, but delegates to
60   * other {@link ViewResolver}s. By default, these other view resolvers are picked up automatically
61   * from the application context, though they can also be set explicitly by using the
62   * {@link #setViewResolvers viewResolvers} property. <strong>Note</strong> that in order for this
63   * view resolver to work properly, the {@link #setOrder order} property needs to be set to a higher
64   * precedence than the others (the default is {@link Ordered#HIGHEST_PRECEDENCE}).
65   *
66   * <p>This view resolver uses the requested {@linkplain MediaType media type} to select a suitable
67   * {@link View} for a request. The requested media type is determined through the configured
68   * {@link ContentNegotiationManager}. Once the requested media type has been determined, this resolver
69   * queries each delegate view resolver for a {@link View} and determines if the requested media type
70   * is {@linkplain MediaType#includes(MediaType) compatible} with the view's
71   * {@linkplain View#getContentType() content type}). The most compatible view is returned.
72   *
73   * <p>Additionally, this view resolver exposes the {@link #setDefaultViews(List) defaultViews} property,
74   * allowing you to override the views provided by the view resolvers. Note that these default views are
75   * offered as candidates, and still need have the content type requested (via file extension, parameter,
76   * or {@code Accept} header, described above).
77   *
78   * <p>For example, if the request path is {@code /view.html}, this view resolver will look for a view
79   * that has the {@code text/html} content type (based on the {@code html} file extension). A request
80   * for {@code /view} with a {@code text/html} request {@code Accept} header has the same result.
81   *
82   * @author Arjen Poutsma
83   * @author Juergen Hoeller
84   * @author Rossen Stoyanchev
85   * @since 3.0
86   * @see ViewResolver
87   * @see InternalResourceViewResolver
88   * @see BeanNameViewResolver
89   */
90  public class ContentNegotiatingViewResolver extends WebApplicationObjectSupport implements ViewResolver, Ordered, InitializingBean {

171 	@Override
172 	protected void initServletContext(ServletContext servletContext) {
173 		Collection<ViewResolver> matchingBeans =
174 				BeanFactoryUtils.beansOfTypeIncludingAncestors(getApplicationContext(), ViewResolver.class).values();
175 		if (this.viewResolvers == null) {
176 			this.viewResolvers = new ArrayList<ViewResolver>(matchingBeans.size());
177 			for (ViewResolver viewResolver : matchingBeans) {
178 				if (this != viewResolver) {
179 					this.viewResolvers.add(viewResolver);
180 				}
181 			}
182 		}
183 		else {
184 			for (int i=0; i < viewResolvers.size(); i++) {
185 				if (matchingBeans.contains(viewResolvers.get(i))) {
186 					continue;
187 				}
188 				String name = viewResolvers.get(i).getClass().getName() + i;
189 				getApplicationContext().getAutowireCapableBeanFactory().initializeBean(viewResolvers.get(i), name);
190 			}
191 
192 		}
193 		if (this.viewResolvers.isEmpty()) {
194 			logger.warn("Did not find any ViewResolvers to delegate to; please configure them using the " +
195 					"'viewResolvers' property on the ContentNegotiatingViewResolver");
196 		}
197 		OrderComparator.sort(this.viewResolvers);
198 		this.cnManagerFactoryBean.setServletContext(servletContext);
199 	}

209 	@Override
210 	public View resolveViewName(String viewName, Locale locale) throws Exception {
211 		RequestAttributes attrs = RequestContextHolder.getRequestAttributes();
212 		Assert.isInstanceOf(ServletRequestAttributes.class, attrs);
213 		List<MediaType> requestedMediaTypes = getMediaTypes(((ServletRequestAttributes) attrs).getRequest());
214 		if (requestedMediaTypes != null) {
215 			List<View> candidateViews = getCandidateViews(viewName, locale, requestedMediaTypes);
216 			View bestView = getBestView(candidateViews, requestedMediaTypes, attrs);
217 			if (bestView != null) {
218 				return bestView;
219 			}
220 		}
221 		if (this.useNotAcceptableStatusCode) {
222 			if (logger.isDebugEnabled()) {
223 				logger.debug("No acceptable view found; returning 406 (Not Acceptable) status code");
224 			}
225 			return NOT_ACCEPTABLE_VIEW;
226 		}
227 		else {
228 			logger.debug("No acceptable view found; returning null");
229 			return null;
230 		}
231 	}

233 	/**
234 	 * Determines the list of {@link MediaType} for the given {@link HttpServletRequest}.
235 	 * @param request the current servlet request
236 	 * @return the list of media types requested, if any
237 	 */
238 	protected List<MediaType> getMediaTypes(HttpServletRequest request) {
239 		try {
240 			ServletWebRequest webRequest = new ServletWebRequest(request);
241 
242 			List<MediaType> acceptableMediaTypes = this.contentNegotiationManager.resolveMediaTypes(webRequest);
243 			acceptableMediaTypes = acceptableMediaTypes.isEmpty() ?
244 					Collections.singletonList(MediaType.ALL) : acceptableMediaTypes;
245 
246 			List<MediaType> producibleMediaTypes = getProducibleMediaTypes(request);
247 			Set<MediaType> compatibleMediaTypes = new LinkedHashSet<MediaType>();
248 			for (MediaType acceptable : acceptableMediaTypes) {
249 				for (MediaType producible : producibleMediaTypes) {
250 					if (acceptable.isCompatibleWith(producible)) {
251 						compatibleMediaTypes.add(getMostSpecificMediaType(acceptable, producible));
252 					}
253 				}
254 			}
255 			List<MediaType> selectedMediaTypes = new ArrayList<MediaType>(compatibleMediaTypes);
256 			MediaType.sortBySpecificityAndQuality(selectedMediaTypes);
257 			if (logger.isDebugEnabled()) {
258 				logger.debug("Requested media types are " + selectedMediaTypes + " based on Accept header types " +
259 						"and producible media types " + producibleMediaTypes + ")");
260 			}
261 			return selectedMediaTypes;
262 		}
263 		catch (HttpMediaTypeNotAcceptableException ex) {
264 			return null;
265 		}
266 	}
267 
268 	@SuppressWarnings("unchecked")
269 	private List<MediaType> getProducibleMediaTypes(HttpServletRequest request) {
270 		Set<MediaType> mediaTypes = (Set<MediaType>)
271 				request.getAttribute(HandlerMapping.PRODUCIBLE_MEDIA_TYPES_ATTRIBUTE);
272 		if (!CollectionUtils.isEmpty(mediaTypes)) {
273 			return new ArrayList<MediaType>(mediaTypes);
274 		}
275 		else {
276 			return Collections.singletonList(MediaType.ALL);
277 		}
278 	}

280 	/**
281 	 * Return the more specific of the acceptable and the producible media types
282 	 * with the q-value of the former.
283 	 */
284 	private MediaType getMostSpecificMediaType(MediaType acceptType, MediaType produceType) {
285 		produceType = produceType.copyQualityValue(acceptType);
286 		return MediaType.SPECIFICITY_COMPARATOR.compare(acceptType, produceType) < 0 ? acceptType : produceType;
287 	}

289 	private List<View> getCandidateViews(String viewName, Locale locale, List<MediaType> requestedMediaTypes)
290 			throws Exception {
291 
292 		List<View> candidateViews = new ArrayList<View>();
293 		for (ViewResolver viewResolver : this.viewResolvers) {
294 			View view = viewResolver.resolveViewName(viewName, locale);
295 			if (view != null) {
296 				candidateViews.add(view);
297 			}
298 			for (MediaType requestedMediaType : requestedMediaTypes) {
299 				List<String> extensions = this.contentNegotiationManager.resolveFileExtensions(requestedMediaType);
300 				for (String extension : extensions) {
301 					String viewNameWithExtension = viewName + "." + extension;
302 					view = viewResolver.resolveViewName(viewNameWithExtension, locale);
303 					if (view != null) {
304 						candidateViews.add(view);
305 					}
306 				}
307 			}
308 		}
309 		if (!CollectionUtils.isEmpty(this.defaultViews)) {
310 			candidateViews.addAll(this.defaultViews);
311 		}
312 		return candidateViews;
313 	}
314 
315 	private View getBestView(List<View> candidateViews, List<MediaType> requestedMediaTypes, RequestAttributes attrs) {
316 		for (View candidateView : candidateViews) {
317 			if (candidateView instanceof SmartView) {
318 				SmartView smartView = (SmartView) candidateView;
319 				if (smartView.isRedirectView()) {
320 					if (logger.isDebugEnabled()) {
321 						logger.debug("Returning redirect view [" + candidateView + "]");
322 					}
323 					return candidateView;
324 				}
325 			}
326 		}
327 		for (MediaType mediaType : requestedMediaTypes) {
328 			for (View candidateView : candidateViews) {
329 				if (StringUtils.hasText(candidateView.getContentType())) {
330 					MediaType candidateContentType = MediaType.parseMediaType(candidateView.getContentType());
331 					if (mediaType.isCompatibleWith(candidateContentType)) {
332 						if (logger.isDebugEnabled()) {
333 							logger.debug("Returning [" + candidateView + "] based on requested media type '"
334 									+ mediaType + "'");
335 						}
336 						attrs.setAttribute(View.SELECTED_CONTENT_TYPE, mediaType, RequestAttributes.SCOPE_REQUEST);
337 						return candidateView;
338 					}
339 				}
340 			}
341 		}
342 		return null;
343 	}
