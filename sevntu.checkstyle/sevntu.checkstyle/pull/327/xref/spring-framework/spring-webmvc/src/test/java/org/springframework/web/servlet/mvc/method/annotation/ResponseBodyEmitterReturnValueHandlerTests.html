66  	@Before
67  	public void setUp() throws Exception {
68  
69  		List<HttpMessageConverter<?>> converters = Arrays.asList(
70  				new StringHttpMessageConverter(), new MappingJackson2HttpMessageConverter());
71  
72  		this.handler = new ResponseBodyEmitterReturnValueHandler(converters);
73  		this.mavContainer = new ModelAndViewContainer();
74  
75  		this.request = new MockHttpServletRequest();
76  		this.response = new MockHttpServletResponse();
77  		this.webRequest = new ServletWebRequest(this.request, this.response);
78  
79  		AsyncWebRequest asyncWebRequest = new StandardServletAsyncWebRequest(this.request, this.response);
80  		WebAsyncUtils.getAsyncManager(this.webRequest).setAsyncWebRequest(asyncWebRequest);
81  		this.request.setAsyncSupported(true);
82  	}

84  	@Test
85  	public void supportsReturnType() throws Exception {
86  		assertTrue(this.handler.supportsReturnType(returnType(TestController.class, "handle")));
87  		assertTrue(this.handler.supportsReturnType(returnType(TestController.class, "handleSse")));
88  		assertTrue(this.handler.supportsReturnType(returnType(TestController.class, "handleResponseEntity")));
89  		assertFalse(this.handler.supportsReturnType(returnType(TestController.class, "handleResponseEntityString")));
90  		assertFalse(this.handler.supportsReturnType(returnType(TestController.class, "handleResponseEntityParameterized")));
91  	}

93  	@Test
94  	public void responseBodyEmitter() throws Exception {
95  		MethodParameter returnType = returnType(TestController.class, "handle");
96  		ResponseBodyEmitter emitter = new ResponseBodyEmitter();
97  		this.handler.handleReturnValue(emitter, returnType, this.mavContainer, this.webRequest);
98  
99  		assertTrue(this.request.isAsyncStarted());
100 		assertEquals("", this.response.getContentAsString());
101 
102 		SimpleBean bean = new SimpleBean();
103 		bean.setId(1L);
104 		bean.setName("Joe");
105 		emitter.send(bean);
106 		emitter.send("\n");
107 
108 		bean.setId(2L);
109 		bean.setName("John");
110 		emitter.send(bean);
111 		emitter.send("\n");
112 
113 		bean.setId(3L);
114 		bean.setName("Jason");
115 		emitter.send(bean);
116 
117 		assertEquals("{\"id\":1,\"name\":\"Joe\"}\n" +
118 						"{\"id\":2,\"name\":\"John\"}\n" +
119 						"{\"id\":3,\"name\":\"Jason\"}",
120 				this.response.getContentAsString());
121 
122 		MockAsyncContext asyncContext = (MockAsyncContext) this.request.getAsyncContext();
123 		assertNull(asyncContext.getDispatchedPath());
124 
125 		emitter.complete();
126 		assertNotNull(asyncContext.getDispatchedPath());
127 	}

129 	@Test
130 	public void sseEmitter() throws Exception {
131 		MethodParameter returnType = returnType(TestController.class, "handleSse");
132 		SseEmitter emitter = new SseEmitter();
133 		this.handler.handleReturnValue(emitter, returnType, this.mavContainer, this.webRequest);
134 
135 		assertTrue(this.request.isAsyncStarted());
136 		assertEquals(200, this.response.getStatus());
137 		assertEquals("text/event-stream", this.response.getContentType());
138 
139 		SimpleBean bean1 = new SimpleBean();
140 		bean1.setId(1L);
141 		bean1.setName("Joe");
142 
143 		SimpleBean bean2 = new SimpleBean();
144 		bean2.setId(2L);
145 		bean2.setName("John");
146 
147 		emitter.send(event().comment("a test").name("update").id("1").reconnectTime(5000L).data(bean1).data(bean2));
148 
149 		assertEquals(":a test\n" +
150 						"name:update\n" +
151 						"id:1\n" +
152 						"retry:5000\n" +
153 						"data:{\"id\":1,\"name\":\"Joe\"}\n" +
154 						"data:{\"id\":2,\"name\":\"John\"}\n" +
155 						"\n",
156 				this.response.getContentAsString());
157 	}

159 	@Test
160 	public void responseEntitySse() throws Exception {
161 		MethodParameter returnType = returnType(TestController.class, "handleResponseEntitySse");
162 		ResponseEntity<SseEmitter> emitter = ResponseEntity.ok().header("foo", "bar").body(new SseEmitter());
163 		this.handler.handleReturnValue(emitter, returnType, this.mavContainer, this.webRequest);
164 
165 		assertTrue(this.request.isAsyncStarted());
166 		assertEquals(200, this.response.getStatus());
167 		assertEquals("text/event-stream", this.response.getContentType());
168 		assertEquals("bar", this.response.getHeader("foo"));
169 	}

173 		MethodParameter returnType = returnType(TestController.class, "handleResponseEntitySse");
174 		ResponseEntity<?> emitter = ResponseEntity.noContent().build();
175 		this.handler.handleReturnValue(emitter, returnType, this.mavContainer, this.webRequest);

177 		assertFalse(this.request.isAsyncStarted());

182 	private MethodParameter returnType(Class<?> clazz, String methodName) throws NoSuchMethodException {
183 		Method method = clazz.getDeclaredMethod(methodName);
184 		return new MethodParameter(method, -1);
185 	}
