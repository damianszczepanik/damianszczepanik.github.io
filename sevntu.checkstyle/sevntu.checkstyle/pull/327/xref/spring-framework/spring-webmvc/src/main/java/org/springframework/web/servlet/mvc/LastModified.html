17  package org.springframework.web.servlet.mvc;
18  
19  import javax.servlet.http.HttpServletRequest;
20  
21  /**
22   * Supports last-modified HTTP requests to facilitate content caching.
23   * Same contract as for the Servlet API's {@code getLastModified} method.
24   *
25   * <p>Delegated to by a {@link org.springframework.web.servlet.HandlerAdapter#getLastModified}
26   * implementation. By default, any Controller or HttpRequestHandler within Spring's
27   * default framework can implement this interface to enable last-modified checking.
28   *
29   * <p><b>Note:</b> Alternative handler implementation approaches have different
30   * last-modified handling styles. For example, Spring 2.5's annotated controller
31   * approach (using {@code @RequestMapping}) provides last-modified support
32   * through the {@link org.springframework.web.context.request.WebRequest#checkNotModified}
33   * method, allowing for last-modified checking within the main handler method.
34   *
35   * @author Rod Johnson
36   * @author Juergen Hoeller
37   * @see javax.servlet.http.HttpServlet#getLastModified
38   * @see Controller
39   * @see SimpleControllerHandlerAdapter
40   * @see org.springframework.web.HttpRequestHandler
41   * @see HttpRequestHandlerAdapter
42   */
43  public interface LastModified {
44  
45  	/**
46  	 * Same contract as for HttpServlet's {@code getLastModified} method.
47  	 * Invoked <b>before</b> request processing.
48  	 * <p>The return value will be sent to the HTTP client as Last-Modified header,
49  	 * and compared with If-Modified-Since headers that the client sends back.
50  	 * The content will only get regenerated if there has been a modification.
51  	 * @param request current HTTP request
52  	 * @return the time the underlying resource was last modified, or -1
53  	 * meaning that the content must always be regenerated
54  	 * @see org.springframework.web.servlet.HandlerAdapter#getLastModified
55  	 * @see javax.servlet.http.HttpServlet#getLastModified
56  	 */
57  	long getLastModified(HttpServletRequest request);
58  
59  }
