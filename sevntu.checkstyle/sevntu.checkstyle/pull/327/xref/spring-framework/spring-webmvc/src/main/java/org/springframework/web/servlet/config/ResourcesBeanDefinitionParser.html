53  /**
54   * {@link org.springframework.beans.factory.xml.BeanDefinitionParser} that parses a
55   * {@code resources} element to register a {@link ResourceHttpRequestHandler} and
56   * register a {@link SimpleUrlHandlerMapping} for mapping resource requests,
57   * and a {@link HttpRequestHandlerAdapter}. Will also create a resource handling
58   * chain with {@link ResourceResolver}s and {@link ResourceTransformer}s.
59   *
60   * @author Keith Donald
61   * @author Jeremy Grelle
62   * @author Brian Clozel
63   * @since 3.0.4
64   */
65  class ResourcesBeanDefinitionParser implements BeanDefinitionParser {
66  
67  	private static final String RESOURCE_CHAIN_CACHE = "spring-resource-chain-cache";
68  
69  	private static final String VERSION_RESOLVER_ELEMENT = "version-resolver";
70  
71  	private static final String VERSION_STRATEGY_ELEMENT = "version-strategy";
72  
73  	private static final String FIXED_VERSION_STRATEGY_ELEMENT = "fixed-version-strategy";
74  
75  	private static final String CONTENT_VERSION_STRATEGY_ELEMENT = "content-version-strategy";
76  
77  	private static final String RESOURCE_URL_PROVIDER = "mvcResourceUrlProvider";
78  
79  
80  	@Override
81  	public BeanDefinition parse(Element element, ParserContext parserContext) {
82  		Object source = parserContext.extractSource(element);
83  
84  		registerUrlProvider(parserContext, source);
85  
86  		String resourceHandlerName = registerResourceHandler(parserContext, element, source);
87  		if (resourceHandlerName == null) {
88  			return null;
89  		}
90  
91  		Map<String, String> urlMap = new ManagedMap<String, String>();
92  		String resourceRequestPath = element.getAttribute("mapping");
93  		if (!StringUtils.hasText(resourceRequestPath)) {
94  			parserContext.getReaderContext().error("The 'mapping' attribute is required.", parserContext.extractSource(element));
95  			return null;
96  		}
97  		urlMap.put(resourceRequestPath, resourceHandlerName);
98  
99  		RuntimeBeanReference pathMatcherRef = MvcNamespaceUtils.registerPathMatcher(null, parserContext, source);
100 		RuntimeBeanReference pathHelperRef = MvcNamespaceUtils.registerUrlPathHelper(null, parserContext, source);
101 
102 		RootBeanDefinition handlerMappingDef = new RootBeanDefinition(SimpleUrlHandlerMapping.class);
103 		handlerMappingDef.setSource(source);
104 		handlerMappingDef.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
105 		handlerMappingDef.getPropertyValues().add("urlMap", urlMap);
106 		handlerMappingDef.getPropertyValues().add("pathMatcher", pathMatcherRef).add("urlPathHelper", pathHelperRef);
107 
108 		String order = element.getAttribute("order");
109 		// Use a default of near-lowest precedence, still allowing for even lower precedence in other mappings
110 		handlerMappingDef.getPropertyValues().add("order", StringUtils.hasText(order) ? order : Ordered.LOWEST_PRECEDENCE - 1);
111 
112 		String beanName = parserContext.getReaderContext().generateBeanName(handlerMappingDef);
113 		parserContext.getRegistry().registerBeanDefinition(beanName, handlerMappingDef);
114 		parserContext.registerComponent(new BeanComponentDefinition(handlerMappingDef, beanName));
115 
116 		// Ensure BeanNameUrlHandlerMapping (SPR-8289) and default HandlerAdapters are not "turned off"
117 		// Register HttpRequestHandlerAdapter
118 		MvcNamespaceUtils.registerDefaultComponents(parserContext, source);
119 
120 		return null;
121 	}
122 
123 	private void registerUrlProvider(ParserContext parserContext, Object source) {
124 		if (!parserContext.getRegistry().containsBeanDefinition(RESOURCE_URL_PROVIDER)) {
125 			RootBeanDefinition urlProvider = new RootBeanDefinition(ResourceUrlProvider.class);
126 			urlProvider.setSource(source);
127 			urlProvider.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
128 			parserContext.getRegistry().registerBeanDefinition(RESOURCE_URL_PROVIDER, urlProvider);
129 			parserContext.registerComponent(new BeanComponentDefinition(urlProvider, RESOURCE_URL_PROVIDER));
130 
131 			RootBeanDefinition interceptor = new RootBeanDefinition(ResourceUrlProviderExposingInterceptor.class);
132 			interceptor.setSource(source);
133 			interceptor.getConstructorArgumentValues().addIndexedArgumentValue(0, urlProvider);
134 
135 			RootBeanDefinition mappedInterceptor = new RootBeanDefinition(MappedInterceptor.class);
136 			mappedInterceptor.setSource(source);
137 			mappedInterceptor.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
138 			mappedInterceptor.getConstructorArgumentValues().addIndexedArgumentValue(0, (Object) null);
139 			mappedInterceptor.getConstructorArgumentValues().addIndexedArgumentValue(1, interceptor);
140 			String mappedInterceptorName = parserContext.getReaderContext().registerWithGeneratedName(mappedInterceptor);
141 			parserContext.registerComponent(new BeanComponentDefinition(mappedInterceptor, mappedInterceptorName));
142 		}
143 	}

145 	private String registerResourceHandler(ParserContext parserContext, Element element, Object source) {
146 		String locationAttr = element.getAttribute("location");
147 		if (!StringUtils.hasText(locationAttr)) {
148 			parserContext.getReaderContext().error("The 'location' attribute is required.", parserContext.extractSource(element));
149 			return null;
150 		}
151 
152 		ManagedList<String> locations = new ManagedList<String>();
153 		locations.addAll(Arrays.asList(StringUtils.commaDelimitedListToStringArray(locationAttr)));
154 
155 		RootBeanDefinition resourceHandlerDef = new RootBeanDefinition(ResourceHttpRequestHandler.class);
156 		resourceHandlerDef.setSource(source);
157 		resourceHandlerDef.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
158 		resourceHandlerDef.getPropertyValues().add("locations", locations);
159 
160 		String cacheSeconds = element.getAttribute("cache-period");
161 		if (StringUtils.hasText(cacheSeconds)) {
162 			resourceHandlerDef.getPropertyValues().add("cacheSeconds", cacheSeconds);
163 		}
164 
165 		Element resourceChainElement = DomUtils.getChildElementByTagName(element, "resource-chain");
166 		if (resourceChainElement != null) {
167 			parseResourceChain(resourceHandlerDef, parserContext, resourceChainElement, source);
168 		}
169 
170 		String beanName = parserContext.getReaderContext().generateBeanName(resourceHandlerDef);
171 		parserContext.getRegistry().registerBeanDefinition(beanName, resourceHandlerDef);
172 		parserContext.registerComponent(new BeanComponentDefinition(resourceHandlerDef, beanName));
173 		return beanName;
174 	}

177 	private void parseResourceChain(RootBeanDefinition resourceHandlerDef, ParserContext parserContext,
178 			Element element, Object source) {
179 
180 		String autoRegistration = element.getAttribute("auto-registration");
181 		boolean isAutoRegistration = !(StringUtils.hasText(autoRegistration) && "false".equals(autoRegistration));
182 
183 		ManagedList<? super Object> resourceResolvers = new ManagedList<Object>();
184 		resourceResolvers.setSource(source);
185 		ManagedList<? super Object> resourceTransformers = new ManagedList<Object>();
186 		resourceTransformers.setSource(source);
187 
188 		parseResourceCache(resourceResolvers, resourceTransformers, element, source);
189 		parseResourceResolversTransformers(isAutoRegistration, resourceResolvers, resourceTransformers,
190 				parserContext, element, source);
191 
192 		if (!resourceResolvers.isEmpty()) {
193 			resourceHandlerDef.getPropertyValues().add("resourceResolvers", resourceResolvers);
194 		}
195 		if (!resourceTransformers.isEmpty()) {
196 			resourceHandlerDef.getPropertyValues().add("resourceTransformers", resourceTransformers);
197 		}
198 	}

200 	private void parseResourceCache(ManagedList<? super Object> resourceResolvers,
201 			ManagedList<? super Object> resourceTransformers, Element element, Object source) {
202 
203 		String resourceCache = element.getAttribute("resource-cache");
204 		if ("true".equals(resourceCache)) {
205 			ConstructorArgumentValues cavs = new ConstructorArgumentValues();
206 
207 			RootBeanDefinition cachingResolverDef = new RootBeanDefinition(CachingResourceResolver.class);
208 			cachingResolverDef.setSource(source);
209 			cachingResolverDef.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
210 			cachingResolverDef.setConstructorArgumentValues(cavs);
211 
212 			RootBeanDefinition cachingTransformerDef = new RootBeanDefinition(CachingResourceTransformer.class);
213 			cachingTransformerDef.setSource(source);
214 			cachingTransformerDef.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
215 			cachingTransformerDef.setConstructorArgumentValues(cavs);
216 
217 			String cacheManagerName = element.getAttribute("cache-manager");
218 			String cacheName = element.getAttribute("cache-name");
219 			if (StringUtils.hasText(cacheManagerName) && StringUtils.hasText(cacheName)) {
220 				RuntimeBeanReference cacheManagerRef = new RuntimeBeanReference(cacheManagerName);
221 				cavs.addIndexedArgumentValue(0, cacheManagerRef);
222 				cavs.addIndexedArgumentValue(1, cacheName);
223 			}
224 			else {
225 				ConstructorArgumentValues cacheCavs = new ConstructorArgumentValues();
226 				cacheCavs.addIndexedArgumentValue(0, RESOURCE_CHAIN_CACHE);
227 				RootBeanDefinition cacheDef = new RootBeanDefinition(ConcurrentMapCache.class);
228 				cacheDef.setSource(source);
229 				cacheDef.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
230 				cacheDef.setConstructorArgumentValues(cacheCavs);
231 				cavs.addIndexedArgumentValue(0, cacheDef);
232 			}
233 			resourceResolvers.add(cachingResolverDef);
234 			resourceTransformers.add(cachingTransformerDef);
235 		}
236 	}

239 	private void parseResourceResolversTransformers(boolean isAutoRegistration,
240 			ManagedList<? super Object> resourceResolvers, ManagedList<? super Object> resourceTransformers,
241 			ParserContext parserContext, Element element, Object source) {
242 
243 		Element resolversElement = DomUtils.getChildElementByTagName(element, "resolvers");
244 		if (resolversElement != null) {
245 			for (Element beanElement : DomUtils.getChildElements(resolversElement)) {
246 				if (VERSION_RESOLVER_ELEMENT.equals(beanElement.getLocalName())) {
247 					RootBeanDefinition versionResolverDef = parseVersionResolver(parserContext, beanElement, source);
248 					versionResolverDef.setSource(source);
249 					resourceResolvers.add(versionResolverDef);
250 					if (isAutoRegistration) {
251 						RootBeanDefinition cssLinkTransformerDef = new RootBeanDefinition(CssLinkResourceTransformer.class);
252 						cssLinkTransformerDef.setSource(source);
253 						cssLinkTransformerDef.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
254 						resourceTransformers.add(cssLinkTransformerDef);
255 					}
256 				}
257 				else {
258 					Object object = parserContext.getDelegate().parsePropertySubElement(beanElement, null);
259 					resourceResolvers.add(object);
260 				}
261 			}
262 		}
263 
264 		if (isAutoRegistration) {
265 			RootBeanDefinition pathResolverDef = new RootBeanDefinition(PathResourceResolver.class);
266 			pathResolverDef.setSource(source);
267 			pathResolverDef.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
268 			resourceResolvers.add(pathResolverDef);
269 		}
270 
271 		Element transformersElement = DomUtils.getChildElementByTagName(element, "transformers");
272 		if (transformersElement != null) {
273 			for (Element beanElement : DomUtils.getChildElementsByTagName(transformersElement, "bean", "ref")) {
274 				Object object = parserContext.getDelegate().parsePropertySubElement(beanElement, null);
275 				resourceTransformers.add(object);
276 			}
277 		}
278 	}

280 	private RootBeanDefinition parseVersionResolver(ParserContext parserContext, Element element, Object source) {
281 		ManagedMap<String, ? super Object> strategyMap = new ManagedMap<String, Object>();
282 		strategyMap.setSource(source);
283 		RootBeanDefinition versionResolverDef = new RootBeanDefinition(VersionResourceResolver.class);
284 		versionResolverDef.setSource(source);
285 		versionResolverDef.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
286 		versionResolverDef.getPropertyValues().addPropertyValue("strategyMap", strategyMap);
287 
288 		for (Element beanElement : DomUtils.getChildElements(element)) {
289 			String[] patterns = StringUtils.commaDelimitedListToStringArray(beanElement.getAttribute("patterns"));
290 			Object strategy = null;
291 			if (FIXED_VERSION_STRATEGY_ELEMENT.equals(beanElement.getLocalName())) {
292 				ConstructorArgumentValues cavs = new ConstructorArgumentValues();
293 				cavs.addIndexedArgumentValue(0, beanElement.getAttribute("version"));
294 				RootBeanDefinition strategyDef = new RootBeanDefinition(FixedVersionStrategy.class);
295 				strategyDef.setSource(source);
296 				strategyDef.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
297 				strategyDef.setConstructorArgumentValues(cavs);
298 				strategy = strategyDef;
299 			}
300 			else if (CONTENT_VERSION_STRATEGY_ELEMENT.equals(beanElement.getLocalName())) {
301 				RootBeanDefinition strategyDef = new RootBeanDefinition(ContentVersionStrategy.class);
302 				strategyDef.setSource(source);
303 				strategyDef.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
304 				strategy = strategyDef;
305 			}
306 			else if (VERSION_STRATEGY_ELEMENT.equals(beanElement.getLocalName())) {
307 				Element childElement = DomUtils.getChildElementsByTagName(beanElement, "bean", "ref").get(0);
308 				strategy = parserContext.getDelegate().parsePropertySubElement(childElement, null);
309 			}
310 			for (String pattern : patterns) {
311 				strategyMap.put(pattern, strategy);
312 			}
313 		}
314 
315 		return versionResolverDef;
316 	}
