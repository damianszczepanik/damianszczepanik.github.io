17  package org.springframework.web.servlet.view.velocity;
18  
19  import java.io.IOException;
20  import javax.servlet.ServletContext;
21  
22  import org.apache.velocity.app.VelocityEngine;
23  import org.apache.velocity.exception.VelocityException;
24  import org.apache.velocity.runtime.resource.loader.ClasspathResourceLoader;
25  
26  import org.springframework.beans.factory.InitializingBean;
27  import org.springframework.context.ResourceLoaderAware;
28  import org.springframework.ui.velocity.VelocityEngineFactory;
29  import org.springframework.web.context.ServletContextAware;
30  
31  /**
32   * JavaBean to configure Velocity for web usage, via the "configLocation"
33   * and/or "velocityProperties" and/or "resourceLoaderPath" bean properties.
34   * The simplest way to use this class is to specify just a "resourceLoaderPath";
35   * you do not need any further configuration then.
36   *
37   * <pre class="code">
38   * &lt;bean id="velocityConfig" class="org.springframework.web.servlet.view.velocity.VelocityConfigurer"&gt;
39   *   &lt;property name="resourceLoaderPath">&lt;value&gt;/WEB-INF/velocity/&lt;/value>&lt;/property&gt;
40   * &lt;/bean&gt;</pre>
41   *
42   * This bean must be included in the application context of any application
43   * using Spring's {@link VelocityView} for web MVC. It exists purely to configure
44   * Velocity; it is not meant to be referenced by application components (just
45   * internally by VelocityView). This class implements {@link VelocityConfig}
46   * in order to be found by VelocityView without depending on the bean name of
47   * this configurer. Each DispatcherServlet may define its own VelocityConfigurer
48   * if desired, potentially with different template loader paths.
49   *
50   * <p>Note that you can also refer to a pre-configured VelocityEngine
51   * instance via the "velocityEngine" property, e.g. set up by
52   * {@link org.springframework.ui.velocity.VelocityEngineFactoryBean},
53   * This allows to share a VelocityEngine for web and email usage, for example.
54   *
55   * <p>This configurer registers the "spring.vm" Velocimacro library for web views
56   * (contained in this package and thus in {@code spring.jar}), which makes
57   * all of Spring's default Velocity macros available to the views.
58   * This allows for using the Spring-provided macros such as follows:
59   *
60   * <pre class="code">
61   * #springBind("person.age")
62   * age is ${status.value}</pre>
63   *
64   * @author Rod Johnson
65   * @author Juergen Hoeller
66   * @author Darren Davison
67   * @see #setConfigLocation
68   * @see #setVelocityProperties
69   * @see #setResourceLoaderPath
70   * @see #setVelocityEngine
71   * @see VelocityView
72   */
73  public class VelocityConfigurer extends VelocityEngineFactory
74  		implements VelocityConfig, InitializingBean, ResourceLoaderAware, ServletContextAware {
75  
76  	/** the name of the resource loader for Spring's bind macros */
77  	private static final String SPRING_MACRO_RESOURCE_LOADER_NAME = "springMacro";
78  
79  	/** the key for the class of Spring's bind macro resource loader */
80  	private static final String SPRING_MACRO_RESOURCE_LOADER_CLASS = "springMacro.resource.loader.class";
81  
82  	/** the name of Spring's default bind macro library */
83  	private static final String SPRING_MACRO_LIBRARY = "org/springframework/web/servlet/view/velocity/spring.vm";
84  
85  
86  	private VelocityEngine velocityEngine;
87  
88  	private ServletContext servletContext;
89  
90  
91  	/**
92  	 * Set a pre-configured VelocityEngine to use for the Velocity web
93  	 * configuration: e.g. a shared one for web and email usage, set up via
94  	 * {@link org.springframework.ui.velocity.VelocityEngineFactoryBean}.
95  	 * <p>Note that the Spring macros will <i>not</i> be enabled automatically in
96  	 * case of an external VelocityEngine passed in here. Make sure to include
97  	 * {@code spring.vm} in your template loader path in such a scenario
98  	 * (if there is an actual need to use those macros).
99  	 * <p>If this is not set, VelocityEngineFactory's properties
100 	 * (inherited by this class) have to be specified.
101 	 */
102 	public void setVelocityEngine(VelocityEngine velocityEngine) {
103 		this.velocityEngine = velocityEngine;
104 	}
105 
106 	@Override
107 	public void setServletContext(ServletContext servletContext) {
108 		this.servletContext = servletContext;
109 	}
110 
111 	/**
112 	 * Initialize VelocityEngineFactory's VelocityEngine
113 	 * if not overridden by a pre-configured VelocityEngine.
114 	 * @see #createVelocityEngine
115 	 * @see #setVelocityEngine
116 	 */
117 	@Override
118 	public void afterPropertiesSet() throws IOException, VelocityException {
119 		if (this.velocityEngine == null) {
120 			this.velocityEngine = createVelocityEngine();
121 		}
122 	}
123 
124 	/**
125 	 * Provides a ClasspathResourceLoader in addition to any default or user-defined
126 	 * loader in order to load the spring Velocity macros from the class path.
127 	 * @see org.apache.velocity.runtime.resource.loader.ClasspathResourceLoader
128 	 */
129 	@Override
130 	protected void postProcessVelocityEngine(VelocityEngine velocityEngine) {
131 		velocityEngine.setApplicationAttribute(ServletContext.class.getName(), this.servletContext);
132 		velocityEngine.setProperty(
133 				SPRING_MACRO_RESOURCE_LOADER_CLASS, ClasspathResourceLoader.class.getName());
134 		velocityEngine.addProperty(
135 				VelocityEngine.RESOURCE_LOADER, SPRING_MACRO_RESOURCE_LOADER_NAME);
136 		velocityEngine.addProperty(
137 				VelocityEngine.VM_LIBRARY, SPRING_MACRO_LIBRARY);
138 
139 		if (logger.isInfoEnabled()) {
140 			logger.info("ClasspathResourceLoader with name '" + SPRING_MACRO_RESOURCE_LOADER_NAME +
141 					"' added to configured VelocityEngine");
142 		}
143 	}
144 
145 	@Override
146 	public VelocityEngine getVelocityEngine() {
147 		return this.velocityEngine;
148 	}
149 
150 }
