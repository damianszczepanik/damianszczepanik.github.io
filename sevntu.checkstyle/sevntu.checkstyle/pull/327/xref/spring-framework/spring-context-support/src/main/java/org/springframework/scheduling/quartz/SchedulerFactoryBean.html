49  /**
50   * {@link FactoryBean} that creates and configures a Quartz {@link org.quartz.Scheduler},
51   * manages its lifecycle as part of the Spring application context, and exposes the
52   * Scheduler as bean reference for dependency injection.
53   *
54   * <p>Allows registration of JobDetails, Calendars and Triggers, automatically
55   * starting the scheduler on initialization and shutting it down on destruction.
56   * In scenarios that just require static registration of jobs at startup, there
57   * is no need to access the Scheduler instance itself in application code.
58   *
59   * <p>For dynamic registration of jobs at runtime, use a bean reference to
60   * this SchedulerFactoryBean to get direct access to the Quartz Scheduler
61   * ({@code org.quartz.Scheduler}). This allows you to create new jobs
62   * and triggers, and also to control and monitor the entire Scheduler.
63   *
64   * <p>Note that Quartz instantiates a new Job for each execution, in
65   * contrast to Timer which uses a TimerTask instance that is shared
66   * between repeated executions. Just JobDetail descriptors are shared.
67   *
68   * <p>When using persistent jobs, it is strongly recommended to perform all
69   * operations on the Scheduler within Spring-managed (or plain JTA) transactions.
70   * Else, database locking will not properly work and might even break.
71   * (See {@link #setDataSource setDataSource} javadoc for details.)
72   *
73   * <p>The preferred way to achieve transactional execution is to demarcate
74   * declarative transactions at the business facade level, which will
75   * automatically apply to Scheduler operations performed within those scopes.
76   * Alternatively, you may add transactional advice for the Scheduler itself.
77   *
78   * <p>Compatible with Quartz 2.1.4 and higher, as of Spring 4.1.
79   *
80   * @author Juergen Hoeller
81   * @since 18.02.2004
82   * @see #setDataSource
83   * @see org.quartz.Scheduler
84   * @see org.quartz.SchedulerFactory
85   * @see org.quartz.impl.StdSchedulerFactory
86   * @see org.springframework.transaction.interceptor.TransactionProxyFactoryBean
87   */
88  public class SchedulerFactoryBean extends SchedulerAccessor implements FactoryBean<Scheduler>,
89  		BeanNameAware, ApplicationContextAware, InitializingBean, DisposableBean, SmartLifecycle {

447 	@Override
448 	public void afterPropertiesSet() throws Exception {
449 		if (this.dataSource == null && this.nonTransactionalDataSource != null) {
450 			this.dataSource = this.nonTransactionalDataSource;
451 		}
452 
453 		if (this.applicationContext != null && this.resourceLoader == null) {
454 			this.resourceLoader = this.applicationContext;
455 		}
456 
457 		// Create SchedulerFactory instance...
458 		SchedulerFactory schedulerFactory = BeanUtils.instantiateClass(this.schedulerFactoryClass);
459 		initSchedulerFactory(schedulerFactory);
460 
461 		if (this.resourceLoader != null) {
462 			// Make given ResourceLoader available for SchedulerFactory configuration.
463 			configTimeResourceLoaderHolder.set(this.resourceLoader);
464 		}
465 		if (this.taskExecutor != null) {
466 			// Make given TaskExecutor available for SchedulerFactory configuration.
467 			configTimeTaskExecutorHolder.set(this.taskExecutor);
468 		}
469 		if (this.dataSource != null) {
470 			// Make given DataSource available for SchedulerFactory configuration.
471 			configTimeDataSourceHolder.set(this.dataSource);
472 		}
473 		if (this.nonTransactionalDataSource != null) {
474 			// Make given non-transactional DataSource available for SchedulerFactory configuration.
475 			configTimeNonTransactionalDataSourceHolder.set(this.nonTransactionalDataSource);
476 		}
477 
478 		// Get Scheduler instance from SchedulerFactory.
479 		try {
480 			this.scheduler = createScheduler(schedulerFactory, this.schedulerName);
481 			populateSchedulerContext();
482 
483 			if (!this.jobFactorySet && !(this.scheduler instanceof RemoteScheduler)) {
484 				// Use AdaptableJobFactory as default for a local Scheduler, unless when
485 				// explicitly given a null value through the "jobFactory" bean property.
486 				this.jobFactory = new AdaptableJobFactory();
487 			}
488 			if (this.jobFactory != null) {
489 				if (this.jobFactory instanceof SchedulerContextAware) {
490 					((SchedulerContextAware) this.jobFactory).setSchedulerContext(this.scheduler.getContext());
491 				}
492 				this.scheduler.setJobFactory(this.jobFactory);
493 			}
494 		}
495 
496 		finally {
497 			if (this.resourceLoader != null) {
498 				configTimeResourceLoaderHolder.remove();
499 			}
500 			if (this.taskExecutor != null) {
501 				configTimeTaskExecutorHolder.remove();
502 			}
503 			if (this.dataSource != null) {
504 				configTimeDataSourceHolder.remove();
505 			}
506 			if (this.nonTransactionalDataSource != null) {
507 				configTimeNonTransactionalDataSourceHolder.remove();
508 			}
509 		}
510 
511 		registerListeners();
512 		registerJobsAndTriggers();
513 	}

516 	/**
517 	 * Load and/or apply Quartz properties to the given SchedulerFactory.
518 	 * @param schedulerFactory the SchedulerFactory to initialize
519 	 */
520 	private void initSchedulerFactory(SchedulerFactory schedulerFactory) throws SchedulerException, IOException {
521 		if (!(schedulerFactory instanceof StdSchedulerFactory)) {
522 			if (this.configLocation != null || this.quartzProperties != null ||
523 					this.taskExecutor != null || this.dataSource != null) {
524 				throw new IllegalArgumentException(
525 						"StdSchedulerFactory required for applying Quartz properties: " + schedulerFactory);
526 			}
527 			// Otherwise assume that no initialization is necessary...
528 			return;
529 		}
530 
531 		Properties mergedProps = new Properties();
532 
533 		if (this.resourceLoader != null) {
534 			mergedProps.setProperty(StdSchedulerFactory.PROP_SCHED_CLASS_LOAD_HELPER_CLASS,
535 					ResourceLoaderClassLoadHelper.class.getName());
536 		}
537 
538 		if (this.taskExecutor != null) {
539 			mergedProps.setProperty(StdSchedulerFactory.PROP_THREAD_POOL_CLASS,
540 					LocalTaskExecutorThreadPool.class.getName());
541 		}
542 		else {
543 			// Set necessary default properties here, as Quartz will not apply
544 			// its default configuration when explicitly given properties.
545 			mergedProps.setProperty(StdSchedulerFactory.PROP_THREAD_POOL_CLASS, SimpleThreadPool.class.getName());
546 			mergedProps.setProperty(PROP_THREAD_COUNT, Integer.toString(DEFAULT_THREAD_COUNT));
547 		}
548 
549 		if (this.configLocation != null) {
550 			if (logger.isInfoEnabled()) {
551 				logger.info("Loading Quartz config from [" + this.configLocation + "]");
552 			}
553 			PropertiesLoaderUtils.fillProperties(mergedProps, this.configLocation);
554 		}
555 
556 		CollectionUtils.mergePropertiesIntoMap(this.quartzProperties, mergedProps);
557 
558 		if (this.dataSource != null) {
559 			mergedProps.put(StdSchedulerFactory.PROP_JOB_STORE_CLASS, LocalDataSourceJobStore.class.getName());
560 		}
561 
562 		// Make sure to set the scheduler name as configured in the Spring configuration.
563 		if (this.schedulerName != null) {
564 			mergedProps.put(StdSchedulerFactory.PROP_SCHED_INSTANCE_NAME, this.schedulerName);
565 		}
566 
567 		((StdSchedulerFactory) schedulerFactory).initialize(mergedProps);
568 	}

570 	/**
571 	 * Create the Scheduler instance for the given factory and scheduler name.
572 	 * Called by {@link #afterPropertiesSet}.
573 	 * <p>The default implementation invokes SchedulerFactory's {@code getScheduler}
574 	 * method. Can be overridden for custom Scheduler creation.
575 	 * @param schedulerFactory the factory to create the Scheduler with
576 	 * @param schedulerName the name of the scheduler to create
577 	 * @return the Scheduler instance
578 	 * @throws SchedulerException if thrown by Quartz methods
579 	 * @see #afterPropertiesSet
580 	 * @see org.quartz.SchedulerFactory#getScheduler
581 	 */
582 	protected Scheduler createScheduler(SchedulerFactory schedulerFactory, String schedulerName)
583 			throws SchedulerException {
584 
585 		// Override thread context ClassLoader to work around naive Quartz ClassLoadHelper loading.
586 		Thread currentThread = Thread.currentThread();
587 		ClassLoader threadContextClassLoader = currentThread.getContextClassLoader();
588 		boolean overrideClassLoader = (this.resourceLoader != null &&
589 				!this.resourceLoader.getClassLoader().equals(threadContextClassLoader));
590 		if (overrideClassLoader) {
591 			currentThread.setContextClassLoader(this.resourceLoader.getClassLoader());
592 		}
593 		try {
594 			SchedulerRepository repository = SchedulerRepository.getInstance();
595 			synchronized (repository) {
596 				Scheduler existingScheduler = (schedulerName != null ? repository.lookup(schedulerName) : null);
597 				Scheduler newScheduler = schedulerFactory.getScheduler();
598 				if (newScheduler == existingScheduler) {
599 					throw new IllegalStateException("Active Scheduler of name '" + schedulerName + "' already registered " +
600 							"in Quartz SchedulerRepository. Cannot create a new Spring-managed Scheduler of the same name!");
601 				}
602 				if (!this.exposeSchedulerInRepository) {
603 					// Need to remove it in this case, since Quartz shares the Scheduler instance by default!
604 					SchedulerRepository.getInstance().remove(newScheduler.getSchedulerName());
605 				}
606 				return newScheduler;
607 			}
608 		}
609 		finally {
610 			if (overrideClassLoader) {
611 				// Reset original thread context ClassLoader.
612 				currentThread.setContextClassLoader(threadContextClassLoader);
613 			}
614 		}
615 	}

617 	/**
618 	 * Expose the specified context attributes and/or the current
619 	 * ApplicationContext in the Quartz SchedulerContext.
620 	 */
621 	private void populateSchedulerContext() throws SchedulerException {
622 		// Put specified objects into Scheduler context.
623 		if (this.schedulerContextMap != null) {
624 			this.scheduler.getContext().putAll(this.schedulerContextMap);
625 		}
626 
627 		// Register ApplicationContext in Scheduler context.
628 		if (this.applicationContextSchedulerContextKey != null) {
629 			if (this.applicationContext == null) {
630 				throw new IllegalStateException(
631 					"SchedulerFactoryBean needs to be set up in an ApplicationContext " +
632 					"to be able to handle an 'applicationContextSchedulerContextKey'");
633 			}
634 			this.scheduler.getContext().put(this.applicationContextSchedulerContextKey, this.applicationContext);
635 		}
636 	}
637 
638 
639 	/**
640 	 * Start the Quartz Scheduler, respecting the "startupDelay" setting.
641 	 * @param scheduler the Scheduler to start
642 	 * @param startupDelay the number of seconds to wait before starting
643 	 * the Scheduler asynchronously
644 	 */
645 	protected void startScheduler(final Scheduler scheduler, final int startupDelay) throws SchedulerException {
646 		if (startupDelay <= 0) {
647 			logger.info("Starting Quartz Scheduler now");
648 			scheduler.start();
649 		}
650 		else {
651 			if (logger.isInfoEnabled()) {
652 				logger.info("Will start Quartz Scheduler [" + scheduler.getSchedulerName() +
653 						"] in " + startupDelay + " seconds");
654 			}
655 			Thread schedulerThread = new Thread() {
656 				@Override
657 				public void run() {
658 					try {
659 						Thread.sleep(startupDelay * 1000);
660 					}
661 					catch (InterruptedException ex) {
662 						// simply proceed
663 					}
664 					if (logger.isInfoEnabled()) {
665 						logger.info("Starting Quartz Scheduler now, after delay of " + startupDelay + " seconds");
666 					}
667 					try {
668 						scheduler.start();
669 					}
670 					catch (SchedulerException ex) {
671 						throw new SchedulingException("Could not start Quartz Scheduler after delay", ex);
672 					}
673 				}
674 			};
675 			schedulerThread.setName("Quartz Scheduler [" + scheduler.getSchedulerName() + "]");
676 			schedulerThread.setDaemon(true);
677 			schedulerThread.start();
678 		}
679 	}

725 		if (this.scheduler != null) {
726 			try {
727 				this.scheduler.standby();
728 			}
729 			catch (SchedulerException ex) {
730 				throw new SchedulingException("Could not stop Quartz Scheduler", ex);
731 			}
732 		}
733 	}

737 		stop();
738 		callback.run();
739 	}

743 		if (this.scheduler != null) {
744 			try {
745 				return !this.scheduler.isInStandbyMode();
746 			}
747 			catch (SchedulerException ex) {
748 				return false;
749 			}
