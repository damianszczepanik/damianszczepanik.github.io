17  package org.springframework.scheduling.quartz;
18  
19  import org.quartz.Job;
20  import org.quartz.JobExecutionContext;
21  import org.quartz.JobExecutionException;
22  import org.quartz.SchedulerException;
23  
24  import org.springframework.beans.BeanWrapper;
25  import org.springframework.beans.MutablePropertyValues;
26  import org.springframework.beans.PropertyAccessorFactory;
27  
28  /**
29   * Simple implementation of the Quartz Job interface, applying the
30   * passed-in JobDataMap and also the SchedulerContext as bean property
31   * values. This is appropriate because a new Job instance will be created
32   * for each execution. JobDataMap entries will override SchedulerContext
33   * entries with the same keys.
34   *
35   * <p>For example, let's assume that the JobDataMap contains a key
36   * "myParam" with value "5": The Job implementation can then expose
37   * a bean property "myParam" of type int to receive such a value,
38   * i.e. a method "setMyParam(int)". This will also work for complex
39   * types like business objects etc.
40   *
41   * <p><b>Note that the preferred way to apply dependency injection
42   * to Job instances is via a JobFactory:</b> that is, to specify
43   * {@link SpringBeanJobFactory} as Quartz JobFactory (typically via
44   * {@link SchedulerFactoryBean#setJobFactory} SchedulerFactoryBean's "jobFactory" property}).
45   * This allows to implement dependency-injected Quartz Jobs without
46   * a dependency on Spring base classes.
47   *
48   * @author Juergen Hoeller
49   * @since 18.02.2004
50   * @see org.quartz.JobExecutionContext#getMergedJobDataMap()
51   * @see org.quartz.Scheduler#getContext()
52   * @see SchedulerFactoryBean#setSchedulerContextAsMap
53   * @see SpringBeanJobFactory
54   * @see SchedulerFactoryBean#setJobFactory
55   */
56  public abstract class QuartzJobBean implements Job {
57  
58  	/**
59  	 * This implementation applies the passed-in job data map as bean property
60  	 * values, and delegates to {@code executeInternal} afterwards.
61  	 * @see #executeInternal
62  	 */
63  	@Override
64  	public final void execute(JobExecutionContext context) throws JobExecutionException {
65  		try {
66  			BeanWrapper bw = PropertyAccessorFactory.forBeanPropertyAccess(this);
67  			MutablePropertyValues pvs = new MutablePropertyValues();
68  			pvs.addPropertyValues(context.getScheduler().getContext());
69  			pvs.addPropertyValues(context.getMergedJobDataMap());
70  			bw.setPropertyValues(pvs, true);
71  		}
72  		catch (SchedulerException ex) {
73  			throw new JobExecutionException(ex);
74  		}
75  		executeInternal(context);
76  	}
77  
78  	/**
79  	 * Execute the actual job. The job data map will already have been
80  	 * applied as bean property values by execute. The contract is
81  	 * exactly the same as for the standard Quartz execute method.
82  	 * @see #execute
83  	 */
84  	protected abstract void executeInternal(JobExecutionContext context) throws JobExecutionException;
85  
86  }
