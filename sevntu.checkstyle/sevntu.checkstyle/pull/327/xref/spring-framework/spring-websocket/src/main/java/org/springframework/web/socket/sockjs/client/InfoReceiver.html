5   /**
6    * A simple contract for executing the SockJS "Info" request before the SockJS
7    * session starts. The request is used to check server capabilities such as
8    * whether it permits use of the WebSocket transport.
9    *
10   * @author Rossen Stoyanchev
11   * @since 4.1
12   */
13  public interface InfoReceiver {
14  
15  	/**
16  	 * Perform an HTTP request to the SockJS "Info" URL.
17  	 * and return the resulting JSON response content, or raise an exception.
18  	 *
19  	 * @param infoUrl the URL to obtain SockJS server information from
20  	 * @return the body of the response
21  	 */
22  	String executeInfoRequest(URI infoUrl);
23  
24  }
