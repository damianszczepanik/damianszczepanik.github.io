1   package org.springframework.web.socket.sockjs.client;

3   import java.net.URI;
4   
5   import org.springframework.web.socket.TextMessage;
6   
7   /**
8    * A SockJS {@link Transport} that uses HTTP requests to simulate a WebSocket
9    * interaction. The {@code connect} method of the base {@code Transport} interface
10   * is used to receive messages from the server while the
11   * {@link #executeSendRequest(java.net.URI, org.springframework.web.socket.TextMessage)
12   * executeSendRequest(URI, TextMessage)} method here is used to send messages.
13   *
14   * @author Rossen Stoyanchev
15   * @since 4.1
16   */
17  public interface XhrTransport extends Transport, InfoReceiver {
18  
19  	/**
20  	 * An {@code XhrTransport} supports both the "xhr_streaming" and "xhr" SockJS
21  	 * server transports. From a client perspective there is no implementation
22  	 * difference.
23  	 *
24  	 * <p>By default an {@code XhrTransport} will be used with "xhr_streaming"
25  	 * first and then with "xhr", if the streaming fails to connect. In some
26  	 * cases it may be useful to suppress streaming so that only "xhr" is used.
27  	 */
28  	boolean isXhrStreamingDisabled();
29  
30  	/**
31  	 * Execute a request to send the message to the server.
32  	 * @param transportUrl the URL for sending messages.
33  	 * @param message the message to send
34  	 */
35  	void executeSendRequest(URI transportUrl, TextMessage message);
36  
37  }
