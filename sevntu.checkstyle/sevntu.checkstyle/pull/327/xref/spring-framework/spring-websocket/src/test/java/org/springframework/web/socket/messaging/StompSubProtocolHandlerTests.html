155 	@Test
156 	public void handleMessageToClientConnectAck() {
157 
158 		StompHeaderAccessor connectHeaders = StompHeaderAccessor.create(StompCommand.CONNECT);
159 		connectHeaders.setHeartbeat(10000, 10000);
160 		connectHeaders.setAcceptVersion("1.0,1.1");
161 		Message<?> connectMessage = MessageBuilder.createMessage(EMPTY_PAYLOAD, connectHeaders.getMessageHeaders());
162 
163 		SimpMessageHeaderAccessor connectAckHeaders = SimpMessageHeaderAccessor.create(SimpMessageType.CONNECT_ACK);
164 		connectAckHeaders.setHeader(SimpMessageHeaderAccessor.CONNECT_MESSAGE_HEADER, connectMessage);
165 		Message<byte[]> connectAckMessage = MessageBuilder.createMessage(EMPTY_PAYLOAD, connectAckHeaders.getMessageHeaders());
166 
167 		this.protocolHandler.handleMessageToClient(this.session, connectAckMessage);
168 
169 		verifyNoMoreInteractions(this.channel);
170 
171 		// Check CONNECTED reply
172 
173 		assertEquals(1, this.session.getSentMessages().size());
174 		TextMessage textMessage = (TextMessage) this.session.getSentMessages().get(0);
175 
176 		List<Message<byte[]>> messages = new StompDecoder().decode(ByteBuffer.wrap(textMessage.getPayload().getBytes()));
177 		assertEquals(1, messages.size());
178 		StompHeaderAccessor replyHeaders = StompHeaderAccessor.wrap(messages.get(0));
179 
180 		assertEquals(StompCommand.CONNECTED, replyHeaders.getCommand());
181 		assertEquals("1.1", replyHeaders.getVersion());
182 		assertArrayEquals(new long[] {0, 0}, replyHeaders.getHeartbeat());
183 		assertEquals("joe", replyHeaders.getNativeHeader("user-name").get(0));
184 	}

198 		TextMessage textMessage = new TextMessage(new StompEncoder().encode(message));
199 		this.protocolHandler.handleMessageFromClient(this.session, textMessage, this.channel);
200 
201 		headers = StompHeaderAccessor.create(StompCommand.CONNECTED);
202 		message = MessageBuilder.createMessage(EMPTY_PAYLOAD, headers.getMessageHeaders());
203 		this.protocolHandler.handleMessageToClient(this.session, message);
204 
205 		headers = StompHeaderAccessor.create(StompCommand.SUBSCRIBE);
206 		message = MessageBuilder.createMessage(EMPTY_PAYLOAD, headers.getMessageHeaders());
207 		textMessage = new TextMessage(new StompEncoder().encode(message));
208 		this.protocolHandler.handleMessageFromClient(this.session, textMessage, this.channel);
209 
210 		headers = StompHeaderAccessor.create(StompCommand.UNSUBSCRIBE);
211 		message = MessageBuilder.createMessage(EMPTY_PAYLOAD, headers.getMessageHeaders());
212 		textMessage = new TextMessage(new StompEncoder().encode(message));
213 		this.protocolHandler.handleMessageFromClient(this.session, textMessage, this.channel);
214 
215 		this.protocolHandler.afterSessionEnded(this.session, CloseStatus.BAD_DATA, this.channel);
216 
217 		assertEquals("Unexpected events " + publisher.events, 5, publisher.events.size());
218 		assertEquals(SessionConnectEvent.class, publisher.events.get(0).getClass());
219 		assertEquals(SessionConnectedEvent.class, publisher.events.get(1).getClass());
220 		assertEquals(SessionSubscribeEvent.class, publisher.events.get(2).getClass());
221 		assertEquals(SessionUnsubscribeEvent.class, publisher.events.get(3).getClass());
222 		assertEquals(SessionDisconnectEvent.class, publisher.events.get(4).getClass());

342 		AtomicReference<Boolean> mutable = new AtomicReference<>();
343 		ExecutorSubscribableChannel channel = new ExecutorSubscribableChannel();
344 		channel.addInterceptor(new ChannelInterceptorAdapter() {
345 			@Override
346 			public Message<?> preSend(Message<?> message, MessageChannel channel) {
347 				mutable.set(MessageHeaderAccessor.getAccessor(message, MessageHeaderAccessor.class).isMutable());
348 				return message;
349 			}
350 		});
351 		channel.addInterceptor(new ImmutableMessageChannelInterceptor());
352 
353 		StompSubProtocolHandler handler = new StompSubProtocolHandler();
354 		handler.afterSessionStarted(this.session, channel);
355 
356 		TextMessage message = StompTextMessageBuilder.create(StompCommand.CONNECT).build();
357 		handler.handleMessageFromClient(this.session, message, channel);
358 		assertNotNull(mutable.get());
359 		assertTrue(mutable.get());
