53  /**
54   * An implementation of {@link WebSocketHandler} that delegates incoming WebSocket
55   * messages to a {@link SubProtocolHandler} along with a {@link MessageChannel} to
56   * which the sub-protocol handler can send messages from WebSocket clients to
57   * the application.
58   * <p>
59   * Also an implementation of {@link MessageHandler} that finds the WebSocket
60   * session associated with the {@link Message} and passes it, along with the message,
61   * to the sub-protocol handler to send messages from the application back to the
62   * client.
63   *
64   * @author Rossen Stoyanchev
65   * @author Andy Wilkinson
66   * @author Artem Bilan
67   * @since 4.0
68   */
69  public class SubProtocolWebSocketHandler implements WebSocketHandler,
70  		SubProtocolCapable, MessageHandler, SmartLifecycle {
71  
72  	/**
73  	 * Sessions connected to this handler use a sub-protocol. Hence we expect to
74  	 * receive some client messages. If we don't receive any within a minute, the
75  	 * connection isn't doing well (proxy issue, slow network?) and can be closed.
76  	 * @see #checkSessions()
77  	 */
78  	private static final int TIME_TO_FIRST_MESSAGE = 60 * 1000;

121 	 * Configure one or more handlers to use depending on the sub-protocol requested by
122 	 * the client in the WebSocket handshake request.
123 	 * @param protocolHandlers the sub-protocol handlers to use
124 	 */
125 	public void setProtocolHandlers(List<SubProtocolHandler> protocolHandlers) {
126 		this.protocolHandlerLookup.clear();
127 		this.protocolHandlers.clear();
128 		for (SubProtocolHandler handler: protocolHandlers) {
129 			addProtocolHandler(handler);
130 		}

138 	/**
139 	 * Register a sub-protocol handler.
140 	 */
141 	public void addProtocolHandler(SubProtocolHandler handler) {
142 		List<String> protocols = handler.getSupportedProtocols();
143 		if (CollectionUtils.isEmpty(protocols)) {
144 			logger.error("No sub-protocols for " + handler + ".");
145 			return;
146 		}
147 		for (String protocol: protocols) {
148 			SubProtocolHandler replaced = this.protocolHandlerLookup.put(protocol, handler);
149 			if ((replaced != null) && (replaced != handler) ) {
150 				throw new IllegalStateException("Can't map " + handler +
151 						" to protocol '" + protocol + "'. Already mapped to " + replaced + ".");
152 			}
153 		}
154 		this.protocolHandlers.add(handler);
155 	}

232 	@Override
233 	public final void start() {
234 		Assert.isTrue(this.defaultProtocolHandler != null || !this.protocolHandlers.isEmpty(), "No handlers");
235 		synchronized (this.lifecycleMonitor) {
236 			this.clientOutboundChannel.subscribe(this);
237 			this.running = true;
238 		}
239 	}
240 
241 	@Override
242 	public final void stop() {
243 		synchronized (this.lifecycleMonitor) {
244 			this.running = false;
245 			this.clientOutboundChannel.unsubscribe(this);
246 			for (WebSocketSessionHolder holder : this.sessions.values()) {
247 				try {
248 					holder.getSession().close(CloseStatus.GOING_AWAY);
249 				}
250 				catch (Throwable t) {
251 					logger.error("Failed to close '" + holder.getSession() + "': " + t.getMessage());
252 				}
253 			}
254 		}
255 	}
256 
257 	@Override
258 	public final void stop(Runnable callback) {
259 		synchronized (this.lifecycleMonitor) {
260 			stop();
261 			callback.run();
262 		}
263 	}

265 	@Override
266 	public void afterConnectionEstablished(WebSocketSession session) throws Exception {
267 		this.stats.incrementSessionCount(session);
268 		session = new ConcurrentWebSocketSessionDecorator(session, getSendTimeLimit(), getSendBufferSizeLimit());
269 		this.sessions.put(session.getId(), new WebSocketSessionHolder(session));
270 		findProtocolHandler(session).afterSessionStarted(session, this.clientInboundChannel);
271 	}
272 
273 	protected final SubProtocolHandler findProtocolHandler(WebSocketSession session) {
274 		String protocol = null;
275 		try {
276 			protocol = session.getAcceptedProtocol();
277 		}
278 		catch (Exception ex) {
279 			// Shouldn't happen
280 			logger.error("Failed to obtain session.getAcceptedProtocol(). Will use the " +
281 					"default protocol handler (if configured).", ex);
282 		}
283 		SubProtocolHandler handler;
284 		if (!StringUtils.isEmpty(protocol)) {
285 			handler = this.protocolHandlerLookup.get(protocol);
286 			Assert.state(handler != null, "No handler for '" + protocol + "' among " + this.protocolHandlerLookup);
287 		}
288 		else {
289 			if (this.defaultProtocolHandler != null) {
290 				handler = this.defaultProtocolHandler;
291 			}
292 			else if (this.protocolHandlers.size() == 1) {
293 				handler = this.protocolHandlers.iterator().next();
294 			}
295 			else {
296 				throw new IllegalStateException("Multiple protocol handlers configured and " +
297 						"no protocol was negotiated. Consider configuring a default SubProtocolHandler.");
298 			}
299 		}
300 		return handler;
301 	}

303 	/**
304 	 * Handle an inbound message from a WebSocket client.
305 	 */
306 	@Override
307 	public void handleMessage(WebSocketSession session, WebSocketMessage<?> message) throws Exception {
308 		SubProtocolHandler protocolHandler = findProtocolHandler(session);
309 		protocolHandler.handleMessageFromClient(session, message, this.clientInboundChannel);
310 		WebSocketSessionHolder holder = this.sessions.get(session.getId());
311 		if (holder != null) {
312 			holder.setHasHandledMessages();
313 		}
314 		checkSessions();
315 	}
316 
317 	/**
318 	 * Handle an outbound Spring Message to a WebSocket client.
319 	 */
320 	@Override
321 	public void handleMessage(Message<?> message) throws MessagingException {
322 		String sessionId = resolveSessionId(message);
323 		if (sessionId == null) {
324 			logger.error("Couldn't find sessionId in " + message);
325 			return;
326 		}
327 		WebSocketSessionHolder holder = this.sessions.get(sessionId);
328 		if (holder == null) {
329 			if (logger.isDebugEnabled()) {
330 				// The broker may not have removed the session yet
331 				logger.debug("No session for " + message);
332 			}
333 			return;
334 		}
335 		WebSocketSession session = holder.getSession();
336 		try {
337 			findProtocolHandler(session).handleMessageToClient(session, message);
338 		}
339 		catch (SessionLimitExceededException ex) {
340 			try {
341 				if (logger.isDebugEnabled()) {
342 					logger.debug("Terminating '" + session + "'", ex);
343 				}
344 				this.stats.incrementLimitExceededCount();
345 				clearSession(session, ex.getStatus()); // clear first, session may be unresponsive
346 				session.close(ex.getStatus());
347 			}
348 			catch (Exception secondException) {
349 				logger.debug("Failure while closing session " + sessionId + ".", secondException);
350 			}
351 		}
352 		catch (Exception e) {
353 			// Could be part of normal workflow (e.g. browser tab closed)
354 			logger.debug("Failed to send message to client in " + session + ": " + message, e);
355 		}
356 	}
357 
358 	private String resolveSessionId(Message<?> message) {
359 		for (SubProtocolHandler handler : this.protocolHandlerLookup.values()) {
360 			String sessionId = handler.resolveSessionId(message);
361 			if (sessionId != null) {
362 				return sessionId;
363 			}
364 		}
365 		if (this.defaultProtocolHandler != null) {
366 			String sessionId = this.defaultProtocolHandler.resolveSessionId(message);
367 			if (sessionId != null) {
368 				return sessionId;
369 			}
370 		}
371 		return null;
372 	}

374 	/**
375 	 * When a session is connected through a higher-level protocol it has a chance
376 	 * to use heartbeat management to shut down sessions that are too slow to send
377 	 * or receive messages. However, after a WebSocketSession is established and
378 	 * before the higher level protocol is fully connected there is a possibility
379 	 * for sessions to hang. This method checks and closes any sessions that have
380 	 * been connected for more than 60 seconds without having received a single
381 	 * message.
382 	 */
383 	private void checkSessions() throws IOException {
384 		long currentTime = System.currentTimeMillis();
385 		if (!isRunning() || (currentTime - this.lastSessionCheckTime < TIME_TO_FIRST_MESSAGE)) {
386 			return;
387 		}
388 		if (this.sessionCheckLock.tryLock()) {
389 			try {
390 				for (WebSocketSessionHolder holder : this.sessions.values()) {
391 					if (holder.hasHandledMessages()) {
392 						continue;
393 					}
394 					long timeSinceCreated = currentTime - holder.getCreateTime();
395 					if (timeSinceCreated < TIME_TO_FIRST_MESSAGE) {
396 						continue;
397 					}
398 					WebSocketSession session = holder.getSession();
399 					if (logger.isErrorEnabled()) {
400 						logger.error("No messages received after " + timeSinceCreated + " ms. " +
401 								"Closing " + holder.getSession() + ".");
402 					}
403 					try {
404 						this.stats.incrementNoMessagesReceivedCount();
405 						session.close(CloseStatus.SESSION_NOT_RELIABLE);
406 					}
407 					catch (Throwable t) {
408 						logger.error("Failure while closing " + session, t);
409 					}
410 				}
411 			}
412 			finally {
413 				this.sessionCheckLock.unlock();
414 			}
415 		}
416 	}

428 	private void clearSession(WebSocketSession session, CloseStatus closeStatus) throws Exception {
429 		if (logger.isDebugEnabled()) {
430 			logger.debug("Clearing session " + session.getId());
431 		}
432 		if (this.sessions.remove(session.getId()) != null) {
433 			this.stats.decrementSessionCount(session);
434 		}
435 		findProtocolHandler(session).afterSessionEnded(session, closeStatus, this.clientInboundChannel);
436 	}

484 	private class Stats {
485 
486 		private final AtomicInteger total = new AtomicInteger();
487 
488 		private final AtomicInteger webSocket = new AtomicInteger();
489 
490 		private final AtomicInteger httpStreaming = new AtomicInteger();
491 
492 		private final AtomicInteger httpPolling = new AtomicInteger();
493 
494 		private final AtomicInteger limitExceeded = new AtomicInteger();
495 
496 		private final AtomicInteger noMessagesReceived = new AtomicInteger();
497 
498 		private final AtomicInteger transportError = new AtomicInteger();
499 
500 
501 		public void incrementSessionCount(WebSocketSession session) {
502 			getCountFor(session).incrementAndGet();
503 			this.total.incrementAndGet();
504 		}
505 
506 		public void decrementSessionCount(WebSocketSession session) {
507 			getCountFor(session).decrementAndGet();
508 		}
509 
510 		public void incrementLimitExceededCount() {
511 			this.limitExceeded.incrementAndGet();
512 		}
513 
514 		public void incrementNoMessagesReceivedCount() {
515 			this.noMessagesReceived.incrementAndGet();
516 		}
517 
518 		public void incrementTransportError() {
519 			this.transportError.incrementAndGet();
520 		}
521 
522 		private AtomicInteger getCountFor(WebSocketSession session) {
523 			if (session instanceof PollingSockJsSession) {
524 				return this.httpPolling;
525 			}
526 			else if (session instanceof StreamingSockJsSession) {
527 				return this.httpStreaming;
528 			}
529 			else {
530 				return this.webSocket;
531 			}
532 		}
533 
534 		public String toString() {
535 			return SubProtocolWebSocketHandler.this.sessions.size() +
536 					" current WS(" + this.webSocket.get() +
537 					")-HttpStream(" + this.httpStreaming.get() +
538 					")-HttpPoll(" + this.httpPolling.get() + "), " +
539 					this.total.get() + " total, " +
540 					(this.limitExceeded.get() + this.noMessagesReceived.get()) + " closed abnormally (" +
541 					this.noMessagesReceived.get() + " connect failure, " +
542 					this.limitExceeded.get() + " send limit, " +
543 					this.transportError.get() + " transport error)";
544 		}
