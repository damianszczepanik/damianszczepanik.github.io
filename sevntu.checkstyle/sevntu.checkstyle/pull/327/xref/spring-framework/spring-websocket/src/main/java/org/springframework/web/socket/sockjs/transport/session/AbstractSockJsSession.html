46  /**
47   * An abstract base class for SockJS sessions implementing {@link SockJsSession}.
48   *
49   * @author Rossen Stoyanchev
50   * @author Sam Brannen
51   * @since 4.0
52   */
53  public abstract class AbstractSockJsSession implements SockJsSession {
54  
55  	private static enum State {NEW, OPEN, CLOSED}
56  
57  
58  	/**
59  	 * Log category to use on network IO exceptions after a client has gone away.
60  	 * <p>The Servlet API does not provide notifications when a client disconnects;
61  	 * see <a href="https://java.net/jira/browse/SERVLET_SPEC-44">SERVLET_SPEC-44</a>.
62  	 * Therefore network IO failures may occur simply because a client has gone away,
63  	 * and that can fill the logs with unnecessary stack traces.
64  	 * <p>We make a best effort to identify such network failures, on a per-server
65  	 * basis, and log them under a separate log category. A simple one-line message
66  	 * is logged at DEBUG level, while a full stack trace is shown at TRACE level.
67  	 * @see #disconnectedClientLogger
68  	 */
69  	public static final String DISCONNECTED_CLIENT_LOG_CATEGORY =
70  			"org.springframework.web.socket.sockjs.DisconnectedClient";
71  
72  	/**
73  	 * Separate logger to use on network IO failure after a client has gone away.
74  	 * @see #DISCONNECTED_CLIENT_LOG_CATEGORY
75  	 */
76  	protected static final Log disconnectedClientLogger = LogFactory.getLog(DISCONNECTED_CLIENT_LOG_CATEGORY);
77  
78  
79  	private static final Set<String> disconnectedClientExceptions;
80  
81  	static {
82  		Set<String> set = new HashSet<String>(2);
83  		set.add("ClientAbortException"); // Tomcat
84  		set.add("EOFException"); // Tomcat
85  		set.add("EofException"); // Jetty
86  		// java.io.IOException "Broken pipe" on WildFly, Glassfish (already covered)
87  		disconnectedClientExceptions = Collections.unmodifiableSet(set);
88  	}

112 	/**
113 	 * Create a new instance.
114 	 * @param id the session ID
115 	 * @param config SockJS service configuration options
116 	 * @param handler the recipient of SockJS messages
117 	 * @param attributes attributes from the HTTP handshake to associate with the WebSocket
118 	 * session; the provided attributes are copied, the original map is not used.
119 	 */
120 	public AbstractSockJsSession(String id, SockJsServiceConfig config, WebSocketHandler handler,
121 			Map<String, Object> attributes) {
122 
123 		Assert.notNull(id, "SessionId must not be null");
124 		Assert.notNull(config, "SockJsConfig must not be null");
125 		Assert.notNull(handler, "WebSocketHandler must not be null");
126 
127 		this.id = id;
128 		this.config = config;
129 		this.handler = handler;
130 
131 		if (attributes != null) {
132 			this.attributes.putAll(attributes);
133 		}
134 	}

158 	public final void sendMessage(WebSocketMessage<?> message) throws IOException {
159 		Assert.state(!isClosed(), "Cannot send a message when session is closed");
160 		Assert.isInstanceOf(TextMessage.class, message, "SockJS supports text messages only: " + message);
161 		sendMessageInternal(((TextMessage) message).getPayload());
162 	}

182 	/**
183 	 * Performs cleanup and notify the {@link WebSocketHandler}.
184 	 */
185 	@Override
186 	public final void close() throws IOException {
187 		close(new CloseStatus(3000, "Go away!"));
188 	}

190 	/**
191 	 * Performs cleanup and notify the {@link WebSocketHandler}.
192 	 */
193 	@Override
194 	public final void close(CloseStatus status) throws IOException {
195 		if (isOpen()) {
196 			if (logger.isDebugEnabled()) {
197 				logger.debug("Closing SockJS session " + getId() + " with " + status);
198 			}
199 			this.state = State.CLOSED;
200 			try {
201 				if (isActive() && !CloseStatus.SESSION_NOT_RELIABLE.equals(status)) {
202 					try {
203 						writeFrameInternal(SockJsFrame.closeFrame(status.getCode(), status.getReason()));
204 					}
205 					catch (Throwable ex) {
206 						logger.debug("Failure while send SockJS close frame", ex);
207 					}
208 				}
209 				updateLastActiveTime();
210 				cancelHeartbeat();
211 				disconnect(status);
212 			}
213 			finally {
214 				try {
215 					this.handler.afterConnectionClosed(this, status);
216 				}
217 				catch (Throwable ex) {
218 					logger.debug("Error from WebSocketHandler.afterConnectionClosed in " + this, ex);
219 				}
220 			}
221 		}
222 	}

224 	@Override
225 	public long getTimeSinceLastActive() {
226 		if (isNew()) {
227 			return (System.currentTimeMillis() - this.timeCreated);
228 		}
229 		else {
230 			return (isActive() ? 0 : System.currentTimeMillis() - this.timeLastActive);
231 		}
232 	}
233 
234 	/**
235 	 * Should be invoked whenever the session becomes inactive.
236 	 */
237 	protected void updateLastActiveTime() {
238 		this.timeLastActive = System.currentTimeMillis();
239 	}

241 	@Override
242 	public void disableHeartbeat() {
243 		this.heartbeatDisabled = true;
244 		cancelHeartbeat();
245 	}
246 
247 	public void sendHeartbeat() throws SockJsTransportFailureException {
248 		if (isActive()) {
249 			writeFrame(SockJsFrame.heartbeatFrame());
250 			scheduleHeartbeat();
251 		}
252 	}
253 
254 	protected void scheduleHeartbeat() {
255 		if (this.heartbeatDisabled) {
256 			return;
257 		}
258 
259 		Assert.state(this.config.getTaskScheduler() != null, "Expected SockJS TaskScheduler");
260 		cancelHeartbeat();
261 		if (!isActive()) {
262 			return;
263 		}
264 
265 		Date time = new Date(System.currentTimeMillis() + this.config.getHeartbeatTime());
266 		this.heartbeatTask = this.config.getTaskScheduler().schedule(new Runnable() {
267 			public void run() {
268 				try {
269 					sendHeartbeat();
270 				}
271 				catch (Throwable ex) {
272 					// ignore
273 				}
274 			}
275 		}, time);
276 		if (logger.isTraceEnabled()) {
277 			logger.trace("Scheduled heartbeat in session " + getId());
278 		}
279 	}
280 
281 	protected void cancelHeartbeat() {
282 		try {
283 			ScheduledFuture<?> task = this.heartbeatTask;
284 			this.heartbeatTask = null;
285 
286 			if ((task != null) && !task.isDone()) {
287 				if (logger.isTraceEnabled()) {
288 					logger.trace("Cancelling heartbeat in session " + getId());
289 				}
290 				task.cancel(false);
291 			}
292 		}
293 		catch (Throwable ex) {
294 			logger.debug("Failure while cancelling heartbeat in session " + getId(), ex);
295 		}
296 	}

298 	/**
299 	 * Polling and Streaming sessions periodically close the current HTTP request and
300 	 * wait for the next request to come through. During this "downtime" the session is
301 	 * still open but inactive and unable to send messages and therefore has to buffer
302 	 * them temporarily. A WebSocket session by contrast is stateful and remain active
303 	 * until closed.
304 	 */
305 	public abstract boolean isActive();
306 
307 	/**
308 	 * Actually close the underlying WebSocket session or in the case of HTTP
309 	 * transports complete the underlying request.
310 	 */
311 	protected abstract void disconnect(CloseStatus status) throws IOException;

314 	// Frame writing
315 
316 	/**
317 	 * For internal use within a TransportHandler and the (TransportHandler-specific)
318 	 * session class.
319 	 */
320 	protected void writeFrame(SockJsFrame frame) throws SockJsTransportFailureException {
321 		if (logger.isTraceEnabled()) {
322 			logger.trace("Preparing to write " + frame);
323 		}
324 		try {
325 			writeFrameInternal(frame);
326 		}
327 		catch (Throwable ex) {
328 			logWriteFrameFailure(ex);
329 			try {
330 				// Force disconnect (so we won't try to send close frame)
331 				disconnect(CloseStatus.SERVER_ERROR);
332 			}
333 			catch (Throwable disconnectFailure) {
334 				// Ignore
335 			}
336 			try {
337 				close(CloseStatus.SERVER_ERROR);
338 			}
339 			catch (Throwable closeFailure) {
340 				// Nothing of consequence, already forced disconnect
341 			}
342 			throw new SockJsTransportFailureException("Failed to write " + frame, this.getId(), ex);
343 		}
344 	}
345 
346 	private void logWriteFrameFailure(Throwable failure) {
347 		@SuppressWarnings("serial")
348 		NestedCheckedException nestedException = new NestedCheckedException("", failure) {};
349 
350 		if ("Broken pipe".equalsIgnoreCase(nestedException.getMostSpecificCause().getMessage()) ||
351 				disconnectedClientExceptions.contains(failure.getClass().getSimpleName())) {
352 
353 			if (disconnectedClientLogger.isTraceEnabled()) {
354 				disconnectedClientLogger.trace("Looks like the client has gone away", failure);
355 			}
356 			else if (disconnectedClientLogger.isDebugEnabled()) {
357 				disconnectedClientLogger.debug("Looks like the client has gone away: " +
358 						nestedException.getMessage() + " (For full stack trace, set the '" +
359 						DISCONNECTED_CLIENT_LOG_CATEGORY + "' log category to TRACE level)");
360 			}
361 		}
362 		else {
363 			logger.debug("Terminating connection after failure to send message to client.", failure);
364 		}
365 	}

367 	protected abstract void writeFrameInternal(SockJsFrame frame) throws IOException;

372 	public void delegateConnectionEstablished() throws Exception {
373 		this.state = State.OPEN;
374 		this.handler.afterConnectionEstablished(this);
375 	}
376 
377 	public void delegateMessages(String... messages) throws SockJsMessageDeliveryException {
378 		List<String> undelivered = new ArrayList<String>(Arrays.asList(messages));
379 		for (String message : messages) {
380 			try {
381 				if (isClosed()) {
382 					throw new SockJsMessageDeliveryException(this.id, undelivered, "Session closed");
383 				}
384 				else {
385 					this.handler.handleMessage(this, new TextMessage(message));
386 					undelivered.remove(0);
387 				}
388 			}
389 			catch (Throwable ex) {
390 				throw new SockJsMessageDeliveryException(this.id, undelivered, ex);
391 			}
392 		}
393 	}
394 
395 	/**
396 	 * Invoked when the underlying connection is closed.
397 	 */
398 	public final void delegateConnectionClosed(CloseStatus status) throws Exception {
399 		if (!isClosed()) {
400 			try {
401 				updateLastActiveTime();
402 				cancelHeartbeat();
403 			}
404 			finally {
405 				this.state = State.CLOSED;
406 				this.handler.afterConnectionClosed(this, status);
407 			}
408 		}
409 	}
410 
411 	/**
412 	 * Close due to error arising from SockJS transport handling.
413 	 */
414 	public void tryCloseWithSockJsTransportError(Throwable error, CloseStatus closeStatus) {
415 		if (logger.isDebugEnabled()) {
416 			logger.debug("Closing due to transport error for " + this);
417 		}
418 		try {
419 			delegateError(error);
420 		}
421 		catch (Throwable delegateException) {
422 			// ignore
423 		}
424 		try {
425 			close(closeStatus);
426 		}
427 		catch (Throwable closeException) {
428 			logger.debug("Failure while closing " + this, closeException);
429 		}
430 	}
431 
432 	public void delegateError(Throwable ex) throws Exception {
433 		this.handler.handleTransportError(this, ex);
434 	}
