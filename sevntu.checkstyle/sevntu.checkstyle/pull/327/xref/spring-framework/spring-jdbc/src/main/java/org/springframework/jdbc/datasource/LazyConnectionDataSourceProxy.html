32  /**
33   * Proxy for a target DataSource, fetching actual JDBC Connections lazily,
34   * i.e. not until first creation of a Statement. Connection initialization
35   * properties like auto-commit mode, transaction isolation and read-only mode
36   * will be kept and applied to the actual JDBC Connection as soon as an
37   * actual Connection is fetched (if ever). Consequently, commit and rollback
38   * calls will be ignored if no Statements have been created.
39   *
40   * <p>This DataSource proxy allows to avoid fetching JDBC Connections from
41   * a pool unless actually necessary. JDBC transaction control can happen
42   * without fetching a Connection from the pool or communicating with the
43   * database; this will be done lazily on first creation of a JDBC Statement.
44   *
45   * <p><b>If you configure both a LazyConnectionDataSourceProxy and a
46   * TransactionAwareDataSourceProxy, make sure that the latter is the outermost
47   * DataSource.</b> In such a scenario, data access code will talk to the
48   * transaction-aware DataSource, which will in turn work with the
49   * LazyConnectionDataSourceProxy.
50   *
51   * <p>Lazy fetching of physical JDBC Connections is particularly beneficial
52   * in a generic transaction demarcation environment. It allows you to demarcate
53   * transactions on all methods that could potentially perform data access,
54   * without paying a performance penalty if no actual data access happens.
55   *
56   * <p>This DataSource proxy gives you behavior analogous to JTA and a
57   * transactional JNDI DataSource (as provided by the J2EE server), even
58   * with a local transaction strategy like DataSourceTransactionManager or
59   * HibernateTransactionManager. It does not add value with Spring's
60   * JtaTransactionManager as transaction strategy.
61   *
62   * <p>Lazy fetching of JDBC Connections is also recommended for read-only
63   * operations with Hibernate, in particular if the chances of resolving the
64   * result in the second-level cache are high. This avoids the need to
65   * communicate with the database at all for such read-only operations.
66   * You will get the same effect with non-transactional reads, but lazy fetching
67   * of JDBC Connections allows you to still perform reads in transactions.
68   *
69   * <p><b>NOTE:</b> This DataSource proxy needs to return wrapped Connections
70   * (which implement the {@link ConnectionProxy} interface) in order to handle
71   * lazy fetching of an actual JDBC Connection. Therefore, the returned Connections
72   * cannot be cast to a native JDBC Connection type such as OracleConnection or
73   * to a connection pool implementation type. Use a corresponding
74   * {@link org.springframework.jdbc.support.nativejdbc.NativeJdbcExtractor}
75   * or JDBC 4's {@link Connection#unwrap} to retrieve the native JDBC Connection.
76   *
77   * @author Juergen Hoeller
78   * @since 1.1.4
79   * @see DataSourceTransactionManager
80   */
81  public class LazyConnectionDataSourceProxy extends DelegatingDataSource {

111 	 * Set the default auto-commit mode to expose when no target Connection
112 	 * has been fetched yet (-> actual JDBC Connection default not known yet).
113 	 * <p>If not specified, the default gets determined by checking a target
114 	 * Connection on startup. If that check fails, the default will be determined
115 	 * lazily on first access of a Connection.
116 	 * @see java.sql.Connection#setAutoCommit
117 	 */
118 	public void setDefaultAutoCommit(boolean defaultAutoCommit) {
119 		this.defaultAutoCommit = defaultAutoCommit;
120 	}

122 	/**
123 	 * Set the default transaction isolation level to expose when no target Connection
124 	 * has been fetched yet (-> actual JDBC Connection default not known yet).
125 	 * <p>This property accepts the int constant value (e.g. 8) as defined in the
126 	 * {@link java.sql.Connection} interface; it is mainly intended for programmatic
127 	 * use. Consider using the "defaultTransactionIsolationName" property for setting
128 	 * the value by name (e.g. "TRANSACTION_SERIALIZABLE").
129 	 * <p>If not specified, the default gets determined by checking a target
130 	 * Connection on startup. If that check fails, the default will be determined
131 	 * lazily on first access of a Connection.
132 	 * @see #setDefaultTransactionIsolationName
133 	 * @see java.sql.Connection#setTransactionIsolation
134 	 */
135 	public void setDefaultTransactionIsolation(int defaultTransactionIsolation) {
136 		this.defaultTransactionIsolation = defaultTransactionIsolation;
137 	}

139 	/**
140 	 * Set the default transaction isolation level by the name of the corresponding
141 	 * constant in {@link java.sql.Connection}, e.g. "TRANSACTION_SERIALIZABLE".
142 	 * @param constantName name of the constant
143 	 * @see #setDefaultTransactionIsolation
144 	 * @see java.sql.Connection#TRANSACTION_READ_UNCOMMITTED
145 	 * @see java.sql.Connection#TRANSACTION_READ_COMMITTED
146 	 * @see java.sql.Connection#TRANSACTION_REPEATABLE_READ
147 	 * @see java.sql.Connection#TRANSACTION_SERIALIZABLE
148 	 */
149 	public void setDefaultTransactionIsolationName(String constantName) {
150 		setDefaultTransactionIsolation(constants.asNumber(constantName).intValue());
151 	}

154 	@Override
155 	public void afterPropertiesSet() {
156 		super.afterPropertiesSet();
157 
158 		// Determine default auto-commit and transaction isolation
159 		// via a Connection from the target DataSource, if possible.
160 		if (this.defaultAutoCommit == null || this.defaultTransactionIsolation == null) {
161 			try {
162 				Connection con = getTargetDataSource().getConnection();
163 				try {
164 					checkDefaultConnectionProperties(con);
165 				}
166 				finally {
167 					con.close();
168 				}
169 			}
170 			catch (SQLException ex) {
171 				logger.warn("Could not retrieve default auto-commit and transaction isolation settings", ex);
172 			}
173 		}
174 	}
175 
176 	/**
177 	 * Check the default connection properties (auto-commit, transaction isolation),
178 	 * keeping them to be able to expose them correctly without fetching an actual
179 	 * JDBC Connection from the target DataSource.
180 	 * <p>This will be invoked once on startup, but also for each retrieval of a
181 	 * target Connection. If the check failed on startup (because the database was
182 	 * down), we'll lazily retrieve those settings.
183 	 * @param con the Connection to use for checking
184 	 * @throws SQLException if thrown by Connection methods
185 	 */
186 	protected synchronized void checkDefaultConnectionProperties(Connection con) throws SQLException {
187 		if (this.defaultAutoCommit == null) {
188 			this.defaultAutoCommit = con.getAutoCommit();
189 		}
190 		if (this.defaultTransactionIsolation == null) {
191 			this.defaultTransactionIsolation = con.getTransactionIsolation();
192 		}
193 	}

211 	 * Return a Connection handle that lazily fetches an actual JDBC Connection
212 	 * when asked for a Statement (or PreparedStatement or CallableStatement).
213 	 * <p>The returned Connection handle implements the ConnectionProxy interface,
214 	 * allowing to retrieve the underlying target Connection.
215 	 * @return a lazy Connection handle
216 	 * @see ConnectionProxy#getTargetConnection()
217 	 */
218 	@Override
219 	public Connection getConnection() throws SQLException {
220 		return (Connection) Proxy.newProxyInstance(
221 				ConnectionProxy.class.getClassLoader(),
222 				new Class<?>[] {ConnectionProxy.class},
223 				new LazyConnectionInvocationHandler());
224 	}

227 	 * Return a Connection handle that lazily fetches an actual JDBC Connection

230 	 * allowing to retrieve the underlying target Connection.
231 	 * @param username the per-Connection username

245 	/**
246 	 * Invocation handler that defers fetching an actual JDBC Connection
247 	 * until first creation of a Statement.
248 	 */
249 	private class LazyConnectionInvocationHandler implements InvocationHandler {
250 
251 		private String username;
252 
253 		private String password;
254 
255 		private Boolean readOnly = Boolean.FALSE;
256 
257 		private Integer transactionIsolation;
258 
259 		private Boolean autoCommit;
260 
261 		private boolean closed = false;
262 
263 		private Connection target;
264 
265 		public LazyConnectionInvocationHandler() {
266 			this.autoCommit = defaultAutoCommit();
267 			this.transactionIsolation = defaultTransactionIsolation();
268 		}
269 
270 		public LazyConnectionInvocationHandler(String username, String password) {
271 			this();
272 			this.username = username;
273 			this.password = password;
274 		}
275 
276 		@Override
277 		public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
278 			// Invocation on ConnectionProxy interface coming in...
279 
280 			if (method.getName().equals("equals")) {
281 				// We must avoid fetching a target Connection for "equals".
282 				// Only consider equal when proxies are identical.
283 				return (proxy == args[0]);
284 			}
285 			else if (method.getName().equals("hashCode")) {
286 				// We must avoid fetching a target Connection for "hashCode",
287 				// and we must return the same hash code even when the target
288 				// Connection has been fetched: use hashCode of Connection proxy.
289 				return System.identityHashCode(proxy);
290 			}
291 			else if (method.getName().equals("unwrap")) {
292 				if (((Class<?>) args[0]).isInstance(proxy)) {
293 					return proxy;
294 				}
295 			}
296 			else if (method.getName().equals("isWrapperFor")) {
297 				if (((Class<?>) args[0]).isInstance(proxy)) {
298 					return true;
299 				}
300 			}
301 			else if (method.getName().equals("getTargetConnection")) {
302 				// Handle getTargetConnection method: return underlying connection.
303 				return getTargetConnection(method);
304 			}
305 
306 			if (!hasTargetConnection()) {
307 				// No physical target Connection kept yet ->
308 				// resolve transaction demarcation methods without fetching
309 				// a physical JDBC Connection until absolutely necessary.
310 
311 				if (method.getName().equals("toString")) {
312 					return "Lazy Connection proxy for target DataSource [" + getTargetDataSource() + "]";
313 				}
314 				else if (method.getName().equals("isReadOnly")) {
315 					return this.readOnly;
316 				}
317 				else if (method.getName().equals("setReadOnly")) {
318 					this.readOnly = (Boolean) args[0];
319 					return null;
320 				}
321 				else if (method.getName().equals("getTransactionIsolation")) {
322 					if (this.transactionIsolation != null) {
323 						return this.transactionIsolation;
324 					}
325 					// Else fetch actual Connection and check there,
326 					// because we didn't have a default specified.
327 				}
328 				else if (method.getName().equals("setTransactionIsolation")) {
329 					this.transactionIsolation = (Integer) args[0];
330 					return null;
331 				}
332 				else if (method.getName().equals("getAutoCommit")) {
333 					if (this.autoCommit != null) {
334 						return this.autoCommit;
335 					}
336 					// Else fetch actual Connection and check there,
337 					// because we didn't have a default specified.
338 				}
339 				else if (method.getName().equals("setAutoCommit")) {
340 					this.autoCommit = (Boolean) args[0];
341 					return null;
342 				}
343 				else if (method.getName().equals("commit")) {
344 					// Ignore: no statements created yet.
345 					return null;
346 				}
347 				else if (method.getName().equals("rollback")) {
348 					// Ignore: no statements created yet.
349 					return null;
350 				}
351 				else if (method.getName().equals("getWarnings")) {
352 					return null;
353 				}
354 				else if (method.getName().equals("clearWarnings")) {
355 					return null;
356 				}
357 				else if (method.getName().equals("close")) {
358 					// Ignore: no target connection yet.
359 					this.closed = true;
360 					return null;
361 				}
362 				else if (method.getName().equals("isClosed")) {
363 					return this.closed;
364 				}
365 				else if (this.closed) {
366 					// Connection proxy closed, without ever having fetched a
367 					// physical JDBC Connection: throw corresponding SQLException.
368 					throw new SQLException("Illegal operation: connection is closed");
369 				}
370 			}
371 
372 			// Target Connection already fetched,
373 			// or target Connection necessary for current operation ->
374 			// invoke method on target connection.
375 			try {
376 				return method.invoke(getTargetConnection(method), args);
377 			}
378 			catch (InvocationTargetException ex) {
379 				throw ex.getTargetException();
380 			}
381 		}
382 
383 		/**
384 		 * Return whether the proxy currently holds a target Connection.
385 		 */
386 		private boolean hasTargetConnection() {
387 			return (this.target != null);
388 		}
389 
390 		/**
391 		 * Return the target Connection, fetching it and initializing it if necessary.
392 		 */
393 		private Connection getTargetConnection(Method operation) throws SQLException {
394 			if (this.target == null) {
395 				// No target Connection held -> fetch one.
396 				if (logger.isDebugEnabled()) {
397 					logger.debug("Connecting to database for operation '" + operation.getName() + "'");
398 				}
399 
400 				// Fetch physical Connection from DataSource.
401 				this.target = (this.username != null) ?
402 						getTargetDataSource().getConnection(this.username, this.password) :
403 						getTargetDataSource().getConnection();
404 
405 				// If we still lack default connection properties, check them now.
406 				checkDefaultConnectionProperties(this.target);
407 
408 				// Apply kept transaction settings, if any.
409 				if (this.readOnly) {
410 					try {
411 						this.target.setReadOnly(this.readOnly);
412 					}
413 					catch (Exception ex) {
414 						// "read-only not supported" -> ignore, it's just a hint anyway
415 						logger.debug("Could not set JDBC Connection read-only", ex);
416 					}
417 				}
418 				if (this.transactionIsolation != null &&
419 						!this.transactionIsolation.equals(defaultTransactionIsolation())) {
420 					this.target.setTransactionIsolation(this.transactionIsolation);
421 				}
422 				if (this.autoCommit != null && this.autoCommit != this.target.getAutoCommit()) {
423 					this.target.setAutoCommit(this.autoCommit);
424 				}
425 			}
426 
427 			else {
428 				// Target Connection already held -> return it.
429 				if (logger.isDebugEnabled()) {
430 					logger.debug("Using existing database connection for operation '" + operation.getName() + "'");
431 				}
432 			}
433 
434 			return this.target;
435 		}
