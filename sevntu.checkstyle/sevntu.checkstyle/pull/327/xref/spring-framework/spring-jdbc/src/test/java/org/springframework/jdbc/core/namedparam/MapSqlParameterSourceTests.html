17  package org.springframework.jdbc.core.namedparam;
18  
19  import org.junit.Test;
20  
21  import org.springframework.jdbc.core.SqlParameterValue;
22  
23  import static org.junit.Assert.*;
24  
25  /**
26   * @author Rick Evans
27   * @author Arjen Poutsma
28   */
29  public final class MapSqlParameterSourceTests {
30  
31  	@Test
32  	public void nullParameterValuesPassedToCtorIsOk() throws Exception {
33  		new MapSqlParameterSource(null);
34  	}
35  
36  	@Test(expected = IllegalArgumentException.class)
37  	public void getValueChokesIfParameterIsNotPresent() throws Exception {
38  		MapSqlParameterSource source = new MapSqlParameterSource();
39  		source.getValue("pechorin was right!");
40  	}
41  
42  	@Test
43  	public void sqlParameterValueRegistersSqlType() throws Exception {
44  		MapSqlParameterSource msps = new MapSqlParameterSource("FOO", new SqlParameterValue(2, "Foo"));
45  		assertEquals("Correct SQL Type not registered", 2, msps.getSqlType("FOO"));
46  		MapSqlParameterSource msps2 = new MapSqlParameterSource();
47  		msps2.addValues(msps.getValues());
48  		assertEquals("Correct SQL Type not registered", 2, msps2.getSqlType("FOO"));
49  	}
50  
51  }
