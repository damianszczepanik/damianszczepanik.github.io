59  	@Before
60  	public void setUp() throws Exception {
61  		this.connection = mock(Connection.class);
62  		this.dataSource = mock(DataSource.class);
63  		this.statement = mock(Statement.class);
64  		this.preparedStatement = mock(PreparedStatement.class);
65  		this.resultSet = mock(ResultSet.class);
66  		this.resultSetMetaData = mock(ResultSetMetaData.class);
67  		this.template = new JdbcTemplate(this.dataSource);
68  		given(this.dataSource.getConnection()).willReturn(this.connection);
69  		given(this.resultSet.getMetaData()).willReturn(this.resultSetMetaData);
70  		given(this.resultSetMetaData.getColumnCount()).willReturn(1);
71  		given(this.resultSetMetaData.getColumnLabel(1)).willReturn("age");
72  		given(this.connection.createStatement()).willReturn(this.statement);
73  		given(this.connection.prepareStatement(anyString())).willReturn(this.preparedStatement);
74  		given(this.preparedStatement.executeQuery()).willReturn(this.resultSet);
75  		given(this.statement.executeQuery(anyString())).willReturn(this.resultSet);
76  	}

78  	@Test
79  	public void testQueryForList() throws Exception {
80  		String sql = "SELECT AGE FROM CUSTMR WHERE ID < 3";
81  		given(this.resultSet.next()).willReturn(true, true, false);
82  		given(this.resultSet.getObject(1)).willReturn(11, 12);
83  		List<Map<String, Object>> li = this.template.queryForList(sql);
84  		assertEquals("All rows returned", 2, li.size());
85  		assertEquals("First row is Integer", 11, ((Integer) li.get(0).get("age")).intValue());
86  		assertEquals("Second row is Integer", 12, ((Integer) li.get(1).get("age")).intValue());
87  		verify(this.resultSet).close();
88  		verify(this.statement).close();
89  	}
