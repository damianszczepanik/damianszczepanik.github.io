17  package org.springframework.jdbc.support.incrementer;
18  
19  import javax.sql.DataSource;
20  
21  /**
22   * {@link DataFieldMaxValueIncrementer} that increments the maximum value of a given Sybase table
23   * with the equivalent of an auto-increment column. Note: If you use this class, your table key
24   * column should <i>NOT</i> be defined as an IDENTITY column, as the sequence table does the job.
25   *
26   * <p>This class is intended to be used with Sybase Adaptive Server.
27   *
28   * <p>The sequence is kept in a table. There should be one sequence table per
29   * table that needs an auto-generated key.
30   *
31   * <p>Example:
32   *
33   * <pre class="code">create table tab (id int not null primary key, text varchar(100))
34   * create table tab_sequence (id bigint identity)
35   * insert into tab_sequence values()</pre>
36   *
37   * If "cacheSize" is set, the intermediate values are served without querying the
38   * database. If the server or your application is stopped or crashes or a transaction
39   * is rolled back, the unused values will never be served. The maximum hole size in
40   * numbering is consequently the value of cacheSize.
41   *
42   * <b>HINT:</b> Since Sybase supports the JDBC 3.0 {@code getGeneratedKeys} method,
43   * it is recommended to use IDENTITY columns directly in the tables and then using a
44   * {@link org.springframework.jdbc.core.simple.SimpleJdbcInsert} or utilizing
45   * a {@link org.springframework.jdbc.support.KeyHolder} when calling the with the
46   * {@code update(PreparedStatementCreator psc, KeyHolder generatedKeyHolder)}
47   * method of the {@link org.springframework.jdbc.core.JdbcTemplate}.
48   *
49   * <p>Thanks to Yinwei Liu for the suggestion!
50   *
51   * @author Thomas Risberg
52   * @author Juergen Hoeller
53   * @since 2.5.5
54   */
55  public class SybaseMaxValueIncrementer extends AbstractIdentityColumnMaxValueIncrementer {
56  
57  	/**
58  	 * Default constructor for bean property style usage.
59  	 * @see #setDataSource
60  	 * @see #setIncrementerName
61  	 * @see #setColumnName
62  	 */
63  	public SybaseMaxValueIncrementer() {
64  	}
65  
66  	/**
67  	 * Convenience constructor.
68  	 * @param dataSource the DataSource to use
69  	 * @param incrementerName the name of the sequence/table to use
70  	 * @param columnName the name of the column in the sequence table to use
71  	 */
72  	public SybaseMaxValueIncrementer(DataSource dataSource, String incrementerName, String columnName) {
73  		super(dataSource, incrementerName, columnName);
74  	}
75  
76  
77  	@Override
78  	protected String getIncrementStatement() {
79  		return "insert into " + getIncrementerName() + " values()";
80  	}
81  
82  	@Override
83  	protected String getIdentityStatement() {
84  		return "select @@identity";
85  	}
86  
87  }
