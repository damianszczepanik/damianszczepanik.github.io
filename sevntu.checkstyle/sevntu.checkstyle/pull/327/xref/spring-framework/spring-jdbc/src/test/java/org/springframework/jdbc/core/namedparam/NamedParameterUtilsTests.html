36  	@Test
37  	public void parseSql() {
38  		String sql = "xxx :a yyyy :b :c :a zzzzz";
39  		ParsedSql psql = NamedParameterUtils.parseSqlStatement(sql);
40  		assertEquals("xxx ? yyyy ? ? ? zzzzz", NamedParameterUtils.substituteNamedParameters(psql, null));
41  		assertEquals("a", psql.getParameterNames().get(0));
42  		assertEquals("c", psql.getParameterNames().get(2));
43  		assertEquals("a", psql.getParameterNames().get(3));
44  		assertEquals(4, psql.getTotalParameterCount());
45  		assertEquals(3, psql.getNamedParameterCount());
46  
47  		String sql2 = "xxx &a yyyy ? zzzzz";
48  		ParsedSql psql2 = NamedParameterUtils.parseSqlStatement(sql2);
49  		assertEquals("xxx ? yyyy ? zzzzz", NamedParameterUtils.substituteNamedParameters(psql2, null));
50  		assertEquals("a", psql2.getParameterNames().get(0));
51  		assertEquals(2, psql2.getTotalParameterCount());
52  		assertEquals(1, psql2.getNamedParameterCount());
53  
54  		String sql3 = "xxx &a+:b" + '\t' + ":c%10 yyyy ? zzzzz";
55  		ParsedSql psql3 = NamedParameterUtils.parseSqlStatement(sql3);
56  		assertEquals("a", psql3.getParameterNames().get(0));
57  		assertEquals("b", psql3.getParameterNames().get(1));
58  		assertEquals("c", psql3.getParameterNames().get(2));
59  	}

62  	public void substituteNamedParameters() {
63  		MapSqlParameterSource namedParams = new MapSqlParameterSource();
64  		namedParams.addValue("a", "a").addValue("b", "b").addValue("c", "c");
65  		assertEquals("xxx ? ? ?", NamedParameterUtils.substituteNamedParameters("xxx :a :b :c", namedParams));
66  		assertEquals("xxx ? ? ? xx ? ?",
67  				NamedParameterUtils.substituteNamedParameters("xxx :a :b :c xx :a :a", namedParams));

70  	@Test
71  	public void convertParamMapToArray() {
72  		Map<String, String> paramMap = new HashMap<String, String>();
73  		paramMap.put("a", "a");
74  		paramMap.put("b", "b");
75  		paramMap.put("c", "c");
76  		assertSame(3, NamedParameterUtils.buildValueArray("xxx :a :b :c", paramMap).length);
77  		assertSame(5, NamedParameterUtils.buildValueArray("xxx :a :b :c xx :a :b", paramMap).length);
78  		assertSame(5, NamedParameterUtils.buildValueArray("xxx :a :a :a xx :a :a", paramMap).length);
79  		assertEquals("b", NamedParameterUtils.buildValueArray("xxx :a :b :c xx :a :b", paramMap)[4]);
80  		try {
81  			NamedParameterUtils.buildValueArray("xxx :a :b ?", paramMap);
82  			fail("mixed named parameters and ? placeholders not detected");
83  		}
84  		catch (InvalidDataAccessApiUsageException expected) {
85  		}
86  	}

143 	@Test
144 	public void parseSqlContainingComments() {
145 		String sql1 = "/*+ HINT */ xxx /* comment ? */ :a yyyy :b :c :a zzzzz -- :xx XX\n";
146 		ParsedSql psql1 = NamedParameterUtils.parseSqlStatement(sql1);
147 		assertEquals("/*+ HINT */ xxx /* comment ? */ ? yyyy ? ? ? zzzzz -- :xx XX\n",
148 				NamedParameterUtils.substituteNamedParameters(psql1, null));
149 		MapSqlParameterSource paramMap = new MapSqlParameterSource();
150 		paramMap.addValue("a", "a");
151 		paramMap.addValue("b", "b");
152 		paramMap.addValue("c", "c");
153 		Object[] params = NamedParameterUtils.buildValueArray(psql1, paramMap, null);
154 		assertEquals(4, params.length);
155 		assertEquals("a", params[0]);
156 		assertEquals("b", params[1]);
157 		assertEquals("c", params[2]);
158 		assertEquals("a", params[3]);
159 
160 		String sql2 = "/*+ HINT */ xxx /* comment ? */ :a yyyy :b :c :a zzzzz -- :xx XX";
161 		ParsedSql psql2 = NamedParameterUtils.parseSqlStatement(sql2);
162 		assertEquals("/*+ HINT */ xxx /* comment ? */ ? yyyy ? ? ? zzzzz -- :xx XX",
163 				NamedParameterUtils.substituteNamedParameters(psql2, null));
164 
165 		String sql3 = "/*+ HINT */ xxx /* comment ? */ :a yyyy :b :c :a zzzzz /* :xx XX*";
166 		ParsedSql psql3 = NamedParameterUtils.parseSqlStatement(sql3);
167 		assertEquals("/*+ HINT */ xxx /* comment ? */ ? yyyy ? ? ? zzzzz /* :xx XX*",
168 				NamedParameterUtils.substituteNamedParameters(psql3, null));
169 
170 		String sql4 = "/*+ HINT */ xxx /* comment :a ? */ :a yyyy :b :c :a zzzzz /* :xx XX*";
171 		ParsedSql psql4 = NamedParameterUtils.parseSqlStatement(sql4);
172 		Map<String, String> parameters = Collections.singletonMap("a", "0");
173 		assertEquals("/*+ HINT */ xxx /* comment :a ? */ ? yyyy ? ? ? zzzzz /* :xx XX*",
174 				NamedParameterUtils.substituteNamedParameters(psql4, new MapSqlParameterSource(parameters)));
175 	}
