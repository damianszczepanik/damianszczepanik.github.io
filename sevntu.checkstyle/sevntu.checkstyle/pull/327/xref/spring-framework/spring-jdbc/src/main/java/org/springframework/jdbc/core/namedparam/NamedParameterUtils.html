44  	/**
45  	 * Set of characters that qualify as parameter separators,
46  	 * indicating that a parameter name in a SQL String has ended.
47  	 */
48  	private static final char[] PARAMETER_SEPARATORS =
49  			new char[] {'"', '\'', ':', '&', ',', ';', '(', ')', '|', '=', '+', '-', '*', '%', '/', '\\', '<', '>', '^'};
50  
51  	/**
52  	 * Set of characters that qualify as comment or quotes starting characters.
53  	 */
54  	private static final String[] START_SKIP =
55  			new String[] {"'", "\"", "--", "/*"};
56  
57  	/**
58  	 * Set of characters that at are the corresponding comment or quotes ending characters.
59  	 */
60  	private static final String[] STOP_SKIP =
61  			new String[] {"'", "\"", "\n", "*/"};
62

68  	/**
69  	 * Parse the SQL statement and locate any placeholders or named parameters.
70  	 * Named parameters are substituted for a JDBC placeholder.
71  	 * @param sql the SQL statement
72  	 * @return the parsed statement, represented as ParsedSql instance
73  	 */
74  	public static ParsedSql parseSqlStatement(final String sql) {
75  		Assert.notNull(sql, "SQL must not be null");
76  
77  		Set<String> namedParameters = new HashSet<String>();
78  		String sqlToUse = sql;
79  		List<ParameterHolder> parameterList = new ArrayList<ParameterHolder>();
80  
81  		char[] statement = sql.toCharArray();
82  		int namedParameterCount = 0;
83  		int unnamedParameterCount = 0;
84  		int totalParameterCount = 0;
85  
86  		int escapes = 0;
87  		int i = 0;
88  		while (i < statement.length) {
89  			int skipToPosition = i;
90  			while (i < statement.length) {
91  				skipToPosition = skipCommentsAndQuotes(statement, i);
92  				if (i == skipToPosition) {
93  					break;
94  				}
95  				else {
96  					i = skipToPosition;
97  				}
98  			}
99  			if (i >= statement.length) {
100 				break;
101 			}
102 			char c = statement[i];
103 			if (c == ':' || c == '&') {
104 				int j = i + 1;
105 				if (j < statement.length && statement[j] == ':' && c == ':') {
106 					// Postgres-style "::" casting operator - to be skipped.
107 					i = i + 2;
108 					continue;
109 				}
110 				String parameter = null;
111 				if (j < statement.length && c == ':' && statement[j] == '{') {
112 					// :{x} style parameter
113 					while (j < statement.length && !('}' == statement[j])) {
114 						j++;
115 						if (':' == statement[j] || '{' == statement[j]) {
116 							throw new InvalidDataAccessApiUsageException("Parameter name contains invalid character '" +
117 									statement[j] + "' at position " + i + " in statement: " + sql);
118 						}
119 					}
120 					if (j >= statement.length) {
121 						throw new InvalidDataAccessApiUsageException(
122 								"Non-terminated named parameter declaration at position " + i + " in statement: " + sql);
123 					}
124 					if (j - i > 3) {
125 						parameter = sql.substring(i + 2, j);
126 						namedParameterCount = addNewNamedParameter(namedParameters, namedParameterCount, parameter);
127 						totalParameterCount = addNamedParameter(parameterList, totalParameterCount, escapes, i, j + 1, parameter);
128 					}
129 					j++;
130 				}
131 				else {
132 					while (j < statement.length && !isParameterSeparator(statement[j])) {
133 						j++;
134 					}
135 					if (j - i > 1) {
136 						parameter = sql.substring(i + 1, j);
137 						namedParameterCount = addNewNamedParameter(namedParameters, namedParameterCount, parameter);
138 						totalParameterCount = addNamedParameter(parameterList, totalParameterCount, escapes, i, j, parameter);
139 					}
140 				}
141 				i = j - 1;
142 			}
143 			else {
144 				if (c == '\\') {
145 					int j = i + 1;
146 					if (j < statement.length && statement[j] == ':') {
147 						// this is an escaped : and should be skipped
148 						sqlToUse = sqlToUse.substring(0, i - escapes) + sqlToUse.substring(i - escapes + 1);
149 						escapes++;
150 						i = i + 2;
151 						continue;
152 					}
153 				}
154 				if (c == '?') {
155 					unnamedParameterCount++;
156 					totalParameterCount++;
157 				}
158 			}
159 			i++;
160 		}
161 		ParsedSql parsedSql = new ParsedSql(sqlToUse);
162 		for (ParameterHolder ph : parameterList) {
163 			parsedSql.addNamedParameter(ph.getParameterName(), ph.getStartIndex(), ph.getEndIndex());
164 		}
165 		parsedSql.setNamedParameterCount(namedParameterCount);
166 		parsedSql.setUnnamedParameterCount(unnamedParameterCount);
167 		parsedSql.setTotalParameterCount(totalParameterCount);
168 		return parsedSql;
169 	}
170 
171 	private static int addNamedParameter(
172 			List<ParameterHolder> parameterList, int totalParameterCount, int escapes, int i, int j, String parameter) {
173 
174 		parameterList.add(new ParameterHolder(parameter, i - escapes, j - escapes));
175 		totalParameterCount++;
176 		return totalParameterCount;
177 	}
178 
179 	private static int addNewNamedParameter(Set<String> namedParameters, int namedParameterCount, String parameter) {
180 		if (!namedParameters.contains(parameter)) {
181 			namedParameters.add(parameter);
182 			namedParameterCount++;
183 		}
184 		return namedParameterCount;
185 	}

187 	/**
188 	 * Skip over comments and quoted names present in an SQL statement
189 	 * @param statement character array containing SQL statement
190 	 * @param position current position of statement
191 	 * @return next position to process after any comments or quotes are skipped
192 	 */
193 	private static int skipCommentsAndQuotes(char[] statement, int position) {
194 		for (int i = 0; i < START_SKIP.length; i++) {
195 			if (statement[position] == START_SKIP[i].charAt(0)) {
196 				boolean match = true;
197 				for (int j = 1; j < START_SKIP[i].length(); j++) {
198 					if (!(statement[position + j] == START_SKIP[i].charAt(j))) {
199 						match = false;
200 						break;
201 					}
202 				}
203 				if (match) {
204 					int offset = START_SKIP[i].length();
205 					for (int m = position + offset; m < statement.length; m++) {
206 						if (statement[m] == STOP_SKIP[i].charAt(0)) {
207 							boolean endMatch = true;
208 							int endPos = m;
209 							for (int n = 1; n < STOP_SKIP[i].length(); n++) {
210 								if (m + n >= statement.length) {
211 									// last comment not closed properly
212 									return statement.length;
213 								}
214 								if (!(statement[m + n] == STOP_SKIP[i].charAt(n))) {
215 									endMatch = false;
216 									break;
217 								}
218 								endPos = m + n;
219 							}
220 							if (endMatch) {
221 								// found character sequence ending comment or quote
222 								return endPos + 1;
223 							}
224 						}
225 					}
226 					// character sequence ending comment or quote not found
227 					return statement.length;
228 				}
229 
230 			}
231 		}
232 		return position;
233 	}

235 	/**
236 	 * Parse the SQL statement and locate any placeholders or named parameters. Named
237 	 * parameters are substituted for a JDBC placeholder, and any select list is expanded
238 	 * to the required number of placeholders. Select lists may contain an array of
239 	 * objects, and in that case the placeholders will be grouped and enclosed with
240 	 * parentheses. This allows for the use of "expression lists" in the SQL statement
241 	 * like: <br /><br />
242 	 * {@code select id, name, state from table where (name, age) in (('John', 35), ('Ann', 50))}
243 	 * <p>The parameter values passed in are used to determine the number of placeholders to
244 	 * be used for a select list. Select lists should be limited to 100 or fewer elements.
245 	 * A larger number of elements is not guaranteed to be supported by the database and
246 	 * is strictly vendor-dependent.
247 	 * @param parsedSql the parsed representation of the SQL statement
248 	 * @param paramSource the source for named parameters
249 	 * @return the SQL statement with substituted parameters
250 	 * @see #parseSqlStatement
251 	 */
252 	public static String substituteNamedParameters(ParsedSql parsedSql, SqlParameterSource paramSource) {
253 		String originalSql = parsedSql.getOriginalSql();
254 		StringBuilder actualSql = new StringBuilder();
255 		List<String> paramNames = parsedSql.getParameterNames();
256 		int lastIndex = 0;
257 		for (int i = 0; i < paramNames.size(); i++) {
258 			String paramName = paramNames.get(i);
259 			int[] indexes = parsedSql.getParameterIndexes(i);
260 			int startIndex = indexes[0];
261 			int endIndex = indexes[1];
262 			actualSql.append(originalSql, lastIndex, startIndex);
263 			if (paramSource != null && paramSource.hasValue(paramName)) {
264 				Object value = paramSource.getValue(paramName);
265 				if (value instanceof SqlParameterValue) {
266 					value = ((SqlParameterValue) value).getValue();
267 				}
268 				if (value instanceof Collection) {
269 					Iterator<?> entryIter = ((Collection<?>) value).iterator();
270 					int k = 0;
271 					while (entryIter.hasNext()) {
272 						if (k > 0) {
273 							actualSql.append(", ");
274 						}
275 						k++;
276 						Object entryItem = entryIter.next();
277 						if (entryItem instanceof Object[]) {
278 							Object[] expressionList = (Object[]) entryItem;
279 							actualSql.append("(");
280 							for (int m = 0; m < expressionList.length; m++) {
281 								if (m > 0) {
282 									actualSql.append(", ");
283 								}
284 								actualSql.append("?");
285 							}
286 							actualSql.append(")");
287 						}
288 						else {
289 							actualSql.append("?");
290 						}
291 					}
292 				}
293 				else {
294 					actualSql.append("?");
295 				}
296 			}
297 			else {
298 				actualSql.append("?");
299 			}
300 			lastIndex = endIndex;
301 		}
302 		actualSql.append(originalSql, lastIndex, originalSql.length());
303 		return actualSql.toString();
304 	}

306 	/**
307 	 * Convert a Map of named parameter values to a corresponding array.
308 	 * @param parsedSql the parsed SQL statement
309 	 * @param paramSource the source for named parameters
310 	 * @param declaredParams the List of declared SqlParameter objects
311 	 * (may be {@code null}). If specified, the parameter metadata will
312 	 * be built into the value array in the form of SqlParameterValue objects.
313 	 * @return the array of values
314 	 */
315 	public static Object[] buildValueArray(
316 			ParsedSql parsedSql, SqlParameterSource paramSource, List<SqlParameter> declaredParams) {
317 
318 		Object[] paramArray = new Object[parsedSql.getTotalParameterCount()];
319 		if (parsedSql.getNamedParameterCount() > 0 && parsedSql.getUnnamedParameterCount() > 0) {
320 			throw new InvalidDataAccessApiUsageException(
321 					"Not allowed to mix named and traditional ? placeholders. You have " +
322 					parsedSql.getNamedParameterCount() + " named parameter(s) and " +
323 					parsedSql.getUnnamedParameterCount() + " traditional placeholder(s) in statement: " +
324 					parsedSql.getOriginalSql());
325 		}
326 		List<String> paramNames = parsedSql.getParameterNames();
327 		for (int i = 0; i < paramNames.size(); i++) {
328 			String paramName = paramNames.get(i);
329 			try {
330 				Object value = paramSource.getValue(paramName);
331 				SqlParameter param = findParameter(declaredParams, paramName, i);
332 				paramArray[i] = (param != null ? new SqlParameterValue(param, value) : value);
333 			}
334 			catch (IllegalArgumentException ex) {
335 				throw new InvalidDataAccessApiUsageException(
336 						"No value supplied for the SQL parameter '" + paramName + "': " + ex.getMessage());
337 			}
338 		}
339 		return paramArray;
340 	}

342 	/**
343 	 * Find a matching parameter in the given list of declared parameters.
344 	 * @param declaredParams the declared SqlParameter objects
345 	 * @param paramName the name of the desired parameter
346 	 * @param paramIndex the index of the desired parameter
347 	 * @return the declared SqlParameter, or {@code null} if none found
348 	 */
349 	private static SqlParameter findParameter(List<SqlParameter> declaredParams, String paramName, int paramIndex) {
350 		if (declaredParams != null) {
351 			// First pass: Look for named parameter match.
352 			for (SqlParameter declaredParam : declaredParams) {
353 				if (paramName.equals(declaredParam.getName())) {
354 					return declaredParam;
355 				}
356 			}
357 			// Second pass: Look for parameter index match.
358 			if (paramIndex < declaredParams.size()) {
359 				SqlParameter declaredParam = declaredParams.get(paramIndex);
360 				// Only accept unnamed parameters for index matches.
361 				if (declaredParam.getName() == null) {
362 					return declaredParam;
363 				}
364 			}
365 		}
366 		return null;
367 	}

369 	/**
370 	 * Determine whether a parameter name ends at the current position,
371 	 * that is, whether the given character qualifies as a separator.
372 	 */
373 	private static boolean isParameterSeparator(char c) {
374 		if (Character.isWhitespace(c)) {
375 			return true;
376 		}
377 		for (char separator : PARAMETER_SEPARATORS) {
378 			if (c == separator) {
379 				return true;
380 			}
381 		}
382 		return false;
383 	}

385 	/**
386 	 * Convert parameter types from an SqlParameterSource into a corresponding int array.
387 	 * This is necessary in order to reuse existing methods on JdbcTemplate.
388 	 * Any named parameter types are placed in the correct position in the
389 	 * Object array based on the parsed SQL statement info.
390 	 * @param parsedSql the parsed SQL statement
391 	 * @param paramSource the source for named parameters
392 	 */
393 	public static int[] buildSqlTypeArray(ParsedSql parsedSql, SqlParameterSource paramSource) {
394 		int[] sqlTypes = new int[parsedSql.getTotalParameterCount()];
395 		List<String> paramNames = parsedSql.getParameterNames();
396 		for (int i = 0; i < paramNames.size(); i++) {
397 			String paramName = paramNames.get(i);
398 			sqlTypes[i] = paramSource.getSqlType(paramName);
399 		}
400 		return sqlTypes;
401 	}

403 	/**
404 	 * Convert parameter declarations from an SqlParameterSource to a corresponding List of SqlParameters.
405 	 * This is necessary in order to reuse existing methods on JdbcTemplate.
406 	 * The SqlParameter for a named parameter is placed in the correct position in the
407 	 * resulting list based on the parsed SQL statement info.
408 	 * @param parsedSql the parsed SQL statement
409 	 * @param paramSource the source for named parameters
410 	 */
411 	public static List<SqlParameter> buildSqlParameterList(ParsedSql parsedSql, SqlParameterSource paramSource) {
412 		List<String> paramNames = parsedSql.getParameterNames();
413 		List<SqlParameter> params = new LinkedList<SqlParameter>();
414 		for (String paramName : paramNames) {
415 			params.add(new SqlParameter(paramName, paramSource.getSqlType(paramName), paramSource.getTypeName(paramName)));
416 		}
417 		return params;
418 	}

468 	private static class ParameterHolder {
469 
470 		private final String parameterName;
471 
472 		private final int startIndex;
473 
474 		private final int endIndex;
475 
476 		public ParameterHolder(String parameterName, int startIndex, int endIndex) {
477 			this.parameterName = parameterName;
478 			this.startIndex = startIndex;
479 			this.endIndex = endIndex;
480 		}
481 
482 		public String getParameterName() {
483 			return this.parameterName;
484 		}
485 
486 		public int getStartIndex() {
487 			return this.startIndex;
488 		}
489 
490 		public int getEndIndex() {
491 			return this.endIndex;
492 		}
