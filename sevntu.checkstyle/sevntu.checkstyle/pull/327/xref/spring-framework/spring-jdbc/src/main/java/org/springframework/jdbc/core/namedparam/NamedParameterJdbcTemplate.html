42  /**
43   * Template class with a basic set of JDBC operations, allowing the use
44   * of named parameters rather than traditional '?' placeholders.
45   *
46   * <p>This class delegates to a wrapped {@link #getJdbcOperations() JdbcTemplate}
47   * once the substitution from named parameters to JDBC style '?' placeholders is
48   * done at execution time. It also allows for expanding a {@link java.util.List}
49   * of values to the appropriate number of placeholders.
50   *
51   * <p>The underlying {@link org.springframework.jdbc.core.JdbcTemplate} is
52   * exposed to allow for convenient access to the traditional
53   * {@link org.springframework.jdbc.core.JdbcTemplate} methods.
54   *
55   * <p><b>NOTE: An instance of this class is thread-safe once configured.</b>
56   *
57   * @author Thomas Risberg
58   * @author Juergen Hoeller
59   * @since 2.0
60   * @see NamedParameterJdbcOperations
61   * @see org.springframework.jdbc.core.JdbcTemplate
62   */
63  public class NamedParameterJdbcTemplate implements NamedParameterJdbcOperations {
64  
65  	/** Default maximum number of entries for this template's SQL cache: 256 */
66  	public static final int DEFAULT_CACHE_LIMIT = 256;
67  
68  
69  	/** The JdbcTemplate we are wrapping */
70  	private final JdbcOperations classicJdbcTemplate;
71  
72  	private volatile int cacheLimit = DEFAULT_CACHE_LIMIT;
73  
74  	/** Cache of original SQL String to ParsedSql representation */
75  	@SuppressWarnings("serial")
76  	private final Map<String, ParsedSql> parsedSqlCache =
77  			new LinkedHashMap<String, ParsedSql>(DEFAULT_CACHE_LIMIT, 0.75f, true) {
78  				@Override
79  				protected boolean removeEldestEntry(Map.Entry<String, ParsedSql> eldest) {
80  					return size() > getCacheLimit();
81  				}
82  			};

86  	 * Create a new NamedParameterJdbcTemplate for the given {@link DataSource}.
87  	 * <p>Creates a classic Spring {@link org.springframework.jdbc.core.JdbcTemplate} and wraps it.
88  	 * @param dataSource the JDBC DataSource to access
89  	 */
90  	public NamedParameterJdbcTemplate(DataSource dataSource) {
91  		Assert.notNull(dataSource, "DataSource must not be null");
92  		this.classicJdbcTemplate = new JdbcTemplate(dataSource);
93  	}

302 	@Override
303 	public int update(
304 			String sql, SqlParameterSource paramSource, KeyHolder generatedKeyHolder, String[] keyColumnNames)
305 			throws DataAccessException {
306 
307 		ParsedSql parsedSql = getParsedSql(sql);
308 		String sqlToUse = NamedParameterUtils.substituteNamedParameters(parsedSql, paramSource);
309 		Object[] params = NamedParameterUtils.buildValueArray(parsedSql, paramSource, null);
310 		List<SqlParameter> declaredParameters = NamedParameterUtils.buildSqlParameterList(parsedSql, paramSource);
311 		PreparedStatementCreatorFactory pscf = new PreparedStatementCreatorFactory(sqlToUse, declaredParameters);
312 		if (keyColumnNames != null) {
313 			pscf.setGeneratedKeysColumnNames(keyColumnNames);
314 		}
315 		else {
316 			pscf.setReturnGeneratedKeys(true);
317 		}
318 		return getJdbcOperations().update(pscf.newPreparedStatementCreator(params), generatedKeyHolder);
319 	}

321 	@Override
322 	public int[] batchUpdate(String sql, Map<String, ?>[] batchValues) {
323 		SqlParameterSource[] batchArgs = new SqlParameterSource[batchValues.length];
324 		int i = 0;
325 		for (Map<String, ?> values : batchValues) {
326 			batchArgs[i] = new MapSqlParameterSource(values);
327 			i++;
328 		}
329 		return batchUpdate(sql, batchArgs);
330 	}

335 		return NamedParameterBatchUpdateUtils.executeBatchUpdateWithNamedParameters(parsedSql, batchArgs, getJdbcOperations());

338 	/**
339 	 * Build a PreparedStatementCreator based on the given SQL and named parameters.
340 	 * <p>Note: Not used for the {@code update} variant with generated key handling.
341 	 * @param sql SQL to execute
342 	 * @param paramSource container of arguments to bind
343 	 * @return the corresponding PreparedStatementCreator
344 	 */
345 	protected PreparedStatementCreator getPreparedStatementCreator(String sql, SqlParameterSource paramSource) {
346 		ParsedSql parsedSql = getParsedSql(sql);
347 		String sqlToUse = NamedParameterUtils.substituteNamedParameters(parsedSql, paramSource);
348 		Object[] params = NamedParameterUtils.buildValueArray(parsedSql, paramSource, null);
349 		List<SqlParameter> declaredParameters = NamedParameterUtils.buildSqlParameterList(parsedSql, paramSource);
350 		PreparedStatementCreatorFactory pscf = new PreparedStatementCreatorFactory(sqlToUse, declaredParameters);
351 		return pscf.newPreparedStatementCreator(params);
352 	}
353 
354 	/**
355 	 * Obtain a parsed representation of the given SQL statement.
356 	 * <p>The default implementation uses an LRU cache with an upper limit
357 	 * of 256 entries.
358 	 * @param sql the original SQL
359 	 * @return a representation of the parsed SQL statement
360 	 */
361 	protected ParsedSql getParsedSql(String sql) {
362 		if (getCacheLimit() <= 0) {
363 			return NamedParameterUtils.parseSqlStatement(sql);
364 		}
365 		synchronized (this.parsedSqlCache) {
366 			ParsedSql parsedSql = this.parsedSqlCache.get(sql);
367 			if (parsedSql == null) {
368 				parsedSql = NamedParameterUtils.parseSqlStatement(sql);
369 				this.parsedSqlCache.put(sql, parsedSql);
370 			}
371 			return parsedSql;
372 		}
373 	}
