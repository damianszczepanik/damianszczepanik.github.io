9   public class CustomerMapper implements RowMapper<Customer> {

11  	private static final String[] COLUMN_NAMES = new String[] {"id", "forename"};
12  
13  	@Override
14  	public Customer mapRow(ResultSet rs, int rownum) throws SQLException {
15  		Customer cust = new Customer();
16  		cust.setId(rs.getInt(COLUMN_NAMES[0]));
17  		cust.setForename(rs.getString(COLUMN_NAMES[1]));
18  		return cust;
19  	}
