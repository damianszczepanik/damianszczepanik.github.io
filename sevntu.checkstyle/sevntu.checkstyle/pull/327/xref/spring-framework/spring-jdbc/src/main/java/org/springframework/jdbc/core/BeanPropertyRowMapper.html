42  /**
43   * {@link RowMapper} implementation that converts a row into a new instance
44   * of the specified mapped target class. The mapped target class must be a
45   * top-level class and it must have a default or no-arg constructor.
46   *
47   * <p>Column values are mapped based on matching the column name as obtained from result set
48   * metadata to public setters for the corresponding properties. The names are matched either
49   * directly or by transforming a name separating the parts with underscores to the same name
50   * using "camel" case.
51   *
52   * <p>Mapping is provided for fields in the target class for many common types, e.g.:
53   * String, boolean, Boolean, byte, Byte, short, Short, int, Integer, long, Long,
54   * float, Float, double, Double, BigDecimal, {@code java.util.Date}, etc.
55   *
56   * <p>To facilitate mapping between columns and fields that don't have matching names,
57   * try using column aliases in the SQL statement like "select fname as first_name from customer".
58   *
59   * <p>For 'null' values read from the databasem, we will attempt to call the setter, but in the case of
60   * Java primitives, this causes a TypeMismatchException. This class can be configured (using the
61   * primitivesDefaultedForNullValue property) to trap this exception and use the primitives default value.
62   * Be aware that if you use the values from the generated bean to update the database the primitive value
63   * will have been set to the primitive's default value instead of null.
64   *
65   * <p>Please note that this class is designed to provide convenience rather than high performance.
66   * For best performance, consider using a custom {@link RowMapper} implementation.
67   *
68   * @author Thomas Risberg
69   * @author Juergen Hoeller
70   * @since 2.5
71   */
72  public class BeanPropertyRowMapper<T> implements RowMapper<T> {
73  
74  	/** Logger available to subclasses */
75  	protected final Log logger = LogFactory.getLog(getClass());
76  
77  	/** The class we are mapping to */
78  	private Class<T> mappedClass;
79  
80  	/** Whether we're strictly validating */
81  	private boolean checkFullyPopulated = false;
82  
83  	/** Whether we're defaulting primitives when mapping a null value */
84  	private boolean primitivesDefaultedForNullValue = false;
85  
86  	/** Map of the fields we provide mapping for */
87  	private Map<String, PropertyDescriptor> mappedFields;
88  
89  	/** Set of bean properties we provide mapping for */
90  	private Set<String> mappedProperties;

101 	/**
102 	 * Create a new BeanPropertyRowMapper, accepting unpopulated properties
103 	 * in the target bean.
104 	 * <p>Consider using the {@link #newInstance} factory method instead,
105 	 * which allows for specifying the mapped type once only.
106 	 * @param mappedClass the class that each row should be mapped to
107 	 */
108 	public BeanPropertyRowMapper(Class<T> mappedClass) {
109 		initialize(mappedClass);

112 	/**
113 	 * Create a new BeanPropertyRowMapper.
114 	 * @param mappedClass the class that each row should be mapped to
115 	 * @param checkFullyPopulated whether we're strictly validating that
116 	 * all bean properties have been mapped from corresponding database fields
117 	 */
118 	public BeanPropertyRowMapper(Class<T> mappedClass, boolean checkFullyPopulated) {
119 		initialize(mappedClass);
120 		this.checkFullyPopulated = checkFullyPopulated;
121 	}

123 
124 	/**
125 	 * Set the class that each row should be mapped to.
126 	 */
127 	public void setMappedClass(Class<T> mappedClass) {
128 		if (this.mappedClass == null) {
129 			initialize(mappedClass);
130 		}
131 		else {
132 			if (!this.mappedClass.equals(mappedClass)) {
133 				throw new InvalidDataAccessApiUsageException("The mapped class can not be reassigned to map to " +
134 						mappedClass + " since it is already providing mapping for " + this.mappedClass);
135 			}
136 		}
137 	}
138 
139 	/**
140 	 * Initialize the mapping metadata for the given class.
141 	 * @param mappedClass the mapped class.
142 	 */
143 	protected void initialize(Class<T> mappedClass) {
144 		this.mappedClass = mappedClass;
145 		this.mappedFields = new HashMap<String, PropertyDescriptor>();
146 		this.mappedProperties = new HashSet<String>();
147 		PropertyDescriptor[] pds = BeanUtils.getPropertyDescriptors(mappedClass);
148 		for (PropertyDescriptor pd : pds) {
149 			if (pd.getWriteMethod() != null) {
150 				this.mappedFields.put(pd.getName().toLowerCase(), pd);
151 				String underscoredName = underscoreName(pd.getName());
152 				if (!pd.getName().toLowerCase().equals(underscoredName)) {
153 					this.mappedFields.put(underscoredName, pd);
154 				}
155 				this.mappedProperties.add(pd.getName());
156 			}
157 		}
158 	}
159 
160 	/**
161 	 * Convert a name in camelCase to an underscored name in lower case.
162 	 * Any upper case letters are converted to lower case with a preceding underscore.
163 	 * @param name the string containing original name
164 	 * @return the converted name
165 	 */
166 	private String underscoreName(String name) {
167 		if (!StringUtils.hasLength(name)) {
168 			return "";
169 		}
170 		StringBuilder result = new StringBuilder();
171 		result.append(name.substring(0, 1).toLowerCase());
172 		for (int i = 1; i < name.length(); i++) {
173 			String s = name.substring(i, i + 1);
174 			String slc = s.toLowerCase();
175 			if (!s.equals(slc)) {
176 				result.append("_").append(slc);
177 			}
178 			else {
179 				result.append(s);
180 			}
181 		}
182 		return result.toString();
183 	}

211 	 * Set whether we're defaulting Java primitives in the case of mapping a null value

214 	 */

217 	}
218 
219 	/**
220 	 * Return whether we're defaulting Java primitives in the case of mapping a null value
221 	 * from corresponding database fields.
222 	 */
223 	public boolean isPrimitivesDefaultedForNullValue() {
224 		return primitivesDefaultedForNullValue;
225 	}

228 	/**
229 	 * Extract the values for all columns in the current row.
230 	 * <p>Utilizes public setters and result set metadata.
231 	 * @see java.sql.ResultSetMetaData
232 	 */
233 	@Override
234 	public T mapRow(ResultSet rs, int rowNumber) throws SQLException {
235 		Assert.state(this.mappedClass != null, "Mapped class was not specified");
236 		T mappedObject = BeanUtils.instantiate(this.mappedClass);
237 		BeanWrapper bw = PropertyAccessorFactory.forBeanPropertyAccess(mappedObject);
238 		initBeanWrapper(bw);
239 
240 		ResultSetMetaData rsmd = rs.getMetaData();
241 		int columnCount = rsmd.getColumnCount();
242 		Set<String> populatedProperties = (isCheckFullyPopulated() ? new HashSet<String>() : null);
243 
244 		for (int index = 1; index <= columnCount; index++) {
245 			String column = JdbcUtils.lookupColumnName(rsmd, index);
246 			PropertyDescriptor pd = this.mappedFields.get(column.replaceAll(" ", "").toLowerCase());
247 			if (pd != null) {
248 				try {
249 					Object value = getColumnValue(rs, index, pd);
250 					if (logger.isDebugEnabled() && rowNumber == 0) {
251 						logger.debug("Mapping column '" + column + "' to property '" +
252 								pd.getName() + "' of type " + pd.getPropertyType());
253 					}
254 					try {
255 						bw.setPropertyValue(pd.getName(), value);
256 					}
257 					catch (TypeMismatchException e) {
258 						if (value == null && primitivesDefaultedForNullValue) {
259 							logger.debug("Intercepted TypeMismatchException for row " + rowNumber +
260 									" and column '" + column + "' with value " + value +
261 									" when setting property '" + pd.getName() + "' of type " + pd.getPropertyType() +
262 									" on object: " + mappedObject);
263 						}
264 						else {
265 							throw e;
266 						}
267 					}
268 					if (populatedProperties != null) {
269 						populatedProperties.add(pd.getName());
270 					}
271 				}
272 				catch (NotWritablePropertyException ex) {
273 					throw new DataRetrievalFailureException(
274 							"Unable to map column " + column + " to property " + pd.getName(), ex);
275 				}
276 			}
277 		}
278 
279 		if (populatedProperties != null && !populatedProperties.equals(this.mappedProperties)) {
280 			throw new InvalidDataAccessApiUsageException("Given ResultSet does not contain all fields " +
281 					"necessary to populate object of class [" + this.mappedClass + "]: " + this.mappedProperties);
282 		}
283 
284 		return mappedObject;
285 	}

296 	/**
297 	 * Retrieve a JDBC object value for the specified column.
298 	 * <p>The default implementation calls
299 	 * {@link JdbcUtils#getResultSetValue(java.sql.ResultSet, int, Class)}.
300 	 * Subclasses may override this to check specific value types upfront,
301 	 * or to post-process values return from {@code getResultSetValue}.
302 	 * @param rs is the ResultSet holding the data
303 	 * @param index is the column index
304 	 * @param pd the bean property that each result object is expected to match
305 	 * (or {@code null} if none specified)
306 	 * @return the Object value
307 	 * @throws SQLException in case of extraction failure
308 	 * @see org.springframework.jdbc.support.JdbcUtils#getResultSetValue(java.sql.ResultSet, int, Class)
309 	 */
310 	protected Object getColumnValue(ResultSet rs, int index, PropertyDescriptor pd) throws SQLException {
311 		return JdbcUtils.getResultSetValue(rs, index, pd.getPropertyType());
312 	}

315 	/**
316 	 * Static factory method to create a new BeanPropertyRowMapper
317 	 * (with the mapped class specified only once).
318 	 * @param mappedClass the class that each row should be mapped to
319 	 */
320 	public static <T> BeanPropertyRowMapper<T> newInstance(Class<T> mappedClass) {
321 		BeanPropertyRowMapper<T> newInstance = new BeanPropertyRowMapper<T>();
322 		newInstance.setMappedClass(mappedClass);
323 		return newInstance;
324 	}
