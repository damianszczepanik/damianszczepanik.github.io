17  package org.springframework.messaging.simp.annotation.support;
18  
19  import java.lang.annotation.Annotation;
20  import java.security.Principal;
21  
22  import org.springframework.core.MethodParameter;
23  import org.springframework.core.annotation.AnnotationUtils;
24  import org.springframework.messaging.Message;
25  import org.springframework.messaging.MessageChannel;
26  import org.springframework.messaging.MessageHeaders;
27  import org.springframework.messaging.handler.DestinationPatternsMessageCondition;
28  import org.springframework.messaging.handler.annotation.SendTo;
29  import org.springframework.messaging.handler.invocation.HandlerMethodReturnValueHandler;
30  import org.springframework.messaging.simp.SimpMessageHeaderAccessor;
31  import org.springframework.messaging.simp.SimpMessageSendingOperations;
32  import org.springframework.messaging.simp.SimpMessageType;
33  import org.springframework.messaging.simp.annotation.SendToUser;
34  import org.springframework.messaging.simp.user.DestinationUserNameProvider;
35  import org.springframework.messaging.support.MessageHeaderInitializer;
36  import org.springframework.util.Assert;
37  import org.springframework.util.ObjectUtils;

39  /**
40   * A {@link HandlerMethodReturnValueHandler} for sending to destinations specified in a
41   * {@link SendTo} or {@link SendToUser} method-level annotations.
42   *
43   * <p>The value returned from the method is converted, and turned to a {@link Message} and
44   * sent through the provided {@link MessageChannel}. The
45   * message is then enriched with the sessionId of the input message as well as the
46   * destination from the annotation(s). If multiple destinations are specified, a copy of
47   * the message is sent to each destination.
48   *
49   * @author Rossen Stoyanchev
50   * @since 4.0
51   */
52  public class SendToMethodReturnValueHandler implements HandlerMethodReturnValueHandler {
53  
54  	private final SimpMessageSendingOperations messagingTemplate;
55  
56  	private final boolean annotationRequired;
57  
58  	private String defaultDestinationPrefix = "/topic";
59  
60  	private String defaultUserDestinationPrefix = "/queue";
61  
62  	private MessageHeaderInitializer headerInitializer;
63  
64  
65  	public SendToMethodReturnValueHandler(SimpMessageSendingOperations messagingTemplate, boolean annotationRequired) {
66  		Assert.notNull(messagingTemplate, "messagingTemplate must not be null");
67  		this.messagingTemplate = messagingTemplate;
68  		this.annotationRequired = annotationRequired;
69  	}
70  
71  
72  	/**
73  	 * Configure a default prefix to add to message destinations in cases where a method
74  	 * is not annotated with {@link SendTo @SendTo} or does not specify any destinations
75  	 * through the annotation's value attribute.
76  	 * <p>By default, the prefix is set to "/topic".
77  	 */
78  	public void setDefaultDestinationPrefix(String defaultDestinationPrefix) {
79  		this.defaultDestinationPrefix = defaultDestinationPrefix;
80  	}
81  
82  	/**
83  	 * Return the configured default destination prefix.
84  	 * @see #setDefaultDestinationPrefix(String)
85  	 */
86  	public String getDefaultDestinationPrefix() {
87  		return this.defaultDestinationPrefix;
88  	}
89  
90  	/**
91  	 * Configure a default prefix to add to message destinations in cases where a
92  	 * method is annotated with {@link SendToUser @SendToUser} but does not specify
93  	 * any destinations through the annotation's value attribute.
94  	 * <p>By default, the prefix is set to "/queue".
95  	 */
96  	public void setDefaultUserDestinationPrefix(String prefix) {
97  		this.defaultUserDestinationPrefix = prefix;
98  	}
99  
100 	/**
101 	 * Return the configured default user destination prefix.
102 	 * @see #setDefaultUserDestinationPrefix(String)
103 	 */
104 	public String getDefaultUserDestinationPrefix() {
105 		return this.defaultUserDestinationPrefix;
106 	}
107 
108 	/**
109 	 * Configure a {@link MessageHeaderInitializer} to apply to the headers of all
110 	 * messages sent to the client outbound channel.
111 	 *
112 	 * <p>By default this property is not set.
113 	 */
114 	public void setHeaderInitializer(MessageHeaderInitializer headerInitializer) {
115 		this.headerInitializer = headerInitializer;
116 	}
117 
118 	/**
119 	 * @return the configured header initializer.
120 	 */
121 	public MessageHeaderInitializer getHeaderInitializer() {
122 		return this.headerInitializer;
123 	}
124 
125 
126 	@Override
127 	public boolean supportsReturnType(MethodParameter returnType) {
128 		if ((returnType.getMethodAnnotation(SendTo.class) != null) ||
129 				(returnType.getMethodAnnotation(SendToUser.class) != null)) {
130 			return true;
131 		}
132 		return (!this.annotationRequired);
133 	}
134 
135 	@Override
136 	public void handleReturnValue(Object returnValue, MethodParameter returnType, Message<?> message) throws Exception {
137 		if (returnValue == null) {
138 			return;
139 		}
140 		MessageHeaders headers = message.getHeaders();
141 		String sessionId = SimpMessageHeaderAccessor.getSessionId(headers);
142 
143 		SendToUser sendToUser = returnType.getMethodAnnotation(SendToUser.class);
144 		if (sendToUser != null) {
145 			boolean broadcast = sendToUser.broadcast();
146 			String user = getUserName(message, headers);
147 			if (user == null) {
148 				if (sessionId == null) {
149 					throw new MissingSessionUserException(message);
150 				}
151 				user = sessionId;
152 				broadcast = false;
153 			}
154 			String[] destinations = getTargetDestinations(sendToUser, message, this.defaultUserDestinationPrefix);
155 			for (String destination : destinations) {
156 				if (broadcast) {
157 					this.messagingTemplate.convertAndSendToUser(user, destination, returnValue);
158 				}
159 				else {
160 					this.messagingTemplate.convertAndSendToUser(user, destination, returnValue, createHeaders(sessionId));
161 				}
162 			}
163 		}
164 		else {
165 			SendTo sendTo = returnType.getMethodAnnotation(SendTo.class);
166 			String[] destinations = getTargetDestinations(sendTo, message, this.defaultDestinationPrefix);
167 			for (String destination : destinations) {
168 				this.messagingTemplate.convertAndSend(destination, returnValue, createHeaders(sessionId));
169 			}
170 		}
171 	}
172 
173 	protected String getUserName(Message<?> message, MessageHeaders headers) {
174 		Principal principal = SimpMessageHeaderAccessor.getUser(headers);
175 		if (principal != null) {
176 			return (principal instanceof DestinationUserNameProvider ?
177 					((DestinationUserNameProvider) principal).getDestinationUserName() : principal.getName());
178 		}
179 		return null;
180 	}
181 
182 	protected String[] getTargetDestinations(Annotation annotation, Message<?> message, String defaultPrefix) {
183 		if (annotation != null) {
184 			String[] value = (String[]) AnnotationUtils.getValue(annotation);
185 			if (!ObjectUtils.isEmpty(value)) {
186 				return value;
187 			}
188 		}
189 		String name = DestinationPatternsMessageCondition.LOOKUP_DESTINATION_HEADER;
190 		String destination = (String) message.getHeaders().get(name);
191 		Assert.hasText(destination, "No lookup destination header in " + message);
192 
193 		return (destination.startsWith("/") ?
194 				new String[] {defaultPrefix + destination} : new String[] {defaultPrefix + "/" + destination});
195 	}
196 
197 	private MessageHeaders createHeaders(String sessionId) {
198 		SimpMessageHeaderAccessor headerAccessor = SimpMessageHeaderAccessor.create(SimpMessageType.MESSAGE);
199 		if (getHeaderInitializer() != null) {
200 			getHeaderInitializer().initHeaders(headerAccessor);
201 		}
202 		headerAccessor.setSessionId(sessionId);
203 		headerAccessor.setLeaveMutable(true);
204 		return headerAccessor.getMessageHeaders();
205 	}
206 
207 
208 	@Override
209 	public String toString() {
210 		return "SendToMethodReturnValueHandler [annotationRequired=" + annotationRequired + "]";
211 	}
