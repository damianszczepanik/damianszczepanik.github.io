36  /**
37   * A default, simple in-memory implementation of {@link SubscriptionRegistry}.
38   *
39   * @author Rossen Stoyanchev
40   * @author Sebastien Deleuze
41   * @since 4.0
42   */
43  public class DefaultSubscriptionRegistry extends AbstractSubscriptionRegistry {
44  
45  	/** Default maximum number of entries for the destination cache: 1024 */
46  	public static final int DEFAULT_CACHE_LIMIT = 1024;
47  
48  
49  	/** The maximum number of entries in the cache */
50  	private volatile int cacheLimit = DEFAULT_CACHE_LIMIT;
51  
52  	private PathMatcher pathMatcher = new AntPathMatcher();
53  
54  	private final DestinationCache destinationCache = new DestinationCache();
55  
56  	private final SessionSubscriptionRegistry subscriptionRegistry = new SessionSubscriptionRegistry();
57  
58  
59  	/**
60  	 * Specify the maximum number of entries for the resolved destination cache.
61  	 * Default is 1024.
62  	 */
63  	public void setCacheLimit(int cacheLimit) {
64  		this.cacheLimit = cacheLimit;
65  	}
66  
67  	/**
68  	 * Return the maximum number of entries for the resolved destination cache.
69  	 */
70  	public int getCacheLimit() {
71  		return this.cacheLimit;
72  	}
73  
74  	/**
75  	 * Specify the {@link PathMatcher} to use.
76  	 */
77  	public void setPathMatcher(PathMatcher pathMatcher) {
78  		this.pathMatcher = pathMatcher;
79  	}
80  
81  	/**
82  	 * Return the configured {@link PathMatcher}.
83  	 */
84  	public PathMatcher getPathMatcher() {
85  		return this.pathMatcher;
86  	}
87  
88  
89  	@Override
90  	protected void addSubscriptionInternal(String sessionId, String subsId, String destination, Message<?> message) {
91  		this.subscriptionRegistry.addSubscription(sessionId, subsId, destination);
92  		this.destinationCache.updateAfterNewSubscription(destination, sessionId, subsId);
93  	}
94  
95  	@Override
96  	protected void removeSubscriptionInternal(String sessionId, String subsId, Message<?> message) {
97  		SessionSubscriptionInfo info = this.subscriptionRegistry.getSubscriptions(sessionId);
98  		if (info != null) {
99  			String destination = info.removeSubscription(subsId);
100 			if (destination != null) {
101 				this.destinationCache.updateAfterRemovedSubscription(sessionId, subsId);
102 			}
103 		}
104 	}
105 
106 	@Override
107 	public void unregisterAllSubscriptions(String sessionId) {
108 		SessionSubscriptionInfo info = this.subscriptionRegistry.removeSubscriptions(sessionId);
109 		if (info != null) {
110 			this.destinationCache.updateAfterRemovedSession(info);
111 		}
112 	}
113 
114 	@Override
115 	protected MultiValueMap<String, String> findSubscriptionsInternal(String destination, Message<?> message) {
116 		MultiValueMap<String, String> result = this.destinationCache.getSubscriptions(destination);
117 		if (result != null) {
118 			return result;
119 		}
120 		result = new LinkedMultiValueMap<String, String>();
121 		for (SessionSubscriptionInfo info : this.subscriptionRegistry.getAllSubscriptions()) {
122 			for (String destinationPattern : info.getDestinations()) {
123 				if (this.pathMatcher.match(destinationPattern, destination)) {
124 					for (String subscriptionId : info.getSubscriptions(destinationPattern)) {
125 						result.add(info.sessionId, subscriptionId);
126 					}
127 				}
128 			}
129 		}
130 		if (!result.isEmpty()) {
131 			this.destinationCache.addSubscriptions(destination, result);
132 		}
133 		return result;
134 	}
135 
136 	@Override
137 	public String toString() {
138 		return "DefaultSubscriptionRegistry[" + this.destinationCache + ", " + this.subscriptionRegistry + "]";
139 	}

142 	/**
143 	 * A cache for destinations previously resolved via
144 	 * {@link DefaultSubscriptionRegistry#findSubscriptionsInternal(String, Message)}
145 	 */
146 	private class DestinationCache {
147 
148 		/** Map from destination -> <sessionId, subscriptionId> for fast look-ups */
149 		private final Map<String, MultiValueMap<String, String>> accessCache =
150 				new ConcurrentHashMap<String, MultiValueMap<String, String>>(DEFAULT_CACHE_LIMIT);
151 
152 		/** Map from destination -> <sessionId, subscriptionId> with locking */
153 		@SuppressWarnings("serial")
154 		private final Map<String, MultiValueMap<String, String>> updateCache =
155 				new LinkedHashMap<String, MultiValueMap<String, String>>(DEFAULT_CACHE_LIMIT, 0.75f, true) {
156 					@Override
157 					protected boolean removeEldestEntry(Map.Entry<String, MultiValueMap<String, String>> eldest) {
158 						return size() > getCacheLimit();
159 					}
160 				};
161 
162 
163 		public MultiValueMap<String, String> getSubscriptions(String destination) {
164 			return this.accessCache.get(destination);
165 		}
166 
167 		public void addSubscriptions(String destination, MultiValueMap<String, String> subscriptions) {
168 			synchronized (this.updateCache) {
169 				this.updateCache.put(destination, new LinkedMultiValueMap<String, String>(subscriptions));
170 				this.accessCache.put(destination, subscriptions);
171 			}
172 		}
173 
174 		public void updateAfterNewSubscription(String destination, String sessionId, String subsId) {
175 			synchronized (this.updateCache) {
176 				for (Map.Entry<String, MultiValueMap<String, String>> entry : this.updateCache.entrySet()) {
177 					String cachedDestination = entry.getKey();
178 					if (getPathMatcher().match(destination, cachedDestination)) {
179 						MultiValueMap<String, String> subs = entry.getValue();
180 						subs.add(sessionId, subsId);
181 						this.accessCache.put(cachedDestination, new LinkedMultiValueMap<String, String>(subs));
182 					}
183 				}
184 			}
185 		}
186 
187 		public void updateAfterRemovedSubscription(String sessionId, String subsId) {
188 			synchronized (this.updateCache) {
189 				Set<String> destinationsToRemove = new HashSet<String>();
190 				for (Map.Entry<String, MultiValueMap<String, String>> entry : this.updateCache.entrySet()) {
191 					String destination = entry.getKey();
192 					MultiValueMap<String, String> sessionMap = entry.getValue();
193 					List<String> subscriptions = sessionMap.get(sessionId);
194 					if (subscriptions != null) {
195 						subscriptions.remove(subsId);
196 						if (subscriptions.isEmpty()) {
197 							sessionMap.remove(sessionId);
198 						}
199 						if (sessionMap.isEmpty()) {
200 							destinationsToRemove.add(destination);
201 						}
202 						else {
203 							this.accessCache.put(destination, new LinkedMultiValueMap<String, String>(sessionMap));
204 						}
205 					}
206 				}
207 				for (String destination : destinationsToRemove) {
208 					this.updateCache.remove(destination);
209 					this.accessCache.remove(destination);
210 				}
211 			}
212 		}
213 
214 		public void updateAfterRemovedSession(SessionSubscriptionInfo info) {
215 			synchronized (this.updateCache) {
216 				Set<String> destinationsToRemove = new HashSet<String>();
217 				for (Map.Entry<String, MultiValueMap<String, String>> entry : this.updateCache.entrySet()) {
218 					String destination = entry.getKey();
219 					MultiValueMap<String, String> sessionMap = entry.getValue();
220 					if (sessionMap.remove(info.getSessionId()) != null) {
221 						if (sessionMap.isEmpty()) {
222 							destinationsToRemove.add(destination);
223 						}
224 						else {
225 							this.accessCache.put(destination, new LinkedMultiValueMap<String, String>(sessionMap));
226 						}
227 					}
228 				}
229 				for (String destination : destinationsToRemove) {
230 					this.updateCache.remove(destination);
231 					this.accessCache.remove(destination);
232 				}
233 			}
234 		}
235 
236 		@Override
237 		public String toString() {
238 			return "cache[" + this.accessCache.size() + " destination(s)]";
239 		}
240 	}

242 	/**
243 	 * Provide access to session subscriptions by sessionId.
244 	 */
245 	private static class SessionSubscriptionRegistry {
246 
247 		// sessionId -> SessionSubscriptionInfo
248 		private final ConcurrentMap<String, SessionSubscriptionInfo> sessions =
249 				new ConcurrentHashMap<String, SessionSubscriptionInfo>();
250 
251 
252 		public SessionSubscriptionInfo getSubscriptions(String sessionId) {
253 			return this.sessions.get(sessionId);
254 		}
255 
256 		public Collection<SessionSubscriptionInfo> getAllSubscriptions() {
257 			return this.sessions.values();
258 		}
259 
260 		public SessionSubscriptionInfo addSubscription(String sessionId, String subscriptionId, String destination) {
261 			SessionSubscriptionInfo info = this.sessions.get(sessionId);
262 			if (info == null) {
263 				info = new SessionSubscriptionInfo(sessionId);
264 				SessionSubscriptionInfo value = this.sessions.putIfAbsent(sessionId, info);
265 				if (value != null) {
266 					info = value;
267 				}
268 			}
269 			info.addSubscription(destination, subscriptionId);
270 			return info;
271 		}
272 
273 		public SessionSubscriptionInfo removeSubscriptions(String sessionId) {
274 			return this.sessions.remove(sessionId);
275 		}
276 
277 		@Override
278 		public String toString() {
279 			return "registry[" + sessions.size() + " sessions]";
280 		}
281 	}

283 	/**
284 	 * Hold subscriptions for a session.
285 	 */
286 	private static class SessionSubscriptionInfo {
287 
288 		private final String sessionId;
289 
290 		// destination -> subscriptionIds
291 		private final Map<String, Set<String>> subscriptions = new ConcurrentHashMap<String, Set<String>>(4);
292 
293 		private final Object monitor = new Object();
294 
295 
296 		public SessionSubscriptionInfo(String sessionId) {
297 			Assert.notNull(sessionId, "sessionId must not be null");
298 			this.sessionId = sessionId;
299 		}
300 
301 		public String getSessionId() {
302 			return this.sessionId;
303 		}
304 
305 		public Set<String> getDestinations() {
306 			return this.subscriptions.keySet();
307 		}
308 
309 		public Set<String> getSubscriptions(String destination) {
310 			return this.subscriptions.get(destination);
311 		}
312 
313 		public void addSubscription(String destination, String subscriptionId) {
314 			Set<String> subs = this.subscriptions.get(destination);
315 			if (subs == null) {
316 				synchronized (this.monitor) {
317 					subs = this.subscriptions.get(destination);
318 					if (subs == null) {
319 						subs = new HashSet<String>(4);
320 						this.subscriptions.put(destination, subs);
321 					}
322 				}
323 			}
324 			subs.add(subscriptionId);
325 		}
326 
327 		public String removeSubscription(String subscriptionId) {
328 			for (String destination : this.subscriptions.keySet()) {
329 				Set<String> subscriptionIds = this.subscriptions.get(destination);
330 				if (subscriptionIds.remove(subscriptionId)) {
331 					synchronized (this.monitor) {
332 						if (subscriptionIds.isEmpty()) {
333 							this.subscriptions.remove(destination);
334 						}
335 					}
336 					return destination;
337 				}
338 			}
339 			return null;
340 		}
341 
342 		@Override
343 		public String toString() {
344 			return "[sessionId=" + this.sessionId + ", subscriptions=" + this.subscriptions + "]";
345 		}
