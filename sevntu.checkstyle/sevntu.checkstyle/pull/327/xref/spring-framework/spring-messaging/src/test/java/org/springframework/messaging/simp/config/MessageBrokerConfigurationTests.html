126 		TestChannel channel = this.simpleBrokerContext.getBean("clientOutboundChannel", TestChannel.class);
127 		SimpAnnotationMethodMessageHandler messageHandler = this.simpleBrokerContext.getBean(SimpAnnotationMethodMessageHandler.class);
128 
129 		StompHeaderAccessor headers = StompHeaderAccessor.create(StompCommand.SUBSCRIBE);
130 		headers.setSessionId("sess1");
131 		headers.setSessionAttributes(new ConcurrentHashMap<>());
132 		headers.setSubscriptionId("subs1");
133 		headers.setDestination("/foo");
134 		Message<?> message = MessageBuilder.withPayload(new byte[0]).setHeaders(headers).build();
135 
136 		messageHandler.handleMessage(message);
137 
138 		message = channel.messages.get(0);
139 		headers = StompHeaderAccessor.wrap(message);
140 
141 		assertEquals(SimpMessageType.MESSAGE, headers.getMessageType());
142 		assertEquals("/foo", headers.getDestination());
143 		assertEquals("bar", new String((byte[]) message.getPayload()));
144 	}

461 	@Configuration
462 	static class CustomConfig extends AbstractMessageBrokerConfiguration {
463 
464 		private ChannelInterceptor interceptor = new ChannelInterceptorAdapter() {};
465 
466 		@Override
467 		protected void configureClientInboundChannel(ChannelRegistration registration) {
468 			registration.setInterceptors(this.interceptor);
469 			registration.taskExecutor(new CustomThreadPoolTaskExecutor())
470 					.corePoolSize(11).maxPoolSize(12).keepAliveSeconds(13).queueCapacity(14);
471 		}
472 
473 		@Override
474 		protected void configureClientOutboundChannel(ChannelRegistration registration) {
475 			registration.setInterceptors(this.interceptor, this.interceptor);
476 			registration.taskExecutor().corePoolSize(21).maxPoolSize(22).keepAliveSeconds(23).queueCapacity(24);
477 		}
478 
479 		@Override
480 		protected void addArgumentResolvers(List<HandlerMethodArgumentResolver> argumentResolvers) {
481 			argumentResolvers.add(mock(HandlerMethodArgumentResolver.class));
482 		}
483 
484 		@Override
485 		protected void addReturnValueHandlers(List<HandlerMethodReturnValueHandler> returnValueHandlers) {
486 			returnValueHandlers.add(mock(HandlerMethodReturnValueHandler.class));
487 		}
488 
489 		@Override
490 		protected void configureMessageBroker(MessageBrokerRegistry registry) {
491 			registry.configureBrokerChannel().setInterceptors(this.interceptor, this.interceptor, this.interceptor);
492 			registry.configureBrokerChannel().taskExecutor().corePoolSize(31).maxPoolSize(32).keepAliveSeconds(33).queueCapacity(34);
493 			registry.setPathMatcher(new AntPathMatcher(".")).enableSimpleBroker("/topic", "/queue");
494 		}
