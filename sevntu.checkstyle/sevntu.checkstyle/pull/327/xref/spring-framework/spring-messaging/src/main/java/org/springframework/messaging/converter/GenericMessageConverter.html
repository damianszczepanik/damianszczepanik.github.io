17  package org.springframework.messaging.converter;
18  
19  import org.springframework.core.convert.ConversionException;
20  import org.springframework.core.convert.ConversionService;
21  import org.springframework.core.convert.support.DefaultConversionService;
22  import org.springframework.messaging.Message;
23  import org.springframework.util.Assert;
24  import org.springframework.util.ClassUtils;
25  
26  /**
27   * An extension of the {@link SimpleMessageConverter} that uses a
28   * {@link ConversionService} to convert the payload of the message
29   * to the requested type.
30   *
31   * <p>Return {@code null} if the conversion service cannot convert
32   * from the payload type to the requested type.
33   *
34   * @author Stephane Nicoll
35   * @since 4.1
36   * @see ConversionService
37   */
38  public class GenericMessageConverter extends SimpleMessageConverter {
39  
40  	private final ConversionService conversionService;
41  
42  
43  	/**
44  	 * Create a new instance with a default {@link ConversionService}.
45  	 */
46  	public GenericMessageConverter() {
47  		this.conversionService = new DefaultConversionService();
48  	}
49  
50  	/**
51  	 * Create a new instance with the given {@link ConversionService}.
52  	 */
53  	public GenericMessageConverter(ConversionService conversionService) {
54  		Assert.notNull(conversionService, "ConversionService must not be null");
55  		this.conversionService = conversionService;
56  	}
57  
58  
59  	@Override
60  	public Object fromMessage(Message<?> message, Class<?> targetClass) {
61  		Object payload = message.getPayload();
62  		if (targetClass == null) {
63  			return payload;
64  		}
65  		if (payload != null && this.conversionService.canConvert(payload.getClass(), targetClass)) {
66  			try {
67  				return this.conversionService.convert(payload, targetClass);
68  			}
69  			catch (ConversionException ex) {
70  				throw new MessageConversionException(message, "Failed to convert message payload '" +
71  						payload + "' to '" + targetClass.getName() + "'", ex);
72  			}
73  		}
74  		return (ClassUtils.isAssignableValue(targetClass, payload) ? payload : null);
75  	}
76  
77  }
