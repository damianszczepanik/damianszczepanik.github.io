17  package org.springframework.messaging.simp.config;
18  
19  import org.springframework.messaging.MessageChannel;
20  import org.springframework.messaging.SubscribableChannel;
21  import org.springframework.messaging.simp.stomp.StompBrokerRelayMessageHandler;
22  import org.springframework.util.Assert;
23  
24  /**
25   * Registration class for configuring a {@link StompBrokerRelayMessageHandler}.
26   *
27   * @author Rossen Stoyanchev
28   * @since 4.0
29   */
30  public class StompBrokerRelayRegistration extends AbstractBrokerRegistration {
31  
32  	private String relayHost = "127.0.0.1";
33  
34  	private int relayPort = 61613;
35  
36  	private String clientLogin = "guest";
37  
38  	private String clientPasscode = "guest";
39  
40  	private String systemLogin = "guest";
41  
42  	private String systemPasscode = "guest";
43  
44  	private Long systemHeartbeatSendInterval;
45  
46  	private Long systemHeartbeatReceiveInterval;
47  
48  	private String virtualHost;
49  
50  	private boolean autoStartup = true;
51  
52  
53  	public StompBrokerRelayRegistration(SubscribableChannel clientInboundChannel,
54  			MessageChannel clientOutboundChannel, String[] destinationPrefixes) {
55  
56  		super(clientInboundChannel, clientOutboundChannel, destinationPrefixes);
57  	}
58  
59  
60  	/**
61  	 * Set the STOMP message broker host.
62  	 */
63  	public StompBrokerRelayRegistration setRelayHost(String relayHost) {
64  		Assert.hasText(relayHost, "relayHost must not be empty");
65  		this.relayHost = relayHost;
66  		return this;
67  	}
68  
69  	/**
70  	 * Set the STOMP message broker port.
71  	 */
72  	public StompBrokerRelayRegistration setRelayPort(int relayPort) {
73  		this.relayPort = relayPort;
74  		return this;
75  	}
76  
77  	/**
78  	 * Set the login to use when creating connections to the STOMP broker on
79  	 * behalf of connected clients.
80  	 * <p>By default this is set to "guest".
81  	 */
82  	public StompBrokerRelayRegistration setClientLogin(String login) {
83  		Assert.hasText(login, "clientLogin must not be empty");
84  		this.clientLogin = login;
85  		return this;
86  	}
87  
88  	/**
89  	 * Set the passcode to use when creating connections to the STOMP broker on
90  	 * behalf of connected clients.
91  	 * <p>By default this is set to "guest".
92  	 */
93  	public StompBrokerRelayRegistration setClientPasscode(String passcode) {
94  		Assert.hasText(passcode, "clientPasscode must not be empty");
95  		this.clientPasscode = passcode;
96  		return this;
97  	}
98  
99  	/**
100 	 * Set the login for the shared "system" connection used to send messages to
101 	 * the STOMP broker from within the application, i.e. messages not associated
102 	 * with a specific client session (e.g. REST/HTTP request handling method).
103 	 * <p>By default this is set to "guest".
104 	 */
105 	public StompBrokerRelayRegistration setSystemLogin(String login) {
106 		Assert.hasText(login, "systemLogin must not be empty");
107 		this.systemLogin = login;
108 		return this;
109 	}
110 
111 	/**
112 	 * Set the passcode for the shared "system" connection used to send messages to
113 	 * the STOMP broker from within the application, i.e. messages not associated
114 	 * with a specific client session (e.g. REST/HTTP request handling method).
115 	 * <p>By default this is set to "guest".
116 	 */
117 	public StompBrokerRelayRegistration setSystemPasscode(String passcode) {
118 		Assert.hasText(passcode, "systemPasscode must not be empty");
119 		this.systemPasscode = passcode;
120 		return this;
121 	}
122 
123 	/**
124 	 * Set the interval, in milliseconds, at which the "system" relay session will,
125 	 * in the absence of any other data being sent, send a heartbeat to the STOMP broker.
126 	 * A value of zero will prevent heartbeats from being sent to the broker.
127 	 * <p>The default value is 10000.
128 	 */
129 	public StompBrokerRelayRegistration setSystemHeartbeatSendInterval(long systemHeartbeatSendInterval) {
130 		this.systemHeartbeatSendInterval = systemHeartbeatSendInterval;
131 		return this;
132 	}
133 
134 	/**
135 	 * Set the maximum interval, in milliseconds, at which the "system" relay session
136 	 * expects, in the absence of any other data, to receive a heartbeat from the STOMP
137 	 * broker. A value of zero will configure the relay session to expect not to receive
138 	 * heartbeats from the broker.
139 	 * <p>The default value is 10000.
140 	 */
141 	public StompBrokerRelayRegistration setSystemHeartbeatReceiveInterval(long heartbeatReceiveInterval) {
142 		this.systemHeartbeatReceiveInterval = heartbeatReceiveInterval;
143 		return this;
144 	}
145 
146 	/**
147 	 * Set the value of the "host" header to use in STOMP CONNECT frames. When this
148 	 * property is configured, a "host" header will be added to every STOMP frame sent to
149 	 * the STOMP broker. This may be useful for example in a cloud environment where the
150 	 * actual host to which the TCP connection is established is different from the host
151 	 * providing the cloud-based STOMP service.
152 	 * <p>By default this property is not set.
153 	 */
154 	public StompBrokerRelayRegistration setVirtualHost(String virtualHost) {
155 		this.virtualHost = virtualHost;
156 		return this;
157 	}
158 
159 	/**
160 	 * Configure whether the {@link StompBrokerRelayMessageHandler} should start
161 	 * automatically when the Spring ApplicationContext is refreshed.
162 	 * <p>The default setting is {@code true}.
163 	 */
164 	public StompBrokerRelayRegistration setAutoStartup(boolean autoStartup) {
165 		this.autoStartup = autoStartup;
166 		return this;
167 	}
168 
169 
170 	protected StompBrokerRelayMessageHandler getMessageHandler(SubscribableChannel brokerChannel) {
171 		StompBrokerRelayMessageHandler handler = new StompBrokerRelayMessageHandler(getClientInboundChannel(),
172 				getClientOutboundChannel(), brokerChannel, getDestinationPrefixes());
173 
174 		handler.setRelayHost(this.relayHost);
175 		handler.setRelayPort(this.relayPort);
176 
177 		handler.setClientLogin(this.clientLogin);
178 		handler.setClientPasscode(this.clientPasscode);
179 
180 		handler.setSystemLogin(this.systemLogin);
181 		handler.setSystemPasscode(this.systemPasscode);
182 
183 		if (this.systemHeartbeatSendInterval != null) {
184 			handler.setSystemHeartbeatSendInterval(this.systemHeartbeatSendInterval);
185 		}
186 		if (this.systemHeartbeatReceiveInterval != null) {
187 			handler.setSystemHeartbeatReceiveInterval(this.systemHeartbeatReceiveInterval);
188 		}
189 		if (this.virtualHost != null) {
190 			handler.setVirtualHost(this.virtualHost);
191 		}
192 
193 		handler.setAutoStartup(this.autoStartup);
194 
195 		return handler;
196 	}
197 
198 }
