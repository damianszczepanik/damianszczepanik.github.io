128 	private void stopActiveMqBrokerAndAwait() throws Exception {
129 		logger.debug("Stopping ActiveMQ broker and will await shutdown");
130 		if (!this.activeMQBroker.isStarted()) {
131 			logger.debug("Broker not running");
132 			return;
133 		}
134 		final CountDownLatch latch = new CountDownLatch(1);
135 		this.activeMQBroker.addShutdownHook(new Runnable() {
136 			public void run() {
137 				latch.countDown();
138 			}
139 		});
140 		this.activeMQBroker.stop();
141 		assertTrue("Broker did not stop", latch.await(5, TimeUnit.SECONDS));
142 		logger.debug("Broker stopped");
143 	}

273 	private static class TestMessageHandler implements MessageHandler {
274 
275 		private final BlockingQueue<Message<?>> queue = new LinkedBlockingQueue<>();
276 
277 		@Override
278 		public void handleMessage(Message<?> message) throws MessagingException {
279 			if (SimpMessageType.HEARTBEAT == SimpMessageHeaderAccessor.getMessageType(message.getHeaders())) {
280 				return;
281 			}
282 			this.queue.add(message);
283 		}
284 
285 		public void expectMessages(MessageExchange... messageExchanges) throws InterruptedException {
286 
287 			List<MessageExchange> expectedMessages =
288 					new ArrayList<MessageExchange>(Arrays.<MessageExchange>asList(messageExchanges));
289 
290 			while (expectedMessages.size() > 0) {
291 				Message<?> message = this.queue.poll(10000, TimeUnit.MILLISECONDS);
292 				assertNotNull("Timed out waiting for messages, expected [" + expectedMessages + "]", message);
293 
294 				MessageExchange match = findMatch(expectedMessages, message);
295 				assertNotNull("Unexpected message=" + message + ", expected [" + expectedMessages + "]", match);
296 
297 				expectedMessages.remove(match);
298 			}
299 		}
300 
301 		private MessageExchange findMatch(List<MessageExchange> expectedMessages, Message<?> message) {
302 			for (MessageExchange exchange : expectedMessages) {
303 				if (exchange.matchMessage(message)) {
304 					return exchange;
305 				}
306 			}
307 			return null;
308 		}
309 	}

311 	/**
312 	 * Holds a message as well as expected and actual messages matched against expectations.
313 	 */
314 	private static class MessageExchange {
315 
316 		private final Message<?> message;
317 
318 		private final MessageMatcher[] expected;
319 
320 		private final Message<?>[] actual;
321 
322 		public MessageExchange(Message<?> message, MessageMatcher... expected) {
323 			this.message = message;
324 			this.expected = expected;
325 			this.actual = new Message<?>[expected.length];
326 		}
327 
328 		public boolean matchMessage(Message<?> message) {
329 			for (int i=0 ; i < this.expected.length; i++) {
330 				if (this.expected[i].match(message)) {
331 					this.actual[i] = message;
332 					return true;
333 				}
334 			}
335 			return false;
336 		}
337 
338 		@Override
339 		public String toString() {
340 			return "Forwarded message:\n" + this.message + "\n" +
341 					"Should receive back:\n" + Arrays.toString(this.expected) + "\n" +
342 					"Actually received:\n" + Arrays.toString(this.actual) + "\n";
343 		}
344 	}

365 			StompHeaderAccessor headers = StompHeaderAccessor.create(StompCommand.CONNECT);

372 			builder.expected.add(new StompConnectedFrameMessageMatcher(sessionId));
373 			return builder;
374 		}

421 		public MessageExchangeBuilder andExpectMessage(String sessionId, String subscriptionId) {
422 			Assert.isTrue(SimpMessageType.MESSAGE.equals(headers.getMessageType()));
423 			String destination = this.headers.getDestination();
424 			Object payload = this.message.getPayload();

436 			this.expected.add(new StompFrameMessageMatcher(StompCommand.ERROR, sessionId));

438 		}

441 			return new MessageExchange(this.message, this.expected.toArray(new MessageMatcher[this.expected.size()]));
442 		}

445 	private static interface MessageMatcher {
446 
447 		boolean match(Message<?> message);
448 
449 	}

451 	private static class StompFrameMessageMatcher implements MessageMatcher {
452 
453 		private final StompCommand command;
454 
455 		private final String sessionId;
456 
457 
458 		public StompFrameMessageMatcher(StompCommand command, String sessionId) {
459 			this.command = command;
460 			this.sessionId = sessionId;
461 		}
462 
463 
464 		@Override
465 		public final boolean match(Message<?> message) {
466 			StompHeaderAccessor headers = StompHeaderAccessor.wrap(message);
467 			if (!this.command.equals(headers.getCommand()) || (this.sessionId != headers.getSessionId())) {
468 				return false;
469 			}
470 			return matchInternal(headers, message.getPayload());
471 		}
472 
473 		protected boolean matchInternal(StompHeaderAccessor headers, Object payload) {
474 			return true;
475 		}
476 
477 		@Override
478 		public String toString() {
479 			return "command=" + this.command  + ", session=\"" + this.sessionId + "\"";
480 		}
481 	}

483 	private static class StompReceiptFrameMessageMatcher extends StompFrameMessageMatcher {
484 
485 		private final String receiptId;
486 
487 		public StompReceiptFrameMessageMatcher(String sessionId, String receipt) {
488 			super(StompCommand.RECEIPT, sessionId);
489 			this.receiptId = receipt;
490 		}
491 
492 		@Override
493 		protected boolean matchInternal(StompHeaderAccessor headers, Object payload) {
494 			return (this.receiptId.equals(headers.getReceiptId()));
495 		}
496 
497 		@Override
498 		public String toString() {
499 			return super.toString() + ", receiptId=\"" + this.receiptId + "\"";
500 		}
501 	}

503 	private static class StompMessageFrameMessageMatcher extends StompFrameMessageMatcher {
504 
505 		private final String subscriptionId;
506 
507 		private final String destination;
508 
509 		private final Object payload;
510 
511 
512 		public StompMessageFrameMessageMatcher(String sessionId, String subscriptionId, String destination, Object payload) {
513 			super(StompCommand.MESSAGE, sessionId);
514 			this.subscriptionId = subscriptionId;
515 			this.destination = destination;
516 			this.payload = payload;
517 		}
518 
519 		@Override
520 		protected boolean matchInternal(StompHeaderAccessor headers, Object payload) {
521 			if (!this.subscriptionId.equals(headers.getSubscriptionId()) ||  !this.destination.equals(headers.getDestination())) {
522 				return false;
523 			}
524 			if (payload instanceof byte[] && this.payload instanceof byte[]) {
525 				return Arrays.equals((byte[]) payload, (byte[]) this.payload);
526 			}
527 			else {
528 				return this.payload.equals(payload);
529 			}
530 		}
531 
532 		@Override
533 		public String toString() {
534 			return super.toString() + ", subscriptionId=\"" + this.subscriptionId
535 					+ "\", destination=\"" + this.destination + "\", payload=\"" + getPayloadAsText() + "\"";
536 		}
537 
538 		protected String getPayloadAsText() {
539 			return (this.payload instanceof byte[])
540 					? new String((byte[]) this.payload, UTF_8) : payload.toString();
541 		}
542 	}
