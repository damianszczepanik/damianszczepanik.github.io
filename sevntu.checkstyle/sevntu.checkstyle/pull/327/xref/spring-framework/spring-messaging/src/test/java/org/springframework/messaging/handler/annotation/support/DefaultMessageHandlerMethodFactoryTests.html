64  	@Test
65  	public void customConversion() throws Exception {
66  		DefaultMessageHandlerMethodFactory instance = createInstance();
67  		GenericConversionService conversionService = new GenericConversionService();
68  		conversionService.addConverter(SampleBean.class, String.class, new Converter<SampleBean, String>() {
69  			@Override
70  			public String convert(SampleBean source) {
71  				return "foo bar";
72  			}
73  		});
74  		instance.setConversionService(conversionService);
75  		instance.afterPropertiesSet();
76  
77  		InvocableHandlerMethod invocableHandlerMethod =
78  				createInvocableHandlerMethod(instance, "simpleString", String.class);
79  
80  		invocableHandlerMethod.invoke(MessageBuilder.withPayload(sample).build());
81  		assertMethodInvocation(sample, "simpleString");
82  	}

129 	@Test
130 	public void overrideArgumentResolvers() throws Exception {
131 		DefaultMessageHandlerMethodFactory instance = createInstance();
132 		List<HandlerMethodArgumentResolver> customResolvers = new ArrayList<HandlerMethodArgumentResolver>();
133 		customResolvers.add(new CustomHandlerMethodArgumentResolver());
134 		instance.setArgumentResolvers(customResolvers);
135 		instance.afterPropertiesSet();
136 
137 		Message<String> message = MessageBuilder.withPayload("sample").build();
138 
139 		// This will work as the local resolver is set
140 		InvocableHandlerMethod invocableHandlerMethod =
141 				createInvocableHandlerMethod(instance, "customArgumentResolver", Locale.class);
142 		invocableHandlerMethod.invoke(message);
143 		assertMethodInvocation(sample, "customArgumentResolver");
144 
145 		// This won't work as no resolver is known for the payload
146 		InvocableHandlerMethod invocableHandlerMethod2 =
147 				createInvocableHandlerMethod(instance, "simpleString", String.class);
148 
149 		thrown.expect(IllegalStateException.class);
150 		thrown.expectMessage("No suitable resolver for");
151 		invocableHandlerMethod2.invoke(message);
152 	}

165 	@Test
166 	public void customValidation() throws Exception {
167 		DefaultMessageHandlerMethodFactory instance = createInstance();
168 		instance.setValidator(new Validator() {
169 			@Override
170 			public boolean supports(Class<?> clazz) {
171 				return String.class.isAssignableFrom(clazz);
172 			}
173 			@Override
174 			public void validate(Object target, Errors errors) {
175 				String value = (String) target;
176 				if ("failure".equals(value)) {
177 					errors.reject("not a valid value");
178 				}
179 			}
180 		});
181 		instance.afterPropertiesSet();
182 
183 		InvocableHandlerMethod invocableHandlerMethod =
184 				createInvocableHandlerMethod(instance, "payloadValidation", String.class);
185 		thrown.expect(MethodArgumentNotValidException.class);
186 		invocableHandlerMethod.invoke(MessageBuilder.withPayload("failure").build());
187 	}

190 	private void assertMethodInvocation(SampleBean bean, String methodName) {
191 		assertTrue("Method " + methodName + " should have been invoked", bean.invocations.get(methodName));
192 	}
193 
194 	private InvocableHandlerMethod createInvocableHandlerMethod(
195 			DefaultMessageHandlerMethodFactory factory, String methodName, Class<?>... parameterTypes) {
196 		return factory.createInvocableHandlerMethod(sample, getListenerMethod(methodName, parameterTypes));
197 	}
198 
199 	private DefaultMessageHandlerMethodFactory createInstance() {
200 		DefaultMessageHandlerMethodFactory factory = new DefaultMessageHandlerMethodFactory();
201 		factory.setBeanFactory(new StaticListableBeanFactory());
202 		return factory;
203 	}
204 
205 	private Method getListenerMethod(String methodName, Class<?>... parameterTypes) {
206 		Method method = ReflectionUtils.findMethod(SampleBean.class, methodName, parameterTypes);
207 		assertNotNull("no method found with name " + methodName + " and parameters " + Arrays.toString(parameterTypes));
208 		return method;
209 	}

212 	static class SampleBean {
213 
214 		private final Map<String, Boolean> invocations = new HashMap<String, Boolean>();
215 
216 		public void simpleString(String value) {
217 			invocations.put("simpleString", true);
218 		}
219 
220 		public void payloadValidation(@Payload @Valid String value) {
221 			invocations.put("payloadValidation", true);
222 		}
223 
224 		public void customArgumentResolver(Locale locale) {
225 			invocations.put("customArgumentResolver", true);
226 			assertEquals("Wrong value for locale", Locale.getDefault(), locale);
227 		}
228 	}

231 	static class CustomHandlerMethodArgumentResolver implements HandlerMethodArgumentResolver {
232 
233 		@Override
234 		public boolean supportsParameter(MethodParameter parameter) {
235 			return parameter.getParameterType().isAssignableFrom(Locale.class);
236 		}
237 
238 		@Override
239 		public Object resolveArgument(MethodParameter parameter, Message<?> message) throws Exception {
240 			return Locale.getDefault();
241 		}
