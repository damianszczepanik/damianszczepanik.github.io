104 	@Test
105 	public void createWithConnectNativeHeaders() {
106 
107 		MultiValueMap<String, String> extHeaders = new LinkedMultiValueMap<>();
108 		extHeaders.add(StompHeaderAccessor.STOMP_LOGIN_HEADER, "joe");
109 		extHeaders.add(StompHeaderAccessor.STOMP_PASSCODE_HEADER, "joe123");
110 
111 		StompHeaderAccessor headerAccessor = StompHeaderAccessor.create(StompCommand.CONNECT, extHeaders);
112 
113 		assertEquals(StompCommand.CONNECT, headerAccessor.getCommand());
114 		assertEquals(SimpMessageType.CONNECT, headerAccessor.getMessageType());
115 		assertNotNull(headerAccessor.getHeader("stompCredentials"));
116 		assertEquals("joe", headerAccessor.getLogin());
117 		assertEquals("joe123", headerAccessor.getPasscode());
118 		assertThat(headerAccessor.toString(), CoreMatchers.containsString("passcode=[PROTECTED]"));
119 
120 		Map<String, List<String>> output = headerAccessor.toNativeHeaderMap();
121 		assertEquals("joe", output.get(StompHeaderAccessor.STOMP_LOGIN_HEADER).get(0));
122 		assertEquals("PROTECTED", output.get(StompHeaderAccessor.STOMP_PASSCODE_HEADER).get(0));
123 	}

152 	public void toNativeHeadersMessageFrame() {
153 
154 		StompHeaderAccessor headers = StompHeaderAccessor.create(StompCommand.MESSAGE);
155 		headers.setSubscriptionId("s1");
156 		headers.setDestination("/d");
157 		headers.setContentType(MimeTypeUtils.APPLICATION_JSON);
158 		headers.updateStompCommandAsServerMessage();
159 
160 		Map<String, List<String>> actual = headers.toNativeHeaderMap();
161 
162 		assertEquals(actual.toString(), 4, actual.size());
163 		assertEquals("s1", actual.get(StompHeaderAccessor.STOMP_SUBSCRIPTION_HEADER).get(0));
164 		assertEquals("/d", actual.get(StompHeaderAccessor.STOMP_DESTINATION_HEADER).get(0));
165 		assertEquals("application/json", actual.get(StompHeaderAccessor.STOMP_CONTENT_TYPE_HEADER).get(0));
166 		assertNotNull("message-id was not created", actual.get(StompHeaderAccessor.STOMP_MESSAGE_ID_HEADER).get(0));
167 	}

198 
199 		MultiValueMap<String, String> extHeaders = new LinkedMultiValueMap<>();
200 		extHeaders.add(StompHeaderAccessor.STOMP_ID_HEADER, "s1");
201 		extHeaders.add(StompHeaderAccessor.STOMP_DESTINATION_HEADER, "/d");
202 		extHeaders.add("accountId", "ABC123");
203 
204 		StompHeaderAccessor headers = StompHeaderAccessor.create(StompCommand.SUBSCRIBE, extHeaders);
205 		String accountId = headers.getFirstNativeHeader("accountId");
206 		headers.setNativeHeader("accountId", accountId.toLowerCase());
207 
208 		Map<String, List<String>> actual = headers.toNativeHeaderMap();
209 		assertEquals(3, actual.size());
210 
211 		assertEquals("s1", actual.get(StompHeaderAccessor.STOMP_ID_HEADER).get(0));
212 		assertEquals("/d", actual.get(StompHeaderAccessor.STOMP_DESTINATION_HEADER).get(0));
213 		assertNotNull("abc123", actual.get("accountId").get(0));
214 	}

246 	@Test
247 	public void getShortLogMessage() {
248 		StompHeaderAccessor accessor = StompHeaderAccessor.create(StompCommand.SEND);
249 		accessor.setDestination("/foo");
250 		accessor.setContentType(MimeTypeUtils.APPLICATION_JSON);
251 		accessor.setSessionId("123");
252 		String actual = accessor.getShortLogMessage("payload".getBytes(Charset.forName("UTF-8")));
253 		assertEquals("SEND /foo session=123 application/json payload=payload", actual);
254 
255 		StringBuilder sb = new StringBuilder();
256 		for (int i = 0; i < 80; i++) {
257 			sb.append("a");
258 		}
259 		final String payload = sb.toString() + " > 80";
260 		actual = accessor.getShortLogMessage(payload.getBytes(UTF_8));
261 		assertEquals("SEND /foo session=123 application/json payload=" + sb + "...(truncated)", actual);
262 	}
