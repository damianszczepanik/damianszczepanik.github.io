55  /**
56   * Provides essential configuration for handling messages with simple messaging
57   * protocols such as STOMP.
58   *
59   * <p>{@link #clientInboundChannel()} and {@link #clientOutboundChannel()} deliver
60   * messages to and from remote clients to several message handlers such as
61   * <ul>
62   * <li>{@link #simpAnnotationMethodMessageHandler()}</li>
63   * <li>{@link #simpleBrokerMessageHandler()}</li>
64   * <li>{@link #stompBrokerRelayMessageHandler()}</li>
65   * <li>{@link #userDestinationMessageHandler()}</li>
66   * </ul>
67   * while {@link #brokerChannel()} delivers messages from within the application to the
68   * the respective message handlers. {@link #brokerMessagingTemplate()} can be injected
69   * into any application component to send messages.
70   *
71   * <p>Subclasses are responsible for the part of the configuration that feed messages
72   * to and from the client inbound/outbound channels (e.g. STOMP over WebSocket).
73   *
74   * @author Rossen Stoyanchev
75   * @author Brian Clozel
76   * @since 4.0
77   */
78  public abstract class AbstractMessageBrokerConfiguration implements ApplicationContextAware {

112 	@Bean

114 		ExecutorSubscribableChannel channel = new ExecutorSubscribableChannel(clientInboundChannelExecutor());

116 		channel.setInterceptors(reg.getInterceptors());

122 		TaskExecutorRegistration reg = getClientInboundChannelRegistration().getOrCreateTaskExecRegistration();
123 		ThreadPoolTaskExecutor executor = reg.getTaskExecutor();
124 		executor.setThreadNamePrefix("clientInboundChannel-");
125 		return executor;
126 	}

128 	protected final ChannelRegistration getClientInboundChannelRegistration() {
129 		if (this.clientInboundChannelRegistration == null) {
130 			ChannelRegistration registration = new ChannelRegistration();
131 			configureClientInboundChannel(registration);
132 			registration.setInterceptors(new ImmutableMessageChannelInterceptor());
133 			this.clientInboundChannelRegistration = registration;
134 		}
135 		return this.clientInboundChannelRegistration;
136 	}

178 	@Bean
179 	public AbstractSubscribableChannel brokerChannel() {
180 		ChannelRegistration reg = getBrokerRegistry().getBrokerChannelRegistration();
181 		ExecutorSubscribableChannel channel = reg.hasTaskExecutor() ?
182 				new ExecutorSubscribableChannel(brokerChannelExecutor()) : new ExecutorSubscribableChannel();
183 		reg.setInterceptors(new ImmutableMessageChannelInterceptor());
184 		channel.setInterceptors(reg.getInterceptors());
185 		return channel;
186 	}
187 
188 	@Bean
189 	public ThreadPoolTaskExecutor brokerChannelExecutor() {
190 		ChannelRegistration reg = getBrokerRegistry().getBrokerChannelRegistration();
191 		ThreadPoolTaskExecutor executor;
192 		if (reg.hasTaskExecutor()) {
193 			executor = reg.taskExecutor().getTaskExecutor();
194 		}
195 		else {
196 			// Should never be used
197 			executor = new ThreadPoolTaskExecutor();
198 			executor.setCorePoolSize(0);
199 			executor.setMaxPoolSize(1);
200 			executor.setQueueCapacity(0);
201 		}
202 		executor.setThreadNamePrefix("brokerChannel-");
203 		return executor;
204 	}

226 	@Bean
227 	public SimpAnnotationMethodMessageHandler simpAnnotationMethodMessageHandler() {
228 		SimpAnnotationMethodMessageHandler handler = new SimpAnnotationMethodMessageHandler(
229 				clientInboundChannel(), clientOutboundChannel(), brokerMessagingTemplate());
230 
231 		handler.setDestinationPrefixes(getBrokerRegistry().getApplicationDestinationPrefixes());
232 		handler.setMessageConverter(brokerMessageConverter());
233 		handler.setValidator(simpValidator());
234 
235 		List<HandlerMethodArgumentResolver> argumentResolvers = new ArrayList<HandlerMethodArgumentResolver>();
236 		addArgumentResolvers(argumentResolvers);
237 		handler.setCustomArgumentResolvers(argumentResolvers);
238 
239 		List<HandlerMethodReturnValueHandler> returnValueHandlers = new ArrayList<HandlerMethodReturnValueHandler>();
240 		addReturnValueHandlers(returnValueHandlers);
241 		handler.setCustomReturnValueHandlers(returnValueHandlers);
242 
243 		PathMatcher pathMatcher = this.getBrokerRegistry().getPathMatcher();
244 		if (pathMatcher != null) {
245 			handler.setPathMatcher(pathMatcher);
246 		}
247 		return handler;
248 	}

284 	@Bean
285 	public CompositeMessageConverter brokerMessageConverter() {
286 		List<MessageConverter> converters = new ArrayList<MessageConverter>();
287 		boolean registerDefaults = configureMessageConverters(converters);
288 		if (registerDefaults) {
289 			converters.add(new StringMessageConverter());
290 			converters.add(new ByteArrayMessageConverter());
291 			if (jackson2Present) {
292 				converters.add(createJacksonConverter());
293 			}
294 		}
295 		return new CompositeMessageConverter(converters);
296 	}

298 	protected MappingJackson2MessageConverter createJacksonConverter() {
299 		DefaultContentTypeResolver resolver = new DefaultContentTypeResolver();
300 		resolver.setDefaultMimeType(MimeTypeUtils.APPLICATION_JSON);
301 		MappingJackson2MessageConverter converter = new MappingJackson2MessageConverter();
302 		converter.setContentTypeResolver(resolver);
303 		return converter;
304 	}

331 	/**
332 	 * Return a {@link org.springframework.validation.Validator}s instance for validating
333 	 * {@code @Payload} method arguments.
334 	 * <p>In order, this method tries to get a Validator instance:
335 	 * <ul>
336 	 * <li>delegating to getValidator() first</li>
337 	 * <li>if none returned, getting an existing instance with its well-known name "mvcValidator",
338 	 * created by an MVC configuration</li>
339 	 * <li>if none returned, checking the classpath for the presence of a JSR-303 implementation
340 	 * before creating a {@code OptionalValidatorFactoryBean}</li>
341 	 * <li>returning a no-op Validator instance</li>
342 	 * </ul>
343 	 */
344 	protected Validator simpValidator() {
345 		Validator validator = getValidator();
346 		if (validator == null) {
347 			if (this.applicationContext.containsBean(MVC_VALIDATOR_NAME)) {
348 				validator = this.applicationContext.getBean(MVC_VALIDATOR_NAME, Validator.class);
349 			}
350 			else if (ClassUtils.isPresent("javax.validation.Validator", getClass().getClassLoader())) {
351 				Class<?> clazz;
352 				try {
353 					String className = "org.springframework.validation.beanvalidation.OptionalValidatorFactoryBean";
354 					clazz = ClassUtils.forName(className, AbstractMessageBrokerConfiguration.class.getClassLoader());
355 				}
356 				catch (Throwable ex) {
357 					throw new BeanInitializationException("Could not find default validator class", ex);
358 				}
359 				validator = (Validator) BeanUtils.instantiate(clazz);
360 			}
361 			else {
362 				validator = new Validator() {
363 					@Override
364 					public boolean supports(Class<?> clazz) {
365 						return false;
366 					}
367 					@Override
368 					public void validate(Object target, Errors errors) {
369 					}
370 				};
371 			}
372 		}
373 		return validator;
374 	}
