51  /**
52   * This class is only used within tests in the spring-orm module.
53   *
54   * <p>Convenient support class for JPA-related tests. Offers the same contract as
55   * AbstractTransactionalDataSourceSpringContextTests and equally good performance,
56   * even when performing the instrumentation required by the JPA specification.
57   *
58   * <p>Exposes an EntityManagerFactory and a shared EntityManager.
59   * Requires an EntityManagerFactory to be injected, plus the DataSource and
60   * JpaTransactionManager through the superclass.
61   *
62   * <p>When using Xerces, make sure a post 2.0.2 version is available on the classpath
63   * to avoid a critical
64   * <a href="http://nagoya.apache.org/bugzilla/show_bug.cgi?id=16014"/>bug</a>
65   * that leads to StackOverflow. Maven users are likely to encounter this problem since
66   * 2.0.2 is used by default.
67   *
68   * <p>A workaround is to explicitly specify the Xerces version inside the Maven POM:
69   * <pre>
70   * &lt;dependency&gt;
71   *   &lt;groupId&gt;xerces&lt;/groupId&gt;
72   *     &lt;artifactId&gt;xercesImpl&lt;/artifactId&gt;
73   *   &lt;version&gt;2.8.1&lt;/version&gt;
74   * &lt;/dependency&gt;
75   * </pre>
76   *
77   * @author Rod Johnson
78   * @author Rob Harrop
79   * @author Juergen Hoeller
80   * @since 2.0
81   * @deprecated as of Spring 3.0, in favor of using the listener-based test context framework
82   * ({@link org.springframework.test.context.junit38.AbstractJUnit38SpringContextTests})
83   */

134 	 * Subclasses should override this method if they wish to disable shadow class loading.

142 	@Override
143 	public void setDirty() {
144 		super.setDirty();
145 		contextCache.remove(cacheKeys());
146 		classLoaderCache.remove(cacheKeys());
147 
148 		// If we are a shadow loader, we need to invoke
149 		// the shadow parent to set it dirty, as
150 		// it is the shadow parent that maintains the cache state,
151 		// not the child
152 		if (this.shadowParent != null) {
153 			try {
154 				Method m = shadowParent.getClass().getMethod("setDirty", (Class[]) null);
155 				m.invoke(shadowParent, (Object[]) null);
156 			}
157 			catch (Exception ex) {
158 				throw new RuntimeException(ex);
159 			}
160 		}
161 	}

164 	@SuppressWarnings({ "rawtypes", "unchecked" })
165 	@Override
166 	public void runBare() throws Throwable {
167 
168 		// getName will return the name of the method being run.
169 		if (isDisabledInThisEnvironment(getName())) {
170 			// Let superclass log that we didn't run the test.
171 			super.runBare();
172 			return;
173 		}
174 
175 		if (!shouldUseShadowLoader()) {
176 			super.runBare();
177 			return;
178 		}
179 
180 		String combinationOfContextLocationsForThisTestClass = cacheKeys();
181 		ClassLoader classLoaderForThisTestClass = getClass().getClassLoader();
182 		// save the TCCL
183 		ClassLoader initialClassLoader = Thread.currentThread().getContextClassLoader();
184 
185 		if (this.shadowParent != null) {
186 			Thread.currentThread().setContextClassLoader(classLoaderForThisTestClass);
187 			super.runBare();
188 		}
189 
190 		else {
191 			ShadowingClassLoader shadowingClassLoader = (ShadowingClassLoader) classLoaderCache.get(combinationOfContextLocationsForThisTestClass);
192 
193 			if (shadowingClassLoader == null) {
194 				shadowingClassLoader = (ShadowingClassLoader) createShadowingClassLoader(classLoaderForThisTestClass);
195 				classLoaderCache.put(combinationOfContextLocationsForThisTestClass, shadowingClassLoader);
196 			}
197 			try {
198 				Thread.currentThread().setContextClassLoader(shadowingClassLoader);
199 				String[] configLocations = getConfigLocations();
200 
201 				// Do not strongly type, to avoid ClassCastException.
202 				Object cachedContext = contextCache.get(combinationOfContextLocationsForThisTestClass);
203 
204 				if (cachedContext == null) {
205 
206 					// Create the LoadTimeWeaver.
207 					Class shadowingLoadTimeWeaverClass = shadowingClassLoader.loadClass(ShadowingLoadTimeWeaver.class.getName());
208 					Constructor constructor = shadowingLoadTimeWeaverClass.getConstructor(ClassLoader.class);
209 					constructor.setAccessible(true);
210 					Object ltw = constructor.newInstance(shadowingClassLoader);
211 
212 					// Create the BeanFactory.
213 					Class beanFactoryClass = shadowingClassLoader.loadClass(DefaultListableBeanFactory.class.getName());
214 					Object beanFactory = BeanUtils.instantiateClass(beanFactoryClass);
215 
216 					// Create the BeanDefinitionReader.
217 					Class beanDefinitionReaderClass = shadowingClassLoader.loadClass(XmlBeanDefinitionReader.class.getName());
218 					Class beanDefinitionRegistryClass = shadowingClassLoader.loadClass(BeanDefinitionRegistry.class.getName());
219 					Object reader = beanDefinitionReaderClass.getConstructor(beanDefinitionRegistryClass).newInstance(beanFactory);
220 
221 					// Load the bean definitions into the BeanFactory.
222 					Method loadBeanDefinitions = beanDefinitionReaderClass.getMethod("loadBeanDefinitions", String[].class);
223 					loadBeanDefinitions.invoke(reader, new Object[] {configLocations});
224 
225 					// Create LoadTimeWeaver-injecting BeanPostProcessor.
226 					Class loadTimeWeaverInjectingBeanPostProcessorClass = shadowingClassLoader.loadClass(LoadTimeWeaverInjectingBeanPostProcessor.class.getName());
227 					Class loadTimeWeaverClass = shadowingClassLoader.loadClass(LoadTimeWeaver.class.getName());
228 					Constructor bppConstructor = loadTimeWeaverInjectingBeanPostProcessorClass.getConstructor(loadTimeWeaverClass);
229 					bppConstructor.setAccessible(true);
230 					Object beanPostProcessor = bppConstructor.newInstance(ltw);
231 
232 					// Add LoadTimeWeaver-injecting BeanPostProcessor.
233 					Class beanPostProcessorClass = shadowingClassLoader.loadClass(BeanPostProcessor.class.getName());
234 					Method addBeanPostProcessor = beanFactoryClass.getMethod("addBeanPostProcessor", beanPostProcessorClass);
235 					addBeanPostProcessor.invoke(beanFactory, beanPostProcessor);
236 
237 					// Create the GenericApplicationContext.
238 					Class genericApplicationContextClass = shadowingClassLoader.loadClass(GenericApplicationContext.class.getName());
239 					Class defaultListableBeanFactoryClass = shadowingClassLoader.loadClass(DefaultListableBeanFactory.class.getName());
240 					cachedContext = genericApplicationContextClass.getConstructor(defaultListableBeanFactoryClass).newInstance(beanFactory);
241 
242 					// Invoke the context's "refresh" method.
243 					genericApplicationContextClass.getMethod("refresh").invoke(cachedContext);
244 
245 					// Store the context reference in the cache.
246 					contextCache.put(combinationOfContextLocationsForThisTestClass, cachedContext);
247 				}
248 				// create the shadowed test
249 				Class shadowedTestClass = shadowingClassLoader.loadClass(getClass().getName());
250 
251 				// So long as JUnit is excluded from shadowing we
252 				// can minimize reflective invocation here
253 				TestCase shadowedTestCase = (TestCase) BeanUtils.instantiateClass(shadowedTestClass);
254 
255 				/* shadowParent = this */
256 				Class thisShadowedClass = shadowingClassLoader.loadClass(AbstractJpaTests.class.getName());
257 				Field shadowed = thisShadowedClass.getDeclaredField("shadowParent");
258 				shadowed.setAccessible(true);
259 				shadowed.set(shadowedTestCase, this);
260 
261 				/* AbstractSpringContextTests.addContext(Object, ApplicationContext) */
262 				Class applicationContextClass = shadowingClassLoader.loadClass(ConfigurableApplicationContext.class.getName());
263 				Method addContextMethod = shadowedTestClass.getMethod("addContext", Object.class, applicationContextClass);
264 				addContextMethod.invoke(shadowedTestCase, configLocations, cachedContext);
265 
266 				// Invoke tests on shadowed test case
267 				shadowedTestCase.setName(getName());
268 				shadowedTestCase.runBare();
269 			}
270 			catch (InvocationTargetException ex) {
271 				// Unwrap this for better exception reporting
272 				// when running tests
273 				throw ex.getTargetException();
274 			}
275 			finally {
276 				Thread.currentThread().setContextClassLoader(initialClassLoader);
277 			}
278 		}
279 	}

281 	protected String cacheKeys() {
282 		return StringUtils.arrayToCommaDelimitedString(getConfigLocations());

286 	 * NB: This method must <b>not</b> have a return type of ShadowingClassLoader as that would cause that
287 	 * class to be loaded eagerly when this test case loads, creating verify errors at runtime.
288 	 */
289 	protected ClassLoader createShadowingClassLoader(ClassLoader classLoader) {

291 				getActualOrmXmlLocation());
292 		customizeResourceOverridingShadowingClassLoader(orxl);
293 		return orxl;

317 	private static class LoadTimeWeaverInjectingBeanPostProcessor extends InstantiationAwareBeanPostProcessorAdapter {
318 
319 		private final LoadTimeWeaver ltw;
320 
321 		@SuppressWarnings("unused")
322 		public LoadTimeWeaverInjectingBeanPostProcessor(LoadTimeWeaver ltw) {
323 			this.ltw = ltw;
324 		}
325 
326 		@Override
327 		public Object postProcessBeforeInitialization(Object bean, String beanName) throws BeansException {
328 			if (bean instanceof LocalContainerEntityManagerFactoryBean) {
329 				((LocalContainerEntityManagerFactoryBean) bean).setLoadTimeWeaver(this.ltw);
330 			}
331 			if (bean instanceof DefaultPersistenceUnitManager) {
332 				((DefaultPersistenceUnitManager) bean).setLoadTimeWeaver(this.ltw);
333 			}
334 			return bean;
335 		}
336 	}

339 	private static class ShadowingLoadTimeWeaver implements LoadTimeWeaver {
340 
341 		private final ClassLoader shadowingClassLoader;
342 
343 		@SuppressWarnings("unused")
344 		public ShadowingLoadTimeWeaver(ClassLoader shadowingClassLoader) {
345 			this.shadowingClassLoader = shadowingClassLoader;
346 		}
347 
348 		@Override
349 		public void addTransformer(ClassFileTransformer transformer) {
350 			try {
351 				Method addClassFileTransformer =
352 						this.shadowingClassLoader.getClass().getMethod("addTransformer", ClassFileTransformer.class);
353 				addClassFileTransformer.setAccessible(true);
354 				addClassFileTransformer.invoke(this.shadowingClassLoader, transformer);
355 			}
356 			catch (Exception ex) {
357 				throw new RuntimeException(ex);
358 			}
359 		}
360 
361 		@Override
362 		public ClassLoader getInstrumentableClassLoader() {
363 			return this.shadowingClassLoader;
364 		}
365 
366 		@Override
367 		public ClassLoader getThrowawayClassLoader() {
368 			// Be sure to copy the same resource overrides and same class file transformers:
369 			// We want the throwaway class loader to behave like the instrumentable class loader.
370 			ResourceOverridingShadowingClassLoader roscl =
371 					new ResourceOverridingShadowingClassLoader(getClass().getClassLoader());
372 			if (this.shadowingClassLoader instanceof ShadowingClassLoader) {
373 				roscl.copyTransformers((ShadowingClassLoader) this.shadowingClassLoader);
374 			}
375 			if (this.shadowingClassLoader instanceof ResourceOverridingShadowingClassLoader) {
376 				roscl.copyOverrides((ResourceOverridingShadowingClassLoader) this.shadowingClassLoader);
377 			}
378 			return roscl;
379 		}
