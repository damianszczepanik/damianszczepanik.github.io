68  		assertNull("No dialect set", cefb.getJpaDialect());
69  
70  		RuntimeException in1 = new RuntimeException("in1");
71  		PersistenceException in2 = new PersistenceException();
72  		assertNull("No translation here", cefb.translateExceptionIfPossible(in1));

74  		assertNotNull(dex);
75  		assertSame(in2, dex.getCause());

115 	@Test
116 	public void testApplicationManagedEntityManagerWithTransaction() throws Exception {
117 		Object testEntity = new Object();
118 
119 		EntityTransaction mockTx = mock(EntityTransaction.class);
120 
121 		// This one's for the tx (shared)
122 		EntityManager sharedEm = mock(EntityManager.class);
123 		given(sharedEm.getTransaction()).willReturn(new NoOpEntityTransaction());
124 
125 		// This is the application-specific one
126 		EntityManager mockEm = mock(EntityManager.class);
127 		given(mockEm.getTransaction()).willReturn(mockTx);
128 
129 		given(mockEmf.createEntityManager()).willReturn(sharedEm, mockEm);
130 
131 		LocalContainerEntityManagerFactoryBean cefb = parseValidPersistenceUnit();
132 
133 		JpaTransactionManager jpatm = new JpaTransactionManager();
134 		jpatm.setEntityManagerFactory(cefb.getObject());
135 
136 		TransactionStatus txStatus = jpatm.getTransaction(new DefaultTransactionAttribute());
137 
138 		EntityManagerFactory emf = cefb.getObject();
139 		assertSame("EntityManagerFactory reference must be cached after init", emf, cefb.getObject());
140 
141 		assertNotSame("EMF must be proxied", mockEmf, emf);
142 		EntityManager em = emf.createEntityManager();
143 		em.joinTransaction();
144 		assertFalse(em.contains(testEntity));
145 
146 		jpatm.commit(txStatus);
147 
148 		cefb.destroy();
149 
150 		verify(mockTx).begin();
151 		verify(mockTx).commit();
152 		verify(mockEm).contains(testEntity);
153 		verify(mockEmf).close();
154 	}

172 
173 		LocalContainerEntityManagerFactoryBean cefb = parseValidPersistenceUnit();
174 
175 		JpaTransactionManager jpatm = new JpaTransactionManager();
176 		jpatm.setEntityManagerFactory(cefb.getObject());
177 
178 		TransactionStatus txStatus = jpatm.getTransaction(new DefaultTransactionAttribute());
179 
180 		EntityManagerFactory emf = cefb.getObject();

260 	protected LocalContainerEntityManagerFactoryBean createEntityManagerFactoryBean(
261 			String persistenceXml, Properties props, String entityManagerName) throws Exception {
262 
263 		// This will be set by DummyPersistenceProvider
264 		actualPui = null;
265 		actualProps = null;
266 
267 		LocalContainerEntityManagerFactoryBean containerEmfb = new LocalContainerEntityManagerFactoryBean();
268 
269 		containerEmfb.setPersistenceUnitName(entityManagerName);
270 		containerEmfb.setPersistenceProviderClass(DummyContainerPersistenceProvider.class);
271 		if (props != null) {
272 			containerEmfb.setJpaProperties(props);
273 		}
274 		containerEmfb.setLoadTimeWeaver(new InstrumentationLoadTimeWeaver());
275 		containerEmfb.setPersistenceXmlLocation(persistenceXml);
276 		containerEmfb.afterPropertiesSet();
277 
278 		assertEquals(entityManagerName, actualPui.getPersistenceUnitName());
279 		if (props != null) {
280 			assertEquals(props, actualProps);
281 		}
282 		//checkInvariants(containerEmfb);
283 
284 		return containerEmfb;
285
