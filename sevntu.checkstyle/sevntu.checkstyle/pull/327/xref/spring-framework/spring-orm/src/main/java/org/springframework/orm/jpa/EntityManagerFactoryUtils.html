91  			Class<Enum> synchronizationTypeClass = (Class<Enum>) ClassUtils.forName(

104 	/**
105 	 * Find an EntityManagerFactory with the given name in the given
106 	 * Spring application context (represented as ListableBeanFactory).
107 	 * <p>The specified unit name will be matched against the configured
108 	 * persistence unit, provided that a discovered EntityManagerFactory
109 	 * implements the {@link EntityManagerFactoryInfo} interface. If not,
110 	 * the persistence unit name will be matched against the Spring bean name,
111 	 * assuming that the EntityManagerFactory bean names follow that convention.
112 	 * <p>If no unit name has been given, this method will search for a default
113 	 * EntityManagerFactory through {@link ListableBeanFactory#getBean(Class)}.
114 	 * @param beanFactory the ListableBeanFactory to search
115 	 * @param unitName the name of the persistence unit (may be {@code null} or empty,
116 	 * in which case a single bean of type EntityManagerFactory will be searched for)
117 	 * @return the EntityManagerFactory
118 	 * @throws NoSuchBeanDefinitionException if there is no such EntityManagerFactory in the context
119 	 * @see EntityManagerFactoryInfo#getPersistenceUnitName()
120 	 */
121 	public static EntityManagerFactory findEntityManagerFactory(
122 			ListableBeanFactory beanFactory, String unitName) throws NoSuchBeanDefinitionException {
123 
124 		Assert.notNull(beanFactory, "ListableBeanFactory must not be null");
125 		if (StringUtils.hasLength(unitName)) {
126 			// See whether we can find an EntityManagerFactory with matching persistence unit name.
127 			String[] candidateNames =
128 					BeanFactoryUtils.beanNamesForTypeIncludingAncestors(beanFactory, EntityManagerFactory.class);
129 			for (String candidateName : candidateNames) {
130 				EntityManagerFactory emf = (EntityManagerFactory) beanFactory.getBean(candidateName);
131 				if (emf instanceof EntityManagerFactoryInfo) {
132 					if (unitName.equals(((EntityManagerFactoryInfo) emf).getPersistenceUnitName())) {
133 						return emf;
134 					}
135 				}
136 			}
137 			// No matching persistence unit found - simply take the EntityManagerFactory
138 			// with the persistence unit name as bean name (by convention).
139 			return beanFactory.getBean(unitName, EntityManagerFactory.class);
140 		}
141 		else {
142 			// Find unique EntityManagerFactory bean in the context, falling back to parent contexts.
143 			return beanFactory.getBean(EntityManagerFactory.class);
144 		}
145 	}

201 	/**
202 	 * Obtain a JPA EntityManager from the given factory. Is aware of a corresponding
203 	 * EntityManager bound to the current thread, e.g. when using JpaTransactionManager.
204 	 * <p>Same as {@code getEntityManager}, but throwing the original PersistenceException.
205 	 * @param emf EntityManagerFactory to create the EntityManager with
206 	 * @param properties the properties to be passed into the {@code createEntityManager}
207 	 * call (may be {@code null})
208 	 * @param synchronizedWithTransaction whether to automatically join ongoing
209 	 * transactions (according to the JPA 2.1 SynchronizationType rules)
210 	 * @return the EntityManager, or {@code null} if none found
211 	 * @throws javax.persistence.PersistenceException if the EntityManager couldn't be created
212 	 * @see #getTransactionalEntityManager(javax.persistence.EntityManagerFactory)
213 	 * @see JpaTransactionManager
214 	 */
215 	public static EntityManager doGetTransactionalEntityManager(
216 			EntityManagerFactory emf, Map<?, ?> properties, boolean synchronizedWithTransaction) throws PersistenceException {
217 
218 		Assert.notNull(emf, "No EntityManagerFactory specified");
219 
220 		EntityManagerHolder emHolder =
221 				(EntityManagerHolder) TransactionSynchronizationManager.getResource(emf);
222 		if (emHolder != null) {
223 			if (synchronizedWithTransaction) {
224 				if (!emHolder.isSynchronizedWithTransaction() &&
225 						TransactionSynchronizationManager.isSynchronizationActive()) {
226 					// Try to explicitly synchronize the EntityManager itself
227 					// with an ongoing JTA transaction, if any.
228 					try {
229 						emHolder.getEntityManager().joinTransaction();
230 					}
231 					catch (TransactionRequiredException ex) {
232 						logger.debug("Could not join transaction because none was actually active", ex);
233 					}
234 					Object transactionData = prepareTransaction(emHolder.getEntityManager(), emf);
235 					TransactionSynchronizationManager.registerSynchronization(
236 							new TransactionalEntityManagerSynchronization(emHolder, emf, transactionData, false));
237 					emHolder.setSynchronizedWithTransaction(true);
238 				}
239 				// Use holder's reference count to track synchronizedWithTransaction access.
240 				// isOpen() check used below to find out about it.
241 				emHolder.requested();
242 				return emHolder.getEntityManager();
243 			}
244 			else {
245 				// unsynchronized EntityManager demanded
246 				if (emHolder.isTransactionActive() && !emHolder.isOpen()) {
247 					if (!TransactionSynchronizationManager.isSynchronizationActive()) {
248 						return null;
249 					}
250 					// EntityManagerHolder with an active transaction coming from JpaTransactionManager,
251 					// with no synchronized EntityManager having been requested by application code before.
252 					// Unbind in order to register a new unsynchronized EntityManager instead.
253 					TransactionSynchronizationManager.unbindResource(emf);
254 				}
255 				else {
256 					// Either a previously bound unsynchronized EntityManager, or the application
257 					// has requested a synchronized EntityManager before and therefore upgraded
258 					// this transaction's EntityManager to synchronized before.
259 					return emHolder.getEntityManager();
260 				}
261 			}
262 		}
263 		else if (!TransactionSynchronizationManager.isSynchronizationActive()) {
264 			// Indicate that we can't obtain a transactional EntityManager.
265 			return null;
266 		}
267 
268 		// Create a new EntityManager for use within the current transaction.
269 		logger.debug("Opening JPA EntityManager");
270 		EntityManager em = null;
271 		if (!synchronizedWithTransaction && createEntityManagerWithSynchronizationTypeMethod != null) {
272 			try {
273 				em = (EntityManager) ReflectionUtils.invokeMethod(createEntityManagerWithSynchronizationTypeMethod,
274 						emf, synchronizationTypeUnsynchronized, properties);
275 			}
276 			catch (AbstractMethodError err) {
277 				// JPA 2.1 API available but method not actually implemented in persistence provider:
278 				// falling back to regular createEntityManager method.
279 			}
280 		}
281 		if (em == null) {
282 			em = (!CollectionUtils.isEmpty(properties) ? emf.createEntityManager(properties) : emf.createEntityManager());
283 		}
284 
285 		// Use same EntityManager for further JPA operations within the transaction.
286 		// Thread-bound object will get removed by synchronization at transaction completion.
287 		logger.debug("Registering transaction synchronization for JPA EntityManager");
288 		emHolder = new EntityManagerHolder(em);
289 		if (synchronizedWithTransaction) {
290 			Object transactionData = prepareTransaction(em, emf);
291 			TransactionSynchronizationManager.registerSynchronization(
292 					new TransactionalEntityManagerSynchronization(emHolder, emf, transactionData, true));
293 			emHolder.setSynchronizedWithTransaction(true);
294 		}
295 		else {
296 			// Unsynchronized - just scope it for the transaction, as demanded by the JPA 2.1 spec...
297 			TransactionSynchronizationManager.registerSynchronization(
298 					new TransactionScopedEntityManagerSynchronization(emHolder, emf));
299 		}
300 		TransactionSynchronizationManager.bindResource(emf, emHolder);
301 
302 		return em;
303 	}

305 	/**
306 	 * Prepare a transaction on the given EntityManager, if possible.
307 	 * @param em the EntityManager to prepare
308 	 * @param emf the EntityManagerFactory that the EntityManager has been created with
309 	 * @return an arbitrary object that holds transaction data, if any
310 	 * (to be passed into cleanupTransaction)
311 	 * @see JpaDialect#prepareTransaction
312 	 */
313 	private static Object prepareTransaction(EntityManager em, EntityManagerFactory emf) {
314 		if (emf instanceof EntityManagerFactoryInfo) {
315 			EntityManagerFactoryInfo emfInfo = (EntityManagerFactoryInfo) emf;
316 			JpaDialect jpaDialect = emfInfo.getJpaDialect();
317 			if (jpaDialect != null) {
318 				return jpaDialect.prepareTransaction(em,
319 						TransactionSynchronizationManager.isCurrentTransactionReadOnly(),
320 						TransactionSynchronizationManager.getCurrentTransactionName());
321 			}
322 		}
323 		return null;
324 	}

326 	/**
327 	 * Prepare a transaction on the given EntityManager, if possible.
328 	 * @param transactionData arbitrary object that holds transaction data, if any
329 	 * (as returned by prepareTransaction)
330 	 * @param emf the EntityManagerFactory that the EntityManager has been created with
331 	 * @see JpaDialect#cleanupTransaction
332 	 */
333 	private static void cleanupTransaction(Object transactionData, EntityManagerFactory emf) {
334 		if (emf instanceof EntityManagerFactoryInfo) {
335 			EntityManagerFactoryInfo emfInfo = (EntityManagerFactoryInfo) emf;
336 			JpaDialect jpaDialect = emfInfo.getJpaDialect();
337 			if (jpaDialect != null) {
338 				jpaDialect.cleanupTransaction(transactionData);
339 			}
340 		}
341 	}

344 	 * Apply the current transaction timeout, if any, to the given JPA Query object.
345 	 * <p>This method sets the JPA 2.0 query hint "javax.persistence.query.timeout" accordingly.
346 	 * @param query the JPA Query object
347 	 * @param emf JPA EntityManagerFactory that the Query was created for
348 	 */
349 	public static void applyTransactionTimeout(Query query, EntityManagerFactory emf) {
350 		EntityManagerHolder emHolder = (EntityManagerHolder) TransactionSynchronizationManager.getResource(emf);
351 		if (emHolder != null && emHolder.hasTimeout()) {
352 			int timeoutValue = (int) emHolder.getTimeToLiveInMillis();
353 			try {
354 				query.setHint("javax.persistence.query.timeout", timeoutValue);
355 			}
356 			catch (IllegalArgumentException ex) {
357 				// oh well, at least we tried...
358 			}
359 		}
360 	}

362 	/**
363 	 * Convert the given runtime exception to an appropriate exception from the
364 	 * {@code org.springframework.dao} hierarchy.
365 	 * Return null if no translation is appropriate: any other exception may
366 	 * have resulted from user code, and should not be translated.
367 	 * <p>The most important cases like object not found or optimistic locking failure
368 	 * are covered here. For more fine-granular conversion, JpaTransactionManager etc
369 	 * support sophisticated translation of exceptions via a JpaDialect.
370 	 * @param ex runtime exception that occurred
371 	 * @return the corresponding DataAccessException instance,
372 	 * or {@code null} if the exception should not be translated
373 	 */
374 	public static DataAccessException convertJpaAccessExceptionIfPossible(RuntimeException ex) {
375 		// Following the JPA specification, a persistence provider can also
376 		// throw these two exceptions, besides PersistenceException.
377 		if (ex instanceof IllegalStateException) {
378 			return new InvalidDataAccessApiUsageException(ex.getMessage(), ex);
379 		}
380 		if (ex instanceof IllegalArgumentException) {
381 			return new InvalidDataAccessApiUsageException(ex.getMessage(), ex);
382 		}
383 
384 		// Check for well-known PersistenceException subclasses.
385 		if (ex instanceof EntityNotFoundException) {
386 			return new JpaObjectRetrievalFailureException((EntityNotFoundException) ex);
387 		}
388 		if (ex instanceof NoResultException) {
389 			return new EmptyResultDataAccessException(ex.getMessage(), 1, ex);
390 		}
391 		if (ex instanceof NonUniqueResultException) {
392 			return new IncorrectResultSizeDataAccessException(ex.getMessage(), 1, ex);
393 		}
394 		if (ex instanceof QueryTimeoutException) {
395 			return new org.springframework.dao.QueryTimeoutException(ex.getMessage(), ex);
396 		}
397 		if (ex instanceof LockTimeoutException) {
398 			return new CannotAcquireLockException(ex.getMessage(), ex);
399 		}
400 		if (ex instanceof PessimisticLockException) {
401 			return new PessimisticLockingFailureException(ex.getMessage(), ex);
402 		}
403 		if (ex instanceof OptimisticLockException) {
404 			return new JpaOptimisticLockingFailureException((OptimisticLockException) ex);
405 		}
406 		if (ex instanceof EntityExistsException) {
407 			return new DataIntegrityViolationException(ex.getMessage(), ex);
408 		}
409 		if (ex instanceof TransactionRequiredException) {
410 			return new InvalidDataAccessApiUsageException(ex.getMessage(), ex);
411 		}
412 
413 		// If we have another kind of PersistenceException, throw it.
414 		if (ex instanceof PersistenceException) {
415 			return new JpaSystemException((PersistenceException) ex);
416 		}
417 
418 		// If we get here, we have an exception that resulted from user code,
419 		// rather than the persistence provider, so we return null to indicate
420 		// that translation should not occur.
421 		return null;
422 	}

424 	/**
425 	 * Close the given JPA EntityManager,
426 	 * catching and logging any cleanup exceptions thrown.
427 	 * @param em the JPA EntityManager to close (may be {@code null})
428 	 * @see javax.persistence.EntityManager#close()
429 	 */
430 	public static void closeEntityManager(EntityManager em) {
431 		if (em != null) {
432 			logger.debug("Closing JPA EntityManager");
433 			try {
434 				if (em.isOpen()) {
435 					em.close();
436 				}
437 			}
438 			catch (PersistenceException ex) {
439 				logger.debug("Could not close JPA EntityManager", ex);
440 			}
441 			catch (Throwable ex) {
442 				logger.debug("Unexpected exception on closing JPA EntityManager", ex);
443 			}
444 		}
445 	}

448 	/**
449 	 * Callback for resource cleanup at the end of a non-JPA transaction
450 	 * (e.g. when participating in a JtaTransactionManager transaction).
451 	 * @see org.springframework.transaction.jta.JtaTransactionManager
452 	 */
453 	private static class TransactionalEntityManagerSynchronization
454 			extends ResourceHolderSynchronization<EntityManagerHolder, EntityManagerFactory>
455 			implements Ordered {
456 
457 		private final Object transactionData;
458 
459 		private final JpaDialect jpaDialect;
460 
461 		private final boolean newEntityManager;
462 
463 		public TransactionalEntityManagerSynchronization(
464 				EntityManagerHolder emHolder, EntityManagerFactory emf, Object txData, boolean newEm) {
465 			super(emHolder, emf);
466 			this.transactionData = txData;
467 			this.jpaDialect = (emf instanceof EntityManagerFactoryInfo ?
468 					((EntityManagerFactoryInfo) emf).getJpaDialect() : null);
469 			this.newEntityManager = newEm;
470 		}
471 
472 		@Override
473 		public int getOrder() {
474 			return ENTITY_MANAGER_SYNCHRONIZATION_ORDER;
475 		}
476 
477 		@Override
478 		protected void flushResource(EntityManagerHolder resourceHolder) {
479 			EntityManager em = resourceHolder.getEntityManager();
480 			if (em instanceof EntityManagerProxy) {
481 				EntityManager target = ((EntityManagerProxy) em).getTargetEntityManager();
482 				if (TransactionSynchronizationManager.hasResource(target)) {
483 					// ExtendedEntityManagerSynchronization active after joinTransaction() call:
484 					// flush synchronization already registered.
485 					return;
486 				}
487 			}
488 			try {
489 				em.flush();
490 			}
491 			catch (RuntimeException ex) {
492 				if (this.jpaDialect != null) {
493 					throw this.jpaDialect.translateExceptionIfPossible(ex);
494 				}
495 				else {
496 					throw convertJpaAccessExceptionIfPossible(ex);
497 				}
498 			}
499 		}
500 
501 		@Override
502 		protected boolean shouldUnbindAtCompletion() {
503 			return this.newEntityManager;
504 		}
505 
506 		@Override
507 		protected void releaseResource(EntityManagerHolder resourceHolder, EntityManagerFactory resourceKey) {
508 			closeEntityManager(resourceHolder.getEntityManager());
509 		}
510 
511 		@Override
512 		protected void cleanupResource(EntityManagerHolder resourceHolder, EntityManagerFactory resourceKey, boolean committed) {
513 			if (!committed) {
514 				// Clear all pending inserts/updates/deletes in the EntityManager.
515 				// Necessary for pre-bound EntityManagers, to avoid inconsistent state.
516 				resourceHolder.getEntityManager().clear();
517 			}
518 			cleanupTransaction(this.transactionData, resourceKey);
519 		}
520 	}

523 	/**
524 	 * Minimal callback that just closes the EntityManager at the end of the transaction.
525 	 */
526 	private static class TransactionScopedEntityManagerSynchronization
527 			extends ResourceHolderSynchronization<EntityManagerHolder, EntityManagerFactory>
528 			implements Ordered {
529 
530 		public TransactionScopedEntityManagerSynchronization(EntityManagerHolder emHolder, EntityManagerFactory emf) {
531 			super(emHolder, emf);
532 		}
533 
534 		@Override
535 		public int getOrder() {
536 			return ENTITY_MANAGER_SYNCHRONIZATION_ORDER + 1;
537 		}
538 
539 		@Override
540 		protected void releaseResource(EntityManagerHolder resourceHolder, EntityManagerFactory resourceKey) {
541 			closeEntityManager(resourceHolder.getEntityManager());
542 		}
543 	}
