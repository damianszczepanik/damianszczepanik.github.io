54  /**
55   * {@link org.springframework.transaction.PlatformTransactionManager} implementation
56   * for a single JPA {@link javax.persistence.EntityManagerFactory}. Binds a JPA
57   * EntityManager from the specified factory to the thread, potentially allowing for
58   * one thread-bound EntityManager per factory. {@link SharedEntityManagerCreator} and
59   * {@code @PersistenceContext} are aware of thread-bound entity managers and participate
60   * in such transactions automatically. Using either is required for JPA access code
61   * supporting this transaction management mechanism.
62   *
63   * <p>This transaction manager is appropriate for applications that use a single
64   * JPA EntityManagerFactory for transactional data access. JTA (usually through
65   * {@link org.springframework.transaction.jta.JtaTransactionManager}) is necessary
66   * for accessing multiple transactional resources within the same transaction.
67   * Note that you need to configure your JPA provider accordingly in order to make
68   * it participate in JTA transactions.
69   *
70   * <p>This transaction manager also supports direct DataSource access within a
71   * transaction (i.e. plain JDBC code working with the same DataSource).
72   * This allows for mixing services which access JPA and services which use plain
73   * JDBC (without being aware of JPA)! Application code needs to stick to the
74   * same simple Connection lookup pattern as with
75   * {@link org.springframework.jdbc.datasource.DataSourceTransactionManager}
76   * (i.e. {@link org.springframework.jdbc.datasource.DataSourceUtils#getConnection}
77   * or going through a
78   * {@link org.springframework.jdbc.datasource.TransactionAwareDataSourceProxy}).
79   * Note that this requires a vendor-specific {@link JpaDialect} to be configured.
80   *
81   * <p>Note: To be able to register a DataSource's Connection for plain JDBC code,
82   * this instance needs to be aware of the DataSource ({@link #setDataSource}).
83   * The given DataSource should obviously match the one used by the given
84   * EntityManagerFactory. This transaction manager will autodetect the DataSource
85   * used as the connection factory of the EntityManagerFactory, so you usually
86   * don't need to explicitly specify the "dataSource" property.
87   *
88   * <p>This transaction manager supports nested transactions via JDBC 3.0 Savepoints.
89   * The {@link #setNestedTransactionAllowed "nestedTransactionAllowed"} flag defaults
90   * to {@code false} though, since nested transactions will just apply to the JDBC
91   * Connection, not to the JPA EntityManager and its cached entity objects and related
92   * context. You can manually set the flag to {@code true} if you want to use nested
93   * transactions for JDBC access code which participates in JPA transactions (provided
94   * that your JDBC driver supports Savepoints). <i>Note that JPA itself does not support
95   * nested transactions! Hence, do not expect JPA access code to semantically
96   * participate in a nested transaction.</i>
97   *
98   * @author Juergen Hoeller
99   * @since 2.0
100  * @see #setEntityManagerFactory
101  * @see #setDataSource
102  * @see LocalEntityManagerFactoryBean
103  * @see org.springframework.orm.jpa.support.SharedEntityManagerBean
104  * @see org.springframework.jdbc.datasource.DataSourceUtils#getConnection
105  * @see org.springframework.jdbc.datasource.DataSourceUtils#releaseConnection
106  * @see org.springframework.jdbc.core.JdbcTemplate
107  * @see org.springframework.jdbc.datasource.DataSourceTransactionManager
108  * @see org.springframework.transaction.jta.JtaTransactionManager
109  */

214 	/**
215 	 * Set the JDBC DataSource that this instance should manage transactions for.
216 	 * The DataSource should match the one used by the JPA EntityManagerFactory:
217 	 * for example, you could specify the same JNDI DataSource for both.
218 	 * <p>If the EntityManagerFactory uses a known DataSource as its connection factory,
219 	 * the DataSource will be autodetected: You can still explicitly specify the
220 	 * DataSource, but you don't need to in this case.
221 	 * <p>A transactional JDBC Connection for this DataSource will be provided to
222 	 * application code accessing this DataSource directly via DataSourceUtils
223 	 * or JdbcTemplate. The Connection will be taken from the JPA EntityManager.
224 	 * <p>Note that you need to use a JPA dialect for a specific JPA implementation
225 	 * to allow for exposing JPA transactions as JDBC transactions.
226 	 * <p>The DataSource specified here should be the target DataSource to manage
227 	 * transactions for, not a TransactionAwareDataSourceProxy. Only data access
228 	 * code may work with TransactionAwareDataSourceProxy, while the transaction
229 	 * manager needs to work on the underlying target DataSource. If there's
230 	 * nevertheless a TransactionAwareDataSourceProxy passed in, it will be
231 	 * unwrapped to extract its target DataSource.
232 	 * @see EntityManagerFactoryInfo#getDataSource()
233 	 * @see #setJpaDialect
234 	 * @see org.springframework.jdbc.datasource.TransactionAwareDataSourceProxy
235 	 * @see org.springframework.jdbc.datasource.DataSourceUtils
236 	 * @see org.springframework.jdbc.core.JdbcTemplate
237 	 */
238 	public void setDataSource(DataSource dataSource) {
239 		if (dataSource instanceof TransactionAwareDataSourceProxy) {
240 			// If we got a TransactionAwareDataSourceProxy, we need to perform transactions
241 			// for its underlying target DataSource, else data access code won't see
242 			// properly exposed transactions (i.e. transactions for the target DataSource).
243 			this.dataSource = ((TransactionAwareDataSourceProxy) dataSource).getTargetDataSource();
244 		}
245 		else {
246 			this.dataSource = dataSource;
247 		}
248 	}

325 	@Override
326 	protected Object doGetTransaction() {
327 		JpaTransactionObject txObject = new JpaTransactionObject();
328 		txObject.setSavepointAllowed(isNestedTransactionAllowed());
329 
330 		EntityManagerHolder emHolder = (EntityManagerHolder)
331 				TransactionSynchronizationManager.getResource(getEntityManagerFactory());
332 		if (emHolder != null) {
333 			if (logger.isDebugEnabled()) {
334 				logger.debug("Found thread-bound EntityManager [" + emHolder.getEntityManager() +
335 						"] for JPA transaction");
336 			}
337 			txObject.setEntityManagerHolder(emHolder, false);
338 		}
339 
340 		if (getDataSource() != null) {
341 			ConnectionHolder conHolder = (ConnectionHolder)
342 					TransactionSynchronizationManager.getResource(getDataSource());
343 			txObject.setConnectionHolder(conHolder);
344 		}
345 
346 		return txObject;
347 	}

354 	@Override
355 	protected void doBegin(Object transaction, TransactionDefinition definition) {
356 		JpaTransactionObject txObject = (JpaTransactionObject) transaction;
357 
358 		if (txObject.hasConnectionHolder() && !txObject.getConnectionHolder().isSynchronizedWithTransaction()) {
359 			throw new IllegalTransactionStateException(
360 					"Pre-bound JDBC Connection found! JpaTransactionManager does not support " +
361 					"running within DataSourceTransactionManager if told to manage the DataSource itself. " +
362 					"It is recommended to use a single JpaTransactionManager for all transactions " +
363 					"on a single DataSource, no matter whether JPA or JDBC access.");
364 		}
365 
366 		try {
367 			if (txObject.getEntityManagerHolder() == null ||
368 					txObject.getEntityManagerHolder().isSynchronizedWithTransaction()) {
369 				EntityManager newEm = createEntityManagerForTransaction();
370 				if (logger.isDebugEnabled()) {
371 					logger.debug("Opened new EntityManager [" + newEm + "] for JPA transaction");
372 				}
373 				txObject.setEntityManagerHolder(new EntityManagerHolder(newEm), true);
374 			}
375 
376 			EntityManager em = txObject.getEntityManagerHolder().getEntityManager();
377 
378 			// Delegate to JpaDialect for actual transaction begin.
379 			final int timeoutToUse = determineTimeout(definition);
380 			Object transactionData = getJpaDialect().beginTransaction(em,
381 					new DelegatingTransactionDefinition(definition) {
382 						@Override
383 						public int getTimeout() {
384 							return timeoutToUse;
385 						}
386 					});
387 			txObject.setTransactionData(transactionData);
388 
389 			// Register transaction timeout.
390 			if (timeoutToUse != TransactionDefinition.TIMEOUT_DEFAULT) {
391 				txObject.getEntityManagerHolder().setTimeoutInSeconds(timeoutToUse);
392 			}
393 
394 			// Register the JPA EntityManager's JDBC Connection for the DataSource, if set.
395 			if (getDataSource() != null) {
396 				ConnectionHandle conHandle = getJpaDialect().getJdbcConnection(em, definition.isReadOnly());
397 				if (conHandle != null) {
398 					ConnectionHolder conHolder = new ConnectionHolder(conHandle);
399 					if (timeoutToUse != TransactionDefinition.TIMEOUT_DEFAULT) {
400 						conHolder.setTimeoutInSeconds(timeoutToUse);
401 					}
402 					if (logger.isDebugEnabled()) {
403 						logger.debug("Exposing JPA transaction as JDBC transaction [" +
404 								conHolder.getConnectionHandle() + "]");
405 					}
406 					TransactionSynchronizationManager.bindResource(getDataSource(), conHolder);
407 					txObject.setConnectionHolder(conHolder);
408 				}
409 				else {
410 					if (logger.isDebugEnabled()) {
411 						logger.debug("Not exposing JPA transaction [" + em + "] as JDBC transaction because " +
412 								"JpaDialect [" + getJpaDialect() + "] does not support JDBC Connection retrieval");
413 					}
414 				}
415 			}
416 
417 			// Bind the entity manager holder to the thread.
418 			if (txObject.isNewEntityManagerHolder()) {
419 				TransactionSynchronizationManager.bindResource(
420 						getEntityManagerFactory(), txObject.getEntityManagerHolder());
421 			}
422 			txObject.getEntityManagerHolder().setSynchronizedWithTransaction(true);
423 		}
424 
425 		catch (TransactionException ex) {
426 			closeEntityManagerAfterFailedBegin(txObject);
427 			throw ex;
428 		}
429 		catch (Throwable ex) {
430 			closeEntityManagerAfterFailedBegin(txObject);
431 			throw new CannotCreateTransactionException("Could not open JPA EntityManager for transaction", ex);
432 		}
433 	}

435 	/**
436 	 * Create a JPA EntityManager to be used for a transaction.
437 	 * <p>The default implementation checks whether the EntityManagerFactory
438 	 * is a Spring proxy and unwraps it first.
439 	 * @see javax.persistence.EntityManagerFactory#createEntityManager()
440 	 * @see EntityManagerFactoryInfo#getNativeEntityManagerFactory()
441 	 */
442 	protected EntityManager createEntityManagerForTransaction() {
443 		EntityManagerFactory emf = getEntityManagerFactory();
444 		if (emf instanceof EntityManagerFactoryInfo) {
445 			emf = ((EntityManagerFactoryInfo) emf).getNativeEntityManagerFactory();
446 		}
447 		Map<String, Object> properties = getJpaPropertyMap();
448 		return (!CollectionUtils.isEmpty(properties) ?
449 				emf.createEntityManager(properties) : emf.createEntityManager());
450 	}

452 	/**
453 	 * Close the current transaction's EntityManager.
454 	 * Called after a transaction begin attempt failed.
455 	 * @param txObject the current transaction
456 	 */
457 	protected void closeEntityManagerAfterFailedBegin(JpaTransactionObject txObject) {
458 		if (txObject.isNewEntityManagerHolder()) {
459 			EntityManager em = txObject.getEntityManagerHolder().getEntityManager();
460 			try {
461 				if (em.getTransaction().isActive()) {
462 					em.getTransaction().rollback();
463 				}
464 			}
465 			catch (Throwable ex) {
466 				logger.debug("Could not rollback EntityManager after failed transaction begin", ex);
467 			}
468 			finally {
469 				EntityManagerFactoryUtils.closeEntityManager(em);
470 			}
471 			txObject.setEntityManagerHolder(null, false);
472 		}
473 	}

475 	@Override
476 	protected Object doSuspend(Object transaction) {
477 		JpaTransactionObject txObject = (JpaTransactionObject) transaction;
478 		txObject.setEntityManagerHolder(null, false);
479 		EntityManagerHolder entityManagerHolder = (EntityManagerHolder)
480 				TransactionSynchronizationManager.unbindResource(getEntityManagerFactory());
481 		txObject.setConnectionHolder(null);
482 		ConnectionHolder connectionHolder = null;
483 		if (getDataSource() != null && TransactionSynchronizationManager.hasResource(getDataSource())) {
484 			connectionHolder = (ConnectionHolder) TransactionSynchronizationManager.unbindResource(getDataSource());
485 		}
486 		return new SuspendedResourcesHolder(entityManagerHolder, connectionHolder);
487 	}

489 	@Override
490 	protected void doResume(Object transaction, Object suspendedResources) {
491 		SuspendedResourcesHolder resourcesHolder = (SuspendedResourcesHolder) suspendedResources;
492 		TransactionSynchronizationManager.bindResource(
493 				getEntityManagerFactory(), resourcesHolder.getEntityManagerHolder());
494 		if (getDataSource() != null && resourcesHolder.getConnectionHolder() != null) {
495 			TransactionSynchronizationManager.bindResource(getDataSource(), resourcesHolder.getConnectionHolder());
496 		}
497 	}

508 	@Override
509 	protected void doCommit(DefaultTransactionStatus status) {
510 		JpaTransactionObject txObject = (JpaTransactionObject) status.getTransaction();
511 		if (status.isDebug()) {
512 			logger.debug("Committing JPA transaction on EntityManager [" +
513 					txObject.getEntityManagerHolder().getEntityManager() + "]");
514 		}
515 		try {
516 			EntityTransaction tx = txObject.getEntityManagerHolder().getEntityManager().getTransaction();
517 			tx.commit();
518 		}
519 		catch (RollbackException ex) {
520 			if (ex.getCause() instanceof RuntimeException) {
521 				DataAccessException dex = getJpaDialect().translateExceptionIfPossible((RuntimeException) ex.getCause());
522 				if (dex != null) {
523 					throw dex;
524 				}
525 			}
526 			throw new TransactionSystemException("Could not commit JPA transaction", ex);
527 		}
528 		catch (RuntimeException ex) {
529 			// Assumably failed to flush changes to database.
530 			throw DataAccessUtils.translateIfNecessary(ex, getJpaDialect());
531 		}
532 	}
533 
534 	@Override
535 	protected void doRollback(DefaultTransactionStatus status) {
536 		JpaTransactionObject txObject = (JpaTransactionObject) status.getTransaction();
537 		if (status.isDebug()) {
538 			logger.debug("Rolling back JPA transaction on EntityManager [" +
539 					txObject.getEntityManagerHolder().getEntityManager() + "]");
540 		}
541 		try {
542 			EntityTransaction tx = txObject.getEntityManagerHolder().getEntityManager().getTransaction();
543 			if (tx.isActive()) {
544 				tx.rollback();
545 			}
546 		}
547 		catch (PersistenceException ex) {
548 			throw new TransactionSystemException("Could not roll back JPA transaction", ex);
549 		}
550 		finally {
551 			if (!txObject.isNewEntityManagerHolder()) {
552 				// Clear all pending inserts/updates/deletes in the EntityManager.
553 				// Necessary for pre-bound EntityManagers, to avoid inconsistent state.
554 				txObject.getEntityManagerHolder().getEntityManager().clear();
555 			}
556 		}
557 	}

559 	@Override
560 	protected void doSetRollbackOnly(DefaultTransactionStatus status) {
561 		JpaTransactionObject txObject = (JpaTransactionObject) status.getTransaction();
562 		if (status.isDebug()) {
563 			logger.debug("Setting JPA transaction on EntityManager [" +
564 					txObject.getEntityManagerHolder().getEntityManager() + "] rollback-only");

569 	@Override
570 	protected void doCleanupAfterCompletion(Object transaction) {
571 		JpaTransactionObject txObject = (JpaTransactionObject) transaction;
572 
573 		// Remove the entity manager holder from the thread, if still there.
574 		// (Could have been removed by EntityManagerFactoryUtils in order
575 		// to replace it with an unsynchronized EntityManager).
576 		if (txObject.isNewEntityManagerHolder()) {
577 			TransactionSynchronizationManager.unbindResourceIfPossible(getEntityManagerFactory());
578 		}
579 		txObject.getEntityManagerHolder().clear();
580 
581 		// Remove the JDBC connection holder from the thread, if exposed.
582 		if (txObject.hasConnectionHolder()) {
583 			TransactionSynchronizationManager.unbindResource(getDataSource());
584 			try {
585 				getJpaDialect().releaseJdbcConnection(txObject.getConnectionHolder().getConnectionHandle(),
586 						txObject.getEntityManagerHolder().getEntityManager());
587 			}
588 			catch (Exception ex) {
589 				// Just log it, to keep a transaction-related exception.
590 				logger.error("Could not close JDBC connection after transaction", ex);
591 			}
592 		}
593 
594 		getJpaDialect().cleanupTransaction(txObject.getTransactionData());
595 
596 		// Remove the entity manager holder from the thread.
597 		if (txObject.isNewEntityManagerHolder()) {
598 			EntityManager em = txObject.getEntityManagerHolder().getEntityManager();
599 			if (logger.isDebugEnabled()) {
600 				logger.debug("Closing JPA EntityManager [" + em + "] after transaction");
601 			}
602 			EntityManagerFactoryUtils.closeEntityManager(em);
603 		}
604 		else {
605 			logger.debug("Not closing pre-bound JPA EntityManager after transaction");
606 		}
607 	}

610 	/**
611 	 * JPA transaction object, representing a EntityManagerHolder.
612 	 * Used as transaction object by JpaTransactionManager.
613 	 */
614 	private class JpaTransactionObject extends JdbcTransactionObjectSupport {
615 
616 		private EntityManagerHolder entityManagerHolder;
617 
618 		private boolean newEntityManagerHolder;
619 
620 		private Object transactionData;
621 
622 		public void setEntityManagerHolder(
623 				EntityManagerHolder entityManagerHolder, boolean newEntityManagerHolder) {
624 			this.entityManagerHolder = entityManagerHolder;
625 			this.newEntityManagerHolder = newEntityManagerHolder;
626 		}
627 
628 		public EntityManagerHolder getEntityManagerHolder() {
629 			return this.entityManagerHolder;
630 		}
631 
632 		public boolean isNewEntityManagerHolder() {
633 			return this.newEntityManagerHolder;
634 		}
635 
636 		public boolean hasTransaction() {
637 			return (this.entityManagerHolder != null && this.entityManagerHolder.isTransactionActive());
638 		}
639 
640 		public void setTransactionData(Object transactionData) {
641 			this.transactionData = transactionData;
642 			this.entityManagerHolder.setTransactionActive(true);
643 			if (transactionData instanceof SavepointManager) {
644 				this.entityManagerHolder.setSavepointManager((SavepointManager) transactionData);
645 			}
646 		}
647 
648 		public Object getTransactionData() {
649 			return this.transactionData;
650 		}
651 
652 		public void setRollbackOnly() {
653 			EntityTransaction tx = this.entityManagerHolder.getEntityManager().getTransaction();
654 			if (tx.isActive()) {
655 				tx.setRollbackOnly();
656 			}
657 			if (hasConnectionHolder()) {
658 				getConnectionHolder().setRollbackOnly();
659 			}
660 		}
661 
662 		@Override
663 		public boolean isRollbackOnly() {
664 			EntityTransaction tx = this.entityManagerHolder.getEntityManager().getTransaction();
665 			return tx.getRollbackOnly();
666 		}
667 
668 		@Override
669 		public void flush() {
670 			try {
671 				this.entityManagerHolder.getEntityManager().flush();
672 			}
673 			catch (RuntimeException ex) {
674 				throw DataAccessUtils.translateIfNecessary(ex, getJpaDialect());
675 			}
676 		}
677 
678 		@Override
679 		public Object createSavepoint() throws TransactionException {
680 			return getSavepointManager().createSavepoint();
681 		}
682 
683 		@Override
684 		public void rollbackToSavepoint(Object savepoint) throws TransactionException {
685 			getSavepointManager().rollbackToSavepoint(savepoint);
686 		}
687 
688 		@Override
689 		public void releaseSavepoint(Object savepoint) throws TransactionException {
690 			getSavepointManager().releaseSavepoint(savepoint);
691 		}
692 
693 		private SavepointManager getSavepointManager() {
694 			if (!isSavepointAllowed()) {
695 				throw new NestedTransactionNotSupportedException(
696 						"Transaction manager does not allow nested transactions");
697 			}
698 			SavepointManager savepointManager = getEntityManagerHolder().getSavepointManager();
699 			if (savepointManager == null) {
700 				throw new NestedTransactionNotSupportedException(
701 						"JpaDialect does not support savepoints - check your JPA provider's capabilities");
702 			}
703 			return savepointManager;
704 		}
705 	}
