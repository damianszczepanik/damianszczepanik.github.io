58  /**
59   * {@link org.springframework.beans.factory.FactoryBean} that creates a
60   * Hibernate {@link org.hibernate.SessionFactory}. This is the usual way to
61   * set up a shared Hibernate SessionFactory in a Spring application context;
62   * the SessionFactory can then be passed to Hibernate-based DAOs via
63   * dependency injection.
64   *
65   * <p>Configuration settings can either be read from a Hibernate XML file,
66   * specified as "configLocation", or completely via this class. A typical
67   * local configuration consists of one or more "mappingResources", various
68   * "hibernateProperties" (not strictly necessary), and a "dataSource" that the
69   * SessionFactory should use. The latter can also be specified via Hibernate
70   * properties, but "dataSource" supports any Spring-configured DataSource,
71   * instead of relying on Hibernate's own connection providers.
72   *
73   * <p>This SessionFactory handling strategy is appropriate for most types of
74   * applications, from Hibernate-only single database apps to ones that need
75   * distributed transactions. Either {@link HibernateTransactionManager} or
76   * {@link org.springframework.transaction.jta.JtaTransactionManager} can be
77   * used for transaction demarcation, with the latter only necessary for
78   * transactions which span multiple databases.
79   *
80   * <p>This factory bean will by default expose a transaction-aware SessionFactory
81   * proxy, letting data access code work with the plain Hibernate SessionFactory
82   * and its {@code getCurrentSession()} method, while still being able to
83   * participate in current Spring-managed transactions: with any transaction
84   * management strategy, either local or JTA / EJB CMT, and any transaction
85   * synchronization mechanism, either Spring or JTA. Furthermore,
86   * {@code getCurrentSession()} will also seamlessly work with
87   * a request-scoped Session managed by
88   * {@link org.springframework.orm.hibernate3.support.OpenSessionInViewFilter} /
89   * {@link org.springframework.orm.hibernate3.support.OpenSessionInViewInterceptor}.
90   *
91   * <p>Requires Hibernate 3.6.x, as of Spring 4.0.
92   * Note that this factory will use "on_close" as default Hibernate connection
93   * release mode, unless in the case of a "jtaTransactionManager" specified,
94   * for the reason that this is appropriate for most Spring-based applications
95   * (in particular when using Spring's HibernateTransactionManager).
96   *
97   * @author Juergen Hoeller
98   * @since 1.2
99   * @see HibernateTemplate#setSessionFactory
100  * @see HibernateTransactionManager#setSessionFactory
101  * @see #setExposeTransactionAwareSessionFactory
102  * @see #setJtaTransactionManager
103  * @see org.hibernate.SessionFactory#getCurrentSession()
104  * @see HibernateTransactionManager
105  */
106 public class LocalSessionFactoryBean extends AbstractSessionFactoryBean implements BeanClassLoaderAware {

501 	@Override
502 	@SuppressWarnings("unchecked")
503 	protected SessionFactory buildSessionFactory() throws Exception {
504 		// Create Configuration instance.
505 		Configuration config = newConfiguration();
506 
507 		DataSource dataSource = getDataSource();
508 		if (dataSource != null) {
509 			// Make given DataSource available for SessionFactory configuration.
510 			configTimeDataSourceHolder.set(dataSource);
511 		}
512 		if (this.jtaTransactionManager != null) {
513 			// Make Spring-provided JTA TransactionManager available.
514 			configTimeTransactionManagerHolder.set(this.jtaTransactionManager);
515 		}
516 		if (this.cacheRegionFactory != null) {
517 			// Make Spring-provided Hibernate RegionFactory available.
518 			configTimeRegionFactoryHolder.set(this.cacheRegionFactory);
519 		}
520 		if (this.lobHandler != null) {
521 			// Make given LobHandler available for SessionFactory configuration.
522 			// Do early because because mapping resource might refer to custom types.
523 			configTimeLobHandlerHolder.set(this.lobHandler);
524 		}
525 
526 		// Analogous to Hibernate EntityManager's Ejb3Configuration:
527 		// Hibernate doesn't allow setting the bean ClassLoader explicitly,
528 		// so we need to expose it as thread context ClassLoader accordingly.
529 		Thread currentThread = Thread.currentThread();
530 		ClassLoader threadContextClassLoader = currentThread.getContextClassLoader();
531 		boolean overrideClassLoader =
532 				(this.beanClassLoader != null && !this.beanClassLoader.equals(threadContextClassLoader));
533 		if (overrideClassLoader) {
534 			currentThread.setContextClassLoader(this.beanClassLoader);
535 		}
536 
537 		try {
538 			if (isExposeTransactionAwareSessionFactory()) {
539 				// Set Hibernate 3.1+ CurrentSessionContext implementation,
540 				// providing the Spring-managed Session as current Session.
541 				// Can be overridden by a custom value for the corresponding Hibernate property.
542 				config.setProperty(
543 						Environment.CURRENT_SESSION_CONTEXT_CLASS, SpringSessionContext.class.getName());
544 			}
545 
546 			if (this.jtaTransactionManager != null) {
547 				// Set Spring-provided JTA TransactionManager as Hibernate property.
548 				config.setProperty(
549 						Environment.TRANSACTION_STRATEGY, JTATransactionFactory.class.getName());
550 				config.setProperty(
551 						Environment.TRANSACTION_MANAGER_STRATEGY, LocalTransactionManagerLookup.class.getName());
552 			}
553 			else {
554 				// Makes the Hibernate Session aware of the presence of a Spring-managed transaction.
555 				// Also sets connection release mode to ON_CLOSE by default.
556 				config.setProperty(
557 						Environment.TRANSACTION_STRATEGY, SpringTransactionFactory.class.getName());
558 			}
559 
560 			if (this.entityInterceptor != null) {
561 				// Set given entity interceptor at SessionFactory level.
562 				config.setInterceptor(this.entityInterceptor);
563 			}
564 
565 			if (this.namingStrategy != null) {
566 				// Pass given naming strategy to Hibernate Configuration.
567 				config.setNamingStrategy(this.namingStrategy);
568 			}
569 
570 			if (this.typeDefinitions != null) {
571 				// Register specified Hibernate type definitions.
572 				Mappings mappings = config.createMappings();
573 				for (TypeDefinitionBean typeDef : this.typeDefinitions) {
574 					mappings.addTypeDef(typeDef.getTypeName(), typeDef.getTypeClass(), typeDef.getParameters());
575 				}
576 			}
577 
578 			if (this.filterDefinitions != null) {
579 				// Register specified Hibernate FilterDefinitions.
580 				for (FilterDefinition filterDef : this.filterDefinitions) {
581 					config.addFilterDefinition(filterDef);
582 				}
583 			}
584 
585 			if (this.configLocations != null) {
586 				for (Resource resource : this.configLocations) {
587 					// Load Hibernate configuration from given location.
588 					config.configure(resource.getURL());
589 				}
590 			}
591 
592 			if (this.hibernateProperties != null) {
593 				// Add given Hibernate properties to Configuration.
594 				config.addProperties(this.hibernateProperties);
595 			}
596 
597 			if (dataSource != null) {
598 				Class<?> providerClass = LocalDataSourceConnectionProvider.class;
599 				if (isUseTransactionAwareDataSource() || dataSource instanceof TransactionAwareDataSourceProxy) {
600 					providerClass = TransactionAwareDataSourceConnectionProvider.class;
601 				}
602 				else if (config.getProperty(Environment.TRANSACTION_MANAGER_STRATEGY) != null) {
603 					providerClass = LocalJtaDataSourceConnectionProvider.class;
604 				}
605 				// Set Spring-provided DataSource as Hibernate ConnectionProvider.
606 				config.setProperty(Environment.CONNECTION_PROVIDER, providerClass.getName());
607 			}
608 
609 			if (this.cacheRegionFactory != null) {
610 				// Expose Spring-provided Hibernate RegionFactory.
611 				config.setProperty(Environment.CACHE_REGION_FACTORY, LocalRegionFactoryProxy.class.getName());
612 			}
613 
614 			if (this.mappingResources != null) {
615 				// Register given Hibernate mapping definitions, contained in resource files.
616 				for (String mapping : this.mappingResources) {
617 					Resource resource = new ClassPathResource(mapping.trim(), this.beanClassLoader);
618 					config.addInputStream(resource.getInputStream());
619 				}
620 			}
621 
622 			if (this.mappingLocations != null) {
623 				// Register given Hibernate mapping definitions, contained in resource files.
624 				for (Resource resource : this.mappingLocations) {
625 					config.addInputStream(resource.getInputStream());
626 				}
627 			}
628 
629 			if (this.cacheableMappingLocations != null) {
630 				// Register given cacheable Hibernate mapping definitions, read from the file system.
631 				for (Resource resource : this.cacheableMappingLocations) {
632 					config.addCacheableFile(resource.getFile());
633 				}
634 			}
635 
636 			if (this.mappingJarLocations != null) {
637 				// Register given Hibernate mapping definitions, contained in jar files.
638 				for (Resource resource : this.mappingJarLocations) {
639 					config.addJar(resource.getFile());
640 				}
641 			}
642 
643 			if (this.mappingDirectoryLocations != null) {
644 				// Register all Hibernate mapping definitions in the given directories.
645 				for (Resource resource : this.mappingDirectoryLocations) {
646 					File file = resource.getFile();
647 					if (!file.isDirectory()) {
648 						throw new IllegalArgumentException(
649 								"Mapping directory location [" + resource + "] does not denote a directory");
650 					}
651 					config.addDirectory(file);
652 				}
653 			}
654 
655 			// Tell Hibernate to eagerly compile the mappings that we registered,
656 			// for availability of the mapping information in further processing.
657 			postProcessMappings(config);
658 			config.buildMappings();
659 
660 			if (this.entityCacheStrategies != null) {
661 				// Register cache strategies for mapped entities.
662 				for (Enumeration<?> classNames = this.entityCacheStrategies.propertyNames(); classNames.hasMoreElements();) {
663 					String className = (String) classNames.nextElement();
664 					String[] strategyAndRegion =
665 							StringUtils.commaDelimitedListToStringArray(this.entityCacheStrategies.getProperty(className));
666 					if (strategyAndRegion.length > 1) {
667 						config.setCacheConcurrencyStrategy(className, strategyAndRegion[0], strategyAndRegion[1]);
668 					}
669 					else if (strategyAndRegion.length > 0) {
670 						config.setCacheConcurrencyStrategy(className, strategyAndRegion[0]);
671 					}
672 				}
673 			}
674 
675 			if (this.collectionCacheStrategies != null) {
676 				// Register cache strategies for mapped collections.
677 				for (Enumeration<?> collRoles = this.collectionCacheStrategies.propertyNames(); collRoles.hasMoreElements();) {
678 					String collRole = (String) collRoles.nextElement();
679 					String[] strategyAndRegion =
680 							StringUtils.commaDelimitedListToStringArray(this.collectionCacheStrategies.getProperty(collRole));
681 					if (strategyAndRegion.length > 1) {
682 						config.setCollectionCacheConcurrencyStrategy(collRole, strategyAndRegion[0], strategyAndRegion[1]);
683 					}
684 					else if (strategyAndRegion.length > 0) {
685 						config.setCollectionCacheConcurrencyStrategy(collRole, strategyAndRegion[0]);
686 					}
687 				}
688 			}
689 
690 			if (this.eventListeners != null) {
691 				// Register specified Hibernate event listeners.
692 				for (Map.Entry<String, Object> entry : this.eventListeners.entrySet()) {
693 					String listenerType = entry.getKey();
694 					Object listenerObject = entry.getValue();
695 					if (listenerObject instanceof Collection) {
696 						Collection<Object> listeners = (Collection<Object>) listenerObject;
697 						EventListeners listenerRegistry = config.getEventListeners();
698 						Object[] listenerArray =
699 								(Object[]) Array.newInstance(listenerRegistry.getListenerClassFor(listenerType), listeners.size());
700 						listenerArray = listeners.toArray(listenerArray);
701 						config.setListeners(listenerType, listenerArray);
702 					}
703 					else {
704 						config.setListener(listenerType, listenerObject);
705 					}
706 				}
707 			}
708 
709 			// Perform custom post-processing in subclasses.
710 			postProcessConfiguration(config);
711 
712 			// Build SessionFactory instance.
713 			logger.info("Building new Hibernate SessionFactory");
714 			this.configuration = config;
715 			return newSessionFactory(config);
716 		}
717 
718 		finally {
719 			if (dataSource != null) {
720 				configTimeDataSourceHolder.remove();
721 			}
722 			if (this.jtaTransactionManager != null) {
723 				configTimeTransactionManagerHolder.remove();
724 			}
725 			if (this.cacheRegionFactory != null) {
726 				configTimeRegionFactoryHolder.remove();
727 			}
728 			if (this.lobHandler != null) {
729 				configTimeLobHandlerHolder.remove();
730 			}
731 			if (overrideClassLoader) {
732 				// Reset original thread context ClassLoader.
733 				currentThread.setContextClassLoader(threadContextClassLoader);
734 			}
735 		}
736 	}

845 	 * used for creating the SessionFactory. A replacement for Hibernate's
846 	 * SchemaUpdate class, for automatically executing schema update scripts
847 	 * on application startup. Can also be invoked manually.
848 	 * <p>Fetch the LocalSessionFactoryBean itself rather than the exposed
849 	 * SessionFactory to be able to invoke this method, e.g. via

859 		logger.info("Updating database schema for Hibernate SessionFactory");
860 		DataSource dataSource = getDataSource();
861 		if (dataSource != null) {
862 			// Make given DataSource available for the schema update.
863 			configTimeDataSourceHolder.set(dataSource);
864 		}
865 		try {
866 			SessionFactory sessionFactory = getSessionFactory();
867 			final Dialect dialect = ((SessionFactoryImplementor) sessionFactory).getDialect();
868 			HibernateTemplate hibernateTemplate = new HibernateTemplate(sessionFactory);
869 			hibernateTemplate.setFlushMode(HibernateTemplate.FLUSH_NEVER);
870 			hibernateTemplate.execute(
871 				new HibernateCallback<Object>() {
872 					@Override
873 					public Object doInHibernate(Session session) throws HibernateException, SQLException {
874 						@SuppressWarnings("deprecation")
875 						Connection con = session.connection();
876 						DatabaseMetadata metadata = new DatabaseMetadata(con, dialect);
877 						String[] sql = getConfiguration().generateSchemaUpdateScript(dialect, metadata);
878 						executeSchemaScript(con, sql);
879 						return null;
880 					}
881 				}
882 			);
883 		}

1013 	 * Execute the given schema script on the given JDBC Connection.
1014 	 * <p>Note that the default implementation will log unsuccessful statements
1015 	 * and continue to execute. Override the {@code executeSchemaStatement}
1016 	 * method to treat failures differently.
1017 	 * @param con the JDBC Connection to execute the script on
1018 	 * @param sql the SQL statements to execute
1019 	 * @throws SQLException if thrown by JDBC methods
1020 	 * @see #executeSchemaStatement
1021 	 */
1022 	protected void executeSchemaScript(Connection con, String[] sql) throws SQLException {
1023 		if (sql != null && sql.length > 0) {
1024 			boolean oldAutoCommit = con.getAutoCommit();
1025 			if (!oldAutoCommit) {
1026 				con.setAutoCommit(true);
1027 			}
1028 			try {
1029 				Statement stmt = con.createStatement();
1030 				try {
1031 					for (String sqlStmt : sql) {
1032 						executeSchemaStatement(stmt, sqlStmt);
1033 					}
1034 				}
1035 				finally {
1036 					JdbcUtils.closeStatement(stmt);
1037 				}
1038 			}
1039 			finally {
1040 				if (!oldAutoCommit) {
1041 					con.setAutoCommit(false);
1042 				}
1043 			}
1044 		}
1045 	}

1052 	 * @param sql the SQL statement to execute
