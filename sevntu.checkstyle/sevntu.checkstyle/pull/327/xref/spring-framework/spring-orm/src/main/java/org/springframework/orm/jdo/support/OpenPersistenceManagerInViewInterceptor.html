17  package org.springframework.orm.jdo.support;
18  
19  import javax.jdo.PersistenceManager;
20  import javax.jdo.PersistenceManagerFactory;
21  
22  import org.apache.commons.logging.Log;
23  import org.apache.commons.logging.LogFactory;
24  
25  import org.springframework.dao.DataAccessException;
26  import org.springframework.orm.jdo.PersistenceManagerFactoryUtils;
27  import org.springframework.orm.jdo.PersistenceManagerHolder;
28  import org.springframework.transaction.support.TransactionSynchronizationManager;
29  import org.springframework.ui.ModelMap;
30  import org.springframework.web.context.request.WebRequest;
31  import org.springframework.web.context.request.WebRequestInterceptor;
32  
33  /**
34   * Spring web request interceptor that binds a JDO PersistenceManager to the
35   * thread for the entire processing of the request. Intended for the "Open
36   * PersistenceManager in View" pattern, i.e. to allow for lazy loading in
37   * web views despite the original transactions already being completed.
38   *
39   * <p>This interceptor makes JDO PersistenceManagers available via the current thread,
40   * which will be autodetected by transaction managers. It is suitable for service
41   * layer transactions via {@link org.springframework.orm.jdo.JdoTransactionManager}
42   * or {@link org.springframework.transaction.jta.JtaTransactionManager} as well
43   * as for non-transactional read-only execution.
44   *
45   * <p>In contrast to {@link OpenPersistenceManagerInViewFilter}, this interceptor
46   * is set up in a Spring application context and can thus take advantage of
47   * bean wiring.
48   *
49   * @author Juergen Hoeller
50   * @since 1.1
51   * @see OpenPersistenceManagerInViewFilter
52   * @see org.springframework.orm.jdo.JdoTransactionManager
53   * @see org.springframework.orm.jdo.PersistenceManagerFactoryUtils#getPersistenceManager
54   * @see org.springframework.transaction.support.TransactionSynchronizationManager
55   */
56  public class OpenPersistenceManagerInViewInterceptor implements WebRequestInterceptor {
57  
58  	/**
59  	 * Suffix that gets appended to the PersistenceManagerFactory toString
60  	 * representation for the "participate in existing persistence manager
61  	 * handling" request attribute.
62  	 * @see #getParticipateAttributeName
63  	 */
64  	public static final String PARTICIPATE_SUFFIX = ".PARTICIPATE";
65  
66  
67  	protected final Log logger = LogFactory.getLog(getClass());
68  
69  	private PersistenceManagerFactory persistenceManagerFactory;
70  
71  
72  	/**
73  	 * Set the JDO PersistenceManagerFactory that should be used to create
74  	 * PersistenceManagers.
75  	 */
76  	public void setPersistenceManagerFactory(PersistenceManagerFactory pmf) {
77  		this.persistenceManagerFactory = pmf;
78  	}
79  
80  	/**
81  	 * Return the JDO PersistenceManagerFactory that should be used to create
82  	 * PersistenceManagers.
83  	 */
84  	public PersistenceManagerFactory getPersistenceManagerFactory() {
85  		return persistenceManagerFactory;
86  	}
87  
88  
89  	@Override
90  	public void preHandle(WebRequest request) throws DataAccessException {
91  		if (TransactionSynchronizationManager.hasResource(getPersistenceManagerFactory())) {
92  			// Do not modify the PersistenceManager: just mark the request accordingly.
93  			String participateAttributeName = getParticipateAttributeName();
94  			Integer count = (Integer) request.getAttribute(participateAttributeName, WebRequest.SCOPE_REQUEST);
95  			int newCount = (count != null ? count + 1 : 1);
96  			request.setAttribute(getParticipateAttributeName(), newCount, WebRequest.SCOPE_REQUEST);
97  		}
98  		else {
99  			logger.debug("Opening JDO PersistenceManager in OpenPersistenceManagerInViewInterceptor");
100 			PersistenceManager pm =
101 					PersistenceManagerFactoryUtils.getPersistenceManager(getPersistenceManagerFactory(), true);
102 			TransactionSynchronizationManager.bindResource(
103 					getPersistenceManagerFactory(), new PersistenceManagerHolder(pm));
104 		}
105 	}
106 
107 	@Override
108 	public void postHandle(WebRequest request, ModelMap model) {
109 	}
110 
111 	@Override
112 	public void afterCompletion(WebRequest request, Exception ex) throws DataAccessException {
113 		String participateAttributeName = getParticipateAttributeName();
114 		Integer count = (Integer) request.getAttribute(participateAttributeName, WebRequest.SCOPE_REQUEST);
115 		if (count != null) {
116 			// Do not modify the PersistenceManager: just clear the marker.
117 			if (count > 1) {
118 				request.setAttribute(participateAttributeName, count - 1, WebRequest.SCOPE_REQUEST);
119 			}
120 			else {
121 				request.removeAttribute(participateAttributeName, WebRequest.SCOPE_REQUEST);
122 			}
123 		}
124 		else {
125 			PersistenceManagerHolder pmHolder = (PersistenceManagerHolder)
126 					TransactionSynchronizationManager.unbindResource(getPersistenceManagerFactory());
127 			logger.debug("Closing JDO PersistenceManager in OpenPersistenceManagerInViewInterceptor");
128 			PersistenceManagerFactoryUtils.releasePersistenceManager(
129 					pmHolder.getPersistenceManager(), getPersistenceManagerFactory());
130 		}
131 	}
132 
133 	/**
134 	 * Return the name of the request attribute that identifies that a request is
135 	 * already filtered. Default implementation takes the toString representation
136 	 * of the PersistenceManagerFactory instance and appends ".FILTERED".
137 	 * @see #PARTICIPATE_SUFFIX
138 	 */
139 	protected String getParticipateAttributeName() {
140 		return getPersistenceManagerFactory().toString() + PARTICIPATE_SUFFIX;
141 	}
142 
143 }
