39  /**
40   * Servlet 2.3 Filter that binds a JPA EntityManager to the thread for the
41   * entire processing of the request. Intended for the "Open EntityManager in
42   * View" pattern, i.e. to allow for lazy loading in web views despite the
43   * original transactions already being completed.
44   *
45   * <p>This filter makes JPA EntityManagers available via the current thread,
46   * which will be autodetected by transaction managers. It is suitable for service
47   * layer transactions via {@link org.springframework.orm.jpa.JpaTransactionManager}
48   * or {@link org.springframework.transaction.jta.JtaTransactionManager} as well
49   * as for non-transactional read-only execution.
50   *
51   * <p>Looks up the EntityManagerFactory in Spring's root web application context.
52   * Supports an "entityManagerFactoryBeanName" filter init-param in {@code web.xml};
53   * the default bean name is "entityManagerFactory". As an alternative, the
54   * "persistenceUnitName" init-param allows for retrieval by logical unit name
55   * (as specified in {@code persistence.xml}).
56   *
57   * @author Juergen Hoeller
58   * @since 2.0
59   * @see OpenEntityManagerInViewInterceptor
60   * @see org.springframework.orm.jpa.JpaTransactionManager
61   * @see org.springframework.orm.jpa.SharedEntityManagerCreator
62   * @see org.springframework.transaction.support.TransactionSynchronizationManager
63   */
64  public class OpenEntityManagerInViewFilter extends OncePerRequestFilter {
65  
66  	/**
67  	 * Default EntityManagerFactory bean name: "entityManagerFactory".
68  	 * Only applies when no "persistenceUnitName" param has been specified.
69  	 * @see #setEntityManagerFactoryBeanName
70  	 * @see #setPersistenceUnitName
71  	 */
72  	public static final String DEFAULT_ENTITY_MANAGER_FACTORY_BEAN_NAME = "entityManagerFactory";
73  
74  	private String entityManagerFactoryBeanName;
75  
76  	private String persistenceUnitName;
77  
78  	private volatile EntityManagerFactory entityManagerFactory;
79  
80  
81  	/**
82  	 * Set the bean name of the EntityManagerFactory to fetch from Spring's
83  	 * root application context.
84  	 * <p>Default is "entityManagerFactory". Note that this default only applies
85  	 * when no "persistenceUnitName" param has been specified.
86  	 * @see #setPersistenceUnitName
87  	 * @see #DEFAULT_ENTITY_MANAGER_FACTORY_BEAN_NAME
88  	 */
89  	public void setEntityManagerFactoryBeanName(String entityManagerFactoryBeanName) {
90  		this.entityManagerFactoryBeanName = entityManagerFactoryBeanName;
91  	}
92  
93  	/**
94  	 * Return the bean name of the EntityManagerFactory to fetch from Spring's
95  	 * root application context.
96  	 */
97  	protected String getEntityManagerFactoryBeanName() {
98  		return this.entityManagerFactoryBeanName;
99  	}
100 
101 	/**
102 	 * Set the name of the persistence unit to access the EntityManagerFactory for.
103 	 * <p>This is an alternative to specifying the EntityManagerFactory by bean name,
104 	 * resolving it by its persistence unit name instead. If no bean name and no persistence
105 	 * unit name have been specified, we'll check whether a bean exists for the default
106 	 * bean name "entityManagerFactory"; if not, a default EntityManagerFactory will
107 	 * be retrieved through finding a single unique bean of type EntityManagerFactory.
108 	 * @see #setEntityManagerFactoryBeanName
109 	 * @see #DEFAULT_ENTITY_MANAGER_FACTORY_BEAN_NAME
110 	 */
111 	public void setPersistenceUnitName(String persistenceUnitName) {
112 		this.persistenceUnitName = persistenceUnitName;
113 	}
114 
115 	/**
116 	 * Return the name of the persistence unit to access the EntityManagerFactory for, if any.
117 	 */
118 	protected String getPersistenceUnitName() {
119 		return this.persistenceUnitName;
120 	}
121 
122 	/**
123 	 * Returns "false" so that the filter may re-bind the opened
124 	 * {@code EntityManager} to each asynchronously dispatched thread and postpone
125 	 * closing it until the very last asynchronous dispatch.
126 	 */
127 	@Override
128 	protected boolean shouldNotFilterAsyncDispatch() {
129 		return false;
130 	}
131 
132 	/**
133 	 * Returns "false" so that the filter may provide an {@code EntityManager}
134 	 * to each error dispatches.
135 	 */
136 	@Override
137 	protected boolean shouldNotFilterErrorDispatch() {
138 		return false;
139 	}
140 
141 	@Override
142 	protected void doFilterInternal(
143 			HttpServletRequest request, HttpServletResponse response, FilterChain filterChain)
144 			throws ServletException, IOException {
145 
146 		EntityManagerFactory emf = lookupEntityManagerFactory(request);
147 		boolean participate = false;
148 
149 		WebAsyncManager asyncManager = WebAsyncUtils.getAsyncManager(request);
150 		String key = getAlreadyFilteredAttributeName();
151 
152 		if (TransactionSynchronizationManager.hasResource(emf)) {
153 			// Do not modify the EntityManager: just set the participate flag.
154 			participate = true;
155 		}
156 		else {
157 			boolean isFirstRequest = !isAsyncDispatch(request);
158 			if (isFirstRequest || !applyEntityManagerBindingInterceptor(asyncManager, key)) {
159 				logger.debug("Opening JPA EntityManager in OpenEntityManagerInViewFilter");
160 				try {
161 					EntityManager em = createEntityManager(emf);
162 					EntityManagerHolder emHolder = new EntityManagerHolder(em);
163 					TransactionSynchronizationManager.bindResource(emf, emHolder);
164 
165 					AsyncRequestInterceptor interceptor = new AsyncRequestInterceptor(emf, emHolder);
166 					asyncManager.registerCallableInterceptor(key, interceptor);
167 					asyncManager.registerDeferredResultInterceptor(key, interceptor);
168 				}
169 				catch (PersistenceException ex) {
170 					throw new DataAccessResourceFailureException("Could not create JPA EntityManager", ex);
171 				}
172 			}
173 		}
174 
175 		try {
176 			filterChain.doFilter(request, response);
177 		}
178 
179 		finally {
180 			if (!participate) {
181 				EntityManagerHolder emHolder = (EntityManagerHolder)
182 						TransactionSynchronizationManager.unbindResource(emf);
183 				if (!isAsyncStarted(request)) {
184 					logger.debug("Closing JPA EntityManager in OpenEntityManagerInViewFilter");
185 					EntityManagerFactoryUtils.closeEntityManager(emHolder.getEntityManager());
186 				}
187 			}
188 		}
189 	}
190 
191 	/**
192 	 * Look up the EntityManagerFactory that this filter should use,
193 	 * taking the current HTTP request as argument.
194 	 * <p>The default implementation delegates to the {@code lookupEntityManagerFactory}
195 	 * without arguments, caching the EntityManagerFactory reference once obtained.
196 	 * @return the EntityManagerFactory to use
197 	 * @see #lookupEntityManagerFactory()
198 	 */
199 	protected EntityManagerFactory lookupEntityManagerFactory(HttpServletRequest request) {
200 		if (this.entityManagerFactory == null) {
201 			this.entityManagerFactory = lookupEntityManagerFactory();
202 		}
203 		return this.entityManagerFactory;
204 	}
205 
206 	/**
207 	 * Look up the EntityManagerFactory that this filter should use.
208 	 * <p>The default implementation looks for a bean with the specified name
209 	 * in Spring's root application context.
210 	 * @return the EntityManagerFactory to use
211 	 * @see #getEntityManagerFactoryBeanName
212 	 */
213 	protected EntityManagerFactory lookupEntityManagerFactory() {
214 		WebApplicationContext wac = WebApplicationContextUtils.getRequiredWebApplicationContext(getServletContext());
215 		String emfBeanName = getEntityManagerFactoryBeanName();
216 		String puName = getPersistenceUnitName();
217 		if (StringUtils.hasLength(emfBeanName)) {
218 			return wac.getBean(emfBeanName, EntityManagerFactory.class);
219 		}
220 		else if (!StringUtils.hasLength(puName) && wac.containsBean(DEFAULT_ENTITY_MANAGER_FACTORY_BEAN_NAME)) {
221 			return wac.getBean(DEFAULT_ENTITY_MANAGER_FACTORY_BEAN_NAME, EntityManagerFactory.class);
222 		}
223 		else {
224 			// Includes fallback search for single EntityManagerFactory bean by type.
225 			return EntityManagerFactoryUtils.findEntityManagerFactory(wac, puName);
226 		}
227 	}
228 
229 	/**
230 	 * Create a JPA EntityManager to be bound to a request.
231 	 * <p>Can be overridden in subclasses.
232 	 * @param emf the EntityManagerFactory to use
233 	 * @see javax.persistence.EntityManagerFactory#createEntityManager()
234 	 */
235 	protected EntityManager createEntityManager(EntityManagerFactory emf) {
236 		return emf.createEntityManager();
237 	}
238 
239 	private boolean applyEntityManagerBindingInterceptor(WebAsyncManager asyncManager, String key) {
240 		if (asyncManager.getCallableInterceptor(key) == null) {
241 			return false;
242 		}
243 		((AsyncRequestInterceptor) asyncManager.getCallableInterceptor(key)).bindSession();
244 		return true;
245 	}
