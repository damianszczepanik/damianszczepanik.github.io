145 		asyncManager.setTaskExecutor(new SyncTaskExecutor());
146 		asyncManager.setAsyncWebRequest(asyncWebRequest);

148 			@Override

153

155 		assertFalse(TransactionSynchronizationManager.hasResource(factory));
156 
157 		// Async dispatch thread
158 
159 		interceptor.preHandle(this.webRequest);
160 		assertTrue(TransactionSynchronizationManager.hasResource(factory));
161 
162 		asyncManager.clearConcurrentResult();
163 
164 		// check that further invocations simply participate
165 		interceptor.preHandle(new ServletWebRequest(request));
166 
167 		interceptor.preHandle(new ServletWebRequest(request));
168 		interceptor.postHandle(new ServletWebRequest(request), null);
169 		interceptor.afterCompletion(new ServletWebRequest(request), null);
170 
171 		interceptor.postHandle(new ServletWebRequest(request), null);
172 		interceptor.afterCompletion(new ServletWebRequest(request), null);
173

175 		interceptor.postHandle(new ServletWebRequest(request), null);
176 		interceptor.afterCompletion(new ServletWebRequest(request), null);
177 
178 		interceptor.postHandle(this.webRequest, null);

254 
255 		final OpenEntityManagerInViewFilter filter = new OpenEntityManagerInViewFilter();
256 		filter.init(filterConfig);

258 		filter2.init(filterConfig2);
259 
260 		final FilterChain filterChain = new FilterChain() {
261 			@Override

292 	@Test
293 	public void testOpenEntityManagerInViewFilterAsyncScenario() throws Exception {
294 		given(manager.isOpen()).willReturn(true);
295 
296 		final EntityManagerFactory factory2 = mock(EntityManagerFactory.class);
297 		final EntityManager manager2 = mock(EntityManager.class);
298 
299 		given(factory2.createEntityManager()).willReturn(manager2);
300 		given(manager2.isOpen()).willReturn(true);
301 
302 		MockServletContext sc = new MockServletContext();
303 		StaticWebApplicationContext wac = new StaticWebApplicationContext();
304 		wac.setServletContext(sc);
305 		wac.getDefaultListableBeanFactory().registerSingleton("entityManagerFactory", factory);
306 		wac.getDefaultListableBeanFactory().registerSingleton("myEntityManagerFactory", factory2);
307 		wac.refresh();
308 		sc.setAttribute(WebApplicationContext.ROOT_WEB_APPLICATION_CONTEXT_ATTRIBUTE, wac);
309 
310 		MockFilterConfig filterConfig = new MockFilterConfig(wac.getServletContext(), "filter");
311 		MockFilterConfig filterConfig2 = new MockFilterConfig(wac.getServletContext(), "filter2");
312 		filterConfig2.addInitParameter("entityManagerFactoryBeanName", "myEntityManagerFactory");
313 
314 		final OpenEntityManagerInViewFilter filter = new OpenEntityManagerInViewFilter();
315 		filter.init(filterConfig);
316 		final OpenEntityManagerInViewFilter filter2 = new OpenEntityManagerInViewFilter();
317 		filter2.init(filterConfig2);
318 
319 		final AtomicInteger count = new AtomicInteger(0);
320 
321 		final FilterChain filterChain = new FilterChain() {
322 			@Override
323 			public void doFilter(ServletRequest servletRequest, ServletResponse servletResponse) {
324 				assertTrue(TransactionSynchronizationManager.hasResource(factory));
325 				servletRequest.setAttribute("invoked", Boolean.TRUE);
326 				count.incrementAndGet();
327 			}
328 		};
329 
330 		final AtomicInteger count2 = new AtomicInteger(0);
331 
332 		final FilterChain filterChain2 = new FilterChain() {
333 			@Override
334 			public void doFilter(ServletRequest servletRequest, ServletResponse servletResponse)
335 				throws IOException, ServletException {
336 				assertTrue(TransactionSynchronizationManager.hasResource(factory2));
337 				filter.doFilter(servletRequest, servletResponse, filterChain);
338 				count2.incrementAndGet();
339 			}
340 		};
341 
342 		FilterChain filterChain3 = new PassThroughFilterChain(filter2, filterChain2);
343 
344 		AsyncWebRequest asyncWebRequest = mock(AsyncWebRequest.class);
345 		given(asyncWebRequest.isAsyncStarted()).willReturn(true);
346 
347 		WebAsyncManager asyncManager = WebAsyncUtils.getAsyncManager(this.request);
348 		asyncManager.setTaskExecutor(new SyncTaskExecutor());
349 		asyncManager.setAsyncWebRequest(asyncWebRequest);
350 		asyncManager.startCallableProcessing(new Callable<String>() {
351 			@Override
352 			public String call() throws Exception {
353 				return "anything";
354 			}
355 		});
356 
357 		assertFalse(TransactionSynchronizationManager.hasResource(factory));
358 		assertFalse(TransactionSynchronizationManager.hasResource(factory2));
359 		filter2.doFilter(this.request, this.response, filterChain3);
360 		assertFalse(TransactionSynchronizationManager.hasResource(factory));
361 		assertFalse(TransactionSynchronizationManager.hasResource(factory2));
362 		assertEquals(1, count.get());
363 		assertEquals(1, count2.get());
364 		assertNotNull(request.getAttribute("invoked"));
365 		verify(asyncWebRequest, times(2)).addCompletionHandler(any(Runnable.class));
366 		verify(asyncWebRequest).addTimeoutHandler(any(Runnable.class));
367 		verify(asyncWebRequest, times(2)).addCompletionHandler(any(Runnable.class));
368 		verify(asyncWebRequest).startAsync();
369 
370 		// Async dispatch after concurrent handling produces result ...
371 
372 		reset(asyncWebRequest);
373 		given(asyncWebRequest.isAsyncStarted()).willReturn(false);
374 
375 		assertFalse(TransactionSynchronizationManager.hasResource(factory));
376 		assertFalse(TransactionSynchronizationManager.hasResource(factory2));
377 		filter.doFilter(this.request, this.response, filterChain3);
378 		assertFalse(TransactionSynchronizationManager.hasResource(factory));
379 		assertFalse(TransactionSynchronizationManager.hasResource(factory2));
380 		assertEquals(2, count.get());
381 		assertEquals(2, count2.get());
382 
383 		verify(this.manager).close();
384 		verify(manager2).close();
385 
386 		wac.close();
387 	}
