57  /**
58   * Abstract {@link org.springframework.beans.factory.FactoryBean} that creates
59   * a local JPA {@link javax.persistence.EntityManagerFactory} instance within
60   * a Spring application context.
61   *
62   * <p>Encapsulates the common functionality between the different JPA bootstrap
63   * contracts (standalone as well as container).
64   *
65   * <p>Implements support for standard JPA configuration conventions as well as
66   * Spring's customizable {@link JpaVendorAdapter} mechanism, and controls the
67   * EntityManagerFactory's lifecycle.
68   *
69   * <p>This class also implements the
70   * {@link org.springframework.dao.support.PersistenceExceptionTranslator}
71   * interface, as autodetected by Spring's
72   * {@link org.springframework.dao.annotation.PersistenceExceptionTranslationPostProcessor},
73   * for AOP-based translation of native exceptions to Spring DataAccessExceptions.
74   * Hence, the presence of e.g. LocalEntityManagerFactoryBean automatically enables
75   * a PersistenceExceptionTranslationPostProcessor to translate JPA exceptions.
76   *
77   * @author Juergen Hoeller
78   * @author Rod Johnson
79   * @since 2.0
80   * @see LocalEntityManagerFactoryBean
81   * @see LocalContainerEntityManagerFactoryBean
82   */
83  @SuppressWarnings("serial")

85  		FactoryBean<EntityManagerFactory>, BeanClassLoaderAware, BeanFactoryAware, BeanNameAware,
86  		InitializingBean, DisposableBean, EntityManagerFactoryInfo, PersistenceExceptionTranslator, Serializable {

287 	@Override
288 	public final void afterPropertiesSet() throws PersistenceException {
289 		if (this.jpaVendorAdapter != null) {
290 			if (this.persistenceProvider == null) {
291 				this.persistenceProvider = this.jpaVendorAdapter.getPersistenceProvider();
292 			}
293 			Map<String, ?> vendorPropertyMap = this.jpaVendorAdapter.getJpaPropertyMap();
294 			if (vendorPropertyMap != null) {
295 				for (Map.Entry<String, ?> entry : vendorPropertyMap.entrySet()) {
296 					if (!this.jpaPropertyMap.containsKey(entry.getKey())) {
297 						this.jpaPropertyMap.put(entry.getKey(), entry.getValue());
298 					}
299 				}
300 			}
301 			if (this.entityManagerFactoryInterface == null) {
302 				this.entityManagerFactoryInterface = this.jpaVendorAdapter.getEntityManagerFactoryInterface();
303 				if (!ClassUtils.isVisible(this.entityManagerFactoryInterface, this.beanClassLoader)) {
304 					this.entityManagerFactoryInterface = EntityManagerFactory.class;
305 				}
306 			}
307 			if (this.entityManagerInterface == null) {
308 				this.entityManagerInterface = this.jpaVendorAdapter.getEntityManagerInterface();
309 				if (!ClassUtils.isVisible(this.entityManagerInterface, this.beanClassLoader)) {
310 					this.entityManagerInterface = EntityManager.class;
311 				}
312 			}
313 			if (this.jpaDialect == null) {
314 				this.jpaDialect = this.jpaVendorAdapter.getJpaDialect();
315 			}
316 		}
317 
318 		this.nativeEntityManagerFactory = createNativeEntityManagerFactory();
319 		if (this.nativeEntityManagerFactory == null) {
320 			throw new IllegalStateException(
321 					"JPA PersistenceProvider returned null EntityManagerFactory - check your JPA provider setup!");
322 		}
323 		if (this.jpaVendorAdapter != null) {
324 			this.jpaVendorAdapter.postProcessEntityManagerFactory(this.nativeEntityManagerFactory);
325 		}
326 
327 		// Wrap the EntityManagerFactory in a factory implementing all its interfaces.
328 		// This allows interception of createEntityManager methods to return an
329 		// application-managed EntityManager proxy that automatically joins
330 		// existing transactions.
331 		this.entityManagerFactory = createEntityManagerFactoryProxy(this.nativeEntityManagerFactory);
332 	}
333 
334 	/**
335 	 * Create a proxy of the given EntityManagerFactory. We do this to be able
336 	 * to return transaction-aware proxies for application-managed
337 	 * EntityManagers, and to introduce the NamedEntityManagerFactory interface
338 	 * @param emf EntityManagerFactory as returned by the persistence provider
339 	 * @return proxy entity manager
340 	 */
341 	protected EntityManagerFactory createEntityManagerFactoryProxy(EntityManagerFactory emf) {
342 		Set<Class<?>> ifcs = new LinkedHashSet<Class<?>>();
343 		if (this.entityManagerFactoryInterface != null) {
344 			ifcs.add(this.entityManagerFactoryInterface);
345 		}
346 		else {
347 			ifcs.addAll(ClassUtils.getAllInterfacesForClassAsSet(emf.getClass(), this.beanClassLoader));
348 		}
349 		ifcs.add(EntityManagerFactoryInfo.class);
350 		try {
351 			return (EntityManagerFactory) Proxy.newProxyInstance(
352 					this.beanClassLoader, ifcs.toArray(new Class<?>[ifcs.size()]),
353 					new ManagedEntityManagerFactoryInvocationHandler(this));
354 		}
355 		catch (IllegalArgumentException ex) {
356 			if (this.entityManagerFactoryInterface != null) {
357 				throw new IllegalStateException("EntityManagerFactory interface [" + this.entityManagerFactoryInterface +
358 						"] seems to conflict with Spring's EntityManagerFactoryInfo mixin - consider resetting the "+
359 						"'entityManagerFactoryInterface' property to plain [javax.persistence.EntityManagerFactory]", ex);
360 			}
361 			else {
362 				throw new IllegalStateException("Conflicting EntityManagerFactory interfaces - " +
363 						"consider specifying the 'jpaVendorAdapter' or 'entityManagerFactoryInterface' property " +
364 						"to select a specific EntityManagerFactory interface to proceed with", ex);
365 			}
366 		}
367 	}

369 	/**
370 	 * Delegate an incoming invocation from the proxy, dispatching to EntityManagerFactoryInfo
371 	 * or the native EntityManagerFactory accordingly.
372 	 */
373 	Object invokeProxyMethod(Method method, Object[] args) throws Throwable {
374 		if (method.getDeclaringClass().isAssignableFrom(EntityManagerFactoryInfo.class)) {
375 			return method.invoke(this, args);
376 		}
377 		else if (method.getName().equals("createEntityManager") && args != null && args.length > 0 &&
378 				args[0] != null && args[0].getClass().isEnum() && "SYNCHRONIZED".equals(args[0].toString())) {
379 			// JPA 2.1's createEntityManager(SynchronizationType, Map)
380 			// Redirect to plain createEntityManager and add synchronization semantics through Spring proxy
381 			EntityManager rawEntityManager = (args.length > 1 ?
382 					this.nativeEntityManagerFactory.createEntityManager((Map<?, ?>) args[1]) :
383 					this.nativeEntityManagerFactory.createEntityManager());
384 			return ExtendedEntityManagerCreator.createApplicationManagedEntityManager(rawEntityManager, this, true);
385 		}
386 
387 		// Standard delegation to the native factory, just post-processing EntityManager return values
388 		Object retVal = method.invoke(this.nativeEntityManagerFactory, args);
389 		if (retVal instanceof EntityManager) {
390 			// Any other createEntityManager variant - expecting non-synchronized semantics
391 			EntityManager rawEntityManager = (EntityManager) retVal;
392 			retVal = ExtendedEntityManagerCreator.createApplicationManagedEntityManager(rawEntityManager, this, false);
393 		}
394 		return retVal;
395 	}

509 	/**
510 	 * Dynamic proxy invocation handler proxying an EntityManagerFactory to
511 	 * return a proxy EntityManager if necessary from createEntityManager()
512 	 * methods.
513 	 */
514 	@SuppressWarnings("serial")
515 	private static class ManagedEntityManagerFactoryInvocationHandler implements InvocationHandler, Serializable {
516 
517 		private final AbstractEntityManagerFactoryBean entityManagerFactoryBean;
518 
519 		public ManagedEntityManagerFactoryInvocationHandler(AbstractEntityManagerFactoryBean emfb) {
520 			this.entityManagerFactoryBean = emfb;
521 		}
522 
523 		@Override
524 		public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
525 			try {
526 				if (method.getName().equals("equals")) {
527 					// Only consider equal when proxies are identical.
528 					return (proxy == args[0]);
529 				}
530 				else if (method.getName().equals("hashCode")) {
531 					// Use hashCode of EntityManagerFactory proxy.
532 					return System.identityHashCode(proxy);
533 				}
534 				else if (method.getName().equals("unwrap")) {
535 					// Handle JPA 2.1 unwrap method - could be a proxy match.
536 					Class<?> targetClass = (Class<?>) args[0];
537 					if (targetClass == null || targetClass.isInstance(proxy)) {
538 						return proxy;
539 					}
540 				}
541 				return this.entityManagerFactoryBean.invokeProxyMethod(method, args);
542 			}
543 			catch (InvocationTargetException ex) {
544 				throw ex.getTargetException();
545 			}
546 		}
