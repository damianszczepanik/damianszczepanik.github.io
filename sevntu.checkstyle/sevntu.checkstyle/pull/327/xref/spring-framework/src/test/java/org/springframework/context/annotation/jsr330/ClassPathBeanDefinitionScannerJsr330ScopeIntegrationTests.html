94  	@Test
95  	public void testSingletonScopeWithNoProxy() {
96  		RequestContextHolder.setRequestAttributes(oldRequestAttributes);
97  		ApplicationContext context = createContext(ScopedProxyMode.NO);
98  		ScopedTestBean bean = (ScopedTestBean) context.getBean("singleton");
99  		assertTrue(context.isSingleton("singleton"));
100 		assertFalse(context.isPrototype("singleton"));
101 
102 		// should not be a proxy
103 		assertFalse(AopUtils.isAopProxy(bean));
104 
105 		assertEquals(DEFAULT_NAME, bean.getName());
106 		bean.setName(MODIFIED_NAME);
107 
108 		RequestContextHolder.setRequestAttributes(newRequestAttributes);
109 		// not a proxy so this should not have changed
110 		assertEquals(MODIFIED_NAME, bean.getName());
111 
112 		// singleton bean, so name should be modified even after lookup
113 		ScopedTestBean bean2 = (ScopedTestBean) context.getBean("singleton");
114 		assertEquals(MODIFIED_NAME, bean2.getName());
115 	}

299 	private ApplicationContext createContext(final ScopedProxyMode scopedProxyMode) {
300 		GenericWebApplicationContext context = new GenericWebApplicationContext();
301 		ClassPathBeanDefinitionScanner scanner = new ClassPathBeanDefinitionScanner(context);
302 		scanner.setIncludeAnnotationConfig(false);
303 		scanner.setScopeMetadataResolver(new ScopeMetadataResolver() {
304 			@Override
305 			public ScopeMetadata resolveScopeMetadata(BeanDefinition definition) {
306 				ScopeMetadata metadata = new ScopeMetadata();
307 				if (definition instanceof AnnotatedBeanDefinition) {
308 					AnnotatedBeanDefinition annDef = (AnnotatedBeanDefinition) definition;
309 					for (String type : annDef.getMetadata().getAnnotationTypes()) {
310 						if (type.equals(javax.inject.Singleton.class.getName())) {
311 							metadata.setScopeName(BeanDefinition.SCOPE_SINGLETON);
312 							break;
313 						}
314 						else if (annDef.getMetadata().getMetaAnnotationTypes(type).contains(javax.inject.Scope.class.getName())) {
315 							metadata.setScopeName(type.substring(type.length() - 13, type.length() - 6).toLowerCase());
316 							metadata.setScopedProxyMode(scopedProxyMode);
317 							break;
318 						}
319 						else if (type.startsWith("javax.inject")) {
320 							metadata.setScopeName(BeanDefinition.SCOPE_PROTOTYPE);
321 						}
322 					}
323 				}
324 				return metadata;
325 			}
326 		});
327 
328 		// Scan twice in order to find errors in the bean definition compatibility check.
329 		scanner.scan(getClass().getPackage().getName());
330 		scanner.scan(getClass().getPackage().getName());
331 
332 		context.registerAlias("classPathBeanDefinitionScannerJsr330ScopeIntegrationTests.SessionScopedTestBean", "session");
333 		context.refresh();
334 		return context;
335 	}
