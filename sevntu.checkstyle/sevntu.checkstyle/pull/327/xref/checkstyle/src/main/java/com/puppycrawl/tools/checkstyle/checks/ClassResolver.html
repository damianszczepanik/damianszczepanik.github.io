19  package com.puppycrawl.tools.checkstyle.checks;
20  
21  import java.util.Set;
22  
23  /**
24   * Utility class to resolve a class name to an actual class. Note that loaded
25   * classes are not initialized.
26   * <p>Limitations: this does not handle inner classes very well.</p>
27   *
28   * @author Oliver Burn
29   */
30  public class ClassResolver
31  {
32      /** name of the package to check if the class belongs to **/
33      private final String pkg;
34      /** set of imports to check against **/
35      private final Set<String> imports;
36      /** use to load classes **/
37      private final ClassLoader loader;
38  
39      /**
40       * Creates a new <code>ClassResolver</code> instance.
41       *
42       * @param loader the ClassLoader to load classes with.
43       * @param pkg the name of the package the class may belong to
44       * @param imports set of imports to check if the class belongs to
45       */
46      public ClassResolver(ClassLoader loader, String pkg, Set<String> imports)
47      {
48          this.loader = loader;
49          this.pkg = pkg;
50          this.imports = imports;
51          imports.add("java.lang.*");
52      }
53  
54      /**
55       * Attempts to resolve the Class for a specified name. The algorithm is
56       * to check:
57       * - fully qualified name
58       * - explicit imports
59       * - enclosing package
60       * - star imports
61       * @param name name of the class to resolve
62       * @param currentClass name of current class (for inner classes).
63       * @return the resolved class
64       * @throws ClassNotFoundException if unable to resolve the class
65       */
66      public Class<?> resolve(String name, String currentClass)
67          throws ClassNotFoundException
68      {
69          // See if the class is full qualified
70          Class<?> clazz = resolveQualifiedName(name);
71          if (clazz != null) {
72              return clazz;
73          }
74  
75          // try matching explicit imports
76          for (String imp : imports) {
77              // Very important to add the "." in the check below. Otherwise you
78              // when checking for "DataException", it will match on
79              // "SecurityDataException". This has been the cause of a very
80              // difficult bug to resolve!
81              if (imp.endsWith("." + name)) {
82                  clazz = resolveQualifiedName(imp);
83                  if (clazz != null) {
84                      return clazz;
85                  }
86  
87              }
88          }
89  
90          // See if in the package
91          if (!"".equals(pkg)) {
92              clazz = resolveQualifiedName(pkg + "." + name);
93              if (clazz != null) {
94                  return clazz;
95              }
96          }
97  
98          //inner class of this class???
99          if (!"".equals(currentClass)) {
100             final String innerClass = (!"".equals(pkg) ? (pkg + ".") : "")
101                 + currentClass + "$" + name;
102             if (isLoadable(innerClass)) {
103                 return safeLoad(innerClass);
104             }
105         }
106 
107         // try star imports
108         for (String imp : imports) {
109             if (imp.endsWith(".*")) {
110                 final String fqn = imp.substring(0, imp.lastIndexOf('.') + 1)
111                     + name;
112                 clazz = resolveQualifiedName(fqn);
113                 if (clazz != null) {
114                     return clazz;
115                 }
116             }
117         }
118 
119         // Giving up, the type is unknown, so load the class to generate an
120         // exception
121         return safeLoad(name);
122     }
123 
124     /**
125      * @return whether a specified class is loadable with safeLoad().
126      * @param name name of the class to check
127      */
128     public boolean isLoadable(String name)
129     {
130         try {
131             safeLoad(name);
132             return true;
133         }
134         catch (final ClassNotFoundException e) {
135             return false;
136         }
137     }
138 
139     /**
140      * Will load a specified class is such a way that it will NOT be
141      * initialised.
142      * @param name name of the class to load
143      * @return the <code>Class</code> for the specified class
144      * @throws ClassNotFoundException if an error occurs
145      */
146     public Class<?> safeLoad(String name)
147         throws ClassNotFoundException
148     {
149         // The next line will load the class using the specified class
150         // loader. The magic is having the "false" parameter. This means the
151         // class will not be initialised. Very, very important.
152         return Class.forName(name, false, loader);
153     }
154 
155     /**
156      * Tries to resolve a class for fully-specified name.
157      * @param name a given name of class.
158      * @return Class object for the given name or null.
159      */
160     private Class<?> resolveQualifiedName(final String name)
161     {
162         try {
163             if (isLoadable(name)) {
164                 return safeLoad(name);
165             }
166             //Perhaps it's fully-qualified inner class
167             final int dot = name.lastIndexOf(".");
168             if (dot != -1) {
169                 final String innerName =
170                     name.substring(0, dot) + "$" + name.substring(dot + 1);
171                 if (isLoadable(innerName)) {
172                     return safeLoad(innerName);
173                 }
174             }
175         }
176         catch (final ClassNotFoundException ex) {
177             // we shouldn't get this exception here,
178             // so this is unexpected runtime exception
179             throw new RuntimeException(ex);
180         }
181 
182         return null;
183     }
184 }
