19  package com.puppycrawl.tools.checkstyle;
20  
21  import java.io.File;
22  
23  /**
24   * Contains utility methods.
25   *
26   * @author <a href="mailto:nesterenko-aleksey@list.ru">Aleksey Nesterenko</a>
27   */
28  public final class Utils
29  {
30  
31      /** stop instances being created **/
32      private Utils()
33      {
34      }
35  
36      /**
37       * Returns whether the file extension matches what we are meant to
38       * process.
39       * @param file the file to be checked.
40       * @param fileExtensions files extensions, empty property in config makes it matches to all.
41       * @return whether there is a match.
42       */
43      public static boolean fileExtensionMatches(File file, String[] fileExtensions)
44      {
45          boolean result = false;
46          if ((fileExtensions == null) || (fileExtensions.length == 0)) {
47              result = true;
48          }
49          else {
50              // normalize extensions so all of them have a leading dot
51              final String[] withDotExtensions = new String[fileExtensions.length];
52              for (int i = 0; i < fileExtensions.length; i++) {
53                  final String extension = fileExtensions[i];
54                  if (extension.startsWith(".")) {
55                      withDotExtensions[i] = extension;
56                  }
57                  else {
58                      withDotExtensions[i] = "." + extension;
59                  }
60              }
61  
62              final String fileName = file.getName();
63              for (final String fileExtension : withDotExtensions) {
64                  if (fileName.endsWith(fileExtension)) {
65                      result = true;
66                  }
67              }
68          }
69  
70          return result;
71      }
72  }
