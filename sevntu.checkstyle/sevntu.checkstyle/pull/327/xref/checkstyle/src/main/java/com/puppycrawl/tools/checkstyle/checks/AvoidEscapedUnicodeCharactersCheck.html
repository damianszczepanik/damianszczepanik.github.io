29  /**
30   * <p>
31   * Restrict using <a href =
32   * "http://docs.oracle.com/javase/specs/jls/se7/html/jls-3.html#jls-3.3">
33   * Unicode escapes</a> (e.g. \u221e).
34   * It is possible to allow using escapes for
35   * <a href="http://en.wiktionary.org/wiki/Appendix:Control_characters">
36   * non-printable(control) characters</a>.
37   * Also, this check can be configured to allow using escapes
38   * if trail comment is present. By the option it is possible to
39   * allow using escapes if literal contains only them. By the option it
40   * is possible to allow using escapes for space literals.
41   * </p>
42   * <p>
43   * Examples of using Unicode:</p>
44   * <pre>
45   * String unitAbbrev = "Î¼s"; //Best: perfectly clear even without a comment.
46   * String unitAbbrev = "\u03bcs"; //Poor: the reader has no idea what this is.
47   * </pre>
48   * <p>
49   * An example of how to configure the check is:
50   * </p>
51   * <pre>
52   * &lt;module name="AvoidEscapedUnicodeCharacters"/&gt;
53   * </pre>
54   * <p>
55   * An example of non-printable(control) characters.
56   * </p>
57   * <pre>
58   * return '\ufeff' + content; // byte order mark
59   * </pre>
60   * <p>
61   * An example of how to configure the check to allow using escapes
62   * for non-printable(control) characters:
63   * </p>
64   * <pre>
65   * &lt;module name="AvoidEscapedUnicodeCharacters"&gt;
66   *     &lt;property name="allowEscapesForControlCharacters" value="true"/&gt;
67   * &lt;/module&gt;
68   * </pre>
69   * <p>
70   * Example of using escapes with trail comment:
71   * </p>
72   * <pre>
73   * String unitAbbrev = "\u03bcs"; // Greek letter mu, "s"
74   * </pre>
75   * <p>An example of how to configure the check to allow using escapes
76   * if trail comment is present:
77   * </p>
78   * <pre>
79   * &lt;module name="AvoidEscapedUnicodeCharacters"&gt;
80   *     &lt;property name="allowByTailComment" value="true"/&gt;
81   * &lt;/module&gt;
82   * </pre>
83   * <p>Example of using escapes if literal contains only them:
84   * </p>
85   * <pre>
86   * String unitAbbrev = "\u03bc\u03bc\u03bc";
87   * </pre>
88   * <p>An example of how to configure the check to allow escapes
89   * if literal contains only them:
90   * </p>
91   * <pre>
92   * &lt;module name="AvoidEscapedUnicodeCharacters"&gt;
93   *    &lt;property name="allowIfAllCharactersEscaped" value="true"/&gt;
94   * &lt;/module&gt;
95   * </pre>
96   * <p>An example of how to configure the check to allow non-printable escapes:
97   * </p>
98   * <pre>
99   * &lt;module name="AvoidEscapedUnicodeCharacters"&gt;
100  *    &lt;property name="allowNonPrintableEscapes" value="true"/&gt;
101  * &lt;/module&gt;
102  * </pre>
103  *
104  * @author maxvetrenko
105  *
106  */
107 public class AvoidEscapedUnicodeCharactersCheck
108     extends Check
109 {
110      /** Regexp for Unicode chars */
111     private static Pattern sUnicodeRegexp =
112             Utils.getPattern("\\\\u[a-fA-F0-9]{4}");
113 
114     /** Regexp Unicode control characters */
115     private static Pattern sUnicodeControl = Utils.getPattern("\\\\(u|U)"
116             + "(00[0-1][0-1A-Fa-f]|00[8-9][0-9A-Fa-f]|034(f|F)|070(f|F)"
117             + "|180(e|E)|200[b-fB-F]|202[b-eB-E]|206[0-4a-fA-F]"
118             + "|[fF]{3}[9a-bA-B]|[fF][eE][fF]{2})");
119 
120     /** Regexp for trail comment */
121     private static Pattern sCommentRegexp = Utils.getPattern(";[ ]*//+"
122             + "[a-zA-Z0-9 ]*|;[ ]*/[*]{1}+[a-zA-Z0-9 ]*");
123 
124     /** Regexp for all escaped chars*/
125     private static Pattern sAllEscapedChars =
126             Utils.getPattern("^((\\\\u)[a-fA-F0-9]{4}"
127                     + "||\\\\b|\\\\t|\\\\n|\\\\f|\\\\r|\\\\|\\\"|\\\')+$");
128 
129     /** Regexp for non-printable unicode chars*/
130     private static Pattern sNonPrintableChars = Utils.getPattern("\\\\u1680|\\\\u2028"
131             + "|\\\\u2029|\\\\u205(f|F)|\\\\u3000|\\\\u2007|\\\\u2000|\\\\u200(a|A)"
132             + "|\\\\u007(F|f)|\\\\u009(f|F)|\\\\u(f|F){4}|\\\\u007(F|f)|\\\\u00(a|A)(d|D)"
133             + "|\\\\u0600|\\\\u061(c|C)|\\\\u06(d|D){2}|\\\\u070(f|F)|\\\\u1680|\\\\u180(e|E)"
134             + "|\\\\u2000|\\\\u2028|\\\\u205(f|F)|\\\\u2066|\\\\u2067|\\\\u2068|\\\\u2069"
135             + "|\\\\u206(a|A)|\\\\u(d|D)800|\\\\u(f|F)(e|E)(f|F){2}|\\\\u(f|F){3}9"
136             + "|\\\\u(f|F){3}(a|A)|\\\\u0020|\\\\u00(a|A)0|\\\\u00(a|A)(d|D)|\\\\u0604"
137             + "|\\\\u061(c|C)|\\\\u06(d|D){2}|\\\\u070(f|F)|\\\\u1680|\\\\u180(e|E)|\\\\u200(f|F)"
138             + "|\\\\u202(f|F)|\\\\u2064|\\\\u2066|\\\\u2067|\\\\u2068|\\\\u2069|\\\\u206(f|F)"
139             + "|\\\\u(f|F)8(f|F){2}|\\\\u(f|F)(e|E)(f|F){2}|\\\\u(f|F){3}9|\\\\u(f|F){3}(b|B)"
140             + "|\\\\u05(d|D)0|\\\\u05(f|F)3|\\\\u0600|\\\\u0750|\\\\u0(e|E)00|\\\\u1(e|E)00"
141             + "|\\\\u2100|\\\\u(f|F)(b|B)50|\\\\u(f|F)(e|E)70|\\\\u(F|f){2}61|\\\\u04(f|F)9"
142             + "|\\\\u05(b|B)(e|E)|\\\\u05(e|E)(a|A)|\\\\u05(f|F)4|\\\\u06(f|F){2}"
143             + "|\\\\u077(f|F)|\\\\u0(e|E)7(f|F)|\\\\u20(a|A)(f|F)|\\\\u213(a|A)|\\\\u0000"
144             + "|\\\\u(f|F)(d|D)(f|F){2}|\\\\u(f|F)(e|E)(f|F){2}|\\\\u(f|F){2}(d|D)(c|C)"
145             + "|\\\\u2002|\\\\u0085|\\\\u200(a|A)|\\\\u2005|\\\\u2000|\\\\u2029|\\\\u000(B|b)"
146             + "|\\\\u2008|\\\\u2003|\\\\u205(f|F)|\\\\u1680|\\\\u0009|\\\\u0020|\\\\u2006"
147             + "|\\\\u2001|\\\\u202(f|F)|\\\\u00(a|A)0|\\\\u000(c|C)|\\\\u2009|\\\\u2004|\\\\u2028"
148             + "|\\\\u2028|\\\\u2007|\\\\u2004|\\\\u2028|\\\\u2007|\\\\u2025"
149             + "|\\\\u(f|F){2}0(e|E)|\\\\u(f|F){2}61");
150 
151     /** Allow use escapes for non-printable(control) characters.  */
152     private boolean allowEscapesForControlCharacters;
153 
154     /** Allow use escapes if trail comment is present*/
155     private boolean allowByTailComment;
156 
157     /** Allow if all characters in literal are excaped*/
158     private boolean allowIfAllCharactersEscaped;
159 
160     /** Allow escapes for space literals*/
161     private boolean allowNonPrintableEscapes;
162 
163     /**
164      * Set allowIfAllCharactersEscaped.
165      * @param allow user's value.
166      */
167     public final void setAllowEscapesForControlCharacters(boolean allow)
168     {
169         allowEscapesForControlCharacters = allow;
170     }
171 
172     /**
173      * Set allowByTailComment.
174      * @param allow user's value.
175      */
176     public final void setAllowByTailComment(boolean allow)
177     {
178         allowByTailComment = allow;
179     }
180 
181     /**
182      * Set allowIfAllCharactersEscaped.
183      * @param allow user's value.
184      */
185     public final void setAllowIfAllCharactersEscaped(boolean allow)
186     {
187         allowIfAllCharactersEscaped = allow;
188     }
189 
190     /**
191      * Set allowSpaceEscapes.
192      * @param allow user's value.
193      */
194     public final void setAllowNonPrintableEscapes(boolean allow)
195     {
196         allowNonPrintableEscapes = allow;
197     }

199     @Override
200     public int[] getDefaultTokens()
201     {
202         return new int[] {TokenTypes.STRING_LITERAL, TokenTypes.CHAR_LITERAL};
203     }
204 
205     @Override
206     public int[] getAcceptableTokens()
207     {
208         return new int[] {TokenTypes.STRING_LITERAL, TokenTypes.CHAR_LITERAL};
209     }
210 
211     @Override
212     public void visitToken(DetailAST ast)
213     {
214 
215         final String literal = ast.getText();
216 
217         if (hasUnicodeChar(literal)) {
218             if (!(allowByTailComment && haastrailComment(ast)
219                     || isAllCharactersEscaped(literal)
220                     || (allowEscapesForControlCharacters
221                             && isOnlyUnicodeValidChars(literal, sUnicodeControl))
222                     || (allowNonPrintableEscapes
223                             && isOnlyUnicodeValidChars(literal, sNonPrintableChars))))
224             {
225                 log(ast.getLineNo(), "forbid.escaped.unicode.char");
226             }
227         }
228     }
229 
230     /**
231      * Checks if literal has Unicode chars.
232      * @param literal String literal.
233      * @return true if literal has Unicode chars.
234      */
235     private boolean hasUnicodeChar(String literal)
236     {
237         return sUnicodeRegexp.matcher(literal).find();
238     }
239 
240     /**
241      * Check if String literal contains Unicode control chars.
242      * @param literal String llteral.
243      * @param pattern RegExp for valid characters.
244      * @return true, if String literal contains Unicode control chars.
245      */
246     private boolean isOnlyUnicodeValidChars(String literal, Pattern pattern)
247     {
248         final int unicodeMatchesCounter =
249                 countMatches(sUnicodeRegexp, literal);
250         final int unicodeValidMatchesCouter =
251                 countMatches(pattern, literal);
252         return unicodeMatchesCounter - unicodeValidMatchesCouter == 0;
253     }
254 
255     /**
256      * Check if trail comment is present after ast token.
257      * @param ast current token.
258      * @return true if trail comment is present after ast token.
259      */
260     private boolean haastrailComment(DetailAST ast)
261     {
262         boolean result = false;
263         final DetailAST variableDef = getVariableDef(ast);
264         DetailAST semi;
265 
266         if (variableDef != null) {
267 
268             semi = variableDef.getNextSibling();
269 
270             if (semi.getType() != TokenTypes.SEMI) {
271                 semi = variableDef.getLastChild();
272             }
273         }
274         else {
275             semi = getSemi(ast);
276         }
277 
278         if (semi != null) {
279             final int lineNo = semi.getLineNo();
280             final String currentLine = getLine(lineNo - 1);
281 
282             if (currentLine != null && sCommentRegexp.matcher(currentLine).find()) {
283                 result = true;
284             }
285         }
286 
287         return result;
288     }
289 
290     /**
291      * Count regexp matchers into String literal.
292      * @param pattern pattern.
293      * @param target String literal.
294      * @return count of regexp matchers.
295      */
296     private int countMatches(Pattern pattern, String target)
297     {
298         int matcherCounter = 0;
299         final Matcher matcher = pattern.matcher(target);
300         while (matcher.find()) {
301             matcherCounter++;
302         }
303         return matcherCounter;
304     }
305 
306     /**
307      * Get variable definition.
308      * @param ast current token.
309      * @return variable definition.
310      */
311     private DetailAST getVariableDef(DetailAST ast)
312     {
313         DetailAST result = ast.getParent();
314         while (result != null
315                 && result.getType() != TokenTypes.VARIABLE_DEF)
316         {
317             result = result.getParent();
318         }
319         return result;
320     }
321 
322     /**
323      * Get semi token.
324      * @param ast current token.
325      * @return semi token or null.
326      */
327     private DetailAST getSemi(DetailAST ast)
328     {
329         DetailAST result = ast.getParent();
330         while (result != null
331                 && result.getLastChild().getType() != TokenTypes.SEMI)
332         {
333             result = result.getParent();
334         }
335         if (result != null) {
336             result = result.getLastChild();
337         }
338         return result;
339     }
340 
341     /**
342      * Checks if all characters in String literal is escaped.
343      * @param literal current literal.
344      * @return true if all characters in String literal is escaped.
345      */
346     private boolean isAllCharactersEscaped(String literal)
347     {
348         return allowIfAllCharactersEscaped
349                 && sAllEscapedChars.matcher(literal.substring(1,
350                         literal.length() - 1)).find();
351     }
