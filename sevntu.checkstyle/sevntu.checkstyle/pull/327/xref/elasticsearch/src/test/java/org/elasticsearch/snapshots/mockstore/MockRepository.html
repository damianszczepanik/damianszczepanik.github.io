83      @Inject
84      public MockRepository(RepositoryName name, RepositorySettings repositorySettings, IndexShardRepository indexShardRepository, ClusterService clusterService) throws IOException {
85          super(name, overrideSettings(repositorySettings, clusterService), indexShardRepository);
86          randomControlIOExceptionRate = repositorySettings.settings().getAsDouble("random_control_io_exception_rate", 0.0);
87          randomDataFileIOExceptionRate = repositorySettings.settings().getAsDouble("random_data_file_io_exception_rate", 0.0);
88          blockOnControlFiles = repositorySettings.settings().getAsBoolean("block_on_control", false);
89          blockOnDataFiles = repositorySettings.settings().getAsBoolean("block_on_data", false);
90          blockOnInitialization = repositorySettings.settings().getAsBoolean("block_on_init", false);
91          randomPrefix = repositorySettings.settings().get("random", "default");
92          waitAfterUnblock = repositorySettings.settings().getAsLong("wait_after_unblock", 0L);
93          logger.info("starting mock repository with random prefix " + randomPrefix);
94          mockBlobStore = new MockBlobStore(super.blobStore());
95      }

105     private static RepositorySettings overrideSettings(RepositorySettings repositorySettings, ClusterService clusterService) {
106         if (repositorySettings.settings().getAsBoolean("localize_location", false)) {
107             return new RepositorySettings(
108                     repositorySettings.globalSettings(),
109                     localizeLocation(repositorySettings.settings(), clusterService));
110         } else {
111             return repositorySettings;
112         }
113     }

115     private static Settings localizeLocation(Settings settings, ClusterService clusterService) {
116         Path location = Paths.get(settings.get("location"));
117         location = location.resolve(clusterService.localNode().getId());
118         return settingsBuilder().put(settings).put("location", location.toAbsolutePath()).build();
119     }

148     public synchronized void unblockExecution() {
149         if (blocked) {
150             blocked = false;
151             // Clean blocking flags, so we wouldn't try to block again
152             blockOnDataFiles = false;
153             blockOnControlFiles = false;
154             blockOnInitialization = false;
155             this.notifyAll();
156         }
157     }
158 
159     public boolean blocked() {
160         return blocked;
161     }
162 
163     private synchronized boolean blockExecution() {
164         logger.debug("Blocking execution");
165         boolean wasBlocked = false;
166         try {
167             while (blockOnDataFiles || blockOnControlFiles || blockOnInitialization) {
168                 blocked = true;
169                 this.wait();
170                 wasBlocked = true;
171             }
172         } catch (InterruptedException ex) {
173             Thread.currentThread().interrupt();
174         }
175         logger.debug("Unblocking execution");
176         return wasBlocked;
177     }

179     public class MockBlobStore extends BlobStoreWrapper {
180         ConcurrentMap<String, AtomicLong> accessCounts = new ConcurrentHashMap<>();
181 
182         private long incrementAndGet(String path) {
183             AtomicLong value = accessCounts.get(path);
184             if (value == null) {
185                 value = accessCounts.putIfAbsent(path, new AtomicLong(1));
186             }
187             if (value != null) {
188                 return value.incrementAndGet();
189             }
190             return 1;
191         }

202         private class MockBlobContainer extends BlobContainerWrapper {
203             private MessageDigest digest;
204 
205             private boolean shouldFail(String blobName, double probability) {
206                 if (probability > 0.0) {
207                     String path = path().add(blobName).buildAsString("/") + "/" + randomPrefix;
208                     path += "/" + incrementAndGet(path);
209                     logger.info("checking [{}] [{}]", path, Math.abs(hashCode(path)) < Integer.MAX_VALUE * probability);
210                     return Math.abs(hashCode(path)) < Integer.MAX_VALUE * probability;
211                 } else {
212                     return false;
213                 }
214             }
215 
216             private int hashCode(String path) {
217                 try {
218                     digest = MessageDigest.getInstance("MD5");
219                     byte[] bytes = digest.digest(path.getBytes("UTF-8"));
220                     int i = 0;
221                     return ((bytes[i++] & 0xFF) << 24) | ((bytes[i++] & 0xFF) << 16)
222                             | ((bytes[i++] & 0xFF) << 8) | (bytes[i++] & 0xFF);
223                 } catch (NoSuchAlgorithmException | UnsupportedEncodingException ex) {
224                     throw new ElasticsearchException("cannot calculate hashcode", ex);
225                 }
226             }
227 
228             private void maybeIOExceptionOrBlock(String blobName) throws IOException {
229                 if (blobName.startsWith("__")) {
230                     if (shouldFail(blobName, randomDataFileIOExceptionRate)) {
231                         logger.info("throwing random IOException for file [{}] at path [{}]", blobName, path());
232                         addFailure();
233                         throw new IOException("Random IOException");
234                     } else if (blockOnDataFiles) {
235                         logger.info("blocking I/O operation for file [{}] at path [{}]", blobName, path());
236                         if (blockExecution() && waitAfterUnblock > 0) {
237                             try {
238                                 // Delay operation after unblocking
239                                 // So, we can start node shutdown while this operation is still running.
240                                 Thread.sleep(waitAfterUnblock);
241                             } catch (InterruptedException ex) {
242                                 //
243                             }
244                         }
245                     }
246                 } else {
247                     if (shouldFail(blobName, randomControlIOExceptionRate)) {
248                         logger.info("throwing random IOException for file [{}] at path [{}]", blobName, path());
249                         addFailure();
250                         throw new IOException("Random IOException");
251                     } else if (blockOnControlFiles) {
252                         logger.info("blocking I/O operation for file [{}] at path [{}]", blobName, path());
253                         if (blockExecution() && waitAfterUnblock > 0) {
254                             try {
255                                 // Delay operation after unblocking
256                                 // So, we can start node shutdown while this operation is still running.
257                                 Thread.sleep(waitAfterUnblock);
258                             } catch (InterruptedException ex) {
259                                 //
260                             }
261                         }
262                     }
263                 }
264             }
