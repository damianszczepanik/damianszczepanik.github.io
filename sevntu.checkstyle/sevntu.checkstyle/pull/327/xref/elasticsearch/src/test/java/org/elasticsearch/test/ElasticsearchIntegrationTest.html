145 /**
146  * {@link ElasticsearchIntegrationTest} is an abstract base class to run integration
147  * tests against a JVM private Elasticsearch Cluster. The test class supports 2 different
148  * cluster scopes.
149  * <ul>
150  * <li>{@link Scope#TEST} - uses a new cluster for each individual test method.</li>
151  * <li>{@link Scope#SUITE} - uses a cluster shared across all test methods in the same suite</li>
152  * </ul>
153  * <p/>
154  * The most common test scope is {@link Scope#SUITE} which shares a cluster per test suite.
155  * <p/>
156  * If the test methods need specific node settings or change persistent and/or transient cluster settings {@link Scope#TEST}
157  * should be used. To configure a scope for the test cluster the {@link ClusterScope} annotation
158  * should be used, here is an example:
159  * <pre>
160  *
161  * @ClusterScope(scope=Scope.TEST) public class SomeIntegrationTest extends ElasticsearchIntegrationTest {
162  * @Test public void testMethod() {}
163  * }
164  * </pre>
165  * <p/>
166  * If no {@link ClusterScope} annotation is present on an integration test the default scope is {@link Scope#SUITE}
167  * <p/>
168  * A test cluster creates a set of nodes in the background before the test starts. The number of nodes in the cluster is
169  * determined at random and can change across tests. The {@link ClusterScope} allows configuring the initial number of nodes 
170  * that are created before the tests start.
171  * <p/>
172  *  <pre>
173  * @ClusterScope(scope=Scope.SUITE, numDataNodes=3)
174  * public class SomeIntegrationTest extends ElasticsearchIntegrationTest {
175  * @Test public void testMethod() {}
176  * }
177  * </pre>
178  * <p/>
179  * Note, the {@link ElasticsearchIntegrationTest} uses randomized settings on a cluster and index level. For instance
180  * each test might use different directory implementation for each test or will return a random client to one of the
181  * nodes in the cluster for each call to {@link #client()}. Test failures might only be reproducible if the correct
182  * system properties are passed to the test execution environment.
183  * <p/>
184  * <p>
185  * This class supports the following system properties (passed with -Dkey=value to the application)
186  * <ul>
187  * <li>-D{@value #TESTS_CLIENT_RATIO} - a double value in the interval [0..1] which defines the ration between node and transport clients used</li>
188  * <li>-D{@value InternalTestCluster#TESTS_ENABLE_MOCK_MODULES} - a boolean value to enable or disable mock modules. This is
189  * useful to test the system without asserting modules that to make sure they don't hide any bugs in production.</li>
190  * <li> - a random seed used to initialize the index random context.
191  * </ul>
192  * </p>
193  */
194 @Ignore
195 @AbstractRandomizedTest.Integration
196 public abstract class ElasticsearchIntegrationTest extends ElasticsearchTestCase {

268     protected final void beforeInternal() throws Exception {
269         assert Thread.getDefaultUncaughtExceptionHandler() instanceof ElasticsearchUncaughtExceptionHandler;
270         try {
271             final Scope currentClusterScope = getCurrentClusterScope();
272             switch (currentClusterScope) {
273                 case SUITE:
274                     assert SUITE_SEED != null : "Suite seed was not initialized";
275                     currentCluster = buildAndPutCluster(currentClusterScope, SUITE_SEED);
276                     break;
277                 case TEST:
278                     currentCluster = buildAndPutCluster(currentClusterScope, randomLong());
279                     break;
280                 default:
281                     fail("Unknown Scope: [" + currentClusterScope + "]");
282             }
283             cluster().beforeTest(getRandom(), getPerTestTransportClientRatio());
284             cluster().wipe();
285             randomIndexTemplate();
286             logger.info("[{}#{}]: before test", getTestClass().getSimpleName(), getTestName());
287         } catch (OutOfMemoryError e) {
288             if (e.getMessage().contains("unable to create new native thread")) {
289                 ElasticsearchTestCase.printStackDump(logger);
290             }
291             throw e;
292         }
293     }

465             // see #7210

467         }

469     }

481     private static ImmutableSettings.Builder setRandomMerge(Random random, ImmutableSettings.Builder builder) {
482         if (random.nextBoolean()) {
483             builder.put(AbstractMergePolicyProvider.INDEX_COMPOUND_FORMAT,
484                     random.nextBoolean() ? random.nextDouble() : random.nextBoolean());
485         }
486         Class<? extends MergePolicyProvider<?>> mergePolicy = TieredMergePolicyProvider.class;
487         switch (random.nextInt(5)) {
488             case 4:
489                 mergePolicy = LogByteSizeMergePolicyProvider.class;
490                 break;
491             case 3:
492                 mergePolicy = LogDocMergePolicyProvider.class;
493                 break;
494             case 0:
495                 mergePolicy = null;
496         }
497         if (mergePolicy != null) {
498             builder.put(MergePolicyModule.MERGE_POLICY_TYPE_KEY, mergePolicy.getName());
499         }
500 
501         switch (random.nextInt(4)) {
502             case 3:
503                 builder.put(MergeSchedulerModule.MERGE_SCHEDULER_TYPE_KEY, ConcurrentMergeSchedulerProvider.class);
504                 final int maxThreadCount = RandomInts.randomIntBetween(random, 1, 4);
505                 final int maxMergeCount = RandomInts.randomIntBetween(random, maxThreadCount, maxThreadCount + 4);
506                 builder.put(ConcurrentMergeSchedulerProvider.MAX_MERGE_COUNT, maxMergeCount);
507                 builder.put(ConcurrentMergeSchedulerProvider.MAX_THREAD_COUNT, maxThreadCount);
508                 break;
509         }
510 
511         return builder;
512     }

549     private TestCluster buildAndPutCluster(Scope currentClusterScope, long seed) throws Exception {
550         final Class<?> clazz = this.getClass();
551         TestCluster testCluster = clusters.remove(clazz); // remove this cluster first
552         clearClusters(); // all leftovers are gone by now... this is really just a double safety if we miss something somewhere
553         switch (currentClusterScope) {
554             case SUITE:
555                 if (testCluster == null) { // only build if it's not there yet
556                     testCluster = buildWithPrivateContext(currentClusterScope, seed);
557                 }
558                 break;
559             case TEST:
560                 // close the previous one and create a new one
561                 IOUtils.closeWhileHandlingException(testCluster);
562                 testCluster = buildTestCluster(currentClusterScope, seed);
563                 break;
564         }
565         clusters.put(clazz, testCluster);
566         return testCluster;
567     }

571             IOUtils.close(clusters.values());
572             clusters.clear();

576     protected final void afterInternal(boolean afterClass) throws Exception {
577         boolean success = false;
578         try {
579             logger.info("[{}#{}]: cleaning up after test", getTestClass().getSimpleName(), getTestName());
580             clearDisruptionScheme();
581             final Scope currentClusterScope = getCurrentClusterScope();
582             try {
583                 if (cluster() != null) {
584                     if (currentClusterScope != Scope.TEST) {
585                         MetaData metaData = client().admin().cluster().prepareState().execute().actionGet().getState().getMetaData();
586                         assertThat("test leaves persistent cluster metadata behind: " + metaData.persistentSettings().getAsMap(), metaData
587                             .persistentSettings().getAsMap().size(), equalTo(0));
588                         assertThat("test leaves transient cluster metadata behind: " + metaData.transientSettings().getAsMap(), metaData
589                             .transientSettings().getAsMap().size(), equalTo(0));
590                     }
591                     ensureClusterSizeConsistency();
592                     cluster().wipe(); // wipe after to make sure we fail in the test that didn't ack the delete
593                     if (afterClass || currentClusterScope == Scope.TEST) {
594                         cluster().close();
595                     }
596                     cluster().assertAfterTest();
597                 }
598             } finally {
599                 if (currentClusterScope == Scope.TEST) {
600                     clearClusters(); // it is ok to leave persistent / transient cluster state behind if scope is TEST
601                 }
602             }
603             logger.info("[{}#{}]: cleaned up after test", getTestClass().getSimpleName(), getTestName());
604             success = true;
605         } finally {
606             if (!success) {
607                 // if we failed here that means that something broke horribly so we should clear all clusters
608                 afterTestRule.forceFailure();
609             }
610         }
611     }

1021     /**
1022      * Waits until at least a give number of document is visible for searchers
1023      *
1024      * @param numDocs         number of documents to wait for
1025      * @param maxWaitTime     if not progress have been made during this time, fail the test
1026      * @param maxWaitTimeUnit the unit in which maxWaitTime is specified
1027      * @param indexer         a {@link org.elasticsearch.test.BackgroundIndexer}. If supplied it will be first checked for documents indexed.
1028      *                        This saves on unneeded searches.
1029      * @return the actual number of docs seen.
1030      * @throws InterruptedException
1031      */
1032     public long waitForDocs(final long numDocs, int maxWaitTime, TimeUnit maxWaitTimeUnit, final @Nullable BackgroundIndexer indexer)
1033             throws InterruptedException {
1034         final AtomicLong lastKnownCount = new AtomicLong(-1);
1035         long lastStartCount = -1;
1036         Predicate<Object> testDocs = new Predicate<Object>() {
1037             @Override
1038             public boolean apply(Object o) {
1039                 if (indexer != null) {
1040                     lastKnownCount.set(indexer.totalIndexedDocs());
1041                 }
1042                 if (lastKnownCount.get() >= numDocs) {
1043                     long count = client().prepareCount().setQuery(matchAllQuery()).execute().actionGet().getCount();
1044                     if (count == lastKnownCount.get()) {
1045                         // no progress - try to refresh for the next time
1046                         client().admin().indices().prepareRefresh().get();
1047                     }
1048                     lastKnownCount.set(count);
1049                     logger.debug("[{}] docs visible for search. waiting for [{}]", lastKnownCount.get(), numDocs);
1050                 } else {
1051                     logger.debug("[{}] docs indexed. waiting for [{}]", lastKnownCount.get(), numDocs);
1052                 }
1053                 return lastKnownCount.get() >= numDocs;
1054             }
1055         };
1056 
1057         while (!awaitBusy(testDocs, maxWaitTime, maxWaitTimeUnit)) {
1058             if (lastStartCount == lastKnownCount.get()) {
1059                 // we didn't make any progress
1060                 fail("failed to reach " + numDocs + "docs");
1061             }
1062             lastStartCount = lastKnownCount.get();
1063         }
1064         return lastKnownCount.get();
1065     }

1300     /**
1301      * Indexes the given {@link IndexRequestBuilder} instances randomly. It shuffles the given builders and either
1302      * indexes they in a blocking or async fashion. This is very useful to catch problems that relate to internal document
1303      * ids or index segment creations. Some features might have bug when a given document is the first or the last in a
1304      * segment or if only one document is in a segment etc. This method prevents issues like this by randomizing the index
1305      * layout.
1306      *
1307      * @param forceRefresh   if <tt>true</tt> all involved indices are refreshed once the documents are indexed.
1308      * @param dummyDocuments if <tt>true</tt> some empty dummy documents may be randomly inserted into the document list and deleted once
1309      *                       all documents are indexed. This is useful to produce deleted documents on the server side.
1310      * @param maybeFlush if <tt>true</tt> this method may randomly execute full flushes after index operations.
1311      * @param builders       the documents to index.
1312      */
1313     public void indexRandom(boolean forceRefresh, boolean dummyDocuments, boolean maybeFlush, List<IndexRequestBuilder> builders) throws InterruptedException, ExecutionException {
1314 
1315         Random random = getRandom();
1316         Set<String> indicesSet = new HashSet<>();
1317         for (IndexRequestBuilder builder : builders) {
1318             indicesSet.add(builder.request().index());
1319         }
1320         Set<Tuple<String, String>> bogusIds = new HashSet<>();
1321         if (random.nextBoolean() && !builders.isEmpty() && dummyDocuments) {
1322             builders = new ArrayList<>(builders);
1323             final String[] indices = indicesSet.toArray(new String[indicesSet.size()]);
1324             // inject some bogus docs
1325             final int numBogusDocs = scaledRandomIntBetween(1, builders.size() * 2);
1326             final int unicodeLen = between(1, 10);
1327             for (int i = 0; i < numBogusDocs; i++) {
1328                 String id = randomRealisticUnicodeOfLength(unicodeLen) + Integer.toString(dummmyDocIdGenerator.incrementAndGet());
1329                 String index = RandomPicks.randomFrom(random, indices);
1330                 bogusIds.add(new Tuple<>(index, id));
1331                 builders.add(client().prepareIndex(index, RANDOM_BOGUS_TYPE, id).setSource("{}"));
1332             }
1333         }
1334         final String[] indices = indicesSet.toArray(new String[indicesSet.size()]);
1335         Collections.shuffle(builders, random);
1336         final CopyOnWriteArrayList<Tuple<IndexRequestBuilder, Throwable>> errors = new CopyOnWriteArrayList<>();
1337         List<CountDownLatch> inFlightAsyncOperations = new ArrayList<>();
1338         // If you are indexing just a few documents then frequently do it one at a time.  If many then frequently in bulk.
1339         if (builders.size() < FREQUENT_BULK_THRESHOLD ? frequently() : builders.size() < ALWAYS_BULK_THRESHOLD ? rarely() : false) {
1340             if (frequently()) {
1341                 logger.info("Index [{}] docs async: [{}] bulk: [{}]", builders.size(), true, false);
1342                 for (IndexRequestBuilder indexRequestBuilder : builders) {
1343                     indexRequestBuilder.execute(new PayloadLatchedActionListener<IndexResponse, IndexRequestBuilder>(indexRequestBuilder, newLatch(inFlightAsyncOperations), errors));
1344                     postIndexAsyncActions(indices, inFlightAsyncOperations, maybeFlush);
1345                 }
1346             } else {
1347                 logger.info("Index [{}] docs async: [{}] bulk: [{}]", builders.size(), false, false);
1348                 for (IndexRequestBuilder indexRequestBuilder : builders) {
1349                     indexRequestBuilder.execute().actionGet();
1350                     postIndexAsyncActions(indices, inFlightAsyncOperations, maybeFlush);
1351                 }
1352             }
1353         } else {
1354             List<List<IndexRequestBuilder>> partition = Lists.partition(builders, Math.min(MAX_BULK_INDEX_REQUEST_SIZE,
1355                     Math.max(1, (int) (builders.size() * randomDouble()))));
1356             logger.info("Index [{}] docs async: [{}] bulk: [{}] partitions [{}]", builders.size(), false, true, partition.size());
1357             for (List<IndexRequestBuilder> segmented : partition) {
1358                 BulkRequestBuilder bulkBuilder = client().prepareBulk();
1359                 for (IndexRequestBuilder indexRequestBuilder : segmented) {
1360                     bulkBuilder.add(indexRequestBuilder);
1361                 }
1362                 BulkResponse actionGet = bulkBuilder.execute().actionGet();
1363                 assertThat(actionGet.hasFailures() ? actionGet.buildFailureMessage() : "", actionGet.hasFailures(), equalTo(false));
1364             }
1365         }
1366         for (CountDownLatch operation : inFlightAsyncOperations) {
1367             operation.await();
1368         }
1369         final List<Throwable> actualErrors = new ArrayList<>();
1370         for (Tuple<IndexRequestBuilder, Throwable> tuple : errors) {
1371             if (ExceptionsHelper.unwrapCause(tuple.v2()) instanceof EsRejectedExecutionException) {
1372                 tuple.v1().execute().actionGet(); // re-index if rejected
1373             } else {
1374                 actualErrors.add(tuple.v2());
1375             }
1376         }
1377         assertThat(actualErrors, emptyIterable());
1378         if (!bogusIds.isEmpty()) {
1379             // delete the bogus types again - it might trigger merges or at least holes in the segments and enforces deleted docs!
1380             for (Tuple<String, String> doc : bogusIds) {
1381                 // see https://github.com/elasticsearch/elasticsearch/issues/8706
1382                 final DeleteResponse deleteResponse = client().prepareDelete(doc.v1(), RANDOM_BOGUS_TYPE, doc.v2()).get();
1383                 if (deleteResponse.isFound() == false) {
1384                     logger.warn("failed to delete a dummy doc [{}][{}]", doc.v1(), doc.v2());
1385                 }
1386             }
1387         }
1388         if (forceRefresh) {
1389             assertNoFailures(client().admin().indices().prepareRefresh(indices).setIndicesOptions(IndicesOptions.lenientExpandOpen()).execute().get());
1390         }
1391     }

1413     /**
1414      * Maybe refresh, optimize, or flush then always make sure there aren't too many in flight async operations.
1415      */
1416     private void postIndexAsyncActions(String[] indices, List<CountDownLatch> inFlightAsyncOperations, boolean maybeFlush) throws InterruptedException {
1417         if (rarely()) {
1418             if (rarely()) {
1419                 client().admin().indices().prepareRefresh(indices).setIndicesOptions(IndicesOptions.lenientExpandOpen()).execute(
1420                         new LatchedActionListener<RefreshResponse>(newLatch(inFlightAsyncOperations)));
1421             } else if (maybeFlush && rarely()) {
1422                 client().admin().indices().prepareFlush(indices).setIndicesOptions(IndicesOptions.lenientExpandOpen()).execute(
1423                         new LatchedActionListener<FlushResponse>(newLatch(inFlightAsyncOperations)));
1424             } else if (rarely()) {
1425                 client().admin().indices().prepareOptimize(indices).setIndicesOptions(IndicesOptions.lenientExpandOpen()).setMaxNumSegments(between(1, 10)).setFlush(maybeFlush && randomBoolean()).execute(
1426                         new LatchedActionListener<OptimizeResponse>(newLatch(inFlightAsyncOperations)));
1427             }
1428         }
1429         while (inFlightAsyncOperations.size() > MAX_IN_FLIGHT_ASYNC_INDEXES) {
1430             int waitFor = between(0, inFlightAsyncOperations.size() - 1);
1431             inFlightAsyncOperations.remove(waitFor).await();
1432         }
