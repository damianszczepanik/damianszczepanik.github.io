20  package org.elasticsearch.benchmark.aliases;
21  
22  import org.elasticsearch.action.admin.indices.alias.IndicesAliasesRequestBuilder;
23  import org.elasticsearch.action.admin.indices.alias.get.GetAliasesResponse;
24  import org.elasticsearch.client.Client;
25  import org.elasticsearch.cluster.metadata.AliasMetaData;
26  import org.elasticsearch.common.Strings;
27  import org.elasticsearch.common.settings.ImmutableSettings;
28  import org.elasticsearch.common.settings.Settings;
29  import org.elasticsearch.indices.IndexAlreadyExistsException;
30  import org.elasticsearch.node.Node;
31  import org.elasticsearch.node.NodeBuilder;
32  
33  import java.io.IOException;
34  import java.util.List;
35  
36  /**
37   */
38  public class AliasesBenchmark {
39  
40      private final static String INDEX_NAME = "my-index";
41  
42      public static void main(String[] args) throws IOException {
43          int NUM_ADDITIONAL_NODES = 0;
44          int BASE_ALIAS_COUNT = 100000;
45          int NUM_ADD_ALIAS_REQUEST = 1000;
46  
47          Settings settings = ImmutableSettings.settingsBuilder()
48                  .put("node.master", false).build();
49          Node node1 = NodeBuilder.nodeBuilder().settings(
50                  ImmutableSettings.settingsBuilder().put(settings).put("node.master", true)
51          ).node();
52  
53          Node[] otherNodes = new Node[NUM_ADDITIONAL_NODES];
54          for (int i = 0; i < otherNodes.length; i++) {
55              otherNodes[i] = NodeBuilder.nodeBuilder().settings(settings).node();
56          }
57  
58          Client client = node1.client();
59          try {
60              client.admin().indices().prepareCreate(INDEX_NAME).execute().actionGet();
61          } catch (IndexAlreadyExistsException e) {}
62          client.admin().cluster().prepareHealth().setWaitForYellowStatus().execute().actionGet();
63          int numberOfAliases = countAliases(client);
64          System.out.println("Number of aliases: " + numberOfAliases);
65  
66          if (numberOfAliases < BASE_ALIAS_COUNT) {
67              int diff = BASE_ALIAS_COUNT - numberOfAliases;
68              System.out.println("Adding " + diff + " more aliases to get to the start amount of " + BASE_ALIAS_COUNT + " aliases");
69              IndicesAliasesRequestBuilder builder = client.admin().indices().prepareAliases();
70              for (int i = 1; i <= diff; i++) {
71                  builder.addAlias(INDEX_NAME, Strings.randomBase64UUID());
72                  if (i % 1000 == 0) {
73                      builder.execute().actionGet();
74                      builder = client.admin().indices().prepareAliases();
75                  }
76              }
77              if (!builder.request().getAliasActions().isEmpty()) {
78                  builder.execute().actionGet();
79              }
80          } else if (numberOfAliases > BASE_ALIAS_COUNT) {
81              IndicesAliasesRequestBuilder builder = client.admin().indices().prepareAliases();
82              int diff = numberOfAliases - BASE_ALIAS_COUNT;
83              System.out.println("Removing " + diff + " aliases to get to the start amount of " + BASE_ALIAS_COUNT + " aliases");
84              List<AliasMetaData> aliases= client.admin().indices().prepareGetAliases("*")
85                      .addIndices(INDEX_NAME)
86                      .execute().actionGet().getAliases().get(INDEX_NAME);
87              for (int i = 0; i <= diff; i++) {
88                  builder.removeAlias(INDEX_NAME, aliases.get(i).alias());
89                  if (i % 1000 == 0) {
90                      builder.execute().actionGet();
91                      builder = client.admin().indices().prepareAliases();
92                  }
93              }
94              if (!builder.request().getAliasActions().isEmpty()) {
95                  builder.execute().actionGet();
96              }
97          }
98  
99          numberOfAliases = countAliases(client);
100         System.out.println("Number of aliases: " + numberOfAliases);
101 
102         long totalTime = 0;
103         int max = numberOfAliases + NUM_ADD_ALIAS_REQUEST;
104         for (int i = numberOfAliases; i <= max; i++) {
105             if (i != numberOfAliases && i % 100 == 0) {
106                 long avgTime = totalTime / 100;
107                 System.out.println("Added [" + (i - numberOfAliases) + "] aliases. Avg create time: "  + avgTime + " ms");
108                 totalTime = 0;
109             }
110 
111             long time = System.currentTimeMillis();
112 //            String filter = termFilter("field" + i, "value" + i).toXContent(XContentFactory.jsonBuilder(), null).string();
113             client.admin().indices().prepareAliases().addAlias(INDEX_NAME, Strings.randomBase64UUID()/*, filter*/)
114                     .execute().actionGet();
115             totalTime += System.currentTimeMillis() - time;
116         }
117         System.out.println("Number of aliases: " + countAliases(client));
118 
119         client.close();
120         node1.close();
121         for (Node otherNode : otherNodes) {
122             otherNode.close();
123         }
124     }
125 
126     private static int countAliases(Client client) {
127         GetAliasesResponse response = client.admin().indices().prepareGetAliases("*")
128                 .addIndices(INDEX_NAME)
129                 .execute().actionGet();
130         if (response.getAliases().isEmpty()) {
131             return 0;
132         } else {
133             return response.getAliases().get(INDEX_NAME).size();
134         }
135     }
136 
137 }
