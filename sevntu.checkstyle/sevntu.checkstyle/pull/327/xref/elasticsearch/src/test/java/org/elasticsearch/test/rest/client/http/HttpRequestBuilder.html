19  package org.elasticsearch.test.rest.client.http;
20  
21  import com.google.common.base.Joiner;
22  import com.google.common.collect.Maps;
23  import org.apache.http.client.methods.*;
24  import org.apache.http.entity.StringEntity;
25  import org.apache.http.impl.client.CloseableHttpClient;
26  import org.elasticsearch.client.support.Headers;
27  import org.elasticsearch.common.Strings;
28  import org.elasticsearch.common.logging.ESLogger;
29  import org.elasticsearch.common.logging.Loggers;
30  import org.elasticsearch.common.transport.InetSocketTransportAddress;
31  import org.elasticsearch.http.HttpServerTransport;
32  
33  import java.io.IOException;
34  import java.io.UnsupportedEncodingException;
35  import java.net.URI;
36  import java.net.URISyntaxException;
37  import java.net.URLEncoder;
38  import java.nio.charset.Charset;
39  import java.util.Map;
40  
41  /**
42   * Executable builder for an http request
43   * Holds an {@link org.apache.http.client.HttpClient} that is used to send the built http request
44   */
45  public class HttpRequestBuilder {
46  
47      private static final ESLogger logger = Loggers.getLogger(HttpRequestBuilder.class);
48  
49      static final Charset DEFAULT_CHARSET = Charset.forName("utf-8");
50  
51      private final CloseableHttpClient httpClient;
52  
53      private String protocol = "http";
54  
55      private String host;
56  
57      private int port;
58  
59      private String path = "";
60  
61      private final Map<String, String> params = Maps.newHashMap();
62  
63      private final Map<String, String> headers = Maps.newHashMap();
64  
65      private String method = HttpGetWithEntity.METHOD_NAME;
66  
67      private String body;
68  
69      public HttpRequestBuilder(CloseableHttpClient httpClient) {
70          this.httpClient = httpClient;
71      }
72  
73      public HttpRequestBuilder host(String host) {
74          this.host = host;
75          return this;
76      }
77  
78      public HttpRequestBuilder httpTransport(HttpServerTransport httpServerTransport) {
79          InetSocketTransportAddress transportAddress = (InetSocketTransportAddress) httpServerTransport.boundAddress().publishAddress();
80          return host(transportAddress.address().getHostName()).port(transportAddress.address().getPort());
81      }
82  
83      public HttpRequestBuilder port(int port) {
84          this.port = port;
85          return this;
86      }
87  
88      public HttpRequestBuilder path(String path) {
89          this.path = path;
90          return this;
91      }
92  
93      public HttpRequestBuilder addParam(String name, String value) {
94          try {
95              //manually url encode params, since URI does it only partially (e.g. '+' stays as is)
96              this.params.put(name, URLEncoder.encode(value, "utf-8"));
97              return this;
98          } catch (UnsupportedEncodingException e) {
99              throw new RuntimeException(e);
100         }
101     }
102 
103     public HttpRequestBuilder addHeaders(Headers headers) {
104         for (String header : headers.headers().names()) {
105             this.headers.put(header, headers.headers().get(header));
106         }
107         return this;
108     }
109 
110     public HttpRequestBuilder addHeader(String name, String value) {
111         this.headers.put(name, value);
112         return this;
113     }
114 
115     public HttpRequestBuilder protocol(String protocol) {
116         this.protocol = protocol;
117         return this;
118     }
119 
120     public HttpRequestBuilder method(String method) {
121         this.method = method;
122         return this;
123     }
124 
125     public HttpRequestBuilder body(String body) {
126         if (Strings.hasLength(body)) {
127             this.body = body;
128         }
129         return this;
130     }
131 
132     public HttpResponse execute() throws IOException {
133         HttpUriRequest httpUriRequest = buildRequest();
134         if (logger.isTraceEnabled()) {
135             StringBuilder stringBuilder = new StringBuilder(httpUriRequest.getMethod()).append(" ").append(httpUriRequest.getURI());
136             if (Strings.hasLength(body)) {
137                 stringBuilder.append("\n").append(body);
138             }
139             logger.trace("sending request \n{}", stringBuilder.toString());
140         }
141         for (Map.Entry<String, String> entry : this.headers.entrySet()) {
142             httpUriRequest.addHeader(entry.getKey(), entry.getValue());
143         }
144         try (CloseableHttpResponse closeableHttpResponse = httpClient.execute(httpUriRequest)) {
145             HttpResponse httpResponse = new HttpResponse(httpUriRequest, closeableHttpResponse);
146             logger.trace("got response \n{}\n{}", closeableHttpResponse, httpResponse.hasBody() ? httpResponse.getBody() : "");
147             return httpResponse;
148         }
149     }
150 
151     private HttpUriRequest buildRequest() {
152 
153         if (HttpGetWithEntity.METHOD_NAME.equalsIgnoreCase(method)) {
154             return addOptionalBody(new HttpGetWithEntity(buildUri()));
155         }
156 
157         if (HttpHead.METHOD_NAME.equalsIgnoreCase(method)) {
158             checkBodyNotSupported();
159             return new HttpHead(buildUri());
160         }
161 
162         if (HttpOptions.METHOD_NAME.equalsIgnoreCase(method)) {
163             checkBodyNotSupported();
164             return new HttpOptions(buildUri());
165         }
166 
167         if (HttpDeleteWithEntity.METHOD_NAME.equalsIgnoreCase(method)) {
168             return addOptionalBody(new HttpDeleteWithEntity(buildUri()));
169         }
170 
171         if (HttpPut.METHOD_NAME.equalsIgnoreCase(method)) {
172             return addOptionalBody(new HttpPut(buildUri()));
173         }
174 
175         if (HttpPost.METHOD_NAME.equalsIgnoreCase(method)) {
176             return addOptionalBody(new HttpPost(buildUri()));
177         }
178 
179         throw new UnsupportedOperationException("method [" + method + "] not supported");
180     }
181 
182     private URI buildUri() {
183         try {
184             //url encode rules for path and query params are different. We use URI to encode the path, but we manually encode each query param through URLEncoder.
185             URI uri = new URI(protocol, null, host, port, path, null, null);
186             //String concatenation FTW. If we use the nicer multi argument URI constructor query parameters will get only partially encoded
187             //(e.g. '+' will stay as is) hence when trying to properly encode params manually they will end up double encoded (+ becomes %252B instead of %2B).
188             StringBuilder uriBuilder = new StringBuilder(protocol).append("://").append(host).append(":").append(port).append(uri.getRawPath());
189             if (params.size() > 0) {
190                 uriBuilder.append("?").append(Joiner.on('&').withKeyValueSeparator("=").join(params));
191             }
192             return URI.create(uriBuilder.toString());
193         } catch(URISyntaxException e) {
194             throw new IllegalArgumentException("unable to build uri", e);
195         }
196     }
197 
198     private HttpEntityEnclosingRequestBase addOptionalBody(HttpEntityEnclosingRequestBase requestBase) {
199         if (Strings.hasText(body)) {
200             requestBase.setEntity(new StringEntity(body, DEFAULT_CHARSET));
201         }
202         return requestBase;
203     }
204 
205     private void checkBodyNotSupported() {
206         if (Strings.hasText(body)) {
207             throw new IllegalArgumentException("request body not supported with head request");
208         }
209     }
210 
211     @Override
212     public String toString() {
213         StringBuilder stringBuilder = new StringBuilder(method).append(" '")
214                 .append(host).append(":").append(port).append(path).append("'");
215         if (!params.isEmpty()) {
216             stringBuilder.append(", params=").append(params);
217         }
218         if (Strings.hasLength(body)) {
219             stringBuilder.append(", body=\n").append(body);
220         }
221         return stringBuilder.toString();
222     }
223 }
