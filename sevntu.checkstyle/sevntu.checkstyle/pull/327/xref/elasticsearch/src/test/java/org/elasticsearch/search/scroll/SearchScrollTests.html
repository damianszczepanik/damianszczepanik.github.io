112     public void testSimpleScrollQueryThenFetchSmallSizeUnevenDistribution() throws Exception {
113         client().admin().indices().prepareCreate("test").setSettings(ImmutableSettings.settingsBuilder().put("index.number_of_shards", 3)).execute().actionGet();
114         client().admin().cluster().prepareHealth().setWaitForEvents(Priority.LANGUID).setWaitForGreenStatus().execute().actionGet();
115 
116         client().admin().cluster().prepareHealth().setWaitForEvents(Priority.LANGUID).setWaitForGreenStatus().execute().actionGet();
117 
118         for (int i = 0; i < 100; i++) {
119             String routing = "0";
120             if (i > 90) {
121                 routing = "1";
122             } else if (i > 60) {
123                 routing = "2";
124             }
125             client().prepareIndex("test", "type1", Integer.toString(i)).setSource("field", i).setRouting(routing).execute().actionGet();
126         }
127 
128         client().admin().indices().prepareRefresh().execute().actionGet();
129 
130         SearchResponse searchResponse = client().prepareSearch()
131                 .setSearchType(SearchType.QUERY_THEN_FETCH)
132                 .setQuery(matchAllQuery())
133                 .setSize(3)
134                 .setScroll(TimeValue.timeValueMinutes(2))
135                 .addSort("field", SortOrder.ASC)
136                 .execute().actionGet();
137         try {
138             long counter = 0;
139     
140             assertThat(searchResponse.getHits().getTotalHits(), equalTo(100l));
141             assertThat(searchResponse.getHits().hits().length, equalTo(3));
142             for (SearchHit hit : searchResponse.getHits()) {
143                 assertThat(((Number) hit.sortValues()[0]).longValue(), equalTo(counter++));
144             }
145     
146             for (int i = 0; i < 32; i++) {
147                 searchResponse = client().prepareSearchScroll(searchResponse.getScrollId())
148                         .setScroll(TimeValue.timeValueMinutes(2))
149                         .execute().actionGet();
150     
151                 assertThat(searchResponse.getHits().getTotalHits(), equalTo(100l));
152                 assertThat(searchResponse.getHits().hits().length, equalTo(3));
153                 for (SearchHit hit : searchResponse.getHits()) {
154                     assertThat(((Number) hit.sortValues()[0]).longValue(), equalTo(counter++));
155                 }
156             }
157     
158             // and now, the last one is one
159             searchResponse = client().prepareSearchScroll(searchResponse.getScrollId())
160                     .setScroll(TimeValue.timeValueMinutes(2))
161                     .execute().actionGet();
162     
163             assertThat(searchResponse.getHits().getTotalHits(), equalTo(100l));
164             assertThat(searchResponse.getHits().hits().length, equalTo(1));
165             for (SearchHit hit : searchResponse.getHits()) {
166                 assertThat(((Number) hit.sortValues()[0]).longValue(), equalTo(counter++));
167             }
168     
169             // a the last is zero
170             searchResponse = client().prepareSearchScroll(searchResponse.getScrollId())
171                     .setScroll(TimeValue.timeValueMinutes(2))
172                     .execute().actionGet();
173     
174             assertThat(searchResponse.getHits().getTotalHits(), equalTo(100l));
175             assertThat(searchResponse.getHits().hits().length, equalTo(0));
176             for (SearchHit hit : searchResponse.getHits()) {
177                 assertThat(((Number) hit.sortValues()[0]).longValue(), equalTo(counter++));
178             }
179             
180         } finally {
181             clearScroll(searchResponse.getScrollId());
182         }
183     }

185     @Test
186     public void testScrollAndUpdateIndex() throws Exception {
187         client().admin().indices().prepareCreate("test").setSettings(ImmutableSettings.settingsBuilder().put("index.number_of_shards", 5)).execute().actionGet();
188         client().admin().cluster().prepareHealth().setWaitForEvents(Priority.LANGUID).setWaitForGreenStatus().execute().actionGet();
189 
190         for (int i = 0; i < 500; i++) {
191             client().prepareIndex("test", "tweet", Integer.toString(i)).setSource(
192                     jsonBuilder().startObject().field("user", "kimchy").field("postDate", System.currentTimeMillis()).field("message", "test").endObject()).execute().actionGet();
193         }
194 
195         client().admin().indices().prepareRefresh().execute().actionGet();
196 
197         assertThat(client().prepareCount().setQuery(matchAllQuery()).execute().actionGet().getCount(), equalTo(500l));
198         assertThat(client().prepareCount().setQuery(termQuery("message", "test")).execute().actionGet().getCount(), equalTo(500l));
199         assertThat(client().prepareCount().setQuery(termQuery("message", "test")).execute().actionGet().getCount(), equalTo(500l));
200         assertThat(client().prepareCount().setQuery(termQuery("message", "update")).execute().actionGet().getCount(), equalTo(0l));
201         assertThat(client().prepareCount().setQuery(termQuery("message", "update")).execute().actionGet().getCount(), equalTo(0l));
202 
203         SearchResponse searchResponse = client().prepareSearch()
204                 .setQuery(queryStringQuery("user:kimchy"))
205                 .setSize(35)
206                 .setScroll(TimeValue.timeValueMinutes(2))
207                 .addSort("postDate", SortOrder.ASC)
208                 .execute().actionGet();
209         try {
210             do {
211                 for (SearchHit searchHit : searchResponse.getHits().hits()) {
212                     Map<String, Object> map = searchHit.sourceAsMap();
213                     map.put("message", "update");
214                     client().prepareIndex("test", "tweet", searchHit.id()).setSource(map).execute().actionGet();
215                 }
216                 searchResponse = client().prepareSearchScroll(searchResponse.getScrollId()).setScroll(TimeValue.timeValueMinutes(2)).execute().actionGet();
217             } while (searchResponse.getHits().hits().length > 0);
218     
219             client().admin().indices().prepareRefresh().execute().actionGet();
220             assertThat(client().prepareCount().setQuery(matchAllQuery()).execute().actionGet().getCount(), equalTo(500l));
221             assertThat(client().prepareCount().setQuery(termQuery("message", "test")).execute().actionGet().getCount(), equalTo(0l));
222             assertThat(client().prepareCount().setQuery(termQuery("message", "test")).execute().actionGet().getCount(), equalTo(0l));
223             assertThat(client().prepareCount().setQuery(termQuery("message", "update")).execute().actionGet().getCount(), equalTo(500l));
224             assertThat(client().prepareCount().setQuery(termQuery("message", "update")).execute().actionGet().getCount(), equalTo(500l));
225         } finally {
226             clearScroll(searchResponse.getScrollId());
227         }
228     }

460     @Test
461     public void testStringSortMissingAscTerminates() throws Exception {
462         assertAcked(prepareCreate("test")
463                 .setSettings(ImmutableSettings.settingsBuilder().put(IndexMetaData.SETTING_NUMBER_OF_SHARDS, 1).put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, 0))
464                 .addMapping("test", "no_field", "type=string", "some_field", "type=string"));
465         client().prepareIndex("test", "test", "1").setSource("some_field", "test").get();
466         refresh();
467 
468         SearchResponse response = client().prepareSearch("test")
469                 .setTypes("test")
470                 .addSort(new FieldSortBuilder("no_field").order(SortOrder.ASC).missing("_last"))
471                 .setScroll("1m")
472                 .get();
473         assertHitCount(response, 1);
474         assertSearchHits(response, "1");
475 
476         response = client().prepareSearchScroll(response.getScrollId()).get();
477         assertSearchResponse(response);
478         assertHitCount(response, 1);
479         assertNoSearchHits(response);
480 
481         response = client().prepareSearch("test")
482                 .setTypes("test")
483                 .addSort(new FieldSortBuilder("no_field").order(SortOrder.ASC).missing("_first"))
484                 .setScroll("1m")
485                 .get();
486         assertHitCount(response, 1);
487         assertSearchHits(response, "1");
488 
489         response = client().prepareSearchScroll(response.getScrollId()).get();
490         assertHitCount(response, 1);
491         assertThat(response.getHits().getHits().length, equalTo(0));
492     }
