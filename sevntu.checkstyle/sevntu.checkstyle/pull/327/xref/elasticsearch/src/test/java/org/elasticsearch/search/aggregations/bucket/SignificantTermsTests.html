172                 .addAggregation(new SignificantTermsBuilder("mySignificantTerms").field("description").executionHint(randomExecutionHint())

174                 .get();
175         assertSearchResponse(response);
176         topTerms = response.getAggregations().get("mySignificantTerms");
177         terms  = new HashSet<>();

293     @Test
294     public void badFilteredAnalysis() throws Exception {
295         // Deliberately using a bad choice of filter here for the background context in order
296         // to test robustness. 
297         // We search for the name of a snowboarder but use music-related content (fact_category:1)
298         // as the background source of term statistics.
299         SearchResponse response = client().prepareSearch("test")
300                 .setSearchType(SearchType.QUERY_AND_FETCH)
301                 .setQuery(new TermQueryBuilder("_all", "terje"))
302                 .setFrom(0).setSize(60).setExplain(true)                
303                 .addAggregation(new SignificantTermsBuilder("mySignificantTerms").field("description")
304                            .minDocCount(2).backgroundFilter(FilterBuilders.termFilter("fact_category", 1)))
305                 .execute()
306                 .actionGet();
307         assertSearchResponse(response);
308         SignificantTerms topTerms = response.getAggregations().get("mySignificantTerms");
309         // We expect at least one of the significant terms to have been selected on the basis
310         // that it is present in the foreground selection but entirely missing from the filtered
311         // background used as context.
312         boolean hasMissingBackgroundTerms = false;
313         for (Bucket topTerm : topTerms) {
314             if (topTerm.getSupersetDf() == 0) {
315                 hasMissingBackgroundTerms = true;
316                 break;
317             }
318         }
319         assertTrue(hasMissingBackgroundTerms);
320     }

323     @Test
324     public void filteredAnalysis() throws Exception {
325         SearchResponse response = client().prepareSearch("test")
326                 .setSearchType(SearchType.QUERY_AND_FETCH)
327                 .setQuery(new TermQueryBuilder("_all", "weller"))
328                 .setFrom(0).setSize(60).setExplain(true)                
329                 .addAggregation(new SignificantTermsBuilder("mySignificantTerms").field("description")
330                            .minDocCount(1).backgroundFilter(FilterBuilders.termsFilter("description",  "paul")))
331                 .execute()
332                 .actionGet();
333         assertSearchResponse(response);
334         SignificantTerms topTerms = response.getAggregations().get("mySignificantTerms");
335         HashSet<String> topWords = new HashSet<String>();
336         for (Bucket topTerm : topTerms) {
337             topWords.add(topTerm.getKeyAsString());
338         }
339         //The word "paul" should be a constant of all docs in the background set and therefore not seen as significant 
340         assertFalse(topWords.contains("paul"));
341         //"Weller" is the only Paul who was in The Jam and therefore this should be identified as a differentiator from the background of all other Pauls. 
342         assertTrue(topWords.contains("jam"));
343     }

345     @Test
346     public void nestedAggs() throws Exception {
347         String[][] expectedKeywordsByCategory={
348                 { "paul", "weller", "jam", "style", "council" },                
349                 { "paul", "smith" },
350                 { "craig", "kelly", "terje", "haakonsen", "burton" }};
351         SearchResponse response = client().prepareSearch("test")
352                 .setSearchType(SearchType.QUERY_AND_FETCH)
353                 .addAggregation(new TermsBuilder("myCategories").field("fact_category").minDocCount(2)
354                         .subAggregation(
355                                    new SignificantTermsBuilder("mySignificantTerms").field("description")
356                                    .executionHint(randomExecutionHint())
357                                    .minDocCount(2)))
358                 .execute()
359                 .actionGet();
360         assertSearchResponse(response);
361         Terms topCategoryTerms = response.getAggregations().get("myCategories");
362         for (org.elasticsearch.search.aggregations.bucket.terms.Terms.Bucket topCategory : topCategoryTerms.getBuckets()) {
363             SignificantTerms topTerms = topCategory.getAggregations().get("mySignificantTerms");
364             HashSet<String> foundTopWords = new HashSet<String>();
365             for (Bucket topTerm : topTerms) {
366                 foundTopWords.add(topTerm.getKeyAsString());
367             }
368             String[] expectedKeywords = expectedKeywordsByCategory[Integer.parseInt(topCategory.getKeyAsString()) - 1];
369             for (String expectedKeyword : expectedKeywords) {
370                 assertTrue(expectedKeyword + " missing from category keywords", foundTopWords.contains(expectedKeyword));
371             }
372         }
373     }

393     private void checkExpectedStringTermsFound(SignificantTerms topTerms) {
394         HashMap<String,Bucket>topWords=new HashMap<>();
395         for (Bucket topTerm : topTerms ){
396             topWords.put(topTerm.getKeyAsString(), topTerm);
397         }
398         assertTrue( topWords.containsKey("haakonsen"));
399         assertTrue( topWords.containsKey("craig"));
400         assertTrue( topWords.containsKey("kelly"));
401         assertTrue( topWords.containsKey("burton"));
402         assertTrue( topWords.containsKey("snowboards"));
403         Bucket kellyTerm=topWords.get("kelly");
404         assertEquals(3, kellyTerm.getSubsetDf());
405         assertEquals(4, kellyTerm.getSupersetDf());
406     }
