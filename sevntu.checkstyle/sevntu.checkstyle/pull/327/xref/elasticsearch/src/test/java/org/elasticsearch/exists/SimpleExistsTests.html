35      @Test
36      public void testExistsRandomPreference() throws Exception {
37          createIndex("test");
38          indexRandom(true, client().prepareIndex("test", "type", "1").setSource("field", "value"),
39                  client().prepareIndex("test", "type", "2").setSource("field", "value"),
40                  client().prepareIndex("test", "type", "3").setSource("field", "value"),
41                  client().prepareIndex("test", "type", "4").setSource("field", "value"),
42                  client().prepareIndex("test", "type", "5").setSource("field", "value"),
43                  client().prepareIndex("test", "type", "6").setSource("field", "value"));
44  
45          int iters = scaledRandomIntBetween(10, 100);
46          for (int i = 0; i < iters; i++) {
47  
48              String randomPreference = randomUnicodeOfLengthBetween(0, 4);
49              // randomPreference should not start with '_' (reserved for known preference types (e.g. _shards, _primary)
50              while (randomPreference.startsWith("_")) {
51                  randomPreference = randomUnicodeOfLengthBetween(0, 4);
52              }
53              // id is not indexed, but lets see that we automatically convert to
54              ExistsResponse existsResponse = client().prepareExists().setQuery(QueryBuilders.matchAllQuery()).setPreference(randomPreference).get();
55              assertExists(existsResponse, true);
56          }
57      }

60      @Test
61      public void simpleIpTests() throws Exception {
62          createIndex("test");
63  
64          client().admin().indices().preparePutMapping("test").setType("type1")
65                  .setSource(XContentFactory.jsonBuilder().startObject().startObject("type1").startObject("properties")
66                          .startObject("from").field("type", "ip").endObject()
67                          .startObject("to").field("type", "ip").endObject()
68                          .endObject().endObject().endObject())
69                  .execute().actionGet();
70  
71          client().prepareIndex("test", "type1", "1").setSource("from", "192.168.0.5", "to", "192.168.0.10").setRefresh(true).execute().actionGet();
72  
73          ExistsResponse existsResponse = client().prepareExists()
74                  .setQuery(boolQuery().must(rangeQuery("from").lt("192.168.0.7")).must(rangeQuery("to").gt("192.168.0.7"))).get();
75  
76          assertExists(existsResponse, true);
77  
78          existsResponse = client().prepareExists().setQuery(boolQuery().must(rangeQuery("from").lt("192.168.0.4")).must(rangeQuery("to").gt("192.168.0.11"))).get();
79  
80          assertExists(existsResponse, false);
81      }

83      @Test
84      public void simpleIdTests() {
85          createIndex("test");
86  
87          client().prepareIndex("test", "type", "XXX1").setSource("field", "value").setRefresh(true).execute().actionGet();
88          // id is not indexed, but lets see that we automatically convert to
89          ExistsResponse existsResponse = client().prepareExists().setQuery(QueryBuilders.termQuery("_id", "XXX1")).execute().actionGet();
90          assertExists(existsResponse, true);
91  
92          existsResponse = client().prepareExists().setQuery(QueryBuilders.queryStringQuery("_id:XXX1")).execute().actionGet();
93          assertExists(existsResponse, true);
94  
95          existsResponse = client().prepareExists().setQuery(QueryBuilders.prefixQuery("_id", "XXX")).execute().actionGet();
96          assertExists(existsResponse, true);
97  
98          existsResponse = client().prepareExists().setQuery(QueryBuilders.queryStringQuery("_id:XXX*").lowercaseExpandedTerms(false)).execute().actionGet();
99          assertExists(existsResponse, true);
100     }

103     public void simpleNonExistenceTests() throws Exception {

105         client().prepareIndex("test", "type1", "1").setSource("field", 2).execute().actionGet();
106         client().prepareIndex("test", "type1", "2").setSource("field", 5).execute().actionGet();
107         client().prepareIndex("test", "type", "XXX1").setSource("field", "value").execute().actionGet();
108         ensureGreen();
109         refresh();
110         ExistsResponse existsResponse = client().prepareExists("test").setQuery(QueryBuilders.rangeQuery("field").gte(6).lte(8)).execute().actionGet();
111         assertExists(existsResponse, false);
112 
113         existsResponse = client().prepareExists("test").setQuery(QueryBuilders.queryStringQuery("_id:XXY*").lowercaseExpandedTerms(false)).execute().actionGet();
114         assertExists(existsResponse, false);
