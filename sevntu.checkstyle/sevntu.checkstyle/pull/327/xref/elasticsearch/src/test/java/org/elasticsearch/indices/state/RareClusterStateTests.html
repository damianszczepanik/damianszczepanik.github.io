65      @Test
66      public void testUnassignedShardAndEmptyNodesInRoutingTable() throws Exception {
67          internalCluster().startNode();
68          createIndex("a");
69          ensureSearchable("a");
70          ClusterState current = clusterService().state();
71          GatewayAllocator allocator = internalCluster().getInstance(GatewayAllocator.class);
72  
73          AllocationDeciders allocationDeciders = new AllocationDeciders(ImmutableSettings.EMPTY, new AllocationDecider[0]);
74          RoutingNodes routingNodes = new RoutingNodes(
75                  ClusterState.builder(current)
76                          .routingTable(RoutingTable.builder(current.routingTable()).remove("a").addAsRecovery(current.metaData().index("a")))
77                          .nodes(DiscoveryNodes.EMPTY_NODES)
78                          .build()
79          );
80          ClusterInfo clusterInfo = new ClusterInfo(ImmutableMap.<String, DiskUsage>of(), ImmutableMap.<String, Long>of());
81  
82          RoutingAllocation routingAllocation = new RoutingAllocation(allocationDeciders, routingNodes, current.nodes(), clusterInfo);
83          allocator.allocateUnassigned(routingAllocation);
84      }

86      @Test
87      @TestLogging(value = "cluster.service:TRACE")
88      public void testDeleteCreateInOneBulk() throws Exception {
89          internalCluster().startNodesAsync(2, ImmutableSettings.builder()
90                  .put(DiscoveryModule.DISCOVERY_TYPE_KEY, "zen")
91                  .build()).get();
92          assertFalse(client().admin().cluster().prepareHealth().setWaitForNodes("2").get().isTimedOut());
93          prepareCreate("test").setSettings(IndexMetaData.SETTING_AUTO_EXPAND_REPLICAS, true).addMapping("type").get();
94          ensureGreen("test");
95  
96          // now that the cluster is stable, remove publishing timeout
97          assertAcked(client().admin().cluster().prepareUpdateSettings().setTransientSettings(ImmutableSettings.builder().put(DiscoverySettings.PUBLISH_TIMEOUT, "0")));
98  
99          Set<String> nodes = new HashSet<>(Arrays.asList(internalCluster().getNodeNames()));
100         nodes.remove(internalCluster().getMasterName());
101 
102         // block none master node.
103         BlockClusterStateProcessing disruption = new BlockClusterStateProcessing(nodes.iterator().next(), getRandom());
104         internalCluster().setDisruptionScheme(disruption);
105         logger.info("--> indexing a doc");
106         index("test", "type", "1");
107         refresh();
108         disruption.startDisrupting();
109         logger.info("--> delete index and recreate it");
110         assertFalse(client().admin().indices().prepareDelete("test").setTimeout("200ms").get().isAcknowledged());
111         assertFalse(prepareCreate("test").setTimeout("200ms").setSettings(IndexMetaData.SETTING_AUTO_EXPAND_REPLICAS, true).get().isAcknowledged());
112         logger.info("--> letting cluster proceed");
113         disruption.stopDisrupting();
114         ensureGreen(TimeValue.timeValueMinutes(30), "test");
115         assertHitCount(client().prepareSearch("test").get(), 0);
116     }
