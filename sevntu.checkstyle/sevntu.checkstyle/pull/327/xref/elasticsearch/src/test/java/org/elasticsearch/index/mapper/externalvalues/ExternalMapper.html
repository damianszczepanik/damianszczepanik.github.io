56  /**
57   * This mapper add a new sub fields
58   * .bin Binary type
59   * .bool Boolean type
60   * .point GeoPoint type
61   * .shape GeoShape type
62   */
63  public class ExternalMapper extends AbstractFieldMapper<Object> {
64      /**
65       * Returns the actual value of the field.
66       *
67       * @param value
68       */
69      @Override
70      public Object value(Object value) {
71          return null;
72      }

74      public static class Names {
75          public static final String FIELD_BIN = "bin";
76          public static final String FIELD_BOOL = "bool";
77          public static final String FIELD_POINT = "point";
78          public static final String FIELD_SHAPE = "shape";
79      }
80  
81      public static class Builder extends AbstractFieldMapper.Builder<Builder, ExternalMapper> {
82  
83          private BinaryFieldMapper.Builder binBuilder = new BinaryFieldMapper.Builder(Names.FIELD_BIN);
84          private BooleanFieldMapper.Builder boolBuilder = new BooleanFieldMapper.Builder(Names.FIELD_BOOL);
85          private GeoPointFieldMapper.Builder pointBuilder = new GeoPointFieldMapper.Builder(Names.FIELD_POINT);
86          private GeoShapeFieldMapper.Builder shapeBuilder = new GeoShapeFieldMapper.Builder(Names.FIELD_SHAPE);
87          private Mapper.Builder stringBuilder;
88          private String generatedValue;
89          private String mapperName;
90  
91          public Builder(String name, String generatedValue, String mapperName) {
92              super(name, new FieldType(Defaults.FIELD_TYPE));
93              this.builder = this;
94              this.stringBuilder = stringField(name).store(false);
95              this.generatedValue = generatedValue;
96              this.mapperName = mapperName;
97          }
98  
99          public Builder string(Mapper.Builder content) {
100             this.stringBuilder = content;
101             return this;
102         }
103 
104         @Override
105         public ExternalMapper build(BuilderContext context) {
106             ContentPath.Type origPathType = context.path().pathType();
107             context.path().pathType(ContentPath.Type.FULL);
108 
109             context.path().add(name);
110             BinaryFieldMapper binMapper = binBuilder.build(context);
111             BooleanFieldMapper boolMapper = boolBuilder.build(context);
112             GeoPointFieldMapper pointMapper = pointBuilder.build(context);
113             GeoShapeFieldMapper shapeMapper = shapeBuilder.build(context);
114             Mapper stringMapper = stringBuilder.build(context);
115             context.path().remove();
116 
117             context.path().pathType(origPathType);
118 
119             return new ExternalMapper(buildNames(context), generatedValue, mapperName, binMapper, boolMapper, pointMapper, shapeMapper, stringMapper,
120                     context.indexSettings(), multiFieldsBuilder.build(this, context), copyTo);
121         }
122     }
123 
124     public static class TypeParser implements Mapper.TypeParser {
125 
126         private String generatedValue;
127         private String mapperName;
128 
129         TypeParser(String mapperName, String generatedValue) {
130             this.mapperName = mapperName;
131             this.generatedValue = generatedValue;
132         }
133 
134         @SuppressWarnings({"unchecked"})
135         @Override
136         public Mapper.Builder parse(String name, Map<String, Object> node, ParserContext parserContext) throws MapperParsingException {
137             ExternalMapper.Builder builder = new ExternalMapper.Builder(name, generatedValue, mapperName);
138             parseField(builder, name, node, parserContext);
139             for (Iterator<Map.Entry<String, Object>> iterator = node.entrySet().iterator(); iterator.hasNext();) {
140                 Map.Entry<String, Object> entry = iterator.next();
141                 String propName = Strings.toUnderscoreCase(entry.getKey());
142                 Object propNode = entry.getValue();
143 
144                 if (parseMultiField(builder, name, parserContext, propName, propNode)) {
145                     iterator.remove();
146                 }
147             }
148 
149             return builder;
150         }
151     }

153     private final String generatedValue;
154     private final String mapperName;

157     private final BooleanFieldMapper boolMapper;
158     private final GeoPointFieldMapper pointMapper;
159     private final GeoShapeFieldMapper shapeMapper;
160     private final Mapper stringMapper;
161 
162     public ExternalMapper(FieldMapper.Names names,
163                           String generatedValue, String mapperName,
164                           BinaryFieldMapper binMapper, BooleanFieldMapper boolMapper, GeoPointFieldMapper pointMapper,
165                           GeoShapeFieldMapper shapeMapper, Mapper stringMapper, Settings indexSettings, MultiFields multiFields, CopyTo copyTo) {
166         super(names, 1.0f, Defaults.FIELD_TYPE, false, null, null, null, null, null, indexSettings,
167                 multiFields, copyTo);
168         this.generatedValue = generatedValue;
169         this.mapperName = mapperName;
170         this.binMapper = binMapper;
171         this.boolMapper = boolMapper;
172         this.pointMapper = pointMapper;
173         this.shapeMapper = shapeMapper;
174         this.stringMapper = stringMapper;
175     }
176 
177     @Override
178     public FieldType defaultFieldType() {
179         return Defaults.FIELD_TYPE;
180     }
181 
182     @Override
183     public FieldDataType defaultFieldDataType() {
184         return null;
185     }
186 
187     @Override
188     public void parse(ParseContext context) throws IOException {
189         byte[] bytes = "Hello world".getBytes(Charset.defaultCharset());
190         binMapper.parse(context.createExternalValueContext(bytes));
191 
192         boolMapper.parse(context.createExternalValueContext(true));
193 
194         // Let's add a Dummy Point
195         Double lat = 42.0;
196         Double lng = 51.0;
197         GeoPoint point = new GeoPoint(lat, lng);
198         pointMapper.parse(context.createExternalValueContext(point));
199 
200         // Let's add a Dummy Shape
201         Point shape = ShapeBuilder.newPoint(-100, 45).build();
202         shapeMapper.parse(context.createExternalValueContext(shape));
203 
204         context = context.createExternalValueContext(generatedValue);
205 
206         // Let's add a Original String
207         stringMapper.parse(context);
208 
209         multiFields.parse(this, context);
210         if (copyTo != null) {
211             copyTo.parse(context);
212         }
213     }
