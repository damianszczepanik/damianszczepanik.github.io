339     @Test
340     public void testNormalizePoint_outsideNormalRange_withOptions() {
341         for (int i = 0; i < 100; i++) {
342             boolean normLat = randomBoolean();
343             boolean normLon = randomBoolean();
344             double normalisedLat = (randomDouble() * 180.0) - 90.0;
345             double normalisedLon = (randomDouble() * 360.0) - 180.0;
346             int shiftLat = randomIntBetween(1, 10000);
347             int shiftLon = randomIntBetween(1, 10000);
348             double testLat = normalisedLat + (180.0 * shiftLat);
349             double testLon = normalisedLon + (360.0 * shiftLon);
350 
351             double expectedLat;
352             double expectedLon;
353             if (normLat) {
354                 expectedLat = normalisedLat * (shiftLat % 2 == 0 ? 1 : -1);
355             } else {
356                 expectedLat = testLat;
357             }
358             if (normLon) {
359                 expectedLon = normalisedLon + ((normLat && shiftLat % 2 == 1) ? 180 : 0);
360                 if (expectedLon > 180.0) {
361                     expectedLon -= 360;
362                 }
363             } else {
364                 double shiftValue = normalisedLon > 0 ? -180 : 180;
365                 expectedLon = testLon + ((normLat && shiftLat % 2 == 1) ? shiftValue : 0);
366             }
367             GeoPoint testPoint = new GeoPoint(testLat, testLon);
368             GeoPoint expectedPoint = new GeoPoint(expectedLat, expectedLon);
369             GeoUtils.normalizePoint(testPoint, normLat, normLon);
370             assertThat("Unexpected Latitude", testPoint.lat(), closeTo(expectedPoint.lat(), MAX_ACCEPTABLE_ERROR));
371             assertThat("Unexpected Longitude", testPoint.lon(), closeTo(expectedPoint.lon(), MAX_ACCEPTABLE_ERROR));
372         }

565     public void testPrefixTreeCellSizes() {
566         assertThat(GeoUtils.EARTH_SEMI_MAJOR_AXIS, equalTo(DistanceUtils.EARTH_EQUATORIAL_RADIUS_KM * 1000));
567         assertThat(GeoUtils.quadTreeCellWidth(0), lessThanOrEqualTo(GeoUtils.EARTH_EQUATOR));
568 
569         SpatialContext spatialContext = new SpatialContext(true);
570 
571         GeohashPrefixTree geohashPrefixTree = new GeohashPrefixTree(spatialContext, GeohashPrefixTree.getMaxLevelsPossible() / 2);
572         Cell gNode = geohashPrefixTree.getWorldCell();
573 
574         for (int i = 0; i < geohashPrefixTree.getMaxLevels(); i++) {
575             double width = GeoUtils.geoHashCellWidth(i);
576             double height = GeoUtils.geoHashCellHeight(i);
577             double size = GeoUtils.geoHashCellSize(i);
578             double degrees = 360.0 * width / GeoUtils.EARTH_EQUATOR;
579             int level = GeoUtils.quadTreeLevelsForPrecision(size);
580 
581             assertThat(GeoUtils.quadTreeCellWidth(level), lessThanOrEqualTo(width));
582             assertThat(GeoUtils.quadTreeCellHeight(level), lessThanOrEqualTo(height));
583             assertThat(GeoUtils.geoHashLevelsForPrecision(size), equalTo(geohashPrefixTree.getLevelForDistance(degrees)));
584 
585             assertThat("width at level " + i, gNode.getShape().getBoundingBox().getWidth(), equalTo(360.d * width / GeoUtils.EARTH_EQUATOR));
586             assertThat("height at level " + i, gNode.getShape().getBoundingBox().getHeight(), equalTo(180.d * height
587                     / GeoUtils.EARTH_POLAR_DISTANCE));
588 
589             gNode = gNode.getNextLevelCells(null).next();
590         }
591 
592         QuadPrefixTree quadPrefixTree = new QuadPrefixTree(spatialContext);
593         Cell qNode = quadPrefixTree.getWorldCell();
594         for (int i = 0; i < quadPrefixTree.getMaxLevels(); i++) {
595 
596             double degrees = 360.0 / (1L << i);
597             double width = GeoUtils.quadTreeCellWidth(i);
598             double height = GeoUtils.quadTreeCellHeight(i);
599             double size = GeoUtils.quadTreeCellSize(i);
600             int level = GeoUtils.quadTreeLevelsForPrecision(size);
601 
602             assertThat(GeoUtils.quadTreeCellWidth(level), lessThanOrEqualTo(width));
603             assertThat(GeoUtils.quadTreeCellHeight(level), lessThanOrEqualTo(height));
604             assertThat(GeoUtils.quadTreeLevelsForPrecision(size), equalTo(quadPrefixTree.getLevelForDistance(degrees)));
605 
606             assertThat("width at level " + i, qNode.getShape().getBoundingBox().getWidth(), equalTo(360.d * width / GeoUtils.EARTH_EQUATOR));
607             assertThat("height at level " + i, qNode.getShape().getBoundingBox().getHeight(), equalTo(180.d * height
608                     / GeoUtils.EARTH_POLAR_DISTANCE));
609 
610             qNode = qNode.getNextLevelCells(null).next();
611         }

614     private static void assertNormalizedPoint(GeoPoint input, GeoPoint expected) {
615         GeoUtils.normalizePoint(input);
616         if (Double.isNaN(expected.lat())) {
617             assertThat("Unexpected Latitude", input.lat(), equalTo(expected.lat()));
618         } else {
619             assertThat("Unexpected Latitude", input.lat(), closeTo(expected.lat(), MAX_ACCEPTABLE_ERROR));
620         }
621         if (Double.isNaN(expected.lon())) {
622             assertThat("Unexpected Longitude", input.lon(), equalTo(expected.lon()));
623         } else {
624             assertThat("Unexpected Longitude", input.lon(), closeTo(expected.lon(), MAX_ACCEPTABLE_ERROR));
625         }
626     }
