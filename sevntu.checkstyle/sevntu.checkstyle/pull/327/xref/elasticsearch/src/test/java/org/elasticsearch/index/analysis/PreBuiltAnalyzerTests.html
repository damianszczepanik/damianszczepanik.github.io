56      @Test
57      public void testThatDefaultAndStandardAnalyzerChangedIn10Beta1() throws IOException {
58          Analyzer currentStandardAnalyzer = PreBuiltAnalyzers.STANDARD.getAnalyzer(Version.V_1_0_0_Beta1);
59          Analyzer currentDefaultAnalyzer = PreBuiltAnalyzers.DEFAULT.getAnalyzer(Version.V_1_0_0_Beta1);
60  
61          // special case, these two are the same instance
62          assertThat(currentDefaultAnalyzer, is(currentStandardAnalyzer));
63          PreBuiltAnalyzers.DEFAULT.getAnalyzer(Version.V_1_0_0_Beta1);
64          final int n = scaledRandomIntBetween(10, 100);
65          Version version = Version.CURRENT;
66          for(int i = 0; i < n; i++) {
67              if (version.equals(Version.V_1_0_0_Beta1)) {
68                  assertThat(currentDefaultAnalyzer, is(PreBuiltAnalyzers.DEFAULT.getAnalyzer(version)));
69              } else {
70                  assertThat(currentDefaultAnalyzer, not(is(PreBuiltAnalyzers.DEFAULT.getAnalyzer(version))));
71              }
72              Analyzer analyzer = PreBuiltAnalyzers.DEFAULT.getAnalyzer(version);
73              TokenStream ts = analyzer.tokenStream("foo", "This is it Dude");
74              ts.reset();
75              CharTermAttribute charTermAttribute = ts.addAttribute(CharTermAttribute.class);
76              List<String> list = new ArrayList<>();
77              while(ts.incrementToken()) {
78                  list.add(charTermAttribute.toString());
79              }
80              if (version.onOrAfter(Version.V_1_0_0_Beta1)) {
81                  assertThat(list.size(), is(4));
82                  assertThat(list, contains("this", "is", "it", "dude"));
83  
84              } else {
85                  assertThat(list.size(), is(1));
86                  assertThat(list, contains("dude"));
87              }
88              ts.close();
89              version = randomVersion();
90          }
91      }

93      @Test
94      public void testAnalyzerChangedIn10RC1() throws IOException {
95          Analyzer pattern = PreBuiltAnalyzers.PATTERN.getAnalyzer(Version.V_1_0_0_RC1);
96          Analyzer standardHtml = PreBuiltAnalyzers.STANDARD_HTML_STRIP.getAnalyzer(Version.V_1_0_0_RC1);
97          final int n = scaledRandomIntBetween(10, 100);
98          Version version = Version.CURRENT;
99          for(int i = 0; i < n; i++) {
100             if (version.equals(Version.V_1_0_0_RC1)) {
101                 assertThat(pattern, is(PreBuiltAnalyzers.PATTERN.getAnalyzer(version)));
102                 assertThat(standardHtml, is(PreBuiltAnalyzers.STANDARD_HTML_STRIP.getAnalyzer(version)));
103             } else {
104                 assertThat(pattern, not(is(PreBuiltAnalyzers.DEFAULT.getAnalyzer(version))));
105                 assertThat(standardHtml, not(is(PreBuiltAnalyzers.DEFAULT.getAnalyzer(version))));
106             }
107             Analyzer analyzer = randomBoolean() ? PreBuiltAnalyzers.PATTERN.getAnalyzer(version) :  PreBuiltAnalyzers.STANDARD_HTML_STRIP.getAnalyzer(version);
108             TokenStream ts = analyzer.tokenStream("foo", "This is it Dude");
109             ts.reset();
110             CharTermAttribute charTermAttribute = ts.addAttribute(CharTermAttribute.class);
111             List<String> list = new ArrayList<>();
112             while(ts.incrementToken()) {
113                 list.add(charTermAttribute.toString());
114             }
115             if (version.onOrAfter(Version.V_1_0_0_RC1)) {
116                 assertThat(list.toString(), list.size(), is(4));
117                 assertThat(list, contains("this", "is", "it", "dude"));
118 
119             } else {
120                 assertThat(list.size(), is(1));
121                 assertThat(list, contains("dude"));
122             }
123             ts.close();
124             version = randomVersion();
125         }
126     }

149     @Test
150     public void testThatAnalyzersAreUsedInMapping() throws IOException {
151         int randomInt = randomInt(PreBuiltAnalyzers.values().length-1);
152         PreBuiltAnalyzers randomPreBuiltAnalyzer = PreBuiltAnalyzers.values()[randomInt];
153         String analyzerName = randomPreBuiltAnalyzer.name().toLowerCase(Locale.ROOT);
154 
155         Version randomVersion = randomVersion();
156         Settings indexSettings = ImmutableSettings.settingsBuilder().put(IndexMetaData.SETTING_VERSION_CREATED, randomVersion).build();
157 
158         NamedAnalyzer namedAnalyzer = new PreBuiltAnalyzerProvider(analyzerName, AnalyzerScope.INDEX, randomPreBuiltAnalyzer.getAnalyzer(randomVersion)).get();
159 
160         String mapping = XContentFactory.jsonBuilder().startObject().startObject("type")
161                 .startObject("properties").startObject("field").field("type", "string").field("analyzer", analyzerName).endObject().endObject()
162                 .endObject().endObject().string();
163         DocumentMapper docMapper = createIndex("test", indexSettings).mapperService().documentMapperParser().parse(mapping);
164 
165         FieldMapper fieldMapper = docMapper.mappers().name("field").mapper();
166         assertThat(fieldMapper.searchAnalyzer(), instanceOf(NamedAnalyzer.class));
167         NamedAnalyzer fieldMapperNamedAnalyzer = (NamedAnalyzer) fieldMapper.searchAnalyzer();
168 
169         assertThat(fieldMapperNamedAnalyzer.analyzer(), is(namedAnalyzer.analyzer()));
170     }
