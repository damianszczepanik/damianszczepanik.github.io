57      @Override
58      public void setupSuiteScopeCluster() throws Exception {
59  
60          prepareCreate("idx").addMapping("type",
61                  jsonBuilder().startObject().startObject("type").startObject("properties")
62                      .startObject("str_value")
63                          .field("type", "string")
64                          .startObject("fields")
65                              .startObject("hash")
66                                  .field("type", "murmur3")
67                              .endObject()
68                          .endObject()
69                      .endObject()
70                      .startObject("str_values")
71                          .field("type", "string")
72                          .startObject("fields")
73                              .startObject("hash")
74                                  .field("type", "murmur3")
75                              .endObject()
76                          .endObject()
77                      .endObject()
78                      .startObject("l_value")
79                          .field("type", "long")
80                          .startObject("fields")
81                              .startObject("hash")
82                                  .field("type", "murmur3")
83                              .endObject()
84                          .endObject()
85                      .endObject()
86                      .startObject("l_values")
87                          .field("type", "long")
88                          .startObject("fields")
89                              .startObject("hash")
90                                  .field("type", "murmur3")
91                              .endObject()
92                          .endObject()
93                      .endObject()
94                          .startObject("d_value")
95                              .field("type", "double")
96                              .startObject("fields")
97                                  .startObject("hash")
98                                      .field("type", "murmur3")
99                                  .endObject()
100                             .endObject()
101                         .endObject()
102                         .startObject("d_values")
103                             .field("type", "double")
104                                 .startObject("fields")
105                                     .startObject("hash")
106                                         .field("type", "murmur3")
107                                     .endObject()
108                                 .endObject()
109                             .endObject()
110                         .endObject()
111                         .endObject().endObject()).execute().actionGet();
112 
113         numDocs = randomIntBetween(2, 100);
114         precisionThreshold = randomIntBetween(0, 1 << randomInt(20));
115         IndexRequestBuilder[] builders = new IndexRequestBuilder[(int) numDocs];
116         for (int i = 0; i < numDocs; ++i) {
117             builders[i] = client().prepareIndex("idx", "type").setSource(jsonBuilder()
118                     .startObject()
119                         .field("str_value", "s" + i)
120                         .field("str_values", new String[]{"s" + (i * 2), "s" + (i * 2 + 1)})
121                         .field("l_value", i)
122                         .field("l_values", new int[] {i * 2, i * 2 + 1})
123                         .field("d_value", i)
124                         .field("d_values", new double[]{i * 2, i * 2 + 1})
125                     .endObject());
126         }
127         indexRandom(true, builders);
128         createIndex("idx_unmapped");
129 
130         IndexRequestBuilder[] dummyDocsBuilder = new IndexRequestBuilder[10];
131         for (int i = 0; i < dummyDocsBuilder.length; i++) {
132             dummyDocsBuilder[i] = client().prepareIndex("idx", "type").setSource("a_field", "1");
133         }
134         indexRandom(true, dummyDocsBuilder);
135 
136         ensureSearchable();
137     }

139     private void assertCount(Cardinality count, long value) {
140         if (value <= precisionThreshold) {
141             // linear counting should be picked, and should be accurate
142             assertEquals(value, count.getValue());
143         } else {
144             // error is not bound, so let's just make sure it is > 0
145             assertThat(count.getValue(), greaterThan(0L));
146         }
147     }
148      private String singleNumericField(boolean hash) {
149         return (randomBoolean() ? "l_value" : "d_value") + (hash ? ".hash" : "");
150     }
151 
152     private String multiNumericField(boolean hash) {
153         return (randomBoolean() ? "l_values" : "d_values") + (hash ? ".hash" : "");
154     }

156     @Test
157     public void unmapped() throws Exception {
158         SearchResponse response = client().prepareSearch("idx_unmapped").setTypes("type")
159                 .addAggregation(cardinality("cardinality").precisionThreshold(precisionThreshold).field("str_value"))
160                 .execute().actionGet();
161 
162         assertSearchResponse(response);
163 
164         Cardinality count = response.getAggregations().get("cardinality");
165         assertThat(count, notNullValue());
166         assertThat(count.getName(), equalTo("cardinality"));
167         assertCount(count, 0);
168     }

226     @Test
227     public void singleValuedNumeric_getProperty() throws Exception {
228 
229         SearchResponse searchResponse = client().prepareSearch("idx").setQuery(matchAllQuery())
230                 .addAggregation(
231                         global("global").subAggregation(
232                                 cardinality("cardinality").precisionThreshold(precisionThreshold).field(singleNumericField(false))))
233                 .execute().actionGet();
234 
235         assertSearchResponse(searchResponse);
236 
237         Global global = searchResponse.getAggregations().get("global");
238         assertThat(global, notNullValue());
239         assertThat(global.getName(), equalTo("global"));
240         // assertThat(global.getDocCount(), equalTo(numDocs));
241         assertThat(global.getAggregations(), notNullValue());
242         assertThat(global.getAggregations().asMap().size(), equalTo(1));
243 
244         Cardinality cardinality = global.getAggregations().get("cardinality");
245         assertThat(cardinality, notNullValue());
246         assertThat(cardinality.getName(), equalTo("cardinality"));
247         long expectedValue = numDocs;
248         assertCount(cardinality, expectedValue);
249         assertThat((Cardinality) global.getProperty("cardinality"), equalTo(cardinality));
250         assertThat((double) global.getProperty("cardinality.value"), equalTo((double) cardinality.getValue()));
251         assertThat((double) cardinality.getProperty("value"), equalTo((double) cardinality.getValue()));
252     }

437     public void asSubAgg() throws Exception {
438         SearchResponse response = client().prepareSearch("idx").setTypes("type")
439                 .addAggregation(terms("terms").field("str_value")
440                         .collectMode(randomFrom(SubAggCollectionMode.values()))
441                         .subAggregation(cardinality("cardinality").precisionThreshold(precisionThreshold).field("str_values")))
442                 .execute().actionGet();
443 
444         assertSearchResponse(response);
445 
446         Terms terms = response.getAggregations().get("terms");

448             Cardinality count = bucket.getAggregations().get("cardinality");
449             assertThat(count, notNullValue());

451             assertCount(count, 2);
452         }
