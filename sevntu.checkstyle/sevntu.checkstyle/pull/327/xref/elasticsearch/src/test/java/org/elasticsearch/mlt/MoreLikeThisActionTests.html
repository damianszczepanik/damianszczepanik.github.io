213     @Test
214     // See issue https://github.com/elasticsearch/elasticsearch/issues/3252
215     public void testNumericField() throws Exception {
216         prepareCreate("test").addMapping("type", jsonBuilder()
217                     .startObject().startObject("type")
218                     .startObject("properties")
219                         .startObject("int_value").field("type", randomNumericType(getRandom())).endObject()
220                         .startObject("string_value").field("type", "string").endObject()
221                         .endObject()
222                     .endObject().endObject()).execute().actionGet();
223         ensureGreen();
224         client().prepareIndex("test", "type", "1")
225                 .setSource(jsonBuilder().startObject().field("string_value", "lucene index").field("int_value", 1).endObject())
226                 .execute().actionGet();
227         client().prepareIndex("test", "type", "2")
228                 .setSource(jsonBuilder().startObject().field("string_value", "elasticsearch index").field("int_value", 42).endObject())
229                 .execute().actionGet();
230 
231         refresh();
232 
233         // Implicit list of fields -> ignore numeric fields
234         SearchResponse searchResponse = client().prepareMoreLikeThis("test", "type", "1").setMinDocFreq(1).setMinTermFreq(1).execute().actionGet();
235         assertHitCount(searchResponse, 1l);
236 
237         // Explicit list of fields including numeric fields -> fail
238         assertThrows(client().prepareMoreLikeThis("test", "type", "1").setField("string_value", "int_value"), SearchPhaseExecutionException.class);
239 
240         // mlt query with no field -> OK
241         searchResponse = client().prepareSearch().setQuery(moreLikeThisQuery().likeText("index").minTermFreq(1).minDocFreq(1)).execute().actionGet();
242         assertHitCount(searchResponse, 2l);
243 
244         // mlt query with string fields
245         searchResponse = client().prepareSearch().setQuery(moreLikeThisQuery("string_value").likeText("index").minTermFreq(1).minDocFreq(1)).execute().actionGet();
246         assertHitCount(searchResponse, 2l);
247 
248         // mlt query with at least a numeric field -> fail by default
249         assertThrows(client().prepareSearch().setQuery(moreLikeThisQuery("string_value", "int_value").likeText("index")), SearchPhaseExecutionException.class);
250 
251         // mlt query with at least a numeric field -> fail by command
252         assertThrows(client().prepareSearch().setQuery(moreLikeThisQuery("string_value", "int_value").likeText("index").failOnUnsupportedField(true)), SearchPhaseExecutionException.class);
253 
254 
255         // mlt query with at least a numeric field but fail_on_unsupported_field set to false
256         searchResponse = client().prepareSearch().setQuery(moreLikeThisQuery("string_value", "int_value").likeText("index").minTermFreq(1).minDocFreq(1).failOnUnsupportedField(false)).get();
257         assertHitCount(searchResponse, 2l);
258 
259         // mlt field query on a numeric field -> failure by default
260         assertThrows(client().prepareSearch().setQuery(moreLikeThisQuery("int_value").likeText("42").minTermFreq(1).minDocFreq(1)), SearchPhaseExecutionException.class);
261 
262         // mlt field query on a numeric field -> failure by command
263         assertThrows(client().prepareSearch().setQuery(moreLikeThisQuery("int_value").likeText("42").minTermFreq(1).minDocFreq(1).failOnUnsupportedField(true)),
264                 SearchPhaseExecutionException.class);
265 
266         // mlt field query on a numeric field but fail_on_unsupported_field set to false
267         searchResponse = client().prepareSearch().setQuery(moreLikeThisQuery("int_value").likeText("42").minTermFreq(1).minDocFreq(1).failOnUnsupportedField(false)).execute().actionGet();
268         assertHitCount(searchResponse, 0l);
269     }

378     @Test
379     public void testCompareMoreLikeThisDSLWithAPI() throws Exception {
380         logger.info("Creating index test");
381         assertAcked(prepareCreate("test").addMapping("type1",
382                 jsonBuilder().startObject().startObject("type1").startObject("properties")
383                         .startObject("text").field("type", "string").endObject()
384                         .endObject().endObject().endObject()));
385 
386         logger.info("Running Cluster Health");
387         assertThat(ensureGreen(), equalTo(ClusterHealthStatus.GREEN));
388 
389         logger.info("Indexing...");
390         String[] texts = new String[] {
391             "Apache Lucene",
392             "free and open source",
393             "information retrieval",
394             "software library",
395             "programmed in Java",
396             "Doug Cutting",
397             "Apache Software Foundation",
398             "Apache Software License",
399             "Lucene programming languages",
400             "Delphi, Perl, C#, C++, Python, Ruby, and PHP"
401         };
402         List<IndexRequestBuilder> builders = new ArrayList<>(10);
403         for (int i = 0; i < texts.length; i++) {
404             builders.add(client().prepareIndex("test", "type1").setSource("text", texts[i]).setId(String.valueOf(i)));
405         }
406         indexRandom(true, false, builders);
407 
408         int iters = between(10, 20);
409         for (int j = 0; j < iters; j++) {
410             logger.info("Running MoreLikeThis DSL with IDs");
411             String id = String.valueOf(getRandom().nextInt(texts.length));
412             Client client = client();
413             MoreLikeThisQueryBuilder queryBuilder = QueryBuilders.moreLikeThisQuery("text").ids(id).minTermFreq(1).minDocFreq(1)
414                     .minimumShouldMatch("0%");
415             SearchResponse mltResponseDSL = client.prepareSearch()
416                     .setSearchType(SearchType.QUERY_THEN_FETCH)
417                     .setTypes("type1")
418                     .setQuery(queryBuilder)
419                     .setSize(texts.length)
420                     .execute().actionGet();
421             assertSearchResponse(mltResponseDSL);
422 
423             logger.info("Running MoreLikeThis API");
424             MoreLikeThisRequest mltRequest = moreLikeThisRequest("test").type("type1").searchSize(texts.length).id(id).minTermFreq(1).minDocFreq(1)
425                     .minimumShouldMatch("0%");
426             SearchResponse mltResponseAPI = client.moreLikeThis(mltRequest).actionGet();
427             assertSearchResponse(mltResponseAPI);
428 
429             logger.info("Ensure the documents and scores returned are the same.");
430             SearchHit[] hitsDSL = mltResponseDSL.getHits().hits();
431             SearchHit[] hitsAPI = mltResponseAPI.getHits().hits();
432 
433             // we have to resort since the results might come from
434             // different shards and docIDs that are used for tie-breaking might not be the same on the shards
435             Comparator<SearchHit> cmp = new Comparator<SearchHit>() {
436 
437                 @Override
438                 public int compare(SearchHit o1, SearchHit o2) {
439                     if (Float.compare(o1.getScore(), o2.getScore()) == 0) {
440                         return o1.getId().compareTo(o2.getId());
441                     }
442                     return Float.compare(o1.getScore(), o2.getScore());
443                 }
444             };
445             ArrayUtil.timSort(hitsDSL, cmp);
446             ArrayUtil.timSort(hitsAPI, cmp);
447             assertThat("Not the same number of results.", hitsAPI.length, equalTo(hitsDSL.length));
448             for (int i = 0; i < hitsDSL.length; i++) {
449                 assertThat("Expected id: " + hitsDSL[i].getId() + " at position " + i + " but wasn't.",
450                         hitsAPI[i].getId(), equalTo(hitsDSL[i].getId()));
451                 assertThat("Expected score: " + hitsDSL[i].getScore() + " at position " + i + " but wasn't.",
452                         hitsAPI[i].getScore(), equalTo(hitsDSL[i].getScore()));
453             }
454         }
455     }
