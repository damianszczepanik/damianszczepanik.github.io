46      /**
47       * Simple upgrade test for function score
48       */
49      @Test
50      public void testSimpleFunctionScoreParsingWorks() throws IOException, ExecutionException, InterruptedException {
51  
52          assertAcked(prepareCreate("test").addMapping(
53                  "type1",
54                  jsonBuilder().startObject()
55                          .startObject("type1")
56                          .startObject("properties")
57                          .startObject("text")
58                          .field("type", "string")
59                          .endObject()
60                          .startObject("loc")
61                          .field("type", "geo_point")
62                          .endObject()
63                          .endObject()
64                          .endObject()
65                          .endObject()));
66          ensureYellow();
67  
68          int numDocs = 10;
69          String[] ids = new String[numDocs];
70          List<IndexRequestBuilder> indexBuilders = new ArrayList<>();
71          for (int i = 0; i < numDocs; i++) {
72              String id = Integer.toString(i);
73              indexBuilders.add(client().prepareIndex()
74                      .setType("type1").setId(id).setIndex("test")
75                      .setSource(
76                              jsonBuilder().startObject()
77                                      .field("text", "value " + (i < 5 ? "boosted" : ""))
78                                      .startObject("loc")
79                                      .field("lat", 10 + i)
80                                      .field("lon", 20)
81                                      .endObject()
82                                      .endObject()));
83              ids[i] = id;
84          }
85          indexRandom(true, indexBuilders);
86          checkFunctionScoreStillWorks(ids);
87          logClusterState();
88          // prevent any kind of allocation during the upgrade we recover from gateway
89          client().admin().indices().prepareUpdateSettings("test").setSettings(ImmutableSettings.builder().put(EnableAllocationDecider.INDEX_ROUTING_ALLOCATION_ENABLE, "none")).get();
90          boolean upgraded;
91          int upgradedNodesCounter = 1;
92          do {
93              logger.debug("function_score bwc: upgrading {}st node", upgradedNodesCounter++);
94              upgraded = backwardsCluster().upgradeOneNode();
95              ensureYellow();
96              logClusterState();
97              checkFunctionScoreStillWorks(ids);
98          } while (upgraded);
99          client().admin().indices().prepareUpdateSettings("test").setSettings(ImmutableSettings.builder().put(EnableAllocationDecider.INDEX_ROUTING_ALLOCATION_ENABLE, "all")).get();
100         logger.debug("done function_score while upgrading");
101     }
102 
103     private void checkFunctionScoreStillWorks(String... ids) throws ExecutionException, InterruptedException, IOException {
104         SearchResponse response = client().search(
105                 searchRequest().source(
106                         searchSource().query(
107                                 functionScoreQuery(termFilter("text", "value"))
108                                         .add(gaussDecayFunction("loc", new GeoPoint(10, 20), "1000km"))
109                                         .add(scriptFunction("_index['text']['value'].tf()"))
110                                         .add(termFilter("text", "boosted"), factorFunction(5))
111                         ))).actionGet();
112         assertSearchResponse(response);
113         assertOrderedSearchHits(response, ids);
114     }
