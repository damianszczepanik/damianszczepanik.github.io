30  import org.elasticsearch.test.rest.client.http.HttpResponse;
31  import org.junit.Test;
32  
33  import java.net.URISyntaxException;
34  import java.nio.file.Path;
35  import java.nio.file.Paths;
36  
37  import static org.elasticsearch.common.settings.ImmutableSettings.settingsBuilder;
38  import static org.elasticsearch.test.ElasticsearchIntegrationTest.Scope;
39  import static org.hamcrest.Matchers.containsString;
40  import static org.hamcrest.Matchers.equalTo;
41  
42  /**
43   * We want to test site plugins
44   */
45  @ClusterScope(scope = Scope.SUITE, numDataNodes = 1)
46  public class SitePluginTests extends ElasticsearchIntegrationTest {
47  
48  
49      @Override
50      protected Settings nodeSettings(int nodeOrdinal) {
51          try {
52              Path pluginDir = Paths.get(SitePluginTests.class.getResource("/org/elasticsearch/plugins").toURI());
53              return settingsBuilder()
54                      .put(super.nodeSettings(nodeOrdinal))
55                      .put("path.plugins", pluginDir.toAbsolutePath())
56                      .put("force.http.enabled", true)
57                      .build();
58          } catch (URISyntaxException ex) {
59              throw new RuntimeException(ex);
60          }
61      }
62  
63      public HttpRequestBuilder httpClient() {
64          RequestConfig.Builder builder = RequestConfig.custom().setRedirectsEnabled(false);
65          CloseableHttpClient httpClient = HttpClients.custom().setDefaultRequestConfig(builder.build()).build();
66          return new HttpRequestBuilder(httpClient).httpTransport(internalCluster().getDataNodeInstance(HttpServerTransport.class));
67      }
68  
69      @Test
70      public void testRedirectSitePlugin() throws Exception {
71          // We use an HTTP Client to test redirection
72          HttpResponse response = httpClient().method("GET").path("/_plugin/dummy").execute();
73          assertThat(response.getStatusCode(), equalTo(RestStatus.MOVED_PERMANENTLY.getStatus()));
74          assertThat(response.getBody(), containsString("/_plugin/dummy/"));
75  
76          // We test the real URL
77          response = httpClient().method("GET").path("/_plugin/dummy/").execute();
78          assertThat(response.getStatusCode(), equalTo(RestStatus.OK.getStatus()));
79          assertThat(response.getBody(), containsString("<title>Dummy Site Plugin</title>"));
80      }
81  
82      /**
83       * Test direct access to an existing file (index.html)
84       */
85      @Test
86      public void testAnyPage() throws Exception {
87          HttpResponse response = httpClient().path("/_plugin/dummy/index.html").execute();
88          assertThat(response.getStatusCode(), equalTo(RestStatus.OK.getStatus()));
89          assertThat(response.getBody(), containsString("<title>Dummy Site Plugin</title>"));
90      }
91  
92      /**
93       * Test case for #4845: https://github.com/elasticsearch/elasticsearch/issues/4845
94       * Serving _site plugins do not pick up on index.html for sub directories
95       */
96      @Test
97      public void testWelcomePageInSubDirs() throws Exception {
98          HttpResponse response = httpClient().path("/_plugin/subdir/dir/").execute();
99          assertThat(response.getStatusCode(), equalTo(RestStatus.OK.getStatus()));
100         assertThat(response.getBody(), containsString("<title>Dummy Site Plugin (subdir)</title>"));
101 
102         response = httpClient().path("/_plugin/subdir/dir_without_index/").execute();
103         assertThat(response.getStatusCode(), equalTo(RestStatus.FORBIDDEN.getStatus()));
104 
105         response = httpClient().path("/_plugin/subdir/dir_without_index/page.html").execute();
106         assertThat(response.getStatusCode(), equalTo(RestStatus.OK.getStatus()));
107         assertThat(response.getBody(), containsString("<title>Dummy Site Plugin (page)</title>"));
108     }
109 }
