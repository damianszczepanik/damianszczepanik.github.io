109     /**
110      * Tests that we can actually recover from a corruption on the primary given that we have replica shards around.
111      */
112     @Test
113     public void testCorruptFileAndRecover() throws ExecutionException, InterruptedException, IOException {
114         int numDocs = scaledRandomIntBetween(100, 1000);
115         assertThat(cluster().numDataNodes(), greaterThanOrEqualTo(2));
116 
117         while (cluster().numDataNodes() < 4) {
118             /**
119              * We need 4 nodes since if we have 2 replicas and only 3 nodes we can't get into green state since
120              * the corrupted node will never be used reallocate a replica since it's marked as corrupted
121              */
122             internalCluster().startNode(ImmutableSettings.builder().put("node.data", true).put("node.client", false).put("node.master", false));
123         }
124         assertThat(cluster().numDataNodes(), greaterThanOrEqualTo(3));
125 
126         final boolean failOnCorruption = randomBoolean();
127         assertAcked(prepareCreate("test").setSettings(ImmutableSettings.builder()
128                         .put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, "1")
129                         .put(MergePolicyModule.MERGE_POLICY_TYPE_KEY, NoMergePolicyProvider.class)
130                         .put(MockFSDirectoryService.CHECK_INDEX_ON_CLOSE, false) // no checkindex - we corrupt shards on purpose
131                         .put(EngineConfig.INDEX_FAIL_ON_CORRUPTION_SETTING, failOnCorruption)
132                         .put(TranslogService.INDEX_TRANSLOG_DISABLE_FLUSH, true) // no translog based flush - it might change the .liv / segments.N files
133                         .put("indices.recovery.concurrent_streams", 10)
134         ));
135         if (failOnCorruption == false) { // test the dynamic setting
136             client().admin().indices().prepareUpdateSettings("test").setSettings(ImmutableSettings.builder()
137                     .put(EngineConfig.INDEX_FAIL_ON_CORRUPTION_SETTING, true)).get();
138         }
139         ensureGreen();
140         disableAllocation("test");
141         IndexRequestBuilder[] builders = new IndexRequestBuilder[numDocs];
142         for (int i = 0; i < builders.length; i++) {
143             builders[i] = client().prepareIndex("test", "type").setSource("field", "value");
144         }
145         indexRandom(true, builders);
146         ensureGreen();
147         assertAllSuccessful(client().admin().indices().prepareFlush().setForce(true).setWaitIfOngoing(true).execute().actionGet());
148         // we have to flush at least once here since we don't corrupt the translog
149         CountResponse countResponse = client().prepareCount().get();
150         assertHitCount(countResponse, numDocs);
151 
152         final int numShards = numShards("test");
153         ShardRouting corruptedShardRouting = corruptRandomFile();
154         enableAllocation("test");
155          /*
156          * we corrupted the primary shard - now lets make sure we never recover from it successfully
157          */
158         Settings build = ImmutableSettings.builder().put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, "2").build();
159         client().admin().indices().prepareUpdateSettings("test").setSettings(build).get();
160         ClusterHealthResponse health = client().admin().cluster()
161                 .health(Requests.clusterHealthRequest("test").waitForGreenStatus()
162                         .timeout("5m") // sometimes due to cluster rebalacing and random settings default timeout is just not enough.
163                         .waitForRelocatingShards(0)).actionGet();
164         if (health.isTimedOut()) {
165             logger.info("cluster state:\n{}\n{}", client().admin().cluster().prepareState().get().getState().prettyPrint(), client().admin().cluster().preparePendingClusterTasks().get().prettyPrint());
166             assertThat("timed out waiting for green state", health.isTimedOut(), equalTo(false));
167         }
168         assertThat(health.getStatus(), equalTo(ClusterHealthStatus.GREEN));
169         final int numIterations = scaledRandomIntBetween(5, 20);
170         for (int i = 0; i < numIterations; i++) {
171             SearchResponse response = client().prepareSearch().setSize(numDocs).get();
172             assertHitCount(response, numDocs);
173         }
174 
175 
176 
177         /*
178          * now hook into the IndicesService and register a close listener to
179          * run the checkindex. if the corruption is still there we will catch it.
180          */
181         final CountDownLatch latch = new CountDownLatch(numShards * 3); // primary + 2 replicas
182         final CopyOnWriteArrayList<Throwable> exception = new CopyOnWriteArrayList<>();
183         final IndicesLifecycle.Listener listener = new IndicesLifecycle.Listener() {
184             @Override
185             public void afterIndexShardClosed(ShardId sid, @Nullable IndexShard indexShard, @IndexSettings Settings indexSettings) {
186                 if (indexShard != null) {
187                     Store store = ((IndexShard) indexShard).store();
188                     store.incRef();
189                     try {
190                         if (!Lucene.indexExists(store.directory()) && indexShard.state() == IndexShardState.STARTED) {
191                             return;
192                         }
193                         try (CheckIndex checkIndex = new CheckIndex(store.directory())) {
194                             BytesStreamOutput os = new BytesStreamOutput();
195                             PrintStream out = new PrintStream(os, false, Charsets.UTF_8.name());
196                             checkIndex.setInfoStream(out);
197                             out.flush();
198                             CheckIndex.Status status = checkIndex.checkIndex();
199                             if (!status.clean) {
200                                 logger.warn("check index [failure]\n{}", new String(os.bytes().toBytes(), Charsets.UTF_8));
201                                 throw new IndexShardException(sid, "index check failure");
202                             }
203                         }
204                     } catch (Throwable t) {
205                         exception.add(t);
206                     } finally {
207                         store.decRef();
208                         latch.countDown();
209                     }
210                 }
211             }
212         };
213 
214         for (IndicesService service : internalCluster().getDataNodeInstances(IndicesService.class)) {
215             service.indicesLifecycle().addListener(listener);
216         }
217         try {
218             client().admin().indices().prepareDelete("test").get();
219             latch.await();
220             assertThat(exception, empty());
221         } finally {
222             for (IndicesService service : internalCluster().getDataNodeInstances(IndicesService.class)) {
223                 service.indicesLifecycle().removeListener(listener);
224             }
225         }
226     }

269                         .health(Requests.clusterHealthRequest("test")).actionGet().getStatus();

296                 corruptedFile = file;
297                 break;
298             }
299         }
300         assertThat(corruptedFile, notNullValue());
301     }

303     /**
304      * This test triggers a corrupt index exception during finalization size if an empty commit point is transferred
305      * during recovery we don't know the version of the segments_N file because it has no segments we can take it from.
306      * This simulates recoveries from old indices or even without checksums and makes sure if we fail during finalization
307      * we also check if the primary is ok. Without the relevant checks this test fails with a RED cluster
308      */
309     public void testCorruptionOnNetworkLayerFinalizingRecovery() throws ExecutionException, InterruptedException, IOException {
310         internalCluster().ensureAtLeastNumDataNodes(2);
311         NodesStatsResponse nodeStats = client().admin().cluster().prepareNodesStats().get();
312         List<NodeStats> dataNodeStats = new ArrayList<>();
313         for (NodeStats stat : nodeStats.getNodes()) {
314             if (stat.getNode().isDataNode()) {
315                 dataNodeStats.add(stat);
316             }
317         }
318 
319         assertThat(dataNodeStats.size(), greaterThanOrEqualTo(2));
320         Collections.shuffle(dataNodeStats, getRandom());
321         NodeStats primariesNode = dataNodeStats.get(0);
322         NodeStats unluckyNode = dataNodeStats.get(1);
323         assertAcked(prepareCreate("test").setSettings(ImmutableSettings.builder()
324                         .put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, "0")
325                         .put(IndexMetaData.SETTING_NUMBER_OF_SHARDS, 1)
326                         .put(EngineConfig.INDEX_FAIL_ON_CORRUPTION_SETTING, true)
327                         .put("index.routing.allocation.include._name", primariesNode.getNode().name())
328                         .put(EnableAllocationDecider.INDEX_ROUTING_REBALANCE_ENABLE, EnableAllocationDecider.Rebalance.NONE)
329 
330         ));
331         ensureGreen(); // allocated with empty commit
332         final AtomicBoolean corrupt = new AtomicBoolean(true);
333         final CountDownLatch hasCorrupted = new CountDownLatch(1);
334         for (NodeStats dataNode : dataNodeStats) {
335             MockTransportService mockTransportService = ((MockTransportService) internalCluster().getInstance(TransportService.class, dataNode.getNode().name()));
336             mockTransportService.addDelegate(internalCluster().getInstance(Discovery.class, unluckyNode.getNode().name()).localNode(), new MockTransportService.DelegateTransport(mockTransportService.original()) {
337 
338                 @Override
339                 public void sendRequest(DiscoveryNode node, long requestId, String action, TransportRequest request, TransportRequestOptions options) throws IOException, TransportException {
340                     if (corrupt.get() && action.equals(RecoveryTarget.Actions.FILE_CHUNK)) {
341                         RecoveryFileChunkRequest req = (RecoveryFileChunkRequest) request;
342                         byte[] array = req.content().array();
343                         int i = randomIntBetween(0, req.content().length() - 1);
344                         array[i] = (byte) ~array[i]; // flip one byte in the content
345                         hasCorrupted.countDown();
346                     }
347                     super.sendRequest(node, requestId, action, request, options);
348                 }
349             });
350         }
351 
352         Settings build = ImmutableSettings.builder()
353                 .put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, "1")
354                 .put("index.routing.allocation.include._name", primariesNode.getNode().name() + "," + unluckyNode.getNode().name()).build();
355         client().admin().indices().prepareUpdateSettings("test").setSettings(build).get();
356         client().admin().cluster().prepareReroute().get();
357         hasCorrupted.await();
358         corrupt.set(false);
359         ensureGreen();
360     }

362     /**
363      * Tests corruption that happens on the network layer and that the primary does not get affected by corruption that happens on the way

366     @Test
367     public void testCorruptionOnNetworkLayer() throws ExecutionException, InterruptedException {
368         int numDocs = scaledRandomIntBetween(100, 1000);

370         if (cluster().numDataNodes() < 3) {

374         List<NodeStats> dataNodeStats = new ArrayList<>();
375         for (NodeStats stat : nodeStats.getNodes()) {
376             if (stat.getNode().isDataNode()) {
377                 dataNodeStats.add(stat);
378             }
379         }
380 
381         assertThat(dataNodeStats.size(), greaterThanOrEqualTo(2));
382         Collections.shuffle(dataNodeStats, getRandom());
383         NodeStats primariesNode = dataNodeStats.get(0);
384         NodeStats unluckyNode = dataNodeStats.get(1);
385 
386 
387         assertAcked(prepareCreate("test").setSettings(ImmutableSettings.builder()
388                         .put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, "0")
389                         .put(IndexMetaData.SETTING_NUMBER_OF_SHARDS, between(1, 4)) // don't go crazy here it must recovery fast
390                         .put(EngineConfig.INDEX_FAIL_ON_CORRUPTION_SETTING, true)
391                                 // This does corrupt files on the replica, so we can't check:
392                         .put(MockFSDirectoryService.CHECK_INDEX_ON_CLOSE, false)
393                         .put("index.routing.allocation.include._name", primariesNode.getNode().name())
394                         .put(EnableAllocationDecider.INDEX_ROUTING_REBALANCE_ENABLE, EnableAllocationDecider.Rebalance.NONE)

396         ensureGreen();
397         IndexRequestBuilder[] builders = new IndexRequestBuilder[numDocs];

410             mockTransportService.addDelegate(internalCluster().getInstance(Discovery.class, unluckyNode.getNode().name()).localNode(), new MockTransportService.DelegateTransport(mockTransportService.original()) {
411

413                 public void sendRequest(DiscoveryNode node, long requestId, String action, TransportRequest request, TransportRequestOptions options) throws IOException, TransportException {
414                     if (action.equals(RecoveryTarget.Actions.FILE_CHUNK)) {
415                         RecoveryFileChunkRequest req = (RecoveryFileChunkRequest) request;
416                         if (truncate && req.length() > 1) {
417                             BytesArray array = new BytesArray(req.content().array(), req.content().arrayOffset(), (int) req.length() - 1);
418                             request = new RecoveryFileChunkRequest(req.recoveryId(), req.shardId(), req.metadata(), req.position(), array, req.lastChunk());
419                         } else {
420                             byte[] array = req.content().array();
421                             int i = randomIntBetween(0, req.content().length() - 1);
422                             array[i] = (byte) ~array[i]; // flip one byte in the content
423                         }
424                     }
425                     super.sendRequest(node, requestId, action, request, options);
426                 }
427             });
428         }
429 
430         Settings build = ImmutableSettings.builder()
431                 .put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, "1")
432                 .put("index.routing.allocation.include._name", "*").build();

436                 .health(Requests.clusterHealthRequest("test").waitForGreenStatus()).actionGet();
437         if (actionGet.isTimedOut()) {
438             logger.info("ensureGreen timed out, cluster state:\n{}\n{}", client().admin().cluster().prepareState().get().getState().prettyPrint(), client().admin().cluster().preparePendingClusterTasks().get().prettyPrint());
439             assertThat("timed out waiting for green state", actionGet.isTimedOut(), equalTo(false));
440         }
441         // we are green so primaries got not corrupted.
442         // ensure that no shard is actually allocated on the unlucky node
443         ClusterStateResponse clusterStateResponse = client().admin().cluster().prepareState().get();
444         for (IndexShardRoutingTable table : clusterStateResponse.getState().routingNodes().getRoutingTable().index("test")) {
445             for (ShardRouting routing : table) {
446                 if (unluckyNode.getNode().getId().equals(routing.currentNodeId())) {
447                     assertThat(routing.state(), not(equalTo(ShardRoutingState.STARTED)));
448                     assertThat(routing.state(), not(equalTo(ShardRoutingState.RELOCATING)));
449                 }
450             }
451         }
452         final int numIterations = scaledRandomIntBetween(5, 20);
453         for (int i = 0; i < numIterations; i++) {
454             SearchResponse response = client().prepareSearch().setSize(numDocs).get();
455             assertHitCount(response, numDocs);
456         }

491         ShardRouting shardRouting = corruptRandomFile(false);
492         // we don't corrupt segments.gen since S/R doesn't snapshot this file
493         // the other problem here why we can't corrupt segments.X files is that the snapshot flushes again before
494         // it snapshots and that will write a new segments.X+1 file
495         logger.info("-->  creating repository");
496         assertAcked(client().admin().cluster().preparePutRepository("test-repo")
497                 .setType("fs").setSettings(settingsBuilder()
498                         .put("location", newTempDirPath(LifecycleScope.SUITE).toAbsolutePath())
499                         .put("compress", randomBoolean())
500                         .put("chunk_size", randomIntBetween(100, 1000))));
501         logger.info("--> snapshot");
502         CreateSnapshotResponse createSnapshotResponse = client().admin().cluster().prepareCreateSnapshot("test-repo", "test-snap").setWaitForCompletion(true).setIndices("test").get();
503         assertThat(createSnapshotResponse.getSnapshotInfo().state(), equalTo(SnapshotState.PARTIAL));
504         logger.info("failed during snapshot -- maybe SI file got corrupted");
505         final List<Path> files = listShardFiles(shardRouting);
506         Path corruptedFile = null;
507         for (Path file : files) {
508             if (file.getFileName().toString().startsWith("corrupted_")) {
509                 corruptedFile = file;
510                 break;
511             }
512         }
513         assertThat(corruptedFile, notNullValue());
514     }

527     private ShardRouting corruptRandomFile(final boolean includePerCommitFiles) throws IOException {
528         ClusterState state = client().admin().cluster().prepareState().get().getState();
529         GroupShardsIterator shardIterators = state.getRoutingNodes().getRoutingTable().activePrimaryShardsGrouped(new String[]{"test"}, false);
530         List<ShardIterator> iterators = Lists.newArrayList(shardIterators);
531         ShardIterator shardIterator = RandomPicks.randomFrom(getRandom(), iterators);
532         ShardRouting shardRouting = shardIterator.nextOrNull();
533         assertNotNull(shardRouting);
534         assertTrue(shardRouting.primary());
535         assertTrue(shardRouting.assignedToNode());
536         String nodeId = shardRouting.currentNodeId();
537         NodesStatsResponse nodeStatses = client().admin().cluster().prepareNodesStats(nodeId).setFs(true).get();
538         Set<Path> files = new TreeSet<>(); // treeset makes sure iteration order is deterministic
539         for (FsStats.Info info : nodeStatses.getNodes()[0].getFs()) {
540             String path = info.getPath();
541             final String relativeDataLocationPath = "indices/test/" + Integer.toString(shardRouting.getId()) + "/index";
542             Path file = Paths.get(path).resolve(relativeDataLocationPath);
543             try (DirectoryStream<Path> stream = Files.newDirectoryStream(file)) {
544                 for (Path item : stream) {
545                     if (Files.isRegularFile(item) && "write.lock".equals(item.getFileName().toString()) == false) {
546                         if (includePerCommitFiles || isPerSegmentFile(item.getFileName().toString())) {
547                             files.add(item);
548                         }
549                     }
550                 }
551             }
552         }
553         pruneOldDeleteGenerations(files);
554         Path fileToCorrupt = null;
555         if (!files.isEmpty()) {
556             fileToCorrupt = RandomPicks.randomFrom(getRandom(), files);
557             try (Directory dir = FSDirectory.open(fileToCorrupt.toAbsolutePath().getParent())) {
558                 long checksumBeforeCorruption;
559                 try (IndexInput input = dir.openInput(fileToCorrupt.getFileName().toString(), IOContext.DEFAULT)) {
560                     checksumBeforeCorruption = CodecUtil.retrieveChecksum(input);
561                 }
562                 try (FileChannel raf = FileChannel.open(fileToCorrupt, StandardOpenOption.READ, StandardOpenOption.WRITE)) {
563                     // read
564                     raf.position(randomIntBetween(0, (int) Math.min(Integer.MAX_VALUE, raf.size() - 1)));
565                     long filePointer = raf.position();
566                     ByteBuffer bb = ByteBuffer.wrap(new byte[1]);
567                     raf.read(bb);
568                     bb.flip();
569                     
570                     // corrupt
571                     byte oldValue = bb.get(0);
572                     byte newValue = (byte) (oldValue + 1);
573                     bb.put(0, newValue);
574                     
575                     // rewrite
576                     raf.position(filePointer);
577                     raf.write(bb);
578                     logger.info("Corrupting file for shard {} --  flipping at position {} from {} to {} file: {}", shardRouting, filePointer, Integer.toHexString(oldValue), Integer.toHexString(newValue), fileToCorrupt.getFileName());
579                 }
580                 long checksumAfterCorruption;
581                 long actualChecksumAfterCorruption;
582                 try (ChecksumIndexInput input = dir.openChecksumInput(fileToCorrupt.getFileName().toString(), IOContext.DEFAULT)) {
583                     assertThat(input.getFilePointer(), is(0l));
584                     input.seek(input.length() - 8); // one long is the checksum... 8 bytes
585                     checksumAfterCorruption = input.getChecksum();
586                     actualChecksumAfterCorruption = input.readLong();
587                 }
588                 // we need to add assumptions here that the checksums actually really don't match there is a small chance to get collisions
589                 // in the checksum which is ok though....
590                 StringBuilder msg = new StringBuilder();
591                 msg.append("Checksum before: [").append(checksumBeforeCorruption).append("]");
592                 msg.append(" after: [").append(checksumAfterCorruption).append("]");
593                 msg.append(" checksum value after corruption: ").append(actualChecksumAfterCorruption).append("]");
594                 msg.append(" file: ").append(fileToCorrupt.getFileName()).append(" length: ").append(dir.fileLength(fileToCorrupt.getFileName().toString()));
595                 logger.info(msg.toString());
596                 assumeTrue("Checksum collision - " + msg.toString(),
597                         checksumAfterCorruption != checksumBeforeCorruption // collision
598                                 || actualChecksumAfterCorruption != checksumBeforeCorruption); // checksum corrupted
599             }
600         }
601         assertThat("no file corrupted", fileToCorrupt, notNullValue());
602         return shardRouting;
603     }

605     private static final boolean isPerCommitFile(String fileName) {
606         // .liv and segments_N are per commit files and might change after corruption
607         return fileName.startsWith("segments") || fileName.endsWith(".liv");
608     }

610     private static final boolean isPerSegmentFile(String fileName) {
611         return isPerCommitFile(fileName) == false;

614     /**
615      * prunes the list of index files such that only the latest del generation files are contained.
616      */
617     private void pruneOldDeleteGenerations(Set<Path> files) {
618         final TreeSet<Path> delFiles = new TreeSet<>();
619         for (Path file : files) {
620             if (file.getFileName().toString().endsWith(".liv")) {
621                 delFiles.add(file);
622             }
623         }
624         Path last = null;
625         for (Path current : delFiles) {
626             if (last != null) {
627                 final String newSegmentName = IndexFileNames.parseSegmentName(current.getFileName().toString());
628                 final String oldSegmentName = IndexFileNames.parseSegmentName(last.getFileName().toString());
629                 if (newSegmentName.equals(oldSegmentName)) {
630                     int oldGen = Integer.parseInt(IndexFileNames.stripExtension(IndexFileNames.stripSegmentName(last.getFileName().toString())).replace("_", ""), Character.MAX_RADIX);
631                     int newGen = Integer.parseInt(IndexFileNames.stripExtension(IndexFileNames.stripSegmentName(current.getFileName().toString())).replace("_", ""), Character.MAX_RADIX);
632                     if (newGen > oldGen) {
633                         files.remove(last);
634                     } else {
635                         files.remove(current);
636                         continue;
637                     }
638                 }
639             }
640             last = current;
641         }
642     }

644     public List<Path> listShardFiles(ShardRouting routing) throws IOException {
645         NodesStatsResponse nodeStatses = client().admin().cluster().prepareNodesStats(routing.currentNodeId()).setFs(true).get();
646 
647         assertThat(routing.toString(), nodeStatses.getNodes().length, equalTo(1));
648         List<Path> files = new ArrayList<>();
649         for (FsStats.Info info : nodeStatses.getNodes()[0].getFs()) {

651             Path file = Paths.get(path).resolve("indices/test/" + Integer.toString(routing.getId()) + "/index");
652             try (DirectoryStream<Path> stream = Files.newDirectoryStream(file)) {
653                 for (Path item : stream) {
654                     files.add(item);
655                 }
656             }
657         }
658         return files;
659     }
