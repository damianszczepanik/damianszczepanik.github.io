19  package org.elasticsearch.test.discovery;

22  import com.google.common.primitives.Ints;
23  import org.elasticsearch.ElasticsearchException;

36  public class ClusterDiscoveryConfiguration extends SettingsSource {
37  
38      static Settings DEFAULT_NODE_SETTINGS = ImmutableSettings.settingsBuilder().put("discovery.type", "zen").build();
39  
40      final int numOfNodes;
41      final Settings nodeSettings;
42      final Settings transportClientSettings;
43  
44      public ClusterDiscoveryConfiguration(int numOfNodes) {
45          this(numOfNodes, ImmutableSettings.EMPTY);
46      }
47  
48      public ClusterDiscoveryConfiguration(int numOfNodes, Settings extraSettings) {
49          this.numOfNodes = numOfNodes;
50          this.nodeSettings = ImmutableSettings.builder().put(DEFAULT_NODE_SETTINGS).put(extraSettings).build();
51          this.transportClientSettings = ImmutableSettings.builder().put(extraSettings).build();
52      }
53  
54      @Override
55      public Settings node(int nodeOrdinal) {
56          return nodeSettings;
57      }
58  
59      @Override
60      public Settings transportClient() {
61          return transportClientSettings;
62      }
63  
64      public static class UnicastZen extends ClusterDiscoveryConfiguration {
65  
66          // this variable is incremented on each bind attempt and will maintain the next port that should be tried
67          private static int nextPort = calcBasePort();
68  
69          private final int[] unicastHostOrdinals;
70          private final int[] unicastHostPorts;
71          private final boolean localMode;
72  
73          public UnicastZen(int numOfNodes) {
74              this(numOfNodes, numOfNodes);
75          }
76  
77          public UnicastZen(int numOfNodes, Settings extraSettings) {
78              this(numOfNodes, numOfNodes, extraSettings);
79          }
80  
81          public UnicastZen(int numOfNodes, int numOfUnicastHosts) {
82              this(numOfNodes, numOfUnicastHosts, ImmutableSettings.EMPTY);
83          }
84  
85          public UnicastZen(int numOfNodes, int numOfUnicastHosts, Settings extraSettings) {
86              super(numOfNodes, extraSettings);
87              if (numOfUnicastHosts == numOfNodes) {
88                  unicastHostOrdinals = new int[numOfNodes];
89                  for (int i = 0; i < numOfNodes; i++) {
90                      unicastHostOrdinals[i] = i;
91                  }
92              } else {
93                  Set<Integer> ordinals = new HashSet<>(numOfUnicastHosts);
94                  while (ordinals.size() != numOfUnicastHosts) {
95                      ordinals.add(RandomizedTest.randomInt(numOfNodes - 1));
96                  }
97                  unicastHostOrdinals = Ints.toArray(ordinals);
98              }
99              this.localMode = nodeSettings.get("node.mode", InternalTestCluster.NODE_MODE).equals("local");
100             this.unicastHostPorts = localMode ? new int[0] : unicastHostPorts(numOfNodes);
101             assert localMode || unicastHostOrdinals.length <= unicastHostPorts.length;
102         }
103 
104         public UnicastZen(int numOfNodes, int[] unicastHostOrdinals) {
105             this(numOfNodes, ImmutableSettings.EMPTY, unicastHostOrdinals);
106         }
107 
108         public UnicastZen(int numOfNodes, Settings extraSettings, int[] unicastHostOrdinals) {
109             super(numOfNodes, extraSettings);
110             this.unicastHostOrdinals = unicastHostOrdinals;
111             this.localMode = nodeSettings.get("node.mode", InternalTestCluster.NODE_MODE).equals("local");
112             this.unicastHostPorts = localMode ? new int[0] : unicastHostPorts(numOfNodes);
113             assert localMode || unicastHostOrdinals.length <= unicastHostPorts.length;
114         }
115 
116         private static int calcBasePort() {
117             // note that this has properly co-exist with the port logic at InternalTestCluster's constructor
118             return 30000 +
119                     1000 * (ElasticsearchIntegrationTest.CHILD_JVM_ID);
120         }
121 
122         @Override
123         public Settings node(int nodeOrdinal) {
124             ImmutableSettings.Builder builder = ImmutableSettings.builder()
125                     .put("discovery.zen.ping.multicast.enabled", false);
126 
127             String[] unicastHosts = new String[unicastHostOrdinals.length];
128             if (localMode) {
129                 builder.put(LocalTransport.TRANSPORT_LOCAL_ADDRESS, "node_" + nodeOrdinal);
130                 for (int i = 0; i < unicastHosts.length; i++) {
131                     unicastHosts[i] = "node_" + unicastHostOrdinals[i];
132                 }
133             } else if (nodeOrdinal >= unicastHostPorts.length) {
134                 throw new ElasticsearchException("nodeOrdinal [" + nodeOrdinal + "] is greater than the number unicast ports [" + unicastHostPorts.length + "]");
135             } else {
136                 // we need to pin the node port & host so we'd know where to point things
137                 builder.put("transport.tcp.port", unicastHostPorts[nodeOrdinal]);
138                 builder.put("transport.host", "localhost");
139                 for (int i = 0; i < unicastHostOrdinals.length; i++) {
140                     unicastHosts[i] = "localhost:" + (unicastHostPorts[unicastHostOrdinals[i]]);
141                 }
142             }
143             builder.putArray("discovery.zen.ping.unicast.hosts", unicastHosts);
144             return builder.put(super.node(nodeOrdinal)).build();
145         }
146 
147         protected synchronized static int[] unicastHostPorts(int numHosts) {
148             int[] unicastHostPorts = new int[numHosts];
149 
150             final int basePort = calcBasePort();
151             final int maxPort = basePort + 1000;
152             int tries = 0;
153             for (int i = 0; i < unicastHostPorts.length; i++) {
154                 boolean foundPortInRange = false;
155                 while (tries < 1000 && !foundPortInRange) {
156                     try (ServerSocket socket = new ServerSocket(nextPort)) {
157                         // bind was a success
158                         foundPortInRange = true;
159                         unicastHostPorts[i] = nextPort;
160                     } catch (IOException e) {
161                         // Do nothing
162                     }
163 
164                     nextPort++;
165                     if (nextPort >= maxPort) {
166                         // Roll back to the beginning of the range and do not go into another JVM's port range
167                         nextPort = basePort;
168                     }
169                     tries++;
170                 }
171 
172                 if (!foundPortInRange) {
173                     throw new ElasticsearchException("could not find enough open ports in range [" + basePort + "-" + maxPort + "]. required [" + unicastHostPorts.length + "] ports");
174                 }
175             }
176             return unicastHostPorts;
177         }
178     }
