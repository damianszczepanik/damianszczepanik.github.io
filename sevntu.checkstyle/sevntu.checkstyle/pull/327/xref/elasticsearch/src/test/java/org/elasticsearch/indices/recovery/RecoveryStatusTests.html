21  import com.google.common.collect.Sets;
22  import org.apache.lucene.store.IndexOutput;
23  import org.elasticsearch.Version;

25  import org.elasticsearch.common.transport.LocalTransportAddress;

27  import org.elasticsearch.index.shard.IndexShard;
28  import org.elasticsearch.index.store.StoreFileMetaData;
29  import org.elasticsearch.test.ElasticsearchSingleNodeTest;
30  
31  import java.io.IOException;
32  import java.util.Set;
33  import java.util.regex.Pattern;
34  
35  /**
36   */
37  public class RecoveryStatusTests extends ElasticsearchSingleNodeTest {
38      
39      public void testRenameTempFiles() throws IOException {
40          IndexService service = createIndex("foo");
41  
42          IndexShard indexShard = service.shard(0);
43          DiscoveryNode node = new DiscoveryNode("foo", new LocalTransportAddress("bar"), Version.CURRENT);
44          RecoveryStatus status = new RecoveryStatus(indexShard, node, new RecoveryTarget.RecoveryListener() {
45              @Override
46              public void onRecoveryDone(RecoveryState state) {
47              }
48  
49              @Override
50              public void onRecoveryFailure(RecoveryState state, RecoveryFailedException e, boolean sendShardFailure) {
51              }
52          });
53          try (IndexOutput indexOutput = status.openAndPutIndexOutput("foo.bar", new StoreFileMetaData("foo.bar", 8), status.store())) {
54              indexOutput.writeInt(1);
55              IndexOutput openIndexOutput = status.getOpenIndexOutput("foo.bar");
56              assertSame(openIndexOutput, indexOutput);
57              openIndexOutput.writeInt(1);
58          }
59          status.removeOpenIndexOutputs("foo.bar");
60          Set<String> strings = Sets.newHashSet(status.store().directory().listAll());
61          String expectedFile = null;
62          for (String file : strings) {
63              if (Pattern.compile("recovery[.]\\d+[.]foo[.]bar").matcher(file).matches()) {
64                  expectedFile = file;
65                  break;
66              }
67          }
68          assertNotNull(expectedFile);
69          indexShard.close("foo", false);// we have to close it here otherwise rename fails since the write.lock is held by the engine
70          status.renameAllTempFiles();
71          strings = Sets.newHashSet(status.store().directory().listAll());
72          assertTrue(strings.toString(), strings.contains("foo.bar"));
73          assertFalse(strings.toString(), strings.contains(expectedFile));
74          // we must fail the recovery because marking it as done will try to move the shard to POST_RECOVERY, which will fail because it's started
75          status.fail(new RecoveryFailedException(status.state(), "end of test. OK.", null), false);
76      }
77  }
