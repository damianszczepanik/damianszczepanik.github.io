40  import static org.elasticsearch.common.settings.ImmutableSettings.settingsBuilder;
41  
42  public class PatternCaptureTokenFilterTests extends ElasticsearchTokenStreamTestCase {
43  
44      @Test
45      public void testPatternCaptureTokenFilter() throws Exception {
46          Index index = new Index("test");
47          Settings settings = settingsBuilder().loadFromClasspath("org/elasticsearch/index/analysis/pattern_capture.json").put(IndexMetaData.SETTING_VERSION_CREATED, Version.CURRENT).build();
48          Injector parentInjector = new ModulesBuilder().add(new SettingsModule(settings), new EnvironmentModule(new Environment(settings)), new IndicesAnalysisModule()).createInjector();
49          Injector injector = new ModulesBuilder().add(
50                  new IndexSettingsModule(index, settings),
51                  new IndexNameModule(index),
52                  new AnalysisModule(settings, parentInjector.getInstance(IndicesAnalysisService.class)))
53                  .createChildInjector(parentInjector);
54  
55          AnalysisService analysisService = injector.getInstance(AnalysisService.class);
56  
57          NamedAnalyzer analyzer1 = analysisService.analyzer("single");
58  
59          assertTokenStreamContents(analyzer1.tokenStream("test", "foobarbaz"), new String[]{"foobarbaz","foobar","foo"});
60  
61          NamedAnalyzer analyzer2 = analysisService.analyzer("multi");
62  
63          assertTokenStreamContents(analyzer2.tokenStream("test", "abc123def"), new String[]{"abc123def","abc","123","def"});
64  
65          NamedAnalyzer analyzer3 = analysisService.analyzer("preserve");
66  
67          assertTokenStreamContents(analyzer3.tokenStream("test", "foobarbaz"), new String[]{"foobar","foo"});
68      }
69      
70      
71      @Test(expected=ElasticsearchIllegalArgumentException.class)
72      public void testNoPatterns() {
73          new PatternCaptureGroupTokenFilterFactory(new Index("test"), settingsBuilder().put(IndexMetaData.SETTING_VERSION_CREATED, Version.CURRENT).build(), "pattern_capture", settingsBuilder().put("pattern", "foobar").build());
74      }
