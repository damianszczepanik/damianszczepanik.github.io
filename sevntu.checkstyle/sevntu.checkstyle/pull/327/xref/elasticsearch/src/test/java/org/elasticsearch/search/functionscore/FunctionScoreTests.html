225     @Test
226     public void simpleWeightedFunctionsTestWithRandomWeightsAndRandomCombineMode() throws IOException, ExecutionException, InterruptedException {
227         assertAcked(prepareCreate(INDEX).addMapping(
228                 TYPE,
229                 MAPPING_WITH_DOUBLE_AND_GEO_POINT_AND_TEXT_FIELD));
230         ensureYellow();
231 
232         XContentBuilder doc = SIMPLE_DOC;
233         index(INDEX, TYPE, "1", doc);
234         refresh();
235         ScoreFunctionBuilder[] scoreFunctionBuilders = getScoreFunctionBuilders();
236         float[] weights = createRandomWeights(scoreFunctionBuilders.length);
237         float[] scores = getScores(scoreFunctionBuilders);
238 
239         String scoreMode = getRandomScoreMode();
240         FunctionScoreQueryBuilder withWeights = functionScoreQuery(termFilter(TEXT_FIELD, "value")).scoreMode(scoreMode);
241         int weightscounter = 0;
242         for (ScoreFunctionBuilder builder : scoreFunctionBuilders) {
243             withWeights.add(builder.setWeight(weights[weightscounter]));
244             weightscounter++;
245         }
246         SearchResponse responseWithWeights = client().search(
247                 searchRequest().source(searchSource().query(withWeights))
248         ).actionGet();
249 
250         double expectedScore = computeExpectedScore(weights, scores, scoreMode);
251         assertThat((float) expectedScore / responseWithWeights.getHits().getAt(0).getScore(), is(1.0f));
252     }

254     protected double computeExpectedScore(float[] weights, float[] scores, String scoreMode) {
255         double expectedScore = 0.0;
256         if ("multiply".equals(scoreMode)) {
257             expectedScore = 1.0;
258         }
259         if ("max".equals(scoreMode)) {
260             expectedScore = Float.MAX_VALUE * -1.0;
261         }
262         if ("min".equals(scoreMode)) {
263             expectedScore = Float.MAX_VALUE;
264         }
265 
266         float weightSum = 0;
267 
268         for (int i = 0; i < weights.length; i++) {
269             double functionScore = (double) weights[i] * scores[i];
270             weightSum += weights[i];
271 
272             if ("avg".equals(scoreMode)) {
273                 expectedScore += functionScore;
274             } else if ("max".equals(scoreMode)) {
275                 expectedScore = Math.max(functionScore, expectedScore);
276             } else if ("min".equals(scoreMode)) {
277                 expectedScore = Math.min(functionScore, expectedScore);
278             } else if ("sum".equals(scoreMode)) {
279                 expectedScore += functionScore;
280             } else if ("multiply".equals(scoreMode)) {
281                 expectedScore *= functionScore;
282             }
283 
284         }
285         if ("avg".equals(scoreMode)) {
286             expectedScore /= weightSum;
287         }
288         return expectedScore;
289     }

323     private String getRandomScoreMode() {
324         String[] scoreModes = {"avg", "sum", "min", "max", "multiply"};
325         return scoreModes[randomInt(scoreModes.length - 1)];
326     }

328     private float[] getScores(ScoreFunctionBuilder... scoreFunctionBuilders) {
329         float[] scores = new float[scoreFunctionBuilders.length];
330         int scorecounter = 0;
331         for (ScoreFunctionBuilder builder : scoreFunctionBuilders) {
332             SearchResponse response = client().search(
333                     searchRequest().source(
334                             searchSource().query(
335                                     functionScoreQuery(termFilter(TEXT_FIELD, "value"))
336                                             .add(builder)
337                             ))).actionGet();
338             scores[scorecounter] = response.getHits().getAt(0).getScore();
339             scorecounter++;
340         }
341         return scores;
342     }

344     private float[] createRandomWeights(int size) {
345         float[] weights = new float[size];
346         for (int i = 0; i < weights.length; i++) {
347             weights[i] = randomFloat() * (randomBoolean() ? 1.0f : -1.0f) * randomInt(100) + 1.e-6f;
348         }
349         return weights;
350     }

516     protected void assertMinScoreSearchResponses(int numDocs, SearchResponse searchResponse, int numMatchingDocs) {
517         assertSearchResponse(searchResponse);
518         assertThat((int) searchResponse.getHits().totalHits(), is(numMatchingDocs));
519         int pos = 0;
520         for (int hitId = numDocs - 1; (numDocs - hitId) < searchResponse.getHits().totalHits(); hitId--) {
521             assertThat(searchResponse.getHits().getAt(pos).getId(), equalTo(Integer.toString(hitId)));
522             pos++;
523         }
