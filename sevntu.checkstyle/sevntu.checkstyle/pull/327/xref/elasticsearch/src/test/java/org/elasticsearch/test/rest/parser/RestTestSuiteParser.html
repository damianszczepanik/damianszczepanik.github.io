19  package org.elasticsearch.test.rest.parser;
20  
21  import org.elasticsearch.common.xcontent.XContentParser;
22  import org.elasticsearch.common.xcontent.yaml.YamlXContent;
23  import org.elasticsearch.test.rest.section.RestTestSuite;
24  import org.elasticsearch.test.rest.section.TestSection;
25  
26  import java.io.IOException;
27  import java.nio.ByteBuffer;
28  import java.nio.channels.FileChannel;
29  import java.nio.file.Files;
30  import java.nio.file.Path;
31  import java.nio.file.StandardOpenOption;
32  
33  /**
34   * Parser for a complete test suite (yaml file)
35   */
36  public class RestTestSuiteParser implements RestTestFragmentParser<RestTestSuite> {
37  
38      public RestTestSuite parse(String api, Path file) throws IOException, RestTestParseException {
39  
40          if (!Files.isRegularFile(file)) {
41              throw new IllegalArgumentException(file.toAbsolutePath() + " is not a file");
42          }
43  
44          String filename = file.getFileName().toString();
45          //remove the file extension
46          int i = filename.lastIndexOf('.');
47          if (i > 0) {
48              filename = filename.substring(0, i);
49          }
50  
51          //our yaml parser seems to be too tolerant. Each yaml suite must end with \n, otherwise clients tests might break.
52          try (FileChannel channel = FileChannel.open(file, StandardOpenOption.READ)) {
53              ByteBuffer bb = ByteBuffer.wrap(new byte[1]);
54              channel.read(bb, channel.size() - 1);
55              if (bb.get(0) != 10) {
56                  throw new RestTestParseException("test suite [" + api + "/" + filename + "] doesn't end with line feed (\\n)");
57              }
58          }
59  
60          XContentParser parser = YamlXContent.yamlXContent.createParser(Files.newInputStream(file));
61          try {
62              RestTestSuiteParseContext testParseContext = new RestTestSuiteParseContext(api, filename, parser);
63              return parse(testParseContext);
64          } catch(Exception e) {
65              throw new RestTestParseException("Error parsing " + api + "/" + filename, e);
66          } finally {
67              parser.close();
68          }
69      }
70  
71      @Override
72      public RestTestSuite parse(RestTestSuiteParseContext parseContext) throws IOException, RestTestParseException {
73          XContentParser parser = parseContext.parser();
74  
75          parser.nextToken();
76          assert parser.currentToken() == XContentParser.Token.START_OBJECT;
77  
78          RestTestSuite restTestSuite = new RestTestSuite(parseContext.getApi(), parseContext.getSuiteName());
79  
80          restTestSuite.setSetupSection(parseContext.parseSetupSection());
81  
82          while(true) {
83              //the "---" section separator is not understood by the yaml parser. null is returned, same as when the parser is closed
84              //we need to somehow distinguish between a null in the middle of a test ("---")
85              // and a null at the end of the file (at least two consecutive null tokens)
86              if(parser.currentToken() == null) {
87                  if (parser.nextToken() == null) {
88                      break;
89                  }
90              }
91  
92              TestSection testSection = parseContext.parseTestSection();
93              if (!restTestSuite.addTestSection(testSection)) {
94                  throw new RestTestParseException("duplicate test section [" + testSection.getName() + "] found in [" + restTestSuite.getPath() + "]");
95              }
96          }
97  
98          return restTestSuite;
99      }
100 }
