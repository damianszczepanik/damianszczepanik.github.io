43      private class RecordingChangeListener extends FileChangesListener {
44  
45          private Path rootDir;
46  
47          private RecordingChangeListener(Path rootDir) {
48              this.rootDir = rootDir;
49          }
50  
51          private String getRelativeFileName(Path file) {
52              return rootDir.toUri().relativize(file.toUri()).getPath();
53          }
54  
55          private List<String> notifications = newArrayList();
56  
57          @Override
58          public void onFileInit(Path file) {
59              notifications.add("onFileInit: " + getRelativeFileName(file));
60          }
61  
62          @Override
63          public void onDirectoryInit(Path file) {
64              notifications.add("onDirectoryInit: " + getRelativeFileName(file));
65          }
66  
67          @Override
68          public void onFileCreated(Path file) {
69              notifications.add("onFileCreated: " + getRelativeFileName(file));
70          }
71  
72          @Override
73          public void onFileDeleted(Path file) {
74              notifications.add("onFileDeleted: " + getRelativeFileName(file));
75          }
76  
77          @Override
78          public void onFileChanged(Path file) {
79              notifications.add("onFileChanged: " + getRelativeFileName(file));
80          }
81  
82          @Override
83          public void onDirectoryCreated(Path file) {
84              notifications.add("onDirectoryCreated: " + getRelativeFileName(file));
85          }
86  
87          @Override
88          public void onDirectoryDeleted(Path file) {
89              notifications.add("onDirectoryDeleted: " + getRelativeFileName(file));
90          }
91  
92          public List<String> notifications() {
93              return notifications;
94          }
95      }

97      @Test
98      public void testSimpleFileOperations() throws IOException {
99          Path tempDir = newTempDirPath(LifecycleScope.TEST);
100         RecordingChangeListener changes = new RecordingChangeListener(tempDir);
101         Path testFile = tempDir.resolve("test.txt");
102         touch(testFile);
103         FileWatcher fileWatcher = new FileWatcher(testFile);
104         fileWatcher.addListener(changes);
105         fileWatcher.init();
106         assertThat(changes.notifications(), contains(equalTo("onFileInit: test.txt")));
107 
108         changes.notifications().clear();
109         fileWatcher.checkAndNotify();
110         assertThat(changes.notifications(), hasSize(0));
111 
112         append("Test", testFile, Charset.defaultCharset());
113         fileWatcher.checkAndNotify();
114         assertThat(changes.notifications(), contains(equalTo("onFileChanged: test.txt")));
115 
116         changes.notifications().clear();
117         fileWatcher.checkAndNotify();
118         assertThat(changes.notifications(), hasSize(0));
119 
120         Files.delete(testFile);
121         fileWatcher.checkAndNotify();
122         assertThat(changes.notifications(), contains(equalTo("onFileDeleted: test.txt")));
123 
124     }

126     @Test
127     public void testSimpleDirectoryOperations() throws IOException {
128         Path tempDir = newTempDirPath(LifecycleScope.TEST);
129         RecordingChangeListener changes = new RecordingChangeListener(tempDir);
130         Path testDir = tempDir.resolve("test-dir");
131         Files.createDirectories(testDir);
132         touch(testDir.resolve("test.txt"));
133         touch(testDir.resolve("test0.txt"));
134 
135         FileWatcher fileWatcher = new FileWatcher(testDir);
136         fileWatcher.addListener(changes);
137         fileWatcher.init();
138         assertThat(changes.notifications(), contains(
139                 equalTo("onDirectoryInit: test-dir/"),
140                 equalTo("onFileInit: test-dir/test.txt"),
141                 equalTo("onFileInit: test-dir/test0.txt")
142         ));
143 
144         changes.notifications().clear();
145         fileWatcher.checkAndNotify();
146         assertThat(changes.notifications(), hasSize(0));
147 
148         for (int i = 0; i < 4; i++) {
149             touch(testDir.resolve("test" + i + ".txt"));
150         }
151         // Make sure that first file is modified
152         append("Test", testDir.resolve("test0.txt"), Charset.defaultCharset());
153 
154         fileWatcher.checkAndNotify();
155         assertThat(changes.notifications(), contains(
156                 equalTo("onFileChanged: test-dir/test0.txt"),
157                 equalTo("onFileCreated: test-dir/test1.txt"),
158                 equalTo("onFileCreated: test-dir/test2.txt"),
159                 equalTo("onFileCreated: test-dir/test3.txt")
160         ));
161 
162         changes.notifications().clear();
163         fileWatcher.checkAndNotify();
164         assertThat(changes.notifications(), hasSize(0));
165 
166         Files.delete(testDir.resolve("test1.txt"));
167         Files.delete(testDir.resolve("test2.txt"));
168 
169         fileWatcher.checkAndNotify();
170         assertThat(changes.notifications(), contains(
171                 equalTo("onFileDeleted: test-dir/test1.txt"),
172                 equalTo("onFileDeleted: test-dir/test2.txt")
173         ));
174 
175         changes.notifications().clear();
176         fileWatcher.checkAndNotify();
177         assertThat(changes.notifications(), hasSize(0));
178 
179         Files.delete(testDir.resolve("test0.txt"));
180         touch(testDir.resolve("test2.txt"));
181         touch(testDir.resolve("test4.txt"));
182         fileWatcher.checkAndNotify();
183 
184         assertThat(changes.notifications(), contains(
185                 equalTo("onFileDeleted: test-dir/test0.txt"),
186                 equalTo("onFileCreated: test-dir/test2.txt"),
187                 equalTo("onFileCreated: test-dir/test4.txt")
188         ));
189 
190 
191         changes.notifications().clear();
192 
193         Files.delete(testDir.resolve("test3.txt"));
194         Files.delete(testDir.resolve("test4.txt"));
195         fileWatcher.checkAndNotify();
196         assertThat(changes.notifications(), contains(
197                 equalTo("onFileDeleted: test-dir/test3.txt"),
198                 equalTo("onFileDeleted: test-dir/test4.txt")
199         ));
200 
201 
202         changes.notifications().clear();
203         if (Files.exists(testDir)) {
204             IOUtils.rm(testDir);
205         }
206         fileWatcher.checkAndNotify();
207 
208         assertThat(changes.notifications(), contains(
209                 equalTo("onFileDeleted: test-dir/test.txt"),
210                 equalTo("onFileDeleted: test-dir/test2.txt"),
211                 equalTo("onDirectoryDeleted: test-dir")
212         ));
213 
214     }

216     @Test
217     public void testNestedDirectoryOperations() throws IOException {
218         Path tempDir = newTempDirPath(LifecycleScope.TEST);
219         RecordingChangeListener changes = new RecordingChangeListener(tempDir);
220         Path testDir = tempDir.resolve("test-dir");
221         Files.createDirectories(testDir);
222         touch(testDir.resolve("test.txt"));
223         Files.createDirectories(testDir.resolve("sub-dir"));
224         touch(testDir.resolve("sub-dir/test0.txt"));
225 
226         FileWatcher fileWatcher = new FileWatcher(testDir);
227         fileWatcher.addListener(changes);
228         fileWatcher.init();
229         assertThat(changes.notifications(), contains(
230                 equalTo("onDirectoryInit: test-dir/"),
231                 equalTo("onDirectoryInit: test-dir/sub-dir/"),
232                 equalTo("onFileInit: test-dir/sub-dir/test0.txt"),
233                 equalTo("onFileInit: test-dir/test.txt")
234         ));
235 
236         changes.notifications().clear();
237         fileWatcher.checkAndNotify();
238         assertThat(changes.notifications(), hasSize(0));
239 
240         // Create new file in subdirectory
241         touch(testDir.resolve("sub-dir/test1.txt"));
242         fileWatcher.checkAndNotify();
243         assertThat(changes.notifications(), contains(
244                 equalTo("onFileCreated: test-dir/sub-dir/test1.txt")
245         ));
246 
247         changes.notifications().clear();
248         fileWatcher.checkAndNotify();
249         assertThat(changes.notifications(), hasSize(0));
250 
251         // Create new subdirectory in subdirectory
252         Files.createDirectories(testDir.resolve("first-level"));
253         touch(testDir.resolve("first-level/file1.txt"));
254         Files.createDirectories(testDir.resolve("first-level/second-level"));
255         touch(testDir.resolve("first-level/second-level/file2.txt"));
256         fileWatcher.checkAndNotify();
257         assertThat(changes.notifications(), contains(
258                 equalTo("onDirectoryCreated: test-dir/first-level/"),
259                 equalTo("onFileCreated: test-dir/first-level/file1.txt"),
260                 equalTo("onDirectoryCreated: test-dir/first-level/second-level/"),
261                 equalTo("onFileCreated: test-dir/first-level/second-level/file2.txt")
262         ));
263 
264         changes.notifications().clear();
265         fileWatcher.checkAndNotify();
266         assertThat(changes.notifications(), hasSize(0));
267 
268         // Delete a directory, check notifications for
269         Path path = testDir.resolve("first-level");
270         if (Files.exists(path)) {
271             IOUtils.rm(path);
272         }
273         fileWatcher.checkAndNotify();
274         assertThat(changes.notifications(), contains(
275                 equalTo("onFileDeleted: test-dir/first-level/file1.txt"),
276                 equalTo("onFileDeleted: test-dir/first-level/second-level/file2.txt"),
277                 equalTo("onDirectoryDeleted: test-dir/first-level/second-level"),
278                 equalTo("onDirectoryDeleted: test-dir/first-level")
279         ));
280     }

282     @Test
283     public void testFileReplacingDirectory() throws IOException {
284         Path tempDir = newTempDirPath(LifecycleScope.TEST);
285         RecordingChangeListener changes = new RecordingChangeListener(tempDir);
286         Path testDir = tempDir.resolve("test-dir");
287         Files.createDirectories(testDir);
288         Path subDir = testDir.resolve("sub-dir");
289         Files.createDirectories(subDir);
290         touch(subDir.resolve("test0.txt"));
291         touch(subDir.resolve("test1.txt"));
292 
293         FileWatcher fileWatcher = new FileWatcher(testDir);
294         fileWatcher.addListener(changes);
295         fileWatcher.init();
296         assertThat(changes.notifications(), contains(
297                 equalTo("onDirectoryInit: test-dir/"),
298                 equalTo("onDirectoryInit: test-dir/sub-dir/"),
299                 equalTo("onFileInit: test-dir/sub-dir/test0.txt"),
300                 equalTo("onFileInit: test-dir/sub-dir/test1.txt")
301         ));
302 
303         changes.notifications().clear();
304 
305         if (Files.exists(subDir)) {
306             IOUtils.rm(subDir);
307         }
308         touch(subDir);
309         fileWatcher.checkAndNotify();
310         assertThat(changes.notifications(), contains(
311                 equalTo("onFileDeleted: test-dir/sub-dir/test0.txt"),
312                 equalTo("onFileDeleted: test-dir/sub-dir/test1.txt"),
313                 equalTo("onDirectoryDeleted: test-dir/sub-dir"),
314                 equalTo("onFileCreated: test-dir/sub-dir")
315         ));
316 
317         changes.notifications().clear();
318 
319         Files.delete(subDir);
320         Files.createDirectories(subDir);
321 
322         fileWatcher.checkAndNotify();
323         assertThat(changes.notifications(), contains(
324                 equalTo("onFileDeleted: test-dir/sub-dir/"),
325                 equalTo("onDirectoryCreated: test-dir/sub-dir/")
326         ));
327     }
