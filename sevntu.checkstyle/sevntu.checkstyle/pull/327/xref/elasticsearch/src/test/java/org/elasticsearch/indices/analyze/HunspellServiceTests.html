42  @ClusterScope(scope= Scope.TEST, numDataNodes=0)

45      @Test
46      public void testLocaleDirectoryWithNodeLevelConfig() throws Exception {
47          Settings settings = ImmutableSettings.settingsBuilder()
48                  .put("path.conf", getResourcePath("/indices/analyze/conf_dir"))
49                  .put(HUNSPELL_LAZY_LOAD, randomBoolean())
50                  .put(HUNSPELL_IGNORE_CASE, true)
51                  .build();
52  
53          internalCluster().startNode(settings);
54          Dictionary dictionary = internalCluster().getInstance(HunspellService.class).getDictionary("en_US");
55          assertThat(dictionary, notNullValue());
56          assertIgnoreCase(true, dictionary);
57      }
58  
59      @Test
60      public void testLocaleDirectoryWithLocaleSpecificConfig() throws Exception {
61          Settings settings = ImmutableSettings.settingsBuilder()
62                  .put("path.conf", getResourcePath("/indices/analyze/conf_dir"))
63                  .put(HUNSPELL_LAZY_LOAD, randomBoolean())
64                  .put(HUNSPELL_IGNORE_CASE, true)
65                  .put("indices.analysis.hunspell.dictionary.en_US.strict_affix_parsing", false)
66                  .put("indices.analysis.hunspell.dictionary.en_US.ignore_case", false)
67                  .build();
68  
69          internalCluster().startNode(settings);
70          Dictionary dictionary = internalCluster().getInstance(HunspellService.class).getDictionary("en_US");
71          assertThat(dictionary, notNullValue());
72          assertIgnoreCase(false, dictionary);
73  
74  
75  
76          // testing that dictionary specific settings override node level settings
77          dictionary = internalCluster().getInstance(HunspellService.class).getDictionary("en_US_custom");
78          assertThat(dictionary, notNullValue());
79          assertIgnoreCase(true, dictionary);
80      }
81  
82      @Test
83      public void testCustomizeLocaleDirectory() throws Exception {
84          Settings settings = ImmutableSettings.settingsBuilder()
85                  .put(HUNSPELL_LOCATION, getResourcePath("/indices/analyze/conf_dir/hunspell"))
86                  .build();
87  
88          internalCluster().startNode(settings);
89          Dictionary dictionary = internalCluster().getInstance(HunspellService.class).getDictionary("en_US");
90          assertThat(dictionary, notNullValue());
91      }

93      @Test
94      @LuceneTestCase.AwaitsFix(bugUrl = "https://github.com/elasticsearch/elasticsearch/issues/9849")
95      public void testDicWithNoAff() throws Exception {
96          Settings settings = ImmutableSettings.settingsBuilder()
97                  .put("path.conf", getResourcePath("/indices/analyze/no_aff_conf_dir"))
98                  .put(HUNSPELL_LAZY_LOAD, randomBoolean())
99                  .build();
100 
101         Dictionary dictionary = null;
102         try {
103             internalCluster().startNode(settings);
104             dictionary = internalCluster().getInstance(HunspellService.class).getDictionary("en_US");
105             fail("Missing affix file didn't throw an error");
106         }
107         catch (Throwable t) {
108             assertNull(dictionary);
109             assertThat(ExceptionsHelper.unwrap(t, ElasticsearchException.class).toString(), Matchers.containsString("Missing affix file"));
110         }
111     }
112 
113     @Test
114     @LuceneTestCase.AwaitsFix(bugUrl = "https://github.com/elasticsearch/elasticsearch/issues/9849")
115     public void testDicWithTwoAffs() throws Exception {
116         Settings settings = ImmutableSettings.settingsBuilder()
117                 .put("path.conf", getResourcePath("/indices/analyze/two_aff_conf_dir"))
118                 .put(HUNSPELL_LAZY_LOAD, randomBoolean())
119                 .build();
120 
121         Dictionary dictionary = null;

123             internalCluster().startNode(settings);

127             assertNull(dictionary);
128             assertThat(ExceptionsHelper.unwrap(t, ElasticsearchException.class).toString(), Matchers.containsString("Too many affix files"));

132     // TODO: open up a getter on Dictionary
133     private void assertIgnoreCase(boolean expected, Dictionary dictionary) throws Exception {
134         Field f = Dictionary.class.getDeclaredField("ignoreCase");
135         f.setAccessible(true);
136         assertEquals(expected, f.getBoolean(dictionary));
137     }
