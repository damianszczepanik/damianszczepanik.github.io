34  import static org.elasticsearch.index.query.QueryBuilders.hasChildQuery;
35  import static org.elasticsearch.index.query.QueryBuilders.matchAllQuery;
36  import static org.elasticsearch.test.hamcrest.ElasticsearchAssertions.*;
37  import static org.hamcrest.core.Is.is;

39  /**
40   */
41  public class ParentChildDeleteByQueryBackwardsCompatibilityTest extends ElasticsearchBackwardsCompatIntegrationTest {
42  
43      @BeforeClass
44      public static void checkVersion() {
45          assumeTrue("parent child queries in delete by query is forbidden from 1.1.2 and up", globalCompatibilityVersion().onOrBefore(Version.V_1_1_1));
46      }
47  
48      @Override
49      public void assertAllShardsOnNodes(String index, String pattern) {
50          super.assertAllShardsOnNodes(index, pattern);
51      }
52  
53      @Override
54      protected Settings externalNodeSettings(int nodeOrdinal) {
55          return ImmutableSettings.builder()
56                  .put(super.externalNodeSettings(nodeOrdinal))
57                  .put("index.translog.disable_flush", true)
58                  .build();
59      }
60  
61      @Test
62      public void testHasChild() throws Exception {
63          assertAcked(prepareCreate("idx")
64                  .setSettings(ImmutableSettings.builder()
65                                  .put(indexSettings())
66                                  .put("index.refresh_interval", "-1")
67                                  .put("index.routing.allocation.exclude._name", backwardsCluster().newNodePattern())
68                  )
69                  .addMapping("parent")
70                  .addMapping("child", "_parent", "type=parent"));
71  
72          List<IndexRequestBuilder> requests = new ArrayList<>();
73          requests.add(client().prepareIndex("idx", "parent", "1").setSource("{}"));
74          requests.add(client().prepareIndex("idx", "child", "1").setParent("1").setSource("{}"));
75          indexRandom(true, requests);
76  
77          SearchResponse response = client().prepareSearch("idx")
78                  .setQuery(hasChildQuery("child", matchAllQuery()))
79                  .get();
80          assertNoFailures(response);
81          assertHitCount(response, 1);
82  
83          client().prepareDeleteByQuery("idx")
84                  .setQuery(hasChildQuery("child", matchAllQuery()))
85                  .get();
86          refresh();
87  
88          response = client().prepareSearch("idx")
89                  .setQuery(hasChildQuery("child", matchAllQuery()))
90                  .get();
91          assertNoFailures(response);
92          assertHitCount(response, 0);
93  
94          client().prepareIndex("idx", "type", "1").setSource("{}").get();
95          assertThat(client().prepareGet("idx", "type", "1").get().isExists(), is(true));
96  
97          backwardsCluster().upgradeAllNodes();
98          backwardsCluster().allowOnAllNodes("idx");
99          ensureGreen("idx");
100 
101         response = client().prepareSearch("idx")
102                 .setQuery(hasChildQuery("child", matchAllQuery()))
103                 .get();
104         assertNoFailures(response);
105         assertHitCount(response, 1); // The delete by query has failed on recovery so that parent doc is still there
106 
107         // But the rest of the recovery did execute, we just skipped over the delete by query with the p/c query.
108         assertThat(client().prepareGet("idx", "type", "1").get().isExists(), is(true));
109         response = client().prepareSearch("idx").setTypes("type").get();
110         assertNoFailures(response);
111         assertHitCount(response, 1);
112     }

114 }
