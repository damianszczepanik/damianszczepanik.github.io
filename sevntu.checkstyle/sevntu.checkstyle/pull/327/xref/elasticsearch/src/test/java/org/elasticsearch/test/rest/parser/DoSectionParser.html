19  package org.elasticsearch.test.rest.parser;
20  
21  import org.elasticsearch.common.xcontent.XContentFactory;
22  import org.elasticsearch.common.xcontent.XContentParser;
23  import org.elasticsearch.common.xcontent.XContentType;
24  import org.elasticsearch.test.rest.section.ApiCallSection;
25  import org.elasticsearch.test.rest.section.DoSection;
26  
27  import java.io.IOException;
28  
29  /**
30   * Parser for do sections
31   */
32  public class DoSectionParser implements RestTestFragmentParser<DoSection> {
33  
34      @Override
35      public DoSection parse(RestTestSuiteParseContext parseContext) throws IOException, RestTestParseException {
36  
37          XContentParser parser = parseContext.parser();
38  
39          String currentFieldName = null;
40          XContentParser.Token token;
41  
42          DoSection doSection = new DoSection();
43  
44          while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
45              if (token == XContentParser.Token.FIELD_NAME) {
46                  currentFieldName = parser.currentName();
47              } else if (token.isValue()) {
48                  if ("catch".equals(currentFieldName)) {
49                      doSection.setCatch(parser.text());
50                  }
51              } else if (token == XContentParser.Token.START_OBJECT) {
52                  if (currentFieldName != null) {
53                      ApiCallSection apiCallSection = new ApiCallSection(currentFieldName);
54                      String paramName = null;
55                      while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
56                          if (token == XContentParser.Token.FIELD_NAME) {
57                              paramName = parser.currentName();
58                          } else if (token.isValue()) {
59                              if ("body".equals(paramName)) {
60                                  String body = parser.text();
61                                  XContentType bodyContentType = XContentFactory.xContentType(body);
62                                  XContentParser bodyParser = XContentFactory.xContent(bodyContentType).createParser(body);
63                                  //multiple bodies are supported e.g. in case of bulk provided as a whole string
64                                  while(bodyParser.nextToken() != null) {
65                                      apiCallSection.addBody(bodyParser.mapOrdered());
66                                  }
67                              } else {
68                                  apiCallSection.addParam(paramName, parser.text());
69                              }
70                          } else if (token == XContentParser.Token.START_OBJECT) {
71                              if ("body".equals(paramName)) {
72                                  apiCallSection.addBody(parser.mapOrdered());
73                              }
74                          }
75                      }
76                      doSection.setApiCallSection(apiCallSection);
77                  }
78              }
79          }
80  
81          parser.nextToken();
82  
83          if (doSection.getApiCallSection() == null) {
84              throw new RestTestParseException("client call section is mandatory within a do section");
85          }
86  
87          return doSection;
88      }
89  }
