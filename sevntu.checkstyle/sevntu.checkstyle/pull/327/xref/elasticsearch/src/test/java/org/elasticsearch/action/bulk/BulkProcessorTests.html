51      @Test
52      public void testThatBulkProcessorCountIsCorrect() throws InterruptedException {
53  
54          final CountDownLatch latch = new CountDownLatch(1);
55          BulkProcessorTestListener listener = new BulkProcessorTestListener(latch);
56  
57          int numDocs = randomIntBetween(10, 100);
58          try (BulkProcessor processor = BulkProcessor.builder(client(), listener).setName("foo")
59                  //let's make sure that the bulk action limit trips, one single execution will index all the documents
60                  .setConcurrentRequests(randomIntBetween(0, 1)).setBulkActions(numDocs)
61                  .setFlushInterval(TimeValue.timeValueHours(24)).setBulkSize(new ByteSizeValue(1, ByteSizeUnit.GB))
62                  .build()) {
63  
64              MultiGetRequestBuilder multiGetRequestBuilder = indexDocs(client(), processor, numDocs);
65  
66              latch.await();
67  
68              assertThat(listener.beforeCounts.get(), equalTo(1));
69              assertThat(listener.afterCounts.get(), equalTo(1));
70              assertThat(listener.bulkFailures.size(), equalTo(0));
71              assertResponseItems(listener.bulkItems, numDocs);
72              assertMultiGetResponse(multiGetRequestBuilder.get(), numDocs);
73          }
74      }

76      @Test
77      public void testBulkProcessorFlush() throws InterruptedException {
78          final CountDownLatch latch = new CountDownLatch(1);
79          BulkProcessorTestListener listener = new BulkProcessorTestListener(latch);
80  
81          int numDocs = randomIntBetween(10, 100);
82  
83          try (BulkProcessor processor = BulkProcessor.builder(client(), listener).setName("foo")
84                  //let's make sure that this bulk won't be automatically flushed
85                  .setConcurrentRequests(randomIntBetween(0, 10)).setBulkActions(numDocs + randomIntBetween(1, 100))
86                  .setFlushInterval(TimeValue.timeValueHours(24)).setBulkSize(new ByteSizeValue(1, ByteSizeUnit.GB)).build()) {
87  
88              MultiGetRequestBuilder multiGetRequestBuilder = indexDocs(client(), processor, numDocs);
89  
90              assertThat(latch.await(randomInt(500), TimeUnit.MILLISECONDS), equalTo(false));
91              //we really need an explicit flush as none of the bulk thresholds was reached
92              processor.flush();
93              latch.await();
94  
95              assertThat(listener.beforeCounts.get(), equalTo(1));
96              assertThat(listener.afterCounts.get(), equalTo(1));
97              assertThat(listener.bulkFailures.size(), equalTo(0));
98              assertResponseItems(listener.bulkItems, numDocs);
99              assertMultiGetResponse(multiGetRequestBuilder.get(), numDocs);
100         }
101     }

103     @Test
104     public void testBulkProcessorConcurrentRequests() throws Exception {
105         int bulkActions = randomIntBetween(10, 100);
106         int numDocs = randomIntBetween(bulkActions, bulkActions + 100);
107         int concurrentRequests = randomIntBetween(0, 10);
108 
109         int expectedBulkActions = numDocs / bulkActions;
110 
111         final CountDownLatch latch = new CountDownLatch(expectedBulkActions);
112         int totalExpectedBulkActions = numDocs % bulkActions == 0 ? expectedBulkActions : expectedBulkActions + 1;
113         final CountDownLatch closeLatch = new CountDownLatch(totalExpectedBulkActions);
114 
115         BulkProcessorTestListener listener = new BulkProcessorTestListener(latch, closeLatch);
116 
117         MultiGetRequestBuilder multiGetRequestBuilder;
118 
119         try (BulkProcessor processor = BulkProcessor.builder(client(), listener)
120                 .setConcurrentRequests(concurrentRequests).setBulkActions(bulkActions)
121                 //set interval and size to high values
122                 .setFlushInterval(TimeValue.timeValueHours(24)).setBulkSize(new ByteSizeValue(1, ByteSizeUnit.GB)).build()) {
123 
124             multiGetRequestBuilder = indexDocs(client(), processor, numDocs);
125 
126             latch.await();
127 
128             assertThat(listener.beforeCounts.get(), equalTo(expectedBulkActions));
129             assertThat(listener.afterCounts.get(), equalTo(expectedBulkActions));
130             assertThat(listener.bulkFailures.size(), equalTo(0));
131             assertThat(listener.bulkItems.size(), equalTo(numDocs - numDocs % bulkActions));
132         }
133 
134         closeLatch.await();
135 
136         assertThat(listener.beforeCounts.get(), equalTo(totalExpectedBulkActions));
137         assertThat(listener.afterCounts.get(), equalTo(totalExpectedBulkActions));
138         assertThat(listener.bulkFailures.size(), equalTo(0));
139         assertThat(listener.bulkItems.size(), equalTo(numDocs));
140 
141         Set<String> ids = new HashSet<>();
142         for (BulkItemResponse bulkItemResponse : listener.bulkItems) {
143             assertThat(bulkItemResponse.isFailed(), equalTo(false));
144             assertThat(bulkItemResponse.getIndex(), equalTo("test"));
145             assertThat(bulkItemResponse.getType(), equalTo("test"));
146             //with concurrent requests > 1 we can't rely on the order of the bulk requests
147             assertThat(Integer.valueOf(bulkItemResponse.getId()), both(greaterThan(0)).and(lessThanOrEqualTo(numDocs)));
148             //we do want to check that we don't get duplicate ids back
149             assertThat(ids.add(bulkItemResponse.getId()), equalTo(true));
150         }
151 
152         assertMultiGetResponse(multiGetRequestBuilder.get(), numDocs);
153     }

195     @Test
196     public void testBulkProcessorWaitOnClose() throws Exception {
197         BulkProcessorTestListener listener = new BulkProcessorTestListener();
198 
199         int numDocs = randomIntBetween(10, 100);
200         BulkProcessor processor = BulkProcessor.builder(client(), listener).setName("foo")
201                 //let's make sure that the bulk action limit trips, one single execution will index all the documents
202                 .setConcurrentRequests(randomIntBetween(0, 1)).setBulkActions(numDocs)
203                 .setFlushInterval(TimeValue.timeValueHours(24)).setBulkSize(new ByteSizeValue(randomIntBetween(1, 10),
204                         (ByteSizeUnit)RandomPicks.randomFrom(getRandom(), ByteSizeUnit.values())))
205                 .build();
206 
207         MultiGetRequestBuilder multiGetRequestBuilder = indexDocs(client(), processor, numDocs);
208         assertThat(processor.isOpen(), is(true));
209         assertThat(processor.awaitClose(1, TimeUnit.MINUTES), is(true));
210         if (randomBoolean()) { // check if we can call it multiple times
211             if (randomBoolean()) {
212                 assertThat(processor.awaitClose(1, TimeUnit.MINUTES), is(true));
213             } else {
214                 processor.close();
215             }
216         }
217         assertThat(processor.isOpen(), is(false));
218 
219         assertThat(listener.beforeCounts.get(), greaterThanOrEqualTo(1));
220         assertThat(listener.afterCounts.get(), greaterThanOrEqualTo(1));
221         assertThat(listener.bulkFailures.size(), equalTo(0));
222         assertResponseItems(listener.bulkItems, numDocs);
223         assertMultiGetResponse(multiGetRequestBuilder.get(), numDocs);
224     }

226     @Test
227     public void testBulkProcessorConcurrentRequestsReadOnlyIndex() throws Exception {
228         createIndex("test-ro");
229         assertAcked(client().admin().indices().prepareUpdateSettings("test-ro")
230                 .setSettings(ImmutableSettings.builder().put(IndexMetaData.SETTING_BLOCKS_WRITE, true)));
231         ensureGreen();
232 
233         int bulkActions = randomIntBetween(10, 100);
234         int numDocs = randomIntBetween(bulkActions, bulkActions + 100);
235         int concurrentRequests = randomIntBetween(0, 10);
236 
237         int expectedBulkActions = numDocs / bulkActions;
238 
239         final CountDownLatch latch = new CountDownLatch(expectedBulkActions);
240         int totalExpectedBulkActions = numDocs % bulkActions == 0 ? expectedBulkActions : expectedBulkActions + 1;
241         final CountDownLatch closeLatch = new CountDownLatch(totalExpectedBulkActions);
242 
243         int testDocs = 0;
244         int testReadOnlyDocs = 0;
245         MultiGetRequestBuilder multiGetRequestBuilder = client().prepareMultiGet();
246         BulkProcessorTestListener listener = new BulkProcessorTestListener(latch, closeLatch);
247 
248         try (BulkProcessor processor = BulkProcessor.builder(client(), listener)
249                 .setConcurrentRequests(concurrentRequests).setBulkActions(bulkActions)
250                         //set interval and size to high values
251                 .setFlushInterval(TimeValue.timeValueHours(24)).setBulkSize(new ByteSizeValue(1, ByteSizeUnit.GB)).build()) {
252 
253             for (int i = 1; i <= numDocs; i++) {
254                 if (randomBoolean()) {
255                     testDocs++;
256                     processor.add(new IndexRequest("test", "test", Integer.toString(testDocs)).source("field", "value"));
257                     multiGetRequestBuilder.add("test", "test", Integer.toString(testDocs));
258                 } else {
259                     testReadOnlyDocs++;
260                     processor.add(new IndexRequest("test-ro", "test", Integer.toString(testReadOnlyDocs)).source("field", "value"));
261                 }
262             }
263         }
264 
265         closeLatch.await();
266 
267         assertThat(listener.beforeCounts.get(), equalTo(totalExpectedBulkActions));
268         assertThat(listener.afterCounts.get(), equalTo(totalExpectedBulkActions));
269         assertThat(listener.bulkFailures.size(), equalTo(0));
270         assertThat(listener.bulkItems.size(), equalTo(testDocs + testReadOnlyDocs));
271 
272         Set<String> ids = new HashSet<>();
273         Set<String> readOnlyIds = new HashSet<>();
274         for (BulkItemResponse bulkItemResponse : listener.bulkItems) {
275             assertThat(bulkItemResponse.getIndex(), either(equalTo("test")).or(equalTo("test-ro")));
276             assertThat(bulkItemResponse.getType(), equalTo("test"));
277             if (bulkItemResponse.getIndex().equals("test")) {
278                 assertThat(bulkItemResponse.isFailed(), equalTo(false));
279                 //with concurrent requests > 1 we can't rely on the order of the bulk requests
280                 assertThat(Integer.valueOf(bulkItemResponse.getId()), both(greaterThan(0)).and(lessThanOrEqualTo(testDocs)));
281                 //we do want to check that we don't get duplicate ids back
282                 assertThat(ids.add(bulkItemResponse.getId()), equalTo(true));
283             } else {
284                 assertThat(bulkItemResponse.isFailed(), equalTo(true));
285                 //with concurrent requests > 1 we can't rely on the order of the bulk requests
286                 assertThat(Integer.valueOf(bulkItemResponse.getId()), both(greaterThan(0)).and(lessThanOrEqualTo(testReadOnlyDocs)));
287                 //we do want to check that we don't get duplicate ids back
288                 assertThat(readOnlyIds.add(bulkItemResponse.getId()), equalTo(true));
289             }
290         }
291 
292         assertMultiGetResponse(multiGetRequestBuilder.get(), testDocs);
293     }

295     private static MultiGetRequestBuilder indexDocs(Client client, BulkProcessor processor, int numDocs) {
296         MultiGetRequestBuilder multiGetRequestBuilder = client.prepareMultiGet();
297         for (int i = 1; i <= numDocs; i++) {
298             processor.add(new IndexRequest("test", "test", Integer.toString(i)).source("field", randomRealisticUnicodeOfLengthBetween(1, 30)));
299             multiGetRequestBuilder.add("test", "test", Integer.toString(i));
300         }
301         return multiGetRequestBuilder;
302     }

304     private static void assertResponseItems(List<BulkItemResponse> bulkItemResponses, int numDocs) {
305         assertThat(bulkItemResponses.size(), is(numDocs));
306         int i = 1;
307         for (BulkItemResponse bulkItemResponse : bulkItemResponses) {
308             assertThat(bulkItemResponse.getIndex(), equalTo("test"));
309             assertThat(bulkItemResponse.getType(), equalTo("test"));
310             assertThat(bulkItemResponse.getId(), equalTo(Integer.toString(i++)));
311             assertThat(bulkItemResponse.isFailed(), equalTo(false));
312         }
313     }
314 
315     private static void assertMultiGetResponse(MultiGetResponse multiGetResponse, int numDocs) {
316         assertThat(multiGetResponse.getResponses().length, equalTo(numDocs));
317         int i = 1;
318         for (MultiGetItemResponse multiGetItemResponse : multiGetResponse) {
319             assertThat(multiGetItemResponse.getIndex(), equalTo("test"));
320             assertThat(multiGetItemResponse.getType(), equalTo("test"));
321             assertThat(multiGetItemResponse.getId(), equalTo(Integer.toString(i++)));
322         }

325     private static class BulkProcessorTestListener implements BulkProcessor.Listener {
326 
327         private final CountDownLatch[] latches;
328         private final AtomicInteger beforeCounts = new AtomicInteger();
329         private final AtomicInteger afterCounts = new AtomicInteger();
330         private final List<BulkItemResponse> bulkItems = new CopyOnWriteArrayList<>();
331         private final List<Throwable> bulkFailures = new CopyOnWriteArrayList<>();
332 
333         private BulkProcessorTestListener(CountDownLatch... latches) {
334             this.latches = latches;
335         }
336 
337         @Override
338         public void beforeBulk(long executionId, BulkRequest request) {
339             beforeCounts.incrementAndGet();
340         }
341 
342         @Override
343         public void afterBulk(long executionId, BulkRequest request, BulkResponse response) {
344             bulkItems.addAll(Arrays.asList(response.getItems()));
345             afterCounts.incrementAndGet();
346             for (CountDownLatch latch : latches) {
347                 latch.countDown();
348             }
349         }
350 
351         @Override
352         public void afterBulk(long executionId, BulkRequest request, Throwable failure) {
353             bulkFailures.add(failure);
354             afterCounts.incrementAndGet();
355             for (CountDownLatch latch : latches) {
356                 latch.countDown();
357             }
358         }
359     }
