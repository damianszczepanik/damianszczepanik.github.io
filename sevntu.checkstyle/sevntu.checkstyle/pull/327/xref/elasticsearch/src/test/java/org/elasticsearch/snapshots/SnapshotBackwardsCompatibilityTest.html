57      @Test
58      public void testSnapshotAndRestore() throws ExecutionException, InterruptedException, IOException {
59          logger.info("-->  creating repository");
60          assertAcked(client().admin().cluster().preparePutRepository("test-repo")
61                  .setType("fs").setSettings(ImmutableSettings.settingsBuilder()
62                          .put("location", newTempDirPath(LifecycleScope.SUITE).toAbsolutePath())
63                          .put("compress", randomBoolean())
64                          .put("chunk_size", randomIntBetween(100, 1000))));
65          String[] indicesBefore = new String[randomIntBetween(2,5)];
66          String[] indicesAfter = new String[randomIntBetween(2,5)];
67          for (int i = 0; i < indicesBefore.length; i++) {
68              indicesBefore[i] = "index_before_" + i;
69              createIndex(indicesBefore[i]);
70          }
71          for (int i = 0; i < indicesAfter.length; i++) {
72              indicesAfter[i] = "index_after_" + i;
73              createIndex(indicesAfter[i]);
74          }
75          String[] indices = new String[indicesBefore.length + indicesAfter.length];
76          System.arraycopy(indicesBefore, 0, indices, 0, indicesBefore.length);
77          System.arraycopy(indicesAfter, 0, indices, indicesBefore.length, indicesAfter.length);
78          ensureYellow();
79          logger.info("--> indexing some data");
80          IndexRequestBuilder[] buildersBefore = new IndexRequestBuilder[randomIntBetween(10, 200)];
81          for (int i = 0; i < buildersBefore.length; i++) {
82              buildersBefore[i] = client().prepareIndex(RandomPicks.randomFrom(getRandom(), indicesBefore), "foo", Integer.toString(i)).setSource("{ \"foo\" : \"bar\" } ");
83          }
84          IndexRequestBuilder[] buildersAfter = new IndexRequestBuilder[randomIntBetween(10, 200)];
85          for (int i = 0; i < buildersAfter.length; i++) {
86              buildersAfter[i] = client().prepareIndex(RandomPicks.randomFrom(getRandom(), indicesBefore), "bar", Integer.toString(i)).setSource("{ \"foo\" : \"bar\" } ");
87          }
88          indexRandom(true, buildersBefore);
89          indexRandom(true, buildersAfter);
90          assertThat(client().prepareCount(indices).get().getCount(), equalTo((long) (buildersBefore.length + buildersAfter.length)));
91          long[] counts = new long[indices.length];
92          for (int i = 0; i < indices.length; i++) {
93              counts[i] = client().prepareCount(indices[i]).get().getCount();
94          }
95  
96          logger.info("--> snapshot subset of indices before upgrage");
97          CreateSnapshotResponse createSnapshotResponse = client().admin().cluster().prepareCreateSnapshot("test-repo", "test-snap-1").setWaitForCompletion(true).setIndices("index_before_*").get();
98          assertThat(createSnapshotResponse.getSnapshotInfo().successfulShards(), greaterThan(0));
99          assertThat(createSnapshotResponse.getSnapshotInfo().successfulShards(), equalTo(createSnapshotResponse.getSnapshotInfo().totalShards()));
100 
101         assertThat(client().admin().cluster().prepareGetSnapshots("test-repo").setSnapshots("test-snap-1").get().getSnapshots().get(0).state(), equalTo(SnapshotState.SUCCESS));
102 
103         logger.info("--> delete some data from indices that were already snapshotted");
104         int howMany = randomIntBetween(1, buildersBefore.length);
105 
106         for (int i = 0; i < howMany; i++) {
107             IndexRequestBuilder indexRequestBuilder = RandomPicks.randomFrom(getRandom(), buildersBefore);
108             IndexRequest request = indexRequestBuilder.request();
109             client().prepareDelete(request.index(), request.type(), request.id()).get();
110         }
111         refresh();
112         final long numDocs = client().prepareCount(indices).get().getCount();
113         assertThat(client().prepareCount(indices).get().getCount(), lessThan((long) (buildersBefore.length + buildersAfter.length)));
114 
115 
116         client().admin().indices().prepareUpdateSettings(indices).setSettings(ImmutableSettings.builder().put(EnableAllocationDecider.INDEX_ROUTING_ALLOCATION_ENABLE, "none")).get();
117         backwardsCluster().allowOnAllNodes(indices);
118         logClusterState();
119         boolean upgraded;
120         do {
121             logClusterState();
122             CountResponse countResponse = client().prepareCount().get();
123             assertHitCount(countResponse, numDocs);
124             upgraded = backwardsCluster().upgradeOneNode();
125             ensureYellow();
126             countResponse = client().prepareCount().get();
127             assertHitCount(countResponse, numDocs);
128         } while (upgraded);
129         client().admin().indices().prepareUpdateSettings(indices).setSettings(ImmutableSettings.builder().put(EnableAllocationDecider.INDEX_ROUTING_ALLOCATION_ENABLE, "all")).get();
130 
131         logger.info("--> close indices");
132         client().admin().indices().prepareClose("index_before_*").get();
133 
134         logger.info("--> verify repository");
135         client().admin().cluster().prepareVerifyRepository("test-repo").get();
136 
137         logger.info("--> restore all indices from the snapshot");
138         RestoreSnapshotResponse restoreSnapshotResponse = client().admin().cluster().prepareRestoreSnapshot("test-repo", "test-snap-1").setWaitForCompletion(true).execute().actionGet();
139         assertThat(restoreSnapshotResponse.getRestoreInfo().totalShards(), greaterThan(0));
140 
141         ensureYellow();
142         assertThat(client().prepareCount(indices).get().getCount(), equalTo((long) (buildersBefore.length + buildersAfter.length)));
143         for (int i = 0; i < indices.length; i++) {
144             assertThat(counts[i], equalTo(client().prepareCount(indices[i]).get().getCount()));
145         }
146 
147         logger.info("--> snapshot subset of indices after upgrade");
148         createSnapshotResponse = client().admin().cluster().prepareCreateSnapshot("test-repo", "test-snap-2").setWaitForCompletion(true).setIndices("index_*").get();
149         assertThat(createSnapshotResponse.getSnapshotInfo().successfulShards(), greaterThan(0));
150         assertThat(createSnapshotResponse.getSnapshotInfo().successfulShards(), equalTo(createSnapshotResponse.getSnapshotInfo().totalShards()));
151 
152         // Test restore after index deletion
153         logger.info("--> delete indices");
154         String index = RandomPicks.randomFrom(getRandom(), indices);
155         cluster().wipeIndices(index);
156         logger.info("--> restore one index after deletion");
157         restoreSnapshotResponse = client().admin().cluster().prepareRestoreSnapshot("test-repo", "test-snap-2").setWaitForCompletion(true).setIndices(index).execute().actionGet();
158         assertThat(restoreSnapshotResponse.getRestoreInfo().totalShards(), greaterThan(0));
159         ensureYellow();
160         assertThat(client().prepareCount(indices).get().getCount(), equalTo((long) (buildersBefore.length + buildersAfter.length)));
161         for (int i = 0; i < indices.length; i++) {
162             assertThat(counts[i], equalTo(client().prepareCount(indices[i]).get().getCount()));
163         }
164     }

166     public void testSnapshotMoreThanOnce() throws ExecutionException, InterruptedException, IOException {
167         Client client = client();
168         final Path tempDir = newTempDirPath(LifecycleScope.SUITE).toAbsolutePath();
169         logger.info("-->  creating repository");
170         assertAcked(client.admin().cluster().preparePutRepository("test-repo")
171                 .setType("fs").setSettings(ImmutableSettings.settingsBuilder()
172                         .put("location", tempDir)
173                         .put("compress", randomBoolean())
174                         .put("chunk_size", randomIntBetween(100, 1000))));
175 
176         // only one shard
177         assertAcked(prepareCreate("test").setSettings(ImmutableSettings.builder()
178                 .put(IndexMetaData.SETTING_NUMBER_OF_SHARDS, 1)
179                 .put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, 0)
180         ));
181         ensureYellow();
182         logger.info("-->  indexing");
183 
184         final int numDocs = randomIntBetween(10, 100);
185         IndexRequestBuilder[] builders = new IndexRequestBuilder[numDocs];
186         for (int i = 0; i < builders.length; i++) {
187             builders[i] = client().prepareIndex("test", "doc", Integer.toString(i)).setSource("foo", "bar" + i);
188         }
189         indexRandom(true, builders);
190         flushAndRefresh();
191         assertNoFailures(client().admin().indices().prepareOptimize("test").setFlush(true).setMaxNumSegments(1).get());
192 
193         CreateSnapshotResponse createSnapshotResponseFirst = client.admin().cluster().prepareCreateSnapshot("test-repo", "test").setWaitForCompletion(true).setIndices("test").get();
194         assertThat(createSnapshotResponseFirst.getSnapshotInfo().successfulShards(), greaterThan(0));
195         assertThat(createSnapshotResponseFirst.getSnapshotInfo().successfulShards(), equalTo(createSnapshotResponseFirst.getSnapshotInfo().totalShards()));
196         assertThat(client.admin().cluster().prepareGetSnapshots("test-repo").setSnapshots("test").get().getSnapshots().get(0).state(), equalTo(SnapshotState.SUCCESS));
197         {
198             SnapshotStatus snapshotStatus = client.admin().cluster().prepareSnapshotStatus("test-repo").setSnapshots("test").get().getSnapshots().get(0);
199             List<SnapshotIndexShardStatus> shards = snapshotStatus.getShards();
200             for (SnapshotIndexShardStatus status : shards) {
201                 assertThat(status.getStats().getProcessedFiles(), greaterThan(1));
202             }
203         }
204         if (frequently()) {
205             logger.info("-->  upgrade");
206             client().admin().indices().prepareUpdateSettings("test").setSettings(ImmutableSettings.builder().put(EnableAllocationDecider.INDEX_ROUTING_ALLOCATION_ENABLE, "none")).get();
207             backwardsCluster().allowOnAllNodes("test");
208             logClusterState();
209             boolean upgraded;
210             do {
211                 logClusterState();
212                 CountResponse countResponse = client().prepareCount().get();
213                 assertHitCount(countResponse, numDocs);
214                 upgraded = backwardsCluster().upgradeOneNode();
215                 ensureYellow();
216                 countResponse = client().prepareCount().get();
217                 assertHitCount(countResponse, numDocs);
218             } while (upgraded);
219             client().admin().indices().prepareUpdateSettings("test").setSettings(ImmutableSettings.builder().put(EnableAllocationDecider.INDEX_ROUTING_ALLOCATION_ENABLE, "all")).get();
220         }
221         if (cluster().numDataNodes() > 1 && randomBoolean()) { // only bump the replicas if we have enough nodes
222             logger.info("--> move from 0 to 1 replica");
223             client().admin().indices().prepareUpdateSettings("test").setSettings(ImmutableSettings.builder().put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, 1)).get();
224         }
225         logger.debug("---> repo exists: " + Files.exists(tempDir.resolve("indices/test/0")) + " files: " + Arrays.toString(FileSystemUtils.files(tempDir.resolve("indices/test/0")))); // it's only one shard!
226         CreateSnapshotResponse createSnapshotResponseSecond = client.admin().cluster().prepareCreateSnapshot("test-repo", "test-1").setWaitForCompletion(true).setIndices("test").get();
227         assertThat(createSnapshotResponseSecond.getSnapshotInfo().successfulShards(), greaterThan(0));
228         assertThat(createSnapshotResponseSecond.getSnapshotInfo().successfulShards(), equalTo(createSnapshotResponseSecond.getSnapshotInfo().totalShards()));
229         assertThat(client.admin().cluster().prepareGetSnapshots("test-repo").setSnapshots("test-1").get().getSnapshots().get(0).state(), equalTo(SnapshotState.SUCCESS));
230         {
231             SnapshotStatus snapshotStatus = client.admin().cluster().prepareSnapshotStatus("test-repo").setSnapshots("test-1").get().getSnapshots().get(0);
232             List<SnapshotIndexShardStatus> shards = snapshotStatus.getShards();
233             for (SnapshotIndexShardStatus status : shards) {
234 
235                 assertThat(status.getStats().getProcessedFiles(), equalTo(1)); // we flush before the snapshot such that we have to process the segments_N files
236             }
237         }
238 
239         client().prepareDelete("test", "doc", "1").get();
240         CreateSnapshotResponse createSnapshotResponseThird = client.admin().cluster().prepareCreateSnapshot("test-repo", "test-2").setWaitForCompletion(true).setIndices("test").get();
241         assertThat(createSnapshotResponseThird.getSnapshotInfo().successfulShards(), greaterThan(0));
242         assertThat(createSnapshotResponseThird.getSnapshotInfo().successfulShards(), equalTo(createSnapshotResponseThird.getSnapshotInfo().totalShards()));
243         assertThat(client.admin().cluster().prepareGetSnapshots("test-repo").setSnapshots("test-2").get().getSnapshots().get(0).state(), equalTo(SnapshotState.SUCCESS));
244         {
245             SnapshotStatus snapshotStatus = client.admin().cluster().prepareSnapshotStatus("test-repo").setSnapshots("test-2").get().getSnapshots().get(0);
246             List<SnapshotIndexShardStatus> shards = snapshotStatus.getShards();
247             for (SnapshotIndexShardStatus status : shards) {
248                 assertThat(status.getStats().getProcessedFiles(), equalTo(2)); // we flush before the snapshot such that we have to process the segments_N files plus the .del file
249             }
250         }
