116     @Test
117     public void testVerifyingIndexOutput() throws IOException {
118         Directory dir = newDirectory();
119         IndexOutput output = dir.createOutput("foo.bar", IOContext.DEFAULT);
120         int iters = scaledRandomIntBetween(10, 100);
121         for (int i = 0; i < iters; i++) {
122             BytesRef bytesRef = new BytesRef(TestUtil.randomRealisticUnicodeString(random(), 10, 1024));
123             output.writeBytes(bytesRef.bytes, bytesRef.offset, bytesRef.length);
124         }
125         CodecUtil.writeFooter(output);
126         output.close();
127         IndexInput indexInput = dir.openInput("foo.bar", IOContext.DEFAULT);
128         String checksum = Store.digestToString(CodecUtil.retrieveChecksum(indexInput));
129         indexInput.seek(0);
130         BytesRef ref = new BytesRef(scaledRandomIntBetween(1, 1024));
131         long length = indexInput.length();
132         IndexOutput verifyingOutput = new Store.LuceneVerifyingIndexOutput(new StoreFileMetaData("foo1.bar", length, checksum), dir.createOutput("foo1.bar", IOContext.DEFAULT));
133         while (length > 0) {
134             if (random().nextInt(10) == 0) {
135                 verifyingOutput.writeByte(indexInput.readByte());
136                 length--;
137             } else {
138                 int min = (int) Math.min(length, ref.bytes.length);
139                 indexInput.readBytes(ref.bytes, ref.offset, min);
140                 verifyingOutput.writeBytes(ref.bytes, ref.offset, min);
141                 length -= min;
142             }
143         }
144         Store.verify(verifyingOutput);
145         verifyingOutput.writeByte((byte) 0x0);
146         try {
147             Store.verify(verifyingOutput);
148             fail("should be a corrupted index");
149         } catch (CorruptIndexException | IndexFormatTooOldException | IndexFormatTooNewException ex) {
150             // ok
151         }
152         IOUtils.close(indexInput, verifyingOutput, dir);
153     }

234     public void testWriteLegacyChecksums() throws IOException {
235         final ShardId shardId = new ShardId(new Index("index"), 1);
236         DirectoryService directoryService = new LuceneManagedDirectoryService(random());
237         Store store = new Store(shardId, ImmutableSettings.EMPTY, directoryService, randomDistributor(directoryService), new DummyShardLock(shardId));
238         // set default codec - all segments need checksums
239         final boolean usesOldCodec = randomBoolean();
240         IndexWriter writer = new IndexWriter(store.directory(), newIndexWriterConfig(random(), new MockAnalyzer(random())).setCodec(usesOldCodec ? new OldSIMockingCodec() : actualDefaultCodec()));
241         int docs = 1 + random().nextInt(100);
242 
243         for (int i = 0; i < docs; i++) {
244             Document doc = new Document();
245             doc.add(new TextField("id", "" + i, random().nextBoolean() ? Field.Store.YES : Field.Store.NO));
246             doc.add(new TextField("body", TestUtil.randomRealisticUnicodeString(random()), random().nextBoolean() ? Field.Store.YES : Field.Store.NO));
247             doc.add(new SortedDocValuesField("dv", new BytesRef(TestUtil.randomRealisticUnicodeString(random()))));
248             writer.addDocument(doc);
249         }
250         if (random().nextBoolean()) {
251             for (int i = 0; i < docs; i++) {
252                 if (random().nextBoolean()) {
253                     Document doc = new Document();
254                     doc.add(new TextField("id", "" + i, random().nextBoolean() ? Field.Store.YES : Field.Store.NO));
255                     doc.add(new TextField("body", TestUtil.randomRealisticUnicodeString(random()), random().nextBoolean() ? Field.Store.YES : Field.Store.NO));
256                     writer.updateDocument(new Term("id", "" + i), doc);
257                 }
258             }
259         }
260         if (random().nextBoolean()) {
261             DirectoryReader.open(writer, random().nextBoolean()).close(); // flush
262         }
263         Store.MetadataSnapshot metadata;
264         // check before we committed
265         try {
266             store.getMetadata();
267             fail("no index present - expected exception");
268         } catch (IndexNotFoundException ex) {
269             // expected
270         }
271         assertThat(store.getMetadataOrEmpty(), is(Store.MetadataSnapshot.EMPTY)); // nothing committed
272 
273         writer.close();
274         Store.LegacyChecksums checksums = new Store.LegacyChecksums();
275         Map<String, StoreFileMetaData> legacyMeta = new HashMap<>();
276         for (String file : store.directory().listAll()) {
277             if (file.equals("write.lock") || file.equals(IndexFileNames.OLD_SEGMENTS_GEN)) {
278                 continue;
279             }
280             BytesRef hash = new BytesRef();
281             if (file.startsWith("segments")) {
282                 hash = Store.MetadataSnapshot.hashFile(store.directory(), file);
283             }
284             StoreFileMetaData storeFileMetaData = new StoreFileMetaData(file, store.directory().fileLength(file), file + "checksum", null, hash);
285             legacyMeta.put(file, storeFileMetaData);
286             checksums.add(storeFileMetaData);
287         }
288         checksums.write(store);
289 
290         metadata = store.getMetadata();
291         Map<String, StoreFileMetaData> stringStoreFileMetaDataMap = metadata.asMap();
292         assertThat(legacyMeta.size(), equalTo(stringStoreFileMetaDataMap.size()));
293         if (usesOldCodec) {
294             for (StoreFileMetaData meta : legacyMeta.values()) {
295                 assertTrue(meta.toString(), stringStoreFileMetaDataMap.containsKey(meta.name()));
296                 assertEquals(meta.name() + "checksum", meta.checksum());
297                 assertTrue(meta + " vs. " + stringStoreFileMetaDataMap.get(meta.name()), stringStoreFileMetaDataMap.get(meta.name()).isSame(meta));
298             }
299         } else {
300 
301             // even if we have a legacy checksum - if we use a new codec we should reuse
302             for (StoreFileMetaData meta : legacyMeta.values()) {
303                 assertTrue(meta.toString(), stringStoreFileMetaDataMap.containsKey(meta.name()));
304                 assertFalse(meta + " vs. " + stringStoreFileMetaDataMap.get(meta.name()), stringStoreFileMetaDataMap.get(meta.name()).isSame(meta));
305                 StoreFileMetaData storeFileMetaData = metadata.get(meta.name());
306                 try (IndexInput input = store.openVerifyingInput(meta.name(), IOContext.DEFAULT, storeFileMetaData)) {
307                     assertTrue(storeFileMetaData.toString(), input instanceof Store.VerifyingIndexInput);
308                     input.seek(meta.length());
309                     Store.verify(input);
310                 }
311             }
312         }
313         assertDeleteContent(store, directoryService);
314         IOUtils.close(store);
315

426                         CodecUtil.retrieveChecksum(input);

432                         }

436                 } else {
437                     String checksum = Store.digestToString(CodecUtil.retrieveChecksum(input));

450             if (meta.hasLegacyChecksum()) {
451                 try (ChecksumIndexInput checksumIndexInput = store.directory().openChecksumInput(meta.name(), IOContext.DEFAULT)) {
452                     checksumIndexInput.seek(meta.length());
453                     assertThat(meta.checksum(), equalTo(Store.digestToString(checksumIndexInput.getChecksum())));
454                 }
455             } else {
456                 try (IndexInput input = store.directory().openInput(meta.name(), IOContext.DEFAULT)) {
457                     String checksum = Store.digestToString(CodecUtil.retrieveChecksum(input));
458                     assertThat("File: " + meta.name() + " has a different checksum", meta.checksum(), equalTo(checksum));
459                     assertThat(meta.hasLegacyChecksum(), equalTo(false));
460                     assertThat(meta.writtenBy(), equalTo(Version.LATEST));
461                 }
462             }
463         }
464         assertConsistent(store, metadata);
465         TestUtil.checkIndex(store.directory());
466         assertDeleteContent(store, directoryService);
467         IOUtils.close(store);
468     }

640     private void readIndexInputFullyWithRandomSeeks(IndexInput indexInput) throws IOException{
641         BytesRef ref = new BytesRef(scaledRandomIntBetween(1, 1024));
642         long pos = 0;
643         while (pos < indexInput.length()) {
644             assertEquals(pos, indexInput.getFilePointer());
645             int op = random().nextInt(5);
646             if (op == 0 ) {
647                 int shift =  100 - randomIntBetween(0, 200);
648                 pos =  Math.min(indexInput.length() - 1, Math.max(0, pos + shift));
649                 indexInput.seek(pos);
650             } else if (op == 1) {
651                 indexInput.readByte();
652                 pos ++;
653             } else {
654                 int min = (int) Math.min(indexInput.length() - pos, ref.bytes.length);
655                 indexInput.readBytes(ref.bytes, ref.offset, min);
656                 pos += min;
657             }
658         }

661     private void corruptFile(Directory dir, String fileIn, String fileOut) throws IOException {
662         IndexInput input = dir.openInput(fileIn, IOContext.READONCE);
663         IndexOutput output = dir.createOutput(fileOut, IOContext.DEFAULT);
664         long len = input.length();
665         byte[] b = new byte[1024];
666         long broken = randomInt((int) len);
667         long pos = 0;
668         while (pos < len) {
669             int min = (int) Math.min(input.length() - pos, b.length);
670             input.readBytes(b, 0, min);
671             if (broken >= pos && broken < pos + min) {
672                 // Flip one byte
673                 int flipPos = (int) (broken - pos);
674                 b[flipPos] = (byte) (b[flipPos] ^ 42);
675             }
676             output.writeBytes(b, min);
677             pos += min;
678         }
679         IOUtils.close(input, output);
