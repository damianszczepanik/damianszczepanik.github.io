51      public void testCleanIndex() throws IOException {
52          MockDirectoryWrapper dir = newMockDirectory();
53          dir.setEnableVirusScanner(false);
54          IndexWriterConfig iwc = newIndexWriterConfig();
55          iwc.setIndexDeletionPolicy(NoDeletionPolicy.INSTANCE);
56          iwc.setMergePolicy(NoMergePolicy.INSTANCE);
57          iwc.setMaxBufferedDocs(2);
58          IndexWriter writer = new IndexWriter(dir, iwc);
59          Document doc = new Document();
60          doc.add(new TextField("id", "1", random().nextBoolean() ? Field.Store.YES : Field.Store.NO));
61          writer.addDocument(doc);
62          writer.commit();
63  
64          doc = new Document();
65          doc.add(new TextField("id", "2", random().nextBoolean() ? Field.Store.YES : Field.Store.NO));
66          writer.addDocument(doc);
67  
68          doc = new Document();
69          doc.add(new TextField("id", "3", random().nextBoolean() ? Field.Store.YES : Field.Store.NO));
70          writer.addDocument(doc);
71  
72          writer.commit();
73          doc = new Document();
74          doc.add(new TextField("id", "4", random().nextBoolean() ? Field.Store.YES : Field.Store.NO));
75          writer.addDocument(doc);
76  
77          writer.deleteDocuments(new Term("id", "2"));
78          writer.commit();
79          try (DirectoryReader open = DirectoryReader.open(writer, true)) {
80              assertEquals(3, open.numDocs());
81              assertEquals(1, open.numDeletedDocs());
82              assertEquals(4, open.maxDoc());
83          }
84          writer.close();
85          if (random().nextBoolean()) {
86              for (String file : dir.listAll()) {
87                  if (file.startsWith("_1")) {
88                      // delete a random file
89                      dir.deleteFile(file);
90                      break;
91                  }
92              }
93          }
94          Lucene.cleanLuceneIndex(dir);
95          if (dir.listAll().length > 0) {
96              assertEquals(dir.listAll().length, 1);
97              assertEquals(dir.listAll()[0], "write.lock");
98          }
99          dir.close();
100     }

102     public void testPruneUnreferencedFiles() throws IOException {
103         MockDirectoryWrapper dir = newMockDirectory();
104         dir.setEnableVirusScanner(false);
105         IndexWriterConfig iwc = newIndexWriterConfig();
106         iwc.setIndexDeletionPolicy(NoDeletionPolicy.INSTANCE);
107         iwc.setMergePolicy(NoMergePolicy.INSTANCE);
108         iwc.setMaxBufferedDocs(2);
109         IndexWriter writer = new IndexWriter(dir, iwc);
110         Document doc = new Document();
111         doc.add(new TextField("id", "1", random().nextBoolean() ? Field.Store.YES : Field.Store.NO));
112         writer.addDocument(doc);
113         writer.commit();
114 
115         doc = new Document();
116         doc.add(new TextField("id", "2", random().nextBoolean() ? Field.Store.YES : Field.Store.NO));
117         writer.addDocument(doc);
118 
119         doc = new Document();
120         doc.add(new TextField("id", "3", random().nextBoolean() ? Field.Store.YES : Field.Store.NO));
121         writer.addDocument(doc);
122 
123         writer.commit();
124         SegmentInfos segmentCommitInfos = Lucene.readSegmentInfos(dir);
125 
126         doc = new Document();
127         doc.add(new TextField("id", "4", random().nextBoolean() ? Field.Store.YES : Field.Store.NO));
128         writer.addDocument(doc);
129 
130         writer.deleteDocuments(new Term("id", "2"));
131         writer.commit();
132         DirectoryReader open = DirectoryReader.open(writer, true);
133         assertEquals(3, open.numDocs());
134         assertEquals(1, open.numDeletedDocs());
135         assertEquals(4, open.maxDoc());
136         open.close();
137         writer.close();
138         SegmentInfos si = Lucene.pruneUnreferencedFiles(segmentCommitInfos.getSegmentsFileName(), dir);
139         assertEquals(si.getSegmentsFileName(), segmentCommitInfos.getSegmentsFileName());
140         open = DirectoryReader.open(dir);
141         assertEquals(3, open.numDocs());
142         assertEquals(0, open.numDeletedDocs());
143         assertEquals(3, open.maxDoc());
144 
145         IndexSearcher s = new IndexSearcher(open);
146         assertEquals(s.search(new TermQuery(new Term("id", "1")), 1).totalHits, 1);
147         assertEquals(s.search(new TermQuery(new Term("id", "2")), 1).totalHits, 1);
148         assertEquals(s.search(new TermQuery(new Term("id", "3")), 1).totalHits, 1);
149         assertEquals(s.search(new TermQuery(new Term("id", "4")), 1).totalHits, 0);
150 
151         for (String file : dir.listAll()) {
152             assertFalse("unexpected file: " + file, file.equals("segments_3") || file.startsWith("_2"));
153         }
154         open.close();
155         dir.close();
156 
157     }

159     public void testFiles() throws IOException {
160         MockDirectoryWrapper dir = newMockDirectory();
161         dir.setEnableVirusScanner(false);
162         IndexWriterConfig iwc = newIndexWriterConfig(new MockAnalyzer(random()));
163         iwc.setMergePolicy(NoMergePolicy.INSTANCE);
164         iwc.setMaxBufferedDocs(2);
165         iwc.setUseCompoundFile(true);
166         IndexWriter writer = new IndexWriter(dir, iwc);
167         Document doc = new Document();
168         doc.add(new TextField("id", "1", random().nextBoolean() ? Field.Store.YES : Field.Store.NO));
169         writer.addDocument(doc);
170         writer.commit();
171         Set<String> files = new HashSet<>();
172         for (String f : Lucene.files(Lucene.readSegmentInfos(dir))) {
173             files.add(f);
174         }
175 
176         assertTrue(files.toString(), files.contains("segments_1"));
177         assertTrue(files.toString(), files.contains("_0.cfs"));
178         assertTrue(files.toString(), files.contains("_0.cfe"));
179         assertTrue(files.toString(), files.contains("_0.si"));
180 
181         doc = new Document();
182         doc.add(new TextField("id", "2", random().nextBoolean() ? Field.Store.YES : Field.Store.NO));
183         writer.addDocument(doc);
184 
185         doc = new Document();
186         doc.add(new TextField("id", "3", random().nextBoolean() ? Field.Store.YES : Field.Store.NO));
187         writer.addDocument(doc);
188         writer.commit();
189 
190         files.clear();
191         for (String f : Lucene.files(Lucene.readSegmentInfos(dir))) {
192             files.add(f);
193         }
194         assertFalse(files.toString(), files.contains("segments_1"));
195         assertTrue(files.toString(), files.contains("segments_2"));
196         assertTrue(files.toString(), files.contains("_0.cfs"));
197         assertTrue(files.toString(), files.contains("_0.cfe"));
198         assertTrue(files.toString(), files.contains("_0.si"));
199 
200         assertTrue(files.toString(), files.contains("_1.cfs"));
201         assertTrue(files.toString(), files.contains("_1.cfe"));
202         assertTrue(files.toString(), files.contains("_1.si"));
203         writer.close();
204         dir.close();
205 
206     }
