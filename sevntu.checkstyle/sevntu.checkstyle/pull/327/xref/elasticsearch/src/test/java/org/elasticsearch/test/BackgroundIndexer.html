46  public class BackgroundIndexer implements AutoCloseable {
47  
48      private final ESLogger logger = Loggers.getLogger(getClass());
49  
50      final Thread[] writers;
51      final CountDownLatch stopLatch;
52      final CopyOnWriteArrayList<Throwable> failures;
53      final AtomicBoolean stop = new AtomicBoolean(false);
54      final AtomicLong idGenerator = new AtomicLong();
55      final AtomicLong indexCounter = new AtomicLong();
56      final CountDownLatch startLatch = new CountDownLatch(1);
57      final AtomicBoolean hasBudget = new AtomicBoolean(false); // when set to true, writers will acquire writes from a semaphore
58      final Semaphore availableBudget = new Semaphore(0);
59  
60      volatile int minFieldSize = 10;
61      volatile int maxFieldSize = 140;

79       * @param type      document type

101     /**
102      * Start indexing in the background using a given number of threads. Indexing will be paused after numOfDocs docs has
103      * been indexed.
104      *
105      * @param index       index name to index into
106      * @param type        document type
107      * @param client      client to use
108      * @param numOfDocs   number of document to index before pausing. Set to -1 to have no limit.
109      * @param writerCount number of indexing threads to use
110      * @param autoStart   set to true to start indexing as soon as all threads have been created.
111      * @param random      random instance to use
112      */
113     public BackgroundIndexer(final String index, final String type, final Client client, final int numOfDocs, final int writerCount,
114                              boolean autoStart, Random random) {
115 
116         if (random == null) {
117             random = RandomizedTest.getRandom();
118         }
119         failures = new CopyOnWriteArrayList<>();
120         writers = new Thread[writerCount];
121         stopLatch = new CountDownLatch(writers.length);
122         logger.info("--> creating {} indexing threads (auto start: [{}], numOfDocs: [{}])", writerCount, autoStart, numOfDocs);
123         for (int i = 0; i < writers.length; i++) {
124             final int indexerId = i;
125             final boolean batch = random.nextBoolean();
126             final Random threadRandom = new Random(random.nextLong());
127             writers[i] = new Thread() {
128                 @Override
129                 public void run() {
130                     long id = -1;
131                     try {
132                         startLatch.await();
133                         logger.info("**** starting indexing thread {}", indexerId);
134                         while (!stop.get()) {
135                             if (batch) {
136                                 int batchSize = threadRandom.nextInt(20) + 1;
137                                 if (hasBudget.get()) {
138                                     batchSize = Math.max(Math.min(batchSize, availableBudget.availablePermits()), 1);// always try to get at least one
139                                     if (!availableBudget.tryAcquire(batchSize, 250, TimeUnit.MILLISECONDS)) {
140                                         // time out -> check if we have to stop.
141                                         continue;
142                                     }
143 
144                                 }
145                                 BulkRequestBuilder bulkRequest = client.prepareBulk();
146                                 for (int i = 0; i < batchSize; i++) {
147                                     id = idGenerator.incrementAndGet();
148                                     bulkRequest.add(client.prepareIndex(index, type, Long.toString(id)).setSource(generateSource(id, threadRandom)));
149                                 }
150                                 BulkResponse bulkResponse = bulkRequest.get();
151                                 for (BulkItemResponse bulkItemResponse : bulkResponse) {
152                                     if (!bulkItemResponse.isFailed()) {
153                                         indexCounter.incrementAndGet();
154                                     } else {
155                                         throw new ElasticsearchException("bulk request failure, id: ["
156                                                 + bulkItemResponse.getFailure().getId() + "] message: " + bulkItemResponse.getFailure().getMessage());
157                                     }
158                                 }
159 
160                             } else {
161 
162                                 if (hasBudget.get() && !availableBudget.tryAcquire(250, TimeUnit.MILLISECONDS)) {
163                                     // time out -> check if we have to stop.
164                                     continue;
165                                 }
166                                 id = idGenerator.incrementAndGet();
167                                 client.prepareIndex(index, type, Long.toString(id) + "-" + indexerId).setSource(generateSource(id, threadRandom)).get();
168                                 indexCounter.incrementAndGet();
169                             }
170                         }
171                         logger.info("**** done indexing thread {}  stop: {} numDocsIndexed: {}", indexerId, stop.get(), indexCounter.get());
172                     } catch (Throwable e) {
173                         failures.add(e);
174                         logger.warn("**** failed indexing thread {} on doc id {}", e, indexerId, id);
175                     } finally {
176                         stopLatch.countDown();
177                     }
178                 }
179             };
180             writers[i].start();
181         }
182 
183         if (autoStart) {
184             start(numOfDocs);
185         }
186     }
187 
188     private XContentBuilder generateSource(long id, Random random) throws IOException {
189         int contentLength = RandomInts.randomIntBetween(random, minFieldSize, maxFieldSize);
190         StringBuilder text = new StringBuilder(contentLength);
191         while (text.length() < contentLength) {
192             int tokenLength = RandomInts.randomIntBetween(random, 1, Math.min(contentLength - text.length(), 10));
193             text.append(" ").append(RandomStrings.randomRealisticUnicodeOfCodepointLength(random, tokenLength));
194         }
195         XContentBuilder builder = XContentFactory.smileBuilder();
196         builder.startObject().field("test", "value" + id)
197                 .field("text", text.toString())
198                 .endObject();
199         return builder;
200 
201     }
202 
203     private void setBudget(int numOfDocs) {
204         logger.debug("updating budget to [{}]", numOfDocs);
205         if (numOfDocs >= 0) {
206             hasBudget.set(true);
207             availableBudget.release(numOfDocs);
208         } else {
209             hasBudget.set(false);
210         }
211 
212     }
213 
214     /** Start indexing with no limit to the number of documents */
215     public void start() {
216         start(-1);
217     }
218 
219     /**
220      * Start indexing
221      *
222      * @param numOfDocs number of document to index before pausing. Set to -1 to have no limit.
223      */
224     public void start(int numOfDocs) {
225         assert !stop.get() : "background indexer can not be started after it has stopped";
226         setBudget(numOfDocs);
227         startLatch.countDown();
228     }
229 
230     /** Pausing indexing by setting current document limit to 0 */
231     public void pauseIndexing() {
232         availableBudget.drainPermits();
233         setBudget(0);
234     }
235 
236     /** Continue indexing after it has paused. No new document limit will be set */
237     public void continueIndexing() {
238         continueIndexing(-1);
239     }
240 
241     /**
242      * Continue indexing after it has paused.
243      *
244      * @param numOfDocs number of document to index before pausing. Set to -1 to have no limit.
245      */
246     public void continueIndexing(int numOfDocs) {
247         setBudget(numOfDocs);
248     }
249 
250     /** Stop all background threads * */
251     public void stop() throws InterruptedException {
252         if (stop.get()) {
253             return;
254         }
255         stop.set(true);
256         Assert.assertThat("timeout while waiting for indexing threads to stop", stopLatch.await(6, TimeUnit.MINUTES), equalTo(true));
257         assertNoFailures();
258     }
259 
260     public long totalIndexedDocs() {
261         return indexCounter.get();
262     }
263 
264     public Throwable[] getFailures() {
265         return failures.toArray(new Throwable[failures.size()]);
266     }
267 
268     public void assertNoFailures() {
269         Assert.assertThat(failures, emptyIterable());
270     }
271 
272     /** the minimum size in code points of a payload field in the indexed documents */
273     public void setMinFieldSize(int fieldSize) {
274         minFieldSize = fieldSize;
275     }
276 
277     /** the minimum size in code points of a payload field in the indexed documents */
278     public void setMaxFieldSize(int fieldSize) {
279         maxFieldSize = fieldSize;
280     }
281 
282     @Override
283     public void close() throws Exception {
284         stop();
285     }
