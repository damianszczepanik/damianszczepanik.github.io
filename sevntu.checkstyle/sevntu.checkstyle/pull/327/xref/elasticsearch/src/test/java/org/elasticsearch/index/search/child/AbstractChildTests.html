51      /**
52       * The name of the field within the child type that stores a score to use in test queries.
53       * <p />
54       * Its type is {@code double}.
55       */
56      protected static String CHILD_SCORE_NAME = "childScore";
57  
58      static SearchContext createSearchContext(String indexName, String parentType, String childType) throws IOException {
59          IndexService indexService = createIndex(indexName);
60          MapperService mapperService = indexService.mapperService();
61          // Parent/child parsers require that the parent and child type to be presented in mapping
62          // Sometimes we want a nested object field in the parent type that triggers nonNestedDocsFilter to be used
63          mapperService.merge(parentType, new CompressedString(PutMappingRequest.buildFromSimplifiedDef(parentType, "nested_field", random().nextBoolean() ? "type=nested" : "type=object").string()), true);
64          mapperService.merge(childType, new CompressedString(PutMappingRequest.buildFromSimplifiedDef(childType, "_parent", "type=" + parentType, CHILD_SCORE_NAME, "type=double").string()), true);
65          return createSearchContext(indexService);
66      }

68      static void assertBitSet(BitSet actual, BitSet expected, IndexSearcher searcher) throws IOException {
69          assertBitSet(new BitDocIdSet(actual), new BitDocIdSet(expected), searcher);
70      }

72      static void assertBitSet(BitDocIdSet actual, BitDocIdSet expected, IndexSearcher searcher) throws IOException {
73          if (!equals(expected, actual)) {
74              Description description = new StringDescription();
75              description.appendText(reason(actual, expected, searcher));
76              description.appendText("\nExpected: ");
77              description.appendValue(expected);
78              description.appendText("\n     got: ");
79              description.appendValue(actual);
80              description.appendText("\n");
81              throw new java.lang.AssertionError(description.toString());
82          }
83      }
84      
85      static boolean equals(BitDocIdSet expected, BitDocIdSet actual) {
86          if (actual == null && expected == null) {
87              return true;
88          } else if (actual == null || expected == null) {
89              return false;
90          }
91          BitSet actualBits = actual.bits();
92          BitSet expectedBits = expected.bits();
93          if (actualBits.length() != expectedBits.length()) {
94              return false;
95          }
96          for (int i = 0; i < expectedBits.length(); i++) {
97              if (expectedBits.get(i) != actualBits.get(i)) {
98                  return false;
99              }
100         }
101         return true;
102     }
103 
104     static String reason(BitDocIdSet actual, BitDocIdSet expected, IndexSearcher indexSearcher) throws IOException {
105         StringBuilder builder = new StringBuilder();
106         builder.append("expected cardinality:").append(expected.bits().cardinality()).append('\n');
107         DocIdSetIterator iterator = expected.iterator();
108         for (int doc = iterator.nextDoc(); doc != DocIdSetIterator.NO_MORE_DOCS; doc = iterator.nextDoc()) {
109             builder.append("Expected doc[").append(doc).append("] with id value ").append(indexSearcher.doc(doc).get(UidFieldMapper.NAME)).append('\n');
110         }
111         builder.append("actual cardinality: ").append(actual.bits().cardinality()).append('\n');
112         iterator = actual.iterator();
113         for (int doc = iterator.nextDoc(); doc != DocIdSetIterator.NO_MORE_DOCS; doc = iterator.nextDoc()) {
114             builder.append("Actual doc[").append(doc).append("] with id value ").append(indexSearcher.doc(doc).get(UidFieldMapper.NAME)).append('\n');
115         }
116         return builder.toString();
117     }

119     static void assertTopDocs(TopDocs actual, TopDocs expected) {
120         assertThat("actual.totalHits != expected.totalHits", actual.totalHits, equalTo(expected.totalHits));
121         assertThat("actual.getMaxScore() != expected.getMaxScore()", actual.getMaxScore(), equalTo(expected.getMaxScore()));
122         assertThat("actual.scoreDocs.length != expected.scoreDocs.length", actual.scoreDocs.length, equalTo(actual.scoreDocs.length));
123         for (int i = 0; i < actual.scoreDocs.length; i++) {
124             ScoreDoc actualHit = actual.scoreDocs[i];
125             ScoreDoc expectedHit = expected.scoreDocs[i];
126             assertThat("actualHit.doc != expectedHit.doc", actualHit.doc, equalTo(expectedHit.doc));
127             assertThat("actualHit.score != expectedHit.score", actualHit.score, equalTo(expectedHit.score));
128         }
129     }

133     }

139     static Query parseQuery(QueryBuilder queryBuilder) throws IOException {
140         QueryParseContext context = new QueryParseContext(new Index("test"), SearchContext.current().queryParserService());
141         XContentParser parser = XContentHelper.createParser(queryBuilder.buildAsBytes());
142         context.reset(parser);
143         return context.parseInnerQuery();
144     }
