20  package org.elasticsearch.index.mapper.boost;
21  
22  import org.elasticsearch.common.xcontent.XContentFactory;
23  import org.elasticsearch.index.mapper.DocumentMapper;
24  import org.elasticsearch.index.mapper.ParsedDocument;
25  import org.elasticsearch.test.ElasticsearchSingleNodeTest;
26  import org.junit.Test;
27  
28  import static org.hamcrest.Matchers.equalTo;
29  
30  public class CustomBoostMappingTests extends ElasticsearchSingleNodeTest {
31  
32      @Test
33      public void testCustomBoostValues() throws Exception {
34          String mapping = XContentFactory.jsonBuilder().startObject().startObject("type").startObject("properties")
35                  .startObject("s_field").field("type", "string").endObject()
36                  .startObject("l_field").field("type", "long").startObject("norms").field("enabled", true).endObject().endObject()
37                  .startObject("i_field").field("type", "integer").startObject("norms").field("enabled", true).endObject().endObject()
38                  .startObject("sh_field").field("type", "short").startObject("norms").field("enabled", true).endObject().endObject()
39                  .startObject("b_field").field("type", "byte").startObject("norms").field("enabled", true).endObject().endObject()
40                  .startObject("d_field").field("type", "double").startObject("norms").field("enabled", true).endObject().endObject()
41                  .startObject("f_field").field("type", "float").startObject("norms").field("enabled", true).endObject().endObject()
42                  .startObject("date_field").field("type", "date").startObject("norms").field("enabled", true).endObject().endObject()
43                  .endObject().endObject().endObject().string();
44  
45          DocumentMapper mapper = createIndex("test").mapperService().documentMapperParser().parse(mapping);
46  
47          ParsedDocument doc = mapper.parse("type", "1", XContentFactory.jsonBuilder().startObject()
48                  .startObject("s_field").field("value", "s_value").field("boost", 2.0f).endObject()
49                  .startObject("l_field").field("value", 1l).field("boost", 3.0f).endObject()
50                  .startObject("i_field").field("value", 1).field("boost", 4.0f).endObject()
51                  .startObject("sh_field").field("value", 1).field("boost", 5.0f).endObject()
52                  .startObject("b_field").field("value", 1).field("boost", 6.0f).endObject()
53                  .startObject("d_field").field("value", 1).field("boost", 7.0f).endObject()
54                  .startObject("f_field").field("value", 1).field("boost", 8.0f).endObject()
55                  .startObject("date_field").field("value", "20100101").field("boost", 9.0f).endObject()
56                  .endObject().bytes());
57  
58          assertThat(doc.rootDoc().getField("s_field").boost(), equalTo(2.0f));
59          assertThat(doc.rootDoc().getField("l_field").boost(), equalTo(3.0f));
60          assertThat(doc.rootDoc().getField("i_field").boost(), equalTo(4.0f));
61          assertThat(doc.rootDoc().getField("sh_field").boost(), equalTo(5.0f));
62          assertThat(doc.rootDoc().getField("b_field").boost(), equalTo(6.0f));
63          assertThat(doc.rootDoc().getField("d_field").boost(), equalTo(7.0f));
64          assertThat(doc.rootDoc().getField("f_field").boost(), equalTo(8.0f));
65          assertThat(doc.rootDoc().getField("date_field").boost(), equalTo(9.0f));
66      }
67  }
