20  package org.elasticsearch.transport;
21  
22  import org.elasticsearch.test.ElasticsearchIntegrationTest;
23  import org.junit.Test;
24  
25  import static org.hamcrest.CoreMatchers.either;
26  import static org.hamcrest.CoreMatchers.startsWith;
27  
28  /**
29   * This test verifies that all of the action names follow our defined naming conventions.
30   * The identified categories are:
31   * - indices:admin: apis that allow to perform administration tasks against indices
32   * - indices:data: apis that are about data
33   * - indices:read: apis that read data
34   * - indices:write: apis that write data
35   * - cluster:admin: cluster apis that allow to perform administration tasks
36   * - cluster:monitor: cluster apis that allow to monitor the system
37   * - internal: internal actions that are used from node to node but not directly exposed to users
38   *
39   * Any transport action belongs to one of the above categories and its name starts with its category, followed by a '/'
40   * and the name of the api itself (e.g. cluster:admin/nodes/restart).
41   * When an api exposes multiple transport handlers, some of which are invoked internally during the execution of the api,
42   * we use the `[n]` suffix to identify node actions and the `[s]` suffix to identify shard actions.
43   */
44  public class ActionNamesTests extends ElasticsearchIntegrationTest {
45  
46      @Test
47      @SuppressWarnings("unchecked")
48      public void testActionNamesCategories() throws NoSuchFieldException, IllegalAccessException {
49          TransportService transportService = internalCluster().getInstance(TransportService.class);
50          for (String action : transportService.serverHandlers.keySet()) {
51              assertThat("action doesn't belong to known category", action, either(startsWith("indices:admin")).or(startsWith("indices:monitor"))
52                      .or(startsWith("indices:data/read")).or(startsWith("indices:data/write"))
53                      .or(startsWith("indices:data/benchmark"))
54                      .or(startsWith("cluster:admin")).or(startsWith("cluster:monitor"))
55                      .or(startsWith("internal:")));
56          }
57      }
58  }
