55  /**
56   * Tests making sure that the reduce is propagated to all aggregations in the hierarchy when executing on a single shard
57   * These tests are based on the date histogram in combination of min_doc_count=0. In order for the date histogram to
58   * compute empty buckets, its {@code reduce()} method must be called. So by adding the date histogram under other buckets,
59   * we can make sure that the reduce is properly propagated by checking that empty buckets were created.
60   */
61  @ElasticsearchIntegrationTest.SuiteScopeTest
62  public class ShardReduceTests extends ElasticsearchIntegrationTest {
63  
64      private IndexRequestBuilder indexDoc(String date, int value) throws Exception {
65          return client().prepareIndex("idx", "type").setSource(jsonBuilder()
66                  .startObject()
67                  .field("value", value)
68                  .field("ip", "10.0.0." + value)
69                  .field("location", GeoHashUtils.encode(52, 5, 12))
70                  .field("date", date)
71                  .field("term-l", 1)
72                  .field("term-d", 1.5)
73                  .field("term-s", "term")
74                  .startObject("nested")
75                  .field("date", date)
76                  .endObject()
77                  .endObject());
78      }
79  
80      @Override
81      public void setupSuiteScopeCluster() throws Exception {
82          assertAcked(prepareCreate("idx")
83                  .addMapping("type", "nested", "type=nested", "ip", "type=ip", "location", "type=geo_point"));
84  
85          indexRandom(true,
86                  indexDoc("2014-01-01", 1),
87                  indexDoc("2014-01-02", 2),
88                  indexDoc("2014-01-04", 3));
89          ensureSearchable();
90      }

92      @Test
93      public void testGlobal() throws Exception {
94  
95          SearchResponse response = client().prepareSearch("idx")
96                  .setQuery(QueryBuilders.matchAllQuery())
97                  .addAggregation(global("global")
98                          .subAggregation(dateHistogram("histo").field("date").interval(DateHistogramInterval.DAY).minDocCount(0)))
99                  .execute().actionGet();
100 
101         assertSearchResponse(response);
102 
103         Global global = response.getAggregations().get("global");
104         Histogram histo = global.getAggregations().get("histo");
105         assertThat(histo.getBuckets().size(), equalTo(4));
106     }
