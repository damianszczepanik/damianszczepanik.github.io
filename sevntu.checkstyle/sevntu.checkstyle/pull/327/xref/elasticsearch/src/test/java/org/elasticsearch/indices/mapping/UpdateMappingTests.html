57      @Test
58      public void dynamicUpdates() throws Exception {
59          client().admin().indices().prepareCreate("test")
60                  .setSettings(
61                          settingsBuilder()
62                                  .put("index.number_of_shards", 1)
63                                  .put("index.number_of_replicas", 0)
64                  ).execute().actionGet();
65          client().admin().cluster().prepareHealth().setWaitForEvents(Priority.LANGUID).setWaitForGreenStatus().execute().actionGet();
66  
67          int recCount = randomIntBetween(200, 600);
68          int numberOfTypes = randomIntBetween(1, 5);
69          List<IndexRequestBuilder> indexRequests = Lists.newArrayList();
70          for (int rec = 0; rec < recCount; rec++) {
71              String type = "type" + (rec % numberOfTypes);
72              String fieldName = "field_" + type + "_" + rec;
73              indexRequests.add(client().prepareIndex("test", type, Integer.toString(rec)).setSource(fieldName, "some_value"));
74          }
75          indexRandom(true, indexRequests);
76  
77          logger.info("checking all the documents are there");
78          RefreshResponse refreshResponse = client().admin().indices().prepareRefresh().execute().actionGet();
79          assertThat(refreshResponse.getFailedShards(), equalTo(0));
80          CountResponse response = client().prepareCount("test").execute().actionGet();
81          assertThat(response.getCount(), equalTo((long) recCount));
82  
83          logger.info("checking all the fields are in the mappings");
84  
85          for (int rec = 0; rec < recCount; rec++) {
86              String type = "type" + (rec % numberOfTypes);
87              String fieldName = "field_" + type + "_" + rec;
88              waitForConcreteMappingsOnAll("test", type, fieldName);
89          }
90      }

213     @Test
214     public void updateIncludeExclude() throws Exception {
215         assertAcked(prepareCreate("test").addMapping("type",
216                 jsonBuilder().startObject().startObject("type").startObject("properties")
217                         .startObject("normal").field("type", "long").endObject()
218                         .startObject("exclude").field("type", "long").endObject()
219                         .startObject("include").field("type", "long").endObject()
220                         .endObject().endObject().endObject()));
221         ensureGreen(); // make sure that replicas are initialized so the refresh command will work them too
222 
223         logger.info("Index doc");
224         index("test", "type", "1", JsonXContent.contentBuilder().startObject()
225                         .field("normal", 1).field("exclude", 1).field("include", 1)
226                         .endObject()
227         );
228         refresh(); // commit it for later testing.
229 
230 
231         logger.info("Adding exclude settings");
232         PutMappingResponse putResponse = client().admin().indices().preparePutMapping("test").setType("type").setSource(
233                 JsonXContent.contentBuilder().startObject().startObject("type")
234                         .startObject("_source")
235                         .startArray("excludes").value("exclude").endArray()
236                         .endObject().endObject()
237         ).get();
238 
239         assertTrue(putResponse.isAcknowledged());
240 
241         // changed mapping doesn't affect indexed documents (checking backward compatibility)
242         GetResponse getResponse = client().prepareGet("test", "type", "1").setRealtime(false).get();
243         assertThat(getResponse.getSource(), hasKey("normal"));
244         assertThat(getResponse.getSource(), hasKey("exclude"));
245         assertThat(getResponse.getSource(), hasKey("include"));
246 
247 
248         logger.info("Index doc again");
249         index("test", "type", "1", JsonXContent.contentBuilder().startObject()
250                         .field("normal", 2).field("exclude", 1).field("include", 2)
251                         .endObject()
252         );
253 
254         // but do affect newly indexed docs
255         getResponse = get("test", "type", "1");
256         assertThat(getResponse.getSource(), hasKey("normal"));
257         assertThat(getResponse.getSource(), not(hasKey("exclude")));
258         assertThat(getResponse.getSource(), hasKey("include"));
259 
260 
261         logger.info("Changing mapping to includes");
262         putResponse = client().admin().indices().preparePutMapping("test").setType("type").setSource(
263                 JsonXContent.contentBuilder().startObject().startObject("type")
264                         .startObject("_source")
265                         .startArray("excludes").endArray()
266                         .startArray("includes").value("include").endArray()
267                         .endObject().endObject()
268         ).get();
269         assertTrue(putResponse.isAcknowledged());
270 
271         GetMappingsResponse getMappingsResponse = client().admin().indices().prepareGetMappings("test").get();
272         MappingMetaData typeMapping = getMappingsResponse.getMappings().get("test").get("type");
273         assertThat((Map<String, Object>) typeMapping.getSourceAsMap().get("_source"), hasKey("includes"));
274         ArrayList<String> includes = (ArrayList<String>) ((Map<String, Object>) typeMapping.getSourceAsMap().get("_source")).get("includes");
275         assertThat(includes, contains("include"));
276         assertThat((Map<String, Object>) typeMapping.getSourceAsMap().get("_source"), hasKey("excludes"));
277         assertThat((ArrayList<String>) ((Map<String, Object>) typeMapping.getSourceAsMap().get("_source")).get("excludes"), emptyIterable());
278 
279 
280         logger.info("Indexing doc yet again");
281         index("test", "type", "1", JsonXContent.contentBuilder().startObject()
282                         .field("normal", 3).field("exclude", 3).field("include", 3)
283                         .endObject()
284         );
285 
286         getResponse = get("test", "type", "1");
287         assertThat(getResponse.getSource(), not(hasKey("normal")));
288         assertThat(getResponse.getSource(), not(hasKey("exclude")));
289         assertThat(getResponse.getSource(), hasKey("include"));
290 
291 
292         logger.info("Adding excludes, but keep includes");
293         putResponse = client().admin().indices().preparePutMapping("test").setType("type").setSource(
294                 JsonXContent.contentBuilder().startObject().startObject("type")
295                         .startObject("_source")
296                         .startArray("excludes").value("*.excludes").endArray()
297                         .endObject().endObject()
298         ).get();
299         assertTrue(putResponse.isAcknowledged());
300 
301         getMappingsResponse = client().admin().indices().prepareGetMappings("test").get();
302         typeMapping = getMappingsResponse.getMappings().get("test").get("type");
303         assertThat((Map<String, Object>) typeMapping.getSourceAsMap().get("_source"), hasKey("includes"));
304         includes = (ArrayList<String>) ((Map<String, Object>) typeMapping.getSourceAsMap().get("_source")).get("includes");
305         assertThat(includes, contains("include"));
306         assertThat((Map<String, Object>) typeMapping.getSourceAsMap().get("_source"), hasKey("excludes"));
307         ArrayList<String> excludes = (ArrayList<String>) ((Map<String, Object>) typeMapping.getSourceAsMap().get("_source")).get("excludes");
308         assertThat(excludes, contains("*.excludes"));
309

314     @Test
315     public void updateDefaultMappingSettings() throws Exception {
316 
317         logger.info("Creating index with _default_ mappings");
318         client().admin().indices().prepareCreate("test").addMapping(MapperService.DEFAULT_MAPPING,
319                 JsonXContent.contentBuilder().startObject().startObject(MapperService.DEFAULT_MAPPING)
320                         .field("date_detection", false)
321                         .endObject().endObject()
322         ).get();
323 
324         GetMappingsResponse getResponse = client().admin().indices().prepareGetMappings("test").addTypes(MapperService.DEFAULT_MAPPING).get();
325         Map<String, Object> defaultMapping = getResponse.getMappings().get("test").get(MapperService.DEFAULT_MAPPING).sourceAsMap();
326         assertThat(defaultMapping, hasKey("date_detection"));
327 
328 
329         logger.info("Emptying _default_ mappings");
330         // now remove it
331         PutMappingResponse putResponse = client().admin().indices().preparePutMapping("test").setType(MapperService.DEFAULT_MAPPING).setSource(
332                 JsonXContent.contentBuilder().startObject().startObject(MapperService.DEFAULT_MAPPING)
333                         .endObject().endObject()
334         ).get();
335         assertThat(putResponse.isAcknowledged(), equalTo(true));
336         logger.info("Done Emptying _default_ mappings");
337 
338         getResponse = client().admin().indices().prepareGetMappings("test").addTypes(MapperService.DEFAULT_MAPPING).get();
339         defaultMapping = getResponse.getMappings().get("test").get(MapperService.DEFAULT_MAPPING).sourceAsMap();
340         assertThat(defaultMapping, not(hasKey("date_detection")));
341 
342         // now test you can change stuff that are normally unchangeable
343         logger.info("Creating _default_ mappings with an analyzed field");
344         putResponse = client().admin().indices().preparePutMapping("test").setType(MapperService.DEFAULT_MAPPING).setSource(
345                 JsonXContent.contentBuilder().startObject().startObject(MapperService.DEFAULT_MAPPING)
346                         .startObject("properties").startObject("f").field("type", "string").field("index", "analyzed").endObject().endObject()
347                         .endObject().endObject()
348         ).get();
349         assertThat(putResponse.isAcknowledged(), equalTo(true));
350 
351 
352         logger.info("Changing _default_ mappings field from analyzed to non-analyzed");
353         putResponse = client().admin().indices().preparePutMapping("test").setType(MapperService.DEFAULT_MAPPING).setSource(
354                 JsonXContent.contentBuilder().startObject().startObject(MapperService.DEFAULT_MAPPING)
355                         .startObject("properties").startObject("f").field("type", "string").field("index", "not_analyzed").endObject().endObject()
356                         .endObject().endObject()
357         ).get();
358         assertThat(putResponse.isAcknowledged(), equalTo(true));
359         logger.info("Done changing _default_ mappings field from analyzed to non-analyzed");
360 
361         getResponse = client().admin().indices().prepareGetMappings("test").addTypes(MapperService.DEFAULT_MAPPING).get();
362         defaultMapping = getResponse.getMappings().get("test").get(MapperService.DEFAULT_MAPPING).sourceAsMap();
363         Map<String, Object> fieldSettings = (Map<String, Object>) ((Map) defaultMapping.get("properties")).get("f");
364         assertThat(fieldSettings, hasEntry("index", (Object) "not_analyzed"));
365 
366         // but we still validate the _default_ type
367         logger.info("Confirming _default_ mappings validation");
368         assertThrows(client().admin().indices().preparePutMapping("test").setType(MapperService.DEFAULT_MAPPING).setSource(
369                 JsonXContent.contentBuilder().startObject().startObject(MapperService.DEFAULT_MAPPING)
370                         .startObject("properties").startObject("f").field("type", "DOESNT_EXIST").endObject().endObject()
371                         .endObject().endObject()
372         ), MapperParsingException.class);
373 
374     }

376     @Test
377     public void updateMappingConcurrently() throws Throwable {
378         createIndex("test1", "test2");
379 
380         // This is important. The test assumes all nodes are aware of all indices. Due to initializing shard throttling
381         // not all shards are allocated with the initial create index. Wait for it..
382         ensureYellow();
383 
384         final Throwable[] threadException = new Throwable[1];
385         final AtomicBoolean stop = new AtomicBoolean(false);
386         Thread[] threads = new Thread[3];
387         final CyclicBarrier barrier = new CyclicBarrier(threads.length);
388         final ArrayList<Client> clientArray = new ArrayList<>();
389         for (Client c : clients()) {
390             clientArray.add(c);
391         }
392 
393         for (int j = 0; j < threads.length; j++) {
394             threads[j] = new Thread(new Runnable() {
395                 @SuppressWarnings("unchecked")
396                 @Override
397                 public void run() {
398                     try {
399                         barrier.await();
400 
401                         for (int i = 0; i < 100; i++) {
402                             if (stop.get()) {
403                                 return;
404                             }
405 
406                             Client client1 = clientArray.get(i % clientArray.size());
407                             Client client2 = clientArray.get((i + 1) % clientArray.size());
408                             String indexName = i % 2 == 0 ? "test2" : "test1";
409                             String typeName = "type" + (i % 10);
410                             String fieldName = Thread.currentThread().getName() + "_" + i;
411 
412                             PutMappingResponse response = client1.admin().indices().preparePutMapping(indexName).setType(typeName).setSource(
413                                     JsonXContent.contentBuilder().startObject().startObject(typeName)
414                                             .startObject("properties").startObject(fieldName).field("type", "string").endObject().endObject()
415                                             .endObject().endObject()
416                             ).get();
417 
418                             assertThat(response.isAcknowledged(), equalTo(true));
419                             GetMappingsResponse getMappingResponse = client2.admin().indices().prepareGetMappings(indexName).get();
420                             ImmutableOpenMap<String, MappingMetaData> mappings = getMappingResponse.getMappings().get(indexName);
421                             assertThat(mappings.containsKey(typeName), equalTo(true));
422                             assertThat(((Map<String, Object>) mappings.get(typeName).getSourceAsMap().get("properties")).keySet(), Matchers.hasItem(fieldName));
423                         }
424                     } catch (Throwable t) {
425                         threadException[0] = t;
426                         stop.set(true);
427                     }
428                 }
429             });
430 
431             threads[j].setName("t_" + j);
432             threads[j].start();
433         }
434 
435         for (Thread t : threads) t.join();
436 
437         if (threadException[0] != null) {
438             throw threadException[0];
439         }
440
