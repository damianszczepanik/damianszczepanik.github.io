20  package org.elasticsearch.cluster.routing.allocation;
21  
22  import org.elasticsearch.Version;
23  import org.elasticsearch.cluster.ClusterState;
24  import org.elasticsearch.cluster.metadata.IndexMetaData;
25  import org.elasticsearch.cluster.metadata.MetaData;
26  import org.elasticsearch.cluster.node.DiscoveryNodes;
27  import org.elasticsearch.cluster.routing.RoutingNodes;
28  import org.elasticsearch.cluster.routing.RoutingTable;
29  import org.elasticsearch.common.logging.ESLogger;
30  import org.elasticsearch.common.logging.Loggers;
31  import org.elasticsearch.test.ElasticsearchAllocationTestCase;
32  import org.junit.Test;
33  
34  import static org.elasticsearch.cluster.routing.ShardRoutingState.*;
35  import static org.elasticsearch.common.settings.ImmutableSettings.settingsBuilder;
36  import static org.hamcrest.Matchers.*;
37  
38  /**
39   *
40   */
41  public class UpdateNumberOfReplicasTests extends ElasticsearchAllocationTestCase {
42  
43      private final ESLogger logger = Loggers.getLogger(UpdateNumberOfReplicasTests.class);
44  
45      @Test
46      public void testUpdateNumberOfReplicas() {
47          AllocationService strategy = createAllocationService(settingsBuilder().put("cluster.routing.allocation.concurrent_recoveries", 10).build());
48  
49          logger.info("Building initial routing table");
50  
51          MetaData metaData = MetaData.builder()
52                  .put(IndexMetaData.builder("test").settings(settings(Version.CURRENT)).numberOfShards(1).numberOfReplicas(1))
53                  .build();
54  
55          RoutingTable routingTable = RoutingTable.builder()
56                  .addAsNew(metaData.index("test"))
57                  .build();
58  
59          ClusterState clusterState = ClusterState.builder(org.elasticsearch.cluster.ClusterName.DEFAULT).metaData(metaData).routingTable(routingTable).build();
60  
61          assertThat(routingTable.index("test").shards().size(), equalTo(1));
62          assertThat(routingTable.index("test").shard(0).size(), equalTo(2));
63          assertThat(routingTable.index("test").shard(0).shards().size(), equalTo(2));
64          assertThat(routingTable.index("test").shard(0).shards().get(0).state(), equalTo(UNASSIGNED));
65          assertThat(routingTable.index("test").shard(0).shards().get(1).state(), equalTo(UNASSIGNED));
66          assertThat(routingTable.index("test").shard(0).shards().get(0).currentNodeId(), nullValue());
67          assertThat(routingTable.index("test").shard(0).shards().get(1).currentNodeId(), nullValue());
68  
69  
70          logger.info("Adding two nodes and performing rerouting");
71          clusterState = ClusterState.builder(clusterState).nodes(DiscoveryNodes.builder().put(newNode("node1")).put(newNode("node2"))).build();
72  
73          RoutingTable prevRoutingTable = routingTable;
74          routingTable = strategy.reroute(clusterState).routingTable();
75          clusterState = ClusterState.builder(clusterState).routingTable(routingTable).build();
76  
77          logger.info("Start all the primary shards");
78          RoutingNodes routingNodes = clusterState.routingNodes();
79          prevRoutingTable = routingTable;
80          routingTable = strategy.applyStartedShards(clusterState, routingNodes.shardsWithState(INITIALIZING)).routingTable();
81          clusterState = ClusterState.builder(clusterState).routingTable(routingTable).build();
82  
83          logger.info("Start all the replica shards");
84          routingNodes = clusterState.routingNodes();
85          prevRoutingTable = routingTable;
86          routingTable = strategy.applyStartedShards(clusterState, routingNodes.shardsWithState(INITIALIZING)).routingTable();
87          clusterState = ClusterState.builder(clusterState).routingTable(routingTable).build();
88          final String nodeHoldingPrimary = routingTable.index("test").shard(0).primaryShard().currentNodeId();
89          final String nodeHoldingReplica = routingTable.index("test").shard(0).replicaShards().get(0).currentNodeId();
90          assertThat(nodeHoldingPrimary, not(equalTo(nodeHoldingReplica)));
91          assertThat(prevRoutingTable != routingTable, equalTo(true));
92          assertThat(routingTable.index("test").shards().size(), equalTo(1));
93          assertThat(routingTable.index("test").shard(0).size(), equalTo(2));
94          assertThat(routingTable.index("test").shard(0).shards().size(), equalTo(2));
95          assertThat(routingTable.index("test").shard(0).primaryShard().state(), equalTo(STARTED));
96          assertThat(routingTable.index("test").shard(0).primaryShard().currentNodeId(), equalTo(nodeHoldingPrimary));
97          assertThat(routingTable.index("test").shard(0).replicaShards().size(), equalTo(1));
98          assertThat(routingTable.index("test").shard(0).replicaShards().get(0).state(), equalTo(STARTED));
99          assertThat(routingTable.index("test").shard(0).replicaShards().get(0).currentNodeId(), equalTo(nodeHoldingReplica));
100 
101 
102         logger.info("add another replica");
103         routingNodes = clusterState.routingNodes();
104         prevRoutingTable = routingTable;
105         routingTable = RoutingTable.builder(routingTable).updateNumberOfReplicas(2).build();
106         metaData = MetaData.builder(clusterState.metaData()).updateNumberOfReplicas(2).build();
107         clusterState = ClusterState.builder(clusterState).routingTable(routingTable).metaData(metaData).build();
108 
109         assertThat(clusterState.metaData().index("test").numberOfReplicas(), equalTo(2));
110 
111         assertThat(prevRoutingTable != routingTable, equalTo(true));
112         assertThat(routingTable.index("test").shards().size(), equalTo(1));
113         assertThat(routingTable.index("test").shard(0).size(), equalTo(3));
114         assertThat(routingTable.index("test").shard(0).primaryShard().state(), equalTo(STARTED));
115         assertThat(routingTable.index("test").shard(0).primaryShard().currentNodeId(), equalTo(nodeHoldingPrimary));
116         assertThat(routingTable.index("test").shard(0).replicaShards().size(), equalTo(2));
117         assertThat(routingTable.index("test").shard(0).replicaShards().get(0).state(), equalTo(STARTED));
118         assertThat(routingTable.index("test").shard(0).replicaShards().get(0).currentNodeId(), equalTo(nodeHoldingReplica));
119         assertThat(routingTable.index("test").shard(0).replicaShards().get(1).state(), equalTo(UNASSIGNED));
120 
121         logger.info("Add another node and start the added replica");
122         clusterState = ClusterState.builder(clusterState).nodes(DiscoveryNodes.builder(clusterState.nodes()).put(newNode("node3"))).build();
123         prevRoutingTable = routingTable;
124         routingTable = strategy.reroute(clusterState).routingTable();
125         clusterState = ClusterState.builder(clusterState).routingTable(routingTable).build();
126 
127         assertThat(prevRoutingTable != routingTable, equalTo(true));
128         assertThat(routingTable.index("test").shards().size(), equalTo(1));
129         assertThat(routingTable.index("test").shard(0).size(), equalTo(3));
130         assertThat(routingTable.index("test").shard(0).primaryShard().state(), equalTo(STARTED));
131         assertThat(routingTable.index("test").shard(0).primaryShard().currentNodeId(), equalTo(nodeHoldingPrimary));
132         assertThat(routingTable.index("test").shard(0).replicaShards().size(), equalTo(2));
133         assertThat(routingTable.index("test").shard(0).replicaShardsWithState(STARTED).size(), equalTo(1));
134         assertThat(routingTable.index("test").shard(0).replicaShardsWithState(STARTED).get(0).currentNodeId(), equalTo(nodeHoldingReplica));
135         assertThat(routingTable.index("test").shard(0).replicaShardsWithState(INITIALIZING).size(), equalTo(1));
136         assertThat(routingTable.index("test").shard(0).replicaShardsWithState(INITIALIZING).get(0).currentNodeId(), equalTo("node3"));
137 
138         routingNodes = clusterState.routingNodes();
139         prevRoutingTable = routingTable;
140         routingTable = strategy.applyStartedShards(clusterState, routingNodes.shardsWithState(INITIALIZING)).routingTable();
141         clusterState = ClusterState.builder(clusterState).routingTable(routingTable).build();
142 
143         assertThat(prevRoutingTable != routingTable, equalTo(true));
144         assertThat(routingTable.index("test").shards().size(), equalTo(1));
145         assertThat(routingTable.index("test").shard(0).size(), equalTo(3));
146         assertThat(routingTable.index("test").shard(0).primaryShard().state(), equalTo(STARTED));
147         assertThat(routingTable.index("test").shard(0).primaryShard().currentNodeId(), equalTo(nodeHoldingPrimary));
148         assertThat(routingTable.index("test").shard(0).replicaShards().size(), equalTo(2));
149         assertThat(routingTable.index("test").shard(0).replicaShardsWithState(STARTED).size(), equalTo(2));
150         assertThat(routingTable.index("test").shard(0).replicaShardsWithState(STARTED).get(0).currentNodeId(), anyOf(equalTo(nodeHoldingReplica), equalTo("node3")));
151         assertThat(routingTable.index("test").shard(0).replicaShardsWithState(STARTED).get(1).currentNodeId(), anyOf(equalTo(nodeHoldingReplica), equalTo("node3")));
152 
153         logger.info("now remove a replica");
154         routingNodes = clusterState.routingNodes();
155         prevRoutingTable = routingTable;
156         routingTable = RoutingTable.builder(routingTable).updateNumberOfReplicas(1).build();
157         metaData = MetaData.builder(clusterState.metaData()).updateNumberOfReplicas(1).build();
158         clusterState = ClusterState.builder(clusterState).routingTable(routingTable).metaData(metaData).build();
159 
160         assertThat(clusterState.metaData().index("test").numberOfReplicas(), equalTo(1));
161 
162         assertThat(prevRoutingTable != routingTable, equalTo(true));
163         assertThat(routingTable.index("test").shards().size(), equalTo(1));
164         assertThat(routingTable.index("test").shard(0).size(), equalTo(2));
165         assertThat(routingTable.index("test").shard(0).primaryShard().state(), equalTo(STARTED));
166         assertThat(routingTable.index("test").shard(0).primaryShard().currentNodeId(), equalTo(nodeHoldingPrimary));
167         assertThat(routingTable.index("test").shard(0).replicaShards().size(), equalTo(1));
168         assertThat(routingTable.index("test").shard(0).replicaShards().get(0).state(), equalTo(STARTED));
169         assertThat(routingTable.index("test").shard(0).replicaShards().get(0).currentNodeId(), anyOf(equalTo(nodeHoldingReplica), equalTo("node3")));
170 
171         logger.info("do a reroute, should remain the same");
172         prevRoutingTable = routingTable;
173         routingTable = strategy.reroute(clusterState).routingTable();
174         clusterState = ClusterState.builder(clusterState).routingTable(routingTable).build();
175 
176         assertThat(prevRoutingTable != routingTable, equalTo(false));
177     }
178 }
