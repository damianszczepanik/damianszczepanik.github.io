55  public class GlobalOrdinalsBenchmark {
56  
57      private static final String INDEX_NAME = "index";
58      private static final String TYPE_NAME = "type";
59      private static final int QUERY_WARMUP = 25;
60      private static final int QUERY_COUNT = 100;
61      private static final int FIELD_START = 1;
62      private static final int FIELD_LIMIT = 1 << 22;
63      private static final boolean USE_DOC_VALUES = false;
64  
65      static long COUNT = SizeValue.parseSizeValue("5m").singles();
66      static Node node;
67      static Client client;
68  
69      public static void main(String[] args) throws Exception {
70          System.setProperty("es.logger.prefix", "");
71          Natives.tryMlockall();
72          Random random = new Random();
73  
74          Settings settings = settingsBuilder()
75                  .put("index.refresh_interval", "-1")
76                  .put(SETTING_NUMBER_OF_SHARDS, 1)
77                  .put(SETTING_NUMBER_OF_REPLICAS, 0)
78                  .put(TransportModule.TRANSPORT_TYPE_KEY, "local")
79                  .build();
80  
81          String clusterName = GlobalOrdinalsBenchmark.class.getSimpleName();
82          node = nodeBuilder().clusterName(clusterName)
83                      .settings(settingsBuilder().put(settings))
84                      .node();
85  
86          client = node.client();
87  
88          try {
89              client.admin().indices().prepareCreate(INDEX_NAME)
90                      .addMapping(TYPE_NAME, jsonBuilder().startObject().startObject(TYPE_NAME)
91                              .startArray("dynamic_templates")
92                                  .startObject()
93                                      .startObject("default")
94                                          .field("match", "*")
95                                          .field("match_mapping_type", "string")
96                                          .startObject("mapping")
97                                              .field("type", "string")
98                                              .field("index", "not_analyzed")
99                                              .startObject("fields")
100                                                 .startObject("doc_values")
101                                                     .field("type", "string")
102                                                     .field("index", "no")
103                                                     .startObject("fielddata")
104                                                         .field("format", "doc_values")
105                                                     .endObject()
106                                                 .endObject()
107                                             .endObject()
108                                         .endObject()
109                                     .endObject()
110                                 .endObject()
111                             .endArray()
112                     .endObject().endObject())
113                     .get();
114             ObjectOpenHashSet<String> uniqueTerms = ObjectOpenHashSet.newInstance();
115             for (int i = 0; i < FIELD_LIMIT; i++) {
116                 boolean added;
117                 do {
118                     added = uniqueTerms.add(RandomStrings.randomAsciiOfLength(random, 16));
119                 } while (!added);
120             }
121             String[] sValues = uniqueTerms.toArray(String.class);
122             uniqueTerms = null;
123 
124             BulkRequestBuilder builder = client.prepareBulk();
125             IntIntOpenHashMap tracker = new IntIntOpenHashMap();
126             for (int i = 0; i < COUNT; i++) {
127                 Map<String, Object> fieldValues = new HashMap<>();
128                 for (int fieldSuffix = 1; fieldSuffix <= FIELD_LIMIT; fieldSuffix <<= 1) {
129                     int index;
130                     if (tracker.containsKey(fieldSuffix)) {
131                         index = tracker.lget();
132                     } else {
133                         tracker.put(fieldSuffix, index = 0);
134                     }
135                     if (index >= fieldSuffix) {
136                         index = random.nextInt(fieldSuffix);
137                         fieldValues.put("field_" + fieldSuffix, sValues[index]);
138                     } else {
139                         fieldValues.put("field_" + fieldSuffix, sValues[index]);
140                         tracker.put(fieldSuffix, ++index);
141                     }
142                 }
143                 builder.add(
144                         client.prepareIndex(INDEX_NAME, TYPE_NAME, String.valueOf(i))
145                         .setSource(fieldValues)
146                 );
147 
148                 if (builder.numberOfActions() >= 1000) {
149                     builder.get();
150                     builder = client.prepareBulk();
151                 }
152             }
153             if (builder.numberOfActions() > 0) {
154                 builder.get();
155             }
156         } catch (IndexAlreadyExistsException e) {
157             System.out.println("--> Index already exists, ignoring indexing phase, waiting for green");
158             ClusterHealthResponse clusterHealthResponse = client.admin().cluster().prepareHealth().setWaitForGreenStatus().setTimeout("10m").execute().actionGet();
159             if (clusterHealthResponse.isTimedOut()) {
160                 System.err.println("--> Timed out waiting for cluster health");
161             }
162         }
163 
164         client.admin().cluster().prepareUpdateSettings()
165                 .setTransientSettings(ImmutableSettings.builder().put("logger.index.fielddata.ordinals", "DEBUG"))
166                 .get();
167 
168         client.admin().indices().prepareRefresh(INDEX_NAME).execute().actionGet();
169         COUNT = client.prepareCount(INDEX_NAME).setQuery(matchAllQuery()).execute().actionGet().getCount();
170         System.out.println("--> Number of docs in index: " + COUNT);
171 
172         List<StatsResult> stats = new ArrayList<>();
173         for (int fieldSuffix = FIELD_START; fieldSuffix <= FIELD_LIMIT; fieldSuffix <<= 1) {
174             String fieldName = "field_" + fieldSuffix;
175             String name = "global_ordinals-" + fieldName;
176             if (USE_DOC_VALUES) {
177                 fieldName = fieldName + ".doc_values";
178                 name = name + "_doc_values"; // can't have . in agg name
179             }
180             stats.add(terms(name, fieldName, "global_ordinals_low_cardinality"));
181         }
182 
183         for (int fieldSuffix = FIELD_START; fieldSuffix <= FIELD_LIMIT; fieldSuffix <<= 1) {
184             String fieldName = "field_" + fieldSuffix;
185             String name = "ordinals-" + fieldName;
186             if (USE_DOC_VALUES) {
187                 fieldName = fieldName + ".doc_values";
188                 name = name + "_doc_values"; // can't have . in agg name
189             }
190             stats.add(terms(name, fieldName, "ordinals"));
191         }
192 
193         System.out.println("------------------ SUMMARY -----------------------------------------");
194         System.out.format(Locale.ENGLISH, "%30s%10s%10s%15s\n", "name", "took", "millis", "fieldata size");
195         for (StatsResult stat : stats) {
196             System.out.format(Locale.ENGLISH, "%30s%10s%10d%15s\n", stat.name, TimeValue.timeValueMillis(stat.took), (stat.took / QUERY_COUNT), stat.fieldDataMemoryUsed);
197         }
198         System.out.println("------------------ SUMMARY -----------------------------------------");
199 
200         client.close();
201         node.close();
202     }

204     private static StatsResult terms(String name, String field, String executionHint) {
205         long totalQueryTime;// LM VALUE
206 
207         client.admin().indices().prepareClearCache().setFieldDataCache(true).execute().actionGet();
208         System.gc();
209 
210         System.out.println("--> Warmup (" + name + ")...");
211         // run just the child query, warm up first
212         for (int j = 0; j < QUERY_WARMUP; j++) {
213             SearchResponse searchResponse = client.prepareSearch(INDEX_NAME)
214                     .setSearchType(SearchType.COUNT)
215                     .setQuery(matchAllQuery())
216                     .addAggregation(AggregationBuilders.terms(name).field(field).executionHint(executionHint))
217                     .get();
218             if (j == 0) {
219                 System.out.println("--> Loading (" + field + "): took: " + searchResponse.getTook());
220             }
221             if (searchResponse.getHits().totalHits() != COUNT) {
222                 System.err.println("--> mismatch on hits");
223             }
224         }
225         System.out.println("--> Warmup (" + name + ") DONE");
226 
227 
228         System.out.println("--> Running (" + name + ")...");
229         totalQueryTime = 0;
230         for (int j = 0; j < QUERY_COUNT; j++) {
231             SearchResponse searchResponse = client.prepareSearch(INDEX_NAME)
232                     .setSearchType(SearchType.COUNT)
233                     .setQuery(matchAllQuery())
234                     .addAggregation(AggregationBuilders.terms(name).field(field).executionHint(executionHint))
235                     .get();
236             if (searchResponse.getHits().totalHits() != COUNT) {
237                 System.err.println("--> mismatch on hits");
238             }
239             totalQueryTime += searchResponse.getTookInMillis();
240         }
241         System.out.println("--> Terms Agg (" + name + "): " + (totalQueryTime / QUERY_COUNT) + "ms");
242 
243         String nodeId = node.injector().getInstance(Discovery.class).localNode().getId();
244         ClusterStatsResponse clusterStateResponse = client.admin().cluster().prepareClusterStats().setNodesIds(nodeId).get();
245         System.out.println("--> Heap used: " + clusterStateResponse.getNodesStats().getJvm().getHeapUsed());
246         ByteSizeValue fieldDataMemoryUsed = clusterStateResponse.getIndicesStats().getFieldData().getMemorySize();
247         System.out.println("--> Fielddata memory size: " + fieldDataMemoryUsed);
248 
249         return new StatsResult(name, totalQueryTime, fieldDataMemoryUsed);
250     }
