46  public class SearchWhileRelocatingTests extends ElasticsearchIntegrationTest {
47  
48  // @LuceneTestCase.AwaitsFix(bugUrl = "problem with search searching on 1 shard (no replica), " +
49  //   "and between getting the cluster state to do the search, and executing it, " +
50  //   "the shard has fully relocated (moved from started on one node, to fully started on another node")
51  //   ^^ the current impl of the test handles this case gracefully since it can happen with 1 replica as well
52  //   we just make sure if we get a partial result without a failure that the postsearch is ok!
53      @Test
54      @Nightly
55      public void testSearchAndRelocateConcurrently0Replicas() throws Exception {
56          testSearchAndRelocateConcurrently(0);
57      }
58  
59      @Test
60      @Nightly
61      public void testSearchAndRelocateConcurrently1Replicas() throws Exception {
62          testSearchAndRelocateConcurrently(1);
63      }
64  
65      @Test
66      public void testSearchAndRelocateConcurrentlyRanodmReplicas() throws Exception {
67          testSearchAndRelocateConcurrently(randomIntBetween(0, 1));
68      }
69  
70      private void testSearchAndRelocateConcurrently(final int numberOfReplicas) throws Exception {
71          final int numShards = between(1, 20);
72          client().admin().indices().prepareCreate("test")
73                  .setSettings(settingsBuilder().put("index.number_of_shards", numShards).put("index.number_of_replicas", numberOfReplicas))
74                  .addMapping("type1", "loc", "type=geo_point", "test", "type=string").execute().actionGet();
75          ensureGreen();
76          List<IndexRequestBuilder> indexBuilders = new ArrayList<>();
77          final int numDocs = between(10, 20);
78          for (int i = 0; i < numDocs; i++) {
79              indexBuilders.add(client().prepareIndex("test", "type", Integer.toString(i))
80                      .setSource(
81                              jsonBuilder().startObject().field("test", "value").startObject("loc").field("lat", 11).field("lon", 21)
82                                      .endObject().endObject()));
83          }
84          indexRandom(true, indexBuilders.toArray(new IndexRequestBuilder[indexBuilders.size()]));
85          assertHitCount(client().prepareSearch().get(), (long) (numDocs));
86          final int numIters = scaledRandomIntBetween(5, 20);
87          for (int i = 0; i < numIters; i++) {
88              final AtomicBoolean stop = new AtomicBoolean(false);
89              final List<Throwable> thrownExceptions = new CopyOnWriteArrayList<>();
90              final List<Throwable> nonCriticalExceptions = new CopyOnWriteArrayList<>();
91  
92              Thread[] threads = new Thread[scaledRandomIntBetween(1, 3)];
93              for (int j = 0; j < threads.length; j++) {
94                  threads[j] = new Thread() {
95                      @Override
96                      public void run() {
97                          boolean criticalException = true;
98                          try {
99                              while (!stop.get()) {
100                                 SearchResponse sr = client().prepareSearch().setSize(numDocs).get();
101                                 // if we did not search all shards but had no failures that is potentially fine
102                                 // if only the hit-count is wrong. this can happen if the cluster-state is behind when the
103                                 // request comes in. It's a small window but a known limitation.
104                                 //
105                                 criticalException = sr.getTotalShards() == sr.getSuccessfulShards() || sr.getFailedShards() > 0;
106                                 assertHitCount(sr, (long) (numDocs));
107                                 criticalException = true;
108                                 final SearchHits sh = sr.getHits();
109                                 assertThat("Expected hits to be the same size the actual hits array", sh.getTotalHits(),
110                                         equalTo((long) (sh.getHits().length)));
111                                 // this is the more critical but that we hit the actual hit array has a different size than the
112                                 // actual number of hits.
113                             }
114                         } catch (SearchPhaseExecutionException ex) {
115                             // it's possible that all shards fail if we have a small number of shards.
116                             // with replicas this should not happen
117                             if (numberOfReplicas == 1 || !ex.getMessage().contains("all shards failed")) {
118                                 thrownExceptions.add(ex);
119                             }
120                         } catch (Throwable t) {
121                             if (!criticalException) {
122                                 nonCriticalExceptions.add(t);
123                             } else {
124                                 thrownExceptions.add(t);
125                             }
126                         }
127                     }
128                 };
129             }
130             for (int j = 0; j < threads.length; j++) {
131                 threads[j].start();
132             }
133             allowNodes("test", between(1, 3));
134             client().admin().cluster().prepareReroute().get();
135             stop.set(true);
136             for (int j = 0; j < threads.length; j++) {
137                 threads[j].join();
138             }
139             // this might time out on some machines if they are really busy and you hit lots of throttling
140             ClusterHealthResponse resp = client().admin().cluster().prepareHealth().setWaitForYellowStatus().setWaitForRelocatingShards(0).setWaitForEvents(Priority.LANGUID).setTimeout("5m").get();
141             assertNoTimeout(resp);
142             if (!thrownExceptions.isEmpty() || !nonCriticalExceptions.isEmpty()) {
143                 Client client = client();
144                 boolean postSearchOK = true;
145                 String verified = "POST SEARCH OK";
146                 for (int j = 0; j < 10; j++) {
147                     if (client.prepareSearch().get().getHits().getTotalHits() != numDocs) {
148                         verified = "POST SEARCH FAIL";
149                         postSearchOK = false;
150                         break;
151                     }
152                 }
153                 assertThat("numberOfReplicas: " + numberOfReplicas + " failed in iteration " + i + ", verification: " + verified, thrownExceptions, Matchers.emptyIterable());
154                 // if we hit only non-critical exceptions we only make sure that the post search works
155                 logger.info("Non-CriticalExceptions: " + nonCriticalExceptions.toString());
156                 assertThat("numberOfReplicas: " + numberOfReplicas + " failed in iteration " + i + ", verification: " + verified, postSearchOK, is(true));
157             }
158         }
159     }
