60      @Test
61      public void indexCleanup() throws Exception {
62          final String masterNode = internalCluster().startNode(ImmutableSettings.builder().put("node.data", false));
63          final String node_1 = internalCluster().startNode(ImmutableSettings.builder().put("node.master", false));
64          final String node_2 = internalCluster().startNode(ImmutableSettings.builder().put("node.master", false));
65          logger.info("--> creating index [test] with one shard and on replica");
66          assertAcked(prepareCreate("test").setSettings(
67                          ImmutableSettings.builder().put(indexSettings())
68                                  .put(IndexMetaData.SETTING_NUMBER_OF_SHARDS, 1)
69                                  .put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, 1))
70          );
71          ensureGreen("test");
72  
73          logger.info("--> making sure that shard and its replica are allocated on node_1 and node_2");
74          assertThat(Files.exists(shardDirectory(node_1, "test", 0)), equalTo(true));
75          assertThat(Files.exists(indexDirectory(node_1, "test")), equalTo(true));
76          assertThat(Files.exists(shardDirectory(node_2, "test", 0)), equalTo(true));
77          assertThat(Files.exists(indexDirectory(node_2, "test")), equalTo(true));
78  
79          logger.info("--> starting node server3");
80          final String node_3 = internalCluster().startNode(ImmutableSettings.builder().put("node.master", false));
81          logger.info("--> running cluster_health");
82          ClusterHealthResponse clusterHealth = client().admin().cluster().prepareHealth()
83                  .setWaitForNodes("4")
84                  .setWaitForRelocatingShards(0)
85                  .get();
86          assertThat(clusterHealth.isTimedOut(), equalTo(false));
87  
88          assertThat(Files.exists(shardDirectory(node_1, "test", 0)), equalTo(true));
89          assertThat(Files.exists(indexDirectory(node_1, "test")), equalTo(true));
90          assertThat(Files.exists(shardDirectory(node_2, "test", 0)), equalTo(true));
91          assertThat(Files.exists(indexDirectory(node_2, "test")), equalTo(true));
92          assertThat(Files.exists(shardDirectory(node_3, "test", 0)), equalTo(false));
93          assertThat(Files.exists(indexDirectory(node_3, "test")), equalTo(false));
94  
95          logger.info("--> move shard from node_1 to node_3, and wait for relocation to finish");
96          internalCluster().client().admin().cluster().prepareReroute().add(new MoveAllocationCommand(new ShardId("test", 0), node_1, node_3)).get();
97          clusterHealth = client().admin().cluster().prepareHealth()
98                  .setWaitForNodes("4")
99                  .setWaitForRelocatingShards(0)
100                 .get();
101         assertThat(clusterHealth.isTimedOut(), equalTo(false));
102 
103         assertThat(waitForShardDeletion(node_1, "test", 0), equalTo(false));
104         assertThat(waitForIndexDeletion(node_1, "test"), equalTo(false));
105         assertThat(Files.exists(shardDirectory(node_2, "test", 0)), equalTo(true));
106         assertThat(Files.exists(indexDirectory(node_2, "test")), equalTo(true));
107         assertThat(Files.exists(shardDirectory(node_3, "test", 0)), equalTo(true));
108         assertThat(Files.exists(indexDirectory(node_3, "test")), equalTo(true));
109     }

111     @Test
112     public void shardsCleanup() throws Exception {
113         final String node_1 = internalCluster().startNode();
114         final String node_2 = internalCluster().startNode();
115         logger.info("--> creating index [test] with one shard and on replica");
116         assertAcked(prepareCreate("test").setSettings(
117                         ImmutableSettings.builder().put(indexSettings())
118                                 .put(IndexMetaData.SETTING_NUMBER_OF_SHARDS, 1)
119                                 .put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, 1))
120         );
121         ensureGreen("test");
122 
123         logger.info("--> making sure that shard and its replica are allocated on node_1 and node_2");
124         assertThat(Files.exists(shardDirectory(node_1, "test", 0)), equalTo(true));
125         assertThat(Files.exists(shardDirectory(node_2, "test", 0)), equalTo(true));
126 
127         logger.info("--> starting node server3");
128         String node_3 = internalCluster().startNode();
129         logger.info("--> running cluster_health");
130         ClusterHealthResponse clusterHealth = client().admin().cluster().prepareHealth()
131                 .setWaitForNodes("3")
132                 .setWaitForRelocatingShards(0)
133                 .get();
134         assertThat(clusterHealth.isTimedOut(), equalTo(false));
135 
136         logger.info("--> making sure that shard is not allocated on server3");
137         assertThat(waitForShardDeletion(node_3, "test", 0), equalTo(false));
138 
139         Path server2Shard = shardDirectory(node_2, "test", 0);
140         logger.info("--> stopping node " + node_2);
141         internalCluster().stopRandomNode(InternalTestCluster.nameFilter(node_2));
142 
143         logger.info("--> running cluster_health");
144         clusterHealth = client().admin().cluster().prepareHealth()
145                 .setWaitForGreenStatus()
146                 .setWaitForNodes("2")
147                 .setWaitForRelocatingShards(0)
148                 .get();
149         assertThat(clusterHealth.isTimedOut(), equalTo(false));
150         logger.info("--> done cluster_health, status " + clusterHealth.getStatus());
151 
152         assertThat(Files.exists(server2Shard), equalTo(true));
153 
154         logger.info("--> making sure that shard and its replica exist on server1, server2 and server3");
155         assertThat(Files.exists(shardDirectory(node_1, "test", 0)), equalTo(true));
156         assertThat(Files.exists(server2Shard), equalTo(true));
157         assertThat(Files.exists(shardDirectory(node_3, "test", 0)), equalTo(true));
158 
159         logger.info("--> starting node node_4");
160         final String node_4 = internalCluster().startNode();
161 
162         logger.info("--> running cluster_health");
163         ensureGreen();
164 
165         logger.info("--> making sure that shard and its replica are allocated on server1 and server3 but not on server2");
166         assertThat(Files.exists(shardDirectory(node_1, "test", 0)), equalTo(true));
167         assertThat(Files.exists(shardDirectory(node_3, "test", 0)), equalTo(true));
168         assertThat(waitForShardDeletion(node_4, "test", 0), equalTo(false));
169     }

171     @Test
172     public void testShardActiveElseWhere() throws Exception {
173         boolean node1IsMasterEligible = randomBoolean();
174         boolean node2IsMasterEligible = !node1IsMasterEligible || randomBoolean();
175         Future<String> node_1_future = internalCluster().startNodeAsync(ImmutableSettings.builder().put("node.master", node1IsMasterEligible).build());
176         Future<String> node_2_future = internalCluster().startNodeAsync(ImmutableSettings.builder().put("node.master", node2IsMasterEligible).build());
177         final String node_1 = node_1_future.get();
178         final String node_2 = node_2_future.get();
179         final String node_1_id = internalCluster().getInstance(DiscoveryService.class, node_1).localNode().getId();
180         final String node_2_id = internalCluster().getInstance(DiscoveryService.class, node_2).localNode().getId();
181 
182         logger.debug("node {} (node_1) is {}master eligible", node_1, node1IsMasterEligible ? "" : "not ");
183         logger.debug("node {} (node_2) is {}master eligible", node_2, node2IsMasterEligible ? "" : "not ");
184         logger.debug("node {} became master", internalCluster().getMasterName());
185         final int numShards = scaledRandomIntBetween(2, 20);
186         assertAcked(prepareCreate("test")
187                         .setSettings(ImmutableSettings.builder().put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, 0).put(IndexMetaData.SETTING_NUMBER_OF_SHARDS, numShards))
188         );
189         ensureGreen("test");
190 
191         waitNoPendingTasksOnAll();
192         ClusterStateResponse stateResponse = client().admin().cluster().prepareState().get();
193 
194         RoutingNode routingNode = stateResponse.getState().routingNodes().node(node_2_id);
195         final int[] node2Shards = new int[routingNode.numberOfOwningShards()];
196         int i = 0;
197         for (MutableShardRouting mutableShardRouting : routingNode) {
198             node2Shards[i] = mutableShardRouting.shardId().id();
199             i++;
200         }
201         logger.info("Node 2 has shards: {}", Arrays.toString(node2Shards));
202         final long shardVersions[] = new long[numShards];
203         final int shardIds[] = new int[numShards];
204         i=0;
205         for (ShardRouting shardRouting : stateResponse.getState().getRoutingTable().allShards("test")) {
206             shardVersions[i] = shardRouting.version();
207             shardIds[i] = shardRouting.getId();
208             i++;
209         }
210         internalCluster().getInstance(ClusterService.class, node_2).submitStateUpdateTask("test", Priority.IMMEDIATE, new ClusterStateUpdateTask() {
211             @Override
212             public ClusterState execute(ClusterState currentState) throws Exception {
213                 IndexRoutingTable.Builder indexRoutingTableBuilder = IndexRoutingTable.builder("test");
214                 for (int i = 0; i < numShards; i++) {
215                    indexRoutingTableBuilder.addIndexShard(
216                            new IndexShardRoutingTable.Builder(new ShardId("test", i), false)
217                                    .addShard(new ImmutableShardRouting("test", i, node_1_id, true, ShardRoutingState.STARTED, shardVersions[shardIds[i]]))
218                                    .build()
219                    );
220                 }
221                 return ClusterState.builder(currentState)
222                         .routingTable(RoutingTable.builder().add(indexRoutingTableBuilder).build())
223                         .build();
224             }
225 
226             public boolean runOnlyOnMaster() {
227                 return false;
228             }
229 
230             @Override
231             public void onFailure(String source, Throwable t) {
232             }
233         });
234         waitNoPendingTasksOnAll();
235         logger.info("Checking if shards aren't removed");
236         for (int shard : node2Shards) {
237             assertTrue(waitForShardDeletion(node_2, "test", shard));
238         }
239     }

241     private Path indexDirectory(String server, String index) {
242         NodeEnvironment env = internalCluster().getInstance(NodeEnvironment.class, server);
243         return env.indexPaths(new Index(index))[0];
244     }
245 
246     private Path shardDirectory(String server, String index, int shard) {
247         NodeEnvironment env = internalCluster().getInstance(NodeEnvironment.class, server);
248         return env.shardPaths(new ShardId(index, shard))[0];
249     }
250 
251     private boolean waitForShardDeletion(final String server, final  String index, final int shard) throws InterruptedException {
252         awaitBusy(new Predicate<Object>() {
253             @Override
254             public boolean apply(Object o) {
255                 return !Files.exists(shardDirectory(server, index, shard));
256             }
257         });
258         return Files.exists(shardDirectory(server, index, shard));
259     }
260 
261     private boolean waitForIndexDeletion(final String server, final  String index) throws InterruptedException {
262         awaitBusy(new Predicate<Object>() {
263             @Override
264             public boolean apply(Object o) {
265                 return !Files.exists(indexDirectory(server, index));
266             }
267         });
268         return Files.exists(indexDirectory(server, index));
269     }
