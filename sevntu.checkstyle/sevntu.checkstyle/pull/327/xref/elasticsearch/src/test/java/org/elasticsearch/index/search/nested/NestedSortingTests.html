73      @Test
74      public void testDuel() throws Exception {
75          final int numDocs = scaledRandomIntBetween(100, 1000);
76          for (int i = 0; i < numDocs; ++i) {
77              final int numChildren = randomInt(2);
78              List<Document> docs = new ArrayList<>(numChildren + 1);
79              for (int j = 0; j < numChildren; ++j) {
80                  Document doc = new Document();
81                  doc.add(new StringField("f", TestUtil.randomSimpleString(getRandom(), 2), Field.Store.NO));
82                  doc.add(new StringField("__type", "child", Field.Store.NO));
83                  docs.add(doc);
84              }
85              if (randomBoolean()) {
86                  docs.add(new Document());
87              }
88              Document parent = new Document();
89              parent.add(new StringField("__type", "parent", Field.Store.NO));
90              docs.add(parent);
91              writer.addDocuments(docs);
92              if (rarely()) { // we need to have a bit more segments than what RandomIndexWriter would do by default
93                  DirectoryReader.open(writer, false).close();
94              }
95          }
96          writer.commit();
97  
98          MultiValueMode sortMode = randomFrom(Arrays.asList(MultiValueMode.MIN, MultiValueMode.MAX));
99          IndexSearcher searcher = new IndexSearcher(DirectoryReader.open(writer, false));
100         PagedBytesIndexFieldData indexFieldData1 = getForField("f");
101         IndexFieldData<?> indexFieldData2 = NoOrdinalsStringFieldDataTests.hideOrdinals(indexFieldData1);
102         final String missingValue = randomBoolean() ? null : TestUtil.randomSimpleString(getRandom(), 2);
103         final int n = randomIntBetween(1, numDocs + 2);
104         final boolean reverse = randomBoolean();
105 
106         final TopDocs topDocs1 = getTopDocs(searcher, indexFieldData1, missingValue, sortMode, n, reverse);
107         final TopDocs topDocs2 = getTopDocs(searcher, indexFieldData2, missingValue, sortMode, n, reverse);
108         for (int i = 0; i < topDocs1.scoreDocs.length; ++i) {
109             final FieldDoc fieldDoc1 = (FieldDoc) topDocs1.scoreDocs[i];
110             final FieldDoc fieldDoc2 = (FieldDoc) topDocs2.scoreDocs[i];
111             assertEquals(fieldDoc1.doc, fieldDoc2.doc);
112             assertArrayEquals(fieldDoc1.fields, fieldDoc2.fields);
113         }
114 
115         searcher.getIndexReader().close();
116     }
117 
118     private TopDocs getTopDocs(IndexSearcher searcher, IndexFieldData<?> indexFieldData, String missingValue, MultiValueMode sortMode, int n, boolean reverse) throws IOException {
119         Filter parentFilter = new TermFilter(new Term("__type", "parent"));
120         Filter childFilter = new TermFilter(new Term("__type", "child"));
121         XFieldComparatorSource nestedComparatorSource = indexFieldData.comparatorSource(missingValue, sortMode, createNested(parentFilter, childFilter));
122         Query query = new ConstantScoreQuery(parentFilter);
123         Sort sort = new Sort(new SortField("f", nestedComparatorSource, reverse));
124         return searcher.search(query, n, sort);
125     }
