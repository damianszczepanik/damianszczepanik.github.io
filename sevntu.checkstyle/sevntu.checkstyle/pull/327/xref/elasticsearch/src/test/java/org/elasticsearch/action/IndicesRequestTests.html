288             indices.add(indexOrAlias);

290         int numUpdateRequests = iterations(1, 10);
291         for (int i = 0; i < numUpdateRequests; i++) {
292             String indexOrAlias = randomIndexOrAlias();
293             bulkRequest.add(new UpdateRequest(indexOrAlias, "type", "id").doc("field1", "value1"));
294             indices.add(indexOrAlias);
295         }
296 
297         internalCluster().clientNodeClient().bulk(bulkRequest).actionGet();

793     private static void assertSameIndices(IndicesRequest originalRequest, boolean optional, String... actions) {
794         for (String action : actions) {
795             List<TransportRequest> requests = consumeTransportRequests(action);
796             if (!optional) {
797                 assertThat("no internal requests intercepted for action [" + action + "]", requests.size(), greaterThan(0));
798             }
799             for (TransportRequest internalRequest : requests) {
800                 assertThat(internalRequest, instanceOf(IndicesRequest.class));
801                 assertThat(internalRequest.getClass().getName(), ((IndicesRequest)internalRequest).indices(), equalTo(originalRequest.indices()));
802                 assertThat(((IndicesRequest)internalRequest).indicesOptions(), equalTo(originalRequest.indicesOptions()));
803             }
804         }
805     }

811     private static void assertSameIndices(String[] indices, boolean optional, String... actions) {

820             }

824     private static void assertIndicesSubset(List<String> indices, String... actions) {
825         //indices returned by each bulk shard request need to be a subset of the original indices
826         for (String action : actions) {
827             List<TransportRequest> requests = consumeTransportRequests(action);
828             assertThat("no internal requests intercepted for action [" + action + "]", requests.size(), greaterThan(0));
829             for (TransportRequest internalRequest : requests) {
830                 assertThat(internalRequest, instanceOf(IndicesRequest.class));
831                 for (String index : ((IndicesRequest) internalRequest).indices()) {
832                     assertThat(indices, hasItem(index));
833                 }
834             }
835         }

847     private String[] randomIndicesOrAliases() {
848         int count = randomIntBetween(1, indices.size() * 2); //every index has an alias
849         String[] indices = new String[count];
850         for (int i = 0; i < count; i++) {
851             indices[i] = randomIndexOrAlias();
852         }
853         return indices;
854     }

856     private String[] randomUniqueIndicesOrAliases() {
857         Set<String> uniqueIndices = new HashSet<>();
858         int count = randomIntBetween(1, this.indices.size());
859         while (uniqueIndices.size() < count) {
860             uniqueIndices.add(randomFrom(this.indices));
861         }
862         String[] indices = new String[count];
863         int i = 0;
864         for (String index : uniqueIndices) {
865             indices[i++] = randomBoolean() ? index + "-alias" : index;
866         }
867         return indices;
868     }

891     private static List<TransportRequest> consumeTransportRequests(String action) {
892         List<TransportRequest> requests = new ArrayList<>();
893         Iterable<TransportService> transportServices = internalCluster().getInstances(TransportService.class);
894         for (TransportService transportService : transportServices) {
895             List<TransportRequest> transportRequests = ((InterceptingTransportService) transportService).consumeRequests(action);
896             if (transportRequests != null) {
897                 requests.addAll(transportRequests);
898             }
899         }
900         return requests;
901     }

903     public static class InterceptingTransportService extends TransportService {
904 
905         private final Set<String> actions = new HashSet<>();
906 
907         private final Map<String, List<TransportRequest>> requests = new HashMap<>();
908 
909         @Inject
910         public InterceptingTransportService(Settings settings, Transport transport, ThreadPool threadPool,
911                                             NodeSettingsService nodeSettingsService, @ClusterDynamicSettings DynamicSettings dynamicSettings) {
912             super(settings, transport, threadPool);
913         }
914 
915         synchronized List<TransportRequest> consumeRequests(String action) {
916             return requests.remove(action);
917         }
918 
919         synchronized void interceptTransportActions(String... actions) {
920             Collections.addAll(this.actions, actions);
921         }
922 
923         synchronized void clearInterceptedActions() {
924             actions.clear();
925         }
926 
927         @Override
928         public void registerHandler(String action, TransportRequestHandler handler) {
929             super.registerHandler(action, new InterceptingRequestHandler(action, handler));
930         }
931 
932         private class InterceptingRequestHandler implements TransportRequestHandler {
933 
934             private final TransportRequestHandler requestHandler;
935             private final String action;
936 
937             InterceptingRequestHandler(String action, TransportRequestHandler requestHandler) {
938                 this.requestHandler = requestHandler;
939                 this.action = action;
940             }
941 
942             @Override
943             public TransportRequest newInstance() {
944                 return requestHandler.newInstance();
945             }
946 
947             @Override
948             public void messageReceived(TransportRequest request, TransportChannel channel) throws Exception {
949                 synchronized (InterceptingTransportService.this) {
950                     if (actions.contains(action)) {
951                         List<TransportRequest> requestList = requests.get(action);
952                         if (requestList == null) {
953                             requestList = new ArrayList<>();
954                             requestList.add(request);
955                             requests.put(action, requestList);
956                         } else {
957                             requestList.add(request);
958                         }
959                     }
960                 }
961                 requestHandler.messageReceived(request, channel);
962             }
963 
964             @Override
965             public String executor() {
966                 return requestHandler.executor();
967             }
968 
969             @Override
970             public boolean isForceExecution() {
971                 return requestHandler.isForceExecution();
972             }
973         }
974     }
