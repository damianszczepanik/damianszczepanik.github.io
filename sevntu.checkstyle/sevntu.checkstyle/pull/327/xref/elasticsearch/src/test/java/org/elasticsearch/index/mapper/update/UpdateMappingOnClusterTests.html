75      @Test
76      public void test_all_with_default() throws Exception {
77          String defaultMapping = jsonBuilder().startObject().startObject("_default_")
78                  .startObject("_all")
79                  .field("enabled", false)
80                  .endObject()
81                  .endObject().endObject().string();
82          client().admin().indices().prepareCreate("index").addMapping("_default_", defaultMapping).get();
83          String docMapping = jsonBuilder().startObject()
84                  .startObject("doc")
85                  .endObject()
86                  .endObject().string();
87          PutMappingResponse response = client().admin().indices().preparePutMapping("index").setType("doc").setSource(docMapping).get();
88          assertTrue(response.isAcknowledged());
89          String docMappingUpdate = jsonBuilder().startObject().startObject("doc")
90                  .startObject("properties")
91                  .startObject("text")
92                  .field("type", "string")
93                  .endObject()
94                  .endObject()
95                  .endObject()
96                  .endObject().string();
97          response = client().admin().indices().preparePutMapping("index").setType("doc").setSource(docMappingUpdate).get();
98          assertTrue(response.isAcknowledged());
99          String docMappingAllExplicitEnabled = jsonBuilder().startObject()
100                 .startObject("doc_all_enabled")
101                 .startObject("_all")
102                 .field("enabled", true)
103                 .endObject()
104                 .endObject()
105                 .endObject().string();
106         response = client().admin().indices().preparePutMapping("index").setType("doc_all_enabled").setSource(docMappingAllExplicitEnabled).get();
107         assertTrue(response.isAcknowledged());
108 
109         GetMappingsResponse mapping = client().admin().indices().prepareGetMappings("index").get();
110         HashMap props = (HashMap)mapping.getMappings().get("index").get("doc").getSourceAsMap().get("_all");
111         assertThat((Boolean)props.get("enabled"), equalTo(false));
112         props = (HashMap)mapping.getMappings().get("index").get("doc").getSourceAsMap().get("properties");
113         assertNotNull(props);
114         assertNotNull(props.get("text"));
115         props = (HashMap)mapping.getMappings().get("index").get("doc_all_enabled").getSourceAsMap().get("_all");
116         assertThat((Boolean)props.get("enabled"), equalTo(true));
117         props = (HashMap)mapping.getMappings().get("index").get("_default_").getSourceAsMap().get("_all");
118         assertThat((Boolean)props.get("enabled"), equalTo(false));
119 
120     }

160     protected void testConflict(String mapping, String mappingUpdate, String... errorMessages) throws InterruptedException {
161         assertAcked(prepareCreate(INDEX).setSource(mapping).get());
162         ensureGreen(INDEX);
163         GetMappingsResponse mappingsBeforeUpdateResponse = client().admin().indices().prepareGetMappings(INDEX).addTypes(TYPE).get();
164         try {
165             client().admin().indices().preparePutMapping(INDEX).setType(TYPE).setSource(mappingUpdate).get();
166             fail();
167         } catch (MergeMappingException e) {
168             for (String errorMessage : errorMessages) {
169                 assertThat(e.getDetailedMessage(), containsString(errorMessage));
170             }
171         }
172         compareMappingOnNodes(mappingsBeforeUpdateResponse);
173 
174     }
175 
176     private void compareMappingOnNodes(GetMappingsResponse previousMapping) {
177         // make sure all nodes have same cluster state
178         for (Client client : cluster()) {
179             GetMappingsResponse currentMapping = client.admin().indices().prepareGetMappings(INDEX).addTypes(TYPE).setLocal(true).get();
180             assertThat(previousMapping.getMappings().get(INDEX).get(TYPE).source(), equalTo(currentMapping.getMappings().get(INDEX).get(TYPE).source()));
181         }
182     }

184     @Test
185     public void testUpdateTimestamp() throws IOException {
186         XContentBuilder mapping = XContentFactory.jsonBuilder().startObject().startObject("type")
187                 .startObject("_timestamp").field("enabled", randomBoolean()).startObject("fielddata").field("loading", "lazy").field("format", "doc_values").endObject().field("store", "no").endObject()
188                 .endObject().endObject();
189         client().admin().indices().prepareCreate("test").addMapping("type", mapping).get();
190         GetMappingsResponse appliedMappings = client().admin().indices().prepareGetMappings("test").get();
191         LinkedHashMap timestampMapping = (LinkedHashMap) appliedMappings.getMappings().get("test").get("type").getSourceAsMap().get("_timestamp");
192         assertThat((Boolean) timestampMapping.get("store"), equalTo(false));
193         assertThat((String)((LinkedHashMap) timestampMapping.get("fielddata")).get("loading"), equalTo("lazy"));
194         assertThat((String)((LinkedHashMap) timestampMapping.get("fielddata")).get("format"), equalTo("doc_values"));
195         mapping = XContentFactory.jsonBuilder().startObject().startObject("type")
196                 .startObject("_timestamp").field("enabled", randomBoolean()).startObject("fielddata").field("loading", "eager").field("format", "array").endObject().field("store", "no").endObject()
197                 .endObject().endObject();
198         PutMappingResponse putMappingResponse = client().admin().indices().preparePutMapping("test").setType("type").setSource(mapping).get();
199         appliedMappings = client().admin().indices().prepareGetMappings("test").get();
200         timestampMapping = (LinkedHashMap) appliedMappings.getMappings().get("test").get("type").getSourceAsMap().get("_timestamp");
201         assertThat((Boolean) timestampMapping.get("store"), equalTo(false));
202         assertThat((String)((LinkedHashMap) timestampMapping.get("fielddata")).get("loading"), equalTo("eager"));
203         assertThat((String)((LinkedHashMap) timestampMapping.get("fielddata")).get("format"), equalTo("array"));
204     }

206     @Test
207     public void testTimestampMergingConflicts() throws Exception {
208         String mapping = XContentFactory.jsonBuilder().startObject().startObject(TYPE)
209                 .startObject("_timestamp").field("enabled", true)
210                 .startObject("fielddata").field("format", "doc_values").endObject()
211                 .field("store", "yes")
212                 .field("index", "analyzed")
213                 .field("path", "foo")
214                 .field("default", "1970-01-01")
215                 .endObject()
216                 .endObject().endObject().string();
217 
218         client().admin().indices().prepareCreate(INDEX).addMapping(TYPE, mapping).get();
219 
220         mapping = XContentFactory.jsonBuilder().startObject().startObject("type")
221                 .startObject("_timestamp").field("enabled", false)
222                 .startObject("fielddata").field("format", "array").endObject()
223                 .field("store", "no")
224                 .field("index", "no")
225                 .field("path", "bar")
226                 .field("default", "1970-01-02")
227                 .endObject()
228                 .endObject().endObject().string();
229         GetMappingsResponse mappingsBeforeUpdateResponse = client().admin().indices().prepareGetMappings(INDEX).addTypes(TYPE).get();
230         try {
231             client().admin().indices().preparePutMapping(INDEX).setType(TYPE).setSource(mapping).get();
232             fail("This should result in conflicts when merging the mapping");
233         } catch (MergeMappingException e) {
234             String[] expectedConflicts = {"mapper [_timestamp] has different index values", "mapper [_timestamp] has different store values", "Cannot update default in _timestamp value. Value is 1970-01-01 now encountering 1970-01-02", "Cannot update path in _timestamp value. Value is foo path in merged mapping is bar"};
235             for (String conflict : expectedConflicts) {
236                 assertThat(e.getDetailedMessage(), containsString(conflict));
237             }
238         }
239         compareMappingOnNodes(mappingsBeforeUpdateResponse);
240     }
