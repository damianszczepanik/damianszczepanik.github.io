56      @Test
57      @TestLogging("action.search.type:TRACE,index.shard:TRACE")
58      public void testRandomDirectoryIOExceptions() throws IOException, InterruptedException, ExecutionException {
59          String mapping = XContentFactory.jsonBuilder().
60                  startObject().
61                  startObject("type").
62                  startObject("properties").
63                  startObject("test")
64                  .field("type", "string")
65                  .field("index", "not_analyzed")
66                  .endObject().
67                          endObject().
68                          endObject()
69                  .endObject().string();
70          final double exceptionRate;
71          final double exceptionOnOpenRate;
72          if (frequently()) {
73              if (randomBoolean()) {
74                  if (randomBoolean()) {
75                      exceptionOnOpenRate = 1.0 / between(5, 100);
76                      exceptionRate = 0.0d;
77                  } else {
78                      exceptionRate = 1.0 / between(5, 100);
79                      exceptionOnOpenRate = 0.0d;
80                  }
81              } else {
82                  exceptionOnOpenRate = 1.0 / between(5, 100);
83                  exceptionRate = 1.0 / between(5, 100);
84              }
85          } else {
86              // rarely no exception
87              exceptionRate = 0d;
88              exceptionOnOpenRate = 0d;
89          }
90          final boolean createIndexWithoutErrors = randomBoolean();
91          int numInitialDocs = 0;
92  
93          if (createIndexWithoutErrors) {
94              Builder settings = settingsBuilder()
95                      .put("index.number_of_replicas", randomIntBetween(0, 1));
96              logger.info("creating index: [test] using settings: [{}]", settings.build().getAsMap());
97              client().admin().indices().prepareCreate("test")
98                      .setSettings(settings)
99                      .addMapping("type", mapping).execute().actionGet();
100             numInitialDocs = between(10, 100);
101             ensureGreen();
102             for (int i = 0; i < numInitialDocs; i++) {
103                 client().prepareIndex("test", "type", "init" + i).setSource("test", "init").get();
104             }
105             client().admin().indices().prepareRefresh("test").execute().get();
106             client().admin().indices().prepareFlush("test").setWaitIfOngoing(true).execute().get();
107             client().admin().indices().prepareClose("test").execute().get();
108             client().admin().indices().prepareUpdateSettings("test").setSettings(settingsBuilder()
109                     .put(MockDirectoryHelper.RANDOM_IO_EXCEPTION_RATE, exceptionRate)
110                     .put(MockDirectoryHelper.RANDOM_IO_EXCEPTION_RATE_ON_OPEN, exceptionOnOpenRate));
111             client().admin().indices().prepareOpen("test").execute().get();
112         } else {
113             Builder settings = settingsBuilder()
114                     .put("index.number_of_replicas", randomIntBetween(0, 1))
115                     .put(MockFSDirectoryService.CHECK_INDEX_ON_CLOSE, false)
116                     .put(MockDirectoryHelper.RANDOM_IO_EXCEPTION_RATE, exceptionRate)
117                     .put(MockDirectoryHelper.RANDOM_IO_EXCEPTION_RATE_ON_OPEN, exceptionOnOpenRate); // we cannot expect that the index will be valid
118             logger.info("creating index: [test] using settings: [{}]", settings.build().getAsMap());
119             client().admin().indices().prepareCreate("test")
120                     .setSettings(settings)
121                     .addMapping("type", mapping).execute().actionGet();
122         }
123         ClusterHealthResponse clusterHealthResponse = client().admin().cluster()
124                 .health(Requests.clusterHealthRequest().waitForYellowStatus().timeout(TimeValue.timeValueSeconds(5))).get(); // it's OK to timeout here 
125         final int numDocs;
126         final boolean expectAllShardsFailed;
127         if (clusterHealthResponse.isTimedOut()) {
128             /* some seeds just won't let you create the index at all and we enter a ping-pong mode
129              * trying one node after another etc. that is ok but we need to make sure we don't wait
130              * forever when indexing documents so we set numDocs = 1 and expecte all shards to fail
131              * when we search below.*/
132             logger.info("ClusterHealth timed out - only index one doc and expect searches to fail");
133             numDocs = 1;
134             expectAllShardsFailed = true;
135         } else {
136             numDocs = between(10, 100);
137             expectAllShardsFailed = false;
138         }
139         int numCreated = 0;
140         boolean[] added = new boolean[numDocs];
141         for (int i = 0; i < numDocs; i++) {
142             added[i] = false;
143             try {
144                 IndexResponse indexResponse = client().prepareIndex("test", "type", Integer.toString(i)).setTimeout(TimeValue.timeValueSeconds(1)).setSource("test", English.intToEnglish(i)).get();
145                 if (indexResponse.isCreated()) {
146                     numCreated++;
147                     added[i] = true;
148                 }
149             } catch (ElasticsearchException ex) {
150             }
151 
152         }
153         NumShards numShards = getNumShards("test");
154         logger.info("Start Refresh");
155         final RefreshResponse refreshResponse = client().admin().indices().prepareRefresh("test").execute().get(); // don't assert on failures here
156         final boolean refreshFailed = refreshResponse.getShardFailures().length != 0 || refreshResponse.getFailedShards() != 0;
157         logger.info("Refresh failed [{}] numShardsFailed: [{}], shardFailuresLength: [{}], successfulShards: [{}], totalShards: [{}] ", refreshFailed, refreshResponse.getFailedShards(), refreshResponse.getShardFailures().length, refreshResponse.getSuccessfulShards(), refreshResponse.getTotalShards());
158         final int numSearches = scaledRandomIntBetween(10, 20);
159         // we don't check anything here really just making sure we don't leave any open files or a broken index behind.
160         for (int i = 0; i < numSearches; i++) {
161             try {
162                 int docToQuery = between(0, numDocs - 1);
163                 int expectedResults = added[docToQuery] ? 1 : 0;
164                 logger.info("Searching for [test:{}]", English.intToEnglish(docToQuery));
165                 SearchResponse searchResponse = client().prepareSearch().setTypes("type").setQuery(QueryBuilders.matchQuery("test", English.intToEnglish(docToQuery)))
166                         .setSize(expectedResults).get();
167                 logger.info("Successful shards: [{}]  numShards: [{}]", searchResponse.getSuccessfulShards(), numShards.numPrimaries);
168                 if (searchResponse.getSuccessfulShards() == numShards.numPrimaries && !refreshFailed) {
169                     assertResultsAndLogOnFailure(expectedResults, searchResponse);
170                 }
171                 // check match all
172                 searchResponse = client().prepareSearch().setTypes("type").setQuery(QueryBuilders.matchAllQuery())
173                         .setSize(numCreated + numInitialDocs).addSort("_uid", SortOrder.ASC).get();
174                 logger.info("Match all Successful shards: [{}]  numShards: [{}]", searchResponse.getSuccessfulShards(), numShards.numPrimaries);
175                 if (searchResponse.getSuccessfulShards() == numShards.numPrimaries && !refreshFailed) {
176                     assertResultsAndLogOnFailure(numCreated + numInitialDocs, searchResponse);
177                 }
178             } catch (SearchPhaseExecutionException ex) {
179                 logger.info("SearchPhaseException: [{}]", ex.getMessage());
180                 // if a scheduled refresh or flush fails all shards we see all shards failed here
181                 if (!(expectAllShardsFailed || refreshResponse.getSuccessfulShards() == 0 || ex.getMessage().contains("all shards failed"))) {
182                     throw ex;
183                 }
184             }
185         }
186 
187         if (createIndexWithoutErrors) {
188             // check the index still contains the records that we indexed without errors
189             client().admin().indices().prepareClose("test").execute().get();
190             client().admin().indices().prepareUpdateSettings("test").setSettings(settingsBuilder()
191                     .put(MockDirectoryHelper.RANDOM_IO_EXCEPTION_RATE, 0)
192                     .put(MockDirectoryHelper.RANDOM_IO_EXCEPTION_RATE_ON_OPEN, 0));
193             client().admin().indices().prepareOpen("test").execute().get();
194             ensureGreen();
195             SearchResponse searchResponse = client().prepareSearch().setTypes("type").setQuery(QueryBuilders.matchQuery("test", "init")).get();
196             assertNoFailures(searchResponse);
197             assertHitCount(searchResponse, numInitialDocs);
198         }
199     }

201     private void assertResultsAndLogOnFailure(long expectedResults, SearchResponse searchResponse) {
202         if (searchResponse.getHits().getTotalHits() != expectedResults) {
203             StringBuilder sb = new StringBuilder("search result contains [");
204             sb.append(searchResponse.getHits().getTotalHits()).append("] results. expected [").append(expectedResults).append("]");
205             String failMsg = sb.toString();
206             for (SearchHit hit : searchResponse.getHits().getHits()) {
207                 sb.append("\n-> _index: [").append(hit.getIndex()).append("] type [").append(hit.getType())
208                         .append("] id [").append(hit.id()).append("]");
209             }
210             logger.warn(sb.toString());
211             fail(failMsg);
212         }
213     }

217         String mapping = XContentFactory.jsonBuilder().
218                 startObject().
219                 startObject("type").
220                 startObject("properties").
221                 startObject("test")
222                 .field("type", "string")
223                 .field("index", "not_analyzed")
224                 .endObject().
225                         endObject().
226                         endObject()
227                 .endObject().string();
228         final double lowLevelRate;
229         final double topLevelRate;
230         if (frequently()) {
231             if (randomBoolean()) {
232                 if (randomBoolean()) {
233                     lowLevelRate = 1.0 / between(2, 10);
234                     topLevelRate = 0.0d;
235                 } else {
236                     topLevelRate = 1.0 / between(2, 10);
237                     lowLevelRate = 0.0d;
238                 }
239             } else {
240                 lowLevelRate = 1.0 / between(2, 10);
241                 topLevelRate = 1.0 / between(2, 10);
242             }
243         } else {
244             // rarely no exception
245             topLevelRate = 0d;
246             lowLevelRate = 0d;
247         }
248 
249         Builder settings = settingsBuilder()
250                 .put(indexSettings())
251                 .put(MockInternalEngine.READER_WRAPPER_TYPE, RandomExceptionDirectoryReaderWrapper.class.getName())
252                 .put(EXCEPTION_TOP_LEVEL_RATIO_KEY, topLevelRate)
253                 .put(EXCEPTION_LOW_LEVEL_RATIO_KEY, lowLevelRate)
254                 .put(MockInternalEngine.WRAP_READER_RATIO, 1.0d);
255         logger.info("creating index: [test] using settings: [{}]", settings.build().getAsMap());
256         assertAcked(prepareCreate("test")
257                 .setSettings(settings)
258                 .addMapping("type", mapping));
259         ensureSearchable();
260         final int numDocs = between(10, 100);
261         int numCreated = 0;
262         boolean[] added = new boolean[numDocs];
263         for (int i = 0; i < numDocs; i++) {
264             try {
265                 IndexResponse indexResponse = client().prepareIndex("test", "type", "" + i).setTimeout(TimeValue.timeValueSeconds(1)).setSource("test", English.intToEnglish(i)).get();
266                 if (indexResponse.isCreated()) {
267                     numCreated++;
268                     added[i] = true;
269                 }
270             } catch (ElasticsearchException ex) {
271             }
272         }
273         logger.info("Start Refresh");
274         RefreshResponse refreshResponse = client().admin().indices().prepareRefresh("test").execute().get(); // don't assert on failures here
275         final boolean refreshFailed = refreshResponse.getShardFailures().length != 0 || refreshResponse.getFailedShards() != 0;
276         logger.info("Refresh failed [{}] numShardsFailed: [{}], shardFailuresLength: [{}], successfulShards: [{}], totalShards: [{}] ", refreshFailed, refreshResponse.getFailedShards(), refreshResponse.getShardFailures().length, refreshResponse.getSuccessfulShards(), refreshResponse.getTotalShards());
277 
278         NumShards test = getNumShards("test");
279         final int numSearches = scaledRandomIntBetween(100, 200);
280         // we don't check anything here really just making sure we don't leave any open files or a broken index behind.
281         for (int i = 0; i < numSearches; i++) {
282             try {
283                 int docToQuery = between(0, numDocs - 1);
284                 int expectedResults = added[docToQuery] ? 1 : 0;
285                 logger.info("Searching for [test:{}]", English.intToEnglish(docToQuery));
286                 SearchResponse searchResponse = client().prepareSearch().setQuery(QueryBuilders.matchQuery("test", English.intToEnglish(docToQuery)))
287                         .setSize(expectedResults).get();
288                 logger.info("Successful shards: [{}]  numShards: [{}]", searchResponse.getSuccessfulShards(), test.numPrimaries);
289                 if (searchResponse.getSuccessfulShards() == test.numPrimaries && !refreshFailed) {
290                     assertResultsAndLogOnFailure(expectedResults, searchResponse);
291                 }
292                 // check match all
293                 searchResponse = client().prepareSearch().setQuery(QueryBuilders.matchAllQuery()).setSize(numCreated).addSort("_id", SortOrder.ASC).get();
294                 logger.info("Match all Successful shards: [{}]  numShards: [{}]", searchResponse.getSuccessfulShards(), test.numPrimaries);
295                 if (searchResponse.getSuccessfulShards() == test.numPrimaries && !refreshFailed) {
296                     assertResultsAndLogOnFailure(numCreated, searchResponse);
297                 }
298 
299             } catch (SearchPhaseExecutionException ex) {
300                 logger.info("expected SearchPhaseException: [{}]", ex.getMessage());
301             }

306     public static final String EXCEPTION_TOP_LEVEL_RATIO_KEY = "index.engine.exception.ratio.top";
307     public static final String EXCEPTION_LOW_LEVEL_RATIO_KEY = "index.engine.exception.ratio.low";

310     public static class RandomExceptionDirectoryReaderWrapper extends MockInternalEngine.DirectoryReaderWrapper {
311         private final Settings settings;
312 
313         static class ThrowingSubReaderWrapper extends SubReaderWrapper implements ThrowingLeafReaderWrapper.Thrower {
314             private final Random random;
315             private final double topLevelRatio;
316             private final double lowLevelRatio;
317 
318             ThrowingSubReaderWrapper(Settings settings) {
319                 final long seed = settings.getAsLong(SETTING_INDEX_SEED, 0l);
320                 this.topLevelRatio = settings.getAsDouble(EXCEPTION_TOP_LEVEL_RATIO_KEY, 0.1d);
321                 this.lowLevelRatio = settings.getAsDouble(EXCEPTION_LOW_LEVEL_RATIO_KEY, 0.1d);
322                 this.random = new Random(seed);
323             }
324 
325             @Override
326             public LeafReader wrap(LeafReader reader) {
327                 return new ThrowingLeafReaderWrapper(reader, this);
328             }
329 
330             @Override
331             public void maybeThrow(ThrowingLeafReaderWrapper.Flags flag) throws IOException {
332                 switch (flag) {
333                     case Fields:
334                     case TermVectors:
335                     case Terms:
336                     case TermsEnum:
337                     case Intersect:
338                     case Norms:
339                     case NumericDocValues:
340                     case BinaryDocValues:
341                     case SortedDocValues:
342                     case SortedSetDocValues:
343                         if (random.nextDouble() < topLevelRatio) {
344                             throw new IOException("Forced top level Exception on [" + flag.name() + "]");
345                         }
346                         break;
347                     case DocsEnum:
348                     case DocsAndPositionsEnum:
349                         if (random.nextDouble() < lowLevelRatio) {
350                             throw new IOException("Forced low level Exception on [" + flag.name() + "]");
351                         }
352                         break;
353                 }
354             }
355 
356             @Override
357             public boolean wrapTerms(String field) {
358                 return true;
359             }
360         }
361 
362         public RandomExceptionDirectoryReaderWrapper(DirectoryReader in, Settings settings) throws IOException {
363             super(in, new ThrowingSubReaderWrapper(settings));
364             this.settings = settings;
365         }
366 
367         @Override
368         protected DirectoryReader doWrapDirectoryReader(DirectoryReader in) throws IOException {
369             return new RandomExceptionDirectoryReaderWrapper(in, settings);
370         }
371     }
