114     private void insertType1(List<String> values1, List<String> values2) throws Exception {
115         XContentBuilder source = jsonBuilder()
116                 .startObject()
117                 .array("field1", values1.toArray())
118                 .startArray("nested1");
119         for (String value1 : values2) {
120             source.startObject().field("field2", value1).endObject();
121         }
122         source.endArray().endObject();
123         indexRandom(false, client().prepareIndex("idx", "type1").setRouting("1").setSource(source));
124     }

126     private void insertType2(String[][] values) throws Exception {
127         XContentBuilder source = jsonBuilder()
128                 .startObject()
129                 .startArray("nested1");
130         for (String[] value : values) {
131             source.startObject().field("field1", value[0]).startArray("nested2");
132             for (int i = 1; i < value.length; i++) {
133                 source.startObject().field("field2", value[i]).endObject();
134             }
135             source.endArray().endObject();
136         }
137         source.endArray().endObject();
138         indexRandom(false, client().prepareIndex("idx", "type2").setRouting("1").setSource(source));
139     }
