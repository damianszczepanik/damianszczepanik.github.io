112     @Test
113     public void testPutWarmerNoAcknowledgement() throws InterruptedException {
114         createIndex("test");
115         // make sure one shard is started so the search during put warmer will not fail
116         index("test", "type", "1", "f", 1);
117 
118         PutWarmerResponse putWarmerResponse = client().admin().indices().preparePutWarmer("custom_warmer").setTimeout("0s")
119                 .setSearchRequest(client().prepareSearch("test").setTypes("test").setQuery(QueryBuilders.matchAllQuery()))
120                 .get();
121         assertThat(putWarmerResponse.isAcknowledged(), equalTo(false));
122         /* Since we don't wait for the ack here we have to wait until the search request has been executed from the master
123          * otherwise the test infra might have already deleted the index and the search request fails on all shards causing
124          * the test to fail too. We simply wait until the the warmer has been installed and also clean it up afterwards.*/
125         assertTrue(awaitBusy(new Predicate<Object>() {
126             @Override
127             public boolean apply(Object input) {
128                 for (Client client : clients()) {
129                     GetWarmersResponse getWarmersResponse = client.admin().indices().prepareGetWarmers().setLocal(true).get();
130                     if (getWarmersResponse.warmers().size() != 1) {
131                         return false;
132                     }
133                 }
134                 return true;
135             }
136         }));
137         assertAcked(client().admin().indices().prepareDeleteWarmer().setIndices("test").setNames("custom_warmer"));
138     }

211     @Test
212     public void testClusterRerouteAcknowledgement() throws InterruptedException {
213         assertAcked(prepareCreate("test").setSettings(ImmutableSettings.builder()
214                 .put(indexSettings())
215                 .put(SETTING_NUMBER_OF_SHARDS, between(cluster().numDataNodes(), DEFAULT_MAX_NUM_SHARDS))
216                 .put(SETTING_NUMBER_OF_REPLICAS, 0)
217         ));
218         ensureGreen();
219 
220         MoveAllocationCommand moveAllocationCommand = getAllocationCommand();
221 
222         assertAcked(client().admin().cluster().prepareReroute().add(moveAllocationCommand));
223 
224         for (Client client : clients()) {
225             ClusterState clusterState = getLocalClusterState(client);
226             for (MutableShardRouting mutableShardRouting : clusterState.routingNodes().routingNodeIter(moveAllocationCommand.fromNode())) {
227                 //if the shard that we wanted to move is still on the same node, it must be relocating
228                 if (mutableShardRouting.shardId().equals(moveAllocationCommand.shardId())) {
229                     assertThat(mutableShardRouting.relocating(), equalTo(true));
230                 }
231 
232             }
233 
234             boolean found = false;
235             for (MutableShardRouting mutableShardRouting : clusterState.routingNodes().routingNodeIter(moveAllocationCommand.toNode())) {
236                 if (mutableShardRouting.shardId().equals(moveAllocationCommand.shardId())) {
237                     assertThat(mutableShardRouting.state(), anyOf(equalTo(ShardRoutingState.INITIALIZING), equalTo(ShardRoutingState.STARTED)));
238                     found = true;
239                     break;
240                 }
241             }
242             assertThat(found, equalTo(true));
243         }
244     }

260     @Test
261     public void testClusterRerouteAcknowledgementDryRun() throws InterruptedException {
262         client().admin().indices().prepareCreate("test")
263                 .setSettings(settingsBuilder()
264                         .put(SETTING_NUMBER_OF_SHARDS, between(cluster().numDataNodes(), DEFAULT_MAX_NUM_SHARDS))
265                         .put(SETTING_NUMBER_OF_REPLICAS, 0)).get();
266         ensureGreen();
267 
268         MoveAllocationCommand moveAllocationCommand = getAllocationCommand();
269 
270         assertAcked(client().admin().cluster().prepareReroute().setDryRun(true).add(moveAllocationCommand));
271 
272         //testing only on master with the latest cluster state as we didn't make any change thus we cannot guarantee that
273         //all nodes hold the same cluster state version. We only know there was no need to change anything, thus no need for ack on this update.
274         ClusterStateResponse clusterStateResponse = client().admin().cluster().prepareState().get();
275         boolean found = false;
276         for (MutableShardRouting mutableShardRouting : clusterStateResponse.getState().routingNodes().routingNodeIter(moveAllocationCommand.fromNode())) {
277             //the shard that we wanted to move is still on the same node, as we had dryRun flag
278             if (mutableShardRouting.shardId().equals(moveAllocationCommand.shardId())) {
279                 assertThat(mutableShardRouting.started(), equalTo(true));
280                 found = true;
281                 break;
282             }
283         }
284         assertThat(found, equalTo(true));
285 
286         for (MutableShardRouting mutableShardRouting : clusterStateResponse.getState().routingNodes().routingNodeIter(moveAllocationCommand.toNode())) {
287             if (mutableShardRouting.shardId().equals(moveAllocationCommand.shardId())) {
288                 fail("shard [" + mutableShardRouting + "] shouldn't be on node [" + moveAllocationCommand.toString() + "]");
289             }
290         }
291     }

308     private MoveAllocationCommand getAllocationCommand() {
309         String fromNodeId = null;
310         String toNodeId = null;
311         MutableShardRouting shardToBeMoved = null;
312         ClusterStateResponse clusterStateResponse = client().admin().cluster().prepareState().get();
313         for (RoutingNode routingNode : clusterStateResponse.getState().routingNodes()) {
314             if (routingNode.node().isDataNode()) {
315                 if (fromNodeId == null && routingNode.numberOfOwningShards() > 0) {
316                     fromNodeId = routingNode.nodeId();
317                     shardToBeMoved = routingNode.get(randomInt(routingNode.size() - 1));
318                 } else {
319                     toNodeId = routingNode.nodeId();
320                 }
321 
322                 if (toNodeId != null && fromNodeId != null) {
323                     break;
324                 }
325             }
326         }
327 
328         assertNotNull(fromNodeId);
329         assertNotNull(toNodeId);
330         assertNotNull(shardToBeMoved);
331 
332         logger.info("==> going to move shard [{}] from [{}] to [{}]", shardToBeMoved, fromNodeId, toNodeId);
333         return new MoveAllocationCommand(shardToBeMoved.shardId(), fromNodeId, toNodeId);
334     }
