104     protected static GeometryCollectionBuilder createGeometryCollection(Random r, Point nearPoint, Rectangle bounds, int numGeometries)
105             throws InvalidShapeException {
106         if (numGeometries <= 0) {
107             // cap geometry collection at 4 shapes (to save test time)
108             numGeometries = RandomInts.randomIntBetween(r, 2, 5);
109         }
110 
111         if (nearPoint == null) {
112             nearPoint = xRandomPoint(r);
113         }
114 
115         if (bounds == null) {
116             bounds = xRandomRectangle(r, nearPoint);
117         }
118 
119         GeometryCollectionBuilder gcb = new GeometryCollectionBuilder();
120         for (int i=0; i<numGeometries;) {
121             ShapeBuilder builder = createShapeWithin(r, bounds);
122             // due to world wrapping, and the possibility for ambiguous polygons, the random shape generation could bail with
123             // a null shape. We catch that situation here, and only increment the counter when a valid shape is returned.
124             // Not the most efficient but its the lesser of the evil alternatives
125             if (builder != null) {
126                 gcb.shape(builder);
127                 ++i;
128             }
129         }
130         return gcb;
131     }

133     private static ShapeBuilder createShape(Random r, Point nearPoint, Rectangle within, ShapeType st) throws InvalidShapeException {
134         return createShape(r, nearPoint, within, st, ST_VALIDATE);
135     }

137     /**
138      * Creates a random shape useful for randomized testing, NOTE: exercise caution when using this to build random GeometryCollections
139      * as creating a large random number of random shapes can result in massive resource consumption
140      * see: {@link org.elasticsearch.search.geo.GeoShapeIntegrationTests#testShapeFilterWithRandomGeoCollection}
141      *
142      * The following options are included
143      * @param nearPoint Create a shape near a provided point
144      * @param within Create a shape within the provided rectangle (note: if not null this will override the provided point)
145      * @param st Create a random shape of the provided type
146      * @return the ShapeBuilder for a random shape
147      */
148     private static ShapeBuilder createShape(Random r, Point nearPoint, Rectangle within, ShapeType st, boolean validate) throws
149             InvalidShapeException {
150 
151         if (st == null) {
152             st = ShapeType.randomType(r);
153         }
154 
155         if (within == null) {
156             within = xRandomRectangle(r, nearPoint);
157         }
158 
159         // NOTE: multipolygon not yet supported. Overlapping polygons are invalid so randomization
160         // requires an approach to avoid overlaps. This could be approached by creating polygons
161         // inside non overlapping bounding rectangles
162         switch (st) {
163             case POINT:
164                 Point p = xRandomPointIn(r, within);
165                 PointBuilder pb = new PointBuilder().coordinate(new Coordinate(p.getX(), p.getY(), Double.NaN));
166                 return pb;
167             case MULTIPOINT:
168             case LINESTRING:
169                 // for random testing having a maximum number of 10 points for a line string is more than sufficient
170                 // if this number gets out of hand, the number of self intersections for a linestring can become
171                 // (n^2-n)/2 and computing the relation intersection matrix will become NP-Hard
172                 int numPoints = RandomInts.randomIntBetween(r, 3, 10);
173                 PointCollection pcb = (st == ShapeType.MULTIPOINT) ? new MultiPointBuilder() : new LineStringBuilder();
174                 for (int i=0; i<numPoints; ++i) {
175                     p = xRandomPointIn(r, within);
176                     pcb.point(p.getX(), p.getY());
177                 }
178                 return pcb;
179             case MULTILINESTRING:
180                 MultiLineStringBuilder mlsb = new MultiLineStringBuilder();
181                 for (int i=0; i<RandomInts.randomIntBetween(r, 1, 10); ++i) {
182                     mlsb.linestring((BaseLineStringBuilder) createShape(r, nearPoint, within, ShapeType.LINESTRING, false));
183                 }
184                 return mlsb;
185             case POLYGON:
186                 numPoints = RandomInts.randomIntBetween(r, 5, 25);
187                 Coordinate[] coordinates = new Coordinate[numPoints];
188                 for (int i=0; i<numPoints; ++i) {
189                     p = (Point) createShape(r, nearPoint, within, ShapeType.POINT, false).build();
190                     coordinates[i] = new Coordinate(p.getX(), p.getY());
191                 }
192                 // random point order or random linestrings can lead to invalid self-crossing polygons,
193                 // compute the convex hull for a set of points to ensure polygon does not self cross
194                 Geometry shell = new ConvexHull(coordinates, ctx.getGeometryFactory()).getConvexHull();
195                 Coordinate[] shellCoords = shell.getCoordinates();
196                 // if points are in a line the convex hull will be 2 points which will also lead to an invalid polygon
197                 // when all else fails, use the bounding box as the polygon
198                 if (shellCoords.length < 3) {
199                     shellCoords = new Coordinate[4];
200                     shellCoords[0] = new Coordinate(within.getMinX(), within.getMinY());
201                     shellCoords[1] = new Coordinate(within.getMinX(), within.getMaxY());
202                     shellCoords[2] = new Coordinate(within.getMaxX(), within.getMaxY());
203                     shellCoords[3] = new Coordinate(within.getMaxX(), within.getMinY());
204                 }
205                 PolygonBuilder pgb = new PolygonBuilder().points(shellCoords).close();
206                 if (validate) {
207                     // This test framework builds semi-random geometry (in the sense that points are not truly random due to spatial
208                     // auto-correlation) As a result of the semi-random nature of the geometry, one can not predict the orientation
209                     // intent for ambiguous polygons. Therefore, an invalid oriented dateline crossing polygon could be built.
210                     // The validate flag will check for these possibilities and bail if an incorrect geometry is created
211                     try {
212                         pgb.build();
213                     } catch (InvalidShapeException e) {
214                         return null;
215                     }
216                 }
217                 return pgb;
218             default:
219                 throw new ElasticsearchException("Unable to create shape of type [" + st + "]");
220         }
221     }

223     protected static Point xRandomPoint(Random r) {
224         return xRandomPointIn(r, ctx.getWorldBounds());
225     }
226 
227     protected static Point xRandomPointIn(Random rand, Rectangle r) {
228         double x = r.getMinX() + rand.nextDouble()*r.getWidth();
229         double y = r.getMinY() + rand.nextDouble()*r.getHeight();
230         x = xNormX(x);
231         y = xNormY(y);
232         Point p = ctx.makePoint(x,y);
233         RandomizedTest.assertEquals(CONTAINS, r.relate(p));
234         return p;
235     }
236 
237     protected static Rectangle xRandomRectangle(Random r, Point nearP) {
238         Rectangle bounds = ctx.getWorldBounds();
239         if (nearP == null)
240             nearP = xRandomPointIn(r, bounds);
241 
242         Range xRange = xRandomRange(r, rarely(r) ? 0 : nearP.getX(), Range.xRange(bounds, ctx));
243         Range yRange = xRandomRange(r, rarely(r) ? 0 : nearP.getY(), Range.yRange(bounds, ctx));
244 
245         return xMakeNormRect(
246                 xDivisible(xRange.getMin()*10e3)/10e3,
247                 xDivisible(xRange.getMax()*10e3)/10e3,
248                 xDivisible(yRange.getMin()*10e3)/10e3,
249                 xDivisible(yRange.getMax()*10e3)/10e3);
250     }
251 
252     private static boolean rarely(Random r) {
253         return RandomInts.randomInt(r, 100) >= 90;
254     }
255 
256     private static Range xRandomRange(Random r, double near, Range bounds) {
257         double mid = near + r.nextGaussian() * bounds.getWidth() / 6;
258         double width = Math.abs(r.nextGaussian()) * bounds.getWidth() / 6;//1/3rd
259         return new Range(mid - width / 2, mid + width / 2);
260     }
261 
262     private static double xDivisible(double v, double divisible) {
263         return (int) (Math.round(v / divisible) * divisible);
264     }
265 
266     private static double xDivisible(double v) {
267         return xDivisible(v, xDIVISIBLE);
268     }
269 
270     protected static Rectangle xMakeNormRect(double minX, double maxX, double minY, double maxY) {
271         minX = DistanceUtils.normLonDEG(minX);
272         maxX = DistanceUtils.normLonDEG(maxX);
273 
274         if (maxX < minX) {
275             double t = minX;
276             minX = maxX;
277             maxX = t;
278         }
279 
280         double minWorldY = ctx.getWorldBounds().getMinY();
281         double maxWorldY = ctx.getWorldBounds().getMaxY();
282         if (minY < minWorldY || minY > maxWorldY) {
283             minY = DistanceUtils.normLatDEG(minY);
284         }
285         if (maxY < minWorldY || maxY > maxWorldY) {
286             maxY = DistanceUtils.normLatDEG(maxY);
287         }
288         if (maxY < minY) {
289             double t = minY;
290             minY = maxY;
291             maxY = t;
292         }
293         return ctx.makeRectangle(minX, maxX, minY, maxY);
294     }
295 
296     protected static double xNormX(double x) {
297         return ctx.isGeo() ? DistanceUtils.normLonDEG(x) : x;
298     }
299 
300     protected static double xNormY(double y) {
301         return ctx.isGeo() ? DistanceUtils.normLatDEG(y) : y;
302     }
