55      static ObjectIntMap<String> expectedDocCountsForGeoHash = null;
56      static ObjectIntMap<String> multiValuedExpectedDocCountsForGeoHash = null;
57      static int highestPrecisionGeohash = 12;
58      static int numDocs = 100;
59  
60      static String smallestGeoHash = null;
61  
62      private static IndexRequestBuilder indexCity(String index, String name, List<String> latLon) throws Exception {
63          XContentBuilder source = jsonBuilder().startObject().field("city", name);
64          if (latLon != null) {
65              source = source.field("location", latLon);
66          }
67          source = source.endObject();
68          return client().prepareIndex(index, "type").setSource(source);
69      }
70  
71      private static IndexRequestBuilder indexCity(String index, String name, String latLon) throws Exception {
72          return indexCity(index, name, Arrays.<String>asList(latLon));
73      }

75      @Override
76      public void setupSuiteScopeCluster() throws Exception {
77          createIndex("idx_unmapped");
78  
79          assertAcked(prepareCreate("idx")
80                  .addMapping("type", "location", "type=geo_point", "city", "type=string,index=not_analyzed"));
81  
82          List<IndexRequestBuilder> cities = new ArrayList<>();
83          Random random = getRandom();
84          expectedDocCountsForGeoHash = new ObjectIntOpenHashMap<>(numDocs * 2);
85          for (int i = 0; i < numDocs; i++) {
86              //generate random point
87              double lat = (180d * random.nextDouble()) - 90d;
88              double lng = (360d * random.nextDouble()) - 180d;
89              String randomGeoHash = GeoHashUtils.encode(lat, lng, highestPrecisionGeohash);
90              //Index at the highest resolution
91              cities.add(indexCity("idx", randomGeoHash, lat + ", " + lng));
92              expectedDocCountsForGeoHash.put(randomGeoHash, expectedDocCountsForGeoHash.getOrDefault(randomGeoHash, 0) + 1);
93              //Update expected doc counts for all resolutions..
94              for (int precision = highestPrecisionGeohash - 1; precision > 0; precision--) {
95                  String hash = GeoHashUtils.encode(lat, lng, precision);
96                  if ((smallestGeoHash == null) || (hash.length() < smallestGeoHash.length())) {
97                      smallestGeoHash = hash;
98                  }
99                  expectedDocCountsForGeoHash.put(hash, expectedDocCountsForGeoHash.getOrDefault(hash, 0) + 1);
100             }
101         }
102         indexRandom(true, cities);
103 
104         assertAcked(prepareCreate("multi_valued_idx")
105                 .addMapping("type", "location", "type=geo_point", "city", "type=string,index=not_analyzed"));
106 
107         cities = new ArrayList<>();
108         multiValuedExpectedDocCountsForGeoHash = new ObjectIntOpenHashMap<>(numDocs * 2);
109         for (int i = 0; i < numDocs; i++) {
110             final int numPoints = random.nextInt(4);
111             List<String> points = new ArrayList<>();
112             Set<String> geoHashes = new HashSet<>();
113             for (int j = 0; j < numPoints; ++j) {
114                 double lat = (180d * random.nextDouble()) - 90d;
115                 double lng = (360d * random.nextDouble()) - 180d;
116                 points.add(lat + "," + lng);
117                 // Update expected doc counts for all resolutions..
118                 for (int precision = highestPrecisionGeohash; precision > 0; precision--) {
119                     final String geoHash = GeoHashUtils.encode(lat, lng, precision);
120                     geoHashes.add(geoHash);
121                 }
122             }
123             cities.add(indexCity("multi_valued_idx", Integer.toString(i), points));
124             for (String hash : geoHashes) {
125                 multiValuedExpectedDocCountsForGeoHash.put(hash, multiValuedExpectedDocCountsForGeoHash.getOrDefault(hash, 0) + 1);
126             }
127         }
128         indexRandom(true, cities);
129 
130         ensureSearchable();
131     }

134     @Test
135     public void simple() throws Exception {
136         for (int precision = 1; precision <= highestPrecisionGeohash; precision++) {
137             SearchResponse response = client().prepareSearch("idx")
138                     .addAggregation(geohashGrid("geohashgrid")
139                             .field("location")
140                             .precision(precision)
141                     )
142                     .execute().actionGet();
143 
144             assertSearchResponse(response);
145 
146             GeoHashGrid geoGrid = response.getAggregations().get("geohashgrid");
147             List<Bucket> buckets = geoGrid.getBuckets();
148             Object[] propertiesKeys = (Object[]) geoGrid.getProperty("_key");
149             Object[] propertiesDocCounts = (Object[]) geoGrid.getProperty("_count");
150             for (int i = 0; i < buckets.size(); i++) {
151                 GeoHashGrid.Bucket cell = buckets.get(i);
152                 String geohash = cell.getKeyAsString();
153 
154                 long bucketCount = cell.getDocCount();
155                 int expectedBucketCount = expectedDocCountsForGeoHash.get(geohash);
156                 assertNotSame(bucketCount, 0);
157                 assertEquals("Geohash " + geohash + " has wrong doc count ",
158                         expectedBucketCount, bucketCount);
159                 GeoPoint geoPoint = (GeoPoint) propertiesKeys[i];
160                 assertThat(GeoHashUtils.encode(geoPoint.lat(), geoPoint.lon(), precision), equalTo(geohash));
161                 assertThat((long) propertiesDocCounts[i], equalTo(bucketCount));
162             }
163         }
164     }

191     @Test
192     public void filtered() throws Exception {
193         GeoBoundingBoxFilterBuilder bbox = new GeoBoundingBoxFilterBuilder("location");
194         bbox.topLeft(smallestGeoHash).bottomRight(smallestGeoHash).filterName("bbox");
195         for (int precision = 1; precision <= highestPrecisionGeohash; precision++) {
196             SearchResponse response = client().prepareSearch("idx")
197                     .addAggregation(
198                             AggregationBuilders.filter("filtered").filter(bbox)
199                                     .subAggregation(
200                                             geohashGrid("geohashgrid")
201                                                     .field("location")
202                                                     .precision(precision)
203                                     )
204                     )
205                     .execute().actionGet();
206 
207             assertSearchResponse(response);
208 
209             Filter filter = response.getAggregations().get("filtered");
210 
211             GeoHashGrid geoGrid = filter.getAggregations().get("geohashgrid");
212             for (GeoHashGrid.Bucket cell : geoGrid.getBuckets()) {
213                 String geohash = cell.getKeyAsString();
214                 long bucketCount = cell.getDocCount();
215                 int expectedBucketCount = expectedDocCountsForGeoHash.get(geohash);
216                 assertNotSame(bucketCount, 0);
217                 assertTrue("Buckets must be filtered", geohash.startsWith(smallestGeoHash));
218                 assertEquals("Geohash " + geohash + " has wrong doc count ",
219                         expectedBucketCount, bucketCount);
220 
221             }
222         }
223     }

268     @Test
269     public void testTopMatch() throws Exception {
270         for (int precision = 1; precision <= highestPrecisionGeohash; precision++) {
271             SearchResponse response = client().prepareSearch("idx")
272                     .addAggregation(geohashGrid("geohashgrid")
273                             .field("location")
274                             .size(1)
275                             .shardSize(100)
276                             .precision(precision)
277                     )
278                     .execute().actionGet();
279 
280             assertSearchResponse(response);
281 
282             GeoHashGrid geoGrid = response.getAggregations().get("geohashgrid");
283             //Check we only have one bucket with the best match for that resolution
284             assertThat(geoGrid.getBuckets().size(), equalTo(1));
285             for (GeoHashGrid.Bucket cell : geoGrid.getBuckets()) {
286                 String geohash = cell.getKeyAsString();
287                 long bucketCount = cell.getDocCount();
288                 int expectedBucketCount = 0;
289                 for (ObjectIntCursor<String> cursor : expectedDocCountsForGeoHash) {
290                     if (cursor.key.length() == precision) {
291                         expectedBucketCount = Math.max(expectedBucketCount, cursor.value);
292                     }
293                 }
294                 assertNotSame(bucketCount, 0);
295                 assertEquals("Geohash " + geohash + " has wrong doc count ",
296                         expectedBucketCount, bucketCount);
297             }
298         }
299     }

301     @Test
302     // making sure this doesn't runs into an OOME
303     public void sizeIsZero() {
304         for (int precision = 1; precision <= highestPrecisionGeohash; precision++) {
305             final int size = randomBoolean() ? 0 : randomIntBetween(1, Integer.MAX_VALUE);
306             final int shardSize = randomBoolean() ? -1 : 0;
307             SearchResponse response = client().prepareSearch("idx")
308                     .addAggregation(geohashGrid("geohashgrid")
309                             .field("location")
310                             .size(size)
311                             .shardSize(shardSize)
312                             .precision(precision)
313                     )
314                     .execute().actionGet();
315 
316             assertSearchResponse(response);
317             GeoHashGrid geoGrid = response.getAggregations().get("geohashgrid");
318             assertThat(geoGrid.getBuckets().size(), greaterThanOrEqualTo(1));
319         }
