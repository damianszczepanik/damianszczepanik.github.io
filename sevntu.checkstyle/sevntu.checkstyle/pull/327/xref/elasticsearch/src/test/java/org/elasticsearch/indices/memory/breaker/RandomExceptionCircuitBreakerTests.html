59      @Test
60      public void testBreakerWithRandomExceptions() throws IOException, InterruptedException, ExecutionException {
61          for (NodeStats node : client().admin().cluster().prepareNodesStats()
62                  .clear().setBreaker(true).execute().actionGet().getNodes()) {
63              assertThat("Breaker is not set to 0", node.getBreaker().getStats(CircuitBreaker.FIELDDATA).getEstimated(), equalTo(0L));
64          }
65  
66          String mapping = XContentFactory.jsonBuilder()
67                  .startObject()
68                  .startObject("type")
69                  .startObject("properties")
70                  .startObject("test-str")
71                  .field("type", "string")
72                  .field("index", "not_analyzed")
73                  .startObject("fielddata")
74                  .field("format", randomBytesFieldDataFormat())
75                  .endObject() // fielddata
76                  .endObject() // test-str
77                  .startObject("test-num")
78                          // I don't use randomNumericType() here because I don't want "byte", and I want "float" and "double"
79                  .field("type", randomFrom(Arrays.asList("float", "long", "double", "short", "integer")))
80                  .startObject("fielddata")
81                  .field("format", randomNumericFieldDataFormat())
82                  .endObject() // fielddata
83                  .endObject() // test-num
84                  .endObject() // properties
85                  .endObject() // type
86                  .endObject() // {}
87                  .string();
88          final double topLevelRate;
89          final double lowLevelRate;
90          if (frequently()) {
91              if (randomBoolean()) {
92                  if (randomBoolean()) {
93                      lowLevelRate = 1.0 / between(2, 10);
94                      topLevelRate = 0.0d;
95                  } else {
96                      topLevelRate = 1.0 / between(2, 10);
97                      lowLevelRate = 0.0d;
98                  }
99              } else {
100                 lowLevelRate = 1.0 / between(2, 10);
101                 topLevelRate = 1.0 / between(2, 10);
102             }
103         } else {
104             // rarely no exception
105             topLevelRate = 0d;
106             lowLevelRate = 0d;
107         }
108 
109         ImmutableSettings.Builder settings = settingsBuilder()
110                 .put(indexSettings())
111                 .put(MockInternalEngine.READER_WRAPPER_TYPE, RandomExceptionDirectoryReaderWrapper.class.getName())
112                 .put(EXCEPTION_TOP_LEVEL_RATIO_KEY, topLevelRate)
113                 .put(EXCEPTION_LOW_LEVEL_RATIO_KEY, lowLevelRate)
114                 .put(MockInternalEngine.WRAP_READER_RATIO, 1.0d);
115         logger.info("creating index: [test] using settings: [{}]", settings.build().getAsMap());
116         client().admin().indices().prepareCreate("test")
117                 .setSettings(settings)
118                 .addMapping("type", mapping).execute().actionGet();
119         ClusterHealthResponse clusterHealthResponse = client().admin().cluster()
120                 .health(Requests.clusterHealthRequest().waitForYellowStatus().timeout(TimeValue.timeValueSeconds(5))).get(); // it's OK to timeout here
121         final int numDocs;
122         if (clusterHealthResponse.isTimedOut()) {
123             /* some seeds just won't let you create the index at all and we enter a ping-pong mode
124              * trying one node after another etc. that is ok but we need to make sure we don't wait
125              * forever when indexing documents so we set numDocs = 1 and expect all shards to fail
126              * when we search below.*/
127             logger.info("ClusterHealth timed out - only index one doc and expect searches to fail");
128             numDocs = 1;
129         } else {
130             numDocs = between(10, 100);
131         }
132         for (int i = 0; i < numDocs; i++) {
133             try {
134                 client().prepareIndex("test", "type", "" + i)
135                         .setTimeout(TimeValue.timeValueSeconds(1)).setSource("test-str", randomUnicodeOfLengthBetween(5, 25), "test-num", i).get();
136             } catch (ElasticsearchException ex) {
137             }
138         }
139         logger.info("Start Refresh");
140         RefreshResponse refreshResponse = client().admin().indices().prepareRefresh("test").execute().get(); // don't assert on failures here
141         final boolean refreshFailed = refreshResponse.getShardFailures().length != 0 || refreshResponse.getFailedShards() != 0;
142         logger.info("Refresh failed: [{}] numShardsFailed: [{}], shardFailuresLength: [{}], successfulShards: [{}], totalShards: [{}] ",
143                 refreshFailed, refreshResponse.getFailedShards(), refreshResponse.getShardFailures().length,
144                 refreshResponse.getSuccessfulShards(), refreshResponse.getTotalShards());
145         final int numSearches = scaledRandomIntBetween(50, 150);
146         NodesStatsResponse resp = client().admin().cluster().prepareNodesStats()
147                 .clear().setBreaker(true).execute().actionGet();
148         for (NodeStats stats : resp.getNodes()) {
149             assertThat("Breaker is set to 0", stats.getBreaker().getStats(CircuitBreaker.FIELDDATA).getEstimated(), equalTo(0L));
150         }
151 
152         for (int i = 0; i < numSearches; i++) {
153             SearchRequestBuilder searchRequestBuilder = client().prepareSearch().setQuery(QueryBuilders.matchAllQuery());
154             switch (randomIntBetween(0, 5)) {
155                 case 5:
156                 case 4:
157                 case 3:
158                     searchRequestBuilder.addSort("test-str", SortOrder.ASC);
159                     // fall through - sometimes get both fields
160                 case 2:
161                 case 1:
162                 default:
163                     searchRequestBuilder.addSort("test-num", SortOrder.ASC);
164 
165             }
166             boolean success = false;
167             try {
168                 // Sort by the string and numeric fields, to load them into field data
169                 searchRequestBuilder.get();
170                 success = true;
171             } catch (SearchPhaseExecutionException ex) {
172                 logger.info("expected SearchPhaseException: [{}]", ex.getMessage());
173             }
174 
175             if (frequently()) {
176                 // Now, clear the cache and check that the circuit breaker has been
177                 // successfully set back to zero. If there is a bug in the circuit
178                 // breaker adjustment code, it should show up here by the breaker
179                 // estimate being either positive or negative.
180                 ensureGreen("test");  // make sure all shards are there - there could be shards that are still starting up.
181                 assertAllSuccessful(client().admin().indices().prepareClearCache("test").setFieldDataCache(true).execute().actionGet());
182 
183                 // Since .cleanUp() is no longer called on cache clear, we need to call it on each node manually
184                 for (String node : internalCluster().getNodeNames()) {
185                     final IndicesFieldDataCache fdCache = internalCluster().getInstance(IndicesFieldDataCache.class, node);
186                     // Clean up the cache, ensuring that entries' listeners have been called
187                     fdCache.getCache().cleanUp();
188                 }
189                 NodesStatsResponse nodeStats = client().admin().cluster().prepareNodesStats()
190                         .clear().setBreaker(true).execute().actionGet();
191                 for (NodeStats stats : nodeStats.getNodes()) {
192                     assertThat("Breaker reset to 0 last search success: " + success + " mapping: " + mapping,
193                             stats.getBreaker().getStats(CircuitBreaker.FIELDDATA).getEstimated(), equalTo(0L));
194                 }
195             }
196         }
197     }

200     public static final String EXCEPTION_TOP_LEVEL_RATIO_KEY = "index.engine.exception.ratio.top";
201     public static final String EXCEPTION_LOW_LEVEL_RATIO_KEY = "index.engine.exception.ratio.low";
202 
203     // TODO: Generalize this class and add it as a utility
204     public static class RandomExceptionDirectoryReaderWrapper extends MockInternalEngine.DirectoryReaderWrapper {
205         private final Settings settings;
206 
207         static class ThrowingSubReaderWrapper extends SubReaderWrapper implements ThrowingLeafReaderWrapper.Thrower {
208             private final Random random;
209             private final double topLevelRatio;
210             private final double lowLevelRatio;
211 
212             ThrowingSubReaderWrapper(Settings settings) {
213                 final long seed = settings.getAsLong(SETTING_INDEX_SEED, 0l);
214                 this.topLevelRatio = settings.getAsDouble(EXCEPTION_TOP_LEVEL_RATIO_KEY, 0.1d);
215                 this.lowLevelRatio = settings.getAsDouble(EXCEPTION_LOW_LEVEL_RATIO_KEY, 0.1d);
216                 this.random = new Random(seed);
217             }
218 
219             @Override
220             public LeafReader wrap(LeafReader reader) {
221                 return new ThrowingLeafReaderWrapper(reader, this);
222             }
223 
224             @Override
225             public void maybeThrow(ThrowingLeafReaderWrapper.Flags flag) throws IOException {
226                 switch (flag) {
227                     case Fields:
228                         break;
229                     case TermVectors:
230                         break;
231                     case Terms:
232                     case TermsEnum:
233                         if (random.nextDouble() < topLevelRatio) {
234                             throw new IOException("Forced top level Exception on [" + flag.name() + "]");
235                         }
236                     case Intersect:
237                         break;
238                     case Norms:
239                         break;
240                     case NumericDocValues:
241                         break;
242                     case BinaryDocValues:
243                         break;
244                     case SortedDocValues:
245                         break;
246                     case SortedSetDocValues:
247                         break;
248                     case DocsEnum:
249                     case DocsAndPositionsEnum:
250                         if (random.nextDouble() < lowLevelRatio) {
251                             throw new IOException("Forced low level Exception on [" + flag.name() + "]");
252                         }
253                         break;
254                 }
255             }
256 
257             @Override
258             public boolean wrapTerms(String field) {
259                 return field.startsWith("test");
260             }
261         }
262 
263 
264         public RandomExceptionDirectoryReaderWrapper(DirectoryReader in, Settings settings) throws IOException {
265             super(in, new ThrowingSubReaderWrapper(settings));
266             this.settings = settings;
267         }
268 
269         @Override
270         protected DirectoryReader doWrapDirectoryReader(DirectoryReader in) throws IOException {
271             return new RandomExceptionDirectoryReaderWrapper(in, settings);
272         }
273     }
