38      @Test
39      public void testV0LegacyTranslogVersion() throws Exception {
40          Path translogFile = getResourcePath("/org/elasticsearch/index/translog/translog-v0.binary");
41          assertThat("test file should exist", Files.exists(translogFile), equalTo(true));
42          TranslogStream stream = TranslogStreams.translogStreamFor(translogFile);
43          assertThat("a version0 stream is returned", stream instanceof LegacyTranslogStream, equalTo(true));
44  
45          StreamInput in = stream.openInput(translogFile);
46  
47          in.readInt();
48          Translog.Operation operation = stream.read(in);
49  
50          assertThat("operation is the correct type correctly", operation.opType() == Translog.Operation.Type.SAVE, equalTo(true));
51          Translog.Index op = (Translog.Index) operation;
52          assertThat(op.id(), equalTo("1"));
53          assertThat(op.type(), equalTo("doc"));
54          assertThat(op.source().toUtf8(), equalTo("{\"body\": \"worda wordb wordc wordd \\\"worde\\\" wordf\"}"));
55          assertThat(op.routing(), equalTo(null));
56          assertThat(op.parent(), equalTo(null));
57          assertThat(op.version(), equalTo(1L));
58          assertThat(op.timestamp(), equalTo(1407312091791L));
59          assertThat(op.ttl(), equalTo(-1L));
60          assertThat(op.versionType(), equalTo(VersionType.INTERNAL));
61  
62          try {
63              in.readInt();
64              stream.read(in);
65              fail("should have been the end of the file");
66          } catch (EOFException e) {
67              // success
68          }
69      }

71      @Test
72      public void testV1ChecksummedTranslogVersion() throws Exception {
73          Path translogFile = getResourcePath("/org/elasticsearch/index/translog/translog-v1.binary");
74          assertThat("test file should exist", Files.exists(translogFile), equalTo(true));
75          TranslogStream stream = TranslogStreams.translogStreamFor(translogFile);
76          assertThat("a version1 stream is returned", stream instanceof ChecksummedTranslogStream, equalTo(true));
77  
78          StreamInput in = stream.openInput(translogFile);
79          Translog.Operation operation = stream.read(in);
80  
81          assertThat("operation is the correct type correctly", operation.opType() == Translog.Operation.Type.CREATE, equalTo(true));
82          Translog.Create op = (Translog.Create) operation;
83          assertThat(op.id(), equalTo("Bwiq98KFSb6YjJQGeSpeiw"));
84          assertThat(op.type(), equalTo("doc"));
85          assertThat(op.source().toUtf8(), equalTo("{\"body\": \"foo\"}"));
86          assertThat(op.routing(), equalTo(null));
87          assertThat(op.parent(), equalTo(null));
88          assertThat(op.version(), equalTo(1L));
89          assertThat(op.timestamp(), equalTo(1408627184844L));
90          assertThat(op.ttl(), equalTo(-1L));
91          assertThat(op.versionType(), equalTo(VersionType.INTERNAL));
92  
93          // There are more operations
94          int opNum = 1;
95          while (true) {
96              try {
97                  stream.read(in);
98                  opNum++;
99              } catch (EOFException e) {
100                 break;
101             }
102         }
103         assertThat("there should be 5 translog operations", opNum, equalTo(5));
104     }

106     @Test
107     public void testCorruptedTranslogs() throws Exception {
108         try {
109             Path translogFile = getResourcePath("/org/elasticsearch/index/translog/translog-v1-corrupted-magic.binary");
110             assertThat("test file should exist", Files.exists(translogFile), equalTo(true));
111             TranslogStream stream = TranslogStreams.translogStreamFor(translogFile);
112             fail("should have thrown an exception about the header being corrupt");
113         } catch (TranslogCorruptedException e) {
114             assertThat("translog corruption from header: " + e.getMessage(),
115                     e.getMessage().contains("translog looks like version 1 or later, but has corrupted header"), equalTo(true));
116         }
117 
118         try {
119             Path translogFile = getResourcePath("/org/elasticsearch/index/translog/translog-invalid-first-byte.binary");
120             assertThat("test file should exist", Files.exists(translogFile), equalTo(true));
121             TranslogStream stream = TranslogStreams.translogStreamFor(translogFile);
122             fail("should have thrown an exception about the header being corrupt");
123         } catch (TranslogCorruptedException e) {
124             assertThat("translog corruption from header: " + e.getMessage(),
125                     e.getMessage().contains("Invalid first byte in translog file, got: 1, expected 0x00 or 0x3f"), equalTo(true));
126         }
127 
128         try {
129             Path translogFile = getResourcePath("/org/elasticsearch/index/translog/translog-v1-corrupted-body.binary");
130             assertThat("test file should exist", Files.exists(translogFile), equalTo(true));
131             TranslogStream stream = TranslogStreams.translogStreamFor(translogFile);
132             try (StreamInput in = stream.openInput(translogFile)) {
133                 while (true) {
134                     try {
135                         stream.read(in);
136                     } catch (EOFException e) {
137                         break;
138                     }
139                 }
140             }
141             fail("should have thrown an exception about the body being corrupted");
142         } catch (TranslogCorruptedException e) {
143             assertThat("translog corruption from body: " + e.getMessage(),
144                     e.getMessage().contains("translog stream is corrupted"), equalTo(true));
145         }
146 
147     }

149     @Test
150     public void testTruncatedTranslog() throws Exception {
151         try {
152             Path translogFile = getResourcePath("/org/elasticsearch/index/translog/translog-v1-truncated.binary");
153             assertThat("test file should exist", Files.exists(translogFile), equalTo(true));
154             TranslogStream stream = TranslogStreams.translogStreamFor(translogFile);
155             try (StreamInput in = stream.openInput(translogFile)) {
156                 while (true) {
157                     try {
158                         stream.read(in);
159                     } catch (EOFException e) {
160                         break;
161                     }
162                 }
163             }
164             fail("should have thrown an exception about the body being truncated");
165         } catch (TruncatedTranslogException e) {
166             assertThat("translog truncated: " + e.getMessage(),
167                     e.getMessage().contains("reached premature end of file, translog is truncated"), equalTo(true));
168         }
