444     @Test
445     public void testHasParentFilter() throws Exception {
446         assertAcked(prepareCreate("test")
447                 .addMapping("parent")
448                 .addMapping("child", "_parent", "type=parent"));
449         ensureGreen();
450         Map<String, Set<String>> parentToChildren = newHashMap();
451         // Childless parent
452         client().prepareIndex("test", "parent", "p0").setSource("p_field", "p0").get();
453         parentToChildren.put("p0", new HashSet<String>());
454 
455         String previousParentId = null;
456         int numChildDocs = 32;
457         int numChildDocsPerParent = 0;
458         List<IndexRequestBuilder> builders = new ArrayList<>();
459         for (int i = 1; i <= numChildDocs; i++) {
460 
461             if (previousParentId == null || i % numChildDocsPerParent == 0) {
462                 previousParentId = "p" + i;
463                 builders.add(client().prepareIndex("test", "parent", previousParentId).setSource("p_field", previousParentId));
464                 numChildDocsPerParent++;
465             }
466 
467             String childId = "c" + i;
468             builders.add(client().prepareIndex("test", "child", childId).setSource("c_field", childId).setParent(previousParentId));
469 
470             if (!parentToChildren.containsKey(previousParentId)) {
471                 parentToChildren.put(previousParentId, new HashSet<String>());
472             }
473             assertThat(parentToChildren.get(previousParentId).add(childId), is(true));
474         }
475         indexRandom(true, builders.toArray(new IndexRequestBuilder[builders.size()]));
476 
477         assertThat(parentToChildren.isEmpty(), equalTo(false));
478         for (Map.Entry<String, Set<String>> parentToChildrenEntry : parentToChildren.entrySet()) {
479             SearchResponse searchResponse = client().prepareSearch("test")
480                     .setQuery(constantScoreQuery(hasParentFilter("parent", termQuery("p_field", parentToChildrenEntry.getKey()))))
