57      @Test
58      public void moveShardCommand() {
59          AllocationService allocation = createAllocationService(settingsBuilder().put("cluster.routing.allocation.concurrent_recoveries", 10).build());
60  
61          logger.info("creating an index with 1 shard, no replica");
62          MetaData metaData = MetaData.builder()
63                  .put(IndexMetaData.builder("test").settings(settings(Version.CURRENT)).numberOfShards(1).numberOfReplicas(0))
64                  .build();
65          RoutingTable routingTable = RoutingTable.builder()
66                  .addAsNew(metaData.index("test"))
67                  .build();
68          ClusterState clusterState = ClusterState.builder(org.elasticsearch.cluster.ClusterName.DEFAULT).metaData(metaData).routingTable(routingTable).build();
69  
70          logger.info("adding two nodes and performing rerouting");
71          clusterState = ClusterState.builder(clusterState).nodes(DiscoveryNodes.builder().put(newNode("node1")).put(newNode("node2"))).build();
72          RoutingAllocation.Result rerouteResult = allocation.reroute(clusterState);
73          clusterState = ClusterState.builder(clusterState).routingTable(rerouteResult.routingTable()).build();
74  
75          logger.info("start primary shard");
76          rerouteResult = allocation.applyStartedShards(clusterState, clusterState.routingNodes().shardsWithState(INITIALIZING));
77          clusterState = ClusterState.builder(clusterState).routingTable(rerouteResult.routingTable()).build();
78  
79          logger.info("move the shard");
80          String existingNodeId = clusterState.routingTable().index("test").shard(0).primaryShard().currentNodeId();
81          String toNodeId;
82          if ("node1".equals(existingNodeId)) {
83              toNodeId = "node2";
84          } else {
85              toNodeId = "node1";
86          }
87          rerouteResult = allocation.reroute(clusterState, new AllocationCommands(new MoveAllocationCommand(new ShardId("test", 0), existingNodeId, toNodeId)));
88          assertThat(rerouteResult.changed(), equalTo(true));
89          clusterState = ClusterState.builder(clusterState).routingTable(rerouteResult.routingTable()).build();
90          assertThat(clusterState.routingNodes().node(existingNodeId).get(0).state(), equalTo(ShardRoutingState.RELOCATING));
91          assertThat(clusterState.routingNodes().node(toNodeId).get(0).state(), equalTo(ShardRoutingState.INITIALIZING));
92  
93          logger.info("finish moving the shard");
94          rerouteResult = allocation.applyStartedShards(clusterState, clusterState.routingNodes().shardsWithState(INITIALIZING));
95          clusterState = ClusterState.builder(clusterState).routingTable(rerouteResult.routingTable()).build();
96  
97          assertThat(clusterState.routingNodes().node(existingNodeId).isEmpty(), equalTo(true));
98          assertThat(clusterState.routingNodes().node(toNodeId).get(0).state(), equalTo(ShardRoutingState.STARTED));
99      }

102     public void allocateCommand() {
103         AllocationService allocation = createAllocationService(settingsBuilder()
104                 .put(DisableAllocationDecider.CLUSTER_ROUTING_ALLOCATION_DISABLE_NEW_ALLOCATION, true)
105                 .put(DisableAllocationDecider.CLUSTER_ROUTING_ALLOCATION_DISABLE_ALLOCATION, true)
106                 .build());
107 
108         logger.info("--> building initial routing table");
109         MetaData metaData = MetaData.builder()
110                 .put(IndexMetaData.builder("test").settings(settings(Version.CURRENT)).numberOfShards(1).numberOfReplicas(1))
111                 .build();
112         RoutingTable routingTable = RoutingTable.builder()
113                 .addAsNew(metaData.index("test"))
114                 .build();
115         ClusterState clusterState = ClusterState.builder(org.elasticsearch.cluster.ClusterName.DEFAULT).metaData(metaData).routingTable(routingTable).build();
116 
117         logger.info("--> adding 3 nodes on same rack and do rerouting");
118         clusterState = ClusterState.builder(clusterState).nodes(DiscoveryNodes.builder()
119                 .put(newNode("node1"))
120                 .put(newNode("node2"))
121                 .put(newNode("node3"))
122                 .put(newNode("node4", ImmutableMap.of("data", Boolean.FALSE.toString())))
123         ).build();
124         RoutingAllocation.Result rerouteResult = allocation.reroute(clusterState);
125         clusterState = ClusterState.builder(clusterState).routingTable(rerouteResult.routingTable()).build();
126         assertThat(clusterState.routingNodes().shardsWithState(INITIALIZING).size(), equalTo(0));
127 
128         logger.info("--> allocating with primary flag set to false, should fail");
129         try {
130             allocation.reroute(clusterState, new AllocationCommands(new AllocateAllocationCommand(new ShardId("test", 0), "node1", false)));
131             fail();
132         } catch (ElasticsearchIllegalArgumentException e) {
133         }
134 
135         logger.info("--> allocating to non-data node, should fail");
136         try {
137             rerouteResult = allocation.reroute(clusterState, new AllocationCommands(new AllocateAllocationCommand(new ShardId("test", 0), "node4", true)));
138             fail();
139         } catch (ElasticsearchIllegalArgumentException e) {
140         }
141 
142         logger.info("--> allocating with primary flag set to true");
143         rerouteResult = allocation.reroute(clusterState, new AllocationCommands(new AllocateAllocationCommand(new ShardId("test", 0), "node1", true)));
144         assertThat(rerouteResult.changed(), equalTo(true));
145         clusterState = ClusterState.builder(clusterState).routingTable(rerouteResult.routingTable()).build();
146         assertThat(clusterState.routingNodes().node("node1").size(), equalTo(1));
147         assertThat(clusterState.routingNodes().node("node1").shardsWithState(INITIALIZING).size(), equalTo(1));
148         assertThat(clusterState.routingNodes().node("node2").size(), equalTo(0));
149 
150         logger.info("--> start the primary shard");
151         rerouteResult = allocation.applyStartedShards(clusterState, clusterState.routingNodes().shardsWithState(INITIALIZING));
152         clusterState = ClusterState.builder(clusterState).routingTable(rerouteResult.routingTable()).build();
153         assertThat(clusterState.routingNodes().node("node1").size(), equalTo(1));
154         assertThat(clusterState.routingNodes().node("node1").shardsWithState(STARTED).size(), equalTo(1));
155         assertThat(clusterState.routingNodes().node("node2").size(), equalTo(0));
156 
157         logger.info("--> allocate the replica shard on the primary shard node, should fail");
158         try {
159             allocation.reroute(clusterState, new AllocationCommands(new AllocateAllocationCommand(new ShardId("test", 0), "node1", false)));
160             fail();
161         } catch (ElasticsearchIllegalArgumentException e) {
162         }
163 
164         logger.info("--> allocate the replica shard on on the second node");
165         rerouteResult = allocation.reroute(clusterState, new AllocationCommands(new AllocateAllocationCommand(new ShardId("test", 0), "node2", false)));
166         assertThat(rerouteResult.changed(), equalTo(true));
167         clusterState = ClusterState.builder(clusterState).routingTable(rerouteResult.routingTable()).build();
168         assertThat(clusterState.routingNodes().node("node1").size(), equalTo(1));
169         assertThat(clusterState.routingNodes().node("node1").shardsWithState(STARTED).size(), equalTo(1));
170         assertThat(clusterState.routingNodes().node("node2").size(), equalTo(1));
171         assertThat(clusterState.routingNodes().node("node2").shardsWithState(INITIALIZING).size(), equalTo(1));
172 
173 
174         logger.info("--> start the replica shard");
175         rerouteResult = allocation.applyStartedShards(clusterState, clusterState.routingNodes().shardsWithState(INITIALIZING));
176         clusterState = ClusterState.builder(clusterState).routingTable(rerouteResult.routingTable()).build();
177         assertThat(clusterState.routingNodes().node("node1").size(), equalTo(1));
178         assertThat(clusterState.routingNodes().node("node1").shardsWithState(STARTED).size(), equalTo(1));
179         assertThat(clusterState.routingNodes().node("node2").size(), equalTo(1));
180         assertThat(clusterState.routingNodes().node("node2").shardsWithState(STARTED).size(), equalTo(1));
181 
182         logger.info("--> verify that we fail when there are no unassigned shards");
183         try {
184             allocation.reroute(clusterState, new AllocationCommands(new AllocateAllocationCommand(new ShardId("test", 0), "node3", false)));
185             fail();
186         } catch (ElasticsearchIllegalArgumentException e) {
187         }
188     }

340     @Test
341     public void serialization() throws Exception {
342         AllocationCommands commands = new AllocationCommands(
343                 new AllocateAllocationCommand(new ShardId("test", 1), "node1", true),
344                 new MoveAllocationCommand(new ShardId("test", 3), "node2", "node3"),
345                 new CancelAllocationCommand(new ShardId("test", 4), "node5", true)
346         );
347         BytesStreamOutput bytes = new BytesStreamOutput();
348         AllocationCommands.writeTo(commands, bytes);
349         AllocationCommands sCommands = AllocationCommands.readFrom(new BytesStreamInput(bytes.bytes()));
350 
351         assertThat(sCommands.commands().size(), equalTo(3));
352         assertThat(((AllocateAllocationCommand) (sCommands.commands().get(0))).shardId(), equalTo(new ShardId("test", 1)));
353         assertThat(((AllocateAllocationCommand) (sCommands.commands().get(0))).node(), equalTo("node1"));
354         assertThat(((AllocateAllocationCommand) (sCommands.commands().get(0))).allowPrimary(), equalTo(true));
355 
356         assertThat(((MoveAllocationCommand) (sCommands.commands().get(1))).shardId(), equalTo(new ShardId("test", 3)));
357         assertThat(((MoveAllocationCommand) (sCommands.commands().get(1))).fromNode(), equalTo("node2"));
358         assertThat(((MoveAllocationCommand) (sCommands.commands().get(1))).toNode(), equalTo("node3"));
359 
360         assertThat(((CancelAllocationCommand) (sCommands.commands().get(2))).shardId(), equalTo(new ShardId("test", 4)));
361         assertThat(((CancelAllocationCommand) (sCommands.commands().get(2))).node(), equalTo("node5"));
362         assertThat(((CancelAllocationCommand) (sCommands.commands().get(2))).allowPrimary(), equalTo(true));
363     }

365     @Test

368                 "    \"commands\" : [\n" +
369                 "        {\"allocate\" : {\"index\" : \"test\", \"shard\" : 1, \"node\" : \"node1\", \"allow_primary\" : true}}\n" +
370                 "       ,{\"move\" : {\"index\" : \"test\", \"shard\" : 3, \"from_node\" : \"node2\", \"to_node\" : \"node3\"}} \n" +
371                 "       ,{\"cancel\" : {\"index\" : \"test\", \"shard\" : 4, \"node\" : \"node5\", \"allow_primary\" : true}} \n" +
372                 "    ]\n" +
373                 "}\n";
374         XContentParser parser = XContentFactory.xContent(XContentType.JSON).createParser(commands);
375         // move two tokens, parser expected to be "on" `commands` field
376         parser.nextToken();
377         parser.nextToken();
378         AllocationCommands sCommands = AllocationCommands.fromXContent(parser);
379 
380         assertThat(sCommands.commands().size(), equalTo(3));
381         assertThat(((AllocateAllocationCommand) (sCommands.commands().get(0))).shardId(), equalTo(new ShardId("test", 1)));
382         assertThat(((AllocateAllocationCommand) (sCommands.commands().get(0))).node(), equalTo("node1"));
383         assertThat(((AllocateAllocationCommand) (sCommands.commands().get(0))).allowPrimary(), equalTo(true));
384 
385         assertThat(((MoveAllocationCommand) (sCommands.commands().get(1))).shardId(), equalTo(new ShardId("test", 3)));
386         assertThat(((MoveAllocationCommand) (sCommands.commands().get(1))).fromNode(), equalTo("node2"));
387         assertThat(((MoveAllocationCommand) (sCommands.commands().get(1))).toNode(), equalTo("node3"));
