109     public void setUp() throws Exception {
110         super.setUp();
111         CodecService codecService = new CodecService(shardId.index());
112         indexConcurrency = randomIntBetween(1, 20);
113         String name = Codec.getDefault().getName();
114         if (Arrays.asList(codecService.availableCodecs()).contains(name)) {
115             // some codecs are read only so we only take the ones that we have in the service and randomly
116             // selected by lucene test case.
117             codecName = name;
118         } else {
119             codecName = "default";
120         }
121         defaultSettings = ImmutableSettings.builder()
122                 .put(EngineConfig.INDEX_COMPOUND_ON_FLUSH, randomBoolean())
123                 .put(EngineConfig.INDEX_GC_DELETES_SETTING, "1h") // make sure this doesn't kick in on us
124                 .put(EngineConfig.INDEX_FAIL_ON_CORRUPTION_SETTING, randomBoolean())
125                 .put(EngineConfig.INDEX_CODEC_SETTING, codecName)
126                 .put(EngineConfig.INDEX_CONCURRENCY_SETTING, indexConcurrency)
127                 .build(); // TODO randomize more settings
128         threadPool = new ThreadPool(getClass().getName());
129         dirPath = newTempDirPath(LifecycleScope.TEST);
130         store = createStore(dirPath);
131         storeReplica = createStore(dirPath);
132         Lucene.cleanLuceneIndex(store.directory());
133         Lucene.cleanLuceneIndex(storeReplica.directory());
134         translog = createTranslog();
135         primaryEngine = createInternalEngine(store, translog);
136         LiveIndexWriterConfig currentIndexWriterConfig = ((InternalEngine)primaryEngine).getCurrentIndexWriterConfig();
137 
138         assertEquals(primaryEngine.config().getCodec().getName(), codecService.codec(codecName).getName());
139         assertEquals(currentIndexWriterConfig.getCodec().getName(), codecService.codec(codecName).getName());
140         if (randomBoolean()) {
141             primaryEngine.config().setEnableGcDeletes(false);
142         }
143 
144         replicaTranslog = createTranslogReplica();
145         replicaEngine = createShadowEngine(storeReplica, replicaTranslog);
146 
147         assertEquals(replicaEngine.config().getCodec().getName(), codecService.codec(codecName).getName());
148         if (randomBoolean()) {
149             replicaEngine.config().setEnableGcDeletes(false);
150         }
151     }

478     @Test
479     public void testShadowEngineIgnoresWriteOperations() throws Exception {
480         // create a document
481         ParseContext.Document document = testDocumentWithTextField();
482         document.add(new Field(SourceFieldMapper.NAME, B_1.toBytes(), SourceFieldMapper.Defaults.FIELD_TYPE));
483         ParsedDocument doc = testParsedDocument("1", "1", "test", null, -1, -1, document, B_1, false);
484         try {
485             replicaEngine.create(new Engine.Create(null, newUid("1"), doc));
486             fail("should have thrown an exception");
487         } catch (UnsupportedOperationException e) {}
488         replicaEngine.refresh("test");
489 
490         // its not there...
491         Engine.Searcher searchResult = replicaEngine.acquireSearcher("test");
492         MatcherAssert.assertThat(searchResult, EngineSearcherTotalHitsMatcher.engineSearcherTotalHits(0));
493         MatcherAssert.assertThat(searchResult, EngineSearcherTotalHitsMatcher.engineSearcherTotalHits(new TermQuery(new Term("value", "test")), 0));
494         searchResult.close();
495         Engine.GetResult getResult = replicaEngine.get(new Engine.Get(true, newUid("1")));
496         assertThat(getResult.exists(), equalTo(false));
497         getResult.release();
498

880             MockDirectoryWrapper wrapper = newMockFSDirectory(dirPath);
881             wrapper.setFailOnOpenInput(randomBoolean());
882             wrapper.setAllowRandomFileNotFoundException(randomBoolean());
883             wrapper.setRandomIOExceptionRate(randomDouble());
884             wrapper.setRandomIOExceptionRateOnOpen(randomDouble());
885             try (Store store = createStore(wrapper)) {
886                 int refCount = store.refCount();
887                 assertTrue("refCount: "+ store.refCount(), store.refCount() > 0);
888                 Translog translog = createTranslog();
889                 ShadowEngine holder;
890                 try {
891                     holder = createShadowEngine(store, translog);
892                 } catch (EngineCreationFailureException ex) {
893                     assertEquals(store.refCount(), refCount);
894                     continue;
895                 }
896                 holder.config().setFailEngineOnCorruption(true);
897                 assertEquals(store.refCount(), refCount+1);
898                 final int numStarts = scaledRandomIntBetween(1, 5);
899                 for (int j = 0; j < numStarts; j++) {
900                     try {
901                         assertEquals(store.refCount(), refCount + 1);
902                         holder.close();
903                         holder = createShadowEngine(store, translog);
904                         holder.config().setFailEngineOnCorruption(true);
905                         assertEquals(store.refCount(), refCount + 1);
906                     } catch (EngineCreationFailureException ex) {
907                         // all is fine
908                         assertEquals(store.refCount(), refCount);
909                         break;
910                     }
911                 }
912                 translog.close();
913                 holder.close();
914                 assertEquals(store.refCount(), refCount);
915             }
