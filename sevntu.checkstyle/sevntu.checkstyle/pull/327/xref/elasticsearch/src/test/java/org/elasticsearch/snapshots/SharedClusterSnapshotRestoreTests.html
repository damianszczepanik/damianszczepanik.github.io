81      public void basicWorkFlowTest() throws Exception {
82          Client client = client();
83  
84          logger.info("-->  creating repository");
85          assertAcked(client.admin().cluster().preparePutRepository("test-repo")
86                  .setType("fs").setSettings(ImmutableSettings.settingsBuilder()
87                          .put("location", newTempDirPath())
88                          .put("compress", randomBoolean())
89                          .put("chunk_size", randomIntBetween(100, 1000))));
90  
91          createIndex("test-idx-1", "test-idx-2", "test-idx-3");
92          ensureGreen();
93  
94          logger.info("--> indexing some data");
95          for (int i = 0; i < 100; i++) {
96              index("test-idx-1", "doc", Integer.toString(i), "foo", "bar" + i);
97              index("test-idx-2", "doc", Integer.toString(i), "foo", "baz" + i);
98              index("test-idx-3", "doc", Integer.toString(i), "foo", "baz" + i);
99          }
100         refresh();
101         assertThat(client.prepareCount("test-idx-1").get().getCount(), equalTo(100L));
102         assertThat(client.prepareCount("test-idx-2").get().getCount(), equalTo(100L));
103         assertThat(client.prepareCount("test-idx-3").get().getCount(), equalTo(100L));
104

106         if (randomBoolean()) {
107             ArrayList<String> indicesToFlush = newArrayList();
108             for (int i = 1; i < 4; i++) {
109                 if (randomBoolean()) {
110                     indicesToFlush.add("test-idx-" + i);
111                 }
112             }
113             if (!indicesToFlush.isEmpty()) {
114                 String[] indices = indicesToFlush.toArray(new String[indicesToFlush.size()]);
115                 logger.info("--> starting asynchronous flush for indices {}", Arrays.toString(indices));

117             }
118         }
119         logger.info("--> snapshot");
120         CreateSnapshotResponse createSnapshotResponse = client.admin().cluster().prepareCreateSnapshot("test-repo", "test-snap").setWaitForCompletion(true).setIndices("test-idx-*", "-test-idx-3").get();

122         assertThat(createSnapshotResponse.getSnapshotInfo().successfulShards(), equalTo(createSnapshotResponse.getSnapshotInfo().totalShards()));
123 
124         assertThat(client.admin().cluster().prepareGetSnapshots("test-repo").setSnapshots("test-snap").get().getSnapshots().get(0).state(), equalTo(SnapshotState.SUCCESS));
125 
126         logger.info("--> delete some data");
127         for (int i = 0; i < 50; i++) {
128             client.prepareDelete("test-idx-1", "doc", Integer.toString(i)).get();
129         }
130         for (int i = 50; i < 100; i++) {
131             client.prepareDelete("test-idx-2", "doc", Integer.toString(i)).get();
132         }
133         for (int i = 0; i < 100; i += 2) {
134             client.prepareDelete("test-idx-3", "doc", Integer.toString(i)).get();

138         assertThat(client.prepareCount("test-idx-2").get().getCount(), equalTo(50L));
139         assertThat(client.prepareCount("test-idx-3").get().getCount(), equalTo(50L));
140 
141         logger.info("--> close indices");
142         client.admin().indices().prepareClose("test-idx-1", "test-idx-2").get();
143 
144         logger.info("--> restore all indices from the snapshot");
145         RestoreSnapshotResponse restoreSnapshotResponse = client.admin().cluster().prepareRestoreSnapshot("test-repo", "test-snap").setWaitForCompletion(true).execute().actionGet();
146         assertThat(restoreSnapshotResponse.getRestoreInfo().totalShards(), greaterThan(0));
147 
148         for (int i=0; i<5; i++) {
149             assertThat(client.prepareCount("test-idx-1").get().getCount(), equalTo(100L));
150             assertThat(client.prepareCount("test-idx-2").get().getCount(), equalTo(100L));

152         }
153

157         logger.info("--> restore one index after deletion");
158         restoreSnapshotResponse = client.admin().cluster().prepareRestoreSnapshot("test-repo", "test-snap").setWaitForCompletion(true).setIndices("test-idx-*", "-test-idx-2").execute().actionGet();
159         assertThat(restoreSnapshotResponse.getRestoreInfo().totalShards(), greaterThan(0));
160 
161         for (int i=0; i<5; i++) {
162             assertThat(client.prepareCount("test-idx-1").get().getCount(), equalTo(100L));
163         }
164         ClusterState clusterState = client.admin().cluster().prepareState().get().getState();
165         assertThat(clusterState.getMetaData().hasIndex("test-idx-1"), equalTo(true));
166         assertThat(clusterState.getMetaData().hasIndex("test-idx-2"), equalTo(false));
167 
168         if (flushResponseFuture != null) {
169             // Finish flush
170             flushResponseFuture.actionGet();
171         }
172     }
