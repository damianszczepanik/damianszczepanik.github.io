41      @Test
42      public void simpleBoundingBoxTest() throws Exception {
43          XContentBuilder xContentBuilder = XContentFactory.jsonBuilder().startObject().startObject("type1")
44                  .startObject("properties").startObject("location").field("type", "geo_point").field("lat_lon", true).endObject().endObject()
45                  .endObject().endObject();
46          assertAcked(prepareCreate("test").addMapping("type1", xContentBuilder));
47          ensureGreen();
48  
49          client().prepareIndex("test", "type1", "1").setSource(jsonBuilder().startObject()
50                  .field("name", "New York")
51                  .startObject("location").field("lat", 40.7143528).field("lon", -74.0059731).endObject()
52                  .endObject()).execute().actionGet();
53  
54          // to NY: 5.286 km
55          client().prepareIndex("test", "type1", "2").setSource(jsonBuilder().startObject()
56                  .field("name", "Times Square")
57                  .startObject("location").field("lat", 40.759011).field("lon", -73.9844722).endObject()
58                  .endObject()).execute().actionGet();
59  
60          // to NY: 0.4621 km
61          client().prepareIndex("test", "type1", "3").setSource(jsonBuilder().startObject()
62                  .field("name", "Tribeca")
63                  .startObject("location").field("lat", 40.718266).field("lon", -74.007819).endObject()
64                  .endObject()).execute().actionGet();
65  
66          // to NY: 1.055 km
67          client().prepareIndex("test", "type1", "4").setSource(jsonBuilder().startObject()
68                  .field("name", "Wall Street")
69                  .startObject("location").field("lat", 40.7051157).field("lon", -74.0088305).endObject()
70                  .endObject()).execute().actionGet();
71  
72          // to NY: 1.258 km
73          client().prepareIndex("test", "type1", "5").setSource(jsonBuilder().startObject()
74                  .field("name", "Soho")
75                  .startObject("location").field("lat", 40.7247222).field("lon", -74).endObject()
76                  .endObject()).execute().actionGet();
77  
78          // to NY: 2.029 km
79          client().prepareIndex("test", "type1", "6").setSource(jsonBuilder().startObject()
80                  .field("name", "Greenwich Village")
81                  .startObject("location").field("lat", 40.731033).field("lon", -73.9962255).endObject()
82                  .endObject()).execute().actionGet();
83  
84          // to NY: 8.572 km
85          client().prepareIndex("test", "type1", "7").setSource(jsonBuilder().startObject()
86                  .field("name", "Brooklyn")
87                  .startObject("location").field("lat", 40.65).field("lon", -73.95).endObject()
88                  .endObject()).execute().actionGet();
89  
90          client().admin().indices().prepareRefresh().execute().actionGet();
91  
92          SearchResponse searchResponse = client().prepareSearch() // from NY
93                  .setQuery(filteredQuery(matchAllQuery(), geoBoundingBoxFilter("location").topLeft(40.73, -74.1).bottomRight(40.717, -73.99)))
94                  .execute().actionGet();
95          assertThat(searchResponse.getHits().getTotalHits(), equalTo(2l));
96          assertThat(searchResponse.getHits().hits().length, equalTo(2));
97          for (SearchHit hit : searchResponse.getHits()) {
98              assertThat(hit.id(), anyOf(equalTo("1"), equalTo("3"), equalTo("5")));
99          }
100 
101         searchResponse = client().prepareSearch() // from NY
102                 .setQuery(filteredQuery(matchAllQuery(), geoBoundingBoxFilter("location").topLeft(40.73, -74.1).bottomRight(40.717, -73.99).type("indexed")))
103                 .execute().actionGet();
104         assertThat(searchResponse.getHits().getTotalHits(), equalTo(2l));
105         assertThat(searchResponse.getHits().hits().length, equalTo(2));
106         for (SearchHit hit : searchResponse.getHits()) {
107             assertThat(hit.id(), anyOf(equalTo("1"), equalTo("3"), equalTo("5")));
108         }
109     }
