61      @Test
62      public void testMarvelHeros() throws IOException {
63          RAMDirectory dir = new RAMDirectory();
64          Map<String, Analyzer> mapping = new HashMap<>();
65          mapping.put("body_ngram", new Analyzer() {
66  
67              @Override
68              protected TokenStreamComponents createComponents(String fieldName) {
69                  Tokenizer t = new StandardTokenizer();
70                  ShingleFilter tf = new ShingleFilter(t, 2, 3);
71                  tf.setOutputUnigrams(false);
72                  return new TokenStreamComponents(t, new LowerCaseFilter(tf));
73              }
74  
75          });
76  
77          mapping.put("body", new Analyzer() {
78  
79              @Override
80              protected TokenStreamComponents createComponents(String fieldName) {
81                  Tokenizer t = new StandardTokenizer();
82                  return new TokenStreamComponents(t, new LowerCaseFilter(t));
83              }
84  
85          });
86          PerFieldAnalyzerWrapper wrapper = new PerFieldAnalyzerWrapper(new WhitespaceAnalyzer(), mapping);
87  
88          IndexWriterConfig conf = new IndexWriterConfig(wrapper);
89          IndexWriter writer = new IndexWriter(dir, conf);
90          BufferedReader reader = new BufferedReader(new InputStreamReader(NoisyChannelSpellCheckerTests.class.getResourceAsStream("/config/names.txt"), Charsets.UTF_8));
91          String line = null;
92          while ((line = reader.readLine()) != null) {
93              Document doc = new Document();
94              doc.add(new Field("body", line, TextField.TYPE_NOT_STORED));
95              doc.add(new Field("body_ngram", line, TextField.TYPE_NOT_STORED));
96              writer.addDocument(doc);
97          }
98  
99          DirectoryReader ir = DirectoryReader.open(writer, false);
100         WordScorer wordScorer = new LaplaceScorer(ir, MultiFields.getTerms(ir, "body_ngram"), "body_ngram", 0.95d, new BytesRef(" "), 0.5f);
101         
102         NoisyChannelSpellChecker suggester = new NoisyChannelSpellChecker();
103         DirectSpellChecker spellchecker = new DirectSpellChecker();
104         spellchecker.setMinQueryLength(1);
105         DirectCandidateGenerator generator = new DirectCandidateGenerator(spellchecker, "body", SuggestMode.SUGGEST_MORE_POPULAR, ir, 0.95, 5);
106         Result result = suggester.getCorrections(wrapper, new BytesRef("american ame"), generator, 1, 1, ir, "body", wordScorer, 1, 2);
107         Correction[] corrections = result.corrections;
108         assertThat(corrections.length, equalTo(1));
109         assertThat(corrections[0].join(space).utf8ToString(), equalTo("american ace"));
110         assertThat(corrections[0].join(space, preTag, postTag).utf8ToString(), equalTo("american <em>ace</em>"));
111         assertThat(result.cutoffScore, greaterThan(0d));
112         
113         result = suggester.getCorrections(wrapper, new BytesRef("american ame"), generator, 1, 1, ir, "body", wordScorer, 0, 1);
114         corrections = result.corrections;
115         assertThat(corrections.length, equalTo(1));
116         assertThat(corrections[0].join(space).utf8ToString(), equalTo("american ame"));
117         assertThat(corrections[0].join(space, preTag, postTag).utf8ToString(), equalTo("american ame"));
118         assertThat(result.cutoffScore, equalTo(Double.MIN_VALUE));
119 
120         suggester = new NoisyChannelSpellChecker(0.85);
121         wordScorer = new LaplaceScorer(ir, MultiFields.getTerms(ir, "body_ngram"), "body_ngram", 0.85d, new BytesRef(" "), 0.5f);
122         corrections = suggester.getCorrections(wrapper, new BytesRef("Xor the Got-Jewel"), generator, 0.5f, 4, ir, "body", wordScorer, 0, 2).corrections;
123         assertThat(corrections.length, equalTo(4));
124         assertThat(corrections[0].join(space).utf8ToString(), equalTo("xorr the god jewel"));
125         assertThat(corrections[1].join(space).utf8ToString(), equalTo("xor the god jewel"));
126         assertThat(corrections[2].join(space).utf8ToString(), equalTo("xorn the god jewel"));
127         assertThat(corrections[3].join(space).utf8ToString(), equalTo("xorr the got jewel"));
128         assertThat(corrections[0].join(space, preTag, postTag).utf8ToString(), equalTo("<em>xorr</em> the <em>god</em> jewel"));
129         assertThat(corrections[1].join(space, preTag, postTag).utf8ToString(), equalTo("xor the <em>god</em> jewel"));
130         assertThat(corrections[2].join(space, preTag, postTag).utf8ToString(), equalTo("<em>xorn</em> the <em>god</em> jewel"));
131         assertThat(corrections[3].join(space, preTag, postTag).utf8ToString(), equalTo("<em>xorr</em> the got jewel"));
132         
133         corrections = suggester.getCorrections(wrapper, new BytesRef("Xor the Got-Jewel"), generator, 0.5f, 4, ir, "body", wordScorer, 1, 2).corrections;
134         assertThat(corrections.length, equalTo(4));
135         assertThat(corrections[0].join(space).utf8ToString(), equalTo("xorr the god jewel"));
136         assertThat(corrections[1].join(space).utf8ToString(), equalTo("xor the god jewel"));
137         assertThat(corrections[2].join(space).utf8ToString(), equalTo("xorn the god jewel"));
138         assertThat(corrections[3].join(space).utf8ToString(), equalTo("xorr the got jewel"));

147         assertThat(corrections[2].join(space).utf8ToString(), equalTo("xorn the god jewel"));
148         assertThat(corrections[3].join(space).utf8ToString(), equalTo("xor teh god jewel"));
149         assertThat(corrections[0].join(space, preTag, postTag).utf8ToString(), equalTo("<em>xorr the god</em> jewel"));
150         assertThat(corrections[1].join(space, preTag, postTag).utf8ToString(), equalTo("xor <em>the god</em> jewel"));
151         assertThat(corrections[2].join(space, preTag, postTag).utf8ToString(), equalTo("<em>xorn the god</em> jewel"));
152         assertThat(corrections[3].join(space, preTag, postTag).utf8ToString(), equalTo("xor teh <em>god</em> jewel"));

154         // test synonyms
155         
156         Analyzer analyzer = new Analyzer() {
157 
158             @Override
159             protected TokenStreamComponents createComponents(String fieldName) {
160                 Tokenizer t = new StandardTokenizer();
161                 TokenFilter filter = new LowerCaseFilter(t);
162                 try {
163                     SolrSynonymParser parser = new SolrSynonymParser(true, false, new WhitespaceAnalyzer());
164                     ((SolrSynonymParser) parser).parse(new StringReader("usa => usa, america, american\nursa => usa, america, american"));
165                     filter = new SynonymFilter(filter, parser.build(), true);
166                 } catch (Exception e) {
167                     throw new RuntimeException(e);
168                 }
169                 return new TokenStreamComponents(t, filter);
170             }
171         };
172         
173         spellchecker.setAccuracy(0.0f);
174         spellchecker.setMinPrefix(1);
175         spellchecker.setMinQueryLength(1);
176         suggester = new NoisyChannelSpellChecker(0.85);
177         wordScorer = new LaplaceScorer(ir, MultiFields.getTerms(ir, "body_ngram"), "body_ngram", 0.85d, new BytesRef(" "), 0.5f);
178         corrections = suggester.getCorrections(analyzer, new BytesRef("captian usa"), generator, 2, 4, ir, "body", wordScorer, 1, 2).corrections;
179         assertThat(corrections[0].join(space).utf8ToString(), equalTo("captain america"));
180         assertThat(corrections[0].join(space, preTag, postTag).utf8ToString(), equalTo("<em>captain america</em>"));
181         
182         generator = new DirectCandidateGenerator(spellchecker, "body", SuggestMode.SUGGEST_MORE_POPULAR, ir, 0.85, 10, null, analyzer, MultiFields.getTerms(ir, "body"));
183         corrections = suggester.getCorrections(analyzer, new BytesRef("captian usw"), generator, 2, 4, ir, "body", wordScorer, 1, 2).corrections;
184         assertThat(corrections[0].join(new BytesRef(" ")).utf8ToString(), equalTo("captain america"));
185         assertThat(corrections[0].join(space, preTag, postTag).utf8ToString(), equalTo("<em>captain america</em>"));
186 
187         // Make sure that user supplied text is not marked as highlighted in the presence of a synonym filter
188         generator = new DirectCandidateGenerator(spellchecker, "body", SuggestMode.SUGGEST_MORE_POPULAR, ir, 0.85, 10, null, analyzer, MultiFields.getTerms(ir, "body"));
189         corrections = suggester.getCorrections(analyzer, new BytesRef("captain usw"), generator, 2, 4, ir, "body", wordScorer, 1, 2).corrections;
190         assertThat(corrections[0].join(new BytesRef(" ")).utf8ToString(), equalTo("captain america"));
191         assertThat(corrections[0].join(space, preTag, postTag).utf8ToString(), equalTo("captain <em>america</em>"));
192     }

195     public void testMarvelHerosMultiGenerator() throws IOException {
196         RAMDirectory dir = new RAMDirectory();
197         Map<String, Analyzer> mapping = new HashMap<>();
198         mapping.put("body_ngram", new Analyzer() {
199 
200             @Override
201             protected TokenStreamComponents createComponents(String fieldName) {
202                 Tokenizer t = new StandardTokenizer();
203                 ShingleFilter tf = new ShingleFilter(t, 2, 3);
204                 tf.setOutputUnigrams(false);
205                 return new TokenStreamComponents(t, new LowerCaseFilter(tf));
206             }
207 
208         });
209 
210         mapping.put("body", new Analyzer() {
211 
212             @Override
213             protected TokenStreamComponents createComponents(String fieldName) {
214                 Tokenizer t = new StandardTokenizer();
215                 return new TokenStreamComponents(t, new LowerCaseFilter(t));
216             }
217 
218         });
219         mapping.put("body_reverse", new Analyzer() {
220 
221             @Override
222             protected TokenStreamComponents createComponents(String fieldName) {
223                 Tokenizer t = new StandardTokenizer();
224                 return new TokenStreamComponents(t, new ReverseStringFilter(new LowerCaseFilter(t)));
225             }
226 
227         });
228         PerFieldAnalyzerWrapper wrapper = new PerFieldAnalyzerWrapper(new WhitespaceAnalyzer(), mapping);
229 
230         IndexWriterConfig conf = new IndexWriterConfig(wrapper);
231         IndexWriter writer = new IndexWriter(dir, conf);
232         BufferedReader reader = new BufferedReader(new InputStreamReader(NoisyChannelSpellCheckerTests.class.getResourceAsStream("/config/names.txt"), Charsets.UTF_8));
233         String line = null;
234         while ((line = reader.readLine()) != null) {
235             Document doc = new Document();
236             doc.add(new Field("body", line, TextField.TYPE_NOT_STORED));
237             doc.add(new Field("body_reverse", line, TextField.TYPE_NOT_STORED));
238             doc.add(new Field("body_ngram", line, TextField.TYPE_NOT_STORED));
239             writer.addDocument(doc);
240         }
241 
242         DirectoryReader ir = DirectoryReader.open(writer, false);
243         LaplaceScorer wordScorer = new LaplaceScorer(ir, MultiFields.getTerms(ir, "body_ngram"), "body_ngram", 0.95d, new BytesRef(" "), 0.5f);
244         NoisyChannelSpellChecker suggester = new NoisyChannelSpellChecker();
245         DirectSpellChecker spellchecker = new DirectSpellChecker();
246         spellchecker.setMinQueryLength(1);
247         DirectCandidateGenerator forward = new DirectCandidateGenerator(spellchecker, "body", SuggestMode.SUGGEST_ALWAYS, ir, 0.95, 10);
248         DirectCandidateGenerator reverse = new DirectCandidateGenerator(spellchecker, "body_reverse", SuggestMode.SUGGEST_ALWAYS, ir, 0.95, 10, wrapper, wrapper,  MultiFields.getTerms(ir, "body_reverse"));
249         CandidateGenerator generator = new MultiCandidateGeneratorWrapper(10, forward, reverse);
250         
251         Correction[] corrections = suggester.getCorrections(wrapper, new BytesRef("american cae"), generator, 1, 1, ir, "body", wordScorer, 1, 2).corrections;
252         assertThat(corrections.length, equalTo(1));
253         assertThat(corrections[0].join(new BytesRef(" ")).utf8ToString(), equalTo("american ace"));
254         
255         generator = new MultiCandidateGeneratorWrapper(5, forward, reverse);
256         corrections = suggester.getCorrections(wrapper, new BytesRef("american ame"), generator, 1, 1, ir, "body", wordScorer, 1, 2).corrections;
257         assertThat(corrections.length, equalTo(1));
258         assertThat(corrections[0].join(new BytesRef(" ")).utf8ToString(), equalTo("american ace"));
259         
260         corrections = suggester.getCorrections(wrapper, new BytesRef("american cae"), forward, 1, 1, ir, "body", wordScorer, 1, 2).corrections;
261         assertThat(corrections.length, equalTo(0)); // only use forward with constant prefix
262         
263         corrections = suggester.getCorrections(wrapper, new BytesRef("america cae"), generator, 2, 1, ir, "body", wordScorer, 1, 2).corrections;
264         assertThat(corrections.length, equalTo(1));
265         assertThat(corrections[0].join(new BytesRef(" ")).utf8ToString(), equalTo("american ace"));
266         
267         corrections = suggester.getCorrections(wrapper, new BytesRef("Zorr the Got-Jewel"), generator, 0.5f, 4, ir, "body", wordScorer, 0, 2).corrections;
268         assertThat(corrections.length, equalTo(4));
269         assertThat(corrections[0].join(new BytesRef(" ")).utf8ToString(), equalTo("xorr the god jewel"));
270         assertThat(corrections[1].join(new BytesRef(" ")).utf8ToString(), equalTo("zorr the god jewel"));
271         assertThat(corrections[2].join(new BytesRef(" ")).utf8ToString(), equalTo("four the god jewel"));
272 
273 
274         corrections = suggester.getCorrections(wrapper, new BytesRef("Zorr the Got-Jewel"), generator, 0.5f, 1, ir, "body", wordScorer, 1.5f, 2).corrections;
275         assertThat(corrections.length, equalTo(1));
276         assertThat(corrections[0].join(new BytesRef(" ")).utf8ToString(), equalTo("xorr the god jewel"));
277         
278         corrections = suggester.getCorrections(wrapper, new BytesRef("Xor the Got-Jewel"), generator, 0.5f, 1, ir, "body", wordScorer, 1.5f, 2).corrections;
279         assertThat(corrections.length, equalTo(1));
280         assertThat(corrections[0].join(new BytesRef(" ")).utf8ToString(), equalTo("xorr the god jewel"));
281 
282     }
