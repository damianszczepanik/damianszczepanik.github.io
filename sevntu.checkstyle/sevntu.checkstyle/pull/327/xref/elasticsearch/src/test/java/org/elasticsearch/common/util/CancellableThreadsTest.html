19  package org.elasticsearch.common.util;
20  
21  import org.elasticsearch.common.util.CancellableThreads.Interruptable;
22  import org.elasticsearch.test.ElasticsearchTestCase;
23  import org.hamcrest.Matchers;
24  import org.junit.Test;
25  
26  import java.util.concurrent.CountDownLatch;
27  
28  public class CancellableThreadsTest extends ElasticsearchTestCase {
29  
30      private static class CustomException extends RuntimeException {
31  
32          public CustomException(String msg) {
33              super(msg);
34          }
35      }
36  
37      private class TestPlan {
38          public final int id;
39          public final boolean busySpin;
40          public final boolean exceptBeforeCancel;
41          public final boolean exitBeforeCancel;
42          public final boolean exceptAfterCancel;
43          public final boolean presetInterrupt;
44  
45          private TestPlan(int id) {
46              this.id = id;
47              this.busySpin = randomBoolean();
48              this.exceptBeforeCancel = randomBoolean();
49              this.exitBeforeCancel = randomBoolean();
50              this.exceptAfterCancel = randomBoolean();
51              this.presetInterrupt = randomBoolean();
52          }
53      }
54  
55  
56      @Test
57      public void testCancellableThreads() throws InterruptedException {
58          Thread[] threads = new Thread[randomIntBetween(3, 10)];
59          final TestPlan[] plans = new TestPlan[threads.length];
60          final Throwable[] throwables = new Throwable[threads.length];
61          final boolean[] interrupted = new boolean[threads.length];
62          final CancellableThreads cancellableThreads = new CancellableThreads();
63          final CountDownLatch readyForCancel = new CountDownLatch(threads.length);
64          for (int i = 0; i < threads.length; i++) {
65              final TestPlan plan = new TestPlan(i);
66              plans[i] = plan;
67              threads[i] = new Thread(new Runnable() {
68                  @Override
69                  public void run() {
70                      try {
71                          if (plan.presetInterrupt) {
72                              Thread.currentThread().interrupt();
73                          }
74                          cancellableThreads.execute(new Interruptable() {
75                              @Override
76                              public void run() throws InterruptedException {
77                                  assertFalse("interrupt thread should have been clear", Thread.currentThread().isInterrupted());
78                                  if (plan.exceptBeforeCancel) {
79                                      throw new CustomException("thread [" + plan.id + "] pre-cancel exception");
80                                  } else if (plan.exitBeforeCancel) {
81                                      return;
82                                  }
83                                  readyForCancel.countDown();
84                                  try {
85                                      if (plan.busySpin) {
86                                          while (!Thread.currentThread().isInterrupted()) {
87                                          }
88                                      } else {
89                                          Thread.sleep(50000);
90                                      }
91                                  } finally {
92                                      if (plan.exceptAfterCancel) {
93                                          throw new CustomException("thread [" + plan.id + "] post-cancel exception");
94                                      }
95                                  }
96                              }
97                          });
98                      } catch (Throwable t) {
99                          throwables[plan.id] = t;
100                     }
101                     if (plan.exceptBeforeCancel || plan.exitBeforeCancel) {
102                         // we have to mark we're ready now (actually done).
103                         readyForCancel.countDown();
104                     }
105                     interrupted[plan.id] = Thread.currentThread().isInterrupted();
106 
107                 }
108             });
109             threads[i].setDaemon(true);
110             threads[i].start();
111         }
112 
113         readyForCancel.await();
114         cancellableThreads.cancel("test");
115         for (Thread thread : threads) {
116             thread.join(20000);
117             assertFalse(thread.isAlive());
118         }
119         for (int i = 0; i < threads.length; i++) {
120             TestPlan plan = plans[i];
121             if (plan.exceptBeforeCancel) {
122                 assertThat(throwables[i], Matchers.instanceOf(CustomException.class));
123             } else if (plan.exitBeforeCancel) {
124                 assertNull(throwables[i]);
125             } else {
126                 // in all other cases, we expect a cancellation exception.
127                 assertThat(throwables[i], Matchers.instanceOf(CancellableThreads.ExecutionCancelledException.class));
128                 if (plan.exceptAfterCancel) {
129                     assertThat(throwables[i].getSuppressed(),
130                             Matchers.arrayContaining(
131                                     Matchers.instanceOf(CustomException.class)
132                             ));
133                 } else {
134                     assertThat(throwables[i].getSuppressed(), Matchers.emptyArray());
135                 }
136             }
137             assertThat(interrupted[plan.id], Matchers.equalTo(plan.presetInterrupt));
138         }
139     }
140
