50  /**
51   * A test that keep a singleton node started for all tests that can be used to get
52   * references to Guice injectors in unit tests.
53   */
54  @Ignore
55  public abstract class ElasticsearchSingleNodeTest extends ElasticsearchTestCase {
56  
57      private static Node NODE = null;
58  
59      private static void reset() {
60          assert NODE != null;
61          stopNode();
62          startNode();
63      }
64  
65      private static void startNode() {
66          assert NODE == null;
67          NODE = newNode();
68      }
69  
70      private static void stopNode() {
71          Node node = NODE;
72          NODE = null;
73          Releasables.close(node);
74      }
75  
76      static void cleanup(boolean resetNode) {
77          assertAcked(client().admin().indices().prepareDelete("*").get());
78          if (resetNode) {
79              reset();
80          }
81          MetaData metaData = client().admin().cluster().prepareState().get().getState().getMetaData();
82          assertThat("test leaves persistent cluster metadata behind: " + metaData.persistentSettings().getAsMap(),
83                  metaData.persistentSettings().getAsMap().size(), equalTo(0));
84          assertThat("test leaves transient cluster metadata behind: " + metaData.transientSettings().getAsMap(),
85                  metaData.transientSettings().getAsMap().size(), equalTo(0));
86      }
87  
88      @After
89      public void tearDown() throws Exception {
90          super.tearDown();
91          cleanup(resetNodeAfterTest());
92      }
93  
94      @BeforeClass
95      public static void setUpClass() throws Exception {
96          stopNode();
97          startNode();
98      }
99  
100     @AfterClass
101     public static void tearDownClass() {
102         stopNode();
103     }

105     /**
106      * This method returns <code>true</code> if the node that is used in the background should be reset
107      * after each test. This is useful if the test changes the cluster state metadata etc. The default is
108      * <code>false</code>.
109      */
110     protected boolean resetNodeAfterTest() {
111         return false;
112     }

114     private static Node newNode() {
115         Node build = NodeBuilder.nodeBuilder().local(true).data(true).settings(ImmutableSettings.builder()
116                 .put(ClusterName.SETTING, clusterName())
117                 .put("node.name", nodeName())
118                 .put(IndexMetaData.SETTING_NUMBER_OF_SHARDS, 1)
119                 .put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, 0)
120                 .put("script.disable_dynamic", false)
121                 .put(EsExecutors.PROCESSORS, 1) // limit the number of threads created
122                 .put("http.enabled", false)
123                 .put("config.ignore_system_properties", true) // make sure we get what we set :)
124         ).build();
125         build.start();
126         assertThat(DiscoveryNode.localNode(build.settings()), is(true));
127         return build;
128     }

194         CreateIndexRequestBuilder createIndexRequestBuilder = client().admin().indices().prepareCreate(index).setSettings(settings);

196             createIndexRequestBuilder.addMapping(type, mappings);
197         }
198         return createIndex(index, createIndexRequestBuilder);

201     protected static IndexService createIndex(String index, CreateIndexRequestBuilder createIndexRequestBuilder) {
202         assertAcked(createIndexRequestBuilder.get());
203         // Wait for the index to be allocated so that cluster state updates don't override
204         // changes that would have been done locally
205         ClusterHealthResponse health = client().admin().cluster()
206                 .health(Requests.clusterHealthRequest(index).waitForYellowStatus().waitForEvents(Priority.LANGUID).waitForRelocatingShards(0)).actionGet();
207         assertThat(health.getStatus(), lessThanOrEqualTo(ClusterHealthStatus.YELLOW));
208         assertThat("Cluster must be a single node cluster", health.getNumberOfDataNodes(), equalTo(1));
209         IndicesService instanceFromNode = getInstanceFromNode(IndicesService.class);
210         return instanceFromNode.indexServiceSafe(index);
211     }

237     /**
238      * Ensures the cluster has a green state via the cluster health API. This method will also wait for relocations.
239      * It is useful to ensure that all action on the cluster have finished and all shards that were currently relocating
240      * are now allocated and started.
241      *
242      * @param timeout time out value to set on {@link org.elasticsearch.action.admin.cluster.health.ClusterHealthRequest}
243      */
244     public ClusterHealthStatus ensureGreen(TimeValue timeout, String... indices) {
245         ClusterHealthResponse actionGet = client().admin().cluster()
246                 .health(Requests.clusterHealthRequest(indices).timeout(timeout).waitForGreenStatus().waitForEvents(Priority.LANGUID).waitForRelocatingShards(0)).actionGet();
247         if (actionGet.isTimedOut()) {
248             logger.info("ensureGreen timed out, cluster state:\n{}\n{}", client().admin().cluster().prepareState().get().getState().prettyPrint(), client().admin().cluster().preparePendingClusterTasks().get().prettyPrint());
249             assertThat("timed out waiting for green state", actionGet.isTimedOut(), equalTo(false));
250         }
251         assertThat(actionGet.getStatus(), equalTo(ClusterHealthStatus.GREEN));
252         logger.debug("indices {} are green", indices.length == 0 ? "[_all]" : indices);
253         return actionGet.getStatus();
254     }
