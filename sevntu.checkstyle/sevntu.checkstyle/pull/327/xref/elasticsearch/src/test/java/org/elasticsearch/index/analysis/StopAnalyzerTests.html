20  package org.elasticsearch.index.analysis;
21  
22  import org.elasticsearch.Version;
23  import org.elasticsearch.cluster.metadata.IndexMetaData;
24  import org.elasticsearch.common.inject.Injector;
25  import org.elasticsearch.common.inject.ModulesBuilder;
26  import org.elasticsearch.common.settings.Settings;
27  import org.elasticsearch.common.settings.SettingsModule;
28  import org.elasticsearch.env.Environment;
29  import org.elasticsearch.env.EnvironmentModule;
30  import org.elasticsearch.index.Index;
31  import org.elasticsearch.index.IndexNameModule;
32  import org.elasticsearch.index.settings.IndexSettingsModule;
33  import org.elasticsearch.indices.analysis.IndicesAnalysisModule;
34  import org.elasticsearch.indices.analysis.IndicesAnalysisService;
35  import org.elasticsearch.test.ElasticsearchTokenStreamTestCase;
36  import org.junit.Test;
37  
38  import static org.elasticsearch.common.settings.ImmutableSettings.settingsBuilder;
39  
40  public class StopAnalyzerTests extends ElasticsearchTokenStreamTestCase {
41  
42      @Test
43      public void testDefaultsCompoundAnalysis() throws Exception {
44          Index index = new Index("test");
45          Settings settings = settingsBuilder().loadFromClasspath("org/elasticsearch/index/analysis/stop.json").put(IndexMetaData.SETTING_VERSION_CREATED, Version.CURRENT).build();
46          Injector parentInjector = new ModulesBuilder().add(new SettingsModule(settings), new EnvironmentModule(new Environment(settings)), new IndicesAnalysisModule()).createInjector();
47          Injector injector = new ModulesBuilder().add(
48                  new IndexSettingsModule(index, settings),
49                  new IndexNameModule(index),
50                  new AnalysisModule(settings, parentInjector.getInstance(IndicesAnalysisService.class)))
51                  .createChildInjector(parentInjector);
52  
53          AnalysisService analysisService = injector.getInstance(AnalysisService.class);
54  
55          NamedAnalyzer analyzer1 = analysisService.analyzer("analyzer1");
56  
57          assertTokenStreamContents(analyzer1.tokenStream("test", "to be or not to be"), new String[0]);
58  
59          NamedAnalyzer analyzer2 = analysisService.analyzer("analyzer2");
60  
61          assertTokenStreamContents(analyzer2.tokenStream("test", "to be or not to be"), new String[0]);
62      }

64  }
