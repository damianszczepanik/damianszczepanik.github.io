192     @Test
193     public void testGeoCircle() {
194         double earthCircumference = 40075016.69;
195         Circle circle = ShapeBuilder.newCircleBuilder().center(0, 0).radius("100m").build();
196         assertEquals((360 * 100) / earthCircumference, circle.getRadius(), 0.00000001);
197         assertEquals((Point) new PointImpl(0, 0, ShapeBuilder.SPATIAL_CONTEXT), circle.getCenter());
198         circle = ShapeBuilder.newCircleBuilder().center(+180, 0).radius("100m").build();
199         assertEquals((360 * 100) / earthCircumference, circle.getRadius(), 0.00000001);
200         assertEquals((Point) new PointImpl(180, 0, ShapeBuilder.SPATIAL_CONTEXT), circle.getCenter());
201         circle = ShapeBuilder.newCircleBuilder().center(-180, 0).radius("100m").build();
202         assertEquals((360 * 100) / earthCircumference, circle.getRadius(), 0.00000001);
203         assertEquals((Point) new PointImpl(-180, 0, ShapeBuilder.SPATIAL_CONTEXT), circle.getCenter());
204         circle = ShapeBuilder.newCircleBuilder().center(0, 90).radius("100m").build();
205         assertEquals((360 * 100) / earthCircumference, circle.getRadius(), 0.00000001);
206         assertEquals((Point) new PointImpl(0, 90, ShapeBuilder.SPATIAL_CONTEXT), circle.getCenter());
207         circle = ShapeBuilder.newCircleBuilder().center(0, -90).radius("100m").build();
208         assertEquals((360 * 100) / earthCircumference, circle.getRadius(), 0.00000001);
209         assertEquals((Point) new PointImpl(0, -90, ShapeBuilder.SPATIAL_CONTEXT), circle.getCenter());
210         double randomLat = (randomDouble() * 180) - 90;
211         double randomLon = (randomDouble() * 360) - 180;
212         double randomRadius = randomIntBetween(1, (int) earthCircumference / 4);
213         circle = ShapeBuilder.newCircleBuilder().center(randomLon, randomLat).radius(randomRadius + "m").build();
214         assertEquals((360 * randomRadius) / earthCircumference, circle.getRadius(), 0.00000001);
215         assertEquals((Point) new PointImpl(randomLon, randomLat, ShapeBuilder.SPATIAL_CONTEXT), circle.getCenter());
216     }

242     @Test
243     public void testDatelineOGC() {
244         // tests that the following shape (defined in counterclockwise OGC order)
245         // https://gist.github.com/anonymous/7f1bb6d7e9cd72f5977c crosses the dateline
246         // expected results: 3 polygons, 1 with a hole
247 
248         // a giant c shape
249         PolygonBuilder builder = ShapeBuilder.newPolygon()
250             .point(174,0)
251             .point(-176,0)
252             .point(-176,3)
253             .point(177,3)
254             .point(177,5)
255             .point(-176,5)
256             .point(-176,8)
257             .point(174,8)
258             .point(174,0);
259 
260         // 3/4 of an embedded 'c', crossing dateline once
261         builder.hole()
262             .point(175, 1)
263             .point(175, 7)
264             .point(-178, 7)
265             .point(-178, 6)
266             .point(176, 6)
267             .point(176, 2)
268             .point(179, 2)
269             .point(179,1)
270             .point(175, 1);
271 
272         // embedded hole right of the dateline
273         builder.hole()
274             .point(-179, 1)
275             .point(-179, 2)
276             .point(-177, 2)
277             .point(-177,1)
278             .point(-179,1);
279 
280         Shape shape = builder.close().build();
281 
282         assertMultiPolygon(shape);
