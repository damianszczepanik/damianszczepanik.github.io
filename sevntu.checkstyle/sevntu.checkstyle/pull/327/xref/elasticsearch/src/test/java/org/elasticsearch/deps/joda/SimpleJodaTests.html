40      @Test
41      public void testMultiParsers() {
42          DateTimeFormatterBuilder builder = new DateTimeFormatterBuilder();
43          DateTimeParser[] parsers = new DateTimeParser[3];
44          parsers[0] = DateTimeFormat.forPattern("MM/dd/yyyy").withZone(DateTimeZone.UTC).getParser();
45          parsers[1] = DateTimeFormat.forPattern("MM-dd-yyyy").withZone(DateTimeZone.UTC).getParser();
46          parsers[2] = DateTimeFormat.forPattern("yyyy-MM-dd HH:mm:ss").withZone(DateTimeZone.UTC).getParser();
47          builder.append(DateTimeFormat.forPattern("MM/dd/yyyy").withZone(DateTimeZone.UTC).getPrinter(), parsers);
48  
49          DateTimeFormatter formatter = builder.toFormatter();
50  
51          formatter.parseMillis("2009-11-15 14:12:12");
52      }

73      @Test
74      public void testIsoDateFormatDateOptionalTimeUTC() {
75          DateTimeFormatter formatter = ISODateTimeFormat.dateOptionalTimeParser().withZone(DateTimeZone.UTC);
76          long millis = formatter.parseMillis("1970-01-01T00:00:00Z");
77          assertThat(millis, equalTo(0l));
78          millis = formatter.parseMillis("1970-01-01T00:00:00.001Z");
79          assertThat(millis, equalTo(1l));
80          millis = formatter.parseMillis("1970-01-01T00:00:00.1Z");
81          assertThat(millis, equalTo(100l));
82          millis = formatter.parseMillis("1970-01-01T00:00:00.1");
83          assertThat(millis, equalTo(100l));
84          millis = formatter.parseMillis("1970-01-01T00:00:00");
85          assertThat(millis, equalTo(0l));
86          millis = formatter.parseMillis("1970-01-01");
87          assertThat(millis, equalTo(0l));
88  
89          millis = formatter.parseMillis("1970");
90          assertThat(millis, equalTo(0l));
91  
92          try {
93              formatter.parseMillis("1970 kuku");
94              fail("formatting should fail");
95          } catch (IllegalArgumentException e) {
96              // all is well
97          }
98  
99          // test offset in format
100         millis = formatter.parseMillis("1970-01-01T00:00:00-02:00");
101         assertThat(millis, equalTo(TimeValue.timeValueHours(2).millis()));
102     }

176     private void expectInvalidPattern(String pattern, String errorMessage) {
177         try {
178             Joda.forPattern(pattern);
179             fail("Pattern " + pattern + " should have thrown an exception but did not");
180         } catch (IllegalArgumentException e) {
181             assertThat(e.getMessage(), containsString(errorMessage));
182         }
183     }

185     @Test
186     public void testRounding() {
187         long TIME = utcTimeInMillis("2009-02-03T01:01:01");
188         MutableDateTime time = new MutableDateTime(DateTimeZone.UTC);
189         time.setMillis(TIME);
190         assertThat(time.monthOfYear().roundFloor().toString(), equalTo("2009-02-01T00:00:00.000Z"));
191         time.setMillis(TIME);
192         assertThat(time.hourOfDay().roundFloor().toString(), equalTo("2009-02-03T01:00:00.000Z"));
193         time.setMillis(TIME);
194         assertThat(time.dayOfMonth().roundFloor().toString(), equalTo("2009-02-03T00:00:00.000Z"));
195     }

197     @Test
198     public void testRoundingSetOnTime() {
199         MutableDateTime time = new MutableDateTime(DateTimeZone.UTC);
200         time.setRounding(time.getChronology().monthOfYear(), MutableDateTime.ROUND_FLOOR);
201         time.setMillis(utcTimeInMillis("2009-02-03T01:01:01"));
202         assertThat(time.toString(), equalTo("2009-02-01T00:00:00.000Z"));
203         assertThat(time.getMillis(), equalTo(utcTimeInMillis("2009-02-01T00:00:00.000Z")));
204 
205         time.setMillis(utcTimeInMillis("2009-05-03T01:01:01"));
206         assertThat(time.toString(), equalTo("2009-05-01T00:00:00.000Z"));
207         assertThat(time.getMillis(), equalTo(utcTimeInMillis("2009-05-01T00:00:00.000Z")));
208 
209         time = new MutableDateTime(DateTimeZone.UTC);
210         time.setRounding(time.getChronology().dayOfMonth(), MutableDateTime.ROUND_FLOOR);
211         time.setMillis(utcTimeInMillis("2009-02-03T01:01:01"));
212         assertThat(time.toString(), equalTo("2009-02-03T00:00:00.000Z"));
213         assertThat(time.getMillis(), equalTo(utcTimeInMillis("2009-02-03T00:00:00.000Z")));
214 
215         time.setMillis(utcTimeInMillis("2009-02-02T23:01:01"));
216         assertThat(time.toString(), equalTo("2009-02-02T00:00:00.000Z"));
217         assertThat(time.getMillis(), equalTo(utcTimeInMillis("2009-02-02T00:00:00.000Z")));
218 
219         time = new MutableDateTime(DateTimeZone.UTC);
220         time.setRounding(time.getChronology().weekOfWeekyear(), MutableDateTime.ROUND_FLOOR);
221         time.setMillis(utcTimeInMillis("2011-05-05T01:01:01"));
222         assertThat(time.toString(), equalTo("2011-05-02T00:00:00.000Z"));
223         assertThat(time.getMillis(), equalTo(utcTimeInMillis("2011-05-02T00:00:00.000Z")));
224     }

226     @Test
227     public void testRoundingWithTimeZone() {
228         MutableDateTime time = new MutableDateTime(DateTimeZone.UTC);
229         time.setZone(DateTimeZone.forOffsetHours(-2));
230         time.setRounding(time.getChronology().dayOfMonth(), MutableDateTime.ROUND_FLOOR);
231 
232         MutableDateTime utcTime = new MutableDateTime(DateTimeZone.UTC);
233         utcTime.setRounding(utcTime.getChronology().dayOfMonth(), MutableDateTime.ROUND_FLOOR);
234 
235         time.setMillis(utcTimeInMillis("2009-02-03T01:01:01"));
236         utcTime.setMillis(utcTimeInMillis("2009-02-03T01:01:01"));
237 
238         assertThat(time.toString(), equalTo("2009-02-02T00:00:00.000-02:00"));
239         assertThat(utcTime.toString(), equalTo("2009-02-03T00:00:00.000Z"));
240         // the time is on the 2nd, and utcTime is on the 3rd, but, because time already encapsulates
241         // time zone, the millis diff is not 24, but 22 hours
242         assertThat(time.getMillis(), equalTo(utcTime.getMillis() - TimeValue.timeValueHours(22).millis()));
243 
244         time.setMillis(utcTimeInMillis("2009-02-04T01:01:01"));
245         utcTime.setMillis(utcTimeInMillis("2009-02-04T01:01:01"));
246         assertThat(time.toString(), equalTo("2009-02-03T00:00:00.000-02:00"));
247         assertThat(utcTime.toString(), equalTo("2009-02-04T00:00:00.000Z"));
248         assertThat(time.getMillis(), equalTo(utcTime.getMillis() - TimeValue.timeValueHours(22).millis()));
249     }

251     private long utcTimeInMillis(String time) {
252         return ISODateTimeFormat.dateOptionalTimeParser().withZone(DateTimeZone.UTC).parseMillis(time);
