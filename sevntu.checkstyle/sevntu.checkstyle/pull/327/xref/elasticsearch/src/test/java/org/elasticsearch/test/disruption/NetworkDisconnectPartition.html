19  package org.elasticsearch.test.disruption;
20  
21  import org.elasticsearch.cluster.node.DiscoveryNode;
22  import org.elasticsearch.common.unit.TimeValue;
23  import org.elasticsearch.test.transport.MockTransportService;
24  
25  import java.util.Random;
26  import java.util.Set;
27  
28  public class NetworkDisconnectPartition extends NetworkPartition {
29  
30  
31      public NetworkDisconnectPartition(Random random) {
32          super(random);
33      }
34  
35      public NetworkDisconnectPartition(String node1, String node2, Random random) {
36          super(node1, node2, random);
37      }
38  
39      public NetworkDisconnectPartition(Set<String> nodesSideOne, Set<String> nodesSideTwo, Random random) {
40          super(nodesSideOne, nodesSideTwo, random);
41      }
42  
43      @Override
44      protected String getPartitionDescription() {
45          return "disconnected";
46      }
47  
48      @Override
49      void applyDisruption(DiscoveryNode node1, MockTransportService transportService1,
50                           DiscoveryNode node2, MockTransportService transportService2) {
51          transportService1.addFailToSendNoConnectRule(node2);
52          transportService2.addFailToSendNoConnectRule(node1);
53      }
54  
55      @Override
56      public TimeValue expectedTimeToHeal() {
57          return TimeValue.timeValueSeconds(0);
58      }
59  }
