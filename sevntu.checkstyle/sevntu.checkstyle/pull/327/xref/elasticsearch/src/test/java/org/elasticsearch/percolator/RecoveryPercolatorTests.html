65      @Test
66      @Slow
67      public void testRestartNodePercolator1() throws Exception {
68          internalCluster().startNode();
69          assertAcked(prepareCreate("test").addMapping("type1", "field1", "type=string").addMapping(PercolatorService.TYPE_NAME, "color", "type=string"));
70  
71          logger.info("--> register a query");
72          client().prepareIndex("test", PercolatorService.TYPE_NAME, "kuku")
73                  .setSource(jsonBuilder().startObject()
74                          .field("color", "blue")
75                          .field("query", termQuery("field1", "value1"))
76                          .endObject())
77                  .setRefresh(true)
78                  .get();
79  
80          PercolateResponse percolate = client().preparePercolate()
81                  .setIndices("test").setDocumentType("type1")
82                  .setSource(jsonBuilder().startObject().startObject("doc")
83                          .field("field1", "value1")
84                          .endObject().endObject())
85                  .get();
86          assertThat(percolate.getMatches(), arrayWithSize(1));
87  
88          internalCluster().rollingRestart();
89  
90          logger.info("Running Cluster Health (wait for the shards to startup)");
91          ensureYellow();
92  
93          percolate = client().preparePercolate()
94                  .setIndices("test").setDocumentType("type1")
95                  .setSource(jsonBuilder().startObject().startObject("doc")
96                          .field("field1", "value1")
97                          .endObject().endObject())
98                  .get();
99          assertMatchCount(percolate, 1l);
100         assertThat(percolate.getMatches(), arrayWithSize(1));
101     }

176     @Test
177     @Slow
178     public void testLoadingPercolateQueriesDuringCloseAndOpen() throws Exception {
179         internalCluster().startNode();
180         internalCluster().startNode();
181 
182         assertAcked(client().admin().indices().prepareCreate("test")
183                 .setSettings(settingsBuilder().put(IndexMetaData.SETTING_NUMBER_OF_SHARDS, 2)
184                     .put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, 1)));
185         ensureGreen();
186 
187         logger.info("--> Add dummy docs");
188         client().prepareIndex("test", "type1", "1").setSource("field1", 0).get();
189         client().prepareIndex("test", "type2", "1").setSource("field1", "0").get();
190         waitForConcreteMappingsOnAll("test", "type1", "field1");
191 
192         logger.info("--> register a queries");
193         for (int i = 1; i <= 100; i++) {
194             client().prepareIndex("test", PercolatorService.TYPE_NAME, Integer.toString(i))
195                     .setSource(jsonBuilder().startObject()
196                             .field("query", rangeQuery("field1").from(0).to(i))
197                                     // The type must be set now, because two fields with the same name exist in different types.
198                                     // Setting the type to `type1`, makes sure that the range query gets parsed to a Lucene NumericRangeQuery.
199                             .field("type", "type1")
200                             .endObject())
201                     .get();
202         }
203         waitForConcreteMappingsOnAll("test", PercolatorService.TYPE_NAME);
204 
205         logger.info("--> Percolate doc with field1=95");
206         PercolateResponse response = client().preparePercolate()
207                 .setIndices("test").setDocumentType("type1")
208                 .setSource(jsonBuilder().startObject().startObject("doc").field("field1", 95).endObject().endObject())
209                 .get();
210         assertMatchCount(response, 6l);
211         assertThat(response.getMatches(), arrayWithSize(6));
212         assertThat(convertFromTextArray(response.getMatches(), "test"), arrayContainingInAnyOrder("95", "96", "97", "98", "99", "100"));
213 
214         logger.info("--> Close and open index to trigger percolate queries loading...");
215         assertAcked(client().admin().indices().prepareClose("test"));
216         assertAcked(client().admin().indices().prepareOpen("test"));
217         ensureGreen();
218 
219         logger.info("--> Percolate doc with field1=100");
220         response = client().preparePercolate()
221                 .setIndices("test").setDocumentType("type1")
222                 .setSource(jsonBuilder().startObject().startObject("doc").field("field1", 100).endObject().endObject()).get();
223 
224         assertMatchCount(response, 1l);
225         assertThat(response.getMatches(), arrayWithSize(1));
226         assertThat(response.getMatches()[0].getId().string(), equalTo("100"));
227     }

241     // 3 nodes, 2 primary + 2 replicas per primary, so each node should have a copy of the data.
242     // We only start and stop nodes 2 and 3, so all requests should succeed and never be partial.
243     private void percolatorRecovery(final boolean multiPercolate) throws Exception {
244         internalCluster().startNode(settingsBuilder().put("node.stay", true));
245         internalCluster().startNode(settingsBuilder().put("node.stay", false));
246         internalCluster().startNode(settingsBuilder().put("node.stay", false));
247         ensureGreen();
248         client().admin().indices().prepareCreate("test")
249                 .setSettings(settingsBuilder()
250                         .put("index.number_of_shards", 2)
251                         .put("index.number_of_replicas", 2)
252                 )
253                 .get();
254         ensureGreen();
255 
256         final Client client = internalCluster().client(new Predicate<Settings>() {
257             @Override
258             public boolean apply(Settings input) {
259                 return input.getAsBoolean("node.stay", true);
260             }
261         });
262         final int numQueries = randomIntBetween(50, 100);
263         logger.info("--> register a queries");
264         for (int i = 0; i < numQueries; i++) {
265             client.prepareIndex("test", PercolatorService.TYPE_NAME, Integer.toString(i))
266                     .setSource(jsonBuilder().startObject().field("query", matchAllQuery()).endObject())
267                     .get();
268         }
269 
270         client.prepareIndex("test", "type", "1")
271                 .setSource(jsonBuilder().startObject().field("field", "a"))
272                 .get();
273 
274         final AtomicBoolean run = new AtomicBoolean(true);
275         final CountDownLatch done = new CountDownLatch(1);
276         final AtomicReference<Throwable> error = new AtomicReference<>();
277         Runnable r = new Runnable() {
278             @Override
279             public void run() {
280                 try {
281                     XContentBuilder doc = jsonBuilder().startObject().field("field", "a").endObject();
282                     while (run.get()) {
283                         NodesInfoResponse nodesInfoResponse = client.admin().cluster().prepareNodesInfo()
284                                 .get();
285                         String node2Id = null;
286                         String node3Id = null;
287                         for (NodeInfo nodeInfo : nodesInfoResponse) {
288                             if ("node2".equals(nodeInfo.getNode().getName())) {
289                                 node2Id = nodeInfo.getNode().id();
290                             } else if ("node3".equals(nodeInfo.getNode().getName())) {
291                                 node3Id = nodeInfo.getNode().id();
292                             }
293                         }
294 
295                         String preference;
296                         if (node2Id == null && node3Id == null) {
297                             preference = "_local";
298                         } else if (node2Id == null || node3Id == null) {
299                             if (node2Id != null) {
300                                 preference = "_prefer_node:" + node2Id;
301                             } else {
302                                 preference = "_prefer_node:" + node3Id;
303                             }
304                         } else {
305                             preference = "_prefer_node:" + (randomBoolean() ? node2Id : node3Id);
306                         }
307 
308                         if (multiPercolate) {
309                             MultiPercolateRequestBuilder builder = client
310                                     .prepareMultiPercolate();
311                             int numPercolateRequest = randomIntBetween(50, 100);
312 
313                             for (int i = 0; i < numPercolateRequest; i++) {
314                                 if (randomBoolean()) {
315                                     builder.add(
316                                             client.preparePercolate()
317                                                     .setPreference(preference)
318                                                     .setGetRequest(Requests.getRequest("test").type("type").id("1"))
319                                                     .setIndices("test").setDocumentType("type")
320                                     );
321                                 } else {
322                                     builder.add(
323                                             client.preparePercolate()
324                                                     .setPreference(preference)
325                                                     .setIndices("test").setDocumentType("type")
326                                                     .setPercolateDoc(docBuilder().setDoc(doc)));
327                                 }
328                             }
329 
330                             MultiPercolateResponse response = builder.get();
331                             assertThat(response.items().length, equalTo(numPercolateRequest));
332                             for (MultiPercolateResponse.Item item : response) {
333                                 assertThat(item.isFailure(), equalTo(false));
334                                 assertNoFailures(item.getResponse());
335                                 assertThat(item.getResponse().getSuccessfulShards(), equalTo(item.getResponse().getTotalShards()));
336                                 assertThat(item.getResponse().getCount(), equalTo((long) numQueries));
337                                 assertThat(item.getResponse().getMatches().length, equalTo(numQueries));
338                             }
339                         } else {
340                             PercolateResponse response;
341                             if (randomBoolean()) {
342                                 response = client.preparePercolate()
343                                         .setIndices("test").setDocumentType("type")
344                                         .setPercolateDoc(docBuilder().setDoc(doc))
345                                         .setPreference(preference)
346                                         .get();
347                             } else {
348                                 response = client.preparePercolate()
349                                         .setGetRequest(Requests.getRequest("test").type("type").id("1"))
350                                         .setIndices("test").setDocumentType("type")
351                                         .setPreference(preference)
352                                         .get();
353                             }
354                             assertNoFailures(response);
355                             assertThat(response.getSuccessfulShards(), equalTo(response.getTotalShards()));
356                             assertThat(response.getCount(), equalTo((long) numQueries));
357                             assertThat(response.getMatches().length, equalTo(numQueries));
358                         }
359                     }
360                 } catch (Throwable t) {
361                     logger.info("Error in percolate thread...", t);
362                     run.set(false);
363                     error.set(t);
364                 } finally {
365                     done.countDown();
366                 }
367             }
368         };
369         new Thread(r).start();
370 
371         Predicate<Settings> nodePredicate = new Predicate<Settings>() {
372             @Override
373             public boolean apply(Settings input) {
374                 return !input.getAsBoolean("node.stay", false);
375             }
376         };
377         try {
378             // 1 index, 2 primaries, 2 replicas per primary
379             for (int i = 0; i < 4; i++) {
380                 internalCluster().stopRandomNode(nodePredicate);
381                 client.admin().cluster().prepareHealth("test")
382                         .setWaitForEvents(Priority.LANGUID)
383                         .setTimeout(TimeValue.timeValueMinutes(2))
384                         .setWaitForYellowStatus()
385                         .setWaitForActiveShards(4) // 2 nodes, so 4 shards (2 primaries, 2 replicas)
386                         .get();
387                 assertThat(error.get(), nullValue());
388                 internalCluster().stopRandomNode(nodePredicate);
389                 client.admin().cluster().prepareHealth("test")
390                         .setWaitForEvents(Priority.LANGUID)
391                         .setTimeout(TimeValue.timeValueMinutes(2))
392                         .setWaitForYellowStatus()
393                         .setWaitForActiveShards(2) // 1 node, so 2 shards (2 primaries, 0 replicas)
394                         .get();
395                 assertThat(error.get(), nullValue());
396                 internalCluster().startNode();
397                 client.admin().cluster().prepareHealth("test")
398                         .setWaitForEvents(Priority.LANGUID)
399                         .setTimeout(TimeValue.timeValueMinutes(2))
400                         .setWaitForYellowStatus()
401                         .setWaitForActiveShards(4)  // 2 nodes, so 4 shards (2 primaries, 2 replicas)
402                         .get();
403                 assertThat(error.get(), nullValue());
404                 internalCluster().startNode();
405                 client.admin().cluster().prepareHealth("test")
406                         .setWaitForEvents(Priority.LANGUID)
407                         .setTimeout(TimeValue.timeValueMinutes(2))
408                         .setWaitForGreenStatus() // We're confirm the shard settings, so green instead of yellow
409                         .setWaitForActiveShards(6) // 3 nodes, so 6 shards (2 primaries, 4 replicas)
410                         .get();
411                 assertThat(error.get(), nullValue());
412             }
413         } finally {
414             run.set(false);
415         }
416         done.await();
417         assertThat(error.get(), nullValue());
418     }
