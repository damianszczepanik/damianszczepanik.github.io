68      @Test
69      public void testIndexActions() throws Exception {
70          createIndex();
71          NumShards numShards = getNumShards(getConcreteIndexName());
72          logger.info("Running Cluster Health");
73          ensureGreen();
74          logger.info("Indexing [type1/1]");
75          IndexResponse indexResponse = client().prepareIndex().setIndex("test").setType("type1").setId("1").setSource(source("1", "test")).setRefresh(true).execute().actionGet();
76          assertThat(indexResponse.getIndex(), equalTo(getConcreteIndexName()));
77          assertThat(indexResponse.getId(), equalTo("1"));
78          assertThat(indexResponse.getType(), equalTo("type1"));
79          logger.info("Refreshing");
80          RefreshResponse refreshResponse = refresh();
81          assertThat(refreshResponse.getSuccessfulShards(), equalTo(numShards.totalNumShards));
82  
83          logger.info("--> index exists?");
84          assertThat(indexExists(getConcreteIndexName()), equalTo(true));
85          logger.info("--> index exists?, fake index");
86          assertThat(indexExists("test1234565"), equalTo(false));
87  
88          logger.info("Clearing cache");
89          ClearIndicesCacheResponse clearIndicesCacheResponse = client().admin().indices().clearCache(clearIndicesCacheRequest("test").recycler(true).fieldDataCache(true).filterCache(true).idCache(true)).actionGet();
90          assertNoFailures(clearIndicesCacheResponse);
91          assertThat(clearIndicesCacheResponse.getSuccessfulShards(), equalTo(numShards.totalNumShards));
92  
93          logger.info("Optimizing");
94          waitForRelocation(ClusterHealthStatus.GREEN);
95          OptimizeResponse optimizeResponse = optimize();
96          assertThat(optimizeResponse.getSuccessfulShards(), equalTo(numShards.totalNumShards));
97  
98          GetResponse getResult;
99  
100         logger.info("Get [type1/1]");
101         for (int i = 0; i < 5; i++) {
102             getResult = client().prepareGet("test", "type1", "1").setOperationThreaded(false).execute().actionGet();
103             assertThat(getResult.getIndex(), equalTo(getConcreteIndexName()));
104             assertThat("cycle #" + i, getResult.getSourceAsString(), equalTo(source("1", "test").string()));
105             assertThat("cycle(map) #" + i, (String) getResult.getSourceAsMap().get("name"), equalTo("test"));
106             getResult = client().get(getRequest("test").type("type1").id("1").operationThreaded(true)).actionGet();
107             assertThat("cycle #" + i, getResult.getSourceAsString(), equalTo(source("1", "test").string()));
108             assertThat(getResult.getIndex(), equalTo(getConcreteIndexName()));
109         }
110 
111         logger.info("Get [type1/1] with script");
112         for (int i = 0; i < 5; i++) {
113             getResult = client().prepareGet("test", "type1", "1").setFields("name").execute().actionGet();
114             assertThat(getResult.getIndex(), equalTo(getConcreteIndexName()));
115             assertThat(getResult.isExists(), equalTo(true));
116             assertThat(getResult.getSourceAsBytes(), nullValue());
117             assertThat(getResult.getField("name").getValues().get(0).toString(), equalTo("test"));
118         }
119 
120         logger.info("Get [type1/2] (should be empty)");
121         for (int i = 0; i < 5; i++) {
122             getResult = client().get(getRequest("test").type("type1").id("2")).actionGet();
123             assertThat(getResult.isExists(), equalTo(false));
124         }
125 
126         logger.info("Delete [type1/1]");
127         DeleteResponse deleteResponse = client().prepareDelete("test", "type1", "1").setReplicationType(ReplicationType.SYNC).execute().actionGet();
128         assertThat(deleteResponse.getIndex(), equalTo(getConcreteIndexName()));
129         assertThat(deleteResponse.getId(), equalTo("1"));
130         assertThat(deleteResponse.getType(), equalTo("type1"));
131         logger.info("Refreshing");
132         client().admin().indices().refresh(refreshRequest("test")).actionGet();
133 
134         logger.info("Get [type1/1] (should be empty)");
135         for (int i = 0; i < 5; i++) {
136             getResult = client().get(getRequest("test").type("type1").id("1")).actionGet();
137             assertThat(getResult.isExists(), equalTo(false));
138         }
139 
140         logger.info("Index [type1/1]");
141         client().index(indexRequest("test").type("type1").id("1").source(source("1", "test"))).actionGet();
142         logger.info("Index [type1/2]");
143         client().index(indexRequest("test").type("type1").id("2").source(source("2", "test2"))).actionGet();
144 
145         logger.info("Flushing");
146         FlushResponse flushResult = client().admin().indices().prepareFlush("test").execute().actionGet();
147         assertThat(flushResult.getSuccessfulShards(), equalTo(numShards.totalNumShards));
148         assertThat(flushResult.getFailedShards(), equalTo(0));
149         logger.info("Refreshing");
150         client().admin().indices().refresh(refreshRequest("test")).actionGet();
151 
152         logger.info("Get [type1/1] and [type1/2]");
153         for (int i = 0; i < 5; i++) {
154             getResult = client().get(getRequest("test").type("type1").id("1")).actionGet();
155             assertThat(getResult.getIndex(), equalTo(getConcreteIndexName()));
156             assertThat("cycle #" + i, getResult.getSourceAsString(), equalTo(source("1", "test").string()));
157             getResult = client().get(getRequest("test").type("type1").id("2")).actionGet();
158             String ste1 = getResult.getSourceAsString();
159             String ste2 = source("2", "test2").string();
160             assertThat("cycle #" + i, ste1, equalTo(ste2));
161             assertThat(getResult.getIndex(), equalTo(getConcreteIndexName()));
162         }
163 
164         logger.info("Count");
165         // check count
166         for (int i = 0; i < 5; i++) {
167             // test successful
168             CountResponse countResponse = client().prepareCount("test").setQuery(termQuery("_type", "type1")).execute().actionGet();
169             assertNoFailures(countResponse);
170             assertThat(countResponse.getCount(), equalTo(2l));
171             assertThat(countResponse.getSuccessfulShards(), equalTo(numShards.numPrimaries));
172             assertThat(countResponse.getFailedShards(), equalTo(0));
173 
174             // test failed (simply query that can't be parsed)
175             countResponse = client().count(countRequest("test").source("{ term : { _type : \"type1 } }".getBytes(Charsets.UTF_8))).actionGet();
176 
177             assertThat(countResponse.getCount(), equalTo(0l));
178             assertThat(countResponse.getSuccessfulShards(), equalTo(0));
179             assertThat(countResponse.getFailedShards(), equalTo(numShards.numPrimaries));
180 
181             // count with no query is a match all one
182             countResponse = client().prepareCount("test").execute().actionGet();
183             assertThat("Failures " + countResponse.getShardFailures(), countResponse.getShardFailures() == null ? 0 : countResponse.getShardFailures().length, equalTo(0));
184             assertThat(countResponse.getCount(), equalTo(2l));
185             assertThat(countResponse.getSuccessfulShards(), equalTo(numShards.numPrimaries));
186             assertThat(countResponse.getFailedShards(), equalTo(0));
187         }
188 
189         logger.info("Delete by query");
190         DeleteByQueryResponse queryResponse = client().prepareDeleteByQuery().setIndices("test").setQuery(termQuery("name", "test2")).execute().actionGet();
191         assertThat(queryResponse.getIndex(getConcreteIndexName()).getShardInfo().getTotal(), greaterThanOrEqualTo(numShards.totalNumShards));
192         assertThat(queryResponse.getIndex(getConcreteIndexName()).getShardInfo().getSuccessful(), greaterThanOrEqualTo(numShards.totalNumShards));
193         assertThat(queryResponse.getIndex(getConcreteIndexName()).getShardInfo().getFailures().length, equalTo(0));
194         client().admin().indices().refresh(refreshRequest("test")).actionGet();
195 
196         logger.info("Get [type1/1] and [type1/2], should be empty");
197         for (int i = 0; i < 5; i++) {
198             getResult = client().get(getRequest("test").type("type1").id("1")).actionGet();
199             assertThat(getResult.getIndex(), equalTo(getConcreteIndexName()));
200             assertThat("cycle #" + i, getResult.getSourceAsString(), equalTo(source("1", "test").string()));
201             getResult = client().get(getRequest("test").type("type1").id("2")).actionGet();
202             assertThat("cycle #" + i, getResult.isExists(), equalTo(false));
203             assertThat(getResult.getIndex(), equalTo(getConcreteIndexName()));
204         }
205     }

207     @Test
208     public void testBulk() throws Exception {
209         createIndex();
210         NumShards numShards = getNumShards(getConcreteIndexName());
211         logger.info("-> running Cluster Health");
212         ensureGreen();
213 
214         BulkResponse bulkResponse = client().prepareBulk()
215                 .add(client().prepareIndex().setIndex("test").setType("type1").setId("1").setSource(source("1", "test")))
216                 .add(client().prepareIndex().setIndex("test").setType("type1").setId("2").setSource(source("2", "test")).setCreate(true))
217                 .add(client().prepareIndex().setIndex("test").setType("type1").setSource(source("3", "test")))
218                 .add(client().prepareDelete().setIndex("test").setType("type1").setId("1"))
219                 .add(client().prepareIndex().setIndex("test").setType("type1").setSource("{ xxx }")) // failure
220                 .execute().actionGet();
221 
222         assertThat(bulkResponse.hasFailures(), equalTo(true));
223         assertThat(bulkResponse.getItems().length, equalTo(5));
224 
225         assertThat(bulkResponse.getItems()[0].isFailed(), equalTo(false));
226         assertThat(bulkResponse.getItems()[0].getOpType(), equalTo("index"));
227         assertThat(bulkResponse.getItems()[0].getIndex(), equalTo(getConcreteIndexName()));
228         assertThat(bulkResponse.getItems()[0].getType(), equalTo("type1"));
229         assertThat(bulkResponse.getItems()[0].getId(), equalTo("1"));
230 
231         assertThat(bulkResponse.getItems()[1].isFailed(), equalTo(false));
232         assertThat(bulkResponse.getItems()[1].getOpType(), equalTo("create"));
233         assertThat(bulkResponse.getItems()[1].getIndex(), equalTo(getConcreteIndexName()));
234         assertThat(bulkResponse.getItems()[1].getType(), equalTo("type1"));
235         assertThat(bulkResponse.getItems()[1].getId(), equalTo("2"));
236 
237         assertThat(bulkResponse.getItems()[2].isFailed(), equalTo(false));
238         assertThat(bulkResponse.getItems()[2].getOpType(), equalTo("create"));
239         assertThat(bulkResponse.getItems()[2].getIndex(), equalTo(getConcreteIndexName()));
240         assertThat(bulkResponse.getItems()[2].getType(), equalTo("type1"));
241         String generatedId3 = bulkResponse.getItems()[2].getId();
242 
243         assertThat(bulkResponse.getItems()[3].isFailed(), equalTo(false));
244         assertThat(bulkResponse.getItems()[3].getOpType(), equalTo("delete"));
245         assertThat(bulkResponse.getItems()[3].getIndex(), equalTo(getConcreteIndexName()));
246         assertThat(bulkResponse.getItems()[3].getType(), equalTo("type1"));
247         assertThat(bulkResponse.getItems()[3].getId(), equalTo("1"));
248 
249         assertThat(bulkResponse.getItems()[4].isFailed(), equalTo(true));
250         assertThat(bulkResponse.getItems()[4].getOpType(), equalTo("create"));
251         assertThat(bulkResponse.getItems()[4].getIndex(), equalTo(getConcreteIndexName()));
252         assertThat(bulkResponse.getItems()[4].getType(), equalTo("type1"));
253 
254         waitForRelocation(ClusterHealthStatus.GREEN);
255         RefreshResponse refreshResponse = client().admin().indices().prepareRefresh("test").execute().actionGet();
256         assertNoFailures(refreshResponse);
257         assertThat(refreshResponse.getSuccessfulShards(), equalTo(numShards.totalNumShards));
258 
259 
260         for (int i = 0; i < 5; i++) {
261             GetResponse getResult = client().get(getRequest("test").type("type1").id("1")).actionGet();
262             assertThat(getResult.getIndex(), equalTo(getConcreteIndexName()));
263             assertThat("cycle #" + i, getResult.isExists(), equalTo(false));
264 
265             getResult = client().get(getRequest("test").type("type1").id("2")).actionGet();
266             assertThat("cycle #" + i, getResult.getSourceAsString(), equalTo(source("2", "test").string()));
267             assertThat(getResult.getIndex(), equalTo(getConcreteIndexName()));
268 
269             getResult = client().get(getRequest("test").type("type1").id(generatedId3)).actionGet();
270             assertThat("cycle #" + i, getResult.getSourceAsString(), equalTo(source("3", "test").string()));
271             assertThat(getResult.getIndex(), equalTo(getConcreteIndexName()));
272         }
273     }

275     @Test
276     public void testDeleteRoutingRequired() throws ExecutionException, InterruptedException, IOException {
277         assertAcked(prepareCreate("test").addMapping("test",
278                 XContentFactory.jsonBuilder().startObject().startObject("test").startObject("_routing").field("required", true).endObject().endObject().endObject()));
279         ensureGreen();
280 
281         int numDocs = iterations(10, 50);
282         IndexRequestBuilder[] indexRequestBuilders = new IndexRequestBuilder[numDocs];
283         for (int i = 0; i < numDocs - 2; i++) {
284             indexRequestBuilders[i] = client().prepareIndex("test", "test", Integer.toString(i))
285                     .setRouting(randomAsciiOfLength(randomIntBetween(1, 10))).setSource("field", "value");
286         }
287         String firstDocId = Integer.toString(numDocs - 2);
288         indexRequestBuilders[numDocs - 2] = client().prepareIndex("test", "test", firstDocId)
289                 .setRouting("routing").setSource("field", "value");
290         String secondDocId = Integer.toString(numDocs - 1);
291         String secondRouting = randomAsciiOfLength(randomIntBetween(1, 10));
292         indexRequestBuilders[numDocs - 1] = client().prepareIndex("test", "test", secondDocId)
293                 .setRouting(secondRouting).setSource("field", "value");
294 
295         indexRandom(true, indexRequestBuilders);
296 
297         SearchResponse searchResponse = client().prepareSearch("test").get();
298         assertNoFailures(searchResponse);
299         assertThat(searchResponse.getHits().totalHits(), equalTo((long) numDocs));
300 
301         //use routing
302         DeleteResponse deleteResponse = client().prepareDelete("test", "test", firstDocId).setRouting("routing").get();
303         assertThat(deleteResponse.isFound(), equalTo(true));
304         GetResponse getResponse = client().prepareGet("test", "test", firstDocId).setRouting("routing").get();
305         assertThat(getResponse.isExists(), equalTo(false));
306         refresh();
307         searchResponse = client().prepareSearch("test").get();
308         assertNoFailures(searchResponse);
309         assertThat(searchResponse.getHits().totalHits(), equalTo((long) numDocs - 1));
310 
311         //don't use routing and trigger a broadcast delete
312         deleteResponse = client().prepareDelete("test", "test", secondDocId).get();
313         assertThat(deleteResponse.isFound(), equalTo(true));
314 
315         getResponse = client().prepareGet("test", "test", secondDocId).setRouting(secondRouting).get();
316         assertThat(getResponse.isExists(), equalTo(false));
317         refresh();
318         searchResponse = client().prepareSearch("test").setSize(numDocs).get();
319         assertNoFailures(searchResponse);
320         assertThat(searchResponse.getHits().totalHits(), equalTo((long) numDocs - 2));
321     }
