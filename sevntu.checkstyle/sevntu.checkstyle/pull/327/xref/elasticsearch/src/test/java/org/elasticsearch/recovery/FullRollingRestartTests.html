41  @ClusterScope(scope = Scope.TEST, numDataNodes = 0, transportClientRatio = 0.0)
42  public class FullRollingRestartTests extends ElasticsearchIntegrationTest {
43  
44      protected void assertTimeout(ClusterHealthRequestBuilder requestBuilder) {
45          ClusterHealthResponse clusterHealth = requestBuilder.get();
46          if (clusterHealth.isTimedOut()) {
47              logger.info("cluster health request timed out:\n{}", clusterHealth);
48              fail("cluster health request timed out");
49          }
50      }
51  
52      @Override
53      protected int numberOfReplicas() {
54          return 1;
55      }
56  
57      @Test
58      @Slow
59      public void testFullRollingRestart() throws Exception {
60          Settings settings = ImmutableSettings.builder().put(ZenDiscovery.SETTING_JOIN_TIMEOUT, "30s").build();
61          internalCluster().startNode(settings);
62          createIndex("test");
63  
64          final String healthTimeout = "1m";
65  
66          for (int i = 0; i < 1000; i++) {
67              client().prepareIndex("test", "type1", Long.toString(i))
68                      .setSource(MapBuilder.<String, Object>newMapBuilder().put("test", "value" + i).map()).execute().actionGet();
69          }
70          flush();
71          for (int i = 1000; i < 2000; i++) {
72              client().prepareIndex("test", "type1", Long.toString(i))
73                      .setSource(MapBuilder.<String, Object>newMapBuilder().put("test", "value" + i).map()).execute().actionGet();
74          }
75  
76          logger.info("--> now start adding nodes");
77          internalCluster().startNodesAsync(2, settings).get();
78  
79          // make sure the cluster state is green, and all has been recovered
80          assertTimeout(client().admin().cluster().prepareHealth().setWaitForEvents(Priority.LANGUID).setTimeout(healthTimeout).setWaitForGreenStatus().setWaitForRelocatingShards(0).setWaitForNodes("3"));
81  
82          logger.info("--> add two more nodes");
83          internalCluster().startNodesAsync(2, settings).get();
84  
85          // We now have 5 nodes
86          setMinimumMasterNodes(3);
87  
88          // make sure the cluster state is green, and all has been recovered
89          assertTimeout(client().admin().cluster().prepareHealth().setWaitForEvents(Priority.LANGUID).setTimeout(healthTimeout).setWaitForGreenStatus().setWaitForRelocatingShards(0).setWaitForNodes("5"));
90  
91          logger.info("--> refreshing and checking data");
92          refresh();
93          for (int i = 0; i < 10; i++) {
94              assertHitCount(client().prepareCount().setQuery(matchAllQuery()).get(), 2000l);
95          }
96  
97          // now start shutting nodes down
98          internalCluster().stopRandomDataNode();
99          // make sure the cluster state is green, and all has been recovered
100         assertTimeout(client().admin().cluster().prepareHealth().setWaitForEvents(Priority.LANGUID).setTimeout(healthTimeout).setWaitForGreenStatus().setWaitForRelocatingShards(0).setWaitForNodes("4"));
101 
102         // going down to 3 nodes. note that the min_master_node may not be in effect when we shutdown the 4th
103         // node, but that's OK as it is set to 3 before.
104         setMinimumMasterNodes(2);
105         internalCluster().stopRandomDataNode();
106         // make sure the cluster state is green, and all has been recovered
107         assertTimeout(client().admin().cluster().prepareHealth().setWaitForEvents(Priority.LANGUID).setTimeout(healthTimeout).setWaitForGreenStatus().setWaitForRelocatingShards(0).setWaitForNodes("3"));
108 
109         logger.info("--> stopped two nodes, verifying data");
110         refresh();
111         for (int i = 0; i < 10; i++) {
112             assertHitCount(client().prepareCount().setQuery(matchAllQuery()).get(), 2000l);
113         }
114 
115         // closing the 3rd node
116         internalCluster().stopRandomDataNode();
117         // make sure the cluster state is green, and all has been recovered
118         assertTimeout(client().admin().cluster().prepareHealth().setWaitForEvents(Priority.LANGUID).setTimeout(healthTimeout).setWaitForGreenStatus().setWaitForRelocatingShards(0).setWaitForNodes("2"));
119 
120         // closing the 2nd node
121         setMinimumMasterNodes(1);
122         internalCluster().stopRandomDataNode();
123 
124         // make sure the cluster state is green, and all has been recovered
125         assertTimeout(client().admin().cluster().prepareHealth().setWaitForEvents(Priority.LANGUID).setTimeout(healthTimeout).setWaitForYellowStatus().setWaitForRelocatingShards(0).setWaitForNodes("1"));
126 
127         logger.info("--> one node left, verifying data");
128         refresh();
129         for (int i = 0; i < 10; i++) {
130             assertHitCount(client().prepareCount().setQuery(matchAllQuery()).get(), 2000l);
131         }
132     }
