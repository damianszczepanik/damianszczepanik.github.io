169     @Test
170     public void simple() throws Exception {
171         SearchResponse response = client().prepareSearch("idx")
172                 .addAggregation(nested("nested").path("nested")
173                         .subAggregation(stats("nested_value_stats").field("nested.value")))
174                 .execute().actionGet();
175 
176         assertSearchResponse(response);
177 
178 
179         double min = Double.POSITIVE_INFINITY;
180         double max = Double.NEGATIVE_INFINITY;
181         long sum = 0;
182         long count = 0;
183         for (int i = 0; i < numParents; ++i) {
184             for (int j = 0; j < numChildren[i]; ++j) {
185                 final long value = i + 1 + j;
186                 min = Math.min(min, value);
187                 max = Math.max(max, value);
188                 sum += value;
189                 ++count;
190             }
191         }
192 
193         Nested nested = response.getAggregations().get("nested");
194         assertThat(nested, notNullValue());
195         assertThat(nested.getName(), equalTo("nested"));
196         assertThat(nested.getDocCount(), equalTo(count));
197         assertThat(nested.getAggregations().asList().isEmpty(), is(false));
198 
199         Stats stats = nested.getAggregations().get("nested_value_stats");
200         assertThat(stats, notNullValue());
201         assertThat(stats.getMin(), equalTo(min));
202         assertThat(stats.getMax(), equalTo(max));
203         assertThat(stats.getCount(), equalTo(count));
204         assertThat(stats.getSum(), equalTo((double) sum));
205         assertThat(stats.getAvg(), equalTo((double) sum / count));
206     }

221     @Test
222     public void nestedWithSubTermsAgg() throws Exception {
223         SearchResponse response = client().prepareSearch("idx")
224                 .addAggregation(nested("nested").path("nested")
225                         .subAggregation(terms("values").field("nested.value").size(100)
226                                 .collectMode(aggCollectionMode)))
227                 .execute().actionGet();
228 
229         assertSearchResponse(response);
230 
231 
232         long docCount = 0;
233         long[] counts = new long[numParents + 6];
234         for (int i = 0; i < numParents; ++i) {
235             for (int j = 0; j < numChildren[i]; ++j) {
236                 final int value = i + 1 + j;
237                 ++counts[value];
238                 ++docCount;
239             }
240         }
241         int uniqueValues = 0;
242         for (long count : counts) {
243             if (count > 0) {
244                 ++uniqueValues;
245             }
246         }
247 
248         Nested nested = response.getAggregations().get("nested");
249         assertThat(nested, notNullValue());
250         assertThat(nested.getName(), equalTo("nested"));
251         assertThat(nested.getDocCount(), equalTo(docCount));
252         assertThat((long) nested.getProperty("_count"), equalTo(docCount));
253         assertThat(nested.getAggregations().asList().isEmpty(), is(false));
254 
255         LongTerms values = nested.getAggregations().get("values");
256         assertThat(values, notNullValue());
257         assertThat(values.getName(), equalTo("values"));
258         assertThat(values.getBuckets(), notNullValue());
259         assertThat(values.getBuckets().size(), equalTo(uniqueValues));
260         for (int i = 0; i < counts.length; ++i) {
261             final String key = Long.toString(i);
262             if (counts[i] == 0) {
263                 assertNull(values.getBucketByKey(key));
264             } else {
265                 Bucket bucket = values.getBucketByKey(key);
266                 assertNotNull(bucket);
267                 assertEquals(counts[i], bucket.getDocCount());
268             }
269         }
270         assertThat((LongTerms) nested.getProperty("values"), sameInstance(values));
271     }

275         SearchResponse response = client().prepareSearch("idx")
276                 .addAggregation(terms("top_values").field("value").size(100)
277                         .collectMode(aggCollectionMode)
