962     private ShapeCollection<Shape> shapeCollection(Shape... shapes) {
963         return new ShapeCollection<>(Arrays.asList(shapes), SPATIAL_CONTEXT);
964     }
965 
966     private ShapeCollection<Shape> shapeCollection(Geometry... geoms) {
967         List<Shape> shapes = new ArrayList<>(geoms.length);
968         for (Geometry geom : geoms) {
969             shapes.add(jtsGeom(geom));
970         }
971         return new ShapeCollection<>(shapes, SPATIAL_CONTEXT);
972     }

974     private JtsGeometry jtsGeom(Geometry geom) {
975         return new JtsGeometry(geom, SPATIAL_CONTEXT, false, false);
976     }
