200     @Test // issue 3246
201     public void ensureThatIndexWarmersCanBeChangedOnRuntime() throws Exception {
202         createIndex("test");
203         ensureGreen();
204 
205         PutWarmerResponse putWarmerResponse = client().admin().indices().preparePutWarmer("custom_warmer")
206                 .setSearchRequest(client().prepareSearch("test").setTypes("test").setQuery(QueryBuilders.matchAllQuery()))
207                 .execute().actionGet();
208         assertThat(putWarmerResponse.isAcknowledged(), equalTo(true));
209 
210         client().prepareIndex("test", "test", "1").setSource("foo", "bar").setRefresh(true).execute().actionGet();

214 
215         long warmerRunsAfterDisabling = getWarmerRuns();
216         assertThat(warmerRunsAfterDisabling, greaterThanOrEqualTo(1L));
217 
218         client().prepareIndex("test", "test", "2").setSource("foo2", "bar2").setRefresh(true).execute().actionGet();
219 
220         assertThat(getWarmerRuns(), equalTo(warmerRunsAfterDisabling));

256     private long getSegmentsMemoryUsage(String idx) {
257         IndicesSegmentResponse response = client().admin().indices().segments(Requests.indicesSegmentsRequest(idx)).actionGet();
258         IndexSegments indicesSegments = response.getIndices().get(idx);
259         long total = 0;
260         for (IndexShardSegments indexShardSegments : indicesSegments) {
261             for (ShardSegments shardSegments : indexShardSegments) {
262                 for (Segment segment : shardSegments) {
263                     logger.debug("+=" + segment.memoryInBytes + " " + indexShardSegments.getShardId() + " " + shardSegments.getIndex());
264                     total += segment.memoryInBytes;
265                 }
266             }
267         }
268         return total;
269     }

271     private enum LoadingMethod {
272         LAZY {
273             @Override
274             CreateIndexRequestBuilder createIndex(String indexName, String type, String fieldName) {
275                 return client().admin().indices().prepareCreate(indexName).setSettings(ImmutableSettings.builder().put(SINGLE_SHARD_NO_REPLICA).put(SearchService.NORMS_LOADING_KEY, Loading.LAZY_VALUE));
276             }
277         },
278         EAGER {
279             @Override
280             CreateIndexRequestBuilder createIndex(String indexName, String type, String fieldName) {
281                 return client().admin().indices().prepareCreate(indexName).setSettings(ImmutableSettings.builder().put(SINGLE_SHARD_NO_REPLICA).put(SearchService.NORMS_LOADING_KEY, Loading.EAGER_VALUE));
282             }
283 
284             @Override
285             boolean isLazy() {
286                 return false;
287             }
288         },
289         EAGER_PER_FIELD {
290             @Override
291             CreateIndexRequestBuilder createIndex(String indexName, String type, String fieldName) throws Exception {
292                 return client().admin().indices().prepareCreate(indexName).setSettings(ImmutableSettings.builder().put(SINGLE_SHARD_NO_REPLICA).put(SearchService.NORMS_LOADING_KEY, Loading.LAZY_VALUE)).addMapping(type, JsonXContent.contentBuilder()
293                                 .startObject()
294                                     .startObject(type)
295                                         .startObject("properties")
296                                             .startObject(fieldName)
297                                                 .field("type", "string")
298                                                 .startObject("norms")
299                                                     .field("loading", Loading.EAGER_VALUE)
300                                                 .endObject()
301                                             .endObject()
302                                         .endObject()
303                                     .endObject()
304                                 .endObject()
305                 );
306             }
307 
308             @Override
309             boolean isLazy() {
310                 return false;
311             }
312         };
313         private static Settings SINGLE_SHARD_NO_REPLICA = ImmutableSettings.builder().put("number_of_shards", 1).put("number_of_replicas", 0).build();
314 
315         abstract CreateIndexRequestBuilder createIndex(String indexName, String type, String fieldName) throws Exception;
316 
317         boolean isLazy() {
318             return true;
319         }
320     }

322     // NOTE: we have to ensure we defeat compression strategies of the default codec...
323     public void testEagerLoading() throws Exception {
324         for (LoadingMethod method : LoadingMethod.values()) {
325             logger.debug("METHOD " + method);
326             String indexName = method.name().toLowerCase(Locale.ROOT);
327             assertAcked(method.createIndex(indexName, "t", "foo"));
328             // index a doc with 1 token, and one with 3 tokens so we dont get CONST compressed (otherwise norms take zero memory usage)
329             client().prepareIndex(indexName, "t", "1").setSource("foo", "bar").execute().actionGet();
330             client().prepareIndex(indexName, "t", "2").setSource("foo", "bar baz foo").setRefresh(true).execute().actionGet();
331             ensureGreen(indexName);
332             long memoryUsage0 = getSegmentsMemoryUsage(indexName);
333             // queries load norms if they were not loaded before
334             client().prepareSearch(indexName).setQuery(QueryBuilders.matchQuery("foo", "bar")).execute().actionGet();
335             long memoryUsage1 = getSegmentsMemoryUsage(indexName);
336             if (method.isLazy()) {
337                 assertThat(memoryUsage1, greaterThan(memoryUsage0));
338             } else {
339                 assertThat(memoryUsage1, equalTo(memoryUsage0));
340             }
341         }
342     }

344     public void testQueryCacheOnWarmer() {
345         createIndex("test");
346         ensureGreen();
347 
348         assertAcked(client().admin().indices().prepareUpdateSettings("test").setSettings(ImmutableSettings.builder().put(IndicesQueryCache.INDEX_CACHE_QUERY_ENABLED, false)));
349         logger.info("register warmer with no query cache, validate no cache is used");
350         assertAcked(client().admin().indices().preparePutWarmer("warmer_1")
351                 .setSearchRequest(client().prepareSearch("test").setTypes("a1").setQuery(QueryBuilders.matchAllQuery()))
352                 .get());
353 
354         client().prepareIndex("test", "type1", "1").setSource("field", "value1").setRefresh(true).execute().actionGet();
355         assertThat(client().admin().indices().prepareStats("test").setQueryCache(true).get().getTotal().getQueryCache().getMemorySizeInBytes(), equalTo(0l));
356 
357         logger.info("register warmer with query cache, validate caching happened");
358         assertAcked(client().admin().indices().preparePutWarmer("warmer_1")
359                 .setSearchRequest(client().prepareSearch("test").setTypes("a1").setQuery(QueryBuilders.matchAllQuery()).setQueryCache(true))
360                 .get());
361 
362         // index again, to make sure it gets refreshed
363         client().prepareIndex("test", "type1", "1").setSource("field", "value1").setRefresh(true).execute().actionGet();
364         assertThat(client().admin().indices().prepareStats("test").setQueryCache(true).get().getTotal().getQueryCache().getMemorySizeInBytes(), greaterThan(0l));
365 
366         client().admin().indices().prepareClearCache().setQueryCache(true).get(); // clean the cache
367         assertThat(client().admin().indices().prepareStats("test").setQueryCache(true).get().getTotal().getQueryCache().getMemorySizeInBytes(), equalTo(0l));
368 
369         logger.info("enable default query caching on the index level, and test that no flag on warmer still caches");
370         assertAcked(client().admin().indices().prepareUpdateSettings("test").setSettings(ImmutableSettings.builder().put(IndicesQueryCache.INDEX_CACHE_QUERY_ENABLED, true)));
371 
372         assertAcked(client().admin().indices().preparePutWarmer("warmer_1")
373                 .setSearchRequest(client().prepareSearch("test").setTypes("a1").setQuery(QueryBuilders.matchAllQuery()))
374                 .get());
375 
376         // index again, to make sure it gets refreshed
377         client().prepareIndex("test", "type1", "1").setSource("field", "value1").setRefresh(true).execute().actionGet();
378         assertThat(client().admin().indices().prepareStats("test").setQueryCache(true).get().getTotal().getQueryCache().getMemorySizeInBytes(), greaterThan(0l));
379     }
