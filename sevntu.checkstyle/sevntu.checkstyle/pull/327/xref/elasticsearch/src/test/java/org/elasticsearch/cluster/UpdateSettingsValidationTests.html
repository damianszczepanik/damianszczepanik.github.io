33  import static org.hamcrest.Matchers.equalTo;

35  /**
36   */
37  @ClusterScope(scope= Scope.TEST, numDataNodes =0)
38  public class UpdateSettingsValidationTests extends ElasticsearchIntegrationTest {
39  
40      @Test
41      public void testUpdateSettingsValidation() throws Exception {
42          List<String> nodes = internalCluster().startNodesAsync(
43                  settingsBuilder().put("node.data", false).build(),
44                  settingsBuilder().put("node.master", false).build(),
45                  settingsBuilder().put("node.master", false).build()
46          ).get();
47          String master = nodes.get(0);
48          String node_1 = nodes.get(1);
49          String node_2 = nodes.get(2);
50  
51          createIndex("test");
52          NumShards test = getNumShards("test");
53  
54          ClusterHealthResponse healthResponse = client().admin().cluster().prepareHealth("test").setWaitForEvents(Priority.LANGUID).setWaitForNodes("3").setWaitForGreenStatus().execute().actionGet();
55          assertThat(healthResponse.isTimedOut(), equalTo(false));
56          assertThat(healthResponse.getIndices().get("test").getActiveShards(), equalTo(test.totalNumShards));
57  
58          client().admin().indices().prepareUpdateSettings("test").setSettings(settingsBuilder().put("index.number_of_replicas", 0)).execute().actionGet();
59          healthResponse = client().admin().cluster().prepareHealth("test").setWaitForEvents(Priority.LANGUID).setWaitForGreenStatus().execute().actionGet();
60          assertThat(healthResponse.isTimedOut(), equalTo(false));
61          assertThat(healthResponse.getIndices().get("test").getActiveShards(), equalTo(test.numPrimaries));
62  
63          try {
64              client().admin().indices().prepareUpdateSettings("test").setSettings(settingsBuilder().put("index.refresh_interval", "")).execute().actionGet();
65              fail();
66          } catch (ElasticsearchIllegalArgumentException ex) {
67              logger.info("Error message: [{}]", ex.getMessage());
68          }
69      }
70  }
