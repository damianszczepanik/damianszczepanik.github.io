246     public void testActualMissingValue(boolean reverse) throws IOException {
247         // missing value is set to an actual value
248         final String[] values = new String[randomIntBetween(2, 30)];
249         for (int i = 1; i < values.length; ++i) {
250             values[i] = TestUtil.randomUnicodeString(getRandom());
251         }
252         final int numDocs = scaledRandomIntBetween(10, 10000);
253         for (int i = 0; i < numDocs; ++i) {
254             final String value = RandomPicks.randomFrom(getRandom(), values);
255             if (value == null) {
256                 writer.addDocument(new Document());
257             } else {
258                 Document d = new Document();
259                 addField(d, "value", value);
260                 writer.addDocument(d);
261             }
262             if (randomInt(10) == 0) {
263                 writer.commit();
264             }
265         }
266 
267         final IndexFieldData indexFieldData = getForField("value");
268         final String missingValue = values[1];
269         IndexSearcher searcher = new IndexSearcher(DirectoryReader.open(writer, true));
270         XFieldComparatorSource comparator = indexFieldData.comparatorSource(missingValue, MultiValueMode.MIN, null);
271         TopFieldDocs topDocs = searcher.search(new MatchAllDocsQuery(), randomBoolean() ? numDocs : randomIntBetween(10, numDocs), new Sort(new SortField("value", comparator, reverse)));
272         assertEquals(numDocs, topDocs.totalHits);
273         BytesRef previousValue = reverse ? UnicodeUtil.BIG_TERM : new BytesRef();
274         for (int i = 0; i < topDocs.scoreDocs.length; ++i) {
275             final String docValue = searcher.doc(topDocs.scoreDocs[i].doc).get("value");
276             final BytesRef value = new BytesRef(docValue == null ? missingValue : docValue);
277             if (reverse) {
278                 assertTrue(previousValue.compareTo(value) >= 0);
279             } else {
280                 assertTrue(previousValue.compareTo(value) <= 0);
281             }
282             previousValue = value;
283         }
284         searcher.getIndexReader().close();
285     }

303     public void testSortMissing(boolean first, boolean reverse) throws IOException {
304         final String[] values = new String[randomIntBetween(2, 10)];
305         for (int i = 1; i < values.length; ++i) {
306             values[i] = TestUtil.randomUnicodeString(getRandom());
307         }
308         final int numDocs = scaledRandomIntBetween(10, 10000);
309         for (int i = 0; i < numDocs; ++i) {
310             final String value = RandomPicks.randomFrom(getRandom(), values);
311             if (value == null) {
312                 writer.addDocument(new Document());
313             } else {
314                 Document d = new Document();
315                 addField(d, "value", value);
316                 writer.addDocument(d);
317             }
318             if (randomInt(10) == 0) {
319                 writer.commit();
320             }
321         }
322         final IndexFieldData indexFieldData = getForField("value");
323         IndexSearcher searcher = new IndexSearcher(DirectoryReader.open(writer, true));
324         XFieldComparatorSource comparator = indexFieldData.comparatorSource(first ? "_first" : "_last", MultiValueMode.MIN, null);
325         TopFieldDocs topDocs = searcher.search(new MatchAllDocsQuery(), randomBoolean() ? numDocs : randomIntBetween(10, numDocs), new Sort(new SortField("value", comparator, reverse)));
326         assertEquals(numDocs, topDocs.totalHits);
327         BytesRef previousValue = first ? null : reverse ? UnicodeUtil.BIG_TERM : new BytesRef();
328         for (int i = 0; i < topDocs.scoreDocs.length; ++i) {
329             final String docValue = searcher.doc(topDocs.scoreDocs[i].doc).get("value");
330             if (first && docValue == null) {
331                 assertNull(previousValue);
332             } else if (!first && docValue != null) {
333                 assertNotNull(previousValue);
334             }
335             final BytesRef value = docValue == null ? null : new BytesRef(docValue);
336             if (previousValue != null && value != null) {
337                 if (reverse) {
338                     assertTrue(previousValue.compareTo(value) >= 0);
339                 } else {
340                     assertTrue(previousValue.compareTo(value) <= 0);
341                 }
342             }
343             previousValue = value;
344         }
345         searcher.getIndexReader().close();
346     }

356     public void testNestedSorting(MultiValueMode sortMode) throws IOException {
357         final String[] values = new String[randomIntBetween(2, 20)];
358         for (int i = 0; i < values.length; ++i) {
359             values[i] = TestUtil.randomSimpleString(getRandom());
360         }
361         final int numParents = scaledRandomIntBetween(10, 10000);
362         List<Document> docs = new ArrayList<>();
363         FixedBitSet parents = new FixedBitSet(64);
364         for (int i = 0; i < numParents; ++i) {
365             docs.clear();
366             final int numChildren = randomInt(4);
367             for (int j = 0; j < numChildren; ++j) {
368                 final Document child = new Document();
369                 final int numValues = randomInt(3);
370                 for (int k = 0; k < numValues; ++k) {
371                     final String value = RandomPicks.randomFrom(getRandom(), values);
372                     addField(child, "text", value);
373                 }
374                 docs.add(child);
375             }
376             final Document parent = new Document();
377             parent.add(new StringField("type", "parent", Store.YES));
378             final String value = RandomPicks.randomFrom(getRandom(), values);
379             if (value != null) {
380                 addField(parent, "text", value);
381             }
382             docs.add(parent);
383             int bit = parents.prevSetBit(parents.length() - 1) + docs.size();
384             parents = FixedBitSet.ensureCapacity(parents, bit);
385             parents.set(bit);
386             writer.addDocuments(docs);
387             if (randomInt(10) == 0) {
388                 writer.commit();
389             }
390         }
391         IndexSearcher searcher = new IndexSearcher(DirectoryReader.open(writer, true));
392         IndexFieldData<?> fieldData = getForField("text");
393         final Object missingValue;
394         switch (randomInt(4)) {
395         case 0:
396             missingValue = "_first";
397             break;
398         case 1:
399             missingValue = "_last";
400             break;
401         case 2:
402             missingValue = new BytesRef(RandomPicks.randomFrom(getRandom(), values));
403             break;
404         default:
405             missingValue = new BytesRef(TestUtil.randomSimpleString(getRandom()));
406             break;
407         }
408         Filter parentFilter = new TermFilter(new Term("type", "parent"));
409         Filter childFilter = new NotFilter(parentFilter);
410         Nested nested = createNested(parentFilter, childFilter);
411         BytesRefFieldComparatorSource nestedComparatorSource = new BytesRefFieldComparatorSource(fieldData, missingValue, sortMode, nested);
412         ToParentBlockJoinQuery query = new ToParentBlockJoinQuery(new FilteredQuery(new MatchAllDocsQuery(), childFilter), new BitDocIdSetCachingWrapperFilter(parentFilter), ScoreMode.None);
413         Sort sort = new Sort(new SortField("text", nestedComparatorSource));
414         TopFieldDocs topDocs = searcher.search(query, randomIntBetween(1, numParents), sort);
415         assertTrue(topDocs.scoreDocs.length > 0);
416         BytesRef previous = null;
417         for (int i = 0; i < topDocs.scoreDocs.length; ++i) {
418             final int docID = topDocs.scoreDocs[i].doc;
419             assertTrue("expected " + docID + " to be a parent", parents.get(docID));
420             BytesRef cmpValue = null;
421             for (int child = parents.prevSetBit(docID - 1) + 1; child < docID; ++child) {
422                 String[] sVals = searcher.doc(child).getValues("text");
423                 final BytesRef[] vals;
424                 if (sVals.length == 0) {
425                     vals = new BytesRef[0];
426                 } else {
427                     vals = new BytesRef[sVals.length];
428                     for (int j = 0; j < vals.length; ++j) {
429                         vals[j] = new BytesRef(sVals[j]);
430                     }
431                 }
432                 for (BytesRef value : vals) {
433                     if (cmpValue == null) {
434                         cmpValue = value;
435                     } else if (sortMode == MultiValueMode.MIN && value.compareTo(cmpValue) < 0) {
436                         cmpValue = value;
437                     } else if (sortMode == MultiValueMode.MAX && value.compareTo(cmpValue) > 0) {
438                         cmpValue = value;
439                     }
440                 }
441             }
442             if (cmpValue == null) {
443                 if ("_first".equals(missingValue)) {
444                     cmpValue = new BytesRef();
445                 } else if ("_last".equals(missingValue)) {
446                     cmpValue = XFieldComparatorSource.MAX_TERM;
447                 } else {
448                     cmpValue = (BytesRef) missingValue;
449                 }
450             }
451             if (previous != null) {
452                 assertTrue(previous.utf8ToString() + "   /   " + cmpValue.utf8ToString(), previous.compareTo(cmpValue) <= 0);
453             }
454             previous = cmpValue;
455         }
456         searcher.getIndexReader().close();
457     }

459     private void assertIteratorConsistentWithRandomAccess(RandomAccessOrds ords, int maxDoc) {
460         for (int doc = 0; doc < maxDoc; ++doc) {
461             ords.setDocument(doc);
462             final int cardinality = ords.cardinality();
463             for (int i = 0; i < cardinality; ++i) {
464                 assertEquals(ords.nextOrd(), ords.ordAt(i));
465             }
466             for (int i = 0; i < 3; ++i) {
467                 assertEquals(ords.nextOrd(), -1);
468             }
469         }
470     }

611         for (Accountable ramUsage : indicesFieldDataCache.getCache().asMap().values()) {
612             if (ramUsage instanceof IndexOrdinalsFieldData) {
613                 cachedInstance = (IndexOrdinalsFieldData) ramUsage;
614                 break;
615             }
616         }
617         assertThat(cachedInstance, sameInstance(globalOrdinals));
618         topLevelReader.close();
619         // Now only 3 segment level entries, only the toplevel reader has been closed, but the segment readers are still used by IW
620         assertThat(indicesFieldDataCache.getCache().size(), equalTo(3l));
621 
622         refreshReader();
623         assertThat(ifd.loadGlobal(topLevelReader), not(sameInstance(globalOrdinals)));
624 
625         ifdService.clear();
626         assertThat(indicesFieldDataCache.getCache().size(), equalTo(0l));
627     }
