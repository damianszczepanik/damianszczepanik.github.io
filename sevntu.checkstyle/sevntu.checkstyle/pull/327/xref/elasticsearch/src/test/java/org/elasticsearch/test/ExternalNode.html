52  /**
53   * Simple helper class to start external nodes to be used within a test cluster
54   */
55  final class ExternalNode implements Closeable {
56  
57      public static final Settings REQUIRED_SETTINGS = ImmutableSettings.builder()
58              .put("config.ignore_system_properties", true)
59              .put(DiscoveryModule.DISCOVERY_TYPE_KEY, "zen")
60              .put("node.mode", "network").build(); // we need network mode for this
61  
62      private final Path path;
63      private final Random random;
64      private final SettingsSource settingsSource;
65      private Process process;
66      private NodeInfo nodeInfo;
67      private final String clusterName;
68      private TransportClient client;
69  
70      private final ESLogger logger = Loggers.getLogger(getClass());
71      private Settings externalNodeSettings;
72  
73  
74      ExternalNode(Path path, long seed, SettingsSource settingsSource) {
75          this(path, null, seed, settingsSource);
76      }
77  
78      ExternalNode(Path path, String clusterName, long seed, SettingsSource settingsSource) {
79          if (!Files.isDirectory(path)) {
80              throw new IllegalArgumentException("path must be a directory");
81          }
82          this.path = path;
83          this.clusterName = clusterName;
84          this.random = new Random(seed);
85          this.settingsSource = settingsSource;
86      }
87  
88      synchronized ExternalNode start(Client localNode, Settings defaultSettings, String nodeName, String clusterName, int nodeOrdinal) throws IOException, InterruptedException {
89          ExternalNode externalNode = new ExternalNode(path, clusterName, random.nextLong(), settingsSource);
90          Settings settings = ImmutableSettings.builder().put(defaultSettings).put(settingsSource.node(nodeOrdinal)).build();
91          externalNode.startInternal(localNode, settings, nodeName, clusterName);
92          return externalNode;
93      }
94  
95      synchronized void startInternal(Client client, Settings settings, String nodeName, String clusterName) throws IOException, InterruptedException {
96          if (process != null) {
97              throw new IllegalStateException("Already started");
98          }
99          List<String> params = new ArrayList<>();
100 
101         if (!Constants.WINDOWS) {
102             params.add("bin/elasticsearch");
103         } else {
104             params.add("bin/elasticsearch.bat");
105         }
106         params.add("-Des.cluster.name=" + clusterName);
107         params.add("-Des.node.name=" + nodeName);
108         ImmutableSettings.Builder externaNodeSettingsBuilder = ImmutableSettings.builder();
109         for (Map.Entry<String, String> entry : settings.getAsMap().entrySet()) {
110             switch (entry.getKey()) {
111                 case "cluster.name":
112                 case "node.name":
113                 case "path.home":
114                 case "node.mode":
115                 case "node.local":
116                 case TransportModule.TRANSPORT_TYPE_KEY:
117                 case DiscoveryModule.DISCOVERY_TYPE_KEY:
118                 case TransportModule.TRANSPORT_SERVICE_TYPE_KEY:
119                 case "config.ignore_system_properties":
120                     continue;
121                 default:
122                     externaNodeSettingsBuilder.put(entry.getKey(), entry.getValue());
123 
124             }
125         }
126         this.externalNodeSettings = externaNodeSettingsBuilder.put(REQUIRED_SETTINGS).build();
127         for (Map.Entry<String, String> entry : externalNodeSettings.getAsMap().entrySet()) {
128             params.add("-Des." + entry.getKey() + "=" + entry.getValue());
129         }
130 
131         params.add("-Des.path.home=" + Paths.get(".").toAbsolutePath());
132         params.add("-Des.path.conf=" + path + "/config");
133 
134         ProcessBuilder builder = new ProcessBuilder(params);
135         builder.directory(path.toFile());
136         builder.inheritIO();
137         boolean success = false;
138         try {
139             logger.debug("starting external node [{}] with: {}", nodeName, builder.command());
140             process = builder.start();
141             this.nodeInfo = null;
142             if (waitForNode(client, nodeName)) {
143                 nodeInfo = nodeInfo(client, nodeName);
144                 assert nodeInfo != null;
145             } else {
146                 throw new IllegalStateException("Node [" + nodeName + "] didn't join the cluster");
147             }
148             success = true;
149         } finally {
150             if (!success) {
151                 stop();
152             }
153         }
154     }
155 
156     static boolean waitForNode(final Client client, final String name) throws InterruptedException {
157         return ElasticsearchTestCase.awaitBusy(new Predicate<Object>() {
158             @Override
159             public boolean apply(java.lang.Object input) {
160                 final NodesInfoResponse nodeInfos = client.admin().cluster().prepareNodesInfo().get();
161                 final NodeInfo[] nodes = nodeInfos.getNodes();
162                 for (NodeInfo info : nodes) {
163                     if (name.equals(info.getNode().getName())) {
164                         return true;
165                     }
166                 }
167                 return false;
168             }
169         }, 30, TimeUnit.SECONDS);
170     }
171 
172     static NodeInfo nodeInfo(final Client client, final String nodeName) {
173         final NodesInfoResponse nodeInfos = client.admin().cluster().prepareNodesInfo().get();
174         final NodeInfo[] nodes = nodeInfos.getNodes();
175         for (NodeInfo info : nodes) {
176             if (nodeName.equals(info.getNode().getName())) {
177                 return info;
178             }
179         }
180         return null;
181     }
182 
183     synchronized TransportAddress getTransportAddress() {
184         if (nodeInfo == null) {
185             throw new IllegalStateException("Node has not started yet");
186         }
187         return nodeInfo.getTransport().getAddress().publishAddress();
188     }
189 
190     synchronized Client getClient() {
191         if (nodeInfo == null) {
192             throw new IllegalStateException("Node has not started yet");
193         }
194         if (client == null) {
195             TransportAddress addr = nodeInfo.getTransport().getAddress().publishAddress();
196             // verify that the end node setting will have network enabled.
197 
198             Settings clientSettings = settingsBuilder().put(externalNodeSettings)
199                     .put("client.transport.nodes_sampler_interval", "1s")
200                     .put("name", "transport_client_" + nodeInfo.getNode().name())
201                     .put(ClusterName.SETTING, clusterName).put("client.transport.sniff", false).build();
202             TransportClient client = new TransportClient(clientSettings);
203             client.addTransportAddress(addr);
204             this.client = client;
205         }
206         return client;
207     }
208 
209     synchronized void reset(long seed) {
210         this.random.setSeed(seed);
211     }
212 
213     synchronized void stop() {
214         stop(false);
215     }
216 
217     synchronized void stop(boolean forceKill) {
218         if (running()) {
219             try {
220                 if (forceKill == false && nodeInfo != null && random.nextBoolean()) {
221                     // sometimes shut down gracefully
222                     getClient().admin().cluster().prepareNodesShutdown(this.nodeInfo.getNode().id()).setExit(random.nextBoolean()).setDelay("0s").get();
223                 }
224                 if (this.client != null) {
225                     client.close();
226                 }
227             } finally {
228                 process.destroy();
229                 try {
230                     process.waitFor();
231                 } catch (InterruptedException e) {
232                     Thread.interrupted();
233                 }
234                 process = null;
235                 nodeInfo = null;
236 
237             }
238         }
239     }
240 
241 
242     synchronized boolean running() {
243         return process != null;
244     }
245 
246     @Override
247     public void close() {
248         stop();
249     }
250 
251     synchronized String getName() {
252         if (nodeInfo == null) {
253             throw new IllegalStateException("Node has not started yet");
254         }
255         return nodeInfo.getNode().getName();
256     }
257 }
