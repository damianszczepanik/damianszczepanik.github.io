22  import org.elasticsearch.test.ElasticsearchTestCase;
23  import org.junit.Test;

25  import static org.hamcrest.Matchers.equalTo;
26  import static org.hamcrest.Matchers.greaterThan;
27  import static org.hamcrest.Matchers.lessThanOrEqualTo;

29  public class RoundingTests extends ElasticsearchTestCase {
30  
31      /**
32       * simple test case to illustrate how Rounding.Interval works on readable input
33       */
34      @Test
35      public void testInterval() {
36          int interval = 10;
37          Rounding.Interval rounding = new Rounding.Interval(interval);
38          int value = 24;
39          final long key = rounding.roundKey(24);
40          final long r = rounding.round(24);
41          String message = "round(" + value + ", interval=" + interval + ") = " + r;
42          assertEquals(value/interval, key);
43          assertEquals(value/interval * interval, r);
44          assertEquals(message, 0, r % interval);
45      }
46  
47      @Test
48      public void testIntervalRandom() {
49          final long interval = randomIntBetween(1, 100);
50          Rounding.Interval rounding = new Rounding.Interval(interval);
51          for (int i = 0; i < 1000; ++i) {
52              long l = Math.max(randomLong(), Long.MIN_VALUE + interval);
53              final long key = rounding.roundKey(l);
54              final long r = rounding.round(l);
55              String message = "round(" + l + ", interval=" + interval + ") = " + r;
56              assertEquals(message, 0, r % interval);
57              assertThat(message, r, lessThanOrEqualTo(l));
58              assertThat(message, r + interval, greaterThan(l));
59              assertEquals(message, r, key*interval);
60          }
61      }

63      /**
64       * Simple test case to illustrate how Rounding.Offset works on readable input.
65       * offset shifts input value back before rounding (so here 6 - 7 -> -1)
66       * then shifts rounded Value back  (here -10 -> -3)
67       */
68      @Test
69      public void testOffsetRounding() {
70          final long interval = 10;
71          final long offset = 7;
72          Rounding.OffsetRounding rounding = new Rounding.OffsetRounding(new Rounding.Interval(interval), offset);
73          assertEquals(-1, rounding.roundKey(6));
74          assertEquals(-3, rounding.round(6));
75          assertEquals(7, rounding.nextRoundingValue(-3));
76          assertEquals(0, rounding.roundKey(7));
77          assertEquals(7, rounding.round(7));
78          assertEquals(17, rounding.nextRoundingValue(7));
79          assertEquals(0, rounding.roundKey(16));
80          assertEquals(7, rounding.round(16));
81          assertEquals(1, rounding.roundKey(17));
82          assertEquals(17, rounding.round(17));
83          assertEquals(27, rounding.nextRoundingValue(17));
84      }
85  
86      /**
87       * test OffsetRounding with an internal interval rounding on random inputs
88       */
89      @Test
90      public void testOffsetRoundingRandom() {
91          for (int i = 0; i < 1000; ++i) {
92              final long interval = randomIntBetween(1, 100);
93              Rounding.Interval internalRounding = new Rounding.Interval(interval);
94              final long offset = randomIntBetween(-100, 100);
95              Rounding.OffsetRounding rounding = new Rounding.OffsetRounding(internalRounding, offset);
96              long safetyMargin = Math.abs(interval) + Math.abs(offset); // to prevent range overflow
97              long value = Math.max(randomLong() - safetyMargin, Long.MIN_VALUE + safetyMargin);
98              final long key = rounding.roundKey(value);
99              final long key_next = rounding.roundKey(value + interval);
100             final long r_value = rounding.round(value);
101             final long nextRoundingValue = rounding.nextRoundingValue(r_value);
102             assertThat("Rounding should be idempotent", r_value, equalTo(rounding.round(r_value)));
103             assertThat("Rounded value smaller than unrounded, regardless of offset", r_value - offset, lessThanOrEqualTo(value - offset));
104             assertThat("Key and next_key should differ by one", key_next - key, equalTo(1L));
105             assertThat("Rounded value <= value < next interval start", r_value + interval, greaterThan(value));
106             assertThat("NextRounding value should be interval from rounded value", r_value + interval, equalTo(nextRoundingValue));
107         }
108     }
