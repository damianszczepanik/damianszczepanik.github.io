56      @Test
57      public void testRegisterRelevantHeaders() throws InterruptedException {
58  
59          final RestController restController = new RestController(ImmutableSettings.EMPTY);
60  
61          int iterations = randomIntBetween(1, 5);
62  
63          Set<String> headers = new HashSet<>();
64          ExecutorService executorService = Executors.newFixedThreadPool(iterations);
65          for (int i = 0; i < iterations; i++) {
66              int headersCount = randomInt(10);
67              final Set<String> newHeaders = new HashSet<>();
68              for (int j = 0; j < headersCount; j++) {
69                  String usefulHeader = randomRealisticUnicodeOfLengthBetween(1, 30);
70                  newHeaders.add(usefulHeader);
71              }
72              headers.addAll(newHeaders);
73  
74              executorService.submit(new Runnable() {
75                  @Override
76                  public void run() {
77                      restController.registerRelevantHeaders(newHeaders.toArray(new String[newHeaders.size()]));
78                  }
79              });
80          }
81  
82          executorService.shutdown();
83          assertThat(executorService.awaitTermination(1, TimeUnit.SECONDS), equalTo(true));
84          String[] relevantHeaders = restController.relevantHeaders().toArray(new String[restController.relevantHeaders().size()]);
85          assertThat(relevantHeaders.length, equalTo(headers.size()));
86  
87          Arrays.sort(relevantHeaders);
88          String[] headersArray = new String[headers.size()];
89          headersArray = headers.toArray(headersArray);
90          Arrays.sort(headersArray);
91          assertThat(relevantHeaders, equalTo(headersArray));
92      }

349     private static Map<String, String> randomHeaders(int count) {
350         return randomMap(count, "header-");
351     }
352 
353     private static Map<String, String> randomContext(int count) {
354         return randomMap(count, "context-");
355     }
356 
357     private static Map<String, String> randomMap(int count, String prefix) {
358         Map<String, String> headers = new HashMap<>();
359         for (int i = 0; i < count; i++) {
360             headers.put(prefix + randomInt(30), randomAsciiOfLength(10));
361         }
362         return headers;
363     }
364 
365     private static Map<String, String> randomHeadersFrom(Map<String, String> headers) {
366         Map<String, String> newHeaders = new HashMap<>();
367         if (headers.isEmpty()) {
368             return newHeaders;
369         }
370         int i = randomInt(headers.size() - 1);
371         for (Map.Entry<String, String> entry : headers.entrySet()) {
372             if (randomInt(i) == 0) {
373                 newHeaders.put(entry.getKey(), entry.getValue());
374             }
375         }
376         return newHeaders;
377     }

379     private static Client client(Client noOpClient, RestRequest restRequest, Set<String> usefulRestHeaders) {
380         return new BaseRestHandler.HeadersAndContextCopyClient(noOpClient, restRequest, usefulRestHeaders);
381     }
382 
383     private static void putHeaders(ActionRequest<?> request, Map<String, String> headers) {
384         for (Map.Entry<String, String> header : headers.entrySet()) {
385             request.putHeader(header.getKey(), header.getValue());
386         }
387     }
388 
389     private static void putContext(ActionRequest<?> request, Map<String, String> context) {
390         for (Map.Entry<String, String> header : context.entrySet()) {
391             request.putInContext(header.getKey(), header.getValue());
392         }
393     }
394 
395     private static void assertHeaders(ActionRequest<?> request, Map<String, String> headers) {
396         if (headers.size() == 0) {
397             assertThat(request.getHeaders() == null || request.getHeaders().size() == 0, equalTo(true));
398         } else {
399             assertThat(request.getHeaders(), notNullValue());
400             assertThat(request.getHeaders().size(), equalTo(headers.size()));
401             for (String key : request.getHeaders()) {
402                 assertThat(headers.get(key), equalTo(request.getHeader(key)));
403             }
404         }
405     }
406 
407     private static void assertContext(ActionRequest<?> request, Map<String, String> context) {
408         if (context.size() == 0) {
409             assertThat(request.isContextEmpty(), is(true));
410         } else {
411             ImmutableOpenMap map = request.getContext();
412             assertThat(map, notNullValue());
413             assertThat(map.size(), equalTo(context.size()));
414             for (Object key : map.keys()) {
415                 assertThat(context.get(key), equalTo(request.getFromContext(key)));
416             }
417         }
418     }
