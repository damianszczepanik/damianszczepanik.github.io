22  import org.elasticsearch.Version;
23  import org.elasticsearch.cluster.ClusterName;
24  import org.elasticsearch.cluster.ClusterState;
25  import org.elasticsearch.cluster.metadata.IndexMetaData;
26  import org.elasticsearch.cluster.metadata.MetaData;
27  import org.elasticsearch.cluster.node.DiscoveryNodes;
28  import org.elasticsearch.cluster.routing.RoutingTable;
29  import org.elasticsearch.common.logging.ESLogger;
30  import org.elasticsearch.common.logging.Loggers;
31  import org.elasticsearch.test.ElasticsearchAllocationTestCase;
32  import org.junit.Test;
33  
34  import static org.elasticsearch.cluster.routing.ShardRoutingState.INITIALIZING;
35  import static org.elasticsearch.common.settings.ImmutableSettings.settingsBuilder;
36  import static org.hamcrest.Matchers.equalTo;
37  
38  /**
39   */
40  public class AllocatePostApiFlagTests extends ElasticsearchAllocationTestCase {
41  
42      private final ESLogger logger = Loggers.getLogger(AllocatePostApiFlagTests.class);
43  
44      @Test
45      public void simpleFlagTests() {
46          AllocationService allocation = createAllocationService(settingsBuilder().put("cluster.routing.allocation.concurrent_recoveries", 10).build());
47  
48          logger.info("creating an index with 1 shard, no replica");
49          MetaData metaData = MetaData.builder()
50                  .put(IndexMetaData.builder("test").settings(settings(Version.CURRENT)).numberOfShards(1).numberOfReplicas(0))
51                  .build();
52          RoutingTable routingTable = RoutingTable.builder()
53                  .addAsNew(metaData.index("test"))
54                  .build();
55          ClusterState clusterState = ClusterState.builder(ClusterName.DEFAULT).metaData(metaData).routingTable(routingTable).build();
56          assertThat(clusterState.routingTable().index("test").shard(0).primaryAllocatedPostApi(), equalTo(false));
57  
58          logger.info("adding two nodes and performing rerouting");
59          clusterState = ClusterState.builder(clusterState).nodes(DiscoveryNodes.builder().put(newNode("node1")).put(newNode("node2"))).build();
60          RoutingAllocation.Result rerouteResult = allocation.reroute(clusterState);
61          clusterState = ClusterState.builder(clusterState).routingTable(rerouteResult.routingTable()).build();
62          assertThat(clusterState.routingTable().index("test").shard(0).primaryAllocatedPostApi(), equalTo(false));
63  
64          logger.info("start primary shard");
65          rerouteResult = allocation.applyStartedShards(clusterState, clusterState.routingNodes().shardsWithState(INITIALIZING));
66          clusterState = ClusterState.builder(clusterState).routingTable(rerouteResult.routingTable()).build();
67          assertThat(clusterState.routingTable().index("test").shard(0).primaryAllocatedPostApi(), equalTo(true));
68      }
69  }
