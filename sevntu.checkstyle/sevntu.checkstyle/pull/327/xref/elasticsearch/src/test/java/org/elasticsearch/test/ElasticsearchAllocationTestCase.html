71      public static AllocationDeciders randomAllocationDeciders(Settings settings, NodeSettingsService nodeSettingsService, Random random) {
72          final ImmutableSet<Class<? extends AllocationDecider>> defaultAllocationDeciders = AllocationDecidersModule.DEFAULT_ALLOCATION_DECIDERS;
73          final List<AllocationDecider> list = new ArrayList<>();
74          for (Class<? extends AllocationDecider> deciderClass : defaultAllocationDeciders) {
75              try {
76                  try {
77                      Constructor<? extends AllocationDecider> constructor = deciderClass.getConstructor(Settings.class, NodeSettingsService.class);
78                      list.add(constructor.newInstance(settings, nodeSettingsService));
79                  } catch (NoSuchMethodException e) {
80                      Constructor<? extends AllocationDecider> constructor = null;
81                      constructor = deciderClass.getConstructor(Settings.class);
82                      list.add(constructor.newInstance(settings));
83                  }
84              } catch (Exception ex) {
85                  throw new RuntimeException(ex);
86              }
87          }
88          assertThat(list.size(), equalTo(defaultAllocationDeciders.size()));
89          for (AllocationDecider d : list) {
90              assertThat(defaultAllocationDeciders.contains(d.getClass()), is(true));
91          }
92          Collections.shuffle(list, random);
93          return new AllocationDeciders(settings, list.toArray(new AllocationDecider[0]));
94  
95      }

113     public static ClusterState startRandomInitializingShard(ClusterState clusterState, AllocationService strategy) {
114         List<MutableShardRouting> initializingShards = clusterState.routingNodes().shardsWithState(INITIALIZING);
115         if (initializingShards.isEmpty()) {
116             return clusterState;
117         }
118         RoutingTable routingTable = strategy.applyStartedShards(clusterState, newArrayList(initializingShards.get(randomInt(initializingShards.size() - 1)))).routingTable();
119         return ClusterState.builder(clusterState).routingTable(routingTable).build();
120     }
