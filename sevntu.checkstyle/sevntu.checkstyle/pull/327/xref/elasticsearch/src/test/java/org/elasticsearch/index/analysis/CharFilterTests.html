41  public class CharFilterTests extends ElasticsearchTokenStreamTestCase {
42  
43      @Test
44      public void testMappingCharFilter() throws Exception {
45          Index index = new Index("test");
46          Settings settings = settingsBuilder()
47                  .put(IndexMetaData.SETTING_VERSION_CREATED, Version.CURRENT)
48                  .put("index.analysis.char_filter.my_mapping.type", "mapping")
49                  .putArray("index.analysis.char_filter.my_mapping.mappings", "ph=>f", "qu=>q")
50                  .put("index.analysis.analyzer.custom_with_char_filter.tokenizer", "standard")
51                  .putArray("index.analysis.analyzer.custom_with_char_filter.char_filter", "my_mapping")
52                  .build();
53          Injector parentInjector = new ModulesBuilder().add(new SettingsModule(settings), new EnvironmentModule(new Environment(settings)), new IndicesAnalysisModule()).createInjector();
54          Injector injector = new ModulesBuilder().add(
55                  new IndexSettingsModule(index, settings),
56                  new IndexNameModule(index),
57                  new AnalysisModule(settings, parentInjector.getInstance(IndicesAnalysisService.class)))
58                  .createChildInjector(parentInjector);
59  
60          AnalysisService analysisService = injector.getInstance(AnalysisService.class);
61  
62          NamedAnalyzer analyzer1 = analysisService.analyzer("custom_with_char_filter");
63          
64          assertTokenStreamContents(analyzer1.tokenStream("test", "jeff quit phish"), new String[]{"jeff", "qit", "fish"});
65  
66          // Repeat one more time to make sure that char filter is reinitialized correctly
67          assertTokenStreamContents(analyzer1.tokenStream("test", "jeff quit phish"), new String[]{"jeff", "qit", "fish"});
68      }
69  
70      @Test
71      public void testHtmlStripCharFilter() throws Exception {
72          Index index = new Index("test");
73          Settings settings = settingsBuilder()
74                  .put(IndexMetaData.SETTING_VERSION_CREATED, Version.CURRENT)
75                  .put("index.analysis.analyzer.custom_with_char_filter.tokenizer", "standard")
76                  .putArray("index.analysis.analyzer.custom_with_char_filter.char_filter", "html_strip")
77                  .build();
78          Injector parentInjector = new ModulesBuilder().add(new SettingsModule(settings), new EnvironmentModule(new Environment(settings)), new IndicesAnalysisModule()).createInjector();
79          Injector injector = new ModulesBuilder().add(
80                  new IndexSettingsModule(index, settings),
81                  new IndexNameModule(index),
82                  new AnalysisModule(settings, parentInjector.getInstance(IndicesAnalysisService.class)))
83                  .createChildInjector(parentInjector);
84  
85          AnalysisService analysisService = injector.getInstance(AnalysisService.class);
86  
87          NamedAnalyzer analyzer1 = analysisService.analyzer("custom_with_char_filter");
88  
89          assertTokenStreamContents(analyzer1.tokenStream("test", "<b>hello</b>!"), new String[]{"hello"});
90  
91          // Repeat one more time to make sure that char filter is reinitialized correctly
92          assertTokenStreamContents(analyzer1.tokenStream("test", "<b>hello</b>!"), new String[]{"hello"});
93      }
