40  @ClusterScope(scope= ElasticsearchIntegrationTest.Scope.TEST, numDataNodes=0, numClientNodes = 1, transportClientRatio = 0)
41  public class PluginServiceTests extends ElasticsearchIntegrationTest {
42  
43      @Test
44      public void testPluginLoadingFromClassName() throws URISyntaxException {
45          Settings settings = settingsBuilder()
46                                  // Defines a plugin in classpath
47                                  .put(PluginsService.LOAD_PLUGIN_FROM_CLASSPATH, true)
48                                  .put(PluginsService.ES_PLUGIN_PROPERTIES_FILE_KEY, "es-plugin-test.properties")
49                                  // Defines a plugin in settings
50                                  .put("plugin.types", InSettingsPlugin.class.getName())
51                              .build();
52  
53          SimpleNodesInfoTests.startNodeWithPlugins(settings, "/org/elasticsearch/plugins/loading/");
54  
55          Plugin plugin = getPlugin("in-settings-plugin");
56          assertNotNull("InSettingsPlugin (defined below in this class) must be loaded", plugin);
57          assertThat(plugin, instanceOf(InSettingsPlugin.class));
58  
59          plugin = getPlugin("in-classpath-plugin");
60          assertNotNull("InClassPathPlugin (defined in package ) must be loaded", plugin);
61          assertThat(plugin, instanceOf(InClassPathPlugin.class));
62  
63          plugin = getPlugin("in-jar-plugin");
64          assertNotNull("InJarPlugin (packaged as a JAR file in a plugins directory) must be loaded", plugin);
65          assertThat(plugin.getClass().getName(), endsWith("InJarPlugin"));
66  
67          plugin = getPlugin("in-zip-plugin");
68          assertNotNull("InZipPlugin (packaged as a Zipped file in a plugins directory) must be loaded", plugin);
69          assertThat(plugin.getClass().getName(), endsWith("InZipPlugin"));
70      }
71  
72      @Test
73      public void testHasLibExtension() {
74          Path p = Paths.get("path", "to", "plugin.jar");
75          assertTrue(PluginsService.hasLibExtension(p));
76  
77          p = Paths.get("path", "to", "plugin.zip");
78          assertTrue(PluginsService.hasLibExtension(p));
79  
80          p = Paths.get("path", "to", "plugin.tar.gz");
81          assertFalse(PluginsService.hasLibExtension(p));
82  
83          p = Paths.get("path", "to", "plugin");
84          assertFalse(PluginsService.hasLibExtension(p));
85      }
86  
87      private Plugin getPlugin(String pluginName) {
88          assertNotNull("cannot check plugin existence with a null plugin's name", pluginName);
89          PluginsService pluginsService = internalCluster().getInstance(PluginsService.class);
90          ImmutableList<Tuple<PluginInfo, Plugin>> plugins = pluginsService.plugins();
91  
92          if ((plugins != null) && (!plugins.isEmpty())) {
93              for (Tuple<PluginInfo, Plugin> plugin:plugins) {
94                  if (pluginName.equals(plugin.v1().getName())) {
95                      return plugin.v2();
96                  }
97              }
98          }
99          return null;
100     }
