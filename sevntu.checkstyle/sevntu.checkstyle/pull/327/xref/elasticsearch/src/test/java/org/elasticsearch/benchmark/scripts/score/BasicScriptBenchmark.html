65      public static class Results {
66          public static final String TIME_PER_DOCIN_MILLIS = "timePerDocinMillis";
67          public static final String NUM_TERMS = "numTerms";
68          public static final String NUM_DOCS = "numDocs";
69          public static final String TIME_PER_QUERY_IN_SEC = "timePerQueryInSec";
70          public static final String TOTAL_TIME_IN_SEC = "totalTimeInSec";
71          Double[] resultSeconds;
72          Double[] resultMSPerQuery;
73          Long[] numDocs;
74          Integer[] numTerms;
75          Double[] timePerDoc;
76          String label;
77          String description;
78          public String lineStyle;
79          public String color;
80  
81          void init(int numVariations, String label, String description, String color, String lineStyle) {
82              resultSeconds = new Double[numVariations];
83              resultMSPerQuery = new Double[numVariations];
84              numDocs = new Long[numVariations];
85              numTerms = new Integer[numVariations];
86              timePerDoc = new Double[numVariations];
87              this.label = label;
88              this.description = description;
89              this.color = color;
90              this.lineStyle = lineStyle;
91          }
92  
93          void set(SearchResponse searchResponse, StopWatch stopWatch, String message, int maxIter, int which, int numTerms) {
94              resultSeconds[which] = (double) ((double) stopWatch.lastTaskTime().getMillis() / (double) 1000);
95              resultMSPerQuery[which] = (double) ((double) stopWatch.lastTaskTime().secondsFrac() / (double) maxIter);
96              numDocs[which] = searchResponse.getHits().totalHits();
97              this.numTerms[which] = numTerms;
98              timePerDoc[which] = resultMSPerQuery[which] / numDocs[which];
99          }

101         public void printResults(BufferedWriter writer) throws IOException {
102             String comma = (writer == null) ? "" : ";";
103             String results = description + "\n" + Results.TOTAL_TIME_IN_SEC + " = " + getResultArray(resultSeconds) + comma + "\n"
104                     + Results.TIME_PER_QUERY_IN_SEC + " = " + getResultArray(resultMSPerQuery) + comma + "\n" + Results.NUM_DOCS + " = "
105                     + getResultArray(numDocs) + comma + "\n" + Results.NUM_TERMS + " = " + getResultArray(numTerms) + comma + "\n"
106                     + Results.TIME_PER_DOCIN_MILLIS + " = " + getResultArray(timePerDoc) + comma + "\n";
107             if (writer != null) {
108                 writer.write(results);
109             } else {
110                 System.out.println(results);
111             }
112 
113         }
114 
115         private String getResultArray(Object[] resultArray) {
116             String result = "[";
117             for (int i = 0; i < resultArray.length; i++) {
118                 result += resultArray[i].toString();
119                 if (i != resultArray.length - 1) {
120                     result += ",";
121                 }
122             }
123             result += "]";
124             return result;
125         }
126     }

160     public static void printOctaveScript(List<Results> allResults, String[] args) throws IOException {
161         if (args.length == 0) {
162             return;
163         }
164         try (BufferedWriter out = Files.newBufferedWriter(Paths.get(args[0]), StandardCharsets.UTF_8)) {
165             out.write("#! /usr/local/bin/octave -qf");
166             out.write("\n\n\n\n");
167             out.write("######################################\n");
168             out.write("# Octave script for plotting results\n");
169             String filename = "scriptScoreBenchmark" + new DateTime().toString();
170             out.write("#Call '" + args[0] + "' from the command line. The plot is then in " + filename + "\n\n");
171 
172             out.write("handleArray = [];\n tagArray = [];\n plot([]);\n hold on;\n");
173             for (Results result : allResults) {
174                 out.write("\n");
175                 out.write("# " + result.description);
176                 result.printResults(out);
177                 out.write("handleArray = [handleArray, addToPlot(" + Results.NUM_TERMS + ", " + Results.TIME_PER_DOCIN_MILLIS + ", '"
178                         + result.color + "','" + result.lineStyle + "',5)];\n");
179                 out.write("tagArray = [tagArray; '" + result.label + "'];\n");
180                 out.write("\n");
181             }
182 
183             out.write("xlabel(\'number of query terms');");
184             out.write("ylabel(\'query time per document');");
185 
186             out.write("legend(handleArray,tagArray);\n");
187 
188             out.write("saveas(gcf,'" + filename + ".png','png')\n");
189             out.write("hold off;\n\n");
190         } catch (IOException e) {
191             System.err.println("Error: " + e.getMessage());
192         }
193         writeHelperFunction();
194     }

201     static void indexData(long numDocs, Client client, boolean randomizeTerms) throws IOException {
202         try {
203             client.admin().indices().prepareDelete("test").execute().actionGet();
204         } catch (Throwable t) {
205             // index might exist already, in this case we do nothing TODO: make
206             // saver in general
207         }
208 
209         XContentBuilder mapping = XContentFactory.jsonBuilder().startObject().startObject("type1").startObject("properties")
210                 .startObject("text").field("type", "string").field("index_options", "offsets").field("analyzer", "payload_float")
211                 .endObject().endObject().endObject().endObject();
212         client.admin()
213                 .indices()
214                 .prepareCreate("test")
215                 .addMapping("type1", mapping)
216                 .setSettings(
217                         ImmutableSettings.settingsBuilder().put("index.analysis.analyzer.payload_float.tokenizer", "whitespace")
218                                 .putArray("index.analysis.analyzer.payload_float.filter", "delimited_float")
219                                 .put("index.analysis.filter.delimited_float.delimiter", "|")
220                                 .put("index.analysis.filter.delimited_float.encoding", "float")
221                                 .put("index.analysis.filter.delimited_float.type", "delimited_payload_filter")
222                                 .put("index.number_of_replicas", 0).put("index.number_of_shards", 1)).execute().actionGet();
223         client.admin().cluster().prepareHealth("test").setWaitForGreenStatus().setTimeout("10s").execute().actionGet();
224         BulkRequestBuilder bulkRequest = client.prepareBulk();
225         Random random = new Random(1);
226         for (int i = 0; i < numDocs; i++) {
227 
228             bulkRequest.add(client.prepareIndex().setType("type1").setIndex("test")
229                     .setSource(jsonBuilder().startObject().field("text", randomText(random, randomizeTerms)).endObject()));
230             if (i % 1000 == 0) {
231                 bulkRequest.execute().actionGet();
232                 bulkRequest = client.prepareBulk();
233             }
234         }
235         bulkRequest.execute().actionGet();
236         client.admin().indices().prepareRefresh("test").execute().actionGet();
237         client.admin().indices().prepareFlush("test").execute().actionGet();
238         System.out.println("Done indexing " + numDocs + " documents");
239 
240     }

242     private static String randomText(Random random, boolean randomizeTerms) {
243         String text = "";
244         for (int i = 0; i < termsList.size(); i++) {
245             if (random.nextInt(5) == 3 || !randomizeTerms) {
246                 text = text + " " + termsList.get(i) + "|1";
247             }
248         }
249         return text;
250     }

259     static List<Entry<String, RequestInfo>> initTermQueries(int minTerms, int maxTerms) {
260         List<Entry<String, RequestInfo>> termSearchRequests = new ArrayList<>();
261         for (int nTerms = minTerms; nTerms < maxTerms; nTerms++) {
262             Map<String, Object> params = new HashMap<>();
263             String[] terms = getTerms(nTerms + 1);
264             params.put("text", terms);
265             SearchRequest request = searchRequest().searchType(SearchType.QUERY_THEN_FETCH).source(
266                     searchSource().explain(false).size(0).query(QueryBuilders.termsQuery("text", terms)));
267             String infoString = "Results for term query with " + (nTerms + 1) + " terms:";
268             termSearchRequests.add(new AbstractMap.SimpleEntry<>(infoString, new RequestInfo(request, nTerms + 1)));
269         }
270         return termSearchRequests;
271     }

273     static List<Entry<String, RequestInfo>> initNativeSearchRequests(int minTerms, int maxTerms, String script, boolean langNative) {
274         List<Entry<String, RequestInfo>> nativeSearchRequests = new ArrayList<>();
275         for (int nTerms = minTerms; nTerms < maxTerms; nTerms++) {
276             Map<String, Object> params = new HashMap<>();
277             String[] terms = getTerms(nTerms + 1);
278             params.put("text", terms);
279             String infoString = "Results for native script with " + (nTerms + 1) + " terms:";
280             ScriptScoreFunctionBuilder scriptFunction = (langNative == true) ? scriptFunction(script, "native", params) : scriptFunction(
281                     script, params);
282             SearchRequest request = searchRequest().searchType(SearchType.QUERY_THEN_FETCH).source(
283                     searchSource()
284                             .explain(false)
285                             .size(0)
286                             .query(functionScoreQuery(FilterBuilders.termsFilter("text", terms), scriptFunction).boostMode(
287                                     CombineFunction.REPLACE)));
288             nativeSearchRequests.add(new AbstractMap.SimpleEntry<>(infoString, new RequestInfo(request, nTerms + 1)));
289         }
290         return nativeSearchRequests;
291     }

293     static List<Entry<String, RequestInfo>> initScriptMatchAllSearchRequests(String script, boolean langNative) {
294         List<Entry<String, RequestInfo>> nativeSearchRequests = new ArrayList<>();
295         String infoString = "Results for constant score script:";
296         ScriptScoreFunctionBuilder scriptFunction = (langNative == true) ? scriptFunction(script, "native") : scriptFunction(script);
297         SearchRequest request = searchRequest().searchType(SearchType.QUERY_THEN_FETCH).source(
298                 searchSource().explain(false).size(0)
299                         .query(functionScoreQuery(FilterBuilders.matchAllFilter(), scriptFunction).boostMode(CombineFunction.REPLACE)));
300         nativeSearchRequests.add(new AbstractMap.SimpleEntry<>(infoString, new RequestInfo(request, 0)));
301 
302         return nativeSearchRequests;
303     }

305     static void runBenchmark(Client client, int maxIter, Results results, List<Entry<String, RequestInfo>> nativeSearchRequests,
306             int minTerms, int warmerIter) throws IOException {
307         int counter = 0;
308         for (Entry<String, RequestInfo> entry : nativeSearchRequests) {
309             SearchResponse searchResponse = null;
310             // warm up
311             for (int i = 0; i < warmerIter; i++) {
312                 searchResponse = client.search(entry.getValue().request).actionGet();
313             }
314             System.gc();
315             // run benchmark
316             StopWatch stopWatch = new StopWatch();
317             stopWatch.start();
318             for (int i = 0; i < maxIter; i++) {
319                 searchResponse = client.search(entry.getValue().request).actionGet();
320             }
321             stopWatch.stop();
322             results.set(searchResponse, stopWatch, entry.getKey(), maxIter, counter, entry.getValue().numTerms);
323             counter++;
324         }
325         results.printResults(null);
326     }
