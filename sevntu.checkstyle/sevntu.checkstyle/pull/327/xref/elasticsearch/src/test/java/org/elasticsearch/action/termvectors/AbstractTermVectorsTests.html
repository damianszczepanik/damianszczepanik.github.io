55      protected static class TestFieldSetting {
56          final public String name;
57          final public boolean storedOffset;
58          final public boolean storedPayloads;
59          final public boolean storedPositions;
60  
61          public TestFieldSetting(String name, boolean storedOffset, boolean storedPayloads, boolean storedPositions) {
62              this.name = name;
63              this.storedOffset = storedOffset;
64              this.storedPayloads = storedPayloads;
65              this.storedPositions = storedPositions;
66          }
67  
68          public void addToMappings(XContentBuilder mappingsBuilder) throws IOException {
69              mappingsBuilder.startObject(name);
70              mappingsBuilder.field("type", "string");
71              String tv_settings;
72              if (storedPositions && storedOffset && storedPayloads) {
73                  tv_settings = "with_positions_offsets_payloads";
74              } else if (storedPositions && storedOffset) {
75                  tv_settings = "with_positions_offsets";
76              } else if (storedPayloads) {
77                  tv_settings = "with_positions_payloads";
78              } else if (storedPositions) {
79                  tv_settings = "with_positions";
80              } else if (storedOffset) {
81                  tv_settings = "with_offsets";
82              } else {
83                  tv_settings = "yes";
84              }
85  
86              mappingsBuilder.field("term_vector", tv_settings);
87  
88              if (storedPayloads) {
89                  mappingsBuilder.field("analyzer", "tv_test");
90              }
91  
92              mappingsBuilder.endObject();
93          }
94  
95          @Override
96          public String toString() {
97              StringBuilder sb = new StringBuilder("name: ").append(name).append(" tv_with:");
98              if (storedPayloads) {
99                  sb.append("payloads,");
100             }
101             if (storedOffset) {
102                 sb.append("offsets,");
103             }
104             if (storedPositions) {
105                 sb.append("positions,");
106             }
107             return sb.toString();
108         }
109     }

202         ensureYellow();

208     protected TestDoc[] generateTestDocs(String index, TestFieldSetting[] fieldSettings) {
209         String[] fieldContentOptions = new String[]{"Generating a random permutation of a sequence (such as when shuffling cards).",
210                 "Selecting a random sample of a population (important in statistical sampling).",
211                 "Allocating experimental units via random assignment to a treatment or control condition.",
212                 "Generating random numbers: see Random number generation.",
213                 "Transforming a data stream (such as when using a scrambler in telecommunications)."};
214 
215         String[] contentArray = new String[fieldSettings.length];
216         Map<String, Object> docSource = new HashMap<>();
217         int totalShards = getNumShards(index).numPrimaries;
218         TestDoc[] testDocs = new TestDoc[totalShards];
219         // this methods wants to send one doc to each shard
220         for (int i = 0; i < totalShards; i++) {
221             docSource.clear();
222             for (int j = 0; j < contentArray.length; j++) {
223                 contentArray[j] = fieldContentOptions[randomInt(fieldContentOptions.length - 1)];
224                 docSource.put(fieldSettings[j].name, contentArray[j]);
225             }
226             final String id = routingKeyForShard(index, "type", i);
227             TestDoc doc = new TestDoc(id, fieldSettings, contentArray.clone());
228             index(doc.index, doc.type, doc.id, docSource);
229             testDocs[i] = doc;
230         }
231 
232         refresh();
233         return testDocs;
234 
235     }

237     protected TestConfig[] generateTestConfigs(int numberOfTests, TestDoc[] testDocs, TestFieldSetting[] fieldSettings) {
238         ArrayList<TestConfig> configs = new ArrayList<>();
239         for (int i = 0; i < numberOfTests; i++) {
240 
241             ArrayList<String> selectedFields = null;
242             if (randomBoolean()) {
243                 // used field selection
244                 selectedFields = new ArrayList<>();
245                 if (randomBoolean()) {
246                     selectedFields.add("Doesnt_exist"); // this will be ignored.
247                 }
248                 for (TestFieldSetting field : fieldSettings)
249                     if (randomBoolean()) {
250                         selectedFields.add(field.name);
251                     }
252 
253                 if (selectedFields.size() == 0) {
254                     selectedFields = null; // 0 length set is not supported.
255                 }
256 
257             }
258             TestConfig config = new TestConfig(testDocs[randomInt(testDocs.length - 1)], selectedFields == null ? null
259                     : selectedFields.toArray(new String[]{}), randomBoolean(), randomBoolean(), randomBoolean());
260 
261             configs.add(config);
262         }
263         // always adds a test that fails
264         configs.add(new TestConfig(new TestDoc("doesnt_exist", new TestFieldSetting[]{}, new String[]{}).index("doesn't_exist").alias("doesn't_exist"),
265                 new String[]{"doesnt_exist"}, true, true, true).expectedException(IndexMissingException.class));
266 
267         refresh();
268 
269         return configs.toArray(new TestConfig[configs.size()]);
270     }

281     protected DirectoryReader indexDocsWithLucene(TestDoc[] testDocs) throws IOException {
282         Map<String, Analyzer> mapping = new HashMap<>();
283         for (TestFieldSetting field : testDocs[0].fieldSettings) {
284             if (field.storedPayloads) {
285                 mapping.put(field.name, new Analyzer() {
286                     @Override
287                     protected TokenStreamComponents createComponents(String fieldName) {
288                         Tokenizer tokenizer = new StandardTokenizer();
289                         TokenFilter filter = new LowerCaseFilter(tokenizer);
290                         filter = new TypeAsPayloadTokenFilter(filter);
291                         return new TokenStreamComponents(tokenizer, filter);
292                     }
293 
294                 });
295             }
296         }
297         PerFieldAnalyzerWrapper wrapper = new PerFieldAnalyzerWrapper(new StandardAnalyzer(CharArraySet.EMPTY_SET), mapping);
298 
299         Directory dir = new RAMDirectory();
300         IndexWriterConfig conf = new IndexWriterConfig(wrapper);
301 
302         conf.setOpenMode(IndexWriterConfig.OpenMode.CREATE);
303         IndexWriter writer = new IndexWriter(dir, conf);
304 
305         for (TestDoc doc : testDocs) {
306             Document d = new Document();
307             d.add(new Field("id", doc.id, StringField.TYPE_STORED));
308             for (int i = 0; i < doc.fieldContent.length; i++) {
309                 FieldType type = new FieldType(TextField.TYPE_STORED);
310                 TestFieldSetting fieldSetting = doc.fieldSettings[i];
311 
312                 type.setStoreTermVectorOffsets(fieldSetting.storedOffset);
313                 type.setStoreTermVectorPayloads(fieldSetting.storedPayloads);
314                 type.setStoreTermVectorPositions(fieldSetting.storedPositions || fieldSetting.storedPayloads || fieldSetting.storedOffset);
315                 type.setStoreTermVectors(true);
316                 type.freeze();
317                 d.add(new Field(fieldSetting.name, doc.fieldContent[i], type));
318             }
319             writer.updateDocument(new Term("id", doc.id), d);
320             writer.commit();
321         }
322         writer.close();
323 
324         return DirectoryReader.open(dir);
325     }

327     protected void validateResponse(TermVectorsResponse esResponse, Fields luceneFields, TestConfig testConfig) throws IOException {
328         assertThat(esResponse.getIndex(), equalTo(testConfig.doc.index));
329         TestDoc testDoc = testConfig.doc;
330         HashSet<String> selectedFields = testConfig.selectedFields == null ? null : new HashSet<>(
331                 Arrays.asList(testConfig.selectedFields));
332         Fields esTermVectorFields = esResponse.getFields();
333         for (TestFieldSetting field : testDoc.fieldSettings) {
334             Terms esTerms = esTermVectorFields.terms(field.name);
335             if (selectedFields != null && !selectedFields.contains(field.name)) {
336                 assertNull(esTerms);
337                 continue;
338             }
339 
340             assertNotNull(esTerms);
341 
342             Terms luceneTerms = luceneFields.terms(field.name);
343             TermsEnum esTermEnum = esTerms.iterator(null);
344             TermsEnum luceneTermEnum = luceneTerms.iterator(null);
345 
346             while (esTermEnum.next() != null) {
347                 assertNotNull(luceneTermEnum.next());
348 
349                 assertThat(esTermEnum.totalTermFreq(), equalTo(luceneTermEnum.totalTermFreq()));
350                 PostingsEnum esDocsPosEnum = esTermEnum.postings(null, null, PostingsEnum.POSITIONS);
351                 PostingsEnum luceneDocsPosEnum = luceneTermEnum.postings(null, null, PostingsEnum.POSITIONS);
352                 if (luceneDocsPosEnum == null) {
353                     // test we expect that...
354                     assertFalse(field.storedOffset);
355                     assertFalse(field.storedPayloads);
356                     assertFalse(field.storedPositions);
357                     continue;
358                 }
359 
360                 String currentTerm = esTermEnum.term().utf8ToString();
361 
362                 assertThat("Token mismatch for field: " + field.name, currentTerm, equalTo(luceneTermEnum.term().utf8ToString()));
363 
364                 esDocsPosEnum.nextDoc();
365                 luceneDocsPosEnum.nextDoc();
366 
367                 int freq = esDocsPosEnum.freq();
368                 assertThat(freq, equalTo(luceneDocsPosEnum.freq()));
369                 for (int i = 0; i < freq; i++) {
370                     String failDesc = " (field:" + field.name + " term:" + currentTerm + ")";
371                     int lucenePos = luceneDocsPosEnum.nextPosition();
372                     int esPos = esDocsPosEnum.nextPosition();
373                     if (field.storedPositions && testConfig.requestPositions) {
374                         assertThat("Position test failed" + failDesc, lucenePos, equalTo(esPos));
375                     } else {
376                         assertThat("Missing position test failed" + failDesc, esPos, equalTo(-1));
377                     }
378                     if (field.storedOffset && testConfig.requestOffsets) {
379                         assertThat("Offset test failed" + failDesc, luceneDocsPosEnum.startOffset(), equalTo(esDocsPosEnum.startOffset()));
380                         assertThat("Offset test failed" + failDesc, luceneDocsPosEnum.endOffset(), equalTo(esDocsPosEnum.endOffset()));
381                     } else {
382                         assertThat("Missing offset test failed" + failDesc, esDocsPosEnum.startOffset(), equalTo(-1));
383                         assertThat("Missing offset test failed" + failDesc, esDocsPosEnum.endOffset(), equalTo(-1));
384                     }
385                     if (field.storedPayloads && testConfig.requestPayloads) {
386                         assertThat("Payload test failed" + failDesc, luceneDocsPosEnum.getPayload(), equalTo(esDocsPosEnum.getPayload()));
387                     } else {
388                         assertThat("Missing payload test failed" + failDesc, esDocsPosEnum.getPayload(), equalTo(null));
389                     }
390                 }
391             }
392             assertNull("Es returned terms are done but lucene isn't", luceneTermEnum.next());
393         }
394     }

402     protected Fields getTermVectorsFromLucene(DirectoryReader directoryReader, TestDoc doc) throws IOException {
403         IndexSearcher searcher = new IndexSearcher(directoryReader);
404         TopDocs search = searcher.search(new TermQuery(new Term("id", doc.id)), 1);
405 
406         ScoreDoc[] scoreDocs = search.scoreDocs;
407         assertEquals(1, scoreDocs.length);
408         return directoryReader.getTermVectors(scoreDocs[0].doc);
