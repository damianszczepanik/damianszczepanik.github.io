42      public static void main(String[] args) throws Exception {
43  
44          Node node = NodeBuilder.nodeBuilder().clusterName(GeoDistanceSearchBenchmark.class.getSimpleName()).node();
45          Client client = node.client();
46  
47          ClusterHealthResponse clusterHealthResponse = client.admin().cluster().prepareHealth().setWaitForGreenStatus().execute().actionGet();
48          if (clusterHealthResponse.isTimedOut()) {
49              System.err.println("Failed to wait for green status, bailing");
50              System.exit(1);
51          }
52  
53          final long NUM_DOCS = SizeValue.parseSizeValue("1m").singles();
54          final long NUM_WARM = 50;
55          final long NUM_RUNS = 100;
56  
57          if (client.admin().indices().prepareExists("test").execute().actionGet().isExists()) {
58              System.out.println("Found an index, count: " + client.prepareCount("test").setQuery(QueryBuilders.matchAllQuery()).execute().actionGet().getCount());
59          } else {
60              String mapping = XContentFactory.jsonBuilder().startObject().startObject("type1")
61                      .startObject("properties").startObject("location").field("type", "geo_point").field("lat_lon", true).endObject().endObject()
62                      .endObject().endObject().string();
63              client.admin().indices().prepareCreate("test")
64                      .setSettings(ImmutableSettings.settingsBuilder().put("index.number_of_shards", 1).put("index.number_of_replicas", 0))
65                      .addMapping("type1", mapping)
66                      .execute().actionGet();
67  
68              System.err.println("--> Indexing [" + NUM_DOCS + "]");
69              for (long i = 0; i < NUM_DOCS; ) {
70                  client.prepareIndex("test", "type1", Long.toString(i++)).setSource(jsonBuilder().startObject()
71                          .field("name", "New York")
72                          .startObject("location").field("lat", 40.7143528).field("lon", -74.0059731).endObject()
73                          .endObject()).execute().actionGet();
74  
75                  // to NY: 5.286 km
76                  client.prepareIndex("test", "type1", Long.toString(i++)).setSource(jsonBuilder().startObject()
77                          .field("name", "Times Square")
78                          .startObject("location").field("lat", 40.759011).field("lon", -73.9844722).endObject()
79                          .endObject()).execute().actionGet();
80  
81                  // to NY: 0.4621 km
82                  client.prepareIndex("test", "type1", Long.toString(i++)).setSource(jsonBuilder().startObject()
83                          .field("name", "Tribeca")
84                          .startObject("location").field("lat", 40.718266).field("lon", -74.007819).endObject()
85                          .endObject()).execute().actionGet();
86  
87                  // to NY: 1.258 km
88                  client.prepareIndex("test", "type1", Long.toString(i++)).setSource(jsonBuilder().startObject()
89                          .field("name", "Soho")
90                          .startObject("location").field("lat", 40.7247222).field("lon", -74).endObject()
91                          .endObject()).execute().actionGet();
92  
93                  // to NY: 8.572 km
94                  client.prepareIndex("test", "type1", Long.toString(i++)).setSource(jsonBuilder().startObject()
95                          .field("name", "Brooklyn")
96                          .startObject("location").field("lat", 40.65).field("lon", -73.95).endObject()
97                          .endObject()).execute().actionGet();
98  
99                  if ((i % 10000) == 0) {
100                     System.err.println("--> indexed " + i);
101                 }
102             }
103             System.err.println("Done indexed");
104             client.admin().indices().prepareFlush("test").execute().actionGet();
105             client.admin().indices().prepareRefresh().execute().actionGet();
106         }
107 
108         System.err.println("--> Warming up (ARC) - optimize_bbox");
109         long start = System.currentTimeMillis();
110         for (int i = 0; i < NUM_WARM; i++) {
111             run(client, GeoDistance.ARC, "memory");
112         }
113         long totalTime = System.currentTimeMillis() - start;
114         System.err.println("--> Warmup (ARC)  - optimize_bbox (memory) " + (totalTime / NUM_WARM) + "ms");
115 
116         System.err.println("--> Perf (ARC) - optimize_bbox (memory)");
117         start = System.currentTimeMillis();
118         for (int i = 0; i < NUM_RUNS; i++) {
119             run(client, GeoDistance.ARC, "memory");
120         }
121         totalTime = System.currentTimeMillis() - start;
122         System.err.println("--> Perf (ARC) - optimize_bbox " + (totalTime / NUM_RUNS) + "ms");
123 
124         System.err.println("--> Warming up (ARC)  - optimize_bbox (indexed)");
125         start = System.currentTimeMillis();
126         for (int i = 0; i < NUM_WARM; i++) {
127             run(client, GeoDistance.ARC, "indexed");
128         }
129         totalTime = System.currentTimeMillis() - start;
130         System.err.println("--> Warmup (ARC) - optimize_bbox (indexed) " + (totalTime / NUM_WARM) + "ms");
131 
132         System.err.println("--> Perf (ARC) - optimize_bbox (indexed)");
133         start = System.currentTimeMillis();
134         for (int i = 0; i < NUM_RUNS; i++) {
135             run(client, GeoDistance.ARC, "indexed");
136         }
137         totalTime = System.currentTimeMillis() - start;
138         System.err.println("--> Perf (ARC) - optimize_bbox (indexed) " + (totalTime / NUM_RUNS) + "ms");
139 
140 
141         System.err.println("--> Warming up (ARC)  - no optimize_bbox");
142         start = System.currentTimeMillis();
143         for (int i = 0; i < NUM_WARM; i++) {
144             run(client, GeoDistance.ARC, "none");
145         }
146         totalTime = System.currentTimeMillis() - start;
147         System.err.println("--> Warmup (ARC) - no optimize_bbox " + (totalTime / NUM_WARM) + "ms");
148 
149         System.err.println("--> Perf (ARC) - no optimize_bbox");
150         start = System.currentTimeMillis();
151         for (int i = 0; i < NUM_RUNS; i++) {
152             run(client, GeoDistance.ARC, "none");
153         }
154         totalTime = System.currentTimeMillis() - start;
155         System.err.println("--> Perf (ARC) - no optimize_bbox " + (totalTime / NUM_RUNS) + "ms");
156 
157         System.err.println("--> Warming up (SLOPPY_ARC)");
158         start = System.currentTimeMillis();
159         for (int i = 0; i < NUM_WARM; i++) {
160             run(client, GeoDistance.SLOPPY_ARC, "memory");
161         }
162         totalTime = System.currentTimeMillis() - start;
163         System.err.println("--> Warmup (SLOPPY_ARC) " + (totalTime / NUM_WARM) + "ms");
164 
165         System.err.println("--> Perf (SLOPPY_ARC)");
166         start = System.currentTimeMillis();
167         for (int i = 0; i < NUM_RUNS; i++) {
168             run(client, GeoDistance.SLOPPY_ARC, "memory");
169         }
170         totalTime = System.currentTimeMillis() - start;
171         System.err.println("--> Perf (SLOPPY_ARC) " + (totalTime / NUM_RUNS) + "ms");
172 
173         System.err.println("--> Warming up (PLANE)");
174         start = System.currentTimeMillis();
175         for (int i = 0; i < NUM_WARM; i++) {
176             run(client, GeoDistance.PLANE, "memory");
177         }
178         totalTime = System.currentTimeMillis() - start;
179         System.err.println("--> Warmup (PLANE) " + (totalTime / NUM_WARM) + "ms");
180 
181         System.err.println("--> Perf (PLANE)");
182         start = System.currentTimeMillis();
183         for (int i = 0; i < NUM_RUNS; i++) {
184             run(client, GeoDistance.PLANE, "memory");
185         }
186         totalTime = System.currentTimeMillis() - start;

188 
189         node.close();
190     }
191 
192     public static void run(Client client, GeoDistance geoDistance, String optimizeBbox) {
193         client.prepareSearch() // from NY
194                 .setSearchType(SearchType.COUNT)
195                 .setQuery(filteredQuery(matchAllQuery(), geoDistanceFilter("location")
196                         .distance("2km")
197                         .optimizeBbox(optimizeBbox)
198                         .geoDistance(geoDistance)
199                         .point(40.7143528, -74.0059731)))
200                 .execute().actionGet();
201     }
