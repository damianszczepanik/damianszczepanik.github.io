61  public class SubAggregationSearchCollectModeBenchmark {
62  
63      static long COUNT = SizeValue.parseSizeValue("2m").singles();
64      static int BATCH = 1000;
65      static int QUERY_WARMUP = 10;
66      static int QUERY_COUNT = 100;
67      static int NUMBER_OF_TERMS = 200;
68      static int NUMBER_OF_MULTI_VALUE_TERMS = 10;
69      static int STRING_TERM_SIZE = 5;
70  
71      static Client client;
72      static Node[] nodes;

74      public static void main(String[] args) throws Exception {
75          Natives.tryMlockall();
76          Random random = new Random();
77  
78          Settings settings = settingsBuilder()
79                  .put("index.refresh_interval", "-1")
80                  .put(SETTING_NUMBER_OF_SHARDS, 1)
81                  .put(SETTING_NUMBER_OF_REPLICAS, 0)
82                  .build();
83  
84          String clusterName = SubAggregationSearchCollectModeBenchmark.class.getSimpleName();
85          nodes = new Node[1];
86          for (int i = 0; i < nodes.length; i++) {
87              nodes[i] = nodeBuilder().clusterName(clusterName)
88                      .settings(settingsBuilder().put(settings).put("name", "node" + i))
89                      .node();
90          }
91  
92          Node clientNode = nodeBuilder()
93                  .clusterName(clusterName)
94                  .settings(settingsBuilder().put(settings).put("name", "client")).client(true).node();
95  
96          client = clientNode.client();
97  
98          Thread.sleep(10000);
99          try {
100             client.admin().indices().create(createIndexRequest("test").mapping("type1", jsonBuilder()
101               .startObject()
102                 .startObject("type1")
103                   .startObject("properties")
104                     .startObject("s_value_dv")
105                       .field("type", "string")
106                       .field("index", "no")
107                       .startObject("fielddata")
108                         .field("format", "doc_values")
109                       .endObject()
110                     .endObject()
111                     .startObject("sm_value_dv")
112                       .field("type", "string")
113                       .field("index", "no")
114                       .startObject("fielddata")
115                         .field("format", "doc_values")
116                       .endObject()
117                     .endObject()
118                     .startObject("l_value_dv")
119                       .field("type", "long")
120                       .field("index", "no")
121                       .startObject("fielddata")
122                         .field("format", "doc_values")
123                       .endObject()
124                     .endObject()
125                     .startObject("lm_value_dv")
126                       .field("type", "long")
127                       .field("index", "no")
128                       .startObject("fielddata")
129                         .field("format", "doc_values")
130                       .endObject()
131                     .endObject()
132                   .endObject()
133                 .endObject()
134               .endObject())).actionGet();
135 
136             long[] lValues = new long[NUMBER_OF_TERMS];
137             for (int i = 0; i < NUMBER_OF_TERMS; i++) {
138                 lValues[i] = ThreadLocalRandom.current().nextLong();
139             }
140             ObjectOpenHashSet<String> uniqueTerms = ObjectOpenHashSet.newInstance();
141             for (int i = 0; i < NUMBER_OF_TERMS; i++) {
142                 boolean added;
143                 do {
144                     added = uniqueTerms.add(RandomStrings.randomAsciiOfLength(random, STRING_TERM_SIZE));
145                 } while (!added);
146             }
147             String[] sValues = uniqueTerms.toArray(String.class);
148             uniqueTerms = null;
149 
150             StopWatch stopWatch = new StopWatch().start();
151 
152             System.out.println("--> Indexing [" + COUNT + "] ...");
153             long ITERS = COUNT / BATCH;
154             long i = 1;
155             int counter = 0;
156             for (; i <= ITERS; i++) {
157                 BulkRequestBuilder request = client.prepareBulk();
158                 for (int j = 0; j < BATCH; j++) {
159                     counter++;
160 
161                     XContentBuilder builder = jsonBuilder().startObject();
162                     builder.field("id", Integer.toString(counter));
163                     final String sValue = sValues[ThreadLocalRandom.current().nextInt(sValues.length)];
164                     final long lValue = lValues[ThreadLocalRandom.current().nextInt(lValues.length)];
165                     builder.field("s_value", sValue);
166                     builder.field("l_value", lValue);
167                     builder.field("s_value_dv", sValue);
168                     builder.field("l_value_dv", lValue);
169 
170                     for (String field : new String[] {"sm_value", "sm_value_dv"}) {
171                         builder.startArray(field);
172                         for (int k = 0; k < NUMBER_OF_MULTI_VALUE_TERMS; k++) {
173                             builder.value(sValues[ThreadLocalRandom.current().nextInt(sValues.length)]);
174                         }
175                         builder.endArray();
176                     }
177 
178                     for (String field : new String[] {"lm_value", "lm_value_dv"}) {
179                         builder.startArray(field);
180                         for (int k = 0; k < NUMBER_OF_MULTI_VALUE_TERMS; k++) {
181                             builder.value(lValues[ThreadLocalRandom.current().nextInt(sValues.length)]);
182                         }
183                         builder.endArray();
184                     }
185 
186                     builder.endObject();
187 
188                     request.add(Requests.indexRequest("test").type("type1").id(Integer.toString(counter))
189                             .source(builder));
190                 }
191                 BulkResponse response = request.execute().actionGet();
192                 if (response.hasFailures()) {
193                     System.err.println("--> failures...");
194                 }
195                 if (((i * BATCH) % 10000) == 0) {
196                     System.out.println("--> Indexed " + (i * BATCH) + " took " + stopWatch.stop().lastTaskTime());
197                     stopWatch.start();
198                 }
199             }
200             System.out.println("--> Indexing took " + stopWatch.totalTime() + ", TPS " + (((double) (COUNT)) / stopWatch.totalTime().secondsFrac()));
201         } catch (Exception e) {
202             System.out.println("--> Index already exists, ignoring indexing phase, waiting for green");
203             ClusterHealthResponse clusterHealthResponse = client.admin().cluster().prepareHealth().setWaitForGreenStatus().setTimeout("10m").execute().actionGet();
204             if (clusterHealthResponse.isTimedOut()) {
205                 System.err.println("--> Timed out waiting for cluster health");
206             }
207         }
208         client.admin().indices().prepareRefresh().execute().actionGet();
209         COUNT = client.prepareCount().setQuery(matchAllQuery()).execute().actionGet().getCount();
210         System.out.println("--> Number of docs in index: " + COUNT);
211 
212         List<StatsResult> stats = Lists.newArrayList();
213         stats.add(runTest("0000", new SubAggCollectionMode[] {SubAggCollectionMode.DEPTH_FIRST,SubAggCollectionMode.DEPTH_FIRST, SubAggCollectionMode.DEPTH_FIRST, SubAggCollectionMode.DEPTH_FIRST}));
214         stats.add(runTest("0001", new SubAggCollectionMode[] {SubAggCollectionMode.DEPTH_FIRST,SubAggCollectionMode.DEPTH_FIRST, SubAggCollectionMode.DEPTH_FIRST, SubAggCollectionMode.BREADTH_FIRST}));
215         stats.add(runTest("0010", new SubAggCollectionMode[] {SubAggCollectionMode.DEPTH_FIRST,SubAggCollectionMode.DEPTH_FIRST, SubAggCollectionMode.BREADTH_FIRST, SubAggCollectionMode.DEPTH_FIRST}));
216         stats.add(runTest("0011", new SubAggCollectionMode[] {SubAggCollectionMode.DEPTH_FIRST,SubAggCollectionMode.DEPTH_FIRST, SubAggCollectionMode.BREADTH_FIRST, SubAggCollectionMode.BREADTH_FIRST}));
217         stats.add(runTest("0100", new SubAggCollectionMode[] {SubAggCollectionMode.DEPTH_FIRST,SubAggCollectionMode.BREADTH_FIRST, SubAggCollectionMode.DEPTH_FIRST, SubAggCollectionMode.DEPTH_FIRST}));
218         stats.add(runTest("0101", new SubAggCollectionMode[] {SubAggCollectionMode.DEPTH_FIRST,SubAggCollectionMode.BREADTH_FIRST, SubAggCollectionMode.DEPTH_FIRST, SubAggCollectionMode.BREADTH_FIRST}));
219         stats.add(runTest("0110", new SubAggCollectionMode[] {SubAggCollectionMode.DEPTH_FIRST,SubAggCollectionMode.BREADTH_FIRST, SubAggCollectionMode.BREADTH_FIRST, SubAggCollectionMode.DEPTH_FIRST}));
220         stats.add(runTest("0111", new SubAggCollectionMode[] {SubAggCollectionMode.DEPTH_FIRST,SubAggCollectionMode.BREADTH_FIRST, SubAggCollectionMode.BREADTH_FIRST, SubAggCollectionMode.BREADTH_FIRST}));
221         stats.add(runTest("1000", new SubAggCollectionMode[] {SubAggCollectionMode.BREADTH_FIRST,SubAggCollectionMode.DEPTH_FIRST, SubAggCollectionMode.DEPTH_FIRST, SubAggCollectionMode.DEPTH_FIRST}));
222         stats.add(runTest("1001", new SubAggCollectionMode[] {SubAggCollectionMode.BREADTH_FIRST,SubAggCollectionMode.DEPTH_FIRST, SubAggCollectionMode.DEPTH_FIRST, SubAggCollectionMode.BREADTH_FIRST}));
223         stats.add(runTest("1010", new SubAggCollectionMode[] {SubAggCollectionMode.BREADTH_FIRST,SubAggCollectionMode.DEPTH_FIRST, SubAggCollectionMode.BREADTH_FIRST, SubAggCollectionMode.DEPTH_FIRST}));
224         stats.add(runTest("1011", new SubAggCollectionMode[] {SubAggCollectionMode.BREADTH_FIRST,SubAggCollectionMode.DEPTH_FIRST, SubAggCollectionMode.BREADTH_FIRST, SubAggCollectionMode.BREADTH_FIRST}));
225         stats.add(runTest("1100", new SubAggCollectionMode[] {SubAggCollectionMode.BREADTH_FIRST,SubAggCollectionMode.BREADTH_FIRST, SubAggCollectionMode.DEPTH_FIRST, SubAggCollectionMode.DEPTH_FIRST}));
226         stats.add(runTest("1101", new SubAggCollectionMode[] {SubAggCollectionMode.BREADTH_FIRST,SubAggCollectionMode.BREADTH_FIRST, SubAggCollectionMode.DEPTH_FIRST, SubAggCollectionMode.BREADTH_FIRST}));
227         stats.add(runTest("1110", new SubAggCollectionMode[] {SubAggCollectionMode.BREADTH_FIRST,SubAggCollectionMode.BREADTH_FIRST, SubAggCollectionMode.BREADTH_FIRST, SubAggCollectionMode.DEPTH_FIRST}));
228         stats.add(runTest("1111", new SubAggCollectionMode[] {SubAggCollectionMode.BREADTH_FIRST,SubAggCollectionMode.BREADTH_FIRST, SubAggCollectionMode.BREADTH_FIRST, SubAggCollectionMode.BREADTH_FIRST}));
229 
230         System.out.println("------------------ SUMMARY ----------------------------------------------");
231         System.out.format(Locale.ENGLISH, "%35s%10s%10s%15s%15s\n", "name", "took", "millis", "fieldata size", "heap used");
232         for (StatsResult stat : stats) {
233             System.out.format(Locale.ENGLISH, "%35s%10s%10d%15s%15s\n", stat.name, TimeValue.timeValueMillis(stat.took), (stat.took / QUERY_COUNT), stat.fieldDataMemoryUsed, stat.heapUsed);
234         }
235         System.out.println("------------------ SUMMARY ----------------------------------------------");
236 
237         clientNode.close();
238 
239         for (Node node : nodes) {
240             node.close();
241         }

244     public static class StatsResult {
245         final String name;
246         final long took;
247         final ByteSizeValue fieldDataMemoryUsed;
248         final ByteSizeValue heapUsed;
249 
250         public StatsResult(String name, long took, ByteSizeValue fieldDataMemoryUsed, ByteSizeValue heapUsed) {
251             this.name = name;
252             this.took = took;
253             this.fieldDataMemoryUsed = fieldDataMemoryUsed;
254             this.heapUsed = heapUsed;
255         }
256     }
257 
258     private static StatsResult runTest(String name, SubAggCollectionMode[] collectionModes) {
259         long totalQueryTime;// LM VALUE
260 
261         client.admin().indices().prepareClearCache().setFieldDataCache(true).execute().actionGet();
262         System.gc();
263 
264         System.out.println("--> Warmup (" + name + ")...");
265         // run just the child query, warm up first
266         for (int j = 0; j < QUERY_WARMUP; j++) {
267             SearchResponse searchResponse = client.prepareSearch("test")
268                     .setSearchType(SearchType.COUNT)
269                     .setQuery(matchAllQuery())
270                     .addAggregation(AggregationBuilders.terms(name + "s_value").field("s_value").collectMode(collectionModes[0])
271                             .subAggregation(AggregationBuilders.terms(name + "l_value").field("l_value").collectMode(collectionModes[1])
272                                     .subAggregation(AggregationBuilders.terms(name + "s_value_dv").field("s_value_dv").collectMode(collectionModes[2])
273                                             .subAggregation(AggregationBuilders.terms(name + "l_value_dv").field("l_value_dv").collectMode(collectionModes[3])))))
274                     .execute().actionGet();
275             if (j == 0) {
276                 System.out.println("--> Loading : took: " + searchResponse.getTook());
277             }
278             if (searchResponse.getHits().totalHits() != COUNT) {
279                 System.err.println("--> mismatch on hits");
280             }
281         }
282         System.out.println("--> Warmup (" + name + ") DONE");
283 
284 
285         System.out.println("--> Running (" + name + ")...");
286         totalQueryTime = 0;
287         for (int j = 0; j < QUERY_COUNT; j++) {
288             SearchResponse searchResponse = client.prepareSearch("test")
289                     .setSearchType(SearchType.COUNT)
290                     .setQuery(matchAllQuery())
291                     .addAggregation(AggregationBuilders.terms(name + "s_value").field("s_value").collectMode(collectionModes[0])
292                             .subAggregation(AggregationBuilders.terms(name + "l_value").field("l_value").collectMode(collectionModes[1])
293                                     .subAggregation(AggregationBuilders.terms(name + "s_value_dv").field("s_value_dv").collectMode(collectionModes[2])
294                                             .subAggregation(AggregationBuilders.terms(name + "l_value_dv").field("l_value_dv").collectMode(collectionModes[3])))))
295                     .execute().actionGet();
296             if (searchResponse.getHits().totalHits() != COUNT) {
297                 System.err.println("--> mismatch on hits");
298             }
299             totalQueryTime += searchResponse.getTookInMillis();
300         }
301         System.out.println("--> Terms Agg (" + name + "): " + (totalQueryTime / QUERY_COUNT) + "ms");
302 
303         String[] nodeIds = new String[nodes.length];
304         for (int i = 0; i < nodeIds.length; i++) {
305             nodeIds[i] = nodes[i].injector().getInstance(Discovery.class).localNode().getId();
306         }
307 
308         ClusterStatsResponse clusterStateResponse = client.admin().cluster().prepareClusterStats().setNodesIds(nodeIds).get();
309         ByteSizeValue heapUsed = clusterStateResponse.getNodesStats().getJvm().getHeapUsed();
310         System.out.println("--> Heap used: " + heapUsed);
311         ByteSizeValue fieldDataMemoryUsed = clusterStateResponse.getIndicesStats().getFieldData().getMemorySize();
312         System.out.println("--> Fielddata memory size: " + fieldDataMemoryUsed);
313 
314         return new StatsResult(name, totalQueryTime, fieldDataMemoryUsed, heapUsed);
315     }
