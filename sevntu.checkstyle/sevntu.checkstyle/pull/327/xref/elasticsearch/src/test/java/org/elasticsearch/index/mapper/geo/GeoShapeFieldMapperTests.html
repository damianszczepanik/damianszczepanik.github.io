39      @Test
40      public void testDefaultConfiguration() throws IOException {
41          String mapping = XContentFactory.jsonBuilder().startObject().startObject("type1")
42                  .startObject("properties").startObject("location")
43                      .field("type", "geo_shape")
44                  .endObject().endObject()
45                  .endObject().endObject().string();
46  
47          DocumentMapper defaultMapper = createIndex("test").mapperService().documentMapperParser().parse(mapping);
48          FieldMapper fieldMapper = defaultMapper.mappers().name("location").mapper();
49          assertThat(fieldMapper, instanceOf(GeoShapeFieldMapper.class));
50  
51          GeoShapeFieldMapper geoShapeFieldMapper = (GeoShapeFieldMapper) fieldMapper;
52          PrefixTreeStrategy strategy = geoShapeFieldMapper.defaultStrategy();
53  
54          assertThat(strategy.getDistErrPct(), equalTo(GeoShapeFieldMapper.Defaults.DISTANCE_ERROR_PCT));
55          assertThat(strategy.getGrid(), instanceOf(GeohashPrefixTree.class));
56          assertThat(strategy.getGrid().getMaxLevels(), equalTo(GeoShapeFieldMapper.Defaults.GEOHASH_LEVELS));
57      }

105     @Test
106     public void testLevelPrecisionConfiguration() throws IOException {
107         DocumentMapperParser parser = createIndex("test").mapperService().documentMapperParser();
108 
109         {
110             String mapping = XContentFactory.jsonBuilder().startObject().startObject("type1")
111                     .startObject("properties").startObject("location")
112                         .field("type", "geo_shape")
113                         .field("tree", "quadtree")
114                         .field("tree_levels", "6")
115                         .field("precision", "70m")
116                         .field("distance_error_pct", "0.5")
117                     .endObject().endObject()
118                     .endObject().endObject().string();
119 
120             
121             DocumentMapper defaultMapper = parser.parse(mapping);
122             FieldMapper fieldMapper = defaultMapper.mappers().name("location").mapper();
123             assertThat(fieldMapper, instanceOf(GeoShapeFieldMapper.class));
124 
125             GeoShapeFieldMapper geoShapeFieldMapper = (GeoShapeFieldMapper) fieldMapper;
126             PrefixTreeStrategy strategy = geoShapeFieldMapper.defaultStrategy();
127 
128             assertThat(strategy.getDistErrPct(), equalTo(0.5));
129             assertThat(strategy.getGrid(), instanceOf(QuadPrefixTree.class));
130             /* 70m is more precise so it wins */
131             assertThat(strategy.getGrid().getMaxLevels(), equalTo(GeoUtils.quadTreeLevelsForPrecision(70d))); 
132         }
133         
134         {
