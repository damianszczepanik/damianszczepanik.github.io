93      @Test
94      public void testRandom() throws Exception {
95          Directory directory = newDirectory();
96          final Random r = random();
97          final IndexWriterConfig iwc = LuceneTestCase.newIndexWriterConfig(r, new MockAnalyzer(r))
98                  .setMaxBufferedDocs(IndexWriterConfig.DISABLE_AUTO_FLUSH)
99                  .setRAMBufferSizeMB(scaledRandomIntBetween(16, 64)); // we might index a lot - don't go crazy here
100         RandomIndexWriter indexWriter = new RandomIndexWriter(r, directory, iwc);
101         int numUniqueChildValues = scaledRandomIntBetween(100, 2000);
102         String[] childValues = new String[numUniqueChildValues];
103         for (int i = 0; i < numUniqueChildValues; i++) {
104             childValues[i] = Integer.toString(i);
105         }
106 
107         IntOpenHashSet filteredOrDeletedDocs = new IntOpenHashSet();
108 
109         int childDocId = 0;
110         int numParentDocs = scaledRandomIntBetween(1, numUniqueChildValues);
111         ObjectObjectOpenHashMap<String, NavigableMap<String, FloatArrayList>> childValueToParentIds = new ObjectObjectOpenHashMap<>();
112         for (int parentDocId = 0; parentDocId < numParentDocs; parentDocId++) {
113             boolean markParentAsDeleted = rarely();
114             boolean filterMe = rarely();
115             String parent = Integer.toString(parentDocId);
116             Document document = new Document();
117             document.add(new StringField(UidFieldMapper.NAME, Uid.createUid("parent", parent), Field.Store.YES));
118             document.add(new StringField(TypeFieldMapper.NAME, "parent", Field.Store.NO));
119             if (markParentAsDeleted) {
120                 filteredOrDeletedDocs.add(parentDocId);
121                 document.add(new StringField("delete", "me", Field.Store.NO));
122             }
123             if (filterMe) {
124                 filteredOrDeletedDocs.add(parentDocId);
125                 document.add(new StringField("filter", "me", Field.Store.NO));
126             }
127             indexWriter.addDocument(document);
128 
129             int numChildDocs = scaledRandomIntBetween(0, 100);
130             for (int i = 0; i < numChildDocs; i++) {
131                 boolean markChildAsDeleted = rarely();
132                 String childValue = childValues[random().nextInt(childValues.length)];
133 
134                 document = new Document();
135                 document.add(new StringField(UidFieldMapper.NAME, Uid.createUid("child", Integer.toString(childDocId++)), Field.Store.NO));
136                 document.add(new StringField(TypeFieldMapper.NAME, "child", Field.Store.NO));
137                 document.add(new StringField(ParentFieldMapper.NAME, Uid.createUid("parent", parent), Field.Store.NO));
138                 document.add(new StringField("field1", childValue, Field.Store.NO));
139                 if (markChildAsDeleted) {
140                     document.add(new StringField("delete", "me", Field.Store.NO));
141                 }
142                 indexWriter.addDocument(document);
143 
144                 if (!markChildAsDeleted) {
145                     NavigableMap<String, FloatArrayList> parentIdToChildScores;
146                     if (childValueToParentIds.containsKey(childValue)) {
147                         parentIdToChildScores = childValueToParentIds.lget();
148                     } else {
149                         childValueToParentIds.put(childValue, parentIdToChildScores = new TreeMap<>());
150                     }
151                     if (!markParentAsDeleted && !filterMe) {
152                         FloatArrayList childScores = parentIdToChildScores.get(parent);
153                         if (childScores == null) {
154                             parentIdToChildScores.put(parent, childScores = new FloatArrayList());
155                         }
156                         childScores.add(1f);
157                     }
158                 }
159             }
160         }
161 
162         // Delete docs that are marked to be deleted.
163         indexWriter.deleteDocuments(new Term("delete", "me"));
164         indexWriter.commit();
165 
166         IndexReader indexReader = DirectoryReader.open(directory);
167         IndexSearcher searcher = new IndexSearcher(indexReader);
168         Engine.Searcher engineSearcher = new Engine.Searcher(
169                 ChildrenQueryTests.class.getSimpleName(), searcher
170         );
171         ((TestSearchContext) SearchContext.current()).setSearcher(new ContextIndexSearcher(SearchContext.current(), engineSearcher));
172 
173         int max = numUniqueChildValues / 4;
174         for (int i = 0; i < max; i++) {
175             // Simulate a parent update
176             if (random().nextBoolean()) {
177                 final int numberOfUpdatableParents = numParentDocs - filteredOrDeletedDocs.size();
178                 int numberOfUpdates = RandomInts.randomIntBetween(random(), 0, Math.min(numberOfUpdatableParents, TEST_NIGHTLY ? 25 : 5));
179                 for (int j = 0; j < numberOfUpdates; j++) {
180                     int parentId;
181                     do {
182                         parentId = random().nextInt(numParentDocs);
183                     } while (filteredOrDeletedDocs.contains(parentId));
184 
185                     String parentUid = Uid.createUid("parent", Integer.toString(parentId));
186                     indexWriter.deleteDocuments(new Term(UidFieldMapper.NAME, parentUid));
187 
188                     Document document = new Document();
189                     document.add(new StringField(UidFieldMapper.NAME, parentUid, Field.Store.YES));
190                     document.add(new StringField(TypeFieldMapper.NAME, "parent", Field.Store.NO));
191                     indexWriter.addDocument(document);
192                 }
193 
194                 indexReader.close();
195                 indexReader = DirectoryReader.open(indexWriter.w, true);
196                 searcher = new IndexSearcher(indexReader);
197                 engineSearcher = new Engine.Searcher(
198                         ChildrenConstantScoreQueryTests.class.getSimpleName(), searcher
199                 );
200                 ((TestSearchContext) SearchContext.current()).setSearcher(new ContextIndexSearcher(SearchContext.current(), engineSearcher));
201             }
202 
203             String childValue = childValues[random().nextInt(numUniqueChildValues)];
204             int shortCircuitParentDocSet = random().nextInt(numParentDocs);
205             ScoreType scoreType = ScoreType.values()[random().nextInt(ScoreType.values().length)];
206             // leave min/max set to 0 half the time
207             int minChildren = random().nextInt(2) * scaledRandomIntBetween(0, 110);
208             int maxChildren = random().nextInt(2) * scaledRandomIntBetween(minChildren, 110);
209 
210             QueryBuilder queryBuilder = hasChildQuery("child", constantScoreQuery(termQuery("field1", childValue)))
211                     .scoreType(scoreType.name().toLowerCase(Locale.ENGLISH))
212                     .minChildren(minChildren)
213                     .maxChildren(maxChildren)
214                     .setShortCircuitCutoff(shortCircuitParentDocSet);
215             // Using a FQ, will invoke / test the Scorer#advance(..) and also let the Weight#scorer not get live docs as acceptedDocs
216             queryBuilder = filteredQuery(queryBuilder, notFilter(termFilter("filter", "me")));
217             Query query = parseQuery(queryBuilder);
218             BitSetCollector collector = new BitSetCollector(indexReader.maxDoc());
219             int numHits = 1 + random().nextInt(25);
220             TopScoreDocCollector actualTopDocsCollector = TopScoreDocCollector.create(numHits);
221             searcher.search(query, MultiCollector.wrap(collector, actualTopDocsCollector));
222             FixedBitSet actualResult = collector.getResult();
223 
224             FixedBitSet expectedResult = new FixedBitSet(indexReader.maxDoc());
225             TopScoreDocCollector expectedTopDocsCollector = TopScoreDocCollector.create(numHits);
226             if (childValueToParentIds.containsKey(childValue)) {
227                 LeafReader slowLeafReader = SlowCompositeReaderWrapper.wrap(indexReader);
228                 final FloatArrayList[] scores = new FloatArrayList[slowLeafReader.maxDoc()];
229                 Terms terms = slowLeafReader.terms(UidFieldMapper.NAME);
230                 if (terms != null) {
231                     NavigableMap<String, FloatArrayList> parentIdToChildScores = childValueToParentIds.lget();
232                     TermsEnum termsEnum = terms.iterator(null);
233                     PostingsEnum docsEnum = null;
234                     for (Map.Entry<String, FloatArrayList> entry : parentIdToChildScores.entrySet()) {
235                         int count = entry.getValue().elementsCount;
236                         if (count >= minChildren && (maxChildren == 0 || count <= maxChildren)) {
237                             TermsEnum.SeekStatus seekStatus = termsEnum.seekCeil(Uid.createUidAsBytes("parent", entry.getKey()));
238                             if (seekStatus == TermsEnum.SeekStatus.FOUND) {
239                                 docsEnum = termsEnum.postings(slowLeafReader.getLiveDocs(), docsEnum, PostingsEnum.NONE);
240                                 expectedResult.set(docsEnum.nextDoc());
241                                 scores[docsEnum.docID()] = new FloatArrayList(entry.getValue());
242                             } else if (seekStatus == TermsEnum.SeekStatus.END) {
243                                 break;
244                             }
245                         }
246                     }
247                 }
248                 MockScorer mockScorer = new MockScorer(scoreType);
249                 final LeafCollector leafCollector = expectedTopDocsCollector.getLeafCollector(slowLeafReader.getContext());
250                 leafCollector.setScorer(mockScorer);
251                 for (int doc = expectedResult.nextSetBit(0); doc < slowLeafReader.maxDoc(); doc = doc + 1 >= expectedResult.length() ? DocIdSetIterator.NO_MORE_DOCS : expectedResult.nextSetBit(doc + 1)) {
252                     mockScorer.scores = scores[doc];
253                     leafCollector.collect(doc);
254                 }
255             }
256 
257             assertBitSet(actualResult, expectedResult, searcher);
258             assertTopDocs(actualTopDocsCollector.topDocs(), expectedTopDocsCollector.topDocs());
259         }
260 
261         indexWriter.close();
262         indexReader.close();
263         directory.close();
264     }

286     /**
287      * Assert that the {@code scoreType} operates as expected and parents are found in the expected order.
288      * <p />
289      * This will use the test index's parent/child types to create parents with multiple children. Each child will have
290      * a randomly generated scored stored in {@link #CHILD_SCORE_NAME}, which is used to score based on the
291      * {@code scoreType} by using a {@link MockScorer} to determine the expected scores.
292      * @param scoreType The score type to use within the query to score parents relative to their children.
293      * @throws IOException if any unexpected error occurs
294      */
295     private void assertScoreType(ScoreType scoreType) throws IOException {
296         SearchContext context = SearchContext.current();
297         Directory directory = newDirectory();
298         IndexWriter writer = new IndexWriter(directory, newIndexWriterConfig(new MockAnalyzer(random())));
299 
300         // calculates the expected score per parent
301         MockScorer scorer = new MockScorer(scoreType);
302         scorer.scores = new FloatArrayList(10);
303 
304         // number of parents to generate
305         int parentDocs = scaledRandomIntBetween(2, 10);
306         // unique child ID
307         int childDocId = 0;
308 
309         // Parent ID to expected score
310         Map<String, Float> parentScores = new TreeMap<>();
311 
312         // Add a few random parents to ensure that the children's score is appropriately taken into account
313         for (int parentDocId = 0; parentDocId < parentDocs; ++parentDocId) {
314             String parent = Integer.toString(parentDocId);
315 
316             // Create the parent
317             Document parentDocument = new Document();
318 
319             parentDocument.add(new StringField(UidFieldMapper.NAME, Uid.createUid("parent", parent), Field.Store.YES));
320             parentDocument.add(new StringField(IdFieldMapper.NAME, parent, Field.Store.YES));
321             parentDocument.add(new StringField(TypeFieldMapper.NAME, "parent", Field.Store.NO));
322 
323             // add the parent to the index
324             writer.addDocument(parentDocument);
325 
326             int numChildDocs = scaledRandomIntBetween(1, 10);
327 
328             // forget any parent's previous scores
329             scorer.scores.clear();
330 
331             // associate children with the parent
332             for (int i = 0; i < numChildDocs; ++i) {
333                 int childScore = random().nextInt(128);
334 
335                 Document childDocument = new Document();
336 
337                 childDocument.add(new StringField(UidFieldMapper.NAME, Uid.createUid("child", Integer.toString(childDocId++)), Field.Store.NO));
338                 childDocument.add(new StringField(TypeFieldMapper.NAME, "child", Field.Store.NO));
339                 // parent association:
340                 childDocument.add(new StringField(ParentFieldMapper.NAME, Uid.createUid("parent", parent), Field.Store.NO));
341                 childDocument.add(new DoubleField(CHILD_SCORE_NAME, childScore, Field.Store.NO));
342 
343                 // remember the score to be calculated
344                 scorer.scores.add(childScore);
345 
346                 // add the associated child to the index
347                 writer.addDocument(childDocument);
348             }
349 
350             // this score that should be returned for this parent
351             parentScores.put(parent, scorer.score());
352         }
353 
354         writer.commit();
355 
356         IndexReader reader = DirectoryReader.open(writer, true);
357         IndexSearcher searcher = new IndexSearcher(reader);
358 
359         // setup to read the parent/child map
360         Engine.Searcher engineSearcher = new Engine.Searcher(ChildrenQueryTests.class.getSimpleName(), searcher);
361         ((TestSearchContext)context).setSearcher(new ContextIndexSearcher(context, engineSearcher));
362 
363         // child query that returns the score as the value of "childScore" for each child document, with the parent's score determined by the score type
364         QueryBuilder childQueryBuilder = functionScoreQuery(typeFilter("child")).add(new FieldValueFactorFunctionBuilder(CHILD_SCORE_NAME));
365         QueryBuilder queryBuilder = hasChildQuery("child", childQueryBuilder)
366                 .scoreType(scoreType.name().toLowerCase(Locale.ENGLISH))
367                 .setShortCircuitCutoff(parentDocs);
368 
369         // Perform the search for the documents using the selected score type
370         TopDocs docs = searcher.search(parseQuery(queryBuilder), parentDocs);
371         assertThat("Expected all parents", docs.totalHits, is(parentDocs));
372 
373         // score should be descending (just a sanity check)
374         float topScore = docs.scoreDocs[0].score;
375 
376         // ensure each score is returned as expected
377         for (int i = 0; i < parentDocs; ++i) {
378             ScoreDoc scoreDoc = docs.scoreDocs[i];
379             // get the ID from the document to get its expected score; remove it so we cannot double-count it
380             float score = parentScores.remove(reader.document(scoreDoc.doc).get(IdFieldMapper.NAME));
381 
382             // expect exact match
383             assertThat("Unexpected score", scoreDoc.score, is(score));
384             assertThat("Not descending", score, lessThanOrEqualTo(topScore));
385 
386             // it had better keep descending
387             topScore = score;
388         }
389 
390         reader.close();
391         writer.close();
392         directory.close();
393     }
