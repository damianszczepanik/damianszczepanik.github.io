45      @Test
46      public void testDiscreteHighlightingPerValue() throws Exception {
47          Directory dir = newDirectory();
48          IndexWriterConfig iwc = newIndexWriterConfig(new MockAnalyzer(random()));
49          iwc.setMergePolicy(newLogMergePolicy());
50          RandomIndexWriter iw = new RandomIndexWriter(random(), dir, iwc);
51  
52          FieldType offsetsType = new FieldType(TextField.TYPE_STORED);
53          offsetsType.setIndexOptions(IndexOptions.DOCS_AND_FREQS_AND_POSITIONS_AND_OFFSETS);
54          Field body = new Field("body", "", offsetsType);
55          final String firstValue = "This is a test. Just a test highlighting from postings highlighter.";
56          Document doc = new Document();
57          doc.add(body);
58          body.setStringValue(firstValue);
59  
60          final String secondValue = "This is the second value to perform highlighting on.";
61          Field body2 = new Field("body", "", offsetsType);
62          doc.add(body2);
63          body2.setStringValue(secondValue);
64  
65          final String thirdValue = "This is the third value to test highlighting with postings.";
66          Field body3 = new Field("body", "", offsetsType);
67          doc.add(body3);
68          body3.setStringValue(thirdValue);
69  
70          iw.addDocument(doc);
71  
72          IndexReader ir = iw.getReader();
73          iw.close();
74  
75          List<Object> fieldValues = new ArrayList<>();
76          fieldValues.add(firstValue);
77          fieldValues.add(secondValue);
78          fieldValues.add(thirdValue);
79  
80  
81          IndexSearcher searcher = newSearcher(ir);
82  
83          Query query = new TermQuery(new Term("body", "highlighting"));
84          BytesRef[] queryTerms = filterTerms(extractTerms(query), "body", true);
85  
86          TopDocs topDocs = searcher.search(query, null, 10, Sort.INDEXORDER);
87          assertThat(topDocs.totalHits, equalTo(1));
88          int docId = topDocs.scoreDocs[0].doc;
89  
90          //highlighting per value, considering whole values (simulating number_of_fragments=0)
91          CustomPostingsHighlighter highlighter = new CustomPostingsHighlighter(new CustomPassageFormatter("<b>", "</b>", new DefaultEncoder()), fieldValues, false, Integer.MAX_VALUE - 1, 0);
92          highlighter.setBreakIterator(new WholeBreakIterator());
93  
94          Snippet[] snippets = highlighter.highlightDoc("body", queryTerms, searcher, docId, 5);
95          assertThat(snippets.length, equalTo(1));
96          assertThat(snippets[0].getText(), equalTo("This is a test. Just a test <b>highlighting</b> from postings highlighter."));
97  
98          snippets = highlighter.highlightDoc("body", queryTerms, searcher, docId, 5);
99          assertThat(snippets.length, equalTo(1));
100         assertThat(snippets[0].getText(), equalTo("This is the second value to perform <b>highlighting</b> on."));
101 
102         snippets = highlighter.highlightDoc("body", queryTerms, searcher, docId, 5);
103         assertThat(snippets.length, equalTo(1));
104         assertThat(snippets[0].getText(), equalTo("This is the third value to test <b>highlighting</b> with postings."));
105 
106 
107         //let's try without whole break iterator as well, to prove that highlighting works the same when working per value (not optimized though)
108         highlighter = new CustomPostingsHighlighter(new CustomPassageFormatter("<b>", "</b>", new DefaultEncoder()), fieldValues, false, Integer.MAX_VALUE - 1, 0);
109 
110         snippets = highlighter.highlightDoc("body", queryTerms, searcher, docId, 5);
111         assertThat(snippets.length, equalTo(1));
112         assertThat(snippets[0].getText(), equalTo("Just a test <b>highlighting</b> from postings highlighter."));
113 
114         snippets = highlighter.highlightDoc("body", queryTerms, searcher, docId, 5);
115         assertThat(snippets.length, equalTo(1));
116         assertThat(snippets[0].getText(), equalTo("This is the second value to perform <b>highlighting</b> on."));

129     @Test
130     public void testDiscreteHighlightingScoring() throws Exception {
131 
132         Directory dir = newDirectory();
133         IndexWriterConfig iwc = newIndexWriterConfig(new MockAnalyzer(random()));
134         iwc.setMergePolicy(newLogMergePolicy());
135         RandomIndexWriter iw = new RandomIndexWriter(random(), dir, iwc);
136 
137         FieldType offsetsType = new FieldType(TextField.TYPE_STORED);
138         offsetsType.setIndexOptions(IndexOptions.DOCS_AND_FREQS_AND_POSITIONS_AND_OFFSETS);
139 
140         //good position but only one match
141         final String firstValue = "This is a test. Just a test1 highlighting from postings highlighter.";
142         Field body = new Field("body", "", offsetsType);
143         Document doc = new Document();
144         doc.add(body);
145         body.setStringValue(firstValue);
146 
147         //two matches, not the best snippet due to its length though
148         final String secondValue = "This is the second highlighting value to perform highlighting on a longer text that gets scored lower.";
149         Field body2 = new Field("body", "", offsetsType);
150         doc.add(body2);
151         body2.setStringValue(secondValue);
152 
153         //two matches and short, will be scored highest
154         final String thirdValue = "This is highlighting the third short highlighting value.";
155         Field body3 = new Field("body", "", offsetsType);
156         doc.add(body3);
157         body3.setStringValue(thirdValue);
158 
159         //one match, same as first but at the end, will be scored lower due to its position
160         final String fourthValue = "Just a test4 highlighting from postings highlighter.";
161         Field body4 = new Field("body", "", offsetsType);
162         doc.add(body4);
163         body4.setStringValue(fourthValue);
164 
165         iw.addDocument(doc);
166 
167         IndexReader ir = iw.getReader();
168         iw.close();
169 
170 
171         String firstHlValue = "Just a test1 <b>highlighting</b> from postings highlighter.";
172         String secondHlValue = "This is the second <b>highlighting</b> value to perform <b>highlighting</b> on a longer text that gets scored lower.";
173         String thirdHlValue = "This is <b>highlighting</b> the third short <b>highlighting</b> value.";
174         String fourthHlValue = "Just a test4 <b>highlighting</b> from postings highlighter.";
175 
176 
177         IndexSearcher searcher = newSearcher(ir);
178         Query query = new TermQuery(new Term("body", "highlighting"));
179         BytesRef[] queryTerms = filterTerms(extractTerms(query), "body", true);
180 
181         TopDocs topDocs = searcher.search(query, null, 10, Sort.INDEXORDER);
182         assertThat(topDocs.totalHits, equalTo(1));
183 
184         int docId = topDocs.scoreDocs[0].doc;
185 
186         List<Object> fieldValues = new ArrayList<>();
187         fieldValues.add(firstValue);
188         fieldValues.add(secondValue);
189         fieldValues.add(thirdValue);
190         fieldValues.add(fourthValue);
191 
192         boolean mergeValues = true;
193         CustomPostingsHighlighter highlighter = new CustomPostingsHighlighter(new CustomPassageFormatter("<b>", "</b>", new DefaultEncoder()), fieldValues, mergeValues, Integer.MAX_VALUE-1, 0);
194         Snippet[] snippets = highlighter.highlightDoc("body", queryTerms, searcher, docId, 5);
195 
196         assertThat(snippets.length, equalTo(4));
197 
198         assertThat(snippets[0].getText(), equalTo(firstHlValue));
199         assertThat(snippets[1].getText(), equalTo(secondHlValue));
200         assertThat(snippets[2].getText(), equalTo(thirdHlValue));
201         assertThat(snippets[3].getText(), equalTo(fourthHlValue));
202 
203 
204         //Let's highlight each separate value and check how the snippets are scored
205         mergeValues = false;
206         highlighter = new CustomPostingsHighlighter(new CustomPassageFormatter("<b>", "</b>", new DefaultEncoder()), fieldValues, mergeValues, Integer.MAX_VALUE-1, 0);
207         List<Snippet> snippets2 = new ArrayList<>();
208         for (int i = 0; i < fieldValues.size(); i++) {
209             snippets2.addAll(Arrays.asList(highlighter.highlightDoc("body", queryTerms, searcher, docId, 5)));
210         }
211 
212         assertThat(snippets2.size(), equalTo(4));
213         assertThat(snippets2.get(0).getText(), equalTo(firstHlValue));
214         assertThat(snippets2.get(1).getText(), equalTo(secondHlValue));
215         assertThat(snippets2.get(2).getText(), equalTo(thirdHlValue));
216         assertThat(snippets2.get(3).getText(), equalTo(fourthHlValue));
217 
218         Comparator <Snippet> comparator = new Comparator<Snippet>() {
219             @Override
220             public int compare(Snippet o1, Snippet o2) {
221                 return (int)Math.signum(o1.getScore() - o2.getScore());
222             }
223         };
224 
225         //sorting both groups of snippets
226         Arrays.sort(snippets, comparator);
227         Collections.sort(snippets2, comparator);
228 
229         //checking that the snippets are in the same order, regardless of whether we used per value discrete highlighting or not
230         //we can't compare the scores directly since they are slightly different due to the multiValued separator added when merging values together
231         //That causes slightly different lengths and start offsets, thus a slightly different score.
232         //Anyways, that's not an issue. What's important is that the score is computed the same way, so that the produced order is always the same.
233         for (int i = 0; i < snippets.length; i++) {
234             assertThat(snippets[i].getText(), equalTo(snippets2.get(i).getText()));
235         }
236 
237         ir.close();
238         dir.close();
239     }

463     private static SortedSet<Term> extractTerms(Query query) {
464         SortedSet<Term> queryTerms = new TreeSet<>();
465         query.extractTerms(queryTerms);
466         return queryTerms;
467     }
468 
469     private static BytesRef[] filterTerms(SortedSet<Term> queryTerms, String field, boolean requireFieldMatch) {
470         SortedSet<Term> fieldTerms;
471         if (requireFieldMatch) {
472             Term floor = new Term(field, "");
473             Term ceiling = new Term(field, UnicodeUtil.BIG_TERM);
474             fieldTerms = queryTerms.subSet(floor, ceiling);
475         } else {
476             fieldTerms = queryTerms;
477         }
478 
479         BytesRef terms[] = new BytesRef[fieldTerms.size()];
480         int termUpto = 0;
481         for(Term term : fieldTerms) {
482             terms[termUpto++] = term.bytes();
483         }
484 
485         return terms;
486     }
