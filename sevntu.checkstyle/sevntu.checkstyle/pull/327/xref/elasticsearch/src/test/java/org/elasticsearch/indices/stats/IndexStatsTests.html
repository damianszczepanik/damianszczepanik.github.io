213     @Test
214     public void testQueryCache() throws Exception {
215         assertAcked(client().admin().indices().prepareCreate("idx").setSettings(IndicesQueryCache.INDEX_CACHE_QUERY_ENABLED, true).get());
216         ensureGreen();
217 
218         // index docs until we have at least one doc on each shard, otherwise, our tests will not work
219         // since refresh will not refresh anything on a shard that has 0 docs and its search response get cached
220         int pageDocs = randomIntBetween(2, 100);
221         int numDocs = 0;
222         int counter = 0;
223         while (true) {
224             IndexRequestBuilder[] builders = new IndexRequestBuilder[pageDocs];
225             for (int i = 0; i < pageDocs; ++i) {
226                 builders[i] = client().prepareIndex("idx", "type", Integer.toString(counter++)).setSource(jsonBuilder()
227                         .startObject()
228                         .field("common", "field")
229                         .field("str_value", "s" + i)
230                         .endObject());
231             }
232             indexRandom(true, builders);
233             numDocs += pageDocs;
234 
235             boolean allHaveDocs = true;
236             for (ShardStats stats : client().admin().indices().prepareStats("idx").setDocs(true).get().getShards()) {
237                 if (stats.getStats().getDocs().getCount() == 0) {
238                     allHaveDocs = false;
239                     break;
240                 }
241             }
242 
243             if (allHaveDocs) {
244                 break;
245             }
246         }
247 
248         assertThat(client().admin().indices().prepareStats("idx").setQueryCache(true).get().getTotal().getQueryCache().getMemorySizeInBytes(), equalTo(0l));
249         assertThat(client().admin().indices().prepareStats("idx").setQueryCache(true).get().getTotal().getQueryCache().getHitCount(), equalTo(0l));
250         assertThat(client().admin().indices().prepareStats("idx").setQueryCache(true).get().getTotal().getQueryCache().getMissCount(), equalTo(0l));
251         for (int i = 0; i < 10; i++) {
252             assertThat(client().prepareSearch("idx").setSearchType(SearchType.COUNT).get().getHits().getTotalHits(), equalTo((long) numDocs));
253             assertThat(client().admin().indices().prepareStats("idx").setQueryCache(true).get().getTotal().getQueryCache().getMemorySizeInBytes(), greaterThan(0l));
254         }
255         assertThat(client().admin().indices().prepareStats("idx").setQueryCache(true).get().getTotal().getQueryCache().getHitCount(), greaterThan(0l));
256         assertThat(client().admin().indices().prepareStats("idx").setQueryCache(true).get().getTotal().getQueryCache().getMissCount(), greaterThan(0l));
257 
258         // index the data again...
259         IndexRequestBuilder[] builders = new IndexRequestBuilder[numDocs];
260         for (int i = 0; i < numDocs; ++i) {
261             builders[i] = client().prepareIndex("idx", "type", Integer.toString(i)).setSource(jsonBuilder()
262                     .startObject()
263                     .field("common", "field")
264                     .field("str_value", "s" + i)
265                     .endObject());
266         }
267         indexRandom(true, builders);
268         refresh();
269         assertBusy(new Runnable() {
270             @Override
271             public void run() {
272                 assertThat(client().admin().indices().prepareStats("idx").setQueryCache(true).get().getTotal().getQueryCache().getMemorySizeInBytes(), equalTo(0l));
273             }
274         });
275 
276         for (int i = 0; i < 10; i++) {
277             assertThat(client().prepareSearch("idx").setSearchType(SearchType.COUNT).get().getHits().getTotalHits(), equalTo((long) numDocs));
278             assertThat(client().admin().indices().prepareStats("idx").setQueryCache(true).get().getTotal().getQueryCache().getMemorySizeInBytes(), greaterThan(0l));
279         }
280 
281         client().admin().indices().prepareClearCache().setQueryCache(true).get(); // clean the cache
282         assertThat(client().admin().indices().prepareStats("idx").setQueryCache(true).get().getTotal().getQueryCache().getMemorySizeInBytes(), equalTo(0l));
283 
284         // test explicit request parameter
285 
286         assertThat(client().prepareSearch("idx").setSearchType(SearchType.COUNT).setQueryCache(false).get().getHits().getTotalHits(), equalTo((long) numDocs));
287         assertThat(client().admin().indices().prepareStats("idx").setQueryCache(true).get().getTotal().getQueryCache().getMemorySizeInBytes(), equalTo(0l));
288 
289         assertThat(client().prepareSearch("idx").setSearchType(SearchType.COUNT).setQueryCache(true).get().getHits().getTotalHits(), equalTo((long) numDocs));
290         assertThat(client().admin().indices().prepareStats("idx").setQueryCache(true).get().getTotal().getQueryCache().getMemorySizeInBytes(), greaterThan(0l));
291 
292         // set the index level setting to false, and see that the reverse works
293 
294         client().admin().indices().prepareClearCache().setQueryCache(true).get(); // clean the cache
295         assertAcked(client().admin().indices().prepareUpdateSettings("idx").setSettings(ImmutableSettings.builder().put(IndicesQueryCache.INDEX_CACHE_QUERY_ENABLED, false)));
296 
297         assertThat(client().prepareSearch("idx").setSearchType(SearchType.COUNT).get().getHits().getTotalHits(), equalTo((long) numDocs));
298         assertThat(client().admin().indices().prepareStats("idx").setQueryCache(true).get().getTotal().getQueryCache().getMemorySizeInBytes(), equalTo(0l));
299 
300         assertThat(client().prepareSearch("idx").setSearchType(SearchType.COUNT).setQueryCache(true).get().getHits().getTotalHits(), equalTo((long) numDocs));
301         assertThat(client().admin().indices().prepareStats("idx").setQueryCache(true).get().getTotal().getQueryCache().getMemorySizeInBytes(), greaterThan(0l));
302     }

339     public void throttleStats() throws Exception {
340         assertAcked(prepareCreate("test")
341                     .setSettings(ImmutableSettings.builder()
342                                  .put(AbstractIndexStore.INDEX_STORE_THROTTLE_TYPE, "merge")
343                                  .put(IndexMetaData.SETTING_NUMBER_OF_SHARDS, "1")
344                                  .put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, "0")
345                                  .put(TieredMergePolicyProvider.INDEX_MERGE_POLICY_MAX_MERGE_AT_ONCE, "2")
346                                  .put(TieredMergePolicyProvider.INDEX_MERGE_POLICY_SEGMENTS_PER_TIER, "2")
347                                  .put(ConcurrentMergeSchedulerProvider.MAX_THREAD_COUNT, "1")
348                                  .put(ConcurrentMergeSchedulerProvider.MAX_MERGE_COUNT, "1")
349                                  .put("index.merge.policy.type", "tiered")
350 
351                                  ));
352         ensureGreen();
353         long termUpto = 0;
354         IndicesStatsResponse stats;
355         // make sure we see throttling kicking in:
356         boolean done = false;
357         long start = System.currentTimeMillis();
358         while (!done) {
359             for(int i=0; i<100; i++) {
360                 // Provoke slowish merging by making many unique terms:
361                 StringBuilder sb = new StringBuilder();
362                 for(int j=0; j<100; j++) {
363                     sb.append(' ');
364                     sb.append(termUpto++);
365                 }
366                 client().prepareIndex("test", "type", ""+termUpto).setSource("field" + (i%10), sb.toString()).get();
367                 if (i % 2 == 0) {
368                     refresh();
369                 }
370             }
371             refresh();
372             stats = client().admin().indices().prepareStats().execute().actionGet();
373             //nodesStats = client().admin().cluster().prepareNodesStats().setIndices(true).get();
374             done = stats.getPrimaries().getIndexing().getTotal().getThrottleTimeInMillis() > 0;
375             if (System.currentTimeMillis() - start > 300*1000) { //Wait 5 minutes for throttling to kick in
376                 fail("index throttling didn't kick in after 5 minutes of intense merging");
377             }
378         }
379 
380         // Optimize & flush and wait; else we sometimes get a "Delete Index failed - not acked"
381         // when ElasticsearchIntegrationTest.after tries to remove indices created by the test:
382         logger.info("test: now optimize");
383         client().admin().indices().prepareOptimize("test").get();
384         flush();
385         logger.info("test: test done");
386     }

390         createIndex("test1", "test2");
391         ensureGreen();
392 
393         client().prepareIndex("test1", "type1", Integer.toString(1)).setSource("field", "value").execute().actionGet();
394         client().prepareIndex("test1", "type2", Integer.toString(1)).setSource("field", "value").execute().actionGet();

396 
397         refresh();
398 
399         NumShards test1 = getNumShards("test1");
400         long test1ExpectedWrites = 2 * test1.dataCopies;
401         NumShards test2 = getNumShards("test2");
402         long test2ExpectedWrites = test2.dataCopies;
403         long totalExpectedWrites = test1ExpectedWrites + test2ExpectedWrites;
404

409         assertThat(stats.getPrimaries().getIndexing().getTotal().isThrottled(), equalTo(false));
410         assertThat(stats.getPrimaries().getIndexing().getTotal().getThrottleTimeInMillis(), equalTo(0l));
411         assertThat(stats.getTotal().getIndexing().getTotal().getIndexCount(), equalTo(totalExpectedWrites));
412         assertThat(stats.getTotal().getStore(), notNullValue());
413         assertThat(stats.getTotal().getMerge(), notNullValue());
414         assertThat(stats.getTotal().getFlush(), notNullValue());
415         assertThat(stats.getTotal().getRefresh(), notNullValue());
416 
417         assertThat(stats.getIndex("test1").getPrimaries().getDocs().getCount(), equalTo(2l));
418         assertThat(stats.getIndex("test1").getTotal().getDocs().getCount(), equalTo(test1ExpectedWrites));
