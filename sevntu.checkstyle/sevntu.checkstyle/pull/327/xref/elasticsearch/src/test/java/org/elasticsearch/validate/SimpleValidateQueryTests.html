67      @BeforeClass
68      public static void enableFilterCache() {
69          assert hasFilterCache == null;
70          hasFilterCache = randomBoolean();
71      }
72  
73      @Override
74      protected Settings nodeSettings(int nodeOrdinal) {
75          ImmutableSettings.Builder builder = ImmutableSettings.settingsBuilder().put(super.nodeSettings(nodeOrdinal));
76          if (hasFilterCache) {
77              // cache everything
78              builder.put(FilterCacheModule.FilterCacheSettings.FILTER_CACHE_TYPE, WeightedFilterCache.class)
79                     .put(AutoFilterCachingPolicy.AGGRESSIVE_CACHING_SETTINGS);
80          } else {
81              builder.put(FilterCacheModule.FilterCacheSettings.FILTER_CACHE_TYPE, NoneFilterCache.class);
82          }
83          return builder.build();
84      }

87      public void simpleValidateQuery() throws Exception {
88          createIndex("test");
89          ensureGreen();
90          client().admin().indices().preparePutMapping("test").setType("type1")
91                  .setSource(XContentFactory.jsonBuilder().startObject().startObject("type1").startObject("properties")
92                          .startObject("foo").field("type", "string").endObject()

94                          .endObject().endObject().endObject())
95                  .execute().actionGet();

97          refresh();
98  
99          assertThat(client().admin().indices().prepareValidateQuery("test").setSource("foo".getBytes(Charsets.UTF_8)).execute().actionGet().isValid(), equalTo(false));
100         assertThat(client().admin().indices().prepareValidateQuery("test").setQuery(QueryBuilders.queryStringQuery("_id:1")).execute().actionGet().isValid(), equalTo(true));
101         assertThat(client().admin().indices().prepareValidateQuery("test").setQuery(QueryBuilders.queryStringQuery("_i:d:1")).execute().actionGet().isValid(), equalTo(false));
102 
103         assertThat(client().admin().indices().prepareValidateQuery("test").setQuery(QueryBuilders.queryStringQuery("foo:1")).execute().actionGet().isValid(), equalTo(true));
104         assertThat(client().admin().indices().prepareValidateQuery("test").setQuery(QueryBuilders.queryStringQuery("bar:hey")).execute().actionGet().isValid(), equalTo(false));
105 
106         assertThat(client().admin().indices().prepareValidateQuery("test").setQuery(QueryBuilders.queryStringQuery("nonexistent:hello")).execute().actionGet().isValid(), equalTo(true));
107 
108         assertThat(client().admin().indices().prepareValidateQuery("test").setQuery(QueryBuilders.queryStringQuery("foo:1 AND")).execute().actionGet().isValid(), equalTo(false));
109     }

111     private static String filter(String uncachedFilter) {
112         String filter = uncachedFilter;
113         if (hasFilterCache) {
114             filter = "cache(" + filter + ")";
115         }
116         return filter;
117     }

119     @Test
120     public void explainValidateQuery() throws Exception {
121         createIndex("test");
122         ensureGreen();
123         client().admin().indices().preparePutMapping("test").setType("type1")
124                 .setSource(XContentFactory.jsonBuilder().startObject().startObject("type1").startObject("properties")
125                         .startObject("foo").field("type", "string").endObject()
126                         .startObject("bar").field("type", "integer").endObject()
127                         .startObject("baz").field("type", "string").field("analyzer", "snowball").endObject()
128                         .startObject("pin").startObject("properties").startObject("location").field("type", "geo_point").endObject().endObject().endObject()
129                         .endObject().endObject().endObject())
130                 .execute().actionGet();
131         client().admin().indices().preparePutMapping("test").setType("child-type")
132                 .setSource(XContentFactory.jsonBuilder().startObject().startObject("child-type")
133                         .startObject("_parent").field("type", "type1").endObject()
134                         .startObject("properties")
135                         .startObject("foo").field("type", "string").endObject()
136                         .endObject()
137                         .endObject().endObject())
138                 .execute().actionGet();
139 
140         refresh();
141 
142         ValidateQueryResponse response;
143         response = client().admin().indices().prepareValidateQuery("test")
144                 .setSource("foo".getBytes(Charsets.UTF_8))
145                 .setExplain(true)
146                 .execute().actionGet();
147         assertThat(response.isValid(), equalTo(false));
148         assertThat(response.getQueryExplanation().size(), equalTo(1));
149         assertThat(response.getQueryExplanation().get(0).getError(), containsString("Failed to parse"));
150         assertThat(response.getQueryExplanation().get(0).getExplanation(), nullValue());
151 
152         final String typeFilter = filter("_type:type1");
153         assertExplanation(QueryBuilders.queryStringQuery("_id:1"), equalTo("filtered(ConstantScore(_uid:type1#1))->" + typeFilter));
154 
155         assertExplanation(QueryBuilders.idsQuery("type1").addIds("1").addIds("2"),
156                 equalTo("filtered(ConstantScore(_uid:type1#1 _uid:type1#2))->" + typeFilter));
157 
158         assertExplanation(QueryBuilders.queryStringQuery("foo"), equalTo("filtered(_all:foo)->" + typeFilter));
159 
160         assertExplanation(QueryBuilders.filteredQuery(
161                 QueryBuilders.termQuery("foo", "1"),
162                 FilterBuilders.orFilter(
163                         FilterBuilders.termFilter("bar", "2"),
164                         FilterBuilders.termFilter("baz", "3")
165                 )
166         ), equalTo("filtered(filtered(foo:1)->" + filter(filter("bar:[2 TO 2]") + " " + filter("baz:3")) + ")->" + typeFilter));
167 
168         assertExplanation(QueryBuilders.filteredQuery(
169                 QueryBuilders.termQuery("foo", "1"),
170                 FilterBuilders.orFilter(
171                         FilterBuilders.termFilter("bar", "2")
172                 )
173         ), equalTo("filtered(filtered(foo:1)->" + filter(filter("bar:[2 TO 2]")) + ")->" + typeFilter));
174 
175         assertExplanation(QueryBuilders.filteredQuery(
176                 QueryBuilders.matchAllQuery(),
177                 FilterBuilders.geoPolygonFilter("pin.location")
178                         .addPoint(40, -70)
179                         .addPoint(30, -80)
180                         .addPoint(20, -90)
181                         .addPoint(40, -70)    // closing polygon
182         ), equalTo("filtered(ConstantScore(" + filter("GeoPolygonFilter(pin.location, [[40.0, -70.0], [30.0, -80.0], [20.0, -90.0], [40.0, -70.0]]))") + ")->" + typeFilter));
183 
184         assertExplanation(QueryBuilders.constantScoreQuery(FilterBuilders.geoBoundingBoxFilter("pin.location")
185                 .topLeft(40, -80)
186                 .bottomRight(20, -70)
187         ), equalTo("filtered(ConstantScore(" + filter("GeoBoundingBoxFilter(pin.location, [40.0, -80.0], [20.0, -70.0]))") + ")->" + typeFilter));
188 
189         assertExplanation(QueryBuilders.constantScoreQuery(FilterBuilders.geoDistanceFilter("pin.location")
190                 .lat(10).lon(20).distance(15, DistanceUnit.DEFAULT).geoDistance(GeoDistance.PLANE)
191         ), equalTo("filtered(ConstantScore(" + filter("GeoDistanceFilter(pin.location, PLANE, 15.0, 10.0, 20.0))") + ")->" + typeFilter));
192 
193         assertExplanation(QueryBuilders.constantScoreQuery(FilterBuilders.geoDistanceFilter("pin.location")
194                 .lat(10).lon(20).distance(15, DistanceUnit.DEFAULT).geoDistance(GeoDistance.PLANE)
195         ), equalTo("filtered(ConstantScore(" + filter("GeoDistanceFilter(pin.location, PLANE, 15.0, 10.0, 20.0))") + ")->" + typeFilter));
196 
197         assertExplanation(QueryBuilders.constantScoreQuery(FilterBuilders.geoDistanceRangeFilter("pin.location")
198                 .lat(10).lon(20).from("15m").to("25m").geoDistance(GeoDistance.PLANE)
199         ), equalTo("filtered(ConstantScore(" + filter("GeoDistanceRangeFilter(pin.location, PLANE, [15.0 - 25.0], 10.0, 20.0))") + ")->" + typeFilter));
200 
201         assertExplanation(QueryBuilders.constantScoreQuery(FilterBuilders.geoDistanceRangeFilter("pin.location")
202                 .lat(10).lon(20).from("15miles").to("25miles").geoDistance(GeoDistance.PLANE)
203         ), equalTo("filtered(ConstantScore(" + filter("GeoDistanceRangeFilter(pin.location, PLANE, [" + DistanceUnit.DEFAULT.convert(15.0, DistanceUnit.MILES) + " - " + DistanceUnit.DEFAULT.convert(25.0, DistanceUnit.MILES) + "], 10.0, 20.0))") + ")->" + typeFilter));
204 
205         assertExplanation(QueryBuilders.filteredQuery(
206                 QueryBuilders.termQuery("foo", "1"),
207                 FilterBuilders.andFilter(
208                         FilterBuilders.termFilter("bar", "2"),
209                         FilterBuilders.termFilter("baz", "3")
210                 )
211         ), equalTo("filtered(filtered(foo:1)->" + filter("+" + filter("bar:[2 TO 2]") + " +" + filter("baz:3")) + ")->" + typeFilter));
212 
213         assertExplanation(QueryBuilders.constantScoreQuery(FilterBuilders.termsFilter("foo", "1", "2", "3")),
214                 equalTo("filtered(ConstantScore(" + filter("foo:1 foo:2 foo:3") + "))->" + typeFilter));
215 
216         assertExplanation(QueryBuilders.constantScoreQuery(FilterBuilders.notFilter(FilterBuilders.termFilter("foo", "bar"))),
217                 equalTo("filtered(ConstantScore(" + filter("NotFilter(" + filter("foo:bar") + ")") + "))->" + typeFilter));
218 
219         assertExplanation(QueryBuilders.filteredQuery(
220                 QueryBuilders.termQuery("foo", "1"),
221                 FilterBuilders.hasChildFilter(
222                         "child-type",
223                         QueryBuilders.matchQuery("foo", "1")
224                 )
225         ), equalTo("filtered(filtered(foo:1)->CustomQueryWrappingFilter(child_filter[child-type/type1](filtered(foo:1)->" + filter("_type:child-type") + ")))->" + typeFilter));
226 
227         assertExplanation(QueryBuilders.filteredQuery(
228                 QueryBuilders.termQuery("foo", "1"),
229                 FilterBuilders.scriptFilter("true")
230         ), equalTo("filtered(filtered(foo:1)->" + filter("ScriptFilter(true)") + ")->" + typeFilter));
231 
232     }

273     @Test //https://github.com/elasticsearch/elasticsearch/issues/3629
274     public void explainDateRangeInQueryString() {
275         assertAcked(prepareCreate("test").setSettings(ImmutableSettings.settingsBuilder()
276                 .put(indexSettings())
277                 .put("index.number_of_shards", 1)));
278 
279         String aMonthAgo = ISODateTimeFormat.yearMonthDay().print(new DateTime(DateTimeZone.UTC).minusMonths(1));
280         String aMonthFromNow = ISODateTimeFormat.yearMonthDay().print(new DateTime(DateTimeZone.UTC).plusMonths(1));
281 
282         client().prepareIndex("test", "type", "1").setSource("past", aMonthAgo, "future", aMonthFromNow).get();
283 
284         refresh();
285 
286         ValidateQueryResponse response = client().admin().indices().prepareValidateQuery()
287                 .setQuery(queryStringQuery("past:[now-2M/d TO now/d]")).setExplain(true).get();
288 
289         assertNoFailures(response);
290         assertThat(response.getQueryExplanation().size(), equalTo(1));
291         assertThat(response.getQueryExplanation().get(0).getError(), nullValue());
292         DateTime twoMonthsAgo = new DateTime(DateTimeZone.UTC).minusMonths(2).withTimeAtStartOfDay();
293         DateTime now = new DateTime(DateTimeZone.UTC).plusDays(1).withTimeAtStartOfDay().minusMillis(1);
294         assertThat(response.getQueryExplanation().get(0).getExplanation(),
295                 equalTo("past:[" + twoMonthsAgo.getMillis() + " TO " + now.getMillis() + "]"));
296         assertThat(response.isValid(), equalTo(true));
297     }

333         assertAcked(prepareCreate("test").setSettings(
334                 ImmutableSettings.settingsBuilder().put(indexSettings())
335                         .put("index.analysis.filter.syns.type", "synonym")
336                         .putArray("index.analysis.filter.syns.synonyms", "one,two")
337                         .put("index.analysis.analyzer.syns.tokenizer", "standard")
338                         .putArray("index.analysis.analyzer.syns.filter", "syns")
339                     ).addMapping("test", "field","type=string,analyzer=syns"));
340         ensureGreen();
341 
342         ValidateQueryResponse validateQueryResponse = client().admin().indices().prepareValidateQuery("test")
343                 .setQuery(QueryBuilders.matchPhrasePrefixQuery("field", "foo")).setExplain(true).get();
344         assertThat(validateQueryResponse.isValid(), equalTo(true));
345         assertThat(validateQueryResponse.getQueryExplanation().size(), equalTo(1));
346         assertThat(validateQueryResponse.getQueryExplanation().get(0).getExplanation(), containsString("field:\"foo*\""));
347 
348         validateQueryResponse = client().admin().indices().prepareValidateQuery("test")
349                 .setQuery(QueryBuilders.matchPhrasePrefixQuery("field", "foo bar")).setExplain(true).get();
350         assertThat(validateQueryResponse.isValid(), equalTo(true));
351         assertThat(validateQueryResponse.getQueryExplanation().size(), equalTo(1));
352         assertThat(validateQueryResponse.getQueryExplanation().get(0).getExplanation(), containsString("field:\"foo bar*\""));
353 
354         // Stacked tokens
355         validateQueryResponse = client().admin().indices().prepareValidateQuery("test")
356                 .setQuery(QueryBuilders.matchPhrasePrefixQuery("field", "one bar")).setExplain(true).get();
357         assertThat(validateQueryResponse.isValid(), equalTo(true));
358         assertThat(validateQueryResponse.getQueryExplanation().size(), equalTo(1));
359         assertThat(validateQueryResponse.getQueryExplanation().get(0).getExplanation(), containsString("field:\"(one two) bar*\""));
360 
361         validateQueryResponse = client().admin().indices().prepareValidateQuery("test")
362                 .setQuery(QueryBuilders.matchPhrasePrefixQuery("field", "foo one")).setExplain(true).get();
363         assertThat(validateQueryResponse.isValid(), equalTo(true));
364         assertThat(validateQueryResponse.getQueryExplanation().size(), equalTo(1));
365         assertThat(validateQueryResponse.getQueryExplanation().get(0).getExplanation(), containsString("field:\"foo (one* two*)\""));
366     }

386     private void assertExplanation(QueryBuilder queryBuilder, Matcher<String> matcher) {
387         ValidateQueryResponse response = client().admin().indices().prepareValidateQuery("test")
388                 .setTypes("type1")
389                 .setQuery(queryBuilder)
