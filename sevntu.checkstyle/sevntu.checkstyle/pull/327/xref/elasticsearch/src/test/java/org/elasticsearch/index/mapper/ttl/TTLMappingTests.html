61      public void testEnabled() throws Exception {
62          String mapping = XContentFactory.jsonBuilder().startObject().startObject("type")
63                  .startObject("_ttl").field("enabled", "yes").endObject()
64                  .endObject().endObject().string();
65          DocumentMapper docMapper = createIndex("test").mapperService().documentMapperParser().parse(mapping);
66          BytesReference source = XContentFactory.jsonBuilder()
67                  .startObject()
68                  .field("field", "value")
69                  .endObject()
70                  .bytes();
71          ParsedDocument doc = docMapper.parse(SourceToParse.source(source).type("type").id("1").ttl(Long.MAX_VALUE));
72  
73          assertThat(doc.rootDoc().getField("_ttl").fieldType().stored(), equalTo(true));
74          assertNotSame(IndexOptions.NONE, doc.rootDoc().getField("_ttl").fieldType().indexOptions());
75          assertThat(doc.rootDoc().getField("_ttl").tokenStream(docMapper.mappers().indexAnalyzer(), null), notNullValue());
76      }

102     @Test
103     public void testThatEnablingTTLFieldOnMergeWorks() throws Exception {
104         String mappingWithoutTtl = XContentFactory.jsonBuilder().startObject().startObject("type")
105                 .startObject("properties").field("field").startObject().field("type", "string").endObject().endObject()
106                 .endObject().endObject().string();

108         String mappingWithTtl = XContentFactory.jsonBuilder().startObject().startObject("type")
109                 .startObject("_ttl")

114 
115         DocumentMapperParser parser = createIndex("test").mapperService().documentMapperParser();
116         DocumentMapper mapperWithoutTtl = parser.parse(mappingWithoutTtl);
117         DocumentMapper mapperWithTtl = parser.parse(mappingWithTtl);
118 
119         DocumentMapper.MergeFlags mergeFlags = DocumentMapper.MergeFlags.mergeFlags().simulate(false);
120         DocumentMapper.MergeResult mergeResult = mapperWithoutTtl.merge(mapperWithTtl, mergeFlags);

168     @Test
169     public void testThatDisablingTTLReportsConflictOnCluster() throws Exception {
170         String mappingWithTtl = getMappingWithTtlEnabled().string();
171         String mappingWithTtlDisabled = getMappingWithTtlDisabled().string();
172         assertAcked(client().admin().indices().prepareCreate("testindex").addMapping("type", mappingWithTtl));
173         GetMappingsResponse mappingsBeforeUpdateResponse = client().admin().indices().prepareGetMappings("testindex").addTypes("type").get();
174         try {
175             client().admin().indices().preparePutMapping("testindex").setSource(mappingWithTtlDisabled).setType("type").get();
176             fail();
177         } catch (MergeMappingException mme) {
178             assertThat(mme.getDetailedMessage(), containsString("_ttl cannot be disabled once it was enabled."));
179         }
180         GetMappingsResponse mappingsAfterUpdateResponse = client().admin().indices().prepareGetMappings("testindex").addTypes("type").get();
181         assertThat(mappingsBeforeUpdateResponse.getMappings().get("testindex").get("type").source(), equalTo(mappingsAfterUpdateResponse.getMappings().get("testindex").get("type").source()));
182     }

185     public void testThatEnablingTTLAfterFirstDisablingWorks() throws Exception {

188         assertAcked(client().admin().indices().prepareCreate("testindex").addMapping("type", withTtlDisabled));

214         XContentBuilder mappingWithTtlEnabled = getMappingWithTtlEnabled("7d");
215         IndexService indexService = createIndex("testindex", ImmutableSettings.settingsBuilder().build(), "type", mappingWithTtlEnabled);
216         XContentBuilder mappingWithOnlyDefaultSet = getMappingWithOnlyTtlDefaultSet("6m");

236     @Test
237     public void testThatSimulatedMergingLeavesStateUntouched() throws Exception {
238 
239         //check if default ttl changed when simulate set to true
240         XContentBuilder mappingWithTtl = getMappingWithTtlEnabled("6d");
241         IndexService indexService = createIndex("testindex", ImmutableSettings.settingsBuilder().build(), "type", mappingWithTtl);
242         CompressedString mappingBeforeMerge = indexService.mapperService().documentMapper("type").mappingSource();
243         XContentBuilder mappingWithTtlDifferentDefault = getMappingWithTtlEnabled("7d");
244         DocumentMapper.MergeResult mergeResult = indexService.mapperService().documentMapper("type").merge(indexService.mapperService().parse("type", new CompressedString(mappingWithTtlDifferentDefault.string()), true), DocumentMapper.MergeFlags.mergeFlags().simulate(true));
245         assertFalse(mergeResult.hasConflicts());
246         // make sure simulate flag actually worked - no mappings applied
247         CompressedString mappingAfterMerge = indexService.mapperService().documentMapper("type").refreshSource();
248         assertThat(mappingAfterMerge, equalTo(mappingBeforeMerge));
249 
250         client().admin().indices().prepareDelete("testindex").get();
251         // check if enabled changed when simulate set to true
252         XContentBuilder mappingWithoutTtl = getMappingWithTtlDisabled();
253         indexService = createIndex("testindex", ImmutableSettings.settingsBuilder().build(), "type", mappingWithoutTtl);
254         mappingBeforeMerge = indexService.mapperService().documentMapper("type").mappingSource();
255         XContentBuilder mappingWithTtlEnabled = getMappingWithTtlEnabled();
256         mergeResult = indexService.mapperService().documentMapper("type").merge(indexService.mapperService().parse("type", new CompressedString(mappingWithTtlEnabled.string()), true), DocumentMapper.MergeFlags.mergeFlags().simulate(true));
257         assertFalse(mergeResult.hasConflicts());
258         // make sure simulate flag actually worked - no mappings applied
259         mappingAfterMerge = indexService.mapperService().documentMapper("type").refreshSource();
260         assertThat(mappingAfterMerge, equalTo(mappingBeforeMerge));
261 
262         client().admin().indices().prepareDelete("testindex").get();
263         // check if enabled changed when simulate set to true
264         mappingWithoutTtl = getMappingWithTtlDisabled("6d");
265         indexService = createIndex("testindex", ImmutableSettings.settingsBuilder().build(), "type", mappingWithoutTtl);
266         mappingBeforeMerge = indexService.mapperService().documentMapper("type").mappingSource();
267         mappingWithTtlEnabled = getMappingWithTtlEnabled("7d");
268         mergeResult = indexService.mapperService().documentMapper("type").merge(indexService.mapperService().parse("type", new CompressedString(mappingWithTtlEnabled.string()), true), DocumentMapper.MergeFlags.mergeFlags().simulate(true));
269         assertFalse(mergeResult.hasConflicts());
270         // make sure simulate flag actually worked - no mappings applied
271         mappingAfterMerge = indexService.mapperService().documentMapper("type").refreshSource();
272         assertThat(mappingAfterMerge, equalTo(mappingBeforeMerge));
273 
274         client().admin().indices().prepareDelete("testindex").get();
275         // check if switching simulate flag off works
276         mappingWithoutTtl = getMappingWithTtlDisabled("6d");
277         indexService = createIndex("testindex", ImmutableSettings.settingsBuilder().build(), "type", mappingWithoutTtl);
278         mappingWithTtlEnabled = getMappingWithTtlEnabled("7d");
279         mergeResult = indexService.mapperService().documentMapper("type").merge(indexService.mapperService().parse("type", new CompressedString(mappingWithTtlEnabled.string()), true), DocumentMapper.MergeFlags.mergeFlags().simulate(false));
280         assertFalse(mergeResult.hasConflicts());
281         // make sure simulate flag actually worked - mappings applied
282         mappingAfterMerge = indexService.mapperService().documentMapper("type").refreshSource();
283         assertThat(mappingAfterMerge, equalTo(new CompressedString("{\"type\":{\"_ttl\":{\"enabled\":true,\"default\":604800000},\"properties\":{\"field\":{\"type\":\"string\"}}}}")));
284 
285         client().admin().indices().prepareDelete("testindex").get();
286         // check if switching simulate flag off works if nothing was applied in the beginning
287         indexService = createIndex("testindex", ImmutableSettings.settingsBuilder().build(), "type");
288         mappingWithTtlEnabled = getMappingWithTtlEnabled("7d");
289         mergeResult = indexService.mapperService().documentMapper("type").merge(indexService.mapperService().parse("type", new CompressedString(mappingWithTtlEnabled.string()), true), DocumentMapper.MergeFlags.mergeFlags().simulate(false));
290         assertFalse(mergeResult.hasConflicts());
291         // make sure simulate flag actually worked - mappings applied
292         mappingAfterMerge = indexService.mapperService().documentMapper("type").refreshSource();
293         assertThat(mappingAfterMerge, equalTo(new CompressedString("{\"type\":{\"_ttl\":{\"enabled\":true,\"default\":604800000},\"properties\":{\"field\":{\"type\":\"string\"}}}}")));
294

305     private org.elasticsearch.common.xcontent.XContentBuilder getMappingWithTtlEnabled(String defaultValue) throws IOException {
306         XContentBuilder mapping = XContentFactory.jsonBuilder().startObject().startObject("type")
307                 .startObject("_ttl")
308                 .field("enabled", true);
309         if (defaultValue != null) {
310             mapping.field("default", defaultValue);
311         }
312         return mapping.endObject()
313                 .startObject("properties").field("field").startObject().field("type", "string").endObject().endObject()
314                 .endObject().endObject();
315     }

324         return mapping.endObject()
325                 .startObject("properties").field("field").startObject().field("type", "string").endObject().endObject()
