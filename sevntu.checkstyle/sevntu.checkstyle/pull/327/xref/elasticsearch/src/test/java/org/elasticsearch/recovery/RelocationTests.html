154     public void testRelocationWhileIndexingRandom() throws Exception {
155         int numberOfRelocations = scaledRandomIntBetween(1, rarely() ? 10 : 4);
156         int numberOfReplicas = randomBoolean() ? 0 : 1;
157         int numberOfNodes = numberOfReplicas == 0 ? 2 : 3;
158 
159         logger.info("testRelocationWhileIndexingRandom(numRelocations={}, numberOfReplicas={}, numberOfNodes={})", numberOfRelocations, numberOfReplicas, numberOfNodes);
160 
161         String[] nodes = new String[numberOfNodes];
162         logger.info("--> starting [node1] ...");
163         nodes[0] = internalCluster().startNode();
164 
165         logger.info("--> creating test index ...");
166         client().admin().indices().prepareCreate("test")
167                 .setSettings(settingsBuilder()
168                                 .put("index.number_of_shards", 1)
169                                 .put("index.number_of_replicas", numberOfReplicas)
170                 ).execute().actionGet();
171 
172 
173         for (int i = 1; i < numberOfNodes; i++) {
174             logger.info("--> starting [node{}] ...", i + 1);
175             nodes[i] = internalCluster().startNode();
176             if (i != numberOfNodes - 1) {
177                 ClusterHealthResponse healthResponse = client().admin().cluster().prepareHealth().setWaitForEvents(Priority.LANGUID)
178                         .setWaitForNodes(Integer.toString(i + 1)).setWaitForGreenStatus().execute().actionGet();
179                 assertThat(healthResponse.isTimedOut(), equalTo(false));
180             }
181         }
182 
183         int numDocs = scaledRandomIntBetween(200, 2500);
184         try (BackgroundIndexer indexer = new BackgroundIndexer("test", "type1", client(), numDocs)) {
185             logger.info("--> waiting for {} docs to be indexed ...", numDocs);
186             waitForDocs(numDocs, indexer);
187             logger.info("--> {} docs indexed", numDocs);
188 
189             logger.info("--> starting relocations...");
190             int nodeShiftBased = numberOfReplicas; // if we have replicas shift those
191             for (int i = 0; i < numberOfRelocations; i++) {
192                 int fromNode = (i % 2);
193                 int toNode = fromNode == 0 ? 1 : 0;
194                 fromNode += nodeShiftBased;
195                 toNode += nodeShiftBased;
196                 numDocs = scaledRandomIntBetween(200, 1000);
197                 logger.debug("--> Allow indexer to index [{}] documents", numDocs);
198                 indexer.continueIndexing(numDocs);
199                 logger.info("--> START relocate the shard from {} to {}", nodes[fromNode], nodes[toNode]);
200                 client().admin().cluster().prepareReroute()
201                         .add(new MoveAllocationCommand(new ShardId("test", 0), nodes[fromNode], nodes[toNode]))
202                         .get();
203                 if (rarely()) {
204                     logger.debug("--> flushing");
205                     client().admin().indices().prepareFlush().get();
206                 }
207                 ClusterHealthResponse clusterHealthResponse = client().admin().cluster().prepareHealth().setWaitForEvents(Priority.LANGUID).setWaitForRelocatingShards(0).setTimeout(ACCEPTABLE_RELOCATION_TIME).execute().actionGet();
208                 assertThat(clusterHealthResponse.isTimedOut(), equalTo(false));
209                 clusterHealthResponse = client().admin().cluster().prepareHealth().setWaitForEvents(Priority.LANGUID).setWaitForRelocatingShards(0).setTimeout(ACCEPTABLE_RELOCATION_TIME).execute().actionGet();
210                 assertThat(clusterHealthResponse.isTimedOut(), equalTo(false));
211                 indexer.pauseIndexing();
212                 logger.info("--> DONE relocate the shard from {} to {}", fromNode, toNode);
213             }
214             logger.info("--> done relocations");
215             logger.info("--> waiting for indexing threads to stop ...");
216             indexer.stop();
217             logger.info("--> indexing threads stopped");
218 
219             logger.info("--> refreshing the index");
220             client().admin().indices().prepareRefresh("test").execute().actionGet();
221             logger.info("--> searching the index");
222             boolean ranOnce = false;
223             for (int i = 0; i < 10; i++) {
224                 try {
225                     logger.info("--> START search test round {}", i + 1);
226                     SearchHits hits = client().prepareSearch("test").setQuery(matchAllQuery()).setSize((int) indexer.totalIndexedDocs()).setNoFields().execute().actionGet().getHits();
227                     ranOnce = true;
228                     if (hits.totalHits() != indexer.totalIndexedDocs()) {
229                         int[] hitIds = new int[(int) indexer.totalIndexedDocs()];
230                         for (int hit = 0; hit < indexer.totalIndexedDocs(); hit++) {
231                             hitIds[hit] = hit + 1;
232                         }
233                         IntOpenHashSet set = IntOpenHashSet.from(hitIds);
234                         for (SearchHit hit : hits.hits()) {
235                             int id = Integer.parseInt(hit.id());
236                             if (!set.remove(id)) {
237                                 logger.error("Extra id [{}]", id);
238                             }
239                         }
240                         set.forEach(new IntProcedure() {
241 
242                             @Override
243                             public void apply(int value) {
244                                 logger.error("Missing id [{}]", value);
245                             }
246 
247                         });
248                     }
249                     assertThat(hits.totalHits(), equalTo(indexer.totalIndexedDocs()));
250                     logger.info("--> DONE search test round {}", i + 1);
251                 } catch (SearchPhaseExecutionException ex) {
252                     // TODO: the first run fails with this failure, waiting for relocating nodes set to 0 is not enough?
253                     logger.warn("Got exception while searching.", ex);
254                 }
255             }
256             if (!ranOnce) {
257                 fail();
258             }
259         }

265         int numberOfRelocations = scaledRandomIntBetween(1, rarely() ? 10 : 4);
266         int numberOfReplicas = randomBoolean() ? 0 : 1;
267         int numberOfNodes = numberOfReplicas == 0 ? 2 : 3;

272         logger.info("--> starting [node_0] ...");
273         nodes[0] = internalCluster().startNode();
274 
275         logger.info("--> creating test index ...");
276         client().admin().indices().prepareCreate("test")
277                 .setSettings(settingsBuilder()
278                         .put("index.number_of_shards", 1)
279                         .put("index.number_of_replicas", numberOfReplicas)
280                         .put("index.refresh_interval", -1) // we want to control refreshes c
281                 ).execute().actionGet();
282 
283         // make sure the first shard is started.
284         ensureYellow();
285 
286         for (int i = 1; i < numberOfNodes; i++) {
287             logger.info("--> starting [node_{}] ...", i);
288             nodes[i] = internalCluster().startNode();
289             if (i != numberOfNodes - 1) {
290                 ClusterHealthResponse healthResponse = client().admin().cluster().prepareHealth().setWaitForEvents(Priority.LANGUID)
291                         .setWaitForNodes(Integer.toString(i + 1)).setWaitForGreenStatus().execute().actionGet();
292                 assertThat(healthResponse.isTimedOut(), equalTo(false));
293             }
294         }
295 
296         final Semaphore postRecoveryShards = new Semaphore(0);
297 
298         for (IndicesLifecycle indicesLifecycle : internalCluster().getInstances(IndicesLifecycle.class)) {
299             indicesLifecycle.addListener(new IndicesLifecycle.Listener() {
300                 @Override
301                 public void indexShardStateChanged(IndexShard indexShard, @Nullable IndexShardState previousState, IndexShardState currentState, @Nullable String reason) {
302                     if (currentState == IndexShardState.POST_RECOVERY) {
303                         postRecoveryShards.release();
304                     }
305                 }
306             });
307         }
308 
309 
310         logger.info("--> starting relocations...");
311         int nodeShiftBased = numberOfReplicas; // if we have replicas shift those
312         for (int i = 0; i < numberOfRelocations; i++) {
313             int fromNode = (i % 2);
314             int toNode = fromNode == 0 ? 1 : 0;
315             fromNode += nodeShiftBased;
316             toNode += nodeShiftBased;
317 
318             List<IndexRequestBuilder> builders1 = new ArrayList<>();
319             for (int numDocs = randomIntBetween(10, 30); numDocs > 0; numDocs--) {
320                 builders1.add(client().prepareIndex("test", "type").setSource("{}"));
321             }
322 
323             List<IndexRequestBuilder> builders2 = new ArrayList<>();
324             for (int numDocs = randomIntBetween(10, 30); numDocs > 0; numDocs--) {
325                 builders2.add(client().prepareIndex("test", "type").setSource("{}"));
326             }
327 
328             logger.info("--> START relocate the shard from {} to {}", nodes[fromNode], nodes[toNode]);
329 
330 
331             client().admin().cluster().prepareReroute()
332                     .add(new MoveAllocationCommand(new ShardId("test", 0), nodes[fromNode], nodes[toNode]))
333                     .get();
334 
335 
336             logger.debug("--> index [{}] documents", builders1.size());
337             indexRandom(false, true, builders1);
338             // wait for shard to reach post recovery
339             postRecoveryShards.acquire(1);
340 
341             logger.debug("--> index [{}] documents", builders2.size());
342             indexRandom(true, true, builders2);
343 
344             // verify cluster was finished.
345             assertFalse(client().admin().cluster().prepareHealth().setWaitForRelocatingShards(0).setWaitForEvents(Priority.LANGUID).setTimeout("30s").get().isTimedOut());
346             logger.info("--> DONE relocate the shard from {} to {}", fromNode, toNode);
347 
348             logger.debug("--> verifying all searches return the same number of docs");
349             long expectedCount = -1;
350             for (Client client : clients()) {
351                 SearchResponse response = client.prepareSearch("test").setPreference("_local").setSearchType(SearchType.COUNT).get();
352                 assertNoFailures(response);
353                 if (expectedCount < 0) {
354                     expectedCount = response.getHits().totalHits();
355                 } else {
356                     assertEquals(expectedCount, response.getHits().totalHits());
357                 }
358             }
359 
360         }
361     }

363     @Test
364     public void testMoveShardsWhileRelocation() throws Exception {
365         final String indexName = "test";
366 
367         ListenableFuture<String> blueFuture = internalCluster().startNodeAsync(ImmutableSettings.builder().put("node.color", "blue").build());
368         ListenableFuture<String> redFuture = internalCluster().startNodeAsync(ImmutableSettings.builder().put("node.color", "red").build());
369         internalCluster().startNode(ImmutableSettings.builder().put("node.color", "green").build());
370         final String blueNodeName = blueFuture.get();
371         final String redNodeName = redFuture.get();
372 
373         ClusterHealthResponse response = client().admin().cluster().prepareHealth().setWaitForNodes(">=3").get();
374         assertThat(response.isTimedOut(), is(false));
375 
376 
377         client().admin().indices().prepareCreate(indexName)
378                 .setSettings(
379                         ImmutableSettings.builder()
380                                 .put(FilterAllocationDecider.INDEX_ROUTING_INCLUDE_GROUP + "color", "blue")
381                                 .put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, 0)
382                 ).get();
383 
384         List<IndexRequestBuilder> requests = new ArrayList<>();
385         int numDocs = scaledRandomIntBetween(25, 250);
386         for (int i = 0; i < numDocs; i++) {
387             requests.add(client().prepareIndex(indexName, "type").setCreate(true).setSource("{}"));
388         }
389         indexRandom(true, requests);
390         ensureSearchable(indexName);
391 
392         ClusterStateResponse stateResponse = client().admin().cluster().prepareState().get();
393         String blueNodeId = internalCluster().getInstance(DiscoveryService.class, blueNodeName).localNode().id();
394 
395         assertFalse(stateResponse.getState().readOnlyRoutingNodes().node(blueNodeId).isEmpty());
396 
397         SearchResponse searchResponse = client().prepareSearch(indexName).get();
398         assertHitCount(searchResponse, numDocs);
399 
400         // Slow down recovery in order to make recovery cancellations more likely
401         IndicesStatsResponse statsResponse = client().admin().indices().prepareStats(indexName).get();
402         long chunkSize = statsResponse.getIndex(indexName).getShards()[0].getStats().getStore().size().bytes() / 10;
403         assertTrue(client().admin().cluster().prepareUpdateSettings()
404                 .setTransientSettings(ImmutableSettings.builder()
405                                 // one chunk per sec..
406                                 .put(RecoverySettings.INDICES_RECOVERY_MAX_BYTES_PER_SEC, chunkSize)
407                                 .put(RecoverySettings.INDICES_RECOVERY_FILE_CHUNK_SIZE, chunkSize)
408                 )
409                 .get().isAcknowledged());
410 
411         client().admin().indices().prepareUpdateSettings(indexName).setSettings(
412                 ImmutableSettings.builder().put(FilterAllocationDecider.INDEX_ROUTING_INCLUDE_GROUP + "color", "red")
413         ).get();
414 
415         // Lets wait a bit and then move again to hopefully trigger recovery cancellations.
416         boolean applied = awaitBusy(
417                 new Predicate<Object>() {
418                     @Override
419                     public boolean apply(Object input) {
420                         RecoveryResponse recoveryResponse = internalCluster().client(redNodeName).admin().indices().prepareRecoveries(indexName)
421                                 .get();
422                         return !recoveryResponse.shardResponses().get(indexName).isEmpty();
423                     }
424                 }
425         );
426         assertTrue(applied);
427         client().admin().indices().prepareUpdateSettings(indexName).setSettings(
428                 ImmutableSettings.builder().put(FilterAllocationDecider.INDEX_ROUTING_INCLUDE_GROUP + "color", "green")
429         ).get();
430 
431         // Restore the recovery speed to not timeout cluster health call
432         assertTrue(client().admin().cluster().prepareUpdateSettings()
433                 .setTransientSettings(ImmutableSettings.builder()
434                                 .put(RecoverySettings.INDICES_RECOVERY_MAX_BYTES_PER_SEC, "20mb")
435                                 .put(RecoverySettings.INDICES_RECOVERY_FILE_CHUNK_SIZE, "512kb")
436                 )
437                 .get().isAcknowledged());
438 
439         // this also waits for all ongoing recoveries to complete:
440         ensureSearchable(indexName);
441         searchResponse = client().prepareSearch(indexName).get();
442         assertHitCount(searchResponse, numDocs);
443 
444         stateResponse = client().admin().cluster().prepareState().get();
445         assertTrue(stateResponse.getState().readOnlyRoutingNodes().node(blueNodeId).isEmpty());
446     }

448     @Test
449     @TestLogging("cluster.service:TRACE,indices.recovery:TRACE")
450     public void testRelocationWithBusyClusterUpdateThread() throws Exception {
451         final String indexName = "test";
452         final Settings settings = ImmutableSettings.builder()
453                 .put("gateway.type", "local")
454                 .put(DiscoverySettings.PUBLISH_TIMEOUT, "1s")
455                 .put("indices.recovery.internal_action_timeout", "1s").build();
456         String master = internalCluster().startNode(settings);
457         ensureGreen();
458         List<String> nodes = internalCluster().startNodesAsync(2, settings).get();
459         final String node1 = nodes.get(0);
460         final String node2 = nodes.get(1);
461         ClusterHealthResponse response = client().admin().cluster().prepareHealth().setWaitForNodes("3").get();
462         assertThat(response.isTimedOut(), is(false));
463 
464 
465         client().admin().indices().prepareCreate(indexName)
466                 .setSettings(
467                         ImmutableSettings.builder()
468                                 .put(FilterAllocationDecider.INDEX_ROUTING_INCLUDE_GROUP + "_name", node1)
469                                 .put(IndexMetaData.SETTING_NUMBER_OF_SHARDS, 1)
470                                 .put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, 0)
471                 ).get();
472 
473 
474         List<IndexRequestBuilder> requests = new ArrayList<>();
475         int numDocs = scaledRandomIntBetween(25, 250);
476         for (int i = 0; i < numDocs; i++) {
477             requests.add(client().prepareIndex(indexName, "type").setCreate(true).setSource("{}"));
478         }
479         indexRandom(true, requests);
480         ensureSearchable(indexName);
481 
482         // capture the incoming state indicate that the replicas have upgraded and assigned
483 
484         final CountDownLatch allReplicasAssigned = new CountDownLatch(1);
485         final CountDownLatch releaseClusterState = new CountDownLatch(1);
486         final CountDownLatch unassignedShardsAfterReplicasAssigned = new CountDownLatch(1);
487         try {
488             internalCluster().getInstance(ClusterService.class, node1).addLast(new ClusterStateListener() {
489                 @Override
490                 public void clusterChanged(ClusterChangedEvent event) {
491                     ClusterState state = event.state();
492                     if (state.routingTable().allShards().size() == 1 || state.routingNodes().hasUnassignedShards()) {
493                         // we have no replicas or they are not assigned yet
494                         return;
495                     }
496 
497                     allReplicasAssigned.countDown();
498                     try {
499                         releaseClusterState.await();
500                     } catch (InterruptedException e) {
501                         //
502                     }
503                 }
504 
505             });
506 
507             internalCluster().getInstance(ClusterService.class, master).addLast(new ClusterStateListener() {
508                 @Override
509                 public void clusterChanged(ClusterChangedEvent event) {
510                     if (event.state().routingNodes().hasUnassigned() && allReplicasAssigned.getCount() == 0) {
511                         unassignedShardsAfterReplicasAssigned.countDown();
512                     }
513                 }
514             });
515 
516             logger.info("--> starting replica recovery");
517             // we don't expect this to be acknowledge by node1 where we block the cluster state thread
518             assertFalse(client().admin().indices().prepareUpdateSettings(indexName)
519                     .setSettings(ImmutableSettings.builder()
520                                     .put(FilterAllocationDecider.INDEX_ROUTING_INCLUDE_GROUP + "_name", node1 + "," + node2)
521                                     .put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, 1)
522 
523                     ).setTimeout("200ms")
524                     .get().isAcknowledged());
525 
526             logger.info("--> waiting for node1 to process replica existence");
527             allReplicasAssigned.await();
528             logger.info("--> waiting for recovery to fail");
529             unassignedShardsAfterReplicasAssigned.await();
530         } finally {
531             logger.info("--> releasing cluster state update thread");
532             releaseClusterState.countDown();
533         }
534         logger.info("--> waiting for recovery to succeed");
535         // force a move.
536         client().admin().cluster().prepareReroute().get();
537         ensureGreen();
538     }

619     class RecoveryCorruption extends MockTransportService.DelegateTransport {
620 
621         private final CountDownLatch corruptionCount;
622 
623         public RecoveryCorruption(Transport transport, CountDownLatch corruptionCount) {
624             super(transport);
625             this.corruptionCount = corruptionCount;
626         }
627 
628         @Override
629         public void sendRequest(DiscoveryNode node, long requestId, String action, TransportRequest request, TransportRequestOptions options) throws IOException, TransportException {
630             if (action.equals(RecoveryTarget.Actions.FILE_CHUNK)) {
631                 RecoveryFileChunkRequest chunkRequest = (RecoveryFileChunkRequest) request;
632                 if (chunkRequest.name().startsWith(IndexFileNames.SEGMENTS)) {
633                     // corrupting the segments_N files in order to make sure future recovery re-send files
634                     logger.debug("corrupting [{}] to {}. file name: [{}]", action, node, chunkRequest.name());
635                     byte[] array = chunkRequest.content().array();
636                     array[0] = (byte) ~array[0]; // flip one byte in the content
637                     corruptionCount.countDown();
638                 }
639                 transport.sendRequest(node, requestId, action, request, options);
640             } else {
641                 transport.sendRequest(node, requestId, action, request, options);
642             }
643         }
