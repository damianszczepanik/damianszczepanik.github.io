35  /**
36   * This test basically verifies that search with a single shard active (cause we indexed to it) and other
37   * shards possibly not active at all (cause they haven't allocated) will still work.
38   */
39  public class SearchWhileCreatingIndexTests extends ElasticsearchIntegrationTest {
40  
41      @Test
42      @Slow
43      public void testIndexCausesIndexCreation() throws Exception {
44          searchWhileCreatingIndex(false, 1); // 1 replica in our default...
45      }
46  
47      @Test
48      @Slow
49      public void testNoReplicas() throws Exception {
50          searchWhileCreatingIndex(true, 0);
51      }
52  
53      @Test
54      @Slow
55      public void testOneReplica() throws Exception {
56          searchWhileCreatingIndex(true, 1);
57      }
58  
59      @Test
60      @Slow
61      public void testTwoReplicas() throws Exception {
62          searchWhileCreatingIndex(true, 2);
63      }
64  
65      private void searchWhileCreatingIndex(boolean createIndex, int numberOfReplicas) throws Exception {
66  
67          // make sure we have enough nodes to guaranty default QUORUM consistency.
68          // TODO: add a smarter choice based on actual consistency (when that is randomized)
69          int shardsNo = numberOfReplicas + 1;
70          int neededNodes = shardsNo <= 2 ? 1 : shardsNo / 2 + 1;
71          internalCluster().ensureAtLeastNumDataNodes(randomIntBetween(neededNodes, shardsNo));
72          for (int i = 0; i < 20; i++) {
73              logger.info("running iteration {}", i);
74              if (createIndex) {
75                  createIndex("test");
76              }
77              client().prepareIndex("test", "type1", randomAsciiOfLength(5)).setSource("field", "test").execute().actionGet();
78              RefreshResponse refreshResponse = client().admin().indices().prepareRefresh("test").execute().actionGet();
79              assertThat(refreshResponse.getSuccessfulShards(), greaterThanOrEqualTo(1)); // at least one shard should be successful when refreshing
80  
81              // we want to make sure that while recovery happens, and a replica gets recovered, its properly refreshed
82              ClusterHealthStatus status = ClusterHealthStatus.RED;
83              while (status != ClusterHealthStatus.GREEN) {
84                  // first, verify that search on the primary search works
85                  SearchResponse searchResponse = client().prepareSearch("test").setPreference("_primary").setQuery(QueryBuilders.termQuery("field", "test")).execute().actionGet();
86                  assertHitCount(searchResponse, 1);
87                  // now, let it go to primary or replica, though in a randomized re-creatable manner
88                  String preference = randomAsciiOfLength(5);
89                  Client client = client();
90                  searchResponse = client.prepareSearch("test").setPreference(preference).setQuery(QueryBuilders.termQuery("field", "test")).execute().actionGet();
91                  if (searchResponse.getHits().getTotalHits() != 1) {
92                      refresh();
93                      SearchResponse searchResponseAfterRefresh = client.prepareSearch("test").setPreference(preference).setQuery(QueryBuilders.termQuery("field", "test")).execute().actionGet();
94                      logger.info("hits count mismatch on any shard search failed, post explicit refresh hits are {}", searchResponseAfterRefresh.getHits().getTotalHits());
95                      ensureGreen();
96                      SearchResponse searchResponseAfterGreen = client.prepareSearch("test").setPreference(preference).setQuery(QueryBuilders.termQuery("field", "test")).execute().actionGet();
97                      logger.info("hits count mismatch on any shard search failed, post explicit wait for green hits are {}", searchResponseAfterGreen.getHits().getTotalHits());
98                      assertHitCount(searchResponse, 1);
99                  }
100                 assertHitCount(searchResponse, 1);
101                 status = client().admin().cluster().prepareHealth("test").get().getStatus();
102                 internalCluster().ensureAtLeastNumDataNodes(numberOfReplicas + 1);
103             }
104             cluster().wipeIndices("test");
105         }
106     }
