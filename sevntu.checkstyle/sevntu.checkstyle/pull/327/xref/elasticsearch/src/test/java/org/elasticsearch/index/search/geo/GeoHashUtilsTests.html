22  import org.elasticsearch.common.geo.GeoHashUtils;

33   */
34  public class GeoHashUtilsTests extends ElasticsearchTestCase {
35  
36      /**
37       * Pass condition: lat=42.6, lng=-5.6 should be encoded as "ezs42e44yx96",
38       * lat=57.64911 lng=10.40744 should be encoded as "u4pruydqqvj8"
39       */
40      @Test
41      public void testEncode() {
42          String hash = GeoHashUtils.encode(42.6, -5.6);
43          assertEquals("ezs42e44yx96", hash);
44  
45          hash = GeoHashUtils.encode(57.64911, 10.40744);
46          assertEquals("u4pruydqqvj8", hash);
47      }
48  
49      /**
50       * Pass condition: lat=52.3738007, lng=4.8909347 should be encoded and then
51       * decoded within 0.00001 of the original value
52       */
53      @Test
54      public void testDecodePreciseLongitudeLatitude() {
55          String hash = GeoHashUtils.encode(52.3738007, 4.8909347);
56  
57          GeoPoint point = GeoHashUtils.decode(hash);
58  
59          assertEquals(52.3738007, point.lat(), 0.00001D);
60          assertEquals(4.8909347, point.lon(), 0.00001D);
61      }
62  
63      /**
64       * Pass condition: lat=84.6, lng=10.5 should be encoded and then decoded
65       * within 0.00001 of the original value
66       */
67      @Test
68      public void testDecodeImpreciseLongitudeLatitude() {
69          String hash = GeoHashUtils.encode(84.6, 10.5);
70  
71          GeoPoint point = GeoHashUtils.decode(hash);
72  
73          assertEquals(84.6, point.lat(), 0.00001D);
74          assertEquals(10.5, point.lon(), 0.00001D);
75      }
76  
77      /*
78      * see https://issues.apache.org/jira/browse/LUCENE-1815 for details
79      */
80  
81      @Test
82      public void testDecodeEncode() {
83          String geoHash = "u173zq37x014";
84          assertEquals(geoHash, GeoHashUtils.encode(52.3738007, 4.8909347));
85          GeoPoint decode = GeoHashUtils.decode(geoHash);
86          assertEquals(52.37380061d, decode.lat(), 0.000001d);
87          assertEquals(4.8909343d, decode.lon(), 0.000001d);
88  
89          assertEquals(geoHash, GeoHashUtils.encode(decode.lat(), decode.lon()));
90      }
