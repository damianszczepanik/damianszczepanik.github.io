54      @Test
55      public void testThatPreBuiltAnalyzersAreNotClosedOnIndexClose() throws Exception {
56          Map<PreBuiltAnalyzers, List<Version>> loadedAnalyzers = Maps.newHashMap();
57          List<String> indexNames = Lists.newArrayList();
58          final int numIndices = scaledRandomIntBetween(2, 4);
59          for (int i = 0; i < numIndices; i++) {
60              String indexName = randomAsciiOfLength(10).toLowerCase(Locale.ROOT);
61              indexNames.add(indexName);
62  
63              int randomInt = randomInt(PreBuiltAnalyzers.values().length-1);
64              PreBuiltAnalyzers preBuiltAnalyzer = PreBuiltAnalyzers.values()[randomInt];
65              String name = preBuiltAnalyzer.name().toLowerCase(Locale.ROOT);
66  
67              Version randomVersion = randomVersion();
68              if (!loadedAnalyzers.containsKey(preBuiltAnalyzer)) {
69                   loadedAnalyzers.put(preBuiltAnalyzer, Lists.<Version>newArrayList());
70              }
71              loadedAnalyzers.get(preBuiltAnalyzer).add(randomVersion);
72  
73              final XContentBuilder mapping = jsonBuilder().startObject()
74                  .startObject("type")
75                      .startObject("properties")
76                          .startObject("foo")
77                              .field("type", "string")
78                              .field("analyzer", name)
79                          .endObject()
80                      .endObject()
81                  .endObject()
82                  .endObject();
83  
84              Settings versionSettings = settings(randomVersion).build();
85              client().admin().indices().prepareCreate(indexName).addMapping("type", mapping).setSettings(versionSettings).get();
86          }
87  
88          ensureGreen();
89  
90          final int numDocs = randomIntBetween(10, 100);
91          // index some amount of data
92          for (int i = 0; i < numDocs; i++) {
93              String randomIndex = indexNames.get(randomInt(indexNames.size()-1));
94              String randomId = randomInt() + "";
95  
96              Map<String, Object> data = Maps.newHashMap();
97              data.put("foo", randomAsciiOfLength(scaledRandomIntBetween(5, 50)));
98  
99              index(randomIndex, "type", randomId, data);
100         }
101 
102         refresh();
103 
104         // close some of the indices
105         int amountOfIndicesToClose = randomInt(numIndices-1);
106         for (int i = 0; i < amountOfIndicesToClose; i++) {
107             String indexName = indexNames.get(i);
108             client().admin().indices().prepareClose(indexName).execute().actionGet();
109         }
110 
111         ensureGreen();
112 
113         // check that all above configured analyzers have been loaded
114         assertThatAnalyzersHaveBeenLoaded(loadedAnalyzers);
115 
116         // check that all of the prebuiltanalyzers are still open
117         assertLuceneAnalyzersAreNotClosed(loadedAnalyzers);
118     }

156     private void assertThatAnalyzersHaveBeenLoaded(Map<PreBuiltAnalyzers, List<Version>> expectedLoadedAnalyzers) {
157         for (Map.Entry<PreBuiltAnalyzers, List<Version>> entry : expectedLoadedAnalyzers.entrySet()) {
158             for (Version version : entry.getValue()) {
159                 // if it is not null in the cache, it has been loaded
160                 assertThat(entry.getKey().getCache().get(version), is(notNullValue()));
161             }
162         }
163     }

165     // the close() method of a lucene analyzer sets the storedValue field to null
166     // we simply check this via reflection - ugly but works
167     private void assertLuceneAnalyzersAreNotClosed(Map<PreBuiltAnalyzers, List<Version>> loadedAnalyzers) throws IllegalAccessException, NoSuchFieldException {
168         for (Map.Entry<PreBuiltAnalyzers, List<Version>> preBuiltAnalyzerEntry : loadedAnalyzers.entrySet()) {
169             PreBuiltAnalyzers preBuiltAnalyzer = preBuiltAnalyzerEntry.getKey();
170             for (Version version : preBuiltAnalyzerEntry.getValue()) {
171                 Analyzer analyzer = preBuiltAnalyzerEntry.getKey().getCache().get(version);
172 
173                 Field field = getFieldFromClass("storedValue", analyzer);
174                 boolean currentAccessible = field.isAccessible();
175                 field.setAccessible(true);
176                 Object storedValue = field.get(analyzer);
177                 field.setAccessible(currentAccessible);
178 
179                 assertThat(String.format(Locale.ROOT, "Analyzer %s in version %s seems to be closed", preBuiltAnalyzer.name(), version), storedValue, is(notNullValue()));
180             }
181         }
182     }

184     /**
185      * Searches for a field until it finds, loops through all superclasses
186      */
187     private Field getFieldFromClass(String fieldName, Object obj) {
188         Field field = null;
189         boolean storedValueFieldFound = false;
190         Class clazz = obj.getClass();
191         while (!storedValueFieldFound) {
192             try {
193                 field = clazz.getDeclaredField(fieldName);
194                 storedValueFieldFound = true;
195             } catch (NoSuchFieldException e) {
196                 clazz = clazz.getSuperclass();
197             }
198 
199             if (Object.class.equals(clazz)) throw new RuntimeException("Could not find storedValue field in class" + clazz);
200         }
201 
202         return field;
203     }
