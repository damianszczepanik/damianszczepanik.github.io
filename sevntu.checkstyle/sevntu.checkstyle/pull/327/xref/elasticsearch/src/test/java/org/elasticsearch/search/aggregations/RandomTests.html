72      // Make sure that unordered, reversed, disjoint and/or overlapping ranges are supported
73      // Duel with filters
74      public void testRandomRanges() throws Exception {
75          final int numDocs = scaledRandomIntBetween(500, 5000);
76          final double[][] docs = new double[numDocs][];
77          for (int i = 0; i < numDocs; ++i) {
78              final int numValues = randomInt(5);
79              docs[i] = new double[numValues];
80              for (int j = 0; j < numValues; ++j) {
81                  docs[i][j] = randomDouble() * 100;
82              }
83          }
84  
85          createIndex("idx");
86          for (int i = 0; i < docs.length; ++i) {
87              XContentBuilder source = jsonBuilder()
88                      .startObject()
89                      .startArray("values");
90              for (int j = 0; j < docs[i].length; ++j) {
91                  source = source.value(docs[i][j]);
92              }
93              source = source.endArray().endObject();
94              client().prepareIndex("idx", "type").setSource(source).execute().actionGet();
95          }
96          assertNoFailures(client().admin().indices().prepareRefresh("idx").setIndicesOptions(IndicesOptions.lenientExpandOpen()).execute().get());
97  
98          final int numRanges = randomIntBetween(1, 20);
99          final double[][] ranges = new double[numRanges][];
100         for (int i = 0; i < ranges.length; ++i) {
101             switch (randomInt(2)) {
102             case 0:
103                 ranges[i] = new double[] { Double.NEGATIVE_INFINITY, randomInt(100) };
104                 break;
105             case 1:
106                 ranges[i] = new double[] { randomInt(100), Double.POSITIVE_INFINITY };
107                 break;
108             case 2:
109                 ranges[i] = new double[] { randomInt(100), randomInt(100) };
110                 break;
111             default:
112                 throw new AssertionError();
113             }
114         }
115 
116         RangeBuilder query = range("range").field("values");
117         for (int i = 0; i < ranges.length; ++i) {
118             String key = Integer.toString(i);
119             if (ranges[i][0] == Double.NEGATIVE_INFINITY) {
120                 query.addUnboundedTo(key, ranges[i][1]);
121             } else if (ranges[i][1] == Double.POSITIVE_INFINITY) {
122                 query.addUnboundedFrom(key, ranges[i][0]);
123             } else {
124                 query.addRange(key, ranges[i][0], ranges[i][1]);
125             }
126         }
127 
128         SearchRequestBuilder reqBuilder = client().prepareSearch("idx").addAggregation(query);
129         for (int i = 0; i < ranges.length; ++i) {
130             RangeFilterBuilder filter = FilterBuilders.rangeFilter("values");
131             if (ranges[i][0] != Double.NEGATIVE_INFINITY) {
132                 filter = filter.from(ranges[i][0]);
133             }
134             if (ranges[i][1] != Double.POSITIVE_INFINITY){
135                 filter = filter.to(ranges[i][1]);
136             }
137             reqBuilder = reqBuilder.addAggregation(filter("filter" + i).filter(filter));
138         }
139 
140         SearchResponse resp = reqBuilder.execute().actionGet();
141         Range range = resp.getAggregations().get("range");
142         List<? extends Bucket> buckets = range.getBuckets();
143 
144         HashMap<String, Bucket> bucketMap = Maps.newHashMapWithExpectedSize(buckets.size());
145         for (Bucket bucket : buckets) {
146             bucketMap.put(bucket.getKeyAsString(), bucket);
147         }
148 
149         for (int i = 0; i < ranges.length; ++i) {
150 
151             long count = 0;
152             for (double[] values : docs) {
153                 for (double value : values) {
154                     if (value >= ranges[i][0] && value < ranges[i][1]) {
155                         ++count;
156                         break;
157                     }
158                 }
159             }
160 
161             final Range.Bucket bucket = bucketMap.get(Integer.toString(i));
162             assertEquals(bucket.getKeyAsString(), Integer.toString(i), bucket.getKeyAsString());
163             assertEquals(bucket.getKeyAsString(), count, bucket.getDocCount());
164 
165             final Filter filter = resp.getAggregations().get("filter" + i);
166             assertThat(filter.getDocCount(), equalTo(count));
167         }
168     }

170     // test long/double/string terms aggs with high number of buckets that require array growth
171     public void testDuelTerms() throws Exception {
172         final int numDocs = scaledRandomIntBetween(1000, 2000);
173         final int maxNumTerms = randomIntBetween(10, 5000);
174 
175         final IntOpenHashSet valuesSet = new IntOpenHashSet();
176         cluster().wipeIndices("idx");
177         prepareCreate("idx")
178                 .addMapping("type", jsonBuilder().startObject()
179                         .startObject("type")
180                         .startObject("properties")
181                         .startObject("string_values")
182                         .field("type", "string")
183                         .field("index", "not_analyzed")
184                         .startObject("fields")
185                             .startObject("doc_values")
186                                 .field("type", "string")
187                                 .field("index", "no")
188                                 .startObject("fielddata")
189                                     .field("format", "doc_values")
190                                 .endObject()
191                             .endObject()
192                         .endObject()
193                         .endObject()
194                         .startObject("long_values")
195                         .field("type", "long")
196                         .endObject()
197                         .startObject("double_values")
198                         .field("type", "double")
199                         .endObject()
200                         .endObject()
201                         .endObject()).execute().actionGet();
202 
203         List<IndexRequestBuilder> indexingRequests = Lists.newArrayList();
204         for (int i = 0; i < numDocs; ++i) {
205             final int[] values = new int[randomInt(4)];
206             for (int j = 0; j < values.length; ++j) {
207                 values[j] = randomInt(maxNumTerms - 1) - 1000;
208                 valuesSet.add(values[j]);
209             }
210             XContentBuilder source = jsonBuilder()
211                     .startObject()
212                     .field("num", randomDouble())
213                     .startArray("long_values");
214             for (int j = 0; j < values.length; ++j) {
215                 source = source.value(values[j]);
216             }
217             source = source.endArray().startArray("double_values");
218             for (int j = 0; j < values.length; ++j) {
219                 source = source.value((double) values[j]);
220             }
221             source = source.endArray().startArray("string_values");
222             for (int j = 0; j < values.length; ++j) {
223                 source = source.value(Integer.toString(values[j]));
224             }
225             source = source.endArray().endObject();
226             indexingRequests.add(client().prepareIndex("idx", "type").setSource(source));
227         }
228         indexRandom(true, indexingRequests);
229 
230         assertNoFailures(client().admin().indices().prepareRefresh("idx").setIndicesOptions(IndicesOptions.lenientExpandOpen()).execute().get());
231 
232         TermsAggregatorFactory.ExecutionMode[] globalOrdinalModes = new TermsAggregatorFactory.ExecutionMode[]{
233                 TermsAggregatorFactory.ExecutionMode.GLOBAL_ORDINALS_HASH,
234                 TermsAggregatorFactory.ExecutionMode.GLOBAL_ORDINALS
235         };
236 
237         SearchResponse resp = client().prepareSearch("idx")
238                 .addAggregation(terms("long").field("long_values").size(maxNumTerms).collectMode(randomFrom(SubAggCollectionMode.values())).subAggregation(min("min").field("num")))
239                 .addAggregation(terms("double").field("double_values").size(maxNumTerms).collectMode(randomFrom(SubAggCollectionMode.values())).subAggregation(max("max").field("num")))
240                 .addAggregation(terms("string_map").field("string_values").collectMode(randomFrom(SubAggCollectionMode.values())).executionHint(TermsAggregatorFactory.ExecutionMode.MAP.toString()).size(maxNumTerms).subAggregation(stats("stats").field("num")))
241                 .addAggregation(terms("string_global_ordinals").field("string_values").collectMode(randomFrom(SubAggCollectionMode.values())).executionHint(globalOrdinalModes[randomInt(globalOrdinalModes.length - 1)].toString()).size(maxNumTerms).subAggregation(extendedStats("stats").field("num")))
242                 .addAggregation(terms("string_global_ordinals_doc_values").field("string_values.doc_values").collectMode(randomFrom(SubAggCollectionMode.values())).executionHint(globalOrdinalModes[randomInt(globalOrdinalModes.length - 1)].toString()).size(maxNumTerms).subAggregation(extendedStats("stats").field("num")))
243                 .execute().actionGet();
244         assertAllSuccessful(resp);
245         assertEquals(numDocs, resp.getHits().getTotalHits());
246 
247         final Terms longTerms = resp.getAggregations().get("long");
248         final Terms doubleTerms = resp.getAggregations().get("double");
249         final Terms stringMapTerms = resp.getAggregations().get("string_map");
250         final Terms stringGlobalOrdinalsTerms = resp.getAggregations().get("string_global_ordinals");
251         final Terms stringGlobalOrdinalsDVTerms = resp.getAggregations().get("string_global_ordinals_doc_values");
252 
253         assertEquals(valuesSet.size(), longTerms.getBuckets().size());
254         assertEquals(valuesSet.size(), doubleTerms.getBuckets().size());
255         assertEquals(valuesSet.size(), stringMapTerms.getBuckets().size());
256         assertEquals(valuesSet.size(), stringGlobalOrdinalsTerms.getBuckets().size());
257         assertEquals(valuesSet.size(), stringGlobalOrdinalsDVTerms.getBuckets().size());
258         for (Terms.Bucket bucket : longTerms.getBuckets()) {
259             final Terms.Bucket doubleBucket = doubleTerms.getBucketByKey(Double.toString(Long.parseLong(bucket.getKeyAsString())));
260             final Terms.Bucket stringMapBucket = stringMapTerms.getBucketByKey(bucket.getKeyAsString());
261             final Terms.Bucket stringGlobalOrdinalsBucket = stringGlobalOrdinalsTerms.getBucketByKey(bucket.getKeyAsString());
262             final Terms.Bucket stringGlobalOrdinalsDVBucket = stringGlobalOrdinalsDVTerms.getBucketByKey(bucket.getKeyAsString());
263             assertNotNull(doubleBucket);
264             assertNotNull(stringMapBucket);
265             assertNotNull(stringGlobalOrdinalsBucket);
266             assertNotNull(stringGlobalOrdinalsDVBucket);
267             assertEquals(bucket.getDocCount(), doubleBucket.getDocCount());
268             assertEquals(bucket.getDocCount(), stringMapBucket.getDocCount());
269             assertEquals(bucket.getDocCount(), stringGlobalOrdinalsBucket.getDocCount());
270             assertEquals(bucket.getDocCount(), stringGlobalOrdinalsDVBucket.getDocCount());
271         }
272     }

274     // Duel between histograms and scripted terms
275     public void testDuelTermsHistogram() throws Exception {
276         createIndex("idx");
277 
278         final int numDocs = scaledRandomIntBetween(500, 5000);
279         final int maxNumTerms = randomIntBetween(10, 2000);
280         final int interval = randomIntBetween(1, 100);
281 
282         final Integer[] values = new Integer[maxNumTerms];
283         for (int i = 0; i < values.length; ++i) {
284             values[i] = randomInt(maxNumTerms * 3) - maxNumTerms;
285         }
286 
287         for (int i = 0; i < numDocs; ++i) {
288             XContentBuilder source = jsonBuilder()
289                     .startObject()
290                     .field("num", randomDouble())
291                     .startArray("values");
292             final int numValues = randomInt(4);
293             for (int j = 0; j < numValues; ++j) {
294                 source = source.value(randomFrom(values));
295             }
296             source = source.endArray().endObject();
297             client().prepareIndex("idx", "type").setSource(source).execute().actionGet();
298         }
299         assertNoFailures(client().admin().indices().prepareRefresh("idx").setIndicesOptions(IndicesOptions.lenientExpandOpen()).execute().get());
300 
301         SearchResponse resp = client().prepareSearch("idx")
302                 .addAggregation(terms("terms").field("values").collectMode(randomFrom(SubAggCollectionMode.values())).script("floor(_value / interval)").param("interval", interval).size(maxNumTerms))
303                 .addAggregation(histogram("histo").field("values").interval(interval))
304                 .execute().actionGet();
305 
306         assertSearchResponse(resp);
307 
308         Terms terms = resp.getAggregations().get("terms");
309         assertThat(terms, notNullValue());
310         Histogram histo = resp.getAggregations().get("histo");
311         assertThat(histo, notNullValue());
312         assertThat(terms.getBuckets().size(), equalTo(histo.getBuckets().size()));
313         for (Histogram.Bucket bucket : histo.getBuckets()) {
314             final double key = ((Number) bucket.getKey()).doubleValue() / interval;
315             final Terms.Bucket termsBucket = terms.getBucketByKey(String.valueOf(key));
316             assertEquals(bucket.getDocCount(), termsBucket.getDocCount());
317         }
318     }

337     // https://github.com/elasticsearch/elasticsearch/issues/6435
338     public void testReduce() throws Exception {
339         createIndex("idx");
340         final int value = randomIntBetween(0, 10);
341         indexRandom(true, client().prepareIndex("idx", "type").setSource("f", value));
342         ensureYellow("idx"); // only one document let's make sure all shards have an active primary
343         SearchResponse response = client().prepareSearch("idx")
344                 .addAggregation(filter("filter").filter(FilterBuilders.matchAllFilter())
345                 .subAggregation(range("range")
346                         .field("f")
347                         .addUnboundedTo(6)
348                         .addUnboundedFrom(6)
349                 .subAggregation(sum("sum").field("f"))))
350                 .execute().actionGet();
351 
352         assertSearchResponse(response);
353 
354         Filter filter = response.getAggregations().get("filter");
355         assertNotNull(filter);
356         assertEquals(1, filter.getDocCount());
357 
358         Range range = filter.getAggregations().get("range");
359         assertThat(range, notNullValue());
360         assertThat(range.getName(), equalTo("range"));
361         List<? extends Bucket> buckets = range.getBuckets();
362         assertThat(buckets.size(), equalTo(2));
363 
364         Range.Bucket bucket = buckets.get(0);
365         assertThat(bucket, notNullValue());
366         assertThat((String) bucket.getKey(), equalTo("*-6.0"));
367         assertThat(((Number) bucket.getFrom()).doubleValue(), equalTo(Double.NEGATIVE_INFINITY));
368         assertThat(((Number) bucket.getTo()).doubleValue(), equalTo(6.0));
369         assertThat(bucket.getDocCount(), equalTo(value < 6 ? 1L : 0L));
370         Sum sum = bucket.getAggregations().get("sum");
371         assertEquals(value < 6 ? value : 0, sum.getValue(), 0d);
372 
373         bucket = buckets.get(1);
374         assertThat(bucket, notNullValue());
375         assertThat((String) bucket.getKey(), equalTo("6.0-*"));
376         assertThat(((Number) bucket.getFrom()).doubleValue(), equalTo(6.0));
377         assertThat(((Number) bucket.getTo()).doubleValue(), equalTo(Double.POSITIVE_INFINITY));
378         assertThat(bucket.getDocCount(), equalTo(value >= 6 ? 1L : 0L));
379         sum = bucket.getAggregations().get("sum");
380         assertEquals(value >= 6 ? value : 0, sum.getValue(), 0d);
381     }
