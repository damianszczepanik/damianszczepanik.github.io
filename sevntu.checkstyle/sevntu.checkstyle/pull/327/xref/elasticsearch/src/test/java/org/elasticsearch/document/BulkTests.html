157         createIndex("test");
158         ensureGreen();
159         BulkResponse bulkResponse = client().prepareBulk()
160                 .add(client().prepareIndex("test", "type", "1").setCreate(true).setSource("field", "1"))
161                 .add(client().prepareIndex("test", "type", "2").setCreate(true).setSource("field", "1"))
162                 .add(client().prepareIndex("test", "type", "1").setSource("field", "2")).get();
163 
164         assertTrue(((IndexResponse) bulkResponse.getItems()[0].getResponse()).isCreated());
165         assertThat(((IndexResponse) bulkResponse.getItems()[0].getResponse()).getVersion(), equalTo(1l));
166         assertTrue(((IndexResponse) bulkResponse.getItems()[1].getResponse()).isCreated());
167         assertThat(((IndexResponse) bulkResponse.getItems()[1].getResponse()).getVersion(), equalTo(1l));
168         assertFalse(((IndexResponse) bulkResponse.getItems()[2].getResponse()).isCreated());
169         assertThat(((IndexResponse) bulkResponse.getItems()[2].getResponse()).getVersion(), equalTo(2l));
170 
171         bulkResponse = client().prepareBulk()
172                 .add(client().prepareUpdate("test", "type", "1").setVersion(4l).setDoc("field", "2"))
173                 .add(client().prepareUpdate("test", "type", "2").setDoc("field", "2"))
174                 .add(client().prepareUpdate("test", "type", "1").setVersion(2l).setDoc("field", "3")).get();
175 
176         assertThat(bulkResponse.getItems()[0].getFailureMessage(), containsString("Version"));
177         assertThat(((UpdateResponse) bulkResponse.getItems()[1].getResponse()).getVersion(), equalTo(2l));
178         assertThat(((UpdateResponse) bulkResponse.getItems()[2].getResponse()).getVersion(), equalTo(3l));
179 
180         bulkResponse = client().prepareBulk()
181                 .add(client().prepareIndex("test", "type", "e1").setCreate(true).setSource("field", "1").setVersion(10).setVersionType(VersionType.EXTERNAL))
182                 .add(client().prepareIndex("test", "type", "e2").setCreate(true).setSource("field", "1").setVersion(10).setVersionType(VersionType.EXTERNAL))
183                 .add(client().prepareIndex("test", "type", "e1").setSource("field", "2").setVersion(12).setVersionType(VersionType.EXTERNAL)).get();
184 
185         assertTrue(((IndexResponse) bulkResponse.getItems()[0].getResponse()).isCreated());
186         assertThat(((IndexResponse) bulkResponse.getItems()[0].getResponse()).getVersion(), equalTo(10l));
187         assertTrue(((IndexResponse) bulkResponse.getItems()[1].getResponse()).isCreated());
188         assertThat(((IndexResponse) bulkResponse.getItems()[1].getResponse()).getVersion(), equalTo(10l));
189         assertFalse(((IndexResponse) bulkResponse.getItems()[2].getResponse()).isCreated());
190         assertThat(((IndexResponse) bulkResponse.getItems()[2].getResponse()).getVersion(), equalTo(12l));
191 
192         bulkResponse = client().prepareBulk()
193                 .add(client().prepareUpdate("test", "type", "e1").setDoc("field", "2").setVersion(10)) // INTERNAL
194                 .add(client().prepareUpdate("test", "type", "e1").setDoc("field", "3").setVersion(20).setVersionType(VersionType.FORCE))
195                 .add(client().prepareUpdate("test", "type", "e1").setDoc("field", "3").setVersion(20).setVersionType(VersionType.INTERNAL)).get();
196 
197         assertThat(bulkResponse.getItems()[0].getFailureMessage(), containsString("Version"));
198         assertThat(((UpdateResponse) bulkResponse.getItems()[1].getResponse()).getVersion(), equalTo(20l));
199         assertThat(((UpdateResponse) bulkResponse.getItems()[2].getResponse()).getVersion(), equalTo(21l));
200     }

387

389                 ImmutableSettings.builder()
390                         .put(indexSettings())

392         ));

396         for (int i = 0; i < numDocs; ) {

405             }
406         }
407

409

412     }
