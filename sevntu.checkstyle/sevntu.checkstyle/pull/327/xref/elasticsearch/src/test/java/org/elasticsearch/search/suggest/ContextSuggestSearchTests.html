335     private void getBackpackSuggestionAndCompare(String contextValue, String... expectedText) {
336         Set<String> expected = Sets.newHashSet(expectedText);
337         CompletionSuggestionBuilder context = SuggestBuilders.completionSuggestion("suggestion").field("suggest_field").text("back").size(10).addContextField("color", contextValue);
338         SuggestRequestBuilder suggestionRequest = client().prepareSuggest(INDEX).addSuggestion(context);
339         SuggestResponse suggestResponse = suggestionRequest.execute().actionGet();
340         Suggest suggest = suggestResponse.getSuggest();
341         assertEquals(suggest.size(), 1);
342         for (Suggestion<? extends Entry<? extends Option>> s : suggest) {
343             CompletionSuggestion suggestion = (CompletionSuggestion) s;
344             for (CompletionSuggestion.Entry entry : suggestion) {
345                 List<CompletionSuggestion.Entry.Option> options = entry.getOptions();
346                 assertEquals(options.size(), expectedText.length);
347                 for (CompletionSuggestion.Entry.Option option : options) {
348                     assertTrue(expected.contains(option.getText().string()));
349                     expected.remove(option.getText().string());
350                 }
351             }
352         }
353     }

937     private void assertSuggestionsMatch(List<String> suggestions, String... hits) {
938         boolean[] suggested = new boolean[hits.length];
939         Arrays.sort(hits);
940         Arrays.fill(suggested, false);
941         int numSuggestions = 0;
942 
943         for (String suggestion : suggestions) {
944             int hitpos = Arrays.binarySearch(hits, suggestion);
945 
946             assertEquals(hits[hitpos], suggestion);
947             assertTrue(hitpos >= 0);
948             assertTrue(!suggested[hitpos]);
949 
950             suggested[hitpos] = true;
951             numSuggestions++;
952 
953         }
954         assertEquals(hits.length, numSuggestions);
955     }
