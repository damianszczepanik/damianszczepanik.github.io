20  package org.elasticsearch.discovery.zen;
21  
22  import org.elasticsearch.Version;
23  import org.elasticsearch.cluster.ClusterName;
24  import org.elasticsearch.cluster.node.DiscoveryNode;
25  import org.elasticsearch.common.transport.DummyTransportAddress;
26  import org.elasticsearch.discovery.zen.ping.ZenPing;
27  import org.elasticsearch.test.ElasticsearchTestCase;
28  import org.junit.Test;
29  
30  import java.util.ArrayList;
31  import java.util.Collections;
32  
33  import static org.hamcrest.Matchers.equalTo;
34  
35  public class ZenPingTests extends ElasticsearchTestCase {
36  
37      @Test
38      public void testPingCollection() {
39          DiscoveryNode[] nodes = new DiscoveryNode[randomIntBetween(1, 30)];
40          long maxIdPerNode[] = new long[nodes.length];
41          DiscoveryNode masterPerNode[] = new DiscoveryNode[nodes.length];
42          boolean hasJoinedOncePerNode[] = new boolean[nodes.length];
43          ArrayList<ZenPing.PingResponse> pings = new ArrayList<>();
44          for (int i = 0; i < nodes.length; i++) {
45              nodes[i] = new DiscoveryNode("" + i, DummyTransportAddress.INSTANCE, Version.CURRENT);
46          }
47  
48          for (int pingCount = scaledRandomIntBetween(10, nodes.length * 10); pingCount > 0; pingCount--) {
49              int node = randomInt(nodes.length - 1);
50              DiscoveryNode masterNode = null;
51              if (randomBoolean()) {
52                  masterNode = nodes[randomInt(nodes.length - 1)];
53              }
54              boolean hasJoinedOnce = randomBoolean();
55              ZenPing.PingResponse ping = new ZenPing.PingResponse(nodes[node], masterNode, ClusterName.DEFAULT, hasJoinedOnce);
56              if (rarely()) {
57                  // ignore some pings
58                  continue;
59              }
60              // update max ping info
61              maxIdPerNode[node] = ping.id();
62              masterPerNode[node] = masterNode;
63              hasJoinedOncePerNode[node] = hasJoinedOnce;
64              pings.add(ping);
65          }
66  
67          // shuffle
68          Collections.shuffle(pings);
69  
70          ZenPing.PingCollection collection = new ZenPing.PingCollection();
71          collection.addPings(pings.toArray(new ZenPing.PingResponse[pings.size()]));
72  
73          ZenPing.PingResponse[] aggregate = collection.toArray();
74  
75          for (ZenPing.PingResponse ping : aggregate) {
76              int nodeId = Integer.parseInt(ping.node().id());
77              assertThat(maxIdPerNode[nodeId], equalTo(ping.id()));
78              assertThat(masterPerNode[nodeId], equalTo(ping.master()));
79              assertThat(hasJoinedOncePerNode[nodeId], equalTo(ping.hasJoinedOnce()));
80  
81              maxIdPerNode[nodeId] = -1; // mark as seen
82          }
83  
84          for (int i = 0; i < maxIdPerNode.length; i++) {
85              assertTrue("node " + i + " had pings but it was not found in collection", maxIdPerNode[i] <= 0);
86          }
87  
88  
89      }
90  }
