1269     protected Suggest searchSuggest(String suggestText, int expectShardsFailed, SuggestionBuilder<?>... suggestions) {
1270         if (randomBoolean()) {
1271             SearchRequestBuilder builder = client().prepareSearch().setSearchType(SearchType.COUNT);
1272             if (suggestText != null) {
1273                 builder.setSuggestText(suggestText);
1274             }
1275             for (SuggestionBuilder<?> suggestion : suggestions) {
1276                 builder.addSuggestion(suggestion);
1277             }
1278             SearchResponse actionGet = builder.execute().actionGet();
1279             assertThat(Arrays.toString(actionGet.getShardFailures()), actionGet.getFailedShards(), equalTo(expectShardsFailed));
1280             return actionGet.getSuggest();
1281         } else {
1282             SuggestRequestBuilder builder = client().prepareSuggest();
1283             if (suggestText != null) {
1284                 builder.setSuggestText(suggestText);
1285             }
1286             for (SuggestionBuilder<?> suggestion : suggestions) {
1287                 builder.addSuggestion(suggestion);
1288             }
1289 
1290             SuggestResponse actionGet = builder.execute().actionGet();
1291             assertThat(Arrays.toString(actionGet.getShardFailures()), actionGet.getFailedShards(), equalTo(expectShardsFailed));
1292             if (expectShardsFailed > 0) {
1293                 throw new SearchPhaseExecutionException("suggest", "Suggest execution failed", new ShardSearchFailure[0]);
