85      private void assertRecoveryStateWithoutStage(RecoveryState state, int shardId, Type type,
86                                                   String sourceNode, String targetNode, boolean hasRestoreSource) {
87          assertThat(state.getShardId().getId(), equalTo(shardId));
88          assertThat(state.getType(), equalTo(type));
89          if (sourceNode == null) {
90              assertNull(state.getSourceNode());
91          } else {
92              assertNotNull(state.getSourceNode());
93              assertThat(state.getSourceNode().getName(), equalTo(sourceNode));
94          }
95          if (targetNode == null) {
96              assertNull(state.getTargetNode());
97          } else {
98              assertNotNull(state.getTargetNode());
99              assertThat(state.getTargetNode().getName(), equalTo(targetNode));
100         }
101         if (hasRestoreSource) {
102             assertNotNull(state.getRestoreSource());
103         } else {
104             assertNull(state.getRestoreSource());
105         }
106 
107     }
108 
109     private void assertRecoveryState(RecoveryState state, int shardId, Type type, Stage stage,
110                                      String sourceNode, String targetNode, boolean hasRestoreSource) {
111         assertRecoveryStateWithoutStage(state, shardId, type, sourceNode, targetNode, hasRestoreSource);
112         assertThat(state.getStage(), equalTo(stage));
113     }
114 
115     private void assertOnGoingRecoveryState(RecoveryState state, int shardId, Type type,
116                                             String sourceNode, String targetNode, boolean hasRestoreSource) {
117         assertRecoveryStateWithoutStage(state, shardId, type, sourceNode, targetNode, hasRestoreSource);
118         assertThat(state.getStage(), not(equalTo(Stage.DONE)));
119     }

121     private void slowDownRecovery(ByteSizeValue shardSize) {
122         long chunkSize = shardSize.bytes() / 10;
123         assertTrue(client().admin().cluster().prepareUpdateSettings()
124                 .setTransientSettings(ImmutableSettings.builder()
125                                 // one chunk per sec..
126                                 .put(RecoverySettings.INDICES_RECOVERY_MAX_BYTES_PER_SEC, chunkSize)
127                                 .put(RecoverySettings.INDICES_RECOVERY_FILE_CHUNK_SIZE, chunkSize)
128                 )
129                 .get().isAcknowledged());
130     }
131 
132     private void restoreRecoverySpeed() {
133         assertTrue(client().admin().cluster().prepareUpdateSettings()
134                 .setTransientSettings(ImmutableSettings.builder()
135                                 .put(RecoverySettings.INDICES_RECOVERY_MAX_BYTES_PER_SEC, "20mb")
136                                 .put(RecoverySettings.INDICES_RECOVERY_FILE_CHUNK_SIZE, "512kb")
137                 )
138                 .get().isAcknowledged());
139     }

186     @Test
187     public void replicaRecoveryTest() throws Exception {
188         logger.info("--> start node A");
189         String nodeA = internalCluster().startNode();
190 
191         logger.info("--> create index on node: {}", nodeA);
192         createAndPopulateIndex(INDEX_NAME, 1, SHARD_COUNT, REPLICA_COUNT);
193 
194         logger.info("--> start node B");
195         String nodeB = internalCluster().startNode();
196         ensureGreen();
197 
198         // force a shard recovery from nodeA to nodeB
199         logger.info("--> bump replica count");
200         client().admin().indices().prepareUpdateSettings(INDEX_NAME)
201                 .setSettings(settingsBuilder().put("number_of_replicas", 1)).execute().actionGet();
202         ensureGreen();
203 
204         logger.info("--> request recoveries");
205         RecoveryResponse response = client().admin().indices().prepareRecoveries(INDEX_NAME).execute().actionGet();
206 
207         // we should now have two total shards, one primary and one replica
208         List<ShardRecoveryResponse> shardResponses = response.shardResponses().get(INDEX_NAME);
209         assertThat(shardResponses.size(), equalTo(2));
210 
211         List<ShardRecoveryResponse> nodeAResponses = findRecoveriesForTargetNode(nodeA, shardResponses);
212         assertThat(nodeAResponses.size(), equalTo(1));
213         List<ShardRecoveryResponse> nodeBResponses = findRecoveriesForTargetNode(nodeB, shardResponses);
214         assertThat(nodeBResponses.size(), equalTo(1));
215 
216         // validate node A recovery
217         ShardRecoveryResponse nodeAShardResponse = nodeAResponses.get(0);
218         assertRecoveryState(nodeAShardResponse.recoveryState(), 0, Type.GATEWAY, Stage.DONE, nodeA, nodeA, false);
219         validateIndexRecoveryState(nodeAShardResponse.recoveryState().getIndex());
220 
221         // validate node B recovery
222         ShardRecoveryResponse nodeBShardResponse = nodeBResponses.get(0);
223         assertRecoveryState(nodeBShardResponse.recoveryState(), 0, Type.REPLICA, Stage.DONE, nodeA, nodeB, false);
224         validateIndexRecoveryState(nodeBShardResponse.recoveryState().getIndex());
225     }

228     @TestLogging("indices.recovery:TRACE")
229     public void rerouteRecoveryTest() throws Exception {
230         logger.info("--> start node A");
231         String nodeA = internalCluster().startNode();
232 
233         logger.info("--> create index on node: {}", nodeA);
234         ByteSizeValue shardSize = createAndPopulateIndex(INDEX_NAME, 1, SHARD_COUNT, REPLICA_COUNT).getShards()[0].getStats().getStore().size();
235 
236         logger.info("--> start node B");
237         String nodeB = internalCluster().startNode();
238 
239         ensureGreen();
240 
241         logger.info("--> slowing down recoveries");
242         slowDownRecovery(shardSize);
243 
244         logger.info("--> move shard from: {} to: {}", nodeA, nodeB);
245         client().admin().cluster().prepareReroute()
246                 .add(new MoveAllocationCommand(new ShardId(INDEX_NAME, 0), nodeA, nodeB))
247                 .execute().actionGet().getState();
248 
249 
250         logger.info("--> request recoveries");
251         RecoveryResponse response = client().admin().indices().prepareRecoveries(INDEX_NAME).execute().actionGet();
252 
253         List<ShardRecoveryResponse> shardResponses = response.shardResponses().get(INDEX_NAME);
254         List<ShardRecoveryResponse> nodeAResponses = findRecoveriesForTargetNode(nodeA, shardResponses);
255         assertThat(nodeAResponses.size(), equalTo(1));
256         List<ShardRecoveryResponse> nodeBResponses = findRecoveriesForTargetNode(nodeB, shardResponses);
257         assertThat(nodeBResponses.size(), equalTo(1));
258 
259         assertRecoveryState(nodeAResponses.get(0).recoveryState(), 0, Type.GATEWAY, Stage.DONE, nodeA, nodeA, false);
260         validateIndexRecoveryState(nodeAResponses.get(0).recoveryState().getIndex());
261 
262         assertOnGoingRecoveryState(nodeBResponses.get(0).recoveryState(), 0, Type.RELOCATION, nodeA, nodeB, false);
263         validateIndexRecoveryState(nodeBResponses.get(0).recoveryState().getIndex());
264 
265         logger.info("--> speeding up recoveries");
266         restoreRecoverySpeed();
267 
268         // wait for it to be finished
269         ensureGreen();
270 
271         response = client().admin().indices().prepareRecoveries(INDEX_NAME).execute().actionGet();
272 
273         shardResponses = response.shardResponses().get(INDEX_NAME);
274         assertThat(shardResponses.size(), equalTo(1));
275 
276         assertRecoveryState(shardResponses.get(0).recoveryState(), 0, Type.RELOCATION, Stage.DONE, nodeA, nodeB, false);
277         validateIndexRecoveryState(shardResponses.get(0).recoveryState().getIndex());
278 
279         logger.info("--> bump replica count");
280         client().admin().indices().prepareUpdateSettings(INDEX_NAME)
281                 .setSettings(settingsBuilder().put("number_of_replicas", 1)).execute().actionGet();
282         ensureGreen();
283 
284         logger.info("--> start node C");
285         String nodeC = internalCluster().startNode();
286         assertFalse(client().admin().cluster().prepareHealth().setWaitForNodes("3").get().isTimedOut());
287 
288         logger.info("--> slowing down recoveries");
289         slowDownRecovery(shardSize);
290 
291         logger.info("--> move replica shard from: {} to: {}", nodeA, nodeC);
292         client().admin().cluster().prepareReroute()
293                 .add(new MoveAllocationCommand(new ShardId(INDEX_NAME, 0), nodeA, nodeC))
294                 .execute().actionGet().getState();
295 
296         response = client().admin().indices().prepareRecoveries(INDEX_NAME).execute().actionGet();
297         shardResponses = response.shardResponses().get(INDEX_NAME);
298 
299         nodeAResponses = findRecoveriesForTargetNode(nodeA, shardResponses);
300         assertThat(nodeAResponses.size(), equalTo(1));
301         nodeBResponses = findRecoveriesForTargetNode(nodeB, shardResponses);
302         assertThat(nodeBResponses.size(), equalTo(1));
303         List<ShardRecoveryResponse> nodeCResponses = findRecoveriesForTargetNode(nodeC, shardResponses);
304         assertThat(nodeCResponses.size(), equalTo(1));
305 
306         assertRecoveryState(nodeAResponses.get(0).recoveryState(), 0, Type.REPLICA, Stage.DONE, nodeB, nodeA, false);
307         validateIndexRecoveryState(nodeAResponses.get(0).recoveryState().getIndex());
308 
309         assertRecoveryState(nodeBResponses.get(0).recoveryState(), 0, Type.RELOCATION, Stage.DONE, nodeA, nodeB, false);
310         validateIndexRecoveryState(nodeBResponses.get(0).recoveryState().getIndex());
311 
312         // relocations of replicas are marked as REPLICA and the source node is the node holding the primary (B)
313         assertOnGoingRecoveryState(nodeCResponses.get(0).recoveryState(), 0, Type.REPLICA, nodeB, nodeC, false);
314         validateIndexRecoveryState(nodeCResponses.get(0).recoveryState().getIndex());
315 
316         logger.info("--> speeding up recoveries");
317         restoreRecoverySpeed();
318         ensureGreen();
319 
320         response = client().admin().indices().prepareRecoveries(INDEX_NAME).execute().actionGet();
321         shardResponses = response.shardResponses().get(INDEX_NAME);
322 
323         nodeAResponses = findRecoveriesForTargetNode(nodeA, shardResponses);
324         assertThat(nodeAResponses.size(), equalTo(0));
325         nodeBResponses = findRecoveriesForTargetNode(nodeB, shardResponses);
326         assertThat(nodeBResponses.size(), equalTo(1));
327         nodeCResponses = findRecoveriesForTargetNode(nodeC, shardResponses);
328         assertThat(nodeCResponses.size(), equalTo(1));
329 
330         assertRecoveryState(nodeBResponses.get(0).recoveryState(), 0, Type.RELOCATION, Stage.DONE, nodeA, nodeB, false);
331         validateIndexRecoveryState(nodeBResponses.get(0).recoveryState().getIndex());
332 
333         // relocations of replicas are marked as REPLICA and the source node is the node holding the primary (B)
334         assertRecoveryState(nodeCResponses.get(0).recoveryState(), 0, Type.REPLICA, Stage.DONE, nodeB, nodeC, false);
335         validateIndexRecoveryState(nodeCResponses.get(0).recoveryState().getIndex());
336     }

338     @Test
339     public void snapshotRecoveryTest() throws Exception {
340         logger.info("--> start node A");
341         String nodeA = internalCluster().startNode();
342 
343         logger.info("--> create repository");
344         assertAcked(client().admin().cluster().preparePutRepository(REPO_NAME)
345                 .setType("fs").setSettings(ImmutableSettings.settingsBuilder()
346                                 .put("location", newTempDir(LifecycleScope.SUITE))
347                                 .put("compress", false)
348                 ).get());
349 
350         ensureGreen();
351 
352         logger.info("--> create index on node: {}", nodeA);
353         createAndPopulateIndex(INDEX_NAME, 1, SHARD_COUNT, REPLICA_COUNT);
354 
355         logger.info("--> snapshot");
356         CreateSnapshotResponse createSnapshotResponse = client().admin().cluster().prepareCreateSnapshot(REPO_NAME, SNAP_NAME)
357                 .setWaitForCompletion(true).setIndices(INDEX_NAME).get();
358         assertThat(createSnapshotResponse.getSnapshotInfo().successfulShards(), greaterThan(0));
359         assertThat(createSnapshotResponse.getSnapshotInfo().successfulShards(), equalTo(createSnapshotResponse.getSnapshotInfo().totalShards()));
360 
361         assertThat(client().admin().cluster().prepareGetSnapshots(REPO_NAME).setSnapshots(SNAP_NAME).get()
362                 .getSnapshots().get(0).state(), equalTo(SnapshotState.SUCCESS));
363 
364         client().admin().indices().prepareClose(INDEX_NAME).execute().actionGet();
365 
366         logger.info("--> restore");
367         RestoreSnapshotResponse restoreSnapshotResponse = client().admin().cluster()
368                 .prepareRestoreSnapshot(REPO_NAME, SNAP_NAME).setWaitForCompletion(true).execute().actionGet();
369         int totalShards = restoreSnapshotResponse.getRestoreInfo().totalShards();
370         assertThat(totalShards, greaterThan(0));
371 
372         ensureGreen();
373 
374         logger.info("--> request recoveries");
375         RecoveryResponse response = client().admin().indices().prepareRecoveries(INDEX_NAME).execute().actionGet();
376 
377         for (Map.Entry<String, List<ShardRecoveryResponse>> shardRecoveryResponse : response.shardResponses().entrySet()) {
378 
379             assertThat(shardRecoveryResponse.getKey(), equalTo(INDEX_NAME));
380             List<ShardRecoveryResponse> shardRecoveryResponses = shardRecoveryResponse.getValue();
381             assertThat(shardRecoveryResponses.size(), equalTo(totalShards));
382 
383             for (ShardRecoveryResponse shardResponse : shardRecoveryResponses) {
384                 assertRecoveryState(shardResponse.recoveryState(), 0, Type.SNAPSHOT, Stage.DONE, null, nodeA, true);
385                 validateIndexRecoveryState(shardResponse.recoveryState().getIndex());
386             }
387         }
388     }

390     private List<ShardRecoveryResponse> findRecoveriesForTargetNode(String nodeName, List<ShardRecoveryResponse> responses) {
391         List<ShardRecoveryResponse> nodeResponses = new ArrayList<>();
392         for (ShardRecoveryResponse response : responses) {
393             if (response.recoveryState().getTargetNode().getName().equals(nodeName)) {
394                 nodeResponses.add(response);
395             }
396         }
397         return nodeResponses;
398     }

400     private IndicesStatsResponse createAndPopulateIndex(String name, int nodeCount, int shardCount, int replicaCount)
401             throws ExecutionException, InterruptedException {
402 
403         logger.info("--> creating test index: {}", name);
404         assertAcked(prepareCreate(name, nodeCount, settingsBuilder().put("number_of_shards", shardCount)
405                 .put("number_of_replicas", replicaCount).put(Store.INDEX_STORE_STATS_REFRESH_INTERVAL, 0)));
406         ensureGreen();
407 
408         logger.info("--> indexing sample data");
409         final int numDocs = between(MIN_DOC_COUNT, MAX_DOC_COUNT);
410         final IndexRequestBuilder[] docs = new IndexRequestBuilder[numDocs];
411 
412         for (int i = 0; i < numDocs; i++) {
413             docs[i] = client().prepareIndex(INDEX_NAME, INDEX_TYPE).
414                     setSource("foo-int", randomInt(),
415                             "foo-string", randomAsciiOfLength(32),
416                             "foo-float", randomFloat());
417         }
418 
419         indexRandom(true, docs);
420         flush();
421         assertThat(client().prepareCount(INDEX_NAME).get().getCount(), equalTo((long) numDocs));
422         return client().admin().indices().prepareStats(INDEX_NAME).execute().actionGet();
423     }

433     @Test
434     public void disconnectsWhileRecoveringTest() throws Exception {
435         final String indexName = "test";
436         final Settings nodeSettings = ImmutableSettings.builder()
437                 .put(RecoverySettings.INDICES_RECOVERY_RETRY_DELAY_NETWORK, "100ms")
438                 .put(RecoverySettings.INDICES_RECOVERY_INTERNAL_ACTION_TIMEOUT, "1s")
439                 .put("cluster.routing.schedule", "100ms") // aggressive reroute post shard failures
440                 .put(TransportModule.TRANSPORT_SERVICE_TYPE_KEY, MockTransportService.class.getName())
441                 .put(MockDirectoryHelper.RANDOM_PREVENT_DOUBLE_WRITE, false) // restarted recoveries will delete temp files and write them again
442                 .build();
443         // start a master node
444         internalCluster().startNode(nodeSettings);
445 
446         ListenableFuture<String> blueFuture = internalCluster().startNodeAsync(ImmutableSettings.builder().put("node.color", "blue").put(nodeSettings).build());
447         ListenableFuture<String> redFuture = internalCluster().startNodeAsync(ImmutableSettings.builder().put("node.color", "red").put(nodeSettings).build());
448         final String blueNodeName = blueFuture.get();
449         final String redNodeName = redFuture.get();
450 
451         ClusterHealthResponse response = client().admin().cluster().prepareHealth().setWaitForNodes(">=3").get();
452         assertThat(response.isTimedOut(), is(false));
453 
454 
455         client().admin().indices().prepareCreate(indexName)
456                 .setSettings(
457                         ImmutableSettings.builder()
458                                 .put(FilterAllocationDecider.INDEX_ROUTING_INCLUDE_GROUP + "color", "blue")
459                                 .put(IndexMetaData.SETTING_NUMBER_OF_SHARDS, 1)
460                                 .put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, 0)
461                 ).get();
462 
463         List<IndexRequestBuilder> requests = new ArrayList<>();
464         int numDocs = scaledRandomIntBetween(25, 250);
465         for (int i = 0; i < numDocs; i++) {
466             requests.add(client().prepareIndex(indexName, "type").setCreate(true).setSource("{}"));
467         }
468         indexRandom(true, requests);
469         ensureSearchable(indexName);
470 
471         ClusterStateResponse stateResponse = client().admin().cluster().prepareState().get();
472         final String blueNodeId = internalCluster().getInstance(DiscoveryService.class, blueNodeName).localNode().id();
473 
474         assertFalse(stateResponse.getState().readOnlyRoutingNodes().node(blueNodeId).isEmpty());
475 
476         SearchResponse searchResponse = client().prepareSearch(indexName).get();
477         assertHitCount(searchResponse, numDocs);
478 
479         String[] recoveryActions = new String[]{
480                 RecoverySource.Actions.START_RECOVERY,
481                 RecoveryTarget.Actions.FILES_INFO,
482                 RecoveryTarget.Actions.FILE_CHUNK,
483                 RecoveryTarget.Actions.CLEAN_FILES,
484                 //RecoveryTarget.Actions.TRANSLOG_OPS, <-- may not be sent if already flushed
485                 RecoveryTarget.Actions.PREPARE_TRANSLOG,
486                 RecoveryTarget.Actions.FINALIZE
487         };
488         final String recoveryActionToBlock = randomFrom(recoveryActions);
489         final boolean dropRequests = randomBoolean();
490         logger.info("--> will {} between blue & red on [{}]", dropRequests ? "drop requests" : "break connection", recoveryActionToBlock);
491 
492         MockTransportService blueMockTransportService = (MockTransportService) internalCluster().getInstance(TransportService.class, blueNodeName);
493         MockTransportService redMockTransportService = (MockTransportService) internalCluster().getInstance(TransportService.class, redNodeName);
494         DiscoveryNode redDiscoNode = internalCluster().getInstance(ClusterService.class, redNodeName).localNode();
495         DiscoveryNode blueDiscoNode = internalCluster().getInstance(ClusterService.class, blueNodeName).localNode();
496         final CountDownLatch requestBlocked = new CountDownLatch(1);
497 
498         blueMockTransportService.addDelegate(redDiscoNode, new RecoveryActionBlocker(dropRequests, recoveryActionToBlock, blueMockTransportService.original(), requestBlocked));
499         redMockTransportService.addDelegate(blueDiscoNode, new RecoveryActionBlocker(dropRequests, recoveryActionToBlock, redMockTransportService.original(), requestBlocked));
500 
501         logger.info("--> starting recovery from blue to red");
502         client().admin().indices().prepareUpdateSettings(indexName).setSettings(
503                 ImmutableSettings.builder()
504                         .put(FilterAllocationDecider.INDEX_ROUTING_INCLUDE_GROUP + "color", "red,blue")
505                         .put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, 1)
506         ).get();
507 
508         requestBlocked.await();
509 
510         logger.info("--> stopping to block recovery");
511         blueMockTransportService.clearAllRules();
512         redMockTransportService.clearAllRules();
513 
514         ensureGreen();
515         searchResponse = client(redNodeName).prepareSearch(indexName).setPreference("_local").get();
516         assertHitCount(searchResponse, numDocs);
517 
518     }

520     private class RecoveryActionBlocker extends MockTransportService.DelegateTransport {
521         private final boolean dropRequests;
522         private final String recoveryActionToBlock;
523         private final CountDownLatch requestBlocked;
524 
525         public RecoveryActionBlocker(boolean dropRequests, String recoveryActionToBlock, Transport delegate, CountDownLatch requestBlocked) {
526             super(delegate);
527             this.dropRequests = dropRequests;
528             this.recoveryActionToBlock = recoveryActionToBlock;
529             this.requestBlocked = requestBlocked;
530         }
531 
532         @Override
533         public void sendRequest(DiscoveryNode node, long requestId, String action, TransportRequest request, TransportRequestOptions options) throws IOException, TransportException {
534             if (recoveryActionToBlock.equals(action) || requestBlocked.getCount() == 0) {
535                 logger.info("--> preventing {} request", action);
536                 requestBlocked.countDown();
537                 if (dropRequests) {
538                     return;
539                 }
540                 throw new ConnectTransportException(node, "DISCONNECT: prevented " + action + " request");
541             }
542             transport.sendRequest(node, requestId, action, request, options);
543         }
