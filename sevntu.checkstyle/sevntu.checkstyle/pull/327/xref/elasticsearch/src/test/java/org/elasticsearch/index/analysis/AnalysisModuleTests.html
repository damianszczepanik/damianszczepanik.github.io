68      public AnalysisService getAnalysisService(Settings settings) {
69          Index index = new Index("test");
70          Injector parentInjector = new ModulesBuilder().add(new SettingsModule(settings), new EnvironmentModule(new Environment(settings)), new IndicesAnalysisModule()).createInjector();
71          injector = new ModulesBuilder().add(
72                  new IndexSettingsModule(index, settings),
73                  new IndexNameModule(index),
74                  new AnalysisModule(settings, parentInjector.getInstance(IndicesAnalysisService.class)))
75                  .createChildInjector(parentInjector);
76  
77          return injector.getInstance(AnalysisService.class);
78      }

104     @Test
105     public void testVersionedAnalyzers() throws Exception {
106         Settings settings2 = settingsBuilder().loadFromClasspath("org/elasticsearch/index/analysis/test1.yml")
107                 .put(IndexMetaData.SETTING_VERSION_CREATED, Version.V_0_90_0).build();
108         AnalysisService analysisService2 = getAnalysisService(settings2);
109 
110         // indicesanalysisservice always has the current version
111         IndicesAnalysisService indicesAnalysisService2 = injector.getInstance(IndicesAnalysisService.class);
112         assertThat(indicesAnalysisService2.analyzer("default"), is(instanceOf(NamedAnalyzer.class)));
113         NamedAnalyzer defaultNamedAnalyzer = (NamedAnalyzer) indicesAnalysisService2.analyzer("default");
114         assertThat(defaultNamedAnalyzer.analyzer(), is(instanceOf(StandardAnalyzer.class)));
115         assertEquals(Version.CURRENT.luceneVersion, defaultNamedAnalyzer.analyzer().getVersion());
116 
117         // analysis service has the expected version
118         assertThat(analysisService2.analyzer("standard").analyzer(), is(instanceOf(StandardAnalyzer.class)));
119         assertEquals(Version.V_0_90_0.luceneVersion, analysisService2.analyzer("standard").analyzer().getVersion());
120         assertEquals(Version.V_0_90_0.luceneVersion, analysisService2.analyzer("thai").analyzer().getVersion());
121     }

123     private void assertTokenFilter(String name, Class clazz) throws IOException {
124         AnalysisService analysisService = AnalysisTestsHelper.createAnalysisServiceFromSettings(ImmutableSettings.settingsBuilder().put(IndexMetaData.SETTING_VERSION_CREATED, Version.CURRENT).build());
125         TokenFilterFactory tokenFilter = analysisService.tokenFilter(name);
126         Tokenizer tokenizer = new WhitespaceTokenizer();
127         tokenizer.setReader(new StringReader("foo bar"));
128         TokenStream stream = tokenFilter.create(tokenizer);
129         assertThat(stream, instanceOf(clazz));
130     }

201     public void testWordListPath() throws Exception {
202         Environment env = new Environment(ImmutableSettings.Builder.EMPTY_SETTINGS);
203         String[] words = new String[]{"donau", "dampf", "schiff", "spargel", "creme", "suppe"};
204 
205         Path wordListFile = generateWordList(words);
206         Settings settings = settingsBuilder().loadFromSource("index: \n  word_list_path: " + wordListFile.toAbsolutePath()).build();
207 
208         Set<?> wordList = Analysis.getWordSet(env, settings, "index.word_list");
209         MatcherAssert.assertThat(wordList.size(), equalTo(6));
210 //        MatcherAssert.assertThat(wordList, hasItems(words));
211         Files.delete(wordListFile);
212     }

214     private Path generateWordList(String[] words) throws Exception {
215         Path wordListFile = newTempDirPath().resolve("wordlist.txt");
216         try (BufferedWriter writer = Files.newBufferedWriter(wordListFile, StandardCharsets.UTF_8)) {
217             for (String word : words) {
218                 writer.write(word);
219                 writer.write('\n');
220             }
221         }
222         return wordListFile;
223     }
