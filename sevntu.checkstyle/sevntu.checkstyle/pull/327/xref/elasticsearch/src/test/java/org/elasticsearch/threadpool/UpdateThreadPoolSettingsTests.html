43      private ThreadPool.Info info(ThreadPool threadPool, String name) {
44          for (ThreadPool.Info info : threadPool.info()) {
45              if (info.getName().equals(name)) {
46                  return info;
47              }
48          }
49          return null;
50      }

52      @Test
53      public void testCachedExecutorType() throws InterruptedException {
54          ThreadPool threadPool = new ThreadPool(
55                  ImmutableSettings.settingsBuilder()
56                          .put("threadpool.search.type", "cached")
57                          .put("name","testCachedExecutorType").build(), null);
58  
59          assertThat(info(threadPool, Names.SEARCH).getType(), equalTo("cached"));
60          assertThat(info(threadPool, Names.SEARCH).getKeepAlive().minutes(), equalTo(5L));
61          assertThat(threadPool.executor(Names.SEARCH), instanceOf(EsThreadPoolExecutor.class));
62  
63          // Replace with different type
64          threadPool.updateSettings(settingsBuilder().put("threadpool.search.type", "same").build());
65          assertThat(info(threadPool, Names.SEARCH).getType(), equalTo("same"));
66          assertThat(threadPool.executor(Names.SEARCH), instanceOf(MoreExecutors.directExecutor().getClass()));
67  
68          // Replace with different type again
69          threadPool.updateSettings(settingsBuilder()
70                  .put("threadpool.search.type", "scaling")
71                  .put("threadpool.search.keep_alive", "10m")
72                  .build());
73          assertThat(info(threadPool, Names.SEARCH).getType(), equalTo("scaling"));
74          assertThat(threadPool.executor(Names.SEARCH), instanceOf(EsThreadPoolExecutor.class));
75          assertThat(((EsThreadPoolExecutor) threadPool.executor(Names.SEARCH)).getCorePoolSize(), equalTo(1));
76          // Make sure keep alive value changed
77          assertThat(info(threadPool, Names.SEARCH).getKeepAlive().minutes(), equalTo(10L));
78          assertThat(((EsThreadPoolExecutor) threadPool.executor(Names.SEARCH)).getKeepAliveTime(TimeUnit.MINUTES), equalTo(10L));
79  
80          // Put old type back
81          threadPool.updateSettings(settingsBuilder().put("threadpool.search.type", "cached").build());
82          assertThat(info(threadPool, Names.SEARCH).getType(), equalTo("cached"));
83          // Make sure keep alive value reused
84          assertThat(info(threadPool, Names.SEARCH).getKeepAlive().minutes(), equalTo(10L));
85          assertThat(threadPool.executor(Names.SEARCH), instanceOf(EsThreadPoolExecutor.class));
86  
87          // Change keep alive
88          Executor oldExecutor = threadPool.executor(Names.SEARCH);
89          threadPool.updateSettings(settingsBuilder().put("threadpool.search.keep_alive", "1m").build());
90          // Make sure keep alive value changed
91          assertThat(info(threadPool, Names.SEARCH).getKeepAlive().minutes(), equalTo(1L));
92          assertThat(((EsThreadPoolExecutor) threadPool.executor(Names.SEARCH)).getKeepAliveTime(TimeUnit.MINUTES), equalTo(1L));
93          // Make sure executor didn't change
94          assertThat(info(threadPool, Names.SEARCH).getType(), equalTo("cached"));
95          assertThat(threadPool.executor(Names.SEARCH), sameInstance(oldExecutor));
96  
97          // Set the same keep alive
98          threadPool.updateSettings(settingsBuilder().put("threadpool.search.keep_alive", "1m").build());
99          // Make sure keep alive value didn't change
100         assertThat(info(threadPool, Names.SEARCH).getKeepAlive().minutes(), equalTo(1L));
101         assertThat(((EsThreadPoolExecutor) threadPool.executor(Names.SEARCH)).getKeepAliveTime(TimeUnit.MINUTES), equalTo(1L));
102         // Make sure executor didn't change
103         assertThat(info(threadPool, Names.SEARCH).getType(), equalTo("cached"));
104         assertThat(threadPool.executor(Names.SEARCH), sameInstance(oldExecutor));
105         terminate(threadPool);
106     }

108     @Test
109     public void testFixedExecutorType() throws InterruptedException {
110         ThreadPool threadPool = new ThreadPool(settingsBuilder()
111                 .put("threadpool.search.type", "fixed")
112                 .put("name","testCachedExecutorType").build(), null);
113 
114         assertThat(threadPool.executor(Names.SEARCH), instanceOf(EsThreadPoolExecutor.class));
115 
116         // Replace with different type
117         threadPool.updateSettings(settingsBuilder()
118                 .put("threadpool.search.type", "scaling")
119                 .put("threadpool.search.keep_alive", "10m")
120                 .put("threadpool.search.min", "2")
121                 .put("threadpool.search.size", "15")
122                 .build());
123         assertThat(info(threadPool, Names.SEARCH).getType(), equalTo("scaling"));
124         assertThat(threadPool.executor(Names.SEARCH), instanceOf(EsThreadPoolExecutor.class));
125         assertThat(((EsThreadPoolExecutor) threadPool.executor(Names.SEARCH)).getCorePoolSize(), equalTo(2));
126         assertThat(((EsThreadPoolExecutor) threadPool.executor(Names.SEARCH)).getMaximumPoolSize(), equalTo(15));
127         assertThat(info(threadPool, Names.SEARCH).getMin(), equalTo(2));
128         assertThat(info(threadPool, Names.SEARCH).getMax(), equalTo(15));
129         // Make sure keep alive value changed
130         assertThat(info(threadPool, Names.SEARCH).getKeepAlive().minutes(), equalTo(10L));
131         assertThat(((EsThreadPoolExecutor) threadPool.executor(Names.SEARCH)).getKeepAliveTime(TimeUnit.MINUTES), equalTo(10L));
132 
133         // Put old type back
134         threadPool.updateSettings(settingsBuilder()
135                 .put("threadpool.search.type", "fixed")
136                 .build());
137         assertThat(info(threadPool, Names.SEARCH).getType(), equalTo("fixed"));
138         // Make sure keep alive value is not used
139         assertThat(info(threadPool, Names.SEARCH).getKeepAlive(), nullValue());
140         // Make sure keep pool size value were reused
141         assertThat(info(threadPool, Names.SEARCH).getMin(), equalTo(15));
142         assertThat(info(threadPool, Names.SEARCH).getMax(), equalTo(15));
143         assertThat(threadPool.executor(Names.SEARCH), instanceOf(EsThreadPoolExecutor.class));
144         assertThat(((EsThreadPoolExecutor) threadPool.executor(Names.SEARCH)).getCorePoolSize(), equalTo(15));
145         assertThat(((EsThreadPoolExecutor) threadPool.executor(Names.SEARCH)).getMaximumPoolSize(), equalTo(15));
146 
147         // Change size
148         Executor oldExecutor = threadPool.executor(Names.SEARCH);
149         threadPool.updateSettings(settingsBuilder().put("threadpool.search.size", "10").build());
150         // Make sure size values changed
151         assertThat(info(threadPool, Names.SEARCH).getMax(), equalTo(10));
152         assertThat(info(threadPool, Names.SEARCH).getMin(), equalTo(10));
153         assertThat(((EsThreadPoolExecutor) threadPool.executor(Names.SEARCH)).getMaximumPoolSize(), equalTo(10));
154         assertThat(((EsThreadPoolExecutor) threadPool.executor(Names.SEARCH)).getCorePoolSize(), equalTo(10));
155         // Make sure executor didn't change
156         assertThat(info(threadPool, Names.SEARCH).getType(), equalTo("fixed"));
157         assertThat(threadPool.executor(Names.SEARCH), sameInstance(oldExecutor));
158 
159         // Change queue capacity
160         threadPool.updateSettings(settingsBuilder()
161                 .put("threadpool.search.queue", "500")
162                 .build());
163 
164         terminate(threadPool);
165     }

168     @Test
169     public void testScalingExecutorType() throws InterruptedException {
170         ThreadPool threadPool = new ThreadPool(settingsBuilder()
171                 .put("threadpool.search.type", "scaling")
172                 .put("threadpool.search.size", 10)
173                 .put("name","testCachedExecutorType").build(), null);
174 
175         assertThat(info(threadPool, Names.SEARCH).getMin(), equalTo(1));
176         assertThat(info(threadPool, Names.SEARCH).getMax(), equalTo(10));
177         assertThat(info(threadPool, Names.SEARCH).getKeepAlive().minutes(), equalTo(5L));
178         assertThat(info(threadPool, Names.SEARCH).getType(), equalTo("scaling"));
179         assertThat(threadPool.executor(Names.SEARCH), instanceOf(EsThreadPoolExecutor.class));
180 
181         // Change settings that doesn't require pool replacement
182         Executor oldExecutor = threadPool.executor(Names.SEARCH);
183         threadPool.updateSettings(settingsBuilder()
184                 .put("threadpool.search.type", "scaling")
185                 .put("threadpool.search.keep_alive", "10m")
186                 .put("threadpool.search.min", "2")
187                 .put("threadpool.search.size", "15")
188                 .build());
189         assertThat(info(threadPool, Names.SEARCH).getType(), equalTo("scaling"));
190         assertThat(threadPool.executor(Names.SEARCH), instanceOf(EsThreadPoolExecutor.class));
191         assertThat(((EsThreadPoolExecutor) threadPool.executor(Names.SEARCH)).getCorePoolSize(), equalTo(2));
192         assertThat(((EsThreadPoolExecutor) threadPool.executor(Names.SEARCH)).getMaximumPoolSize(), equalTo(15));
193         assertThat(info(threadPool, Names.SEARCH).getMin(), equalTo(2));
194         assertThat(info(threadPool, Names.SEARCH).getMax(), equalTo(15));
195         // Make sure keep alive value changed
196         assertThat(info(threadPool, Names.SEARCH).getKeepAlive().minutes(), equalTo(10L));
197         assertThat(((EsThreadPoolExecutor) threadPool.executor(Names.SEARCH)).getKeepAliveTime(TimeUnit.MINUTES), equalTo(10L));
198         assertThat(threadPool.executor(Names.SEARCH), sameInstance(oldExecutor));
199 
200         terminate(threadPool);
201     }

203     @Test(timeout = 10000)
204     public void testShutdownDownNowDoesntBlock() throws Exception {
205         ThreadPool threadPool = new ThreadPool(ImmutableSettings.settingsBuilder()
206                 .put("threadpool.search.type", "cached")
207                 .put("name","testCachedExecutorType").build(), null);
208 
209         final CountDownLatch latch = new CountDownLatch(1);
210         Executor oldExecutor = threadPool.executor(Names.SEARCH);
211         threadPool.executor(Names.SEARCH).execute(new Runnable() {
212             @Override
213             public void run() {
214                 try {
215                     Thread.sleep(20000);
216                 } catch (InterruptedException ex) {
217                     latch.countDown();
218                     Thread.currentThread().interrupt();
219                 }
220             }
221         });
222         threadPool.updateSettings(settingsBuilder().put("threadpool.search.type", "fixed").build());
223         assertThat(threadPool.executor(Names.SEARCH), not(sameInstance(oldExecutor)));
224         assertThat(((ThreadPoolExecutor) oldExecutor).isShutdown(), equalTo(true));
225         assertThat(((ThreadPoolExecutor) oldExecutor).isTerminating(), equalTo(true));
226         assertThat(((ThreadPoolExecutor) oldExecutor).isTerminated(), equalTo(false));
227         threadPool.shutdownNow(); // interrupt the thread
228         latch.await();
229         terminate(threadPool);
230     }

232     @Test
233     public void testCustomThreadPool() throws Exception {
234         ThreadPool threadPool = new ThreadPool(ImmutableSettings.settingsBuilder()
235                 .put("threadpool.my_pool1.type", "cached")
236                 .put("threadpool.my_pool2.type", "fixed")
237                 .put("threadpool.my_pool2.size", "1")
238                 .put("threadpool.my_pool2.queue_size", "1")
239                 .put("name", "testCustomThreadPool").build(), null);
240 
241         ThreadPoolInfo groups = threadPool.info();
242         boolean foundPool1 = false;
243         boolean foundPool2 = false;
244         outer: for (ThreadPool.Info info : groups) {
245             if ("my_pool1".equals(info.getName())) {
246                 foundPool1 = true;
247                 assertThat(info.getType(), equalTo("cached"));
248             } else if ("my_pool2".equals(info.getName())) {
249                 foundPool2 = true;
250                 assertThat(info.getType(), equalTo("fixed"));
251                 assertThat(info.getMin(), equalTo(1));
252                 assertThat(info.getMax(), equalTo(1));
253                 assertThat(info.getQueueSize().singles(), equalTo(1l));
254             } else {
255                 for (Field field : Names.class.getFields()) {
256                     if (info.getName().equalsIgnoreCase(field.getName())) {
257                         // This is ok it is a default thread pool
258                         continue outer;
259                     }
260                 }
261                 fail("Unexpected pool name: " + info.getName());
262             }
263         }
264         assertThat(foundPool1, is(true));
265         assertThat(foundPool2, is(true));
266 
267         // Updating my_pool2
268         Settings settings = ImmutableSettings.builder()
269                 .put("threadpool.my_pool2.size", "10")
270                 .build();
271         threadPool.updateSettings(settings);
272 
273         groups = threadPool.info();
274         foundPool1 = false;
275         foundPool2 = false;
276         outer: for (ThreadPool.Info info : groups) {
277             if ("my_pool1".equals(info.getName())) {
278                 foundPool1 = true;
279                 assertThat(info.getType(), equalTo("cached"));
280             } else if ("my_pool2".equals(info.getName())) {
281                 foundPool2 = true;
282                 assertThat(info.getMax(), equalTo(10));
283                 assertThat(info.getMin(), equalTo(10));
284                 assertThat(info.getQueueSize().singles(), equalTo(1l));
285                 assertThat(info.getType(), equalTo("fixed"));
286             } else {
287                 for (Field field : Names.class.getFields()) {
288                     if (info.getName().equalsIgnoreCase(field.getName())) {
289                         // This is ok it is a default thread pool
290                         continue outer;
291                     }
292                 }
293                 fail("Unexpected pool name: " + info.getName());
294             }
295         }
296         assertThat(foundPool1, is(true));
297         assertThat(foundPool2, is(true));
298         terminate(threadPool);
299     }
