50      @SuppressWarnings("unchecked")
51      @Test
52      public void testCopyToFieldsParsing() throws Exception {
53          String mapping = jsonBuilder().startObject().startObject("type1").startObject("properties")
54                  .startObject("copy_test")
55                  .field("type", "string")
56                  .array("copy_to", "another_field", "cyclic_test")
57                  .endObject()
58  
59                  .startObject("another_field")
60                  .field("type", "string")
61                  .endObject()
62  
63                  .startObject("cyclic_test")
64                  .field("type", "string")
65                  .array("copy_to", "copy_test")
66                  .endObject()
67  
68                  .startObject("int_to_str_test")
69                  .field("type", "integer")
70                  .array("copy_to",  "another_field", "new_field")
71                  .endObject()
72                  .endObject().endObject().endObject().string();
73  
74          DocumentMapper docMapper = createIndex("test").mapperService().documentMapperParser().parse(mapping);
75          FieldMapper fieldMapper = docMapper.mappers().name("copy_test").mapper();
76          assertThat(fieldMapper, instanceOf(StringFieldMapper.class));
77  
78          // Check json serialization
79          StringFieldMapper stringFieldMapper = (StringFieldMapper) fieldMapper;
80          XContentBuilder builder = jsonBuilder().startObject();
81          stringFieldMapper.toXContent(builder, ToXContent.EMPTY_PARAMS).endObject();
82          builder.close();
83          Map<String, Object> serializedMap = JsonXContent.jsonXContent.createParser(builder.bytes()).mapAndClose();
84          Map<String, Object> copyTestMap = (Map<String, Object>) serializedMap.get("copy_test");
85          assertThat(copyTestMap.get("type").toString(), is("string"));
86          List<String> copyToList = (List<String>) copyTestMap.get("copy_to");
87          assertThat(copyToList.size(), equalTo(2));
88          assertThat(copyToList.get(0).toString(), equalTo("another_field"));
89          assertThat(copyToList.get(1).toString(), equalTo("cyclic_test"));
90  
91          // Check data parsing
92          BytesReference json = jsonBuilder().startObject()
93                  .field("copy_test", "foo")
94                  .field("cyclic_test", "bar")
95                  .field("int_to_str_test", 42)
96                  .endObject().bytes();
97  
98          ParseContext.Document doc = docMapper.parse("type1", "1", json).rootDoc();
99          assertThat(doc.getFields("copy_test").length, equalTo(2));
100         assertThat(doc.getFields("copy_test")[0].stringValue(), equalTo("foo"));
101         assertThat(doc.getFields("copy_test")[1].stringValue(), equalTo("bar"));
102 
103         assertThat(doc.getFields("another_field").length, equalTo(2));
104         assertThat(doc.getFields("another_field")[0].stringValue(), equalTo("foo"));
105         assertThat(doc.getFields("another_field")[1].stringValue(), equalTo("42"));
106 
107         assertThat(doc.getFields("cyclic_test").length, equalTo(2));
108         assertThat(doc.getFields("cyclic_test")[0].stringValue(), equalTo("foo"));
109         assertThat(doc.getFields("cyclic_test")[1].stringValue(), equalTo("bar"));
110 
111         assertThat(doc.getFields("int_to_str_test").length, equalTo(1));
112         assertThat(doc.getFields("int_to_str_test")[0].numericValue().intValue(), equalTo(42));
113 
114         assertThat(doc.getFields("new_field").length, equalTo(1));
115         assertThat(doc.getFields("new_field")[0].numericValue().intValue(), equalTo(42));
116 
117         fieldMapper = docMapper.mappers().name("new_field").mapper();
118         assertThat(fieldMapper, instanceOf(LongFieldMapper.class));
119     }

184     @Test
185     public void testCopyToFieldMerge() throws Exception {
186 
187         String mappingBefore = jsonBuilder().startObject().startObject("type1").startObject("properties")
188 
189                 .startObject("copy_test")
190                 .field("type", "string")
191                 .array("copy_to", "foo", "bar")
192                 .endObject()
193 
194                 .endObject().endObject().endObject().string();
195 
196         String mappingAfter = jsonBuilder().startObject().startObject("type1").startObject("properties")
197 
198                 .startObject("copy_test")
199                 .field("type", "string")
200                 .array("copy_to", "baz", "bar")
201                 .endObject()
202 
203                 .endObject().endObject().endObject().string();
204 
205         DocumentMapperParser parser = createIndex("test").mapperService().documentMapperParser();
206         DocumentMapper docMapperBefore = parser.parse(mappingBefore);
207 
208         ImmutableList<String> fields = docMapperBefore.mappers().name("copy_test").mapper().copyTo().copyToFields();
209 
210         assertThat(fields.size(), equalTo(2));
211         assertThat(fields.get(0), equalTo("foo"));
212         assertThat(fields.get(1), equalTo("bar"));
213 
214 
215         DocumentMapper docMapperAfter = parser.parse(mappingAfter);
216 
217         DocumentMapper.MergeResult mergeResult = docMapperBefore.merge(docMapperAfter, mergeFlags().simulate(true));
218 
219         assertThat(Arrays.toString(mergeResult.conflicts()), mergeResult.hasConflicts(), equalTo(false));
220 
221         docMapperBefore.merge(docMapperAfter, mergeFlags().simulate(false));
222 
223         fields = docMapperBefore.mappers().name("copy_test").mapper().copyTo().copyToFields();
224 
225         assertThat(fields.size(), equalTo(2));
226         assertThat(fields.get(0), equalTo("baz"));
227         assertThat(fields.get(1), equalTo("bar"));
228     }

230     public void testCopyToNestedField() throws Exception {
231         IndexService indexService = createIndex("test");
232         DocumentMapperParser parser = indexService.mapperService().documentMapperParser();
233         for (boolean mapped : new boolean[] {true, false}) {
234             XContentBuilder mapping = jsonBuilder().startObject()
235                     .startObject("type")
236                         .startObject("properties")
237                             .startObject("n1")
238                                 .field("type", "nested")
239                                 .startObject("properties")
240                                     .startObject("n2")
241                                         .field("type", "nested")
242                                         .startObject("properties")
243                                             .startObject("source")
244                                                 .field("type", "long")
245                                                 .startArray("copy_to")
246                                                     .value("target") // should go to the root doc
247                                                     .value("n1.target") // should go to the parent doc
248                                                     .value("n1.n2.target") // should go to the current doc
249                                                 .endArray()
250                                             .endObject();
251             for (int i = 0; i < 3; ++i) {
252                 if (mapped) {
253                     mapping = mapping.startObject("target").field("type", "long").endObject();
254                 }
255                 mapping = mapping.endObject().endObject();
256             }
257             mapping = mapping.endObject();
258 
259             DocumentMapper mapper = parser.parse(mapping.string());
260 
261             XContentBuilder jsonDoc = XContentFactory.jsonBuilder()
262                     .startObject()
263                         .startArray("n1")
264                             .startObject()
265                                 .startArray("n2")
266                                     .startObject()
267                                         .field("source", 3)
268                                     .endObject()
269                                     .startObject()
270                                         .field("source", 5)
271                                     .endObject()
272                                 .endArray()
273                             .endObject()
274                             .startObject()
275                                 .startArray("n2")
276                                     .startObject()
277                                         .field("source", 7)
278                                     .endObject()
279                                 .endArray()
280                             .endObject()
281                         .endArray()
282                     .endObject();
283 
284             ParsedDocument doc = mapper.parse("type", "1", jsonDoc.bytes());
285             assertEquals(6, doc.docs().size());
286 
287             Document nested = doc.docs().get(0);
288             assertFieldValue(nested, "n1.n2.target", 7L);
289             assertFieldValue(nested, "n1.target");
290             assertFieldValue(nested, "target");
291 
292             nested = doc.docs().get(2);
293             assertFieldValue(nested, "n1.n2.target", 5L);
294             assertFieldValue(nested, "n1.target");
295             assertFieldValue(nested, "target");
296 
297             nested = doc.docs().get(3);
298             assertFieldValue(nested, "n1.n2.target", 3L);
299             assertFieldValue(nested, "n1.target");
300             assertFieldValue(nested, "target");
301 
302             Document parent = doc.docs().get(1);
303             assertFieldValue(parent, "target");
304             assertFieldValue(parent, "n1.target", 7L);
305             assertFieldValue(parent, "n1.n2.target");
306 
307             parent = doc.docs().get(4);
308             assertFieldValue(parent, "target");
309             assertFieldValue(parent, "n1.target", 3L, 5L);
310             assertFieldValue(parent, "n1.n2.target");
311 
312             Document root = doc.docs().get(5);
313             assertFieldValue(root, "target", 3L, 5L, 7L);
314             assertFieldValue(root, "n1.target");
315             assertFieldValue(root, "n1.n2.target");
316         }
317     }

319     private void assertFieldValue(Document doc, String field, Number... expected) {
320         IndexableField[] values = doc.getFields(field);
321         if (values == null) {
322             values = new IndexableField[0];
323         }
324         Number[] actual = new Number[values.length];
325         for (int i = 0; i < values.length; ++i) {
326             actual[i] = values[i].numericValue();
327         }
328         assertArrayEquals(expected, actual);
329     }
