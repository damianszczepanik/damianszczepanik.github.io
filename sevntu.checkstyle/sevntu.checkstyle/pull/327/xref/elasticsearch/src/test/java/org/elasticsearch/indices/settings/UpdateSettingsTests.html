51      @Test
52      public void testOpenCloseUpdateSettings() throws Exception {
53          createIndex("test");
54          try {
55              client().admin().indices().prepareUpdateSettings("test")
56                      .setSettings(ImmutableSettings.settingsBuilder()
57                              .put("index.refresh_interval", -1) // this one can change
58                              .put("index.cache.filter.type", "none") // this one can't
59                      )
60                      .execute().actionGet();
61              fail();
62          } catch (ElasticsearchIllegalArgumentException e) {
63              // all is well
64          }
65  
66          IndexMetaData indexMetaData = client().admin().cluster().prepareState().execute().actionGet().getState().metaData().index("test");
67          assertThat(indexMetaData.settings().get("index.refresh_interval"), nullValue());
68          assertThat(indexMetaData.settings().get("index.cache.filter.type"), nullValue());
69  
70          // Now verify via dedicated get settings api:
71          GetSettingsResponse getSettingsResponse = client().admin().indices().prepareGetSettings("test").get();
72          assertThat(getSettingsResponse.getSetting("test", "index.refresh_interval"), nullValue());
73          assertThat(getSettingsResponse.getSetting("test", "index.cache.filter.type"), nullValue());
74  
75          client().admin().indices().prepareUpdateSettings("test")
76                  .setSettings(ImmutableSettings.settingsBuilder()
77                          .put("index.refresh_interval", -1) // this one can change
78                  )
79                  .execute().actionGet();
80  
81          indexMetaData = client().admin().cluster().prepareState().execute().actionGet().getState().metaData().index("test");
82          assertThat(indexMetaData.settings().get("index.refresh_interval"), equalTo("-1"));
83          // Now verify via dedicated get settings api:
84          getSettingsResponse = client().admin().indices().prepareGetSettings("test").get();
85          assertThat(getSettingsResponse.getSetting("test", "index.refresh_interval"), equalTo("-1"));
86  
87          // now close the index, change the non dynamic setting, and see that it applies
88  
89          // Wait for the index to turn green before attempting to close it
90          ClusterHealthResponse health = client().admin().cluster().prepareHealth().setTimeout("30s").setWaitForEvents(Priority.LANGUID).setWaitForGreenStatus().execute().actionGet();
91          assertThat(health.isTimedOut(), equalTo(false));
92  
93          client().admin().indices().prepareClose("test").execute().actionGet();
94  
95          client().admin().indices().prepareUpdateSettings("test")
96                  .setSettings(ImmutableSettings.settingsBuilder()
97                          .put("index.refresh_interval", "1s") // this one can change
98                          .put("index.cache.filter.type", "none") // this one can't
99                  )
100                 .execute().actionGet();
101 
102         indexMetaData = client().admin().cluster().prepareState().execute().actionGet().getState().metaData().index("test");
103         assertThat(indexMetaData.settings().get("index.refresh_interval"), equalTo("1s"));
104         assertThat(indexMetaData.settings().get("index.cache.filter.type"), equalTo("none"));
105 
106         // Now verify via dedicated get settings api:
107         getSettingsResponse = client().admin().indices().prepareGetSettings("test").get();
108         assertThat(getSettingsResponse.getSetting("test", "index.refresh_interval"), equalTo("1s"));
109         assertThat(getSettingsResponse.getSetting("test", "index.cache.filter.type"), equalTo("none"));
110     }

113     public void testEngineGCDeletesSetting() throws InterruptedException {
114         createIndex("test");
115         client().prepareIndex("test", "type", "1").setSource("f", 1).get(); // set version to 1
116         client().prepareDelete("test", "type", "1").get(); // sets version to 2
117         client().prepareIndex("test", "type", "1").setSource("f", 2).setVersion(2).get(); // delete is still in cache this should work & set version to 3
118         client().admin().indices().prepareUpdateSettings("test")
119                 .setSettings(ImmutableSettings.settingsBuilder()
120                         .put("index.gc_deletes", 0)
121                 ).get();
122 
123         client().prepareDelete("test", "type", "1").get(); // sets version to 4
124         Thread.sleep(300); // wait for cache time to change TODO: this needs to be solved better. To be discussed.
125         assertThrows(client().prepareIndex("test", "type", "1").setSource("f", 3).setVersion(4), VersionConflictEngineException.class); // delete is should not be in cache

129     // #6626: make sure we can update throttle settings and the changes take effect
130     @Test
131     @Slow
132     public void testUpdateThrottleSettings() {
133 
134         // No throttling at first, only 1 non-replicated shard, force lots of merging:
135         assertAcked(prepareCreate("test")
136                     .setSettings(ImmutableSettings.builder()
137                                  .put(AbstractIndexStore.INDEX_STORE_THROTTLE_TYPE, "none")
138                                  .put(IndexMetaData.SETTING_NUMBER_OF_SHARDS, "1")
139                                  .put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, "0")
140                                  .put(TieredMergePolicyProvider.INDEX_MERGE_POLICY_MAX_MERGE_AT_ONCE, "2")
141                                  .put(TieredMergePolicyProvider.INDEX_MERGE_POLICY_SEGMENTS_PER_TIER, "2")
142                                  .put(ConcurrentMergeSchedulerProvider.MAX_THREAD_COUNT, "1")
143                                  .put(ConcurrentMergeSchedulerProvider.MAX_MERGE_COUNT, "2")
144                                  .put(Store.INDEX_STORE_STATS_REFRESH_INTERVAL, 0) // get stats all the time - no caching
145                                  ));
146         ensureGreen();
147         long termUpto = 0;
148         for(int i=0;i<100;i++) {
149             // Provoke slowish merging by making many unique terms:
150             StringBuilder sb = new StringBuilder();
151             for(int j=0;j<100;j++) {
152                 sb.append(' ');
153                 sb.append(termUpto++);
154             }
155             client().prepareIndex("test", "type", ""+termUpto).setSource("field" + (i%10), sb.toString()).get();
156             if (i % 2 == 0) {
157                 refresh();
158             }
159         }
160 
161         // No merge IO throttling should have happened:
162         NodesStatsResponse nodesStats = client().admin().cluster().prepareNodesStats().setIndices(true).get();
163         for(NodeStats stats : nodesStats.getNodes()) {
164             assertThat(stats.getIndices().getStore().getThrottleTime().getMillis(), equalTo(0l));
165         }
166 
167         logger.info("test: set low merge throttling");
168 
169         // Now updates settings to turn on merge throttling lowish rate
170         client()
171             .admin()
172             .indices()
173             .prepareUpdateSettings("test")
174             .setSettings(ImmutableSettings.builder()
175                          .put(AbstractIndexStore.INDEX_STORE_THROTTLE_TYPE, "merge")
176                          .put(AbstractIndexStore.INDEX_STORE_THROTTLE_MAX_BYTES_PER_SEC, "1mb"))
177             .get();
178 
179         // Make sure setting says it is in fact changed:
180         GetSettingsResponse getSettingsResponse = client().admin().indices().prepareGetSettings("test").get();
181         assertThat(getSettingsResponse.getSetting("test", AbstractIndexStore.INDEX_STORE_THROTTLE_TYPE), equalTo("merge"));
182 
183         // Also make sure we see throttling kicking in:
184         boolean done = false;
185         while (done == false) {
186             // Provoke slowish merging by making many unique terms:
187             for(int i=0;i<5;i++) {
188                 StringBuilder sb = new StringBuilder();
189                 for(int j=0;j<100;j++) {
190                     sb.append(' ');
191                     sb.append(termUpto++);
192                     sb.append(" some random text that keeps repeating over and over again hambone");
193                 }
194                 client().prepareIndex("test", "type", ""+termUpto).setSource("field" + (i%10), sb.toString()).get();
195             }
196             refresh();
197             nodesStats = client().admin().cluster().prepareNodesStats().setIndices(true).get();
198             for(NodeStats stats : nodesStats.getNodes()) {
199                 long throttleMillis = stats.getIndices().getStore().getThrottleTime().getMillis();
200                 if (throttleMillis > 0) {
201                     done = true;
202                     break;
203                 }
204             }
205         }
206 
207         logger.info("test: disable merge throttling");
208         
209         // Now updates settings to disable merge throttling
210         client()
211             .admin()
212             .indices()
213             .prepareUpdateSettings("test")
214             .setSettings(ImmutableSettings.builder()
215                          .put(AbstractIndexStore.INDEX_STORE_THROTTLE_TYPE, "none"))
216             .get();
217 
218         // Optimize does a waitForMerges, which we must do to make sure all in-flight (throttled) merges finish:
219         logger.info("test: optimize");
220         client().admin().indices().prepareOptimize("test").get();
221         logger.info("test: optimize done");
222 
223         // Record current throttling so far
224         long sumThrottleTime = 0;
225         nodesStats = client().admin().cluster().prepareNodesStats().setIndices(true).get();
226         for(NodeStats stats : nodesStats.getNodes()) {
227             sumThrottleTime += stats.getIndices().getStore().getThrottleTime().getMillis();
228         }
229 
230         // Make sure no further throttling happens:
231         for(int i=0;i<100;i++) {
232             // Provoke slowish merging by making many unique terms:
233             StringBuilder sb = new StringBuilder();
234             for(int j=0;j<100;j++) {
235                 sb.append(' ');
236                 sb.append(termUpto++);
237             }
238             client().prepareIndex("test", "type", ""+termUpto).setSource("field" + (i%10), sb.toString()).get();
239             if (i % 2 == 0) {
240                 refresh();
241             }
242         }
243         logger.info("test: done indexing after disabling throttling");
244 
245         long newSumThrottleTime = 0;
246         nodesStats = client().admin().cluster().prepareNodesStats().setIndices(true).get();
247         for(NodeStats stats : nodesStats.getNodes()) {
248             newSumThrottleTime += stats.getIndices().getStore().getThrottleTime().getMillis();
249         }
250 
251         // No additional merge IO throttling should have happened:
252         assertEquals(sumThrottleTime, newSumThrottleTime);
253 
254         // Optimize & flush and wait; else we sometimes get a "Delete Index failed - not acked"
255         // when ElasticsearchIntegrationTest.after tries to remove indices created by the test:
256 
257         // Wait for merges to finish
258         client().admin().indices().prepareOptimize("test").get();
259         flush();
260 
261         logger.info("test: test done");
262     }

264     private static class MockAppender extends AppenderSkeleton {
265         public boolean sawIndexWriterMessage;
266         public boolean sawFlushDeletes;
267         public boolean sawMergeThreadPaused;
268         public boolean sawUpdateMaxThreadCount;
269         public boolean sawUpdateAutoThrottle;
270 
271         @Override
272         protected void append(LoggingEvent event) {
273             String message = event.getMessage().toString();
274             if (event.getLevel() == Level.TRACE &&
275                 event.getLoggerName().endsWith("lucene.iw")) {
276                 sawFlushDeletes |= message.contains("IW: apply all deletes during flush");
277                 sawMergeThreadPaused |= message.contains("CMS: pause thread");
278             }
279             if (event.getLevel() == Level.INFO && message.contains("updating [index.merge.scheduler.max_thread_count] from [10000] to [1]")) {
280                 sawUpdateMaxThreadCount = true;
281             }
282             if (event.getLevel() == Level.INFO && message.contains("updating [index.merge.scheduler.auto_throttle] from [true] to [false]")) {
283                 sawUpdateAutoThrottle = true;
284             }
285         }
286 
287         @Override
288         public boolean requiresLayout() {
289             return false;
290         }
291 
292         @Override
293         public void close() {
294         }
295     }

297     @Test
298     public void testUpdateAutoThrottleSettings() {
299 
300         MockAppender mockAppender = new MockAppender();
301         Logger rootLogger = Logger.getRootLogger();
302         Level savedLevel = rootLogger.getLevel();
303         rootLogger.addAppender(mockAppender);
304         rootLogger.setLevel(Level.TRACE);
305 
306         try {
307             // No throttling at first, only 1 non-replicated shard, force lots of merging:
308             assertAcked(prepareCreate("test")
309                         .setSettings(ImmutableSettings.builder()
310                                      .put(IndexMetaData.SETTING_NUMBER_OF_SHARDS, "1")
311                                      .put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, "0")
312                                      .put(TieredMergePolicyProvider.INDEX_MERGE_POLICY_MAX_MERGE_AT_ONCE, "2")
313                                      .put(TieredMergePolicyProvider.INDEX_MERGE_POLICY_SEGMENTS_PER_TIER, "2")
314                                      .put(ConcurrentMergeSchedulerProvider.MAX_THREAD_COUNT, "1")
315                                      .put(ConcurrentMergeSchedulerProvider.MAX_MERGE_COUNT, "2")
316                                      .put(ConcurrentMergeSchedulerProvider.AUTO_THROTTLE, "true")
317                                      ));
318 
319             // Disable auto throttle:
320             client()
321                 .admin()
322                 .indices()
323                 .prepareUpdateSettings("test")
324                 .setSettings(ImmutableSettings.builder()
325                              .put(ConcurrentMergeSchedulerProvider.AUTO_THROTTLE, "no"))
326                 .get();
327 
328             // Make sure we log the change:
329             assertTrue(mockAppender.sawUpdateAutoThrottle);
330 
331             // Make sure setting says it is in fact changed:
332             GetSettingsResponse getSettingsResponse = client().admin().indices().prepareGetSettings("test").get();
333             assertThat(getSettingsResponse.getSetting("test", ConcurrentMergeSchedulerProvider.AUTO_THROTTLE), equalTo("no"));
334         } finally {
335             rootLogger.removeAppender(mockAppender);
336             rootLogger.setLevel(savedLevel);
337         }
338     }

371                              .put(ConcurrentMergeSchedulerProvider.MAX_THREAD_COUNT, "1")

373                 .get();

375             // Make sure we log the change:
376             assertTrue(mockAppender.sawUpdateMaxThreadCount);
377 
378             // Make sure setting says it is in fact changed:
379             GetSettingsResponse getSettingsResponse = client().admin().indices().prepareGetSettings("test").get();
380             assertThat(getSettingsResponse.getSetting("test", ConcurrentMergeSchedulerProvider.MAX_THREAD_COUNT), equalTo("1"));
381 
382         } finally {
383             rootLogger.removeAppender(mockAppender);
384             rootLogger.setLevel(savedLevel);
385         }
