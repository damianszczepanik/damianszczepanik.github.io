75      public enum Method {
76          AGGREGATION {
77              @Override
78              SearchRequestBuilder addTermsAgg(SearchRequestBuilder builder, String name, String field, String executionHint) {
79                  return builder.addAggregation(AggregationBuilders.terms(name).executionHint(executionHint).field(field));
80              }
81  
82              @Override
83              SearchRequestBuilder addTermsStatsAgg(SearchRequestBuilder builder, String name, String keyField, String valueField) {
84                  return builder.addAggregation(AggregationBuilders.terms(name).field(keyField).subAggregation(AggregationBuilders.stats("stats").field(valueField)));
85              }
86          },
87          AGGREGATION_DEFERRED {
88              @Override
89              SearchRequestBuilder addTermsAgg(SearchRequestBuilder builder, String name, String field, String executionHint) {
90                  return builder.addAggregation(AggregationBuilders.terms(name).executionHint(executionHint).field(field).collectMode(SubAggCollectionMode.BREADTH_FIRST));
91              }
92  
93              @Override
94              SearchRequestBuilder addTermsStatsAgg(SearchRequestBuilder builder, String name, String keyField, String valueField) {
95                  return builder.addAggregation(AggregationBuilders.terms(name).field(keyField).collectMode(SubAggCollectionMode.BREADTH_FIRST).subAggregation(AggregationBuilders.stats("stats").field(valueField)));
96              }
97          };
98          abstract SearchRequestBuilder addTermsAgg(SearchRequestBuilder builder, String name, String field, String executionHint);
99          abstract SearchRequestBuilder addTermsStatsAgg(SearchRequestBuilder builder, String name, String keyField, String valueField);
100     }

303     public static class StatsResult {
304         final String name;
305         final long took;
306         final ByteSizeValue fieldDataMemoryUsed;

315     private static StatsResult terms(String name, Method method, String field, String executionHint) {
316         long totalQueryTime;// LM VALUE
317 
318         client.admin().indices().prepareClearCache().setFieldDataCache(true).execute().actionGet();
319         System.gc();
320 
321         System.out.println("--> Warmup (" + name + ")...");
322         // run just the child query, warm up first
323         for (int j = 0; j < QUERY_WARMUP; j++) {
324             SearchResponse searchResponse = method.addTermsAgg(client.prepareSearch("test")
325                     .setSearchType(SearchType.COUNT)
326                     .setQuery(matchAllQuery()), name, field, executionHint)
327                     .execute().actionGet();
328             if (j == 0) {
329                 System.out.println("--> Loading (" + field + "): took: " + searchResponse.getTook());
330             }
331             if (searchResponse.getHits().totalHits() != COUNT) {
332                 System.err.println("--> mismatch on hits");
333             }
334         }
335         System.out.println("--> Warmup (" + name + ") DONE");
336 
337 
338         System.out.println("--> Running (" + name + ")...");
339         totalQueryTime = 0;
340         for (int j = 0; j < QUERY_COUNT; j++) {
341             SearchResponse searchResponse = method.addTermsAgg(client.prepareSearch()
342                     .setSearchType(SearchType.COUNT)
343                     .setQuery(matchAllQuery()), name, field, executionHint)
344                     .execute().actionGet();
345             if (searchResponse.getHits().totalHits() != COUNT) {
346                 System.err.println("--> mismatch on hits");
347             }
348             totalQueryTime += searchResponse.getTookInMillis();
349         }
350         System.out.println("--> Terms Agg (" + name + "): " + (totalQueryTime / QUERY_COUNT) + "ms");
351 
352         String[] nodeIds = new String[nodes.length];
353         for (int i = 0; i < nodeIds.length; i++) {
354             nodeIds[i] = nodes[i].injector().getInstance(Discovery.class).localNode().getId();
355         }
356 
357         ClusterStatsResponse clusterStateResponse = client.admin().cluster().prepareClusterStats().setNodesIds(nodeIds).get();
358         System.out.println("--> Heap used: " + clusterStateResponse.getNodesStats().getJvm().getHeapUsed());
359         ByteSizeValue fieldDataMemoryUsed = clusterStateResponse.getIndicesStats().getFieldData().getMemorySize();
360         System.out.println("--> Fielddata memory size: " + fieldDataMemoryUsed);
361 
362         return new StatsResult(name, totalQueryTime, fieldDataMemoryUsed);
363     }

365     private static StatsResult termsStats(String name, Method method, String keyField, String valueField, String executionHint) {
366         long totalQueryTime;
367 
368         client.admin().indices().prepareClearCache().setFieldDataCache(true).execute().actionGet();
369         System.gc();
370 
371         System.out.println("--> Warmup (" + name + ")...");
372         // run just the child query, warm up first
373         for (int j = 0; j < QUERY_WARMUP; j++) {
374             SearchResponse searchResponse = method.addTermsStatsAgg(client.prepareSearch()
375                     .setSearchType(SearchType.COUNT)
376                     .setQuery(matchAllQuery()), name, keyField, valueField)
377                     .execute().actionGet();
378             if (j == 0) {
379                 System.out.println("--> Loading (" + name + "): took: " + searchResponse.getTook());
380             }
381             if (searchResponse.getHits().totalHits() != COUNT) {
382                 System.err.println("--> mismatch on hits");
383             }
384         }
385         System.out.println("--> Warmup (" + name + ") DONE");
386 
387 
388         System.out.println("--> Running (" + name + ")...");
389         totalQueryTime = 0;
390         for (int j = 0; j < QUERY_COUNT; j++) {
391             SearchResponse searchResponse = method.addTermsStatsAgg(client.prepareSearch()
392                     .setSearchType(SearchType.COUNT)
393                     .setQuery(matchAllQuery()), name, keyField, valueField)
394                     .execute().actionGet();
395             if (searchResponse.getHits().totalHits() != COUNT) {
396                 System.err.println("--> mismatch on hits");
397             }
398             totalQueryTime += searchResponse.getTookInMillis();
399         }
400         System.out.println("--> Terms stats agg (" + name + "): " + (totalQueryTime / QUERY_COUNT) + "ms");
401         return new StatsResult(name, totalQueryTime, ByteSizeValue.parseBytesSizeValue("0b"));
402     }
