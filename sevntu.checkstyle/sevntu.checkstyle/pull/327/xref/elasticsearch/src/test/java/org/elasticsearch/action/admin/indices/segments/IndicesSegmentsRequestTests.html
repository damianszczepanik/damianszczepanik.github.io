20  package org.elasticsearch.action.admin.indices.segments;
21  
22  import org.elasticsearch.action.ListenableActionFuture;
23  import org.elasticsearch.action.support.IndicesOptions;
24  import org.elasticsearch.common.settings.ImmutableSettings;
25  import org.elasticsearch.common.settings.Settings;
26  import org.elasticsearch.index.engine.Segment;
27  import org.elasticsearch.indices.IndexClosedException;
28  import org.elasticsearch.test.ElasticsearchSingleNodeTest;
29  import org.junit.Before;
30  import org.junit.Test;
31  
32  import java.util.List;
33  
34  public class IndicesSegmentsRequestTests extends ElasticsearchSingleNodeTest {
35      
36      @Before
37      public void setupIndex() {
38          Settings settings = ImmutableSettings.builder()
39              // don't allow any merges so that the num docs is the expected segments
40              .put("index.merge.policy.segments_per_tier", 1000000f)
41              .build();
42          createIndex("test", settings);
43  
44          int numDocs = scaledRandomIntBetween(100, 1000);
45          for (int j = 0; j < numDocs; ++j) {
46              String id = Integer.toString(j);
47              client().prepareIndex("test", "type1", id).setSource("text", "sometext").get();
48          }
49          client().admin().indices().prepareFlush("test").get();
50      }
51  
52      public void testBasic() {
53          IndicesSegmentResponse rsp = client().admin().indices().prepareSegments("test").get();
54          List<Segment> segments = rsp.getIndices().get("test").iterator().next().getShards()[0].getSegments();
55          assertNull(segments.get(0).ramTree);
56      }
57      
58      public void testVerbose() {
59          IndicesSegmentResponse rsp = client().admin().indices().prepareSegments("test").setVerbose(true).get();
60          List<Segment> segments = rsp.getIndices().get("test").iterator().next().getShards()[0].getSegments();
61          assertNotNull(segments.get(0).ramTree);
62      }
63  
64      /**
65       * with the default IndicesOptions inherited from BroadcastOperationRequest this will raise an exception
66       */
67      @Test(expected=org.elasticsearch.indices.IndexClosedException.class)
68      public void testRequestOnClosedIndex() {
69          client().admin().indices().prepareClose("test").get();
70          client().admin().indices().prepareSegments("test").get();
71      }
72  
73      /**
74       * setting the "ignoreUnavailable" option prevents IndexClosedException
75       */
76      public void testRequestOnClosedIndexIgnoreUnavailable() {
77          client().admin().indices().prepareClose("test").get();
78          IndicesOptions defaultOptions = new IndicesSegmentsRequest().indicesOptions();
79          IndicesOptions testOptions = IndicesOptions.fromOptions(true, true, true, false, defaultOptions);
80          IndicesSegmentResponse rsp = client().admin().indices().prepareSegments("test").setIndicesOptions(testOptions).get();
81          assertEquals(0, rsp.getIndices().size());
82      }
83  
84      /**
85       * by default IndicesOptions setting IndicesSegmentsRequest should not throw exception when no index present
86       */
87      public void testAllowNoIndex() {
88          client().admin().indices().prepareDelete("test").get();
89          IndicesSegmentResponse rsp = client().admin().indices().prepareSegments().get();
90          assertEquals(0, rsp.getIndices().size());
91      }
92  }
