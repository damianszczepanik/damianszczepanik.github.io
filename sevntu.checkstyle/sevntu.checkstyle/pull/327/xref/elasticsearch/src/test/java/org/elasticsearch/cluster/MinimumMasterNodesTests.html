54  
55          Settings settings = settingsBuilder()
56                  .put("discovery.type", "zen")
57                  .put("discovery.zen.minimum_master_nodes", 2)
58                  .put("discovery.zen.ping_timeout", "200ms")
59                  .put("discovery.initial_state_timeout", "500ms")
60                  .build();
61  
62          logger.info("--> start first node");
63          internalCluster().startNode(settings);
64  
65          logger.info("--> should be blocked, no master...");

67          assertThat(state.blocks().hasGlobalBlock(DiscoverySettings.NO_MASTER_BLOCK_ID), equalTo(true));

70          logger.info("--> start second node, cluster should be formed");
71          internalCluster().startNode(settings);

172     @Test
173     public void multipleNodesShutdownNonMasterNodes() throws Exception {
174         Settings settings = settingsBuilder()
175                 .put("discovery.type", "zen")
176                 .put("discovery.zen.minimum_master_nodes", 3)
177                 .put("discovery.zen.ping_timeout", "1s")
178                 .put("discovery.initial_state_timeout", "500ms")
179                 .build();
180 
181         logger.info("--> start first 2 nodes");
182         internalCluster().startNode(settings);
183         internalCluster().startNode(settings);
184 
185         ClusterState state;
186 
187         awaitBusy(new Predicate<Object>() {
188             @Override
189             public boolean apply(Object obj) {
190                 ClusterState state = client().admin().cluster().prepareState().setLocal(true).execute().actionGet().getState();
191                 return state.blocks().hasGlobalBlock(DiscoverySettings.NO_MASTER_BLOCK_ID);
192             }
193         });
194 
195         awaitBusy(new Predicate<Object>() {
196             @Override
197             public boolean apply(Object obj) {
198                 ClusterState state = client().admin().cluster().prepareState().setLocal(true).execute().actionGet().getState();
199                 return state.blocks().hasGlobalBlock(DiscoverySettings.NO_MASTER_BLOCK_ID);
200             }
201         });
202 
203         state = client().admin().cluster().prepareState().setLocal(true).execute().actionGet().getState();
204         assertThat(state.blocks().hasGlobalBlock(DiscoverySettings.NO_MASTER_BLOCK_ID), equalTo(true));
205         state = client().admin().cluster().prepareState().setLocal(true).execute().actionGet().getState();
206         assertThat(state.blocks().hasGlobalBlock(DiscoverySettings.NO_MASTER_BLOCK_ID), equalTo(true));
207 
208         logger.info("--> start two more nodes");
209         internalCluster().startNodesAsync(2, settings).get();
210 
211         ClusterHealthResponse clusterHealthResponse = client().admin().cluster().prepareHealth().setWaitForEvents(Priority.LANGUID).setWaitForNodes("4").execute().actionGet();
212         assertThat(clusterHealthResponse.isTimedOut(), equalTo(false));
213 
214         state = client().admin().cluster().prepareState().execute().actionGet().getState();
215         assertThat(state.nodes().size(), equalTo(4));
216 
217         createIndex("test");
218         NumShards numShards = getNumShards("test");
219         logger.info("--> indexing some data");
220         for (int i = 0; i < 100; i++) {
221             client().prepareIndex("test", "type1", Integer.toString(i)).setSource("field", "value").execute().actionGet();
222         }
223         // make sure that all shards recovered before trying to flush
224         assertThat(client().admin().cluster().prepareHealth("test").setWaitForActiveShards(numShards.totalNumShards).execute().actionGet().isTimedOut(), equalTo(false));
225         // flush for simpler debugging
226         client().admin().indices().prepareFlush().execute().actionGet();
227 
228         client().admin().indices().prepareRefresh().execute().actionGet();
229         logger.info("--> verify we the data back");
230         for (int i = 0; i < 10; i++) {
231             assertThat(client().prepareCount().setQuery(QueryBuilders.matchAllQuery()).execute().actionGet().getCount(), equalTo(100l));
232         }
233 
234         internalCluster().stopRandomNonMasterNode();
235         internalCluster().stopRandomNonMasterNode();
236 
237         logger.info("--> verify that there is no master anymore on remaining nodes");
238         // spin here to wait till the state is set
239         assertNoMasterBlockOnAllNodes();
240 
241         logger.info("--> start back the 2 nodes ");
242         String[] newNodes = internalCluster().startNodesAsync(2, settings).get().toArray(Strings.EMPTY_ARRAY);
243 
244         clusterHealthResponse = client().admin().cluster().prepareHealth().setWaitForNodes("4").execute().actionGet();
245         assertThat(clusterHealthResponse.isTimedOut(), equalTo(false));
246 
247         logger.info("--> running Cluster Health");
248         ClusterHealthResponse clusterHealth = client().admin().cluster().health(clusterHealthRequest().waitForGreenStatus()).actionGet();
249         logger.info("--> done Cluster Health, status " + clusterHealth.getStatus());
250         assertThat(clusterHealth.isTimedOut(), equalTo(false));
251         assertThat(clusterHealth.getStatus(), equalTo(ClusterHealthStatus.GREEN));
252 
253         state = client().admin().cluster().prepareState().execute().actionGet().getState();
254         assertThat(state.nodes().size(), equalTo(4));
255         // we prefer to elect up and running nodes
256         assertThat(state.nodes().masterNodeId(), not(isOneOf(newNodes)));
257 
258         logger.info("--> verify we the data back");
259         for (int i = 0; i < 10; i++) {
260             assertThat(client().prepareCount().setQuery(QueryBuilders.matchAllQuery()).execute().actionGet().getCount(), equalTo(100l));
261         }
262     }

264     @Test
265     public void dynamicUpdateMinimumMasterNodes() throws InterruptedException, IOException {
266         Settings settings = settingsBuilder()
267                 .put("discovery.type", "zen")
268                 .put("discovery.zen.ping_timeout", "400ms")
269                 .put("discovery.initial_state_timeout", "500ms")
270                 .build();
271 
272         logger.info("--> start 2 nodes");
273         internalCluster().startNode(settings);
274         internalCluster().startNode(settings);
275 
276         // wait until second node join the cluster
277         ClusterHealthResponse clusterHealthResponse = client().admin().cluster().prepareHealth().setWaitForEvents(Priority.LANGUID).setWaitForNodes("2").get();
278         assertThat(clusterHealthResponse.isTimedOut(), equalTo(false));
279 
280         logger.info("--> setting minimum master node to 2");
281         setMinimumMasterNodes(2);
282 
283         // make sure it has been processed on all nodes (master node spawns a secondary cluster state update task)
284         for (Client client : internalCluster()) {
285             assertThat(client.admin().cluster().prepareHealth().setWaitForEvents(Priority.LANGUID).setLocal(true).get().isTimedOut(),
286                     equalTo(false));
287         }
288 
289         logger.info("--> stopping a node");
290         internalCluster().stopRandomDataNode();
291         logger.info("--> verifying min master node has effect");
292         assertNoMasterBlockOnAllNodes();
293 
294         logger.info("--> bringing another node up");
295         internalCluster().startNode(settingsBuilder().put(settings).put(ElectMasterService.DISCOVERY_ZEN_MINIMUM_MASTER_NODES, 2).build());
296         clusterHealthResponse = client().admin().cluster().prepareHealth().setWaitForEvents(Priority.LANGUID).setWaitForNodes("2").get();
297         assertThat(clusterHealthResponse.isTimedOut(), equalTo(false));
298     }

300     private void assertNoMasterBlockOnAllNodes() throws InterruptedException {
301         assertThat(awaitBusy(new Predicate<Object>() {
302             @Override
303             public boolean apply(Object obj) {
304                 boolean success = true;
305                 for (Client client : internalCluster()) {
306                     ClusterState state = client.admin().cluster().prepareState().setLocal(true).execute().actionGet().getState();
307                     success &= state.blocks().hasGlobalBlock(DiscoverySettings.NO_MASTER_BLOCK_ID);
308                     if (logger.isDebugEnabled()) {
309                         logger.debug("Checking for NO_MASTER_BLOCK on client: {} NO_MASTER_BLOCK: [{}]", client, state.blocks().hasGlobalBlock(DiscoverySettings.NO_MASTER_BLOCK_ID));
310                     }
311                 }
312                 return success;
313             }
314         }, 20, TimeUnit.SECONDS), equalTo(true));
315     }

317     @Test
318     public void testCanNotBringClusterDown() throws ExecutionException, InterruptedException {
319         int nodeCount = scaledRandomIntBetween(1, 5);
320         ImmutableSettings.Builder settings = settingsBuilder()
321                 .put("discovery.type", "zen")
322                 .put("discovery.zen.ping_timeout", "200ms")
323                 .put("discovery.initial_state_timeout", "500ms");
324 
325         // set an initial value which is at least quorum to avoid split brains during initial startup
326         int initialMinMasterNodes = randomIntBetween(nodeCount / 2 + 1, nodeCount);
327         settings.put(ElectMasterService.DISCOVERY_ZEN_MINIMUM_MASTER_NODES, initialMinMasterNodes);
328 
329 
330         logger.info("--> starting [{}] nodes. min_master_nodes set to [{}]", nodeCount, initialMinMasterNodes);
331         internalCluster().startNodesAsync(nodeCount, settings.build()).get();
332 
333         logger.info("--> waiting for nodes to join");
334         assertFalse(client().admin().cluster().prepareHealth().setWaitForNodes(Integer.toString(nodeCount)).get().isTimedOut());
335 
336         int updateCount = randomIntBetween(1, nodeCount);
337 
338         logger.info("--> updating [{}] to [{}]", ElectMasterService.DISCOVERY_ZEN_MINIMUM_MASTER_NODES, updateCount);
339         assertAcked(client().admin().cluster().prepareUpdateSettings()
340                 .setPersistentSettings(settingsBuilder().put(ElectMasterService.DISCOVERY_ZEN_MINIMUM_MASTER_NODES, updateCount)));
341 
342         logger.info("--> verifying no node left and master is up");
343         assertFalse(client().admin().cluster().prepareHealth().setWaitForNodes(Integer.toString(nodeCount)).get().isTimedOut());
344 
345         updateCount = nodeCount + randomIntBetween(1, 2000);
346         logger.info("--> trying to updating [{}] to [{}]", ElectMasterService.DISCOVERY_ZEN_MINIMUM_MASTER_NODES, updateCount);
347         assertThat(client().admin().cluster().prepareUpdateSettings()
348                         .setPersistentSettings(settingsBuilder().put(ElectMasterService.DISCOVERY_ZEN_MINIMUM_MASTER_NODES, updateCount))
349                         .get().getPersistentSettings().getAsMap().keySet(),
350                 empty());
351 
352         logger.info("--> verifying no node left and master is up");
353         assertFalse(client().admin().cluster().prepareHealth().setWaitForNodes(Integer.toString(nodeCount)).get().isTimedOut());
354     }
