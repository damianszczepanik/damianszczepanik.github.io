85      public static class BlockingActionFilter extends org.elasticsearch.action.support.ActionFilter.Simple {
86  
87          ImmutableSet<String> blockedActions = ImmutableSet.of();
88  
89          @Inject
90          public BlockingActionFilter(Settings settings) {
91              super(settings);
92          }
93  
94          @Override
95          protected boolean apply(String action, ActionRequest request, ActionListener listener) {
96              if (blockedActions.contains(action)) {
97                  throw new ElasticsearchException("force exception on [" + action + "]");
98              }
99              return true;
100         }
101 
102         @Override
103         protected boolean apply(String action, ActionResponse response, ActionListener listener) {
104             return true;
105         }
106 
107         @Override
108         public int order() {
109             return 0;
110         }
111 
112         public void blockActions(String... actions) {
113             blockedActions = ImmutableSet.copyOf(actions);
114         }
115     }

117     static class InfoListener implements ClusterInfoService.Listener {
118         final AtomicReference<CountDownLatch> collected = new AtomicReference<>(new CountDownLatch(1));
119         volatile ClusterInfo lastInfo = null;
120 
121         @Override
122         public void onNewInfo(ClusterInfo info) {
123             lastInfo = info;
124             CountDownLatch latch = collected.get();
125             latch.countDown();
126         }
127 
128         public void reset() {
129             lastInfo = null;
130             collected.set(new CountDownLatch(1));
131         }
132 
133         public ClusterInfo get() throws InterruptedException {
134             CountDownLatch latch = collected.get();
135             if (!latch.await(10, TimeUnit.SECONDS)) {
136                 fail("failed to get a new cluster info");
137             }
138             return lastInfo;
139         }
140     }

152     @Test
153     public void testClusterInfoServiceCollectsInformation() throws Exception {
154         internalCluster().startNodesAsync(2,
155                 ImmutableSettings.builder().put(InternalClusterInfoService.INTERNAL_CLUSTER_INFO_UPDATE_INTERVAL, "200ms").build())
156                 .get();
157         assertAcked(prepareCreate("test").setSettings(settingsBuilder().put(Store.INDEX_STORE_STATS_REFRESH_INTERVAL, 0).build()));
158         ensureGreen("test");
159         InternalTestCluster internalTestCluster = internalCluster();
160         // Get the cluster info service on the master node
161         final InternalClusterInfoService infoService = (InternalClusterInfoService) internalTestCluster.getInstance(ClusterInfoService.class, internalTestCluster.getMasterName());
162         InfoListener listener = new InfoListener();
163         infoService.addListener(listener);
164         ClusterInfo info = listener.get();
165         assertNotNull("info should not be null", info);
166         Map<String, DiskUsage> usages = info.getNodeDiskUsages();
167         Map<String, Long> shardSizes = info.getShardSizes();
168         assertNotNull(usages);
169         assertNotNull(shardSizes);
170         assertThat("some usages are populated", usages.values().size(), Matchers.equalTo(2));
171         assertThat("some shard sizes are populated", shardSizes.values().size(), greaterThan(0));
172         for (DiskUsage usage : usages.values()) {
173             logger.info("--> usage: {}", usage);
174             assertThat("usage has be retrieved", usage.getFreeBytes(), greaterThan(0L));
175         }
176         for (Long size : shardSizes.values()) {
177             logger.info("--> shard size: {}", size);
178             assertThat("shard size is greater than 0", size, greaterThan(0L));
179         }
180     }

182     @Test
183     public void testClusterInfoServiceInformationClearOnError() throws InterruptedException, ExecutionException {
184         internalCluster().startNodesAsync(2,
185                 // manually control publishing
186                 ImmutableSettings.builder().put(InternalClusterInfoService.INTERNAL_CLUSTER_INFO_UPDATE_INTERVAL, "60m").build())
187                 .get();
188         prepareCreate("test").setSettings(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, 1).get();
189         ensureGreen("test");
190         InternalTestCluster internalTestCluster = internalCluster();
191         InternalClusterInfoService infoService = (InternalClusterInfoService) internalTestCluster.getInstance(ClusterInfoService.class, internalTestCluster.getMasterName());
192         InfoListener listener = new InfoListener();
193         infoService.addListener(listener);
194 
195         // get one healthy sample
196         infoService.updateOnce();
197         ClusterInfo info = listener.get();
198         assertNotNull("failed to collect info", info);
199         assertThat("some usages are populated", info.getNodeDiskUsages().size(), Matchers.equalTo(2));
200         assertThat("some shard sizes are populated", info.getShardSizes().size(), greaterThan(0));
201 
202 
203         MockTransportService mockTransportService = (MockTransportService) internalCluster().getInstance(TransportService.class, internalTestCluster.getMasterName());
204 
205         final AtomicBoolean timeout = new AtomicBoolean(false);
206         final Set<String> blockedActions = ImmutableSet.of(NodesStatsAction.NAME, NodesStatsAction.NAME + "[n]", IndicesStatsAction.NAME, IndicesStatsAction.NAME + "[s]");
207         // drop all outgoing stats requests to force a timeout.
208         for (DiscoveryNode node : internalTestCluster.clusterService().state().getNodes()) {
209             mockTransportService.addDelegate(node, new MockTransportService.DelegateTransport(mockTransportService.original()) {
210                 @Override
211                 public void sendRequest(DiscoveryNode node, long requestId, String action, TransportRequest request,
212                                         TransportRequestOptions options) throws IOException, TransportException {
213                     if (blockedActions.contains(action)) {
214                         if (timeout.get()) {
215                             logger.info("dropping [{}] to [{}]", action, node);
216                             return;
217                         }
218                     }
219                     super.sendRequest(node, requestId, action, request, options);
220                 }
221             });
222         }
223 
224         // timeouts shouldn't clear the info
225         timeout.set(true);
226         listener.reset();
227         infoService.updateOnce();
228         info = listener.get();
229         assertNotNull("info should not be null", info);
230         // node info will time out both on the request level on the count down latch. this means
231         // it is likely to update the node disk usage based on the one response that came be from local
232         // node.
233         assertThat(info.getNodeDiskUsages().size(), greaterThanOrEqualTo(1));
234         // indices is guaranteed to time out on the latch, not updating anything.
235         assertThat(info.getShardSizes().size(), greaterThan(1));
236 
237         // now we cause an exception
238         timeout.set(false);
239         ActionFilters actionFilters = internalTestCluster.getInstance(ActionFilters.class, internalTestCluster.getMasterName());
240         BlockingActionFilter blockingActionFilter = null;
241         for (ActionFilter filter : actionFilters.filters()) {
242             if (filter instanceof BlockingActionFilter) {
243                 blockingActionFilter = (BlockingActionFilter) filter;
244                 break;
245             }
246         }
247 
248         assertNotNull("failed to find BlockingActionFilter", blockingActionFilter);
249         blockingActionFilter.blockActions(blockedActions.toArray(Strings.EMPTY_ARRAY));
250         listener.reset();
251         infoService.updateOnce();
252         info = listener.get();
253         assertNotNull("info should not be null", info);
254         assertThat(info.getNodeDiskUsages().size(), equalTo(0));
255         assertThat(info.getShardSizes().size(), equalTo(0));
256 
257         // check we recover
258         blockingActionFilter.blockActions();
259         listener.reset();
260         infoService.updateOnce();
261         info = listener.get();
262         assertNotNull("info should not be null", info);
263         assertThat(info.getNodeDiskUsages().size(), equalTo(2));
264         assertThat(info.getShardSizes().size(), greaterThan(0));
265 
266     }
