20  package org.elasticsearch.benchmark.search.aggregations;
21  
22  import org.elasticsearch.action.admin.cluster.health.ClusterHealthResponse;
23  import org.elasticsearch.action.bulk.BulkRequestBuilder;
24  import org.elasticsearch.action.bulk.BulkResponse;
25  import org.elasticsearch.action.search.SearchResponse;
26  import org.elasticsearch.client.Client;
27  import org.elasticsearch.client.Requests;
28  import org.elasticsearch.common.StopWatch;
29  import org.elasticsearch.common.settings.Settings;
30  import org.elasticsearch.common.unit.SizeValue;
31  import org.elasticsearch.common.xcontent.XContentBuilder;
32  import org.elasticsearch.node.Node;
33  
34  import java.util.Date;
35  import java.util.Random;
36  
37  import static org.elasticsearch.cluster.metadata.IndexMetaData.SETTING_NUMBER_OF_REPLICAS;
38  import static org.elasticsearch.cluster.metadata.IndexMetaData.SETTING_NUMBER_OF_SHARDS;
39  import static org.elasticsearch.common.settings.ImmutableSettings.settingsBuilder;
40  import static org.elasticsearch.common.xcontent.XContentFactory.jsonBuilder;
41  import static org.elasticsearch.index.query.QueryBuilders.matchAllQuery;
42  import static org.elasticsearch.node.NodeBuilder.nodeBuilder;
43  import static org.elasticsearch.search.aggregations.AggregationBuilders.*;
44  
45  /**
46   *
47   */
48  public class HistogramAggregationSearchBenchmark {
49  
50      static final long COUNT = SizeValue.parseSizeValue("20m").singles();
51      static final int BATCH = 1000;
52      static final int QUERY_WARMUP = 5;
53      static final int QUERY_COUNT = 20;
54      static final int NUMBER_OF_TERMS = 1000;
55  
56      public static void main(String[] args) throws Exception {
57          Settings settings = settingsBuilder()
58                  .put("refresh_interval", "-1")
59                  .put(SETTING_NUMBER_OF_SHARDS, 1)
60                  .put(SETTING_NUMBER_OF_REPLICAS, 0)
61                  .build();
62  
63          String clusterName = HistogramAggregationSearchBenchmark.class.getSimpleName();
64          Node node1 = nodeBuilder()
65                  .clusterName(clusterName)
66                  .settings(settingsBuilder().put(settings).put("name", "node1")).node();
67  
68          //Node clientNode = nodeBuilder().clusterName(clusterName).settings(settingsBuilder().put(settings).put("name", "client")).client(true).node();
69  
70          Client client = node1.client();
71  
72          long[] lValues = new long[NUMBER_OF_TERMS];
73          for (int i = 0; i < NUMBER_OF_TERMS; i++) {
74              lValues[i] = i;
75          }
76  
77          Random r = new Random();
78          try {
79              client.admin().indices().prepareCreate("test")
80                      .setSettings(settingsBuilder().put(settings))
81                      .addMapping("type1", jsonBuilder()
82                          .startObject()
83                              .startObject("type1")
84                                  .startObject("properties")
85                                      .startObject("l_value")
86                                          .field("type", "long")
87                                      .endObject()
88                                      .startObject("i_value")
89                                          .field("type", "integer")
90                                      .endObject()
91                                      .startObject("s_value")
92                                          .field("type", "short")
93                                      .endObject()
94                                      .startObject("b_value")
95                                          .field("type", "byte")
96                                      .endObject()
97                                  .endObject()
98                              .endObject()
99                          .endObject())
100                     .execute().actionGet();
101 
102             StopWatch stopWatch = new StopWatch().start();
103 
104             System.out.println("--> Indexing [" + COUNT + "] ...");
105             long iters = COUNT / BATCH;
106             long i = 1;
107             int counter = 0;
108             for (; i <= iters; i++) {
109                 BulkRequestBuilder request = client.prepareBulk();
110                 for (int j = 0; j < BATCH; j++) {
111                     counter++;
112                     final long value = lValues[r.nextInt(lValues.length)];
113                     XContentBuilder source = jsonBuilder().startObject()
114                             .field("id", Integer.valueOf(counter))
115                             .field("l_value", value)
116                             .field("i_value", (int) value)
117                             .field("s_value", (short) value)
118                             .field("b_value", (byte) value)
119                             .field("date", new Date())
120                             .endObject();
121                     request.add(Requests.indexRequest("test").type("type1").id(Integer.toString(counter))
122                             .source(source));
123                 }
124                 BulkResponse response = request.execute().actionGet();
125                 if (response.hasFailures()) {
126                     System.err.println("--> failures...");
127                 }
128                 if (((i * BATCH) % 10000) == 0) {
129                     System.out.println("--> Indexed " + (i * BATCH) + " took " + stopWatch.stop().lastTaskTime());
130                     stopWatch.start();
131                 }
132             }
133             client.admin().indices().prepareFlush("test").execute().actionGet();
134             System.out.println("--> Indexing took " + stopWatch.totalTime() + ", TPS " + (((double) (COUNT)) / stopWatch.totalTime().secondsFrac()));
135         } catch (Exception e) {
136             System.out.println("--> Index already exists, ignoring indexing phase, waiting for green");
137             ClusterHealthResponse clusterHealthResponse = client.admin().cluster().prepareHealth().setWaitForGreenStatus().setTimeout("10m").execute().actionGet();
138             if (clusterHealthResponse.isTimedOut()) {
139                 System.err.println("--> Timed out waiting for cluster health");
140             }
141         }
142         if (client.prepareCount().setQuery(matchAllQuery()).execute().actionGet().getCount() != COUNT) {
143             throw new Error();
144         }
145         System.out.println("--> Number of docs in index: " + COUNT);
146 
147         System.out.println("--> Warmup...");
148         // run just the child query, warm up first
149         for (int j = 0; j < QUERY_WARMUP; j++) {
150             SearchResponse searchResponse = client.prepareSearch()
151                     .setQuery(matchAllQuery())
152                     .addAggregation(histogram("l_value").field("l_value").interval(4))
153                     .addAggregation(histogram("i_value").field("i_value").interval(4))
154                     .addAggregation(histogram("s_value").field("s_value").interval(4))
155                     .addAggregation(histogram("b_value").field("b_value").interval(4))
156                     .addAggregation(histogram("date").field("date").interval(1000))
157                     .execute().actionGet();
158             if (j == 0) {
159                 System.out.println("--> Warmup took: " + searchResponse.getTook());
160             }
161             if (searchResponse.getHits().totalHits() != COUNT) {
162                 System.err.println("--> mismatch on hits");
163             }
164         }
165         System.out.println("--> Warmup DONE");
166 
167         long totalQueryTime = 0;
168         for (String field : new String[] {"b_value", "s_value", "i_value", "l_value"}) {
169             totalQueryTime = 0;
170             for (int j = 0; j < QUERY_COUNT; j++) {
171                 SearchResponse searchResponse = client.prepareSearch()
172                         .setQuery(matchAllQuery())
173                         .addAggregation(histogram(field).field(field).interval(4))
174                         .execute().actionGet();
175                 if (searchResponse.getHits().totalHits() != COUNT) {
176                     System.err.println("--> mismatch on hits");
177                 }
178                 totalQueryTime += searchResponse.getTookInMillis();
179             }
180             System.out.println("--> Histogram Aggregation (" + field + ") " + (totalQueryTime / QUERY_COUNT) + "ms");
181 
182             totalQueryTime = 0;
183             for (int j = 0; j < QUERY_COUNT; j++) {
184                 SearchResponse searchResponse = client.prepareSearch()
185                         .setQuery(matchAllQuery())
186                         .addAggregation(histogram(field).field(field).subAggregation(stats(field).field(field)).interval(4))
187                         .execute().actionGet();
188                 if (searchResponse.getHits().totalHits() != COUNT) {
189                     System.err.println("--> mismatch on hits");
190                 }
191                 totalQueryTime += searchResponse.getTookInMillis();
192             }
193             System.out.println("--> Histogram Aggregation (" + field + "/" + field + ") " + (totalQueryTime / QUERY_COUNT) + "ms");
194         }
195 
196         totalQueryTime = 0;
197         for (int j = 0; j < QUERY_COUNT; j++) {
198             SearchResponse searchResponse = client.prepareSearch()
199                     .setQuery(matchAllQuery())
200                     .addAggregation(dateHistogram("date").field("date").interval(1000))
201                     .execute().actionGet();
202             if (searchResponse.getHits().totalHits() != COUNT) {
203                 System.err.println("--> mismatch on hits");
204             }
205             totalQueryTime += searchResponse.getTookInMillis();
206         }
207         System.out.println("--> Histogram Aggregation (date) " + (totalQueryTime / QUERY_COUNT) + "ms");
208 
209         totalQueryTime = 0;
210         for (int j = 0; j < QUERY_COUNT; j++) {
211             SearchResponse searchResponse = client.prepareSearch()
212                     .setQuery(matchAllQuery())
213                     .addAggregation(dateHistogram("date").field("date").interval(1000).subAggregation(stats("stats").field("l_value")))
214                     .execute().actionGet();
215             if (searchResponse.getHits().totalHits() != COUNT) {
216                 System.err.println("--> mismatch on hits");
217             }
218             totalQueryTime += searchResponse.getTookInMillis();
219         }
220         System.out.println("--> Histogram Aggregation (date/l_value) " + (totalQueryTime / QUERY_COUNT) + "ms");
221 
222         node1.close();
223     }
224 }
