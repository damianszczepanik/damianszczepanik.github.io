227         assertThat(((NonSnapshottableGatewayMetadata) metaData.custom(NonSnapshottableGatewayMetadata.TYPE)).getData(), equalTo("after_snapshot_ns_gw"));

243         // But should still be in state
244         metaData = internalCluster().getInstance(ClusterService.class).state().metaData();

248     private void updateClusterState(final ClusterStateUpdater updater) throws InterruptedException {
249         final CountDownLatch countDownLatch = new CountDownLatch(1);
250         final ClusterService clusterService = internalCluster().getInstance(ClusterService.class);
251         clusterService.submitStateUpdateTask("test", new ProcessedClusterStateUpdateTask() {
252             @Override
253             public ClusterState execute(ClusterState currentState) throws Exception {
254                 return updater.execute(currentState);
255             }
256 
257             @Override
258             public void onFailure(String source, @Nullable Throwable t) {
259                 countDownLatch.countDown();
260             }
261 
262             @Override
263             public void clusterStateProcessed(String source, ClusterState oldState, ClusterState newState) {
264                 countDownLatch.countDown();
265             }
266         });
267         countDownLatch.await();
268     }

320     @Test
321     public void snapshotWithStuckNodeTest() throws Exception {
322         logger.info("--> start 2 nodes");
323         ArrayList<String> nodes = newArrayList();
324         nodes.add(internalCluster().startNode());
325         nodes.add(internalCluster().startNode());
326         Client client = client();
327 
328         assertAcked(prepareCreate("test-idx", 2, settingsBuilder().put("number_of_shards", 2).put("number_of_replicas", 0)));
329         ensureGreen();
330 
331         logger.info("--> indexing some data");
332         for (int i = 0; i < 100; i++) {
333             index("test-idx", "doc", Integer.toString(i), "foo", "bar" + i);
334         }
335         refresh();
336         assertThat(client.prepareCount("test-idx").get().getCount(), equalTo(100L));
337 
338         logger.info("--> creating repository");
339         Path repo = newTempDirPath(LifecycleScope.TEST);
340         PutRepositoryResponse putRepositoryResponse = client.admin().cluster().preparePutRepository("test-repo")
341                 .setType(MockRepositoryModule.class.getCanonicalName()).setSettings(
342                         ImmutableSettings.settingsBuilder()
343                                 .put("location", repo)
344                                 .put("random", randomAsciiOfLength(10))
345                                 .put("wait_after_unblock", 200)
346                 ).get();
347         assertThat(putRepositoryResponse.isAcknowledged(), equalTo(true));
348 
349         // Pick one node and block it
350         String blockedNode = blockNodeWithIndex("test-idx");
351         // Remove it from the list of available nodes
352         nodes.remove(blockedNode);
353 
354         int numberOfFilesBeforeSnapshot = numberOfFiles(repo);
355         logger.info("--> snapshot");
356         client.admin().cluster().prepareCreateSnapshot("test-repo", "test-snap").setWaitForCompletion(false).setIndices("test-idx").get();
357 
358         logger.info("--> waiting for block to kick in");
359         waitForBlock(blockedNode, "test-repo", TimeValue.timeValueSeconds(60));
360 
361         logger.info("--> execution was blocked on node [{}], aborting snapshot", blockedNode);
362 
363         ListenableActionFuture<DeleteSnapshotResponse> deleteSnapshotResponseFuture = internalCluster().client(nodes.get(0)).admin().cluster().prepareDeleteSnapshot("test-repo", "test-snap").execute();
364         // Make sure that abort makes some progress
365         Thread.sleep(100);
366         unblockNode(blockedNode);
367         logger.info("--> stopping node", blockedNode);
368         stopNode(blockedNode);
369         try {
370             DeleteSnapshotResponse deleteSnapshotResponse = deleteSnapshotResponseFuture.actionGet();
371             assertThat(deleteSnapshotResponse.isAcknowledged(), equalTo(true));
372         } catch (SnapshotMissingException ex) {
373             // When master node is closed during this test, it sometime manages to delete the snapshot files before
374             // completely stopping. In this case the retried delete snapshot operation on the new master can fail
375             // with SnapshotMissingException
376         }
377 
378         logger.info("--> making sure that snapshot no longer exists");
379         assertThrows(client().admin().cluster().prepareGetSnapshots("test-repo").setSnapshots("test-snap").execute(), SnapshotMissingException.class);
380         // Subtract index file from the count
381         assertThat("not all files were deleted during snapshot cancellation", numberOfFilesBeforeSnapshot, equalTo(numberOfFiles(repo) - 1));
382         logger.info("--> done");
383     }

433 
434         assertThat(createSnapshotResponse.getSnapshotInfo().state(), equalTo(SnapshotState.FAILED));
435 
436         if (randomBoolean()) {
437             logger.info("checking snapshot completion using status");

439             awaitBusy(new Predicate<Object>() {
440                 @Override
441                 public boolean apply(Object o) {
442                     SnapshotsStatusResponse snapshotsStatusResponse = client().admin().cluster().prepareSnapshotStatus("test-repo").setSnapshots("test-snap-2").get();
443                     ImmutableList<SnapshotStatus> snapshotStatuses = snapshotsStatusResponse.getSnapshots();
444                     if (snapshotStatuses.size() == 1) {

448                     return false;

452             ImmutableList<SnapshotStatus> snapshotStatuses = snapshotsStatusResponse.getSnapshots();
453             assertThat(snapshotStatuses.size(), equalTo(1));
454             SnapshotStatus snapshotStatus = snapshotStatuses.get(0);
455             logger.info("State: [{}], Reason: [{}]", createSnapshotResponse.getSnapshotInfo().state(), createSnapshotResponse.getSnapshotInfo().reason());

457             assertThat(snapshotStatus.getShardsStats().getDoneShards(), lessThan(12));
458             assertThat(snapshotStatus.getShardsStats().getDoneShards(), greaterThan(6));
459 
460             // There is slight delay between snapshot being marked as completed in the cluster state and on the file system
461             // After it was marked as completed in the cluster state - we need to check if it's completed on the file system as well
462             awaitBusy(new Predicate<Object>() {
463                 @Override
464                 public boolean apply(Object o) {
465                     GetSnapshotsResponse response = client().admin().cluster().prepareGetSnapshots("test-repo").setSnapshots("test-snap-2").get();
466                     assertThat(response.getSnapshots().size(), equalTo(1));
467                     SnapshotInfo snapshotInfo = response.getSnapshots().get(0);
468                     if (snapshotInfo.state().completed()) {
469                         assertThat(snapshotInfo.state(), equalTo(SnapshotState.PARTIAL));
470                         return true;
471                     }
472                     return false;
473                 }
474             });
475         } else {
476             logger.info("checking snapshot completion using wait_for_completion flag");

682     private boolean snapshotIsDone(String repository, String snapshot) {
683         try {
684             SnapshotsStatusResponse snapshotsStatusResponse = client().admin().cluster().prepareSnapshotStatus(repository).setSnapshots(snapshot).get();
685             if (snapshotsStatusResponse.getSnapshots().isEmpty()) {
686                 return false;
687             }
688             for (SnapshotStatus snapshotStatus : snapshotsStatusResponse.getSnapshots()) {
689                 if (snapshotStatus.getState().completed()) {
690                     return true;
691                 }
692             }
693             return false;
694         } catch (SnapshotMissingException ex) {
695             return false;
696         }
697     }

744         public static abstract class TestCustomMetaDataFactory<T extends TestCustomMetaData> extends MetaData.Custom.Factory<T> {
745 
746             protected abstract TestCustomMetaData newTestCustomMetaData(String data);
747 
748             @Override
749             public T readFrom(StreamInput in) throws IOException {
750                 return (T) newTestCustomMetaData(in.readString());
751             }
752 
753             @Override
754             public void writeTo(T metadata, StreamOutput out) throws IOException {
755                 out.writeString(metadata.getData());
756             }
757 
758             @Override
759             public T fromXContent(XContentParser parser) throws IOException {
760                 XContentParser.Token token;
761                 String data = null;
762                 while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
763                     if (token == XContentParser.Token.FIELD_NAME) {
764                         String currentFieldName = parser.currentName();
765                         if ("data".equals(currentFieldName)) {
766                             if (parser.nextToken() != XContentParser.Token.VALUE_STRING) {
767                                 throw new ElasticsearchParseException("failed to parse snapshottable metadata, invalid data type");
768                             }
769                             data = parser.text();
770                         } else {
771                             throw new ElasticsearchParseException("failed to parse snapshottable metadata, unknown field [" + currentFieldName + "]");
772                         }
773                     } else {
774                         throw new ElasticsearchParseException("failed to parse snapshottable metadata");
775                     }
776                 }
777                 if (data == null) {
778                     throw new ElasticsearchParseException("failed to parse snapshottable metadata, data not found");
779                 }
780                 return (T) newTestCustomMetaData(data);
781             }
782 
783             @Override
784             public void toXContent(T metadata, XContentBuilder builder, ToXContent.Params params) throws IOException {
785                 builder.field("data", metadata.getData());
786             }
787         }
