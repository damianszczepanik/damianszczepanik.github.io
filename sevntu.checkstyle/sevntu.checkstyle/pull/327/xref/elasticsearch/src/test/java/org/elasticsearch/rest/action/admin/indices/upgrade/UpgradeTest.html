68      public void testUpgrade() throws Exception {
69          // allow the cluster to rebalance quickly - 2 concurrent rebalance are default we can do higher
70          ImmutableSettings.Builder builder = ImmutableSettings.builder();
71          builder.put(ConcurrentRebalanceAllocationDecider.CLUSTER_ROUTING_ALLOCATION_CLUSTER_CONCURRENT_REBALANCE, 100);
72          client().admin().cluster().prepareUpdateSettings().setPersistentSettings(builder).get();
73  
74          int numIndexes = randomIntBetween(2, 4);
75          String[] indexNames = new String[numIndexes];
76          for (int i = 0; i < numIndexes; ++i) {
77              final String indexName = "test" + i;
78              indexNames[i] = indexName;
79              
80              Settings settings = ImmutableSettings.builder()
81                  .put("index.routing.allocation.exclude._name", backwardsCluster().newNodePattern())
82                  // don't allow any merges so that we can check segments are upgraded
83                  // by the upgrader, and not just regular merging
84                  .put("index.merge.policy.segments_per_tier", 1000000f)
85                  .put(indexSettings())
86                  .build();
87  
88              assertAcked(prepareCreate(indexName).setSettings(settings));
89              ensureGreen(indexName);
90              assertAllShardsOnNodes(indexName, backwardsCluster().backwardsNodePattern());
91  
92              int numDocs = scaledRandomIntBetween(100, 1000);
93              List<IndexRequestBuilder> docs = new ArrayList<>();
94              for (int j = 0; j < numDocs; ++j) {
95                  String id = Integer.toString(j);
96                  docs.add(client().prepareIndex(indexName, "type1", id).setSource("text", "sometext"));
97              }
98              indexRandom(true, docs);
99              ensureGreen(indexName);
100             if (globalCompatibilityVersion().before(Version.V_1_4_0_Beta1)) {
101                 // before 1.4 and the wait_if_ongoing flag, flushes could fail randomly, so we
102                 // need to continue to try flushing until all shards succeed
103                 assertTrue(awaitBusy(new Predicate<Object>() {
104                     @Override
105                     public boolean apply(Object o) {
106                         return flush(indexName).getFailedShards() == 0;
107                     }
108                 }));
109             } else {
110                 assertEquals(0, flush(indexName).getFailedShards());
111             }
112             
113             // index more docs that won't be flushed
114             numDocs = scaledRandomIntBetween(100, 1000);
115             docs = new ArrayList<>();
116             for (int j = 0; j < numDocs; ++j) {
117                 String id = Integer.toString(j);
118                 docs.add(client().prepareIndex(indexName, "type2", id).setSource("text", "someothertext"));
119             }
120             indexRandom(true, docs);
121             ensureGreen(indexName);
122         }
123         logger.debug("--> Upgrading nodes");
124         backwardsCluster().allowOnAllNodes(indexNames);
125         ensureGreen();
126         // disable allocation entirely until all nodes are upgraded
127         builder = ImmutableSettings.builder();
128         builder.put(EnableAllocationDecider.CLUSTER_ROUTING_ALLOCATION_ENABLE, EnableAllocationDecider.Allocation.NONE);
129         client().admin().cluster().prepareUpdateSettings().setTransientSettings(builder).get();
130         backwardsCluster().upgradeAllNodes();
131         builder = ImmutableSettings.builder();
132         // disable rebalanceing entirely for the time being otherwise we might get relocations / rebalance from nodes with old segments
133         builder.put(EnableAllocationDecider.CLUSTER_ROUTING_REBALANCE_ENABLE, EnableAllocationDecider.Rebalance.NONE);
134         builder.put(EnableAllocationDecider.CLUSTER_ROUTING_ALLOCATION_ENABLE, EnableAllocationDecider.Allocation.ALL);
135         client().admin().cluster().prepareUpdateSettings().setTransientSettings(builder).get();
136         ensureGreen();
137         logger.info("--> Nodes upgrade complete");
138         logSegmentsState();
139         
140         final HttpRequestBuilder httpClient = httpClient();
141 
142         assertNotUpgraded(httpClient, null);
143         final String indexToUpgrade = "test" + randomInt(numIndexes - 1);
144         
145         logger.info("--> Running upgrade on index " + indexToUpgrade);
146         runUpgrade(httpClient, indexToUpgrade);
147         awaitBusy(new Predicate<Object>() {
148             @Override
149             public boolean apply(Object o) {
150                 try {
151                     return isUpgraded(httpClient, indexToUpgrade);
152                 } catch (Exception e) {
153                     throw ExceptionsHelper.convertToRuntime(e);
154                 }
155             }
156         });
157         logger.info("--> Single index upgrade complete");
158         
159         logger.info("--> Running upgrade on the rest of the indexes");
160         runUpgrade(httpClient, null);
161         logSegmentsState();
162         logger.info("--> Full upgrade complete");
163         assertUpgraded(httpClient, null);
164     }

166     static String upgradePath(String index) {
167         String path = "/_upgrade";
168         if (index != null) {
169             path = "/" + index + path;
170         }
171         return path;
172     }
173     
174     public static void assertNotUpgraded(HttpRequestBuilder httpClient, String index) throws Exception {
175         for (UpgradeStatus status : getUpgradeStatus(httpClient, upgradePath(index))) {
176             assertTrue("index " + status.indexName + " should not be zero sized", status.totalBytes != 0);
177             // TODO: it would be better for this to be strictly greater, but sometimes an extra flush
178             // mysteriously happens after the second round of docs are indexed
179             assertTrue("index " + status.indexName + " should have recovered some segments from transaction log",
180                        status.totalBytes >= status.toUpgradeBytes);
181             assertTrue("index " + status.indexName + " should need upgrading", status.toUpgradeBytes != 0);
182         }
183     }
184 
185     public static void assertUpgraded(HttpRequestBuilder httpClient, String index) throws Exception {
186         for (UpgradeStatus status : getUpgradeStatus(httpClient, upgradePath(index))) {
187             assertTrue("index " + status.indexName + " should not be zero sized", status.totalBytes != 0);
188             assertEquals("index " + status.indexName + " should be upgraded",
189                 0, status.toUpgradeBytes);
190         }
191         
192         // double check using the segments api that all segments are actually upgraded
193         IndicesSegmentResponse segsRsp;
194         if (index == null) {
195             segsRsp = client().admin().indices().prepareSegments().execute().actionGet();
196         } else {
197             segsRsp = client().admin().indices().prepareSegments(index).execute().actionGet();
198         }
199         for (IndexSegments indexSegments : segsRsp.getIndices().values()) {
200             for (IndexShardSegments shard : indexSegments) {
201                 for (ShardSegments segs : shard.getShards()) {
202                     for (Segment seg : segs.getSegments()) {
203                         assertEquals("Index " + indexSegments.getIndex() + " has unupgraded segment " + seg.toString(),
204                                      Version.CURRENT.luceneVersion.major, seg.version.major);
205                         assertEquals("Index " + indexSegments.getIndex() + " has unupgraded segment " + seg.toString(),
206                                      Version.CURRENT.luceneVersion.minor, seg.version.minor);
207                     }
208                 }
209             }
210         }
211     }

213     static boolean isUpgraded(HttpRequestBuilder httpClient, String index) throws Exception {
214         ESLogger logger = Loggers.getLogger(UpgradeTest.class);
215         int toUpgrade = 0;
216         for (UpgradeStatus status : getUpgradeStatus(httpClient, upgradePath(index))) {
217             logger.info("Index: " + status.indexName + ", total: " + status.totalBytes + ", toUpgrade: " + status.toUpgradeBytes);
218             toUpgrade += status.toUpgradeBytes;
219         }
220         return toUpgrade == 0;
221     }

223     static class UpgradeStatus {
224         public final String indexName;
225         public final int totalBytes;
226         public final int toUpgradeBytes;
227         
228         public UpgradeStatus(String indexName, int totalBytes, int toUpgradeBytes) {
229             this.indexName = indexName;
230             this.totalBytes = totalBytes;
231             this.toUpgradeBytes = toUpgradeBytes;
232         }
233     }
234     
235     public static void runUpgrade(HttpRequestBuilder httpClient, String index, String... params) throws Exception {
236         assert params.length % 2 == 0;
237         HttpRequestBuilder builder = httpClient.method("POST").path(upgradePath(index));
238         for (int i = 0; i < params.length; i += 2) {
239             builder.addParam(params[i], params[i + 1]);
240         }
241         HttpResponse rsp = builder.execute();
242         assertNotNull(rsp);
243         assertEquals(200, rsp.getStatusCode());
244     }
245 
246     @SuppressWarnings("unchecked")
247     static List<UpgradeStatus> getUpgradeStatus(HttpRequestBuilder httpClient, String path) throws Exception {
248         HttpResponse rsp = httpClient.method("GET").path(path).execute();
249         Map<String,Object> data = validateAndParse(rsp);
250         List<UpgradeStatus> ret = new ArrayList<>();
251         for (String index : data.keySet()) {
252             Map<String, Object> status = (Map<String,Object>)data.get(index);
253             assertTrue("missing key size_in_bytes for index " + index, status.containsKey("size_in_bytes"));
254             Object totalBytes = status.get("size_in_bytes");
255             assertTrue("size_in_bytes for index " + index + " is not an integer", totalBytes instanceof Integer);
256             assertTrue("missing key size_to_upgrade_in_bytes for index " + index, status.containsKey("size_to_upgrade_in_bytes"));
257             Object toUpgradeBytes = status.get("size_to_upgrade_in_bytes");
258             assertTrue("size_to_upgrade_in_bytes for index " + index + " is not an integer", toUpgradeBytes instanceof Integer);
259             ret.add(new UpgradeStatus(index, (Integer)totalBytes, (Integer)toUpgradeBytes));
260         }
261         return ret;
262     }
263 
264     @SuppressWarnings("unchecked")
265     static Map<String, Object> validateAndParse(HttpResponse rsp) throws Exception {
266         assertNotNull(rsp);
267         assertEquals(200, rsp.getStatusCode());
268         assertTrue(rsp.hasBody());
269         return (Map<String,Object>)new JsonPath(rsp.getBody()).evaluate("");
270     }

272     HttpRequestBuilder httpClient() {
273         InetSocketAddress[] addresses = cluster().httpAddresses();
274         InetSocketAddress address = addresses[randomInt(addresses.length - 1)];
275         return new HttpRequestBuilder(HttpClients.createDefault()).host(address.getHostName()).port(address.getPort());
276     }
