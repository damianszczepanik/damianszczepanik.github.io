77          SignificanceHeuristicStreams.registerStream(ChiSquare.STREAM, ChiSquare.STREAM.getName());
78          SignificanceHeuristicStreams.registerStream(ScriptHeuristic.STREAM, ScriptHeuristic.STREAM.getName());
79          Version version = ElasticsearchIntegrationTest.randomVersion();
80          InternalSignificantTerms[] sigTerms = getRandomSignificantTerms(getRandomSignificanceheuristic());
81  
82          // write
83          ByteArrayOutputStream outBuffer = new ByteArrayOutputStream();
84          OutputStreamStreamOutput out = new OutputStreamStreamOutput(outBuffer);
85          out.setVersion(version);
86  
87          sigTerms[0].writeTo(out);
88  
89          // read
90          ByteArrayInputStream inBuffer = new ByteArrayInputStream(outBuffer.toByteArray());
91          InputStreamStreamInput in = new InputStreamStreamInput(inBuffer);
92          in.setVersion(version);
93  
94          sigTerms[1].readFrom(in);
95  
96          assertTrue(sigTerms[1].significanceHeuristic.equals(sigTerms[0].significanceHeuristic));
97      }

127     // 1. The output of the builders can actually be parsed

129     @Test
130     public void testBuilderAndParser() throws Exception {
131 
132         Set<SignificanceHeuristicParser> parsers = new HashSet<>();
133         parsers.add(new JLHScore.JLHScoreParser());
134         parsers.add(new MutualInformation.MutualInformationParser());
135         parsers.add(new GND.GNDParser());
136         parsers.add(new ChiSquare.ChiSquareParser());
137         SignificanceHeuristicParserMapper heuristicParserMapper = new SignificanceHeuristicParserMapper(parsers);
138         SearchContext searchContext = new SignificantTermsTestSearchContext();
139 
140         // test jlh with string
141         assertTrue(parseFromString(heuristicParserMapper, searchContext, "\"jlh\":{}") instanceof JLHScore);
142         // test gnd with string
143         assertTrue(parseFromString(heuristicParserMapper, searchContext, "\"gnd\":{}") instanceof GND);
144         // test mutual information with string
145         boolean includeNegatives = randomBoolean();
146         boolean backgroundIsSuperset = randomBoolean();
147         assertThat(parseFromString(heuristicParserMapper, searchContext, "\"mutual_information\":{\"include_negatives\": " + includeNegatives + ", \"background_is_superset\":" + backgroundIsSuperset + "}"), equalTo((SignificanceHeuristic) (new MutualInformation(includeNegatives, backgroundIsSuperset))));
148         assertThat(parseFromString(heuristicParserMapper, searchContext, "\"chi_square\":{\"include_negatives\": " + includeNegatives + ", \"background_is_superset\":" + backgroundIsSuperset + "}"), equalTo((SignificanceHeuristic) (new ChiSquare(includeNegatives, backgroundIsSuperset))));
149 
150         // test with builders
151         assertTrue(parseFromBuilder(heuristicParserMapper, searchContext, new JLHScore.JLHScoreBuilder()) instanceof JLHScore);
152         assertTrue(parseFromBuilder(heuristicParserMapper, searchContext, new GND.GNDBuilder(backgroundIsSuperset)) instanceof GND);
153         assertThat(parseFromBuilder(heuristicParserMapper, searchContext, new MutualInformation.MutualInformationBuilder(includeNegatives, backgroundIsSuperset)), equalTo((SignificanceHeuristic) new MutualInformation(includeNegatives, backgroundIsSuperset)));
154         assertThat(parseFromBuilder(heuristicParserMapper, searchContext, new ChiSquare.ChiSquareBuilder(includeNegatives, backgroundIsSuperset)), equalTo((SignificanceHeuristic) new ChiSquare(includeNegatives, backgroundIsSuperset)));
155 
156         // test exceptions
157         String faultyHeuristicdefinition = "\"mutual_information\":{\"include_negatives\": false, \"some_unknown_field\": false}";
158         String expectedError = "unknown for mutual_information";
159         checkParseException(heuristicParserMapper, searchContext, faultyHeuristicdefinition, expectedError);
160 
161         faultyHeuristicdefinition = "\"chi_square\":{\"unknown_field\": true}";
162         expectedError = "unknown for chi_square";
163         checkParseException(heuristicParserMapper, searchContext, faultyHeuristicdefinition, expectedError);
164 
165         faultyHeuristicdefinition = "\"jlh\":{\"unknown_field\": true}";
166         expectedError = "expected }, got ";
167         checkParseException(heuristicParserMapper, searchContext, faultyHeuristicdefinition, expectedError);
168 
169         faultyHeuristicdefinition = "\"gnd\":{\"unknown_field\": true}";
170         expectedError = "unknown for gnd";
171         checkParseException(heuristicParserMapper, searchContext, faultyHeuristicdefinition, expectedError);
172     }

174     protected void checkParseException(SignificanceHeuristicParserMapper heuristicParserMapper, SearchContext searchContext, String faultyHeuristicDefinition, String expectedError) throws IOException {
175         try {
176             XContentParser stParser = JsonXContent.jsonXContent.createParser("{\"field\":\"text\", " + faultyHeuristicDefinition + ",\"min_doc_count\":200}");
177             stParser.nextToken();
178             new SignificantTermsParser(heuristicParserMapper).parse("testagg", stParser, searchContext);
179             fail();
180         } catch (ElasticsearchParseException e) {
181             assertTrue(e.getMessage().contains(expectedError));
182         }
183     }

185     protected SignificanceHeuristic parseFromBuilder(SignificanceHeuristicParserMapper heuristicParserMapper, SearchContext searchContext, SignificanceHeuristicBuilder significanceHeuristicBuilder) throws IOException {
186         SignificantTermsBuilder stBuilder = new SignificantTermsBuilder("testagg");
187         stBuilder.significanceHeuristic(significanceHeuristicBuilder).field("text").minDocCount(200);
188         XContentBuilder stXContentBuilder = XContentFactory.jsonBuilder();
189         stBuilder.internalXContent(stXContentBuilder, null);
190         XContentParser stParser = JsonXContent.jsonXContent.createParser(stXContentBuilder.string());
191         return parseSignificanceHeuristic(heuristicParserMapper, searchContext, stParser);
192     }

194     private SignificanceHeuristic parseSignificanceHeuristic(SignificanceHeuristicParserMapper heuristicParserMapper, SearchContext searchContext, XContentParser stParser) throws IOException {
195         stParser.nextToken();
196         SignificantTermsAggregatorFactory aggregatorFactory = (SignificantTermsAggregatorFactory) new SignificantTermsParser(heuristicParserMapper).parse("testagg", stParser, searchContext);
197         stParser.nextToken();
198         assertThat(aggregatorFactory.getBucketCountThresholds().getMinDocCount(), equalTo(200l));
199         assertThat(stParser.currentToken(), equalTo(null));
200         stParser.close();
201         return aggregatorFactory.getSignificanceHeuristic();
202     }

204     protected SignificanceHeuristic parseFromString(SignificanceHeuristicParserMapper heuristicParserMapper, SearchContext searchContext, String heuristicString) throws IOException {
205         XContentParser stParser = JsonXContent.jsonXContent.createParser("{\"field\":\"text\", " + heuristicString + ", \"min_doc_count\":200}");
206         return parseSignificanceHeuristic(heuristicParserMapper, searchContext, stParser);
207     }

209     void testBackgroundAssertions(SignificanceHeuristic heuristicIsSuperset, SignificanceHeuristic heuristicNotSuperset) {
210         try {
211             heuristicIsSuperset.getScore(2, 3, 1, 4);
212             fail();
213         } catch (ElasticsearchIllegalArgumentException illegalArgumentException) {
214             assertNotNull(illegalArgumentException.getMessage());
215             assertTrue(illegalArgumentException.getMessage().contains("subsetFreq > supersetFreq"));
216         }
217         try {
218             heuristicIsSuperset.getScore(1, 4, 2, 3);
219             fail();
220         } catch (ElasticsearchIllegalArgumentException illegalArgumentException) {
221             assertNotNull(illegalArgumentException.getMessage());
222             assertTrue(illegalArgumentException.getMessage().contains("subsetSize > supersetSize"));
223         }
224         try {
225             heuristicIsSuperset.getScore(2, 1, 3, 4);
226             fail();
227         } catch (ElasticsearchIllegalArgumentException illegalArgumentException) {
228             assertNotNull(illegalArgumentException.getMessage());
229             assertTrue(illegalArgumentException.getMessage().contains("subsetFreq > subsetSize"));
230         }
231         try {
232             heuristicIsSuperset.getScore(1, 2, 4, 3);
233             fail();
234         } catch (ElasticsearchIllegalArgumentException illegalArgumentException) {
235             assertNotNull(illegalArgumentException.getMessage());
236             assertTrue(illegalArgumentException.getMessage().contains("supersetFreq > supersetSize"));
237         }
238         try {
239             heuristicIsSuperset.getScore(1, 3, 4, 4);
240             fail();
241         } catch (ElasticsearchIllegalArgumentException assertionError) {
242             assertNotNull(assertionError.getMessage());
243             assertTrue(assertionError.getMessage().contains("supersetFreq - subsetFreq > supersetSize - subsetSize"));
244         }
245         try {
246             int idx = randomInt(3);
247             long[] values = {1, 2, 3, 4};
248             values[idx] *= -1;
249             heuristicIsSuperset.getScore(values[0], values[1], values[2], values[3]);
250             fail();
251         } catch (ElasticsearchIllegalArgumentException illegalArgumentException) {
252             assertNotNull(illegalArgumentException.getMessage());
253             assertTrue(illegalArgumentException.getMessage().contains("Frequencies of subset and superset must be positive"));
254         }
255         try {
256             heuristicNotSuperset.getScore(2, 1, 3, 4);
257             fail();
258         } catch (ElasticsearchIllegalArgumentException illegalArgumentException) {
259             assertNotNull(illegalArgumentException.getMessage());
260             assertTrue(illegalArgumentException.getMessage().contains("subsetFreq > subsetSize"));
261         }
262         try {
263             heuristicNotSuperset.getScore(1, 2, 4, 3);
264             fail();
265         } catch (ElasticsearchIllegalArgumentException illegalArgumentException) {
266             assertNotNull(illegalArgumentException.getMessage());
267             assertTrue(illegalArgumentException.getMessage().contains("supersetFreq > supersetSize"));
268         }
269         try {
270             int idx = randomInt(3);
271             long[] values = {1, 2, 3, 4};
272             values[idx] *= -1;
273             heuristicNotSuperset.getScore(values[0], values[1], values[2], values[3]);
274             fail();
275         } catch (ElasticsearchIllegalArgumentException illegalArgumentException) {
276             assertNotNull(illegalArgumentException.getMessage());
277             assertTrue(illegalArgumentException.getMessage().contains("Frequencies of subset and superset must be positive"));
278         }
279     }

347     @Test
348     public void scoreMutual() throws Exception {
349         SignificanceHeuristic heuristic = new MutualInformation(true, true);
350         assertThat(heuristic.getScore(1, 1, 1, 3), greaterThan(0.0));
351         assertThat(heuristic.getScore(1, 1, 2, 3), lessThan(heuristic.getScore(1, 1, 1, 3)));
352         assertThat(heuristic.getScore(2, 2, 2, 4), equalTo(1.0));
353         assertThat(heuristic.getScore(0, 2, 2, 4), equalTo(1.0));
354         assertThat(heuristic.getScore(2, 2, 4, 4), equalTo(0.0));
355         assertThat(heuristic.getScore(1, 2, 2, 4), equalTo(0.0));
356         assertThat(heuristic.getScore(3, 6, 9, 18), equalTo(0.0));
357 
358         double score = 0.0;
359         try {
360             long a = randomLong();
361             long b = randomLong();
362             long c = randomLong();
363             long d = randomLong();
364             score = heuristic.getScore(a, b, c, d);
365         } catch (ElasticsearchIllegalArgumentException e) {
366         }
367         assertThat(score, lessThanOrEqualTo(1.0));
368         assertThat(score, greaterThanOrEqualTo(0.0));
369         heuristic = new MutualInformation(false, true);
370         assertThat(heuristic.getScore(0, 1, 2, 3), equalTo(Double.NEGATIVE_INFINITY));
371 
372         heuristic = new MutualInformation(true, false);
373         score = heuristic.getScore(2, 3, 1, 4);
374         assertThat(score, greaterThanOrEqualTo(0.0));
375         assertThat(score, lessThanOrEqualTo(1.0));
376         score = heuristic.getScore(1, 4, 2, 3);
377         assertThat(score, greaterThanOrEqualTo(0.0));
378         assertThat(score, lessThanOrEqualTo(1.0));
379         score = heuristic.getScore(1, 3, 4, 4);
380         assertThat(score, greaterThanOrEqualTo(0.0));
381         assertThat(score, lessThanOrEqualTo(1.0));
382     }

384     @Test
385     public void testGNDCornerCases() throws Exception {
386         GND gnd = new GND(true);
387         //term is only in the subset, not at all in the other set but that is because the other set is empty.
388         // this should actually not happen because only terms that are in the subset are considered now,
389         // however, in this case the score should be 0 because a term that does not exist cannot be relevant...
390         assertThat(gnd.getScore(0, randomIntBetween(1, 2), 0, randomIntBetween(2,3)), equalTo(0.0));
391         // the terms do not co-occur at all - should be 0
392         assertThat(gnd.getScore(0, randomIntBetween(1, 2), randomIntBetween(2, 3), randomIntBetween(5,6)), equalTo(0.0));
393         // comparison between two terms that do not exist - probably not relevant
394         assertThat(gnd.getScore(0, 0, 0, randomIntBetween(1,2)), equalTo(0.0));
395         // terms co-occur perfectly - should be 1
396         assertThat(gnd.getScore(1, 1, 1, 1), equalTo(1.0));
397         gnd = new GND(false);
398         assertThat(gnd.getScore(0, 0, 0, 0), equalTo(0.0));
399     }
