472     private String createRandomDelimiter(String[] tokens) {
473         String delimiter = "";
474         boolean isTokenOrWhitespace = true;
475         while(isTokenOrWhitespace) {
476             isTokenOrWhitespace = false;
477             delimiter = randomUnicodeOfLength(1);
478             for(String token:tokens) {
479                 if(token.contains(delimiter)) {
480                     isTokenOrWhitespace = true;
481                 }
482             }
483             if(Character.isWhitespace(delimiter.charAt(0))) {
484                 isTokenOrWhitespace = true;
485             }
486         }
487         return delimiter;
488     }

490     private String createString(String[] tokens, Map<String, List<BytesRef>> payloads, int encoding, char delimiter) {
491         String resultString = "";
492         ObjectIntOpenHashMap<String> payloadCounter = new ObjectIntOpenHashMap<>();
493         for (String token : tokens) {
494             if (!payloadCounter.containsKey(token)) {
495                 payloadCounter.putIfAbsent(token, 0);
496             } else {
497                 payloadCounter.put(token, payloadCounter.get(token) + 1);
498             }
499             resultString = resultString + token;
500             BytesRef payload = payloads.get(token).get(payloadCounter.get(token));
501             if (payload.length > 0) {
502                 resultString = resultString + delimiter;
503                 switch (encoding) {
504                 case 0: {
505                     resultString = resultString + Float.toString(PayloadHelper.decodeFloat(payload.bytes, payload.offset));
506                     break;
507                 }
508                 case 1: {
509                     resultString = resultString + Integer.toString(PayloadHelper.decodeInt(payload.bytes, payload.offset));
510                     break;
511                 }
512                 case 2: {
513                     resultString = resultString + payload.utf8ToString();
514                     break;
515                 }
516                 default: {
517                     throw new ElasticsearchException("unsupported encoding type");
518                 }
519                 }
520             }
521             resultString = resultString + " ";
522         }
523         return resultString;
524     }

526     private Map<String, List<BytesRef>> createPayloads(String[] tokens, int encoding) {
527         Map<String, List<BytesRef>> payloads = new HashMap<>();
528         for (String token : tokens) {
529             if (payloads.get(token) == null) {
530                 payloads.put(token, new ArrayList<BytesRef>());
531             }
532             boolean createPayload = randomBoolean();
533             if (createPayload) {
534                 switch (encoding) {
535                 case 0: {
536                     float theFloat = randomFloat();
537                     payloads.get(token).add(new BytesRef(PayloadHelper.encodeFloat(theFloat)));
538                     break;
539                 }
540                 case 1: {
541                     payloads.get(token).add(new BytesRef(PayloadHelper.encodeInt(randomInt())));
542                     break;
543                 }
544                 case 2: {
545                     String payload = randomUnicodeOfLengthBetween(50, 100);
546                     for (int c = 0; c < payload.length(); c++) {
547                         if (Character.isWhitespace(payload.charAt(c))) {
548                             payload = payload.replace(payload.charAt(c), 'w');
549                         }
550                     }
551                     payloads.get(token).add(new BytesRef(payload));
552                     break;
553                 }
554                 default: {
555                     throw new ElasticsearchException("unsupported encoding type");
556                 }
557                 }
558             } else {
559                 payloads.get(token).add(new BytesRef());
560             }
561         }
562         return payloads;
563     }

565     private String[] crateRandomTokens() {

629     private void checkBrownFoxTermVector(Fields fields, String fieldName, boolean withPayloads) throws ElasticsearchException, IOException {
630         String[] values = {"brown", "dog", "fox", "jumps", "lazy", "over", "quick", "the"};
631         int[] freq = {1, 1, 1, 1, 1, 1, 1, 2};
632         int[][] pos = {{2}, {8}, {3}, {4}, {7}, {5}, {1}, {0, 6}};
633         int[][] startOffset = {{10}, {40}, {16}, {20}, {35}, {26}, {4}, {0, 31}};
634         int[][] endOffset = {{15}, {43}, {19}, {25}, {39}, {30}, {9}, {3, 34}};
635 
636         Terms terms = fields.terms(fieldName);
637         assertThat(terms.size(), equalTo(8l));
638         TermsEnum iterator = terms.iterator(null);
639         for (int j = 0; j < values.length; j++) {
640             String string = values[j];
641             BytesRef next = iterator.next();
642             assertThat(next, notNullValue());
643             assertThat("expected " + string, string, equalTo(next.utf8ToString()));
644             assertThat(next, notNullValue());
645             // do not test ttf or doc frequency, because here we have many
646             // shards and do not know how documents are distributed
647             PostingsEnum docsAndPositions = iterator.postings(null, null, PostingsEnum.ALL);
648             assertThat(docsAndPositions.nextDoc(), equalTo(0));
649             assertThat(freq[j], equalTo(docsAndPositions.freq()));
650             int[] termPos = pos[j];
651             int[] termStartOffset = startOffset[j];
652             int[] termEndOffset = endOffset[j];
653             assertThat(termPos.length, equalTo(freq[j]));
654             assertThat(termStartOffset.length, equalTo(freq[j]));
655             assertThat(termEndOffset.length, equalTo(freq[j]));
656             for (int k = 0; k < freq[j]; k++) {
657                 int nextPosition = docsAndPositions.nextPosition();
658                 assertThat("term: " + string, nextPosition, equalTo(termPos[k]));
659                 assertThat("term: " + string, docsAndPositions.startOffset(), equalTo(termStartOffset[k]));
660                 assertThat("term: " + string, docsAndPositions.endOffset(), equalTo(termEndOffset[k]));
661                 if (withPayloads) {
662                     assertThat("term: " + string, docsAndPositions.getPayload(), equalTo(new BytesRef("word")));
663                 }
664             }
665         }
666         assertThat(iterator.next(), nullValue());
667     }

717     private void compareTermVectors(String fieldName, Fields fields0, Fields fields1) throws IOException {

719         Terms terms1 = fields1.terms(fieldName);
720         assertThat(terms0, notNullValue());
721         assertThat(terms1, notNullValue());
722         assertThat(terms0.size(), equalTo(terms1.size()));
723 
724         TermsEnum iter0 = terms0.iterator(null);
725         TermsEnum iter1 = terms1.iterator(null);
726         for (int i = 0; i < terms0.size(); i++) {
727             BytesRef next0 = iter0.next();
728             assertThat(next0, notNullValue());
729             BytesRef next1 = iter1.next();
730             assertThat(next1, notNullValue());
731 
732             // compare field value

734             String string1 = next1.utf8ToString();

736 
737             // compare df and ttf
738             assertThat("term: " + string0, iter0.docFreq(), equalTo(iter1.docFreq()));
739             assertThat("term: " + string0, iter0.totalTermFreq(), equalTo(iter1.totalTermFreq()));
740 
741             // compare freq and docs
742             PostingsEnum docsAndPositions0 = iter0.postings(null, null, PostingsEnum.ALL);
743             PostingsEnum docsAndPositions1 = iter1.postings(null, null, PostingsEnum.ALL);
744             assertThat("term: " + string0, docsAndPositions0.nextDoc(), equalTo(docsAndPositions1.nextDoc()));
745             assertThat("term: " + string0, docsAndPositions0.freq(), equalTo(docsAndPositions1.freq()));

749                 assertThat("term: " + string0, docsAndPositions0.nextPosition(), equalTo(docsAndPositions1.nextPosition()));
750                 assertThat("term: " + string0, docsAndPositions0.startOffset(), equalTo(docsAndPositions1.startOffset()));

752             }

1075         Map<String, Object> stats = JsonXContent.jsonXContent.createParser(xContentBuilder.bytes()).map();
1076         assertThat("number of fields expected:", fields.size(), equalTo(stats.size()));
1077         for (String fieldName : fields) {
1078             logger.info("Checking field statistics for field: {}", fieldName);

1080             Map<String, Integer> fieldStatistics = getFieldStatistics(stats, fieldName);
1081             String msg = "field: " + fieldName + " ";

1083                     (int) terms.getSumDocFreq(),
1084                     equalOrLessThanTo(fieldStatistics.get("sum_doc_freq"), isEqual));
1085             assertThat(msg + "doc_count:",
1086                     terms.getDocCount(),
1087                     equalOrLessThanTo(fieldStatistics.get("doc_count"), isEqual));
1088             assertThat(msg + "sum_ttf:",
1089                     (int) terms.getSumTotalTermFreq(),
1090                     equalOrLessThanTo(fieldStatistics.get("sum_ttf"), isEqual));

1098                 msg = "term: (" + fieldName + "," + term + ") ";

1100                         termsEnum.docFreq(),
1101                         equalOrLessThanTo(termStatistics.get("doc_freq"), isEqual));
1102                 assertThat(msg + "ttf:",
