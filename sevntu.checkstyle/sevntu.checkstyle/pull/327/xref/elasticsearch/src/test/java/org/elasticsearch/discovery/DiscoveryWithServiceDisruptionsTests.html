109     private List<String> startCluster(int numberOfNodes, int minimumMasterNode) throws ExecutionException, InterruptedException {
110         configureCluster(numberOfNodes, minimumMasterNode);
111         List<String> nodes = internalCluster().startNodesAsync(numberOfNodes).get();
112         ensureStableCluster(numberOfNodes);
113 
114         // TODO: this is a temporary solution so that nodes will not base their reaction to a partition based on previous successful results
115         for (ZenPingService pingService : internalCluster().getInstances(ZenPingService.class)) {
116             for (ZenPing zenPing : pingService.zenPings()) {
117                 if (zenPing instanceof UnicastZenPing) {
118                     ((UnicastZenPing) zenPing).clearTemporalResponses();
119                 }
120             }
121         }
122         return nodes;

152     private void configureCluster(int numberOfNodes, int minimumMasterNode) throws ExecutionException, InterruptedException {
153         if (randomBoolean()) {

155         } else {

161     private void configureMulticastCluster(int numberOfNodes, int minimumMasterNode) throws ExecutionException, InterruptedException {
162         if (minimumMasterNode < 0) {
163             minimumMasterNode = numberOfNodes / 2 + 1;
164         }
165         // TODO: Rarely use default settings form some of these
166         Settings settings = ImmutableSettings.builder()
167                 .put(DEFAULT_SETTINGS)
168                 .put(ElectMasterService.DISCOVERY_ZEN_MINIMUM_MASTER_NODES, minimumMasterNode)
169                 .build();
170 
171         if (discoveryConfig == null) {
172             discoveryConfig = new ClusterDiscoveryConfiguration(numberOfNodes, settings);
173         }
174     }

176     private void configureUnicastCluster(int numberOfNodes, @Nullable int[] unicastHostsOrdinals, int minimumMasterNode) throws ExecutionException, InterruptedException {
177         if (minimumMasterNode < 0) {
178             minimumMasterNode = numberOfNodes / 2 + 1;
179         }
180         // TODO: Rarely use default settings form some of these
181         Settings nodeSettings = ImmutableSettings.builder()
182                 .put(DEFAULT_SETTINGS)
183                 .put(ElectMasterService.DISCOVERY_ZEN_MINIMUM_MASTER_NODES, minimumMasterNode)
184                 .build();
185 
186         if (discoveryConfig == null) {
187             if (unicastHostsOrdinals == null) {
188                 discoveryConfig = new ClusterDiscoveryConfiguration.UnicastZen(numberOfNodes, nodeSettings);
189             } else {
190                 discoveryConfig = new ClusterDiscoveryConfiguration.UnicastZen(numberOfNodes, nodeSettings, unicastHostsOrdinals);
191             }

197      * Test that no split brain occurs under partial network partition. See https://github.com/elasticsearch/elasticsearch/issues/2488
198      *
199      * @throws Exception
200      */
201     @Test
202     public void failWithMinimumMasterNodesConfigured() throws Exception {
203 
204         List<String> nodes = startCluster(3);
205 
206         // Figure out what is the elected master node
207         final String masterNode = internalCluster().getMasterName();
208         logger.info("---> legit elected master node=" + masterNode);
209 
210         // Pick a node that isn't the elected master.
211         Set<String> nonMasters = new HashSet<>(nodes);
212         nonMasters.remove(masterNode);
213         final String unluckyNode = randomFrom(nonMasters.toArray(Strings.EMPTY_ARRAY));
214 
215 
216         // Simulate a network issue between the unlucky node and elected master node in both directions.
217 
218         NetworkDisconnectPartition networkDisconnect = new NetworkDisconnectPartition(masterNode, unluckyNode, getRandom());
219         setDisruptionScheme(networkDisconnect);
220         networkDisconnect.startDisrupting();
221 
222         // Wait until elected master has removed that the unlucky node...
223         ensureStableCluster(2, masterNode);
224 
225         // The unlucky node must report *no* master node, since it can't connect to master and in fact it should
226         // continuously ping until network failures have been resolved. However
227         // It may a take a bit before the node detects it has been cut off from the elected master
228         assertNoMaster(unluckyNode);
229 
230         networkDisconnect.stopDisrupting();
231 
232         // Wait until the master node sees all 3 nodes again.
233         ensureStableCluster(3);
234 
235         // The elected master shouldn't have changed, since the unlucky node never could have elected himself as
236         // master since m_m_n of 2 could never be satisfied.
237         assertMaster(masterNode, nodes);
238     }

269     @Test
270     public void testVerifyApiBlocksDuringPartition() throws Exception {
271         startCluster(3);
272 
273         // Makes sure that the get request can be executed on each node locally:
274         assertAcked(prepareCreate("test").setSettings(ImmutableSettings.builder()
275                         .put(IndexMetaData.SETTING_NUMBER_OF_SHARDS, 1)
276                         .put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, 2)
277         ));
278 
279         // Everything is stable now, it is now time to simulate evil...
280         // but first make sure we have no initializing shards and all is green
281         // (waiting for green here, because indexing / search in a yellow index is fine as long as no other nodes go down)
282         ensureGreen("test");
283 
284         NetworkPartition networkPartition = addRandomPartition();
285 
286         final String isolatedNode = networkPartition.getMinoritySide().get(0);
287         final String nonIsolatedNode = networkPartition.getMajoritySide().get(0);
288 
289         // Simulate a network issue between the unlucky node and the rest of the cluster.
290         networkPartition.startDisrupting();
291 
292 
293         // The unlucky node must report *no* master node, since it can't connect to master and in fact it should
294         // continuously ping until network failures have been resolved. However
295         // It may a take a bit before the node detects it has been cut off from the elected master
296         logger.info("waiting for isolated node [{}] to have no master", isolatedNode);
297         assertNoMaster(isolatedNode, DiscoverySettings.NO_MASTER_BLOCK_WRITES, TimeValue.timeValueSeconds(10));
298 
299 
300         logger.info("wait until elected master has been removed and a new 2 node cluster was from (via [{}])", isolatedNode);
301         ensureStableCluster(2, nonIsolatedNode);
302 
303         for (String node : networkPartition.getMajoritySide()) {
304             ClusterState nodeState = getNodeClusterState(node);
305             boolean success = true;
306             if (nodeState.nodes().getMasterNode() == null) {
307                 success = false;
308             }
309             if (!nodeState.blocks().global().isEmpty()) {
310                 success = false;
311             }
312             if (!success) {
313                 fail("node [" + node + "] has no master or has blocks, despite of being on the right side of the partition. State dump:\n"
314                         + nodeState.prettyPrint());
315             }
316         }
317 
318 
319         networkPartition.stopDisrupting();
320 
321         // Wait until the master node sees al 3 nodes again.
322         ensureStableCluster(3, new TimeValue(DISRUPTION_HEALING_OVERHEAD.millis() + networkPartition.expectedTimeToHeal().millis()));
323 
324         logger.info("Verify no master block with {} set to {}", DiscoverySettings.NO_MASTER_BLOCK, "all");
325         client().admin().cluster().prepareUpdateSettings()
326                 .setTransientSettings(ImmutableSettings.builder().put(DiscoverySettings.NO_MASTER_BLOCK, "all"))
327                 .get();
328 
329         networkPartition.startDisrupting();
330 
331 
332         // The unlucky node must report *no* master node, since it can't connect to master and in fact it should
333         // continuously ping until network failures have been resolved. However
334         // It may a take a bit before the node detects it has been cut off from the elected master
335         logger.info("waiting for isolated node [{}] to have no master", isolatedNode);
336         assertNoMaster(isolatedNode, DiscoverySettings.NO_MASTER_BLOCK_ALL, TimeValue.timeValueSeconds(10));
337 
338         // make sure we have stable cluster & cross partition recoveries are canceled by the removal of the missing node
339         // the unresponsive partition causes recoveries to only time out after 15m (default) and these will cause
340         // the test to fail due to unfreed resources
341         ensureStableCluster(2, nonIsolatedNode);
342 
343     }

368 
369         // make sure isolated need picks up on things.
370         assertNoMaster(isolatedNode, TimeValue.timeValueSeconds(40));
371 
372         // restore isolation
373         networkPartition.stopDisrupting();
374 
375         ensureStableCluster(3, new TimeValue(DISRUPTION_HEALING_OVERHEAD.millis() + networkPartition.expectedTimeToHeal().millis()));
376 
377         logger.info("issue a reroute");
378         // trigger a reroute now, instead of waiting for the background reroute of RerouteService
379         assertAcked(client().admin().cluster().prepareReroute());
380         // and wait for it to finish and for the cluster to stabilize
381         ensureGreen("test");
382 
383         // verify all cluster states are the same
384         ClusterState state = null;
385         for (String node : nodes) {
386             ClusterState nodeState = getNodeClusterState(node);
387             if (state == null) {
388                 state = nodeState;
389                 continue;
390             }
391             // assert nodes are identical
392             try {
393                 assertEquals("unequal versions", state.version(), nodeState.version());
394                 assertEquals("unequal node count", state.nodes().size(), nodeState.nodes().size());
395                 assertEquals("different masters ", state.nodes().masterNodeId(), nodeState.nodes().masterNodeId());
396                 assertEquals("different meta data version", state.metaData().version(), nodeState.metaData().version());
397                 if (!state.routingTable().prettyPrint().equals(nodeState.routingTable().prettyPrint())) {
398                     fail("different routing");
399                 }
400             } catch (AssertionError t) {
401                 fail("failed comparing cluster state: " + t.getMessage() + "\n" +
402                         "--- cluster state of node [" + nodes.get(0) + "]: ---\n" + state.prettyPrint() +
403                         "\n--- cluster state [" + node + "]: ---\n" + nodeState.prettyPrint());
404             }
405 
406         }

409     /**
410      * Test that we do not loose document whose indexing request was successful, under a randomly selected disruption scheme
411      * We also collect & report the type of indexing failures that occur.
412      *
413      * This test is a superset of tests run in the Jepsen test suite, with the exception of versioned updates
414      */
415     @Test
416     // NOTE: if you remove the awaitFix, make sure to port the test to the 1.x branch
417     @LuceneTestCase.AwaitsFix(bugUrl = "needs some more work to stabilize")
418     @TestLogging("action.index:TRACE,action.get:TRACE,discovery:TRACE,cluster.service:TRACE,indices.recovery:TRACE,indices.cluster:TRACE")
419     public void testAckedIndexing() throws Exception {
420         // TODO: add node count randomizaion
421         final List<String> nodes = startCluster(3);
422 
423         assertAcked(prepareCreate("test")
424                 .setSettings(ImmutableSettings.builder()
425                                 .put(IndexMetaData.SETTING_NUMBER_OF_SHARDS, 1 + randomInt(2))
426                                 .put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, randomInt(2))
427                 ));
428         ensureGreen();
429 
430         ServiceDisruptionScheme disruptionScheme = addRandomDisruptionScheme();
431         logger.info("disruption scheme [{}] added", disruptionScheme);
432 
433         final ConcurrentHashMap<String, String> ackedDocs = new ConcurrentHashMap<>(); // id -> node sent.
434 
435         final AtomicBoolean stop = new AtomicBoolean(false);
436         List<Thread> indexers = new ArrayList<>(nodes.size());
437         List<Semaphore> semaphores = new ArrayList<>(nodes.size());
438         final AtomicInteger idGenerator = new AtomicInteger(0);
439         final AtomicReference<CountDownLatch> countDownLatchRef = new AtomicReference<>();
440         final List<Exception> exceptedExceptions = Collections.synchronizedList(new ArrayList<Exception>());
441 
442         logger.info("starting indexers");
443         try {
444             for (final String node : nodes) {
445                 final Semaphore semaphore = new Semaphore(0);
446                 semaphores.add(semaphore);
447                 final Client client = client(node);
448                 final String name = "indexer_" + indexers.size();
449                 final int numPrimaries = getNumShards("test").numPrimaries;
450                 Thread thread = new Thread(new Runnable() {
451                     @Override
452                     public void run() {
453                         while (!stop.get()) {
454                             String id = null;
455                             try {
456                                 if (!semaphore.tryAcquire(10, TimeUnit.SECONDS)) {
457                                     continue;
458                                 }
459                                 logger.info("[{}] Acquired semaphore and it has {} permits left", name, semaphore.availablePermits());
460                                 try {
461                                     id = Integer.toString(idGenerator.incrementAndGet());
462                                     int shard = ((InternalTestCluster) cluster()).getInstance(DjbHashFunction.class).hash(id) % numPrimaries;
463                                     logger.trace("[{}] indexing id [{}] through node [{}] targeting shard [{}]", name, id, node, shard);
464                                     IndexResponse response = client.prepareIndex("test", "type", id).setSource("{}").setTimeout("1s").get();
465                                     assertThat(response.getVersion(), equalTo(1l));
466                                     ackedDocs.put(id, node);
467                                     logger.trace("[{}] indexed id [{}] through node [{}]", name, id, node);
468                                 } catch (ElasticsearchException e) {
469                                     exceptedExceptions.add(e);
470                                     logger.trace("[{}] failed id [{}] through node [{}]", e, name, id, node);
471                                 } finally {
472                                     countDownLatchRef.get().countDown();
473                                     logger.trace("[{}] decreased counter : {}", name, countDownLatchRef.get().getCount());
474                                 }
475                             } catch (InterruptedException e) {
476                                 // fine - semaphore interrupt
477                             } catch (Throwable t) {
478                                 logger.info("unexpected exception in background thread of [{}]", t, node);
479                             }
480                         }
481                     }
482                 });
483 
484                 thread.setName(name);
485                 thread.setDaemon(true);
486                 thread.start();
487                 indexers.add(thread);
488             }
489 
490             int docsPerIndexer = randomInt(3);
491             logger.info("indexing " + docsPerIndexer + " docs per indexer before partition");
492             countDownLatchRef.set(new CountDownLatch(docsPerIndexer * indexers.size()));
493             for (Semaphore semaphore : semaphores) {
494                 semaphore.release(docsPerIndexer);
495             }
496             assertTrue(countDownLatchRef.get().await(1, TimeUnit.MINUTES));
497 
498             for (int iter = 1 + randomInt(2); iter > 0; iter--) {
499                 logger.info("starting disruptions & indexing (iteration [{}])", iter);
500                 disruptionScheme.startDisrupting();
501 
502                 docsPerIndexer = 1 + randomInt(5);
503                 logger.info("indexing " + docsPerIndexer + " docs per indexer during partition");
504                 countDownLatchRef.set(new CountDownLatch(docsPerIndexer * indexers.size()));
505                 Collections.shuffle(semaphores);
506                 for (Semaphore semaphore : semaphores) {
507                     assertThat(semaphore.availablePermits(), equalTo(0));
508                     semaphore.release(docsPerIndexer);
509                 }
510                 assertTrue(countDownLatchRef.get().await(60000 + disruptionScheme.expectedTimeToHeal().millis() * (docsPerIndexer * indexers.size()), TimeUnit.MILLISECONDS));
511 
512                 logger.info("stopping disruption");
513                 disruptionScheme.stopDisrupting();
514                 ensureStableCluster(3, TimeValue.timeValueMillis(disruptionScheme.expectedTimeToHeal().millis() + DISRUPTION_HEALING_OVERHEAD.millis()));
515                 ensureGreen("test");
516 
517                 logger.info("validating successful docs");
518                 for (String node : nodes) {
519                     try {
520                         logger.debug("validating through node [{}]", node);
521                         for (String id : ackedDocs.keySet()) {
522                             assertTrue("doc [" + id + "] indexed via node [" + ackedDocs.get(id) + "] not found",
523                                     client(node).prepareGet("test", "type", id).setPreference("_local").get().isExists());
524                         }
525                     } catch (AssertionError e) {
526                         throw new AssertionError(e.getMessage() + " (checked via node [" + node + "]", e);
527                     }
528                 }
529 
530                 logger.info("done validating (iteration [{}])", iter);
531             }
532         } finally {
533             if (exceptedExceptions.size() > 0) {
534                 StringBuilder sb = new StringBuilder("Indexing exceptions during disruption:");
535                 for (Exception e : exceptedExceptions) {
536                     sb.append("\n").append(e.getMessage());
537                 }
538                 logger.debug(sb.toString());
539             }
540             logger.info("shutting down indexers");
541             stop.set(true);
542             for (Thread indexer : indexers) {
543                 indexer.interrupt();
544                 indexer.join(60000);
545             }
546         }
547     }

549     /**
550      * Test that cluster recovers from a long GC on master that causes other nodes to elect a new one
551      */
552     @Test
553     public void testMasterNodeGCs() throws Exception {
554         // TODO: on mac OS multicast threads are shared between nodes and we therefore we can't simulate GC and stop pinging for just one node
555         // find a way to block thread creation in the generic thread pool to avoid this.
556         List<String> nodes = startUnicastCluster(3, null, -1);
557 
558         String oldMasterNode = internalCluster().getMasterName();
559         // a very long GC, but it's OK as we remove the disruption when it has had an effect
560         SingleNodeDisruption masterNodeDisruption = new IntermittentLongGCDisruption(oldMasterNode, getRandom(), 100, 200, 30000, 60000);
561         internalCluster().setDisruptionScheme(masterNodeDisruption);
562         masterNodeDisruption.startDisrupting();
563 
564         Set<String> oldNonMasterNodesSet = new HashSet<>(nodes);
565         oldNonMasterNodesSet.remove(oldMasterNode);
566 
567         List<String> oldNonMasterNodes = new ArrayList<>(oldNonMasterNodesSet);
568 
569         logger.info("waiting for nodes to de-elect master [{}]", oldMasterNode);
570         for (String node : oldNonMasterNodesSet) {
571             assertDifferentMaster(node, oldMasterNode);
572         }
573 
574         logger.info("waiting for nodes to elect a new master");
575         ensureStableCluster(2, oldNonMasterNodes.get(0));
576 
577         logger.info("waiting for any pinging to stop");
578         assertDiscoveryCompleted(oldNonMasterNodes);
579 
580         // restore GC
581         masterNodeDisruption.stopDisrupting();
582         ensureStableCluster(3, new TimeValue(DISRUPTION_HEALING_OVERHEAD.millis() + masterNodeDisruption.expectedTimeToHeal().millis()),
583                 oldNonMasterNodes.get(0));
584

591     /**
592      * Tests that emulates a frozen elected master node that unfreezes and pushes his cluster state to other nodes
593      * that already are following another elected master node. These nodes should reject this cluster state and prevent
594      * them from following the stale master.
595      */
596     @Test
597     public void testStaleMasterNotHijackingMajority() throws Exception {
598         // TODO: on mac OS multicast threads are shared between nodes and we therefore we can't simulate GC and stop pinging for just one node
599         // find a way to block thread creation in the generic thread pool to avoid this.
600         // 3 node cluster with unicast discovery and minimum_master_nodes set to 2:
601         final List<String> nodes = startUnicastCluster(3, null, 2);
602 
603         // Save the current master node as old master node, because that node will get frozen
604         final String oldMasterNode = internalCluster().getMasterName();
605         for (String node : nodes) {
606             ensureStableCluster(3, node);
607         }
608         assertMaster(oldMasterNode, nodes);
609 
610         // Simulating a painful gc by suspending all threads for a long time on the current elected master node.
611         SingleNodeDisruption masterNodeDisruption = new LongGCDisruption(getRandom(), oldMasterNode);
612 
613         // Save the majority side
614         final List<String> majoritySide = new ArrayList<>(nodes);
615         majoritySide.remove(oldMasterNode);
616 
617         // Keeps track of the previous and current master when a master node transition took place on each node on the majority side:
618         final Map<String, List<Tuple<String, String>>> masters = Collections.synchronizedMap(new HashMap<String, List<Tuple<String, String>>>());
619         for (final String node : majoritySide) {
620             masters.put(node, new ArrayList<Tuple<String, String>>());
621             internalCluster().getInstance(ClusterService.class, node).add(new ClusterStateListener() {
622                 @Override
623                 public void clusterChanged(ClusterChangedEvent event) {
624                     DiscoveryNode previousMaster = event.previousState().nodes().getMasterNode();
625                     DiscoveryNode currentMaster = event.state().nodes().getMasterNode();
626                     if (!Objects.equals(previousMaster, currentMaster)) {
627                         logger.info("node {} received new cluster state: {} \n and had previous cluster state: {}", node, event.state(), event.previousState());
628                         String previousMasterNodeName = previousMaster != null ? previousMaster.name() : null;
629                         String currentMasterNodeName = currentMaster != null ? currentMaster.name() : null;
630                         masters.get(node).add(new Tuple<>(previousMasterNodeName, currentMasterNodeName));
631                     }
632                 }
633             });
634         }
635 
636         final CountDownLatch oldMasterNodeSteppedDown = new CountDownLatch(1);
637         internalCluster().getInstance(ClusterService.class, oldMasterNode).add(new ClusterStateListener() {
638             @Override
639             public void clusterChanged(ClusterChangedEvent event) {
640                 if (event.state().nodes().masterNodeId() == null) {
641                     oldMasterNodeSteppedDown.countDown();
642                 }
643             }
644         });
645 
646         internalCluster().setDisruptionScheme(masterNodeDisruption);
647         logger.info("freezing node [{}]", oldMasterNode);
648         masterNodeDisruption.startDisrupting();
649 
650         // Wait for the majority side to get stable
651         assertDifferentMaster(majoritySide.get(0), oldMasterNode);
652         assertDifferentMaster(majoritySide.get(1), oldMasterNode);
653         assertDiscoveryCompleted(majoritySide);
654 
655         // The old master node is frozen, but here we submit a cluster state update task that doesn't get executed,
656         // but will be queued and once the old master node un-freezes it gets executed.
657         // The old master node will send this update + the cluster state where he is flagged as master to the other
658         // nodes that follow the new master. These nodes should ignore this update.
659         internalCluster().getInstance(ClusterService.class, oldMasterNode).submitStateUpdateTask("sneaky-update", Priority.IMMEDIATE, new ClusterStateUpdateTask() {
660             @Override
661             public ClusterState execute(ClusterState currentState) throws Exception {
662                 return ClusterState.builder(currentState).build();
663             }
664 
665             @Override
666             public void onFailure(String source, Throwable t) {
667                 logger.warn("failure [{}]", t, source);
668             }
669         });
670 
671         // Save the new elected master node
672         final String newMasterNode = internalCluster().getMasterName(majoritySide.get(0));
673         logger.info("new detected master node [{}]", newMasterNode);
674 
675         // Stop disruption
676         logger.info("Unfreeze node [{}]", oldMasterNode);
677         masterNodeDisruption.stopDisrupting();
678 
679         oldMasterNodeSteppedDown.await(30, TimeUnit.SECONDS);
680         // Make sure that the end state is consistent on all nodes:
681         assertDiscoveryCompleted(nodes);
682         // Use assertBusy(...) because the unfrozen node may take a while to actually join the cluster.
683         // The assertDiscoveryCompleted(...) can't know if all nodes have the old master node in all of the local cluster states
684         assertBusy(new Runnable() {
685             @Override
686             public void run() {
687                 assertMaster(newMasterNode, nodes);
688             }
689         });
690 
691 
692         assertThat(masters.size(), equalTo(2));
693         for (Map.Entry<String, List<Tuple<String, String>>> entry : masters.entrySet()) {
694             String nodeName = entry.getKey();
695             List<Tuple<String, String>> recordedMasterTransition = entry.getValue();
696             assertThat("[" + nodeName + "] Each node should only record two master node transitions", recordedMasterTransition.size(), equalTo(2));
697             assertThat("[" + nodeName + "] First transition's previous master should be [null]", recordedMasterTransition.get(0).v1(), equalTo(oldMasterNode));
698             assertThat("[" + nodeName + "] First transition's current master should be [" + newMasterNode + "]", recordedMasterTransition.get(0).v2(), nullValue());
699             assertThat("[" + nodeName + "] Second transition's previous master should be [null]", recordedMasterTransition.get(1).v1(), nullValue());
700             assertThat("[" + nodeName + "] Second transition's current master should be [" + newMasterNode + "]", recordedMasterTransition.get(1).v2(), equalTo(newMasterNode));
701         }
702     }
