47      @Test
48      public void testBasics() throws Exception {
49          assertAcked(prepareCreate("test").addMapping("type", "field1", "type=string"));
50          ensureGreen();
51  
52          logger.info("--> register a queries");
53          client().prepareIndex("test", PercolatorService.TYPE_NAME, "1")
54                  .setSource(jsonBuilder().startObject().field("query", matchQuery("field1", "b")).field("a", "b").endObject())
55                  .execute().actionGet();
56          client().prepareIndex("test", PercolatorService.TYPE_NAME, "2")
57                  .setSource(jsonBuilder().startObject().field("query", matchQuery("field1", "c")).endObject())
58                  .execute().actionGet();
59          client().prepareIndex("test", PercolatorService.TYPE_NAME, "3")
60                  .setSource(jsonBuilder().startObject().field("query", boolQuery()
61                          .must(matchQuery("field1", "b"))
62                          .must(matchQuery("field1", "c"))
63                  ).endObject())
64                  .execute().actionGet();
65          client().prepareIndex("test", PercolatorService.TYPE_NAME, "4")
66                  .setSource(jsonBuilder().startObject().field("query", matchAllQuery()).endObject())
67                  .execute().actionGet();
68  
69          MultiPercolateResponse response = client().prepareMultiPercolate()
70                  .add(client().preparePercolate()
71                          .setIndices("test").setDocumentType("type")
72                          .setPercolateDoc(docBuilder().setDoc(jsonBuilder().startObject().field("field1", "b").endObject())))
73                  .add(client().preparePercolate()
74                          .setIndices("test").setDocumentType("type")
75                          .setPercolateDoc(docBuilder().setDoc(yamlBuilder().startObject().field("field1", "c").endObject())))
76                  .add(client().preparePercolate()
77                          .setIndices("test").setDocumentType("type")
78                          .setPercolateDoc(docBuilder().setDoc(smileBuilder().startObject().field("field1", "b c").endObject())))
79                  .add(client().preparePercolate()
80                          .setIndices("test").setDocumentType("type")
81                          .setPercolateDoc(docBuilder().setDoc(jsonBuilder().startObject().field("field1", "d").endObject())))
82                  .add(client().preparePercolate() // non existing doc, so error element
83                          .setIndices("test").setDocumentType("type")
84                          .setGetRequest(Requests.getRequest("test").type("type").id("5")))
85                  .execute().actionGet();
86  
87          MultiPercolateResponse.Item item = response.getItems()[0];
88          assertMatchCount(item.response(), 2l);
89          assertThat(item.getResponse().getMatches(), arrayWithSize(2));
90          assertThat(item.errorMessage(), nullValue());
91          assertThat(convertFromTextArray(item.getResponse().getMatches(), "test"), arrayContainingInAnyOrder("1", "4"));
92  
93          item = response.getItems()[1];
94          assertThat(item.errorMessage(), nullValue());
95  
96          assertMatchCount(item.response(), 2l);
97          assertThat(item.getResponse().getMatches(), arrayWithSize(2));
98          assertThat(convertFromTextArray(item.getResponse().getMatches(), "test"), arrayContainingInAnyOrder("2", "4"));
99  
100         item = response.getItems()[2];
101         assertThat(item.errorMessage(), nullValue());
102         assertMatchCount(item.response(), 4l);
103         assertThat(convertFromTextArray(item.getResponse().getMatches(), "test"), arrayContainingInAnyOrder("1", "2", "3", "4"));
104 
105         item = response.getItems()[3];
106         assertThat(item.errorMessage(), nullValue());
107         assertMatchCount(item.response(), 1l);
108         assertThat(item.getResponse().getMatches(), arrayWithSize(1));
109         assertThat(convertFromTextArray(item.getResponse().getMatches(), "test"), arrayContaining("4"));
110 
111         item = response.getItems()[4];
112         assertThat(item.getResponse(), nullValue());
113         assertThat(item.errorMessage(), notNullValue());
114         assertThat(item.errorMessage(), containsString("document missing"));
115     }

196     @Test
197     public void testExistingDocsOnly() throws Exception {
198         createIndex("test");
199 
200         int numQueries = randomIntBetween(50, 100);
201         logger.info("--> register a queries");
202         for (int i = 0; i < numQueries; i++) {
203             client().prepareIndex("test", PercolatorService.TYPE_NAME, Integer.toString(i))
204                     .setSource(jsonBuilder().startObject().field("query", matchAllQuery()).endObject())
205                     .execute().actionGet();
206         }
207 
208         client().prepareIndex("test", "type", "1")
209                 .setSource(jsonBuilder().startObject().field("field", "a"))
210                 .execute().actionGet();
211 
212         MultiPercolateRequestBuilder builder = client().prepareMultiPercolate();
213         int numPercolateRequest = randomIntBetween(50, 100);
214         for (int i = 0; i < numPercolateRequest; i++) {
215             builder.add(
216                     client().preparePercolate()
217                             .setGetRequest(Requests.getRequest("test").type("type").id("1"))
218                             .setIndices("test").setDocumentType("type"));
219         }
220 
221         MultiPercolateResponse response = builder.execute().actionGet();
222         assertThat(response.items().length, equalTo(numPercolateRequest));
223         for (MultiPercolateResponse.Item item : response) {
224             assertThat(item.isFailure(), equalTo(false));
225             assertMatchCount(item.response(), numQueries);
226             assertThat(item.getResponse().getMatches().length, equalTo(numQueries));
227         }
228 
229         // Non existing doc
230         builder = client().prepareMultiPercolate();
231         for (int i = 0; i < numPercolateRequest; i++) {
232             builder.add(
233                     client().preparePercolate()
234                             .setGetRequest(Requests.getRequest("test").type("type").id("2"))
235                             .setIndices("test").setDocumentType("type"));
236         }
237 
238         response = builder.execute().actionGet();
239         assertThat(response.items().length, equalTo(numPercolateRequest));
240         for (MultiPercolateResponse.Item item : response) {
241             assertThat(item.isFailure(), equalTo(true));
242             assertThat(item.errorMessage(), containsString("document missing"));
243             assertThat(item.getResponse(), nullValue());
244         }
245 
246         // One existing doc
247         builder = client().prepareMultiPercolate();
248         for (int i = 0; i < numPercolateRequest; i++) {
249             builder.add(
250                     client().preparePercolate()
251                             .setGetRequest(Requests.getRequest("test").type("type").id("2"))
252                             .setIndices("test").setDocumentType("type"));
253         }
254         builder.add(
255                 client().preparePercolate()
256                         .setGetRequest(Requests.getRequest("test").type("type").id("1"))
257                         .setIndices("test").setDocumentType("type"));
258 
259         response = builder.execute().actionGet();
260         assertThat(response.items().length, equalTo(numPercolateRequest + 1));
261         assertThat(response.items()[numPercolateRequest].isFailure(), equalTo(false));
262         assertMatchCount(response.items()[numPercolateRequest].response(), numQueries);
263         assertThat(response.items()[numPercolateRequest].getResponse().getMatches().length, equalTo(numQueries));
264     }

266     @Test
267     public void testWithDocsOnly() throws Exception {
268         createIndex("test");
269         ensureGreen();
270 
271         NumShards test = getNumShards("test");
272 
273         int numQueries = randomIntBetween(50, 100);
274         logger.info("--> register a queries");
275         for (int i = 0; i < numQueries; i++) {
276             client().prepareIndex("test", PercolatorService.TYPE_NAME, Integer.toString(i))
277                     .setSource(jsonBuilder().startObject().field("query", matchAllQuery()).endObject())
278                     .execute().actionGet();
279         }
280 
281         MultiPercolateRequestBuilder builder = client().prepareMultiPercolate();
282         int numPercolateRequest = randomIntBetween(50, 100);
283         for (int i = 0; i < numPercolateRequest; i++) {
284             builder.add(
285                     client().preparePercolate()
286                             .setIndices("test").setDocumentType("type")
287                             .setPercolateDoc(docBuilder().setDoc(jsonBuilder().startObject().field("field", "a").endObject())));
288         }
289 
290         MultiPercolateResponse response = builder.execute().actionGet();
291         assertThat(response.items().length, equalTo(numPercolateRequest));
292         for (MultiPercolateResponse.Item item : response) {
293             assertThat(item.isFailure(), equalTo(false));
294             assertMatchCount(item.response(), numQueries);
295             assertThat(item.getResponse().getMatches().length, equalTo(numQueries));
296         }
297 
298         // All illegal json
299         builder = client().prepareMultiPercolate();
300         for (int i = 0; i < numPercolateRequest; i++) {
301             builder.add(
302                     client().preparePercolate()
303                             .setIndices("test").setDocumentType("type")
304                             .setSource("illegal json"));
305         }
306 
307         response = builder.execute().actionGet();
308         assertThat(response.items().length, equalTo(numPercolateRequest));
309         for (MultiPercolateResponse.Item item : response) {
310             assertThat(item.isFailure(), equalTo(false));
311             assertThat(item.getResponse().getSuccessfulShards(), equalTo(0));
312             assertThat(item.getResponse().getShardFailures().length, equalTo(test.numPrimaries));
313             for (ShardOperationFailedException shardFailure : item.getResponse().getShardFailures()) {
314                 assertThat(shardFailure.reason(), containsString("Failed to derive xcontent from"));
315                 assertThat(shardFailure.status().getStatus(), equalTo(500));
316             }
317         }
318 
319         // one valid request
320         builder = client().prepareMultiPercolate();
321         for (int i = 0; i < numPercolateRequest; i++) {
322             builder.add(
323                     client().preparePercolate()
324                             .setIndices("test").setDocumentType("type")
325                             .setSource("illegal json"));
326         }
327         builder.add(
328                 client().preparePercolate()
329                         .setIndices("test").setDocumentType("type")
330                         .setPercolateDoc(docBuilder().setDoc(jsonBuilder().startObject().field("field", "a").endObject())));
331 
332         response = builder.execute().actionGet();
333         assertThat(response.items().length, equalTo(numPercolateRequest + 1));
334         assertThat(response.items()[numPercolateRequest].isFailure(), equalTo(false));
335         assertMatchCount(response.items()[numPercolateRequest].response(), numQueries);
336         assertThat(response.items()[numPercolateRequest].getResponse().getMatches().length, equalTo(numQueries));
337     }

340     @Test
341     public void testNestedMultiPercolation() throws IOException {
342         initNestedIndexAndPercolation();
343         MultiPercolateRequestBuilder mpercolate= client().prepareMultiPercolate();
344         mpercolate.add(client().preparePercolate().setPercolateDoc(new PercolateSourceBuilder.DocBuilder().setDoc(getNotMatchingNestedDoc())).setIndices("nestedindex").setDocumentType("company"));
345         mpercolate.add(client().preparePercolate().setPercolateDoc(new PercolateSourceBuilder.DocBuilder().setDoc(getMatchingNestedDoc())).setIndices("nestedindex").setDocumentType("company"));
346         MultiPercolateResponse response = mpercolate.get();
347         assertEquals(response.getItems()[0].getResponse().getMatches().length, 0);
348         assertEquals(response.getItems()[1].getResponse().getMatches().length, 1);
349         assertEquals(response.getItems()[1].getResponse().getMatches()[0].getId().string(), "Q");
350     }
351 
352     void initNestedIndexAndPercolation() throws IOException {
353         XContentBuilder mapping = XContentFactory.jsonBuilder();
354         mapping.startObject().startObject("properties").startObject("companyname").field("type", "string").endObject()
355                 .startObject("employee").field("type", "nested").startObject("properties")
356                 .startObject("name").field("type", "string").endObject().endObject().endObject().endObject()
357                 .endObject();
358 
359         assertAcked(client().admin().indices().prepareCreate("nestedindex").addMapping("company", mapping));
360         ensureGreen("nestedindex");
361 
362         client().prepareIndex("nestedindex", PercolatorService.TYPE_NAME, "Q").setSource(jsonBuilder().startObject()
363                 .field("query", QueryBuilders.nestedQuery("employee", QueryBuilders.matchQuery("employee.name", "virginia potts").operator(MatchQueryBuilder.Operator.AND)).scoreMode("avg")).endObject()).get();
364 
365         refresh();
366 
367     }

369     XContentBuilder getMatchingNestedDoc() throws IOException {
370         XContentBuilder doc = XContentFactory.jsonBuilder();
371         doc.startObject().field("companyname", "stark").startArray("employee")
372                 .startObject().field("name", "virginia potts").endObject()
373                 .startObject().field("name", "tony stark").endObject()
374                 .endArray().endObject();
375         return doc;
376     }

381                 .startObject().field("name", "virginia stark").endObject()
382                 .startObject().field("name", "tony potts").endObject()
383                 .endArray().endObject();
