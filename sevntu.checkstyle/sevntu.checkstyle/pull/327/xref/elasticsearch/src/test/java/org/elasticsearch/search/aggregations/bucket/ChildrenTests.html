54      @Override
55      public void setupSuiteScopeCluster() throws Exception {
56          assertAcked(
57                  prepareCreate("test")
58                      .addMapping("article")
59                      .addMapping("comment", "_parent", "type=article")
60          );
61  
62          List<IndexRequestBuilder> requests = new ArrayList<>();
63          String[] uniqueCategories = new String[randomIntBetween(1, 25)];
64          for (int i = 0; i < uniqueCategories.length; i++) {
65              uniqueCategories[i] = Integer.toString(i);
66          }
67          int catIndex = 0;
68  
69          int numParentDocs = randomIntBetween(uniqueCategories.length, uniqueCategories.length * 5);
70          for (int i = 0; i < numParentDocs; i++) {
71              String id = Integer.toString(i);
72  
73              // TODO: this array is always of length 1, and testChildrenAggs fails if this is changed
74              String[] categories = new String[randomIntBetween(1,1)];
75              for (int j = 0; j < categories.length; j++) {
76                  String category = categories[j] = uniqueCategories[catIndex++ % uniqueCategories.length];
77                  Control control = categoryToControl.get(category);
78                  if (control == null) {
79                      categoryToControl.put(category, control = new Control(category));
80                  }
81                  control.articleIds.add(id);
82              }
83  
84              requests.add(client().prepareIndex("test", "article", id).setCreate(true).setSource("category", categories, "randomized", true));
85          }
86  
87          String[] commenters = new String[randomIntBetween(5, 50)];
88          for (int i = 0; i < commenters.length; i++) {
89              commenters[i] = Integer.toString(i);
90          }
91  
92          int id = 0;
93          for (Control control : categoryToControl.values()) {
94              for (String articleId : control.articleIds) {
95                  int numChildDocsPerParent = randomIntBetween(0, 5);
96                  for (int i = 0; i < numChildDocsPerParent; i++) {
97                      String commenter = commenters[id % commenters.length];
98                      String idValue = Integer.toString(id++);
99                      control.commentIds.add(idValue);
100                     Set<String> ids = control.commenterToCommentId.get(commenter);
101                     if (ids == null) {
102                         control.commenterToCommentId.put(commenter, ids = new HashSet<>());
103                     }
104                     ids.add(idValue);
105                     requests.add(client().prepareIndex("test", "comment", idValue).setCreate(true).setParent(articleId).setSource("commenter", commenter));
106                 }
107             }
108         }
109 
110         requests.add(client().prepareIndex("test", "article", "a").setSource("category", new String[]{"a"}, "randomized", false));
111         requests.add(client().prepareIndex("test", "article", "b").setSource("category", new String[]{"a", "b"}, "randomized", false));
112         requests.add(client().prepareIndex("test", "article", "c").setSource("category", new String[]{"a", "b", "c"}, "randomized", false));
113         requests.add(client().prepareIndex("test", "article", "d").setSource("category", new String[]{"c"}, "randomized", false));
114         requests.add(client().prepareIndex("test", "comment", "a").setParent("a").setSource("{}"));
115         requests.add(client().prepareIndex("test", "comment", "c").setParent("c").setSource("{}"));
116 
117         indexRandom(true, requests);
118         ensureSearchable("test");
119     }

121     @Test
122     public void testChildrenAggs() throws Exception {
123         SearchResponse searchResponse = client().prepareSearch("test")
124                 .setQuery(matchQuery("randomized", true))
125                 .addAggregation(
126                         terms("category").field("category").size(0).subAggregation(
127                                 children("to_comment").childType("comment").subAggregation(
128                                         terms("commenters").field("commenter").size(0).subAggregation(
129                                                 topHits("top_comments")
130                                         ))
131                         )
132                 ).get();
133         assertSearchResponse(searchResponse);
134 
135         Terms categoryTerms = searchResponse.getAggregations().get("category");
136         assertThat(categoryTerms.getBuckets().size(), equalTo(categoryToControl.size()));
137         for (Map.Entry<String, Control> entry1 : categoryToControl.entrySet()) {
138             Terms.Bucket categoryBucket = categoryTerms.getBucketByKey(entry1.getKey());
139             assertThat(categoryBucket.getKeyAsString(), equalTo(entry1.getKey()));
140             assertThat(categoryBucket.getDocCount(), equalTo((long) entry1.getValue().articleIds.size()));
141 
142             Children childrenBucket = categoryBucket.getAggregations().get("to_comment");
143             assertThat(childrenBucket.getName(), equalTo("to_comment"));
144             assertThat(childrenBucket.getDocCount(), equalTo((long) entry1.getValue().commentIds.size()));
145             assertThat((long) childrenBucket.getProperty("_count"), equalTo((long) entry1.getValue().commentIds.size()));
146 
147             Terms commentersTerms = childrenBucket.getAggregations().get("commenters");
148             assertThat((Terms) childrenBucket.getProperty("commenters"), sameInstance(commentersTerms));
149             assertThat(commentersTerms.getBuckets().size(), equalTo(entry1.getValue().commenterToCommentId.size()));
150             for (Map.Entry<String, Set<String>> entry2 : entry1.getValue().commenterToCommentId.entrySet()) {
151                 Terms.Bucket commentBucket = commentersTerms.getBucketByKey(entry2.getKey());
152                 assertThat(commentBucket.getKeyAsString(), equalTo(entry2.getKey()));
153                 assertThat(commentBucket.getDocCount(), equalTo((long) entry2.getValue().size()));
154 
155                 TopHits topHits = commentBucket.getAggregations().get("top_comments");
156                 for (SearchHit searchHit : topHits.getHits().getHits()) {
157                     assertThat(entry2.getValue().contains(searchHit.getId()), is(true));
158                 }
159             }
160         }
161     }

163     @Test
164     public void testParentWithMultipleBuckets() throws Exception {
165         SearchResponse searchResponse = client().prepareSearch("test")
166                 .setQuery(matchQuery("randomized", false))
167                 .addAggregation(
168                         terms("category").field("category").size(0).subAggregation(
169                                 children("to_comment").childType("comment").subAggregation(topHits("top_comments").addSort("_uid", SortOrder.ASC))
170                         )
171                 ).get();
172         assertSearchResponse(searchResponse);
173 
174         Terms categoryTerms = searchResponse.getAggregations().get("category");
175         assertThat(categoryTerms.getBuckets().size(), equalTo(3));
176 
177         for (Terms.Bucket bucket : categoryTerms.getBuckets()) {
178             logger.info("bucket=" + bucket.getKey());
179             Children childrenBucket = bucket.getAggregations().get("to_comment");
180             TopHits topHits = childrenBucket.getAggregations().get("top_comments");
181             logger.info("total_hits={}", topHits.getHits().getTotalHits());
182             for (SearchHit searchHit : topHits.getHits()) {
183                 logger.info("hit= {} {} {}", searchHit.sortValues()[0], searchHit.getType(), searchHit.getId());
184             }
185         }
186 
187         Terms.Bucket categoryBucket = categoryTerms.getBucketByKey("a");
188         assertThat(categoryBucket.getKeyAsString(), equalTo("a"));
189         assertThat(categoryBucket.getDocCount(), equalTo(3l));
190 
191         Children childrenBucket = categoryBucket.getAggregations().get("to_comment");
192         assertThat(childrenBucket.getName(), equalTo("to_comment"));
193         assertThat(childrenBucket.getDocCount(), equalTo(2l));
194         TopHits topHits = childrenBucket.getAggregations().get("top_comments");
195         assertThat(topHits.getHits().totalHits(), equalTo(2l));
196         assertThat(topHits.getHits().getAt(0).getId(), equalTo("a"));
197         assertThat(topHits.getHits().getAt(0).getType(), equalTo("comment"));
198         assertThat(topHits.getHits().getAt(1).getId(), equalTo("c"));
199         assertThat(topHits.getHits().getAt(1).getType(), equalTo("comment"));
200 
201         categoryBucket = categoryTerms.getBucketByKey("b");
202         assertThat(categoryBucket.getKeyAsString(), equalTo("b"));
203         assertThat(categoryBucket.getDocCount(), equalTo(2l));
204 
205         childrenBucket = categoryBucket.getAggregations().get("to_comment");
206         assertThat(childrenBucket.getName(), equalTo("to_comment"));
207         assertThat(childrenBucket.getDocCount(), equalTo(1l));
208         topHits = childrenBucket.getAggregations().get("top_comments");
209         assertThat(topHits.getHits().totalHits(), equalTo(1l));
210         assertThat(topHits.getHits().getAt(0).getId(), equalTo("c"));
211         assertThat(topHits.getHits().getAt(0).getType(), equalTo("comment"));
212 
213         categoryBucket = categoryTerms.getBucketByKey("c");
214         assertThat(categoryBucket.getKeyAsString(), equalTo("c"));
215         assertThat(categoryBucket.getDocCount(), equalTo(2l));
216 
217         childrenBucket = categoryBucket.getAggregations().get("to_comment");
218         assertThat(childrenBucket.getName(), equalTo("to_comment"));
219         assertThat(childrenBucket.getDocCount(), equalTo(1l));
220         topHits = childrenBucket.getAggregations().get("top_comments");
221         assertThat(topHits.getHits().totalHits(), equalTo(1l));
222         assertThat(topHits.getHits().getAt(0).getId(), equalTo("c"));
223         assertThat(topHits.getHits().getAt(0).getType(), equalTo("comment"));
224     }

226     @Test
227     public void testWithDeletes() throws Exception {
228         String indexName = "xyz";
229         assertAcked(
230                 prepareCreate(indexName)
231                         .addMapping("parent")
232                         .addMapping("child", "_parent", "type=parent", "count", "type=long")
233         );
234 
235         List<IndexRequestBuilder> requests = new ArrayList<>();
236         requests.add(client().prepareIndex(indexName, "parent", "1").setSource("{}"));
237         requests.add(client().prepareIndex(indexName, "child", "0").setParent("1").setSource("count", 1));
238         requests.add(client().prepareIndex(indexName, "child", "1").setParent("1").setSource("count", 1));
239         requests.add(client().prepareIndex(indexName, "child", "2").setParent("1").setSource("count", 1));
240         requests.add(client().prepareIndex(indexName, "child", "3").setParent("1").setSource("count", 1));
241         indexRandom(true, requests);
242 
243         for (int i = 0; i < 10; i++) {
244             SearchResponse searchResponse = client().prepareSearch(indexName)
245                     .addAggregation(children("children").childType("child").subAggregation(sum("counts").field("count")))
246                     .get();
247 
248             assertNoFailures(searchResponse);
249             Children children = searchResponse.getAggregations().get("children");
250             assertThat(children.getDocCount(), equalTo(4l));
251 
252             Sum count = children.getAggregations().get("counts");
253             assertThat(count.getValue(), equalTo(4.));
254 
255             String idToUpdate = Integer.toString(randomInt(3));
256             UpdateResponse updateResponse = client().prepareUpdate(indexName, "child", idToUpdate)
257                     .setRouting("1")
258                     .setDoc("count", 1)
259                     .get();
260             assertThat(updateResponse.getVersion(), greaterThan(1l));
261             refresh();
262         }
263     }

278     @Test
279     public void testPostCollection() throws Exception {
280         String indexName = "prodcatalog";
281         String masterType = "masterprod";
282         String childType = "variantsku";
283         assertAcked(
284                 prepareCreate(indexName)
285                         .addMapping(masterType, "brand", "type=string", "name", "type=string", "material", "type=string")
286                         .addMapping(childType, "_parent", "type=masterprod", "color", "type=string", "size", "type=string")
287         );
288 
289         List<IndexRequestBuilder> requests = new ArrayList<>();
290         requests.add(client().prepareIndex(indexName, masterType, "1").setSource("brand", "Levis", "name", "Style 501", "material", "Denim"));
291         requests.add(client().prepareIndex(indexName, childType, "0").setParent("1").setSource("color", "blue", "size", "32"));
292         requests.add(client().prepareIndex(indexName, childType, "1").setParent("1").setSource("color", "blue", "size", "34"));
293         requests.add(client().prepareIndex(indexName, childType, "2").setParent("1").setSource("color", "blue", "size", "36"));
294         requests.add(client().prepareIndex(indexName, childType, "3").setParent("1").setSource("color", "black", "size", "38"));
295         requests.add(client().prepareIndex(indexName, childType, "4").setParent("1").setSource("color", "black", "size", "40"));
296         requests.add(client().prepareIndex(indexName, childType, "5").setParent("1").setSource("color", "gray", "size", "36"));
297 
298         requests.add(client().prepareIndex(indexName, masterType, "2").setSource("brand", "Wrangler", "name", "Regular Cut", "material", "Leather"));
299         requests.add(client().prepareIndex(indexName, childType, "6").setParent("2").setSource("color", "blue", "size", "32"));
300         requests.add(client().prepareIndex(indexName, childType, "7").setParent("2").setSource("color", "blue", "size", "34"));
301         requests.add(client().prepareIndex(indexName, childType, "8").setParent("2").setSource("color", "black", "size", "36"));
302         requests.add(client().prepareIndex(indexName, childType, "9").setParent("2").setSource("color", "black", "size", "38"));
303         requests.add(client().prepareIndex(indexName, childType, "10").setParent("2").setSource("color", "black", "size", "40"));
304         requests.add(client().prepareIndex(indexName, childType, "11").setParent("2").setSource("color", "orange", "size", "36"));
305         requests.add(client().prepareIndex(indexName, childType, "12").setParent("2").setSource("color", "green", "size", "44"));
306         indexRandom(true, requests);
307 
308         SearchResponse response = client().prepareSearch(indexName).setTypes(masterType)
309                 .setQuery(hasChildQuery(childType, termQuery("color", "orange")))
310                 .addAggregation(children("my-refinements")
311                                 .childType(childType)
312                                 .subAggregation(terms("my-colors").field("color"))
313                                 .subAggregation(terms("my-sizes").field("size"))
314                 ).get();
315         assertNoFailures(response);
316         assertHitCount(response, 1);
317 
318         Children childrenAgg = response.getAggregations().get("my-refinements");
319         assertThat(childrenAgg.getDocCount(), equalTo(7l));
320 
321         Terms termsAgg = childrenAgg.getAggregations().get("my-colors");
322         assertThat(termsAgg.getBuckets().size(), equalTo(4));
323         assertThat(termsAgg.getBucketByKey("black").getDocCount(), equalTo(3l));
324         assertThat(termsAgg.getBucketByKey("blue").getDocCount(), equalTo(2l));
325         assertThat(termsAgg.getBucketByKey("green").getDocCount(), equalTo(1l));
326         assertThat(termsAgg.getBucketByKey("orange").getDocCount(), equalTo(1l));
327 
328         termsAgg = childrenAgg.getAggregations().get("my-sizes");
329         assertThat(termsAgg.getBuckets().size(), equalTo(6));
330         assertThat(termsAgg.getBucketByKey("36").getDocCount(), equalTo(2l));
331         assertThat(termsAgg.getBucketByKey("32").getDocCount(), equalTo(1l));
332         assertThat(termsAgg.getBucketByKey("34").getDocCount(), equalTo(1l));
333         assertThat(termsAgg.getBucketByKey("38").getDocCount(), equalTo(1l));
334         assertThat(termsAgg.getBucketByKey("40").getDocCount(), equalTo(1l));
335         assertThat(termsAgg.getBucketByKey("44").getDocCount(), equalTo(1l));
336     }
