69      @Test
70      public void testStoredFields() throws Exception {
71          createIndex("test");
72          client().admin().cluster().prepareHealth().setWaitForEvents(Priority.LANGUID).setWaitForYellowStatus().execute().actionGet();
73  
74          String mapping = XContentFactory.jsonBuilder().startObject().startObject("type1")
75                  // _timestamp and _size are randomly enabled via templates but we don't want it here to test stored fields behaviour
76                  .startObject("_timestamp").field("enabled", false).endObject()
77                  .startObject("_size").field("enabled", false).endObject()
78                  .startObject("properties")
79                  .startObject("field1").field("type", "string").field("store", "yes").endObject()
80                  .startObject("field2").field("type", "string").field("store", "no").endObject()
81                  .startObject("field3").field("type", "string").field("store", "yes").endObject()
82                  .endObject().endObject().endObject().string();
83  
84          client().admin().indices().preparePutMapping().setType("type1").setSource(mapping).execute().actionGet();
85  
86          client().prepareIndex("test", "type1", "1").setSource(jsonBuilder().startObject()
87                  .field("field1", "value1")
88                  .field("field2", "value2")
89                  .field("field3", "value3")
90                  .endObject()).execute().actionGet();
91  
92          client().admin().indices().prepareRefresh().execute().actionGet();
93  
94          SearchResponse searchResponse = client().prepareSearch().setQuery(matchAllQuery()).addField("field1").execute().actionGet();
95          assertThat(searchResponse.getHits().getTotalHits(), equalTo(1l));
96          assertThat(searchResponse.getHits().hits().length, equalTo(1));
97          assertThat(searchResponse.getHits().getAt(0).fields().size(), equalTo(1));
98          assertThat(searchResponse.getHits().getAt(0).fields().get("field1").value().toString(), equalTo("value1"));
99  
100         // field2 is not stored, check that it gets extracted from source
101         searchResponse = client().prepareSearch().setQuery(matchAllQuery()).addField("field2").execute().actionGet();
102         assertThat(searchResponse.getHits().getTotalHits(), equalTo(1l));
103         assertThat(searchResponse.getHits().hits().length, equalTo(1));
104         assertThat(searchResponse.getHits().getAt(0).fields().size(), equalTo(1));
105         assertThat(searchResponse.getHits().getAt(0).fields().get("field2").value().toString(), equalTo("value2"));
106 
107         searchResponse = client().prepareSearch().setQuery(matchAllQuery()).addField("field3").execute().actionGet();
108         assertThat(searchResponse.getHits().getTotalHits(), equalTo(1l));
109         assertThat(searchResponse.getHits().hits().length, equalTo(1));
110         assertThat(searchResponse.getHits().getAt(0).fields().size(), equalTo(1));
111         assertThat(searchResponse.getHits().getAt(0).fields().get("field3").value().toString(), equalTo("value3"));
112 
113         searchResponse = client().prepareSearch().setQuery(matchAllQuery()).addField("*").execute().actionGet();
114         assertThat(searchResponse.getHits().getTotalHits(), equalTo(1l));
115         assertThat(searchResponse.getHits().hits().length, equalTo(1));
116         assertThat(searchResponse.getHits().getAt(0).source(), nullValue());
117         assertThat(searchResponse.getHits().getAt(0).fields().size(), equalTo(2));
118         assertThat(searchResponse.getHits().getAt(0).fields().get("field1").value().toString(), equalTo("value1"));
119         assertThat(searchResponse.getHits().getAt(0).fields().get("field3").value().toString(), equalTo("value3"));
120 
121         searchResponse = client().prepareSearch().setQuery(matchAllQuery()).addField("*").addField("_source").execute().actionGet();
122         assertThat(searchResponse.getHits().getTotalHits(), equalTo(1l));
123         assertThat(searchResponse.getHits().hits().length, equalTo(1));
124         assertThat(searchResponse.getHits().getAt(0).source(), notNullValue());
125         assertThat(searchResponse.getHits().getAt(0).fields().size(), equalTo(2));

134 
135         String mapping = XContentFactory.jsonBuilder().startObject().startObject("type1").startObject("properties")
136                 .startObject("num1").field("type", "double").field("store", "yes").endObject()
137                 .endObject().endObject().endObject().string();
138 
139         client().admin().indices().preparePutMapping().setType("type1").setSource(mapping).execute().actionGet();
140 
141         client().prepareIndex("test", "type1", "1")
142                 .setSource(jsonBuilder().startObject().field("test", "value beck").field("num1", 1.0f).field("date", "1970-01-01T00:00:00").endObject())
143                 .execute().actionGet();
144         client().admin().indices().prepareFlush().execute().actionGet();
145         client().prepareIndex("test", "type1", "2")
146                 .setSource(jsonBuilder().startObject().field("test", "value beck").field("num1", 2.0f).field("date", "1970-01-01T00:00:25").endObject())
147                 .execute().actionGet();
148         client().admin().indices().prepareFlush().execute().actionGet();
149         client().prepareIndex("test", "type1", "3")
150                 .setSource(jsonBuilder().startObject().field("test", "value beck").field("num1", 3.0f).field("date", "1970-01-01T00:02:00").endObject())
151                 .execute().actionGet();
152         client().admin().indices().refresh(refreshRequest()).actionGet();
153 
154         logger.info("running doc['num1'].value");
155         SearchResponse response = client().prepareSearch()
156                 .setQuery(matchAllQuery())
157                 .addSort("num1", SortOrder.ASC)
158                 .addScriptField("sNum1", "doc['num1'].value")
159                 .addScriptField("sNum1_field", "_fields['num1'].value")
160                 .addScriptField("date1", "doc['date'].date.millis")
161                 .execute().actionGet();
162 
163         assertNoFailures(response);
164 
165         assertThat(response.getHits().totalHits(), equalTo(3l));
166         assertThat(response.getHits().getAt(0).isSourceEmpty(), equalTo(true));
167         assertThat(response.getHits().getAt(0).id(), equalTo("1"));
168         assertThat(response.getHits().getAt(0).fields().size(), equalTo(3));
169         assertThat((Double) response.getHits().getAt(0).fields().get("sNum1").values().get(0), equalTo(1.0));
170         assertThat((Double) response.getHits().getAt(0).fields().get("sNum1_field").values().get(0), equalTo(1.0));
171         assertThat((Long) response.getHits().getAt(0).fields().get("date1").values().get(0), equalTo(0l));
172         assertThat(response.getHits().getAt(1).id(), equalTo("2"));
173         assertThat(response.getHits().getAt(1).fields().size(), equalTo(3));
174         assertThat((Double) response.getHits().getAt(1).fields().get("sNum1").values().get(0), equalTo(2.0));
175         assertThat((Double) response.getHits().getAt(1).fields().get("sNum1_field").values().get(0), equalTo(2.0));
176         assertThat((Long) response.getHits().getAt(1).fields().get("date1").values().get(0), equalTo(25000l));
177         assertThat(response.getHits().getAt(2).id(), equalTo("3"));
178         assertThat(response.getHits().getAt(2).fields().size(), equalTo(3));
179         assertThat((Double) response.getHits().getAt(2).fields().get("sNum1").values().get(0), equalTo(3.0));
180         assertThat((Double) response.getHits().getAt(2).fields().get("sNum1_field").values().get(0), equalTo(3.0));
181         assertThat((Long) response.getHits().getAt(2).fields().get("date1").values().get(0), equalTo(120000l));
182 
183         logger.info("running doc['num1'].value * factor");
184         Map<String, Object> params = MapBuilder.<String, Object>newMapBuilder().put("factor", 2.0).map();
185         response = client().prepareSearch()
186                 .setQuery(matchAllQuery())
187                 .addSort("num1", SortOrder.ASC)
188                 .addScriptField("sNum1", "doc['num1'].value * factor", params)
189                 .execute().actionGet();
190 
191         assertThat(response.getHits().totalHits(), equalTo(3l));
192         assertThat(response.getHits().getAt(0).id(), equalTo("1"));
193         assertThat(response.getHits().getAt(0).fields().size(), equalTo(1));
194         assertThat((Double) response.getHits().getAt(0).fields().get("sNum1").values().get(0), equalTo(2.0));
195         assertThat(response.getHits().getAt(1).id(), equalTo("2"));
196         assertThat(response.getHits().getAt(1).fields().size(), equalTo(1));
197         assertThat((Double) response.getHits().getAt(1).fields().get("sNum1").values().get(0), equalTo(4.0));
198         assertThat(response.getHits().getAt(2).id(), equalTo("3"));
199         assertThat(response.getHits().getAt(2).fields().size(), equalTo(1));
200         assertThat((Double) response.getHits().getAt(2).fields().get("sNum1").values().get(0), equalTo(6.0));
201     }

205         prepareCreate("test").addMapping("type1", "num1", "type=long").execute().actionGet();
206         ensureYellow();
207 
208         int numDocs = randomIntBetween(1, 30);
209         IndexRequestBuilder[] indexRequestBuilders = new IndexRequestBuilder[numDocs];

211             indexRequestBuilders[i] = client().prepareIndex("test", "type1", Integer.toString(i))

213         }
214         indexRandom(true, indexRequestBuilders);
215

217                 .setQuery(matchAllQuery()).addSort("num1", SortOrder.ASC).setSize(numDocs)
218                 .addScriptField("uid", "_fields._uid.value").get();

220         assertNoFailures(response);
221 
222         assertThat(response.getHits().totalHits(), equalTo((long)numDocs));
223         for (int i = 0; i < numDocs; i++) {
224             assertThat(response.getHits().getAt(i).id(), equalTo(Integer.toString(i)));
225             assertThat(response.getHits().getAt(i).fields().size(), equalTo(1));
226             assertThat((String)response.getHits().getAt(i).fields().get("uid").value(), equalTo("type1#" + Integer.toString(i)));
227         }
228 
229         response = client().prepareSearch()
230                 .setQuery(matchAllQuery()).addSort("num1", SortOrder.ASC).setSize(numDocs)
231                 .addScriptField("id", "_fields._id.value").get();
232 
233         assertNoFailures(response);
234 
235         assertThat(response.getHits().totalHits(), equalTo((long)numDocs));
236         for (int i = 0; i < numDocs; i++) {
237             assertThat(response.getHits().getAt(i).id(), equalTo(Integer.toString(i)));
238             assertThat(response.getHits().getAt(i).fields().size(), equalTo(1));
239             assertThat((String)response.getHits().getAt(i).fields().get("id").value(), equalTo(Integer.toString(i)));

267             assertThat((String)response.getHits().getAt(i).fields().get("uid").value(), equalTo("type1#" + Integer.toString(i)));
268             assertThat((String)response.getHits().getAt(i).fields().get("type").value(), equalTo("type1"));
269             assertThat((String)response.getHits().getAt(i).fields().get("id").value(), equalTo(Integer.toString(i)));
270         }

576     public void testScriptFields() throws Exception {
577         assertAcked(prepareCreate("index").addMapping("type",
578                 "s", "type=string,index=not_analyzed",
579                 "l", "type=long",
580                 "d", "type=double",
581                 "ms", "type=string,index=not_analyzed",
582                 "ml", "type=long",
583                 "md", "type=double").get());
584         final int numDocs = randomIntBetween(3, 8);
585         List<IndexRequestBuilder> reqs = new ArrayList<>();
586         for (int i = 0; i < numDocs; ++i) {
587             reqs.add(client().prepareIndex("index", "type", Integer.toString(i)).setSource(
588                     "s", Integer.toString(i),
589                     "ms", new String[] {Integer.toString(i), Integer.toString(i+1)},
590                     "l", i,
591                     "ml", new long[] {i, i+1},
592                     "d", i,
593                     "md", new double[] {i, i+1}));
594         }
595         indexRandom(true, reqs);
596         ensureSearchable();
597         SearchRequestBuilder req = client().prepareSearch("index");
598         for (String field : Arrays.asList("s", "ms", "l", "ml", "d", "md")) {
599             req.addScriptField(field, "doc['" + field + "'].values");
600         }
601         SearchResponse resp = req.get();
602         assertSearchResponse(resp);
603         for (SearchHit hit : resp.getHits().getHits()) {
604             final int id = Integer.parseInt(hit.getId());
605             Map<String, SearchHitField> fields = hit.getFields();
606             assertThat(fields.get("s").getValues(), equalTo(Collections.<Object>singletonList(Integer.toString(id))));
607             assertThat(fields.get("l").getValues(), equalTo(Collections.<Object>singletonList((long) id)));
608             assertThat(fields.get("d").getValues(), equalTo(Collections.<Object>singletonList((double) id)));
609             assertThat(fields.get("ms").getValues(), equalTo(Arrays.<Object>asList(Integer.toString(id), Integer.toString(id + 1))));
610             assertThat(fields.get("ml").getValues(), equalTo(Arrays.<Object>asList((long) id, id+1L)));
611             assertThat(fields.get("md").getValues(), equalTo(Arrays.<Object>asList((double) id, id+1d)));
612         }
613     }
