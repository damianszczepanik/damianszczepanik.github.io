60  public class TermsAggregationSearchAndIndexingBenchmark {
61  
62      static String indexName = "test";
63      static String typeName = "type1";
64      static Random random = new Random();
65  
66      static long COUNT = SizeValue.parseSizeValue("2m").singles();
67      static int BATCH = 1000;
68      static int NUMBER_OF_TERMS = (int) SizeValue.parseSizeValue("100k").singles();
69      static int NUMBER_OF_MULTI_VALUE_TERMS = 10;
70      static int STRING_TERM_SIZE = 5;
71  
72      static Node[] nodes;
73  
74      public static void main(String[] args) throws Exception {
75          Natives.tryMlockall();
76          Settings settings = settingsBuilder()
77                  .put("refresh_interval", "-1")
78                  .put(SETTING_NUMBER_OF_SHARDS, 1)
79                  .put(SETTING_NUMBER_OF_REPLICAS, 0)
80                  .build();
81  
82          String clusterName = TermsAggregationSearchAndIndexingBenchmark.class.getSimpleName();
83          nodes = new Node[1];
84          for (int i = 0; i < nodes.length; i++) {
85              nodes[i] = nodeBuilder().settings(settingsBuilder().put(settings).put("name", "node1"))
86                      .clusterName(clusterName)
87                      .node();
88          }
89          Client client = nodes[0].client();
90  
91          client.admin().cluster().prepareHealth(indexName).setWaitForGreenStatus().setTimeout("10s").execute().actionGet();
92          try {
93              client.admin().indices().prepareCreate(indexName)
94                      .addMapping(typeName, generateMapping("eager", "lazy"))
95                      .get();
96              Thread.sleep(5000);
97  
98              long startTime = System.currentTimeMillis();
99              ObjectOpenHashSet<String> uniqueTerms = ObjectOpenHashSet.newInstance();
100             for (int i = 0; i < NUMBER_OF_TERMS; i++) {
101                 boolean added;
102                 do {
103                     added = uniqueTerms.add(RandomStrings.randomAsciiOfLength(random, STRING_TERM_SIZE));
104                 } while (!added);
105             }
106             String[] sValues = uniqueTerms.toArray(String.class);
107             long ITERS = COUNT / BATCH;
108             long i = 1;
109             int counter = 0;
110             for (; i <= ITERS; i++) {
111                 BulkRequestBuilder request = client.prepareBulk();
112                 for (int j = 0; j < BATCH; j++) {
113                     counter++;
114 
115                     XContentBuilder builder = jsonBuilder().startObject();
116                     builder.field("id", Integer.toString(counter));
117                     final String sValue = sValues[counter % sValues.length];
118                     builder.field("s_value", sValue);
119                     builder.field("s_value_dv", sValue);
120 
121                     for (String field : new String[] {"sm_value", "sm_value_dv"}) {
122                         builder.startArray(field);
123                         for (int k = 0; k < NUMBER_OF_MULTI_VALUE_TERMS; k++) {
124                             builder.value(sValues[ThreadLocalRandom.current().nextInt(sValues.length)]);
125                         }
126                         builder.endArray();
127                     }
128 
129                     request.add(Requests.indexRequest(indexName).type("type1").id(Integer.toString(counter))
130                             .source(builder));
131                 }
132                 BulkResponse response = request.execute().actionGet();
133                 if (response.hasFailures()) {
134                     System.err.println("--> failures...");
135                 }
136                 if (((i * BATCH) % 10000) == 0) {
137                     System.out.println("--> Indexed " + (i * BATCH));
138                 }
139             }
140 
141             System.out.println("--> Indexing took " + ((System.currentTimeMillis() - startTime) / 1000) + " seconds.");
142         } catch (IndexAlreadyExistsException e) {
143             System.out.println("--> Index already exists, ignoring indexing phase, waiting for green");
144             ClusterHealthResponse clusterHealthResponse = client.admin().cluster().prepareHealth(indexName).setWaitForGreenStatus().setTimeout("10m").execute().actionGet();
145             if (clusterHealthResponse.isTimedOut()) {
146                 System.err.println("--> Timed out waiting for cluster health");
147             }
148         }
149         client.admin().indices().preparePutMapping(indexName)
150                 .setType(typeName)
151                 .setSource(generateMapping("lazy", "lazy"))
152                 .get();
153         client.admin().indices().prepareRefresh().execute().actionGet();
154         System.out.println("--> Number of docs in index: " + client.prepareCount().setQuery(matchAllQuery()).execute().actionGet().getCount());
155 
156 
157         String[] nodeIds = new String[nodes.length];
158         for (int i = 0; i < nodeIds.length; i++) {
159             nodeIds[i] = nodes[i].injector().getInstance(Discovery.class).localNode().getId();
160         }
161 
162         List<TestRun> testRuns = new ArrayList<>();
163         testRuns.add(new TestRun("Regular field ordinals", "eager", "lazy", "s_value", "ordinals"));
164         testRuns.add(new TestRun("Docvalues field ordinals", "lazy", "eager", "s_value_dv", "ordinals"));
165         testRuns.add(new TestRun("Regular field global ordinals", "eager_global_ordinals", "lazy", "s_value", null));
166         testRuns.add(new TestRun("Docvalues field global", "lazy", "eager_global_ordinals", "s_value_dv", null));
167 
168         List<TestResult> testResults = new ArrayList<>();
169         for (TestRun testRun : testRuns) {
170             client.admin().indices().preparePutMapping(indexName).setType(typeName)
171                     .setSource(generateMapping(testRun.indexedFieldEagerLoading, testRun.docValuesEagerLoading)).get();
172             client.admin().indices().prepareClearCache(indexName).setFieldDataCache(true).get();
173             SearchThread searchThread = new SearchThread(client, testRun.termsAggsField, testRun.termsAggsExecutionHint);
174             RefreshThread refreshThread = new RefreshThread(client);
175             System.out.println("--> Running '" + testRun.name + "' round...");
176             new Thread(refreshThread).start();
177             new Thread(searchThread).start();
178             Thread.sleep(2 * 60 * 1000);
179             refreshThread.stop();
180             searchThread.stop();
181 
182             System.out.println("--> Avg refresh time: " + refreshThread.avgRefreshTime + " ms");
183             System.out.println("--> Avg query time: " + searchThread.avgQueryTime + " ms");
184 
185             ClusterStatsResponse clusterStateResponse = client.admin().cluster().prepareClusterStats().setNodesIds(nodeIds).get();
186             System.out.println("--> Heap used: " + clusterStateResponse.getNodesStats().getJvm().getHeapUsed());
187             ByteSizeValue fieldDataMemoryUsed = clusterStateResponse.getIndicesStats().getFieldData().getMemorySize();
188             System.out.println("--> Fielddata memory size: " + fieldDataMemoryUsed);
189             testResults.add(new TestResult(testRun.name, refreshThread.avgRefreshTime, searchThread.avgQueryTime, fieldDataMemoryUsed));
190         }
191 
192         System.out.println("----------------------------------------- SUMMARY ----------------------------------------------");
193         System.out.format(Locale.ENGLISH, "%30s%18s%15s%15s\n", "name", "avg refresh time", "avg query time", "fieldata size");
194         for (TestResult testResult : testResults) {
195             System.out.format(Locale.ENGLISH, "%30s%18s%15s%15s\n", testResult.name, testResult.avgRefreshTime, testResult.avgQueryTime, testResult.fieldDataSizeInMemory);
196         }
197         System.out.println("----------------------------------------- SUMMARY ----------------------------------------------");
198 
199         client.close();
200         for (Node node : nodes) {
201             node.close();
202         }
203     }

205     static class RefreshThread implements Runnable {
206 
207         private final Client client;
208         private volatile boolean run = true;
209         private volatile boolean stopped = false;
210         private volatile long avgRefreshTime = 0;
211 
212         RefreshThread(Client client) throws IOException {
213             this.client = client;
214         }
215 
216         @Override
217         public void run() {
218             long totalRefreshTime = 0;
219             int numExecutedRefreshed = 0;
220             while (run) {
221                 long docIdLimit = COUNT;
222                 for (long docId = 1; run && docId < docIdLimit;) {
223                     try {
224                         for (int j = 0; j < 8; j++) {
225                             GetResponse getResponse = client
226                                     .prepareGet(indexName, "type1", String.valueOf(++docId))
227                                     .get();
228                             client.prepareIndex(indexName, "type1", getResponse.getId())
229                                     .setSource(getResponse.getSource())
230                                     .get();
231                         }
232                         long startTime = System.currentTimeMillis();
233                         client.admin().indices().prepareRefresh(indexName).execute().actionGet();
234                         totalRefreshTime += System.currentTimeMillis() - startTime;
235                         numExecutedRefreshed++;
236                         Thread.sleep(500);
237                     } catch (Throwable e) {
238                         e.printStackTrace();
239                     }
240                 }
241             }
242             avgRefreshTime = totalRefreshTime / numExecutedRefreshed;
243             stopped = true;
244         }
245 
246         public void stop() throws InterruptedException {
247             run = false;
248             while (!stopped) {
249                 Thread.sleep(100);
250             }
251         }
252 
253     }

255     private static class TestRun {
256 
257         final String name;
258         final String indexedFieldEagerLoading;
259         final String docValuesEagerLoading;
260         final String termsAggsField;
261         final String termsAggsExecutionHint;

263         private TestRun(String name, String indexedFieldEagerLoading, String docValuesEagerLoading, String termsAggsField, String termsAggsExecutionHint) {
264             this.name = name;
265             this.indexedFieldEagerLoading = indexedFieldEagerLoading;

267             this.termsAggsField = termsAggsField;
268             this.termsAggsExecutionHint = termsAggsExecutionHint;
269         }

272     private static class TestResult {
273 
274         final String name;
275         final TimeValue avgRefreshTime;
276         final TimeValue avgQueryTime;
277         final ByteSizeValue fieldDataSizeInMemory;
278 
279         private TestResult(String name, long avgRefreshTime, long avgQueryTime, ByteSizeValue fieldDataSizeInMemory) {
280             this.name = name;
281             this.avgRefreshTime = TimeValue.timeValueMillis(avgRefreshTime);
282             this.avgQueryTime = TimeValue.timeValueMillis(avgQueryTime);
283             this.fieldDataSizeInMemory = fieldDataSizeInMemory;
284         }

287     static class SearchThread implements Runnable {
288 
289         private final Client client;
290         private final String field;
291         private final String executionHint;
292         private volatile boolean run = true;
293         private volatile boolean stopped = false;
294         private volatile long avgQueryTime = 0;
295 
296         SearchThread(Client client, String field, String executionHint) {
297             this.client = client;
298             this.field = field;
299             this.executionHint = executionHint;
300         }
301 
302         @Override
303         public void run() {
304             long totalQueryTime = 0;
305             int numExecutedQueries = 0;
306             while (run) {
307                 try {
308                     SearchResponse searchResponse = Method.AGGREGATION.addTermsAgg(client.prepareSearch()
309                             .setSearchType(SearchType.COUNT)
310                             .setQuery(matchAllQuery()), "test", field, executionHint)
311                             .execute().actionGet();
312                     if (searchResponse.getHits().totalHits() != COUNT) {
313                         System.err.println("--> mismatch on hits");
314                     }
315                     totalQueryTime += searchResponse.getTookInMillis();
316                     numExecutedQueries++;
317                 } catch (Throwable e) {
318                     e.printStackTrace();
319                 }
320             }
321             avgQueryTime = totalQueryTime / numExecutedQueries;
322             stopped = true;
323         }
324 
325         public void stop() throws InterruptedException {
326             run = false;
327             while (!stopped) {
328                 Thread.sleep(100);
329             }
330         }
331 
332     }

334     private static XContentBuilder generateMapping(String loading1, String loading2) throws IOException {
335         return jsonBuilder().startObject().startObject("type1").startObject("properties")
336                 .startObject("s_value")
337                 .field("type", "string")
338                 .field("index", "not_analyzed")
339                 .startObject("fielddata")
340                 .field("loading", loading1)
341                 .endObject()
342                 .endObject()
343                 .startObject("s_value_dv")
344                 .field("type", "string")
345                 .field("index", "no")
346                 .startObject("fielddata")
347                 .field("loading", loading2)
348                 .field("format", "doc_values")
349                 .endObject()
350                 .endObject()
351                 .endObject().endObject().endObject();
352     }
