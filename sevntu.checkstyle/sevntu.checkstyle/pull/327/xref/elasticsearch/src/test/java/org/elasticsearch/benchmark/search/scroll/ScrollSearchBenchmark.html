20  package org.elasticsearch.benchmark.search.scroll;
21  
22  import org.elasticsearch.action.admin.cluster.health.ClusterHealthResponse;
23  import org.elasticsearch.action.bulk.BulkRequestBuilder;
24  import org.elasticsearch.action.search.SearchResponse;
25  import org.elasticsearch.client.Client;
26  import org.elasticsearch.common.settings.Settings;
27  import org.elasticsearch.common.unit.SizeValue;
28  import org.elasticsearch.indices.IndexAlreadyExistsException;
29  import org.elasticsearch.monitor.jvm.JvmStats;
30  import org.elasticsearch.node.Node;
31  import org.elasticsearch.search.SearchHit;
32  import org.elasticsearch.search.sort.SortOrder;
33  
34  import java.util.Locale;
35  
36  import static org.elasticsearch.cluster.metadata.IndexMetaData.SETTING_NUMBER_OF_REPLICAS;
37  import static org.elasticsearch.cluster.metadata.IndexMetaData.SETTING_NUMBER_OF_SHARDS;
38  import static org.elasticsearch.common.settings.ImmutableSettings.settingsBuilder;
39  import static org.elasticsearch.node.NodeBuilder.nodeBuilder;
40  
41  /**
42   */
43  public class ScrollSearchBenchmark {
44  
45      // Run with: -Xms1G -Xms1G
46      public static void main(String[] args) {
47          String indexName = "test";
48          String typeName = "type";
49          String clusterName = ScrollSearchBenchmark.class.getSimpleName();
50          long numDocs = SizeValue.parseSizeValue("300k").singles();
51          int requestSize = 50;
52  
53          Settings settings = settingsBuilder()
54                  .put(SETTING_NUMBER_OF_SHARDS, 3)
55                  .put(SETTING_NUMBER_OF_REPLICAS, 0)
56                  .build();
57  
58          Node[] nodes = new Node[3];
59          for (int i = 0; i < nodes.length; i++) {
60              nodes[i] = nodeBuilder()
61                      .clusterName(clusterName)
62                      .settings(settingsBuilder().put(settings).put("name", "node" + i))
63                      .node();
64          }
65  
66          Client client = nodes[0].client();
67  
68          try {
69              client.admin().indices().prepareCreate(indexName).get();
70              for (int counter = 1; counter <= numDocs;) {
71                  BulkRequestBuilder bulkRequestBuilder = client.prepareBulk();
72                  for (int bulkCounter = 0; bulkCounter < 100; bulkCounter++) {
73                      if (counter > numDocs) {
74                          break;
75                      }
76                      bulkRequestBuilder.add(
77                              client.prepareIndex(indexName, typeName, String.valueOf(counter))
78                                      .setSource("field1", counter++)
79                      );
80                  }
81                  int indexedDocs = counter - 1;
82                  if (indexedDocs % 100000 == 0) {
83                       System.out.printf(Locale.ENGLISH, "--> Indexed %d so far\n", indexedDocs);
84                  }
85                  bulkRequestBuilder.get();
86              }
87          } catch (IndexAlreadyExistsException e) {
88              System.out.println("--> Index already exists, ignoring indexing phase, waiting for green");
89              ClusterHealthResponse clusterHealthResponse = client.admin().cluster().prepareHealth(indexName).setWaitForGreenStatus().setTimeout("10m").execute().actionGet();
90              if (clusterHealthResponse.isTimedOut()) {
91                  System.err.println("--> Timed out waiting for cluster health");
92              }
93          }
94  
95          client.admin().indices().prepareRefresh(indexName).get();
96          System.out.printf(Locale.ENGLISH, "--> Number of docs in index: %d\n", client.prepareCount().get().getCount());
97  
98          Long counter = numDocs;
99          SearchResponse searchResponse = client.prepareSearch(indexName)
100                 .addSort("field1", SortOrder.DESC)
101                 .setSize(requestSize)
102                 .setScroll("10m").get();
103 
104         if (searchResponse.getHits().getTotalHits() != numDocs) {
105             System.err.printf(Locale.ENGLISH, "Expected total hits [%d] but got [%d]\n", numDocs, searchResponse.getHits().getTotalHits());
106         }
107 
108         if (searchResponse.getHits().hits().length != requestSize) {
109             System.err.printf(Locale.ENGLISH, "Expected hits length [%d] but got [%d]\n", requestSize, searchResponse.getHits().hits().length);
110         }
111 
112         for (SearchHit hit : searchResponse.getHits()) {
113             if (!hit.sortValues()[0].equals(counter--)) {
114                 System.err.printf(Locale.ENGLISH, "Expected sort value [%d] but got [%s]\n", counter + 1, hit.sortValues()[0]);
115             }
116         }
117         String scrollId = searchResponse.getScrollId();
118         int scrollRequestCounter = 0;
119         long sumTimeSpent = 0;
120         while (true) {
121             long timeSpent = System.currentTimeMillis();
122             searchResponse = client.prepareSearchScroll(scrollId).setScroll("10m").get();
123             sumTimeSpent += (System.currentTimeMillis() - timeSpent);
124             scrollRequestCounter++;
125             if (searchResponse.getHits().getTotalHits() != numDocs) {
126                 System.err.printf(Locale.ENGLISH, "Expected total hits [%d] but got [%d]\n", numDocs, searchResponse.getHits().getTotalHits());
127             }
128             if (scrollRequestCounter % 20 == 0) {
129                 long avgTimeSpent = sumTimeSpent / 20;
130                 JvmStats.Mem mem = JvmStats.jvmStats().mem();
131                 System.out.printf(Locale.ENGLISH, "Cursor location=%d, avg time spent=%d ms\n", (requestSize * scrollRequestCounter), (avgTimeSpent));
132                 System.out.printf(Locale.ENGLISH, "heap max=%s, used=%s, percentage=%d\n", mem.getHeapMax(), mem.getHeapUsed(), mem.getHeapUsedPrecent());
133                 sumTimeSpent = 0;
134             }
135             if (searchResponse.getHits().hits().length == 0) {
136                 break;
137             }
138             if (searchResponse.getHits().hits().length != requestSize) {
139                 System.err.printf(Locale.ENGLISH, "Expected hits length [%d] but got [%d]\n", requestSize, searchResponse.getHits().hits().length);
140             }
141             for (SearchHit hit : searchResponse.getHits()) {
142                 if (!hit.sortValues()[0].equals(counter--)) {
143                     System.err.printf(Locale.ENGLISH, "Expected sort value [%d] but got [%s]\n", counter + 1, hit.sortValues()[0]);
144                 }
145             }
146             scrollId = searchResponse.getScrollId();
147         }
148         if (counter != 0) {
149             System.err.printf(Locale.ENGLISH, "Counter should be 0 because scroll has been consumed\n");
150         }
151 
152         for (Node node : nodes) {
153             node.close();
154         }
155     }
156 
157 }
