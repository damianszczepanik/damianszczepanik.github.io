211     @Test
212     public void testSingleField() throws NoSuchFieldException, IllegalAccessException {
213         SearchResponse searchResponse = client().prepareSearch("test")
214                 .setQuery(randomizeType(multiMatchQuery("15", "skill"))).get();
215         assertNoFailures(searchResponse);
216         assertFirstHit(searchResponse, hasId("theone"));
217         String[] fields = {"full_name", "first_name", "last_name", "last_name_phrase", "first_name_phrase", "category_phrase", "category"};
218 
219         String[] query = {"marvel","hero", "captain",  "america", "15", "17", "1", "5", "ultimate", "Man",
220                 "marvel", "wolferine", "ninja"};
221 
222         // check if it's equivalent to a match query.
223         int numIters = scaledRandomIntBetween(10, 100);
224         for (int i = 0; i < numIters; i++) {
225             String field = RandomPicks.randomFrom(getRandom(), fields);
226             int numTerms = randomIntBetween(1, query.length);
227             StringBuilder builder = new StringBuilder();
228             for (int j = 0; j < numTerms; j++) {
229                 builder.append(RandomPicks.randomFrom(getRandom(), query)).append(" ");
230             }
231             MultiMatchQueryBuilder multiMatchQueryBuilder = randomizeType(multiMatchQuery(builder.toString(), field));
232             SearchResponse multiMatchResp = client().prepareSearch("test")
233                     // _uid sort field is a tie, in case hits have the same score,
234                     // the hits will be sorted the same consistently
235                     .addSort("_score", SortOrder.DESC)
236                     .addSort("_uid", SortOrder.ASC)
237                     .setQuery(multiMatchQueryBuilder).get();
238             MatchQueryBuilder matchQueryBuilder = QueryBuilders.matchQuery(field, builder.toString());
239             if (getType(multiMatchQueryBuilder) != null) {
240                 matchQueryBuilder.type(MatchQueryBuilder.Type.valueOf(getType(multiMatchQueryBuilder).matchQueryType().toString()));
241             }
242             SearchResponse matchResp = client().prepareSearch("test")
243                     // _uid tie sort
244                     .addSort("_score", SortOrder.DESC)
245                     .addSort("_uid", SortOrder.ASC)
246                     .setQuery(matchQueryBuilder).get();
247             assertThat("field: " + field + " query: " + builder.toString(), multiMatchResp.getHits().getTotalHits(), equalTo(matchResp.getHits().getTotalHits()));
248             SearchHits hits = multiMatchResp.getHits();
249             for (int j = 0; j < hits.hits().length; j++) {
250                 assertThat(hits.getHits()[j].score(), equalTo(matchResp.getHits().getHits()[j].score()));
251                 assertThat(hits.getHits()[j].getId(), equalTo(matchResp.getHits().getHits()[j].getId()));
252             }
253         }

320 
321         final int numDocs = (int) client().prepareCount("test")
322                 .setQuery(matchAllQuery()).get().getCount();
323         int numIters = scaledRandomIntBetween(5, 10);
324         for (int i = 0; i < numIters; i++) {
325             {
326                 MatchQueryBuilder.Type type = randomBoolean() ? null : MatchQueryBuilder.Type.BOOLEAN;
327                 MultiMatchQueryBuilder multiMatchQueryBuilder = randomBoolean() ? multiMatchQuery("marvel hero captain america", "full_name", "first_name", "last_name", "category") :

329                 SearchResponse left = client().prepareSearch("test").setSize(numDocs)
330                         .addSort(SortBuilders.scoreSort()).addSort(SortBuilders.fieldSort("_uid"))
331                         .setQuery(randomizeType(multiMatchQueryBuilder

333 
334                 SearchResponse right = client().prepareSearch("test").setSize(numDocs)

337                                 add(matchQuery("full_name", "marvel hero captain america"))
338                                 .add(matchQuery("first_name", "marvel hero captain america"))
339                                 .add(matchQuery("last_name", "marvel hero captain america"))
340                                 .add(matchQuery("category", "marvel hero captain america"))

345             {

522     private static final void assertEquivalent(String query, SearchResponse left, SearchResponse right) {
523         assertNoFailures(left);
524         assertNoFailures(right);
525         SearchHits leftHits = left.getHits();
526         SearchHits rightHits = right.getHits();
527         assertThat(leftHits.getTotalHits(), equalTo(rightHits.getTotalHits()));
528         assertThat(leftHits.getHits().length, equalTo(rightHits.getHits().length));
529         SearchHit[] hits = leftHits.getHits();
530         SearchHit[] rHits = rightHits.getHits();
531         for (int i = 0; i < hits.length; i++) {
532             assertThat("query: " + query + " hit: " + i, (double) hits[i].getScore(), closeTo(rHits[i].getScore(), 0.00001d));
533         }
534         for (int i = 0; i < hits.length; i++) {
535             if (hits[i].getScore() == hits[hits.length - 1].getScore()) {
536                 return; // we need to cut off here since this is the tail of the queue and we might not have fetched enough docs
537             }
538             assertThat("query: " + query, hits[i].getId(), equalTo(rHits[i].getId()));
539         }
540     }

567     public MultiMatchQueryBuilder randomizeType(MultiMatchQueryBuilder builder) {
568         try {
569             MultiMatchQueryBuilder.Type type = getType(builder);
570             if (type == null && randomBoolean()) {
571                 return builder;
572             }
573             if (type == null) {
574                 type = MultiMatchQueryBuilder.Type.BEST_FIELDS;
575             }
576             if (randomBoolean()) {
577                 builder.type(type);
578             } else {
579                 Object oType = type;
580                 switch (type) {
581                     case BEST_FIELDS:
582                         if (randomBoolean()) {
583                             oType = MatchQueryBuilder.Type.BOOLEAN;
584                         }
585                         break;
586                     case MOST_FIELDS:
587                         if (randomBoolean()) {
588                             oType = MatchQueryBuilder.Type.BOOLEAN;
589                         }
590                         break;
591                     case CROSS_FIELDS:
592                         break;
593                     case PHRASE:
594                         if (randomBoolean()) {
595                             oType = MatchQueryBuilder.Type.PHRASE;
596                         }
597                         break;
598                     case PHRASE_PREFIX:
599                         if (randomBoolean()) {
600                             oType = MatchQueryBuilder.Type.PHRASE_PREFIX;
601                         }
602                         break;
603                 }
604                 builder.type(oType);
605             }
606             return builder;
607         } catch (Exception ex) {
608             throw new RuntimeException(ex);
609         }
610     }

612     private MultiMatchQueryBuilder.Type getType(MultiMatchQueryBuilder builder) throws NoSuchFieldException, IllegalAccessException {
613         Field field = MultiMatchQueryBuilder.class.getDeclaredField("type");
614         field.setAccessible(true);
615         return (MultiMatchQueryBuilder.Type) field.get(builder);
