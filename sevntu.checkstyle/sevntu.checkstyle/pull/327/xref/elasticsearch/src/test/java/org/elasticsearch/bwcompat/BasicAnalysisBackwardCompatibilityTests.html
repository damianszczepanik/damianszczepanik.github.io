47      /**
48       * Simple upgrade test for analyzers to make sure they analyze to the same tokens after upgrade
49       * TODO we need this for random tokenizers / tokenfilters as well
50       */
51      @Test
52      public void testAnalyzerTokensAfterUpgrade() throws IOException, ExecutionException, InterruptedException {
53          int numFields = randomIntBetween(PreBuiltAnalyzers.values().length, PreBuiltAnalyzers.values().length * 10);
54          StringBuilder builder = new StringBuilder();
55          String[] fields = new String[numFields * 2];
56          int fieldId = 0;
57          for (int i = 0; i < fields.length; i++) {
58              fields[i++] = "field_" + fieldId++;
59              String analyzer = randomAnalyzer();
60              fields[i] = "type=string,analyzer=" + analyzer;
61          }
62          assertAcked(prepareCreate("test")
63                  .addMapping("type", fields)
64                  .setSettings(indexSettings()));
65          ensureYellow();
66          InputOutput[] inout = new InputOutput[numFields];
67          for (int i = 0; i < numFields; i++) {
68              String input;
69              Matcher matcher;
70              do {
71                  // In Lucene 4.10, a bug was fixed in StandardTokenizer which was causing breaks on complex characters.
72                  // The bug was fixed without backcompat Version handling, so testing between >=4.10 vs <= 4.9 can
73                  // cause differences when the random string generated contains these complex characters. To mitigate
74                  // the problem, we skip any strings containing these characters.
75                  // TODO: only skip strings containing complex chars when comparing against ES <= 1.3.x
76                  input = TestUtil.randomAnalysisString(getRandom(), 100, false);
77                  matcher = complexUnicodeChars.matcher(input);
78              } while (matcher.find());
79  
80              AnalyzeResponse test = client().admin().indices().prepareAnalyze("test", input).setField("field_" + i).get();
81              inout[i] = new InputOutput(test, input, "field_" + i);
82          }
83  
84          logClusterState();
85          boolean upgraded;
86          do {
87              logClusterState();
88              upgraded = backwardsCluster().upgradeOneNode();
89              ensureYellow();
90          } while (upgraded);
91  
92          for (int i = 0; i < inout.length; i++) {
93              InputOutput inputOutput = inout[i];
94              AnalyzeResponse test = client().admin().indices().prepareAnalyze("test", inputOutput.input).setField(inputOutput.field).get();
95              List<AnalyzeResponse.AnalyzeToken> tokens = test.getTokens();
96              List<AnalyzeResponse.AnalyzeToken>  expectedTokens = inputOutput.response.getTokens();
97              assertThat("size mismatch field: " + fields[i*2] + " analyzer: " + fields[i*2 + 1] + " input: " + BaseTokenStreamTestCase.escape(inputOutput.input), expectedTokens.size(), equalTo(tokens.size()));
98              for (int j = 0; j < tokens.size(); j++) {
99                  String msg = "failed for term: " + expectedTokens.get(j).getTerm() + " field: " + fields[i*2] + " analyzer: " + fields[i*2 + 1] + " input: " + BaseTokenStreamTestCase.escape(inputOutput.input);
100                 assertThat(msg,  BaseTokenStreamTestCase.escape(expectedTokens.get(j).getTerm()), equalTo(BaseTokenStreamTestCase.escape(tokens.get(j).getTerm())));
101                 assertThat(msg, expectedTokens.get(j).getPosition(), equalTo(tokens.get(j).getPosition()));
102                 assertThat(msg, expectedTokens.get(j).getStartOffset(), equalTo(tokens.get(j).getStartOffset()));
103                 assertThat(msg, expectedTokens.get(j).getEndOffset(), equalTo(tokens.get(j).getEndOffset()));
104                 assertThat(msg, expectedTokens.get(j).getType(), equalTo(tokens.get(j).getType()));
105             }
106         }
107     }

109     private String randomAnalyzer() {
110         while(true) {
111             PreBuiltAnalyzers preBuiltAnalyzers = RandomPicks.randomFrom(getRandom(), PreBuiltAnalyzers.values());
112             if (preBuiltAnalyzers == PreBuiltAnalyzers.SORANI && compatibilityVersion().before(Version.V_1_3_0)) {
113                 continue; // SORANI was added in 1.3.0
114             }
115             return preBuiltAnalyzers.name().toLowerCase(Locale.ROOT);
116         }
