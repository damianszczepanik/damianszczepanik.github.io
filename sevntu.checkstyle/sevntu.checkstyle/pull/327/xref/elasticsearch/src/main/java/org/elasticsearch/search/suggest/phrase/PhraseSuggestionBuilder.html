55      /**
56       * Sets the gram size for the n-gram model used for this suggester. The
57       * default value is <tt>1</tt> corresponding to <tt>unigrams</tt>. Use
58       * <tt>2</tt> for <tt>bigrams</tt> and <tt>3</tt> for <tt>trigrams</tt>.
59       */
60      public PhraseSuggestionBuilder gramSize(int gramSize) {
61          if (gramSize < 1) {
62              throw new ElasticsearchIllegalArgumentException("gramSize must be >= 1");
63          }
64          this.gramSize = gramSize;
65          return this;
66      }

68      /**
69       * Sets the maximum percentage of the terms that at most considered to be
70       * misspellings in order to form a correction. This method accepts a float
71       * value in the range [0..1) as a fraction of the actual query terms a
72       * number <tt>&gt;=1</tt> as an absolut number of query terms.
73       * 
74       * The default is set to <tt>1.0</tt> which corresponds to that only
75       * corrections with at most 1 missspelled term are returned.
76       */
77      public PhraseSuggestionBuilder maxErrors(Float maxErrors) {
78          this.maxErrors = maxErrors;
79          return this;
80      }

88          return this;

101     /**
102      * Sets the confidence level for this suggester. The confidence level
103      * defines a factor applied to the input phrases score which is used as a
104      * threshold for other suggest candidates. Only candidates that score higher
105      * than the threshold will be included in the result. For instance a
106      * confidence level of <tt>1.0</tt> will only return suggestions that score
107      * higher than the input phrase. If set to <tt>0.0</tt> the top N candidates
108      * are returned. The default is <tt>1.0</tt>
109      */
110     public PhraseSuggestionBuilder confidence(Float confidence) {
111         this.confidence = confidence;
112         return this;
113     }

115     /**
116      * Adds a {@link CandidateGenerator} to this suggester. The
117      * {@link CandidateGenerator} is used to draw candidates for each individual
118      * phrase term before the candidates are scored.
119      */
120     public PhraseSuggestionBuilder addCandidateGenerator(CandidateGenerator generator) {
121         List<CandidateGenerator> list = this.generators.get(generator.getType());
122         if (list == null) {
123             list = new ArrayList<>();
124             this.generators.put(generator.getType(), list);
125         }
126         list.add(generator);
127         return this;
128     }

131      * Clear the candidate generators.
132      */
133     public PhraseSuggestionBuilder clearCandidateGenerators() {
134         this.generators.clear();
135         return this;
136     }

214     @Override
215     public XContentBuilder innerToXContent(XContentBuilder builder, Params params) throws IOException {
216         if (realWordErrorLikelihood != null) {
217             builder.field("real_word_error_likelihood", realWordErrorLikelihood);
218         }
219         if (confidence != null) {
220             builder.field("confidence", confidence);
221         }
222         if (separator != null) {
223             builder.field("separator", separator);
224         }
225         if (maxErrors != null) {
226             builder.field("max_errors", maxErrors);
227         }
228         if (gramSize != null) {
229             builder.field("gram_size", gramSize);
230         }
231         if (forceUnigrams != null) {
232             builder.field("force_unigrams", forceUnigrams);
233         }
234         if (tokenLimit != null) {
235             builder.field("token_limit", tokenLimit);
236         }
237         if (!generators.isEmpty()) {
238             Set<Entry<String, List<CandidateGenerator>>> entrySet = generators.entrySet();
239             for (Entry<String, List<CandidateGenerator>> entry : entrySet) {
240                 builder.startArray(entry.getKey());
241                 for (CandidateGenerator generator : entry.getValue()) {
242                     generator.toXContent(builder, params);
243                 }
244                 builder.endArray();
245             }
246         }
247         if (model != null) {
248             builder.startObject("smoothing");
249             model.toXContent(builder, params);
250             builder.endObject();
251         }
252         if (preTag != null) {
253             builder.startObject("highlight");
254             builder.field("pre_tag", preTag);
255             builder.field("post_tag", postTag);
256             builder.endObject();
257         }
258         if (collateQuery != null || collateFilter != null) {
259             builder.startObject("collate");
260             if (collateQuery != null) {
261                 builder.field("query", collateQuery);
262             }
263             if (collateFilter != null) {
264                 builder.field("filter", collateFilter);
265             }
266             if (collatePreference != null) {
267                 builder.field("preference", collatePreference);
268             }
269             if (collateParams != null) {
270                 builder.field("params", collateParams);
271             }
272             if (collatePrune != null) {
273                 builder.field("prune", collatePrune.booleanValue());
274             }
275             builder.endObject();
276         }
277         return builder;
278     }

290     /**
291      * A "stupid-backoff" smoothing model simialr to <a
292      * href="http://en.wikipedia.org/wiki/Katz's_back-off_model"> Katz's
293      * Backoff</a>. This model is used as the default if no model is configured.
294      * <p>
295      * See <a
296      * href="http://en.wikipedia.org/wiki/N-gram#Smoothing_techniques">N-Gram
297      * Smoothing</a> for details.
298      * </p>
299      */
300     public static final class StupidBackoff extends SmoothingModel {
301         private final double discount;
302 
303         /**
304          * Creates a Stupid-Backoff smoothing model.
305          * 
306          * @param discount
307          *            the discount given to lower order ngrams if the higher order ngram doesn't exits
308          */
309         public StupidBackoff(double discount) {
310             super("stupid_backoff");
311             this.discount = discount;
312         }
313 
314         @Override
315         protected XContentBuilder innerToXContent(XContentBuilder builder, Params params) throws IOException {
316             builder.field("discount", discount);
317             return builder;
318         }
319     }

351     public static abstract class SmoothingModel implements ToXContent {
352         private final String type;
353 
354         protected SmoothingModel(String type) {
355             this.type = type;
356         }
357 
358         @Override
359         public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
360             builder.startObject(type);
361             innerToXContent(builder,params);
362             builder.endObject();
363             return builder;
364         }
365         
366         protected abstract XContentBuilder innerToXContent(XContentBuilder builder, Params params) throws IOException;
367     }

369     /**
370      * Linear interpolation smoothing model.
371      * <p>
372      * See <a
373      * href="http://en.wikipedia.org/wiki/N-gram#Smoothing_techniques">N-Gram
374      * Smoothing</a> for details.
375      * </p>
376      */
377     public static final class LinearInterpolation extends SmoothingModel {
378         private final double trigramLambda;
379         private final double bigramLambda;
380         private final double unigramLambda;
381 
382         /**
383          * Creates a linear interpolation smoothing model.
384          * 
385          * Note: the lambdas must sum up to one.
386          * 
387          * @param trigramLambda
388          *            the trigram lambda
389          * @param bigramLambda
390          *            the bigram lambda
391          * @param unigramLambda
392          *            the unigram lambda
393          */
394         public LinearInterpolation(double trigramLambda, double bigramLambda, double unigramLambda) {
395             super("linear");
396             this.trigramLambda = trigramLambda;
397             this.bigramLambda = bigramLambda;
398             this.unigramLambda = unigramLambda;
399         }
400 
401         @Override
402         protected XContentBuilder innerToXContent(XContentBuilder builder, Params params) throws IOException {
403             builder.field("trigram_lambda", trigramLambda);
404             builder.field("bigram_lambda", bigramLambda);
405             builder.field("unigram_lambda", unigramLambda);
406             return builder;
407         }
408     }

430     public static final class DirectCandidateGenerator extends CandidateGenerator {
431         private final String field;
432         private String preFilter;
433         private String postFilter;
434         private String suggestMode;
435         private Float accuracy;
436         private Integer size;
437         private String sort;
438         private String stringDistance;
439         private Integer maxEdits;
440         private Integer maxInspections;
441         private Float maxTermFreq;
442         private Integer prefixLength;
443         private Integer minWordLength;
444         private Float minDocFreq;
445 
446         /**
447          * Sets from what field to fetch the candidate suggestions from. This is
448          * an required option and needs to be set via this setter or
449          * {@link org.elasticsearch.search.suggest.SuggestBuilder.TermSuggestionBuilder#setField(String)}
450          * method
451          */
452         public DirectCandidateGenerator(String field) {
453             super("direct_generator");
454             this.field = field;
455         }
456 
457         /**
458          * The global suggest mode controls what suggested terms are included or
459          * controls for what suggest text tokens, terms should be suggested for.
460          * Three possible values can be specified:
461          * <ol>
462          * <li><code>missing</code> - Only suggest terms in the suggest text
463          * that aren't in the index. This is the default.
464          * <li><code>popular</code> - Only suggest terms that occur in more docs
465          * then the original suggest text term.
466          * <li><code>always</code> - Suggest any matching suggest terms based on
467          * tokens in the suggest text.
468          * </ol>
469          */
470         public DirectCandidateGenerator suggestMode(String suggestMode) {
471             this.suggestMode = suggestMode;
472             return this;
473         }
474 
475         /**
476          * Sets how similar the suggested terms at least need to be compared to
477          * the original suggest text tokens. A value between 0 and 1 can be
478          * specified. This value will be compared to the string distance result
479          * of each candidate spelling correction.
480          * <p/>
481          * Default is <tt>0.5</tt>
482          */
483         public DirectCandidateGenerator accuracy(float accuracy) {
484             this.accuracy = accuracy;
485             return this;
486         }
487 
488         /**
489          * Sets the maximum suggestions to be returned per suggest text term.
490          */
491         public DirectCandidateGenerator size(int size) {
492             if (size <= 0) {
493                 throw new ElasticsearchIllegalArgumentException("Size must be positive");
494             }
495             this.size = size;
496             return this;
497         }
498 
499         /**
500          * Sets how to sort the suggest terms per suggest text token. Two
501          * possible values:
502          * <ol>
503          * <li><code>score</code> - Sort should first be based on score, then
504          * document frequency and then the term itself.
505          * <li><code>frequency</code> - Sort should first be based on document
506          * frequency, then scotr and then the term itself.
507          * </ol>
508          * <p/>
509          * What the score is depends on the suggester being used.
510          */
511         public DirectCandidateGenerator sort(String sort) {
512             this.sort = sort;
513             return this;
514         }
515 
516         /**
517          * Sets what string distance implementation to use for comparing how
518          * similar suggested terms are. Four possible values can be specified:
519          * <ol>
520          * <li><code>internal</code> - This is the default and is based on
521          * <code>damerau_levenshtein</code>, but highly optimized for comparing
522          * string distance for terms inside the index.
523          * <li><code>damerau_levenshtein</code> - String distance algorithm
524          * based on Damerau-Levenshtein algorithm.
525          * <li><code>levenstein</code> - String distance algorithm based on
526          * Levenstein edit distance algorithm.
527          * <li><code>jarowinkler</code> - String distance algorithm based on
528          * Jaro-Winkler algorithm.
529          * <li><code>ngram</code> - String distance algorithm based on character
530          * n-grams.
531          * </ol>
532          */
533         public DirectCandidateGenerator stringDistance(String stringDistance) {
534             this.stringDistance = stringDistance;
535             return this;
536         }
537 
538         /**
539          * Sets the maximum edit distance candidate suggestions can have in
540          * order to be considered as a suggestion. Can only be a value between 1
541          * and 2. Any other value result in an bad request error being thrown.
542          * Defaults to <tt>2</tt>.
543          */
544         public DirectCandidateGenerator maxEdits(Integer maxEdits) {
545             this.maxEdits = maxEdits;
546             return this;
547         }
548 
549         /**
550          * A factor that is used to multiply with the size in order to inspect
551          * more candidate suggestions. Can improve accuracy at the cost of
552          * performance. Defaults to <tt>5</tt>.
553          */
554         public DirectCandidateGenerator maxInspections(Integer maxInspections) {
555             this.maxInspections = maxInspections;
556             return this;
557         }
558 
559         /**
560          * Sets a maximum threshold in number of documents a suggest text token
561          * can exist in order to be corrected. Can be a relative percentage
562          * number (e.g 0.4) or an absolute number to represent document
563          * frequencies. If an value higher than 1 is specified then fractional
564          * can not be specified. Defaults to <tt>0.01</tt>.
565          * <p/>
566          * This can be used to exclude high frequency terms from being
567          * suggested. High frequency terms are usually spelled correctly on top
568          * of this this also improves the suggest performance.
569          */
570         public DirectCandidateGenerator maxTermFreq(float maxTermFreq) {
571             this.maxTermFreq = maxTermFreq;
572             return this;
573         }
574 
575         /**
576          * Sets the number of minimal prefix characters that must match in order
577          * be a candidate suggestion. Defaults to 1. Increasing this number
578          * improves suggest performance. Usually misspellings don't occur in the
579          * beginning of terms.
580          */
581         public DirectCandidateGenerator prefixLength(int prefixLength) {
582             this.prefixLength = prefixLength;
583             return this;
584         }
585 
586         /**
587          * The minimum length a suggest text term must have in order to be
588          * corrected. Defaults to <tt>4</tt>.
589          */
590         public DirectCandidateGenerator minWordLength(int minWordLength) {
591             this.minWordLength = minWordLength;
592             return this;
593         }
594 
595         /**
596          * Sets a minimal threshold in number of documents a suggested term
597          * should appear in. This can be specified as an absolute number or as a
598          * relative percentage of number of documents. This can improve quality
599          * by only suggesting high frequency terms. Defaults to 0f and is not
600          * enabled. If a value higher than 1 is specified then the number cannot
601          * be fractional.
602          */
603         public DirectCandidateGenerator minDocFreq(float minDocFreq) {
604             this.minDocFreq = minDocFreq;
605             return this;
606         }
607 
608         /**
609          * Sets a filter (analyzer) that is applied to each of the tokens passed to this candidate generator.
610          * This filter is applied to the original token before candidates are generated.
611          */
612         public DirectCandidateGenerator preFilter(String preFilter) {
613             this.preFilter = preFilter;
614             return this;
615         }
616 
617         /**
618          * Sets a filter (analyzer) that is applied to each of the generated tokens
619          * before they are passed to the actual phrase scorer.
620          */
621         public DirectCandidateGenerator postFilter(String postFilter) {
622             this.postFilter = postFilter;
623             return this;
624         }
625         
626         @Override
627         public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
628             builder.startObject();
629             if (field != null) {
630                 builder.field("field", field);
631             }
632             if (suggestMode != null) {
633                 builder.field("suggest_mode", suggestMode);
634             }
635             if (accuracy != null) {
636                 builder.field("accuracy", accuracy);
637             }
638             if (size != null) {
639                 builder.field("size", size);
640             }
641             if (sort != null) {
642                 builder.field("sort", sort);
643             }
644             if (stringDistance != null) {
645                 builder.field("string_distance", stringDistance);
646             }
647             if (maxEdits != null) {
648                 builder.field("max_edits", maxEdits);
649             }
650             if (maxInspections != null) {
651                 builder.field("max_inspections", maxInspections);
652             }
653             if (maxTermFreq != null) {
654                 builder.field("max_term_freq", maxTermFreq);
655             }
656             if (prefixLength != null) {
657                 builder.field("prefix_length", prefixLength);
658             }
659             if (minWordLength != null) {
660                 builder.field("min_word_length", minWordLength);
661             }
662             if (minDocFreq != null) {
663                 builder.field("min_doc_freq", minDocFreq);
664             }
665             if (preFilter != null) {
666                 builder.field("pre_filter", preFilter);
667             }
668             if (postFilter != null) {
669                 builder.field("post_filter", postFilter);
670             }
671             builder.endObject();
672             return builder;
673         }
