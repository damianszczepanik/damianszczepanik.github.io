20  package org.elasticsearch.action.admin.indices.settings.get;
21  
22  import org.elasticsearch.ElasticsearchException;
23  import org.elasticsearch.action.ActionListener;
24  import org.elasticsearch.action.admin.cluster.state.ClusterStateRequest;
25  import org.elasticsearch.action.support.ActionFilters;
26  import org.elasticsearch.action.support.master.TransportMasterNodeReadOperationAction;
27  import org.elasticsearch.cluster.ClusterService;
28  import org.elasticsearch.cluster.ClusterState;
29  import org.elasticsearch.cluster.block.ClusterBlockException;
30  import org.elasticsearch.cluster.block.ClusterBlockLevel;
31  import org.elasticsearch.cluster.metadata.IndexMetaData;
32  import org.elasticsearch.common.collect.ImmutableOpenMap;
33  import org.elasticsearch.common.inject.Inject;
34  import org.elasticsearch.common.regex.Regex;
35  import org.elasticsearch.common.settings.ImmutableSettings;
36  import org.elasticsearch.common.settings.Settings;
37  import org.elasticsearch.common.settings.SettingsFilter;
38  import org.elasticsearch.common.util.CollectionUtils;
39  import org.elasticsearch.threadpool.ThreadPool;
40  import org.elasticsearch.transport.TransportService;
41  
42  import java.util.Map;
43  
44  /**
45   */
46  public class TransportGetSettingsAction extends TransportMasterNodeReadOperationAction<GetSettingsRequest, GetSettingsResponse> {
47  
48      private final SettingsFilter settingsFilter;
49  
50      @Inject
51      public TransportGetSettingsAction(Settings settings, TransportService transportService, ClusterService clusterService,
52                                        ThreadPool threadPool, SettingsFilter settingsFilter, ActionFilters actionFilters) {
53          super(settings, GetSettingsAction.NAME, transportService, clusterService, threadPool, actionFilters);
54          this.settingsFilter = settingsFilter;
55      }
56  
57      @Override
58      protected String executor() {
59          // Very lightweight operation
60          return ThreadPool.Names.SAME;
61      }
62  
63      @Override
64      protected ClusterBlockException checkBlock(GetSettingsRequest request, ClusterState state) {
65          return state.blocks().indicesBlockedException(ClusterBlockLevel.METADATA, state.metaData().concreteIndices(request.indicesOptions(), request.indices()));
66      }
67  
68  
69      @Override
70      protected GetSettingsRequest newRequest() {
71          return new GetSettingsRequest();
72      }
73  
74      @Override
75      protected GetSettingsResponse newResponse() {
76          return new GetSettingsResponse();
77      }
78  
79      @Override
80      protected void masterOperation(GetSettingsRequest request, ClusterState state, ActionListener<GetSettingsResponse> listener) throws ElasticsearchException {
81          String[] concreteIndices = state.metaData().concreteIndices(request.indicesOptions(), request.indices());
82          ImmutableOpenMap.Builder<String, Settings> indexToSettingsBuilder = ImmutableOpenMap.builder();
83          for (String concreteIndex : concreteIndices) {
84              IndexMetaData indexMetaData = state.getMetaData().index(concreteIndex);
85              if (indexMetaData == null) {
86                  continue;
87              }
88  
89              Settings settings = SettingsFilter.filterSettings(settingsFilter.getPatterns(), indexMetaData.settings());
90              if (!CollectionUtils.isEmpty(request.names())) {
91                  ImmutableSettings.Builder settingsBuilder = ImmutableSettings.builder();
92                  for (Map.Entry<String, String> entry : settings.getAsMap().entrySet()) {
93                      if (Regex.simpleMatch(request.names(), entry.getKey())) {
94                          settingsBuilder.put(entry.getKey(), entry.getValue());
95                      }
96                  }
97                  settings = settingsBuilder.build();
98              }
99              indexToSettingsBuilder.put(concreteIndex, settings);
100         }
101         listener.onResponse(new GetSettingsResponse(indexToSettingsBuilder.build()));
102     }
103 }
