197     public FilterCachingPolicy parseFilterCachePolicy() throws IOException {
198         final String text = parser.textOrNull();
199         if (text == null || text.equals("auto")) {
200             return autoFilterCachePolicy();
201         } else if (parser.booleanValue()) {
202             // cache without conditions on how many times the filter has been
203             // used or what the produced DocIdSet looks like, but ONLY on large
204             // segments to not pollute the cache
205             return FilterCachingPolicy.CacheOnLargeSegments.DEFAULT;
206         } else {
207             return null;
208         }
209     }

224     public Filter cacheFilter(Filter filter, final @Nullable HashedBytesRef cacheKey, final FilterCachingPolicy cachePolicy) {
225         if (filter == null) {
226             return null;
227         }
228         if (this.disableFilterCaching || this.propagateNoCache || filter instanceof NoCacheFilter) {
229             return filter;
230         }
231         if (filter instanceof ResolvableFilter) {
232             final ResolvableFilter resolvableFilter = (ResolvableFilter) filter;
233             // We need to wrap it another filter, because this method is invoked at query parse time, which
234             // may not be during search execution time. (for example index alias filter and percolator)
235             return new Filter() {
236                 @Override
237                 public DocIdSet getDocIdSet(LeafReaderContext atomicReaderContext, Bits bits) throws IOException {
238                     Filter filter = resolvableFilter.resolve();
239                     if (filter == null) {
240                         return null;
241                     }
242                     filter = indexQueryParser.indexCache.filter().cache(filter, cacheKey, cachePolicy);
243                     return filter.getDocIdSet(atomicReaderContext, bits);
244                 }
245 
246                 @Override
247                 public String toString(String field) {
248                     return "AnonymousResolvableFilter"; // TODO: not sure what is going on here
249                 }
250             };
251         } else {
252             return indexQueryParser.indexCache.filter().cache(filter, cacheKey, cachePolicy);
253         }
254     }

288     @Nullable
289     public Query parseInnerQuery() throws IOException, QueryParsingException {
290         // move to START object
291         XContentParser.Token token;
292         if (parser.currentToken() != XContentParser.Token.START_OBJECT) {
293             token = parser.nextToken();
294             if (token != XContentParser.Token.START_OBJECT) {
295                 throw new QueryParsingException(index, "[_na] query malformed, must start with start_object");
296             }
297         }
298         token = parser.nextToken();
299         if (token != XContentParser.Token.FIELD_NAME) {
300             throw new QueryParsingException(index, "[_na] query malformed, no field after start_object");
301         }
302         String queryName = parser.currentName();
303         // move to the next START_OBJECT
304         token = parser.nextToken();
305         if (token != XContentParser.Token.START_OBJECT && token != XContentParser.Token.START_ARRAY) {
306             throw new QueryParsingException(index, "[_na] query malformed, no field after start_object");
307         }
308 
309         QueryParser queryParser = indexQueryParser.queryParser(queryName);
310         if (queryParser == null) {
311             throw new QueryParsingException(index, "No query registered for [" + queryName + "]");
312         }
313         Query result = queryParser.parse(this);
314         if (parser.currentToken() == XContentParser.Token.END_OBJECT || parser.currentToken() == XContentParser.Token.END_ARRAY) {
315             // if we are at END_OBJECT, move to the next one...
316             parser.nextToken();
317         }
318         if (result instanceof NoCacheQuery) {
319             propagateNoCache = true;
320         }
321         if (CustomQueryWrappingFilter.shouldUseCustomQueryWrappingFilter(result)) {
322             requireCustomQueryWrappingFilter = true;
323             // If later on, either directly or indirectly this query gets wrapped in a query filter it must never
324             // get cached even if a filter higher up the chain is configured to do this. This will happen, because
325             // the result filter will be instance of NoCacheFilter (CustomQueryWrappingFilter) which will in
326             // #executeFilterParser() set propagateNoCache to true.
327         }
328         return result;
329     }

331     @Nullable
332     public Filter parseInnerFilter() throws IOException, QueryParsingException {
333         // move to START object
334         XContentParser.Token token;
335         if (parser.currentToken() != XContentParser.Token.START_OBJECT) {
336             token = parser.nextToken();
337             if (token != XContentParser.Token.START_OBJECT) {
338                 throw new QueryParsingException(index, "[_na] filter malformed, must start with start_object");
339             }
340         }
341         token = parser.nextToken();
342         if (token != XContentParser.Token.FIELD_NAME) {
343             // empty filter
344             if (token == XContentParser.Token.END_OBJECT || token == XContentParser.Token.VALUE_NULL) {
345                 return null;
346             }
347             throw new QueryParsingException(index, "[_na] filter malformed, no field after start_object");
348         }
349         String filterName = parser.currentName();
350         // move to the next START_OBJECT or START_ARRAY
351         token = parser.nextToken();
352         if (token != XContentParser.Token.START_OBJECT && token != XContentParser.Token.START_ARRAY) {
353             throw new QueryParsingException(index, "[_na] filter malformed, no field after start_object");
354         }
355 
356         FilterParser filterParser = indexQueryParser.filterParser(filterName);
357         if (filterParser == null) {
358             throw new QueryParsingException(index, "No filter registered for [" + filterName + "]");
359         }
360         Filter result = executeFilterParser(filterParser);
361         if (parser.currentToken() == XContentParser.Token.END_OBJECT || parser.currentToken() == XContentParser.Token.END_ARRAY) {
362             // if we are at END_OBJECT, move to the next one...
363             parser.nextToken();
364         }
365         return result;
366     }

368     public Filter parseInnerFilter(String filterName) throws IOException, QueryParsingException {
369         FilterParser filterParser = indexQueryParser.filterParser(filterName);
370         if (filterParser == null) {
371             throw new QueryParsingException(index, "No filter registered for [" + filterName + "]");
372         }
373         return executeFilterParser(filterParser);
374     }
375 
376     private Filter executeFilterParser(FilterParser filterParser) throws IOException {
377         final boolean propagateNoCache = this.propagateNoCache; // first safe the state that we need to restore
378         this.propagateNoCache = false; // parse the subfilter with caching, that's fine
379         Filter result = filterParser.parse(this);
380         // now make sure we set propagateNoCache to true if it is true already or if the result is
381         // an instance of NoCacheFilter or if we used to be true! all filters above will
382         // be not cached ie. wrappers of this filter!
383         this.propagateNoCache |= (result instanceof NoCacheFilter) || propagateNoCache;
384         return result;
385     }

423     private MapperService.SmartNameFieldMappers failIfFieldMappingNotFound(String name, MapperService.SmartNameFieldMappers fieldMapping) {
424         if (allowUnmappedFields) {
425             return fieldMapping;
426         } else if (mapUnmappedFieldAsString){
427             StringFieldMapper.Builder builder = MapperBuilders.stringField(name);
428             // it would be better to pass the real index settings, but they are not easily accessible from here...
429             Settings settings = ImmutableSettings.builder().put(IndexMetaData.SETTING_VERSION_CREATED, indexQueryParser.getIndexCreatedVersion()).build();
430             StringFieldMapper stringFieldMapper = builder.build(new Mapper.BuilderContext(settings, new ContentPath(1)));
431             return new MapperService.SmartNameFieldMappers(mapperService(), new FieldMappers(stringFieldMapper), null, false);
432         } else {
433             Version indexCreatedVersion = indexQueryParser.getIndexCreatedVersion();
434             if (fieldMapping == null && indexCreatedVersion.onOrAfter(Version.V_1_4_0_Beta1)) {
435                 throw new QueryParsingException(index, "Strict field resolution and no field mapping can be found for the field with name [" + name + "]");
436             } else {
437                 return fieldMapping;
438             }
439         }
440     }

442     /**
443      * Returns the narrowed down explicit types, or, if not set, all types.
444      */
445     public Collection<String> queryTypes() {
446         String[] types = getTypes();
447         if (types == null || types.length == 0) {
448             return mapperService().types();
449         }
450         if (types.length == 1 && types[0].equals("_all")) {
451             return mapperService().types();
452         }
453         return Arrays.asList(types);
454     }

458     public SearchLookup lookup() {
459         SearchContext current = SearchContext.current();
460         if (current != null) {
461             return current.lookup();
462         }
463         if (lookup == null) {
464             lookup = new SearchLookup(mapperService(), indexQueryParser.fieldDataService, null);
465         }
466         return lookup;
467     }
