40  /**
41   * A bulk processor is a thread safe bulk processing class, allowing to easily set when to "flush" a new bulk request
42   * (either based on number of actions, based on the size, or time), and to easily control the number of concurrent bulk
43   * requests allowed to be executed in parallel.
44   * <p/>
45   * In order to create a new bulk processor, use the {@link Builder}.
46   */
47  public class BulkProcessor implements Closeable {
48  
49      /**
50       * A listener for the execution.
51       */
52      public static interface Listener {
53  
54          /**
55           * Callback before the bulk is executed.
56           */
57          void beforeBulk(long executionId, BulkRequest request);
58  
59          /**
60           * Callback after a successful execution of bulk request.
61           */
62          void afterBulk(long executionId, BulkRequest request, BulkResponse response);
63  
64          /**
65           * Callback after a failed execution of bulk request.
66           */
67          void afterBulk(long executionId, BulkRequest request, Throwable failure);
68      }
69  
70      /**
71       * A builder used to create a build an instance of a bulk processor.
72       */
73      public static class Builder {
74  
75          private final Client client;
76          private final Listener listener;
77  
78          private String name;
79          private int concurrentRequests = 1;
80          private int bulkActions = 1000;
81          private ByteSizeValue bulkSize = new ByteSizeValue(5, ByteSizeUnit.MB);
82          private TimeValue flushInterval = null;
83  
84          /**
85           * Creates a builder of bulk processor with the client to use and the listener that will be used
86           * to be notified on the completion of bulk requests.
87           */
88          public Builder(Client client, Listener listener) {
89              this.client = client;
90              this.listener = listener;
91          }
92  
93          /**
94           * Sets an optional name to identify this bulk processor.
95           */
96          public Builder setName(String name) {
97              this.name = name;
98              return this;
99          }
100 
101         /**
102          * Sets the number of concurrent requests allowed to be executed. A value of 0 means that only a single
103          * request will be allowed to be executed. A value of 1 means 1 concurrent request is allowed to be executed
104          * while accumulating new bulk requests. Defaults to <tt>1</tt>.
105          */
106         public Builder setConcurrentRequests(int concurrentRequests) {
107             this.concurrentRequests = concurrentRequests;
108             return this;
109         }
110 
111         /**
112          * Sets when to flush a new bulk request based on the number of actions currently added. Defaults to
113          * <tt>1000</tt>. Can be set to <tt>-1</tt> to disable it.
114          */
115         public Builder setBulkActions(int bulkActions) {
116             this.bulkActions = bulkActions;
117             return this;
118         }
119 
120         /**
121          * Sets when to flush a new bulk request based on the size of actions currently added. Defaults to
122          * <tt>5mb</tt>. Can be set to <tt>-1</tt> to disable it.
123          */
124         public Builder setBulkSize(ByteSizeValue bulkSize) {
125             this.bulkSize = bulkSize;
126             return this;
127         }
128 
129         /**
130          * Sets a flush interval flushing *any* bulk actions pending if the interval passes. Defaults to not set.
131          * <p/>
132          * Note, both {@link #setBulkActions(int)} and {@link #setBulkSize(org.elasticsearch.common.unit.ByteSizeValue)}
133          * can be set to <tt>-1</tt> with the flush interval set allowing for complete async processing of bulk actions.
134          */
135         public Builder setFlushInterval(TimeValue flushInterval) {
136             this.flushInterval = flushInterval;
137             return this;
138         }
139 
140         /**
141          * Builds a new bulk processor.
142          */
143         public BulkProcessor build() {
144             return new BulkProcessor(client, listener, name, concurrentRequests, bulkActions, bulkSize, flushInterval);
145         }
146     }
147 
148     public static Builder builder(Client client, Listener listener) {
149         return new Builder(client, listener);
150     }

152     private final Client client;
153     private final Listener listener;
154 
155     private final String name;
156 
157     private final int concurrentRequests;
158     private final int bulkActions;
159     private final long bulkSize;
160     private final TimeValue flushInterval;
161 
162     private final Semaphore semaphore;
163     private final ScheduledThreadPoolExecutor scheduler;
164     private final ScheduledFuture scheduledFuture;
165 
166     private final AtomicLong executionIdGen = new AtomicLong();
167 
168     private BulkRequest bulkRequest;
169 
170     private volatile boolean closed = false;
171 
172     BulkProcessor(Client client, Listener listener, @Nullable String name, int concurrentRequests, int bulkActions, ByteSizeValue bulkSize, @Nullable TimeValue flushInterval) {
173         this.client = client;
174         this.listener = listener;
175         this.name = name;
176         this.concurrentRequests = concurrentRequests;
177         this.bulkActions = bulkActions;
178         this.bulkSize = bulkSize.bytes();
179 
180         this.semaphore = new Semaphore(concurrentRequests);
181         this.bulkRequest = new BulkRequest();
182 
183         this.flushInterval = flushInterval;
184         if (flushInterval != null) {
185             this.scheduler = (ScheduledThreadPoolExecutor) Executors.newScheduledThreadPool(1, EsExecutors.daemonThreadFactory(client.settings(), (name != null ? "[" + name + "]" : "") + "bulk_processor"));
186             this.scheduler.setExecuteExistingDelayedTasksAfterShutdownPolicy(false);
187             this.scheduler.setContinueExistingPeriodicTasksAfterShutdownPolicy(false);
188             this.scheduledFuture = this.scheduler.scheduleWithFixedDelay(new Flush(), flushInterval.millis(), flushInterval.millis(), TimeUnit.MILLISECONDS);
189         } else {
190             this.scheduler = null;
191             this.scheduledFuture = null;
192         }
193     }
194 
195     /**
196      * Closes the processor. If flushing by time is enabled, then it's shutdown. Any remaining bulk actions are flushed.
197      */
198     @Override
199     public void close() {
200         try {
201             awaitClose(0, TimeUnit.NANOSECONDS);
202         } catch(InterruptedException exc) {
203             Thread.currentThread().interrupt();
204         }
205     }
206 
207     /**
208      * Closes the processor. If flushing by time is enabled, then it's shutdown. Any remaining bulk actions are flushed.
209      *
210      * If concurrent requests are not enabled, returns {@code true} immediately.
211      * If concurrent requests are enabled, waits for up to the specified timeout for all bulk requests to complete then returns {@code true},
212      * If the specified waiting time elapses before all bulk requests complete, {@code false} is returned.
213      *
214      * @param timeout The maximum time to wait for the bulk requests to complete
215      * @param unit The time unit of the {@code timeout} argument
216      * @return {@code true} if all bulk requests completed and {@code false} if the waiting time elapsed before all the bulk requests completed
217      * @throws InterruptedException If the current thread is interrupted
218      */
219     public synchronized boolean awaitClose(long timeout, TimeUnit unit) throws InterruptedException {
220         if (closed) {
221             return true;
222         }
223         closed = true;
224         if (this.scheduledFuture != null) {
225             FutureUtils.cancel(this.scheduledFuture);
226             this.scheduler.shutdown();
227         }
228         if (bulkRequest.numberOfActions() > 0) {
229             execute();
230         }
231         if (this.concurrentRequests < 1) {
232             return true;
233         }
234         if (semaphore.tryAcquire(this.concurrentRequests, timeout, unit)) {
235             semaphore.release(this.concurrentRequests);
236             return true;
237         }
238         return false;
239     }

268     boolean isOpen() {
269         return closed == false;
270     }
271 
272     protected void ensureOpen() {
273         if (closed) {
274             throw new ElasticsearchIllegalStateException("bulk process already closed");
275         }
276     }
277 
278     private synchronized void internalAdd(ActionRequest request, @Nullable Object payload) {
279         ensureOpen();
280         bulkRequest.add(request, payload);
281         executeIfNeeded();
282     }

284     public BulkProcessor add(BytesReference data, boolean contentUnsafe, @Nullable String defaultIndex, @Nullable String defaultType) throws Exception {
285         return add(data, contentUnsafe, defaultIndex, defaultType, null);
286     }
287 
288     public synchronized BulkProcessor add(BytesReference data, boolean contentUnsafe, @Nullable String defaultIndex, @Nullable String defaultType, @Nullable Object payload) throws Exception {
289         bulkRequest.add(data, contentUnsafe, defaultIndex, defaultType, null, payload, true);
290         executeIfNeeded();
291         return this;
292     }
293 
294     private void executeIfNeeded() {
295         ensureOpen();
296         if (!isOverTheLimit()) {
297             return;
298         }
299         execute();
300     }
301 
302     // (currently) needs to be executed under a lock
303     private void execute() {
304         final BulkRequest bulkRequest = this.bulkRequest;
305         final long executionId = executionIdGen.incrementAndGet();
306 
307         this.bulkRequest = new BulkRequest();
308 
309         if (concurrentRequests == 0) {
310             // execute in a blocking fashion...
311             boolean afterCalled = false;
312             try {
313                 listener.beforeBulk(executionId, bulkRequest);
314                 BulkResponse bulkItemResponses = client.bulk(bulkRequest).actionGet();
315                 afterCalled = true;
316                 listener.afterBulk(executionId, bulkRequest, bulkItemResponses);
317             } catch (Exception e) {
318                 if (!afterCalled) {
319                     listener.afterBulk(executionId, bulkRequest, e);
320                 }
321             }
322         } else {
323             boolean success = false;
324             try {
325                 listener.beforeBulk(executionId, bulkRequest);
326                 semaphore.acquire();
327                 client.bulk(bulkRequest, new ActionListener<BulkResponse>() {
328                     @Override
329                     public void onResponse(BulkResponse response) {
330                         try {
331                             listener.afterBulk(executionId, bulkRequest, response);
332                         } finally {
333                             semaphore.release();
334                         }
335                     }
336 
337                     @Override
338                     public void onFailure(Throwable e) {
339                         try {
340                             listener.afterBulk(executionId, bulkRequest, e);
341                         } finally {
342                             semaphore.release();
343                         }
344                     }
345                 });
346                 success = true;
347             } catch (InterruptedException e) {
348                 Thread.interrupted();
349                 listener.afterBulk(executionId, bulkRequest, e);
350             } catch (Throwable t) {
351                 listener.afterBulk(executionId, bulkRequest, t);
352             } finally {
353                  if (!success) {  // if we fail on client.bulk() release the semaphore
354                      semaphore.release();
355                  }
356             }
357         }
358     }
359 
360     private boolean isOverTheLimit() {
361         if (bulkActions != -1 && bulkRequest.numberOfActions() >= bulkActions) {
362             return true;
363         }
364         if (bulkSize != -1 && bulkRequest.estimatedSizeInBytes() >= bulkSize) {
365             return true;
366         }
367         return false;
368     }
369 
370     /**
371      * Flush pending delete or index requests.
372      */
373     public synchronized void flush() {
374         ensureOpen();
375         if (bulkRequest.numberOfActions() > 0) {
376             execute();
377         }
378     }
379 
380     class Flush implements Runnable {
381 
382         @Override
383         public void run() {
384             synchronized (BulkProcessor.this) {
385                 if (closed) {
386                     return;
387                 }
388                 if (bulkRequest.numberOfActions() == 0) {
389                     return;
390                 }
391                 execute();
392             }
393         }
394     }
