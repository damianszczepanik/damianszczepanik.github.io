63      @Override
64      public void doRequest(final RestRequest request, final RestChannel channel, final Client client) {
65          final String[] indices = Strings.splitStringByCommaToArray(request.param("index"));
66          final ClusterStateRequest clusterStateRequest = new ClusterStateRequest();
67          clusterStateRequest.clear().indices(indices).metaData(true);
68          clusterStateRequest.local(request.paramAsBoolean("local", clusterStateRequest.local()));
69          clusterStateRequest.masterNodeTimeout(request.paramAsTime("master_timeout", clusterStateRequest.masterNodeTimeout()));
70  
71          client.admin().cluster().state(clusterStateRequest, new RestActionListener<ClusterStateResponse>(channel) {
72              @Override
73              public void processResponse(final ClusterStateResponse clusterStateResponse) {
74                  final String[] concreteIndices = clusterStateResponse.getState().metaData().concreteIndices(IndicesOptions.fromOptions(false, true, true, true), indices);
75                  final String[] openIndices = clusterStateResponse.getState().metaData().concreteIndices(IndicesOptions.lenientExpandOpen(), indices);
76                  ClusterHealthRequest clusterHealthRequest = Requests.clusterHealthRequest(openIndices);
77                  clusterHealthRequest.local(request.paramAsBoolean("local", clusterHealthRequest.local()));
78                  client.admin().cluster().health(clusterHealthRequest, new RestActionListener<ClusterHealthResponse>(channel) {
79                      @Override
80                      public void processResponse(final ClusterHealthResponse clusterHealthResponse) {
81                          IndicesStatsRequest indicesStatsRequest = new IndicesStatsRequest();
82                          indicesStatsRequest.all();
83                          client.admin().indices().stats(indicesStatsRequest, new RestResponseListener<IndicesStatsResponse>(channel) {
84                              @Override
85                              public RestResponse buildResponse(IndicesStatsResponse indicesStatsResponse) throws Exception {
86                                  Table tab = buildTable(request, concreteIndices, clusterHealthResponse, indicesStatsResponse, clusterStateResponse.getState().metaData());
87                                  return RestTable.buildResponse(tab, channel);
88                              }
89                          });
90  
91                      }
92                  });
93              }
94          });
95      }

293     private Table buildTable(RestRequest request, String[] indices, ClusterHealthResponse health, IndicesStatsResponse stats, MetaData indexMetaDatas) {
294         Table table = getTableWithHeader(request);
295 
296         for (String index : indices) {
297             ClusterIndexHealth indexHealth = health.getIndices().get(index);
298             IndexStats indexStats = stats.getIndices().get(index);
299             IndexMetaData indexMetaData = indexMetaDatas.getIndices().get(index);
300             IndexMetaData.State state = indexMetaData.getState();
301 
302             table.startRow();
303             table.addCell(state == IndexMetaData.State.OPEN ? (indexHealth == null ? "red*" : indexHealth.getStatus().toString().toLowerCase(Locale.ROOT)) : null);
304             table.addCell(state.toString().toLowerCase(Locale.ROOT));
305             table.addCell(index);
306             table.addCell(indexHealth == null ? null : indexHealth.getNumberOfShards());
307             table.addCell(indexHealth == null ? null : indexHealth.getNumberOfReplicas());
308             table.addCell(indexStats == null ? null : indexStats.getPrimaries().getDocs().getCount());
309             table.addCell(indexStats == null ? null : indexStats.getPrimaries().getDocs().getDeleted());
310 
311             table.addCell(indexStats == null ? null : indexStats.getTotal().getStore().size());
312             table.addCell(indexStats == null ? null : indexStats.getPrimaries().getStore().size());
