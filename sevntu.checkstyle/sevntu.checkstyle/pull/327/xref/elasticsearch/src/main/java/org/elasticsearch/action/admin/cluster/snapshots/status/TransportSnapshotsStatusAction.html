51  /**
52   */
53  public class TransportSnapshotsStatusAction extends TransportMasterNodeOperationAction<SnapshotsStatusRequest, SnapshotsStatusResponse> {
54  
55      private final SnapshotsService snapshotsService;
56  
57      private final TransportNodesSnapshotsStatus transportNodesSnapshotsStatus;
58  
59      @Inject
60      public TransportSnapshotsStatusAction(Settings settings, TransportService transportService, ClusterService clusterService, ThreadPool threadPool, SnapshotsService snapshotsService, TransportNodesSnapshotsStatus transportNodesSnapshotsStatus, ActionFilters actionFilters) {
61          super(settings, SnapshotsStatusAction.NAME, transportService, clusterService, threadPool, actionFilters);
62          this.snapshotsService = snapshotsService;
63          this.transportNodesSnapshotsStatus = transportNodesSnapshotsStatus;
64      }
65  
66      @Override
67      protected String executor() {
68          return ThreadPool.Names.GENERIC;
69      }
70  
71      @Override
72      protected ClusterBlockException checkBlock(SnapshotsStatusRequest request, ClusterState state) {
73          return state.blocks().globalBlockedException(ClusterBlockLevel.METADATA);
74      }
75  
76      @Override
77      protected SnapshotsStatusRequest newRequest() {
78          return new SnapshotsStatusRequest();
79      }
80  
81      @Override
82      protected SnapshotsStatusResponse newResponse() {
83          return new SnapshotsStatusResponse();
84      }
85  
86      @Override
87      protected void masterOperation(final SnapshotsStatusRequest request,
88                                     final ClusterState state,
89                                     final ActionListener<SnapshotsStatusResponse> listener) throws Exception {
90          ImmutableList<SnapshotMetaData.Entry> currentSnapshots = snapshotsService.currentSnapshots(request.repository(), request.snapshots());
91  
92          if (currentSnapshots.isEmpty()) {
93              listener.onResponse(buildResponse(request, currentSnapshots, null));
94              return;
95          }
96  
97          Set<String> nodesIds = newHashSet();
98          for (SnapshotMetaData.Entry entry : currentSnapshots) {
99              for (SnapshotMetaData.ShardSnapshotStatus status : entry.shards().values()) {
100                 if (status.nodeId() != null) {
101                     nodesIds.add(status.nodeId());
102                 }
103             }
104         }
105 
106         if (!nodesIds.isEmpty()) {
107             // There are still some snapshots running - check their progress
108             SnapshotId[] snapshotIds = new SnapshotId[currentSnapshots.size()];
109             for (int i = 0; i < currentSnapshots.size(); i++) {
110                 snapshotIds[i] = currentSnapshots.get(i).snapshotId();
111             }
112 
113             TransportNodesSnapshotsStatus.Request nodesRequest = new TransportNodesSnapshotsStatus.Request(request, nodesIds.toArray(new String[nodesIds.size()]))
114                     .snapshotIds(snapshotIds).timeout(request.masterNodeTimeout());
115             transportNodesSnapshotsStatus.execute(nodesRequest, new ActionListener<TransportNodesSnapshotsStatus.NodesSnapshotStatus>() {
116                         @Override
117                         public void onResponse(TransportNodesSnapshotsStatus.NodesSnapshotStatus nodeSnapshotStatuses) {
118                             try {
119                                 ImmutableList<SnapshotMetaData.Entry> currentSnapshots =
120                                         snapshotsService.currentSnapshots(request.repository(), request.snapshots());
121                                 listener.onResponse(buildResponse(request, currentSnapshots, nodeSnapshotStatuses));
122                             } catch (Throwable e) {
123                                 listener.onFailure(e);
124                             }
125                         }
126 
127                         @Override
128                         public void onFailure(Throwable e) {
129                             listener.onFailure(e);
130                         }
131                     });
132         } else {
133             // We don't have any in-progress shards, just return current stats
134             listener.onResponse(buildResponse(request, currentSnapshots, null));
135         }
136 
137     }
138 
139     private SnapshotsStatusResponse buildResponse(SnapshotsStatusRequest request, ImmutableList<SnapshotMetaData.Entry> currentSnapshots,
140                                                   TransportNodesSnapshotsStatus.NodesSnapshotStatus nodeSnapshotStatuses) throws IOException {
141         // First process snapshot that are currently processed
142         ImmutableList.Builder<SnapshotStatus> builder = ImmutableList.builder();
143         Set<SnapshotId> currentSnapshotIds = newHashSet();
144         if (!currentSnapshots.isEmpty()) {
145             Map<String, TransportNodesSnapshotsStatus.NodeSnapshotStatus> nodeSnapshotStatusMap;
146             if (nodeSnapshotStatuses != null) {
147                 nodeSnapshotStatusMap = nodeSnapshotStatuses.getNodesMap();
148             } else {
149                 nodeSnapshotStatusMap = newHashMap();
150             }
151 
152             for (SnapshotMetaData.Entry entry : currentSnapshots) {
153                 currentSnapshotIds.add(entry.snapshotId());
154                 ImmutableList.Builder<SnapshotIndexShardStatus> shardStatusBuilder = ImmutableList.builder();
155                 for (ImmutableMap.Entry<ShardId, SnapshotMetaData.ShardSnapshotStatus> shardEntry : entry.shards().entrySet()) {
156                     SnapshotMetaData.ShardSnapshotStatus status = shardEntry.getValue();
157                     if (status.nodeId() != null) {
158                         // We should have information about this shard from the shard:
159                         TransportNodesSnapshotsStatus.NodeSnapshotStatus nodeStatus = nodeSnapshotStatusMap.get(status.nodeId());
160                         if (nodeStatus != null) {
161                             ImmutableMap<ShardId, SnapshotIndexShardStatus> shardStatues = nodeStatus.status().get(entry.snapshotId());
162                             if (shardStatues != null) {
163                                 SnapshotIndexShardStatus shardStatus = shardStatues.get(shardEntry.getKey());
164                                 if (shardStatus != null) {
165                                     // We have full information about this shard
166                                     shardStatusBuilder.add(shardStatus);
167                                     continue;
168                                 }
169                             }
170                         }
171                     }
172                     final SnapshotIndexShardStage stage;
173                     switch (shardEntry.getValue().state()) {
174                         case FAILED:
175                         case ABORTED:
176                         case MISSING:
177                             stage = SnapshotIndexShardStage.FAILURE;
178                             break;
179                         case INIT:
180                         case WAITING:
181                         case STARTED:
182                             stage = SnapshotIndexShardStage.STARTED;
183                             break;
184                         case SUCCESS:
185                             stage = SnapshotIndexShardStage.DONE;
186                             break;
187                         default:
188                             throw new ElasticsearchIllegalArgumentException("Unknown snapshot state " + shardEntry.getValue().state());
189                     }
190                     SnapshotIndexShardStatus shardStatus = new SnapshotIndexShardStatus(shardEntry.getKey(), stage);
191                     shardStatusBuilder.add(shardStatus);
192                 }
193                 builder.add(new SnapshotStatus(entry.snapshotId(), entry.state(), shardStatusBuilder.build()));
194             }
195         }
196         // Now add snapshots on disk that are not currently running
197         if (Strings.hasText(request.repository())) {
198             if (request.snapshots() != null && request.snapshots().length > 0) {
199                 for (String snapshotName : request.snapshots()) {
200                     SnapshotId snapshotId = new SnapshotId(request.repository(), snapshotName);
201                     if (currentSnapshotIds.contains(snapshotId)) {
202                         // This is a snapshot the is currently running - skipping
203                         continue;
204                     }
205                     Snapshot snapshot = snapshotsService.snapshot(snapshotId);
206                     ImmutableList.Builder<SnapshotIndexShardStatus> shardStatusBuilder = ImmutableList.builder();
207                     if (snapshot.state().completed()) {
208                         ImmutableMap<ShardId, IndexShardSnapshotStatus> shardStatues = snapshotsService.snapshotShards(snapshotId);
209                         for (ImmutableMap.Entry<ShardId, IndexShardSnapshotStatus> shardStatus : shardStatues.entrySet()) {
210                             shardStatusBuilder.add(new SnapshotIndexShardStatus(shardStatus.getKey(), shardStatus.getValue()));
211                         }
212                         final SnapshotMetaData.State state;
213                         switch (snapshot.state()) {
214                             case FAILED:
215                                 state = SnapshotMetaData.State.FAILED;
216                                 break;
217                             case SUCCESS:
218                             case PARTIAL:
219                                 // Translating both PARTIAL and SUCCESS to SUCCESS for now
220                                 // TODO: add the differentiation on the metadata level in the next major release
221                                 state = SnapshotMetaData.State.SUCCESS;
222                                 break;
223                             default:
224                                 throw new ElasticsearchIllegalArgumentException("Unknown snapshot state " + snapshot.state());
225                         }
226                         builder.add(new SnapshotStatus(snapshotId, state, shardStatusBuilder.build()));
227                     }
228                 }
229             }
230         }
231 
232         return new SnapshotsStatusResponse(builder.build());
233     }
