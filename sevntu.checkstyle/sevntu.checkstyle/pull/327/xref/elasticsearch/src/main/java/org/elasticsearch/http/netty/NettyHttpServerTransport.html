135     @Inject
136     public NettyHttpServerTransport(Settings settings, NetworkService networkService, BigArrays bigArrays) {
137         super(settings);
138         this.networkService = networkService;
139         this.bigArrays = bigArrays;
140 
141         if (settings.getAsBoolean("netty.epollBugWorkaround", false)) {
142             System.setProperty("org.jboss.netty.epollBugWorkaround", "true");
143         }
144 
145         ByteSizeValue maxContentLength = settings.getAsBytesSize("http.netty.max_content_length", settings.getAsBytesSize("http.max_content_length", new ByteSizeValue(100, ByteSizeUnit.MB)));
146         this.maxChunkSize = settings.getAsBytesSize("http.netty.max_chunk_size", settings.getAsBytesSize("http.max_chunk_size", new ByteSizeValue(8, ByteSizeUnit.KB)));
147         this.maxHeaderSize = settings.getAsBytesSize("http.netty.max_header_size", settings.getAsBytesSize("http.max_header_size", new ByteSizeValue(8, ByteSizeUnit.KB)));
148         this.maxInitialLineLength = settings.getAsBytesSize("http.netty.max_initial_line_length", settings.getAsBytesSize("http.max_initial_line_length", new ByteSizeValue(4, ByteSizeUnit.KB)));
149         // don't reset cookies by default, since I don't think we really need to
150         // note, parsing cookies was fixed in netty 3.5.1 regarding stack allocation, but still, currently, we don't need cookies
151         this.resetCookies = settings.getAsBoolean("http.netty.reset_cookies", settings.getAsBoolean("http.reset_cookies", false));
152         this.maxCumulationBufferCapacity = settings.getAsBytesSize("http.netty.max_cumulation_buffer_capacity", null);
153         this.maxCompositeBufferComponents = settings.getAsInt("http.netty.max_composite_buffer_components", -1);
154         this.workerCount = settings.getAsInt("http.netty.worker_count", EsExecutors.boundedNumberOfProcessors(settings) * 2);
155         this.blockingServer = settings.getAsBoolean("http.netty.http.blocking_server", settings.getAsBoolean(TCP_BLOCKING_SERVER, settings.getAsBoolean(TCP_BLOCKING, false)));
156         this.port = settings.get("http.netty.port", settings.get("http.port", "9200-9300"));
157         this.bindHost = settings.get("http.netty.bind_host", settings.get("http.bind_host", settings.get("http.host")));
158         this.publishHost = settings.get("http.netty.publish_host", settings.get("http.publish_host", settings.get("http.host")));
159         this.publishPort = settings.getAsInt("http.netty.publish_port", settings.getAsInt("http.publish_port", 0));
160         this.tcpNoDelay = settings.get("http.netty.tcp_no_delay", settings.get(TCP_NO_DELAY, "true"));
161         this.tcpKeepAlive = settings.get("http.netty.tcp_keep_alive", settings.get(TCP_KEEP_ALIVE, "true"));
162         this.reuseAddress = settings.getAsBoolean("http.netty.reuse_address", settings.getAsBoolean(TCP_REUSE_ADDRESS, NetworkUtils.defaultReuseAddress()));
163         this.tcpSendBufferSize = settings.getAsBytesSize("http.netty.tcp_send_buffer_size", settings.getAsBytesSize(TCP_SEND_BUFFER_SIZE, TCP_DEFAULT_SEND_BUFFER_SIZE));
164         this.tcpReceiveBufferSize = settings.getAsBytesSize("http.netty.tcp_receive_buffer_size", settings.getAsBytesSize(TCP_RECEIVE_BUFFER_SIZE, TCP_DEFAULT_RECEIVE_BUFFER_SIZE));
165 
166         long defaultReceiverPredictor = 512 * 1024;
167         if (JvmInfo.jvmInfo().mem().directMemoryMax().bytes() > 0) {
168             // we can guess a better default...
169             long l = (long) ((0.3 * JvmInfo.jvmInfo().mem().directMemoryMax().bytes()) / workerCount);
170             defaultReceiverPredictor = Math.min(defaultReceiverPredictor, Math.max(l, 64 * 1024));
171         }
172 
173         // See AdaptiveReceiveBufferSizePredictor#DEFAULT_XXX for default values in netty..., we can use higher ones for us, even fixed one
174         ByteSizeValue receivePredictorMin = settings.getAsBytesSize("http.netty.receive_predictor_min", settings.getAsBytesSize("http.netty.receive_predictor_size", new ByteSizeValue(defaultReceiverPredictor)));
175         ByteSizeValue receivePredictorMax = settings.getAsBytesSize("http.netty.receive_predictor_max", settings.getAsBytesSize("http.netty.receive_predictor_size", new ByteSizeValue(defaultReceiverPredictor)));
176         if (receivePredictorMax.bytes() == receivePredictorMin.bytes()) {
177             receiveBufferSizePredictorFactory = new FixedReceiveBufferSizePredictorFactory((int) receivePredictorMax.bytes());
178         } else {
179             receiveBufferSizePredictorFactory = new AdaptiveReceiveBufferSizePredictorFactory((int) receivePredictorMin.bytes(), (int) receivePredictorMin.bytes(), (int) receivePredictorMax.bytes());
180         }
181 
182         this.compression = settings.getAsBoolean(SETTING_HTTP_COMPRESSION, false);
183         this.compressionLevel = settings.getAsInt(SETTING_HTTP_COMPRESSION_LEVEL, 6);
184         this.pipelining = settings.getAsBoolean(SETTING_PIPELINING, DEFAULT_SETTING_PIPELINING);
185         this.pipeliningMaxEvents = settings.getAsInt(SETTING_PIPELINING_MAX_EVENTS, DEFAULT_SETTING_PIPELINING_MAX_EVENTS);
186 
187         // validate max content length
188         if (maxContentLength.bytes() > Integer.MAX_VALUE) {
189             logger.warn("maxContentLength[" + maxContentLength + "] set to high value, resetting it to [100mb]");
190             maxContentLength = new ByteSizeValue(100, ByteSizeUnit.MB);
191         }
192         this.maxContentLength = maxContentLength;
193 
194         logger.debug("using max_chunk_size[{}], max_header_size[{}], max_initial_line_length[{}], max_content_length[{}], receive_predictor[{}->{}], pipelining[{}], pipelining_max_events[{}]",
195                 maxChunkSize, maxHeaderSize, maxInitialLineLength, this.maxContentLength, receivePredictorMin, receivePredictorMax, pipelining, pipeliningMaxEvents);
196     }

207     @Override
208     protected void doStart() throws ElasticsearchException {
209         this.serverOpenChannels = new OpenChannelsHandler(logger);
210 
211         if (blockingServer) {
212             serverBootstrap = new ServerBootstrap(new OioServerSocketChannelFactory(
213                     Executors.newCachedThreadPool(daemonThreadFactory(settings, "http_server_boss")),
214                     Executors.newCachedThreadPool(daemonThreadFactory(settings, "http_server_worker"))
215             ));
216         } else {
217             serverBootstrap = new ServerBootstrap(new NioServerSocketChannelFactory(
218                     Executors.newCachedThreadPool(daemonThreadFactory(settings, "http_server_boss")),
219                     Executors.newCachedThreadPool(daemonThreadFactory(settings, "http_server_worker")),
220                     workerCount));
221         }
222 
223         serverBootstrap.setPipelineFactory(configureServerChannelPipelineFactory());
224 
225         if (!"default".equals(tcpNoDelay)) {
226             serverBootstrap.setOption("child.tcpNoDelay", Booleans.parseBoolean(tcpNoDelay, null));
227         }
228         if (!"default".equals(tcpKeepAlive)) {
229             serverBootstrap.setOption("child.keepAlive", Booleans.parseBoolean(tcpKeepAlive, null));
230         }
231         if (tcpSendBufferSize != null && tcpSendBufferSize.bytes() > 0) {
232             serverBootstrap.setOption("child.sendBufferSize", tcpSendBufferSize.bytes());
233         }
234         if (tcpReceiveBufferSize != null && tcpReceiveBufferSize.bytes() > 0) {
235             serverBootstrap.setOption("child.receiveBufferSize", tcpReceiveBufferSize.bytes());
236         }
237         serverBootstrap.setOption("receiveBufferSizePredictorFactory", receiveBufferSizePredictorFactory);
238         serverBootstrap.setOption("child.receiveBufferSizePredictorFactory", receiveBufferSizePredictorFactory);
239         if (reuseAddress != null) {
240             serverBootstrap.setOption("reuseAddress", reuseAddress);
241             serverBootstrap.setOption("child.reuseAddress", reuseAddress);
242         }
243 
244         // Bind and start to accept incoming connections.
245         InetAddress hostAddressX;
246         try {
247             hostAddressX = networkService.resolveBindHostAddress(bindHost);
248         } catch (IOException e) {
249             throw new BindHttpException("Failed to resolve host [" + bindHost + "]", e);
250         }
251         final InetAddress hostAddress = hostAddressX;
252 
253         PortsRange portsRange = new PortsRange(port);
254         final AtomicReference<Exception> lastException = new AtomicReference<>();
255         boolean success = portsRange.iterate(new PortsRange.PortCallback() {
256             @Override
257             public boolean onPortNumber(int portNumber) {
258                 try {
259                     serverChannel = serverBootstrap.bind(new InetSocketAddress(hostAddress, portNumber));
260                 } catch (Exception e) {
261                     lastException.set(e);
262                     return false;
263                 }
264                 return true;
265             }
266         });
267         if (!success) {
268             throw new BindHttpException("Failed to bind to [" + port + "]", lastException.get());
269         }
270 
271         InetSocketAddress boundAddress = (InetSocketAddress) serverChannel.getLocalAddress();
272         InetSocketAddress publishAddress;
273         if (0 == publishPort) {
274             publishPort = boundAddress.getPort();
275         }
276         try {
277             publishAddress = new InetSocketAddress(networkService.resolvePublishHostAddress(publishHost), publishPort);
278         } catch (Exception e) {
279             throw new BindTransportException("Failed to resolve publish address", e);
280         }
281         this.boundAddress = new BoundTransportAddress(new InetSocketTransportAddress(boundAddress), new InetSocketTransportAddress(publishAddress));
282     }

284     @Override
285     protected void doStop() throws ElasticsearchException {
286         if (serverChannel != null) {
287             serverChannel.close().awaitUninterruptibly();
288             serverChannel = null;
289         }
290 
291         if (serverOpenChannels != null) {
292             serverOpenChannels.close();
293             serverOpenChannels = null;
294         }
295 
296         if (serverBootstrap != null) {
297             serverBootstrap.releaseExternalResources();
298             serverBootstrap = null;
299         }
300     }

330     protected void exceptionCaught(ChannelHandlerContext ctx, ExceptionEvent e) throws Exception {
331         if (e.getCause() instanceof ReadTimeoutException) {
332             if (logger.isTraceEnabled()) {
333                 logger.trace("Connection timeout [{}]", ctx.getChannel().getRemoteAddress());
334             }
335             ctx.getChannel().close();
336         } else {
337             if (!lifecycle.started()) {
338                 // ignore
339                 return;
340             }
341             if (!NetworkExceptionHelper.isCloseConnectionException(e.getCause())) {
342                 logger.warn("Caught exception while handling client http traffic, closing connection {}", e.getCause(), ctx.getChannel());
343                 ctx.getChannel().close();
344             } else {
345                 logger.debug("Caught exception while handling client http traffic, closing connection {}", e.getCause(), ctx.getChannel());
346                 ctx.getChannel().close();
347             }
348         }

355     protected static class HttpChannelPipelineFactory implements ChannelPipelineFactory {
356 
357         protected final NettyHttpServerTransport transport;
358         protected final HttpRequestHandler requestHandler;
359 
360         public HttpChannelPipelineFactory(NettyHttpServerTransport transport) {
361             this.transport = transport;
362             this.requestHandler = new HttpRequestHandler(transport);
363         }
364 
365         @Override
366         public ChannelPipeline getPipeline() throws Exception {
367             ChannelPipeline pipeline = Channels.pipeline();
368             pipeline.addLast("openChannels", transport.serverOpenChannels);
369             HttpRequestDecoder requestDecoder = new HttpRequestDecoder(
370                     (int) transport.maxInitialLineLength.bytes(),
371                     (int) transport.maxHeaderSize.bytes(),
372                     (int) transport.maxChunkSize.bytes()
373             );
374             if (transport.maxCumulationBufferCapacity != null) {
375                 if (transport.maxCumulationBufferCapacity.bytes() > Integer.MAX_VALUE) {
376                     requestDecoder.setMaxCumulationBufferCapacity(Integer.MAX_VALUE);
377                 } else {
378                     requestDecoder.setMaxCumulationBufferCapacity((int) transport.maxCumulationBufferCapacity.bytes());
379                 }
380             }
381             if (transport.maxCompositeBufferComponents != -1) {
382                 requestDecoder.setMaxCumulationBufferComponents(transport.maxCompositeBufferComponents);
383             }
384             pipeline.addLast("decoder", requestDecoder);
385             pipeline.addLast("decoder_compress", new ESHttpContentDecompressor(transport.compression));
386             HttpChunkAggregator httpChunkAggregator = new HttpChunkAggregator((int) transport.maxContentLength.bytes());
387             if (transport.maxCompositeBufferComponents != -1) {
388                 httpChunkAggregator.setMaxCumulationBufferComponents(transport.maxCompositeBufferComponents);
389             }
390             pipeline.addLast("aggregator", httpChunkAggregator);
391             pipeline.addLast("encoder", new ESHttpResponseEncoder());
392             if (transport.compression) {
393                 pipeline.addLast("encoder_compress", new HttpContentCompressor(transport.compressionLevel));
394             }
395             if (transport.pipelining) {
396                 pipeline.addLast("pipelining", new HttpPipeliningHandler(transport.pipeliningMaxEvents));
397             }
398             pipeline.addLast("handler", requestHandler);
399             return pipeline;
400         }
