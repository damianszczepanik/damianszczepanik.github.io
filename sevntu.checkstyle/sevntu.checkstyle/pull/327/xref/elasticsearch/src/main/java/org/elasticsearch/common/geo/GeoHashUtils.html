48      /**
49       * Encodes the given latitude and longitude into a geohash
50       *
51       * @param latitude  Latitude to encode
52       * @param longitude Longitude to encode
53       * @return Geohash encoding of the longitude and latitude
54       */
55      public static String encode(double latitude, double longitude, int precision) {
56  //        double[] latInterval = {-90.0, 90.0};
57  //        double[] lngInterval = {-180.0, 180.0};
58          double latInterval0 = -90.0;
59          double latInterval1 = 90.0;
60          double lngInterval0 = -180.0;
61          double lngInterval1 = 180.0;
62  
63          final StringBuilder geohash = new StringBuilder();
64          boolean isEven = true;
65  
66          int bit = 0;
67          int ch = 0;
68  
69          while (geohash.length() < precision) {
70              double mid = 0.0;
71              if (isEven) {
72  //                mid = (lngInterval[0] + lngInterval[1]) / 2D;
73                  mid = (lngInterval0 + lngInterval1) / 2D;
74                  if (longitude > mid) {
75                      ch |= BITS[bit];
76  //                    lngInterval[0] = mid;
77                      lngInterval0 = mid;
78                  } else {
79  //                    lngInterval[1] = mid;
80                      lngInterval1 = mid;
81                  }
82              } else {
83  //                mid = (latInterval[0] + latInterval[1]) / 2D;
84                  mid = (latInterval0 + latInterval1) / 2D;
85                  if (latitude > mid) {
86                      ch |= BITS[bit];
87  //                    latInterval[0] = mid;
88                      latInterval0 = mid;
89                  } else {
90  //                    latInterval[1] = mid;
91                      latInterval1 = mid;
92                  }
93              }
94  
95              isEven = !isEven;
96  
97              if (bit < 4) {
98                  bit++;
99              } else {
100                 geohash.append(BASE_32[ch]);
101                 bit = 0;
102                 ch = 0;
103             }
104         }
105 
106         return geohash.toString();
107     }

109     private static final char encode(int x, int y) {
110         return BASE_32[((x & 1) + ((y & 1) * 2) + ((x & 2) * 2) + ((y & 2) * 4) + ((x & 4) * 4)) % 32];
111     }

113     /**
114      * Calculate all neighbors of a given geohash cell.
115      *
116      * @param geohash Geohash of the defined cell
117      * @return geohashes of all neighbor cells

119     public static Collection<? extends CharSequence> neighbors(String geohash) {
120         return addNeighbors(geohash, geohash.length(), new ArrayList<CharSequence>(8));
121     }

123     /**
124      * Calculate the geohash of a neighbor of a geohash
125      *
126      * @param geohash the geohash of a cell
127      * @param level   level of the geohash
128      * @param dx      delta of the first grid coordinate (must be -1, 0 or +1)
129      * @param dy      delta of the second grid coordinate (must be -1, 0 or +1)
130      * @return geohash of the defined cell
131      */
132     private final static String neighbor(String geohash, int level, int dx, int dy) {
133         int cell = decode(geohash.charAt(level - 1));
134 
135         // Decoding the Geohash bit pattern to determine grid coordinates
136         int x0 = cell & 1;  // first bit of x
137         int y0 = cell & 2;  // first bit of y
138         int x1 = cell & 4;  // second bit of x
139         int y1 = cell & 8;  // second bit of y
140         int x2 = cell & 16; // third bit of x
141 
142         // combine the bitpattern to grid coordinates.
143         // note that the semantics of x and y are swapping
144         // on each level
145         int x = x0 + (x1 / 2) + (x2 / 4);
146         int y = (y0 / 2) + (y1 / 4);
147 
148         if (level == 1) {
149             // Root cells at north (namely "bcfguvyz") or at
150             // south (namely "0145hjnp") do not have neighbors
151             // in north/south direction
152             if ((dy < 0 && y == 0) || (dy > 0 && y == 3)) {
153                 return null;
154             } else {
155                 return Character.toString(encode(x + dx, y + dy));
156             }
157         } else {
158             // define grid coordinates for next level
159             final int nx = ((level % 2) == 1) ? (x + dx) : (x + dy);
160             final int ny = ((level % 2) == 1) ? (y + dy) : (y + dx);
161 
162             // if the defined neighbor has the same parent a the current cell
163             // encode the cell directly. Otherwise find the cell next to this
164             // cell recursively. Since encoding wraps around within a cell
165             // it can be encoded here.
166             // xLimit and YLimit must always be respectively 7 and 3
167             // since x and y semantics are swapping on each level.
168             if (nx >= 0 && nx <= 7 && ny >= 0 && ny <= 3) {
169                 return geohash.substring(0, level - 1) + encode(nx, ny);
170             } else {
171                 String neighbor = neighbor(geohash, level - 1, dx, dy);
172                 if(neighbor != null) {
173                     return neighbor + encode(nx, ny); 
174                 } else {
175                     return null;
176                 }
177             }
178         }
179     }

181     /**
182      * Add all geohashes of the cells next to a given geohash to a list.
183      *
184      * @param geohash   Geohash of a specified cell
185      * @param neighbors list to add the neighbors to
186      * @return the given list
187      */
188     public static final <E extends Collection<? super String>> E addNeighbors(String geohash, E neighbors) {
189         return addNeighbors(geohash, geohash.length(), neighbors);
190     }
191     
192     /**
193      * Add all geohashes of the cells next to a given geohash to a list.
194      *
195      * @param geohash   Geohash of a specified cell
196      * @param length    level of the given geohash
197      * @param neighbors list to add the neighbors to
198      * @return the given list
199      */
200     public static final <E extends Collection<? super String>> E addNeighbors(String geohash, int length, E neighbors) {
201         String south = neighbor(geohash, length, 0, -1);
202         String north = neighbor(geohash, length, 0, +1);
203         if (north != null) {
204             neighbors.add(neighbor(north, length, -1, 0));
205             neighbors.add(north);
206             neighbors.add(neighbor(north, length, +1, 0));
207         }
208 
209         neighbors.add(neighbor(geohash, length, -1, 0));
210         neighbors.add(neighbor(geohash, length, +1, 0));
211 
212         if (south != null) {
213             neighbors.add(neighbor(south, length, -1, 0));
214             neighbors.add(south);
215             neighbors.add(neighbor(south, length, +1, 0));
216         }
217 
218         return neighbors;
219     }

302     /**
303      * Decodes the given geohash into a latitude and longitude
304      *
305      * @param geohash Geohash to decocde
306      * @return the given {@link GeoPoint} reseted to the center of
307      *         cell, given by the geohash
308      */
309     public static GeoPoint decode(String geohash, GeoPoint ret) {
310         double[] interval = decodeCell(geohash);
311         return ret.reset((interval[0] + interval[1]) / 2D, (interval[2] + interval[3]) / 2D);
312     }

314     private static double[] decodeCell(String geohash) {
315         double[] interval = {-90.0, 90.0, -180.0, 180.0};
316         boolean isEven = true;
317 
318         for (int i = 0; i < geohash.length(); i++) {
319             final int cd = decode(geohash.charAt(i));
320 
321             for (int mask : BITS) {
322                 if (isEven) {
323                     if ((cd & mask) != 0) {
324                         interval[2] = (interval[2] + interval[3]) / 2D;
325                     } else {
326                         interval[3] = (interval[2] + interval[3]) / 2D;
327                     }
328                 } else {
329                     if ((cd & mask) != 0) {
330                         interval[0] = (interval[0] + interval[1]) / 2D;
331                     } else {
332                         interval[1] = (interval[0] + interval[1]) / 2D;
333                     }
334                 }
335                 isEven = !isEven;
336             }
337         }
338         return interval;
339     }

344     /**
345      * Encodes latitude and longitude information into a single long with variable precision.
346      * Up to 12 levels of precision are supported which should offer sub-metre resolution.
347      *
348      * @param latitude
349      * @param longitude
350      * @param precision The required precision between 1 and 12
351      * @return A single long where 4 bits are used for holding the precision and the remaining 
352      * 60 bits are reserved for 5 bit cell identifiers giving up to 12 layers. 
353      */
354     public static long encodeAsLong(double latitude, double longitude, int precision) {
355         if((precision>12)||(precision<1))
356         {
357             throw new ElasticsearchIllegalArgumentException("Illegal precision length of "+precision+
358                     ". Long-based geohashes only support precisions between 1 and 12");
359         }
360         double latInterval0 = -90.0;
361         double latInterval1 = 90.0;
362         double lngInterval0 = -180.0;
363         double lngInterval1 = 180.0;
364 
365         long geohash = 0l;
366         boolean isEven = true;
367 
368         int bit = 0;
369         int ch = 0;
370 
371         int geohashLength=0;
372         while (geohashLength < precision) {
373             double mid = 0.0;
374             if (isEven) {
375                 mid = (lngInterval0 + lngInterval1) / 2D;
376                 if (longitude > mid) {
377                     ch |= BITS[bit];
378                     lngInterval0 = mid;
379                 } else {
380                     lngInterval1 = mid;
381                 }
382             } else {
383                 mid = (latInterval0 + latInterval1) / 2D;
384                 if (latitude > mid) {
385                     ch |= BITS[bit];
386                     latInterval0 = mid;
387                 } else {
388                     latInterval1 = mid;
389                 }
390             }
391 
392             isEven = !isEven;
393 
394             if (bit < 4) {
395                 bit++;
396             } else {
397                 geohashLength++;
398                 geohash|=ch;
399                 if(geohashLength<precision){
400                     geohash<<=5;
401                 }
402                 bit = 0;
403                 ch = 0;
404             }
405         }
406         geohash<<=4;
407         geohash|=precision;
408         return geohash;
409     }

411     /**
412      * Formats a geohash held as a long as a more conventional 
413      * String-based geohash
414      * @param geohashAsLong a geohash encoded as a long
415      * @return A traditional base32-based String representation of a geohash 
416      */
417     public static String toString(long geohashAsLong)
418     {
419         int precision = (int) (geohashAsLong&15);
420         char[] chars = new char[precision];
421         geohashAsLong >>= 4;
422         for (int i = precision - 1; i >= 0 ; i--) {
423             chars[i] =  BASE_32[(int) (geohashAsLong & 31)];
424             geohashAsLong >>= 5;
425         }
426         return new String(chars);        
427     }

449     private static double[] decodeCell(long geohash) {
450         double[] interval = {-90.0, 90.0, -180.0, 180.0};
451         boolean isEven = true;
452         
453         int precision= (int) (geohash&15);
454         geohash>>=4;
455         int[]cds=new int[precision];
456         for (int i = precision-1; i >=0 ; i--) {            
457             cds[i] = (int) (geohash&31);
458             geohash>>=5;
459         }
460 
461         for (int i = 0; i <cds.length ; i++) {            
462             final int cd = cds[i];
463             for (int mask : BITS) {
464                 if (isEven) {
465                     if ((cd & mask) != 0) {
466                         interval[2] = (interval[2] + interval[3]) / 2D;
467                     } else {
468                         interval[3] = (interval[2] + interval[3]) / 2D;
469                     }
470                 } else {
471                     if ((cd & mask) != 0) {
472                         interval[0] = (interval[0] + interval[1]) / 2D;
473                     } else {
474                         interval[1] = (interval[0] + interval[1]) / 2D;
475                     }
476                 }
477                 isEven = !isEven;
478             }
479         }
480         return interval;
