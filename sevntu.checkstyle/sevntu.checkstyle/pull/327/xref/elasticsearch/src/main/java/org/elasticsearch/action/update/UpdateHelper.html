71      /**
72       * Prepares an update request by converting it into an index or delete request or an update response (no action).
73       */
74      @SuppressWarnings("unchecked")
75      public Result prepare(UpdateRequest request, IndexShard indexShard) {
76          long getDate = System.currentTimeMillis();
77          final GetResult getResult = indexShard.getService().get(request.type(), request.id(),
78                  new String[]{RoutingFieldMapper.NAME, ParentFieldMapper.NAME, TTLFieldMapper.NAME, TimestampFieldMapper.NAME},
79                  true, request.version(), request.versionType(), FetchSourceContext.FETCH_SOURCE, false);
80  
81          if (!getResult.isExists()) {
82              if (request.upsertRequest() == null && !request.docAsUpsert()) {
83                  throw new DocumentMissingException(new ShardId(indexShard.indexService().index().name(), request.shardId()), request.type(), request.id());
84              }
85              Long ttl = null;
86              IndexRequest indexRequest = request.docAsUpsert() ? request.doc() : request.upsertRequest();
87              if (request.scriptedUpsert() && (request.script() != null)) {
88                  // Run the script to perform the create logic
89                  IndexRequest upsert = request.upsertRequest();               
90                  Map<String, Object> upsertDoc = upsert.sourceAsMap();
91                  Map<String, Object> ctx = new HashMap<>(2);
92                  // Tell the script that this is a create and not an update
93                  ctx.put("op", "create");
94                  ctx.put("_source", upsertDoc);
95                  try {
96                      ExecutableScript script = scriptService.executable(request.scriptLang, request.script, request.scriptType, request.scriptParams);
97                      script.setNextVar("ctx", ctx);
98                      script.run();
99                      // we need to unwrap the ctx...
100                     ctx = (Map<String, Object>) script.unwrap(ctx);
101                 } catch (Exception e) {
102                     throw new ElasticsearchIllegalArgumentException("failed to execute script", e);
103                 }                
104                 //Allow the script to set TTL using ctx._ttl
105                 ttl = getTTLFromScriptContext(ctx);
106                 //Allow the script to abort the create by setting "op" to "none"
107                 String scriptOpChoice = (String) ctx.get("op");
108                 
109                 // Only valid options for an upsert script are "create"
110                 // (the default) or "none", meaning abort upsert
111                 if (!"create".equals(scriptOpChoice)) {
112                     if (!"none".equals(scriptOpChoice)) {
113                         logger.warn("Used upsert operation [{}] for script [{}], doing nothing...", scriptOpChoice, request.script);
114                     }
115                     UpdateResponse update = new UpdateResponse(getResult.getIndex(), getResult.getType(), getResult.getId(),
116                             getResult.getVersion(), false);
117                     update.setGetResult(getResult);
118                     return new Result(update, Operation.NONE, upsertDoc, XContentType.JSON);
119                 }
120                 indexRequest.source((Map)ctx.get("_source"));
121             }
122 
123             indexRequest.index(request.index()).type(request.type()).id(request.id())
124                     // it has to be a "create!"
125                     .create(true)                    
126                     .routing(request.routing())
127                     .ttl(ttl)
128                     .refresh(request.refresh())
129                     .replicationType(request.replicationType()).consistencyLevel(request.consistencyLevel());
130             indexRequest.operationThreaded(false);
131             if (request.versionType() != VersionType.INTERNAL) {
132                 // in all but the internal versioning mode, we want to create the new document using the given version.
133                 indexRequest.version(request.version()).versionType(request.versionType());
134             }
135             return new Result(indexRequest, Operation.UPSERT, null, null);
136         }
137 
138         long updateVersion = getResult.getVersion();
139 
140         if (request.versionType() != VersionType.INTERNAL) {
141             assert request.versionType() == VersionType.FORCE;
142             updateVersion = request.version(); // remember, match_any is excluded by the conflict test
143         }
144 
145         if (getResult.internalSourceRef() == null) {
146             // no source, we can't do nothing, through a failure...
147             throw new DocumentSourceMissingException(new ShardId(indexShard.indexService().index().name(), request.shardId()), request.type(), request.id());
148         }
149 
150         Tuple<XContentType, Map<String, Object>> sourceAndContent = XContentHelper.convertToMap(getResult.internalSourceRef(), true);
151         String operation = null;
152         String timestamp = null;
153         Long ttl = null;
154         final Map<String, Object> updatedSourceAsMap;
155         final XContentType updateSourceContentType = sourceAndContent.v1();
156         String routing = getResult.getFields().containsKey(RoutingFieldMapper.NAME) ? getResult.field(RoutingFieldMapper.NAME).getValue().toString() : null;
157         String parent = getResult.getFields().containsKey(ParentFieldMapper.NAME) ? getResult.field(ParentFieldMapper.NAME).getValue().toString() : null;
158 
159         if (request.script() == null && request.doc() != null) {
160             IndexRequest indexRequest = request.doc();
161             updatedSourceAsMap = sourceAndContent.v2();
162             if (indexRequest.ttl() > 0) {
163                 ttl = indexRequest.ttl();
164             }
165             timestamp = indexRequest.timestamp();
166             if (indexRequest.routing() != null) {
167                 routing = indexRequest.routing();
168             }
169             if (indexRequest.parent() != null) {
170                 parent = indexRequest.parent();
171             }
172             boolean noop = !XContentHelper.update(updatedSourceAsMap, indexRequest.sourceAsMap(), request.detectNoop());
173             // noop could still be true even if detectNoop isn't because update detects empty maps as noops.  BUT we can only
174             // actually turn the update into a noop if detectNoop is true to preserve backwards compatibility and to handle
175             // cases where users repopulating multi-fields or adding synonyms, etc.
176             if (request.detectNoop() && noop) {
177                 operation = "none";
178             }
179         } else {
180             Map<String, Object> ctx = new HashMap<>(16);
181             Long originalTtl = getResult.getFields().containsKey(TTLFieldMapper.NAME) ? (Long) getResult.field(TTLFieldMapper.NAME).getValue() : null;
182             Long originalTimestamp = getResult.getFields().containsKey(TimestampFieldMapper.NAME) ? (Long) getResult.field(TimestampFieldMapper.NAME).getValue() : null;
183             ctx.put("_index", getResult.getIndex());
184             ctx.put("_type", getResult.getType());
185             ctx.put("_id", getResult.getId());
186             ctx.put("_version", getResult.getVersion());
187             ctx.put("_routing", routing);
188             ctx.put("_parent", parent);
189             ctx.put("_timestamp", originalTimestamp);
190             ctx.put("_ttl", originalTtl);
191             ctx.put("_source", sourceAndContent.v2());
192 
193             try {
194                 ExecutableScript script = scriptService.executable(request.scriptLang, request.script, request.scriptType, request.scriptParams);
195                 script.setNextVar("ctx", ctx);
196                 script.run();
197                 // we need to unwrap the ctx...
198                 ctx = (Map<String, Object>) script.unwrap(ctx);
199             } catch (Exception e) {
200                 throw new ElasticsearchIllegalArgumentException("failed to execute script", e);
201             }
202 
203             operation = (String) ctx.get("op");
204 
205             Object fetchedTimestamp = ctx.get("_timestamp");
206             if (fetchedTimestamp != null) {
207                 timestamp = fetchedTimestamp.toString();
208             } else if (originalTimestamp != null) {
209                 // No timestamp has been given in the update script, so we keep the previous timestamp if there is one
210                 timestamp = originalTimestamp.toString();
211             }
212 
213             ttl = getTTLFromScriptContext(ctx);
214             
215             updatedSourceAsMap = (Map<String, Object>) ctx.get("_source");
216         }
217 
218         // apply script to update the source
219         // No TTL has been given in the update script so we keep previous TTL value if there is one
220         if (ttl == null) {
221             ttl = getResult.getFields().containsKey(TTLFieldMapper.NAME) ? (Long) getResult.field(TTLFieldMapper.NAME).getValue() : null;
222             if (ttl != null) {
223                 ttl = ttl - (System.currentTimeMillis() - getDate); // It is an approximation of exact TTL value, could be improved
224             }
225         }
226 
227         if (operation == null || "index".equals(operation)) {
228             final IndexRequest indexRequest = Requests.indexRequest(request.index()).type(request.type()).id(request.id()).routing(routing).parent(parent)
229                     .source(updatedSourceAsMap, updateSourceContentType)
230                     .version(updateVersion).versionType(request.versionType())
231                     .replicationType(request.replicationType()).consistencyLevel(request.consistencyLevel())
232                     .timestamp(timestamp).ttl(ttl)
233                     .refresh(request.refresh());
234             indexRequest.operationThreaded(false);
235             return new Result(indexRequest, Operation.INDEX, updatedSourceAsMap, updateSourceContentType);
236         } else if ("delete".equals(operation)) {
237             DeleteRequest deleteRequest = Requests.deleteRequest(request.index()).type(request.type()).id(request.id()).routing(routing).parent(parent)
238                     .version(updateVersion).versionType(request.versionType())
239                     .replicationType(request.replicationType()).consistencyLevel(request.consistencyLevel());
240             deleteRequest.operationThreaded(false);
241             return new Result(deleteRequest, Operation.DELETE, updatedSourceAsMap, updateSourceContentType);
242         } else if ("none".equals(operation)) {
243             UpdateResponse update = new UpdateResponse(getResult.getIndex(), getResult.getType(), getResult.getId(), getResult.getVersion(), false);
244             update.setGetResult(extractGetResult(request, indexShard.indexService().index().name(), getResult.getVersion(), updatedSourceAsMap, updateSourceContentType, getResult.internalSourceRef()));
245             return new Result(update, Operation.NONE, updatedSourceAsMap, updateSourceContentType);
246         } else {
247             logger.warn("Used update operation [{}] for script [{}], doing nothing...", operation, request.script);
248             UpdateResponse update = new UpdateResponse(getResult.getIndex(), getResult.getType(), getResult.getId(), getResult.getVersion(), false);
249             return new Result(update, Operation.NONE, updatedSourceAsMap, updateSourceContentType);
250         }
251     }
252 
253     private Long getTTLFromScriptContext(Map<String, Object> ctx) {
254         Long ttl = null;
255         Object fetchedTTL = ctx.get("_ttl");
256         if (fetchedTTL != null) {
257             if (fetchedTTL instanceof Number) {
258                 ttl = ((Number) fetchedTTL).longValue();
259             } else {
260                 ttl = TimeValue.parseTimeValue((String) fetchedTTL, null).millis();
261             }
262         }
263         return ttl;
264     }

266     /**
267      * Extracts the fields from the updated document to be returned in a update response
268      */
269     public GetResult extractGetResult(final UpdateRequest request, String concreteIndex, long version, final Map<String, Object> source, XContentType sourceContentType, @Nullable final BytesReference sourceAsBytes) {
270         if (request.fields() == null || request.fields().length == 0) {
271             return null;
272         }
273         boolean sourceRequested = false;
274         Map<String, GetField> fields = null;
275         if (request.fields() != null && request.fields().length > 0) {
276             SourceLookup sourceLookup = new SourceLookup();
277             sourceLookup.setNextSource(source);
278             for (String field : request.fields()) {
279                 if (field.equals("_source")) {
280                     sourceRequested = true;
281                     continue;
282                 }
283                 Object value = sourceLookup.extractValue(field);
284                 if (value != null) {
285                     if (fields == null) {
286                         fields = newHashMapWithExpectedSize(2);
287                     }
288                     GetField getField = fields.get(field);
289                     if (getField == null) {
290                         getField = new GetField(field, new ArrayList<>(2));
291                         fields.put(field, getField);
292                     }
293                     getField.getValues().add(value);
294                 }
295             }
296         }
297 
298         // TODO when using delete/none, we can still return the source as bytes by generating it (using the sourceContentType)
299         return new GetResult(concreteIndex, request.type(), request.id(), version, true, sourceRequested ? sourceAsBytes : null, fields);
300     }

302     public static class Result {
303 
304         private final Streamable action;
305         private final Operation operation;
306         private final Map<String, Object> updatedSourceAsMap;
307         private final XContentType updateSourceContentType;
308 
309         public Result(Streamable action, Operation operation, Map<String, Object> updatedSourceAsMap, XContentType updateSourceContentType) {
310             this.action = action;
311             this.operation = operation;
312             this.updatedSourceAsMap = updatedSourceAsMap;
313             this.updateSourceContentType = updateSourceContentType;
314         }
315 
316         @SuppressWarnings("unchecked")
317         public <T extends Streamable> T action() {
318             return (T) action;
319         }
320 
321         public Operation operation() {
322             return operation;
323         }
324 
325         public Map<String, Object> updatedSourceAsMap() {
326             return updatedSourceAsMap;
327         }
328 
329         public XContentType updateSourceContentType() {
330             return updateSourceContentType;
331         }
332     }

334     public static enum Operation {
335 
336         UPSERT,
337         INDEX,
338         DELETE,
339         NONE

341     }
