17  package org.elasticsearch.common.inject.internal;
18  
19  import org.elasticsearch.common.Classes;
20  import org.elasticsearch.common.inject.BindingAnnotation;
21  import org.elasticsearch.common.inject.Key;
22  import org.elasticsearch.common.inject.ScopeAnnotation;
23  import org.elasticsearch.common.inject.TypeLiteral;
24  
25  import java.lang.annotation.Annotation;
26  import java.lang.annotation.Retention;
27  import java.lang.annotation.RetentionPolicy;
28  import java.lang.reflect.Member;
29  
30  /**
31   * Annotation utilities.
32   *
33   * @author crazybob@google.com (Bob Lee)
34   */
35  public class Annotations {
36  
37      /**
38       * Returns true if the given annotation is retained at runtime.
39       */
40      public static boolean isRetainedAtRuntime(Class<? extends Annotation> annotationType) {
41          Retention retention = annotationType.getAnnotation(Retention.class);
42          return retention != null && retention.value() == RetentionPolicy.RUNTIME;
43      }
44  
45      /**
46       * Returns the scope annotation on {@code type}, or null if none is specified.
47       */
48      public static Class<? extends Annotation> findScopeAnnotation(
49              Errors errors, Class<?> implementation) {
50          return findScopeAnnotation(errors, implementation.getAnnotations());
51      }
52  
53      /**
54       * Returns the scoping annotation, or null if there isn't one.
55       */
56      public static Class<? extends Annotation> findScopeAnnotation(Errors errors, Annotation[] annotations) {
57          Class<? extends Annotation> found = null;
58  
59          for (Annotation annotation : annotations) {
60              if (annotation.annotationType().getAnnotation(ScopeAnnotation.class) != null) {
61                  if (found != null) {
62                      errors.duplicateScopeAnnotations(found, annotation.annotationType());
63                  } else {
64                      found = annotation.annotationType();
65                  }
66              }
67          }
68  
69          return found;
70      }
71  
72      public static boolean isScopeAnnotation(Class<? extends Annotation> annotationType) {
73          return annotationType.getAnnotation(ScopeAnnotation.class) != null;
74      }
75  
76      /**
77       * Adds an error if there is a misplaced annotations on {@code type}. Scoping
78       * annotations are not allowed on abstract classes or interfaces.
79       */
80      public static void checkForMisplacedScopeAnnotations(
81              Class<?> type, Object source, Errors errors) {
82          if (Classes.isConcrete(type)) {
83              return;
84          }
85  
86          Class<? extends Annotation> scopeAnnotation = findScopeAnnotation(errors, type);
87          if (scopeAnnotation != null) {
88              errors.withSource(type).scopeAnnotationOnAbstractType(scopeAnnotation, type, source);
89          }
90      }
91  
92      /**
93       * Gets a key for the given type, member and annotations.
94       */
95      public static Key<?> getKey(TypeLiteral<?> type, Member member, Annotation[] annotations,
96                                  Errors errors) throws ErrorsException {
97          int numErrorsBefore = errors.size();
98          Annotation found = findBindingAnnotation(errors, member, annotations);
99          errors.throwIfNewErrors(numErrorsBefore);
100         return found == null ? Key.get(type) : Key.get(type, found);
101     }
102 
103     /**
104      * Returns the binding annotation on {@code member}, or null if there isn't one.
105      */
106     public static Annotation findBindingAnnotation(
107             Errors errors, Member member, Annotation[] annotations) {
108         Annotation found = null;
109 
110         for (Annotation annotation : annotations) {
111             if (annotation.annotationType().getAnnotation(BindingAnnotation.class) != null) {
112                 if (found != null) {
113                     errors.duplicateBindingAnnotations(member,
114                             found.annotationType(), annotation.annotationType());
115                 } else {
116                     found = annotation;
117                 }
118             }
119         }
120 
121         return found;
122     }
123 }
