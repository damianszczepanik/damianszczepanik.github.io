189     protected NumberFieldMapper(Names names, int precisionStep, float boost, FieldType fieldType, Boolean docValues,
190                                 Explicit<Boolean> ignoreMalformed, Explicit<Boolean> coerce, NamedAnalyzer indexAnalyzer,
191                                 NamedAnalyzer searchAnalyzer, SimilarityProvider similarity,
192                                 Loading normsLoading, @Nullable Settings fieldDataSettings, Settings indexSettings,
193                                 MultiFields multiFields, CopyTo copyTo) {
194         // LUCENE 4 UPGRADE: Since we can't do anything before the super call, we have to push the boost check down to subclasses
195         super(names, boost, fieldType, docValues, indexAnalyzer, searchAnalyzer,
196                 similarity, normsLoading, fieldDataSettings, indexSettings, multiFields, copyTo);
197         if (precisionStep <= 0 || precisionStep >= maxPrecisionStep()) {
198             this.precisionStep = Integer.MAX_VALUE;
199         } else {
200             this.precisionStep = precisionStep;
201         }
202         this.ignoreMalformed = ignoreMalformed;
203         this.coerce = coerce;
204         Version v = Version.indexCreated(indexSettings);
205         this.useSortedNumericDocValues = v.onOrAfter(Version.V_1_4_0_Beta1);
206     }

251     protected final void addDocValue(ParseContext context, List<Field> fields, long value) {
252         if (useSortedNumericDocValues) {
253             fields.add(new SortedNumericDocValuesField(names().indexName(), value));
254         } else {
255             CustomLongNumericDocValuesField field = (CustomLongNumericDocValuesField) context.doc().getByKey(names().indexName());
256             if (field != null) {
257                 field.add(value);
258             } else {
259                 field = new CustomLongNumericDocValuesField(names().indexName(), value);
260                 context.doc().addWithKey(names().indexName(), field);
261             }
262         }
263     }

305     /**
306      * A terms filter based on the field data cache for numeric fields.
307      */
308     @Override
309     public Filter fieldDataTermsFilter(List values, @Nullable QueryParseContext context) {
310         IndexNumericFieldData fieldData = context.getForField(this);
311         if (fieldData.getNumericType().isFloatingPoint()) {
312             // create with initial size large enough to avoid rehashing
313             DoubleOpenHashSet terms =
314                     new DoubleOpenHashSet((int) (values.size() * (1 + DoubleOpenHashSet.DEFAULT_LOAD_FACTOR)));
315             for (int i = 0, len = values.size(); i < len; i++) {
316                 terms.add(parseDoubleValue(values.get(i)));
317             }
318 
319             return FieldDataTermsFilter.newDoubles(fieldData, terms);
320         } else {
321             // create with initial size large enough to avoid rehashing
322             LongOpenHashSet terms =
323                     new LongOpenHashSet((int) (values.size() * (1 + LongOpenHashSet.DEFAULT_LOAD_FACTOR)));
324             for (int i = 0, len = values.size(); i < len; i++) {
325                 terms.add(parseLongValue(values.get(i)));
326             }
327 
328             return FieldDataTermsFilter.newLongs(fieldData, terms);
329         }
330     }

332     /**
333      * Converts an object value into a double
334      */
335     public double parseDoubleValue(Object value) {
336         if (value instanceof Number) {
337             return ((Number) value).doubleValue();
338         }
339 
340         if (value instanceof BytesRef) {
341             return Double.parseDouble(((BytesRef) value).utf8ToString());
342         }
343 
344         return Double.parseDouble(value.toString());
345     }

347     /**
348      * Converts an object value into a long
349      */
350     public long parseLongValue(Object value) {
351         if (value instanceof Number) {
352             return ((Number) value).longValue();
353         }
354 
355         if (value instanceof BytesRef) {
356             return Long.parseLong(((BytesRef) value).utf8ToString());
357         }
358 
359         return Long.parseLong(value.toString());
360     }

372     @Override
373     public void merge(Mapper mergeWith, MergeContext mergeContext) throws MergeMappingException {
374         super.merge(mergeWith, mergeContext);
375         if (!this.getClass().equals(mergeWith.getClass())) {
376             return;
377         }
378         if (!mergeContext.mergeFlags().simulate()) {
379             NumberFieldMapper nfmMergeWith = (NumberFieldMapper) mergeWith;
380             this.precisionStep = nfmMergeWith.precisionStep;
381             this.includeInAll = nfmMergeWith.includeInAll;
382             if (nfmMergeWith.ignoreMalformed.explicit()) {
383                 this.ignoreMalformed = nfmMergeWith.ignoreMalformed;
384             }
385             if (nfmMergeWith.coerce.explicit()) {
386                 this.coerce = nfmMergeWith.coerce;
387             }
388         }
389     }

395     protected NumericTokenStream popCachedStream() {
396         if (precisionStep == 4) {
397             return tokenStream4.get();
398         } else if (precisionStep == 8) {
399             return tokenStream8.get();
400         } else if (precisionStep == 16) {
401             return tokenStream16.get();
402         } else if (precisionStep == Integer.MAX_VALUE) {
403             return tokenStreamMax.get();
404         }
405         return tokenStream.get();
406     }

485     public static class CustomLongNumericDocValuesField extends CustomNumericDocValuesField {
486 
487         public static final FieldType TYPE = new FieldType();
488         static {
489           TYPE.setDocValuesType(DocValuesType.BINARY);
490           TYPE.freeze();
491         }
492 
493         private final LongArrayList values;
494 
495         public CustomLongNumericDocValuesField(String  name, long value) {
496             super(name);
497             values = new LongArrayList();
498             add(value);
499         }
500 
501         public void add(long value) {
502             values.add(value);
503         }
504 
505         @Override
506         public BytesRef binaryValue() {
507             CollectionUtils.sortAndDedup(values);
508 
509             // here is the trick:
510             //  - the first value is zig-zag encoded so that eg. -5 would become positive and would be better compressed by vLong
511             //  - for other values, we only encode deltas using vLong
512             final byte[] bytes = new byte[values.size() * ByteUtils.MAX_BYTES_VLONG];
513             final ByteArrayDataOutput out = new ByteArrayDataOutput(bytes);
514             ByteUtils.writeVLong(out, ByteUtils.zigZagEncode(values.get(0)));
515             for (int i = 1; i < values.size(); ++i) {
516                 final long delta = values.get(i) - values.get(i - 1);
517                 ByteUtils.writeVLong(out, delta);
518             }
519             return new BytesRef(bytes, 0, out.getPosition());
520         }
521 
522     }
