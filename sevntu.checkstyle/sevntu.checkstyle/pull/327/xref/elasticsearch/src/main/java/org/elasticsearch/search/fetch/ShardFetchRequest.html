20  package org.elasticsearch.search.fetch;
21  
22  import com.carrotsearch.hppc.IntArrayList;
23  import org.apache.lucene.search.FieldDoc;
24  import org.apache.lucene.search.ScoreDoc;
25  import org.elasticsearch.Version;
26  import org.elasticsearch.action.search.SearchScrollRequest;
27  import org.elasticsearch.action.search.type.ParsedScrollId;
28  import org.elasticsearch.common.io.stream.StreamInput;
29  import org.elasticsearch.common.io.stream.StreamOutput;
30  import org.elasticsearch.common.lucene.Lucene;
31  import org.elasticsearch.transport.TransportRequest;
32  
33  import java.io.IOException;
34  
35  /**
36   * Shard level fetch base request. Holds all the info needed to execute a fetch.
37   * Used with search scroll as the original request doesn't hold indices.
38   */
39  public class ShardFetchRequest extends TransportRequest {
40  
41      private long id;
42  
43      private int[] docIds;
44  
45      private int size;
46  
47      private ScoreDoc lastEmittedDoc;
48  
49      public ShardFetchRequest() {
50      }
51  
52      public ShardFetchRequest(SearchScrollRequest request, long id, IntArrayList list, ScoreDoc lastEmittedDoc) {
53          super(request);
54          this.id = id;
55          this.docIds = list.buffer;
56          this.size = list.size();
57          this.lastEmittedDoc = lastEmittedDoc;
58      }
59  
60      protected ShardFetchRequest(TransportRequest originalRequest, long id, IntArrayList list, ScoreDoc lastEmittedDoc) {
61          super(originalRequest);
62          this.id = id;
63          this.docIds = list.buffer;
64          this.size = list.size();
65          this.lastEmittedDoc = lastEmittedDoc;
66      }
67  
68      public long id() {
69          return id;
70      }
71  
72      public int[] docIds() {
73          return docIds;
74      }
75  
76      public int docIdsSize() {
77          return size;
78      }
79  
80      public ScoreDoc lastEmittedDoc() {
81          return lastEmittedDoc;
82      }
83  
84      @Override
85      public void readFrom(StreamInput in) throws IOException {
86          super.readFrom(in);
87          id = in.readLong();
88          size = in.readVInt();
89          docIds = new int[size];
90          for (int i = 0; i < size; i++) {
91              docIds[i] = in.readVInt();
92          }
93          byte flag = in.readByte();
94          if (flag == 1) {
95              lastEmittedDoc = Lucene.readFieldDoc(in);
96          } else if (flag == 2) {
97              lastEmittedDoc = Lucene.readScoreDoc(in);
98          } else if (flag != 0) {
99              throw new IOException("Unknown flag: " + flag);
100         }
101     }
102 
103     @Override
104     public void writeTo(StreamOutput out) throws IOException {
105         super.writeTo(out);
106         out.writeLong(id);
107         out.writeVInt(size);
108         for (int i = 0; i < size; i++) {
109             out.writeVInt(docIds[i]);
110         }
111         if (lastEmittedDoc == null) {
112             out.writeByte((byte) 0);
113         } else if (lastEmittedDoc instanceof FieldDoc) {
114             out.writeByte((byte) 1);
115             Lucene.writeFieldDoc(out, (FieldDoc) lastEmittedDoc);
116         } else {
117             out.writeByte((byte) 2);
118             Lucene.writeScoreDoc(out, lastEmittedDoc);
119         }
120     }
121 }
