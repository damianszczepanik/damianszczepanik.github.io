53      private AliasMetaData(String alias, CompressedString filter, String indexRouting, String searchRouting) {
54          this.alias = alias;
55          this.filter = filter;
56          this.indexRouting = indexRouting;
57          this.searchRouting = searchRouting;
58          if (searchRouting != null) {
59              searchRoutingValues = Collections.unmodifiableSet(Strings.splitStringByCommaToSet(searchRouting));
60          } else {
61              searchRoutingValues = ImmutableSet.of();
62          }
63      }
64  
65      private AliasMetaData(AliasMetaData aliasMetaData, String alias) {
66          this(alias, aliasMetaData.filter(), aliasMetaData.indexRouting(), aliasMetaData.searchRouting());
67      }

124     @Override
125     public boolean equals(Object o) {
126         if (this == o) return true;
127         if (o == null || getClass() != o.getClass()) return false;
128 
129         AliasMetaData that = (AliasMetaData) o;
130 
131         if (alias != null ? !alias.equals(that.alias) : that.alias != null) return false;
132         if (filter != null ? !filter.equals(that.filter) : that.filter != null) return false;
133         if (indexRouting != null ? !indexRouting.equals(that.indexRouting) : that.indexRouting != null) return false;
134         if (searchRouting != null ? !searchRouting.equals(that.searchRouting) : that.searchRouting != null)
135             return false;
136 
137         return true;
138     }

143         result = 31 * result + (filter != null ? filter.hashCode() : 0);

145         result = 31 * result + (searchRouting != null ? searchRouting.hashCode() : 0);

167             indexRouting = aliasMetaData.indexRouting();
168             searchRouting = aliasMetaData.searchRouting();
169         }

172             return alias;

176             this.filter = filter;
177             return this;
178         }
179 
180         public Builder filter(String filter) {
181             if (!Strings.hasLength(filter)) {
182                 this.filter = null;
183                 return this;
184             }
185             try {
186                 try (XContentParser parser = XContentFactory.xContent(filter).createParser(filter)) {
187                     filter(parser.mapOrdered());
188                 }
189                 return this;
190             } catch (IOException e) {
191                 throw new ElasticsearchGenerationException("Failed to generate [" + filter + "]", e);
192             }
193         }
194 
195         public Builder filter(Map<String, Object> filter) {
196             if (filter == null || filter.isEmpty()) {
197                 this.filter = null;
198                 return this;
199             }
200             try {
201                 XContentBuilder builder = XContentFactory.jsonBuilder().map(filter);
202                 this.filter = new CompressedString(builder.bytes());
203                 return this;
204             } catch (IOException e) {
205                 throw new ElasticsearchGenerationException("Failed to build json for alias request", e);
206             }
207         }
208 
209         public Builder filter(XContentBuilder filterBuilder) {
210             try {
211                 return filter(filterBuilder.string());
212             } catch (IOException e) {
213                 throw new ElasticsearchGenerationException("Failed to build json for alias request", e);
214             }
215         }
216 
217         public Builder routing(String routing) {
218             this.indexRouting = routing;
219             this.searchRouting = routing;
220             return this;
221         }

237         public static void toXContent(AliasMetaData aliasMetaData, XContentBuilder builder, ToXContent.Params params) throws IOException {
238             builder.startObject(aliasMetaData.alias(), XContentBuilder.FieldCaseConversion.NONE);
239 
240             boolean binary = params.paramAsBoolean("binary", false);
241 
242             if (aliasMetaData.filter() != null) {
243                 if (binary) {
244                     builder.field("filter", aliasMetaData.filter.compressed());
245                 } else {
246                     byte[] data = aliasMetaData.filter().uncompressed();
247                     XContentParser parser = XContentFactory.xContent(data).createParser(data);
248                     Map<String, Object> filter = parser.mapOrdered();
249                     parser.close();
250                     builder.field("filter", filter);
251                 }
252             }
253             if (aliasMetaData.indexRouting() != null) {
254                 builder.field("index_routing", aliasMetaData.indexRouting());
255             }
256             if (aliasMetaData.searchRouting() != null) {
257                 builder.field("search_routing", aliasMetaData.searchRouting());
258             }
259 
260             builder.endObject();
261         }

263         public static AliasMetaData fromXContent(XContentParser parser) throws IOException {
264             Builder builder = new Builder(parser.currentName());
265 
266             String currentFieldName = null;
267             XContentParser.Token token = parser.nextToken();
268             if (token == null) {
269                 // no data...
270                 return builder.build();
271             }
272             while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
273                 if (token == XContentParser.Token.FIELD_NAME) {
274                     currentFieldName = parser.currentName();
275                 } else if (token == XContentParser.Token.START_OBJECT) {
276                     if ("filter".equals(currentFieldName)) {
277                         Map<String, Object> filter = parser.mapOrdered();
278                         builder.filter(filter);
279                     }
280                 } else if (token == XContentParser.Token.VALUE_EMBEDDED_OBJECT) {
281                     if ("filter".equals(currentFieldName)) {
282                         builder.filter(new CompressedString(parser.binaryValue()));
283                     }
284                 } else if (token == XContentParser.Token.VALUE_STRING) {
285                     if ("routing".equals(currentFieldName)) {
286                         builder.routing(parser.text());
287                     } else if ("index_routing".equals(currentFieldName) || "indexRouting".equals(currentFieldName)) {
288                         builder.indexRouting(parser.text());
289                     } else if ("search_routing".equals(currentFieldName) || "searchRouting".equals(currentFieldName)) {
290                         builder.searchRouting(parser.text());
291                     }
292                 }
293             }
294             return builder.build();
295         }

297         public static void writeTo(AliasMetaData aliasMetaData, StreamOutput out) throws IOException {
298             out.writeString(aliasMetaData.alias());
299             if (aliasMetaData.filter() != null) {
300                 out.writeBoolean(true);
301                 aliasMetaData.filter.writeTo(out);
302             } else {
303                 out.writeBoolean(false);
304             }
305             if (aliasMetaData.indexRouting() != null) {
306                 out.writeBoolean(true);
307                 out.writeString(aliasMetaData.indexRouting());
308             } else {
309                 out.writeBoolean(false);
310             }
311             if (aliasMetaData.searchRouting() != null) {
312                 out.writeBoolean(true);
313                 out.writeString(aliasMetaData.searchRouting());
314             } else {
315                 out.writeBoolean(false);
316             }
317 
318         }

320         public static AliasMetaData readFrom(StreamInput in) throws IOException {
321             String alias = in.readString();
322             CompressedString filter = null;
323             if (in.readBoolean()) {
324                 filter = CompressedString.readCompressedString(in);
325             }
326             String indexRouting = null;
327             if (in.readBoolean()) {
328                 indexRouting = in.readString();
329             }
330             String searchRouting = null;
331             if (in.readBoolean()) {
332                 searchRouting = in.readString();
333             }
334             return new AliasMetaData(alias, filter, indexRouting, searchRouting);
335         }
