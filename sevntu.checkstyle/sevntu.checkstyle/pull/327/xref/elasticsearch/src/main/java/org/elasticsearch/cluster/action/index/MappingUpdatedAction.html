64  /**
65   * Called by shards in the cluster when their mapping was dynamically updated and it needs to be updated
66   * in the cluster state meta data (and broadcast to all members).
67   */
68  public class MappingUpdatedAction extends TransportMasterNodeOperationAction<MappingUpdatedAction.MappingUpdatedRequest, MappingUpdatedAction.MappingUpdatedResponse> {
69  
70      public static final String INDICES_MAPPING_ADDITIONAL_MAPPING_CHANGE_TIME = "indices.mapping.additional_mapping_change_time";
71      public static final String ACTION_NAME = "internal:cluster/mapping_updated";
72  
73      private final AtomicLong mappingUpdateOrderGen = new AtomicLong();
74      private final MetaDataMappingService metaDataMappingService;
75  
76      private volatile MasterMappingUpdater masterMappingUpdater;
77  
78      private volatile TimeValue additionalMappingChangeTime;
79  
80      class ApplySettings implements NodeSettingsService.Listener {
81          @Override
82          public void onRefreshSettings(Settings settings) {
83              final TimeValue current = MappingUpdatedAction.this.additionalMappingChangeTime;
84              final TimeValue newValue = settings.getAsTime(INDICES_MAPPING_ADDITIONAL_MAPPING_CHANGE_TIME, current);
85              if (!current.equals(newValue)) {
86                  logger.info("updating " + INDICES_MAPPING_ADDITIONAL_MAPPING_CHANGE_TIME + " from [{}] to [{}]", current, newValue);
87                  MappingUpdatedAction.this.additionalMappingChangeTime = newValue;
88              }
89          }
90      }
91  
92      @Inject
93      public MappingUpdatedAction(Settings settings, TransportService transportService, ClusterService clusterService, ThreadPool threadPool,
94                                  MetaDataMappingService metaDataMappingService, NodeSettingsService nodeSettingsService, ActionFilters actionFilters) {
95          super(settings, ACTION_NAME, transportService, clusterService, threadPool, actionFilters);
96          this.metaDataMappingService = metaDataMappingService;
97          // this setting should probably always be 0, just add the option to wait for more changes within a time window
98          this.additionalMappingChangeTime = settings.getAsTime(INDICES_MAPPING_ADDITIONAL_MAPPING_CHANGE_TIME, TimeValue.timeValueMillis(0));
99          nodeSettingsService.addListener(new ApplySettings());
100     }
101 
102     public void start() {
103         this.masterMappingUpdater = new MasterMappingUpdater(EsExecutors.threadName(settings, "master_mapping_updater"));
104         this.masterMappingUpdater.start();
105     }
106 
107     public void stop() {
108         this.masterMappingUpdater.close();
109         this.masterMappingUpdater = null;
110     }
111 
112     public void updateMappingOnMaster(String index, DocumentMapper documentMapper, String indexUUID) {
113         updateMappingOnMaster(index, documentMapper, indexUUID, null);
114     }
115 
116     public void updateMappingOnMaster(String index, DocumentMapper documentMapper, String indexUUID, MappingUpdateListener listener) {
117         assert !documentMapper.type().equals(MapperService.DEFAULT_MAPPING) : "_default_ mapping should not be updated";
118         masterMappingUpdater.add(new MappingChange(documentMapper, index, indexUUID, listener));
119     }

123         // internal call by other nodes, no need to check for blocks
124         return null;
125     }

143     @Override
144     protected void masterOperation(final MappingUpdatedRequest request, final ClusterState state, final ActionListener<MappingUpdatedResponse> listener) throws ElasticsearchException {
145         metaDataMappingService.updateMapping(request.index(), request.indexUUID(), request.type(), request.mappingSource(), request.order, request.nodeId, new ActionListener<ClusterStateUpdateResponse>() {
146             @Override
147             public void onResponse(ClusterStateUpdateResponse response) {
148                 listener.onResponse(new MappingUpdatedResponse());
149             }
150 
151             @Override
152             public void onFailure(Throwable t) {
153                 logger.warn("[{}] update-mapping [{}] failed to dynamically update the mapping in cluster_state from shard", t, request.index(), request.type());
154                 listener.onFailure(t);
155             }
156         });
157     }

289     /**
290      * The master mapping updater removes the overhead of refreshing the mapping (refreshSource) on the
291      * indexing thread.
292      * <p/>
293      * It also allows to reduce multiple mapping updates on the same index(UUID) and type into one update
294      * (refreshSource + sending to master), which allows to offload the number of times mappings are updated
295      * and sent to master for heavy single index requests that each introduce a new mapping, and when
296      * multiple shards exists on the same nodes, allowing to work on the index level in this case.
297      */
298     private class MasterMappingUpdater extends Thread {
299 
300         private volatile boolean running = true;
301         private final BlockingQueue<MappingChange> queue = ConcurrentCollections.newBlockingQueue();
302 
303         public MasterMappingUpdater(String name) {
304             super(name);
305         }
306 
307         public void add(MappingChange change) {
308             queue.add(change);
309         }
310 
311         public void close() {
312             running = false;
313             this.interrupt();
314         }
315 
316         class UpdateKey {
317             public final String indexUUID;
318             public final String type;
319 
320             UpdateKey(String indexUUID, String type) {
321                 this.indexUUID = indexUUID;
322                 this.type = type;
323             }
324 
325             @Override
326             public boolean equals(Object o) {
327                 if (this == o) {
328                     return true;
329                 }
330                 if (o == null || getClass() != o.getClass()) {
331                     return false;
332                 }
333 
334                 UpdateKey updateKey = (UpdateKey) o;
335 
336                 if (!indexUUID.equals(updateKey.indexUUID)) {
337                     return false;
338                 }
339                 if (!type.equals(updateKey.type)) {
340                     return false;
341                 }
342 
343                 return true;
344             }
345 
346             @Override
347             public int hashCode() {
348                 int result = indexUUID.hashCode();
349                 result = 31 * result + type.hashCode();
350                 return result;
351             }
352         }
353 
354         class UpdateValue {
355             public final MappingChange mainChange;
356             public final List<MappingUpdateListener> listeners = Lists.newArrayList();
357 
358             UpdateValue(MappingChange mainChange) {
359                 this.mainChange = mainChange;
360             }
361 
362             public void notifyListeners(@Nullable Throwable t) {
363                 for (MappingUpdateListener listener : listeners) {
364                     try {
365                         if (t == null) {
366                             listener.onMappingUpdate();
367                         } else {
368                             listener.onFailure(t);
369                         }
370                     } catch (Throwable lisFailure) {
371                         logger.warn("unexpected failure on mapping update listener callback [{}]", lisFailure, listener);
372                     }
373                 }
374             }
375         }
376 
377         @Override
378         public void run() {
379             Map<UpdateKey, UpdateValue> pendingUpdates = Maps.newHashMap();
380             while (running) {
381                 try {
382                     MappingChange polledChange = queue.poll(10, TimeUnit.MINUTES);
383                     if (polledChange == null) {
384                         continue;
385                     }
386                     List<MappingChange> changes = Lists.newArrayList(polledChange);
387                     if (additionalMappingChangeTime.millis() > 0) {
388                         Thread.sleep(additionalMappingChangeTime.millis());
389                     }
390                     queue.drainTo(changes);
391                     Collections.reverse(changes); // process then in newest one to oldest
392                     // go over and add to pending updates map
393                     for (MappingChange change : changes) {
394                         UpdateKey key = new UpdateKey(change.indexUUID, change.documentMapper.type());
395                         UpdateValue updateValue = pendingUpdates.get(key);
396                         if (updateValue == null) {
397                             updateValue = new UpdateValue(change);
398                             pendingUpdates.put(key, updateValue);
399                         }
400                         if (change.listener != null) {
401                             updateValue.listeners.add(change.listener);
402                         }
403                     }
404 
405                     for (Iterator<UpdateValue> iterator = pendingUpdates.values().iterator(); iterator.hasNext(); ) {
406                         final UpdateValue updateValue = iterator.next();
407                         iterator.remove();
408                         MappingChange change = updateValue.mainChange;
409 
410                         final MappingUpdatedAction.MappingUpdatedRequest mappingRequest;
411                         try {
412                             // we generate the order id before we get the mapping to send and refresh the source, so
413                             // if 2 happen concurrently, we know that the later order will include the previous one
414                             long orderId = mappingUpdateOrderGen.incrementAndGet();
415                             change.documentMapper.refreshSource();
416                             DiscoveryNode node = clusterService.localNode();
417                             mappingRequest = new MappingUpdatedAction.MappingUpdatedRequest(
418                                     change.index, change.indexUUID, change.documentMapper.type(), change.documentMapper.mappingSource(), orderId, node != null ? node.id() : null
419                             );
420                         } catch (Throwable t) {
421                             logger.warn("Failed to update master on updated mapping for index [" + change.index + "], type [" + change.documentMapper.type() + "]", t);
422                             updateValue.notifyListeners(t);
423                             continue;
424                         }
425                         logger.trace("sending mapping updated to master: {}", mappingRequest);
426                         execute(mappingRequest, new ActionListener<MappingUpdatedAction.MappingUpdatedResponse>() {
427                             @Override
428                             public void onResponse(MappingUpdatedAction.MappingUpdatedResponse mappingUpdatedResponse) {
429                                 logger.debug("successfully updated master with mapping update: {}", mappingRequest);
430                                 updateValue.notifyListeners(null);
431                             }
432 
433                             @Override
434                             public void onFailure(Throwable e) {
435                                 logger.warn("failed to update master on updated mapping for {}", e, mappingRequest);
436                                 updateValue.notifyListeners(e);
437                             }
438                         });
439 
440                     }
441                 } catch (Throwable t) {
442                     if (t instanceof InterruptedException && !running) {
443                         // all is well, we are shutting down
444                     } else {
445                         logger.warn("failed to process mapping updates", t);
446                     }
447                     // cleanup all pending update callbacks that were not processed due to a global failure...
448                     for (Iterator<Map.Entry<UpdateKey, UpdateValue>> iterator = pendingUpdates.entrySet().iterator(); iterator.hasNext(); ) {
449                         Map.Entry<UpdateKey, UpdateValue> entry = iterator.next();
450                         iterator.remove();
451                         entry.getValue().notifyListeners(t);
452                     }
453                 }
454             }
455         }
456     }
