20  package org.elasticsearch.common.cli;
21  
22  import com.google.common.collect.Maps;
23  import org.elasticsearch.common.settings.Settings;
24  import org.elasticsearch.env.Environment;
25  
26  import java.io.IOException;
27  import java.nio.file.Files;
28  import java.nio.file.Path;
29  import java.nio.file.attribute.PosixFileAttributeView;
30  import java.nio.file.attribute.PosixFileAttributes;
31  import java.nio.file.attribute.PosixFilePermission;
32  import java.nio.file.attribute.PosixFilePermissions;
33  import java.util.Map;
34  import java.util.Set;
35  
36  /**
37   * A helper command that checks if configured paths have been changed when running a CLI command.
38   * It is only executed in case of specified paths by the command and if the paths underlying filesystem
39   * supports posix permissions.
40   *
41   * If this is the case, a warn message is issued whenever an owner, a group or the file permissions is changed by
42   * the command being executed and not configured back to its prior state, which should be the task of the command
43   * being executed.
44   *
45   */
46  public abstract class CheckFileCommand extends CliTool.Command {
47  
48      public CheckFileCommand(Terminal terminal) {
49          super(terminal);
50      }
51  
52      /**
53       * abstract method, which should implement the same logic as CliTool.Command.execute(), but is wrapped
54       */
55      public abstract CliTool.ExitStatus doExecute(Settings settings, Environment env) throws Exception;
56  
57      /**
58       * Returns the array of paths, that should be checked if the permissions, user or groups have changed
59       * before and after execution of the command
60       *
61       */
62      protected abstract Path[] pathsForPermissionsCheck(Settings settings, Environment env) throws Exception;
63  
64      @Override
65      public CliTool.ExitStatus execute(Settings settings, Environment env) throws Exception {
66          Path[] paths = pathsForPermissionsCheck(settings, env);
67  
68          if (paths == null || paths.length == 0) {
69              return doExecute(settings, env);
70          }
71  
72          Map<Path, Set<PosixFilePermission>> permissions = Maps.newHashMapWithExpectedSize(paths.length);
73          Map<Path, String> owners = Maps.newHashMapWithExpectedSize(paths.length);
74          Map<Path, String> groups = Maps.newHashMapWithExpectedSize(paths.length);
75  
76          if (paths != null && paths.length > 0) {
77              for (Path path : paths) {
78                  try {
79                      boolean supportsPosixPermissions = Files.getFileStore(path).supportsFileAttributeView(PosixFileAttributeView.class);
80                      if (supportsPosixPermissions) {
81                          PosixFileAttributes attributes = Files.readAttributes(path, PosixFileAttributes.class);
82                          permissions.put(path, attributes.permissions());
83                          owners.put(path, attributes.owner().getName());
84                          groups.put(path, attributes.group().getName());
85                      }
86                  } catch (IOException e) {
87                      // silently swallow if not supported, no need to log things
88                  }
89              }
90          }
91  
92          CliTool.ExitStatus status = doExecute(settings, env);
93  
94          // check if permissions differ
95          for (Map.Entry<Path, Set<PosixFilePermission>> entry : permissions.entrySet()) {
96              if (!Files.exists(entry.getKey())) {
97                  continue;
98              }
99  
100             Set<PosixFilePermission> permissionsBeforeWrite = entry.getValue();
101             Set<PosixFilePermission> permissionsAfterWrite = Files.getPosixFilePermissions(entry.getKey());
102             if (!permissionsBeforeWrite.equals(permissionsAfterWrite)) {
103                 terminal.printWarn("The file permissions of [%s] have changed from [%s] to [%s]",
104                         entry.getKey(), PosixFilePermissions.toString(permissionsBeforeWrite), PosixFilePermissions.toString(permissionsAfterWrite));
105                 terminal.printWarn("Please ensure that the user account running Elasticsearch has read access to this file!");
106             }
107         }
108 
109         // check if owner differs
110         for (Map.Entry<Path, String> entry : owners.entrySet()) {
111             if (!Files.exists(entry.getKey())) {
112                 continue;
113             }
114 
115             String ownerBeforeWrite = entry.getValue();
116             String ownerAfterWrite = Files.getOwner(entry.getKey()).getName();
117             if (!ownerAfterWrite.equals(ownerBeforeWrite)) {
118                 terminal.printWarn("WARN: Owner of file [%s] used to be [%s], but now is [%s]", entry.getKey(), ownerBeforeWrite, ownerAfterWrite);
119             }
120         }
121 
122         // check if group differs
123         for (Map.Entry<Path, String> entry : groups.entrySet()) {
124             if (!Files.exists(entry.getKey())) {
125                 continue;
126             }
127 
128             String groupBeforeWrite = entry.getValue();
129             String groupAfterWrite = Files.readAttributes(entry.getKey(), PosixFileAttributes.class).group().getName();
130             if (!groupAfterWrite.equals(groupBeforeWrite)) {
131                 terminal.printWarn("WARN: Group of file [%s] used to be [%s], but now is [%s]", entry.getKey(), groupBeforeWrite, groupAfterWrite);
132             }
133         }
134 
135         return status;
136     }
137 }
