93       * Returns the binding for {@code key}
94       */
95      @Override
96      public <T> BindingImpl<T> getBinding(Key<T> key) {
97          Errors errors = new Errors(key);
98          try {
99              BindingImpl<T> result = getBindingOrThrow(key, errors);
100             errors.throwConfigurationExceptionIfErrorsExist();
101             return result;
102         } catch (ErrorsException e) {
103             throw new ConfigurationException(errors.merge(e.getErrors()).getMessages());
104         }

107     /**
108      * Gets a binding implementation.  First, it check to see if the parent has a binding.  If the
109      * parent has a binding and the binding is scoped, it will use that binding.  Otherwise, this
110      * checks for an explicit binding. If no explicit binding is found, it looks for a just-in-time
111      * binding.
112      */
113     public <T> BindingImpl<T> getBindingOrThrow(Key<T> key, Errors errors)
114             throws ErrorsException {
115         // Check explicit bindings, i.e. bindings created by modules.
116         BindingImpl<T> binding = state.getExplicitBinding(key);
117         if (binding != null) {
118             return binding;
119         }
120 
121         // Look for an on-demand binding.
122         return getJustInTimeBinding(key, errors);
123     }

148     /**
149      * Returns a just-in-time binding for {@code key}, creating it if necessary.
150      *
151      * @throws ErrorsException if the binding could not be created.
152      */
153     private <T> BindingImpl<T> getJustInTimeBinding(Key<T> key, Errors errors)
154             throws ErrorsException {
155         synchronized (state.lock()) {
156             // first try to find a JIT binding that we've already created
157             for (InjectorImpl injector = this; injector != null; injector = injector.parent) {
158                 @SuppressWarnings("unchecked") // we only store bindings that match their key
159                         BindingImpl<T> binding = (BindingImpl<T>) injector.jitBindings.get(key);
160 
161                 if (binding != null) {
162                     return binding;
163                 }
164             }
165 
166             return createJustInTimeBindingRecursive(key, errors);
167         }
168     }

186             Key<MembersInjector<T>> key, Errors errors) throws ErrorsException {
187         Type membersInjectorType = key.getTypeLiteral().getType();
188         if (!(membersInjectorType instanceof ParameterizedType)) {
189             throw errors.cannotInjectRawMembersInjector().toException();
190         }
191 
192         @SuppressWarnings("unchecked") // safe because T came from Key<MembersInjector<T>>
193                 TypeLiteral<T> instanceType = (TypeLiteral<T>) TypeLiteral.get(
194                 ((ParameterizedType) membersInjectorType).getActualTypeArguments()[0]);
195         MembersInjector<T> membersInjector = membersInjectorStore.get(instanceType, errors);
196 
197         InternalFactory<MembersInjector<T>> factory = new ConstantFactory<>(
198                 Initializables.of(membersInjector));
199 
200 
201         return new InstanceBindingImpl<>(this, key, SourceProvider.UNKNOWN_SOURCE,
202                 factory, ImmutableSet.<InjectionPoint>of(), membersInjector);
203     }

205     /**
206      * Creates a synthetic binding to {@code Provider<T>}, i.e. a binding to the provider from
207      * {@code Binding<T>}.
208      */
209     private <T> BindingImpl<Provider<T>> createProviderBinding(Key<Provider<T>> key, Errors errors)
210             throws ErrorsException {
211         Type providerType = key.getTypeLiteral().getType();
212 
213         // If the Provider has no type parameter (raw Provider)...
214         if (!(providerType instanceof ParameterizedType)) {
215             throw errors.cannotInjectRawProvider().toException();
216         }
217 
218         Type entryType = ((ParameterizedType) providerType).getActualTypeArguments()[0];
219 
220         @SuppressWarnings("unchecked") // safe because T came from Key<Provider<T>>
221                 Key<T> providedKey = (Key<T>) key.ofType(entryType);
222 
223         BindingImpl<T> delegate = getBindingOrThrow(providedKey, errors);
224         return new ProviderBindingImpl<>(this, key, delegate);
225     }

271     /**
272      * Converts a constant string binding to the required type.
273      *
274      * @return the binding if it could be resolved, or null if the binding doesn't exist
275      * @throws org.elasticsearch.common.inject.internal.ErrorsException
276      *          if there was an error resolving the binding
277      */
278     private <T> BindingImpl<T> convertConstantStringBinding(Key<T> key, Errors errors)
279             throws ErrorsException {
280         // Find a constant string binding.
281         Key<String> stringKey = key.ofType(String.class);
282         BindingImpl<String> stringBinding = state.getExplicitBinding(stringKey);
283         if (stringBinding == null || !stringBinding.isConstant()) {
284             return null;
285         }
286 
287         String stringValue = stringBinding.getProvider().get();
288         Object source = stringBinding.getSource();
289 
290         // Find a matching type converter.
291         TypeLiteral<T> type = key.getTypeLiteral();
292         MatcherAndConverter matchingConverter = state.getConverter(stringValue, type, errors, source);
293 
294         if (matchingConverter == null) {
295             // No converter can handle the given type.
296             return null;
297         }
298 
299         // Try to convert the string. A failed conversion results in an error.
300         try {
301             @SuppressWarnings("unchecked") // This cast is safe because we double check below.
302                     T converted = (T) matchingConverter.getTypeConverter().convert(stringValue, type);
303 
304             if (converted == null) {
305                 throw errors.converterReturnedNull(stringValue, source, type, matchingConverter)
306                         .toException();
307             }
308 
309             if (!type.getRawType().isInstance(converted)) {
310                 throw errors.conversionTypeError(stringValue, source, type, matchingConverter, converted)
311                         .toException();
312             }
313 
314             return new ConvertedConstantBindingImpl<>(this, key, converted, stringBinding);
315         } catch (ErrorsException e) {
316             throw e;
317         } catch (RuntimeException e) {
318             throw errors.conversionError(stringValue, source, type, matchingConverter, e)
319                     .toException();
320         }
321     }

378     <T> void initializeBinding(BindingImpl<T> binding, Errors errors) throws ErrorsException {
379         // Put the partially constructed binding in the map a little early. This enables us to handle
380         // circular dependencies. Example: FooImpl -> BarImpl -> FooImpl.
381         // Note: We don't need to synchronize on state.lock() during injector creation.
382         // TODO: for the above example, remove the binding for BarImpl if the binding for FooImpl fails
383         if (binding instanceof ConstructorBindingImpl<?>) {
384             Key<T> key = binding.getKey();
385             jitBindings.put(key, binding);
386             boolean successful = false;
387             try {
388                 ((ConstructorBindingImpl) binding).initialize(this, errors);
389                 successful = true;
390             } finally {
391                 if (!successful) {
392                     jitBindings.remove(key);
393                 }
394             }
395         }
396     }

398     /**
399      * Creates a binding for an injectable type with the given scope. Looks for a scope on the type if
400      * none is specified.
401      */
402     <T> BindingImpl<T> createUnitializedBinding(Key<T> key, Scoping scoping, Object source,
403                                                 Errors errors) throws ErrorsException {
404         Class<?> rawType = key.getTypeLiteral().getRawType();
405 
406         // Don't try to inject arrays, or enums.
407         if (rawType.isArray() || rawType.isEnum()) {
408             throw errors.missingImplementation(key).toException();
409         }
410 
411         // Handle TypeLiteral<T> by binding the inner type
412         if (rawType == TypeLiteral.class) {
413             @SuppressWarnings("unchecked") // we have to fudge the inner type as Object
414                     BindingImpl<T> binding = (BindingImpl<T>) createTypeLiteralBinding(
415                     (Key<TypeLiteral<Object>>) key, errors);
416             return binding;
417         }
418 
419         // Handle @ImplementedBy
420         ImplementedBy implementedBy = rawType.getAnnotation(ImplementedBy.class);
421         if (implementedBy != null) {
422             Annotations.checkForMisplacedScopeAnnotations(rawType, source, errors);
423             return createImplementedByBinding(key, scoping, implementedBy, errors);
424         }
425 
426         // Handle @ProvidedBy.
427         ProvidedBy providedBy = rawType.getAnnotation(ProvidedBy.class);
428         if (providedBy != null) {
429             Annotations.checkForMisplacedScopeAnnotations(rawType, source, errors);
430             return createProvidedByBinding(key, scoping, providedBy, errors);
431         }
432 
433         // We can't inject abstract classes.
434         // TODO: Method interceptors could actually enable us to implement
435         // abstract types. Should we remove this restriction?
436         if (Modifier.isAbstract(rawType.getModifiers())) {
437             throw errors.missingImplementation(key).toException();
438         }
439 
440         // Error: Inner class.
441         if (Classes.isInnerClass(rawType)) {
442             throw errors.cannotInjectInnerClass(rawType).toException();
443         }
444 
445         if (!scoping.isExplicitlyScoped()) {
446             Class<? extends Annotation> scopeAnnotation = findScopeAnnotation(errors, rawType);
447             if (scopeAnnotation != null) {
448                 scoping = Scopes.makeInjectable(Scoping.forAnnotation(scopeAnnotation),
449                         this, errors.withSource(rawType));
450             }
451         }
452 
453         return ConstructorBindingImpl.create(this, key, source, scoping);
454     }

456     /**
457      * Converts a binding for a {@code Key<TypeLiteral<T>>} to the value {@code TypeLiteral<T>}. It's
458      * a bit awkward because we have to pull out the inner type in the type literal.
459      */
460     private <T> BindingImpl<TypeLiteral<T>> createTypeLiteralBinding(
461             Key<TypeLiteral<T>> key, Errors errors) throws ErrorsException {
462         Type typeLiteralType = key.getTypeLiteral().getType();
463         if (!(typeLiteralType instanceof ParameterizedType)) {
464             throw errors.cannotInjectRawTypeLiteral().toException();
465         }
466 
467         ParameterizedType parameterizedType = (ParameterizedType) typeLiteralType;
468         Type innerType = parameterizedType.getActualTypeArguments()[0];
469 
470         // this is unforunate. We don't support building TypeLiterals for type variable like 'T'. If
471         // this proves problematic, we can probably fix TypeLiteral to support type variables
472         if (!(innerType instanceof Class)
473                 && !(innerType instanceof GenericArrayType)
474                 && !(innerType instanceof ParameterizedType)) {
475             throw errors.cannotInjectTypeLiteralOf(innerType).toException();
476         }
477 
478         @SuppressWarnings("unchecked") // by definition, innerType == T, so this is safe
479                 TypeLiteral<T> value = (TypeLiteral<T>) TypeLiteral.get(innerType);
480         InternalFactory<TypeLiteral<T>> factory = new ConstantFactory<>(
481                 Initializables.of(value));
482         return new InstanceBindingImpl<>(this, key, SourceProvider.UNKNOWN_SOURCE,
483                 factory, ImmutableSet.<InjectionPoint>of(), value);
484     }

486     /**
487      * Creates a binding for a type annotated with @ProvidedBy.
488      */
489     <T> BindingImpl<T> createProvidedByBinding(Key<T> key, Scoping scoping,
490                                                ProvidedBy providedBy, Errors errors) throws ErrorsException {
491         final Class<?> rawType = key.getTypeLiteral().getRawType();
492         final Class<? extends Provider<?>> providerType = providedBy.value();
493 
494         // Make sure it's not the same type. TODO: Can we check for deeper loops?
495         if (providerType == rawType) {
496             throw errors.recursiveProviderType().toException();
497         }
498 
499         // Assume the provider provides an appropriate type. We double check at runtime.
500         @SuppressWarnings("unchecked")
501         final Key<? extends Provider<T>> providerKey
502                 = (Key<? extends Provider<T>>) Key.get(providerType);
503         final BindingImpl<? extends Provider<?>> providerBinding
504                 = getBindingOrThrow(providerKey, errors);
505 
506         InternalFactory<T> internalFactory = new InternalFactory<T>() {
507             @Override
508             public T get(Errors errors, InternalContext context, Dependency dependency)
509                     throws ErrorsException {
510                 errors = errors.withSource(providerKey);
511                 Provider<?> provider = providerBinding.getInternalFactory().get(
512                         errors, context, dependency);
513                 try {
514                     Object o = provider.get();
515                     if (o != null && !rawType.isInstance(o)) {
516                         throw errors.subtypeNotProvided(providerType, rawType).toException();
517                     }
518                     @SuppressWarnings("unchecked") // protected by isInstance() check above
519                             T t = (T) o;
520                     return t;
521                 } catch (RuntimeException e) {
522                     throw errors.errorInProvider(e).toException();
523                 }
524             }
525         };
526 
527         return new LinkedProviderBindingImpl<>(
528                 this,
529                 key,
530                 rawType /* source */,
531                 Scopes.<T>scope(key, this, internalFactory, scoping),
532                 scoping,
533                 providerKey);
534     }

536     /**
537      * Creates a binding for a type annotated with @ImplementedBy.
538      */
539     <T> BindingImpl<T> createImplementedByBinding(Key<T> key, Scoping scoping,
540                                                   ImplementedBy implementedBy, Errors errors)
541             throws ErrorsException {
542         Class<?> rawType = key.getTypeLiteral().getRawType();
543         Class<?> implementationType = implementedBy.value();
544 
545         // Make sure it's not the same type. TODO: Can we check for deeper cycles?
546         if (implementationType == rawType) {
547             throw errors.recursiveImplementationType().toException();
548         }
549 
550         // Make sure implementationType extends type.
551         if (!rawType.isAssignableFrom(implementationType)) {
552             throw errors.notASubtype(implementationType, rawType).toException();
553         }
554 
555         @SuppressWarnings("unchecked") // After the preceding check, this cast is safe.
556                 Class<? extends T> subclass = (Class<? extends T>) implementationType;
557 
558         // Look up the target binding.
559         final Key<? extends T> targetKey = Key.get(subclass);
560         final BindingImpl<? extends T> targetBinding = getBindingOrThrow(targetKey, errors);
561 
562         InternalFactory<T> internalFactory = new InternalFactory<T>() {
563             @Override
564             public T get(Errors errors, InternalContext context, Dependency<?> dependency)
565                     throws ErrorsException {
566                 return targetBinding.getInternalFactory().get(
567                         errors.withSource(targetKey), context, dependency);
568             }
569         };
570 
571         return new LinkedBindingImpl<>(
572                 this,
573                 key,
574                 rawType /* source */,
575                 Scopes.<T>scope(key, this, internalFactory, scoping),
576                 scoping,
577                 targetKey);
578     }

580     /**
581      * Attempts to create a just-in-time binding for {@code key} in the root injector, falling back to
582      * other ancestor injectors until this injector is tried.
583      */
584     private <T> BindingImpl<T> createJustInTimeBindingRecursive(Key<T> key, Errors errors)
585             throws ErrorsException {
586         // ask the parent to create the JIT binding
587         if (parent != null && !parent.readOnly /* ES: don't check on parent if its read only, its already created all the bindings it can*/) {
588             try {
589                 return parent.createJustInTimeBindingRecursive(key, new Errors());
590             } catch (ErrorsException ignored) {
591             }
592         }
593 
594         if (state.isBlacklisted(key)) {
595             throw errors.childBindingAlreadySet(key).toException();
596         }
597 
598         BindingImpl<T> binding = createJustInTimeBinding(key, errors);
599         state.parent().blacklist(key);
600         jitBindings.put(key, binding);
601         return binding;
602     }

604     /**
605      * Returns a new just-in-time binding created by resolving {@code key}. The strategies used to
606      * create just-in-time bindings are:
607      * <ol>
608      * <li>Internalizing Providers. If the requested binding is for {@code Provider<T>}, we delegate
609      * to the binding for {@code T}.
610      * <li>Converting constants.
611      * <li>ImplementedBy and ProvidedBy annotations. Only for unannotated keys.
612      * <li>The constructor of the raw type. Only for unannotated keys.
613      * </ol>
614      *
615      * @throws org.elasticsearch.common.inject.internal.ErrorsException
616      *          if the binding cannot be created.
617      */
618     <T> BindingImpl<T> createJustInTimeBinding(Key<T> key, Errors errors) throws ErrorsException {
619         if (state.isBlacklisted(key)) {
620             throw errors.childBindingAlreadySet(key).toException();
621         }
622 
623         // Handle cases where T is a Provider<?>.
624         if (isProvider(key)) {
625             // These casts are safe. We know T extends Provider<X> and that given Key<Provider<X>>,
626             // createProviderBinding() will return BindingImpl<Provider<X>>.
627             @SuppressWarnings("unchecked")
628             BindingImpl binding = createProviderBinding((Key) key, errors);
629             return binding;
630         }
631 
632         // Handle cases where T is a MembersInjector<?>
633         if (isMembersInjector(key)) {
634             // These casts are safe. T extends MembersInjector<X> and that given Key<MembersInjector<X>>,
635             // createMembersInjectorBinding() will return BindingImpl<MembersInjector<X>>.
636             @SuppressWarnings("unchecked")
637             BindingImpl binding = createMembersInjectorBinding((Key) key, errors);
638             return binding;
639         }
640 
641         // Try to convert a constant string binding to the requested type.
642         BindingImpl<T> convertedBinding = convertConstantStringBinding(key, errors);
643         if (convertedBinding != null) {
644             return convertedBinding;
645         }
646 
647         // If the key has an annotation...
648         if (key.hasAnnotationType()) {
649             // Look for a binding without annotation attributes or return null.
650             if (key.hasAttributes()) {
651                 try {
652                     Errors ignored = new Errors();
653                     return getBindingOrThrow(key.withoutAttributes(), ignored);
654                 } catch (ErrorsException ignored) {
655                     // throw with a more appropriate message below
656                 }
657             }
658             throw errors.missingImplementation(key).toException();
659         }
660 
661         Object source = key.getTypeLiteral().getRawType();
662         BindingImpl<T> binding = createUnitializedBinding(key, Scoping.UNSCOPED, source, errors);
663         initializeBinding(binding, errors);
664         return binding;
665     }

678     private static class BindingsMultimap {
679         final Map<TypeLiteral<?>, List<Binding<?>>> multimap = Maps.newHashMap();

681         <T> void put(TypeLiteral<T> type, Binding<T> binding) {

701     /**
702      * Returns parameter injectors, or {@code null} if there are no parameters.
703      */
704     SingleParameterInjector<?>[] getParametersInjectors(
705             List<Dependency<?>> parameters, Errors errors) throws ErrorsException {
706         if (parameters.isEmpty()) {
707             return null;
708         }
709 
710         int numErrorsBefore = errors.size();
711         SingleParameterInjector<?>[] result = new SingleParameterInjector<?>[parameters.size()];
712         int i = 0;
713         for (Dependency<?> parameter : parameters) {
714             try {
715                 result[i++] = createParameterInjector(parameter, errors.withSource(parameter));
716             } catch (ErrorsException rethrownBelow) {
717                 // rethrown below
718             }
719         }
720 
721         errors.throwIfNewErrors(numErrorsBefore);
722         return result;
723     }

776     <T> Provider<T> getProviderOrThrow(final Key<T> key, Errors errors) throws ErrorsException {
777         final InternalFactory<? extends T> factory = getInternalFactory(key, errors);
778         // ES: optimize for a common case of read only instance getting from the parent...
779         if (factory instanceof InternalFactory.Instance) {
780             return new Provider<T>() {
781                 @Override
782                 public T get() {
783                     try {
784                         return (T) ((InternalFactory.Instance) factory).get(null, null, null);
785                     } catch (ErrorsException e) {
786                         // ignore
787                     }
788                     // should never happen...
789                     assert false;
790                     return null;
791                 }
792             };
793         }
794 
795         final Dependency<T> dependency = Dependency.get(key);
796         return new Provider<T>() {
797             @Override
798             public T get() {
799                 final Errors errors = new Errors(dependency);
800                 try {
801                     T t = callInContext(new ContextualCallable<T>() {
802                         @Override
803                         public T call(InternalContext context) throws ErrorsException {
804                             context.setDependency(dependency);
805                             try {
806                                 return factory.get(errors, context, dependency);
807                             } finally {
808                                 context.setDependency(null);
809                             }
810                         }
811                     });
812                     errors.throwIfNewErrors(0);
813                     return t;
814                 } catch (ErrorsException e) {
815                     throw new ProvisionException(errors.merge(e.getErrors()).getMessages());
816                 }
817             }
818 
819             @Override
820             public String toString() {
821                 return factory.toString();
822             }
823         };
824     }

826     @Override

848     final ThreadLocal<Object[]> localContext;
849 
850     /**
851      * Looks up thread local context. Creates (and removes) a new context if necessary.
852      */
853     <T> T callInContext(ContextualCallable<T> callable) throws ErrorsException {
854         Object[] reference = localContext.get();
855         if (reference[0] == null) {
856             reference[0] = new InternalContext();
857             try {
858                 return callable.call((InternalContext) reference[0]);
859             } finally {
860                 // Only clear the context if this call created it.
861                 reference[0] = null;
862             }
863         } else {
864             // Someone else will clean up this context.
865             return callable.call((InternalContext) reference[0]);
866         }
867     }
