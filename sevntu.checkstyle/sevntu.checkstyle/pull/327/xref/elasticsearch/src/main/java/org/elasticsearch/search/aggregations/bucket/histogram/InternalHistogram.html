66      private final static BucketStreams.Stream<Bucket> BUCKET_STREAM = new BucketStreams.Stream<Bucket>() {
67          @Override
68          public Bucket readResult(StreamInput in, BucketStreamContext context) throws IOException {
69              Factory<?> factory = (Factory<?>) context.attributes().get("factory");
70              if (factory == null) {
71                  throw new ElasticsearchIllegalStateException("No factory found for histogram buckets");
72              }
73              Bucket histogram = new Bucket(context.keyed(), context.formatter(), factory);
74              histogram.readFrom(in);
75              return histogram;
76          }

80              BucketStreamContext context = new BucketStreamContext();
81              context.formatter(bucket.formatter);

83              return context;
84          }

140         <B extends Bucket> B reduce(List<B> buckets, ReduceContext context) {
141             List<InternalAggregations> aggregations = new ArrayList<>(buckets.size());
142             long docCount = 0;
143             for (Bucket bucket : buckets) {
144                 docCount += bucket.docCount;
145                 aggregations.add((InternalAggregations) bucket.getAggregations());
146             }
147             InternalAggregations aggs = InternalAggregations.reduce(aggregations, context);
148             return (B) getFactory().createBucket(key, docCount, aggs, keyed, formatter);
149         }
150 
151         @Override
152         public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
153             if (formatter != null && formatter != ValueFormatter.RAW) {
154                 Text keyTxt = new StringText(formatter.format(key));
155                 if (keyed) {
156                     builder.startObject(keyTxt.string());
157                 } else {
158                     builder.startObject();
159                 }
160                 builder.field(CommonFields.KEY_AS_STRING, keyTxt);
161             } else {
162                 if (keyed) {
163                     builder.startObject(String.valueOf(getKey()));
164                 } else {
165                     builder.startObject();
166                 }
167             }
168             builder.field(CommonFields.KEY, key);
169             builder.field(CommonFields.DOC_COUNT, docCount);
170             aggregations.toXContentInternal(builder, params);
171             builder.endObject();
172             return builder;
173         }
174 
175         @Override
176         public void readFrom(StreamInput in) throws IOException {
177             key = in.readLong();
178             docCount = in.readVLong();
179             aggregations = InternalAggregations.readAggregations(in);
180         }
181 
182         @Override
183         public void writeTo(StreamOutput out) throws IOException {
184             out.writeLong(key);
185             out.writeVLong(docCount);
186             aggregations.writeTo(out);
187         }

190     static class EmptyBucketInfo {
191 
192         final Rounding rounding;
193         final InternalAggregations subAggregations;
194         final ExtendedBounds bounds;
195 
196         EmptyBucketInfo(Rounding rounding, InternalAggregations subAggregations) {
197             this(rounding, subAggregations, null);
198         }
199 
200         EmptyBucketInfo(Rounding rounding, InternalAggregations subAggregations, ExtendedBounds bounds) {
201             this.rounding = rounding;
202             this.subAggregations = subAggregations;
203             this.bounds = bounds;
204         }
205 
206         public static EmptyBucketInfo readFrom(StreamInput in) throws IOException {
207             Rounding rounding = Rounding.Streams.read(in);
208             InternalAggregations aggs = InternalAggregations.readAggregations(in);
209             if (in.readBoolean()) {
210                 return new EmptyBucketInfo(rounding, aggs, ExtendedBounds.readFrom(in));
211             }
212             return new EmptyBucketInfo(rounding, aggs);
213         }
214 
215         public static void writeTo(EmptyBucketInfo info, StreamOutput out) throws IOException {
216             Rounding.Streams.write(info.rounding, out);
217             info.subAggregations.writeTo(out);
218             out.writeBoolean(info.bounds != null);
219             if (info.bounds != null) {
220                 info.bounds.writeTo(out);
221             }
222         }

288     private static class IteratorAndCurrent<B> {
289 
290         private final Iterator<B> iterator;
291         private B current;
292 
293         IteratorAndCurrent(Iterator<B> iterator) {
294             this.iterator = iterator;
295             current = iterator.next();
296         }
297 
298     }
299 
300     private List<B> reduceBuckets(ReduceContext reduceContext) {
301         List<InternalAggregation> aggregations = reduceContext.aggregations();
302 
303         final PriorityQueue<IteratorAndCurrent<B>> pq = new PriorityQueue<IteratorAndCurrent<B>>(aggregations.size()) {
304             @Override
305             protected boolean lessThan(IteratorAndCurrent<B> a, IteratorAndCurrent<B> b) {
306                 return a.current.key < b.current.key;
307             }
308         };
309         for (InternalAggregation aggregation : aggregations) {
310             InternalHistogram<B> histogram = (InternalHistogram) aggregation;
311             if (histogram.buckets.isEmpty() == false) {
312                 pq.add(new IteratorAndCurrent<>(histogram.buckets.iterator()));
313             }
314         }
315 
316         List<B> reducedBuckets = new ArrayList<>();
317         if (pq.size() > 0) {
318             // list of buckets coming from different shards that have the same key
319             List<B> currentBuckets = new ArrayList<>();
320             long key = pq.top().current.key;
321 
322             do {
323                 final IteratorAndCurrent<B> top = pq.top();
324 
325                 if (top.current.key != key) {
326                     // the key changes, reduce what we already buffered and reset the buffer for current buckets
327                     final B reduced = currentBuckets.get(0).reduce(currentBuckets, reduceContext);
328                     if (reduced.getDocCount() >= minDocCount) {
329                         reducedBuckets.add(reduced);
330                     }
331                     currentBuckets.clear();
332                     key = top.current.key;
333                 }
334 
335                 currentBuckets.add(top.current);
336 
337                 if (top.iterator.hasNext()) {
338                     final B next = top.iterator.next();
339                     assert next.key > top.current.key : "shards must return data sorted by key";
340                     top.current = next;
341                     pq.updateTop();
342                 } else {
343                     pq.pop();
344                 }
345             } while (pq.size() > 0);
346 
347             if (currentBuckets.isEmpty() == false) {
348                 final B reduced = currentBuckets.get(0).reduce(currentBuckets, reduceContext);
349                 if (reduced.getDocCount() >= minDocCount) {
350                     reducedBuckets.add(reduced);
351                 }
352             }
353         }
354 
355         return reducedBuckets;
356     }

358     private void addEmptyBuckets(List<B> list) {
359         B lastBucket = null;
360         ExtendedBounds bounds = emptyBucketInfo.bounds;
361         ListIterator<B> iter = list.listIterator();
362 
363         // first adding all the empty buckets *before* the actual data (based on th extended_bounds.min the user requested)
364         if (bounds != null) {
365             B firstBucket = iter.hasNext() ? list.get(iter.nextIndex()) : null;
366             if (firstBucket == null) {
367                 if (bounds.min != null && bounds.max != null) {
368                     long key = bounds.min;
369                     long max = bounds.max;
370                     while (key <= max) {
371                         iter.add(getFactory().createBucket(key, 0, emptyBucketInfo.subAggregations, keyed, formatter));
372                         key = emptyBucketInfo.rounding.nextRoundingValue(key);
373                     }
374                 }
375             } else {
376                 if (bounds.min != null) {
377                     long key = bounds.min;
378                     if (key < firstBucket.key) {
379                         while (key < firstBucket.key) {
380                             iter.add(getFactory().createBucket(key, 0, emptyBucketInfo.subAggregations, keyed, formatter));
381                             key = emptyBucketInfo.rounding.nextRoundingValue(key);
382                         }
383                     }
384                 }
385             }
386         }
387 
388         // now adding the empty buckets within the actual data,
389         // e.g. if the data series is [1,2,3,7] there're 3 empty buckets that will be created for 4,5,6
390         while (iter.hasNext()) {
391             B nextBucket = list.get(iter.nextIndex());
392             if (lastBucket != null) {
393                 long key = emptyBucketInfo.rounding.nextRoundingValue(lastBucket.key);
394                 while (key < nextBucket.key) {
395                     iter.add(getFactory().createBucket(key, 0, emptyBucketInfo.subAggregations, keyed, formatter));
396                     key = emptyBucketInfo.rounding.nextRoundingValue(key);
397                 }
398                 assert key == nextBucket.key;
399             }
400             lastBucket = iter.next();
401         }
402 
403         // finally, adding the empty buckets *after* the actual data (based on the extended_bounds.max requested by the user)
404         if (bounds != null && lastBucket != null && bounds.max != null && bounds.max > lastBucket.key) {
405             long key = emptyBucketInfo.rounding.nextRoundingValue(lastBucket.key);
406             long max = bounds.max;
407             while (key <= max) {
408                 iter.add(getFactory().createBucket(key, 0, emptyBucketInfo.subAggregations, keyed, formatter));
409                 key = emptyBucketInfo.rounding.nextRoundingValue(key);
410             }
411         }
412     }

414     @Override
415     public InternalAggregation reduce(ReduceContext reduceContext) {
416         List<B> reducedBuckets = reduceBuckets(reduceContext);
417 
418         // adding empty buckets if needed
419         if (minDocCount == 0) {
420             addEmptyBuckets(reducedBuckets);
421         }
422 
423         if (order == InternalOrder.KEY_ASC) {
424             // nothing to do, data are already sorted since shards return
425             // sorted buckets and the merge-sort performed by reduceBuckets
426             // maintains order
427         } else if (order == InternalOrder.KEY_DESC) {
428             // we just need to reverse here...
429             reducedBuckets = Lists.reverse(reducedBuckets);
430         } else {
431             // sorted by sub-aggregation, need to fall back to a costly n*log(n) sort
432             CollectionUtil.introSort(reducedBuckets, order.comparator());
433         }
434 
435         return getFactory().create(getName(), reducedBuckets, order, minDocCount, emptyBucketInfo, formatter, keyed, getMetaData());
436     }

438     @Override
439     protected void doReadFrom(StreamInput in) throws IOException {
440         this.factory = resolveFactory(in.readString());
441         order = InternalOrder.Streams.readOrder(in);
442         minDocCount = in.readVLong();
443         if (minDocCount == 0) {
444             emptyBucketInfo = EmptyBucketInfo.readFrom(in);
445         }
446         formatter = ValueFormatterStreams.readOptional(in);
447         keyed = in.readBoolean();
448         int size = in.readVInt();
449         List<B> buckets = new ArrayList<>(size);
450         for (int i = 0; i < size; i++) {
451             B bucket = getFactory().createEmptyBucket(keyed, formatter);
452             bucket.readFrom(in);
453             buckets.add(bucket);
454         }
455         this.buckets = buckets;
456     }

458     @SuppressWarnings("unchecked")
459     private static <B extends InternalHistogram.Bucket> Factory<B> resolveFactory(String factoryType) {
460         if (factoryType.equals(InternalDateHistogram.TYPE.name())) {
461             return (Factory<B>) new InternalDateHistogram.Factory();
462         } else if (factoryType.equals(TYPE.name())) {
463             return new Factory<>();
464         } else {
465             throw new ElasticsearchIllegalStateException("Invalid histogram factory type [" + factoryType + "]");
466         }
467     }
468 
469     @Override
470     protected void doWriteTo(StreamOutput out) throws IOException {
471         out.writeString(factory.type());
472         InternalOrder.Streams.writeOrder(order, out);
473         out.writeVLong(minDocCount);
474         if (minDocCount == 0) {
475             EmptyBucketInfo.writeTo(emptyBucketInfo, out);
476         }
477         ValueFormatterStreams.writeOptional(formatter, out);
478         out.writeBoolean(keyed);
479         out.writeVInt(buckets.size());
480         for (B bucket : buckets) {
481             bucket.writeTo(out);
482         }
483     }

485     @Override
486     public XContentBuilder doXContentBody(XContentBuilder builder, Params params) throws IOException {
487         if (keyed) {
488             builder.startObject(CommonFields.BUCKETS);
489         } else {
490             builder.startArray(CommonFields.BUCKETS);
491         }
492         for (B bucket : buckets) {
493             bucket.toXContent(builder, params);
494         }
495         if (keyed) {
496             builder.endObject();
497         } else {
498             builder.endArray();
499         }
500         return builder;
501     }
