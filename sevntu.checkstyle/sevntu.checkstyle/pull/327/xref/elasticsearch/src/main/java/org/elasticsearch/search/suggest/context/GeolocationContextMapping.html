48  /**
49   * The {@link GeolocationContextMapping} allows to take GeoInfomation into account
50   * during building suggestions. The mapping itself works with geohashes
51   * explicitly and is configured by three parameters:
52   * <ul>
53   * <li><code>precision</code>: length of the geohash indexed as prefix of the
54   * completion field</li>
55   * <li><code>neighbors</code>: Should the neighbor cells of the deepest geohash
56   * level also be indexed as alternatives to the actual geohash</li>
57   * <li><code>location</code>: (optional) location assumed if it is not provided</li>
58   * </ul>
59   * Internally this mapping wraps the suggestions into a form
60   * <code>[geohash][suggestion]</code>. If the neighbor option is set the cells
61   * next to the cell on the deepest geohash level ( <code>precision</code>) will
62   * be indexed as well. The {@link TokenStream} used to build the {@link FST} for
63   * suggestion will be wrapped into a {@link PrefixTokenFilter} managing these
64   * geohases as prefixes.
65   */
66  public class GeolocationContextMapping extends ContextMapping {
67

91          super(TYPE, name);

96          this.defaultConfig = new GeoConfig(this, defaultLocations);

99      /**
100      * load a {@link GeolocationContextMapping} by configuration. Such a configuration
101      * can set the parameters
102      * <ul>
103      * <li>precision [<code>String</code>, <code>Double</code>,
104      * <code>Float</code> or <code>Integer</code>] defines the length of the
105      * underlying geohash</li>
106      * <li>defaultLocation [<code>String</code>] defines the location to use if
107      * it is not provided by the document</li>
108      * <li>neighbors [<code>Boolean</code>] defines if the last level of the
109      * geohash should be extended by neighbor cells</li>
110      * </ul>
111      * 
112      * @param config
113      *            Configuration for {@link GeolocationContextMapping}
114      * @return new {@link GeolocationContextMapping} configured by the parameters of
115      *         <code>config</code>
116      */
117     protected static GeolocationContextMapping load(String name, Map<String, Object> config) {
118         if (!config.containsKey(FIELD_PRECISION)) {
119             throw new ElasticsearchParseException("field [precision] is missing");
120         }
121 
122         final GeolocationContextMapping.Builder builder = new GeolocationContextMapping.Builder(name);
123 
124         if (config != null) {
125             final Object configPrecision = config.get(FIELD_PRECISION);
126             if (configPrecision == null) {
127                 // ignore precision
128             } else if (configPrecision instanceof Integer) {
129                 builder.precision((Integer) configPrecision);
130                 config.remove(FIELD_PRECISION);
131             } else if (configPrecision instanceof Long) {
132                 builder.precision((Long) configPrecision);
133                 config.remove(FIELD_PRECISION);
134             } else if (configPrecision instanceof Double) {
135                 builder.precision((Double) configPrecision);
136                 config.remove(FIELD_PRECISION);
137             } else if (configPrecision instanceof Float) {
138                 builder.precision((Float) configPrecision);
139                 config.remove(FIELD_PRECISION);
140             } else if (configPrecision instanceof Iterable) {
141                 for (Object precision : (Iterable)configPrecision) {
142                     if (precision instanceof Integer) {
143                         builder.precision((Integer) precision);
144                     } else if (precision instanceof Long) {
145                         builder.precision((Long) precision);
146                     } else if (precision instanceof Double) {
147                         builder.precision((Double) precision);
148                     } else if (precision instanceof Float) {
149                         builder.precision((Float) precision);
150                     } else {
151                         builder.precision(precision.toString());
152                     }
153                 }
154                 config.remove(FIELD_PRECISION);
155             } else {
156                 builder.precision(configPrecision.toString());
157                 config.remove(FIELD_PRECISION);
158             }
159 
160             final Object configNeighbors = config.get(FIELD_NEIGHBORS);
161             if (configNeighbors != null) {
162                 builder.neighbors((Boolean) configNeighbors);
163                 config.remove(FIELD_NEIGHBORS);
164             }
165 
166             final Object def = config.get(FIELD_MISSING);
167             if (def != null) {
168                 if (def instanceof Iterable) {
169                     for (Object location : (Iterable)def) {
170                         builder.addDefaultLocation(location.toString());    
171                     }
172                 } else if (def instanceof String) {
173                     builder.addDefaultLocation(def.toString());
174                 } else if (def instanceof Map) {
175                     Map<String, Object> latlonMap = (Map<String, Object>) def;
176                     if (!latlonMap.containsKey("lat") || !(latlonMap.get("lat") instanceof Double)) {
177                         throw new ElasticsearchParseException("field [" + FIELD_MISSING + "] map must have field lat and a valid latitude");
178                     }
179                     if (!latlonMap.containsKey("lon") || !(latlonMap.get("lon") instanceof Double)) {
180                         throw new ElasticsearchParseException("field [" + FIELD_MISSING + "] map must have field lon and a valid longitude");
181                     }
182                     builder.addDefaultLocation(Double.valueOf(latlonMap.get("lat").toString()), Double.valueOf(latlonMap.get("lon").toString()));
183                 } else {
184                     throw new ElasticsearchParseException("field [" + FIELD_MISSING + "] must be of type string or list");
185                 }
186                 config.remove(FIELD_MISSING);
187             }
188 
189             final Object fieldName = config.get(FIELD_FIELDNAME);
190             if (fieldName != null) {
191                 builder.field(fieldName.toString());
192                 config.remove(FIELD_FIELDNAME);
193             }
194         }
195         return builder.build();
196     }

198     @Override
199     protected XContentBuilder toInnerXContent(XContentBuilder builder, Params params) throws IOException {
200         builder.field(FIELD_PRECISION, precision);
201         builder.field(FIELD_NEIGHBORS, neighbors);
202         if (defaultLocations != null) {
203             builder.startArray(FIELD_MISSING);
204             for (String defaultLocation : defaultLocations) {
205                 builder.value(defaultLocation);
206             }
207             builder.endArray();
208         }
209         if (fieldName != null) {
210             builder.field(FIELD_FIELDNAME, fieldName);
211         }
212         return builder;
213     }

215     protected static Collection<String> parseSinglePointOrList(XContentParser parser) throws IOException {
216         Token token = parser.currentToken();
217         if(token == Token.START_ARRAY) {
218             token = parser.nextToken();
219             // Test if value is a single point in <code>[lon, lat]</code> format
220             if(token == Token.VALUE_NUMBER) {
221                 double lon = parser.doubleValue();
222                 if(parser.nextToken() == Token.VALUE_NUMBER) {
223                     double lat = parser.doubleValue();
224                     if(parser.nextToken() == Token.END_ARRAY) {
225                         return Collections.singleton(GeoHashUtils.encode(lat, lon));
226                     } else {
227                         throw new ElasticsearchParseException("only two values expected");
228                     }
229                 } else {
230                     throw new ElasticsearchParseException("latitue must be a numeric value");
231                 }
232             } else {
233                 // otherwise it's a list of locations
234                 ArrayList<String> result = Lists.newArrayList();
235                 while (token != Token.END_ARRAY) {
236                     result.add(GeoUtils.parseGeoPoint(parser).geohash());
237                     token = parser.nextToken(); //infinite loop without this line
238                 }
239                 return result;
240             }
241         } else {
242             // or a single location
243             return Collections.singleton(GeoUtils.parseGeoPoint(parser).geohash());
244         }
245     }

252     @Override
253     public ContextConfig parseContext(ParseContext parseContext, XContentParser parser) throws IOException, ElasticsearchParseException {
254 
255         if(fieldName != null) {
256             FieldMapper<?> mapper = parseContext.docMapper().mappers().fullName(fieldName).mapper();
257             if(!(mapper instanceof GeoPointFieldMapper)) {
258                 throw new ElasticsearchParseException("referenced field must be mapped to geo_point");
259             }
260         }
261         
262         Collection<String> locations;
263         if(parser.currentToken() == Token.VALUE_NULL) {
264             locations = null;
265         } else {
266             locations = parseSinglePointOrList(parser);
267         }
268         return new GeoConfig(this, locations);
269     }

296         int precisionInts[] = new int[precisions.length];
297         for (int i = 0 ; i < precisions.length; i++) {

299         }

314     private static final int parsePrecision(XContentParser parser) throws IOException, ElasticsearchParseException {
315         switch (parser.currentToken()) {
316         case VALUE_STRING:
317             return GeoUtils.geoHashLevelsForPrecision(parser.text());
318         case VALUE_NUMBER:
319             switch (parser.numberType()) {
320             case INT:
321             case LONG:
322                 return parser.intValue();
323             default:
324                 return GeoUtils.geoHashLevelsForPrecision(parser.doubleValue());
325             }
326         default:
327             throw new ElasticsearchParseException("invalid precision value");
328         }
329     }

331     @Override
332     public GeoQuery parseQuery(String name, XContentParser parser) throws IOException, ElasticsearchParseException {
333         if (parser.currentToken() == Token.START_OBJECT) {
334             double lat = Double.NaN;
335             double lon = Double.NaN;
336             GeoPoint point = null;
337             int[] precision = null;
338             
339             while (parser.nextToken() != Token.END_OBJECT) {
340                 final String fieldName = parser.text();
341                 if("lat".equals(fieldName)) {
342                     if(point == null) {
343                         parser.nextToken();
344                         switch (parser.currentToken()) {
345                             case VALUE_NUMBER:
346                             case VALUE_STRING:
347                                 lat = parser.doubleValue(true);
348                                 break;
349                             default:
350                                 throw new ElasticsearchParseException("latitude must be a number");
351                         }
352                     } else {
353                         throw new ElasticsearchParseException("only lat/lon or [" + FIELD_VALUE + "] is allowed");
354                     }
355                 } else if ("lon".equals(fieldName)) {
356                     if(point == null) {
357                         parser.nextToken();
358                         switch (parser.currentToken()) {
359                             case VALUE_NUMBER:
360                             case VALUE_STRING:
361                                 lon = parser.doubleValue(true);
362                                 break;
363                             default:
364                                 throw new ElasticsearchParseException("longitude must be a number");
365                         }
366                     } else {
367                         throw new ElasticsearchParseException("only lat/lon or [" + FIELD_VALUE + "] is allowed");
368                     }
369                 } else if (FIELD_PRECISION.equals(fieldName)) {
370                     if(parser.nextToken() == Token.START_ARRAY) {
371                         IntOpenHashSet precisions = new IntOpenHashSet();
372                         while(parser.nextToken() != Token.END_ARRAY) {
373                             precisions.add(parsePrecision(parser));
374                         }
375                         precision = precisions.toArray();
376                     } else {
377                         precision = new int[] { parsePrecision(parser) };
378                     }
379                 } else if (FIELD_VALUE.equals(fieldName)) {
380                     if(Double.isNaN(lon) && Double.isNaN(lat)) {
381                         parser.nextToken();
382                         point = GeoUtils.parseGeoPoint(parser);
383                     } else {
384                         throw new ElasticsearchParseException("only lat/lon or [" + FIELD_VALUE + "] is allowed");
385                     }
386                 } else {
387                     throw new ElasticsearchParseException("unexpected fieldname [" + fieldName + "]");
388                 }
389             }
390 
391             if (point == null) {
392                 if (Double.isNaN(lat) || Double.isNaN(lon)) {
393                     throw new ElasticsearchParseException("location is missing");
394                 } else {
395                     point = new GeoPoint(lat, lon);
396                 }
397             }
398 
399             if (precision == null || precision.length == 0) {
400                 precision = this.precision;
401             }
402 
403             return new GeoQuery(name, point.geohash(), precision);
404         } else {
405             return new GeoQuery(name, GeoUtils.parseGeoPoint(parser).getGeohash(), precision);
406         }
407     }

409     @Override
410     public int hashCode() {
411         final int prime = 31;
412         int result = 1;
413         result = prime * result + ((defaultLocations == null) ? 0 : defaultLocations.hashCode());
414         result = prime * result + ((fieldName == null) ? 0 : fieldName.hashCode());
415         result = prime * result + (neighbors ? 1231 : 1237);
416         result = prime * result + Arrays.hashCode(precision);
417         return result;
418     }

420     @Override
421     public boolean equals(Object obj) {
422         if (this == obj)
423             return true;
424         if (obj == null)
425             return false;
426         if (getClass() != obj.getClass())
427             return false;
428         GeolocationContextMapping other = (GeolocationContextMapping) obj;
429         if (defaultLocations == null) {
430             if (other.defaultLocations != null)
431                 return false;
432         } else if (!defaultLocations.equals(other.defaultLocations))
433             return false;
434         if (fieldName == null) {
435             if (other.fieldName != null)
436                 return false;
437         } else if (!fieldName.equals(other.fieldName))
438             return false;
439         if (neighbors != other.neighbors)
440             return false;
441         if (!Arrays.equals(precision, other.precision))
442             return false;
443         return true;
444     }

450 
451         private IntOpenHashSet precisions = new IntOpenHashSet();
452         private boolean neighbors; // take neighbor cell on the lowest level into account
453         private HashSet<String> defaultLocations = new HashSet<>();
454         private String fieldName = null;
455         
456         protected Builder(String name) {
457             this(name, true, null);
458         }
459 
460         protected Builder(String name, boolean neighbors, int...levels) {
461             super(name);
462             neighbors(neighbors);
463             if (levels != null) {
464                 for (int level : levels) {
465                     precision(level);
466                 }
467             }
468         }
469 
470         /**
471          * Set the precision use o make suggestions
472          * 
473          * @param precision
474          *            precision as distance with {@link DistanceUnit}. Default:
475          *            meters
476          * @return this
477          */
478         public Builder precision(String precision) {
479             return precision(DistanceUnit.parse(precision, DistanceUnit.METERS, DistanceUnit.METERS));
480         }
481 
482         /**
483          * Set the precision use o make suggestions
484          * 
485          * @param precision
486          *            precision value
487          * @param unit
488          *            {@link DistanceUnit} to use
489          * @return this
490          */
491         public Builder precision(double precision, DistanceUnit unit) {
492             return precision(unit.toMeters(precision));
493         }
494 
495         /**
496          * Set the precision use o make suggestions
497          * 
498          * @param meters
499          *            precision as distance in meters
500          * @return this
501          */
502         public Builder precision(double meters) {
503             int level = GeoUtils.geoHashLevelsForPrecision(meters);
504             // Ceiling precision: we might return more results 
505             if (GeoUtils.geoHashCellSize(level) < meters) {
506                level = Math.max(1, level - 1); 
507             }
508             return precision(level);
509         }
510 
511         /**
512          * Set the precision use o make suggestions
513          * 
514          * @param level
515          *            maximum length of geohashes
516          * @return this
517          */
518         public Builder precision(int level) {
519             this.precisions.add(level);
520             return this;
521         }

525          * 
526          * @param neighbors

528          * @return this
529          */

531             this.neighbors = neighbors;
532             return this;
533         }

574         }

577          * Set a default location that should be used, if no location is
578          * provided by the query
579          * 
580          * @param point
581          *            location
582          * @return this
583          */
584         public Builder defaultLocation(GeoPoint point) {
585             this.defaultLocations.add(point.geohash());

587         }

589         /**
590          * Set the name of the field containing a geolocation to use

609     private static class GeoConfig extends ContextConfig {
610 
611         private final GeolocationContextMapping mapping;
612         private final Collection<String> locations;
613 
614         public GeoConfig(GeolocationContextMapping mapping, Collection<String> locations) {
615             this.locations = locations;
616             this.mapping = mapping;
617         }
618 
619         @Override
620         protected TokenStream wrapTokenStream(Document doc, TokenStream stream) {
621             Collection<String> geohashes;
622 
623             if (locations == null || locations.size() == 0) {
624                 if(mapping.fieldName != null) {
625                     IndexableField[] fields = doc.getFields(mapping.fieldName);
626                     if(fields.length == 0) {
627                         IndexableField[] lonFields = doc.getFields(mapping.fieldName + ".lon");
628                         IndexableField[] latFields = doc.getFields(mapping.fieldName + ".lat");
629                         if (lonFields.length > 0 && latFields.length > 0) {
630                             geohashes = new ArrayList<>(fields.length);
631                             GeoPoint spare = new GeoPoint();
632                             for (int i = 0 ; i < lonFields.length ; i++) {
633                                 IndexableField lonField = lonFields[i];
634                                 IndexableField latField = latFields[i];
635                                 assert lonField.fieldType().docValuesType() == latField.fieldType().docValuesType();
636                                 // we write doc values fields differently: one field for all values, so we need to only care about indexed fields
637                                 if (lonField.fieldType().docValuesType() == DocValuesType.NONE) {
638                                     spare.reset(latField.numericValue().doubleValue(), lonField.numericValue().doubleValue());
639                                     geohashes.add(spare.geohash());
640                                 }
641                             }
642                         } else {
643                             geohashes = mapping.defaultLocations;
644                         }
645                     } else {
646                         geohashes = new ArrayList<>(fields.length);
647                         GeoPoint spare = new GeoPoint();
648                         for (IndexableField field : fields) {
649                             spare.resetFromString(field.stringValue());
650                             geohashes.add(spare.geohash());
651                         }
652                     }
653                 } else {
654                     geohashes = mapping.defaultLocations;
655                 }
656             } else {
657                 geohashes = locations;
658             }
659 
660             Collection<String> locations = new HashSet<>();
661             for (String geohash : geohashes) {
662                 for (int p : mapping.precision) {
663                     int precision = Math.min(p, geohash.length());
664                     String truncatedGeohash = geohash.substring(0, precision);
665                     if(mapping.neighbors) {
666                         GeoHashUtils.addNeighbors(truncatedGeohash, precision, locations);
667                     }
668                     locations.add(truncatedGeohash);
669                 }
670             }
671 
672             return new PrefixTokenFilter(stream, ContextMapping.SEPARATOR, locations);
673         }
674 
675         @Override
676         public String toString() {
677             StringBuilder sb = new StringBuilder("GeoConfig(location = [");
678             Iterator<? extends CharSequence> location = this.locations.iterator();
679             if (location.hasNext()) {
680                 sb.append(location.next());
681                 while (location.hasNext()) {
682                     sb.append(", ").append(location.next());
683                 }
684             }
685             return sb.append("])").toString();
686         }
687     }

689     private static class GeoQuery extends ContextQuery {
690         private final String location;
691         private final int[] precisions;

693         public GeoQuery(String name, String location, int...precisions) {
694             super(name);
695             this.location = location;

698 
699         @Override
700         public Automaton toAutomaton() {
701             Automaton automaton;
702             if(precisions == null || precisions.length == 0) {
703                  automaton = Automata.makeString(location);
704             } else {
705                 automaton = Automata.makeString(location.substring(0, Math.max(1, Math.min(location.length(), precisions[0]))));
706                 for (int i = 1; i < precisions.length; i++) {
707                     final String cell = location.substring(0, Math.max(1, Math.min(location.length(), precisions[i])));
708                     automaton = Operations.union(automaton, Automata.makeString(cell));
709                 }
710             }
711             return automaton;
712         }
