20  package org.elasticsearch.action.termvectors;
21  
22  import org.elasticsearch.ElasticsearchException;
23  import org.elasticsearch.ExceptionsHelper;
24  import org.elasticsearch.action.support.ActionFilters;
25  import org.elasticsearch.action.support.TransportActions;
26  import org.elasticsearch.action.support.single.shard.TransportShardSingleOperationAction;
27  import org.elasticsearch.cluster.ClusterService;
28  import org.elasticsearch.cluster.ClusterState;
29  import org.elasticsearch.cluster.routing.ShardIterator;
30  import org.elasticsearch.common.inject.Inject;
31  import org.elasticsearch.common.settings.Settings;
32  import org.elasticsearch.index.IndexService;
33  import org.elasticsearch.index.shard.IndexShard;
34  import org.elasticsearch.index.shard.ShardId;
35  import org.elasticsearch.indices.IndicesService;
36  import org.elasticsearch.threadpool.ThreadPool;
37  import org.elasticsearch.transport.TransportService;
38  
39  public class TransportShardMultiTermsVectorAction extends TransportShardSingleOperationAction<MultiTermVectorsShardRequest, MultiTermVectorsShardResponse> {
40  
41      private final IndicesService indicesService;
42  
43      private static final String ACTION_NAME = MultiTermVectorsAction.NAME + "[shard]";
44  
45      @Inject
46      public TransportShardMultiTermsVectorAction(Settings settings, ClusterService clusterService, TransportService transportService,
47                                                  IndicesService indicesService, ThreadPool threadPool, ActionFilters actionFilters) {
48          super(settings, ACTION_NAME, threadPool, clusterService, transportService, actionFilters);
49          this.indicesService = indicesService;
50      }
51  
52      @Override
53      protected boolean isSubAction() {
54          return true;
55      }
56  
57      @Override
58      protected String executor() {
59          return ThreadPool.Names.GET;
60      }
61  
62      @Override
63      protected MultiTermVectorsShardRequest newRequest() {
64          return new MultiTermVectorsShardRequest();
65      }
66  
67      @Override
68      protected MultiTermVectorsShardResponse newResponse() {
69          return new MultiTermVectorsShardResponse();
70      }
71  
72      @Override
73      protected boolean resolveIndex() {
74          return false;
75      }
76  
77      @Override
78      protected ShardIterator shards(ClusterState state, InternalRequest request) {
79          return clusterService.operationRouting()
80                  .getShards(state, request.concreteIndex(), request.request().shardId(), request.request().preference());
81      }
82  
83      @Override
84      protected MultiTermVectorsShardResponse shardOperation(MultiTermVectorsShardRequest request, ShardId shardId) throws ElasticsearchException {
85          MultiTermVectorsShardResponse response = new MultiTermVectorsShardResponse();
86          for (int i = 0; i < request.locations.size(); i++) {
87              TermVectorsRequest termVectorsRequest = request.requests.get(i);
88              try {
89                  IndexService indexService = indicesService.indexServiceSafe(request.index());
90                  IndexShard indexShard = indexService.shardSafe(shardId.id());
91                  TermVectorsResponse termVectorsResponse = indexShard.termVectorsService().getTermVectors(termVectorsRequest, shardId.getIndex());
92                  termVectorsResponse.updateTookInMillis(termVectorsRequest.startTime());
93                  response.add(request.locations.get(i), termVectorsResponse);
94              } catch (Throwable t) {
95                  if (TransportActions.isShardNotAvailableException(t)) {
96                      throw (ElasticsearchException) t;
97                  } else {
98                      logger.debug("{} failed to execute multi term vectors for [{}]/[{}]", t, shardId, termVectorsRequest.type(), termVectorsRequest.id());
99                      response.add(request.locations.get(i),
100                             new MultiTermVectorsResponse.Failure(request.index(), termVectorsRequest.type(), termVectorsRequest.id(), ExceptionsHelper.detailedMessage(t)));
101                 }
102             }
103         }
104 
105         return response;
106     }
107 }
