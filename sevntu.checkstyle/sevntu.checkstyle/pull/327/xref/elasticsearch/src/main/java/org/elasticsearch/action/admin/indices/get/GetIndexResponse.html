37  /**
38   * A response for a delete index action.
39   */
40  public class GetIndexResponse extends ActionResponse {
41  
42      private ImmutableOpenMap<String, ImmutableList<IndexWarmersMetaData.Entry>> warmers = ImmutableOpenMap.of();
43      private ImmutableOpenMap<String, ImmutableOpenMap<String, MappingMetaData>> mappings = ImmutableOpenMap.of();
44      private ImmutableOpenMap<String, ImmutableList<AliasMetaData>> aliases = ImmutableOpenMap.of();
45      private ImmutableOpenMap<String, Settings> settings = ImmutableOpenMap.of();
46      private String[] indices;
47  
48      GetIndexResponse(String[] indices, ImmutableOpenMap<String, ImmutableList<IndexWarmersMetaData.Entry>> warmers,
49              ImmutableOpenMap<String, ImmutableOpenMap<String, MappingMetaData>> mappings,
50              ImmutableOpenMap<String, ImmutableList<AliasMetaData>> aliases, ImmutableOpenMap<String, Settings> settings) {
51          this.indices = indices;
52          if (warmers != null) {
53              this.warmers = warmers;
54          }
55          if (mappings != null) {
56              this.mappings = mappings;
57          }
58          if (aliases != null) {
59              this.aliases = aliases;
60          }
61          if (settings != null) {
62              this.settings = settings;
63          }
64      }

109     @Override
110     public void readFrom(StreamInput in) throws IOException {
111         super.readFrom(in);
112         this.indices = in.readStringArray();
113         int warmersSize = in.readVInt();
114         ImmutableOpenMap.Builder<String, ImmutableList<IndexWarmersMetaData.Entry>> warmersMapBuilder = ImmutableOpenMap.builder();
115         for (int i = 0; i < warmersSize; i++) {
116             String key = in.readString();
117             int valueSize = in.readVInt();
118             ImmutableList.Builder<IndexWarmersMetaData.Entry> warmerEntryBuilder = ImmutableList.builder();
119             for (int j = 0; j < valueSize; j++) {
120                 warmerEntryBuilder.add(new IndexWarmersMetaData.Entry(
121                         in.readString(),
122                         in.readStringArray(),
123                         in.readOptionalBoolean(),
124                         in.readBytesReference())
125                 );
126             }
127             warmersMapBuilder.put(key, warmerEntryBuilder.build());
128         }
129         warmers = warmersMapBuilder.build();
130         int mappingsSize = in.readVInt();
131         ImmutableOpenMap.Builder<String, ImmutableOpenMap<String, MappingMetaData>> mappingsMapBuilder = ImmutableOpenMap.builder();
132         for (int i = 0; i < mappingsSize; i++) {
133             String key = in.readString();
134             int valueSize = in.readVInt();
135             ImmutableOpenMap.Builder<String, MappingMetaData> mappingEntryBuilder = ImmutableOpenMap.builder();
136             for (int j = 0; j < valueSize; j++) {
137                 mappingEntryBuilder.put(in.readString(), MappingMetaData.readFrom(in));
138             }
139             mappingsMapBuilder.put(key, mappingEntryBuilder.build());
140         }
141         mappings = mappingsMapBuilder.build();
142         int aliasesSize = in.readVInt();
143         ImmutableOpenMap.Builder<String, ImmutableList<AliasMetaData>> aliasesMapBuilder = ImmutableOpenMap.builder();
144         for (int i = 0; i < aliasesSize; i++) {
145             String key = in.readString();
146             int valueSize = in.readVInt();
147             ImmutableList.Builder<AliasMetaData> aliasEntryBuilder = ImmutableList.builder();
148             for (int j = 0; j < valueSize; j++) {
149                 aliasEntryBuilder.add(AliasMetaData.Builder.readFrom(in));
150             }
151             aliasesMapBuilder.put(key, aliasEntryBuilder.build());
152         }
153         aliases = aliasesMapBuilder.build();
154         int settingsSize = in.readVInt();
155         ImmutableOpenMap.Builder<String, Settings> settingsMapBuilder = ImmutableOpenMap.builder();
156         for (int i = 0; i < settingsSize; i++) {
157             String key = in.readString();
158             settingsMapBuilder.put(key, ImmutableSettings.readSettingsFromStream(in));
159         }
160         settings = settingsMapBuilder.build();
161     }
162 
163     @Override
164     public void writeTo(StreamOutput out) throws IOException {
165         super.writeTo(out);
166         out.writeStringArray(indices);
167         out.writeVInt(warmers.size());
168         for (ObjectObjectCursor<String, ImmutableList<IndexWarmersMetaData.Entry>> indexEntry : warmers) {
169             out.writeString(indexEntry.key);
170             out.writeVInt(indexEntry.value.size());
171             for (IndexWarmersMetaData.Entry warmerEntry : indexEntry.value) {
172                 out.writeString(warmerEntry.name());
173                 out.writeStringArray(warmerEntry.types());
174                 out.writeOptionalBoolean(warmerEntry.queryCache());
175                 out.writeBytesReference(warmerEntry.source());
176             }
177         }
178         out.writeVInt(mappings.size());
179         for (ObjectObjectCursor<String, ImmutableOpenMap<String, MappingMetaData>> indexEntry : mappings) {
180             out.writeString(indexEntry.key);
181             out.writeVInt(indexEntry.value.size());
182             for (ObjectObjectCursor<String, MappingMetaData> mappingEntry : indexEntry.value) {
183                 out.writeString(mappingEntry.key);
184                 MappingMetaData.writeTo(mappingEntry.value, out);
185             }
186         }
187         out.writeVInt(aliases.size());
188         for (ObjectObjectCursor<String, ImmutableList<AliasMetaData>> indexEntry : aliases) {
189             out.writeString(indexEntry.key);
190             out.writeVInt(indexEntry.value.size());
191             for (AliasMetaData aliasEntry : indexEntry.value) {
192                 AliasMetaData.Builder.writeTo(aliasEntry, out);
193             }
194         }
195         out.writeVInt(settings.size());
196         for (ObjectObjectCursor<String, Settings> indexEntry : settings) {
197             out.writeString(indexEntry.key);
198             ImmutableSettings.writeSettingsToStream(indexEntry.value, out);
199         }
200     }
