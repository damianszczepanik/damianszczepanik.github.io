43  /**
44   * The {@link DiskThresholdDecider} checks that the node a shard is potentially
45   * being allocated to has enough disk space.
46   *
47   * It has three configurable settings, all of which can be changed dynamically:
48   *
49   * <code>cluster.routing.allocation.disk.watermark.low</code> is the low disk
50   * watermark. New shards will not allocated to a node with usage higher than this,
51   * although this watermark may be passed by allocating a shard. It defaults to
52   * 0.85 (85.0%).
53   *
54   * <code>cluster.routing.allocation.disk.watermark.high</code> is the high disk
55   * watermark. If a node has usage higher than this, shards are not allowed to
56   * remain on the node. In addition, if allocating a shard to a node causes the
57   * node to pass this watermark, it will not be allowed. It defaults to
58   * 0.90 (90.0%).
59   *
60   * Both watermark settings are expressed in terms of used disk percentage, or
61   * exact byte values for free space (like "500mb")
62   *
63   * <code>cluster.routing.allocation.disk.threshold_enabled</code> is used to
64   * enable or disable this decider. It defaults to false (disabled).
65   */
66  public class DiskThresholdDecider extends AllocationDecider {
67  
68      public static final String NAME = "disk_threshold";
69  
70      private volatile Double freeDiskThresholdLow;
71      private volatile Double freeDiskThresholdHigh;
72      private volatile ByteSizeValue freeBytesThresholdLow;
73      private volatile ByteSizeValue freeBytesThresholdHigh;
74      private volatile boolean includeRelocations;
75      private volatile boolean enabled;
76      private volatile TimeValue rerouteInterval;

78      public static final String CLUSTER_ROUTING_ALLOCATION_DISK_THRESHOLD_ENABLED = "cluster.routing.allocation.disk.threshold_enabled";
79      public static final String CLUSTER_ROUTING_ALLOCATION_LOW_DISK_WATERMARK = "cluster.routing.allocation.disk.watermark.low";
80      public static final String CLUSTER_ROUTING_ALLOCATION_HIGH_DISK_WATERMARK = "cluster.routing.allocation.disk.watermark.high";
81      public static final String CLUSTER_ROUTING_ALLOCATION_INCLUDE_RELOCATIONS = "cluster.routing.allocation.disk.include_relocations";
82      public static final String CLUSTER_ROUTING_ALLOCATION_REROUTE_INTERVAL = "cluster.routing.allocation.disk.reroute_interval";

84      class ApplySettings implements NodeSettingsService.Listener {
85          @Override
86          public void onRefreshSettings(Settings settings) {
87              String newLowWatermark = settings.get(CLUSTER_ROUTING_ALLOCATION_LOW_DISK_WATERMARK, null);
88              String newHighWatermark = settings.get(CLUSTER_ROUTING_ALLOCATION_HIGH_DISK_WATERMARK, null);
89              Boolean newRelocationsSetting = settings.getAsBoolean(CLUSTER_ROUTING_ALLOCATION_INCLUDE_RELOCATIONS, null);
90              Boolean newEnableSetting =  settings.getAsBoolean(CLUSTER_ROUTING_ALLOCATION_DISK_THRESHOLD_ENABLED, null);
91              TimeValue newRerouteInterval = settings.getAsTime(CLUSTER_ROUTING_ALLOCATION_REROUTE_INTERVAL, null);
92  
93              if (newEnableSetting != null) {
94                  logger.info("updating [{}] from [{}] to [{}]", CLUSTER_ROUTING_ALLOCATION_DISK_THRESHOLD_ENABLED,
95                          DiskThresholdDecider.this.enabled, newEnableSetting);
96                  DiskThresholdDecider.this.enabled = newEnableSetting;
97              }
98              if (newRelocationsSetting != null) {
99                  logger.info("updating [{}] from [{}] to [{}]", CLUSTER_ROUTING_ALLOCATION_INCLUDE_RELOCATIONS,
100                         DiskThresholdDecider.this.includeRelocations, newRelocationsSetting);
101                 DiskThresholdDecider.this.includeRelocations = newRelocationsSetting;
102             }
103             if (newLowWatermark != null) {
104                 if (!validWatermarkSetting(newLowWatermark)) {
105                     throw new ElasticsearchParseException("Unable to parse low watermark: [" + newLowWatermark + "]");
106                 }
107                 logger.info("updating [{}] to [{}]", CLUSTER_ROUTING_ALLOCATION_LOW_DISK_WATERMARK, newLowWatermark);
108                 DiskThresholdDecider.this.freeDiskThresholdLow = 100.0 - thresholdPercentageFromWatermark(newLowWatermark);
109                 DiskThresholdDecider.this.freeBytesThresholdLow = thresholdBytesFromWatermark(newLowWatermark);
110             }
111             if (newHighWatermark != null) {
112                 if (!validWatermarkSetting(newHighWatermark)) {
113                     throw new ElasticsearchParseException("Unable to parse high watermark: [" + newHighWatermark + "]");
114                 }
115                 logger.info("updating [{}] to [{}]", CLUSTER_ROUTING_ALLOCATION_HIGH_DISK_WATERMARK, newHighWatermark);
116                 DiskThresholdDecider.this.freeDiskThresholdHigh = 100.0 - thresholdPercentageFromWatermark(newHighWatermark);
117                 DiskThresholdDecider.this.freeBytesThresholdHigh = thresholdBytesFromWatermark(newHighWatermark);
118             }
119             if (newRerouteInterval != null) {
120                 logger.info("updating [{}] to [{}]", CLUSTER_ROUTING_ALLOCATION_REROUTE_INTERVAL, newRerouteInterval);
121                 DiskThresholdDecider.this.rerouteInterval = newRerouteInterval;
122             }
123         }

126     /**
127      * Listens for a node to go over the high watermark and kicks off an empty
128      * reroute if it does. Also responsible for logging about nodes that have
129      * passed the disk watermarks
130      */
131     class DiskListener implements ClusterInfoService.Listener {
132         private final Client client;
133         private long lastRun;
134 
135         DiskListener(Client client) {
136             this.client = client;
137         }
138 
139         /**
140          * Warn about the given disk usage if the low or high watermark has been passed
141          */
142         private void warnAboutDiskIfNeeded(DiskUsage usage) {
143             // Check absolute disk values
144             if (usage.getFreeBytes() < DiskThresholdDecider.this.freeBytesThresholdHigh.bytes()) {
145                 logger.warn("high disk watermark [{}] exceeded on {}, shards will be relocated away from this node",
146                         DiskThresholdDecider.this.freeBytesThresholdHigh, usage);
147             } else if (usage.getFreeBytes() < DiskThresholdDecider.this.freeBytesThresholdLow.bytes()) {
148                 logger.info("low disk watermark [{}] exceeded on {}, replicas will not be assigned to this node",
149                         DiskThresholdDecider.this.freeBytesThresholdLow, usage);
150             }
151 
152             // Check percentage disk values
153             if (usage.getFreeDiskAsPercentage() < DiskThresholdDecider.this.freeDiskThresholdHigh) {
154                 logger.warn("high disk watermark [{}] exceeded on {}, shards will be relocated away from this node",
155                         Strings.format1Decimals(DiskThresholdDecider.this.freeDiskThresholdHigh, "%"), usage);
156             } else if (usage.getFreeDiskAsPercentage() < DiskThresholdDecider.this.freeDiskThresholdLow) {
157                 logger.info("low disk watermark [{}] exceeded on {}, replicas will not be assigned to this node",
158                         Strings.format1Decimals(DiskThresholdDecider.this.freeDiskThresholdLow, "%"), usage);
159             }
160         }
161 
162         @Override
163         public void onNewInfo(ClusterInfo info) {
164             Map<String, DiskUsage> usages = info.getNodeDiskUsages();
165             if (usages != null) {
166                 boolean reroute = false;
167                 for (DiskUsage entry : usages.values()) {
168                     warnAboutDiskIfNeeded(entry);
169                     if (entry.getFreeBytes() < DiskThresholdDecider.this.freeBytesThresholdHigh.bytes() ||
170                             entry.getFreeDiskAsPercentage() < DiskThresholdDecider.this.freeDiskThresholdHigh) {
171                         if ((System.currentTimeMillis() - lastRun) > DiskThresholdDecider.this.rerouteInterval.millis()) {
172                             lastRun = System.currentTimeMillis();
173                             reroute = true;
174                         } else {
175                             logger.debug("high disk watermark exceeded on {} but an automatic reroute has occurred in the last [{}], skipping reroute",
176                                     entry, DiskThresholdDecider.this.rerouteInterval);
177                         }
178                     }
179                 }
180                 if (reroute) {
181                     logger.info("high disk watermark exceeded on one or more nodes, rerouting shards");
182                     // Execute an empty reroute, but don't block on the response
183                     client.admin().cluster().prepareReroute().execute();
184                 }
185             }
186         }
187     }

196     @Inject
197     public DiskThresholdDecider(Settings settings, NodeSettingsService nodeSettingsService, ClusterInfoService infoService, Client client) {
198         super(settings);
199         String lowWatermark = settings.get(CLUSTER_ROUTING_ALLOCATION_LOW_DISK_WATERMARK, "85%");
200         String highWatermark = settings.get(CLUSTER_ROUTING_ALLOCATION_HIGH_DISK_WATERMARK, "90%");
201 
202         if (!validWatermarkSetting(lowWatermark)) {
203             throw new ElasticsearchParseException("Unable to parse low watermark: [" + lowWatermark + "]");
204         }
205         if (!validWatermarkSetting(highWatermark)) {
206             throw new ElasticsearchParseException("Unable to parse high watermark: [" + highWatermark + "]");
207         }
208         // Watermark is expressed in terms of used data, but we need "free" data watermark
209         this.freeDiskThresholdLow = 100.0 - thresholdPercentageFromWatermark(lowWatermark);
210         this.freeDiskThresholdHigh = 100.0 - thresholdPercentageFromWatermark(highWatermark);
211 
212         this.freeBytesThresholdLow = thresholdBytesFromWatermark(lowWatermark);
213         this.freeBytesThresholdHigh = thresholdBytesFromWatermark(highWatermark);
214         this.includeRelocations = settings.getAsBoolean(CLUSTER_ROUTING_ALLOCATION_INCLUDE_RELOCATIONS, true);
215         this.rerouteInterval = settings.getAsTime(CLUSTER_ROUTING_ALLOCATION_REROUTE_INTERVAL, TimeValue.timeValueSeconds(60));
216 
217         this.enabled = settings.getAsBoolean(CLUSTER_ROUTING_ALLOCATION_DISK_THRESHOLD_ENABLED, true);
218         nodeSettingsService.addListener(new ApplySettings());
219         infoService.addListener(new DiskListener(client));
220     }

262     /**
263      * Returns the size of all shards that are currently being relocated to
264      * the node, but may not be finished transfering yet.
265      *
266      * If subtractShardsMovingAway is set then the size of shards moving away is subtracted from the total size
267      * of all shards
268      */
269     public long sizeOfRelocatingShards(RoutingNode node, Map<String, Long> shardSizes, boolean subtractShardsMovingAway) {
270         long totalSize = 0;
271         for (ShardRouting routing : node.shardsWithState(ShardRoutingState.RELOCATING, ShardRoutingState.INITIALIZING)) {
272             if (routing.initializing() && routing.relocatingNodeId() != null) {
273                 totalSize += getShardSize(routing, shardSizes);
274             } else if (subtractShardsMovingAway && routing.relocating()) {
275                 totalSize -= getShardSize(routing, shardSizes);
276             }
277         }
278         return totalSize;
279     }
280 
281     private long getShardSize(ShardRouting routing, Map<String, Long> shardSizes) {
282         Long shardSize = shardSizes.get(shardIdentifierFromRouting(routing));
283         return shardSize == null ? 0 : shardSize;
284     }

286     @Override
287     public Decision canAllocate(ShardRouting shardRouting, RoutingNode node, RoutingAllocation allocation) {
288 
289         // Always allow allocation if the decider is disabled
290         if (!enabled) {
291             return allocation.decision(Decision.YES, NAME, "disk threshold decider disabled");
292         }
293 
294         // Allow allocation regardless if only a single node is available
295         if (allocation.nodes().size() <= 1) {
296             if (logger.isTraceEnabled()) {
297                 logger.trace("Only a single node is present, allowing allocation");
298             }
299             return allocation.decision(Decision.YES, NAME, "only a single node is present");
300         }
301 
302         // Fail open there is no info available
303         ClusterInfo clusterInfo = allocation.clusterInfo();
304         if (clusterInfo == null) {
305             if (logger.isTraceEnabled()) {
306                 logger.trace("Cluster info unavailable for disk threshold decider, allowing allocation.");
307             }
308             return allocation.decision(Decision.YES, NAME, "cluster info unavailable");
309         }
310 
311         // Fail open if there are no disk usages available
312         Map<String, DiskUsage> usages = clusterInfo.getNodeDiskUsages();
313         Map<String, Long> shardSizes = clusterInfo.getShardSizes();
314         if (usages.isEmpty()) {
315             if (logger.isTraceEnabled()) {
316                 logger.trace("Unable to determine disk usages for disk-aware allocation, allowing allocation");
317             }
318             return allocation.decision(Decision.YES, NAME, "disk usages unavailable");
319         }
320 
321         DiskUsage usage = usages.get(node.nodeId());
322         if (usage == null) {
323             // If there is no usage, and we have other nodes in the cluster,
324             // use the average usage for all nodes as the usage for this node
325             usage = averageUsage(node, usages);
326             if (logger.isDebugEnabled()) {
327                 logger.debug("Unable to determine disk usage for [{}], defaulting to average across nodes [{} total] [{} free] [{}% free]",
328                         node.nodeId(), usage.getTotalBytes(), usage.getFreeBytes(), usage.getFreeDiskAsPercentage());
329             }
330         }
331 
332         if (includeRelocations) {
333             long relocatingShardsSize = sizeOfRelocatingShards(node, shardSizes, false);
334             DiskUsage usageIncludingRelocations = new DiskUsage(node.nodeId(), node.node().name(),
335                     usage.getTotalBytes(), usage.getFreeBytes() - relocatingShardsSize);
336             if (logger.isTraceEnabled()) {
337                 logger.trace("usage without relocations: {}", usage);
338                 logger.trace("usage with relocations: [{} bytes] {}", relocatingShardsSize, usageIncludingRelocations);
339             }
340             usage = usageIncludingRelocations;
341         }
342 
343         // First, check that the node currently over the low watermark
344         double freeDiskPercentage = usage.getFreeDiskAsPercentage();
345         long freeBytes = usage.getFreeBytes();
346         if (logger.isTraceEnabled()) {
347             logger.trace("Node [{}] has {}% free disk", node.nodeId(), freeDiskPercentage);
348         }
349 
350         // a flag for whether the primary shard has been previously allocated
351         boolean primaryHasBeenAllocated = allocation.routingTable().index(shardRouting.index()).shard(shardRouting.id()).primaryAllocatedPostApi();
352 
353         // checks for exact byte comparisons
354         if (freeBytes < freeBytesThresholdLow.bytes()) {
355             // If the shard is a replica or has a primary that has already been allocated before, check the low threshold
356             if (!shardRouting.primary() || (shardRouting.primary() && primaryHasBeenAllocated)) {
357                 if (logger.isDebugEnabled()) {
358                     logger.debug("Less than the required {} free bytes threshold ({} bytes free) on node {}, preventing allocation",
359                             freeBytesThresholdLow, freeBytes, node.nodeId());
360                 }
361                 return allocation.decision(Decision.NO, NAME, "less than required [%s] free on node, free: [%s]",
362                         freeBytesThresholdLow, new ByteSizeValue(freeBytes));
363             } else if (freeBytes > freeBytesThresholdHigh.bytes()) {
364                 // Allow the shard to be allocated because it is primary that
365                 // has never been allocated if it's under the high watermark
366                 if (logger.isDebugEnabled()) {
367                     logger.debug("Less than the required {} free bytes threshold ({} bytes free) on node {}, " +
368                                     "but allowing allocation because primary has never been allocated",
369                             freeBytesThresholdLow, freeBytes, node.nodeId());
370                 }
371                 return allocation.decision(Decision.YES, NAME, "primary has never been allocated before");
372             } else {
373                 // Even though the primary has never been allocated, the node is
374                 // above the high watermark, so don't allow allocating the shard
375                 if (logger.isDebugEnabled()) {
376                     logger.debug("Less than the required {} free bytes threshold ({} bytes free) on node {}, " +
377                                     "preventing allocation even though primary has never been allocated",
378                             freeBytesThresholdHigh, freeBytes, node.nodeId());
379                 }
380                 return allocation.decision(Decision.NO, NAME, "less than required [%s] free on node, free: [%s]",
381                         freeBytesThresholdHigh, new ByteSizeValue(freeBytes));
382             }
383         }
384 
385         // checks for percentage comparisons
386         if (freeDiskPercentage < freeDiskThresholdLow) {
387             // If the shard is a replica or has a primary that has already been allocated before, check the low threshold
388             if (!shardRouting.primary() || (shardRouting.primary() && primaryHasBeenAllocated)) {
389                 if (logger.isDebugEnabled()) {
390                     logger.debug("Less than the required {} free disk threshold ({} free) on node [{}], preventing allocation",
391                             Strings.format1Decimals(freeDiskThresholdLow, "%"),
392                             Strings.format1Decimals(freeDiskPercentage, "%"), node.nodeId());
393                 }
394                 return allocation.decision(Decision.NO, NAME, "less than required [%s%%] free disk on node, free: [%s%%]",
395                         freeDiskThresholdLow, freeDiskPercentage);
396             } else if (freeDiskPercentage > freeDiskThresholdHigh) {
397                 // Allow the shard to be allocated because it is primary that
398                 // has never been allocated if it's under the high watermark
399                 if (logger.isDebugEnabled()) {
400                     logger.debug("Less than the required {} free disk threshold ({} free) on node [{}], " +
401                                     "but allowing allocation because primary has never been allocated",
402                             Strings.format1Decimals(freeDiskThresholdLow, "%"),
403                             Strings.format1Decimals(freeDiskPercentage, "%"), node.nodeId());
404                 }
405                 return allocation.decision(Decision.YES, NAME, "primary has never been allocated before");
406             } else {
407                 // Even though the primary has never been allocated, the node is
408                 // above the high watermark, so don't allow allocating the shard
409                 if (logger.isDebugEnabled()) {
410                     logger.debug("Less than the required {} free bytes threshold ({} bytes free) on node {}, " +
411                                     "preventing allocation even though primary has never been allocated",
412                             Strings.format1Decimals(freeDiskThresholdHigh, "%"),
413                             Strings.format1Decimals(freeDiskPercentage, "%"), node.nodeId());
414                 }
415                 return allocation.decision(Decision.NO, NAME, "less than required [%s%%] free disk on node, free: [%s%%]",
416                         freeDiskThresholdLow, freeDiskPercentage);
417             }
418         }
419 
420         // Secondly, check that allocating the shard to this node doesn't put it above the high watermark
421         Long shardSize = shardSizes.get(shardIdentifierFromRouting(shardRouting));
422         shardSize = shardSize == null ? 0 : shardSize;
423         double freeSpaceAfterShard = this.freeDiskPercentageAfterShardAssigned(usage, shardSize);
424         long freeBytesAfterShard = freeBytes - shardSize;
425         if (freeBytesAfterShard < freeBytesThresholdHigh.bytes()) {
426             logger.warn("After allocating, node [{}] would have less than the required {} free bytes threshold ({} bytes free), preventing allocation",
427                     node.nodeId(), freeBytesThresholdHigh, freeBytesAfterShard);
428             return allocation.decision(Decision.NO, NAME, "after allocation less than required [%s] free on node, free: [%s]",
429                     freeBytesThresholdLow, new ByteSizeValue(freeBytesAfterShard));
430         }
431         if (freeSpaceAfterShard < freeDiskThresholdHigh) {
432             logger.warn("After allocating, node [{}] would have less than the required {} free disk threshold ({} free), preventing allocation",
433                     node.nodeId(), Strings.format1Decimals(freeDiskThresholdHigh, "%"), Strings.format1Decimals(freeSpaceAfterShard, "%"));
434             return allocation.decision(Decision.NO, NAME, "after allocation less than required [%s%%] free disk on node, free: [%s%%]",
435                     freeDiskThresholdLow, freeSpaceAfterShard);
436         }
437 
438         return allocation.decision(Decision.YES, NAME, "enough disk for shard on node, free: [%s]", new ByteSizeValue(freeBytes));
439     }

441     @Override
442     public Decision canRemain(ShardRouting shardRouting, RoutingNode node, RoutingAllocation allocation) {
443         if (!enabled) {
444             return allocation.decision(Decision.YES, NAME, "disk threshold decider disabled");
445         }
446         // Allow allocation regardless if only a single node is available
447         if (allocation.nodes().size() <= 1) {
448             return allocation.decision(Decision.YES, NAME, "only a single node is present");
449         }
450 
451         ClusterInfo clusterInfo = allocation.clusterInfo();
452         if (clusterInfo == null) {
453             if (logger.isTraceEnabled()) {
454                 logger.trace("Cluster info unavailable for disk threshold decider, allowing allocation.");
455             }
456             return allocation.decision(Decision.YES, NAME, "cluster info unavailable");
457         }
458 
459         Map<String, DiskUsage> usages = clusterInfo.getNodeDiskUsages();
460         if (usages.isEmpty()) {
461             if (logger.isTraceEnabled()) {
462                 logger.trace("Unable to determine disk usages for disk-aware allocation, allowing allocation");
463             }
464             return allocation.decision(Decision.YES, NAME, "disk usages unavailable");
465         }
466 
467         DiskUsage usage = usages.get(node.nodeId());
468         if (usage == null) {
469             // If there is no usage, and we have other nodes in the cluster,
470             // use the average usage for all nodes as the usage for this node
471             usage = averageUsage(node, usages);
472             if (logger.isDebugEnabled()) {
473                 logger.debug("Unable to determine disk usage for {}, defaulting to average across nodes [{} total] [{} free] [{}% free]",
474                         node.nodeId(), usage.getTotalBytes(), usage.getFreeBytes(), usage.getFreeDiskAsPercentage());
475             }
476         }
477 
478         if (includeRelocations) {
479             Map<String, Long> shardSizes = clusterInfo.getShardSizes();
480             long relocatingShardsSize = sizeOfRelocatingShards(node, shardSizes, true);
481             DiskUsage usageIncludingRelocations = new DiskUsage(node.nodeId(), node.node().name(),
482                     usage.getTotalBytes(), usage.getFreeBytes() - relocatingShardsSize);
483             if (logger.isTraceEnabled()) {
484                 logger.trace("usage without relocations: {}", usage);
485                 logger.trace("usage with relocations: [{} bytes] {}", relocatingShardsSize, usageIncludingRelocations);
486             }
487             usage = usageIncludingRelocations;
488         }
489 
490         // If this node is already above the high threshold, the shard cannot remain (get it off!)
491         double freeDiskPercentage = usage.getFreeDiskAsPercentage();
492         long freeBytes = usage.getFreeBytes();
493         if (logger.isDebugEnabled()) {
494             logger.debug("Node [{}] has {}% free disk ({} bytes)", node.nodeId(), freeDiskPercentage, freeBytes);
495         }
496         if (freeBytes < freeBytesThresholdHigh.bytes()) {
497             if (logger.isDebugEnabled()) {
498                 logger.debug("Less than the required {} free bytes threshold ({} bytes free) on node {}, shard cannot remain",
499                         freeBytesThresholdHigh, freeBytes, node.nodeId());
500             }
501             return allocation.decision(Decision.NO, NAME, "after allocation less than required [%s] free on node, free: [%s]",
502                     freeBytesThresholdHigh, new ByteSizeValue(freeBytes));
503         }
504         if (freeDiskPercentage < freeDiskThresholdHigh) {
505             if (logger.isDebugEnabled()) {
506                 logger.debug("Less than the required {}% free disk threshold ({}% free) on node {}, shard cannot remain",
507                         freeDiskThresholdHigh, freeDiskPercentage, node.nodeId());
508             }
509             return allocation.decision(Decision.NO, NAME, "after allocation less than required [%s%%] free disk on node, free: [%s%%]",
510                     freeDiskThresholdHigh, freeDiskPercentage);
511         }
512 
513         return allocation.decision(Decision.YES, NAME, "enough disk for shard to remain on node, free: [%s]", new ByteSizeValue(freeBytes));
514     }

516     /**
517      * Returns a {@link DiskUsage} for the {@link RoutingNode} using the
518      * average usage of other nodes in the disk usage map.
519      * @param node Node to return an averaged DiskUsage object for
520      * @param usages Map of nodeId to DiskUsage for all known nodes
521      * @return DiskUsage representing given node using the average disk usage
522      */
523     public DiskUsage averageUsage(RoutingNode node, Map<String, DiskUsage> usages) {
524         if (usages.size() == 0) {
525             return new DiskUsage(node.nodeId(), node.node().name(), 0, 0);
526         }
527         long totalBytes = 0;
528         long freeBytes = 0;
529         for (DiskUsage du : usages.values()) {
530             totalBytes += du.getTotalBytes();
531             freeBytes += du.getFreeBytes();
532         }
533         return new DiskUsage(node.nodeId(), node.node().name(), totalBytes / usages.size(), freeBytes / usages.size());
534     }

536     /**
537      * Given the DiskUsage for a node and the size of the shard, return the
538      * percentage of free disk if the shard were to be allocated to the node.
539      * @param usage A DiskUsage for the node to have space computed for
540      * @param shardSize Size in bytes of the shard
541      * @return Percentage of free space after the shard is assigned to the node
542      */
543     public double freeDiskPercentageAfterShardAssigned(DiskUsage usage, Long shardSize) {
544         shardSize = (shardSize == null) ? 0 : shardSize;
545         DiskUsage newUsage = new DiskUsage(usage.getNodeId(), usage.getNodeName(),
546                 usage.getTotalBytes(), usage.getFreeBytes() - shardSize);
547         return newUsage.getFreeDiskAsPercentage();
548     }

550     /**
551      * Attempts to parse the watermark into a percentage, returning 100.0% if
552      * it cannot be parsed.
553      */
554     public double thresholdPercentageFromWatermark(String watermark) {
555         try {
556             return RatioValue.parseRatioValue(watermark).getAsPercent();
557         } catch (ElasticsearchParseException ex) {
558             return 100.0;
559         }
560     }

562     /**
563      * Attempts to parse the watermark into a {@link ByteSizeValue}, returning
564      * a ByteSizeValue of 0 bytes if the value cannot be parsed.
565      */
566     public ByteSizeValue thresholdBytesFromWatermark(String watermark) {
567         try {
568             return ByteSizeValue.parseBytesSizeValue(watermark);
569         } catch (ElasticsearchParseException ex) {
570             return ByteSizeValue.parseBytesSizeValue("0b");
571         }

574     /**
575      * Checks if a watermark string is a valid percentage or byte size value,
576      * returning true if valid, false if invalid.
577      */
578     public boolean validWatermarkSetting(String watermark) {
579         try {
580             RatioValue.parseRatioValue(watermark);
581             return true;
582         } catch (ElasticsearchParseException e) {
583             try {
584                 ByteSizeValue.parseBytesSizeValue(watermark);
585                 return true;
586             } catch (ElasticsearchParseException ex) {
587                 return false;
588             }
589         }
