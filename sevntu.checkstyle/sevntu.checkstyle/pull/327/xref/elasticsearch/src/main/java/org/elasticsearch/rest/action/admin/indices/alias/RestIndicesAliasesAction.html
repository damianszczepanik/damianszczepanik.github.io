20  package org.elasticsearch.rest.action.admin.indices.alias;
21  
22  import org.elasticsearch.ElasticsearchIllegalArgumentException;
23  import org.elasticsearch.action.admin.indices.alias.IndicesAliasesRequest;
24  import org.elasticsearch.action.admin.indices.alias.IndicesAliasesResponse;
25  import org.elasticsearch.client.Client;
26  import org.elasticsearch.cluster.metadata.AliasAction;
27  import org.elasticsearch.common.inject.Inject;
28  import org.elasticsearch.common.settings.Settings;
29  import org.elasticsearch.common.xcontent.XContentFactory;
30  import org.elasticsearch.common.xcontent.XContentParser;
31  import org.elasticsearch.rest.*;
32  import org.elasticsearch.rest.action.support.AcknowledgedRestListener;
33  
34  import java.util.Map;
35  
36  import static org.elasticsearch.cluster.metadata.AliasAction.newAddAliasAction;
37  import static org.elasticsearch.rest.RestRequest.Method.POST;
38  
39  /**
40   *
41   */
42  public class RestIndicesAliasesAction extends BaseRestHandler {
43  
44      @Inject
45      public RestIndicesAliasesAction(Settings settings, RestController controller, Client client) {
46          super(settings, controller, client);
47          controller.registerHandler(POST, "/_aliases", this);
48      }
49  
50      @Override
51      public void handleRequest(final RestRequest request, final RestChannel channel, final Client client) throws Exception {
52          IndicesAliasesRequest indicesAliasesRequest = new IndicesAliasesRequest();
53          indicesAliasesRequest.listenerThreaded(false);
54          indicesAliasesRequest.masterNodeTimeout(request.paramAsTime("master_timeout", indicesAliasesRequest.masterNodeTimeout()));
55          try (XContentParser parser = XContentFactory.xContent(request.content()).createParser(request.content())) {
56              // {
57              //     actions : [
58              //         { add : { index : "test1", alias : "alias1", filter : {"user" : "kimchy"} } }
59              //         { remove : { index : "test1", alias : "alias1" } }
60              //     ]
61              // }
62              indicesAliasesRequest.timeout(request.paramAsTime("timeout", indicesAliasesRequest.timeout()));
63              XContentParser.Token token = parser.nextToken();
64              if (token == null) {
65                  throw new ElasticsearchIllegalArgumentException("No action is specified");
66              }
67              while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
68                  if (token == XContentParser.Token.START_ARRAY) {
69                      while ((token = parser.nextToken()) != XContentParser.Token.END_ARRAY) {
70                          if (token == XContentParser.Token.FIELD_NAME) {
71                              String action = parser.currentName();
72                              AliasAction.Type type;
73                              if ("add".equals(action)) {
74                                  type = AliasAction.Type.ADD;
75                              } else if ("remove".equals(action)) {
76                                  type = AliasAction.Type.REMOVE;
77                              } else {
78                                  throw new ElasticsearchIllegalArgumentException("Alias action [" + action + "] not supported");
79                              }
80                              String index = null;
81                              String alias = null;
82                              Map<String, Object> filter = null;
83                              String routing = null;
84                              boolean routingSet = false;
85                              String indexRouting = null;
86                              boolean indexRoutingSet = false;
87                              String searchRouting = null;
88                              boolean searchRoutingSet = false;
89                              String currentFieldName = null;
90                              while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
91                                  if (token == XContentParser.Token.FIELD_NAME) {
92                                      currentFieldName = parser.currentName();
93                                  } else if (token.isValue()) {
94                                      if ("index".equals(currentFieldName)) {
95                                          index = parser.text();
96                                      } else if ("alias".equals(currentFieldName)) {
97                                          alias = parser.text();
98                                      } else if ("routing".equals(currentFieldName)) {
99                                          routing = parser.textOrNull();
100                                         routingSet = true;
101                                     } else if ("indexRouting".equals(currentFieldName) || "index-routing".equals(currentFieldName) || "index_routing".equals(currentFieldName)) {
102                                         indexRouting = parser.textOrNull();
103                                         indexRoutingSet = true;
104                                     } else if ("searchRouting".equals(currentFieldName) || "search-routing".equals(currentFieldName) || "search_routing".equals(currentFieldName)) {
105                                         searchRouting = parser.textOrNull();
106                                         searchRoutingSet = true;
107                                     }
108                                 } else if (token == XContentParser.Token.START_OBJECT) {
109                                     if ("filter".equals(currentFieldName)) {
110                                         filter = parser.mapOrdered();
111                                     }
112                                 }
113                             }
114 
115                             if (type == AliasAction.Type.ADD) {
116                                 AliasAction aliasAction = newAddAliasAction(index, alias).filter(filter);
117                                 if (routingSet) {
118                                     aliasAction.routing(routing);
119                                 }
120                                 if (indexRoutingSet) {
121                                     aliasAction.indexRouting(indexRouting);
122                                 }
123                                 if (searchRoutingSet) {
124                                     aliasAction.searchRouting(searchRouting);
125                                 }
126                                 indicesAliasesRequest.addAliasAction(aliasAction);
127                             } else if (type == AliasAction.Type.REMOVE) {
128                                 indicesAliasesRequest.removeAlias(index, alias);
129                             }
130                         }
131                     }
132                 }
133             }
134         }
135         client.admin().indices().aliases(indicesAliasesRequest, new AcknowledgedRestListener<IndicesAliasesResponse>(channel));
136     }
137 }
