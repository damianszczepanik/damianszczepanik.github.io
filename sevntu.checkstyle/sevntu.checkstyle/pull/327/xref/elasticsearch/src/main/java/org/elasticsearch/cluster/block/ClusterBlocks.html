36  /**
37   * Represents current cluster level blocks to block dirty operations done against the cluster.
38   */
39  public class ClusterBlocks {
40  
41      public static final ClusterBlocks EMPTY_CLUSTER_BLOCK = new ClusterBlocks(ImmutableSet.<ClusterBlock>of(), ImmutableMap.<String, ImmutableSet<ClusterBlock>>of());
42  
43      private final ImmutableSet<ClusterBlock> global;
44  
45      private final ImmutableMap<String, ImmutableSet<ClusterBlock>> indicesBlocks;
46  
47      private final ImmutableLevelHolder[] levelHolders;
48  
49      ClusterBlocks(ImmutableSet<ClusterBlock> global, ImmutableMap<String, ImmutableSet<ClusterBlock>> indicesBlocks) {
50          this.global = global;
51          this.indicesBlocks = indicesBlocks;
52  
53          levelHolders = new ImmutableLevelHolder[ClusterBlockLevel.values().length];
54          for (ClusterBlockLevel level : ClusterBlockLevel.values()) {
55              ImmutableSet.Builder<ClusterBlock> globalBuilder = ImmutableSet.builder();
56              for (ClusterBlock block : global) {
57                  if (block.contains(level)) {
58                      globalBuilder.add(block);
59                  }
60              }
61  
62  
63              ImmutableMap.Builder<String, ImmutableSet<ClusterBlock>> indicesBuilder = ImmutableMap.builder();
64              for (Map.Entry<String, ImmutableSet<ClusterBlock>> entry : indicesBlocks.entrySet()) {
65                  ImmutableSet.Builder<ClusterBlock> indexBuilder = ImmutableSet.builder();
66                  for (ClusterBlock block : entry.getValue()) {
67                      if (block.contains(level)) {
68                          indexBuilder.add(block);
69                      }
70                  }
71  
72                  indicesBuilder.put(entry.getKey(), indexBuilder.build());
73              }
74  
75              levelHolders[level.id()] = new ImmutableLevelHolder(globalBuilder.build(), indicesBuilder.build());
76          }
77      }
78  
79      public ImmutableSet<ClusterBlock> global() {
80          return global;
81      }
82  
83      public ImmutableMap<String, ImmutableSet<ClusterBlock>> indices() {
84          return indicesBlocks;
85      }
86  
87      public ImmutableSet<ClusterBlock> global(ClusterBlockLevel level) {
88          return levelHolders[level.id()].global();
89      }
90  
91      public ImmutableMap<String, ImmutableSet<ClusterBlock>> indices(ClusterBlockLevel level) {
92          return levelHolders[level.id()].indices();
93      }
94  
95      /**
96       * Returns <tt>true</tt> if one of the global blocks as its disable state persistence flag set.
97       */
98      public boolean disableStatePersistence() {
99          for (ClusterBlock clusterBlock : global) {
100             if (clusterBlock.disableStatePersistence()) {
101                 return true;
102             }
103         }
104         return false;
105     }
106 
107     public boolean hasGlobalBlock(ClusterBlock block) {
108         return global.contains(block);
109     }
110 
111     public boolean hasGlobalBlock(int blockId) {
112         for (ClusterBlock clusterBlock : global) {
113             if (clusterBlock.id() == blockId) {
114                 return true;
115             }
116         }
117         return false;
118     }
119 
120     public boolean hasGlobalBlock(ClusterBlockLevel level) {
121         return global(level).size() > 0;
122     }
123 
124     /**
125      * Is there a global block with the provided status?
126      */
127     public boolean hasGlobalBlock(RestStatus status) {
128         for (ClusterBlock clusterBlock : global) {
129             if (clusterBlock.status().equals(status)) {
130                 return true;
131             }
132         }
133         return false;
134     }
135 
136     public boolean hasIndexBlock(String index, ClusterBlock block) {
137         return indicesBlocks.containsKey(index) && indicesBlocks.get(index).contains(block);
138     }
139 
140     public void globalBlockedRaiseException(ClusterBlockLevel level) throws ClusterBlockException {
141         ClusterBlockException blockException = globalBlockedException(level);
142         if (blockException != null) {
143             throw blockException;
144         }
145     }
146 
147     public ClusterBlockException globalBlockedException(ClusterBlockLevel level) {
148         if (global(level).isEmpty()) {
149             return null;
150         }
151         return new ClusterBlockException(ImmutableSet.copyOf(global(level)));
152     }
153 
154     public void indexBlockedRaiseException(ClusterBlockLevel level, String index) throws ClusterBlockException {
155         ClusterBlockException blockException = indexBlockedException(level, index);
156         if (blockException != null) {
157             throw blockException;
158         }
159     }
160 
161     public ClusterBlockException indexBlockedException(ClusterBlockLevel level, String index) {
162         if (!indexBlocked(level, index)) {
163             return null;
164         }
165         ImmutableSet.Builder<ClusterBlock> builder = ImmutableSet.builder();
166         builder.addAll(global(level));
167         ImmutableSet<ClusterBlock> indexBlocks = indices(level).get(index);
168         if (indexBlocks != null) {
169             builder.addAll(indexBlocks);
170         }
171         return new ClusterBlockException(builder.build());
172     }
173 
174     public boolean indexBlocked(ClusterBlockLevel level, String index) {
175         if (!global(level).isEmpty()) {
176             return true;
177         }
178         ImmutableSet<ClusterBlock> indexBlocks = indices(level).get(index);
179         if (indexBlocks != null && !indexBlocks.isEmpty()) {
180             return true;
181         }
182         return false;
183     }
184 
185     public ClusterBlockException indicesBlockedException(ClusterBlockLevel level, String[] indices) {
186         boolean indexIsBlocked = false;
187         for (String index : indices) {
188             if (indexBlocked(level, index)) {
189                 indexIsBlocked = true;
190             }
191         }
192         if (!indexIsBlocked) {
193             return null;
194         }
195         ImmutableSet.Builder<ClusterBlock> builder = ImmutableSet.builder();
196         builder.addAll(global(level));
197         for (String index : indices) {
198             ImmutableSet<ClusterBlock> indexBlocks = indices(level).get(index);
199             if (indexBlocks != null) {
200                 builder.addAll(indexBlocks);
201             }
202         }
203         return new ClusterBlockException(builder.build());
204     }

206     static class ImmutableLevelHolder {
207 
208         static final ImmutableLevelHolder EMPTY = new ImmutableLevelHolder(ImmutableSet.<ClusterBlock>of(), ImmutableMap.<String, ImmutableSet<ClusterBlock>>of());
209 
210         private final ImmutableSet<ClusterBlock> global;
211         private final ImmutableMap<String, ImmutableSet<ClusterBlock>> indices;
212 
213         ImmutableLevelHolder(ImmutableSet<ClusterBlock> global, ImmutableMap<String, ImmutableSet<ClusterBlock>> indices) {
214             this.global = global;
215             this.indices = indices;
216         }
217 
218         public ImmutableSet<ClusterBlock> global() {
219             return global;
220         }
221 
222         public ImmutableMap<String, ImmutableSet<ClusterBlock>> indices() {
223             return indices;
224         }
225     }
226 
227     public static Builder builder() {
228         return new Builder();
229     }
230 
231     public static class Builder {
232 
233         private Set<ClusterBlock> global = Sets.newHashSet();
234 
235         private Map<String, Set<ClusterBlock>> indices = Maps.newHashMap();
236 
237         public Builder() {
238         }
239 
240         public Builder blocks(ClusterBlocks blocks) {
241             global.addAll(blocks.global());
242             for (Map.Entry<String, ImmutableSet<ClusterBlock>> entry : blocks.indices().entrySet()) {
243                 if (!indices.containsKey(entry.getKey())) {
244                     indices.put(entry.getKey(), Sets.<ClusterBlock>newHashSet());
245                 }
246                 indices.get(entry.getKey()).addAll(entry.getValue());
247             }
248             return this;
249         }
250 
251         public Builder addBlocks(IndexMetaData indexMetaData) {
252             if (indexMetaData.state() == IndexMetaData.State.CLOSE) {
253                 addIndexBlock(indexMetaData.index(), MetaDataIndexStateService.INDEX_CLOSED_BLOCK);
254             }
255             if (indexMetaData.settings().getAsBoolean(IndexMetaData.SETTING_READ_ONLY, false)) {
256                 addIndexBlock(indexMetaData.index(), IndexMetaData.INDEX_READ_ONLY_BLOCK);
257             }
258             if (indexMetaData.settings().getAsBoolean(IndexMetaData.SETTING_BLOCKS_READ, false)) {
259                 addIndexBlock(indexMetaData.index(), IndexMetaData.INDEX_READ_BLOCK);
260             }
261             if (indexMetaData.settings().getAsBoolean(IndexMetaData.SETTING_BLOCKS_WRITE, false)) {
262                 addIndexBlock(indexMetaData.index(), IndexMetaData.INDEX_WRITE_BLOCK);
263             }
264             if (indexMetaData.settings().getAsBoolean(IndexMetaData.SETTING_BLOCKS_METADATA, false)) {
265                 addIndexBlock(indexMetaData.index(), IndexMetaData.INDEX_METADATA_BLOCK);
266             }
267             return this;
268         }
269 
270         public Builder addGlobalBlock(ClusterBlock block) {
271             global.add(block);
272             return this;
273         }
274 
275         public Builder removeGlobalBlock(ClusterBlock block) {
276             global.remove(block);
277             return this;
278         }
279 
280         public Builder addIndexBlock(String index, ClusterBlock block) {
281             if (!indices.containsKey(index)) {
282                 indices.put(index, Sets.<ClusterBlock>newHashSet());
283             }
284             indices.get(index).add(block);
285             return this;
286         }
287 
288         public Builder removeIndexBlocks(String index) {
289             if (!indices.containsKey(index)) {
290                 return this;
291             }
292             indices.remove(index);
293             return this;
294         }
295 
296         public Builder removeIndexBlock(String index, ClusterBlock block) {
297             if (!indices.containsKey(index)) {
298                 return this;
299             }
300             indices.get(index).remove(block);
301             if (indices.get(index).isEmpty()) {
302                 indices.remove(index);
303             }
304             return this;
305         }
306 
307         public ClusterBlocks build() {
308             ImmutableMap.Builder<String, ImmutableSet<ClusterBlock>> indicesBuilder = ImmutableMap.builder();
309             for (Map.Entry<String, Set<ClusterBlock>> entry : indices.entrySet()) {
310                 indicesBuilder.put(entry.getKey(), ImmutableSet.copyOf(entry.getValue()));
311             }
312             return new ClusterBlocks(ImmutableSet.copyOf(global), indicesBuilder.build());
313         }
314 
315         public static ClusterBlocks readClusterBlocks(StreamInput in) throws IOException {
316             ImmutableSet<ClusterBlock> global = readBlockSet(in);
317             ImmutableMap.Builder<String, ImmutableSet<ClusterBlock>> indicesBuilder = ImmutableMap.builder();
318             int size = in.readVInt();
319             for (int j = 0; j < size; j++) {
320                 indicesBuilder.put(in.readString().intern(), readBlockSet(in));
321             }
322             return new ClusterBlocks(global, indicesBuilder.build());
323         }
324 
325         public static void writeClusterBlocks(ClusterBlocks blocks, StreamOutput out) throws IOException {
326             writeBlockSet(blocks.global(), out);
327             out.writeVInt(blocks.indices().size());
328             for (Map.Entry<String, ImmutableSet<ClusterBlock>> entry : blocks.indices().entrySet()) {
329                 out.writeString(entry.getKey());
330                 writeBlockSet(entry.getValue(), out);
331             }
332         }
333 
334         private static void writeBlockSet(ImmutableSet<ClusterBlock> blocks, StreamOutput out) throws IOException {
335             out.writeVInt(blocks.size());
336             for (ClusterBlock block : blocks) {
337                 block.writeTo(out);
338             }
339         }
340 
341         private static ImmutableSet<ClusterBlock> readBlockSet(StreamInput in) throws IOException {
342             ImmutableSet.Builder<ClusterBlock> builder = ImmutableSet.builder();
343             int size = in.readVInt();
344             for (int i = 0; i < size; i++) {
345                 builder.add(ClusterBlock.readClusterBlock(in));
346             }
347             return builder.build();
348         }
349     }
