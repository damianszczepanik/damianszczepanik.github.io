20  package org.elasticsearch.cluster.routing.allocation.decider;

22  import com.google.common.collect.ImmutableMap;
23  import org.elasticsearch.cluster.metadata.IndexMetaData;
24  import org.elasticsearch.cluster.node.DiscoveryNodeFilters;
25  import org.elasticsearch.cluster.routing.RoutingNode;
26  import org.elasticsearch.cluster.routing.ShardRouting;
27  import org.elasticsearch.cluster.routing.allocation.RoutingAllocation;
28  import org.elasticsearch.common.inject.Inject;
29  import org.elasticsearch.common.settings.Settings;
30  import org.elasticsearch.node.settings.NodeSettingsService;
31  
32  import static org.elasticsearch.cluster.node.DiscoveryNodeFilters.OpType.AND;
33  import static org.elasticsearch.cluster.node.DiscoveryNodeFilters.OpType.OR;
34  
35  /**
36   * This {@link AllocationDecider} control shard allocation by include and
37   * exclude filters via dynamic cluster and index routing settings.
38   * <p>
39   * This filter is used to make explicit decision on which nodes certain shard
40   * can / should be allocated. The decision if a shard can be allocated, must not
41   * be allocated or should be allocated is based on either cluster wide dynamic
42   * settings (<tt>cluster.routing.allocation.*</tt>) or index specific dynamic
43   * settings (<tt>index.routing.allocation.*</tt>). All of those settings can be
44   * changed at runtime via the cluster or the index update settings API.
45   * </p>
46   * Note: Cluster settings are applied first and will override index specific
47   * settings such that if a shard can be allocated according to the index routing
48   * settings it wont be allocated on a node if the cluster specific settings
49   * would disallow the allocation. Filters are applied in the following order:
50   * <ol>
51   * <li><tt>required</tt> - filters required allocations.
52   * If any <tt>required</tt> filters are set the allocation is denied if the index is <b>not</b> in the set of <tt>required</tt> to allocate on the filtered node</li>
53   * <p/>
54   * <li><tt>include</tt> - filters "allowed" allocations.
55   * If any <tt>include</tt> filters are set the allocation is denied if the index is <b>not</b> in the set of <tt>include</tt> filters for the filtered node</li>
56   * <p/>
57   * <li><tt>exclude</tt> - filters "prohibited" allocations.
58   * If any <tt>exclude</tt> filters are set the allocation is denied if the index is in the set of <tt>exclude</tt> filters for the filtered node</li>
59   * </ol>
60   */
61  public class FilterAllocationDecider extends AllocationDecider {
62  
63      public static final String NAME = "filter";
64  
65      public static final String INDEX_ROUTING_REQUIRE_GROUP = "index.routing.allocation.require.";
66      public static final String INDEX_ROUTING_INCLUDE_GROUP = "index.routing.allocation.include.";
67      public static final String INDEX_ROUTING_EXCLUDE_GROUP = "index.routing.allocation.exclude.";
68  
69      public static final String CLUSTER_ROUTING_REQUIRE_GROUP = "cluster.routing.allocation.require.";
70      public static final String CLUSTER_ROUTING_INCLUDE_GROUP = "cluster.routing.allocation.include.";
71      public static final String CLUSTER_ROUTING_EXCLUDE_GROUP = "cluster.routing.allocation.exclude.";
72  
73      private volatile DiscoveryNodeFilters clusterRequireFilters;
74      private volatile DiscoveryNodeFilters clusterIncludeFilters;
75      private volatile DiscoveryNodeFilters clusterExcludeFilters;
76  
77      @Inject
78      public FilterAllocationDecider(Settings settings, NodeSettingsService nodeSettingsService) {
79          super(settings);
80          ImmutableMap<String, String> requireMap = settings.getByPrefix(CLUSTER_ROUTING_REQUIRE_GROUP).getAsMap();
81          if (requireMap.isEmpty()) {
82              clusterRequireFilters = null;
83          } else {
84              clusterRequireFilters = DiscoveryNodeFilters.buildFromKeyValue(AND, requireMap);
85          }
86          ImmutableMap<String, String> includeMap = settings.getByPrefix(CLUSTER_ROUTING_INCLUDE_GROUP).getAsMap();
87          if (includeMap.isEmpty()) {
88              clusterIncludeFilters = null;
89          } else {
90              clusterIncludeFilters = DiscoveryNodeFilters.buildFromKeyValue(OR, includeMap);
91          }
92          ImmutableMap<String, String> excludeMap = settings.getByPrefix(CLUSTER_ROUTING_EXCLUDE_GROUP).getAsMap();
93          if (excludeMap.isEmpty()) {
94              clusterExcludeFilters = null;
95          } else {
96              clusterExcludeFilters = DiscoveryNodeFilters.buildFromKeyValue(OR, excludeMap);
97          }
98          nodeSettingsService.addListener(new ApplySettings());
99      }
100 
101     @Override
102     public Decision canAllocate(ShardRouting shardRouting, RoutingNode node, RoutingAllocation allocation) {
103         return shouldFilter(shardRouting, node, allocation);
104     }
105 
106     @Override
107     public Decision canRemain(ShardRouting shardRouting, RoutingNode node, RoutingAllocation allocation) {
108         return shouldFilter(shardRouting, node, allocation);
109     }
110 
111     private Decision shouldFilter(ShardRouting shardRouting, RoutingNode node, RoutingAllocation allocation) {
112         if (clusterRequireFilters != null) {
113             if (!clusterRequireFilters.match(node.node())) {
114                 return allocation.decision(Decision.NO, NAME, "node does not match global required filters [%s]", clusterRequireFilters);
115             }
116         }
117         if (clusterIncludeFilters != null) {
118             if (!clusterIncludeFilters.match(node.node())) {
119                 return allocation.decision(Decision.NO, NAME, "node does not match global include filters [%s]", clusterIncludeFilters);
120             }
121         }
122         if (clusterExcludeFilters != null) {
123             if (clusterExcludeFilters.match(node.node())) {
124                 return allocation.decision(Decision.NO, NAME, "node matches global exclude filters [%s]", clusterExcludeFilters);
125             }
126         }
127 
128         IndexMetaData indexMd = allocation.routingNodes().metaData().index(shardRouting.index());
129         if (indexMd.requireFilters() != null) {
130             if (!indexMd.requireFilters().match(node.node())) {
131                 return allocation.decision(Decision.NO, NAME, "node does not match index required filters [%s]", indexMd.requireFilters());
132             }
133         }
134         if (indexMd.includeFilters() != null) {
135             if (!indexMd.includeFilters().match(node.node())) {
136                 return allocation.decision(Decision.NO, NAME, "node does not match index include filters [%s]", indexMd.includeFilters());
137             }
138         }
139         if (indexMd.excludeFilters() != null) {
140             if (indexMd.excludeFilters().match(node.node())) {
141                 return allocation.decision(Decision.NO, NAME, "node matches index exclude filters [%s]", indexMd.excludeFilters());
142             }
143         }
144 
145         return allocation.decision(Decision.YES, NAME, "node passes include/exclude/require filters");
146     }
147 
148     class ApplySettings implements NodeSettingsService.Listener {
149         @Override
150         public void onRefreshSettings(Settings settings) {
151             ImmutableMap<String, String> requireMap = settings.getByPrefix(CLUSTER_ROUTING_REQUIRE_GROUP).getAsMap();
152             if (!requireMap.isEmpty()) {
153                 clusterRequireFilters = DiscoveryNodeFilters.buildFromKeyValue(AND, requireMap);
154             }
155             ImmutableMap<String, String> includeMap = settings.getByPrefix(CLUSTER_ROUTING_INCLUDE_GROUP).getAsMap();
156             if (!includeMap.isEmpty()) {
157                 clusterIncludeFilters = DiscoveryNodeFilters.buildFromKeyValue(OR, includeMap);
158             }
159             ImmutableMap<String, String> excludeMap = settings.getByPrefix(CLUSTER_ROUTING_EXCLUDE_GROUP).getAsMap();
160             if (!excludeMap.isEmpty()) {
161                 clusterExcludeFilters = DiscoveryNodeFilters.buildFromKeyValue(OR, excludeMap);
162             }
163         }
164     }
165 }
