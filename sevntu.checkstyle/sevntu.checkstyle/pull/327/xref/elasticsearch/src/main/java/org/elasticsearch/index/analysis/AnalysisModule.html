123     public AnalysisModule(Settings settings, IndicesAnalysisService indicesAnalysisService) {
124         this.settings = settings;
125         this.indicesAnalysisService = indicesAnalysisService;
126         processors.add(new DefaultProcessor());
127         try {
128             processors.add(new ExtendedProcessor());
129         } catch (Throwable t) {
130             // ignore. no extended ones
131         }
132     }

134     public AnalysisModule addProcessor(AnalysisBinderProcessor processor) {
135         processors.addFirst(processor);
136         return this;
137     }

159     @Override
160     protected void configure() {
161         MapBinder<String, CharFilterFactoryFactory> charFilterBinder
162                 = MapBinder.newMapBinder(binder(), String.class, CharFilterFactoryFactory.class);
163 
164         // CHAR FILTERS
165 
166         AnalysisBinderProcessor.CharFiltersBindings charFiltersBindings = new AnalysisBinderProcessor.CharFiltersBindings();
167         for (AnalysisBinderProcessor processor : processors) {
168             processor.processCharFilters(charFiltersBindings);
169         }
170         charFiltersBindings.charFilters.putAll(charFilters);
171 
172         Map<String, Settings> charFiltersSettings = settings.getGroups("index.analysis.char_filter");
173         for (Map.Entry<String, Settings> entry : charFiltersSettings.entrySet()) {
174             String charFilterName = entry.getKey();
175             Settings charFilterSettings = entry.getValue();
176 
177             Class<? extends CharFilterFactory> type = null;
178             try {
179                 type = charFilterSettings.getAsClass("type", null, "org.elasticsearch.index.analysis.", "CharFilterFactory");
180             } catch (NoClassSettingsException e) {
181                 // nothing found, see if its in bindings as a binding name
182                 if (charFilterSettings.get("type") != null) {
183                     type = charFiltersBindings.charFilters.get(Strings.toUnderscoreCase(charFilterSettings.get("type")));
184                     if (type == null) {
185                         type = charFiltersBindings.charFilters.get(Strings.toCamelCase(charFilterSettings.get("type")));
186                     }
187                 }
188                 if (type == null) {
189                     throw new ElasticsearchIllegalArgumentException("failed to find char filter type [" + charFilterSettings.get("type") + "] for [" + charFilterName + "]", e);
190                 }
191             }
192             if (type == null) {
193                 // nothing found, see if its in bindings as a binding name
194                 throw new ElasticsearchIllegalArgumentException("Char Filter [" + charFilterName + "] must have a type associated with it");
195             }
196             charFilterBinder.addBinding(charFilterName).toProvider(FactoryProvider.newFactory(CharFilterFactoryFactory.class, type)).in(Scopes.SINGLETON);
197         }
198         // go over the char filters in the bindings and register the ones that are not configured
199         for (Map.Entry<String, Class<? extends CharFilterFactory>> entry : charFiltersBindings.charFilters.entrySet()) {
200             String charFilterName = entry.getKey();
201             Class<? extends CharFilterFactory> clazz = entry.getValue();
202             // we don't want to re-register one that already exists
203             if (charFiltersSettings.containsKey(charFilterName)) {
204                 continue;
205             }
206             // check, if it requires settings, then don't register it, we know default has no settings...
207             if (clazz.getAnnotation(AnalysisSettingsRequired.class) != null) {
208                 continue;
209             }
210             // register it as default under the name
211             if (indicesAnalysisService != null && indicesAnalysisService.hasCharFilter(charFilterName)) {
212                 // don't register it here, we will use explicitly register it in the AnalysisService
213                 //charFilterBinder.addBinding(charFilterName).toInstance(indicesAnalysisService.charFilterFactoryFactory(charFilterName));
214             } else {
215                 charFilterBinder.addBinding(charFilterName).toProvider(FactoryProvider.newFactory(CharFilterFactoryFactory.class, clazz)).in(Scopes.SINGLETON);
216             }
217         }
218 
219 
220         // TOKEN FILTERS
221 
222         MapBinder<String, TokenFilterFactoryFactory> tokenFilterBinder
223                 = MapBinder.newMapBinder(binder(), String.class, TokenFilterFactoryFactory.class);
224 
225         // initial default bindings
226         AnalysisBinderProcessor.TokenFiltersBindings tokenFiltersBindings = new AnalysisBinderProcessor.TokenFiltersBindings();
227         for (AnalysisBinderProcessor processor : processors) {
228             processor.processTokenFilters(tokenFiltersBindings);
229         }
230         tokenFiltersBindings.tokenFilters.putAll(tokenFilters);
231 
232         Map<String, Settings> tokenFiltersSettings = settings.getGroups("index.analysis.filter");
233         for (Map.Entry<String, Settings> entry : tokenFiltersSettings.entrySet()) {
234             String tokenFilterName = entry.getKey();
235             Settings tokenFilterSettings = entry.getValue();
236 
237             Class<? extends TokenFilterFactory> type = null;
238             try {
239                 type = tokenFilterSettings.getAsClass("type", null, "org.elasticsearch.index.analysis.", "TokenFilterFactory");
240             } catch (NoClassSettingsException e) {
241                 // nothing found, see if its in bindings as a binding name
242                 if (tokenFilterSettings.get("type") != null) {
243                     type = tokenFiltersBindings.tokenFilters.get(Strings.toUnderscoreCase(tokenFilterSettings.get("type")));
244                     if (type == null) {
245                         type = tokenFiltersBindings.tokenFilters.get(Strings.toCamelCase(tokenFilterSettings.get("type")));
246                     }
247                 }
248                 if (type == null) {
249                     throw new ElasticsearchIllegalArgumentException("failed to find token filter type [" + tokenFilterSettings.get("type") + "] for [" + tokenFilterName + "]", e);
250                 }
251             }
252             if (type == null) {
253                 throw new ElasticsearchIllegalArgumentException("token filter [" + tokenFilterName + "] must have a type associated with it");
254             }
255             tokenFilterBinder.addBinding(tokenFilterName).toProvider(FactoryProvider.newFactory(TokenFilterFactoryFactory.class, type)).in(Scopes.SINGLETON);
256         }
257         // go over the filters in the bindings and register the ones that are not configured
258         for (Map.Entry<String, Class<? extends TokenFilterFactory>> entry : tokenFiltersBindings.tokenFilters.entrySet()) {
259             String tokenFilterName = entry.getKey();
260             Class<? extends TokenFilterFactory> clazz = entry.getValue();
261             // we don't want to re-register one that already exists
262             if (tokenFiltersSettings.containsKey(tokenFilterName)) {
263                 continue;
264             }
265             // check, if it requires settings, then don't register it, we know default has no settings...
266             if (clazz.getAnnotation(AnalysisSettingsRequired.class) != null) {
267                 continue;
268             }
269             // register it as default under the name
270             if (indicesAnalysisService != null && indicesAnalysisService.hasTokenFilter(tokenFilterName)) {
271                 // don't register it here, we will use explicitly register it in the AnalysisService
272                 // tokenFilterBinder.addBinding(tokenFilterName).toInstance(indicesAnalysisService.tokenFilterFactoryFactory(tokenFilterName));
273             } else {
274                 tokenFilterBinder.addBinding(tokenFilterName).toProvider(FactoryProvider.newFactory(TokenFilterFactoryFactory.class, clazz)).in(Scopes.SINGLETON);
275             }
276         }
277 
278         // TOKENIZER
279 
280         MapBinder<String, TokenizerFactoryFactory> tokenizerBinder
281                 = MapBinder.newMapBinder(binder(), String.class, TokenizerFactoryFactory.class);
282 
283         // initial default bindings
284         AnalysisBinderProcessor.TokenizersBindings tokenizersBindings = new AnalysisBinderProcessor.TokenizersBindings();
285         for (AnalysisBinderProcessor processor : processors) {
286             processor.processTokenizers(tokenizersBindings);
287         }
288         tokenizersBindings.tokenizers.putAll(tokenizers);
289 
290         Map<String, Settings> tokenizersSettings = settings.getGroups("index.analysis.tokenizer");
291         for (Map.Entry<String, Settings> entry : tokenizersSettings.entrySet()) {
292             String tokenizerName = entry.getKey();
293             Settings tokenizerSettings = entry.getValue();
294 
295 
296             Class<? extends TokenizerFactory> type = null;
297             try {
298                 type = tokenizerSettings.getAsClass("type", null, "org.elasticsearch.index.analysis.", "TokenizerFactory");
299             } catch (NoClassSettingsException e) {
300                 // nothing found, see if its in bindings as a binding name
301                 if (tokenizerSettings.get("type") != null) {
302                     type = tokenizersBindings.tokenizers.get(Strings.toUnderscoreCase(tokenizerSettings.get("type")));
303                     if (type == null) {
304                         type = tokenizersBindings.tokenizers.get(Strings.toCamelCase(tokenizerSettings.get("type")));
305                     }
306                 }
307                 if (type == null) {
308                     throw new ElasticsearchIllegalArgumentException("failed to find tokenizer type [" + tokenizerSettings.get("type") + "] for [" + tokenizerName + "]", e);
309                 }
310             }
311             if (type == null) {
312                 throw new ElasticsearchIllegalArgumentException("token filter [" + tokenizerName + "] must have a type associated with it");
313             }
314             tokenizerBinder.addBinding(tokenizerName).toProvider(FactoryProvider.newFactory(TokenizerFactoryFactory.class, type)).in(Scopes.SINGLETON);
315         }
316         // go over the tokenizers in the bindings and register the ones that are not configured
317         for (Map.Entry<String, Class<? extends TokenizerFactory>> entry : tokenizersBindings.tokenizers.entrySet()) {
318             String tokenizerName = entry.getKey();
319             Class<? extends TokenizerFactory> clazz = entry.getValue();
320             // we don't want to re-register one that already exists
321             if (tokenizersSettings.containsKey(tokenizerName)) {
322                 continue;
323             }
324             // check, if it requires settings, then don't register it, we know default has no settings...
325             if (clazz.getAnnotation(AnalysisSettingsRequired.class) != null) {
326                 continue;
327             }
328             // register it as default under the name
329             if (indicesAnalysisService != null && indicesAnalysisService.hasTokenizer(tokenizerName)) {
330                 // don't register it here, we will use explicitly register it in the AnalysisService
331                 // tokenizerBinder.addBinding(tokenizerName).toProvider(FactoryProvider.newFactory(TokenizerFactoryFactory.class, clazz)).in(Scopes.SINGLETON);
332             } else {
333                 tokenizerBinder.addBinding(tokenizerName).toProvider(FactoryProvider.newFactory(TokenizerFactoryFactory.class, clazz)).in(Scopes.SINGLETON);
334             }
335         }
336 
337         // ANALYZER
338 
339         MapBinder<String, AnalyzerProviderFactory> analyzerBinder
340                 = MapBinder.newMapBinder(binder(), String.class, AnalyzerProviderFactory.class);
341 
342         // initial default bindings
343         AnalysisBinderProcessor.AnalyzersBindings analyzersBindings = new AnalysisBinderProcessor.AnalyzersBindings();
344         for (AnalysisBinderProcessor processor : processors) {
345             processor.processAnalyzers(analyzersBindings);
346         }
347         analyzersBindings.analyzers.putAll(analyzers);
348 
349         Map<String, Settings> analyzersSettings = settings.getGroups("index.analysis.analyzer");
350         for (Map.Entry<String, Settings> entry : analyzersSettings.entrySet()) {
351             String analyzerName = entry.getKey();
352             Settings analyzerSettings = entry.getValue();
353 
354             Class<? extends AnalyzerProvider> type = null;
355             try {
356                 type = analyzerSettings.getAsClass("type", null, "org.elasticsearch.index.analysis.", "AnalyzerProvider");
357             } catch (NoClassSettingsException e) {
358                 // nothing found, see if its in bindings as a binding name
359                 if (analyzerSettings.get("type") != null) {
360                     type = analyzersBindings.analyzers.get(Strings.toUnderscoreCase(analyzerSettings.get("type")));
361                     if (type == null) {
362                         type = analyzersBindings.analyzers.get(Strings.toCamelCase(analyzerSettings.get("type")));
363                     }
364                 }
365                 if (type == null) {
366                     // no specific type, check if it has a tokenizer associated with it
367                     String tokenizerName = analyzerSettings.get("tokenizer");
368                     if (tokenizerName != null) {
369                         // we have a tokenizer, use the CustomAnalyzer
370                         type = CustomAnalyzerProvider.class;
371                     } else {
372                         throw new ElasticsearchIllegalArgumentException("failed to find analyzer type [" + analyzerSettings.get("type") + "] or tokenizer for [" + analyzerName + "]", e);
373                     }
374                 }
375             }
376             if (type == null) {
377                 // no specific type, check if it has a tokenizer associated with it
378                 String tokenizerName = analyzerSettings.get("tokenizer");
379                 if (tokenizerName != null) {
380                     // we have a tokenizer, use the CustomAnalyzer
381                     type = CustomAnalyzerProvider.class;
382                 } else {
383                     throw new ElasticsearchIllegalArgumentException("failed to find analyzer type [" + analyzerSettings.get("type") + "] or tokenizer for [" + analyzerName + "]");
384                 }
385             }
386             analyzerBinder.addBinding(analyzerName).toProvider(FactoryProvider.newFactory(AnalyzerProviderFactory.class, type)).in(Scopes.SINGLETON);
387         }
388 
389 
390         // go over the analyzers in the bindings and register the ones that are not configured
391         for (Map.Entry<String, Class<? extends AnalyzerProvider>> entry : analyzersBindings.analyzers.entrySet()) {
392             String analyzerName = entry.getKey();
393             Class<? extends AnalyzerProvider> clazz = entry.getValue();
394             // we don't want to re-register one that already exists
395             if (analyzersSettings.containsKey(analyzerName)) {
396                 continue;
397             }
398             // check, if it requires settings, then don't register it, we know default has no settings...
399             if (clazz.getAnnotation(AnalysisSettingsRequired.class) != null) {
400                 continue;
401             }
402             // register it as default under the name
403             if (indicesAnalysisService != null && indicesAnalysisService.hasAnalyzer(analyzerName)) {
404                 // don't register it here, we will use explicitly register it in the AnalysisService
405                 // analyzerBinder.addBinding(analyzerName).toProvider(FactoryProvider.newFactory(AnalyzerProviderFactory.class, clazz)).in(Scopes.SINGLETON);
406             } else {
407                 analyzerBinder.addBinding(analyzerName).toProvider(FactoryProvider.newFactory(AnalyzerProviderFactory.class, clazz)).in(Scopes.SINGLETON);
408             }
409         }
410 
411         bind(AnalysisService.class).in(Scopes.SINGLETON);
412     }
