20  package org.elasticsearch.action.search.type;
21  
22  import com.google.common.collect.ImmutableMap;
23  import org.elasticsearch.action.ActionListener;
24  import org.elasticsearch.action.search.SearchRequest;
25  import org.elasticsearch.action.search.SearchResponse;
26  import org.elasticsearch.action.support.ActionFilters;
27  import org.elasticsearch.cluster.ClusterService;
28  import org.elasticsearch.cluster.node.DiscoveryNode;
29  import org.elasticsearch.common.inject.Inject;
30  import org.elasticsearch.common.settings.Settings;
31  import org.elasticsearch.common.util.concurrent.AtomicArray;
32  import org.elasticsearch.search.action.SearchServiceListener;
33  import org.elasticsearch.search.action.SearchServiceTransportAction;
34  import org.elasticsearch.search.controller.SearchPhaseController;
35  import org.elasticsearch.search.fetch.FetchSearchResultProvider;
36  import org.elasticsearch.search.internal.InternalSearchResponse;
37  import org.elasticsearch.search.internal.ShardSearchTransportRequest;
38  import org.elasticsearch.search.query.QuerySearchResult;
39  import org.elasticsearch.threadpool.ThreadPool;
40  
41  import static org.elasticsearch.action.search.type.TransportSearchHelper.buildScrollId;
42  
43  public class TransportSearchScanAction extends TransportSearchTypeAction {
44  
45      @Inject
46      public TransportSearchScanAction(Settings settings, ThreadPool threadPool, ClusterService clusterService,
47                                       SearchServiceTransportAction searchService, SearchPhaseController searchPhaseController, ActionFilters actionFilters) {
48          super(settings, threadPool, clusterService, searchService, searchPhaseController, actionFilters);
49      }
50  
51      @Override
52      protected void doExecute(SearchRequest searchRequest, ActionListener<SearchResponse> listener) {
53          new AsyncAction(searchRequest, listener).start();
54      }
55  
56      private class AsyncAction extends BaseAsyncAction<QuerySearchResult> {
57  
58          private AsyncAction(SearchRequest request, ActionListener<SearchResponse> listener) {
59              super(request, listener);
60          }
61  
62          @Override
63          protected String firstPhaseName() {
64              return "init_scan";
65          }
66  
67          @Override
68          protected void sendExecuteFirstPhase(DiscoveryNode node, ShardSearchTransportRequest request, SearchServiceListener<QuerySearchResult> listener) {
69              searchService.sendExecuteScan(node, request, listener);
70          }
71  
72          @Override
73          protected void moveToSecondPhase() throws Exception {
74              final InternalSearchResponse internalResponse = searchPhaseController.merge(SearchPhaseController.EMPTY_DOCS, firstResults, (AtomicArray<? extends FetchSearchResultProvider>) AtomicArray.empty());
75              String scrollId = null;
76              if (request.scroll() != null) {
77                  scrollId = buildScrollId(request.searchType(), firstResults, ImmutableMap.of("total_hits", Long.toString(internalResponse.hits().totalHits())));
78              }
79              listener.onResponse(new SearchResponse(internalResponse, scrollId, expectedSuccessfulOps, successfulOps.get(), buildTookInMillis(), buildShardFailures()));
80          }
81      }
82  }
