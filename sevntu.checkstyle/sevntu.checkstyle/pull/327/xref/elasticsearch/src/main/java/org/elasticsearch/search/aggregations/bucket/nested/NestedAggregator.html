50  public class NestedAggregator extends SingleBucketAggregator {
51  
52      private BitDocIdSetFilter parentFilter;
53      private final Filter childFilter;
54  
55      private DocIdSetIterator childDocs;
56      private BitSet parentDocs;
57  
58      public NestedAggregator(String name, AggregatorFactories factories, ObjectMapper objectMapper, AggregationContext aggregationContext, Aggregator parentAggregator, Map<String, Object> metaData, FilterCachingPolicy filterCachingPolicy) throws IOException {
59          super(name, factories, aggregationContext, parentAggregator, metaData);
60          childFilter = aggregationContext.searchContext().filterCache().cache(objectMapper.nestedTypeFilter(), null, filterCachingPolicy);
61      }
62  
63      @Override
64      public LeafBucketCollector getLeafCollector(final LeafReaderContext ctx, final LeafBucketCollector sub) throws IOException {
65          // Reset parentFilter, so we resolve the parentDocs for each new segment being searched
66          this.parentFilter = null;
67          // In ES if parent is deleted, then also the children are deleted. Therefore acceptedDocs can also null here.
68          DocIdSet childDocIdSet = childFilter.getDocIdSet(ctx, null);
69          if (DocIdSets.isEmpty(childDocIdSet)) {
70              childDocs = null;
71          } else {
72              childDocs = childDocIdSet.iterator();
73          }
74  
75          return new LeafBucketCollectorBase(sub, null) {
76              @Override
77              public void collect(int parentDoc, long bucket) throws IOException {
78                  // here we translate the parent doc to a list of its nested docs, and then call super.collect for evey one of them so they'll be collected
79  
80                  // if parentDoc is 0 then this means that this parent doesn't have child docs (b/c these appear always before the parent doc), so we can skip:
81                  if (parentDoc == 0 || childDocs == null) {
82                      return;
83                  }
84                  if (parentFilter == null) {
85                      // The aggs are instantiated in reverse, first the most inner nested aggs and lastly the top level aggs
86                      // So at the time a nested 'nested' aggs is parsed its closest parent nested aggs hasn't been constructed.
87                      // So the trick is to set at the last moment just before needed and we can use its child filter as the
88                      // parent filter.
89  
90                      // Additional NOTE: Before this logic was performed in the setNextReader(...) method, but the the assumption
91                      // that aggs instances are constructed in reverse doesn't hold when buckets are constructed lazily during
92                      // aggs execution
93                      Filter parentFilterNotCached = findClosestNestedPath(parent());
94                      if (parentFilterNotCached == null) {
95                          parentFilterNotCached = NonNestedDocsFilter.INSTANCE;
96                      }
97                      parentFilter = context.searchContext().bitsetFilterCache().getBitDocIdSetFilter(parentFilterNotCached);
98                      BitDocIdSet parentSet = parentFilter.getDocIdSet(ctx);
99                      if (DocIdSets.isEmpty(parentSet)) {
100                         // There are no parentDocs in the segment, so return and set childDocs to null, so we exit early for future invocations.
101                         childDocs = null;
102                         return;
103                     } else {
104                         parentDocs = parentSet.bits();
105                     }
106                 }
107 
108                 final int prevParentDoc = parentDocs.prevSetBit(parentDoc - 1);
109                 int childDocId = childDocs.docID();
110                 if (childDocId <= prevParentDoc) {
111                     childDocId = childDocs.advance(prevParentDoc + 1);
112                 }
113 
114                 for (; childDocId < parentDoc; childDocId = childDocs.nextDoc()) {
115                     collectBucket(sub, childDocId, bucket);
116                 }
117             }
118         };
119     }
120         
121     @Override
122     public InternalAggregation buildAggregation(long owningBucketOrdinal) throws IOException {
123         return new InternalNested(name, bucketDocCount(owningBucketOrdinal), bucketAggregations(owningBucketOrdinal), metaData());
124     }
125 
126         @Override
127     public InternalAggregation buildEmptyAggregation() {
128         return new InternalNested(name, 0, buildEmptySubAggregations(), metaData());
129     }
130 
131     private static Filter findClosestNestedPath(Aggregator parent) {
132         for (; parent != null; parent = parent.parent()) {
133             if (parent instanceof NestedAggregator) {
134                 return ((NestedAggregator) parent).childFilter;
135             } else if (parent instanceof ReverseNestedAggregator) {
136                 return ((ReverseNestedAggregator) parent).getParentFilter();
137             }
138         }
139         return null;
140     }
141 
142     public static class Factory extends AggregatorFactory {
143 
144         private final String path;
145         private final FilterCachingPolicy filterCachingPolicy;
146 
147         public Factory(String name, String path, FilterCachingPolicy filterCachingPolicy) {
148             super(name, InternalNested.TYPE.name());
149             this.path = path;
150             this.filterCachingPolicy = filterCachingPolicy;
151         }
152 
153         @Override
154         public Aggregator createInternal(AggregationContext context, Aggregator parent, boolean collectsFromSingleBucket, Map<String, Object> metaData) throws IOException {
155             if (collectsFromSingleBucket == false) {
156                 return asMultiBucketAggregator(this, context, parent);
157             }
158             MapperService.SmartNameObjectMapper mapper = context.searchContext().smartNameObjectMapper(path);
159             if (mapper == null) {
160                 return new Unmapped(name, context, parent, metaData);
161             }
162             ObjectMapper objectMapper = mapper.mapper();
163             if (objectMapper == null) {
164                 return new Unmapped(name, context, parent, metaData);
165             }
166             if (!objectMapper.nested().isNested()) {
167                 throw new AggregationExecutionException("[nested] nested path [" + path + "] is not nested");
168             }
169             return new NestedAggregator(name, factories, objectMapper, context, parent, metaData, filterCachingPolicy);
170         }
171 
172         private final static class Unmapped extends NonCollectingAggregator {
173 
174             public Unmapped(String name, AggregationContext context, Aggregator parent, Map<String, Object> metaData) throws IOException {
175                 super(name, context, parent, metaData);
176             }
177 
178             @Override
179             public InternalAggregation buildEmptyAggregation() {
180                 return new InternalNested(name, 0, buildEmptySubAggregations(), metaData());
181             }
182         }
183     }
