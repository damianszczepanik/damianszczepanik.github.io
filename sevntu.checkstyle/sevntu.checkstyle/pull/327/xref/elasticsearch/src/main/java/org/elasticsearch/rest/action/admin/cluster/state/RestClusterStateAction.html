20  package org.elasticsearch.rest.action.admin.cluster.state;
21  
22  import org.elasticsearch.action.admin.cluster.state.ClusterStateRequest;
23  import org.elasticsearch.action.admin.cluster.state.ClusterStateResponse;
24  import org.elasticsearch.action.support.IndicesOptions;
25  import org.elasticsearch.client.Client;
26  import org.elasticsearch.client.Requests;
27  import org.elasticsearch.cluster.ClusterState;
28  import org.elasticsearch.common.Strings;
29  import org.elasticsearch.common.inject.Inject;
30  import org.elasticsearch.common.settings.Settings;
31  import org.elasticsearch.common.settings.SettingsFilter;
32  import org.elasticsearch.common.xcontent.XContentBuilder;
33  import org.elasticsearch.common.xcontent.XContentBuilderString;
34  import org.elasticsearch.rest.*;
35  import org.elasticsearch.rest.action.support.RestBuilderListener;
36  
37  import java.util.EnumSet;
38  
39  
40  /**
41   *
42   */
43  public class RestClusterStateAction extends BaseRestHandler {
44  
45      private final SettingsFilter settingsFilter;
46  
47      @Inject
48      public RestClusterStateAction(Settings settings, RestController controller, Client client, SettingsFilter settingsFilter) {
49          super(settings, controller, client);
50          controller.registerHandler(RestRequest.Method.GET, "/_cluster/state", this);
51          controller.registerHandler(RestRequest.Method.GET, "/_cluster/state/{metric}", this);
52          controller.registerHandler(RestRequest.Method.GET, "/_cluster/state/{metric}/{indices}", this);
53  
54          this.settingsFilter = settingsFilter;
55      }
56  
57      @Override
58      public void handleRequest(final RestRequest request, final RestChannel channel, final Client client) {
59          final ClusterStateRequest clusterStateRequest = Requests.clusterStateRequest();
60          clusterStateRequest.listenerThreaded(false);
61          clusterStateRequest.indicesOptions(IndicesOptions.fromRequest(request, clusterStateRequest.indicesOptions()));
62          clusterStateRequest.local(request.paramAsBoolean("local", clusterStateRequest.local()));
63          clusterStateRequest.masterNodeTimeout(request.paramAsTime("master_timeout", clusterStateRequest.masterNodeTimeout()));
64  
65          final String[] indices = Strings.splitStringByCommaToArray(request.param("indices", "_all"));
66          boolean isAllIndicesOnly = indices.length == 1 && "_all".equals(indices[0]);
67          if (!isAllIndicesOnly) {
68              clusterStateRequest.indices(indices);
69          }
70  
71          if (request.hasParam("metric")) {
72              EnumSet<ClusterState.Metric> metrics = ClusterState.Metric.parseString(request.param("metric"), true);
73              // do not ask for what we do not need.
74              clusterStateRequest.nodes(metrics.contains(ClusterState.Metric.NODES) || metrics.contains(ClusterState.Metric.MASTER_NODE));
75              clusterStateRequest.routingTable(metrics.contains(ClusterState.Metric.ROUTING_TABLE));
76              clusterStateRequest.metaData(metrics.contains(ClusterState.Metric.METADATA));
77              clusterStateRequest.blocks(metrics.contains(ClusterState.Metric.BLOCKS));
78          }
79          settingsFilter.addFilterSettingParams(request);
80  
81          client.admin().cluster().state(clusterStateRequest, new RestBuilderListener<ClusterStateResponse>(channel) {
82              @Override
83              public RestResponse buildResponse(ClusterStateResponse response, XContentBuilder builder) throws Exception {
84                  builder.startObject();
85                  builder.field(Fields.CLUSTER_NAME, response.getClusterName().value());
86                  response.getState().toXContent(builder, request);
87                  builder.endObject();
88                  return new BytesRestResponse(RestStatus.OK, builder);
89              }
90          });
91      }
92  
93      static final class Fields {
94          static final XContentBuilderString CLUSTER_NAME = new XContentBuilderString("cluster_name");
95      }
96  }
