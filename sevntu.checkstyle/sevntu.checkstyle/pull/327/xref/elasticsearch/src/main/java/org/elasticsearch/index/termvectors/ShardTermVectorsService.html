79      public TermVectorsResponse getTermVectors(TermVectorsRequest request, String concreteIndex) {
80          final TermVectorsResponse termVectorsResponse = new TermVectorsResponse(concreteIndex, request.type(), request.id());
81          final Term uidTerm = new Term(UidFieldMapper.NAME, Uid.createUidAsBytes(request.type(), request.id()));
82  
83          Engine.GetResult get = indexShard.get(new Engine.Get(request.realtime(), uidTerm).version(request.version()).versionType(request.versionType()));
84  
85          boolean docFromTranslog = get.source() != null;
86          AggregatedDfs dfs = null;
87  
88          /* fetched from translog is treated as an artificial document */
89          if (docFromTranslog) {
90              request.doc(get.source().source, false);
91              termVectorsResponse.setDocVersion(get.version());
92          }
93  
94          /* handle potential wildcards in fields */
95          if (request.selectedFields() != null) {
96              handleFieldWildcards(request);
97          }
98  
99          final Engine.Searcher searcher = indexShard.acquireSearcher("term_vector");
100         try {
101             Fields topLevelFields = MultiFields.getFields(get.searcher() != null ? get.searcher().reader() : searcher.reader());
102             Versions.DocIdAndVersion docIdAndVersion = get.docIdAndVersion();
103             /* from an artificial document */
104             if (request.doc() != null) {
105                 Fields termVectorsByField = generateTermVectorsFromDoc(request, !docFromTranslog);
106                 // if no document indexed in shard, take the queried document itself for stats
107                 if (topLevelFields == null) {
108                     topLevelFields = termVectorsByField;
109                 }
110                 if (termVectorsByField != null && useDfs(request)) {
111                     dfs = getAggregatedDfs(termVectorsByField, request);
112                 }
113                 termVectorsResponse.setFields(termVectorsByField, request.selectedFields(), request.getFlags(), topLevelFields, dfs);
114                 termVectorsResponse.setExists(true);
115                 termVectorsResponse.setArtificial(!docFromTranslog);
116             }
117             /* or from an existing document */
118             else if (docIdAndVersion != null) {
119                 // fields with stored term vectors
120                 Fields termVectorsByField = docIdAndVersion.context.reader().getTermVectors(docIdAndVersion.docId);
121                 Set<String> selectedFields = request.selectedFields();
122                 // generate tvs for fields where analyzer is overridden
123                 if (selectedFields == null && request.perFieldAnalyzer() != null) {
124                     selectedFields = getFieldsToGenerate(request.perFieldAnalyzer(), termVectorsByField);
125                 }
126                 // fields without term vectors
127                 if (selectedFields != null) {
128                     termVectorsByField = addGeneratedTermVectors(get, termVectorsByField, request, selectedFields);
129                 }
130                 if (termVectorsByField != null && useDfs(request)) {
131                     dfs = getAggregatedDfs(termVectorsByField, request);
132                 }
133                 termVectorsResponse.setFields(termVectorsByField, request.selectedFields(), request.getFlags(), topLevelFields, dfs);
134                 termVectorsResponse.setDocVersion(docIdAndVersion.version);
135                 termVectorsResponse.setExists(true);
136             } else {
137                 termVectorsResponse.setExists(false);
138             }
139         } catch (Throwable ex) {
140             throw new ElasticsearchException("failed to execute term vector request", ex);
141         } finally {
142             searcher.close();
143             get.release();
144         }
145         return termVectorsResponse;
146     }

148     private void handleFieldWildcards(TermVectorsRequest request) {
149         Set<String> fieldNames = new HashSet<>();
150         for (String pattern : request.selectedFields()) {
151             fieldNames.addAll(indexShard.mapperService().simpleMatchToIndexNames(pattern));
152         }
153         request.selectedFields(fieldNames.toArray(Strings.EMPTY_ARRAY));
154     }

156     private boolean isValidField(FieldMapper field) {
157         // must be a string
158         if (!(field instanceof StringFieldMapper)) {
159             return false;
160         }
161         // and must be indexed
162         if (field.fieldType().indexOptions() == IndexOptions.NONE) {
163             return false;
164         }
165         return true;
166     }

168     private Fields addGeneratedTermVectors(Engine.GetResult get, Fields termVectorsByField, TermVectorsRequest request, Set<String> selectedFields) throws IOException {
169         /* only keep valid fields */
170         Set<String> validFields = new HashSet<>();
171         for (String field : selectedFields) {
172             FieldMapper fieldMapper = indexShard.mapperService().smartNameFieldMapper(field);
173             if (!isValidField(fieldMapper)) {
174                 continue;
175             }
176             // already retrieved, only if the analyzer hasn't been overridden at the field
177             if (fieldMapper.fieldType().storeTermVectors() &&
178                     (request.perFieldAnalyzer() == null || !request.perFieldAnalyzer().containsKey(field))) {
179                 continue;
180             }
181             validFields.add(field);
182         }
183 
184         if (validFields.isEmpty()) {
185             return termVectorsByField;
186         }
187 
188         /* generate term vectors from fetched document fields */
189         GetResult getResult = indexShard.getService().get(
190                 get, request.id(), request.type(), validFields.toArray(Strings.EMPTY_ARRAY), null, false);
191         Fields generatedTermVectors = generateTermVectors(getResult.getFields().values(), request.offsets(), request.perFieldAnalyzer());
192 
193         /* merge with existing Fields */
194         if (termVectorsByField == null) {
195             return generatedTermVectors;
196         } else {
197             return mergeFields(termVectorsByField, generatedTermVectors);
198         }
199     }

201     private Analyzer getAnalyzerAtField(String field, @Nullable Map<String, String> perFieldAnalyzer) {
202         MapperService mapperService = indexShard.mapperService();
203         Analyzer analyzer;
204         if (perFieldAnalyzer != null && perFieldAnalyzer.containsKey(field)) {
205             analyzer = mapperService.analysisService().analyzer(perFieldAnalyzer.get(field).toString());
206         } else {
207             analyzer = mapperService.smartNameFieldMapper(field).indexAnalyzer();
208         }
209         if (analyzer == null) {
210             analyzer = mapperService.analysisService().defaultIndexAnalyzer();
211         }
212         return analyzer;
213     }

215     private Set<String> getFieldsToGenerate(Map<String, String> perAnalyzerField, Fields fieldsObject) {
216         Set<String> selectedFields = new HashSet<>();
217         for (String fieldName : fieldsObject) {
218             if (perAnalyzerField.containsKey(fieldName)) {
219                 selectedFields.add(fieldName);
220             }
221         }
222         return selectedFields;
223     }

225     private Fields generateTermVectors(Collection<GetField> getFields, boolean withOffsets, @Nullable Map<String, String> perFieldAnalyzer)
226             throws IOException {
227         /* store document in memory index */
228         MemoryIndex index = new MemoryIndex(withOffsets);
229         for (GetField getField : getFields) {
230             String field = getField.getName();
231             Analyzer analyzer = getAnalyzerAtField(field, perFieldAnalyzer);
232             for (Object text : getField.getValues()) {
233                 index.addField(field, text.toString(), analyzer);
234             }
235         }
236         /* and read vectors from it */
237         return MultiFields.getFields(index.createSearcher().getIndexReader());
238     }

240     private Fields generateTermVectorsFromDoc(TermVectorsRequest request, boolean doAllFields) throws IOException {
241         // parse the document, at the moment we do update the mapping, just like percolate
242         ParsedDocument parsedDocument = parseDocument(indexShard.shardId().getIndex(), request.type(), request.doc());
243 
244         // select the right fields and generate term vectors
245         ParseContext.Document doc = parsedDocument.rootDoc();
246         Collection<String> seenFields = new HashSet<>();
247         Collection<GetField> getFields = new HashSet<>();
248         for (IndexableField field : doc.getFields()) {
249             FieldMapper fieldMapper = indexShard.mapperService().smartNameFieldMapper(field.name());
250             if (seenFields.contains(field.name())) {
251                 continue;
252             }
253             else {
254                 seenFields.add(field.name());
255             }
256             if (!isValidField(fieldMapper)) {
257                 continue;
258             }
259             if (request.selectedFields() == null && !doAllFields && !fieldMapper.fieldType().storeTermVectors()) {
260                 continue;
261             }
262             if (request.selectedFields() != null && !request.selectedFields().contains(field.name())) {
263                 continue;
264             }
265             String[] values = doc.getValues(field.name());
266             getFields.add(new GetField(field.name(), Arrays.asList((Object[]) values)));
267         }
268         return generateTermVectors(getFields, request.offsets(), request.perFieldAnalyzer());
269     }

271     private ParsedDocument parseDocument(String index, String type, BytesReference doc) {
272         MapperService mapperService = indexShard.mapperService();
273         IndexService indexService = indexShard.indexService();
274 
275         // TODO: make parsing not dynamically create fields not in the original mapping
276         Tuple<DocumentMapper, Boolean> docMapper = mapperService.documentMapperWithAutoCreate(type);
277         ParsedDocument parsedDocument = docMapper.v1().parse(source(doc).type(type).flyweight(true)).setMappingsModified(docMapper);
278         if (parsedDocument.mappingsModified()) {
279             mappingUpdatedAction.updateMappingOnMaster(index, docMapper.v1(), indexService.indexUUID());
280         }
281         return parsedDocument;
282     }

284     private Fields mergeFields(Fields fields1, Fields fields2) throws IOException {
285         ParallelFields parallelFields = new ParallelFields();
286         for (String fieldName : fields2) {
287             Terms terms = fields2.terms(fieldName);
288             if (terms != null) {
289                 parallelFields.addField(fieldName, terms);
290             }
291         }
292         for (String fieldName : fields1) {
293             if (parallelFields.fields.containsKey(fieldName)) {
294                 continue;
295             }
296             Terms terms = fields1.terms(fieldName);
297             if (terms != null) {
298                 parallelFields.addField(fieldName, terms);
299             }
300         }
301         return parallelFields;
302     }

304     // Poached from Lucene ParallelLeafReader
305     private static final class ParallelFields extends Fields {
306         final Map<String,Terms> fields = new TreeMap<>();
307 
308         ParallelFields() {
309         }
310 
311         void addField(String fieldName, Terms terms) {
312             fields.put(fieldName, terms);
313         }
314 
315         @Override
316         public Iterator<String> iterator() {
317             return Collections.unmodifiableSet(fields.keySet()).iterator();
318         }
319 
320         @Override
321         public Terms terms(String field) {
322             return fields.get(field);
323         }
324 
325         @Override
326         public int size() {
327             return fields.size();
328         }
329     }

331     private boolean useDfs(TermVectorsRequest request) {
332         return request.dfs() && (request.fieldStatistics() || request.termStatistics());
333     }
334 
335     private AggregatedDfs getAggregatedDfs(Fields termVectorsFields, TermVectorsRequest request) throws IOException {
336         DfsOnlyRequest dfsOnlyRequest = new DfsOnlyRequest(termVectorsFields, new String[]{request.index()},
337                 new String[]{request.type()}, request.selectedFields());
338         DfsOnlyResponse response = dfsAction.execute(dfsOnlyRequest).actionGet();
339         return response.getDfs();
340     }
