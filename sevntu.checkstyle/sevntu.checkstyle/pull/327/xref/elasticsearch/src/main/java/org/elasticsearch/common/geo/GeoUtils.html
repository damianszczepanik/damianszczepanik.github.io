41      /** Earth ellipsoid major axis defined by WGS 84 in meters */
42      public static final double EARTH_SEMI_MAJOR_AXIS = 6378137.0;      // meters (WGS 84)
43  
44      /** Earth ellipsoid minor axis defined by WGS 84 in meters */
45      public static final double EARTH_SEMI_MINOR_AXIS = 6356752.314245; // meters (WGS 84)
46  
47      /** Earth mean radius defined by WGS 84 in meters */
48      public static final double EARTH_MEAN_RADIUS = 6371008.7714D;      // meters (WGS 84)
49  
50      /** Earth axis ratio defined by WGS 84 (0.996647189335) */
51      public static final double EARTH_AXIS_RATIO = EARTH_SEMI_MINOR_AXIS / EARTH_SEMI_MAJOR_AXIS;
52  
53      /** Earth ellipsoid equator length in meters */
54      public static final double EARTH_EQUATOR = 2*Math.PI * EARTH_SEMI_MAJOR_AXIS;
55  
56      /** Earth ellipsoid polar distance in meters */
57      public static final double EARTH_POLAR_DISTANCE = Math.PI * EARTH_SEMI_MINOR_AXIS;

63          // SloppyMath impl returns a result in kilometers

67      /**
68       * Calculate the width (in meters) of geohash cells at a specific level 
69       * @param level geohash level must be greater or equal to zero 
70       * @return the width of cells at level in meters  
71       */
72      public static double geoHashCellWidth(int level) {
73          assert level>=0;
74          // Geohash cells are split into 32 cells at each level. the grid
75          // alternates at each level between a 8x4 and a 4x8 grid 
76          return EARTH_EQUATOR / (1L<<((((level+1)/2)*3) + ((level/2)*2)));
77      }
78  
79      /**
80       * Calculate the width (in meters) of quadtree cells at a specific level 
81       * @param level quadtree level must be greater or equal to zero 
82       * @return the width of cells at level in meters  
83       */
84      public static double quadTreeCellWidth(int level) {
85          assert level >=0;
86          return EARTH_EQUATOR / (1L<<level);
87      }

89      /**
90       * Calculate the height (in meters) of geohash cells at a specific level 
91       * @param level geohash level must be greater or equal to zero 
92       * @return the height of cells at level in meters  
93       */
94      public static double geoHashCellHeight(int level) {
95          assert level>=0;
96          // Geohash cells are split into 32 cells at each level. the grid
97          // alternates at each level between a 8x4 and a 4x8 grid 
98          return EARTH_POLAR_DISTANCE / (1L<<((((level+1)/2)*2) + ((level/2)*3)));
99      }

102      * Calculate the height (in meters) of quadtree cells at a specific level 
103      * @param level quadtree level must be greater or equal to zero 
104      * @return the height of cells at level in meters  
105      */
106     public static double quadTreeCellHeight(int level) {
107         assert level>=0;
108         return EARTH_POLAR_DISTANCE / (1L<<level);
109     }

111     /**
112      * Calculate the size (in meters) of geohash cells at a specific level 
113      * @param level geohash level must be greater or equal to zero 
114      * @return the size of cells at level in meters  
115      */
116     public static double geoHashCellSize(int level) {
117         assert level>=0;
118         final double w = geoHashCellWidth(level);
119         final double h = geoHashCellHeight(level);
120         return Math.sqrt(w*w + h*h);
121     }

123     /**
124      * Calculate the size (in meters) of quadtree cells at a specific level 
125      * @param level quadtree level must be greater or equal to zero 
126      * @return the size of cells at level in meters  
127      */
128     public static double quadTreeCellSize(int level) {
129         assert level>=0;
130         return Math.sqrt(EARTH_POLAR_DISTANCE*EARTH_POLAR_DISTANCE + EARTH_EQUATOR*EARTH_EQUATOR) / (1L<<level);
131     }

133     /**
134      * Calculate the number of levels needed for a specific precision. Quadtree
135      * cells will not exceed the specified size (diagonal) of the precision.
136      * @param meters Maximum size of cells in meters (must greater than zero)
137      * @return levels need to achieve precision  
138      */
139     public static int quadTreeLevelsForPrecision(double meters) {
140         assert meters >= 0;
141         if(meters == 0) {
142             return QuadPrefixTree.MAX_LEVELS_POSSIBLE;
143         } else {
144             final double ratio = 1+(EARTH_POLAR_DISTANCE / EARTH_EQUATOR); // cell ratio
145             final double width = Math.sqrt((meters*meters)/(ratio*ratio)); // convert to cell width
146             final long part = Math.round(Math.ceil(EARTH_EQUATOR / width));
147             final int level = Long.SIZE - Long.numberOfLeadingZeros(part)-1; // (log_2)
148             return (part<=(1l<<level)) ?level :(level+1); // adjust level
149         }
150     }

152     /**
153      * Calculate the number of levels needed for a specific precision. QuadTree
154      * cells will not exceed the specified size (diagonal) of the precision.
155      * @param distance Maximum size of cells as unit string (must greater or equal to zero)
156      * @return levels need to achieve precision  
157      */
158     public static int quadTreeLevelsForPrecision(String distance) {
159         return quadTreeLevelsForPrecision(DistanceUnit.METERS.parse(distance, DistanceUnit.DEFAULT));
160     }

162     /**
163      * Calculate the number of levels needed for a specific precision. GeoHash
164      * cells will not exceed the specified size (diagonal) of the precision.
165      * @param meters Maximum size of cells in meters (must greater or equal to zero)
166      * @return levels need to achieve precision  
167      */
168     public static int geoHashLevelsForPrecision(double meters) {
169         assert meters >= 0;
170         
171         if(meters == 0) {
172             return GeohashPrefixTree.getMaxLevelsPossible();
173         } else {
174             final double ratio = 1+(EARTH_POLAR_DISTANCE / EARTH_EQUATOR); // cell ratio
175             final double width = Math.sqrt((meters*meters)/(ratio*ratio)); // convert to cell width
176             final double part = Math.ceil(EARTH_EQUATOR / width);
177             if(part == 1)
178                 return 1;
179             final int bits = (int)Math.round(Math.ceil(Math.log(part) / Math.log(2)));
180             final int full = bits / 5;                // number of 5 bit subdivisions    
181             final int left = bits - full*5;           // bit representing the last level
182             final int even = full + (left>0?1:0);     // number of even levels
183             final int odd = full + (left>3?1:0);      // number of odd levels
184             return even+odd;
185         }
186     }

193      */
194     public static int geoHashLevelsForPrecision(String distance) {

198     /**
199      * Normalize longitude to lie within the -180 (exclusive) to 180 (inclusive) range.
200      *
201      * @param lon Longitude to normalize
202      * @return The normalized longitude.
203      */
204     public static double normalizeLon(double lon) {
205         return centeredModulus(lon, 360);
206     }

208     /**
209      * Normalize latitude to lie within the -90 to 90 (both inclusive) range.
210      * <p/>
211      * Note: You should not normalize longitude and latitude separately,
212      * because when normalizing latitude it may be necessary to
213      * add a shift of 180&deg; in the longitude.
214      * For this purpose, you should call the
215      * {@link #normalizePoint(GeoPoint)} function.
216      *
217      * @param lat Latitude to normalize
218      * @return The normalized latitude.
219      * @see #normalizePoint(GeoPoint)
220      */
221     public static double normalizeLat(double lat) {
222         lat = centeredModulus(lat, 360);
223         if (lat < -90) {
224             lat = -180 - lat;
225         } else if (lat > 90) {
226             lat = 180 - lat;
227         }
228         return lat;
229     }

231     /**
232      * Normalize the geo {@code Point} for its coordinates to lie within their
233      * respective normalized ranges.
234      * <p/>
235      * Note: A shift of 180&deg; is applied in the longitude if necessary,
236      * in order to normalize properly the latitude.
237      *
238      * @param point The point to normalize in-place.
239      */
240     public static void normalizePoint(GeoPoint point) {
241         normalizePoint(point, true, true);
242     }
243 
244     /**
245      * Normalize the geo {@code Point} for the given coordinates to lie within
246      * their respective normalized ranges.
247      * <p/>
248      * You can control which coordinate gets normalized with the two flags.
249      * <p/>
250      * Note: A shift of 180&deg; is applied in the longitude if necessary,
251      * in order to normalize properly the latitude.
252      * If normalizing latitude but not longitude, it is assumed that
253      * the longitude is in the form x+k*360, with x in ]-180;180],
254      * and k is meaningful to the application.
255      * Therefore x will be adjusted while keeping k preserved.
256      *
257      * @param point   The point to normalize in-place.
258      * @param normLat Whether to normalize latitude or leave it as is.
259      * @param normLon Whether to normalize longitude.
260      */
261     public static void normalizePoint(GeoPoint point, boolean normLat, boolean normLon) {
262         double lat = point.lat();
263         double lon = point.lon();
264         
265         normLat = normLat && (lat>90 || lat <= -90);
266         normLon = normLon && (lon>180 || lon <= -180);
267         
268         if (normLat) {
269             lat = centeredModulus(lat, 360);
270             boolean shift = true;
271             if (lat < -90) {
272                 lat = -180 - lat;
273             } else if (lat > 90) {
274                 lat = 180 - lat;
275             } else {
276                 // No need to shift the longitude, and the latitude is normalized
277                 shift = false;
278             }
279             if (shift) {
280                 if (normLon) {
281                     lon += 180;
282                 } else {
283                     // Longitude won't be normalized,
284                     // keep it in the form x+k*360 (with x in ]-180;180])
285                     // by only changing x, assuming k is meaningful for the user application.
286                     lon += normalizeLon(lon) > 0 ? -180 : 180;
287                 }
288             }
289         }
290         if (normLon) {
291             lon = centeredModulus(lon, 360);
292         }
293         point.reset(lat, lon);
294     }
295 
296     private static double centeredModulus(double dividend, double divisor) {
297         double rtn = dividend % divisor;
298         if (rtn <= 0) {
299             rtn += divisor;
300         }
301         if (rtn > divisor / 2) {
302             rtn -= divisor;
303         }
304         return rtn;
305     }

319     /**
320      * Parse a {@link GeoPoint} with a {@link XContentParser}. A geopoint has one of the following forms:
321      * 
322      * <ul>
323      *     <li>Object: <pre>{&quot;lat&quot;: <i>&lt;latitude&gt;</i>, &quot;lon&quot;: <i>&lt;longitude&gt;</i>}</pre></li>
324      *     <li>String: <pre>&quot;<i>&lt;latitude&gt;</i>,<i>&lt;longitude&gt;</i>&quot;</pre></li>
325      *     <li>Geohash: <pre>&quot;<i>&lt;geohash&gt;</i>&quot;</pre></li>
326      *     <li>Array: <pre>[<i>&lt;longitude&gt;</i>,<i>&lt;latitude&gt;</i>]</pre></li>
327      * </ul>
328      * 
329      * @param parser {@link XContentParser} to parse the value from
330      * @param point A {@link GeoPoint} that will be reset by the values parsed
331      * @return new {@link GeoPoint} parsed from the parse
332      * 
333      * @throws IOException
334      * @throws org.elasticsearch.ElasticsearchParseException
335      */
336     public static GeoPoint parseGeoPoint(XContentParser parser, GeoPoint point) throws IOException, ElasticsearchParseException {
337         double lat = Double.NaN;
338         double lon = Double.NaN;
339         String geohash = null;
340         
341         if(parser.currentToken() == Token.START_OBJECT) {
342             while(parser.nextToken() != Token.END_OBJECT) {
343                 if(parser.currentToken() == Token.FIELD_NAME) {
344                     String field = parser.text();
345                     if(LATITUDE.equals(field)) {
346                         parser.nextToken();
347                         switch (parser.currentToken()) {
348                             case VALUE_NUMBER:
349                             case VALUE_STRING:
350                                 lat = parser.doubleValue(true);
351                                 break;
352                             default:
353                                 throw new ElasticsearchParseException("latitude must be a number");
354                         }
355                     } else if (LONGITUDE.equals(field)) {
356                         parser.nextToken();
357                         switch (parser.currentToken()) {
358                             case VALUE_NUMBER:
359                             case VALUE_STRING:
360                                 lon = parser.doubleValue(true);
361                                 break;
362                             default:
363                                 throw new ElasticsearchParseException("longitude must be a number");
364                         }
365                     } else if (GEOHASH.equals(field)) {
366                         if(parser.nextToken() == Token.VALUE_STRING) {
367                             geohash = parser.text();
368                         } else {
369                             throw new ElasticsearchParseException("geohash must be a string");
370                         }
371                     } else {
372                         throw new ElasticsearchParseException("field must be either '" + LATITUDE + "', '" + LONGITUDE + "' or '" + GEOHASH + "'");
373                     }
374                 } else {
375                     throw new ElasticsearchParseException("Token '"+parser.currentToken()+"' not allowed");
376                 }
377             }
378 
379             if (geohash != null) {
380                 if(!Double.isNaN(lat) || !Double.isNaN(lon)) {
381                     throw new ElasticsearchParseException("field must be either lat/lon or geohash");
382                 } else {
383                     return point.resetFromGeoHash(geohash);
384                 }
385             } else if (Double.isNaN(lat)) {
386                 throw new ElasticsearchParseException("field [" + LATITUDE + "] missing");
387             } else if (Double.isNaN(lon)) {
388                 throw new ElasticsearchParseException("field [" + LONGITUDE + "] missing");
389             } else {
390                 return point.reset(lat, lon);
391             }
392             
393         } else if(parser.currentToken() == Token.START_ARRAY) {
394             int element = 0;
395             while(parser.nextToken() != Token.END_ARRAY) {
396                 if(parser.currentToken() == Token.VALUE_NUMBER) {
397                     element++;
398                     if(element == 1) {
399                         lon = parser.doubleValue();
400                     } else if(element == 2) {
401                         lat = parser.doubleValue();
402                     } else {
403                         throw new ElasticsearchParseException("only two values allowed");
404                     }
405                 } else {
406                     throw new ElasticsearchParseException("Numeric value expected");
407                 }
408             }
409             return point.reset(lat, lon);
410         } else if(parser.currentToken() == Token.VALUE_STRING) {
411             String data = parser.text();
412             int comma = data.indexOf(',');
413             if(comma > 0) {
414                 lat = Double.parseDouble(data.substring(0, comma).trim());
415                 lon = Double.parseDouble(data.substring(comma + 1).trim());
416                 return point.reset(lat, lon);
417             } else {
418                 return point.resetFromGeoHash(data);
419             }
420         } else {
421             throw new ElasticsearchParseException("geo_point expected");
422         }
423     }
