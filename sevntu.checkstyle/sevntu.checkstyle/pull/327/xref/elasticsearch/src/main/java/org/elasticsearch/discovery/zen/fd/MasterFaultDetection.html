46  /**
47   * A fault detection that pings the master periodically to see if its alive.
48   */
49  public class MasterFaultDetection extends FaultDetection {
50  
51      public static final String MASTER_PING_ACTION_NAME = "internal:discovery/zen/fd/master_ping";
52  
53      public static interface Listener {
54  
55          /** called when pinging the master failed, like a timeout, transport disconnects etc */
56          void onMasterFailure(DiscoveryNode masterNode, String reason);
57  
58      }
59  
60      private final ClusterService clusterService;
61      private final CopyOnWriteArrayList<Listener> listeners = new CopyOnWriteArrayList<>();
62  
63      private volatile MasterPinger masterPinger;
64  
65      private final Object masterNodeMutex = new Object();
66  
67      private volatile DiscoveryNode masterNode;
68  
69      private volatile int retryCount;
70  
71      private final AtomicBoolean notifiedMasterFailure = new AtomicBoolean();
72  
73      public MasterFaultDetection(Settings settings, ThreadPool threadPool, TransportService transportService,
74                                  ClusterName clusterName, ClusterService clusterService) {
75          super(settings, threadPool, transportService, clusterName);
76          this.clusterService = clusterService;
77  
78          logger.debug("[master] uses ping_interval [{}], ping_timeout [{}], ping_retries [{}]", pingInterval, pingRetryTimeout, pingRetryCount);
79  
80          transportService.registerHandler(MASTER_PING_ACTION_NAME, new MasterPingRequestHandler());
81      }
82  
83      public DiscoveryNode masterNode() {
84          return this.masterNode;
85      }
86  
87      public void addListener(Listener listener) {
88          listeners.add(listener);
89      }
90  
91      public void removeListener(Listener listener) {
92          listeners.remove(listener);
93      }
94  
95      public void restart(DiscoveryNode masterNode, String reason) {
96          synchronized (masterNodeMutex) {
97              if (logger.isDebugEnabled()) {
98                  logger.debug("[master] restarting fault detection against master [{}], reason [{}]", masterNode, reason);
99              }
100             innerStop();
101             innerStart(masterNode);
102         }
103     }
104 
105     public void start(final DiscoveryNode masterNode, String reason) {
106         synchronized (masterNodeMutex) {
107             if (logger.isDebugEnabled()) {
108                 logger.debug("[master] starting fault detection against master [{}], reason [{}]", masterNode, reason);
109             }
110             innerStart(masterNode);
111         }
112     }
113 
114     private void innerStart(final DiscoveryNode masterNode) {
115         this.masterNode = masterNode;
116         this.retryCount = 0;
117         this.notifiedMasterFailure.set(false);
118 
119         // try and connect to make sure we are connected
120         try {
121             transportService.connectToNode(masterNode);
122         } catch (final Exception e) {
123             // notify master failure (which stops also) and bail..
124             notifyMasterFailure(masterNode, "failed to perform initial connect [" + e.getMessage() + "]");
125             return;
126         }
127         if (masterPinger != null) {
128             masterPinger.stop();
129         }
130         this.masterPinger = new MasterPinger();
131 
132         // we start pinging slightly later to allow the chosen master to complete it's own master election
133         threadPool.schedule(pingInterval, ThreadPool.Names.SAME, masterPinger);
134     }
135 
136     public void stop(String reason) {
137         synchronized (masterNodeMutex) {
138             if (masterNode != null) {
139                 if (logger.isDebugEnabled()) {
140                     logger.debug("[master] stopping fault detection against master [{}], reason [{}]", masterNode, reason);
141                 }
142             }
143             innerStop();
144         }
145     }
146 
147     private void innerStop() {
148         // also will stop the next ping schedule
149         this.retryCount = 0;
150         if (masterPinger != null) {
151             masterPinger.stop();
152             masterPinger = null;
153         }
154         this.masterNode = null;
155     }
156 
157     @Override
158     public void close() {
159         super.close();
160         stop("closing");
161         this.listeners.clear();
162         transportService.removeHandler(MASTER_PING_ACTION_NAME);
163     }
164 
165     @Override
166     protected void handleTransportDisconnect(DiscoveryNode node) {
167         synchronized (masterNodeMutex) {
168             if (!node.equals(this.masterNode)) {
169                 return;
170             }
171             if (connectOnNetworkDisconnect) {
172                 try {
173                     transportService.connectToNode(node);
174                     // if all is well, make sure we restart the pinger
175                     if (masterPinger != null) {
176                         masterPinger.stop();
177                     }
178                     this.masterPinger = new MasterPinger();
179                     // we use schedule with a 0 time value to run the pinger on the pool as it will run on later
180                     threadPool.schedule(TimeValue.timeValueMillis(0), ThreadPool.Names.SAME, masterPinger);
181                 } catch (Exception e) {
182                     logger.trace("[master] [{}] transport disconnected (with verified connect)", masterNode);
183                     notifyMasterFailure(masterNode, "transport disconnected (with verified connect)");
184                 }
185             } else {
186                 logger.trace("[master] [{}] transport disconnected", node);
187                 notifyMasterFailure(node, "transport disconnected");
188             }
189         }
190     }
191 
192     private void notifyMasterFailure(final DiscoveryNode masterNode, final String reason) {
193         if (notifiedMasterFailure.compareAndSet(false, true)) {
194             threadPool.generic().execute(new Runnable() {
195                 @Override
196                 public void run() {
197                     for (Listener listener : listeners) {
198                         listener.onMasterFailure(masterNode, reason);
199                     }
200                 }
201             });
202             stop("master failure, " + reason);
203         }
204     }
205 
206     private class MasterPinger implements Runnable {
207 
208         private volatile boolean running = true;
209 
210         public void stop() {
211             this.running = false;
212         }
213 
214         @Override
215         public void run() {
216             if (!running) {
217                 // return and don't spawn...
218                 return;
219             }
220             final DiscoveryNode masterToPing = masterNode;
221             if (masterToPing == null) {
222                 // master is null, should not happen, but we are still running, so reschedule
223                 threadPool.schedule(pingInterval, ThreadPool.Names.SAME, MasterPinger.this);
224                 return;
225             }
226             final MasterPingRequest request = new MasterPingRequest(clusterService.localNode().id(), masterToPing.id(), clusterName);
227             final TransportRequestOptions options = options().withType(TransportRequestOptions.Type.PING).withTimeout(pingRetryTimeout);
228             transportService.sendRequest(masterToPing, MASTER_PING_ACTION_NAME, request, options, new BaseTransportResponseHandler<MasterPingResponseResponse>() {
229 
230                         @Override
231                         public MasterPingResponseResponse newInstance() {
232                             return new MasterPingResponseResponse();
233                         }
234 
235                         @Override
236                         public void handleResponse(MasterPingResponseResponse response) {
237                             if (!running) {
238                                 return;
239                             }
240                             // reset the counter, we got a good result
241                             MasterFaultDetection.this.retryCount = 0;
242                             // check if the master node did not get switched on us..., if it did, we simply return with no reschedule
243                             if (masterToPing.equals(MasterFaultDetection.this.masterNode())) {
244                                 // we don't stop on disconnection from master, we keep pinging it
245                                 threadPool.schedule(pingInterval, ThreadPool.Names.SAME, MasterPinger.this);
246                             }
247                         }
248 
249                         @Override
250                         public void handleException(TransportException exp) {
251                             if (!running) {
252                                 return;
253                             }
254                             synchronized (masterNodeMutex) {
255                                 // check if the master node did not get switched on us...
256                                 if (masterToPing.equals(MasterFaultDetection.this.masterNode())) {
257                                     if (exp instanceof ConnectTransportException || exp.getCause() instanceof ConnectTransportException) {
258                                         handleTransportDisconnect(masterToPing);
259                                         return;
260                                     } else if (exp.getCause() instanceof NotMasterException) {
261                                         logger.debug("[master] pinging a master {} that is no longer a master", masterNode);
262                                         notifyMasterFailure(masterToPing, "no longer master");
263                                         return;
264                                     } else if (exp.getCause() instanceof ThisIsNotTheMasterYouAreLookingForException) {
265                                         logger.debug("[master] pinging a master {} that is not the master", masterNode);
266                                         notifyMasterFailure(masterToPing, "not master");
267                                         return;
268                                     } else if (exp.getCause() instanceof NodeDoesNotExistOnMasterException) {
269                                         logger.debug("[master] pinging a master {} but we do not exists on it, act as if its master failure", masterNode);
270                                         notifyMasterFailure(masterToPing, "do not exists on master, act as master failure");
271                                         return;
272                                     }
273 
274                                     int retryCount = ++MasterFaultDetection.this.retryCount;
275                                     logger.trace("[master] failed to ping [{}], retry [{}] out of [{}]", exp, masterNode, retryCount, pingRetryCount);
276                                     if (retryCount >= pingRetryCount) {
277                                         logger.debug("[master] failed to ping [{}], tried [{}] times, each with maximum [{}] timeout", masterNode, pingRetryCount, pingRetryTimeout);
278                                         // not good, failure
279                                         notifyMasterFailure(masterToPing, "failed to ping, tried [" + pingRetryCount + "] times, each with  maximum [" + pingRetryTimeout + "] timeout");
280                                     } else {
281                                         // resend the request, not reschedule, rely on send timeout
282                                         transportService.sendRequest(masterToPing, MASTER_PING_ACTION_NAME, request, options, this);
283                                     }
284                                 }
285                             }
286                         }
287 
288                         @Override
289                         public String executor() {
290                             return ThreadPool.Names.SAME;
291                         }
292                     }
293             );
294         }
295     }

297     /** Thrown when a ping reaches the wrong node */
298     static class ThisIsNotTheMasterYouAreLookingForException extends ElasticsearchIllegalStateException {
299 
300         ThisIsNotTheMasterYouAreLookingForException(String msg) {
301             super(msg);
302         }
303 
304         ThisIsNotTheMasterYouAreLookingForException() {
305         }
306 
307         @Override
308         public Throwable fillInStackTrace() {
309             return null;
310         }
311     }

313     static class NodeDoesNotExistOnMasterException extends ElasticsearchIllegalStateException {
314         @Override
315         public Throwable fillInStackTrace() {
316             return null;
317         }

320     private class MasterPingRequestHandler extends BaseTransportRequestHandler<MasterPingRequest> {
321 
322         @Override
323         public MasterPingRequest newInstance() {
324             return new MasterPingRequest();
325         }
326 
327         @Override
328         public void messageReceived(final MasterPingRequest request, final TransportChannel channel) throws Exception {
329             final DiscoveryNodes nodes = clusterService.state().nodes();
330             // check if we are really the same master as the one we seemed to be think we are
331             // this can happen if the master got "kill -9" and then another node started using the same port
332             if (!request.masterNodeId.equals(nodes.localNodeId())) {
333                 throw new ThisIsNotTheMasterYouAreLookingForException();
334             }
335 
336             // ping from nodes of version < 1.4.0 will have the clustername set to null
337             if (request.clusterName != null && !request.clusterName.equals(clusterName)) {
338                 logger.trace("master fault detection ping request is targeted for a different [{}] cluster then us [{}]", request.clusterName, clusterName);
339                 throw new ThisIsNotTheMasterYouAreLookingForException("master fault detection ping request is targeted for a different [" + request.clusterName + "] cluster then us [" + clusterName + "]");
340             }
341 
342             // when we are elected as master or when a node joins, we use a cluster state update thread
343             // to incorporate that information in the cluster state. That cluster state is published
344             // before we make it available locally. This means that a master ping can come from a node
345             // that has already processed the new CS but it is not known locally.
346             // Therefore, if we fail we have to check again under a cluster state thread to make sure
347             // all processing is finished.
348             //
349 
350             if (!nodes.localNodeMaster() || !nodes.nodeExists(request.nodeId)) {
351                 logger.trace("checking ping from [{}] under a cluster state thread", request.nodeId);
352                 clusterService.submitStateUpdateTask("master ping (from: [" + request.nodeId + "])", new ProcessedClusterStateNonMasterUpdateTask() {
353 
354                     @Override
355                     public ClusterState execute(ClusterState currentState) throws Exception {
356                         // if we are no longer master, fail...
357                         DiscoveryNodes nodes = currentState.nodes();
358                         if (!nodes.localNodeMaster()) {
359                             throw new NotMasterException();
360                         }
361                         if (!nodes.nodeExists(request.nodeId)) {
362                             throw new NodeDoesNotExistOnMasterException();
363                         }
364                         return currentState;
365                     }
366 
367                     @Override
368                     public void onFailure(String source, @Nullable Throwable t) {
369                         if (t == null) {
370                             t = new ElasticsearchException("unknown error while processing ping");
371                         }
372                         try {
373                             channel.sendResponse(t);
374                         } catch (IOException e) {
375                             logger.warn("error while sending ping response", e);
376                         }
377                     }
378 
379                     @Override
380                     public void clusterStateProcessed(String source, ClusterState oldState, ClusterState newState) {
381                         try {
382                             channel.sendResponse(new MasterPingResponseResponse());
383                         } catch (IOException e) {
384                             logger.warn("error while sending ping response", e);
385                         }
386                     }
387                 });
388             } else {
389                 // send a response, and note if we are connected to the master or not
390                 channel.sendResponse(new MasterPingResponseResponse());
391             }
392         }
393 
394         @Override
395         public String executor() {
396             return ThreadPool.Names.SAME;
397         }
398     }

401     private static class MasterPingRequest extends TransportRequest {
402 
403         private String nodeId;
404 
405         private String masterNodeId;
406         private ClusterName clusterName;
407 
408         private MasterPingRequest() {
409         }
410 
411         private MasterPingRequest(String nodeId, String masterNodeId, ClusterName clusterName) {
412             this.nodeId = nodeId;
413             this.masterNodeId = masterNodeId;
414             this.clusterName = clusterName;
415         }
416 
417         @Override
418         public void readFrom(StreamInput in) throws IOException {
419             super.readFrom(in);
420             nodeId = in.readString();
421             masterNodeId = in.readString();
422             clusterName = ClusterName.readClusterName(in);
423         }
424 
425         @Override
426         public void writeTo(StreamOutput out) throws IOException {
427             super.writeTo(out);
428             out.writeString(nodeId);
429             out.writeString(masterNodeId);
430             clusterName.writeTo(out);
431         }
432     }
433 
434     private static class MasterPingResponseResponse extends TransportResponse {
435 
436         private MasterPingResponseResponse() {
437         }
438 
439         @Override
440         public void readFrom(StreamInput in) throws IOException {
441             super.readFrom(in);
442             if (in.getVersion().onOrBefore(Version.V_1_4_0_Beta1)) {
443                 // old listedOnMaster
444                 in.readBoolean();
445             }
446         }
447 
448         @Override
449         public void writeTo(StreamOutput out) throws IOException {
450             super.writeTo(out);
451             if (out.getVersion().onOrBefore(Version.V_1_4_0_Beta1)) {
452                 // old listedOnMaster
453                 out.writeBoolean(true);
454             }
455         }
456     }
