20  package org.elasticsearch.action.bench;
21  
22  import org.elasticsearch.ElasticsearchIllegalArgumentException;
23  import org.elasticsearch.action.ActionResponse;
24  import org.elasticsearch.common.Strings;
25  import org.elasticsearch.common.io.stream.StreamInput;
26  import org.elasticsearch.common.io.stream.StreamOutput;
27  import org.elasticsearch.common.io.stream.Streamable;
28  import org.elasticsearch.common.xcontent.ToXContent;
29  import org.elasticsearch.common.xcontent.XContentBuilder;
30  import org.elasticsearch.common.xcontent.XContentBuilderString;
31  import org.elasticsearch.common.xcontent.XContentFactory;
32  
33  import java.io.IOException;
34  import java.util.*;
35  
36  /**
37   * Benchmark response.
38   *
39   * A benchmark response will contain a mapping of names to results for each competition.
40   */
41  public class BenchmarkResponse extends ActionResponse implements Streamable, ToXContent {
42  
43      private String benchmarkName;
44      private State state = State.RUNNING;
45      private boolean verbose;
46      private String[] errors = Strings.EMPTY_ARRAY;
47  
48      Map<String, CompetitionResult> competitionResults;
49  
50      public BenchmarkResponse() {
51          competitionResults = new HashMap<>();
52      }
53  
54      public BenchmarkResponse(String benchmarkName, Map<String, CompetitionResult> competitionResults) {
55          this.benchmarkName = benchmarkName;
56          this.competitionResults = competitionResults;
57      }
58  
59      /**
60       * Benchmarks can be in one of:
61       *  RUNNING     - executing normally
62       *  COMPLETE    - completed normally
63       *  ABORTED     - aborted
64       *  FAILED      - execution failed
65       */
66      public static enum State {
67          RUNNING((byte) 0),
68          COMPLETE((byte) 1),
69          ABORTED((byte) 2),
70          FAILED((byte) 3);
71  
72          private final byte id;
73          private static final State[] STATES = new State[State.values().length];
74  
75          static {
76              for (State state : State.values()) {
77                  assert state.id() < STATES.length && state.id() >= 0;
78                  STATES[state.id] = state;
79              }
80          }
81  
82          State(byte id) {
83              this.id = id;
84          }
85  
86          public byte id() {
87              return id;
88          }
89  
90          public static State fromId(byte id) throws ElasticsearchIllegalArgumentException {
91              if (id < 0 || id >= STATES.length) {
92                  throw new ElasticsearchIllegalArgumentException("No mapping for id [" + id + "]");
93              }
94              return STATES[id];
95          }
96      }
97  
98      /**
99       * Name of the benchmark
100      * @return  Name of the benchmark
101      */
102     public String benchmarkName() {
103         return benchmarkName;
104     }
105 
106     /**
107      * Sets the benchmark name
108      * @param benchmarkName Benchmark name
109      */
110     public void benchmarkName(String benchmarkName) {
111         this.benchmarkName = benchmarkName;
112     }
113 
114     /**
115      * Benchmark state
116      * @return  Benchmark state
117      */
118     public State state() {
119         return state;
120     }
121 
122     /**
123      * Sets the state of the benchmark
124      * @param state State
125      */
126     public void state(State state) {
127         this.state = state;
128     }
129 
130     /**
131      * Possibly replace the existing state with the new state depending on the severity
132      * of the new state. More severe states, such as FAILED, will over-write less severe
133      * ones, such as COMPLETED.
134      * @param newState  New candidate state
135      * @return          The merged state
136      */
137     public State mergeState(State newState) {
138         if (state.compareTo(newState) < 0) {
139             state = newState;
140         }
141         return state;
142     }
143 
144     /**
145      * Map of competition names to competition results
146      * @return  Map of competition names to competition results
147      */
148     public Map<String, CompetitionResult> competitionResults() {
149         return competitionResults;
150     }
151 
152     /**
153      * Whether to report verbose statistics
154      */
155     public boolean verbose() {
156         return verbose;
157     }
158 
159     /**
160      * Sets whether to report verbose statistics
161      */
162     public void verbose(boolean verbose) {
163         this.verbose = verbose;
164     }
165 
166     /**
167      * Whether the benchmark encountered error conditions
168      * @return  Whether the benchmark encountered error conditions
169      */
170     public boolean hasErrors() {
171         return (errors != null && errors.length > 0);
172     }
173 
174     /**
175      * Error messages
176      * @return  Error messages
177      */
178     public String[] errors() {
179         return this.errors;
180     }
181 
182     /**
183      * Sets error messages
184      * @param errors    Error messages
185      */
186     public void errors(String... errors) {
187         this.errors = (errors == null) ? Strings.EMPTY_ARRAY : errors;
188     }
189 
190     @Override
191     public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
192         builder.field(Fields.STATUS, state.toString());
193         if (errors != null) {
194             builder.array(Fields.ERRORS, errors);
195         }
196         builder.startObject(Fields.COMPETITORS);
197         if (competitionResults != null) {
198             for (Map.Entry<String, CompetitionResult> entry : competitionResults.entrySet()) {
199                 entry.getValue().verbose(verbose);
200                 entry.getValue().toXContent(builder, params);
201             }
202         }
203         builder.endObject();
204         return builder;
205     }
206 
207     @Override
208     public void readFrom(StreamInput in) throws IOException {
209         super.readFrom(in);
210         benchmarkName = in.readString();
211         state = State.fromId(in.readByte());
212         errors = in.readStringArray();
213         int size = in.readVInt();
214         for (int i = 0; i < size; i++) {
215             String s = in.readString();
216             CompetitionResult cr = new CompetitionResult();
217             cr.readFrom(in);
218             competitionResults.put(s, cr);
219         }
220     }
221 
222     @Override
223     public void writeTo(StreamOutput out) throws IOException {
224         super.writeTo(out);
225         out.writeString(benchmarkName);
226         out.writeByte(state.id());
227         out.writeStringArray(errors);
228         out.write(competitionResults.size());
229         for (Map.Entry<String, CompetitionResult> entry : competitionResults.entrySet()) {
230             out.writeString(entry.getKey());
231             entry.getValue().writeTo(out);
232         }
233     }
234 
235     @Override
236     public String toString() {
237         try {
238             XContentBuilder builder = XContentFactory.jsonBuilder().prettyPrint();
239             builder.startObject();
240             toXContent(builder, EMPTY_PARAMS);
241             builder.endObject();
242             return builder.string();
243         } catch (IOException e) {
244             return "{ \"error\" : \"" + e.getMessage() + "\"}";
245         }
246     }
247 
248     static final class Fields {
249         static final XContentBuilderString STATUS = new XContentBuilderString("status");
250         static final XContentBuilderString ERRORS = new XContentBuilderString("errors");
251         static final XContentBuilderString COMPETITORS = new XContentBuilderString("competitors");
252     }
253 }
