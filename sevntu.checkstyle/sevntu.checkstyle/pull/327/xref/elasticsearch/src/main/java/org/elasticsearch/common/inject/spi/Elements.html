37  /**
38   * Exposes elements of a module so they can be inspected, validated or {@link
39   * Element#applyTo(Binder) rewritten}.
40   *
41   * @author jessewilson@google.com (Jesse Wilson)
42   * @since 2.0
43   */
44  public final class Elements {
45      private static final BindingTargetVisitor<Object, Object> GET_INSTANCE_VISITOR
46              = new DefaultBindingTargetVisitor<Object, Object>() {
47          @Override
48          public Object visit(InstanceBinding<?> binding) {
49              return binding.getInstance();
50          }
51  
52          @Override
53          protected Object visitOther(Binding<?> binding) {
54              throw new IllegalArgumentException();
55          }
56      };

58      /**
59       * Records the elements executed by {@code modules}.
60       */
61      public static List<Element> getElements(Module... modules) {
62          return getElements(Stage.DEVELOPMENT, Arrays.asList(modules));
63      }
64  
65      /**
66       * Records the elements executed by {@code modules}.
67       */
68      public static List<Element> getElements(Stage stage, Module... modules) {
69          return getElements(stage, Arrays.asList(modules));
70      }
71  
72      /**
73       * Records the elements executed by {@code modules}.
74       */
75      public static List<Element> getElements(Iterable<? extends Module> modules) {
76          return getElements(Stage.DEVELOPMENT, modules);
77      }
78  
79      /**
80       * Records the elements executed by {@code modules}.
81       */
82      public static List<Element> getElements(Stage stage, Iterable<? extends Module> modules) {
83          RecordingBinder binder = new RecordingBinder(stage);
84          for (Module module : modules) {
85              binder.install(module);
86          }
87          return Collections.unmodifiableList(binder.elements);
88      }
89  
90      /**
91       * Returns the module composed of {@code elements}.
92       */
93      public static Module getModule(final Iterable<? extends Element> elements) {
94          return new Module() {
95              @Override
96              public void configure(Binder binder) {
97                  for (Element element : elements) {
98                      element.applyTo(binder);
99                  }
100             }
101         };
102     }

104     @SuppressWarnings("unchecked")
105     static <T> BindingTargetVisitor<T, T> getInstanceVisitor() {
106         return (BindingTargetVisitor<T, T>) GET_INSTANCE_VISITOR;
107     }

109     private static class RecordingBinder implements Binder, PrivateBinder {
110         private final Stage stage;
111         private final Set<Module> modules;
112         private final List<Element> elements;
113         private final Object source;
114         private final SourceProvider sourceProvider;
115 
116         /**
117          * The binder where exposed bindings will be created
118          */
119         private final RecordingBinder parent;
120         private final PrivateElementsImpl privateElements;
121 
122         private RecordingBinder(Stage stage) {
123             this.stage = stage;
124             this.modules = Sets.newHashSet();
125             this.elements = Lists.newArrayList();
126             this.source = null;
127             this.sourceProvider = new SourceProvider().plusSkippedClasses(
128                     Elements.class, RecordingBinder.class, AbstractModule.class,
129                     ConstantBindingBuilderImpl.class, AbstractBindingBuilder.class, BindingBuilder.class);
130             this.parent = null;
131             this.privateElements = null;
132         }
133 
134         /**
135          * Creates a recording binder that's backed by {@code prototype}.
136          */
137         private RecordingBinder(
138                 RecordingBinder prototype, Object source, SourceProvider sourceProvider) {
139             checkArgument(source == null ^ sourceProvider == null);
140 
141             this.stage = prototype.stage;
142             this.modules = prototype.modules;
143             this.elements = prototype.elements;
144             this.source = source;
145             this.sourceProvider = sourceProvider;
146             this.parent = prototype.parent;
147             this.privateElements = prototype.privateElements;
148         }
149 
150         /**
151          * Creates a private recording binder.
152          */
153         private RecordingBinder(RecordingBinder parent, PrivateElementsImpl privateElements) {
154             this.stage = parent.stage;
155             this.modules = Sets.newHashSet();
156             this.elements = privateElements.getElementsMutable();
157             this.source = parent.source;
158             this.sourceProvider = parent.sourceProvider;
159             this.parent = parent;
160             this.privateElements = privateElements;
161         }
162 
163         @Override
164         public void bindScope(Class<? extends Annotation> annotationType, Scope scope) {
165             elements.add(new ScopeBinding(getSource(), annotationType, scope));
166         }
167 
168         @Override
169         @SuppressWarnings("unchecked") // it is safe to use the type literal for the raw type
170         public void requestInjection(Object instance) {
171             requestInjection((TypeLiteral) TypeLiteral.get(instance.getClass()), instance);
172         }
173 
174         @Override
175         public <T> void requestInjection(TypeLiteral<T> type, T instance) {
176             elements.add(new InjectionRequest<>(getSource(), type, instance));
177         }
178 
179         @Override
180         public <T> MembersInjector<T> getMembersInjector(final TypeLiteral<T> typeLiteral) {
181             final MembersInjectorLookup<T> element
182                     = new MembersInjectorLookup<>(getSource(), typeLiteral);
183             elements.add(element);
184             return element.getMembersInjector();
185         }
186 
187         @Override
188         public <T> MembersInjector<T> getMembersInjector(Class<T> type) {
189             return getMembersInjector(TypeLiteral.get(type));
190         }
191 
192         @Override
193         public void bindListener(Matcher<? super TypeLiteral<?>> typeMatcher, TypeListener listener) {
194             elements.add(new TypeListenerBinding(getSource(), listener, typeMatcher));
195         }
196 
197         @Override
198         public void requestStaticInjection(Class<?>... types) {
199             for (Class<?> type : types) {
200                 elements.add(new StaticInjectionRequest(getSource(), type));
201             }
202         }
203 
204         @Override
205         public void install(Module module) {
206             if (modules.add(module)) {
207                 Binder binder = this;
208                 if (module instanceof PrivateModule) {
209                     binder = binder.newPrivateBinder();
210                 }
211 
212                 try {
213                     module.configure(binder);
214                 } catch (RuntimeException e) {
215                     Collection<Message> messages = Errors.getMessagesFromThrowable(e);
216                     if (!messages.isEmpty()) {
217                         elements.addAll(messages);
218                     } else {
219                         addError(e);
220                     }
221                 }
222                 binder.install(ProviderMethodsModule.forModule(module));
223             }
224         }
225 
226         @Override
227         public Stage currentStage() {
228             return stage;
229         }
230 
231         @Override
232         public void addError(String message, Object... arguments) {
233             elements.add(new Message(getSource(), Errors.format(message, arguments)));
234         }
235 
236         @Override
237         public void addError(Throwable t) {
238             String message = "An exception was caught and reported. Message: " + t.getMessage();
239             elements.add(new Message(ImmutableList.of(getSource()), message, t));
240         }
241 
242         @Override
243         public void addError(Message message) {
244             elements.add(message);
245         }
246 
247         @Override
248         public <T> AnnotatedBindingBuilder<T> bind(Key<T> key) {
249             return new BindingBuilder<>(this, elements, getSource(), key);
250         }
251 
252         @Override
253         public <T> AnnotatedBindingBuilder<T> bind(TypeLiteral<T> typeLiteral) {
254             return bind(Key.get(typeLiteral));
255         }
256 
257         @Override
258         public <T> AnnotatedBindingBuilder<T> bind(Class<T> type) {
259             return bind(Key.get(type));
260         }
261 
262         @Override
263         public AnnotatedConstantBindingBuilder bindConstant() {
264             return new ConstantBindingBuilderImpl<Void>(this, elements, getSource());
265         }
266 
267         @Override
268         public <T> Provider<T> getProvider(final Key<T> key) {
269             final ProviderLookup<T> element = new ProviderLookup<>(getSource(), key);
270             elements.add(element);
271             return element.getProvider();
272         }
273 
274         @Override
275         public <T> Provider<T> getProvider(Class<T> type) {
276             return getProvider(Key.get(type));
277         }
278 
279         @Override
280         public void convertToTypes(Matcher<? super TypeLiteral<?>> typeMatcher,
281                                    TypeConverter converter) {
282             elements.add(new TypeConverterBinding(getSource(), typeMatcher, converter));
283         }
284 
285         @Override
286         public RecordingBinder withSource(final Object source) {
287             return new RecordingBinder(this, source, null);
288         }
289 
290         @Override
291         public RecordingBinder skipSources(Class... classesToSkip) {
292             // if a source is specified explicitly, we don't need to skip sources
293             if (source != null) {
294                 return this;
295             }
296 
297             SourceProvider newSourceProvider = sourceProvider.plusSkippedClasses(classesToSkip);
298             return new RecordingBinder(this, null, newSourceProvider);
299         }
300 
301         @Override
302         public PrivateBinder newPrivateBinder() {
303             PrivateElementsImpl privateElements = new PrivateElementsImpl(getSource());
304             elements.add(privateElements);
305             return new RecordingBinder(this, privateElements);
306         }
307 
308         @Override
309         public void expose(Key<?> key) {
310             exposeInternal(key);
311         }
312 
313         @Override
314         public AnnotatedElementBuilder expose(Class<?> type) {
315             return exposeInternal(Key.get(type));
316         }
317 
318         @Override
319         public AnnotatedElementBuilder expose(TypeLiteral<?> type) {
320             return exposeInternal(Key.get(type));
321         }
322 
323         private <T> AnnotatedElementBuilder exposeInternal(Key<T> key) {
324             if (privateElements == null) {
325                 addError("Cannot expose %s on a standard binder. "
326                         + "Exposed bindings are only applicable to private binders.", key);
327                 return new AnnotatedElementBuilder() {
328                     @Override
329                     public void annotatedWith(Class<? extends Annotation> annotationType) {
330                     }
331 
332                     @Override
333                     public void annotatedWith(Annotation annotation) {
334                     }
335                 };
336             }
337 
338             ExposureBuilder<T> builder = new ExposureBuilder<>(this, getSource(), key);
339             privateElements.addExposureBuilder(builder);
340             return builder;
341         }
342 
343         private static ESLogger logger = Loggers.getLogger(Bootstrap.class);
344 
345         protected Object getSource() {
346             Object ret;
347             if (logger.isDebugEnabled()) {
348                 ret = sourceProvider != null
349                         ? sourceProvider.get()
350                         : source;
351             } else {
352                 ret = source;
353             }
354             return ret == null ? "_unknown_" : ret;
355         }
356 
357         @Override
358         public String toString() {
359             return "Binder";
360         }
361     }
