38  /**
39   * {@link IndexShardRoutingTable} encapsulates all instances of a single shard.
40   * Each Elasticsearch index consists of multiple shards, each shard encapsulates
41   * a disjoint set of the index data and each shard has one or more instances
42   * referred to as replicas of a shard. Given that, this class encapsulates all
43   * replicas (instances) for a single index shard.
44   */
45  public class IndexShardRoutingTable implements Iterable<ShardRouting> {
46  
47      final ShardShuffler shuffler;
48      final ShardId shardId;
49  
50      final ShardRouting primary;
51      final ImmutableList<ShardRouting> primaryAsList;
52      final ImmutableList<ShardRouting> replicas;
53      final ImmutableList<ShardRouting> shards;
54      final ImmutableList<ShardRouting> activeShards;
55      final ImmutableList<ShardRouting> assignedShards;
56      final boolean allShardsStarted;
57  
58      /**
59       * The initializing list, including ones that are initializing on a target node because of relocation.
60       * If we can come up with a better variable name, it would be nice...
61       */
62      final ImmutableList<ShardRouting> allInitializingShards;
63  
64      final boolean primaryAllocatedPostApi;
65  
66      IndexShardRoutingTable(ShardId shardId, ImmutableList<ShardRouting> shards, boolean primaryAllocatedPostApi) {
67          this.shardId = shardId;
68          this.shuffler = new RotationShardShuffler(ThreadLocalRandom.current().nextInt());
69          this.shards = shards;
70          this.primaryAllocatedPostApi = primaryAllocatedPostApi;
71  
72          ShardRouting primary = null;
73          ImmutableList.Builder<ShardRouting> replicas = ImmutableList.builder();
74          ImmutableList.Builder<ShardRouting> activeShards = ImmutableList.builder();
75          ImmutableList.Builder<ShardRouting> assignedShards = ImmutableList.builder();
76          ImmutableList.Builder<ShardRouting> allInitializingShards = ImmutableList.builder();
77          boolean allShardsStarted = true;
78          for (ShardRouting shard : shards) {
79              if (shard.primary()) {
80                  primary = shard;
81              } else {
82                  replicas.add(shard);
83              }
84              if (shard.active()) {
85                  activeShards.add(shard);
86              }
87              if (shard.initializing()) {
88                  allInitializingShards.add(shard);
89              }
90              if (shard.relocating()) {
91                  // create the target initializing shard routing on the node the shard is relocating to
92                  allInitializingShards.add(shard.targetRoutingIfRelocating());
93              }
94              if (shard.assignedToNode()) {
95                  assignedShards.add(shard);
96              }
97              if (shard.state() != ShardRoutingState.STARTED) {
98                  allShardsStarted = false;
99              }
100         }
101         this.allShardsStarted = allShardsStarted;
102 
103         this.primary = primary;
104         if (primary != null) {
105             this.primaryAsList = ImmutableList.of(primary);
106         } else {
107             this.primaryAsList = ImmutableList.of();
108         }
109         this.replicas = replicas.build();
110         this.activeShards = activeShards.build();
111         this.assignedShards = assignedShards.build();
112         this.allInitializingShards = allInitializingShards.build();
113     }

115     /**
116      * Normalizes all shard routings to the same version.
117      */
118     public IndexShardRoutingTable normalizeVersions() {
119         if (shards.isEmpty()) {
120             return this;
121         }
122         if (shards.size() == 1) {
123             return this;
124         }
125         long highestVersion = shards.get(0).version();
126         boolean requiresNormalization = false;
127         for (int i = 1; i < shards.size(); i++) {
128             if (shards.get(i).version() != highestVersion) {
129                 requiresNormalization = true;
130             }
131             if (shards.get(i).version() > highestVersion) {
132                 highestVersion = shards.get(i).version();
133             }
134         }
135         if (!requiresNormalization) {
136             return this;
137         }
138         List<ShardRouting> shardRoutings = new ArrayList<>(shards.size());
139         for (int i = 0; i < shards.size(); i++) {
140             if (shards.get(i).version() == highestVersion) {
141                 shardRoutings.add(shards.get(i));
142             } else {
143                 shardRoutings.add(new ImmutableShardRouting(shards.get(i), highestVersion));
144             }
145         }
146         return new IndexShardRoutingTable(shardId, ImmutableList.copyOf(shardRoutings), primaryAllocatedPostApi);
147     }

273         if (allInitializingShards.isEmpty()) {
274             return new PlainShardIterator(shardId, shuffler.shuffle(activeShards, seed));
275         }
276         ArrayList<ShardRouting> ordered = new ArrayList<>(activeShards.size() + allInitializingShards.size());

278         ordered.addAll(allInitializingShards);
279         return new PlainShardIterator(shardId, ordered);

297     public ShardIterator primaryFirstActiveInitializingShardsIt() {
298         ArrayList<ShardRouting> ordered = new ArrayList<>(activeShards.size() + allInitializingShards.size());
299         // fill it in a randomized fashion
300         for (ShardRouting shardRouting : shuffler.shuffle(activeShards)) {
301             ordered.add(shardRouting);
302             if (shardRouting.primary()) {
303                 // switch, its the matching node id
304                 ordered.set(ordered.size() - 1, ordered.get(0));
305                 ordered.set(0, shardRouting);
306             }
307         }
308         // no need to worry about primary first here..., its temporal
309         if (!allInitializingShards.isEmpty()) {
310             ordered.addAll(allInitializingShards);
311         }
312         return new PlainShardIterator(shardId, ordered);
313     }

315     public ShardIterator onlyNodeActiveInitializingShardsIt(String nodeId) {
316         ArrayList<ShardRouting> ordered = new ArrayList<>(activeShards.size() + allInitializingShards.size());
317         // fill it in a randomized fashion
318         for (int i = 0; i < activeShards.size(); i++) {
319             ShardRouting shardRouting = activeShards.get(i);
320             if (nodeId.equals(shardRouting.currentNodeId())) {
321                 ordered.add(shardRouting);
322             }
323         }
324         for (int i = 0; i < allInitializingShards.size(); i++) {
325             ShardRouting shardRouting = allInitializingShards.get(i);
326             if (nodeId.equals(shardRouting.currentNodeId())) {
327                 ordered.add(shardRouting);
328             }
329         }
330         return new PlainShardIterator(shardId, ordered);
331     }

376     static class AttributesRoutings {
377 
378         public final ImmutableList<ShardRouting> withSameAttribute;
379         public final ImmutableList<ShardRouting> withoutSameAttribute;
380         public final int totalSize;
381 
382         AttributesRoutings(ImmutableList<ShardRouting> withSameAttribute, ImmutableList<ShardRouting> withoutSameAttribute) {
383             this.withSameAttribute = withSameAttribute;
384             this.withoutSameAttribute = withoutSameAttribute;
385             this.totalSize = withoutSameAttribute.size() + withSameAttribute.size();
386         }
387     }
388 
389     private volatile Map<AttributesKey, AttributesRoutings> activeShardsByAttributes = ImmutableMap.of();
390     private volatile Map<AttributesKey, AttributesRoutings> initializingShardsByAttributes = ImmutableMap.of();
391     private final Object shardsByAttributeMutex = new Object();
392 
393     private AttributesRoutings getActiveAttribute(AttributesKey key, DiscoveryNodes nodes) {
394         AttributesRoutings shardRoutings = activeShardsByAttributes.get(key);
395         if (shardRoutings == null) {
396             synchronized (shardsByAttributeMutex) {
397                 ArrayList<ShardRouting> from = new ArrayList<>(activeShards);
398                 ImmutableList<ShardRouting> to = collectAttributeShards(key, nodes, from);
399 
400                 shardRoutings = new AttributesRoutings(to, ImmutableList.copyOf(from));
401                 activeShardsByAttributes = MapBuilder.newMapBuilder(activeShardsByAttributes).put(key, shardRoutings).immutableMap();
402             }
403         }
404         return shardRoutings;
405     }
406 
407     private AttributesRoutings getInitializingAttribute(AttributesKey key, DiscoveryNodes nodes) {
408         AttributesRoutings shardRoutings = initializingShardsByAttributes.get(key);
409         if (shardRoutings == null) {
410             synchronized (shardsByAttributeMutex) {
411                 ArrayList<ShardRouting> from = new ArrayList<>(allInitializingShards);
412                 ImmutableList<ShardRouting> to = collectAttributeShards(key, nodes, from);
413                 shardRoutings = new AttributesRoutings(to, ImmutableList.copyOf(from));
414                 initializingShardsByAttributes = MapBuilder.newMapBuilder(initializingShardsByAttributes).put(key, shardRoutings).immutableMap();
415             }
416         }
417         return shardRoutings;
418     }
419 
420     private static ImmutableList<ShardRouting> collectAttributeShards(AttributesKey key, DiscoveryNodes nodes, ArrayList<ShardRouting> from) {
421         final ArrayList<ShardRouting> to = new ArrayList<>();
422         for (final String attribute : key.attributes) {
423             final String localAttributeValue = nodes.localNode().attributes().get(attribute);
424             if (localAttributeValue != null) {
425                 for (Iterator<ShardRouting> iterator = from.iterator(); iterator.hasNext(); ) {
426                     ShardRouting fromShard = iterator.next();
427                     final DiscoveryNode discoveryNode = nodes.get(fromShard.currentNodeId());
428                     if (discoveryNode == null) {
429                         iterator.remove(); // node is not present anymore - ignore shard
430                     } else if (localAttributeValue.equals(discoveryNode.attributes().get(attribute))) {
431                         iterator.remove();
432                         to.add(fromShard);
433                     }
434                 }
435             }
436         }
437         return ImmutableList.copyOf(to);
438     }

444     public ShardIterator preferAttributesActiveInitializingShardsIt(String[] attributes, DiscoveryNodes nodes, int seed) {
445         AttributesKey key = new AttributesKey(attributes);
446         AttributesRoutings activeRoutings = getActiveAttribute(key, nodes);
447         AttributesRoutings initializingRoutings = getInitializingAttribute(key, nodes);
448 
449         // we now randomize, once between the ones that have the same attributes, and once for the ones that don't
450         // we don't want to mix between the two!
451         ArrayList<ShardRouting> ordered = new ArrayList<>(activeRoutings.totalSize + initializingRoutings.totalSize);
452         ordered.addAll(shuffler.shuffle(activeRoutings.withSameAttribute, seed));
453         ordered.addAll(shuffler.shuffle(activeRoutings.withoutSameAttribute, seed));
454         ordered.addAll(shuffler.shuffle(initializingRoutings.withSameAttribute, seed));
455         ordered.addAll(shuffler.shuffle(initializingRoutings.withoutSameAttribute, seed));
456         return new PlainShardIterator(shardId, ordered);
457     }

467     public List<ShardRouting> replicaShardsWithState(ShardRoutingState... states) {
468         List<ShardRouting> shards = newArrayList();
469         for (ShardRouting shardEntry : replicas) {
470             for (ShardRoutingState state : states) {
471                 if (shardEntry.state() == state) {
472                     shards.add(shardEntry);
473                 }
474             }
475         }
476         return shards;
477     }

479     public List<ShardRouting> shardsWithState(ShardRoutingState state) {
480         if (state == ShardRoutingState.INITIALIZING) {
481             return allInitializingShards;
482         }
483         List<ShardRouting> shards = newArrayList();
484         for (ShardRouting shardEntry : this) {
485             if (shardEntry.state() == state) {
486                 shards.add(shardEntry);
487             }
488         }
489         return shards;
490     }

492     public static class Builder {
493 
494         private ShardId shardId;
495 
496         private final List<ShardRouting> shards;
497 
498         private boolean primaryAllocatedPostApi;
499 
500         public Builder(IndexShardRoutingTable indexShard) {
501             this.shardId = indexShard.shardId;
502             this.shards = newArrayList(indexShard.shards);
503             this.primaryAllocatedPostApi = indexShard.primaryAllocatedPostApi();
504         }
505 
506         public Builder(ShardId shardId, boolean primaryAllocatedPostApi) {
507             this.shardId = shardId;
508             this.shards = newArrayList();
509             this.primaryAllocatedPostApi = primaryAllocatedPostApi;
510         }
511 
512         public Builder addShard(ImmutableShardRouting shardEntry) {
513             for (ShardRouting shard : shards) {
514                 // don't add two that map to the same node id
515                 // we rely on the fact that a node does not have primary and backup of the same shard
516                 if (shard.assignedToNode() && shardEntry.assignedToNode()
517                         && shard.currentNodeId().equals(shardEntry.currentNodeId())) {
518                     return this;
519                 }
520             }
521             shards.add(shardEntry);
522             return this;
523         }
524 
525         public Builder removeShard(ShardRouting shardEntry) {
526             shards.remove(shardEntry);
527             return this;
528         }
529 
530         public IndexShardRoutingTable build() {
531             // we can automatically set allocatedPostApi to true if the primary is active
532             if (!primaryAllocatedPostApi) {
533                 for (ShardRouting shardRouting : shards) {
534                     if (shardRouting.primary() && shardRouting.active()) {
535                         primaryAllocatedPostApi = true;
536                     }
537                 }
538             }
539             return new IndexShardRoutingTable(shardId, ImmutableList.copyOf(shards), primaryAllocatedPostApi);
540         }

542         public static IndexShardRoutingTable readFrom(StreamInput in) throws IOException {
543             String index = in.readString();
544             return readFromThin(in, index);
545         }
546 
547         public static IndexShardRoutingTable readFromThin(StreamInput in, String index) throws IOException {
548             int iShardId = in.readVInt();
549             boolean allocatedPostApi = in.readBoolean();
550             Builder builder = new Builder(new ShardId(index, iShardId), allocatedPostApi);
551 
552             int size = in.readVInt();
553             for (int i = 0; i < size; i++) {
554                 ImmutableShardRouting shard = ImmutableShardRouting.readShardRoutingEntry(in, index, iShardId);
555                 builder.addShard(shard);
556             }
557 
558             return builder.build();
559         }
560 
561         public static void writeTo(IndexShardRoutingTable indexShard, StreamOutput out) throws IOException {
562             out.writeString(indexShard.shardId().index().name());
563             writeToThin(indexShard, out);
564         }
565 
566         public static void writeToThin(IndexShardRoutingTable indexShard, StreamOutput out) throws IOException {
567             out.writeVInt(indexShard.shardId.id());
568             out.writeBoolean(indexShard.primaryAllocatedPostApi());
569 
570             out.writeVInt(indexShard.shards.size());
571             for (ShardRouting entry : indexShard) {
572                 entry.writeToThin(out);
573             }
574         }
