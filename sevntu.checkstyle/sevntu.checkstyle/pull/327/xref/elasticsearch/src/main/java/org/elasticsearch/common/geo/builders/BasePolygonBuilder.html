33   * The {@link BasePolygonBuilder} implements the groundwork to create polygons. This contains
34   * Methods to wrap polygons at the dateline and building shapes from the data held by the
35   * builder.
36   * Since this Builder can be embedded to other builders (i.e. {@link MultiPolygonBuilder}) 
37   * the class of the embedding builder is given by the generic argument <code>E</code>

114     /**
115      * The coordinates setup by the builder will be assembled to a polygon. The result will consist of
116      * a set of polygons. Each of these components holds a list of linestrings defining the polygon: the
117      * first set of coordinates will be used as the shell of the polygon. The others are defined to holes
118      * within the polygon.
119      * This Method also wraps the polygons at the dateline. In order to this fact the result may
120      * contains more polygons and less holes than defined in the builder it self.
121      * 
122      * @return coordinates of the polygon
123      */
124     public Coordinate[][][] coordinates() {
125         int numEdges = shell.points.size()-1; // Last point is repeated 
126         for (int i = 0; i < holes.size(); i++) {
127             numEdges += holes.get(i).points.size()-1;
128         }
129 
130         Edge[] edges = new Edge[numEdges];
131         Edge[] holeComponents = new Edge[holes.size()];
132         int offset = createEdges(0, orientation, shell, null, edges, 0);
133         for (int i = 0; i < holes.size(); i++) {
134             int length = createEdges(i+1, orientation, shell, this.holes.get(i), edges, offset);
135             holeComponents[i] = edges[offset];
136             offset += length;
137         }
138 
139         int numHoles = holeComponents.length;
140 
141         numHoles = merge(edges, 0, intersections(+DATELINE, edges), holeComponents, numHoles);
142         numHoles = merge(edges, 0, intersections(-DATELINE, edges), holeComponents, numHoles);
143 
144         return compose(edges, holeComponents, numHoles);
145     }

171     public Geometry buildGeometry(GeometryFactory factory, boolean fixDateline) {
172         if(fixDateline) {
173             Coordinate[][][] polygons = coordinates();
174             return polygons.length == 1
175                     ? polygon(factory, polygons[0])
176                     : multipolygon(factory, polygons);
177         } else {
178             return toPolygon(factory);
179         }
180     }

182     public Polygon toPolygon() {
183         return toPolygon(FACTORY);
184     }

186     protected Polygon toPolygon(GeometryFactory factory) {
187         final LinearRing shell = linearRing(factory, this.shell.points);
188         final LinearRing[] holes = new LinearRing[this.holes.size()];
189         Iterator<BaseLineStringBuilder<?>> iterator = this.holes.iterator();
190         for (int i = 0; iterator.hasNext(); i++) {
191             holes[i] = linearRing(factory, iterator.next().points);
192         }
193         return factory.createPolygon(shell, holes);
194     }

205     protected static Polygon polygon(GeometryFactory factory, Coordinate[][] polygon) {
206         LinearRing shell = factory.createLinearRing(polygon[0]);
207         LinearRing[] holes;
208         
209         if(polygon.length > 1) {
210             holes = new LinearRing[polygon.length-1];
211             for (int i = 0; i < holes.length; i++) {
212                 holes[i] = factory.createLinearRing(polygon[i+1]);
213             }
214         } else {
215             holes = null;
216         }
217         return factory.createPolygon(shell, holes);
218     }

220     /**
221      * Create a Multipolygon from a set of coordinates. Each primary array contains a polygon which
222      * in turn contains an array of linestrings. These line Strings are represented as an array of
223      * coordinates. The first linestring will be the shell of the polygon the others define holes
224      * within the polygon.
225      *      
226      * @param factory {@link GeometryFactory} to use
227      * @param polygons definition of polygons
228      * @return a new Multipolygon
229      */
230     protected static MultiPolygon multipolygon(GeometryFactory factory, Coordinate[][][] polygons) {
231         Polygon[] polygonSet = new Polygon[polygons.length];
232         for (int i = 0; i < polygonSet.length; i++) {
233             polygonSet[i] = polygon(factory, polygons[i]);
234         }
235         return factory.createMultiPolygon(polygonSet);
236     }

238     /**
239      * This method sets the component id of all edges in a ring to a given id and shifts the
240      * coordinates of this component according to the dateline 
241      * 
242      * @param edge An arbitrary edge of the component
243      * @param id id to apply to the component
244      * @param edges a list of edges to which all edges of the component will be added (could be <code>null</code>)
245      * @return number of edges that belong to this component
246      */
247     private static int component(final Edge edge, final int id, final ArrayList<Edge> edges) {
248         // find a coordinate that is not part of the dateline 
249         Edge any = edge;
250         while(any.coordinate.x == +DATELINE || any.coordinate.x == -DATELINE) {
251             if((any = any.next) == edge) {
252                 break;   
253             }
254         }
255 
256         double shift = any.coordinate.x > DATELINE ? DATELINE : (any.coordinate.x < -DATELINE ? -DATELINE : 0);
257         if (debugEnabled()) {
258             LOGGER.debug("shift: {[]}", shift);
259         }
260 
261         // run along the border of the component, collect the
262         // edges, shift them according to the dateline and
263         // update the component id
264         int length = 0;
265         Edge current = edge;
266         do {
267 
268             current.coordinate = shift(current.coordinate, shift); 
269             current.component = id;
270             if(edges != null) {
271                 edges.add(current);
272             }
273 
274             length++;
275         } while((current = current.next) != edge);
276 
277         return length;
278     }

280     /**
281      * Compute all coordinates of a component
282      * @param component an arbitrary edge of the component
283      * @param coordinates Array of coordinates to write the result to
284      * @return the coordinates parameter
285      */
286     private static Coordinate[] coordinates(Edge component, Coordinate[] coordinates) {
287         for (int i = 0; i < coordinates.length; i++) {
288             coordinates[i] = (component = component.next).coordinate;
289         }
290         return coordinates;
291     }

293     private static Coordinate[][][] buildCoordinates(ArrayList<ArrayList<Coordinate[]>> components) {
294         Coordinate[][][] result = new Coordinate[components.size()][][];
295         for (int i = 0; i < result.length; i++) {
296             ArrayList<Coordinate[]> component = components.get(i);
297             result[i] = component.toArray(new Coordinate[component.size()][]);
298         }
299 
300         if(debugEnabled()) {
301             for (int i = 0; i < result.length; i++) {
302                 LOGGER.debug("Component {[]}:", i);
303                 for (int j = 0; j < result[i].length; j++) {
304                     LOGGER.debug("\t" + Arrays.toString(result[i][j]));
305                 }
306             }
307         }
308 
309         return result;
310     }

312     private static final Coordinate[][] EMPTY = new Coordinate[0][];
313 
314     private static Coordinate[][] holes(Edge[] holes, int numHoles) {
315         if (numHoles == 0) {
316             return EMPTY;
317         }
318         final Coordinate[][] points = new Coordinate[numHoles][];
319 
320         for (int i = 0; i < numHoles; i++) {
321             int length = component(holes[i], -(i+1), null); // mark as visited by inverting the sign
322             points[i] = coordinates(holes[i], new Coordinate[length+1]);
323         }
324 
325         return points;
326     }

328     private static Edge[] edges(Edge[] edges, int numHoles, ArrayList<ArrayList<Coordinate[]>> components) {
329         ArrayList<Edge> mainEdges = new ArrayList<>(edges.length);
330 
331         for (int i = 0; i < edges.length; i++) {
332             if (edges[i].component >= 0) {
333                 int length = component(edges[i], -(components.size()+numHoles+1), mainEdges);
334                 ArrayList<Coordinate[]> component = new ArrayList<>();
335                 component.add(coordinates(edges[i], new Coordinate[length+1]));
336                 components.add(component);
337             }
338         }
339 
340         return mainEdges.toArray(new Edge[mainEdges.size()]);
341     }

343     private static Coordinate[][][] compose(Edge[] edges, Edge[] holes, int numHoles) {
344         final ArrayList<ArrayList<Coordinate[]>> components = new ArrayList<>();
345         assign(holes, holes(holes, numHoles), numHoles, edges(edges, numHoles, components), components);
346         return buildCoordinates(components);
347     }

349     private static void assign(Edge[] holes, Coordinate[][] points, int numHoles, Edge[] edges, ArrayList<ArrayList<Coordinate[]>> components) {
350         // Assign Hole to related components
351         // To find the new component the hole belongs to all intersections of the
352         // polygon edges with a vertical line are calculated. This vertical line
353         // is an arbitrary point of the hole. The polygon edge next to this point
354         // is part of the polygon the hole belongs to.
355         if (debugEnabled()) {
356             LOGGER.debug("Holes: " + Arrays.toString(holes));
357         }
358         for (int i = 0; i < numHoles; i++) {
359             final Edge current = new Edge(holes[i].coordinate, holes[i].next);
360             // the edge intersects with itself at its own coordinate.  We need intersect to be set this way so the binary search 
361             // will get the correct position in the edge list and therefore the correct component to add the hole
362             current.intersect = current.coordinate;
363             final int intersections = intersections(current.coordinate.x, edges);
364             // if no intersection is found then the hole is not within the polygon, so
365             // don't waste time calling a binary search
366             final int pos;
367             if (intersections == 0 ||
368                (pos = Arrays.binarySearch(edges, 0, intersections, current, INTERSECTION_ORDER)) >= 0) {
369                 throw new ElasticsearchParseException("Invalid shape: Hole is not within polygon");
370             }
371             final int index = -(pos+2);
372             final int component = -edges[index].component - numHoles - 1;
373 
374             if(debugEnabled()) {
375                 LOGGER.debug("\tposition ("+index+") of edge "+current+": " + edges[index]);
376                 LOGGER.debug("\tComponent: " + component);
377                 LOGGER.debug("\tHole intersections ("+current.coordinate.x+"): " + Arrays.toString(edges));
378             }
379 
380             components.get(component).add(points[i]);
381         }
382     }

384     private static int merge(Edge[] intersections, int offset, int length, Edge[] holes, int numHoles) {
385         // Intersections appear pairwise. On the first edge the inner of
386         // of the polygon is entered. On the second edge the outer face
387         // is entered. Other kinds of intersections are discard by the
388         // intersection function
389 
390         for (int i = 0; i < length; i += 2) {
391             Edge e1 = intersections[offset + i + 0];
392             Edge e2 = intersections[offset + i + 1];
393 
394             // If two segments are connected maybe a hole must be deleted
395             // Since Edges of components appear pairwise we need to check
396             // the second edge only (the first edge is either polygon or
397             // already handled)
398             if (e2.component > 0) {
399                 //TODO: Check if we could save the set null step
400                 numHoles--;
401                 holes[e2.component-1] = holes[numHoles];
402                 holes[numHoles] = null;
403             }
404             // only connect edges if intersections are pairwise 
405             // 1. per the comment above, the edge array is sorted by y-value of the intersection
406             // with the dateline.  Two edges have the same y intercept when they cross the 
407             // dateline thus they appear sequentially (pairwise) in the edge array. Two edges
408             // do not have the same y intercept when we're forming a multi-poly from a poly
409             // that wraps the dateline (but there are 2 ordered intercepts).  
410             // The connect method creates a new edge for these paired edges in the linked list. 
411             // For boundary conditions (e.g., intersect but not crossing) there is no sibling edge 
412             // to connect. Thus the first logic check enforces the pairwise rule
413             // 2. the second logic check ensures the two candidate edges aren't already connected by an
414             //    existing edge along the dateline - this is necessary due to a logic change in
415             //    ShapeBuilder.intersection that computes dateline edges as valid intersect points 
416             //    in support of OGC standards
417             if (e1.intersect != Edge.MAX_COORDINATE && e2.intersect != Edge.MAX_COORDINATE 
418                     && !(e1.next.next.coordinate.equals3D(e2.coordinate) && Math.abs(e1.next.coordinate.x) == DATELINE
419                     && Math.abs(e2.coordinate.x) == DATELINE) ) {
420                 connect(e1, e2);
421             }
422         }
423         return numHoles;
424     }

426     private static void connect(Edge in, Edge out) {
427         assert in != null && out != null;
428         assert in != out;
429         // Connecting two Edges by inserting the point at
430         // dateline intersection and connect these by adding
431         // two edges between this points. One per direction
432         if(in.intersect != in.next.coordinate) {
433             // NOTE: the order of the object creation is crucial here! Don't change it!
434             // first edge has no point on dateline
435             Edge e1 = new Edge(in.intersect, in.next);
436             
437             if(out.intersect != out.next.coordinate) {
438                 // second edge has no point on dateline
439                 Edge e2 = new Edge(out.intersect, out.next);
440                 in.next = new Edge(in.intersect, e2, in.intersect);
441             } else {
442                 // second edge intersects with dateline
443                 in.next = new Edge(in.intersect, out.next, in.intersect);
444             }
445             out.next = new Edge(out.intersect, e1, out.intersect);
446         } else if (in.next != out && in.coordinate != out.intersect) {
447             // first edge intersects with dateline
448             Edge e2 = new Edge(out.intersect, in.next, out.intersect);
449 
450             if(out.intersect != out.next.coordinate) {
451                 // second edge has no point on dateline
452                 Edge e1 = new Edge(out.intersect, out.next);
453                 in.next = new Edge(in.intersect, e1, in.intersect);
454                 
455             } else {
456                 // second edge intersects with dateline
457                 in.next = new Edge(in.intersect, out.next, in.intersect);
458             }
459             out.next = e2;
460         }
461     }

463     private static int createEdges(int component, Orientation orientation, BaseLineStringBuilder<?> shell,
464                                    BaseLineStringBuilder<?> hole,
465                                    Edge[] edges, int offset) {
466         // inner rings (holes) have an opposite direction than the outer rings
467         // XOR will invert the orientation for outer ring cases (Truth Table:, T/T = F, T/F = T, F/T = T, F/F = F)
468         boolean direction = (component != 0 ^ orientation == Orientation.RIGHT);
469         // set the points array accordingly (shell or hole)
470         Coordinate[] points = (hole != null) ? hole.coordinates(false) : shell.coordinates(false);
471         Edge.ring(component, direction, orientation == Orientation.LEFT, shell, points, 0, edges, offset, points.length-1);
472         return points.length-1;
473     }

475     public static class Ring<P extends ShapeBuilder> extends BaseLineStringBuilder<Ring<P>> {
476 
477         private final P parent;
478 
479         protected Ring(P parent) {
480             this(parent, new ArrayList<Coordinate>());
481         }
482 
483         protected Ring(P parent, ArrayList<Coordinate> points) {
484             super(points);
485             this.parent = parent;
486         }
487 
488         public P close() {
489             Coordinate start = points.get(0);
490             Coordinate end = points.get(points.size()-1);
491             if(start.x != end.x || start.y != end.y) {
492                 points.add(start);
493             }
494             return parent;
495         }
496 
497         @Override
498         public GeoShapeType type() {
499             return null;
500         }
