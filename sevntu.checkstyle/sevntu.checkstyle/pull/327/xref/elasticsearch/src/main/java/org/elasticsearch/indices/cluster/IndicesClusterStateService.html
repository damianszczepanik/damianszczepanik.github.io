147     @Override
148     public void clusterChanged(final ClusterChangedEvent event) {
149         if (!indicesService.changesAllowed()) {
150             return;
151         }
152 
153         if (!lifecycle.started()) {
154             return;
155         }
156 
157         synchronized (mutex) {
158             // we need to clean the shards and indices we have on this node, since we
159             // are going to recover them again once state persistence is disabled (no master / not recovered)
160             // TODO: this feels a bit hacky here, a block disables state persistence, and then we clean the allocated shards, maybe another flag in blocks?
161             if (event.state().blocks().disableStatePersistence()) {
162                 for (IndexService indexService : indicesService) {
163                     String index = indexService.index().getName();
164                     for (Integer shardId : indexService.shardIds()) {
165                         logger.debug("[{}][{}] removing shard (disabled block persistence)", index, shardId);
166                         try {
167                             indexService.removeShard(shardId, "removing shard (disabled block persistence)");
168                         } catch (Throwable e) {
169                             logger.warn("[{}] failed to remove shard (disabled block persistence)", e, index);
170                         }
171                     }
172                     removeIndex(index, "cleaning index (disabled block persistence)");
173                 }
174                 return;
175             }
176 
177             cleanFailedShards(event);
178 
179             applyDeletedIndices(event);
180             applyNewIndices(event);
181             applyMappings(event);
182             applyAliases(event);
183             applyNewOrUpdatedShards(event);
184             applyDeletedShards(event);
185             applyCleanedIndices(event);
186             applySettings(event);
187         }
188     }

190     private void applyCleanedIndices(final ClusterChangedEvent event) {
191         // handle closed indices, since they are not allocated on a node once they are closed
192         // so applyDeletedIndices might not take them into account
193         for (IndexService indexService : indicesService) {
194             String index = indexService.index().getName();
195             IndexMetaData indexMetaData = event.state().metaData().index(index);
196             if (indexMetaData != null && indexMetaData.state() == IndexMetaData.State.CLOSE) {
197                 for (Integer shardId : indexService.shardIds()) {
198                     logger.debug("[{}][{}] removing shard (index is closed)", index, shardId);
199                     try {
200                         indexService.removeShard(shardId, "removing shard (index is closed)");
201                     } catch (Throwable e) {
202                         logger.warn("[{}] failed to remove shard (index is closed)", e, index);
203                     }
204                 }
205             }
206         }
207         for (IndexService indexService : indicesService) {
208             String index = indexService.index().getName();
209             if (indexService.shardIds().isEmpty()) {
210                 if (logger.isDebugEnabled()) {
211                     logger.debug("[{}] cleaning index (no shards allocated)", index);
212                 }
213                 // clean the index
214                 removeIndex(index, "removing index (no shards allocated)");
215             }
216         }
217     }

219     private void applyDeletedIndices(final ClusterChangedEvent event) {
220         final ClusterState previousState = event.previousState();
221         final String localNodeId = event.state().nodes().localNodeId();
222         assert localNodeId != null;
223 
224         for (IndexService indexService : indicesService) {
225             IndexMetaData indexMetaData = event.state().metaData().index(indexService.index().name());
226             if (indexMetaData != null) {
227                 if (!indexMetaData.isSameUUID(indexService.indexUUID())) {
228                     logger.debug("[{}] mismatch on index UUIDs between cluster state and local state, cleaning the index so it will be recreated", indexMetaData.index());
229                     deleteIndex(indexMetaData.index(), "mismatch on index UUIDs between cluster state and local state, cleaning the index so it will be recreated");
230                 }
231             }
232         }
233 
234         for (String index : event.indicesDeleted()) {
235             if (logger.isDebugEnabled()) {
236                 logger.debug("[{}] cleaning index, no longer part of the metadata", index);
237             }
238             final Settings indexSettings;
239             final IndexService idxService = indicesService.indexService(index);
240             if (idxService != null) {
241                 indexSettings = idxService.getIndexSettings();
242                 deleteIndex(index, "index no longer part of the metadata");
243             } else {
244                 final IndexMetaData metaData = previousState.metaData().index(index);
245                 assert metaData != null;
246                 indexSettings = metaData.settings();
247                 indicesService.deleteClosedIndex("closed index no longer part of the metadata", metaData, event.state());
248             }
249             try {
250                 nodeIndexDeletedAction.nodeIndexDeleted(event.state(), index, indexSettings, localNodeId);
251             } catch (Throwable e) {
252                 logger.debug("failed to send to master index {} deleted event", e, index);
253             }
254         }
255

259     private void applyDeletedShards(final ClusterChangedEvent event) {
260         RoutingNodes.RoutingNodeIterator routingNode = event.state().readOnlyRoutingNodes().routingNodeIter(event.state().nodes().localNodeId());
261         if (routingNode == null) {
262             return;
263         }
264         IntOpenHashSet newShardIds = new IntOpenHashSet();
265         for (IndexService indexService : indicesService) {
266             String index = indexService.index().name();
267             IndexMetaData indexMetaData = event.state().metaData().index(index);
268             if (indexMetaData == null) {
269                 continue;
270             }
271             // now, go over and delete shards that needs to get deleted
272             newShardIds.clear();
273             for (MutableShardRouting shard : routingNode) {
274                 if (shard.index().equals(index)) {
275                     newShardIds.add(shard.id());
276                 }
277             }
278             for (Integer existingShardId : indexService.shardIds()) {
279                 if (!newShardIds.contains(existingShardId)) {
280                     if (indexMetaData.state() == IndexMetaData.State.CLOSE) {
281                         if (logger.isDebugEnabled()) {
282                             logger.debug("[{}][{}] removing shard (index is closed)", index, existingShardId);
283                         }
284                         indexService.removeShard(existingShardId, "removing shard (index is closed)");
285                     } else {
286                         // we can just remove the shard, without cleaning it locally, since we will clean it
287                         // when all shards are allocated in the IndicesStore
288                         if (logger.isDebugEnabled()) {
289                             logger.debug("[{}][{}] removing shard (not allocated)", index, existingShardId);
290                         }
291                         indexService.removeShard(existingShardId, "removing shard (not allocated)");
292                     }
293                 }
294             }
295         }
296     }

298     private void applyNewIndices(final ClusterChangedEvent event) {
299         // we only create indices for shards that are allocated
300         RoutingNodes.RoutingNodeIterator routingNode = event.state().readOnlyRoutingNodes().routingNodeIter(event.state().nodes().localNodeId());
301         if (routingNode == null) {
302             return;
303         }
304         for (MutableShardRouting shard : routingNode) {
305             if (!indicesService.hasIndex(shard.index())) {
306                 final IndexMetaData indexMetaData = event.state().metaData().index(shard.index());
307                 if (logger.isDebugEnabled()) {
308                     logger.debug("[{}] creating index", indexMetaData.index());
309                 }
310                 try {
311                     indicesService.createIndex(indexMetaData.index(), indexMetaData.settings(), event.state().nodes().localNode().id());
312                 } catch (Exception e) {
313                     logger.warn("[{}] failed to create index", e, indexMetaData.index());
314                 }
315             }
316         }
317     }

319     private void applySettings(ClusterChangedEvent event) {
320         if (!event.metaDataChanged()) {
321             return;
322         }
323         for (IndexMetaData indexMetaData : event.state().metaData()) {
324             if (!indicesService.hasIndex(indexMetaData.index())) {
325                 // we only create / update here
326                 continue;
327             }
328             // if the index meta data didn't change, no need check for refreshed settings
329             if (!event.indexMetaDataChanged(indexMetaData)) {
330                 continue;
331             }
332             String index = indexMetaData.index();
333             IndexService indexService = indicesService.indexService(index);
334             if (indexService == null) {
335                 // already deleted on us, ignore it
336                 continue;
337             }
338             IndexSettingsService indexSettingsService = indexService.injector().getInstance(IndexSettingsService.class);
339             indexSettingsService.refreshSettings(indexMetaData.settings());
340         }
341     }

344     private void applyMappings(ClusterChangedEvent event) {
345         // go over and update mappings
346         for (IndexMetaData indexMetaData : event.state().metaData()) {
347             if (!indicesService.hasIndex(indexMetaData.index())) {
348                 // we only create / update here
349                 continue;
350             }
351             List<String> typesToRefresh = Lists.newArrayList();
352             String index = indexMetaData.index();
353             IndexService indexService = indicesService.indexService(index);
354             if (indexService == null) {
355                 // got deleted on us, ignore (closing the node)
356                 return;
357             }
358             MapperService mapperService = indexService.mapperService();
359             // first, go over and update the _default_ mapping (if exists)
360             if (indexMetaData.mappings().containsKey(MapperService.DEFAULT_MAPPING)) {
361                 boolean requireRefresh = processMapping(index, mapperService, MapperService.DEFAULT_MAPPING, indexMetaData.mapping(MapperService.DEFAULT_MAPPING).source());
362                 if (requireRefresh) {
363                     typesToRefresh.add(MapperService.DEFAULT_MAPPING);
364                 }
365             }
366 
367             // go over and add the relevant mappings (or update them)
368             for (ObjectCursor<MappingMetaData> cursor : indexMetaData.mappings().values()) {
369                 MappingMetaData mappingMd = cursor.value;
370                 String mappingType = mappingMd.type();
371                 CompressedString mappingSource = mappingMd.source();
372                 if (mappingType.equals(MapperService.DEFAULT_MAPPING)) { // we processed _default_ first
373                     continue;
374                 }
375                 boolean requireRefresh = processMapping(index, mapperService, mappingType, mappingSource);
376                 if (requireRefresh) {
377                     typesToRefresh.add(mappingType);
378                 }
379             }
380             if (!typesToRefresh.isEmpty() && sendRefreshMapping) {
381                 nodeMappingRefreshAction.nodeMappingRefresh(event.state(),
382                         new NodeMappingRefreshAction.NodeMappingRefreshRequest(index, indexMetaData.uuid(),
383                                 typesToRefresh.toArray(new String[typesToRefresh.size()]), event.state().nodes().localNodeId())
384                 );
385             }
386             // go over and remove mappings
387             for (DocumentMapper documentMapper : mapperService.docMappers(true)) {
388                 if (seenMappings.containsKey(new Tuple<>(index, documentMapper.type())) && !indexMetaData.mappings().containsKey(documentMapper.type())) {
389                     // we have it in our mappings, but not in the metadata, and we have seen it in the cluster state, remove it
390                     mapperService.remove(documentMapper.type());
391                     seenMappings.remove(new Tuple<>(index, documentMapper.type()));
392                 }
393             }
394         }
395     }

397     private boolean processMapping(String index, MapperService mapperService, String mappingType, CompressedString mappingSource) {
398         if (!seenMappings.containsKey(new Tuple<>(index, mappingType))) {
399             seenMappings.put(new Tuple<>(index, mappingType), true);
400         }
401 
402         // refresh mapping can happen for 2 reasons. The first is less urgent, and happens when the mapping on this
403         // node is ahead of what there is in the cluster state (yet an update-mapping has been sent to it already,
404         // it just hasn't been processed yet and published). Eventually, the mappings will converge, and the refresh
405         // mapping sent is more of a safe keeping (assuming the update mapping failed to reach the master, ...)
406         // the second case is where the parsing/merging of the mapping from the metadata doesn't result in the same
407         // mapping, in this case, we send to the master to refresh its own version of the mappings (to conform with the
408         // merge version of it, which it does when refreshing the mappings), and warn log it.
409         boolean requiresRefresh = false;
410         try {
411             if (!mapperService.hasMapping(mappingType)) {
412                 if (logger.isDebugEnabled() && mappingSource.compressed().length < 512) {
413                     logger.debug("[{}] adding mapping [{}], source [{}]", index, mappingType, mappingSource.string());
414                 } else if (logger.isTraceEnabled()) {
415                     logger.trace("[{}] adding mapping [{}], source [{}]", index, mappingType, mappingSource.string());
416                 } else {
417                     logger.debug("[{}] adding mapping [{}] (source suppressed due to length, use TRACE level if needed)", index, mappingType);
418                 }
419                 // we don't apply default, since it has been applied when the mappings were parsed initially
420                 mapperService.merge(mappingType, mappingSource, false);
421                 if (!mapperService.documentMapper(mappingType).mappingSource().equals(mappingSource)) {
422                     logger.debug("[{}] parsed mapping [{}], and got different sources\noriginal:\n{}\nparsed:\n{}", index, mappingType, mappingSource, mapperService.documentMapper(mappingType).mappingSource());
423                     requiresRefresh = true;
424                 }
425             } else {
426                 DocumentMapper existingMapper = mapperService.documentMapper(mappingType);
427                 if (!mappingSource.equals(existingMapper.mappingSource())) {
428                     // mapping changed, update it
429                     if (logger.isDebugEnabled() && mappingSource.compressed().length < 512) {
430                         logger.debug("[{}] updating mapping [{}], source [{}]", index, mappingType, mappingSource.string());
431                     } else if (logger.isTraceEnabled()) {
432                         logger.trace("[{}] updating mapping [{}], source [{}]", index, mappingType, mappingSource.string());
433                     } else {
434                         logger.debug("[{}] updating mapping [{}] (source suppressed due to length, use TRACE level if needed)", index, mappingType);
435                     }
436                     // we don't apply default, since it has been applied when the mappings were parsed initially
437                     mapperService.merge(mappingType, mappingSource, false);
438                     if (!mapperService.documentMapper(mappingType).mappingSource().equals(mappingSource)) {
439                         requiresRefresh = true;
440                         logger.debug("[{}] parsed mapping [{}], and got different sources\noriginal:\n{}\nparsed:\n{}", index, mappingType, mappingSource, mapperService.documentMapper(mappingType).mappingSource());
441                     }
442                 }
443             }
444         } catch (Throwable e) {
445             logger.warn("[{}] failed to add mapping [{}], source [{}]", e, index, mappingType, mappingSource);
446         }
447         return requiresRefresh;
448     }

450     private boolean aliasesChanged(ClusterChangedEvent event) {
451         return !event.state().metaData().aliases().equals(event.previousState().metaData().aliases()) ||
452                 !event.state().routingTable().equals(event.previousState().routingTable());
453     }

455     private void applyAliases(ClusterChangedEvent event) {
456         // check if aliases changed
457         if (aliasesChanged(event)) {
458             // go over and update aliases
459             for (IndexMetaData indexMetaData : event.state().metaData()) {
460                 String index = indexMetaData.index();
461                 IndexService indexService = indicesService.indexService(index);
462                 if (indexService == null) {
463                     // we only create / update here
464                     continue;
465                 }
466                 IndexAliasesService indexAliasesService = indexService.aliasesService();
467                 processAliases(index, indexMetaData.aliases().values(), indexAliasesService);
468                 // go over and remove aliases
469                 for (IndexAlias indexAlias : indexAliasesService) {
470                     if (!indexMetaData.aliases().containsKey(indexAlias.alias())) {
471                         // we have it in our aliases, but not in the metadata, remove it
472                         indexAliasesService.remove(indexAlias.alias());
473                     }
474                 }
475             }
476         }
477     }

479     private void processAliases(String index, ObjectContainer<AliasMetaData> aliases, IndexAliasesService indexAliasesService) {
480         HashMap<String, IndexAlias> newAliases = newHashMap();
481         for (ObjectCursor<AliasMetaData> cursor : aliases) {
482             AliasMetaData aliasMd = cursor.value;
483             String alias = aliasMd.alias();
484             CompressedString filter = aliasMd.filter();
485             try {
486                 if (!indexAliasesService.hasAlias(alias)) {
487                     if (logger.isDebugEnabled()) {
488                         logger.debug("[{}] adding alias [{}], filter [{}]", index, alias, filter);
489                     }
490                     newAliases.put(alias, indexAliasesService.create(alias, filter));
491                 } else {
492                     if ((filter == null && indexAliasesService.alias(alias).filter() != null) ||
493                             (filter != null && !filter.equals(indexAliasesService.alias(alias).filter()))) {
494                         if (logger.isDebugEnabled()) {
495                             logger.debug("[{}] updating alias [{}], filter [{}]", index, alias, filter);
496                         }
497                         newAliases.put(alias, indexAliasesService.create(alias, filter));
498                     }
499                 }
500             } catch (Throwable e) {
501                 logger.warn("[{}] failed to add alias [{}], filter [{}]", e, index, alias, filter);
502             }
503         }
504         indexAliasesService.addAll(newAliases);
505     }

507     private void applyNewOrUpdatedShards(final ClusterChangedEvent event) throws ElasticsearchException {
508         if (!indicesService.changesAllowed()) {
509             return;
510         }
511 
512         RoutingTable routingTable = event.state().routingTable();
513         RoutingNodes.RoutingNodeIterator routingNode = event.state().readOnlyRoutingNodes().routingNodeIter(event.state().nodes().localNodeId());
514 
515         if (routingNode == null) {
516             failedShards.clear();
517             return;
518         }
519         DiscoveryNodes nodes = event.state().nodes();
520 
521         for (final ShardRouting shardRouting : routingNode) {
522             final IndexService indexService = indicesService.indexService(shardRouting.index());
523             if (indexService == null) {
524                 // got deleted on us, ignore
525                 continue;
526             }
527             final IndexMetaData indexMetaData = event.state().metaData().index(shardRouting.index());
528             if (indexMetaData == null) {
529                 // the index got deleted on the metadata, we will clean it later in the apply deleted method call
530                 continue;
531             }
532 
533             final int shardId = shardRouting.id();
534 
535             if (!indexService.hasShard(shardId) && shardRouting.started()) {
536                 if (failedShards.containsKey(shardRouting.shardId())) {
537                     if (nodes.masterNode() != null) {
538                         shardStateAction.resendShardFailed(shardRouting, indexMetaData.getUUID(),
539                                 "master " + nodes.masterNode() + " marked shard as started, but shard has previous failed. resending shard failure.",
540                                 nodes.masterNode()
541                         );
542                     }
543                 } else {
544                     // the master thinks we are started, but we don't have this shard at all, mark it as failed
545                     logger.warn("[{}][{}] master [{}] marked shard as started, but shard has not been created, mark shard as failed", shardRouting.index(), shardId, nodes.masterNode());
546                     failedShards.put(shardRouting.shardId(), new FailedShard(shardRouting.version()));
547                     if (nodes.masterNode() != null) {
548                         shardStateAction.shardFailed(shardRouting, indexMetaData.getUUID(),
549                                 "master " + nodes.masterNode() + " marked shard as started, but shard has not been created, mark shard as failed",
550                                 nodes.masterNode()
551                         );
552                     }
553                 }
554                 continue;
555             }
556 
557             IndexShard indexShard = indexService.shard(shardId);
558             if (indexShard != null) {
559                 ShardRouting currentRoutingEntry = indexShard.routingEntry();
560                 // if the current and global routing are initializing, but are still not the same, its a different "shard" being allocated
561                 // for example: a shard that recovers from one node and now needs to recover to another node,
562                 //              or a replica allocated and then allocating a primary because the primary failed on another node
563                 boolean shardHasBeenRemoved = false;
564                 if (currentRoutingEntry.initializing() && shardRouting.initializing() && !currentRoutingEntry.equals(shardRouting)) {
565                     logger.debug("[{}][{}] removing shard (different instance of it allocated on this node, current [{}], global [{}])", shardRouting.index(), shardRouting.id(), currentRoutingEntry, shardRouting);
566                     // closing the shard will also cancel any ongoing recovery.
567                     indexService.removeShard(shardRouting.id(), "removing shard (different instance of it allocated on this node)");
568                     shardHasBeenRemoved = true;
569                 } else if (isPeerRecovery(shardRouting)) {
570                     // check if there is an existing recovery going, and if so, and the source node is not the same, cancel the recovery to restart it
571                     RecoveryState recoveryState = recoveryTarget.recoveryState(indexShard);
572                     if (recoveryState != null && recoveryState.getStage() != RecoveryState.Stage.DONE) {
573                         // we have an ongoing recovery, find the source based on current routing and compare them
574                         DiscoveryNode sourceNode = findSourceNodeForPeerRecovery(routingTable, nodes, shardRouting);
575                         if (!recoveryState.getSourceNode().equals(sourceNode)) {
576                             logger.debug("[{}][{}] removing shard (recovery source changed), current [{}], global [{}])", shardRouting.index(), shardRouting.id(), currentRoutingEntry, shardRouting);
577                             // closing the shard will also cancel any ongoing recovery.
578                             indexService.removeShard(shardRouting.id(), "removing shard (recovery source node changed)");
579                             shardHasBeenRemoved = true;
580                         }
581                     }
582                 }
583                 if (shardHasBeenRemoved == false && !shardRouting.equals(indexShard.routingEntry())) {
584                     if (shardRouting.primary() && indexShard.routingEntry().primary() == false && shardRouting.initializing() && indexShard.allowsPrimaryPromotion() == false) {
585                         logger.debug("{} reinitialize shard on primary promotion", indexShard.shardId());
586                         indexService.removeShard(shardId, "promoted to primary");
587                     } else {
588                         // if we happen to remove the shardRouting by id above we don't need to jump in here!
589                         indexShard.routingEntry(shardRouting);
590                         indexService.shardInjectorSafe(shardId).getInstance(IndexShardGatewayService.class).routingStateChanged();
591                     }
592                 }
593             }
594 
595             if (shardRouting.initializing()) {
596                 applyInitializingShard(routingTable, nodes, indexMetaData, routingTable.index(shardRouting.index()).shard(shardRouting.id()), shardRouting);
597             }
598         }
599     }

601     private void cleanFailedShards(final ClusterChangedEvent event) {
602         RoutingTable routingTable = event.state().routingTable();
603         RoutingNodes.RoutingNodeIterator routingNode = event.state().readOnlyRoutingNodes().routingNodeIter(event.state().nodes().localNodeId());
604         if (routingNode == null) {
605             failedShards.clear();
606             return;
607         }
608         DiscoveryNodes nodes = event.state().nodes();
609         long now = System.currentTimeMillis();
610         String localNodeId = nodes.localNodeId();
611         Iterator<Map.Entry<ShardId, FailedShard>> iterator = failedShards.entrySet().iterator();
612         shards:
613         while (iterator.hasNext()) {
614             Map.Entry<ShardId, FailedShard> entry = iterator.next();
615             FailedShard failedShard = entry.getValue();
616             IndexRoutingTable indexRoutingTable = routingTable.index(entry.getKey().getIndex());
617             if (indexRoutingTable != null) {
618                 IndexShardRoutingTable shardRoutingTable = indexRoutingTable.shard(entry.getKey().id());
619                 if (shardRoutingTable != null) {
620                     for (ShardRouting shardRouting : shardRoutingTable.assignedShards()) {
621                         if (localNodeId.equals(shardRouting.currentNodeId())) {
622                             // we have a timeout here just to make sure we don't have dangled failed shards for some reason
623                             // its just another safely layer
624                             if (shardRouting.version() == failedShard.version && ((now - failedShard.timestamp) < TimeValue.timeValueMinutes(60).millis())) {
625                                 // It's the same failed shard - keep it if it hasn't timed out
626                                 continue shards;
627                             } else {
628                                 // Different version or expired, remove it
629                                 break;
630                             }
631                         }
632                     }
633                 }
634             }
635             iterator.remove();
636         }
637     }

639     private void applyInitializingShard(final RoutingTable routingTable, final DiscoveryNodes nodes, final IndexMetaData indexMetaData, final IndexShardRoutingTable indexShardRouting, final ShardRouting shardRouting) throws ElasticsearchException {
640         final IndexService indexService = indicesService.indexService(shardRouting.index());
641         if (indexService == null) {
642             // got deleted on us, ignore
643             return;
644         }
645         final int shardId = shardRouting.id();
646 
647         if (indexService.hasShard(shardId)) {
648             IndexShard indexShard = indexService.shardSafe(shardId);
649             if (indexShard.state() == IndexShardState.STARTED || indexShard.state() == IndexShardState.POST_RECOVERY) {
650                 // the master thinks we are initializing, but we are already started or on POST_RECOVERY and waiting
651                 // for master to confirm a shard started message (either master failover, or a cluster event before
652                 // we managed to tell the master we started), mark us as started
653                 if (logger.isTraceEnabled()) {
654                     logger.trace("{} master marked shard as initializing, but shard has state [{}], resending shard started to {}",
655                             indexShard.shardId(), indexShard.state(), nodes.masterNode());
656                 }
657                 if (nodes.masterNode() != null) {
658                     shardStateAction.shardStarted(shardRouting, indexMetaData.getUUID(),
659                             "master " + nodes.masterNode() + " marked shard as initializing, but shard state is [" + indexShard.state() + "], mark shard as started",
660                             nodes.masterNode());
661                 }
662                 return;
663             } else {
664                 if (indexShard.ignoreRecoveryAttempt()) {
665                     logger.trace("ignoring recovery instruction for an existing shard {} (shard state: [{}])", indexShard.shardId(), indexShard.state());
666                     return;
667                 }
668             }
669         }
670 
671         // if we're in peer recovery, try to find out the source node now so in case it fails, we will not create the index shard
672         DiscoveryNode sourceNode = null;
673         if (isPeerRecovery(shardRouting)) {
674             sourceNode = findSourceNodeForPeerRecovery(routingTable, nodes, shardRouting);
675             if (sourceNode == null) {
676                 logger.trace("ignoring initializing shard {} - no source node can be found.", shardRouting.shardId());
677                 return;
678             }
679         }
680 
681         // if there is no shard, create it
682         if (!indexService.hasShard(shardId)) {
683             if (failedShards.containsKey(shardRouting.shardId())) {
684                 if (nodes.masterNode() != null) {
685                     shardStateAction.resendShardFailed(shardRouting, indexMetaData.getUUID(),
686                             "master " + nodes.masterNode() + " marked shard as initializing, but shard is marked as failed, resend shard failure",
687                             nodes.masterNode());
688                 }
689                 return;
690             }
691             try {
692                 if (logger.isDebugEnabled()) {
693                     logger.debug("[{}][{}] creating shard", shardRouting.index(), shardId);
694                 }
695                 IndexShard indexShard = indexService.createShard(shardId, shardRouting.primary());
696                 indexShard.routingEntry(shardRouting);
697                 indexShard.addFailedEngineListener(failedEngineHandler);
698             } catch (IndexShardAlreadyExistsException e) {
699                 // ignore this, the method call can happen several times
700             } catch (Throwable e) {
701                 logger.warn("[{}][{}] failed to create shard", e, shardRouting.index(), shardRouting.id());
702                 try {
703                     indexService.removeShard(shardId, "failed to create [" + ExceptionsHelper.detailedMessage(e) + "]");
704                 } catch (IndexShardMissingException e1) {
705                     // ignore
706                 } catch (Throwable e1) {
707                     logger.warn("[{}][{}] failed to remove shard after failed creation", e1, shardRouting.index(), shardRouting.id());
708                 }
709                 failedShards.put(shardRouting.shardId(), new FailedShard(shardRouting.version()));
710                 if (nodes.masterNode() != null) {
711                     shardStateAction.shardFailed(shardRouting, indexMetaData.getUUID(), "Failed to create shard, message [" + detailedMessage(e) + "]",
712                             nodes.masterNode()
713                     );
714                 } else {
715                     logger.debug("can't send shard failed for {} as there is no current master", shardRouting.shardId());
716                 }
717                 return;
718             }
719         }
720         final IndexShard indexShard = indexService.shardSafe(shardId);
721 
722         if (indexShard.ignoreRecoveryAttempt()) {
723             // we are already recovering (we can get to this state since the cluster event can happen several
724             // times while we recover)
725             logger.trace("ignoring recovery instruction for shard {} (shard state: [{}])", indexShard.shardId(), indexShard.state());
726             return;
727         }
728 
729         if (isPeerRecovery(shardRouting)) {
730             try {
731 
732                 assert sourceNode != null : "peer recovery started but sourceNode is null";
733 
734                 // we don't mark this one as relocated at the end.
735                 // For primaries: requests in any case are routed to both when its relocating and that way we handle
736                 //    the edge case where its mark as relocated, and we might need to roll it back...
737                 // For replicas: we are recovering a backup from a primary
738                 RecoveryState.Type type = shardRouting.primary() ? RecoveryState.Type.RELOCATION : RecoveryState.Type.REPLICA;
739                 recoveryTarget.startRecovery(indexShard, type, sourceNode, new PeerRecoveryListener(shardRouting, indexService, indexMetaData));
740             } catch (Throwable e) {
741                 indexShard.failShard("corrupted preexisting index", e);
742                 handleRecoveryFailure(indexService, indexMetaData, shardRouting, true, e);
743             }
744         } else {
745             // we are the first primary, recover from the gateway
746             // if its post api allocation, the index should exists
747             boolean indexShouldExists = indexShardRouting.primaryAllocatedPostApi();
748             IndexShardGatewayService shardGatewayService = indexService.shardInjectorSafe(shardId).getInstance(IndexShardGatewayService.class);
749             shardGatewayService.recover(indexShouldExists, new IndexShardGatewayService.RecoveryListener() {
750                 @Override
751                 public void onRecoveryDone() {
752                     shardStateAction.shardStarted(shardRouting, indexMetaData.getUUID(), "after recovery from gateway");
753                 }
754 
755                 @Override
756                 public void onIgnoreRecovery(String reason) {
757                 }
758 
759                 @Override
760                 public void onRecoveryFailed(IndexShardGatewayRecoveryException e) {
761                     handleRecoveryFailure(indexService, indexMetaData, shardRouting, true, e);
762                 }
763             });
764         }
765     }

767     /**
768      * Finds the routing source node for peer recovery, return null if its not found. Note, this method expects the shard
769      * routing to *require* peer recovery, use {@link #isPeerRecovery(org.elasticsearch.cluster.routing.ShardRouting)} to
770      * check if its needed or not.
771      */
772     private DiscoveryNode findSourceNodeForPeerRecovery(RoutingTable routingTable, DiscoveryNodes nodes, ShardRouting shardRouting) {
773         DiscoveryNode sourceNode = null;
774         if (!shardRouting.primary()) {
775             IndexShardRoutingTable shardRoutingTable = routingTable.index(shardRouting.index()).shard(shardRouting.id());
776             for (ShardRouting entry : shardRoutingTable) {
777                 if (entry.primary() && entry.started()) {
778                     // only recover from started primary, if we can't find one, we will do it next round
779                     sourceNode = nodes.get(entry.currentNodeId());
780                     if (sourceNode == null) {
781                         logger.trace("can't find replica source node because primary shard {} is assigned to an unknown node.", entry);
782                         return null;
783                     }
784                     break;
785                 }
786             }
787 
788             if (sourceNode == null) {
789                 logger.trace("can't find replica source node for {} because a primary shard can not be found.", shardRouting.shardId());
790             }
791         } else if (shardRouting.relocatingNodeId() != null) {
792             sourceNode = nodes.get(shardRouting.relocatingNodeId());
793             if (sourceNode == null) {
794                 logger.trace("can't find relocation source node for shard {} because it is assigned to an unknown node [{}].", shardRouting.shardId(), shardRouting.relocatingNodeId());
795             }
796         } else {
797             throw new ElasticsearchIllegalStateException("trying to find source node for peer recovery when routing state means no peer recovery: " + shardRouting);
798         }
799         return sourceNode;
800     }
801 
802     private boolean isPeerRecovery(ShardRouting shardRouting) {
803         return !shardRouting.primary() || shardRouting.relocatingNodeId() != null;
804     }

806     private class PeerRecoveryListener implements RecoveryTarget.RecoveryListener {
807 
808         private final ShardRouting shardRouting;
809         private final IndexService indexService;
810         private final IndexMetaData indexMetaData;
811 
812         private PeerRecoveryListener(ShardRouting shardRouting, IndexService indexService, IndexMetaData indexMetaData) {
813             this.shardRouting = shardRouting;
814             this.indexService = indexService;
815             this.indexMetaData = indexMetaData;
816         }
817 
818         @Override
819         public void onRecoveryDone(RecoveryState state) {
820             shardStateAction.shardStarted(shardRouting, indexMetaData.getUUID(), "after recovery (replica) from node [" + state.getSourceNode() + "]");
821         }
822 
823         @Override
824         public void onRecoveryFailure(RecoveryState state, RecoveryFailedException e, boolean sendShardFailure) {
825             handleRecoveryFailure(indexService, indexMetaData, shardRouting, sendShardFailure, e);
826         }
827     }

829     private void handleRecoveryFailure(IndexService indexService, IndexMetaData indexMetaData, ShardRouting shardRouting, boolean sendShardFailure, Throwable failure) {
830         synchronized (mutex) {
831             if (indexService.hasShard(shardRouting.shardId().id())) {
832                 try {
833                     logger.debug("[{}][{}] removing shard on failed recovery [{}]", shardRouting.index(), shardRouting.shardId().id(), failure.getMessage());
834                     indexService.removeShard(shardRouting.shardId().id(), "recovery failure [" + ExceptionsHelper.detailedMessage(failure) + "]");
835                 } catch (IndexShardMissingException e) {
836                     // the node got closed on us, ignore it
837                 } catch (Throwable e1) {
838                     logger.warn("[{}][{}] failed to delete shard after recovery failure", e1, indexService.index().name(), shardRouting.shardId().id());
839                 }
840             }
841             if (sendShardFailure) {
842                 logger.warn("[{}][{}] sending failed shard after recovery failure", failure, indexService.index().name(), shardRouting.shardId().id());
843                 try {
844                     failedShards.put(shardRouting.shardId(), new FailedShard(shardRouting.version()));
845                     shardStateAction.shardFailed(shardRouting, indexMetaData.getUUID(), "Failed to start shard, message [" + detailedMessage(failure) + "]");
846                 } catch (Throwable e1) {
847                     logger.warn("[{}][{}] failed to mark shard as failed after a failed start", e1, indexService.index().name(), shardRouting.id());
848                 }
849             }
850         }
851     }

883     private class FailedEngineHandler implements Engine.FailedEngineListener {
884         @Override
885         public void onFailedEngine(final ShardId shardId, final String reason, final @Nullable Throwable failure) {
886             ShardRouting shardRouting = null;
887             final IndexService indexService = indicesService.indexService(shardId.index().name());
888             if (indexService != null) {
889                 IndexShard indexShard = indexService.shard(shardId.id());
890                 if (indexShard != null) {
891                     shardRouting = indexShard.routingEntry();
892                 }
893             }
894             if (shardRouting == null) {
895                 logger.warn("[{}][{}] engine failed, but can't find index shard. failure reason: [{}]",
896                         shardId.index().name(), shardId.id(), reason);
897                 return;
898             }
899             final ShardRouting fShardRouting = shardRouting;
900             final String indexUUID = indexService.indexUUID(); // we know indexService is not null here.
901             final String failureMessage = "engine failure, message [" + reason + "]" +
902                     (failure == null ? "" : "[" + detailedMessage(failure) + "]");
903             threadPool.generic().execute(new Runnable() {
904                 @Override
905                 public void run() {
906                     synchronized (mutex) {
907                         if (indexService.hasShard(shardId.id())) {
908                             try {
909 
910                                 indexService.removeShard(shardId.id(), failureMessage);
911                             } catch (IndexShardMissingException e) {
912                                 // the node got closed on us, ignore it
913                             } catch (Throwable e1) {
914                                 logger.warn("[{}][{}] failed to delete shard after failed engine ([{}])", e1, indexService.index().name(), shardId.id(), reason);
915                             }
916                         }
917                         try {
918                             failedShards.put(fShardRouting.shardId(), new FailedShard(fShardRouting.version()));
919                             shardStateAction.shardFailed(fShardRouting, indexUUID, failureMessage);
920                         } catch (Throwable e1) {
921                             logger.warn("[{}][{}] failed to mark shard as failed after a failed engine ([{}])", e1, indexService.index().name(), shardId.id(), reason);
922                         }
923                     }
924                 }
925             });
926         }
