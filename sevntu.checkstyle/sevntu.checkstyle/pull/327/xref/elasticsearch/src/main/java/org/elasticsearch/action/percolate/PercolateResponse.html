40  /**
41   * Encapsulates the response of a percolator request.
42   */
43  public class PercolateResponse extends BroadcastOperationResponse implements Iterable<PercolateResponse.Match>, ToXContent {
44  
45      public static final Match[] EMPTY = new Match[0];
46  
47      private long tookInMillis;
48      private Match[] matches;
49      private long count;
50      private InternalAggregations aggregations;
51  
52      PercolateResponse(int totalShards, int successfulShards, int failedShards, List<ShardOperationFailedException> shardFailures,
53                               Match[] matches, long count, long tookInMillis, InternalAggregations aggregations) {
54          super(totalShards, successfulShards, failedShards, shardFailures);
55          this.tookInMillis = tookInMillis;
56          this.matches = matches;
57          this.count = count;
58          this.aggregations = aggregations;
59      }
60  
61      PercolateResponse(int totalShards, int successfulShards, int failedShards, List<ShardOperationFailedException> shardFailures, long tookInMillis, Match[] matches) {
62          super(totalShards, successfulShards, failedShards, shardFailures);
63          this.tookInMillis = tookInMillis;
64          this.matches = matches;
65      }

67      PercolateResponse() {
68      }

71       * How long the percolate took.

105     @Override
106     public Iterator<Match> iterator() {
107         return Arrays.asList(matches).iterator();
108     }

110     @Override
111     public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
112         builder.field(Fields.TOOK, tookInMillis);
113         RestActions.buildBroadcastShardsHeader(builder, this);
114 
115         builder.field(Fields.TOTAL, count);
116         if (matches != null) {
117             builder.startArray(Fields.MATCHES);
118             boolean justIds = "ids".equals(params.param("percolate_format"));
119             if (justIds) {
120                 for (PercolateResponse.Match match : matches) {
121                     builder.value(match.getId());
122                 }
123             } else {
124                 for (PercolateResponse.Match match : matches) {
125                     builder.startObject();
126                     builder.field(Fields._INDEX, match.getIndex());
127                     builder.field(Fields._ID, match.getId());
128                     float score = match.getScore();
129                     if (score != PercolatorService.NO_SCORE) {
130                         builder.field(Fields._SCORE, match.getScore());
131                     }
132                     if (match.getHighlightFields() != null) {
133                         builder.startObject(Fields.HIGHLIGHT);
134                         for (HighlightField field : match.getHighlightFields().values()) {
135                             builder.field(field.name());
136                             if (field.fragments() == null) {
137                                 builder.nullValue();
138                             } else {
139                                 builder.startArray();
140                                 for (Text fragment : field.fragments()) {
141                                     builder.value(fragment);
142                                 }
143                                 builder.endArray();
144                             }
145                         }
146                         builder.endObject();
147                     }
148                     builder.endObject();
149                 }
150             }
151             builder.endArray();
152         }
153         if (aggregations != null) {
154             aggregations.toXContent(builder, params);
155         }
156         return builder;
157     }
158 
159     @Override
160     public void readFrom(StreamInput in) throws IOException {
161         super.readFrom(in);
162         tookInMillis = in.readVLong();
163         count = in.readVLong();
164         int size = in.readVInt();
165         if (size != -1) {
166             matches = new Match[size];
167             for (int i = 0; i < size; i++) {
168                 matches[i] = new Match();
169                 matches[i].readFrom(in);
170             }
171         }
172         aggregations = InternalAggregations.readOptionalAggregations(in);
173     }
174 
175     @Override
176     public void writeTo(StreamOutput out) throws IOException {
177         super.writeTo(out);
178         out.writeVLong(tookInMillis);
179         out.writeVLong(count);
180         if (matches == null) {
181             out.writeVInt(-1);
182         } else {
183             out.writeVInt(matches.length);
184             for (Match match : matches) {
185                 match.writeTo(out);
186             }
187         }
188         out.writeOptionalStreamable(aggregations);
189     }

191     /**
192      * Represents a query that has matched with the document that was percolated.
193      */
194     public static class Match implements Streamable {
195 
196         private Text index;
197         private Text id;
198         private float score;
199         private Map<String, HighlightField> hl;
200 
201         /**
202          * Constructor only for internal usage.
203          */
204         public Match(Text index, Text id, float score, Map<String, HighlightField> hl) {
205             this.id = id;
206             this.score = score;
207             this.index = index;
208             this.hl = hl;
209         }
210 
211         /**
212          * Constructor only for internal usage.
213          */
214         public Match(Text index, Text id, float score) {
215             this.id = id;
216             this.score = score;
217             this.index = index;
218         }
219 
220         Match() {
221         }
222 
223         /**
224          * @return The index that the matched percolator query resides in.
225          */
226         public Text getIndex() {
227             return index;
228         }
229 
230         /**
231          * @return The id of the matched percolator query.
232          */
233         public Text getId() {
234             return id;
235         }
236 
237         /**
238          * @return If in the percolate request a query was specified this returns the score representing how well that
239          * query matched on the metadata associated with the matching query otherwise {@link Float#NaN} is returned.
240          */
241         public float getScore() {
242             return score;
243         }
244 
245         /**
246          * @return If highlighting was specified in the percolate request the this returns highlight snippets for each
247          * matching field in the document being percolated based on this query otherwise <code>null</code> is returned.
248          */
249         @Nullable
250         public Map<String, HighlightField> getHighlightFields() {
251             return hl;
252         }
253 
254         @Override
255         public void readFrom(StreamInput in) throws IOException {
256             id = in.readText();
257             index = in.readText();
258             score = in.readFloat();
259             int size = in.readVInt();
260             if (size > 0) {
261                 hl = new HashMap<>(size);
262                 for (int j = 0; j < size; j++) {
263                     hl.put(in.readString(), HighlightField.readHighlightField(in));
264                 }
265             }
266         }
267 
268         @Override
269         public void writeTo(StreamOutput out) throws IOException {
270             out.writeText(id);
271             out.writeText(index);
272             out.writeFloat(score);
273             if (hl != null) {
274                 out.writeVInt(hl.size());
275                 for (Map.Entry<String, HighlightField> entry : hl.entrySet()) {
276                     out.writeString(entry.getKey());
277                     entry.getValue().writeTo(out);
278                 }
279             } else {
280                 out.writeVInt(0);
281             }
282         }
283     }

287         static final XContentBuilderString TOTAL = new XContentBuilderString("total");
288         static final XContentBuilderString MATCHES = new XContentBuilderString("matches");
289         static final XContentBuilderString _INDEX = new XContentBuilderString("_index");
290         static final XContentBuilderString _ID = new XContentBuilderString("_id");
291         static final XContentBuilderString _SCORE = new XContentBuilderString("_score");
292         static final XContentBuilderString HIGHLIGHT = new XContentBuilderString("highlight");
293     }
