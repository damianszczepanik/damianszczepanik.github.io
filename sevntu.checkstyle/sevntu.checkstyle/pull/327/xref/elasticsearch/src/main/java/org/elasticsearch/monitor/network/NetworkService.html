20  package org.elasticsearch.monitor.network;
21  
22  import org.elasticsearch.common.component.AbstractComponent;
23  import org.elasticsearch.common.inject.Inject;
24  import org.elasticsearch.common.settings.Settings;
25  import org.elasticsearch.common.unit.TimeValue;
26  import org.elasticsearch.common.util.SingleObjectCache;
27  
28  import java.net.InetAddress;
29  import java.net.NetworkInterface;
30  import java.util.Enumeration;
31  
32  /**
33   *
34   */
35  public final class NetworkService extends AbstractComponent {
36  
37      private final NetworkProbe probe;
38  
39      private final NetworkInfo info;
40  
41      private final SingleObjectCache<NetworkStats> networkStatsCache;
42  
43      @Inject
44      public NetworkService(Settings settings, NetworkProbe probe) {
45          super(settings);
46          this.probe = probe;
47  
48          TimeValue refreshInterval = settings.getAsTime("monitor.network.refresh_interval", TimeValue.timeValueSeconds(5));
49  
50          logger.debug("Using probe [{}] with refresh_interval [{}]", probe, refreshInterval);
51  
52          this.info = probe.networkInfo();
53          this.info.refreshInterval = refreshInterval.millis();
54          networkStatsCache = new NetworkStatsCache(refreshInterval, probe.networkStats());
55          if (logger.isDebugEnabled()) {
56              StringBuilder netDebug = new StringBuilder("net_info");
57              try {
58                  Enumeration<NetworkInterface> interfaces = NetworkInterface.getNetworkInterfaces();
59                  String hostName = InetAddress.getLocalHost().getHostName();
60                  netDebug.append("\nhost [").append(hostName).append("]\n");
61                  while (interfaces.hasMoreElements()) {
62                      NetworkInterface net = interfaces.nextElement();
63  
64                      netDebug.append(net.getName()).append('\t').append("display_name [").append(net.getDisplayName()).append("]\n");
65                      Enumeration<InetAddress> addresses = net.getInetAddresses();
66                      netDebug.append("\t\taddress ");
67                      while (addresses.hasMoreElements()) {
68                          netDebug.append("[").append(addresses.nextElement()).append("] ");
69                      }
70                      netDebug.append('\n');
71                      netDebug.append("\t\tmtu [").append(net.getMTU()).append("] multicast [").append(net.supportsMulticast()).append("] ptp [").append(net.isPointToPoint())
72                              .append("] loopback [").append(net.isLoopback()).append("] up [").append(net.isUp()).append("] virtual [").append(net.isVirtual()).append("]")
73                              .append('\n');
74  
75                      Enumeration<NetworkInterface> subInterfaces = net.getSubInterfaces();
76                      if (subInterfaces != null && subInterfaces.hasMoreElements()) {
77                          netDebug.append("\t\t\tsub interfaces:\n");
78  
79                          while (subInterfaces.hasMoreElements()) {
80  
81                              net = subInterfaces.nextElement();
82  
83                              netDebug.append("\t\t\t").append(net.getName()).append("\t").append("display_name [").append(net.getDisplayName()).append("]\n");
84                              addresses = net.getInetAddresses();
85                              netDebug.append("\t\t\t\t\taddress ");
86                              while (addresses.hasMoreElements()) {
87                                  netDebug.append("[").append(addresses.nextElement()).append("] ");
88                              }
89                              netDebug.append('\n');
90                              netDebug.append("\t\t\t\t\tmtu [").append(net.getMTU()).append("] multicast [").append(net.supportsMulticast()).append("] ptp [").append(net.isPointToPoint())
91                                      .append("] loopback [").append(net.isLoopback()).append("] up [").append(net.isUp()).append("] virtual [").append(net.isVirtual()).append("]")
92                                      .append('\n');
93                          }
94                      }
95                  }
96              } catch (Exception ex) {
97                  netDebug.append("failed to get Network Interface Info [" + ex.getMessage() + "]");
98              }
99              logger.debug(netDebug.toString());
100         }
101 
102         if (logger.isTraceEnabled()) {
103             logger.trace("ifconfig\n\n" + ifconfig());
104         }
105         stats(); // pull the stats one time
106     }
107 
108     public NetworkInfo info() {
109         return this.info;
110     }
111 
112     public  NetworkStats stats() {
113         return networkStatsCache.getOrRefresh();
114     }
115 
116     private class NetworkStatsCache extends SingleObjectCache<NetworkStats> {
117         public NetworkStatsCache(TimeValue interval, NetworkStats initValue) {
118             super(interval, initValue);
119         }
120 
121         @Override
122         protected NetworkStats refresh() {
123             return probe.networkStats();
124         }
125     }
126 
127     public String ifconfig() {
128         return probe.ifconfig();
129     }
130 }
