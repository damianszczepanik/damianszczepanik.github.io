132     @Override
133     protected Tuple<BulkShardResponse, BulkShardRequest> shardOperationOnPrimary(ClusterState clusterState, PrimaryOperationRequest shardRequest) {
134         final BulkShardRequest request = shardRequest.request;
135         IndexService indexService = indicesService.indexServiceSafe(request.index());
136         IndexShard indexShard = indexService.shardSafe(shardRequest.shardId.id());
137         Engine.IndexingOperation[] ops = null;
138         final Set<String> mappingTypesToUpdate = Sets.newHashSet();
139 
140         long[] preVersions = new long[request.items().length];
141         VersionType[] preVersionTypes = new VersionType[request.items().length];
142         for (int requestIndex = 0; requestIndex < request.items().length; requestIndex++) {
143             BulkItemRequest item = request.items()[requestIndex];
144             if (item.request() instanceof IndexRequest) {
145                 IndexRequest indexRequest = (IndexRequest) item.request();
146                 preVersions[requestIndex] = indexRequest.version();
147                 preVersionTypes[requestIndex] = indexRequest.versionType();
148                 try {
149                     try {
150                         WriteResult result = shardIndexOperation(request, indexRequest, clusterState, indexShard, true);
151                         // add the response
152                         IndexResponse indexResponse = result.response();
153                         setResponse(item, new BulkItemResponse(item.id(), indexRequest.opType().lowercase(), indexResponse));
154                         if (result.mappingTypeToUpdate != null) {
155                             mappingTypesToUpdate.add(result.mappingTypeToUpdate);
156                         }
157                         if (result.op != null) {
158                             if (ops == null) {
159                                 ops = new Engine.IndexingOperation[request.items().length];
160                             }
161                             ops[requestIndex] = result.op;
162                         }
163                     } catch (WriteFailureException e) {
164                         if (e.getMappingTypeToUpdate() != null) {
165                             mappingTypesToUpdate.add(e.getMappingTypeToUpdate());
166                         }
167                         throw e.getCause();
168                     }
169                 } catch (Throwable e) {
170                     // rethrow the failure if we are going to retry on primary and let parent failure to handle it
171                     if (retryPrimaryException(e)) {
172                         // restore updated versions...
173                         for (int j = 0; j < requestIndex; j++) {
174                             applyVersion(request.items()[j], preVersions[j], preVersionTypes[j]);
175                         }
176                         for (String mappingTypeToUpdate : mappingTypesToUpdate) {
177                             DocumentMapper docMapper = indexService.mapperService().documentMapper(mappingTypeToUpdate);
178                             if (docMapper != null) {
179                                 mappingUpdatedAction.updateMappingOnMaster(indexService.index().name(), docMapper, indexService.indexUUID());
180                             }
181                         }
182                         throw (ElasticsearchException) e;
183                     }
184                     if (e instanceof ElasticsearchException && ((ElasticsearchException) e).status() == RestStatus.CONFLICT) {
185                         logger.trace("{} failed to execute bulk item (index) {}", e, shardRequest.shardId, indexRequest);
186                     } else {
187                         logger.debug("{} failed to execute bulk item (index) {}", e, shardRequest.shardId, indexRequest);
188                     }
189                     // if its a conflict failure, and we already executed the request on a primary (and we execute it
190                     // again, due to primary relocation and only processing up to N bulk items when the shard gets closed)
191                     // then just use the response we got from the successful execution
192                     if (item.getPrimaryResponse() != null && isConflictException(e)) {
193                         setResponse(item, item.getPrimaryResponse());
194                     } else {
195                         setResponse(item, new BulkItemResponse(item.id(), indexRequest.opType().lowercase(),
196                                 new BulkItemResponse.Failure(request.index(), indexRequest.type(), indexRequest.id(), e)));
197                     }
198                 }
199             } else if (item.request() instanceof DeleteRequest) {
200                 DeleteRequest deleteRequest = (DeleteRequest) item.request();
201                 preVersions[requestIndex] = deleteRequest.version();
202                 preVersionTypes[requestIndex] = deleteRequest.versionType();
203 
204                 try {
205                     // add the response
206                     DeleteResponse deleteResponse = shardDeleteOperation(request, deleteRequest, indexShard).response();
207                     setResponse(item, new BulkItemResponse(item.id(), OP_TYPE_DELETE, deleteResponse));
208                 } catch (Throwable e) {
209                     // rethrow the failure if we are going to retry on primary and let parent failure to handle it
210                     if (retryPrimaryException(e)) {
211                         // restore updated versions...
212                         for (int j = 0; j < requestIndex; j++) {
213                             applyVersion(request.items()[j], preVersions[j], preVersionTypes[j]);
214                         }
215                         throw (ElasticsearchException) e;
216                     }
217                     if (e instanceof ElasticsearchException && ((ElasticsearchException) e).status() == RestStatus.CONFLICT) {
218                         logger.trace("{} failed to execute bulk item (delete) {}", e, shardRequest.shardId, deleteRequest);
219                     } else {
220                         logger.debug("{} failed to execute bulk item (delete) {}", e, shardRequest.shardId, deleteRequest);
221                     }
222                     // if its a conflict failure, and we already executed the request on a primary (and we execute it
223                     // again, due to primary relocation and only processing up to N bulk items when the shard gets closed)
224                     // then just use the response we got from the successful execution
225                     if (item.getPrimaryResponse() != null && isConflictException(e)) {
226                         setResponse(item, item.getPrimaryResponse());
227                     } else {
228                         setResponse(item, new BulkItemResponse(item.id(), OP_TYPE_DELETE,
229                                 new BulkItemResponse.Failure(request.index(), deleteRequest.type(), deleteRequest.id(), e)));
230                     }
231                 }
232             } else if (item.request() instanceof UpdateRequest) {
233                 UpdateRequest updateRequest = (UpdateRequest) item.request();
234                 preVersions[requestIndex] = updateRequest.version();
235                 preVersionTypes[requestIndex] = updateRequest.versionType();
236                 //  We need to do the requested retries plus the initial attempt. We don't do < 1+retry_on_conflict because retry_on_conflict may be Integer.MAX_VALUE
237                 for (int updateAttemptsCount = 0; updateAttemptsCount <= updateRequest.retryOnConflict(); updateAttemptsCount++) {
238                     UpdateResult updateResult;
239                     try {
240                         updateResult = shardUpdateOperation(clusterState, request, updateRequest, indexShard);
241                     } catch (Throwable t) {
242                         updateResult = new UpdateResult(null, null, false, t, null);
243                     }
244                     if (updateResult.success()) {
245                         switch (updateResult.result.operation()) {
246                             case UPSERT:
247                             case INDEX:
248                                 WriteResult result = updateResult.writeResult;
249                                 IndexRequest indexRequest = updateResult.request();
250                                 BytesReference indexSourceAsBytes = indexRequest.source();
251                                 // add the response
252                                 IndexResponse indexResponse = result.response();
253                                 UpdateResponse updateResponse = new UpdateResponse(indexResponse.getShardInfo(), indexResponse.getIndex(), indexResponse.getType(), indexResponse.getId(), indexResponse.getVersion(), indexResponse.isCreated());
254                                 if (updateRequest.fields() != null && updateRequest.fields().length > 0) {
255                                     Tuple<XContentType, Map<String, Object>> sourceAndContent = XContentHelper.convertToMap(indexSourceAsBytes, true);
256                                     updateResponse.setGetResult(updateHelper.extractGetResult(updateRequest, shardRequest.request.index(), indexResponse.getVersion(), sourceAndContent.v2(), sourceAndContent.v1(), indexSourceAsBytes));
257                                 }
258                                 item = request.items()[requestIndex] = new BulkItemRequest(request.items()[requestIndex].id(), indexRequest);
259                                 setResponse(item, new BulkItemResponse(item.id(), OP_TYPE_UPDATE, updateResponse));
260                                 if (result.mappingTypeToUpdate != null) {
261                                     mappingTypesToUpdate.add(result.mappingTypeToUpdate);
262                                 }
263                                 if (result.op != null) {
264                                     if (ops == null) {
265                                         ops = new Engine.IndexingOperation[request.items().length];
266                                     }
267                                     ops[requestIndex] = result.op;
268                                 }
269                                 // Replace the update request to the translated index request to execute on the replica.
270                                 break;
271                             case DELETE:
272                                 DeleteResponse response = updateResult.writeResult.response();
273                                 DeleteRequest deleteRequest = updateResult.request();
274                                 updateResponse = new UpdateResponse(response.getShardInfo(), response.getIndex(), response.getType(), response.getId(), response.getVersion(), false);
275                                 updateResponse.setGetResult(updateHelper.extractGetResult(updateRequest, shardRequest.request.index(), response.getVersion(), updateResult.result.updatedSourceAsMap(), updateResult.result.updateSourceContentType(), null));
276                                 // Replace the update request to the translated delete request to execute on the replica.
277                                 item = request.items()[requestIndex] = new BulkItemRequest(request.items()[requestIndex].id(), deleteRequest);
278                                 setResponse(item, new BulkItemResponse(item.id(), OP_TYPE_UPDATE, updateResponse));
279                                 break;
280                             case NONE:
281                                 setResponse(item, new BulkItemResponse(item.id(), OP_TYPE_UPDATE, updateResult.noopResult));
282                                 item.setIgnoreOnReplica(); // no need to go to the replica
283                                 break;
284                         }
285                         // NOTE: Breaking out of the retry_on_conflict loop!
286                         break;
287                     } else if (updateResult.failure()) {
288                         Throwable t = updateResult.error;
289                         if (updateResult.retry) {
290                             // updateAttemptCount is 0 based and marks current attempt, if it's equal to retryOnConflict we are going out of the iteration
291                             if (updateAttemptsCount >= updateRequest.retryOnConflict()) {
292                                 setResponse(item, new BulkItemResponse(item.id(), OP_TYPE_UPDATE,
293                                         new BulkItemResponse.Failure(request.index(), updateRequest.type(), updateRequest.id(), t)));
294                             }
295                         } else {
296                             // rethrow the failure if we are going to retry on primary and let parent failure to handle it
297                             if (retryPrimaryException(t)) {
298                                 // restore updated versions...
299                                 for (int j = 0; j < requestIndex; j++) {
300                                     applyVersion(request.items()[j], preVersions[j], preVersionTypes[j]);
301                                 }
302                                 throw (ElasticsearchException) t;
303                             }
304                             // if its a conflict failure, and we already executed the request on a primary (and we execute it
305                             // again, due to primary relocation and only processing up to N bulk items when the shard gets closed)
306                             // then just use the response we got from the successful execution
307                             if (item.getPrimaryResponse() != null && isConflictException(t)) {
308                                 setResponse(item, item.getPrimaryResponse());
309                             } else if (updateResult.result == null) {
310                                 setResponse(item, new BulkItemResponse(item.id(), OP_TYPE_UPDATE, new BulkItemResponse.Failure(shardRequest.request.index(), updateRequest.type(), updateRequest.id(), t)));
311                             } else {
312                                 switch (updateResult.result.operation()) {
313                                     case UPSERT:
314                                     case INDEX:
315                                         IndexRequest indexRequest = updateResult.request();
316                                         if (t instanceof ElasticsearchException && ((ElasticsearchException) t).status() == RestStatus.CONFLICT) {
317                                             logger.trace("{} failed to execute bulk item (index) {}", t, shardRequest.shardId, indexRequest);
318                                         } else {
319                                             logger.debug("{} failed to execute bulk item (index) {}", t, shardRequest.shardId, indexRequest);
320                                         }
321                                         setResponse(item, new BulkItemResponse(item.id(), OP_TYPE_UPDATE,
322                                                 new BulkItemResponse.Failure(request.index(), indexRequest.type(), indexRequest.id(), t)));
323                                         break;
324                                     case DELETE:
325                                         DeleteRequest deleteRequest = updateResult.request();
326                                         if (t instanceof ElasticsearchException && ((ElasticsearchException) t).status() == RestStatus.CONFLICT) {
327                                             logger.trace("{} failed to execute bulk item (delete) {}", t, shardRequest.shardId, deleteRequest);
328                                         } else {
329                                             logger.debug("{} failed to execute bulk item (delete) {}", t, shardRequest.shardId, deleteRequest);
330                                         }
331                                         setResponse(item, new BulkItemResponse(item.id(), OP_TYPE_DELETE,
332                                                 new BulkItemResponse.Failure(request.index(), deleteRequest.type(), deleteRequest.id(), t)));
333                                         break;
334                                 }
335                             }
336                             // NOTE: Breaking out of the retry_on_conflict loop!
337                             break;
338                         }
339 
340                     }
341                 }
342             }
343 
344             assert item.getPrimaryResponse() != null;
345             assert preVersionTypes[requestIndex] != null;
346         }
347 
348         for (String mappingTypToUpdate : mappingTypesToUpdate) {
349             DocumentMapper docMapper = indexService.mapperService().documentMapper(mappingTypToUpdate);
350             if (docMapper != null) {
351                 mappingUpdatedAction.updateMappingOnMaster(indexService.index().name(), docMapper, indexService.indexUUID());
352             }
353         }
354 
355         if (request.refresh()) {
356             try {
357                 indexShard.refresh("refresh_flag_bulk");
358             } catch (Throwable e) {
359                 // ignore
360             }
361         }
362         BulkItemResponse[] responses = new BulkItemResponse[request.items().length];
363         BulkItemRequest[] items = request.items();
364         for (int i = 0; i < items.length; i++) {
365             responses[i] = items[i].getPrimaryResponse();
366         }
367         return new Tuple<>(new BulkShardResponse(shardRequest.shardId, responses), shardRequest.request);
368     }

370     private void setResponse(BulkItemRequest request, BulkItemResponse response) {
371         request.setPrimaryResponse(response);
372         if (response.isFailed()) {
373             request.setIgnoreOnReplica();
374         }
375     }

399     private WriteResult shardIndexOperation(BulkShardRequest request, IndexRequest indexRequest, ClusterState clusterState,
400                                             IndexShard indexShard, boolean processed) {
401 
402         // validate, if routing is required, that we got routing
403         MappingMetaData mappingMd = clusterState.metaData().index(request.index()).mappingOrDefault(indexRequest.type());
404         if (mappingMd != null && mappingMd.routing().required()) {
405             if (indexRequest.routing() == null) {
406                 throw new RoutingMissingException(request.index(), indexRequest.type(), indexRequest.id());
407             }
408         }
409 
410         if (!processed) {
411             indexRequest.process(clusterState.metaData(), mappingMd, allowIdGeneration, request.index());
412         }
413 
414         SourceToParse sourceToParse = SourceToParse.source(SourceToParse.Origin.PRIMARY, indexRequest.source()).type(indexRequest.type()).id(indexRequest.id())
415                 .routing(indexRequest.routing()).parent(indexRequest.parent()).timestamp(indexRequest.timestamp()).ttl(indexRequest.ttl());
416 
417         // update mapping on master if needed, we won't update changes to the same type, since once its changed, it won't have mappers added
418         String mappingTypeToUpdate = null;
419 
420         long version;
421         boolean created;
422         Engine.IndexingOperation op;
423         try {
424             if (indexRequest.opType() == IndexRequest.OpType.INDEX) {
425                 Engine.Index index = indexShard.prepareIndex(sourceToParse, indexRequest.version(), indexRequest.versionType(), Engine.Operation.Origin.PRIMARY, request.canHaveDuplicates() || indexRequest.canHaveDuplicates());
426                 if (index.parsedDoc().mappingsModified()) {
427                     mappingTypeToUpdate = indexRequest.type();
428                 }
429                 indexShard.index(index);
430                 version = index.version();
431                 op = index;
432                 created = index.created();
433             } else {
434                 Engine.Create create = indexShard.prepareCreate(sourceToParse, indexRequest.version(), indexRequest.versionType(), Engine.Operation.Origin.PRIMARY,
435                         request.canHaveDuplicates() || indexRequest.canHaveDuplicates(), indexRequest.autoGeneratedId());
436                 if (create.parsedDoc().mappingsModified()) {
437                     mappingTypeToUpdate = indexRequest.type();
438                 }
439                 indexShard.create(create);
440                 version = create.version();
441                 op = create;
442                 created = true;
443             }
444             // update the version on request so it will happen on the replicas
445             indexRequest.versionType(indexRequest.versionType().versionTypeForReplicationAndRecovery());
446             indexRequest.version(version);
447         } catch (Throwable t) {
448             throw new WriteFailureException(t, mappingTypeToUpdate);
449         }
450 
451         assert indexRequest.versionType().validateVersionForWrites(indexRequest.version());
452 
453 
454         IndexResponse indexResponse = new IndexResponse(request.index(), indexRequest.type(), indexRequest.id(), version, created);
455         return new WriteResult(indexResponse, mappingTypeToUpdate, op);
456     }

458     private WriteResult shardDeleteOperation(BulkShardRequest request, DeleteRequest deleteRequest, IndexShard indexShard) {
459         Engine.Delete delete = indexShard.prepareDelete(deleteRequest.type(), deleteRequest.id(), deleteRequest.version(), deleteRequest.versionType(), Engine.Operation.Origin.PRIMARY);
460         indexShard.delete(delete);
461         // update the request with the version so it will go to the replicas
462         deleteRequest.versionType(delete.versionType().versionTypeForReplicationAndRecovery());
463         deleteRequest.version(delete.version());
464 
465         assert deleteRequest.versionType().validateVersionForWrites(deleteRequest.version());
466 
467         DeleteResponse deleteResponse = new DeleteResponse(request.index(), deleteRequest.type(), deleteRequest.id(), delete.version(), delete.found());
468         return new WriteResult(deleteResponse, null, null);
469     }

524     private UpdateResult shardUpdateOperation(ClusterState clusterState, BulkShardRequest bulkShardRequest, UpdateRequest updateRequest, IndexShard indexShard) {
525         UpdateHelper.Result translate = updateHelper.prepare(updateRequest, indexShard);
526         switch (translate.operation()) {
527             case UPSERT:
528             case INDEX:
529                 IndexRequest indexRequest = translate.action();
530                 try {
531                     WriteResult result = shardIndexOperation(bulkShardRequest, indexRequest, clusterState, indexShard, false);
532                     return new UpdateResult(translate, indexRequest, result);
533                 } catch (Throwable t) {
534                     t = ExceptionsHelper.unwrapCause(t);
535                     boolean retry = false;
536                     if (t instanceof VersionConflictEngineException || (t instanceof DocumentAlreadyExistsException && translate.operation() == UpdateHelper.Operation.UPSERT)) {
537                         retry = true;
538                     }
539                     return new UpdateResult(translate, indexRequest, retry, t, null);
540                 }
541             case DELETE:
542                 DeleteRequest deleteRequest = translate.action();
543                 try {
544                     WriteResult result = shardDeleteOperation(bulkShardRequest, deleteRequest, indexShard);
545                     return new UpdateResult(translate, deleteRequest, result);
546                 } catch (Throwable t) {
547                     t = ExceptionsHelper.unwrapCause(t);
548                     boolean retry = false;
549                     if (t instanceof VersionConflictEngineException) {
550                         retry = true;
551                     }
552                     return new UpdateResult(translate, deleteRequest, retry, t, null);
553                 }
554             case NONE:
555                 UpdateResponse updateResponse = translate.action();
556                 indexShard.indexingService().noopUpdate(updateRequest.type());
557                 return new UpdateResult(translate, updateResponse);
558             default:
559                 throw new ElasticsearchIllegalStateException("Illegal update operation " + translate.operation());
560         }
561     }

564     @Override
565     protected void shardOperationOnReplica(ReplicaOperationRequest shardRequest) {
566         IndexShard indexShard = indicesService.indexServiceSafe(shardRequest.shardId.getIndex()).shardSafe(shardRequest.shardId.id());
567         final BulkShardRequest request = shardRequest.request;
568         for (int i = 0; i < request.items().length; i++) {
569             BulkItemRequest item = request.items()[i];
570             if (item == null || item.isIgnoreOnReplica()) {
571                 continue;
572             }
573             if (item.request() instanceof IndexRequest) {
574                 IndexRequest indexRequest = (IndexRequest) item.request();
575                 try {
576                     SourceToParse sourceToParse = SourceToParse.source(SourceToParse.Origin.REPLICA, indexRequest.source()).type(indexRequest.type()).id(indexRequest.id())
577                             .routing(indexRequest.routing()).parent(indexRequest.parent()).timestamp(indexRequest.timestamp()).ttl(indexRequest.ttl());
578 
579                     if (indexRequest.opType() == IndexRequest.OpType.INDEX) {
580                         Engine.Index index = indexShard.prepareIndex(sourceToParse, indexRequest.version(), indexRequest.versionType(), Engine.Operation.Origin.REPLICA, request.canHaveDuplicates() || indexRequest.canHaveDuplicates());
581                         indexShard.index(index);
582                     } else {
583                         Engine.Create create = indexShard.prepareCreate(sourceToParse,
584                                 indexRequest.version(), indexRequest.versionType(),
585                                 Engine.Operation.Origin.REPLICA, request.canHaveDuplicates() || indexRequest.canHaveDuplicates(), indexRequest.autoGeneratedId());
586                         indexShard.create(create);
587                     }
588                 } catch (Throwable e) {
589                     // if its not an ignore replica failure, we need to make sure to bubble up the failure
590                     // so we will fail the shard
591                     if (!ignoreReplicaException(e)) {
592                         throw e;
593                     }
594                 }
595             } else if (item.request() instanceof DeleteRequest) {
596                 DeleteRequest deleteRequest = (DeleteRequest) item.request();
597                 try {
598                     Engine.Delete delete = indexShard.prepareDelete(deleteRequest.type(), deleteRequest.id(), deleteRequest.version(), deleteRequest.versionType(), Engine.Operation.Origin.REPLICA);
599                     indexShard.delete(delete);
600                 } catch (Throwable e) {
601                     // if its not an ignore replica failure, we need to make sure to bubble up the failure
602                     // so we will fail the shard
603                     if (!ignoreReplicaException(e)) {
604                         throw e;
605                     }
606                 }
607             }
608         }
609 
610         if (request.refresh()) {
611             try {
612                 indexShard.refresh("refresh_flag_bulk");
613             } catch (Throwable e) {
614                 // ignore
615             }
616         }
617     }

619     private void applyVersion(BulkItemRequest item, long version, VersionType versionType) {
620         if (item.request() instanceof IndexRequest) {
621             ((IndexRequest) item.request()).version(version).versionType(versionType);
622         } else if (item.request() instanceof DeleteRequest) {
623             ((DeleteRequest) item.request()).version(version).versionType();
624         } else if (item.request() instanceof UpdateRequest) {
625             ((UpdateRequest) item.request()).version(version).versionType();
626         } else {
627             // log?
628         }
629     }
