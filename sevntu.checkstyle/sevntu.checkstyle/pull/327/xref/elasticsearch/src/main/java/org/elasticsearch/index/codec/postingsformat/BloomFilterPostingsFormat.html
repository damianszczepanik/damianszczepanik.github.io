33  /**
34   * <p>
35   * A {@link PostingsFormat} useful for low doc-frequency fields such as primary
36   * keys. Bloom filters are maintained in a ".blm" file which offers "fast-fail"
37   * for reads in segments known to have no record of the key. A choice of
38   * delegate PostingsFormat is used to record all other Postings data.
39   * </p>
40   * <p>
41   * This is a special bloom filter version, based on {@link org.elasticsearch.common.util.BloomFilter} and inspired
42   * by Lucene {@link org.apache.lucene.codecs.bloom.BloomFilteringPostingsFormat}.
43   * </p>
44   * @deprecated only for reading old segments
45   */
46  @Deprecated
47  public class BloomFilterPostingsFormat extends PostingsFormat {
48  
49      public static final String BLOOM_CODEC_NAME = "XBloomFilter"; // the Lucene one is named BloomFilter
50      public static final int BLOOM_CODEC_VERSION = 1;
51      public static final int BLOOM_CODEC_VERSION_CHECKSUM = 2;
52      public static final int BLOOM_CODEC_VERSION_CURRENT = BLOOM_CODEC_VERSION_CHECKSUM;
53  
54      /**
55       * Extension of Bloom Filters file
56       */
57      static final String BLOOM_EXTENSION = "blm";
58  
59      private BloomFilter.Factory bloomFilterFactory = BloomFilter.Factory.DEFAULT;
60      private PostingsFormat delegatePostingsFormat;
61  
62      /**
63       * Creates Bloom filters for a selection of fields created in the index. This
64       * is recorded as a set of Bitsets held as a segment summary in an additional
65       * "blm" file. This PostingsFormat delegates to a choice of delegate
66       * PostingsFormat for encoding all other postings data.
67       *
68       * @param delegatePostingsFormat The PostingsFormat that records all the non-bloom filter data i.e.
69       *                               postings info.
70       * @param bloomFilterFactory     The {@link BloomFilter.Factory} responsible for sizing BloomFilters
71       *                               appropriately
72       */
73      public BloomFilterPostingsFormat(PostingsFormat delegatePostingsFormat,
74                                       BloomFilter.Factory bloomFilterFactory) {
75          super(BLOOM_CODEC_NAME);
76          this.delegatePostingsFormat = delegatePostingsFormat;
77          this.bloomFilterFactory = bloomFilterFactory;
78      }
79  
80      // Used only by core Lucene at read-time via Service Provider instantiation -
81      // do not use at Write-time in application code.
82      public BloomFilterPostingsFormat() {
83          super(BLOOM_CODEC_NAME);
84      }
85  
86      @Override
87      public BloomFilteredFieldsConsumer fieldsConsumer(SegmentWriteState state) throws IOException {
88          throw new UnsupportedOperationException("this codec can only be used for reading");
89      }
90  
91      @Override
92      public BloomFilteredFieldsProducer fieldsProducer(SegmentReadState state)
93              throws IOException {
94          return new BloomFilteredFieldsProducer(state);
95      }
96  
97      public PostingsFormat getDelegate() {
98          return delegatePostingsFormat;
99      }

101     private final class LazyBloomLoader implements Accountable {
102         private final long offset;
103         private final IndexInput indexInput;
104         private BloomFilter filter;
105 
106         private LazyBloomLoader(long offset, IndexInput origial) {
107             this.offset = offset;
108             this.indexInput = origial.clone();
109         }
110 
111         synchronized BloomFilter get() throws IOException {
112             if (filter == null) {
113                 try (final IndexInput input = indexInput) {
114                     input.seek(offset);
115                     this.filter = BloomFilter.deserialize(input);
116                 }
117             }
118             return filter;
119         }
120 
121         @Override
122         public long ramBytesUsed() {
123             return filter == null ? 0l : filter.getSizeInBytes();
124         }
125 
126         @Override
127         public Collection<Accountable> getChildResources() {
128             return Collections.singleton(Accountables.namedAccountable("bloom", ramBytesUsed()));
129         }
130     }

132     public final class BloomFilteredFieldsProducer extends FieldsProducer {
133         private FieldsProducer delegateFieldsProducer;
134         HashMap<String, LazyBloomLoader> bloomsByFieldName = new HashMap<>();
135         private final int version;
136         private final IndexInput data;
137 
138         // for internal use only
139         FieldsProducer getDelegate() {
140             return delegateFieldsProducer;
141         }
142 
143         public BloomFilteredFieldsProducer(SegmentReadState state)
144                 throws IOException {
145 
146             final String bloomFileName = IndexFileNames.segmentFileName(
147                     state.segmentInfo.name, state.segmentSuffix, BLOOM_EXTENSION);
148             final Directory directory = state.directory;
149             IndexInput dataInput = directory.openInput(bloomFileName, state.context);
150             try {
151                 ChecksumIndexInput bloomIn = new BufferedChecksumIndexInput(dataInput.clone());
152                 version = CodecUtil.checkHeader(bloomIn, BLOOM_CODEC_NAME, BLOOM_CODEC_VERSION,
153                         BLOOM_CODEC_VERSION_CURRENT);
154                 // // Load the hash function used in the BloomFilter
155                 // hashFunction = HashFunction.forName(bloomIn.readString());
156                 // Load the delegate postings format
157                final String delegatePostings = bloomIn.readString();
158                 this.delegateFieldsProducer = PostingsFormat.forName(delegatePostings)
159                         .fieldsProducer(state);
160                 this.data = dataInput;
161                 dataInput = null; // null it out such that we don't close it
162             } finally {
163                 IOUtils.closeWhileHandlingException(dataInput);
164             }
165         }
166 
167         @Override
168         public Iterator<String> iterator() {
169             return delegateFieldsProducer.iterator();
170         }
171 
172         @Override
173         public void close() throws IOException {
174             IOUtils.close(data, delegateFieldsProducer);
175         }
176 
177         @Override
178         public Terms terms(String field) throws IOException {
179             LazyBloomLoader filter = bloomsByFieldName.get(field);
180             if (filter == null) {
181                 return delegateFieldsProducer.terms(field);
182             } else {
183                 Terms result = delegateFieldsProducer.terms(field);
184                 if (result == null) {
185                     return null;
186                 }
187                 return new BloomFilteredTerms(result, filter.get());
188             }
189         }
190 
191         @Override
192         public int size() {
193             return delegateFieldsProducer.size();
194         }
195 
196         @Override
197         public long ramBytesUsed() {
198             long size = delegateFieldsProducer.ramBytesUsed();
199             for (LazyBloomLoader bloomFilter : bloomsByFieldName.values()) {
200                 size += bloomFilter.ramBytesUsed();
201             }
202             return size;
203         }
204 
205         @Override
206         public Collection<Accountable> getChildResources() {
207             List<Accountable> resources = new ArrayList<>();
208             resources.addAll(Accountables.namedAccountables("field", bloomsByFieldName));
209             if (delegateFieldsProducer != null) {
210                 resources.add(Accountables.namedAccountable("delegate", delegateFieldsProducer));
211             }
212             return Collections.unmodifiableList(resources);
213         }
214 
215         @Override
216         public void checkIntegrity() throws IOException {
217             delegateFieldsProducer.checkIntegrity();
218             if (version >= BLOOM_CODEC_VERSION_CHECKSUM) {
219                 CodecUtil.checksumEntireFile(data);
220             }
221         }
222 
223         @Override
224         public FieldsProducer getMergeInstance() throws IOException {
225             return delegateFieldsProducer.getMergeInstance();
226         }
227     }

229     public static final class BloomFilteredTerms extends FilterLeafReader.FilterTerms {
230         private BloomFilter filter;
231 
232         public BloomFilteredTerms(Terms terms, BloomFilter filter) {
233             super(terms);
234             this.filter = filter;
235         }
236 
237         public BloomFilter getFilter() {
238             return filter;
239         }
240 
241         @Override
242         public TermsEnum iterator(TermsEnum reuse) throws IOException {
243             TermsEnum result;
244             if ((reuse != null) && (reuse instanceof BloomFilteredTermsEnum)) {
245                 // recycle the existing BloomFilteredTermsEnum by asking the delegate
246                 // to recycle its contained TermsEnum
247                 BloomFilteredTermsEnum bfte = (BloomFilteredTermsEnum) reuse;
248                 if (bfte.filter == filter) {
249                     bfte.reset(this.in);
250                     return bfte;
251                 }
252                 reuse = bfte.reuse;
253             }
254             // We have been handed something we cannot reuse (either null, wrong
255             // class or wrong filter) so allocate a new object
256             result = new BloomFilteredTermsEnum(this.in, reuse, filter);
257             return result;
258         }
259     }

261     static final class BloomFilteredTermsEnum extends TermsEnum {
262 
263         private Terms delegateTerms;
264         private TermsEnum delegateTermsEnum;
265         private TermsEnum reuse;
266         private BloomFilter filter;
267 
268         public BloomFilteredTermsEnum(Terms other, TermsEnum reuse, BloomFilter filter) {
269             this.delegateTerms = other;
270             this.reuse = reuse;
271             this.filter = filter;
272         }
273 
274         void reset(Terms others) {
275             reuse = this.delegateTermsEnum;
276             this.delegateTermsEnum = null;
277             this.delegateTerms = others;
278         }
279 
280         private TermsEnum getDelegate() throws IOException {
281             if (delegateTermsEnum == null) {
282                 /* pull the iterator only if we really need it -
283                  * this can be a relatively heavy operation depending on the 
284                  * delegate postings format and they underlying directory
285                  * (clone IndexInput) */
286                 delegateTermsEnum = delegateTerms.iterator(reuse);
287             }
288             return delegateTermsEnum;
289         }
290 
291         @Override
292         public final BytesRef next() throws IOException {
293             return getDelegate().next();
294         }
295 
296         @Override
297         public final boolean seekExact(BytesRef text)
298                 throws IOException {
299             // The magical fail-fast speed up that is the entire point of all of
300             // this code - save a disk seek if there is a match on an in-memory
301             // structure
302             // that may occasionally give a false positive but guaranteed no false
303             // negatives
304             if (!filter.mightContain(text)) {
305                 return false;
306             }
307             return getDelegate().seekExact(text);
308         }
309 
310         @Override
311         public final SeekStatus seekCeil(BytesRef text)
312                 throws IOException {
313             return getDelegate().seekCeil(text);
314         }
315 
316         @Override
317         public final void seekExact(long ord) throws IOException {
318             getDelegate().seekExact(ord);
319         }
320 
321         @Override
322         public final BytesRef term() throws IOException {
323             return getDelegate().term();
324         }
325 
326         @Override
327         public final long ord() throws IOException {
328             return getDelegate().ord();
329         }
330 
331         @Override
332         public final int docFreq() throws IOException {
333             return getDelegate().docFreq();
334         }
335 
336         @Override
337         public final long totalTermFreq() throws IOException {
338             return getDelegate().totalTermFreq();
339         }
340 
341 
342         @Override
343         public PostingsEnum postings(Bits liveDocs, PostingsEnum reuse, int flags) throws IOException {
344             return getDelegate().postings(liveDocs, reuse, flags);
345         }

348     // TODO: would be great to move this out to test code, but the interaction between es090 and bloom is complex
349     // at least it is not accessible via SPI
350     public final class BloomFilteredFieldsConsumer extends FieldsConsumer {
351         private final FieldsConsumer delegateFieldsConsumer;
352         private final Map<FieldInfo, BloomFilter> bloomFilters = new HashMap<>();
353         private final SegmentWriteState state;
354         private boolean closed = false;
355 
356         // private PostingsFormat delegatePostingsFormat;
357 
358         public BloomFilteredFieldsConsumer(FieldsConsumer fieldsConsumer,
359                                            SegmentWriteState state, PostingsFormat delegatePostingsFormat) {
360             this.delegateFieldsConsumer = fieldsConsumer;
361             // this.delegatePostingsFormat=delegatePostingsFormat;
362             this.state = state;
363         }
364 
365         // for internal use only
366         public FieldsConsumer getDelegate() {
367             return delegateFieldsConsumer;
368         }
369 
370 
371         @Override
372         public void write(Fields fields) throws IOException {
373 
374             // Delegate must write first: it may have opened files
375             // on creating the class
376             // (e.g. Lucene41PostingsConsumer), and write() will
377             // close them; alternatively, if we delayed pulling
378             // the fields consumer until here, we could do it
379             // afterwards:
380             delegateFieldsConsumer.write(fields);
381 
382             for(String field : fields) {
383                 Terms terms = fields.terms(field);
384                 if (terms == null) {
385                     continue;
386                 }
387                 FieldInfo fieldInfo = state.fieldInfos.fieldInfo(field);
388                 TermsEnum termsEnum = terms.iterator(null);
389 
390                 BloomFilter bloomFilter = null;
391 
392                 PostingsEnum postings = null;
393                 while (true) {
394                     BytesRef term = termsEnum.next();
395                     if (term == null) {
396                         break;
397                     }
398                     if (bloomFilter == null) {
399                         bloomFilter = bloomFilterFactory.createFilter(state.segmentInfo.getDocCount());
400                         assert bloomFilters.containsKey(field) == false;
401                         bloomFilters.put(fieldInfo, bloomFilter);
402                     }
403                     // Make sure there's at least one doc for this term:
404                     postings = termsEnum.postings(null, postings, 0);
405                     if (postings.nextDoc() != DocIdSetIterator.NO_MORE_DOCS) {
406                         bloomFilter.put(term);
407                     }
408                 }
409             }
410         }
411 
412         @Override
413         public void close() throws IOException {
414             if (closed) {
415                 return;
416             }
417             closed = true;
418             delegateFieldsConsumer.close();
419             // Now we are done accumulating values for these fields
420             List<Entry<FieldInfo, BloomFilter>> nonSaturatedBlooms = new ArrayList<>();
421 
422             for (Entry<FieldInfo, BloomFilter> entry : bloomFilters.entrySet()) {
423                 nonSaturatedBlooms.add(entry);
424             }
425             String bloomFileName = IndexFileNames.segmentFileName(
426                     state.segmentInfo.name, state.segmentSuffix, BLOOM_EXTENSION);
427             IndexOutput bloomOutput = null;
428             try {
429                 bloomOutput = state.directory
430                         .createOutput(bloomFileName, state.context);
431                 CodecUtil.writeHeader(bloomOutput, BLOOM_CODEC_NAME,
432                         BLOOM_CODEC_VERSION_CURRENT);
433                 // remember the name of the postings format we will delegate to
434                 bloomOutput.writeString(delegatePostingsFormat.getName());
435 
436                 // First field in the output file is the number of fields+blooms saved
437                 bloomOutput.writeInt(nonSaturatedBlooms.size());
438                 for (Entry<FieldInfo, BloomFilter> entry : nonSaturatedBlooms) {
439                     FieldInfo fieldInfo = entry.getKey();
440                     BloomFilter bloomFilter = entry.getValue();
441                     bloomOutput.writeInt(fieldInfo.number);
442                     saveAppropriatelySizedBloomFilter(bloomOutput, bloomFilter, fieldInfo);
443                 }
444                 CodecUtil.writeFooter(bloomOutput);
445             } finally {
446                 IOUtils.close(bloomOutput);
447             }
448             //We are done with large bitsets so no need to keep them hanging around
449             bloomFilters.clear();
450         }
451 
452         private void saveAppropriatelySizedBloomFilter(IndexOutput bloomOutput,
453                                                        BloomFilter bloomFilter, FieldInfo fieldInfo) throws IOException {
454             BloomFilter.serilaize(bloomFilter, bloomOutput);
455         }
