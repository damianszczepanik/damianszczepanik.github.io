335     public void sortValues(Object[] sortValues) {
336         // LUCENE 4 UPGRADE: There must be a better way
337         // we want to convert to a Text object here, and not BytesRef
338 
339         // Don't write into sortValues! Otherwise the fields in FieldDoc is modified, which may be used in other places. (SearchContext#lastEmitedDoc)
340         Object[] sortValuesCopy = new Object[sortValues.length];
341         System.arraycopy(sortValues, 0, sortValuesCopy, 0, sortValues.length);
342         if (sortValues != null) {
343             for (int i = 0; i < sortValues.length; i++) {
344                 if (sortValues[i] instanceof BytesRef) {
345                     sortValuesCopy[i] = new StringAndBytesText(new BytesArray((BytesRef) sortValues[i]));
346                 }
347             }
348         }
349         this.sortValues = sortValuesCopy;
350     }

414     public static class Fields {
415         static final XContentBuilderString _INDEX = new XContentBuilderString("_index");
416         static final XContentBuilderString _TYPE = new XContentBuilderString("_type");
417         static final XContentBuilderString _ID = new XContentBuilderString("_id");
418         static final XContentBuilderString _VERSION = new XContentBuilderString("_version");
419         static final XContentBuilderString _SCORE = new XContentBuilderString("_score");
420         static final XContentBuilderString FIELDS = new XContentBuilderString("fields");
421         static final XContentBuilderString HIGHLIGHT = new XContentBuilderString("highlight");
422         static final XContentBuilderString SORT = new XContentBuilderString("sort");
423         static final XContentBuilderString MATCHED_QUERIES = new XContentBuilderString("matched_queries");
424         static final XContentBuilderString _EXPLANATION = new XContentBuilderString("_explanation");
425         static final XContentBuilderString VALUE = new XContentBuilderString("value");
426         static final XContentBuilderString DESCRIPTION = new XContentBuilderString("description");
427         static final XContentBuilderString DETAILS = new XContentBuilderString("details");
428         static final XContentBuilderString INNER_HITS = new XContentBuilderString("inner_hits");

431     @Override
432     public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
433         builder.startObject();
434         // For inner_hit hits shard is null and that is ok, because the parent search hit has all this information.
435         // Even if this was included in the inner_hit hits this would be the same, so better leave it out.
436         if (explanation() != null && shard != null) {
437             builder.field("_shard", shard.shardId());
438             builder.field("_node", shard.nodeIdText());
439         }
440         if (shard != null) {
441             builder.field(Fields._INDEX, shard.indexText());
442         }
443         builder.field(Fields._TYPE, type);
444         builder.field(Fields._ID, id);
445         if (nestedIdentity != null) {
446             nestedIdentity.toXContent(builder, params);
447         }
448         if (version != -1) {
449             builder.field(Fields._VERSION, version);
450         }
451         if (Float.isNaN(score)) {
452             builder.nullField(Fields._SCORE);
453         } else {
454             builder.field(Fields._SCORE, score);
455         }
456         if (source != null) {
457             XContentHelper.writeRawField("_source", source, builder, params);
458         }
459         if (fields != null && !fields.isEmpty()) {
460             builder.startObject(Fields.FIELDS);
461             for (SearchHitField field : fields.values()) {
462                 if (field.values().isEmpty()) {
463                     continue;
464                 }
465                 String fieldName = field.getName();
466                 if (field.isMetadataField()) {
467                     builder.field(fieldName, field.value());
468                 } else {
469                     builder.startArray(fieldName);
470                     for (Object value : field.getValues()) {
471                         builder.value(value);
472                     }
473                     builder.endArray();
474                 }
475             }
476             builder.endObject();
477         }
478         if (highlightFields != null && !highlightFields.isEmpty()) {
479             builder.startObject(Fields.HIGHLIGHT);
480             for (HighlightField field : highlightFields.values()) {
481                 builder.field(field.name());
482                 if (field.fragments() == null) {
483                     builder.nullValue();
484                 } else {
485                     builder.startArray();
486                     for (Text fragment : field.fragments()) {
487                         builder.value(fragment);
488                     }
489                     builder.endArray();
490                 }
491             }
492             builder.endObject();
493         }
494         if (sortValues != null && sortValues.length > 0) {
495             builder.startArray(Fields.SORT);
496             for (Object sortValue : sortValues) {
497                 if (sortValue != null && sortValue.equals(MAX_TERM_AS_TEXT)) {
498                     // We don't display MAX_TERM in JSON responses in case some clients have UTF-8 parsers that wouldn't accept a
499                     // non-character in the response, even though this is valid UTF-8
500                     builder.nullValue();
501                 } else {
502                     builder.value(sortValue);
503                 }
504             }
505             builder.endArray();
506         }
507         if (matchedQueries.length > 0) {
508             builder.startArray(Fields.MATCHED_QUERIES);
509             for (String matchedFilter : matchedQueries) {
510                 builder.value(matchedFilter);
511             }
512             builder.endArray();
513         }
514         if (explanation() != null) {
515             builder.field(Fields._EXPLANATION);
516             buildExplanation(builder, explanation());
517         }
518         if (innerHits != null) {
519             builder.startObject(Fields.INNER_HITS);
520             for (Map.Entry<String, InternalSearchHits> entry : innerHits.entrySet()) {
521                 builder.startObject(entry.getKey());
522                 entry.getValue().toXContent(builder, params);
523                 builder.endObject();
524             }
525             builder.endObject();
526         }
527         builder.endObject();
528         return builder;
529     }

531     private void buildExplanation(XContentBuilder builder, Explanation explanation) throws IOException {
532         builder.startObject();
533         builder.field(Fields.VALUE, explanation.getValue());
534         builder.field(Fields.DESCRIPTION, explanation.getDescription());
535         Explanation[] innerExps = explanation.getDetails();
536         if (innerExps != null) {
537             builder.startArray(Fields.DETAILS);
538             for (Explanation exp : innerExps) {
539                 buildExplanation(builder, exp);
540             }
541             builder.endArray();
542         }
543         builder.endObject();
544     }

557     public void readFrom(StreamInput in, InternalSearchHits.StreamContext context) throws IOException {
558         score = in.readFloat();
559         id = in.readText();
560         type = in.readText();
561         nestedIdentity = in.readOptionalStreamable(new InternalNestedIdentity());
562         version = in.readLong();
563         source = in.readBytesReference();
564         if (source.length() == 0) {
565             source = null;
566         }
567         if (in.readBoolean()) {
568             explanation = readExplanation(in);
569         }
570         int size = in.readVInt();
571         if (size == 0) {
572             fields = ImmutableMap.of();
573         } else if (size == 1) {
574             SearchHitField hitField = readSearchHitField(in);
575             fields = ImmutableMap.of(hitField.name(), hitField);
576         } else if (size == 2) {
577             SearchHitField hitField1 = readSearchHitField(in);
578             SearchHitField hitField2 = readSearchHitField(in);
579             fields = ImmutableMap.of(hitField1.name(), hitField1, hitField2.name(), hitField2);
580         } else if (size == 3) {
581             SearchHitField hitField1 = readSearchHitField(in);
582             SearchHitField hitField2 = readSearchHitField(in);
583             SearchHitField hitField3 = readSearchHitField(in);
584             fields = ImmutableMap.of(hitField1.name(), hitField1, hitField2.name(), hitField2, hitField3.name(), hitField3);
585         } else if (size == 4) {
586             SearchHitField hitField1 = readSearchHitField(in);
587             SearchHitField hitField2 = readSearchHitField(in);
588             SearchHitField hitField3 = readSearchHitField(in);
589             SearchHitField hitField4 = readSearchHitField(in);
590             fields = ImmutableMap.of(hitField1.name(), hitField1, hitField2.name(), hitField2, hitField3.name(), hitField3, hitField4.name(), hitField4);
591         } else if (size == 5) {
592             SearchHitField hitField1 = readSearchHitField(in);
593             SearchHitField hitField2 = readSearchHitField(in);
594             SearchHitField hitField3 = readSearchHitField(in);
595             SearchHitField hitField4 = readSearchHitField(in);
596             SearchHitField hitField5 = readSearchHitField(in);
597             fields = ImmutableMap.of(hitField1.name(), hitField1, hitField2.name(), hitField2, hitField3.name(), hitField3, hitField4.name(), hitField4, hitField5.name(), hitField5);
598         } else {
599             ImmutableMap.Builder<String, SearchHitField> builder = ImmutableMap.builder();
600             for (int i = 0; i < size; i++) {
601                 SearchHitField hitField = readSearchHitField(in);
602                 builder.put(hitField.name(), hitField);
603             }
604             fields = builder.build();
605         }
606 
607         size = in.readVInt();
608         if (size == 0) {
609             highlightFields = ImmutableMap.of();
610         } else if (size == 1) {
611             HighlightField field = readHighlightField(in);
612             highlightFields = ImmutableMap.of(field.name(), field);
613         } else if (size == 2) {
614             HighlightField field1 = readHighlightField(in);
615             HighlightField field2 = readHighlightField(in);
616             highlightFields = ImmutableMap.of(field1.name(), field1, field2.name(), field2);
617         } else if (size == 3) {
618             HighlightField field1 = readHighlightField(in);
619             HighlightField field2 = readHighlightField(in);
620             HighlightField field3 = readHighlightField(in);
621             highlightFields = ImmutableMap.of(field1.name(), field1, field2.name(), field2, field3.name(), field3);
622         } else if (size == 4) {
623             HighlightField field1 = readHighlightField(in);
624             HighlightField field2 = readHighlightField(in);
625             HighlightField field3 = readHighlightField(in);
626             HighlightField field4 = readHighlightField(in);
627             highlightFields = ImmutableMap.of(field1.name(), field1, field2.name(), field2, field3.name(), field3, field4.name(), field4);
628         } else {
629             ImmutableMap.Builder<String, HighlightField> builder = ImmutableMap.builder();
630             for (int i = 0; i < size; i++) {
631                 HighlightField field = readHighlightField(in);
632                 builder.put(field.name(), field);
633             }
634             highlightFields = builder.build();
635         }
636 
637         size = in.readVInt();
638         if (size > 0) {
639             sortValues = new Object[size];
640             for (int i = 0; i < sortValues.length; i++) {
641                 byte type = in.readByte();
642                 if (type == 0) {
643                     sortValues[i] = null;
644                 } else if (type == 1) {
645                     sortValues[i] = in.readString();
646                 } else if (type == 2) {
647                     sortValues[i] = in.readInt();
648                 } else if (type == 3) {
649                     sortValues[i] = in.readLong();
650                 } else if (type == 4) {
651                     sortValues[i] = in.readFloat();
652                 } else if (type == 5) {
653                     sortValues[i] = in.readDouble();
654                 } else if (type == 6) {
655                     sortValues[i] = in.readByte();
656                 } else if (type == 7) {
657                     sortValues[i] = in.readShort();
658                 } else if (type == 8) {
659                     sortValues[i] = in.readBoolean();
660                 } else if (type == 9) {
661                     sortValues[i] = in.readText();
662                 } else {
663                     throw new IOException("Can't match type [" + type + "]");
664                 }
665             }
666         }
667 
668         size = in.readVInt();
669         if (size > 0) {
670             matchedQueries = new String[size];
671             for (int i = 0; i < size; i++) {
672                 matchedQueries[i] = in.readString();
673             }
674         }
675 
676         if (context.streamShardTarget() == InternalSearchHits.StreamContext.ShardTargetType.STREAM) {
677             if (in.readBoolean()) {
678                 shard = readSearchShardTarget(in);
679             }
680         } else if (context.streamShardTarget() == InternalSearchHits.StreamContext.ShardTargetType.LOOKUP) {
681             int lookupId = in.readVInt();
682             if (lookupId > 0) {
683                 shard = context.handleShardLookup().get(lookupId);
684             }
685         }
686 
687         size = in.readVInt();
688         if (size > 0) {
689             innerHits = new HashMap<>(size);
690             for (int i = 0; i < size; i++) {
691                 String key = in.readString();
692                 InternalSearchHits value = InternalSearchHits.readSearchHits(in, InternalSearchHits.streamContext().streamShardTarget(InternalSearchHits.StreamContext.ShardTargetType.NO_STREAM));
693                 innerHits.put(key, value);
694             }
695         }
696     }

703     public void writeTo(StreamOutput out, InternalSearchHits.StreamContext context) throws IOException {
704         out.writeFloat(score);
705         out.writeText(id);
706         out.writeText(type);
707         out.writeOptionalStreamable(nestedIdentity);
708         out.writeLong(version);
709         out.writeBytesReference(source);
710         if (explanation == null) {
711             out.writeBoolean(false);
712         } else {
713             out.writeBoolean(true);
714             writeExplanation(out, explanation);
715         }
716         if (fields == null) {
717             out.writeVInt(0);
718         } else {
719             out.writeVInt(fields.size());
720             for (SearchHitField hitField : fields().values()) {
721                 hitField.writeTo(out);
722             }
723         }
724         if (highlightFields == null) {
725             out.writeVInt(0);
726         } else {
727             out.writeVInt(highlightFields.size());
728             for (HighlightField highlightField : highlightFields.values()) {
729                 highlightField.writeTo(out);
730             }
731         }
732 
733         if (sortValues.length == 0) {
734             out.writeVInt(0);
735         } else {
736             out.writeVInt(sortValues.length);
737             for (Object sortValue : sortValues) {
738                 if (sortValue == null) {
739                     out.writeByte((byte) 0);
740                 } else {
741                     Class type = sortValue.getClass();
742                     if (type == String.class) {
743                         out.writeByte((byte) 1);
744                         out.writeString((String) sortValue);
745                     } else if (type == Integer.class) {
746                         out.writeByte((byte) 2);
747                         out.writeInt((Integer) sortValue);
748                     } else if (type == Long.class) {
749                         out.writeByte((byte) 3);
750                         out.writeLong((Long) sortValue);
751                     } else if (type == Float.class) {
752                         out.writeByte((byte) 4);
753                         out.writeFloat((Float) sortValue);
754                     } else if (type == Double.class) {
755                         out.writeByte((byte) 5);
756                         out.writeDouble((Double) sortValue);
757                     } else if (type == Byte.class) {
758                         out.writeByte((byte) 6);
759                         out.writeByte((Byte) sortValue);
760                     } else if (type == Short.class) {
761                         out.writeByte((byte) 7);
762                         out.writeShort((Short) sortValue);
763                     } else if (type == Boolean.class) {
764                         out.writeByte((byte) 8);
765                         out.writeBoolean((Boolean) sortValue);
766                     } else if (sortValue instanceof Text) {
767                         out.writeByte((byte) 9);
768                         out.writeText((Text) sortValue);
769                     } else {
770                         throw new IOException("Can't handle sort field value of type [" + type + "]");
771                     }
772                 }
773             }
774         }
775 
776         if (matchedQueries.length == 0) {
777             out.writeVInt(0);
778         } else {
779             out.writeVInt(matchedQueries.length);
780             for (String matchedFilter : matchedQueries) {
781                 out.writeString(matchedFilter);
782             }
783         }
784 
785         if (context.streamShardTarget() == InternalSearchHits.StreamContext.ShardTargetType.STREAM) {
786             if (shard == null) {
787                 out.writeBoolean(false);
788             } else {
789                 out.writeBoolean(true);
790                 shard.writeTo(out);
791             }
792         } else if (context.streamShardTarget() == InternalSearchHits.StreamContext.ShardTargetType.LOOKUP) {
793             if (shard == null) {
794                 out.writeVInt(0);
795             } else {
796                 out.writeVInt(context.shardHandleLookup().get(shard));
797             }
798         }
799 
800         if (innerHits == null) {
801             out.writeVInt(0);
802         } else {
803             out.writeVInt(innerHits.size());
804             for (Map.Entry<String, InternalSearchHits> entry : innerHits.entrySet()) {
805                 out.writeString(entry.getKey());
806                 entry.getValue().writeTo(out, InternalSearchHits.streamContext().streamShardTarget(InternalSearchHits.StreamContext.ShardTargetType.NO_STREAM));
807             }
808         }
809     }

812 
813         private Text field;
814         private int offset;
815         private InternalNestedIdentity child;
816 
817         public InternalNestedIdentity(String field, int offset, InternalNestedIdentity child) {
818             this.field = new StringAndBytesText(field);
819             this.offset = offset;
820             this.child = child;
821         }
822 
823         InternalNestedIdentity() {
824         }
825 
826         @Override

829         }

843             field = in.readOptionalText();
844             offset = in.readInt();
845             child = in.readOptionalStreamable(new InternalNestedIdentity());
846         }
847 
848         @Override
849         public void writeTo(StreamOutput out) throws IOException {
850             out.writeOptionalText(field);
851             out.writeInt(offset);
852             out.writeOptionalStreamable(child);
853         }

855         @Override
856         public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
857             builder.startObject(Fields._NESTED);
858             if (field != null) {
859                 builder.field(Fields._NESTED_FIELD, field);
860             }
861             if (offset != -1) {
862                 builder.field(Fields._NESTED_OFFSET, offset);
863             }
864             if (child != null) {
865                 builder = child.toXContent(builder, params);
866             }
867             builder.endObject();
868             return builder;
869         }
