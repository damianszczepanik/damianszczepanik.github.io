57  public class BooleanFieldMapper extends AbstractFieldMapper<Boolean> {
58  
59      public static final String CONTENT_TYPE = "boolean";
60  
61      public static class Defaults extends AbstractFieldMapper.Defaults {
62          public static final FieldType FIELD_TYPE = new FieldType(AbstractFieldMapper.Defaults.FIELD_TYPE);
63  
64          static {
65              FIELD_TYPE.setOmitNorms(true);
66              FIELD_TYPE.setIndexOptions(IndexOptions.DOCS);
67              FIELD_TYPE.setTokenized(false);
68              FIELD_TYPE.freeze();
69          }
70  
71          public static final Boolean NULL_VALUE = null;
72      }
73  
74      public static class Values {
75          public final static BytesRef TRUE = new BytesRef("T");
76          public final static BytesRef FALSE = new BytesRef("F");
77      }
78  
79      public static class Builder extends AbstractFieldMapper.Builder<Builder, BooleanFieldMapper> {
80  
81          private Boolean nullValue = Defaults.NULL_VALUE;
82  
83          public Builder(String name) {
84              super(name, new FieldType(Defaults.FIELD_TYPE));
85              this.builder = this;
86          }
87  
88          public Builder nullValue(boolean nullValue) {
89              this.nullValue = nullValue;
90              return this;
91          }
92  
93          @Override
94          public Builder tokenized(boolean tokenized) {
95              if (tokenized) {
96                  throw new ElasticsearchIllegalArgumentException("bool field can't be tokenized");
97              }
98              return super.tokenized(tokenized);
99          }
100 
101         @Override
102         public BooleanFieldMapper build(BuilderContext context) {
103             return new BooleanFieldMapper(buildNames(context), boost, fieldType, nullValue,
104                     similarity, normsLoading, fieldDataSettings, context.indexSettings(), multiFieldsBuilder.build(this, context), copyTo);
105         }
106     }
107 
108     public static class TypeParser implements Mapper.TypeParser {
109         @Override
110         public Mapper.Builder parse(String name, Map<String, Object> node, ParserContext parserContext) throws MapperParsingException {
111             BooleanFieldMapper.Builder builder = booleanField(name);
112             parseField(builder, name, node, parserContext);
113             for (Iterator<Map.Entry<String, Object>> iterator = node.entrySet().iterator(); iterator.hasNext();) {
114                 Map.Entry<String, Object> entry = iterator.next();
115                 String propName = Strings.toUnderscoreCase(entry.getKey());
116                 Object propNode = entry.getValue();
117                 if (propName.equals("null_value")) {
118                     if (propNode == null) {
119                         throw new MapperParsingException("Property [null_value] cannot be null.");
120                     }
121                     builder.nullValue(nodeBooleanValue(propNode));
122                     iterator.remove();
123                 }
124             }
125             return builder;
126         }
127     }
128 
129     private Boolean nullValue;
130 
131     protected BooleanFieldMapper(Names names, float boost, FieldType fieldType, Boolean nullValue,
132                                  SimilarityProvider similarity, Loading normsLoading,
133                                  @Nullable Settings fieldDataSettings, Settings indexSettings, MultiFields multiFields, CopyTo copyTo) {
134         super(names, boost, fieldType, null, Lucene.KEYWORD_ANALYZER, Lucene.KEYWORD_ANALYZER, similarity, normsLoading, fieldDataSettings, indexSettings, multiFields, copyTo);
135         this.nullValue = nullValue;
136     }

138     @Override
139     public FieldType defaultFieldType() {
140         return Defaults.FIELD_TYPE;
141     }
142 
143     @Override
144     public FieldDataType defaultFieldDataType() {
145         // TODO have a special boolean type?
146         return new FieldDataType("string");

149     @Override
150     public boolean useTermQueryWithQueryString() {
151         return true;
152     }
153 
154     @Override
155     public Boolean value(Object value) {
156         if (value == null) {
157             return Boolean.FALSE;
158         }
159         String sValue = value.toString();
160         if (sValue.length() == 0) {
161             return Boolean.FALSE;
162         }
163         if (sValue.length() == 1 && sValue.charAt(0) == 'F') {
164             return Boolean.FALSE;
165         }
166         if (Booleans.parseBoolean(sValue, false)) {
167             return Boolean.TRUE;
168         }
169         return Boolean.FALSE;
170     }
171 
172     @Override
173     public Object valueForSearch(Object value) {
174         return value(value);
175     }
176 
177     @Override
178     public BytesRef indexedValueForSearch(Object value) {
179         if (value == null) {
180             return Values.FALSE;
181         }
182         if (value instanceof Boolean) {
183             return ((Boolean) value) ? Values.TRUE : Values.FALSE;
184         }
185         String sValue;
186         if (value instanceof BytesRef) {
187             sValue = ((BytesRef) value).utf8ToString();
188         } else {
189             sValue = value.toString();
190         }
191         if (sValue.length() == 0) {
192             return Values.FALSE;
193         }
194         if (sValue.length() == 1 && sValue.charAt(0) == 'F') {
195             return Values.FALSE;
196         }
197         if (Booleans.parseBoolean(sValue, false)) {
198             return Values.TRUE;
199         }
200         return Values.FALSE;
201     }
202 
203     @Override
204     public Filter nullValueFilter() {
205         if (nullValue == null) {
206             return null;
207         }
208         return new TermFilter(names().createIndexNameTerm(nullValue ? Values.TRUE : Values.FALSE));
209     }
210 
211     @Override
212     protected void parseCreateField(ParseContext context, List<Field> fields) throws IOException {
213         if (fieldType().indexOptions() == IndexOptions.NONE && !fieldType().stored()) {
214             return;
215         }
216 
217         Boolean value = context.parseExternalValue(Boolean.class);
218         if (value == null) {
219             XContentParser.Token token = context.parser().currentToken();
220             if (token == XContentParser.Token.VALUE_NULL) {
221                 if (nullValue != null) {
222                     value = nullValue;
223                 }
224             } else {
225                 value = context.parser().booleanValue();
226             }
227         }
228 
229         if (value == null) {
230             return;
231         }
232         fields.add(new Field(names.indexName(), value ? "T" : "F", fieldType));
233     }
234 
235     @Override
236     public void merge(Mapper mergeWith, MergeContext mergeContext) throws MergeMappingException {
237         super.merge(mergeWith, mergeContext);
238         if (!this.getClass().equals(mergeWith.getClass())) {
239             return;
240         }
241 
242         if (!mergeContext.mergeFlags().simulate()) {
243             this.nullValue = ((BooleanFieldMapper) mergeWith).nullValue;
244         }
245     }
246 
247     @Override
248     protected String contentType() {
249         return CONTENT_TYPE;
250     }
251 
252     @Override
253     protected void doXContentBody(XContentBuilder builder, boolean includeDefaults, Params params) throws IOException {
254         super.doXContentBody(builder, includeDefaults, params);
255         if (includeDefaults || nullValue != null) {
256             builder.field("null_value", nullValue);
257         }
258     }
259 
260     @Override
261     public boolean hasDocValues() {
262         return false;
263     }
