65  /**
66   * A node level service that delete expired docs on node primary shards.
67   */
68  public class IndicesTTLService extends AbstractLifecycleComponent<IndicesTTLService> {
69  
70      public static final String INDICES_TTL_INTERVAL = "indices.ttl.interval";
71      public static final String INDEX_TTL_DISABLE_PURGE = "index.ttl.disable_purge";
72  
73      private final ClusterService clusterService;
74      private final IndicesService indicesService;
75      private final TransportBulkAction bulkAction;
76  
77      private final int bulkSize;
78      private PurgerThread purgerThread;
79  
80      @Inject
81      public IndicesTTLService(Settings settings, ClusterService clusterService, IndicesService indicesService, NodeSettingsService nodeSettingsService, TransportBulkAction bulkAction) {
82          super(settings);
83          this.clusterService = clusterService;
84          this.indicesService = indicesService;
85          TimeValue interval = this.settings.getAsTime("indices.ttl.interval", TimeValue.timeValueSeconds(60));
86          this.bulkAction = bulkAction;
87          this.bulkSize = this.settings.getAsInt("indices.ttl.bulk_size", 10000);
88          this.purgerThread = new PurgerThread(EsExecutors.threadName(settings, "[ttl_expire]"), interval);
89  
90          nodeSettingsService.addListener(new ApplySettings());
91      }
92  
93      @Override
94      protected void doStart() throws ElasticsearchException {
95          this.purgerThread.start();
96      }
97  
98      @Override
99      protected void doStop() throws ElasticsearchException {
100         try {
101             this.purgerThread.shutdown();
102         } catch (InterruptedException e) {
103             Thread.interrupted();
104         }
105     }

111     private class PurgerThread extends Thread {
112         private final AtomicBoolean running = new AtomicBoolean(true);
113         private final Notifier notifier;
114         private final CountDownLatch shutdownLatch = new CountDownLatch(1);
115 
116 
117         public PurgerThread(String name, TimeValue interval) {
118             super(name);
119             setDaemon(true);
120             this.notifier = new Notifier(interval);
121         }
122 
123         public void shutdown() throws InterruptedException {
124             if (running.compareAndSet(true, false)) {
125                 notifier.doNotify();
126                 shutdownLatch.await();
127             }
128 
129         }
130 
131         public void resetInterval(TimeValue interval) {
132             notifier.setTimeout(interval);
133         }
134 
135         @Override
136         public void run() {
137             try {
138                 while (running.get()) {
139                     try {
140                         List<IndexShard> shardsToPurge = getShardsToPurge();
141                         purgeShards(shardsToPurge);
142                     } catch (Throwable e) {
143                         if (running.get()) {
144                             logger.warn("failed to execute ttl purge", e);
145                         }
146                     }
147                     if (running.get()) {
148                         notifier.await();
149                     }
150                 }
151             } finally {
152                 shutdownLatch.countDown();
153             }
154         }
155 
156         /**
157          * Returns the shards to purge, i.e. the local started primary shards that have ttl enabled and disable_purge to false
158          */
159         private List<IndexShard> getShardsToPurge() {
160             List<IndexShard> shardsToPurge = new ArrayList<>();
161             MetaData metaData = clusterService.state().metaData();
162             for (IndexService indexService : indicesService) {
163                 // check the value of disable_purge for this index
164                 IndexMetaData indexMetaData = metaData.index(indexService.index().name());
165                 if (indexMetaData == null) {
166                     continue;
167                 }
168                 boolean disablePurge = indexMetaData.settings().getAsBoolean(INDEX_TTL_DISABLE_PURGE, false);
169                 if (disablePurge) {
170                     continue;
171                 }
172 
173                 // should be optimized with the hasTTL flag
174                 FieldMappers ttlFieldMappers = indexService.mapperService().fullName(TTLFieldMapper.NAME);
175                 if (ttlFieldMappers == null) {
176                     continue;
177                 }
178                 // check if ttl is enabled for at least one type of this index
179                 boolean hasTTLEnabled = false;
180                 for (FieldMapper ttlFieldMapper : ttlFieldMappers) {
181                     if (((TTLFieldMapper) ttlFieldMapper).enabled()) {
182                         hasTTLEnabled = true;
183                         break;
184                     }
185                 }
186                 if (hasTTLEnabled) {
187                     for (IndexShard indexShard : indexService) {
188                         if (indexShard.state() == IndexShardState.STARTED && indexShard.routingEntry().primary() && indexShard.routingEntry().started()) {
189                             shardsToPurge.add(indexShard);
190                         }
191                     }
192                 }
193             }
194             return shardsToPurge;
195         }
196 
197         public TimeValue getInterval() {
198             return notifier.getTimeout();
199         }
200     }
201 
202     private void purgeShards(List<IndexShard> shardsToPurge) {
203         for (IndexShard shardToPurge : shardsToPurge) {
204             Query query = shardToPurge.indexService().mapperService().smartNameFieldMapper(TTLFieldMapper.NAME).rangeQuery(null, System.currentTimeMillis(), false, true, null);
205             Engine.Searcher searcher = shardToPurge.acquireSearcher("indices_ttl");
206             try {
207                 logger.debug("[{}][{}] purging shard", shardToPurge.routingEntry().index(), shardToPurge.routingEntry().id());
208                 ExpiredDocsCollector expiredDocsCollector = new ExpiredDocsCollector();
209                 searcher.searcher().search(query, expiredDocsCollector);
210                 List<DocToPurge> docsToPurge = expiredDocsCollector.getDocsToPurge();
211 
212                 BulkRequest bulkRequest = new BulkRequest();
213                 for (DocToPurge docToPurge : docsToPurge) {
214 
215                     bulkRequest.add(new DeleteRequest().index(shardToPurge.routingEntry().index()).type(docToPurge.type).id(docToPurge.id).version(docToPurge.version).routing(docToPurge.routing));
216                     bulkRequest = processBulkIfNeeded(bulkRequest, false);
217                 }
218                 processBulkIfNeeded(bulkRequest, true);
219             } catch (Exception e) {
220                 logger.warn("failed to purge", e);
221             } finally {
222                 searcher.close();
223             }
224         }
225     }

227     private static class DocToPurge {
228         public final String type;
229         public final String id;
230         public final long version;
231         public final String routing;
232 
233         public DocToPurge(String type, String id, long version, String routing) {
234             this.type = type;
235             this.id = id;
236             this.version = version;
237             this.routing = routing;
238         }
239     }

241     private class ExpiredDocsCollector extends SimpleCollector {
242         private LeafReaderContext context;
243         private List<DocToPurge> docsToPurge = new ArrayList<>();
244 
245         public ExpiredDocsCollector() {
246         }
247 
248         @Override
249         public void setScorer(Scorer scorer) {
250         }
251 
252         @Override
253         public boolean needsScores() {
254             return false;
255         }
256 
257         @Override
258         public void collect(int doc) {
259             try {
260                 UidAndRoutingFieldsVisitor fieldsVisitor = new UidAndRoutingFieldsVisitor();
261                 context.reader().document(doc, fieldsVisitor);
262                 Uid uid = fieldsVisitor.uid();
263                 final long version = Versions.loadVersion(context.reader(), new Term(UidFieldMapper.NAME, uid.toBytesRef()));
264                 docsToPurge.add(new DocToPurge(uid.type(), uid.id(), version, fieldsVisitor.routing()));
265             } catch (Exception e) {
266                 logger.trace("failed to collect doc", e);
267             }
268         }
269 
270         @Override
271         public void doSetNextReader(LeafReaderContext context) throws IOException {
272             this.context = context;
273         }
274 
275         public List<DocToPurge> getDocsToPurge() {
276             return this.docsToPurge;
277         }
278     }

280     private BulkRequest processBulkIfNeeded(BulkRequest bulkRequest, boolean force) {
281         if ((force && bulkRequest.numberOfActions() > 0) || bulkRequest.numberOfActions() >= bulkSize) {
282             try {
283                 bulkAction.executeBulk(bulkRequest, new ActionListener<BulkResponse>() {
284                     @Override
285                     public void onResponse(BulkResponse bulkResponse) {
286                         logger.trace("bulk took " + bulkResponse.getTookInMillis() + "ms");
287                     }
288 
289                     @Override
290                     public void onFailure(Throwable e) {
291                         logger.warn("failed to execute bulk");
292                     }
293                 });
294             } catch (Exception e) {
295                 logger.warn("failed to process bulk", e);
296             }
297             bulkRequest = new BulkRequest();
298         }
299         return bulkRequest;
300     }

302     class ApplySettings implements NodeSettingsService.Listener {
303         @Override
304         public void onRefreshSettings(Settings settings) {
305             final TimeValue currentInterval = IndicesTTLService.this.purgerThread.getInterval();
306             final TimeValue interval = settings.getAsTime(INDICES_TTL_INTERVAL, currentInterval);
307             if (!interval.equals(currentInterval)) {
308                 logger.info("updating indices.ttl.interval from [{}] to [{}]",currentInterval, interval);
309                 IndicesTTLService.this.purgerThread.resetInterval(interval);
310 
311             }
312         }
313     }

316     private static final class Notifier {
317 
318         private final ReentrantLock lock = new ReentrantLock();
319         private final Condition condition = lock.newCondition();
320         private volatile TimeValue timeout;
321 
322         public Notifier(TimeValue timeout) {
323             assert timeout != null;
324             this.timeout = timeout;
325         }
326 
327         public void await() {
328             lock.lock();
329             try {
330                 condition.await(timeout.millis(), TimeUnit.MILLISECONDS);
331             } catch (InterruptedException e) {
332                 Thread.interrupted();
333             } finally {
334                 lock.unlock();
335             }
336 
337         }
338 
339         public void setTimeout(TimeValue timeout) {
340             assert timeout != null;
341             this.timeout = timeout;
342             doNotify();
343         }
344 
345         public TimeValue getTimeout() {
346             return timeout;
347         }
348 
349         public void doNotify() {
350             lock.lock();
351             try {
352                 condition.signalAll();
353             } finally {
354                 lock.unlock();
355             }
356         }
357     }
