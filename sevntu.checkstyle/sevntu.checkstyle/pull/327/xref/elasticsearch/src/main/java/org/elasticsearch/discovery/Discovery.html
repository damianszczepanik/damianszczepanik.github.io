20  package org.elasticsearch.discovery;
21  
22  import org.elasticsearch.cluster.ClusterState;
23  import org.elasticsearch.cluster.node.DiscoveryNode;
24  import org.elasticsearch.cluster.routing.allocation.AllocationService;
25  import org.elasticsearch.common.Nullable;
26  import org.elasticsearch.common.component.LifecycleComponent;
27  import org.elasticsearch.node.service.NodeService;
28  
29  /**
30   * A pluggable module allowing to implement discovery of other nodes, publishing of the cluster
31   * state to all nodes, electing a master of the cluster that raises cluster state change
32   * events.
33   */
34  public interface Discovery extends LifecycleComponent<Discovery> {
35  
36      DiscoveryNode localNode();
37  
38      void addListener(InitialStateDiscoveryListener listener);
39  
40      void removeListener(InitialStateDiscoveryListener listener);
41  
42      String nodeDescription();
43  
44      /**
45       * Here as a hack to solve dep injection problem...
46       */
47      void setNodeService(@Nullable NodeService nodeService);
48  
49      /**
50       * Another hack to solve dep injection problem..., note, this will be called before
51       * any start is called.
52       */
53      void setAllocationService(AllocationService allocationService);
54  
55      /**
56       * Publish all the changes to the cluster from the master (can be called just by the master). The publish
57       * process should not publish this state to the master as well! (the master is sending it...).
58       *
59       * The {@link AckListener} allows to keep track of the ack received from nodes, and verify whether
60       * they updated their own cluster state or not.
61       */
62      void publish(ClusterState clusterState, AckListener ackListener);
63  
64      public static interface AckListener {
65          void onNodeAck(DiscoveryNode node, @Nullable Throwable t);
66          void onTimeout();
67      }
68  }
