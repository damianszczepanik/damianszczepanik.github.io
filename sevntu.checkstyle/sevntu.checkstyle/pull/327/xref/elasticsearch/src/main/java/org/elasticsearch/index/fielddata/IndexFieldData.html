48      public static class CommonSettings {
49          public static final String SETTING_MEMORY_STORAGE_HINT = "memory_storage_hint";
50  
51          public enum MemoryStorageFormat {
52              ORDINALS, PACKED, PAGED;
53  
54              public static MemoryStorageFormat fromString(String string) {
55                  for (MemoryStorageFormat e : MemoryStorageFormat.values()) {
56                      if (e.name().equalsIgnoreCase(string)) {
57                          return e;
58                      }
59                  }
60                  return null;
61              }
62          }
63  
64          /**
65           * Gets a memory storage hint that should be honored if possible but is not mandatory
66           */
67          public static MemoryStorageFormat getMemoryStorageHint(FieldDataType fieldDataType) {
68              // backwards compatibility
69              String s = fieldDataType.getSettings().get("ordinals");
70              if (s != null) {
71                  return "always".equals(s) ? MemoryStorageFormat.ORDINALS : null;
72              }
73              return MemoryStorageFormat.fromString(fieldDataType.getSettings().get(SETTING_MEMORY_STORAGE_HINT));
74          }
75      }

109     // we need this extended source we we have custom comparators to reuse our field data
110     // in this case, we need to reduce type that will be used when search results are reduced
111     // on another node (we don't have the custom source them...)
112     public abstract class XFieldComparatorSource extends FieldComparatorSource {
113 
114         /** UTF-8 term containing a single code point: {@link Character#MAX_CODE_POINT} which will compare greater than all other index terms
115          *  since {@link Character#MAX_CODE_POINT} is a noncharacter and thus shouldn't appear in an index term. */
116         public static final BytesRef MAX_TERM;
117         static {
118             BytesRefBuilder builder = new BytesRefBuilder();
119             final char[] chars = Character.toChars(Character.MAX_CODE_POINT);
120             builder.copyChars(chars, 0, chars.length);
121             MAX_TERM = builder.toBytesRef();
122         }
123 
124         /**
125          * Simple wrapper class around a filter that matches parent documents
126          * and a filter that matches child documents. For every root document R,
127          * R will be in the parent filter and its children documents will be the
128          * documents that are contained in the inner set between the previous
129          * parent + 1, or 0 if there is no previous parent, and R (excluded).
130          */
131         public static class Nested {
132 
133             private final BitDocIdSetFilter rootFilter;
134             private final Filter innerFilter;
135 
136             public Nested(BitDocIdSetFilter rootFilter, Filter innerFilter) {
137                 this.rootFilter = rootFilter;
138                 this.innerFilter = innerFilter;
139             }
140 
141             /**
142              * Get a {@link BitDocIdSet} that matches the root documents.
143              */
144             public BitDocIdSet rootDocs(LeafReaderContext ctx) throws IOException {
145                 return rootFilter.getDocIdSet(ctx);
146             }
147 
148             /**
149              * Get a {@link DocIdSet} that matches the inner documents.
150              */
151             public DocIdSet innerDocs(LeafReaderContext ctx) throws IOException {
152                 return innerFilter.getDocIdSet(ctx, null);
153             }
154         }
155 
156         /** Whether missing values should be sorted first. */
157         protected final boolean sortMissingFirst(Object missingValue) {
158             return "_first".equals(missingValue);
159         }
160 
161         /** Whether missing values should be sorted last, this is the default. */
162         protected final boolean sortMissingLast(Object missingValue) {
163             return missingValue == null || "_last".equals(missingValue);
164         }
165 
166         /** Return the missing object value according to the reduced type of the comparator. */
167         protected final Object missingObject(Object missingValue, boolean reversed) {
168             if (sortMissingFirst(missingValue) || sortMissingLast(missingValue)) {
169                 final boolean min = sortMissingFirst(missingValue) ^ reversed;
170                 switch (reducedType()) {
171                 case INT:
172                     return min ? Integer.MIN_VALUE : Integer.MAX_VALUE;
173                 case LONG:
174                     return min ? Long.MIN_VALUE : Long.MAX_VALUE;
175                 case FLOAT:
176                     return min ? Float.NEGATIVE_INFINITY : Float.POSITIVE_INFINITY;
177                 case DOUBLE:
178                     return min ? Double.NEGATIVE_INFINITY : Double.POSITIVE_INFINITY;
179                 case STRING:
180                 case STRING_VAL:
181                     return min ? null : MAX_TERM;
182                 default:
183                     throw new UnsupportedOperationException("Unsupported reduced type: " + reducedType());
184                 }
185             } else {
186                 switch (reducedType()) {
187                 case INT:
188                     if (missingValue instanceof Number) {
189                         return ((Number) missingValue).intValue();
190                     } else {
191                         return Integer.parseInt(missingValue.toString());
192                     }
193                 case LONG:
194                     if (missingValue instanceof Number) {
195                         return ((Number) missingValue).longValue();
196                     } else {
197                         return Long.parseLong(missingValue.toString());
198                     }
199                 case FLOAT:
200                     if (missingValue instanceof Number) {
201                         return ((Number) missingValue).floatValue();
202                     } else {
203                         return Float.parseFloat(missingValue.toString());
204                     }
205                 case DOUBLE:
206                     if (missingValue instanceof Number) {
207                         return ((Number) missingValue).doubleValue();
208                     } else {
209                         return Double.parseDouble(missingValue.toString());
210                     }
211                 case STRING:
212                 case STRING_VAL:
213                     if (missingValue instanceof BytesRef) {
214                         return (BytesRef) missingValue;
215                     } else if (missingValue instanceof byte[]) {
216                         return new BytesRef((byte[]) missingValue);
217                     } else {
218                         return new BytesRef(missingValue.toString());
219                     }
220                 default:
221                     throw new UnsupportedOperationException("Unsupported reduced type: " + reducedType());
222                 }
223             }
224         }
225 
226         public abstract SortField.Type reducedType();
227     }
