106     /**
107      * Enum defining the different dynamic settings for scripting, either
108      * ONLY_DISK_ALLOWED (scripts must be placed on disk), EVERYTHING_ALLOWED
109      * (all dynamic scripting is enabled), or SANDBOXED_ONLY (only sandboxed
110      * scripting languages are allowed)
111      */
112     enum DynamicScriptDisabling {
113         EVERYTHING_ALLOWED,
114         ONLY_DISK_ALLOWED,
115         SANDBOXED_ONLY;
116 
117         static DynamicScriptDisabling parse(String s) {
118             switch (s.toLowerCase(Locale.ROOT)) {
119                 // true for "disable_dynamic" means only on-disk scripts are enabled
120                 case "true":
121                 case "all":
122                     return ONLY_DISK_ALLOWED;
123                 // false for "disable_dynamic" means all scripts are enabled
124                 case "false":
125                 case "none":
126                     return EVERYTHING_ALLOWED;
127                 // only sandboxed scripting is enabled
128                 case "sandbox":
129                 case "sandboxed":
130                     return SANDBOXED_ONLY;
131                 default:
132                     throw new ElasticsearchIllegalArgumentException("Unrecognized script allowance setting: [" + s + "]");
133             }
134         }
135     }

142     @Inject
143     public ScriptService(Settings settings, Environment env, Set<ScriptEngineService> scriptEngines,
144                          ResourceWatcherService resourceWatcherService, NodeSettingsService nodeSettingsService) throws IOException {
145         super(settings);
146 
147         int cacheMaxSize = settings.getAsInt(SCRIPT_CACHE_SIZE_SETTING, 100);
148         TimeValue cacheExpire = settings.getAsTime(SCRIPT_CACHE_EXPIRE_SETTING, null);
149         logger.debug("using script cache with max_size [{}], expire [{}]", cacheMaxSize, cacheExpire);
150 
151         this.defaultLang = settings.get(DEFAULT_SCRIPTING_LANGUAGE_SETTING, DEFAULT_LANG);
152         this.dynamicScriptingDisabled = DynamicScriptDisabling.parse(settings.get(DISABLE_DYNAMIC_SCRIPTING_SETTING, DISABLE_DYNAMIC_SCRIPTING_DEFAULT));
153 
154         CacheBuilder cacheBuilder = CacheBuilder.newBuilder();
155         if (cacheMaxSize >= 0) {
156             cacheBuilder.maximumSize(cacheMaxSize);
157         }
158         if (cacheExpire != null) {
159             cacheBuilder.expireAfterAccess(cacheExpire.nanos(), TimeUnit.NANOSECONDS);
160         }
161         cacheBuilder.removalListener(new ScriptCacheRemovalListener());
162         this.cache = cacheBuilder.build();
163 
164         ImmutableMap.Builder<String, ScriptEngineService> builder = ImmutableMap.builder();
165         for (ScriptEngineService scriptEngine : scriptEngines) {
166             for (String type : scriptEngine.types()) {
167                 builder.put(type, scriptEngine);
168             }
169         }
170         this.scriptEngines = builder.build();
171 
172         // add file watcher for static scripts
173         scriptsDirectory = env.configFile().resolve("scripts");
174         if (logger.isTraceEnabled()) {
175             logger.trace("Using scripts directory [{}] ", scriptsDirectory);
176         }
177         this.fileWatcher = new FileWatcher(scriptsDirectory);
178         fileWatcher.addListener(new ScriptChangesListener());
179 
180         if (settings.getAsBoolean(SCRIPT_AUTO_RELOAD_ENABLED_SETTING, true)) {
181             // automatic reload is enabled - register scripts
182             resourceWatcherService.add(fileWatcher);
183         } else {
184             // automatic reload is disable just load scripts once
185             fileWatcher.init();
186         }
187         nodeSettingsService.addListener(new ApplySettings());
188     }

202     /**
203      * Clear both the in memory and on disk compiled script caches. Files on
204      * disk will be treated as if they are new and recompiled.
205      * */
206     public void clearCache() {
207         logger.debug("clearing script cache");
208         // Clear the in-memory script caches
209         this.cache.invalidateAll();
210         this.cache.cleanUp();
211         // Clear the cache of on-disk scripts
212         this.staticCache.clear();
213         // Clear the file watcher's state so it re-compiles on-disk scripts
214         this.fileWatcher.clearState();
215     }

218         ScriptEngineService scriptEngineService = scriptEngines.get(lang);
219         if (scriptEngineService == null) {
220             throw new ElasticsearchIllegalArgumentException("script_lang not supported [" + lang + "]");
221         }
222         return scriptEngineService;

225     /**
226      * Compiles a script straight-away, or returns the previously compiled and cached script, without checking if it can be executed based on settings.
227      */
228     public CompiledScript compile(String lang,  String script, ScriptType scriptType) {
229         if (lang == null) {
230             lang = defaultLang;
231         }
232         if (logger.isTraceEnabled()) {
233             logger.trace("Compiling lang: [{}] type: [{}] script: {}", lang, scriptType, script);
234         }
235 
236         if (scriptType == ScriptType.FILE) {
237             CompiledScript compiled = staticCache.get(script); //On disk scripts will be loaded into the staticCache by the listener
238             if (compiled == null) {
239                 throw new ElasticsearchIllegalArgumentException("Unable to find on disk script " + script);
240             }
241             return compiled;
242         }
243 
244         ScriptEngineService scriptEngineService = getScriptEngineService(lang);
245         verifyDynamicScripting(lang, scriptEngineService);
246 
247         if (scriptType == ScriptType.INDEXED) {
248             if (client == null) {
249                 throw new ElasticsearchIllegalArgumentException("Got an indexed script with no Client registered.");
250             }
251             final IndexedScript indexedScript = new IndexedScript(lang, script);
252             script = getScriptFromIndex(client, indexedScript.lang, indexedScript.id);
253         }
254 
255         CacheKey cacheKey = new CacheKey(lang, script);
256         CompiledScript compiled = cache.getIfPresent(cacheKey);
257         if (compiled == null) {
258             //Either an un-cached inline script or an indexed script
259             // not the end of the world if we compile it twice...
260             compiled = new CompiledScript(lang, scriptEngineService.compile(script));
261             //Since the cache key is the script content itself we don't need to
262             //invalidate/check the cache if an indexed script changes.
263             cache.put(cacheKey, compiled);
264         }
265         return compiled;
266     }

268     private void verifyDynamicScripting(String lang, ScriptEngineService scriptEngineService) {
269         if (!dynamicScriptEnabled(lang, scriptEngineService)) {
270             throw new ScriptException("dynamic scripting for [" + lang + "] disabled");
271         }
272     }

274     public void queryScriptIndex(GetIndexedScriptRequest request, final ActionListener<GetResponse> listener) {
275         String scriptLang = validateScriptLanguage(request.scriptLang());
276         GetRequest getRequest = new GetRequest(request, SCRIPT_INDEX).type(scriptLang).id(request.id())
277                 .version(request.version()).versionType(request.versionType())
278                 .preference("_local"); //Set preference for no forking
279         client.get(getRequest, listener);
280     }

282     private String validateScriptLanguage(String scriptLang) {
283         if (scriptLang == null) {
284             scriptLang = defaultLang;
285         } else if (!scriptEngines.containsKey(scriptLang)) {
286             throw new ElasticsearchIllegalArgumentException("script_lang not supported ["+scriptLang+"]");
287         }
288         return scriptLang;
289     }

291     private String getScriptFromIndex(Client client, String scriptLang, String id) {
292         scriptLang = validateScriptLanguage(scriptLang);
293         GetRequest getRequest = new GetRequest(SCRIPT_INDEX, scriptLang, id);
294         GetResponse responseFields = client.get(getRequest).actionGet();
295         if (responseFields.isExists()) {
296             return getScriptFromResponse(responseFields);
297         }
298         throw new ElasticsearchIllegalArgumentException("Unable to find script [" + SCRIPT_INDEX + "/"
299                 + scriptLang + "/" + id + "]");
300     }

302     private void validate(BytesReference scriptBytes, String scriptLang) {
303         try {
304             XContentParser parser = XContentFactory.xContent(scriptBytes).createParser(scriptBytes);
305             TemplateQueryParser.TemplateContext context = TemplateQueryParser.parse(parser, "params", "script", "template");
306             if (Strings.hasLength(context.template())){
307                 //Just try and compile it
308                 //This will have the benefit of also adding the script to the cache if it compiles
309                 try {
310                     CompiledScript compiledScript = compile(scriptLang, context.template(), ScriptType.INLINE);
311                     if (compiledScript == null) {
312                         throw new ElasticsearchIllegalArgumentException("Unable to parse [" + context.template() +
313                                 "] lang [" + scriptLang + "] (ScriptService.compile returned null)");
314                     }
315                 } catch (Exception e) {
316                     throw new ElasticsearchIllegalArgumentException("Unable to parse [" + context.template() +
317                             "] lang [" + scriptLang + "]", e);
318                 }
319             } else {
320                 throw new ElasticsearchIllegalArgumentException("Unable to find script in : " + scriptBytes.toUtf8());
321             }
322         } catch (IOException e) {
323             throw new ElasticsearchIllegalArgumentException("failed to parse template script", e);
324         }
325     }

327     public void putScriptToIndex(PutIndexedScriptRequest request, ActionListener<IndexResponse> listener) {
328         String scriptLang = validateScriptLanguage(request.scriptLang());
329         //verify that the script compiles
330         validate(request.safeSource(), scriptLang);
331 
332         IndexRequest indexRequest = new IndexRequest(request).index(SCRIPT_INDEX).type(scriptLang).id(request.id())
333                 .version(request.version()).versionType(request.versionType())
334                 .source(request.safeSource(), true).opType(request.opType()).refresh(true); //Always refresh after indexing a template
335         client.index(indexRequest, listener);
336     }

338     public void deleteScriptFromIndex(DeleteIndexedScriptRequest request, ActionListener<DeleteResponse> listener) {
339         String scriptLang = validateScriptLanguage(request.scriptLang());
340         DeleteRequest deleteRequest = new DeleteRequest(request).index(SCRIPT_INDEX).type(scriptLang).id(request.id())
341                 .refresh(true).version(request.version()).versionType(request.versionType());
342         client.delete(deleteRequest, listener);

345     @SuppressWarnings("unchecked")
346     public static String getScriptFromResponse(GetResponse responseFields) {
347         Map<String, Object> source = responseFields.getSourceAsMap();
348         if (source.containsKey("template")) {
349             try {
350                 XContentBuilder builder = XContentFactory.contentBuilder(XContentType.JSON);
351                 Object template = source.get("template");
352                 if (template instanceof Map ){
353                     builder.map((Map<String, Object>)template);
354                     return builder.string();
355                 } else {
356                     return template.toString();
357                 }
358             } catch (IOException | ClassCastException e) {
359                 throw new ElasticsearchIllegalStateException("Unable to parse "  + responseFields.getSourceAsString() + " as json", e);
360             }
361         } else  if (source.containsKey("script")) {
362             return source.get("script").toString();
363         } else {
364             try {
365                 XContentBuilder builder = XContentFactory.contentBuilder(XContentType.JSON);
366                 builder.map(responseFields.getSource());
367                 return builder.string();
368             } catch (IOException|ClassCastException e) {
369                 throw new ElasticsearchIllegalStateException("Unable to parse "  + responseFields.getSourceAsString() + " as json", e);
370             }
371         }
372     }

396     private boolean dynamicScriptEnabled(String lang, ScriptEngineService scriptEngineService) {
397         // Templating languages (mustache) and native scripts are always
398         // allowed, "native" executions are registered through plugins
399         if (this.dynamicScriptingDisabled == DynamicScriptDisabling.EVERYTHING_ALLOWED ||
400                 NativeScriptEngineService.NAME.equals(lang) || MustacheScriptEngineService.NAME.equals(lang)) {
401             return true;
402         }
403         if (this.dynamicScriptingDisabled == DynamicScriptDisabling.ONLY_DISK_ALLOWED) {
404             return false;
405         }
406         return scriptEngineService.sandboxed();
407     }

409     /**
410      * A small listener for the script cache that calls each
411      * {@code ScriptEngineService}'s {@code scriptRemoved} method when the
412      * script has been removed from the cache
413      */
414     private class ScriptCacheRemovalListener implements RemovalListener<CacheKey, CompiledScript> {
415 
416         @Override
417         public void onRemoval(RemovalNotification<CacheKey, CompiledScript> notification) {
418             if (logger.isDebugEnabled()) {
419                 logger.debug("notifying script services of script removal due to: [{}]", notification.getCause());
420             }
421             for (ScriptEngineService service : scriptEngines.values()) {
422                 try {
423                     service.scriptRemoved(notification.getValue());
424                 } catch (Exception e) {
425                     logger.warn("exception calling script removal listener for script service", e);
426                     // We don't rethrow because Guava would just catch the
427                     // exception and log it, which we have already done
428                 }
429             }
430         }
431     }

433     private class ScriptChangesListener extends FileChangesListener {
434 
435         private Tuple<String, String> scriptNameExt(Path file) {
436             Path scriptPath = scriptsDirectory.relativize(file);
437             int extIndex = scriptPath.toString().lastIndexOf('.');
438             if (extIndex != -1) {
439                 String ext = scriptPath.toString().substring(extIndex + 1);
440                 String scriptName = scriptPath.toString().substring(0, extIndex).replace(scriptPath.getFileSystem().getSeparator(), "_");
441                 return new Tuple<>(scriptName, ext);
442             } else {
443                 return null;
444             }
445         }
446 
447         @Override
448         public void onFileInit(Path file) {
449             if (logger.isTraceEnabled()) {
450                 logger.trace("Loading script file : [{}]", file);
451             }
452             Tuple<String, String> scriptNameExt = scriptNameExt(file);
453             if (scriptNameExt != null) {
454                 boolean found = false;
455                 for (ScriptEngineService engineService : scriptEngines.values()) {
456                     for (String s : engineService.extensions()) {
457                         if (s.equals(scriptNameExt.v2())) {
458                             found = true;
459                             try {
460                                 logger.info("compiling script file [{}]", file.toAbsolutePath());
461                                 String script = Streams.copyToString(new InputStreamReader(Files.newInputStream(file), Charsets.UTF_8));
462                                 staticCache.put(scriptNameExt.v1(), new CompiledScript(engineService.types()[0], engineService.compile(script)));
463                             } catch (Throwable e) {
464                                 logger.warn("failed to load/compile script [{}]", e, scriptNameExt.v1());
465                             }
466                             break;
467                         }
468                     }
469                     if (found) {
470                         break;
471                     }
472                 }
473                 if (!found) {
474                     logger.warn("no script engine found for [{}]", scriptNameExt.v2());
475                 }
476             }
477         }
478 
479         @Override
480         public void onFileCreated(Path file) {
481             onFileInit(file);
482         }
483 
484         @Override
485         public void onFileDeleted(Path file) {
486             Tuple<String, String> scriptNameExt = scriptNameExt(file);
487             if (scriptNameExt != null) {
488                 logger.info("removing script file [{}]", file.toAbsolutePath());
489                 staticCache.remove(scriptNameExt.v1());
490             }
491         }
492 
493         @Override
494         public void onFileChanged(Path file) {
495             onFileInit(file);
496         }

500     /**
501      * The type of a script, more specifically where it gets loaded from:
502      * - provided dynamically at request time
503      * - loaded from an index
504      * - loaded from file
505      */
506     public static enum ScriptType {
507 
508         INLINE,
509         INDEXED,
510         FILE;
511 
512         private static final int INLINE_VAL = 0;
513         private static final int INDEXED_VAL = 1;
514         private static final int FILE_VAL = 2;
515 
516         public static ScriptType readFrom(StreamInput in) throws IOException {
517             int scriptTypeVal = in.readVInt();
518             switch (scriptTypeVal) {
519                 case INDEXED_VAL:
520                     return INDEXED;
521                 case INLINE_VAL:
522                     return INLINE;
523                 case FILE_VAL:
524                     return FILE;
525                 default:
526                     throw new ElasticsearchIllegalArgumentException("Unexpected value read for ScriptType got [" + scriptTypeVal +
527                             "] expected one of [" + INLINE_VAL + "," + INDEXED_VAL + "," + FILE_VAL + "]");
528             }
529         }
530 
531         public static void writeTo(ScriptType scriptType, StreamOutput out) throws IOException{
532             if (scriptType != null) {
533                 switch (scriptType){
534                     case INDEXED:
535                         out.writeVInt(INDEXED_VAL);
536                         return;
537                     case INLINE:
538                         out.writeVInt(INLINE_VAL);
539                         return;
540                     case FILE:
541                         out.writeVInt(FILE_VAL);
542                         return;
543                     default:
544                         throw new ElasticsearchIllegalStateException("Unknown ScriptType " + scriptType);
545                 }
546             } else {
547                 out.writeVInt(INLINE_VAL); //Default to inline
548             }
549         }
550     }

552     private static class CacheKey {
553         public final String lang;
554         public final String script;
555 
556         public CacheKey(String lang, String script) {
557             this.lang = lang;
558             this.script = script;
559         }
560 
561         @Override
562         public boolean equals(Object o) {
563             if (! (o instanceof  CacheKey)) {
564                 return false;
565             }
566             CacheKey other = (CacheKey) o;
567             return lang.equals(other.lang) && script.equals(other.script);
568         }
569 
570         @Override
571         public int hashCode() {
572             return lang.hashCode() + 31 * script.hashCode();
573         }
574     }

576     private static class IndexedScript {
577         private final String lang;
578         private final String id;
579 
580         IndexedScript(String lang, String script) {
581             this.lang = lang;
582             final String[] parts = script.split("/");
583             if (parts.length == 1) {
584                 this.id = script;
585             } else {
586                 if (parts.length != 3) {
587                     throw new ElasticsearchIllegalArgumentException("Illegal index script format [" + script + "]" +
588                             " should be /lang/id");
589                 } else {
590                     if (!parts[1].equals(this.lang)) {
591                         throw new ElasticsearchIllegalStateException("Conflicting script language, found [" + parts[1] + "] expected + ["+ this.lang + "]");
592                     }
593                     this.id = parts[2];
594                 }
595             }
596         }
597     }

599     private class ApplySettings implements NodeSettingsService.Listener {
600         @Override
601         public void onRefreshSettings(Settings settings) {
602             GroovyScriptEngineService engine = (GroovyScriptEngineService) ScriptService.this.scriptEngines.get("groovy");
603             if (engine != null) {
604                 String[] patches = settings.getAsArray(GroovyScriptEngineService.GROOVY_SCRIPT_BLACKLIST_PATCH, Strings.EMPTY_ARRAY);
605                 boolean blacklistChanged = engine.addToBlacklist(patches);
606                 if (blacklistChanged) {
607                     logger.info("adding {} to [{}], new blacklisted methods: {}", patches,
608                             GroovyScriptEngineService.GROOVY_SCRIPT_BLACKLIST_PATCH, engine.blacklistAdditions());
609                     engine.reloadConfig();
610                     // Because the GroovyScriptEngineService knows nothing about the
611                     // cache, we need to clear it here if the setting changes
612                     ScriptService.this.clearCache();
613                 }
614             }
615         }
