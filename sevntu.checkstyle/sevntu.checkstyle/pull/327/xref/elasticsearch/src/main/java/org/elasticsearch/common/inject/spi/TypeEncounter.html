17  package org.elasticsearch.common.inject.spi;
18  
19  import org.elasticsearch.common.inject.Key;
20  import org.elasticsearch.common.inject.MembersInjector;
21  import org.elasticsearch.common.inject.Provider;
22  import org.elasticsearch.common.inject.TypeLiteral;
23  
24  /**
25   * Context of an injectable type encounter. Enables reporting errors, registering injection
26   * listeners and binding method interceptors for injectable type {@code I}. It is an error to use
27   * an encounter after the {@link TypeListener#hear(TypeLiteral, TypeEncounter) hear()} method has
28   * returned.
29   *
30   * @param <I> the injectable type encountered
31   * @since 2.0
32   */
33  public interface TypeEncounter<I> {
34  
35      /**
36       * Records an error message for type {@code I} which will be presented to the user at a later
37       * time. Unlike throwing an exception, this enable us to continue configuring the Injector and
38       * discover more errors. Uses {@link String#format(String, Object[])} to insert the arguments
39       * into the message.
40       */
41      void addError(String message, Object... arguments);
42  
43      /**
44       * Records an exception for type {@code I}, the full details of which will be logged, and the
45       * message of which will be presented to the user at a later time. If your type listener calls
46       * something that you worry may fail, you should catch the exception and pass it to this method.
47       */
48      void addError(Throwable t);
49  
50      /**
51       * Records an error message to be presented to the user at a later time.
52       */
53      void addError(Message message);
54  
55      /**
56       * Returns the provider used to obtain instances for the given injection key. The returned
57       * provider will not be valid until the injector has been created. The provider will throw an
58       * {@code IllegalStateException} if you try to use it beforehand.
59       */
60      <T> Provider<T> getProvider(Key<T> key);
61  
62      /**
63       * Returns the provider used to obtain instances for the given injection type. The returned
64       * provider will not be valid until the injetor has been created. The provider will throw an
65       * {@code IllegalStateException} if you try to use it beforehand.
66       */
67      <T> Provider<T> getProvider(Class<T> type);
68  
69      /**
70       * Returns the members injector used to inject dependencies into methods and fields on instances
71       * of the given type {@code T}. The returned members injector will not be valid until the main
72       * injector has been created. The members injector will throw an {@code IllegalStateException}
73       * if you try to use it beforehand.
74       *
75       * @param typeLiteral type to get members injector for
76       */
77      <T> MembersInjector<T> getMembersInjector(TypeLiteral<T> typeLiteral);
78  
79      /**
80       * Returns the members injector used to inject dependencies into methods and fields on instances
81       * of the given type {@code T}. The returned members injector will not be valid until the main
82       * injector has been created. The members injector will throw an {@code IllegalStateException}
83       * if you try to use it beforehand.
84       *
85       * @param type type to get members injector for
86       */
87      <T> MembersInjector<T> getMembersInjector(Class<T> type);
88  
89      /**
90       * Registers a members injector for type {@code I}. Guice will use the members injector after its
91       * performed its own injections on an instance of {@code I}.
92       */
93      void register(MembersInjector<? super I> membersInjector);
94  
95      /**
96       * Registers an injection listener for type {@code I}. Guice will notify the listener after all
97       * injections have been performed on an instance of {@code I}.
98       */
99      void register(InjectionListener<? super I> listener);
100 }
