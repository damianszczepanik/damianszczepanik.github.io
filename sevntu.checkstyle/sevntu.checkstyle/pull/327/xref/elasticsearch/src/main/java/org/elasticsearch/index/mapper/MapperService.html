216     private URL getMappingUrl(Settings indexSettings, Environment environment, String mappingLocation, String configString, String resourceLocation) {
217         URL mappingUrl;
218         if (mappingLocation == null) {
219             try {
220                 mappingUrl = environment.resolveConfig(configString);
221             } catch (FailedToResolveConfigException e) {
222                 // not there, default to the built in one
223                 mappingUrl = indexSettings.getClassLoader().getResource(resourceLocation);
224                 if (mappingUrl == null) {
225                     mappingUrl = MapperService.class.getClassLoader().getResource(resourceLocation);
226                 }
227             }
228         } else {
229             try {
230                 mappingUrl = environment.resolveConfig(mappingLocation);
231             } catch (FailedToResolveConfigException e) {
232                 // not there, default to the built in one
233                 try {
234                     mappingUrl = Paths.get(mappingLocation).toUri().toURL();
235                 } catch (MalformedURLException e1) {
236                     throw new FailedToResolveConfigException("Failed to resolve dynamic mapping location [" + mappingLocation + "]");
237                 }
238             }
239         }
240         return mappingUrl;
241     }

253     /**
254      * returns an immutable iterator over current document mappers.
255      *
256      * @param includingDefaultMapping indicates whether the iterator should contain the {@link #DEFAULT_MAPPING} document mapper.
257      *                                As is this not really an active type, you would typically set this to false
258      */
259     public Iterable<DocumentMapper> docMappers(final boolean includingDefaultMapping) {
260         return  new Iterable<DocumentMapper>() {
261             @Override
262             public Iterator<DocumentMapper> iterator() {
263                 final Iterator<DocumentMapper> iterator;
264                 if (includingDefaultMapping) {
265                     iterator = mappers.values().iterator();
266                 } else {
267                     iterator = Iterators.filter(mappers.values().iterator(), NOT_A_DEFAULT_DOC_MAPPER);
268                 }
269                 return Iterators.unmodifiableIterator(iterator);
270             }
271         };
272     }

278         }

297     public DocumentMapper merge(String type, CompressedString mappingSource, boolean applyDefault) {
298         if (DEFAULT_MAPPING.equals(type)) {
299             // verify we can parse it
300             DocumentMapper mapper = documentParser.parseCompressed(type, mappingSource);
301             // still add it as a document mapper so we have it registered and, for example, persisted back into
302             // the cluster meta data if needed, or checked for existence
303             synchronized (typeMutex) {
304                 mappers = newMapBuilder(mappers).put(type, mapper).map();
305             }
306             try {
307                 defaultMappingSource = mappingSource.string();
308             } catch (IOException e) {
309                 throw new ElasticsearchGenerationException("failed to un-compress", e);
310             }
311             return mapper;
312         } else {
313             return merge(parse(type, mappingSource, applyDefault));
314         }
315     }

317     // never expose this to the outside world, we need to reparse the doc mapper so we get fresh
318     // instances of field mappers to properly remove existing doc mapper
319     private DocumentMapper merge(DocumentMapper mapper) {
320         synchronized (typeMutex) {
321             if (mapper.type().length() == 0) {
322                 throw new InvalidTypeNameException("mapping type name is empty");
323             }
324             if (mapper.type().charAt(0) == '_') {
325                 throw new InvalidTypeNameException("mapping type name [" + mapper.type() + "] can't start with '_'");
326             }
327             if (mapper.type().contains("#")) {
328                 throw new InvalidTypeNameException("mapping type name [" + mapper.type() + "] should not include '#' in it");
329             }
330             if (mapper.type().contains(",")) {
331                 throw new InvalidTypeNameException("mapping type name [" + mapper.type() + "] should not include ',' in it");
332             }
333             if (mapper.type().contains(".") && !PercolatorService.TYPE_NAME.equals(mapper.type())) {
334                 logger.warn("Type [{}] contains a '.', it is recommended not to include it within a type name", mapper.type());
335             }
336             // we can add new field/object mappers while the old ones are there
337             // since we get new instances of those, and when we remove, we remove
338             // by instance equality
339             DocumentMapper oldMapper = mappers.get(mapper.type());
340 
341             if (oldMapper != null) {
342                 DocumentMapper.MergeResult result = oldMapper.merge(mapper, mergeFlags().simulate(false));
343                 if (result.hasConflicts()) {
344                     // TODO: What should we do???
345                     if (logger.isDebugEnabled()) {
346                         logger.debug("merging mapping for type [{}] resulted in conflicts: [{}]", mapper.type(), Arrays.toString(result.conflicts()));
347                     }
348                 }
349                 fieldDataService.onMappingUpdate();
350                 return oldMapper;
351             } else {
352                 FieldMapperListener.Aggregator fieldMappersAgg = new FieldMapperListener.Aggregator();
353                 mapper.traverse(fieldMappersAgg);
354                 addFieldMappers(fieldMappersAgg.mappers);
355                 mapper.addFieldMapperListener(fieldMapperListener);
356 
357                 ObjectMapperListener.Aggregator objectMappersAgg = new ObjectMapperListener.Aggregator();
358                 mapper.traverse(objectMappersAgg);
359                 addObjectMappers(objectMappersAgg.mappers.toArray(new ObjectMapper[objectMappersAgg.mappers.size()]));
360                 mapper.addObjectMapperListener(objectMapperListener);
361 
362                 for (DocumentTypeListener typeListener : typeListeners) {
363                     typeListener.beforeCreate(mapper);
364                 }
365                 mappers = newMapBuilder(mappers).put(mapper.type(), mapper).map();
366                 return mapper;
367             }
368         }
369     }

371     private void addObjectMappers(ObjectMapper[] objectMappers) {
372         synchronized (mappersMutex) {
373             ImmutableOpenMap.Builder<String, ObjectMappers> fullPathObjectMappers = ImmutableOpenMap.builder(this.fullPathObjectMappers);
374             for (ObjectMapper objectMapper : objectMappers) {
375                 ObjectMappers mappers = fullPathObjectMappers.get(objectMapper.fullPath());
376                 if (mappers == null) {
377                     mappers = new ObjectMappers(objectMapper);
378                 } else {
379                     mappers = mappers.concat(objectMapper);
380                 }
381                 fullPathObjectMappers.put(objectMapper.fullPath(), mappers);
382                 // update the hasNested flag
383                 if (objectMapper.nested().isNested()) {
384                     hasNested = true;
385                 }
386             }
387             this.fullPathObjectMappers = fullPathObjectMappers.build();
388         }
389     }

397     public void remove(String type) {
398         synchronized (typeMutex) {
399             DocumentMapper docMapper = mappers.get(type);
400             if (docMapper == null) {
401                 return;
402             }
403             docMapper.close();
404             mappers = newMapBuilder(mappers).remove(type).map();
405             removeObjectAndFieldMappers(docMapper);
406             for (DocumentTypeListener typeListener : typeListeners) {
407                 typeListener.afterRemove(docMapper);
408             }
409         }
410     }

412     private void removeObjectAndFieldMappers(DocumentMapper docMapper) {
413         synchronized (mappersMutex) {
414             fieldMappers = fieldMappers.copyAndRemoveAll(docMapper.mappers());
415 
416             ImmutableOpenMap.Builder<String, ObjectMappers> fullPathObjectMappers = ImmutableOpenMap.builder(this.fullPathObjectMappers);
417             for (ObjectMapper mapper : docMapper.objectMappers().values()) {
418                 ObjectMappers mappers = fullPathObjectMappers.get(mapper.fullPath());
419                 if (mappers != null) {
420                     mappers = mappers.remove(mapper);
421                     if (mappers.isEmpty()) {
422                         fullPathObjectMappers.remove(mapper.fullPath());
423                     } else {
424                         fullPathObjectMappers.put(mapper.fullPath(), mappers);
425                     }
426                 }
427             }
428 
429             this.fullPathObjectMappers = fullPathObjectMappers.build();
430         }

455     /**
456      * Returns the document mapper created, including if the document mapper ended up
457      * being actually created or not in the second tuple value.
458      */
459     public Tuple<DocumentMapper, Boolean> documentMapperWithAutoCreate(String type) {
460         DocumentMapper mapper = mappers.get(type);
461         if (mapper != null) {
462             return Tuple.tuple(mapper, Boolean.FALSE);
463         }
464         if (!dynamic) {
465             throw new TypeMissingException(index, type, "trying to auto create mapping, but dynamic mapping is disabled");
466         }
467         // go ahead and dynamically create it
468         synchronized (typeMutex) {
469             mapper = mappers.get(type);
470             if (mapper != null) {
471                 return Tuple.tuple(mapper, Boolean.FALSE);
472             }
473             merge(type, null, true);
474             return Tuple.tuple(mappers.get(type), Boolean.TRUE);
475         }
476     }

478     /**
479      * A filter for search. If a filter is required, will return it, otherwise, will return <tt>null</tt>.
480      */
481     @Nullable
482     public Filter searchFilter(String... types) {
483         boolean filterPercolateType = hasMapping(PercolatorService.TYPE_NAME);
484         if (types != null && filterPercolateType) {
485             for (String type : types) {
486                 if (PercolatorService.TYPE_NAME.equals(type)) {
487                     filterPercolateType = false;
488                     break;
489                 }
490             }
491         }
492         Filter excludePercolatorType = null;
493         if (filterPercolateType) {
494             excludePercolatorType = new NotFilter(documentMapper(PercolatorService.TYPE_NAME).typeFilter());
495         }
496 
497         if (types == null || types.length == 0) {
498             if (hasNested && filterPercolateType) {
499                 return new AndFilter(ImmutableList.of(excludePercolatorType, NonNestedDocsFilter.INSTANCE));
500             } else if (hasNested) {
501                 return NonNestedDocsFilter.INSTANCE;
502             } else if (filterPercolateType) {
503                 return excludePercolatorType;
504             } else {
505                 return null;
506             }
507         }
508         // if we filter by types, we don't need to filter by non nested docs
509         // since they have different types (starting with __)
510         if (types.length == 1) {
511             DocumentMapper docMapper = documentMapper(types[0]);
512             Filter filter = docMapper != null ? docMapper.typeFilter() : new TermFilter(new Term(types[0]));
513             if (hasNested) {
514                 return new AndFilter(ImmutableList.of(filter, NonNestedDocsFilter.INSTANCE));
515             } else {
516                 return filter;
517             }
518         }
519         // see if we can use terms filter
520         boolean useTermsFilter = true;
521         for (String type : types) {
522             DocumentMapper docMapper = documentMapper(type);
523             if (docMapper == null) {
524                 useTermsFilter = false;
525                 break;
526             }
527             if (docMapper.typeMapper().fieldType().indexOptions() == IndexOptions.NONE) {
528                 useTermsFilter = false;
529                 break;
530             }
531         }
532 
533         // We only use terms filter is there is a type filter, this means we don't need to check for hasNested here
534         if (useTermsFilter) {
535             BytesRef[] typesBytes = new BytesRef[types.length];
536             for (int i = 0; i < typesBytes.length; i++) {
537                 typesBytes[i] = new BytesRef(types[i]);
538             }
539             TermsFilter termsFilter = new TermsFilter(TypeFieldMapper.NAME, typesBytes);
540             if (filterPercolateType) {
541                 return new AndFilter(ImmutableList.of(excludePercolatorType, termsFilter));
542             } else {
543                 return termsFilter;
544             }
545         } else {
546             // Current bool filter requires that at least one should clause matches, even with a must clause.
547             XBooleanFilter bool = new XBooleanFilter();
548             for (String type : types) {
549                 DocumentMapper docMapper = documentMapper(type);
550                 if (docMapper == null) {
551                     bool.add(new FilterClause(new TermFilter(new Term(TypeFieldMapper.NAME, type)), BooleanClause.Occur.SHOULD));
552                 } else {
553                     bool.add(new FilterClause(docMapper.typeFilter(), BooleanClause.Occur.SHOULD));
554                 }
555             }
556             if (filterPercolateType) {
557                 bool.add(excludePercolatorType, BooleanClause.Occur.MUST);
558             }
559             if (hasNested) {
560                 bool.add(NonNestedDocsFilter.INSTANCE, BooleanClause.Occur.MUST);
561             }
562 
563             return bool;
564         }
565     }

603     /**
604      * Returns all the fields that match the given pattern, with an optional narrowing
605      * based on a list of types.
606      */
607     public List<String> simpleMatchToIndexNames(String pattern, @Nullable String[] types) {
608         if (Regex.isSimpleMatchPattern(pattern) == false) {
609             // no wildcards
610             return ImmutableList.of(pattern);
611         }
612         
613         if (types == null || types.length == 0 || types.length == 1 && types[0].equals("_all")) {
614             return fieldMappers.simpleMatchToIndexNames(pattern);
615         }
616 
617         List<String> fields = Lists.newArrayList();
618         for (String type : types) {
619             DocumentMapper possibleDocMapper = mappers.get(type);
620             if (possibleDocMapper != null) {
621                 for (String indexName : possibleDocMapper.mappers().simpleMatchToIndexNames(pattern)) {
622                     fields.add(indexName);
623                 }
624             }
625         }
626         return fields;
627     }

630         if (types == null || types.length == 0 || types.length == 1 && types[0].equals("_all")) {
631             ObjectMappers mappers = objectMapper(smartName);
632             if (mappers != null) {
633                 return new SmartNameObjectMapper(mappers.mapper(), guessDocMapper(smartName));

641                 if (mapper != null) {
642                     return new SmartNameObjectMapper(mapper, possibleDocMapper);
643                 }
644             }
645         }
646         return null;
647     }

728     /**
729      * Returns smart field mappers based on a smart name. A smart name is one that can optionally be prefixed
730      * with a type (and then a '.'). If it is, then the {@link MapperService.SmartNameFieldMappers}
731      * will have the doc mapper set.
732      * <p/>
733      * <p>It also (without the optional type prefix) try and find the {@link FieldMappers} for the specific
734      * name. It will first try to find it based on the full name (with the dots if its a compound name). If
735      * it is not found, will try and find it based on the indexName (which can be controlled in the mapping),
736      * and last, will try it based no the name itself.
737      * <p/>
738      * <p>If nothing is found, returns null.
739      */
740     public SmartNameFieldMappers smartName(String smartName) {
741         FieldMappers fieldMappers = fullName(smartName);
742         if (fieldMappers != null) {
743             return new SmartNameFieldMappers(this, fieldMappers, null, false);
744         }
745         fieldMappers = indexName(smartName);
746         if (fieldMappers != null) {
747             return new SmartNameFieldMappers(this, fieldMappers, null, false);
748         }
749         return null;
750     }

752     /**
753      * Given a type (eg. long, string, ...), return an anonymous field mapper that can be used for search operations.
754      */
755     public FieldMapper<?> unmappedFieldMapper(String type) {
756         final ImmutableMap<String, FieldMapper<?>> unmappedFieldMappers = this.unmappedFieldMappers;
757         FieldMapper<?> mapper = unmappedFieldMappers.get(type);
758         if (mapper == null) {
759             final Mapper.TypeParser.ParserContext parserContext = documentMapperParser().parserContext();
760             Mapper.TypeParser typeParser = parserContext.typeParser(type);
761             if (typeParser == null) {
762                 throw new ElasticsearchIllegalArgumentException("No mapper found for type [" + type + "]");
763             }
764             final Mapper.Builder<?, ?> builder = typeParser.parse("__anonymous_" + type, ImmutableMap.<String, Object>of(), parserContext);
765             final BuilderContext builderContext = new BuilderContext(indexSettings, new ContentPath(1));
766             mapper = (FieldMapper<?>) builder.build(builderContext);
767 
768             // There is no need to synchronize writes here. In the case of concurrent access, we could just
769             // compute some mappers several times, which is not a big deal
770             this.unmappedFieldMappers = ImmutableMap.<String, FieldMapper<?>>builder()
771                     .putAll(unmappedFieldMappers)
772                     .put(type, mapper)
773                     .build();
774         }
775         return mapper;
776     }

786     /**
787      * Resolves the closest inherited {@link ObjectMapper} that is nested.
788      */
789     public ObjectMapper resolveClosestNestedObjectMapper(String fieldName) {
790         int indexOf = fieldName.lastIndexOf('.');
791         if (indexOf == -1) {
792             return null;
793         } else {
794             do {
795                 String objectPath = fieldName.substring(0, indexOf);
796                 ObjectMappers objectMappers = objectMapper(objectPath);
797                 if (objectMappers == null) {
798                     indexOf = objectPath.lastIndexOf('.');
799                     continue;
800                 }
801 
802                 if (objectMappers.hasNested()) {
803                     for (ObjectMapper objectMapper : objectMappers) {
804                         if (objectMapper.nested().isNested()) {
805                             return objectMapper;
806                         }
807                     }
808                 }
809 
810                 indexOf = objectPath.lastIndexOf('.');
811             } while (indexOf != -1);
812         }
813 
814         return null;
815     }
