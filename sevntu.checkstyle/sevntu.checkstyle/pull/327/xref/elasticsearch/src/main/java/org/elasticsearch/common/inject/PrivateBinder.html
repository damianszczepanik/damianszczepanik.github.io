17  package org.elasticsearch.common.inject;
18  
19  import org.elasticsearch.common.inject.binder.AnnotatedElementBuilder;
20  
21  /**
22   * Returns a binder whose configuration information is hidden from its environment by default. See
23   * {@link org.elasticsearch.common.inject.PrivateModule PrivateModule} for details.
24   *
25   * @author jessewilson@google.com (Jesse Wilson)
26   * @since 2.0
27   */
28  public interface PrivateBinder extends Binder {
29  
30      /**
31       * Makes the binding for {@code key} available to the enclosing environment
32       */
33      void expose(Key<?> key);
34  
35      /**
36       * Makes a binding for {@code type} available to the enclosing environment. Use {@link
37       * org.elasticsearch.common.inject.binder.AnnotatedElementBuilder#annotatedWith(Class) annotatedWith()} to expose {@code type} with a
38       * binding annotation.
39       */
40      AnnotatedElementBuilder expose(Class<?> type);
41  
42      /**
43       * Makes a binding for {@code type} available to the enclosing environment. Use {@link
44       * AnnotatedElementBuilder#annotatedWith(Class) annotatedWith()} to expose {@code type} with a
45       * binding annotation.
46       */
47      AnnotatedElementBuilder expose(TypeLiteral<?> type);
48  
49      @Override
50      PrivateBinder withSource(Object source);
51  
52      @Override
53      PrivateBinder skipSources(Class... classesToSkip);
54  }
