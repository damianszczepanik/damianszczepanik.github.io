17  package org.elasticsearch.common.inject;
18  
19  import org.elasticsearch.common.inject.InjectorImpl.MethodInvoker;
20  import org.elasticsearch.common.inject.internal.Errors;
21  import org.elasticsearch.common.inject.internal.ErrorsException;
22  import org.elasticsearch.common.inject.internal.InternalContext;
23  import org.elasticsearch.common.inject.spi.InjectionPoint;
24  
25  import java.lang.reflect.InvocationTargetException;
26  import java.lang.reflect.Method;
27  import java.lang.reflect.Modifier;
28  
29  /**
30   * Invokes an injectable method.
31   */
32  class SingleMethodInjector implements SingleMemberInjector {
33      final MethodInvoker methodInvoker;
34      final SingleParameterInjector<?>[] parameterInjectors;
35      final InjectionPoint injectionPoint;
36  
37      public SingleMethodInjector(InjectorImpl injector, InjectionPoint injectionPoint, Errors errors)
38              throws ErrorsException {
39          this.injectionPoint = injectionPoint;
40          final Method method = (Method) injectionPoint.getMember();
41          methodInvoker = createMethodInvoker(method);
42          parameterInjectors = injector.getParametersInjectors(injectionPoint.getDependencies(), errors);
43      }
44  
45      private MethodInvoker createMethodInvoker(final Method method) {
46  
47          // We can't use FastMethod if the method is private.
48          int modifiers = method.getModifiers();
49          if (!Modifier.isPrivate(modifiers) && !Modifier.isProtected(modifiers)) {
50          }
51  
52          if (!Modifier.isPublic(modifiers)) {
53              method.setAccessible(true);
54          }
55  
56          return new MethodInvoker() {
57              @Override
58              public Object invoke(Object target, Object... parameters)
59                      throws IllegalAccessException, InvocationTargetException {
60                  return method.invoke(target, parameters);
61              }
62          };
63      }
64  
65      @Override
66      public InjectionPoint getInjectionPoint() {
67          return injectionPoint;
68      }
69  
70      @Override
71      public void inject(Errors errors, InternalContext context, Object o) {
72          Object[] parameters;
73          try {
74              parameters = SingleParameterInjector.getAll(errors, context, parameterInjectors);
75          } catch (ErrorsException e) {
76              errors.merge(e.getErrors());
77              return;
78          }
79  
80          try {
81              methodInvoker.invoke(o, parameters);
82          } catch (IllegalAccessException e) {
83              throw new AssertionError(e); // a security manager is blocking us, we're hosed
84          } catch (InvocationTargetException userException) {
85              Throwable cause = userException.getCause() != null
86                      ? userException.getCause()
87                      : userException;
88              errors.withSource(injectionPoint).errorInjectingMethod(cause);
89          }
90      }
91  }
