20  package org.elasticsearch.cluster.routing.allocation.decider;
21  
22  import org.elasticsearch.cluster.routing.MutableShardRouting;
23  import org.elasticsearch.cluster.routing.RoutingNode;
24  import org.elasticsearch.cluster.routing.RoutingNodes;
25  import org.elasticsearch.cluster.routing.ShardRouting;
26  import org.elasticsearch.cluster.routing.allocation.RoutingAllocation;
27  import org.elasticsearch.common.inject.Inject;
28  import org.elasticsearch.common.settings.Settings;
29  
30  /**
31   * An allocation decider that prevents relocation or allocation from nodes
32   * that might not be version compatible. If we relocate from a node that runs
33   * a newer version than the node we relocate to this might cause {@link org.apache.lucene.index.IndexFormatTooNewException}
34   * on the lowest level since it might have already written segments that use a new postings format or codec that is not
35   * available on the target node.
36   */
37  public class NodeVersionAllocationDecider extends AllocationDecider {
38  
39      public static final String NAME = "node_version";
40  
41      @Inject
42      public NodeVersionAllocationDecider(Settings settings) {
43          super(settings);
44      }
45  
46      @Override
47      public Decision canAllocate(ShardRouting shardRouting, RoutingNode node, RoutingAllocation allocation) {
48          String sourceNodeId = shardRouting.currentNodeId();
49          /* if sourceNodeId is not null we do a relocation and just check the version of the node
50           * that we are currently allocate on. If not we are initializing and recover from primary.*/
51          if (sourceNodeId == null) { // we allocate - check primary
52              if (shardRouting.primary()) {
53                  // we are the primary we can allocate wherever
54                  return allocation.decision(Decision.YES, NAME, "primary shard can be allocated anywhere");
55              }
56              final MutableShardRouting primary = allocation.routingNodes().activePrimary(shardRouting);
57              if (primary == null) { // we have a primary - it's a start ;)
58                  return allocation.decision(Decision.YES, NAME, "no active primary shard yet");
59              }
60              sourceNodeId = primary.currentNodeId();
61          }
62          return isVersionCompatible(allocation.routingNodes(), sourceNodeId, node, allocation);
63  
64      }
65  
66      private Decision isVersionCompatible(final RoutingNodes routingNodes, final String sourceNodeId, final RoutingNode target, RoutingAllocation allocation) {
67          final RoutingNode source = routingNodes.node(sourceNodeId);
68          if (target.node().version().onOrAfter(source.node().version())) {
69              /* we can allocate if we can recover from a node that is younger or on the same version
70               * if the primary is already running on a newer version that won't work due to possible
71               * differences in the lucene index format etc.*/
72              return allocation.decision(Decision.YES, NAME, "target node version [%s] is same or newer than source node version [%s]",
73                      target.node().version(), source.node().version());
74          } else {
75              return allocation.decision(Decision.NO, NAME, "target node version [%s] is older than source node version [%s]",
76                      target.node().version(), source.node().version());
77          }
78      }
79  }
