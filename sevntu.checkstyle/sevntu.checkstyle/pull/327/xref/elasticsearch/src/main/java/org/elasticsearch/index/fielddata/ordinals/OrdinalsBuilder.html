36  /**
37   * Simple class to build document ID <-> ordinal mapping. Note: Ordinals are
38   * <tt>1</tt> based monotonically increasing positive integers. <tt>0</tt>
39   * donates the missing value in this context.
40   */
41  public final class OrdinalsBuilder implements Closeable {
42  
43      /**
44       * Whether to for the use of {@link MultiOrdinals} to store the ordinals for testing purposes.
45       */
46      public static final String FORCE_MULTI_ORDINALS = "force_multi_ordinals";
47  
48      /**
49       * Default acceptable overhead ratio. {@link OrdinalsBuilder} memory usage is mostly transient so it is likely a better trade-off to
50       * trade memory for speed in order to resize less often.
51       */
52      public static final float DEFAULT_ACCEPTABLE_OVERHEAD_RATIO = PackedInts.FAST;
53  
54      /**
55       * The following structure is used to store ordinals. The idea is to store ords on levels of increasing sizes. Level 0 stores
56       * 1 value and 1 pointer to level 1. Level 1 stores 2 values and 1 pointer to level 2, ..., Level n stores 2**n values and
57       * 1 pointer to level n+1. If at some point an ordinal or a pointer has 0 as a value, this means that there are no remaining
58       * values. On the first level, ordinals.get(docId) is the first ordinal for docId or 0 if the document has no ordinals. On
59       * subsequent levels, the first 2^level slots are reserved and all have 0 as a value.
60       * <pre>
61       * Example for an index of 3 docs (O=ordinal, P = pointer)
62       * Level 0:
63       *   ordinals           [1] [4] [2]
64       *   nextLevelSlices    2  0  1
65       * Level 1:
66       *   ordinals           [0  0] [2  0] [3  4]
67       *   nextLevelSlices    0  0  1
68       * Level 2:
69       *   ordinals           [0  0  0  0] [5  0  0  0]
70       *   nextLevelSlices    0  0
71       * </pre>
72       * On level 0, all documents have an ordinal: 0 has 1, 1 has 4 and 2 has 2 as a first ordinal, this means that we need to read
73       * nextLevelEntries to get the index of their ordinals on the next level. The entry for document 1 is 0, meaning that we have
74       * already read all its ordinals. On the contrary 0 and 2 have more ordinals which are stored at indices 2 and 1. Let's continue
75       * with document 2: it has 2 more ordinals on level 1: 3 and 4 and its next level index is 1 meaning that there are remaining
76       * ordinals on the next level. On level 2 at index 1, we can read [5  0  0  0] meaning that 5 is an ordinal as well, but the
77       * fact that it is followed by zeros means that there are no more ordinals. In the end, document 2 has 2, 3, 4 and 5 as ordinals.
78       * <p/>
79       * In addition to these structures, there is another array which stores the current position (level + slice + offset in the slice)
80       * in order to be able to append data in constant time.
81       */
82      private static class OrdinalsStore {
83  
84          private static final int PAGE_SIZE = 1 << 12;
85  
86          /**
87           * Number of slots at <code>level</code>
88           */
89          private static int numSlots(int level) {
90              return 1 << level;
91          }
92  
93          private static int slotsMask(int level) {
94              return numSlots(level) - 1;
95          }
96  
97          /**
98           * Encode the position for the given level and offset. The idea is to encode the level using unary coding in the lower bits and
99           * then the offset in the higher bits.
100          */
101         private static long position(int level, long offset) {
102             assert level >= 1;
103             return (1 << (level - 1)) | (offset << level);
104         }
105 
106         /**
107          * Decode the level from an encoded position.
108          */
109         private static int level(long position) {
110             return 1 + Long.numberOfTrailingZeros(position);
111         }
112 
113         /**
114          * Decode the offset from the position.
115          */
116         private static long offset(long position, int level) {
117             return position >>> level;
118         }
119 
120         /**
121          * Get the ID of the slice given an offset.
122          */
123         private static long sliceID(int level, long offset) {
124             return offset >>> level;
125         }
126 
127         /**
128          * Compute the first offset of the given slice.
129          */
130         private static long startOffset(int level, long slice) {
131             return slice << level;
132         }
133 
134         /**
135          * Compute the number of ordinals stored for a value given its current position.
136          */
137         private static int numOrdinals(int level, long offset) {
138             return (1 << level) + (int) (offset & slotsMask(level));
139         }
140 
141         // Current position
142         private PagedGrowableWriter positions;
143         // First level (0) of ordinals and pointers to the next level
144         private final GrowableWriter firstOrdinals;
145         private PagedGrowableWriter firstNextLevelSlices;
146         // Ordinals and pointers for other levels +1
147         private final PagedGrowableWriter[] ordinals;
148         private final PagedGrowableWriter[] nextLevelSlices;
149         private final int[] sizes;
150 
151         private final int startBitsPerValue;
152         private final float acceptableOverheadRatio;
153 
154         OrdinalsStore(int maxDoc, int startBitsPerValue, float acceptableOverheadRatio) {
155             this.startBitsPerValue = startBitsPerValue;
156             this.acceptableOverheadRatio = acceptableOverheadRatio;
157             positions = new PagedGrowableWriter(maxDoc, PAGE_SIZE, startBitsPerValue, acceptableOverheadRatio);
158             firstOrdinals = new GrowableWriter(startBitsPerValue, maxDoc, acceptableOverheadRatio);
159             // over allocate in order to never worry about the array sizes, 24 entries would allow to store several millions of ordinals per doc...
160             ordinals = new PagedGrowableWriter[24];
161             nextLevelSlices = new PagedGrowableWriter[24];
162             sizes = new int[24];
163             Arrays.fill(sizes, 1); // reserve the 1st slice on every level
164         }
165 
166         /**
167          * Allocate a new slice and return its ID.
168          */
169         private long newSlice(int level) {
170             final long newSlice = sizes[level]++;
171             // Lazily allocate ordinals
172             if (ordinals[level] == null) {
173                 ordinals[level] = new PagedGrowableWriter(8L * numSlots(level), PAGE_SIZE, startBitsPerValue, acceptableOverheadRatio);
174             } else {
175                 ordinals[level] = ordinals[level].grow(sizes[level] * numSlots(level));
176                 if (nextLevelSlices[level] != null) {
177                     nextLevelSlices[level] = nextLevelSlices[level].grow(sizes[level]);
178                 }
179             }
180             return newSlice;
181         }
182 
183         public int addOrdinal(int docID, long ordinal) {
184             final long position = positions.get(docID);
185 
186             if (position == 0L) { // on the first level
187                 // 0 or 1 ordinal
188                 if (firstOrdinals.get(docID) == 0L) {
189                     firstOrdinals.set(docID, ordinal + 1);
190                     return 1;
191                 } else {
192                     final long newSlice = newSlice(1);
193                     if (firstNextLevelSlices == null) {
194                         firstNextLevelSlices = new PagedGrowableWriter(firstOrdinals.size(), PAGE_SIZE, 3, acceptableOverheadRatio);
195                     }
196                     firstNextLevelSlices.set(docID, newSlice);
197                     final long offset = startOffset(1, newSlice);
198                     ordinals[1].set(offset, ordinal + 1);
199                     positions.set(docID, position(1, offset)); // current position is on the 1st level and not allocated yet
200                     return 2;
201                 }
202             } else {
203                 int level = level(position);
204                 long offset = offset(position, level);
205                 assert offset != 0L;
206                 if (((offset + 1) & slotsMask(level)) == 0L) {
207                     // reached the end of the slice, allocate a new one on the next level
208                     final long newSlice = newSlice(level + 1);
209                     if (nextLevelSlices[level] == null) {
210                         nextLevelSlices[level] = new PagedGrowableWriter(sizes[level], PAGE_SIZE, 1, acceptableOverheadRatio);
211                     }
212                     nextLevelSlices[level].set(sliceID(level, offset), newSlice);
213                     ++level;
214                     offset = startOffset(level, newSlice);
215                     assert (offset & slotsMask(level)) == 0L;
216                 } else {
217                     // just go to the next slot
218                     ++offset;
219                 }
220                 ordinals[level].set(offset, ordinal + 1);
221                 final long newPosition = position(level, offset);
222                 positions.set(docID, newPosition);
223                 return numOrdinals(level, offset);
224             }
225         }
226 
227         public void appendOrdinals(int docID, LongsRef ords) {
228             // First level
229             final long firstOrd = firstOrdinals.get(docID);
230             if (firstOrd == 0L) {
231                 return;
232             }
233             ords.longs = ArrayUtil.grow(ords.longs, ords.offset + ords.length + 1);
234             ords.longs[ords.offset + ords.length++] = firstOrd - 1;
235             if (firstNextLevelSlices == null) {
236                 return;
237             }
238             long sliceID = firstNextLevelSlices.get(docID);
239             if (sliceID == 0L) {
240                 return;
241             }
242             // Other levels
243             for (int level = 1; ; ++level) {
244                 final int numSlots = numSlots(level);
245                 ords.longs = ArrayUtil.grow(ords.longs, ords.offset + ords.length + numSlots);
246                 final long offset = startOffset(level, sliceID);
247                 for (int j = 0; j < numSlots; ++j) {
248                     final long ord = ordinals[level].get(offset + j);
249                     if (ord == 0L) {
250                         return;
251                     }
252                     ords.longs[ords.offset + ords.length++] = ord - 1;
253                 }
254                 if (nextLevelSlices[level] == null) {
255                     return;
256                 }
257                 sliceID = nextLevelSlices[level].get(sliceID);
258                 if (sliceID == 0L) {
259                     return;
260                 }
261             }
262         }
263 
264     }

266     private final int maxDoc;
267     private long currentOrd = -1;
268     private int numDocsWithValue = 0;
269     private int numMultiValuedDocs = 0;
270     private int totalNumOrds = 0;
271 
272     private OrdinalsStore ordinals;
273     private final LongsRef spare;
274 
275     public OrdinalsBuilder(long numTerms, int maxDoc, float acceptableOverheadRatio) throws IOException {
276         this.maxDoc = maxDoc;
277         int startBitsPerValue = 8;
278         if (numTerms >= 0) {
279             startBitsPerValue = PackedInts.bitsRequired(numTerms);
280         }
281         ordinals = new OrdinalsStore(maxDoc, startBitsPerValue, acceptableOverheadRatio);
282         spare = new LongsRef();
283     }
284 
285     public OrdinalsBuilder(int maxDoc, float acceptableOverheadRatio) throws IOException {
286         this(-1, maxDoc, acceptableOverheadRatio);
287     }
288 
289     public OrdinalsBuilder(int maxDoc) throws IOException {
290         this(maxDoc, DEFAULT_ACCEPTABLE_OVERHEAD_RATIO);
291     }

293     /**
294      * Returns a shared {@link LongsRef} instance for the given doc ID holding all ordinals associated with it.
295      */
296     public LongsRef docOrds(int docID) {
297         spare.offset = spare.length = 0;
298         ordinals.appendOrdinals(docID, spare);
299         return spare;
300     }

322         return currentOrd;

325     /**
326      * Associates the given document id with the current ordinal.
327      */
328     public OrdinalsBuilder addDoc(int doc) {
329         totalNumOrds++;
330         final int numValues = ordinals.addOrdinal(doc, currentOrd);
331         if (numValues == 1) {
332             ++numDocsWithValue;
333         } else if (numValues == 2) {
334             ++numMultiValuedDocs;
335         }
336         return this;
337     }

341      */

344     }

381     /**
382      * Builds a {@link BitSet} where each documents bit is that that has one or more ordinals associated with it.
383      * if every document has an ordinal associated with it this method returns <code>null</code>
384      */
385     public BitSet buildDocsWithValuesSet() {
386         if (numDocsWithValue == maxDoc) {
387             return null;
388         }
389         final FixedBitSet bitSet = new FixedBitSet(maxDoc);
390         for (int docID = 0; docID < maxDoc; ++docID) {
391             if (ordinals.firstOrdinals.get(docID) != 0) {
392                 bitSet.set(docID);
393             }
394         }
395         return bitSet;
396     }

398     /**
399      * Builds an {@link Ordinals} instance from the builders current state.
400      */
401     public Ordinals build(Settings settings) {
402         final float acceptableOverheadRatio = settings.getAsFloat("acceptable_overhead_ratio", PackedInts.FASTEST);
403         final boolean forceMultiOrdinals = settings.getAsBoolean(FORCE_MULTI_ORDINALS, false);
404         if (forceMultiOrdinals || numMultiValuedDocs > 0 || MultiOrdinals.significantlySmallerThanSinglePackedOrdinals(maxDoc, numDocsWithValue, getValueCount(), acceptableOverheadRatio)) {
405             // MultiOrdinals can be smaller than SinglePackedOrdinals for sparse fields
406             return new MultiOrdinals(this, acceptableOverheadRatio);
407         } else {
408             return new SinglePackedOrdinals(this, acceptableOverheadRatio);
409         }
410     }

419     /**
420      * A {@link TermsEnum} that iterates only full precision prefix coded 64 bit values.
421      *
422      * @see #buildFromTerms(TermsEnum, Bits)
423      */
424     public static TermsEnum wrapNumeric64Bit(TermsEnum termsEnum) {
425         return new FilteredTermsEnum(termsEnum, false) {
426             @Override
427             protected AcceptStatus accept(BytesRef term) throws IOException {
428                 // we stop accepting terms once we moved across the prefix codec terms - redundant values!
429                 return NumericUtils.getPrefixCodedLongShift(term) == 0 ? AcceptStatus.YES : AcceptStatus.END;
430             }
431         };
432     }

434     /**
435      * A {@link TermsEnum} that iterates only full precision prefix coded 32 bit values.
436      *
437      * @see #buildFromTerms(TermsEnum, Bits)
438      */
439     public static TermsEnum wrapNumeric32Bit(TermsEnum termsEnum) {
440         return new FilteredTermsEnum(termsEnum, false) {
441 
442             @Override
443             protected AcceptStatus accept(BytesRef term) throws IOException {
444                 // we stop accepting terms once we moved across the prefix codec terms - redundant values!
445                 return NumericUtils.getPrefixCodedIntShift(term) == 0 ? AcceptStatus.YES : AcceptStatus.END;
446             }
447         };
448     }

450     /**
451      * This method iterates all terms in the given {@link TermsEnum} and
452      * associates each terms ordinal with the terms documents. The caller must
453      * exhaust the returned {@link BytesRefIterator} which returns all values
454      * where the first returned value is associted with the ordinal <tt>1</tt>
455      * etc.
456      * <p>
457      * If the {@link TermsEnum} contains prefix coded numerical values the terms
458      * enum should be wrapped with either {@link #wrapNumeric32Bit(TermsEnum)}
459      * or {@link #wrapNumeric64Bit(TermsEnum)} depending on its precision. If
460      * the {@link TermsEnum} is not wrapped the returned
461      * {@link BytesRefIterator} will contain partial precision terms rather than
462      * only full-precision terms.
463      * </p>
464      */
465     public BytesRefIterator buildFromTerms(final TermsEnum termsEnum) throws IOException {
466         return new BytesRefIterator() {
467             private PostingsEnum docsEnum = null;
468 
469             @Override
470             public BytesRef next() throws IOException {
471                 BytesRef ref;
472                 if ((ref = termsEnum.next()) != null) {
473                     docsEnum = termsEnum.postings(null, docsEnum, PostingsEnum.NONE);
474                     nextOrdinal();
475                     int docId;
476                     while ((docId = docsEnum.nextDoc()) != DocIdSetIterator.NO_MORE_DOCS) {
477                         addDoc(docId);
478                     }
479                 }
480                 return ref;
481             }
482         };
483     }
