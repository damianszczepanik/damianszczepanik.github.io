69      public RoutingNodes(ClusterState clusterState) {
70          this.metaData = clusterState.metaData();
71          this.blocks = clusterState.blocks();
72          this.routingTable = clusterState.routingTable();
73  
74          Map<String, List<MutableShardRouting>> nodesToShards = newHashMap();
75          // fill in the nodeToShards with the "live" nodes
76          for (ObjectCursor<DiscoveryNode> cursor : clusterState.nodes().dataNodes().values()) {
77              nodesToShards.put(cursor.value.id(), new ArrayList<MutableShardRouting>());
78          }
79  
80          // fill in the inverse of node -> shards allocated
81          // also fill replicaSet information
82          for (IndexRoutingTable indexRoutingTable : routingTable.indicesRouting().values()) {
83              for (IndexShardRoutingTable indexShard : indexRoutingTable) {
84                  for (ShardRouting shard : indexShard) {
85                      // to get all the shards belonging to an index, including the replicas,
86                      // we define a replica set and keep track of it. A replica set is identified
87                      // by the ShardId, as this is common for primary and replicas.
88                      // A replica Set might have one (and not more) replicas with the state of RELOCATING.
89                      if (shard.assignedToNode()) {
90                          List<MutableShardRouting> entries = nodesToShards.get(shard.currentNodeId());
91                          if (entries == null) {
92                              entries = newArrayList();
93                              nodesToShards.put(shard.currentNodeId(), entries);
94                          }
95                          MutableShardRouting sr = new MutableShardRouting(shard);
96                          entries.add(sr);
97                          assignedShardsAdd(sr);
98                          if (shard.relocating()) {
99                              entries = nodesToShards.get(shard.relocatingNodeId());
100                             relocatingShards++;
101                             if (entries == null) {
102                                 entries = newArrayList();
103                                 nodesToShards.put(shard.relocatingNodeId(), entries);
104                             }
105                             // add the counterpart shard with relocatingNodeId reflecting the source from which
106                             // it's relocating from.
107                             sr = new MutableShardRouting(shard.index(), shard.id(), shard.relocatingNodeId(),
108                                     shard.currentNodeId(), shard.primary(), ShardRoutingState.INITIALIZING, shard.version());
109                             entries.add(sr);
110                             assignedShardsAdd(sr);
111                         } else if (!shard.active()) { // shards that are initializing without being relocated
112                             if (shard.primary()) {
113                                 inactivePrimaryCount++;
114                             }
115                             inactiveShardCount++;
116                         }
117                     } else {
118                         MutableShardRouting sr = new MutableShardRouting(shard);
119                         assignedShardsAdd(sr);
120                         unassignedShards.add(sr);
121                     }
122                 }
123             }
124         }
125         for (Map.Entry<String, List<MutableShardRouting>> entry : nodesToShards.entrySet()) {
126             String nodeId = entry.getKey();
127             this.nodesToShards.put(nodeId, new RoutingNode(nodeId, clusterState.nodes().get(nodeId), entry.getValue()));
128         }
129     }

160     public int requiredAverageNumberOfShardsPerNode() {
161         int totalNumberOfShards = 0;
162         // we need to recompute to take closed shards into account
163         for (ObjectCursor<IndexMetaData> cursor : metaData.indices().values()) {
164             IndexMetaData indexMetaData = cursor.value;
165             if (indexMetaData.state() == IndexMetaData.State.OPEN) {
166                 totalNumberOfShards += indexMetaData.totalNumberOfShards();
167             }
168         }
169         return totalNumberOfShards / nodesToShards.size();
170     }

211     public ObjectIntOpenHashMap<String> nodesPerAttributesCounts(String attributeName) {
212         ObjectIntOpenHashMap<String> nodesPerAttributesCounts = nodesPerAttributeNames.get(attributeName);
213         if (nodesPerAttributesCounts != null) {
214             return nodesPerAttributesCounts;
215         }
216         nodesPerAttributesCounts = new ObjectIntOpenHashMap<>();
217         for (RoutingNode routingNode : this) {
218             String attrValue = routingNode.node().attributes().get(attributeName);
219             nodesPerAttributesCounts.addTo(attrValue, 1);

222         return nodesPerAttributesCounts;

279     /**
280      * Returns <code>true</code> iff all replicas are active for the given shard routing. Otherwise <code>false</code>
281      */
282     public boolean allReplicasActive(ShardRouting shardRouting) {
283         final List<MutableShardRouting> shards = assignedShards(shardRouting.shardId());
284         if (shards.isEmpty() || shards.size() < this.routingTable.index(shardRouting.index()).shard(shardRouting.id()).size()) {
285             return false; // if we are empty nothing is active if we have less than total at least one is unassigned
286         }
287         for (MutableShardRouting shard : shards) {
288             if (!shard.active()) {
289                 return false;
290             }
291         }
292         return true;
293     }

295     public List<MutableShardRouting> shards(Predicate<MutableShardRouting> predicate) {
296         List<MutableShardRouting> shards = newArrayList();
297         for (RoutingNode routingNode : this) {
298             for (MutableShardRouting shardRouting : routingNode) {
299                 if (predicate.apply(shardRouting)) {

307     public List<MutableShardRouting> shardsWithState(ShardRoutingState... state) {
308         // TODO these are used on tests only - move into utils class
309         List<MutableShardRouting> shards = newArrayList();
310         for (RoutingNode routingNode : this) {
311             shards.addAll(routingNode.shardsWithState(state));
312         }

316                 break;
317             }

322     public List<MutableShardRouting> shardsWithState(String index, ShardRoutingState... state) {
323         // TODO these are used on tests only - move into utils class
324         List<MutableShardRouting> shards = newArrayList();
325         for (RoutingNode routingNode : this) {
326             shards.addAll(routingNode.shardsWithState(index, state));
327         }
328         for (ShardRoutingState s : state) {
329             if (s == ShardRoutingState.UNASSIGNED) {
330                 for (MutableShardRouting unassignedShard : unassignedShards) {
331                     if (unassignedShard.index().equals(index)) {
332                         shards.add(unassignedShard);
333                     }
334                 }
335                 break;
336             }
337         }
338         return shards;

353     /**
354      * Assign a shard to a node. This will increment the inactiveShardCount counter
355      * and the inactivePrimaryCount counter if the shard is the primary.
356      * In case the shard is already assigned and started, it will be marked as 
357      * relocating, which is accounted for, too, so the number of concurrent relocations
358      * can be retrieved easily.
359      * This method can be called several times for the same shard, only the first time
360      * will change the state.
361      *
362      * INITIALIZING => INITIALIZING
363      * UNASSIGNED   => INITIALIZING
364      * STARTED      => RELOCATING
365      * RELOCATING   => RELOCATING
366      *
367      * @param shard the shard to be assigned
368      * @param nodeId the nodeId this shard should initialize on or relocate from
369      */
370     public void assign(MutableShardRouting shard, String nodeId) {
371         // state will not change if the shard is already initializing.
372         ShardRoutingState oldState = shard.state();
373         shard.assignToNode(nodeId);
374         node(nodeId).add(shard);
375         if (oldState == ShardRoutingState.UNASSIGNED) {
376             inactiveShardCount++;
377             if (shard.primary()) {
378                 inactivePrimaryCount++;
379             }
380         }
381 
382         if (shard.state() == ShardRoutingState.RELOCATING) {
383             relocatingShards++;
384         }
385         assignedShardsAdd(shard);
386     }

396     /**
397      * Mark a shard as started and adjusts internal statistics.
398      */
399     public void started(MutableShardRouting shard) {
400         if (!shard.active() && shard.relocatingNodeId() == null) {
401             inactiveShardCount--;
402             if (shard.primary()) {
403                 inactivePrimaryCount--;
404             }
405         } else if (shard.relocating()) {
406             relocatingShards--;
407         }
408         assert !shard.started();
409         shard.moveToStarted();
410     }

420     /**
421      * swaps the status of a shard, making replicas primary and vice versa.
422      *
423      * @param shards the shard to have its primary status swapped.
424      */
425     public void swapPrimaryFlag(MutableShardRouting... shards) {
426         for (MutableShardRouting shard : shards) {
427             if (shard.primary()) {
428                 shard.moveFromPrimary();
429                 if (shard.unassigned()) {
430                     unassignedShards.primaries--;
431                 }
432             } else {
433                 shard.moveToPrimary();
434                 if (shard.unassigned()) {
435                     unassignedShards.primaries++;
436                 }
437             }
438         }
439     }

441     private static final List<MutableShardRouting> EMPTY = Collections.emptyList();

443     private List<MutableShardRouting> assignedShards(ShardId shardId) {
444         final List<MutableShardRouting> replicaSet = assignedShards.get(shardId);
445         return replicaSet == null ? EMPTY : Collections.unmodifiableList(replicaSet);
446     }

448     /**
449      * Cancels the give shard from the Routing nodes internal statistics and cancels
450      * the relocation if the shard is relocating.
451      * @param shard
452      */
453     private void remove(MutableShardRouting shard) {
454         if (!shard.active() && shard.relocatingNodeId() == null) {
455             inactiveShardCount--;
456             assert inactiveShardCount >= 0;
457             if (shard.primary()) {
458                 inactivePrimaryCount--;
459             }
460         } else if (shard.relocating()) {
461             cancelRelocation(shard);
462         }
463         assignedShardsRemove(shard);
464     }
465 
466     private void assignedShardsAdd(MutableShardRouting shard) {
467         if (shard.unassigned()) {
468             // no unassigned
469             return;
470         }
471         List<MutableShardRouting> shards = assignedShards.get(shard.shardId());
472         if (shards == null) {
473             shards = Lists.newArrayList();
474             assignedShards.put(shard.shardId(), shards);
475         }
476         assert  assertInstanceNotInList(shard, shards);
477         shards.add(shard);
478     }
479 
480     private boolean assertInstanceNotInList(MutableShardRouting shard, List<MutableShardRouting> shards) {
481         for (MutableShardRouting s : shards) {
482             assert s != shard;
483         }
484         return true;
485     }
486 
487     private void assignedShardsRemove(MutableShardRouting shard) {
488         final List<MutableShardRouting> replicaSet = assignedShards.get(shard.shardId());
489         if (replicaSet != null) {
490             final Iterator<MutableShardRouting> iterator = replicaSet.iterator();
491             while(iterator.hasNext()) {
492                 // yes we check identity here
493                 if (shard == iterator.next()) {
494                     iterator.remove();
495                     return;
496                 }
497             }
498             assert false : "Illegal state";
499         }
500     }

533     public final static class UnassignedShards implements Iterable<MutableShardRouting>  {
534 
535         private final List<MutableShardRouting> unassigned;
536 
537         private int primaries = 0;
538         private long transactionId = 0;
539         private final UnassignedShards source;
540         private final long sourceTransactionId;
541 
542         public UnassignedShards(UnassignedShards other) {
543             source = other;
544             sourceTransactionId = other.transactionId;
545             unassigned = new ArrayList<>(other.unassigned);
546             primaries = other.primaries;
547         }
548 
549         public UnassignedShards() {
550             unassigned = new ArrayList<>();
551             source = null;
552             sourceTransactionId = -1;
553         }
554 
555         public void add(MutableShardRouting mutableShardRouting) {
556             if(mutableShardRouting.primary()) {
557                 primaries++;
558             }
559             unassigned.add(mutableShardRouting);
560             transactionId++;
561         }
562 
563         public void addAll(Collection<MutableShardRouting> mutableShardRoutings) {
564             for (MutableShardRouting r : mutableShardRoutings) {
565                 add(r);
566             }
567         }
568 
569         public int size() {
570             return unassigned.size();
571         }
572 
573         public int numPrimaries() {
574             return primaries;
575         }
576 
577         @Override
578         public Iterator<MutableShardRouting> iterator() {
579             final Iterator<MutableShardRouting> iterator = unassigned.iterator();
580             return new Iterator<MutableShardRouting>() {
581                 private  MutableShardRouting current;
582                 @Override
583                 public boolean hasNext() {
584                     return iterator.hasNext();
585                 }
586 
587                 @Override
588                 public MutableShardRouting next() {
589                     return current = iterator.next();
590                 }
591 
592                 @Override
593                 public void remove() {
594                     iterator.remove();
595                     if (current.primary()) {
596                         primaries--;
597                     }
598                     transactionId++;
599                 }
600             };
601         }
602 
603         public boolean isEmpty() {
604             return unassigned.isEmpty();
605         }
606 
607         public void shuffle() {
608             Collections.shuffle(unassigned);
609         }
610 
611         public void clear() {
612             transactionId++;
613             unassigned.clear();
614             primaries = 0;
615         }
616 
617         public void transactionEnd(UnassignedShards shards) {
618            assert shards.source == this && shards.sourceTransactionId == transactionId :
619                    "Expected ID: " + shards.sourceTransactionId + " actual: " + transactionId + " Expected Source: " + shards.source + " actual: " + this;
620            transactionId++;
621            this.unassigned.clear();
622            this.unassigned.addAll(shards.unassigned);
623            this.primaries = shards.primaries;
624         }
625 
626         public UnassignedShards transactionBegin() {
627             return new UnassignedShards(this);
628         }
629 
630         public MutableShardRouting[] drain() {
631             MutableShardRouting[] mutableShardRoutings = unassigned.toArray(new MutableShardRouting[unassigned.size()]);
632             unassigned.clear();
633             primaries = 0;
634             transactionId++;
635             return mutableShardRoutings;
636         }
637     }

640     /**
641      * Calculates RoutingNodes statistics by iterating over all {@link MutableShardRouting}s
642      * in the cluster to ensure the book-keeping is correct.
643      * For performance reasons, this should only be called from asserts
644      *
645      * @return this method always returns <code>true</code> or throws an assertion error. If assertion are not enabled
646      *         this method does nothing.
647      */
648     public static boolean assertShardStats(RoutingNodes routingNodes) {
649         boolean run = false;
650         assert (run = true); // only run if assertions are enabled!
651         if (!run) {
652             return true;
653         }
654         int unassignedPrimaryCount = 0;
655         int inactivePrimaryCount = 0;
656         int inactiveShardCount = 0;
657         int relocating = 0;
658         final Set<ShardId> seenShards = newHashSet();
659         Map<String, Integer> indicesAndShards = new HashMap<>();
660         for (RoutingNode node : routingNodes) {
661             for (MutableShardRouting shard : node) {
662                 if (!shard.active() && shard.relocatingNodeId() == null) {
663                     if (!shard.relocating()) {
664                         inactiveShardCount++;
665                         if (shard.primary()) {
666                             inactivePrimaryCount++;
667                         }
668                     }
669                 }
670                 if (shard.relocating()) {
671                     relocating++;
672                 }
673                 seenShards.add(shard.shardId());
674                 Integer i = indicesAndShards.get(shard.index());
675                 if (i == null) {
676                     i = shard.id();
677                 }
678                 indicesAndShards.put(shard.index(), Math.max(i, shard.id()));
679             }
680         }
681         // Assert that the active shard routing are identical.
682         Set<Map.Entry<String, Integer>> entries = indicesAndShards.entrySet();
683         final List<MutableShardRouting> shards = newArrayList();
684         for (Map.Entry<String, Integer> e : entries) {
685             String index = e.getKey();
686             for (int i = 0; i < e.getValue(); i++) {
687                 for (RoutingNode routingNode : routingNodes) {
688                     for (MutableShardRouting shardRouting : routingNode) {
689                         if (shardRouting.index().equals(index) && shardRouting.id() == i) {
690                             shards.add(shardRouting);
691                         }
692                     }
693                 }
694                 List<MutableShardRouting> mutableShardRoutings = routingNodes.assignedShards(new ShardId(index, i));
695                 assert mutableShardRoutings.size() == shards.size();
696                 for (MutableShardRouting r : mutableShardRoutings) {
697                     assert shards.contains(r);
698                     shards.remove(r);
699                 }
700                 assert shards.isEmpty();
701             }
702         }
703 
704         for (MutableShardRouting shard : routingNodes.unassigned()) {
705             if (shard.primary()) {
706                 unassignedPrimaryCount++;
707             }
708             seenShards.add(shard.shardId());
709         }
710 
711         assert unassignedPrimaryCount == routingNodes.unassignedShards.numPrimaries() :
712                 "Unassigned primaries is [" + unassignedPrimaryCount + "] but RoutingNodes returned unassigned primaries [" + routingNodes.unassigned().numPrimaries() + "]";
713         assert inactivePrimaryCount == routingNodes.inactivePrimaryCount :
714                 "Inactive Primary count [" + inactivePrimaryCount + "] but RoutingNodes returned inactive primaries [" + routingNodes.inactivePrimaryCount + "]";
715         assert inactiveShardCount == routingNodes.inactiveShardCount :
716                 "Inactive Shard count [" + inactiveShardCount + "] but RoutingNodes returned inactive shards [" + routingNodes.inactiveShardCount + "]";
717         assert routingNodes.getRelocatingShardCount() == relocating : "Relocating shards mismatch [" + routingNodes.getRelocatingShardCount() + "] but expected [" + relocating + "]";
718         return true;
719     }
