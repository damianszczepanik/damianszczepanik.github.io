72  /**
73   * RecoverySourceHandler handles the three phases of shard recovery, which is
74   * everything relating to copying the segment files as well as sending translog
75   * operations across the wire once the segments have been copied.
76   */
77  public class RecoverySourceHandler implements Engine.RecoveryHandler {
78  
79      protected final ESLogger logger;
80      // Shard that is going to be recovered (the "source")
81      private final IndexShard shard;
82      private final String indexName;
83      private final int shardId;
84      // Request containing source and target node information
85      private final StartRecoveryRequest request;
86      private final RecoverySettings recoverySettings;
87      private final TransportService transportService;
88      private final ClusterService clusterService;
89      private final IndexService indexService;
90      private final MappingUpdatedAction mappingUpdatedAction;
91  
92      private final RecoveryResponse response;
93      private final CancellableThreads cancellableThreads = new CancellableThreads() {
94          @Override
95          protected void onCancel(String reason, @Nullable Throwable suppressedException) {
96              RuntimeException e;
97              if (shard.state() == IndexShardState.CLOSED) { // check if the shard got closed on us
98                  e = new IndexShardClosedException(shard.shardId(), "shard is closed and recovery was canceled reason [" + reason + "]");
99              } else {
100                 e = new ExecutionCancelledException("recovery was canceled reason [" + reason + "]");
101             }
102             if (suppressedException != null) {
103                 e.addSuppressed(suppressedException);
104             }
105             throw e;
106         }
107     };

134     /**
135      * Perform phase1 of the recovery operations. Once this {@link SnapshotIndexCommit}
136      * snapshot has been performed no commit operations (files being fsync'd)
137      * are effectively allowed on this index until all recovery phases are done
138      *
139      * Phase1 examines the segment files on the target node and copies over the
140      * segments that are missing. Only segments that have the same size and
141      * checksum can be reused
142      *
143      * {@code InternalEngine#recover} is responsible for snapshotting the index
144      * and releasing the snapshot once all 3 phases of recovery are complete
145      */
146     @Override
147     public void phase1(final SnapshotIndexCommit snapshot) throws ElasticsearchException {
148         cancellableThreads.checkForCancel();
149         // Total size of segment files that are recovered
150         long totalSize = 0;
151         // Total size of segment files that were able to be re-used
152         long existingTotalSize = 0;
153         final Store store = shard.store();
154         store.incRef();
155         try {
156             StopWatch stopWatch = new StopWatch().start();
157             final Store.MetadataSnapshot recoverySourceMetadata = store.getMetadata(snapshot);
158             for (String name : snapshot.getFiles()) {
159                 final StoreFileMetaData md = recoverySourceMetadata.get(name);
160                 if (md == null) {
161                     logger.info("Snapshot differs from actual index for file: {} meta: {}", name, recoverySourceMetadata.asMap());
162                     throw new CorruptIndexException("Snapshot differs from actual index - maybe index was removed metadata has " +
163                             recoverySourceMetadata.asMap().size() + " files", name);
164                 }
165             }
166             // Generate a "diff" of all the identical, different, and missing
167             // segment files on the target node, using the existing files on
168             // the source node
169             final Store.RecoveryDiff diff = recoverySourceMetadata.recoveryDiff(new Store.MetadataSnapshot(request.existingFiles()));
170             for (StoreFileMetaData md : diff.identical) {
171                 response.phase1ExistingFileNames.add(md.name());
172                 response.phase1ExistingFileSizes.add(md.length());
173                 existingTotalSize += md.length();
174                 if (logger.isTraceEnabled()) {
175                     logger.trace("[{}][{}] recovery [phase1] to {}: not recovering [{}], exists in local store and has checksum [{}], size [{}]",
176                             indexName, shardId, request.targetNode(), md.name(), md.checksum(), md.length());
177                 }
178                 totalSize += md.length();
179             }
180             for (StoreFileMetaData md : Iterables.concat(diff.different, diff.missing)) {
181                 if (request.existingFiles().containsKey(md.name())) {
182                     logger.trace("[{}][{}] recovery [phase1] to {}: recovering [{}], exists in local store, but is different: remote [{}], local [{}]",
183                             indexName, shardId, request.targetNode(), md.name(), request.existingFiles().get(md.name()), md);
184                 } else {
185                     logger.trace("[{}][{}] recovery [phase1] to {}: recovering [{}], does not exists in remote",
186                             indexName, shardId, request.targetNode(), md.name());
187                 }
188                 response.phase1FileNames.add(md.name());
189                 response.phase1FileSizes.add(md.length());
190                 totalSize += md.length();
191             }
192             response.phase1TotalSize = totalSize;
193             response.phase1ExistingTotalSize = existingTotalSize;
194 
195             logger.trace("[{}][{}] recovery [phase1] to {}: recovering_files [{}] with total_size [{}], reusing_files [{}] with total_size [{}]",
196                     indexName, shardId, request.targetNode(), response.phase1FileNames.size(),
197                     new ByteSizeValue(totalSize), response.phase1ExistingFileNames.size(), new ByteSizeValue(existingTotalSize));
198             cancellableThreads.execute(new Interruptable() {
199                 @Override
200                 public void run() throws InterruptedException {
201                     RecoveryFilesInfoRequest recoveryInfoFilesRequest = new RecoveryFilesInfoRequest(request.recoveryId(), request.shardId(),
202                             response.phase1FileNames, response.phase1FileSizes, response.phase1ExistingFileNames, response.phase1ExistingFileSizes);
203                     transportService.submitRequest(request.targetNode(), RecoveryTarget.Actions.FILES_INFO, recoveryInfoFilesRequest,
204                             TransportRequestOptions.options().withTimeout(recoverySettings.internalActionTimeout()),
205                             EmptyTransportResponseHandler.INSTANCE_SAME).txGet();
206                 }
207             });
208 
209 
210             // This latch will be used to wait until all files have been transferred to the target node
211             final CountDownLatch latch = new CountDownLatch(response.phase1FileNames.size());
212             final CopyOnWriteArrayList<Throwable> exceptions = new CopyOnWriteArrayList<>();
213             final AtomicReference<Throwable> corruptedEngine = new AtomicReference<>();
214             int fileIndex = 0;
215             ThreadPoolExecutor pool;
216             for (final String name : response.phase1FileNames) {
217                 long fileSize = response.phase1FileSizes.get(fileIndex);
218 
219                 // Files are split into two categories, files that are "small"
220                 // (under 5mb) and other files. Small files are transferred
221                 // using a separate thread pool dedicated to small files.
222                 //
223                 // The idea behind this is that while we are transferring an
224                 // older, large index, a user may create a new index, but that
225                 // index will not be able to recover until the large index
226                 // finishes, by using two different thread pools we can allow
227                 // tiny files (like segments for a brand new index) to be
228                 // recovered while ongoing large segment recoveries are
229                 // happening. It also allows these pools to be configured
230                 // separately.
231                 if (fileSize > recoverySettings.SMALL_FILE_CUTOFF_BYTES) {
232                     pool = recoverySettings.concurrentStreamPool();
233                 } else {
234                     pool = recoverySettings.concurrentSmallFileStreamPool();
235                 }
236 
237                 pool.execute(new AbstractRunnable() {
238                     @Override
239                     public void onFailure(Throwable t) {
240                         // we either got rejected or the store can't be incremented / we are canceled
241                         logger.debug("Failed to transfer file [" + name + "] on recovery");
242                     }
243 
244                     @Override
245                     public void onAfter() {
246                         // Signify this file has completed by decrementing the latch
247                         latch.countDown();
248                     }
249 
250                     @Override
251                     protected void doRun() {
252                         cancellableThreads.checkForCancel();
253                         store.incRef();
254                         final StoreFileMetaData md = recoverySourceMetadata.get(name);
255                         try (final IndexInput indexInput = store.directory().openInput(name, IOContext.READONCE)) {
256                             final int BUFFER_SIZE = (int) recoverySettings.fileChunkSize().bytes();
257                             final byte[] buf = new byte[BUFFER_SIZE];
258                             boolean shouldCompressRequest = recoverySettings.compress();
259                             if (CompressorFactory.isCompressed(indexInput)) {
260                                 shouldCompressRequest = false;
261                             }
262 
263                             final long len = indexInput.length();
264                             long readCount = 0;
265                             final TransportRequestOptions requestOptions = TransportRequestOptions.options()
266                                     .withCompress(shouldCompressRequest)
267                                     .withType(TransportRequestOptions.Type.RECOVERY)
268                                     .withTimeout(recoverySettings.internalActionTimeout());
269 
270                             while (readCount < len) {
271                                 if (shard.state() == IndexShardState.CLOSED) { // check if the shard got closed on us
272                                     throw new IndexShardClosedException(shard.shardId());
273                                 }
274                                 int toRead = readCount + BUFFER_SIZE > len ? (int) (len - readCount) : BUFFER_SIZE;
275                                 final long position = indexInput.getFilePointer();
276 
277                                 // Pause using the rate limiter, if desired, to throttle the recovery
278                                 if (recoverySettings.rateLimiter() != null) {
279                                     recoverySettings.rateLimiter().pause(toRead);
280                                 }
281 
282                                 indexInput.readBytes(buf, 0, toRead, false);
283                                 final BytesArray content = new BytesArray(buf, 0, toRead);
284                                 readCount += toRead;
285                                 final boolean lastChunk = readCount == len;
286                                 cancellableThreads.execute(new Interruptable() {
287                                     @Override
288                                     public void run() throws InterruptedException {
289                                         // Actually send the file chunk to the target node, waiting for it to complete
290                                         transportService.submitRequest(request.targetNode(), RecoveryTarget.Actions.FILE_CHUNK,
291                                                 new RecoveryFileChunkRequest(request.recoveryId(), request.shardId(), md, position, content, lastChunk),
292                                                 requestOptions, EmptyTransportResponseHandler.INSTANCE_SAME).txGet();
293                                     }
294                                 });
295 
296                             }
297                         } catch (Throwable e) {
298                             final Throwable corruptIndexException;
299                             if ((corruptIndexException = ExceptionsHelper.unwrapCorruption(e)) != null) {
300                                 if (store.checkIntegrityNoException(md) == false) { // we are corrupted on the primary -- fail!
301                                     logger.warn("{} Corrupted file detected {} checksum mismatch", shard.shardId(), md);
302                                     if (corruptedEngine.compareAndSet(null, corruptIndexException) == false) {
303                                         // if we are not the first exception, add ourselves as suppressed to the main one:
304                                         corruptedEngine.get().addSuppressed(e);
305                                     }
306                                 } else { // corruption has happened on the way to replica
307                                     RemoteTransportException exception = new RemoteTransportException("File corruption occurred on recovery but checksums are ok", null);
308                                     exception.addSuppressed(e);
309                                     exceptions.add(0, exception); // last exception first
310                                     logger.warn("{} Remote file corruption on node {}, recovering {}. local checksum OK",
311                                             corruptIndexException, shard.shardId(), request.targetNode(), md);
312 
313                                 }
314                             } else {
315                                 exceptions.add(0, e); // last exceptions first
316                             }
317                         } finally {
318                             store.decRef();
319 
320                         }
321                     }
322                 });
323                 fileIndex++;
324             }
325 
326             cancellableThreads.execute(new Interruptable() {
327                 @Override
328                 public void run() throws InterruptedException {
329                     // Wait for all files that need to be transferred to finish transferring
330                     latch.await();
331                 }
332             });
333 
334             if (corruptedEngine.get() != null) {
335                 throw corruptedEngine.get();
336             } else {
337                 ExceptionsHelper.rethrowAndSuppress(exceptions);
338             }
339 
340             cancellableThreads.execute(new Interruptable() {
341                 @Override
342                 public void run() throws InterruptedException {
343                     // Send the CLEAN_FILES request, which takes all of the files that
344                     // were transferred and renames them from their temporary file
345                     // names to the actual file names. It also writes checksums for
346                     // the files after they have been renamed.
347                     //
348                     // Once the files have been renamed, any other files that are not
349                     // related to this recovery (out of date segments, for example)
350                     // are deleted
351                     try {
352                         transportService.submitRequest(request.targetNode(), RecoveryTarget.Actions.CLEAN_FILES,
353                                 new RecoveryCleanFilesRequest(request.recoveryId(), shard.shardId(), recoverySourceMetadata),
354                                 TransportRequestOptions.options().withTimeout(recoverySettings.internalActionTimeout()),
355                                 EmptyTransportResponseHandler.INSTANCE_SAME).txGet();
356                     } catch (RemoteTransportException remoteException) {
357                         final IOException corruptIndexException;
358                         // we realized that after the index was copied and we wanted to finalize the recovery
359                         // the index was corrupted:
360                         //   - maybe due to a broken segments file on an empty index (transferred with no checksum)
361                         //   - maybe due to old segments without checksums or length only checks
362                         if ((corruptIndexException = ExceptionsHelper.unwrapCorruption(remoteException)) != null) {
363                             try {
364                                 final Store.MetadataSnapshot recoverySourceMetadata = store.getMetadata(snapshot);
365                                 StoreFileMetaData[] metadata = Iterables.toArray(recoverySourceMetadata, StoreFileMetaData.class);
366                                 ArrayUtil.timSort(metadata, new Comparator<StoreFileMetaData>() {
367                                     @Override
368                                     public int compare(StoreFileMetaData o1, StoreFileMetaData o2) {
369                                         return Long.compare(o1.length(), o2.length()); // check small files first
370                                     }
371                                 });
372                                 for (StoreFileMetaData md : metadata) {
373                                     logger.debug("{} checking integrity for file {} after remove corruption exception", shard.shardId(), md);
374                                     if (store.checkIntegrityNoException(md) == false) { // we are corrupted on the primary -- fail!
375                                         logger.warn("{} Corrupted file detected {} checksum mismatch", shard.shardId(), md);
376                                         throw corruptIndexException;
377                                     }
378                                 }
379                             } catch (IOException ex) {
380                                 remoteException.addSuppressed(ex);
381                                 throw remoteException;
382                             }
383                             // corruption has happened on the way to replica
384                             RemoteTransportException exception = new RemoteTransportException("File corruption occurred on recovery but checksums are ok", null);
385                             exception.addSuppressed(remoteException);
386                             logger.warn("{} Remote file corruption during finalization on node {}, recovering {}. local checksum OK",
387                                     corruptIndexException, shard.shardId(), request.targetNode());
388                         } else {
389                             throw remoteException;
390                         }
391                     }
392                 }
393             });
394             stopWatch.stop();
395             logger.trace("[{}][{}] recovery [phase1] to {}: took [{}]", indexName, shardId, request.targetNode(), stopWatch.totalTime());
396             response.phase1Time = stopWatch.totalTime().millis();
397         } catch (Throwable e) {
398             throw new RecoverFilesRecoveryException(request.shardId(), response.phase1FileNames.size(), new ByteSizeValue(totalSize), e);
399         } finally {
400             store.decRef();
401         }
402     }

404     /**
405      * Perform phase2 of the recovery process
406      *
407      * Phase2 takes a snapshot of the current translog *without* acquiring the
408      * write lock (however, the translog snapshot is a point-in-time view of
409      * the translog). It then sends each translog operation to the target node
410      * so it can be replayed into the new shard.
411      *
412      * {@code InternalEngine#recover} is responsible for taking the snapshot
413      * of the translog and releasing it once all 3 phases of recovery are complete
414      */
415     @Override
416     public void phase2(Translog.Snapshot snapshot) throws ElasticsearchException {
417         if (shard.state() == IndexShardState.CLOSED) {
418             throw new IndexShardClosedException(request.shardId());
419         }
420         cancellableThreads.checkForCancel();
421         logger.trace("{} recovery [phase2] to {}: start", request.shardId(), request.targetNode());
422         StopWatch stopWatch = new StopWatch().start();
423         cancellableThreads.execute(new Interruptable() {
424             @Override
425             public void run() throws InterruptedException {
426                 // Send a request preparing the new shard's translog to receive
427                 // operations. This ensures the shard engine is started and disables
428                 // garbage collection (not the JVM's GC!) of tombstone deletes
429                 transportService.submitRequest(request.targetNode(), RecoveryTarget.Actions.PREPARE_TRANSLOG,
430                         new RecoveryPrepareForTranslogOperationsRequest(request.recoveryId(), request.shardId()),
431                         TransportRequestOptions.options().withTimeout(recoverySettings.internalActionTimeout()), EmptyTransportResponseHandler.INSTANCE_SAME).txGet();
432             }
433         });
434 
435         stopWatch.stop();
436         response.startTime = stopWatch.totalTime().millis();
437         logger.trace("{} recovery [phase2] to {}: start took [{}]",
438                 request.shardId(), request.targetNode(), stopWatch.totalTime());
439 
440 
441         logger.trace("{} recovery [phase2] to {}: updating current mapping to master", request.shardId(), request.targetNode());
442         // Ensure that the mappings are synced with the master node
443         updateMappingOnMaster();
444 
445         logger.trace("{} recovery [phase2] to {}: sending transaction log operations", request.shardId(), request.targetNode());
446         stopWatch = new StopWatch().start();
447         // Send all the snapshot's translog operations to the target
448         int totalOperations = sendSnapshot(snapshot);
449         stopWatch.stop();
450         logger.trace("{} recovery [phase2] to {}: took [{}]", request.shardId(), request.targetNode(), stopWatch.totalTime());
451         response.phase2Time = stopWatch.totalTime().millis();
452         response.phase2Operations = totalOperations;
453     }
454 
455     /**
456      * Perform phase 3 of the recovery process
457      *
458      * Phase3 again takes a snapshot of the translog, however this time the
459      * snapshot is acquired under a write lock. The translog operations are
460      * sent to the target node where they are replayed.
461      *
462      * {@code InternalEngine#recover} is responsible for taking the snapshot
463      * of the translog, and after phase 3 completes the snapshots from all
464      * three phases are released.
465      */
466     @Override
467     public void phase3(Translog.Snapshot snapshot) throws ElasticsearchException {
468         if (shard.state() == IndexShardState.CLOSED) {
469             throw new IndexShardClosedException(request.shardId());
470         }
471         cancellableThreads.checkForCancel();
472         StopWatch stopWatch = new StopWatch().start();
473         final int totalOperations;
474         logger.trace("[{}][{}] recovery [phase3] to {}: sending transaction log operations", indexName, shardId, request.targetNode());
475 
476         // Send the translog operations to the target node
477         totalOperations = sendSnapshot(snapshot);
478 
479         cancellableThreads.execute(new Interruptable() {
480             @Override
481             public void run() throws InterruptedException {
482                 // Send the FINALIZE request to the target node. The finalize request
483                 // clears unreferenced translog files, refreshes the engine now that
484                 // new segments are available, and enables garbage collection of
485                 // tombstone files. The shard is also moved to the POST_RECOVERY phase
486                 // during this time
487                 transportService.submitRequest(request.targetNode(), RecoveryTarget.Actions.FINALIZE,
488                         new RecoveryFinalizeRecoveryRequest(request.recoveryId(), request.shardId()),
489                         TransportRequestOptions.options().withTimeout(recoverySettings.internalActionLongTimeout()),
490                         EmptyTransportResponseHandler.INSTANCE_SAME).txGet();
491             }
492         });
493 
494 
495         if (request.markAsRelocated()) {
496             // TODO what happens if the recovery process fails afterwards, we need to mark this back to started
497             try {
498                 shard.relocated("to " + request.targetNode());
499             } catch (IllegalIndexShardStateException e) {
500                 // we can ignore this exception since, on the other node, when it moved to phase3
501                 // it will also send shard started, which might cause the index shard we work against
502                 // to move be closed by the time we get to the the relocated method
503             }
504         }
505         stopWatch.stop();
506         logger.trace("[{}][{}] recovery [phase3] to {}: took [{}]",
507                 indexName, shardId, request.targetNode(), stopWatch.totalTime());
508         response.phase3Time = stopWatch.totalTime().millis();
509         response.phase3Operations = totalOperations;
510     }

512     /**
513      * Ensures that the mapping in the cluster state is the same as the mapping
514      * in our mapper service. If the mapping is not in sync, sends a request
515      * to update it in the cluster state and blocks until it has finished
516      * being updated.
517      */
518     private void updateMappingOnMaster() {
519         // we test that the cluster state is in sync with our in memory mapping stored by the mapperService
520         // we have to do it under the "cluster state update" thread to make sure that one doesn't modify it
521         // while we're checking
522         final BlockingQueue<DocumentMapper> documentMappersToUpdate = ConcurrentCollections.newBlockingQueue();
523         final CountDownLatch latch = new CountDownLatch(1);
524         final AtomicReference<Throwable> mappingCheckException = new AtomicReference<>();
525 
526         // we use immediate as this is a very light weight check and we don't wait to delay recovery
527         clusterService.submitStateUpdateTask("recovery_mapping_check", Priority.IMMEDIATE, new MappingUpdateTask(clusterService, indexService, recoverySettings, latch, documentMappersToUpdate, mappingCheckException, this.cancellableThreads));
528         cancellableThreads.execute(new Interruptable() {
529             @Override
530             public void run() throws InterruptedException {
531                 latch.await();
532             }
533         });
534         if (mappingCheckException.get() != null) {
535             logger.warn("error during mapping check, failing recovery", mappingCheckException.get());
536             throw new ElasticsearchException("error during mapping check", mappingCheckException.get());
537         }
538         if (documentMappersToUpdate.isEmpty()) {
539             return;
540         }
541         final CountDownLatch updatedOnMaster = new CountDownLatch(documentMappersToUpdate.size());
542         MappingUpdatedAction.MappingUpdateListener listener = new MappingUpdatedAction.MappingUpdateListener() {
543             @Override
544             public void onMappingUpdate() {
545                 updatedOnMaster.countDown();
546             }
547 
548             @Override
549             public void onFailure(Throwable t) {
550                 logger.debug("{} recovery to {}: failed to update mapping on master", request.shardId(), request.targetNode(), t);
551                 updatedOnMaster.countDown();
552             }
553         };
554         for (DocumentMapper documentMapper : documentMappersToUpdate) {
555             mappingUpdatedAction.updateMappingOnMaster(indexService.index().getName(), documentMapper, indexService.indexUUID(), listener);
556         }
557         cancellableThreads.execute(new Interruptable() {
558             @Override
559             public void run() throws InterruptedException {
560                 try {
561                     if (!updatedOnMaster.await(recoverySettings.internalActionTimeout().millis(), TimeUnit.MILLISECONDS)) {
562                         logger.debug("[{}][{}] recovery [phase2] to {}: waiting on pending mapping update timed out. waited [{}]",
563                                 indexName, shardId, request.targetNode(), recoverySettings.internalActionTimeout());
564                     }
565                 } catch (InterruptedException e) {
566                     Thread.currentThread().interrupt();
567                     logger.debug("interrupted while waiting for mapping to update on master");
568                 }
569             }
570         });
571     }

573     /**
574      * Send the given snapshot's operations to this handler's target node.
575      *
576      * Operations are bulked into a single request depending on an operation
577      * count limit or size-in-bytes limit
578      *
579      * @return the total number of translog operations that were sent
580      */
581     protected int sendSnapshot(Translog.Snapshot snapshot) throws ElasticsearchException {
582         int ops = 0;
583         long size = 0;
584         int totalOperations = 0;
585         final List<Translog.Operation> operations = Lists.newArrayList();
586         Translog.Operation operation = snapshot.next();
587 
588         final TransportRequestOptions recoveryOptions = TransportRequestOptions.options()
589                 .withCompress(recoverySettings.compress())
590                 .withType(TransportRequestOptions.Type.RECOVERY)
591                 .withTimeout(recoverySettings.internalActionLongTimeout());
592 
593         while (operation != null) {
594             if (shard.state() == IndexShardState.CLOSED) {
595                 throw new IndexShardClosedException(request.shardId());
596             }
597             cancellableThreads.checkForCancel();
598             operations.add(operation);
599             ops += 1;
600             size += operation.estimateSize();
601             totalOperations++;
602 
603             // Check if this request is past the size or bytes threshold, and
604             // if so, send it off
605             if (ops >= recoverySettings.translogOps() || size >= recoverySettings.translogSize().bytes()) {
606 
607                 // don't throttle translog, since we lock for phase3 indexing,
608                 // so we need to move it as fast as possible. Note, since we
609                 // index docs to replicas while the index files are recovered
610                 // the lock can potentially be removed, in which case, it might
611                 // make sense to re-enable throttling in this phase
612 //                if (recoverySettings.rateLimiter() != null) {
613 //                    recoverySettings.rateLimiter().pause(size);
614 //                }
615 
616                 cancellableThreads.execute(new Interruptable() {
617                     @Override
618                     public void run() throws InterruptedException {
619                         final RecoveryTranslogOperationsRequest translogOperationsRequest = new RecoveryTranslogOperationsRequest(request.recoveryId(), request.shardId(), operations);
620                         transportService.submitRequest(request.targetNode(), RecoveryTarget.Actions.TRANSLOG_OPS, translogOperationsRequest,
621                                 recoveryOptions, EmptyTransportResponseHandler.INSTANCE_SAME).txGet();
622                     }
623                 });
624 
625                 ops = 0;
626                 size = 0;
627                 operations.clear();
628             }
629             operation = snapshot.next();
630         }
631         // send the leftover
632         if (!operations.isEmpty()) {
633             cancellableThreads.execute(new Interruptable() {
634                 @Override
635                 public void run() throws InterruptedException {
636                     RecoveryTranslogOperationsRequest translogOperationsRequest = new RecoveryTranslogOperationsRequest(request.recoveryId(), request.shardId(), operations);
637                     transportService.submitRequest(request.targetNode(), RecoveryTarget.Actions.TRANSLOG_OPS, translogOperationsRequest,
638                             recoveryOptions, EmptyTransportResponseHandler.INSTANCE_SAME).txGet();
639                 }
640             });
641 
642         }
643         return totalOperations;
644     }

646     /**
647      * Cancels the recovery and interrupts all eligible threads.
648      */
649     public void cancel(String reason) {
650         cancellableThreads.cancel(reason);
651     }

662     // this is a static class since we are holding an instance to the IndexShard
663     // on ShardRecoveryHandler which can not be GCed if the recovery is canceled
664     // but this task is still stuck in the queue. This can be problematic if the
665     // queue piles up and recoveries fail and can lead to OOM or memory pressure if lots of shards
666     // are created and removed.
667     private static class MappingUpdateTask extends TimeoutClusterStateUpdateTask {
668         private final CountDownLatch latch;
669         private final BlockingQueue<DocumentMapper> documentMappersToUpdate;
670         private final AtomicReference<Throwable> mappingCheckException;
671         private final CancellableThreads cancellableThreads;
672         private ClusterService clusterService;
673         private IndexService indexService;
674         private RecoverySettings recoverySettings;
675 
676         public MappingUpdateTask(ClusterService clusterService, IndexService indexService, RecoverySettings recoverySettings, CountDownLatch latch, BlockingQueue<DocumentMapper> documentMappersToUpdate, AtomicReference<Throwable> mappingCheckException, CancellableThreads cancellableThreads) {
677             this.latch = latch;
678             this.documentMappersToUpdate = documentMappersToUpdate;
679             this.mappingCheckException = mappingCheckException;
680             this.clusterService = clusterService;
681             this.indexService = indexService;
682             this.recoverySettings = recoverySettings;
683             this.cancellableThreads = cancellableThreads;
684         }
685 
686         @Override
687         public boolean runOnlyOnMaster() {
688             return false;
689         }
690 
691         @Override
692         public TimeValue timeout() {
693             return recoverySettings.internalActionTimeout();
694         }
695 
696         @Override
697         public void clusterStateProcessed(String source, ClusterState oldState, ClusterState newState) {
698             latch.countDown();
699         }
700 
701         @Override
702         public ClusterState execute(ClusterState currentState) throws Exception {
703             if (cancellableThreads.isCancelled() == false) { // no need to run this if recovery is canceled
704                 IndexMetaData indexMetaData = clusterService.state().metaData().getIndices().get(indexService.index().getName());
705                 ImmutableOpenMap<String, MappingMetaData> metaDataMappings = null;
706                 if (indexMetaData != null) {
707                     metaDataMappings = indexMetaData.getMappings();
708                 }
709                 // default mapping should not be sent back, it can only be updated by put mapping API, and its
710                 // a full in place replace, we don't want to override a potential update coming into it
711                 for (DocumentMapper documentMapper : indexService.mapperService().docMappers(false)) {
712 
713                     MappingMetaData mappingMetaData = metaDataMappings == null ? null : metaDataMappings.get(documentMapper.type());
714                     if (mappingMetaData == null || !documentMapper.refreshSource().equals(mappingMetaData.source())) {
715                         // not on master yet in the right form
716                         documentMappersToUpdate.add(documentMapper);
717                     }
718                 }
719             }
720             return currentState;
721         }
722 
723         @Override
724         public void onFailure(String source, Throwable t) {
725             mappingCheckException.set(t);
726             latch.countDown();
727         }
728     }
