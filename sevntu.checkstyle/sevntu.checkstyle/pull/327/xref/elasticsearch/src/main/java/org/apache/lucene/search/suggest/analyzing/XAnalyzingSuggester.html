47  /**
48   * Suggester that first analyzes the surface form, adds the
49   * analyzed form to a weighted FST, and then does the same
50   * thing at lookup time.  This means lookup is based on the
51   * analyzed form while suggestions are still the surface
52   * form(s).
53   *
54   * <p>
55   * This can result in powerful suggester functionality.  For
56   * example, if you use an analyzer removing stop words, 
57   * then the partial text "ghost chr..." could see the
58   * suggestion "The Ghost of Christmas Past".  Note that
59   * position increments MUST NOT be preserved for this example
60   * to work, so you should call the constructor with
61   * <code>preservePositionIncrements</code> parameter set to
62   * false
63   *
64   * <p>
65   * If SynonymFilter is used to map wifi and wireless network to
66   * hotspot then the partial text "wirele..." could suggest
67   * "wifi router".  Token normalization like stemmers, accent
68   * removal, etc., would allow suggestions to ignore such
69   * variations.
70   *
71   * <p>
72   * When two matching suggestions have the same weight, they
73   * are tie-broken by the analyzed form.  If their analyzed
74   * form is the same then the order is undefined.
75   *
76   * <p>
77   * There are some limitations:
78   * <ul>
79   *
80   *   <li> A lookup from a query like "net" in English won't
81   *        be any different than "net " (ie, user added a
82   *        trailing space) because analyzers don't reflect
83   *        when they've seen a token separator and when they
84   *        haven't.
85   *
86   *   <li> If you're using {@code StopFilter}, and the user will
87   *        type "fast apple", but so far all they've typed is
88   *        "fast a", again because the analyzer doesn't convey whether
89   *        it's seen a token separator after the "a",
90   *        {@code StopFilter} will remove that "a" causing
91   *        far more matches than you'd expect.
92   *
93   *   <li> Lookups with the empty string return no results
94   *        instead of all results.
95   * </ul>
96   * 
97   * @lucene.experimental
98   */
99  public class XAnalyzingSuggester extends Lookup {

227       throw new IllegalArgumentException("options should only contain EXACT_FIRST and PRESERVE_SEP; got " + options);

262   // Replaces SEP with epsilon or remaps them if
263   // we were asked to preserve them:
264   private Automaton replaceSep(Automaton a) {
265 
266       Automaton result = new Automaton();
267 
268       // Copy all states over
269       int numStates = a.getNumStates();
270       for(int s=0;s<numStates;s++) {
271         result.createState();
272         result.setAccept(s, a.isAccept(s));
273       }
274 
275       // Go in reverse topo sort so we know we only have to
276       // make one pass:
277       Transition t = new Transition();
278       int[] topoSortStates = topoSortStates(a);
279       for(int i=0;i<topoSortStates.length;i++) {
280         int state = topoSortStates[topoSortStates.length-1-i];
281         int count = a.initTransition(state, t);
282         for(int j=0;j<count;j++) {
283           a.getNextTransition(t);
284           if (t.min == TokenStreamToAutomaton.POS_SEP) {
285             assert t.max == TokenStreamToAutomaton.POS_SEP;
286             if (preserveSep) {
287               // Remap to SEP_LABEL:
288               result.addTransition(state, t.dest, SEP_LABEL);
289             } else {
290               result.addEpsilon(state, t.dest);
291             }
292           } else if (t.min == TokenStreamToAutomaton.HOLE) {
293             assert t.max == TokenStreamToAutomaton.HOLE;
294 
295             // Just remove the hole: there will then be two
296             // SEP tokens next to each other, which will only
297             // match another hole at search time.  Note that
298             // it will also match an empty-string token ... if
299             // that's somehow a problem we can always map HOLE
300             // to a dedicated byte (and escape it in the
301             // input).
302             result.addEpsilon(state, t.dest);
303           } else {
304             result.addTransition(state, t.dest, t.min, t.max);
305           }
306         }
307       }
308 
309       result.finishState();
310 
311       return result;
312   }

314   protected Automaton convertAutomaton(Automaton a) {
315     if (queryPrefix != null) {
316       a = Operations.concatenate(Arrays.asList(queryPrefix, a));
317       // This automaton should not blow up during determinize:
318       a = Operations.determinize(a, Integer.MAX_VALUE);
319     }
320     return a;
321   }
322   
323   private int[] topoSortStates(Automaton a) {
324       int[] states = new int[a.getNumStates()];
325       final Set<Integer> visited = new HashSet<>();
326       final LinkedList<Integer> worklist = new LinkedList<>();
327       worklist.add(0);
328       visited.add(0);
329       int upto = 0;
330       states[upto] = 0;
331       upto++;
332       Transition t = new Transition();
333       while (worklist.size() > 0) {
334         int s = worklist.removeFirst();
335         int count = a.initTransition(s, t);
336         for (int i=0;i<count;i++) {
337           a.getNextTransition(t);
338           if (!visited.contains(t.dest)) {
339             visited.add(t.dest);
340             worklist.add(t.dest);
341             states[upto++] = t.dest;
342           }
343         }
344       }
345       return states;
346     }

348   /** Just escapes the 0xff byte (which we still for SEP). */
349   private static final class  EscapingTokenStreamToAutomaton extends TokenStreamToAutomaton {
350 
351     final BytesRefBuilder spare = new BytesRefBuilder();
352     private char sepLabel;
353 
354     public EscapingTokenStreamToAutomaton(char sepLabel) {
355       this.sepLabel = sepLabel;
356     }
357 
358     @Override
359     protected BytesRef changeToken(BytesRef in) {
360       int upto = 0;
361       for(int i=0;i<in.length;i++) {
362         byte b = in.bytes[in.offset+i];
363         if (b == (byte) sepLabel) {
364           spare.grow(upto+2);
365           spare.setByteAt(upto++, (byte) sepLabel);
366           spare.setByteAt(upto++, b);
367         } else {
368           spare.grow(upto+1);
369           spare.setByteAt(upto++, b);
370         }
371       }
372       spare.setLength(upto);
373       return spare.get();
374     }
375   }
376 
377   public TokenStreamToAutomaton getTokenStreamToAutomaton() {
378     final TokenStreamToAutomaton tsta;
379     if (preserveSep) {
380       tsta = new EscapingTokenStreamToAutomaton((char) sepLabel);
381     } else {
382       // When we're not preserving sep, we don't steal 0xff
383       // byte, so we don't need to do any escaping:
384       tsta = new TokenStreamToAutomaton();
385     }
386     tsta.setPreservePositionIncrements(preservePositionIncrements);
387     return tsta;
388   }

390   private static class AnalyzingComparator implements Comparator<BytesRef> {
391 
392     private final boolean hasPayloads;
393 
394     public AnalyzingComparator(boolean hasPayloads) {
395       this.hasPayloads = hasPayloads;
396     }
397 
398     private final ByteArrayDataInput readerA = new ByteArrayDataInput();
399     private final ByteArrayDataInput readerB = new ByteArrayDataInput();
400     private final BytesRef scratchA = new BytesRef();
401     private final BytesRef scratchB = new BytesRef();
402 
403     @Override
404     public int compare(BytesRef a, BytesRef b) {
405 
406       // First by analyzed form:
407       readerA.reset(a.bytes, a.offset, a.length);
408       scratchA.length = readerA.readShort();
409       scratchA.bytes = a.bytes;
410       scratchA.offset = readerA.getPosition();
411 
412       readerB.reset(b.bytes, b.offset, b.length);
413       scratchB.bytes = b.bytes;
414       scratchB.length = readerB.readShort();
415       scratchB.offset = readerB.getPosition();
416 
417       int cmp = scratchA.compareTo(scratchB);
418       if (cmp != 0) {
419         return cmp;
420       }
421       readerA.skipBytes(scratchA.length);
422       readerB.skipBytes(scratchB.length);
423       // Next by cost:
424       long aCost = readerA.readInt();
425       long bCost = readerB.readInt();
426       if (aCost < bCost) {
427         return -1;
428       } else if (aCost > bCost) {
429         return 1;
430       }
431 
432       // Finally by surface form:
433       if (hasPayloads) {
434         scratchA.length = readerA.readShort();
435         scratchA.offset = readerA.getPosition();
436         scratchB.length = readerB.readShort();
437         scratchB.offset = readerB.getPosition();
438       } else {
439         scratchA.offset = readerA.getPosition();
440         scratchA.length = a.length - scratchA.offset;
441         scratchB.offset = readerB.getPosition();
442         scratchB.length = b.length - scratchB.offset;
443       }
444       return scratchA.compareTo(scratchB);
445     }
446   }

448   @Override
449   public void build(InputIterator iterator) throws IOException {
450     String prefix = getClass().getSimpleName();
451     Path directory = OfflineSorter.defaultTempDir();
452     Path tempInput = Files.createTempFile(directory, prefix, ".input");
453     Path tempSorted = Files.createTempFile(directory, prefix, ".sorted");
454 
455     hasPayloads = iterator.hasPayloads();
456 
457     OfflineSorter.ByteSequencesWriter writer = new OfflineSorter.ByteSequencesWriter(tempInput);
458     OfflineSorter.ByteSequencesReader reader = null;
459     BytesRefBuilder scratch = new BytesRefBuilder();
460 
461     TokenStreamToAutomaton ts2a = getTokenStreamToAutomaton();
462 
463     boolean success = false;
464     count = 0;
465     byte buffer[] = new byte[8];
466     try {
467       ByteArrayDataOutput output = new ByteArrayDataOutput(buffer);
468       BytesRef surfaceForm;
469 
470       while ((surfaceForm = iterator.next()) != null) {
471         Set<IntsRef> paths = toFiniteStrings(surfaceForm, ts2a);
472         
473         maxAnalyzedPathsForOneInput = Math.max(maxAnalyzedPathsForOneInput, paths.size());
474 
475         for (IntsRef path : paths) {
476 
477           Util.toBytesRef(path, scratch);
478           
479           // length of the analyzed text (FST input)
480           if (scratch.length() > Short.MAX_VALUE-2) {
481             throw new IllegalArgumentException("cannot handle analyzed forms > " + (Short.MAX_VALUE-2) + " in length (got " + scratch.length() + ")");
482           }
483           short analyzedLength = (short) scratch.length();
484 
485           // compute the required length:
486           // analyzed sequence + weight (4) + surface + analyzedLength (short)
487           int requiredLength = analyzedLength + 4 + surfaceForm.length + 2;
488 
489           BytesRef payload;
490 
491           if (hasPayloads) {
492             if (surfaceForm.length > (Short.MAX_VALUE-2)) {
493               throw new IllegalArgumentException("cannot handle surface form > " + (Short.MAX_VALUE-2) + " in length (got " + surfaceForm.length + ")");
494             }
495             payload = iterator.payload();
496             // payload + surfaceLength (short)
497             requiredLength += payload.length + 2;
498           } else {
499             payload = null;
500           }
501           
502           buffer = ArrayUtil.grow(buffer, requiredLength);
503           
504           output.reset(buffer);
505 
506           output.writeShort(analyzedLength);
507 
508           output.writeBytes(scratch.bytes(), 0, scratch.length());
509 
510           output.writeInt(encodeWeight(iterator.weight()));
511 
512           if (hasPayloads) {
513             for(int i=0;i<surfaceForm.length;i++) {
514               if (surfaceForm.bytes[i] == payloadSep) {
515                 throw new IllegalArgumentException("surface form cannot contain unit separator character U+001F; this character is reserved");
516               }
517             }
518             output.writeShort((short) surfaceForm.length);
519             output.writeBytes(surfaceForm.bytes, surfaceForm.offset, surfaceForm.length);
520             output.writeBytes(payload.bytes, payload.offset, payload.length);
521           } else {
522             output.writeBytes(surfaceForm.bytes, surfaceForm.offset, surfaceForm.length);
523           }
524 
525           assert output.getPosition() == requiredLength: output.getPosition() + " vs " + requiredLength;
526 
527           writer.write(buffer, 0, output.getPosition());
528         }
529         count++;
530       }
531       writer.close();
532 
533       // Sort all input/output pairs (required by FST.Builder):
534       new OfflineSorter(new AnalyzingComparator(hasPayloads)).sort(tempInput, tempSorted);
535 
536       // Free disk space:
537       Files.delete(tempInput);
538 
539       reader = new OfflineSorter.ByteSequencesReader(tempSorted);
540      
541       PairOutputs<Long,BytesRef> outputs = new PairOutputs<>(PositiveIntOutputs.getSingleton(), ByteSequenceOutputs.getSingleton());
542       Builder<Pair<Long,BytesRef>> builder = new Builder<>(FST.INPUT_TYPE.BYTE1, outputs);
543 
544       // Build FST:
545       BytesRefBuilder previousAnalyzed = null;
546       BytesRefBuilder analyzed = new BytesRefBuilder();
547       BytesRef surface = new BytesRef();
548       IntsRefBuilder scratchInts = new IntsRefBuilder();
549       ByteArrayDataInput input = new ByteArrayDataInput();
550 
551       // Used to remove duplicate surface forms (but we
552       // still index the hightest-weight one).  We clear
553       // this when we see a new analyzed form, so it cannot
554       // grow unbounded (at most 256 entries):
555       Set<BytesRef> seenSurfaceForms = new HashSet<>();
556 
557       int dedup = 0;
558       while (reader.read(scratch)) {
559         input.reset(scratch.bytes(), 0, scratch.length());
560         short analyzedLength = input.readShort();
561         analyzed.grow(analyzedLength+2);
562         input.readBytes(analyzed.bytes(), 0, analyzedLength);
563         analyzed.setLength(analyzedLength);
564 
565         long cost = input.readInt();
566 
567         surface.bytes = scratch.bytes();
568         if (hasPayloads) {
569           surface.length = input.readShort();
570           surface.offset = input.getPosition();
571         } else {
572           surface.offset = input.getPosition();
573           surface.length = scratch.length() - surface.offset;
574         }
575         
576         if (previousAnalyzed == null) {
577           previousAnalyzed = new BytesRefBuilder();
578           previousAnalyzed.copyBytes(analyzed);
579           seenSurfaceForms.add(BytesRef.deepCopyOf(surface));
580         } else if (analyzed.get().equals(previousAnalyzed.get())) {
581           dedup++;
582           if (dedup >= maxSurfaceFormsPerAnalyzedForm) {
583             // More than maxSurfaceFormsPerAnalyzedForm
584             // dups: skip the rest:
585             continue;
586           }
587           if (seenSurfaceForms.contains(surface)) {
588             continue;
589           }
590           seenSurfaceForms.add(BytesRef.deepCopyOf(surface));
591         } else {
592           dedup = 0;
593           previousAnalyzed.copyBytes(analyzed);
594           seenSurfaceForms.clear();
595           seenSurfaceForms.add(BytesRef.deepCopyOf(surface));
596         }
597 
598         // TODO: I think we can avoid the extra 2 bytes when
599         // there is no dup (dedup==0), but we'd have to fix
600         // the exactFirst logic ... which would be sort of
601         // hairy because we'd need to special case the two
602         // (dup/not dup)...
603 
604         // NOTE: must be byte 0 so we sort before whatever
605         // is next
606         analyzed.append((byte) 0);
607         analyzed.append((byte) dedup);
608 
609         Util.toIntsRef(analyzed.get(), scratchInts);
610         //System.out.println("ADD: " + scratchInts + " -> " + cost + ": " + surface.utf8ToString());
611         if (!hasPayloads) {
612           builder.add(scratchInts.get(), outputs.newPair(cost, BytesRef.deepCopyOf(surface)));
613         } else {
614           int payloadOffset = input.getPosition() + surface.length;
615           int payloadLength = scratch.length() - payloadOffset;
616           BytesRef br = new BytesRef(surface.length + 1 + payloadLength);
617           System.arraycopy(surface.bytes, surface.offset, br.bytes, 0, surface.length);
618           br.bytes[surface.length] = (byte) payloadSep;
619           System.arraycopy(scratch.bytes(), payloadOffset, br.bytes, surface.length+1, payloadLength);
620           br.length = br.bytes.length;
621           builder.add(scratchInts.get(), outputs.newPair(cost, br));
622         }
623       }
624       fst = builder.finish();
625 
626       //PrintWriter pw = new PrintWriter("/tmp/out.dot");
627       //Util.toDot(fst, pw, true, true);
628       //pw.close();
629 
630       success = true;
631     } finally {
632       IOUtils.closeWhileHandlingException(reader, writer);
633

677   private LookupResult getLookupResult(Long output1, BytesRef output2, CharsRefBuilder spare) {
678     LookupResult result;
679     if (hasPayloads) {
680       int sepIndex = -1;
681       for(int i=0;i<output2.length;i++) {
682         if (output2.bytes[output2.offset+i] == payloadSep) {
683           sepIndex = i;
684           break;
685         }
686       }
687       assert sepIndex != -1;
688       final int payloadLen = output2.length - sepIndex - 1;
689       spare.copyUTF8Bytes(output2.bytes, output2.offset, sepIndex);
690       BytesRef payload = new BytesRef(payloadLen);
691       System.arraycopy(output2.bytes, sepIndex+1, payload.bytes, 0, payloadLen);
692       payload.length = payloadLen;
693       result = new LookupResult(spare.toString(), decodeWeight(output1), payload);
694     } else {
695       spare.copyUTF8Bytes(output2);
696       result = new LookupResult(spare.toString(), decodeWeight(output1));
697     }
698 
699     return result;
700   }

702   private boolean sameSurfaceForm(BytesRef key, BytesRef output2) {
703     if (hasPayloads) {
704       // output2 has at least PAYLOAD_SEP byte:
705       if (key.length >= output2.length) {
706         return false;
707       }
708       for(int i=0;i<key.length;i++) {
709         if (key.bytes[key.offset+i] != output2.bytes[output2.offset+i]) {
710           return false;
711         }
712       }
713       return output2.bytes[output2.offset + key.length] == payloadSep;
714     } else {
715       return key.bytesEquals(output2);
716     }
717   }

719   @Override
720   public List<LookupResult> lookup(final CharSequence key, Set<BytesRef> contexts, boolean onlyMorePopular, int num) {
721     assert num > 0;
722 
723     if (onlyMorePopular) {
724       throw new IllegalArgumentException("this suggester only works with onlyMorePopular=false");
725     }
726     if (fst == null) {
727       return Collections.emptyList();
728     }
729 
730     //System.out.println("lookup key=" + key + " num=" + num);
731     for (int i = 0; i < key.length(); i++) {
732       if (key.charAt(i) == holeCharacter) {
733         throw new IllegalArgumentException("lookup key cannot contain HOLE character U+001E; this character is reserved");
734       }
735       if (key.charAt(i) == sepLabel) {
736         throw new IllegalArgumentException("lookup key cannot contain unit separator character U+001F; this character is reserved");
737       }
738     }
739     final BytesRef utf8Key = new BytesRef(key);
740     try {
741 
742       Automaton lookupAutomaton = toLookupAutomaton(key);
743 
744       final CharsRefBuilder spare = new CharsRefBuilder();
745 
746       //System.out.println("  now intersect exactFirst=" + exactFirst);
747     
748       // Intersect automaton w/ suggest wFST and get all
749       // prefix starting nodes & their outputs:
750       //final PathIntersector intersector = getPathIntersector(lookupAutomaton, fst);
751 
752       //System.out.println("  prefixPaths: " + prefixPaths.size());
753 
754       BytesReader bytesReader = fst.getBytesReader();
755 
756       FST.Arc<Pair<Long,BytesRef>> scratchArc = new FST.Arc<>();
757 
758       final List<LookupResult> results = new ArrayList<>();
759 
760       List<FSTUtil.Path<Pair<Long,BytesRef>>> prefixPaths = FSTUtil.intersectPrefixPaths(convertAutomaton(lookupAutomaton), fst);
761 
762       if (exactFirst) {
763 
764         int count = 0;
765         for (FSTUtil.Path<Pair<Long,BytesRef>> path : prefixPaths) {
766           if (fst.findTargetArc(endByte, path.fstNode, scratchArc, bytesReader) != null) {
767             // This node has END_BYTE arc leaving, meaning it's an
768             // "exact" match:
769             count++;
770           }
771         }
772 
773         // Searcher just to find the single exact only
774         // match, if present:
775         Util.TopNSearcher<Pair<Long,BytesRef>> searcher;
776         searcher = new Util.TopNSearcher<>(fst, count * maxSurfaceFormsPerAnalyzedForm, count * maxSurfaceFormsPerAnalyzedForm, weightComparator);
777 
778         // NOTE: we could almost get away with only using
779         // the first start node.  The only catch is if
780         // maxSurfaceFormsPerAnalyzedForm had kicked in and
781         // pruned our exact match from one of these nodes
782         // ...:
783         for (FSTUtil.Path<Pair<Long,BytesRef>> path : prefixPaths) {
784           if (fst.findTargetArc(endByte, path.fstNode, scratchArc, bytesReader) != null) {
785             // This node has END_BYTE arc leaving, meaning it's an
786             // "exact" match:
787             searcher.addStartPaths(scratchArc, fst.outputs.add(path.output, scratchArc.output), false, path.input);
788           }
789         }
790 
791         Util.TopResults<Pair<Long,BytesRef>> completions = searcher.search();
792 
793         // NOTE: this is rather inefficient: we enumerate
794         // every matching "exactly the same analyzed form"
795         // path, and then do linear scan to see if one of
796         // these exactly matches the input.  It should be
797         // possible (though hairy) to do something similar
798         // to getByOutput, since the surface form is encoded
799         // into the FST output, so we more efficiently hone
800         // in on the exact surface-form match.  Still, I
801         // suspect very little time is spent in this linear
802         // seach: it's bounded by how many prefix start
803         // nodes we have and the
804         // maxSurfaceFormsPerAnalyzedForm:
805         for(Result<Pair<Long,BytesRef>> completion : completions) {
806           BytesRef output2 = completion.output.output2;
807           if (sameSurfaceForm(utf8Key, output2)) {
808             results.add(getLookupResult(completion.output.output1, output2, spare));
809             break;
810           }
811         }
812 
813         if (results.size() == num) {
814           // That was quick:
815           return results;
816         }
817       }
818 
819       Util.TopNSearcher<Pair<Long,BytesRef>> searcher;
820       searcher = new Util.TopNSearcher<Pair<Long,BytesRef>>(fst,
821                                                             num - results.size(),
822                                                             num * maxAnalyzedPathsForOneInput,
823                                                             weightComparator) {
824         private final Set<BytesRef> seen = new HashSet<>();
825 
826         @Override
827         protected boolean acceptResult(IntsRef input, Pair<Long,BytesRef> output) {
828 
829           // Dedup: when the input analyzes to a graph we
830           // can get duplicate surface forms:
831           if (seen.contains(output.output2)) {
832             return false;
833           }
834           seen.add(output.output2);
835           
836           if (!exactFirst) {
837             return true;
838           } else {
839             // In exactFirst mode, don't accept any paths
840             // matching the surface form since that will
841             // create duplicate results:
842             if (sameSurfaceForm(utf8Key, output.output2)) {
843               // We found exact match, which means we should
844               // have already found it in the first search:
845               assert results.size() == 1;
846               return false;
847             } else {
848               return true;
849             }
850           }
851         }
852       };
853 
854       prefixPaths = getFullPrefixPaths(prefixPaths, lookupAutomaton, fst);
855       
856       for (FSTUtil.Path<Pair<Long,BytesRef>> path : prefixPaths) {
857         searcher.addStartPaths(path.fstNode, path.output, true, path.input);
858       }
859 
860       TopResults<Pair<Long,BytesRef>> completions = searcher.search();
861 
862       for(Result<Pair<Long,BytesRef>> completion : completions) {
863 
864         LookupResult result = getLookupResult(completion.output.output1, completion.output.output2, spare);
865 
866         // TODO: for fuzzy case would be nice to return
867         // how many edits were required
868 
869         //System.out.println("    result=" + result);
870         results.add(result);
871 
872         if (results.size() == num) {
873           // In the exactFirst=true case the search may
874           // produce one extra path
875           break;
876         }
877       }
878 
879       return results;
880     } catch (IOException bogus) {
881       throw new RuntimeException(bogus);
882     }
883   }

1004         private int maxSurfaceFormsPerAnalyzedForm;
1005         private IntsRefBuilder scratchInts = new IntsRefBuilder();
1006         private final PairOutputs<Long, BytesRef> outputs;
1007         private boolean hasPayloads;
1008         private BytesRefBuilder analyzed = new BytesRefBuilder();
1009         private final SurfaceFormAndPayload[] surfaceFormsAndPayload;
1010         private int count;
1011         private ObjectIntOpenHashMap<BytesRef> seenSurfaceForms = HppcMaps.Object.Integer.ensureNoNullKeys(256, 0.75f);
1012         private int payloadSep;

1014         public XBuilder(int maxSurfaceFormsPerAnalyzedForm, boolean hasPayloads, int payloadSep) {
1015             this.payloadSep = payloadSep;
1016             this.outputs = new PairOutputs<>(PositiveIntOutputs.getSingleton(), ByteSequenceOutputs.getSingleton());
1017             this.builder = new Builder<>(FST.INPUT_TYPE.BYTE1, outputs);
1018             this.maxSurfaceFormsPerAnalyzedForm = maxSurfaceFormsPerAnalyzedForm;

1028         private final static class SurfaceFormAndPayload implements Comparable<SurfaceFormAndPayload> {
1029             BytesRef payload;
1030             long weight;
1031             
1032             public SurfaceFormAndPayload(BytesRef payload, long cost) {
1033                 super();
1034                 this.payload = payload;
1035                 this.weight = cost;
1036             }
1037 
1038             @Override
1039             public int compareTo(SurfaceFormAndPayload o) {
1040                 int res = compare(weight, o.weight);
1041                 if (res == 0 ){
1042                     return payload.compareTo(o.payload);
1043                 }
1044                 return res;
1045             }
1046             public static int compare(long x, long y) {
1047                 return (x < y) ? -1 : ((x == y) ? 0 : 1);
1048             }
1049         }

1051         public void addSurface(BytesRef surface, BytesRef payload, long cost) throws IOException {
1052             int surfaceIndex = -1;
1053             long encodedWeight = cost == -1 ? cost : encodeWeight(cost);
1054             /*
1055              * we need to check if we have seen this surface form, if so only use the 
1056              * the surface form with the highest weight and drop the rest no matter if 
1057              * the payload differs.
1058              */
1059             if (count >= maxSurfaceFormsPerAnalyzedForm) {
1060                 // More than maxSurfaceFormsPerAnalyzedForm
1061                 // dups: skip the rest:
1062                 return;
1063             }
1064             BytesRef surfaceCopy;
1065             if (count > 0 && seenSurfaceForms.containsKey(surface)) {
1066                 surfaceIndex = seenSurfaceForms.lget();
1067                 SurfaceFormAndPayload surfaceFormAndPayload = surfaceFormsAndPayload[surfaceIndex];
1068                 if (encodedWeight >= surfaceFormAndPayload.weight) {
1069                     return;
1070                 }
1071                 surfaceCopy = BytesRef.deepCopyOf(surface);
1072             } else {
1073                 surfaceIndex = count++;
1074                 surfaceCopy = BytesRef.deepCopyOf(surface);
1075                 seenSurfaceForms.put(surfaceCopy, surfaceIndex);
1076             }
1077            
1078             BytesRef payloadRef;
1079             if (!hasPayloads) {
1080                 payloadRef = surfaceCopy;
1081             } else {
1082                 int len = surface.length + 1 + payload.length;
1083                 final BytesRef br = new BytesRef(len);
1084                 System.arraycopy(surface.bytes, surface.offset, br.bytes, 0, surface.length);
1085                 br.bytes[surface.length] = (byte) payloadSep;
1086                 System.arraycopy(payload.bytes, payload.offset, br.bytes, surface.length + 1, payload.length);
1087                 br.length = len;
1088                 payloadRef = br;
1089             }
1090             if (surfaceFormsAndPayload[surfaceIndex] == null) {
1091                 surfaceFormsAndPayload[surfaceIndex] = new SurfaceFormAndPayload(payloadRef, encodedWeight);
1092             } else {
1093                 surfaceFormsAndPayload[surfaceIndex].payload = payloadRef;
1094                 surfaceFormsAndPayload[surfaceIndex].weight = encodedWeight;
1095             }
1096         }

1098         public void finishTerm(long defaultWeight) throws IOException {
1099             ArrayUtil.timSort(surfaceFormsAndPayload, 0, count);
1100             int deduplicator = 0;
1101             analyzed.append((byte) 0);
1102             analyzed.setLength(analyzed.length() + 1);
1103             analyzed.grow(analyzed.length());
1104             for (int i = 0; i < count; i++) {
1105                 analyzed.setByteAt(analyzed.length() - 1, (byte) deduplicator++);
1106                 Util.toIntsRef(analyzed.get(), scratchInts);
1107                 SurfaceFormAndPayload candiate = surfaceFormsAndPayload[i];
1108                 long cost = candiate.weight == -1 ? encodeWeight(Math.min(Integer.MAX_VALUE, defaultWeight)) : candiate.weight;
1109                 builder.add(scratchInts.get(), outputs.newPair(cost, candiate.payload));
1110             }
1111             seenSurfaceForms.clear();
1112             count = 0;
1113         }
