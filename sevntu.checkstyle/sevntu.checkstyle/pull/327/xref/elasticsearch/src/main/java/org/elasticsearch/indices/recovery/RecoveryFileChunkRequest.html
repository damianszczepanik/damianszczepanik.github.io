20  package org.elasticsearch.indices.recovery;
21  
22  import org.apache.lucene.util.Version;
23  import org.elasticsearch.common.Nullable;
24  import org.elasticsearch.common.bytes.BytesReference;
25  import org.elasticsearch.common.io.stream.StreamInput;
26  import org.elasticsearch.common.io.stream.StreamOutput;
27  import org.elasticsearch.common.lucene.Lucene;
28  import org.elasticsearch.index.shard.ShardId;
29  import org.elasticsearch.index.store.StoreFileMetaData;
30  import org.elasticsearch.transport.TransportRequest;
31  
32  import java.io.IOException;
33  
34  /**
35   *
36   */
37  public final class RecoveryFileChunkRequest extends TransportRequest {  // public for testing
38      private boolean lastChunk;
39      private long recoveryId;
40      private ShardId shardId;
41      private long position;
42      private BytesReference content;
43      private StoreFileMetaData metaData;
44  
45      RecoveryFileChunkRequest() {
46      }
47  
48      public RecoveryFileChunkRequest(long recoveryId, ShardId shardId, StoreFileMetaData metaData, long position, BytesReference content, boolean lastChunk) {
49          this.recoveryId = recoveryId;
50          this.shardId = shardId;
51          this.metaData = metaData;
52          this.position = position;
53          this.content = content;
54          this.lastChunk = lastChunk;
55      }
56  
57      public long recoveryId() {
58          return this.recoveryId;
59      }
60  
61      public ShardId shardId() {
62          return shardId;
63      }
64  
65      public String name() {
66          return metaData.name();
67      }
68  
69      public long position() {
70          return position;
71      }
72  
73      @Nullable
74      public String checksum() {
75          return metaData.checksum();
76      }
77  
78      public long length() {
79          return metaData.length();
80      }
81  
82      public BytesReference content() {
83          return content;
84      }
85  
86      @Override
87      public void readFrom(StreamInput in) throws IOException {
88          super.readFrom(in);
89          recoveryId = in.readLong();
90          shardId = ShardId.readShardId(in);
91          String name = in.readString();
92          position = in.readVLong();
93          long length = in.readVLong();
94          String checksum = in.readOptionalString();
95          content = in.readBytesReference();
96          Version writtenBy = null;
97          String versionString = in.readOptionalString();
98          writtenBy = Lucene.parseVersionLenient(versionString, null);
99          metaData = new StoreFileMetaData(name, length, checksum, writtenBy);
100         lastChunk = in.readBoolean();
101     }
102 
103     @Override
104     public void writeTo(StreamOutput out) throws IOException {
105         super.writeTo(out);
106         out.writeLong(recoveryId);
107         shardId.writeTo(out);
108         out.writeString(metaData.name());
109         out.writeVLong(position);
110         out.writeVLong(metaData.length());
111         out.writeOptionalString(metaData.checksum());
112         out.writeBytesReference(content);
113         out.writeOptionalString(metaData.writtenBy() == null ? null : metaData.writtenBy().toString());
114         out.writeBoolean(lastChunk);
115     }
116 
117     @Override
118     public String toString() {
119         return shardId + ": name='" + name() + '\'' +
120                 ", position=" + position +
121                 ", length=" + length();
122     }
123 
124     public StoreFileMetaData metadata() {
125         return metaData;
126     }
127 
128     /**
129      * Returns <code>true</code> if this chunk is the last chunk in the stream.
130      */
131     public boolean lastChunk() {
132         return lastChunk;
133     }
134 }
