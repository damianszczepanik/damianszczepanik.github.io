20  package org.elasticsearch.search.aggregations.metrics.percentiles;
21  
22  import com.carrotsearch.hppc.DoubleArrayList;
23  
24  import org.elasticsearch.common.xcontent.XContentParser;
25  import org.elasticsearch.search.SearchParseException;
26  import org.elasticsearch.search.aggregations.Aggregator;
27  import org.elasticsearch.search.aggregations.AggregatorFactory;
28  import org.elasticsearch.search.aggregations.support.ValuesSource;
29  import org.elasticsearch.search.aggregations.support.ValuesSource.Numeric;
30  import org.elasticsearch.search.aggregations.support.ValuesSourceConfig;
31  import org.elasticsearch.search.aggregations.support.ValuesSourceParser;
32  import org.elasticsearch.search.internal.SearchContext;
33  
34  import java.io.IOException;
35  import java.util.Arrays;
36  
37  public abstract class AbstractPercentilesParser implements Aggregator.Parser {
38  
39      private boolean formattable;
40  
41      public AbstractPercentilesParser(boolean formattable) {
42          this.formattable = formattable;
43      }
44  
45      @Override
46      public AggregatorFactory parse(String aggregationName, XContentParser parser, SearchContext context) throws IOException {
47      
48          ValuesSourceParser<ValuesSource.Numeric> vsParser = ValuesSourceParser.numeric(aggregationName, InternalPercentiles.TYPE, context)
49                  .formattable(formattable).build();
50      
51          double[] keys = null;
52          boolean keyed = true;
53          double compression = 100;
54      
55          XContentParser.Token token;
56          String currentFieldName = null;
57          while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
58              if (token == XContentParser.Token.FIELD_NAME) {
59                  currentFieldName = parser.currentName();
60              } else if (vsParser.token(currentFieldName, token, parser)) {
61                  continue;
62              } else if (token == XContentParser.Token.START_ARRAY) {
63                  if (keysFieldName().equals(currentFieldName)) {
64                      DoubleArrayList values = new DoubleArrayList(10);
65                      while ((token = parser.nextToken()) != XContentParser.Token.END_ARRAY) {
66                          double value = parser.doubleValue();
67                          values.add(value);
68                      }
69                      keys = values.toArray();
70                      Arrays.sort(keys);
71                  } else {
72                      throw new SearchParseException(context, "Unknown key for a " + token + " in [" + aggregationName + "]: [" + currentFieldName + "].");
73                  }
74              } else if (token == XContentParser.Token.VALUE_BOOLEAN) {
75                  if ("keyed".equals(currentFieldName)) {
76                      keyed = parser.booleanValue();
77                  } else {
78                      throw new SearchParseException(context, "Unknown key for a " + token + " in [" + aggregationName + "]: [" + currentFieldName + "].");
79                  }
80              } else if (token == XContentParser.Token.VALUE_NUMBER) {
81                  if ("compression".equals(currentFieldName)) {
82                      compression = parser.doubleValue();
83                  } else {
84                      throw new SearchParseException(context, "Unknown key for a " + token + " in [" + aggregationName + "]: [" + currentFieldName + "].");
85                  }
86              } else {
87                  throw new SearchParseException(context, "Unexpected token " + token + " in [" + aggregationName + "].");
88              }
89          }
90          return buildFactory(context, aggregationName, vsParser.config(), keys, compression, keyed);
91      }
92  
93      protected abstract AggregatorFactory buildFactory(SearchContext context, String aggregationName, ValuesSourceConfig<Numeric> config, double[] cdfValues,
94              double compression, boolean keyed);
95  
96      protected abstract String keysFieldName();
97  
98  }
