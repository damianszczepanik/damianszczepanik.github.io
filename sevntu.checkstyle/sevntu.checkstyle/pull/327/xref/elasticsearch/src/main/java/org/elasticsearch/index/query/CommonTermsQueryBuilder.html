20  package org.elasticsearch.index.query;
21  
22  import org.elasticsearch.ElasticsearchIllegalArgumentException;
23  import org.elasticsearch.common.xcontent.XContentBuilder;
24  
25  import java.io.IOException;
26  
27  /**
28   * CommonTermsQuery query is a query that executes high-frequency terms in a
29   * optional sub-query to prevent slow queries due to "common" terms like
30   * stopwords. This query basically builds 2 queries off the {@link #add(Term)
31   * added} terms where low-frequency terms are added to a required boolean clause
32   * and high-frequency terms are added to an optional boolean clause. The
33   * optional clause is only executed if the required "low-frequency' clause
34   * matches. Scores produced by this query will be slightly different to plain
35   * {@link BooleanQuery} scorer mainly due to differences in the
36   * {@link Similarity#coord(int,int) number of leave queries} in the required
37   * boolean clause. In the most cases high-frequency terms are unlikely to
38   * significantly contribute to the document score unless at least one of the
39   * low-frequency terms are matched such that this query can improve query
40   * execution times significantly if applicable.
41   * <p>
42   */
43  public class CommonTermsQueryBuilder extends BaseQueryBuilder implements BoostableQueryBuilder<CommonTermsQueryBuilder> {
44  
45      public static enum Operator {
46          OR, AND
47      }
48  
49      private final String name;
50  
51      private final Object text;
52  
53      private Operator highFreqOperator = null;
54  
55      private Operator lowFreqOperator = null;
56  
57      private String analyzer = null;
58  
59      private Float boost = null;
60  
61      private String lowFreqMinimumShouldMatch = null;
62  
63      private String highFreqMinimumShouldMatch = null;
64  
65      private Boolean disableCoords = null;
66  
67      private Float cutoffFrequency = null;
68  
69      private String queryName;
70  
71      /**
72       * Constructs a new common terms query.
73       */
74      public CommonTermsQueryBuilder(String name, Object text) {
75          if (name == null) {
76              throw new ElasticsearchIllegalArgumentException("Field name must not be null");
77          }
78          if (text == null) {
79              throw new ElasticsearchIllegalArgumentException("Query must not be null");
80          }
81          this.text = text;
82          this.name = name;
83      }
84  
85      /**
86       * Sets the operator to use for terms with a high document frequency
87       * (greater than or equal to {@link #cutoffFrequency(float)}. Defaults to
88       * <tt>AND</tt>.
89       */
90      public CommonTermsQueryBuilder highFreqOperator(Operator operator) {
91          this.highFreqOperator = operator;
92          return this;
93      }
94  
95      /**
96       * Sets the operator to use for terms with a low document frequency (less
97       * than {@link #cutoffFrequency(float)}. Defaults to <tt>AND</tt>.
98       */
99      public CommonTermsQueryBuilder lowFreqOperator(Operator operator) {
100         this.lowFreqOperator = operator;
101         return this;
102     }
103 
104     /**
105      * Explicitly set the analyzer to use. Defaults to use explicit mapping
106      * config for the field, or, if not set, the default search analyzer.
107      */
108     public CommonTermsQueryBuilder analyzer(String analyzer) {
109         this.analyzer = analyzer;
110         return this;
111     }
112 
113     /**
114      * Set the boost to apply to the query.
115      */
116     @Override
117     public CommonTermsQueryBuilder boost(float boost) {
118         this.boost = boost;
119         return this;
120     }
121 
122     /**
123      * Sets the cutoff document frequency for high / low frequent terms. A value
124      * in [0..1] (or absolute number >=1) representing the maximum threshold of
125      * a terms document frequency to be considered a low frequency term.
126      * Defaults to
127      * <tt>{@value CommonTermsQueryParser#DEFAULT_MAX_TERM_DOC_FREQ}</tt>
128      */
129     public CommonTermsQueryBuilder cutoffFrequency(float cutoffFrequency) {
130         this.cutoffFrequency = cutoffFrequency;
131         return this;
132     }
133 
134     /**
135      * Sets the minimum number of high frequent query terms that need to match in order to
136      * produce a hit when there are no low frequen terms.
137      */
138     public CommonTermsQueryBuilder highFreqMinimumShouldMatch(String highFreqMinimumShouldMatch) {
139         this.highFreqMinimumShouldMatch = highFreqMinimumShouldMatch;
140         return this;
141     }
142 
143     /**
144      * Sets the minimum number of low frequent query terms that need to match in order to
145      * produce a hit.
146      */
147     public CommonTermsQueryBuilder lowFreqMinimumShouldMatch(String lowFreqMinimumShouldMatch) {
148         this.lowFreqMinimumShouldMatch = lowFreqMinimumShouldMatch;
149         return this;
150     }
151 
152     /**
153      * Sets the query name for the filter that can be used when searching for matched_filters per hit.
154      */
155     public CommonTermsQueryBuilder queryName(String queryName) {
156         this.queryName = queryName;
157         return this;
158     }
159 
160     @Override
161     public void doXContent(XContentBuilder builder, Params params) throws IOException {
162         builder.startObject(CommonTermsQueryParser.NAME);
163         builder.startObject(name);
164 
165         builder.field("query", text);
166         if (disableCoords != null) {
167             builder.field("disable_coords", disableCoords);
168         }
169         if (highFreqOperator != null) {
170             builder.field("high_freq_operator", highFreqOperator.toString());
171         }
172         if (lowFreqOperator != null) {
173             builder.field("low_freq_operator", lowFreqOperator.toString());
174         }
175         if (analyzer != null) {
176             builder.field("analyzer", analyzer);
177         }
178         if (boost != null) {
179             builder.field("boost", boost);
180         }
181         if (cutoffFrequency != null) {
182             builder.field("cutoff_frequency", cutoffFrequency);
183         }
184         if (lowFreqMinimumShouldMatch != null || highFreqMinimumShouldMatch != null) {
185             builder.startObject("minimum_should_match");
186             if (lowFreqMinimumShouldMatch != null) {
187                 builder.field("low_freq", lowFreqMinimumShouldMatch);
188             }
189             if (highFreqMinimumShouldMatch != null) {
190                 builder.field("high_freq", highFreqMinimumShouldMatch);
191             }
192             builder.endObject();
193         }
194         if (queryName != null) {
195             builder.field("_name", queryName);
196         }
197 
198         builder.endObject();
199         builder.endObject();
200     }
201 }
