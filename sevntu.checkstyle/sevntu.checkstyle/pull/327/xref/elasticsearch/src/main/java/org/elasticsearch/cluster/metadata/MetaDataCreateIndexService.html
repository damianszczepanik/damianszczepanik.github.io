106     @Inject
107     public MetaDataCreateIndexService(Settings settings, Environment environment, ThreadPool threadPool, ClusterService clusterService,
108                                       IndicesService indicesService, AllocationService allocationService, MetaDataService metaDataService,
109                                       Version version, @RiverIndexName String riverIndexName, AliasValidator aliasValidator,
110                                       Set<IndexTemplateFilter> indexTemplateFilters, NodeEnvironment nodeEnv) {
111         super(settings);
112         this.environment = environment;
113         this.threadPool = threadPool;
114         this.clusterService = clusterService;
115         this.indicesService = indicesService;
116         this.allocationService = allocationService;
117         this.metaDataService = metaDataService;
118         this.version = version;
119         this.riverIndexName = riverIndexName;
120         this.aliasValidator = aliasValidator;
121         this.nodeEnv = nodeEnv;
122 
123         if (indexTemplateFilters.isEmpty()) {
124             this.indexTemplateFilter = DEFAULT_INDEX_TEMPLATE_FILTER;
125         } else {
126             IndexTemplateFilter[] templateFilters = new IndexTemplateFilter[indexTemplateFilters.size() + 1];
127             templateFilters[0] = DEFAULT_INDEX_TEMPLATE_FILTER;
128             int i = 1;
129             for (IndexTemplateFilter indexTemplateFilter : indexTemplateFilters) {
130                 templateFilters[i++] = indexTemplateFilter;
131             }
132             this.indexTemplateFilter = new IndexTemplateFilter.Compound(templateFilters);
133         }
134     }

136     public void createIndex(final CreateIndexClusterStateUpdateRequest request, final ActionListener<ClusterStateUpdateResponse> listener) {
137 
138         // we lock here, and not within the cluster service callback since we don't want to
139         // block the whole cluster state handling
140         final Semaphore mdLock = metaDataService.indexMetaDataLock(request.index());
141 
142         // quick check to see if we can acquire a lock, otherwise spawn to a thread pool
143         if (mdLock.tryAcquire()) {
144             createIndex(request, listener, mdLock);
145             return;
146         }
147         threadPool.executor(ThreadPool.Names.MANAGEMENT).execute(new ActionRunnable(listener) {
148             @Override
149             public void doRun() throws InterruptedException {
150                 if (!mdLock.tryAcquire(request.masterNodeTimeout().nanos(), TimeUnit.NANOSECONDS)) {
151                     listener.onFailure(new ProcessClusterEventTimeoutException(request.masterNodeTimeout(), "acquire index lock"));
152                     return;
153                 }
154                 createIndex(request, listener, mdLock);
155             }
156         });
157     }

159     public void validateIndexName(String index, ClusterState state) throws ElasticsearchException {
160         if (state.routingTable().hasIndex(index)) {
161             throw new IndexAlreadyExistsException(new Index(index));
162         }
163         if (state.metaData().hasIndex(index)) {
164             throw new IndexAlreadyExistsException(new Index(index));
165         }
166         if (!Strings.validFileName(index)) {
167             throw new InvalidIndexNameException(new Index(index), index, "must not contain the following characters " + Strings.INVALID_FILENAME_CHARS);
168         }
169         if (index.contains("#")) {
170             throw new InvalidIndexNameException(new Index(index), index, "must not contain '#'");
171         }
172         if (!index.equals(riverIndexName) && index.charAt(0) == '_') {
173             throw new InvalidIndexNameException(new Index(index), index, "must not start with '_'");
174         }
175         if (!index.toLowerCase(Locale.ROOT).equals(index)) {
176             throw new InvalidIndexNameException(new Index(index), index, "must be lowercase");
177         }
178         int byteCount = 0;
179         try {
180             byteCount = index.getBytes("UTF-8").length;
181         } catch (UnsupportedEncodingException e) {
182             // UTF-8 should always be supported, but rethrow this if it is not for some reason
183             throw new ElasticsearchException("Unable to determine length of index name", e);
184         }
185         if (byteCount > MAX_INDEX_NAME_BYTES) {
186             throw new InvalidIndexNameException(new Index(index), index,
187                     "index name is too long, (" + byteCount +
188                     " > " + MAX_INDEX_NAME_BYTES + ")");
189         }
190         if (state.metaData().aliases().containsKey(index)) {
191             throw new InvalidIndexNameException(new Index(index), index, "already exists as alias");
192         }
193     }

195     private void createIndex(final CreateIndexClusterStateUpdateRequest request, final ActionListener<ClusterStateUpdateResponse> listener, final Semaphore mdLock) {
196 
197         ImmutableSettings.Builder updatedSettingsBuilder = ImmutableSettings.settingsBuilder();
198         for (Map.Entry<String, String> entry : request.settings().getAsMap().entrySet()) {
199             if (!entry.getKey().startsWith("index.")) {
200                 updatedSettingsBuilder.put("index." + entry.getKey(), entry.getValue());
201             } else {
202                 updatedSettingsBuilder.put(entry.getKey(), entry.getValue());
203             }
204         }
205         request.settings(updatedSettingsBuilder.build());
206 
207         clusterService.submitStateUpdateTask("create-index [" + request.index() + "], cause [" + request.cause() + "]", Priority.URGENT, new AckedClusterStateUpdateTask<ClusterStateUpdateResponse>(request, listener) {
208 
209             @Override
210             protected ClusterStateUpdateResponse newResponse(boolean acknowledged) {
211                 return new ClusterStateUpdateResponse(acknowledged);
212             }
213 
214             @Override
215             public void onAllNodesAcked(@Nullable Throwable t) {
216                 mdLock.release();
217                 super.onAllNodesAcked(t);
218             }
219 
220             @Override
221             public void onAckTimeout() {
222                 mdLock.release();
223                 super.onAckTimeout();
224             }
225 
226             @Override
227             public void onFailure(String source, Throwable t) {
228                 mdLock.release();
229                 super.onFailure(source, t);
230             }
231 
232             @Override
233             public ClusterState execute(ClusterState currentState) throws Exception {
234                 boolean indexCreated = false;
235                 String removalReason = null;
236                 try {
237                     validate(request, currentState);
238 
239                     for (Alias alias : request.aliases()) {
240                         aliasValidator.validateAlias(alias, request.index(), currentState.metaData());
241                     }
242 
243                     // we only find a template when its an API call (a new index)
244                     // find templates, highest order are better matching
245                     List<IndexTemplateMetaData> templates = findTemplates(request, currentState, indexTemplateFilter);
246 
247                     Map<String, Custom> customs = Maps.newHashMap();
248 
249                     // add the request mapping
250                     Map<String, Map<String, Object>> mappings = Maps.newHashMap();
251 
252                     Map<String, AliasMetaData> templatesAliases = Maps.newHashMap();
253 
254                     List<String> templateNames = Lists.newArrayList();
255 
256                     for (Map.Entry<String, String> entry : request.mappings().entrySet()) {
257                         mappings.put(entry.getKey(), parseMapping(entry.getValue()));
258                     }
259 
260                     for (Map.Entry<String, Custom> entry : request.customs().entrySet()) {
261                         customs.put(entry.getKey(), entry.getValue());
262                     }
263 
264                     // apply templates, merging the mappings into the request mapping if exists
265                     for (IndexTemplateMetaData template : templates) {
266                         templateNames.add(template.getName());
267                         for (ObjectObjectCursor<String, CompressedString> cursor : template.mappings()) {
268                             if (mappings.containsKey(cursor.key)) {
269                                 XContentHelper.mergeDefaults(mappings.get(cursor.key), parseMapping(cursor.value.string()));
270                             } else {
271                                 mappings.put(cursor.key, parseMapping(cursor.value.string()));
272                             }
273                         }
274                         // handle custom
275                         for (ObjectObjectCursor<String, Custom> cursor : template.customs()) {
276                             String type = cursor.key;
277                             IndexMetaData.Custom custom = cursor.value;
278                             IndexMetaData.Custom existing = customs.get(type);
279                             if (existing == null) {
280                                 customs.put(type, custom);
281                             } else {
282                                 IndexMetaData.Custom merged = IndexMetaData.lookupFactorySafe(type).merge(existing, custom);
283                                 customs.put(type, merged);
284                             }
285                         }
286                         //handle aliases
287                         for (ObjectObjectCursor<String, AliasMetaData> cursor : template.aliases()) {
288                             AliasMetaData aliasMetaData = cursor.value;
289                             //if an alias with same name came with the create index request itself,
290                             // ignore this one taken from the index template
291                             if (request.aliases().contains(new Alias(aliasMetaData.alias()))) {
292                                 continue;
293                             }
294                             //if an alias with same name was already processed, ignore this one
295                             if (templatesAliases.containsKey(cursor.key)) {
296                                 continue;
297                             }
298 
299                             //Allow templatesAliases to be templated by replacing a token with the name of the index that we are applying it to
300                             if (aliasMetaData.alias().contains("{index}")) {
301                                 String templatedAlias = aliasMetaData.alias().replace("{index}", request.index());
302                                 aliasMetaData = AliasMetaData.newAliasMetaData(aliasMetaData, templatedAlias);
303                             }
304 
305                             aliasValidator.validateAliasMetaData(aliasMetaData, request.index(), currentState.metaData());
306                             templatesAliases.put(aliasMetaData.alias(), aliasMetaData);
307                         }
308                     }
309 
310                     // now add config level mappings
311                     Path mappingsDir = environment.configFile().resolve("mappings");
312                     if (Files.isDirectory(mappingsDir)) {
313                         // first index level
314                         Path indexMappingsDir = mappingsDir.resolve(request.index());
315                         if (Files.isDirectory(indexMappingsDir)) {
316                             addMappings(mappings, indexMappingsDir);
317                         }
318 
319                         // second is the _default mapping
320                         Path defaultMappingsDir = mappingsDir.resolve("_default");
321                         if (Files.isDirectory(defaultMappingsDir)) {
322                             addMappings(mappings, defaultMappingsDir);
323                         }
324                     }
325 
326                     ImmutableSettings.Builder indexSettingsBuilder = settingsBuilder();
327                     // apply templates, here, in reverse order, since first ones are better matching
328                     for (int i = templates.size() - 1; i >= 0; i--) {
329                         indexSettingsBuilder.put(templates.get(i).settings());
330                     }
331                     // now, put the request settings, so they override templates
332                     indexSettingsBuilder.put(request.settings());
333                     if (request.index().equals(ScriptService.SCRIPT_INDEX)) {
334                         indexSettingsBuilder.put(SETTING_NUMBER_OF_SHARDS, settings.getAsInt(SETTING_NUMBER_OF_SHARDS, 1));
335                     } else {
336                         if (indexSettingsBuilder.get(SETTING_NUMBER_OF_SHARDS) == null) {
337                             if (request.index().equals(riverIndexName)) {
338                                 indexSettingsBuilder.put(SETTING_NUMBER_OF_SHARDS, settings.getAsInt(SETTING_NUMBER_OF_SHARDS, 1));
339                             } else {
340                                 indexSettingsBuilder.put(SETTING_NUMBER_OF_SHARDS, settings.getAsInt(SETTING_NUMBER_OF_SHARDS, 5));
341                             }
342                         }
343                     }
344                     if (request.index().equals(ScriptService.SCRIPT_INDEX)) {
345                         indexSettingsBuilder.put(SETTING_NUMBER_OF_REPLICAS, settings.getAsInt(SETTING_NUMBER_OF_REPLICAS, 0));
346                         indexSettingsBuilder.put(SETTING_AUTO_EXPAND_REPLICAS, "0-all");
347                     }
348                     else {
349                         if (indexSettingsBuilder.get(SETTING_NUMBER_OF_REPLICAS) == null) {
350                             if (request.index().equals(riverIndexName)) {
351                                 indexSettingsBuilder.put(SETTING_NUMBER_OF_REPLICAS, settings.getAsInt(SETTING_NUMBER_OF_REPLICAS, 1));
352                             } else {
353                                 indexSettingsBuilder.put(SETTING_NUMBER_OF_REPLICAS, settings.getAsInt(SETTING_NUMBER_OF_REPLICAS, 1));
354                             }
355                         }
356                     }
357 
358                     if (settings.get(SETTING_AUTO_EXPAND_REPLICAS) != null && indexSettingsBuilder.get(SETTING_AUTO_EXPAND_REPLICAS) == null) {
359                         indexSettingsBuilder.put(SETTING_AUTO_EXPAND_REPLICAS, settings.get(SETTING_AUTO_EXPAND_REPLICAS));
360                     }
361 
362                     if (indexSettingsBuilder.get(SETTING_VERSION_CREATED) == null) {
363                         DiscoveryNodes nodes = currentState.nodes();
364                         final Version createdVersion = Version.smallest(version, nodes.smallestNonClientNodeVersion());
365                         indexSettingsBuilder.put(SETTING_VERSION_CREATED, createdVersion);
366                     }
367 
368                     if (indexSettingsBuilder.get(SETTING_CREATION_DATE) == null) {
369                         indexSettingsBuilder.put(SETTING_CREATION_DATE, System.currentTimeMillis());
370                     }
371 
372                     indexSettingsBuilder.put(SETTING_UUID, Strings.randomBase64UUID());
373 
374                     Settings actualIndexSettings = indexSettingsBuilder.build();
375 
376                     // Set up everything, now locally create the index to see that things are ok, and apply
377 
378                     // create the index here (on the master) to validate it can be created, as well as adding the mapping
379                     indicesService.createIndex(request.index(), actualIndexSettings, clusterService.localNode().id());
380                     indexCreated = true;
381                     // now add the mappings
382                     IndexService indexService = indicesService.indexServiceSafe(request.index());
383                     MapperService mapperService = indexService.mapperService();
384                     // first, add the default mapping
385                     if (mappings.containsKey(MapperService.DEFAULT_MAPPING)) {
386                         try {
387                             mapperService.merge(MapperService.DEFAULT_MAPPING, new CompressedString(XContentFactory.jsonBuilder().map(mappings.get(MapperService.DEFAULT_MAPPING)).string()), false);
388                         } catch (Exception e) {
389                             removalReason = "failed on parsing default mapping on index creation";
390                             throw new MapperParsingException("mapping [" + MapperService.DEFAULT_MAPPING + "]", e);
391                         }
392                     }
393                     for (Map.Entry<String, Map<String, Object>> entry : mappings.entrySet()) {
394                         if (entry.getKey().equals(MapperService.DEFAULT_MAPPING)) {
395                             continue;
396                         }
397                         try {
398                             // apply the default here, its the first time we parse it
399                             mapperService.merge(entry.getKey(), new CompressedString(XContentFactory.jsonBuilder().map(entry.getValue()).string()), true);
400                         } catch (Exception e) {
401                             removalReason = "failed on parsing mappings on index creation";
402                             throw new MapperParsingException("mapping [" + entry.getKey() + "]", e);
403                         }
404                     }
405 
406                     IndexQueryParserService indexQueryParserService = indexService.queryParserService();
407                     for (Alias alias : request.aliases()) {
408                         if (Strings.hasLength(alias.filter())) {
409                             aliasValidator.validateAliasFilter(alias.name(), alias.filter(), indexQueryParserService);
410                         }
411                     }
412                     for (AliasMetaData aliasMetaData : templatesAliases.values()) {
413                         if (aliasMetaData.filter() != null) {
414                             aliasValidator.validateAliasFilter(aliasMetaData.alias(), aliasMetaData.filter().uncompressed(), indexQueryParserService);
415                         }
416                     }
417 
418                     // now, update the mappings with the actual source
419                     Map<String, MappingMetaData> mappingsMetaData = Maps.newHashMap();
420                     for (DocumentMapper mapper : mapperService.docMappers(true)) {
421                         MappingMetaData mappingMd = new MappingMetaData(mapper);
422                         mappingsMetaData.put(mapper.type(), mappingMd);
423                     }
424 
425                     final IndexMetaData.Builder indexMetaDataBuilder = IndexMetaData.builder(request.index()).settings(actualIndexSettings);
426                     for (MappingMetaData mappingMd : mappingsMetaData.values()) {
427                         indexMetaDataBuilder.putMapping(mappingMd);
428                     }
429 
430                     for (AliasMetaData aliasMetaData : templatesAliases.values()) {
431                         indexMetaDataBuilder.putAlias(aliasMetaData);
432                     }
433                     for (Alias alias : request.aliases()) {
434                         AliasMetaData aliasMetaData = AliasMetaData.builder(alias.name()).filter(alias.filter())
435                                 .indexRouting(alias.indexRouting()).searchRouting(alias.searchRouting()).build();
436                         indexMetaDataBuilder.putAlias(aliasMetaData);
437                     }
438 
439                     for (Map.Entry<String, Custom> customEntry : customs.entrySet()) {
440                         indexMetaDataBuilder.putCustom(customEntry.getKey(), customEntry.getValue());
441                     }
442 
443                     indexMetaDataBuilder.state(request.state());
444 
445                     final IndexMetaData indexMetaData;
446                     try {
447                         indexMetaData = indexMetaDataBuilder.build();
448                     } catch (Exception e) {
449                         removalReason = "failed to build index metadata";
450                         throw e;
451                     }
452 
453                     indexService.indicesLifecycle().beforeIndexAddedToCluster(new Index(request.index()),
454                             indexMetaData.settings());
455 
456                     MetaData newMetaData = MetaData.builder(currentState.metaData())
457                             .put(indexMetaData, false)
458                             .build();
459 
460                     logger.info("[{}] creating index, cause [{}], templates {}, shards [{}]/[{}], mappings {}", request.index(), request.cause(), templateNames, indexMetaData.numberOfShards(), indexMetaData.numberOfReplicas(), mappings.keySet());
461 
462                     ClusterBlocks.Builder blocks = ClusterBlocks.builder().blocks(currentState.blocks());
463                     if (!request.blocks().isEmpty()) {
464                         for (ClusterBlock block : request.blocks()) {
465                             blocks.addIndexBlock(request.index(), block);
466                         }
467                     }
468                     if (request.state() == State.CLOSE) {
469                         blocks.addIndexBlock(request.index(), MetaDataIndexStateService.INDEX_CLOSED_BLOCK);
470                     }
471 
472                     ClusterState updatedState = ClusterState.builder(currentState).blocks(blocks).metaData(newMetaData).build();
473 
474                     if (request.state() == State.OPEN) {
475                         RoutingTable.Builder routingTableBuilder = RoutingTable.builder(updatedState.routingTable())
476                                 .addAsNew(updatedState.metaData().index(request.index()));
477                         RoutingAllocation.Result routingResult = allocationService.reroute(ClusterState.builder(updatedState).routingTable(routingTableBuilder).build());
478                         updatedState = ClusterState.builder(updatedState).routingResult(routingResult).build();
479                     }
480                     removalReason = "cleaning up after validating index on master";
481                     return updatedState;
482                 } finally {
483                     if (indexCreated) {
484                         // Index was already partially created - need to clean up
485                         indicesService.removeIndex(request.index(), removalReason != null ? removalReason : "failed to create index");
486                     }
487                 }
488             }
489         });
490     }

492     private Map<String, Object> parseMapping(String mappingSource) throws Exception {
493         return XContentFactory.xContent(mappingSource).createParser(mappingSource).mapAndClose();
494     }
495 
496     private void addMappings(Map<String, Map<String, Object>> mappings, Path mappingsDir) throws IOException {
497         try (DirectoryStream<Path> stream = Files.newDirectoryStream(mappingsDir)) {
498             for (Path mappingFile : stream) {
499                 final String fileName = mappingFile.getFileName().toString();
500                 if (Files.isHidden(mappingFile)) {
501                     continue;
502                 }
503                 int lastDotIndex = fileName.lastIndexOf('.');
504                 String mappingType = lastDotIndex != -1 ? mappingFile.getFileName().toString().substring(0, lastDotIndex) : mappingFile.getFileName().toString();
505                 try (BufferedReader reader = Files.newBufferedReader(mappingFile, Charsets.UTF_8)) {
506                     String mappingSource = Streams.copyToString(reader);
507                     if (mappings.containsKey(mappingType)) {
508                         XContentHelper.mergeDefaults(mappings.get(mappingType), parseMapping(mappingSource));
509                     } else {
510                         mappings.put(mappingType, parseMapping(mappingSource));
511                     }
512                 } catch (Exception e) {
513                     logger.warn("failed to read / parse mapping [" + mappingType + "] from location [" + mappingFile + "], ignoring...", e);
514                 }
515             }
516         }
517     }

519     private List<IndexTemplateMetaData> findTemplates(CreateIndexClusterStateUpdateRequest request, ClusterState state, IndexTemplateFilter indexTemplateFilter) throws IOException {
520         List<IndexTemplateMetaData> templates = Lists.newArrayList();
521         for (ObjectCursor<IndexTemplateMetaData> cursor : state.metaData().templates().values()) {
522             IndexTemplateMetaData template = cursor.value;
523             if (indexTemplateFilter.apply(request, template)) {
524                 templates.add(template);
525             }
526         }
527 
528         // see if we have templates defined under config
529         final Path templatesDir = environment.configFile().resolve("templates");
530         if (Files.isDirectory(templatesDir)) {
531             try (DirectoryStream<Path> stream = Files.newDirectoryStream(templatesDir)) {
532                 for (Path templatesFile : stream) {
533                     if (Files.isRegularFile(templatesFile)) {
534                         XContentParser parser = null;
535                         try {
536                             final byte[] templatesData = Files.readAllBytes(templatesFile);
537                             parser = XContentHelper.createParser(templatesData, 0, templatesData.length);
538                             IndexTemplateMetaData template = IndexTemplateMetaData.Builder.fromXContent(parser, templatesFile.getFileName().toString());
539                             if (indexTemplateFilter.apply(request, template)) {
540                                 templates.add(template);
541                             }
542                         } catch (Exception e) {
543                             logger.warn("[{}] failed to read template [{}] from config", e, request.index(), templatesFile.toAbsolutePath());
544                         } finally {
545                             Releasables.closeWhileHandlingException(parser);
546                         }
547                     }
548                 }
549             }
550         }
551 
552         CollectionUtil.timSort(templates, new Comparator<IndexTemplateMetaData>() {
553             @Override
554             public int compare(IndexTemplateMetaData o1, IndexTemplateMetaData o2) {
555                 return o2.order() - o1.order();
556             }
557         });
558         return templates;
559     }

561     private void validate(CreateIndexClusterStateUpdateRequest request, ClusterState state) throws ElasticsearchException {
562         validateIndexName(request.index(), state);
563         String customPath = request.settings().get(IndexMetaData.SETTING_DATA_PATH, null);
564         if (customPath != null && nodeEnv.isCustomPathsEnabled() == false) {
565             throw new IndexCreationException(new Index(request.index()),
566                     new ElasticsearchIllegalArgumentException("custom data_paths for indices is disabled"));
567         }
568     }

570     private static class DefaultIndexTemplateFilter implements IndexTemplateFilter {
571         @Override
572         public boolean apply(CreateIndexClusterStateUpdateRequest request, IndexTemplateMetaData template) {
573             return Regex.simpleMatch(template.template(), request.index());
574         }
