38       * The package separator character '.'
39       */
40      private static final char PACKAGE_SEPARATOR = '.';

42      /**
43       * Return the default ClassLoader to use: typically the thread context
44       * ClassLoader, if available; the ClassLoader that loaded the ClassUtils
45       * class will be used as fallback.
46       * <p/>
47       * <p>Call this method if you intend to use the thread context ClassLoader
48       * in a scenario where you absolutely need a non-null ClassLoader reference:
49       * for example, for class path resource loading (but not necessarily for
50       * <code>Class.forName</code>, which accepts a <code>null</code> ClassLoader
51       * reference as well).
52       *
53       * @return the default ClassLoader (never <code>null</code>)
54       * @see java.lang.Thread#getContextClassLoader()
55       */
56      public static ClassLoader getDefaultClassLoader() {
57          ClassLoader cl = null;
58          try {
59              cl = Thread.currentThread().getContextClassLoader();
60          } catch (Throwable ex) {
61              // Cannot access thread context ClassLoader - falling back to system class loader...
62          }
63          if (cl == null) {
64              // No thread context class loader -> use class loader of this class.
65              cl = Classes.class.getClassLoader();
66          }
67          return cl;
68      }
69  
70      /**
71       * Determine the name of the package of the given class:
72       * e.g. "java.lang" for the <code>java.lang.String</code> class.
73       *
74       * @param clazz the class
75       * @return the package name, or the empty String if the class
76       *         is defined in the default package
77       */
78      public static String getPackageName(Class<?> clazz) {
79          String className = clazz.getName();
80          int lastDotIndex = className.lastIndexOf(PACKAGE_SEPARATOR);
81          return (lastDotIndex != -1 ? className.substring(0, lastDotIndex) : "");
82      }
83  
84      public static String getPackageNameNoDomain(Class<?> clazz) {
85          String fullPackage = getPackageName(clazz);
86          if (fullPackage.startsWith("org.") || fullPackage.startsWith("com.") || fullPackage.startsWith("net.")) {
87              return fullPackage.substring(4);
88          }
89          return fullPackage;
90      }

92      public static boolean isInnerClass(Class<?> clazz) {
93          return !Modifier.isStatic(clazz.getModifiers())
94                  && clazz.getEnclosingClass() != null;
95      }
96  
97      public static boolean isConcrete(Class<?> clazz) {
98          int modifiers = clazz.getModifiers();
99          return !clazz.isInterface() && !Modifier.isAbstract(modifiers);
100     }

102     public static <T> Class<? extends T> loadClass(ClassLoader classLoader, String className, String prefixPackage, String suffixClassName) {
103         return loadClass(classLoader, className, prefixPackage, suffixClassName, null);
104     }
105 
106     @SuppressWarnings({"unchecked"})
107     public static <T> Class<? extends T> loadClass(ClassLoader classLoader, String className, String prefixPackage, String suffixClassName, String errorPrefix) {
108         Throwable t = null;
109         String[] classNames = classNames(className, prefixPackage, suffixClassName);
110         for (String fullClassName : classNames) {
111             try {
112                 return (Class<? extends T>) classLoader.loadClass(fullClassName);
113             } catch (ClassNotFoundException ex) {
114                 t = ex;
115             } catch (NoClassDefFoundError er) {
116                 t = er;
117             }
118         }
119         if (errorPrefix == null) {
120             errorPrefix = "failed to load class";
121         }
122         throw new NoClassSettingsException(errorPrefix + " with value [" + className + "]; tried " + Arrays.toString(classNames), t);
123     }
124 
125     private static String[] classNames(String className, String prefixPackage, String suffixClassName) {
126         String prefixValue = prefixPackage;
127         int packageSeparator = className.lastIndexOf('.');
128         String classNameValue = className;
129         // If class name contains package use it as package prefix instead of specified default one
130         if (packageSeparator > 0) {
131             prefixValue = className.substring(0, packageSeparator + 1);
132             classNameValue = className.substring(packageSeparator + 1);
133         }
134         return new String[]{
135                 className,
136                 prefixValue + Strings.capitalize(toCamelCase(classNameValue)) + suffixClassName,
137                 prefixValue + toCamelCase(classNameValue) + "." + Strings.capitalize(toCamelCase(classNameValue)) + suffixClassName,
138                 prefixValue + toCamelCase(classNameValue).toLowerCase(Locale.ROOT) + "." + Strings.capitalize(toCamelCase(classNameValue)) + suffixClassName,
139         };
140     }
