256     public synchronized void close(final String reason, boolean delete) {
257         if (closed.compareAndSet(false, true)) {
258             deleted.compareAndSet(false, delete);
259             final Set<Integer> shardIds = shardIds();
260             for (final int shardId : shardIds) {
261                 try {
262                     removeShard(shardId, reason);
263                 } catch (Throwable t) {
264                     logger.warn("failed to close shard", t);
265                 }
266             }
267         }
268     }

285     public synchronized IndexShard createShard(int sShardId, boolean primary) throws ElasticsearchException {
286         /*
287          * TODO: we execute this in parallel but it's a synced method. Yet, we might
288          * be able to serialize the execution via the cluster state in the future. for now we just
289          * keep it synced.
290          */
291         if (closed.get()) {
292             throw new ElasticsearchIllegalStateException("Can't create shard [" + index.name() + "][" + sShardId + "], closed");
293         }
294         final ShardId shardId = new ShardId(index, sShardId);
295         ShardLock lock = null;
296         boolean success = false;
297         Injector shardInjector = null;
298         try {
299             lock = nodeEnv.shardLock(shardId, TimeUnit.SECONDS.toMillis(5));
300             if (shards.containsKey(shardId.id())) {
301                 throw new IndexShardAlreadyExistsException(shardId + " already exists");
302             }
303 
304             indicesLifecycle.beforeIndexShardCreated(shardId, indexSettings);
305 
306             logger.debug("creating shard_id {}", shardId);
307             // if we are on a shared FS we only own the shard (ie. we can safely delete it) if we are the primary.
308             final boolean canDeleteShardContent = IndexMetaData.isOnSharedFilesystem(indexSettings) == false ||
309                     (primary && IndexMetaData.isOnSharedFilesystem(indexSettings));
310             ModulesBuilder modules = new ModulesBuilder();
311             modules.add(new ShardsPluginsModule(indexSettings, pluginsService));
312             modules.add(new IndexShardModule(shardId, primary, indexSettings));
313             modules.add(new ShardIndexingModule());
314             modules.add(new ShardSearchModule());
315             modules.add(new ShardGetModule());
316             modules.add(new StoreModule(indexSettings, injector.getInstance(IndexStore.class), lock,
317                     new StoreCloseListener(shardId, canDeleteShardContent)));
318             modules.add(new DeletionPolicyModule(indexSettings));
319             modules.add(new MergePolicyModule(indexSettings));
320             modules.add(new MergeSchedulerModule(indexSettings));
321             modules.add(new ShardFilterCacheModule());
322             modules.add(new ShardQueryCacheModule());
323             modules.add(new ShardBitsetFilterCacheModule());
324             modules.add(new ShardFieldDataModule());
325             modules.add(new TranslogModule(indexSettings));
326             modules.add(new IndexShardGatewayModule());
327             modules.add(new PercolatorShardModule());
328             modules.add(new ShardTermVectorsModule());
329             modules.add(new IndexShardSnapshotModule());
330             modules.add(new SuggestShardModule());
331             try {
332                 shardInjector = modules.createChildInjector(injector);
333             } catch (CreationException e) {
334                 throw new IndexShardCreationException(shardId, Injectors.getFirstErrorFailure(e));
335             } catch (Throwable e) {
336                 throw new IndexShardCreationException(shardId, e);
337             }
338 
339             IndexShard indexShard = shardInjector.getInstance(IndexShard.class);
340             indicesLifecycle.indexShardStateChanged(indexShard, null, "shard created");
341             indicesLifecycle.afterIndexShardCreated(indexShard);
342 
343             shards = newMapBuilder(shards).put(shardId.id(), new Tuple<>(indexShard, shardInjector)).immutableMap();
344             success = true;
345             return indexShard;
346         } catch (IOException ex) {
347             throw new IndexShardCreationException(shardId, ex);
348         } finally {
349             if (success == false) {
350                 IOUtils.closeWhileHandlingException(lock);
351                 if (shardInjector != null) {
352                     IndexShard indexShard = shardInjector.getInstance(IndexShard.class);
353                     closeShardInjector("initialization failed", shardId, shardInjector, indexShard);
354                 }
355             }
356         }
357     }

359     public synchronized void removeShard(int shardId, String reason) throws ElasticsearchException {
360         final ShardId sId = new ShardId(index, shardId);
361         final Injector shardInjector;
362         final IndexShard indexShard;
363         if (shards.containsKey(shardId) == false) {
364             return;
365         }
366         logger.debug("[{}] closing... (reason: [{}])", shardId, reason);
367         HashMap<Integer, Tuple<IndexShard, Injector>> tmpShardsMap = newHashMap(shards);
368         Tuple<IndexShard, Injector> tuple = tmpShardsMap.remove(shardId);
369         indexShard = tuple.v1();
370         shardInjector = tuple.v2();
371         shards = ImmutableMap.copyOf(tmpShardsMap);
372         closeShardInjector(reason, sId, shardInjector, indexShard);
373         logger.debug("[{}] closed (reason: [{}])", shardId, reason);
374     }

376     private void closeShardInjector(String reason, ShardId sId, Injector shardInjector, IndexShard indexShard) {
377         final int shardId = sId.id();
378         try {
379             try {
380                 indicesLifecycle.beforeIndexShardClosed(sId, indexShard, indexSettings);
381             } finally {
382                 // close everything else even if the beforeIndexShardClosed threw an exception
383                 for (Class<? extends Closeable> closeable : pluginsService.shardServices()) {
384                     try {
385                         shardInjector.getInstance(closeable).close();
386                     } catch (Throwable e) {
387                         logger.debug("[{}] failed to clean plugin shard service [{}]", e, shardId, closeable);
388                     }
389                 }
390                 // now we can close the translog service, we need to close it before the we close the shard
391                 closeInjectorResource(sId, shardInjector, TranslogService.class);
392                 // this logic is tricky, we want to close the engine so we rollback the changes done to it
393                 // and close the shard so no operations are allowed to it
394                 if (indexShard != null) {
395                     try {
396                         final boolean flushEngine = deleted.get() == false && closed.get(); // only flush we are we closed (closed index or shutdown) and if we are not deleted
397                         indexShard.close(reason, flushEngine);
398                     } catch (Throwable e) {
399                         logger.debug("[{}] failed to close index shard", e, shardId);
400                         // ignore
401                     }
402                 }
403                 closeInjectorResource(sId, shardInjector,
404                         MergeSchedulerProvider.class,
405                         MergePolicyProvider.class,
406                         IndexShardGatewayService.class,
407                         Translog.class,
408                         PercolatorQueriesRegistry.class);
409 
410                 // call this before we close the store, so we can release resources for it
411                 indicesLifecycle.afterIndexShardClosed(sId, indexShard, indexSettings);
412             }
413         } finally {
414             try {
415                 shardInjector.getInstance(Store.class).close();
416             } catch (Throwable e) {
417                 logger.warn("[{}] failed to close store on shard removal (reason: [{}])", e, shardId, reason);
418             }
419         }
420     }

422     /**
423      * This method gets an instance for each of the given classes passed and calls #close() on the returned instance.
424      * NOTE: this method swallows all exceptions thrown from the close method of the injector and logs them as debug log
425      */
426     private void closeInjectorResource(ShardId shardId, Injector shardInjector, Class<? extends Closeable>... toClose) {
427         for (Class<? extends Closeable> closeable : toClose) {
428             try {
429                 final Closeable instance = shardInjector.getInstance(closeable);
430                 if (instance == null) {
431                     throw new NullPointerException("No instance available for " + closeable.getName());
432                 }
433                 IOUtils.close(instance);
434             } catch (Throwable t) {
435                 logger.debug("{} failed to close {}", t, shardId, Strings.toUnderscoreCase(closeable.getSimpleName()));
436             }
437         }
438     }

440     private void onShardClose(ShardLock lock, boolean ownsShard) {
441         if (deleted.get()) { // we remove that shards content if this index has been deleted
442             try {
443                 if (ownsShard) {
444                     indicesServices.deleteShardStore("delete index", lock, indexSettings);
445                 }
446             } catch (IOException e) {
447                 indicesServices.addPendingDelete(lock.getShardId(), indexSettings);
448                 logger.debug("{} failed to delete shard content - scheduled a retry", e, lock.getShardId().id());
449             }
450         }
451     }

453     private class StoreCloseListener implements Store.OnClose {
454         private final ShardId shardId;
455         private final boolean ownsShard;
456 
457         public StoreCloseListener(ShardId shardId, boolean ownsShard) {
458             this.shardId = shardId;
459             this.ownsShard = ownsShard;
460         }
461 
462         @Override
463         public void handle(ShardLock lock) {
464             assert lock.getShardId().equals(shardId) : "shard Id mismatch, expected: "  + shardId + " but got: " + lock.getShardId();
465             onShardClose(lock, ownsShard);
466         }
467     }
