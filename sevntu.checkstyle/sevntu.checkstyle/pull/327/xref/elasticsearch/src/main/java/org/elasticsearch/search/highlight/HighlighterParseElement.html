37  import static com.google.common.collect.Lists.newArrayList;

39  /**
40   * <pre>
41   * highlight : {
42   *  tags_schema : "styled",
43   *  pre_tags : ["tag1", "tag2"],
44   *  post_tags : ["tag1", "tag2"],
45   *  order : "score",
46   *  highlight_filter : true,
47   *  fields : {
48   *      field1 : {  },
49   *      field2 : { fragment_size : 100, number_of_fragments : 2 },
50   *      field3 : { number_of_fragments : 5, order : "simple", tags_schema : "styled" },
51   *      field4 : { number_of_fragments: 0, pre_tags : ["openingTagA", "openingTagB"], post_tags : ["closingTag"] }
52   *  }
53   * }
54   * </pre>
55   */
56  public class HighlighterParseElement implements SearchParseElement {
57  
58      private static final String[] DEFAULT_PRE_TAGS = new String[]{"<em>"};
59      private static final String[] DEFAULT_POST_TAGS = new String[]{"</em>"};
60  
61      private static final String[] STYLED_PRE_TAG = {
62              "<em class=\"hlt1\">", "<em class=\"hlt2\">", "<em class=\"hlt3\">",
63              "<em class=\"hlt4\">", "<em class=\"hlt5\">", "<em class=\"hlt6\">",
64              "<em class=\"hlt7\">", "<em class=\"hlt8\">", "<em class=\"hlt9\">",
65              "<em class=\"hlt10\">"
66      };
67      private static final String[] STYLED_POST_TAGS = {"</em>"};
68  
69      @Override
70      public void parse(XContentParser parser, SearchContext context) throws Exception {
71          try {
72              context.highlight(parse(parser, context.queryParserService()));
73          } catch (ElasticsearchIllegalArgumentException ex) {
74              throw new SearchParseException(context, "Error while trying to parse Highlighter element in request");
75          }
76      }
77  
78      public SearchContextHighlight parse(XContentParser parser, IndexQueryParserService queryParserService) throws IOException {
79          XContentParser.Token token;
80          String topLevelFieldName = null;
81          final List<Tuple<String, SearchContextHighlight.FieldOptions.Builder>> fieldsOptions = newArrayList();
82  
83          final SearchContextHighlight.FieldOptions.Builder globalOptionsBuilder = new SearchContextHighlight.FieldOptions.Builder()
84                  .preTags(DEFAULT_PRE_TAGS).postTags(DEFAULT_POST_TAGS).scoreOrdered(false).highlightFilter(false)
85                  .requireFieldMatch(false).forceSource(false).fragmentCharSize(100).numberOfFragments(5)
86                  .encoder("default").boundaryMaxScan(SimpleBoundaryScanner.DEFAULT_MAX_SCAN)
87                  .boundaryChars(SimpleBoundaryScanner.DEFAULT_BOUNDARY_CHARS)
88                  .noMatchSize(0).phraseLimit(256);
89  
90          while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
91              if (token == XContentParser.Token.FIELD_NAME) {
92                  topLevelFieldName = parser.currentName();
93              } else if (token == XContentParser.Token.START_ARRAY) {
94                  if ("pre_tags".equals(topLevelFieldName) || "preTags".equals(topLevelFieldName)) {
95                      List<String> preTagsList = Lists.newArrayList();
96                      while ((token = parser.nextToken()) != XContentParser.Token.END_ARRAY) {
97                          preTagsList.add(parser.text());
98                      }
99                      globalOptionsBuilder.preTags(preTagsList.toArray(new String[preTagsList.size()]));
100                 } else if ("post_tags".equals(topLevelFieldName) || "postTags".equals(topLevelFieldName)) {
101                     List<String> postTagsList = Lists.newArrayList();
102                     while ((token = parser.nextToken()) != XContentParser.Token.END_ARRAY) {
103                         postTagsList.add(parser.text());
104                     }
105                     globalOptionsBuilder.postTags(postTagsList.toArray(new String[postTagsList.size()]));
106                 } else if ("fields".equals(topLevelFieldName)) {
107                     while ((token = parser.nextToken()) != XContentParser.Token.END_ARRAY) {
108                         if (token == XContentParser.Token.START_OBJECT) {
109                             String highlightFieldName = null;
110                             while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
111                                 if (token == XContentParser.Token.FIELD_NAME) {
112                                     if (highlightFieldName != null) {
113                                         throw new ElasticsearchIllegalArgumentException("If highlighter fields is an array it must contain objects containing a single field");
114                                     }
115                                     highlightFieldName = parser.currentName();
116                                 } else if (token == XContentParser.Token.START_OBJECT) {
117                                     fieldsOptions.add(Tuple.tuple(highlightFieldName, parseFields(parser, queryParserService)));
118                                 }
119                             }
120                         } else {
121                             throw new ElasticsearchIllegalArgumentException("If highlighter fields is an array it must contain objects containing a single field");
122                         }
123                     }
124                 }
125             } else if (token.isValue()) {
126                 if ("order".equals(topLevelFieldName)) {
127                     globalOptionsBuilder.scoreOrdered("score".equals(parser.text()));
128                 } else if ("tags_schema".equals(topLevelFieldName) || "tagsSchema".equals(topLevelFieldName)) {
129                     String schema = parser.text();
130                     if ("styled".equals(schema)) {
131                         globalOptionsBuilder.preTags(STYLED_PRE_TAG);
132                         globalOptionsBuilder.postTags(STYLED_POST_TAGS);
133                     }
134                 } else if ("highlight_filter".equals(topLevelFieldName) || "highlightFilter".equals(topLevelFieldName)) {
135                     globalOptionsBuilder.highlightFilter(parser.booleanValue());
136                 } else if ("fragment_size".equals(topLevelFieldName) || "fragmentSize".equals(topLevelFieldName)) {
137                     globalOptionsBuilder.fragmentCharSize(parser.intValue());
138                 } else if ("number_of_fragments".equals(topLevelFieldName) || "numberOfFragments".equals(topLevelFieldName)) {
139                     globalOptionsBuilder.numberOfFragments(parser.intValue());
140                 } else if ("encoder".equals(topLevelFieldName)) {
141                     globalOptionsBuilder.encoder(parser.text());
142                 } else if ("require_field_match".equals(topLevelFieldName) || "requireFieldMatch".equals(topLevelFieldName)) {
143                     globalOptionsBuilder.requireFieldMatch(parser.booleanValue());
144                 } else if ("boundary_max_scan".equals(topLevelFieldName) || "boundaryMaxScan".equals(topLevelFieldName)) {
145                     globalOptionsBuilder.boundaryMaxScan(parser.intValue());
146                 } else if ("boundary_chars".equals(topLevelFieldName) || "boundaryChars".equals(topLevelFieldName)) {
147                     char[] charsArr = parser.text().toCharArray();
148                     Character[] globalBoundaryChars = new Character[charsArr.length];
149                     for (int i = 0; i < charsArr.length; i++) {
150                         globalBoundaryChars[i] = charsArr[i];
151                     }
152                     globalOptionsBuilder.boundaryChars(globalBoundaryChars);
153                 } else if ("type".equals(topLevelFieldName)) {
154                     globalOptionsBuilder.highlighterType(parser.text());
155                 } else if ("fragmenter".equals(topLevelFieldName)) {
156                     globalOptionsBuilder.fragmenter(parser.text());
157                 } else if ("no_match_size".equals(topLevelFieldName) || "noMatchSize".equals(topLevelFieldName)) {
158                     globalOptionsBuilder.noMatchSize(parser.intValue());
159                 } else if ("force_source".equals(topLevelFieldName) || "forceSource".equals(topLevelFieldName)) {
160                     globalOptionsBuilder.forceSource(parser.booleanValue());
161                 } else if ("phrase_limit".equals(topLevelFieldName) || "phraseLimit".equals(topLevelFieldName)) {
162                     globalOptionsBuilder.phraseLimit(parser.intValue());
163                 }
164             } else if (token == XContentParser.Token.START_OBJECT && "options".equals(topLevelFieldName)) {
165                 globalOptionsBuilder.options(parser.map());
166             } else if (token == XContentParser.Token.START_OBJECT) {
167                 if ("fields".equals(topLevelFieldName)) {
168                     String highlightFieldName = null;
169                     while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
170                         if (token == XContentParser.Token.FIELD_NAME) {
171                             highlightFieldName = parser.currentName();
172                         } else if (token == XContentParser.Token.START_OBJECT) {
173                             fieldsOptions.add(Tuple.tuple(highlightFieldName, parseFields(parser, queryParserService)));
174                         }
175                     }
176                 } else if ("highlight_query".equals(topLevelFieldName) || "highlightQuery".equals(topLevelFieldName)) {
177                     globalOptionsBuilder.highlightQuery(queryParserService.parse(parser).query());
178                 }
179             }
180         }
181 
182         final SearchContextHighlight.FieldOptions globalOptions = globalOptionsBuilder.build();
183         if (globalOptions.preTags() != null && globalOptions.postTags() == null) {
184             throw new ElasticsearchIllegalArgumentException("Highlighter global preTags are set, but global postTags are not set");
185         }
186 
187         final List<SearchContextHighlight.Field> fields = Lists.newArrayList();
188         // now, go over and fill all fieldsOptions with default values from the global state
189         for (final Tuple<String, SearchContextHighlight.FieldOptions.Builder> tuple : fieldsOptions) {
190             fields.add(new SearchContextHighlight.Field(tuple.v1(), tuple.v2().merge(globalOptions).build()));
191         }
192         return new SearchContextHighlight(fields);
193     }
194 
195     protected SearchContextHighlight.FieldOptions.Builder parseFields(XContentParser parser, IndexQueryParserService queryParserService) throws IOException {
196         XContentParser.Token token;
197 
198         final SearchContextHighlight.FieldOptions.Builder fieldOptionsBuilder = new SearchContextHighlight.FieldOptions.Builder();
199         String fieldName = null;
200         while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
201             if (token == XContentParser.Token.FIELD_NAME) {
202                 fieldName = parser.currentName();
203             } else if (token == XContentParser.Token.START_ARRAY) {
204                 if ("pre_tags".equals(fieldName) || "preTags".equals(fieldName)) {
205                     List<String> preTagsList = Lists.newArrayList();
206                     while ((token = parser.nextToken()) != XContentParser.Token.END_ARRAY) {
207                         preTagsList.add(parser.text());
208                     }
209                     fieldOptionsBuilder.preTags(preTagsList.toArray(new String[preTagsList.size()]));
210                 } else if ("post_tags".equals(fieldName) || "postTags".equals(fieldName)) {
211                     List<String> postTagsList = Lists.newArrayList();
212                     while ((token = parser.nextToken()) != XContentParser.Token.END_ARRAY) {
213                         postTagsList.add(parser.text());
214                     }
215                     fieldOptionsBuilder.postTags(postTagsList.toArray(new String[postTagsList.size()]));
216                 } else if ("matched_fields".equals(fieldName) || "matchedFields".equals(fieldName)) {
217                     Set<String> matchedFields = Sets.newHashSet();
218                     while ((token = parser.nextToken()) != XContentParser.Token.END_ARRAY) {
219                         matchedFields.add(parser.text());
220                     }
221                     fieldOptionsBuilder.matchedFields(matchedFields);
222                 }
223             } else if (token.isValue()) {
224                 if ("fragment_size".equals(fieldName) || "fragmentSize".equals(fieldName)) {
225                     fieldOptionsBuilder.fragmentCharSize(parser.intValue());
226                 } else if ("number_of_fragments".equals(fieldName) || "numberOfFragments".equals(fieldName)) {
227                     fieldOptionsBuilder.numberOfFragments(parser.intValue());
228                 } else if ("fragment_offset".equals(fieldName) || "fragmentOffset".equals(fieldName)) {
229                     fieldOptionsBuilder.fragmentOffset(parser.intValue());
230                 } else if ("highlight_filter".equals(fieldName) || "highlightFilter".equals(fieldName)) {
231                     fieldOptionsBuilder.highlightFilter(parser.booleanValue());
232                 } else if ("order".equals(fieldName)) {
233                     fieldOptionsBuilder.scoreOrdered("score".equals(parser.text()));
234                 } else if ("require_field_match".equals(fieldName) || "requireFieldMatch".equals(fieldName)) {
235                     fieldOptionsBuilder.requireFieldMatch(parser.booleanValue());
236                 } else if ("boundary_max_scan".equals(fieldName) || "boundaryMaxScan".equals(fieldName)) {
237                     fieldOptionsBuilder.boundaryMaxScan(parser.intValue());
238                 } else if ("boundary_chars".equals(fieldName) || "boundaryChars".equals(fieldName)) {
239                     char[] charsArr = parser.text().toCharArray();
240                     Character[] boundaryChars = new Character[charsArr.length];
241                     for (int i = 0; i < charsArr.length; i++) {
242                         boundaryChars[i] = charsArr[i];
243                     }
244                     fieldOptionsBuilder.boundaryChars(boundaryChars);
245                 } else if ("type".equals(fieldName)) {
246                     fieldOptionsBuilder.highlighterType(parser.text());
247                 } else if ("fragmenter".equals(fieldName)) {
248                     fieldOptionsBuilder.fragmenter(parser.text());
249                 } else if ("no_match_size".equals(fieldName) || "noMatchSize".equals(fieldName)) {
250                     fieldOptionsBuilder.noMatchSize(parser.intValue());
251                 } else if ("force_source".equals(fieldName) || "forceSource".equals(fieldName)) {
252                     fieldOptionsBuilder.forceSource(parser.booleanValue());
253                 } else if ("phrase_limit".equals(fieldName) || "phraseLimit".equals(fieldName)) {
254                     fieldOptionsBuilder.phraseLimit(parser.intValue());
255                 }
256             } else if (token == XContentParser.Token.START_OBJECT) {
257                 if ("highlight_query".equals(fieldName) || "highlightQuery".equals(fieldName)) {
258                     fieldOptionsBuilder.highlightQuery(queryParserService.parse(parser).query());
259                 } else if ("options".equals(fieldName)) {
260                     fieldOptionsBuilder.options(parser.map());
261                 }
262             }
263         }
264         return fieldOptionsBuilder;
265     }
