65  /**
66   * A Store provides plain access to files written by an elasticsearch index shard. Each shard
67   * has a dedicated store that is uses to access Lucene's Directory which represents the lowest level
68   * of file abstraction in Lucene used to read and write Lucene indices.
69   * This class also provides access to metadata information like checksums for committed files. A committed
70   * file is a file that belongs to a segment written by a Lucene commit. Files that have not been committed
71   * ie. created during a merge or a shard refresh / NRT reopen are not considered in the MetadataSnapshot.
72   * <p/>
73   * Note: If you use a store it's reference count should be increased before using it by calling #incRef and a
74   * corresponding #decRef must be called in a try/finally block to release the store again ie.:
75   * <pre>
76   *      store.incRef();
77   *      try {
78   *        // use the store...
79   *
80   *      } finally {
81   *          store.decRef();
82   *      }
83   * </pre>
84   */
85  public class Store extends AbstractIndexShardComponent implements Closeable, RefCounted {

146      * @throws IOException if the index is corrupted or the segments file is not present

148     private static SegmentInfos readSegmentsInfo(IndexCommit commit, Directory directory) throws IOException {
149         try {
150             return commit == null ? Lucene.readSegmentInfos(directory) : Lucene.readSegmentInfos(commit, directory);
151         } catch (EOFException eof) {
152             // TODO this should be caught by lucene - EOF is almost certainly an index corruption
153             throw new CorruptIndexException("Read past EOF while reading segment infos", "commit(" + commit + ")", eof);
154         } catch (IOException exception) {
155             throw exception; // IOExceptions like too many open files are not necessarily a corruption - just bubble it up
156         } catch (Exception ex) {
157             throw new CorruptIndexException("Hit unexpected exception while reading segment infos", "commit(" + commit + ")", ex);
158         }

224     /**
225      * Renames all the given files form the key of the map to the
226      * value of the map. All successfully renamed files are removed from the map in-place.
227      */
228     public void renameTempFilesSafe(Map<String, String> tempFileMap) throws IOException {
229         // this works just like a lucene commit - we rename all temp files and once we successfully
230         // renamed all the segments we rename the commit to ensure we don't leave half baked commits behind.
231         final Map.Entry<String, String>[] entries = tempFileMap.entrySet().toArray(new Map.Entry[tempFileMap.size()]);
232         ArrayUtil.timSort(entries, new Comparator<Map.Entry<String, String>>() {
233             @Override
234             public int compare(Map.Entry<String, String> o1, Map.Entry<String, String> o2) {
235                 String left = o1.getValue();
236                 String right = o2.getValue();
237                 if (left.startsWith(IndexFileNames.SEGMENTS) || right.startsWith(IndexFileNames.SEGMENTS)) {
238                     if (left.startsWith(IndexFileNames.SEGMENTS) == false) {
239                         return -1;
240                     } else if (right.startsWith(IndexFileNames.SEGMENTS) == false) {
241                         return 1;
242                     }
243                 }
244                 return left.compareTo(right);
245             }
246         });
247         metadataLock.writeLock().lock();
248         // we make sure that nobody fetches the metadata while we do this rename operation here to ensure we don't
249         // get exceptions if files are still open.
250         try (Lock writeLock = directory.makeLock(IndexWriter.WRITE_LOCK_NAME)) {
251             if (!writeLock.obtain(IndexWriterConfig.getDefaultWriteLockTimeout())) { // obtain write lock
252                 throw new LockObtainFailedException("Index locked for write: " + writeLock);
253             }
254             for (Map.Entry<String, String> entry : entries) {
255                 String tempFile = entry.getKey();
256                 String origFile = entry.getValue();
257                 // first, go and delete the existing ones
258                 try {
259                     directory.deleteFile(origFile);
260                 } catch (FileNotFoundException | NoSuchFileException e) {
261                 } catch (Throwable ex) {
262                     logger.debug("failed to delete file [{}]", ex, origFile);
263                 }
264                 // now, rename the files... and fail it it won't work
265                 this.renameFile(tempFile, origFile);
266                 final String remove = tempFileMap.remove(tempFile);
267                 assert remove != null;
268             }
269         } finally {
270             metadataLock.writeLock().unlock();
271         }
272 
273     }

342     private void closeInternal() {
343         try {
344             try {
345                 directory.innerClose(); // this closes the distributorDirectory as well
346             } finally {
347                 onClose.handle(shardLock);
348             }
349         } catch (IOException e) {
350             logger.debug("failed to close directory", e);
351         } finally {
352             IOUtils.closeWhileHandlingException(shardLock);
353         }
354     }

358      * Reads a MetadataSnapshot from the given index locations or returns an empty snapshot if it can't be read.
359      *
360      * @throws IOException if the index we try to read is corrupted
361      */
362     public static MetadataSnapshot readMetadataSnapshot(Path[] indexLocations, ESLogger logger) throws IOException {
363         final Directory[] dirs = new Directory[indexLocations.length];
364         try {
365             for (int i = 0; i < indexLocations.length; i++) {
366                 dirs[i] = new SimpleFSDirectory(indexLocations[i]);
367             }
368             DistributorDirectory dir = new DistributorDirectory(dirs);
369             failIfCorrupted(dir, new ShardId("", 1));
370             return new MetadataSnapshot(null, dir, logger);
371         } catch (IndexNotFoundException ex) {
372             // that's fine - happens all the time no need to log
373         } catch (FileNotFoundException | NoSuchFileException ex) {
374             logger.info("Failed to open / find files while reading metadata snapshot");
375         } finally {
376             IOUtils.close(dirs);
377         }
378         return MetadataSnapshot.EMPTY;
379     }

381     /**
382      * The returned IndexOutput might validate the files checksum if the file has been written with a newer lucene version
383      * and the metadata holds the necessary information to detect that it was been written by Lucene 4.8 or newer. If it has only
384      * a legacy checksum, returned IndexOutput will not verify the checksum.
385      * <p/>
386      * Note: Checksums are calculated nevertheless since lucene does it by default sicne version 4.8.0. This method only adds the
387      * verification against the checksum in the given metadata and does not add any significant overhead.
388      */
389     public IndexOutput createVerifyingOutput(String fileName, final StoreFileMetaData metadata, final IOContext context) throws IOException {
390         IndexOutput output = directory().createOutput(fileName, context);
391         boolean success = false;
392         try {
393             if (metadata.hasLegacyChecksum()) {
394                 logger.debug("create legacy adler32 output for {}", fileName);
395                 output = new LegacyVerification.Adler32VerifyingIndexOutput(output, metadata.checksum(), metadata.length());
396             } else if (metadata.checksum() == null) {
397                 // TODO: when the file is a segments_N, we can still CRC-32 + length for more safety
398                 // its had that checksum forever.
399                 logger.debug("create legacy length-only output for {}", fileName);
400                 output = new LegacyVerification.LengthVerifyingIndexOutput(output, metadata.length());
401             } else {
402                 assert metadata.writtenBy() != null;
403                 assert metadata.writtenBy().onOrAfter(Version.LUCENE_4_8);
404                 output = new LuceneVerifyingIndexOutput(metadata, output);
405             }
406             success = true;
407         } finally {
408             if (success == false) {
409                 IOUtils.closeWhileHandlingException(output);
410             }
411         }
412         return output;
413     }

450     public static void checkIntegrity(final StoreFileMetaData md, final Directory directory) throws IOException {
451         try (IndexInput input = directory.openInput(md.name(), IOContext.READONCE)) {
452             if (input.length() != md.length()) { // first check the length no matter how old this file is
453                 throw new CorruptIndexException("expected length=" + md.length() + " != actual length: " + input.length() + " : file truncated?", input);
454             }
455             if (md.writtenBy() != null && md.writtenBy().onOrAfter(Version.LUCENE_4_8_0)) {
456                 // throw exception if the file is corrupt
457                 String checksum = Store.digestToString(CodecUtil.checksumEntireFile(input));
458                 // throw exception if metadata is inconsistent
459                 if (!checksum.equals(md.checksum())) {
460                     throw new CorruptIndexException("inconsistent metadata: lucene checksum=" + checksum +
461                                                                        ", metadata checksum=" + md.checksum(), input);
462                 }
463             } else if (md.hasLegacyChecksum()) {
464                 // legacy checksum verification - no footer that we need to omit in the checksum!
465                 final Checksum checksum = new Adler32();
466                 final byte[] buffer = new byte[md.length() > 4096 ? 4096 : (int) md.length()];
467                 final long len = input.length();
468                 long read = 0;
469                 while (len > read) {
470                     final long bytesLeft = len - read;
471                     final int bytesToRead = bytesLeft < buffer.length ? (int) bytesLeft : buffer.length;
472                     input.readBytes(buffer, 0, bytesToRead, false);
473                     checksum.update(buffer, 0, bytesToRead);
474                     read += bytesToRead;
475                 }
476                 String adler32 = Store.digestToString(checksum.getValue());
477                 if (!adler32.equals(md.checksum())) {
478                     throw new CorruptIndexException("checksum failed (hardware problem?) : expected=" + md.checksum() +
479                                                                                            " actual=" + adler32, input);
480                 }
481             }
482         }
483     }

504     private static final void failIfCorrupted(Directory directory, ShardId shardId) throws IOException {
505         final String[] files = directory.listAll();
506         List<CorruptIndexException> ex = new ArrayList<>();
507         for (String file : files) {
508             if (file.startsWith(CORRUPTED)) {
509                 try (ChecksumIndexInput input = directory.openChecksumInput(file, IOContext.READONCE)) {
510                     int version = CodecUtil.checkHeader(input, CODEC, VERSION_START, VERSION);
511                     String msg = input.readString();
512                     StringBuilder builder = new StringBuilder(shardId.toString());
513                     builder.append(" Preexisting corrupted index [");
514                     builder.append(file).append("] caused by: ");
515                     builder.append(msg);
516                     if (version == VERSION_STACK_TRACE) {
517                         builder.append(System.lineSeparator());
518                         builder.append(input.readString());
519                     }
520                     ex.add(new CorruptIndexException(builder.toString(), "preexisting_corruption"));
521                     CodecUtil.checkFooter(input);
522                 }
523             }
524         }
525         if (ex.isEmpty() == false) {
526             ExceptionsHelper.rethrowAndSuppress(ex);
527         }
528     }

530     /**
531      * This method deletes every file in this store that is not contained in the given source meta data or is a
532      * legacy checksum file. After the delete it pulls the latest metadata snapshot from the store and compares it
533      * to the given snapshot. If the snapshots are inconsistent an illegal state exception is thrown
534      *
535      * @param reason         the reason for this cleanup operation logged for each deleted file
536      * @param sourceMetaData the metadata used for cleanup. all files in this metadata should be kept around.
537      * @throws IOException                        if an IOException occurs
538      * @throws ElasticsearchIllegalStateException if the latest snapshot in this store differs from the given one after the cleanup.
539      */
540     public void cleanupAndVerify(String reason, MetadataSnapshot sourceMetaData) throws IOException {
541         failIfCorrupted();
542         metadataLock.writeLock().lock();
543         try (Lock writeLock = directory.makeLock(IndexWriter.WRITE_LOCK_NAME)) {
544             if (!writeLock.obtain(IndexWriterConfig.getDefaultWriteLockTimeout())) { // obtain write lock
545                 throw new LockObtainFailedException("Index locked for write: " + writeLock);
546             }
547             final StoreDirectory dir = directory;
548             for (String existingFile : dir.listAll()) {
549                 if (existingFile.equals(IndexWriter.WRITE_LOCK_NAME) || Store.isChecksum(existingFile) || sourceMetaData.contains(existingFile)) {
550                     continue; // don't delete snapshot file, or the checksums file (note, this is extra protection since the Store won't delete checksum)
551                 }
552                 try {
553                     dir.deleteFile(reason, existingFile);
554                     // FNF should not happen since we hold a write lock?
555                 } catch (IOException ex) {
556                     if (existingFile.startsWith(IndexFileNames.SEGMENTS)
557                         || existingFile.equals(IndexFileNames.OLD_SEGMENTS_GEN)) {
558                         // TODO do we need to also fail this if we can't delete the pending commit file?
559                         // if one of those files can't be deleted we better fail the cleanup otherwise we might leave an old commit point around?
560                         throw new ElasticsearchIllegalStateException("Can't delete " + existingFile + " - cleanup failed", ex);
561                     }
562                     logger.debug("failed to delete file [{}]", ex, existingFile);
563                     // ignore, we don't really care, will get deleted later on
564                 }
565             }
566             final Store.MetadataSnapshot metadataOrEmpty = getMetadata();
567             verifyAfterCleanup(sourceMetaData, metadataOrEmpty);
568         } finally {
569             metadataLock.writeLock().unlock();
570         }
571     }

573     // pkg private for testing
574     final void verifyAfterCleanup(MetadataSnapshot sourceMetaData, MetadataSnapshot targetMetaData) {
575         final RecoveryDiff recoveryDiff = targetMetaData.recoveryDiff(sourceMetaData);
576         if (recoveryDiff.identical.size() != recoveryDiff.size()) {
577             if (recoveryDiff.missing.isEmpty()) {
578                 for (StoreFileMetaData meta : recoveryDiff.different) {
579                     StoreFileMetaData local = targetMetaData.get(meta.name());
580                     StoreFileMetaData remote = sourceMetaData.get(meta.name());
581                     // if we have different files the they must have no checksums otherwise something went wrong during recovery.
582                     // we have that problem when we have an empty index is only a segments_1 file then we can't tell if it's a Lucene 4.8 file
583                     // and therefore no checksum. That isn't much of a problem since we simply copy it over anyway but those files come out as
584                     // different in the diff. That's why we have to double check here again if the rest of it matches.
585 
586                     // all is fine this file is just part of a commit or a segment that is different
587                     final boolean same = local.isSame(remote);
588 
589                     // this check ensures that the two files are consistent ie. if we don't have checksums only the rest needs to match we are just
590                     // verifying that we are consistent on both ends source and target
591                     final boolean hashAndLengthEqual = (
592                             local.checksum() == null
593                                     && remote.checksum() == null
594                                     && local.hash().equals(remote.hash())
595                                     && local.length() == remote.length());
596                     final boolean consistent = hashAndLengthEqual || same;
597                     if (consistent == false) {
598                         logger.debug("Files are different on the recovery target: {} ", recoveryDiff);
599                         throw new ElasticsearchIllegalStateException("local version: " + local + " is different from remote version after recovery: " + remote, null);
600                     }
601                 }
602             } else {
603                 logger.debug("Files are missing on the recovery target: {} ", recoveryDiff);
604                 throw new ElasticsearchIllegalStateException("Files are missing on the recovery target: [different="
605                         + recoveryDiff.different + ", missing=" + recoveryDiff.missing + ']', null);
606             }
607         }
608     }

685         ImmutableMap<String, StoreFileMetaData> buildMetadata(IndexCommit commit, Directory directory, ESLogger logger) throws IOException {
686             ImmutableMap.Builder<String, StoreFileMetaData> builder = ImmutableMap.builder();
687             Map<String, String> checksumMap = readLegacyChecksums(directory).v1();
688             try {
689                 final SegmentInfos segmentCommitInfos = Store.readSegmentsInfo(commit, directory);
690                 Version maxVersion = Version.LUCENE_4_0; // we don't know which version was used to write so we take the max version.
691                 for (SegmentCommitInfo info : segmentCommitInfos) {
692                     final Version version = info.info.getVersion();
693                     if (version == null) {
694                         // version is written since 3.1+: we should have already hit IndexFormatTooOld.
695                         throw new IllegalArgumentException("expected valid version value: " + info.info.toString());
696                     }
697                     if (version.onOrAfter(maxVersion)) {
698                         maxVersion = version;
699                     }
700                     for (String file : info.files()) {
701                         String legacyChecksum = checksumMap.get(file);
702                         if (version.onOrAfter(FIRST_LUCENE_CHECKSUM_VERSION)) {
703                             checksumFromLuceneFile(directory, file, builder, logger, version, SEGMENT_INFO_EXTENSION.equals(IndexFileNames.getExtension(file)));
704                         } else {
705                             builder.put(file, new StoreFileMetaData(file, directory.fileLength(file), legacyChecksum, version));
706                         }
707                     }
708                 }
709                 final String segmentsFile = segmentCommitInfos.getSegmentsFileName();
710                 String legacyChecksum = checksumMap.get(segmentsFile);
711                 if (maxVersion.onOrAfter(FIRST_LUCENE_CHECKSUM_VERSION)) {
712                     checksumFromLuceneFile(directory, segmentsFile, builder, logger, maxVersion, true);
713                 } else {
714                     final BytesRefBuilder fileHash = new BytesRefBuilder();
715                     final long length;
716                     try (final IndexInput in = directory.openInput(segmentsFile, IOContext.READONCE)) {
717                         length = in.length();
718                         hashFile(fileHash, new InputStreamIndexInput(in, length), length);
719                     }
720                     builder.put(segmentsFile, new StoreFileMetaData(segmentsFile, length, legacyChecksum, maxVersion, fileHash.get()));
721                 }
722             } catch (CorruptIndexException | IndexNotFoundException | IndexFormatTooOldException | IndexFormatTooNewException ex) {
723                 // we either know the index is corrupted or it's just not there
724                 throw ex;
725             } catch (Throwable ex) {
726                 try {
727                     // Lucene checks the checksum after it tries to lookup the codec etc.
728                     // in that case we might get only IAE or similar exceptions while we are really corrupt...
729                     // TODO we should check the checksum in lucene if we hit an exception
730                     logger.warn("failed to build store metadata. checking segment info integrity (with commit [{}])",
731                             ex, commit == null ? "no" : "yes");
732                     Lucene.checkSegmentInfoIntegrity(directory);
733                 } catch (CorruptIndexException | IndexFormatTooOldException | IndexFormatTooNewException cex) {
734                     cex.addSuppressed(ex);
735                     throw cex;
736                 } catch (Throwable e) {
737                     // ignore...
738                 }
739 
740                 throw ex;
741             }
742             return builder.build();
743         }

746          * Reads legacy checksum files found in the directory.
747          * <p/>
748          * Files are expected to start with _checksums- prefix
749          * followed by long file version. Only file with the highest version is read, all other files are ignored.
750          *

752          * @return a map of file checksums and the checksum file version

755         static Tuple<Map<String, String>, Long> readLegacyChecksums(Directory directory) throws IOException {
756             synchronized (directory) {
757                 long lastFound = -1;
758                 for (String name : directory.listAll()) {
759                     if (!isChecksum(name)) {
760                         continue;
761                     }
762                     long current = Long.parseLong(name.substring(CHECKSUMS_PREFIX.length()));
763                     if (current > lastFound) {
764                         lastFound = current;
765                     }
766                 }
767                 if (lastFound > -1) {
768                     try (IndexInput indexInput = directory.openInput(CHECKSUMS_PREFIX + lastFound, IOContext.READONCE)) {
769                         indexInput.readInt(); // version
770                         return new Tuple(indexInput.readStringStringMap(), lastFound);
771                     }
772                 }
773                 return new Tuple(new HashMap<>(), -1l);
774             }

801         private static void checksumFromLuceneFile(Directory directory, String file, ImmutableMap.Builder<String, StoreFileMetaData> builder, ESLogger logger, Version version, boolean readFileAsHash) throws IOException {
802             final String checksum;
803             final BytesRefBuilder fileHash = new BytesRefBuilder();
804             try (final IndexInput in = directory.openInput(file, IOContext.READONCE)) {
805                 final long length;
806                 try {
807                     length = in.length();
808                     if (length < CodecUtil.footerLength()) {
809                         // truncated files trigger IAE if we seek negative... these files are really corrupted though
810                         throw new CorruptIndexException("Can't retrieve checksum from file: " + file + " file length must be >= " + CodecUtil.footerLength() + " but was: " + in.length(), in);
811                     }
812                     if (readFileAsHash) {
813                         final VerifyingIndexInput verifyingIndexInput = new VerifyingIndexInput(in); // additional safety we checksum the entire file we read the hash for...
814                         hashFile(fileHash, new InputStreamIndexInput(verifyingIndexInput, length), length);
815                         checksum = digestToString(verifyingIndexInput.verify());
816                     } else {
817                         checksum = digestToString(CodecUtil.retrieveChecksum(in));
818                     }
819 
820                 } catch (Throwable ex) {
821                     logger.debug("Can retrieve checksum from file [{}]", ex, file);
822                     throw ex;
823                 }
824                 builder.put(file, new StoreFileMetaData(file, length, checksum, version, fileHash.get()));
825             }
826         }

870         /**
871          * Returns a diff between the two snapshots that can be used for recovery. The given snapshot is treated as the
872          * recovery target and this snapshot as the source. The returned diff will hold a list of files that are:
873          * <ul>
874          * <li>identical: they exist in both snapshots and they can be considered the same ie. they don't need to be recovered</li>
875          * <li>different: they exist in both snapshots but their they are not identical</li>
876          * <li>missing: files that exist in the source but not in the target</li>
877          * </ul>
878          * This method groups file into per-segment files and per-commit files. A file is treated as
879          * identical if and on if all files in it's group are identical. On a per-segment level files for a segment are treated
880          * as identical iff:
881          * <ul>
882          * <li>all files in this segment have the same checksum</li>
883          * <li>all files in this segment have the same length</li>
884          * <li>the segments <tt>.si</tt> files hashes are byte-identical Note: This is a using a perfect hash function, The metadata transfers the <tt>.si</tt> file content as it's hash</li>
885          * </ul>
886          * <p/>
887          * The <tt>.si</tt> file contains a lot of diagnostics including a timestamp etc. in the future there might be
888          * unique segment identifiers in there hardening this method further.
889          * <p/>
890          * The per-commit files handles very similar. A commit is composed of the <tt>segments_N</tt> files as well as generational files like
891          * deletes (<tt>_x_y.del</tt>) or field-info (<tt>_x_y.fnm</tt>) files. On a per-commit level files for a commit are treated
892          * as identical iff:
893          * <ul>
894          * <li>all files belonging to this commit have the same checksum</li>
895          * <li>all files belonging to this commit have the same length</li>
896          * <li>the segments file <tt>segments_N</tt> files hashes are byte-identical Note: This is a using a perfect hash function, The metadata transfers the <tt>segments_N</tt> file content as it's hash</li>
897          * </ul>
898          * <p/>
899          * NOTE: this diff will not contain the <tt>segments.gen</tt> file. This file is omitted on recovery.
900          */
901         public RecoveryDiff recoveryDiff(MetadataSnapshot recoveryTargetSnapshot) {
902             final ImmutableList.Builder<StoreFileMetaData> identical = ImmutableList.builder();
903             final ImmutableList.Builder<StoreFileMetaData> different = ImmutableList.builder();
904             final ImmutableList.Builder<StoreFileMetaData> missing = ImmutableList.builder();
905             final Map<String, List<StoreFileMetaData>> perSegment = new HashMap<>();
906             final List<StoreFileMetaData> perCommitStoreFiles = new ArrayList<>();
907 
908             for (StoreFileMetaData meta : this) {
909                 if (IndexFileNames.OLD_SEGMENTS_GEN.equals(meta.name())) { // legacy
910                     continue; // we don't need that file at all
911                 }
912                 final String segmentId = IndexFileNames.parseSegmentName(meta.name());
913                 final String extension = IndexFileNames.getExtension(meta.name());
914                 assert FIELD_INFOS_FILE_EXTENSION.equals(extension) == false || IndexFileNames.stripExtension(IndexFileNames.stripSegmentName(meta.name())).isEmpty() : "FieldInfos are generational but updateable DV are not supported in elasticsearch";
915                 if (IndexFileNames.SEGMENTS.equals(segmentId) || DEL_FILE_EXTENSION.equals(extension) || LIV_FILE_EXTENSION.equals(extension)) {
916                     // only treat del files as per-commit files fnm files are generational but only for upgradable DV
917                     perCommitStoreFiles.add(meta);
918                 } else {
919                     List<StoreFileMetaData> perSegStoreFiles = perSegment.get(segmentId);
920                     if (perSegStoreFiles == null) {
921                         perSegStoreFiles = new ArrayList<>();
922                         perSegment.put(segmentId, perSegStoreFiles);
923                     }
924                     perSegStoreFiles.add(meta);
925                 }
926             }
927             final ArrayList<StoreFileMetaData> identicalFiles = new ArrayList<>();
928             for (List<StoreFileMetaData> segmentFiles : Iterables.concat(perSegment.values(), Collections.singleton(perCommitStoreFiles))) {
929                 identicalFiles.clear();
930                 boolean consistent = true;
931                 for (StoreFileMetaData meta : segmentFiles) {
932                     StoreFileMetaData storeFileMetaData = recoveryTargetSnapshot.get(meta.name());
933                     if (storeFileMetaData == null) {
934                         consistent = false;
935                         missing.add(meta);
936                     } else if (storeFileMetaData.isSame(meta) == false) {
937                         consistent = false;
938                         different.add(meta);
939                     } else {
940                         identicalFiles.add(meta);
941                     }
942                 }
943                 if (consistent) {
944                     identical.addAll(identicalFiles);
945                 } else {
946                     // make sure all files are added - this can happen if only the deletes are different
947                     different.addAll(identicalFiles);
948                 }
949             }
950             RecoveryDiff recoveryDiff = new RecoveryDiff(identical.build(), different.build(), missing.build());
951             assert recoveryDiff.size() == this.metadata.size() - (metadata.containsKey(IndexFileNames.OLD_SEGMENTS_GEN) ? 1 : 0)
952                     : "some files are missing recoveryDiff size: [" + recoveryDiff.size() + "] metadata size: [" + this.metadata.size() + "] contains  segments.gen: [" + metadata.containsKey(IndexFileNames.OLD_SEGMENTS_GEN) + "]";
953             return recoveryDiff;
954         }

1125         private final StoreFileMetaData metadata;
1126         private long writtenBytes;
1127         private final long checksumPosition;
1128         private String actualChecksum;
1129 
1130         LuceneVerifyingIndexOutput(StoreFileMetaData metadata, IndexOutput out) {
1131             super(out);
1132             this.metadata = metadata;
1133             checksumPosition = metadata.length() - 8; // the last 8 bytes are the checksum
1134         }

1136         @Override
1137         public void verify() throws IOException {
1138             if (metadata.checksum().equals(actualChecksum) && writtenBytes == metadata.length()) {
1139                 return;
1140             }
1141             throw new CorruptIndexException("verification failed (hardware problem?) : expected=" + metadata.checksum() +
1142                     " actual=" + actualChecksum + " writtenLength=" + writtenBytes + " expectedLength=" + metadata.length() +
1143                     " (resource=" + metadata.toString() + ")", "VerifyingIndexOutput(" + metadata.name() + ")");
1144         }
1145 
1146         @Override
1147         public void writeByte(byte b) throws IOException {
1148             if (writtenBytes++ == checksumPosition) {
1149                 readAndCompareChecksum();
1150             }
1151             out.writeByte(b);
1152         }
1153 
1154         private void readAndCompareChecksum() throws IOException {
1155             actualChecksum = digestToString(getChecksum());
1156             if (!metadata.checksum().equals(actualChecksum)) {
1157                 throw new CorruptIndexException("checksum failed (hardware problem?) : expected=" + metadata.checksum() +
1158                         " actual=" + actualChecksum +
1159                         " (resource=" + metadata.toString() + ")", "VerifyingIndexOutput(" + metadata.name() + ")");
1160             }
1161         }
1162 
1163         @Override
1164         public void writeBytes(byte[] b, int offset, int length) throws IOException {
1165             if (writtenBytes + length > checksumPosition && actualChecksum == null) {
1166                 assert writtenBytes <= checksumPosition;
1167                 final int bytesToWrite = (int) (checksumPosition - writtenBytes);
1168                 out.writeBytes(b, offset, bytesToWrite);
1169                 readAndCompareChecksum();
1170                 offset += bytesToWrite;
1171                 length -= bytesToWrite;
1172                 writtenBytes += bytesToWrite;
1173             }
1174             out.writeBytes(b, offset, length);
1175             writtenBytes += length;
1176         }

1180     /**
1181      * Index input that calculates checksum as data is read from the input.
1182      * <p/>
1183      * This class supports random access (it is possible to seek backward and forward) in order to accommodate retry
1184      * mechanism that is used in some repository plugins (S3 for example). However, the checksum is only calculated on
1185      * the first read. All consecutive reads of the same data are not used to calculate the checksum.
1186      */
1187     static class VerifyingIndexInput extends ChecksumIndexInput {
1188         private final IndexInput input;
1189         private final Checksum digest;
1190         private final long checksumPosition;
1191         private final byte[] checksum = new byte[8];
1192         private long verifiedPosition = 0;
1193 
1194         public VerifyingIndexInput(IndexInput input) {
1195             this(input, new BufferedChecksum(new CRC32()));
1196         }
1197 
1198         public VerifyingIndexInput(IndexInput input, Checksum digest) {
1199             super("VerifyingIndexInput(" + input + ")");
1200             this.input = input;
1201             this.digest = digest;
1202             checksumPosition = input.length() - 8;
1203         }
1204 
1205         @Override
1206         public byte readByte() throws IOException {
1207             long pos = input.getFilePointer();
1208             final byte b = input.readByte();
1209             pos++;
1210             if (pos > verifiedPosition) {
1211                 if (pos <= checksumPosition) {
1212                     digest.update(b);
1213                 } else {
1214                     checksum[(int) (pos - checksumPosition - 1)] = b;
1215                 }
1216                 verifiedPosition = pos;
1217             }
1218             return b;
1219         }
1220 
1221         @Override
1222         public void readBytes(byte[] b, int offset, int len)
1223                 throws IOException {
1224             long pos = input.getFilePointer();
1225             input.readBytes(b, offset, len);
1226             if (pos + len > verifiedPosition) {
1227                 // Conversion to int is safe here because (verifiedPosition - pos) can be at most len, which is integer
1228                 int alreadyVerified = (int) Math.max(0, verifiedPosition - pos);
1229                 if (pos < checksumPosition) {
1230                     if (pos + len < checksumPosition) {
1231                         digest.update(b, offset + alreadyVerified, len - alreadyVerified);
1232                     } else {
1233                         int checksumOffset = (int) (checksumPosition - pos);
1234                         if (checksumOffset - alreadyVerified > 0) {
1235                             digest.update(b, offset + alreadyVerified, checksumOffset - alreadyVerified);
1236                         }
1237                         System.arraycopy(b, offset + checksumOffset, checksum, 0, len - checksumOffset);
1238                     }
1239                 } else {
1240                     // Conversion to int is safe here because checksumPosition is (file length - 8) so
1241                     // (pos - checksumPosition) cannot be bigger than 8 unless we are reading after the end of file
1242                     assert pos - checksumPosition < 8;
1243                     System.arraycopy(b, offset, checksum, (int) (pos - checksumPosition), len);
1244                 }
1245                 verifiedPosition = pos + len;
1246             }
1247         }
1248 
1249         @Override
1250         public long getChecksum() {
1251             return digest.getValue();
1252         }
1253 
1254         @Override
1255         public void seek(long pos) throws IOException {
1256             if (pos < verifiedPosition) {
1257                 // going within verified region - just seek there
1258                 input.seek(pos);
1259             } else {
1260                 if (verifiedPosition > getFilePointer()) {
1261                     // portion of the skip region is verified and portion is not
1262                     // skipping the verified portion
1263                     input.seek(verifiedPosition);
1264                     // and checking unverified
1265                     skipBytes(pos - verifiedPosition);
1266                 } else {
1267                     skipBytes(pos - getFilePointer());
1268                 }
1269             }
1270         }
1271 
1272         @Override
1273         public void close() throws IOException {
1274             input.close();
1275         }
1276 
1277         @Override
1278         public long getFilePointer() {
1279             return input.getFilePointer();
1280         }
1281 
1282         @Override
1283         public long length() {
1284             return input.length();
1285         }
1286 
1287         @Override
1288         public IndexInput clone() {
1289             throw new UnsupportedOperationException();
1290         }
1291 
1292         @Override
1293         public IndexInput slice(String sliceDescription, long offset, long length) throws IOException {
1294             throw new UnsupportedOperationException();
1295         }
1296 
1297         public long getStoredChecksum() {
1298             return new ByteArrayDataInput(checksum).readLong();
1299         }
1300 
1301         public long verify() throws CorruptIndexException {
1302             long storedChecksum = getStoredChecksum();
1303             if (getChecksum() == storedChecksum) {
1304                 return storedChecksum;
1305             }
1306             throw new CorruptIndexException("verification failed : calculated=" + Store.digestToString(getChecksum()) +
1307                     " stored=" + Store.digestToString(storedChecksum), this);
1308         }
1309 
1310     }

1322     /**
1323      * Marks this store as corrupted. This method writes a <tt>corrupted_${uuid}</tt> file containing the given exception
1324      * message. If a store contains a <tt>corrupted_${uuid}</tt> file {@link #isMarkedCorrupted()} will return <code>true</code>.
1325      */
1326     public void markStoreCorrupted(IOException exception) throws IOException {
1327         ensureOpen();
1328         if (!isMarkedCorrupted()) {
1329             String uuid = CORRUPTED + Strings.randomBase64UUID();
1330             try (IndexOutput output = this.directory().createOutput(uuid, IOContext.DEFAULT)) {
1331                 CodecUtil.writeHeader(output, CODEC, VERSION);
1332                 output.writeString(ExceptionsHelper.detailedMessage(exception, true, 0)); // handles null exception
1333                 output.writeString(ExceptionsHelper.stackTrace(exception));
1334                 CodecUtil.writeFooter(output);
1335             } catch (IOException ex) {
1336                 logger.warn("Can't mark store as corrupted", ex);
1337             }
1338             directory().sync(Collections.singleton(uuid));
1339         }

1379             for (String file : files) {
