172         private long addLong(long current, long other) {
173             if (other == -1) {
174                 return current;
175             }
176             if (current == -1) {
177                 return other;
178             }
179             return current + other;
180         }
181 
182         private double addDouble(double current, double other) {
183             if (other == -1) {
184                 return current;
185             }
186             if (current == -1) {
187                 return other;
188             }
189             return current + other;
190         }

195             available = addLong(available, info.available);

198             diskReadBytes = addLong(diskReadBytes, info.diskReadBytes);
199             diskWriteBytes = addLong(diskWriteBytes, info.diskWriteBytes);
200             diskQueue = addDouble(diskQueue, info.diskQueue);
201             diskServiceTime = addDouble(diskServiceTime, info.diskServiceTime);
202         }

204         static final class Fields {
205             static final XContentBuilderString PATH = new XContentBuilderString("path");
206             static final XContentBuilderString MOUNT = new XContentBuilderString("mount");
207             static final XContentBuilderString DEV = new XContentBuilderString("dev");
208             static final XContentBuilderString TOTAL = new XContentBuilderString("total");
209             static final XContentBuilderString TOTAL_IN_BYTES = new XContentBuilderString("total_in_bytes");
210             static final XContentBuilderString FREE = new XContentBuilderString("free");
211             static final XContentBuilderString FREE_IN_BYTES = new XContentBuilderString("free_in_bytes");
212             static final XContentBuilderString AVAILABLE = new XContentBuilderString("available");
213             static final XContentBuilderString AVAILABLE_IN_BYTES = new XContentBuilderString("available_in_bytes");
214             static final XContentBuilderString DISK_READS = new XContentBuilderString("disk_reads");
215             static final XContentBuilderString DISK_WRITES = new XContentBuilderString("disk_writes");
216             static final XContentBuilderString DISK_IO_OP = new XContentBuilderString("disk_io_op");
217             static final XContentBuilderString DISK_READ_SIZE = new XContentBuilderString("disk_read_size");
218             static final XContentBuilderString DISK_READ_SIZE_IN_BYTES = new XContentBuilderString("disk_read_size_in_bytes");
219             static final XContentBuilderString DISK_WRITE_SIZE = new XContentBuilderString("disk_write_size");
220             static final XContentBuilderString DISK_WRITE_SIZE_IN_BYTES = new XContentBuilderString("disk_write_size_in_bytes");
221             static final XContentBuilderString DISK_IO_SIZE = new XContentBuilderString("disk_io_size");
222             static final XContentBuilderString DISK_IO_IN_BYTES = new XContentBuilderString("disk_io_size_in_bytes");
223             static final XContentBuilderString DISK_QUEUE = new XContentBuilderString("disk_queue");
224             static final XContentBuilderString DISK_SERVICE_TIME = new XContentBuilderString("disk_service_time");
225         }

227         @Override
228         public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
229             builder.startObject();
230             if (path != null) {
231                 builder.field(Fields.PATH, path, XContentBuilder.FieldCaseConversion.NONE);
232             }
233             if (mount != null) {
234                 builder.field(Fields.MOUNT, mount, XContentBuilder.FieldCaseConversion.NONE);
235             }
236             if (dev != null) {
237                 builder.field(Fields.DEV, dev, XContentBuilder.FieldCaseConversion.NONE);
238             }
239 
240             if (total != -1) {
241                 builder.byteSizeField(Fields.TOTAL_IN_BYTES, Fields.TOTAL, total);
242             }
243             if (free != -1) {
244                 builder.byteSizeField(Fields.FREE_IN_BYTES, Fields.FREE, free);
245             }
246             if (available != -1) {
247                 builder.byteSizeField(Fields.AVAILABLE_IN_BYTES, Fields.AVAILABLE, available);
248             }
249 
250             long iop = -1;
251 
252             if (diskReads != -1) {
253                 iop = diskReads;
254                 builder.field(Fields.DISK_READS, diskReads);
255             }
256             if (diskWrites != -1) {
257                 if (iop != -1) {
258                     iop += diskWrites;
259                 } else {
260                     iop = diskWrites;
261                 }
262                 builder.field(Fields.DISK_WRITES, diskWrites);
263             }
264 
265             if (iop != -1) {
266                 builder.field(Fields.DISK_IO_OP, iop);
267             }
268 
269             long ioBytes = -1;
270 
271             if (diskReadBytes != -1) {
272                 ioBytes = diskReadBytes;
273                 builder.byteSizeField(Fields.DISK_READ_SIZE_IN_BYTES, Fields.DISK_READ_SIZE, diskReadBytes);
274             }
275             if (diskWriteBytes != -1) {
276                 if (ioBytes != -1) {
277                     ioBytes += diskWriteBytes;
278                 } else {
279                     ioBytes = diskWriteBytes;
280                 }
281                 builder.byteSizeField(Fields.DISK_WRITE_SIZE_IN_BYTES, Fields.DISK_WRITE_SIZE, diskWriteBytes);
282             }
283 
284             if (ioBytes != -1) {
285                 builder.byteSizeField(Fields.DISK_IO_IN_BYTES, Fields.DISK_IO_SIZE, ioBytes);
286             }
287 
288             if (diskQueue != -1) {
289                 builder.field(Fields.DISK_QUEUE, Strings.format1Decimals(diskQueue, ""));
290             }
291             if (diskServiceTime != -1) {
292                 builder.field(Fields.DISK_SERVICE_TIME, Strings.format1Decimals(diskServiceTime, ""));
293             }
294 
295             builder.endObject();
296             return builder;
297         }

308     public FsStats(long timestamp, Info[] infos) {
309         this.timestamp = timestamp;
310         this.infos = infos;
311         this.total = null;
312     }
313 
314     public Info getTotal() {
315         return total();
316     }
317 
318     public Info total() {
319         if (total != null) {
320             return total;
321         }
322         Info res = new Info();
323         Set<String> seenDevices = new HashSet<>(infos.length);
324         for (Info subInfo : infos) {
325             if (subInfo.dev != null) {
326                 if (!seenDevices.add(subInfo.dev)) {
327                     continue; // already added numbers for this device;
328                 }
329             }
330             res.add(subInfo);
331         }
332         total = res;
333         return res;
334     }

337         return timestamp;
338     }

345     public static FsStats readFsStats(StreamInput in) throws IOException {
346         FsStats stats = new FsStats();
347         stats.readFrom(in);
348         return stats;
349     }

351     @Override
352     public void readFrom(StreamInput in) throws IOException {
353         timestamp = in.readVLong();
354         infos = new Info[in.readVInt()];
355         for (int i = 0; i < infos.length; i++) {
356             infos[i] = Info.readInfoFrom(in);
357         }
358     }
359 
360     @Override
361     public void writeTo(StreamOutput out) throws IOException {
362         out.writeVLong(timestamp);
363         out.writeVInt(infos.length);
364         for (Info info : infos) {
365             info.writeTo(out);
366         }
367     }

369     static final class Fields {
370         static final XContentBuilderString FS = new XContentBuilderString("fs");
371         static final XContentBuilderString TIMESTAMP = new XContentBuilderString("timestamp");
372         static final XContentBuilderString DATA = new XContentBuilderString("data");
373         static final XContentBuilderString TOTAL = new XContentBuilderString("total");
374     }
375 
376     @Override
377     public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
378         builder.startObject(Fields.FS);
379         builder.field(Fields.TIMESTAMP, timestamp);
380         builder.field(Fields.TOTAL);
381         total().toXContent(builder, params);
382         builder.startArray(Fields.DATA);
383         for (Info info : infos) {
384             info.toXContent(builder, params);
385         }
386         builder.endArray();
387         builder.endObject();
388         return builder;
389     }
