175 public class ActionModule extends AbstractModule {
176 
177     private final Map<String, ActionEntry> actions = Maps.newHashMap();
178     private final List<Class<? extends ActionFilter>> actionFilters = Lists.newArrayList();
179 
180     static class ActionEntry<Request extends ActionRequest, Response extends ActionResponse> {
181         public final GenericAction<Request, Response> action;
182         public final Class<? extends TransportAction<Request, Response>> transportAction;
183         public final Class[] supportTransportActions;
184 
185         ActionEntry(GenericAction<Request, Response> action, Class<? extends TransportAction<Request, Response>> transportAction, Class... supportTransportActions) {
186             this.action = action;
187             this.transportAction = transportAction;
188             this.supportTransportActions = supportTransportActions;
189         }
190 
191 
192     }
193 
194     private final boolean proxy;
195 
196     public ActionModule(boolean proxy) {
197         this.proxy = proxy;
198     }
199 
200     /**
201      * Registers an action.
202      *
203      * @param action                  The action type.
204      * @param transportAction         The transport action implementing the actual action.
205      * @param supportTransportActions Any support actions that are needed by the transport action.
206      * @param <Request>               The request type.
207      * @param <Response>              The response type.
208      */
209     public <Request extends ActionRequest, Response extends ActionResponse> void registerAction(GenericAction<Request, Response> action, Class<? extends TransportAction<Request, Response>> transportAction, Class... supportTransportActions) {
210         actions.put(action.name(), new ActionEntry<>(action, transportAction, supportTransportActions));
211     }
212 
213     public ActionModule registerFilter(Class<? extends ActionFilter> actionFilter) {
214         actionFilters.add(actionFilter);
215         return this;
216     }
217

340                 bind(entry.getValue().transportAction).asEagerSingleton();

342                 for (Class supportAction : entry.getValue().supportTransportActions) {
343                     bind(supportAction).asEagerSingleton();
344                 }
345             }
346         }
