45  /**
46   * Service responsible for submitting update index settings requests
47   */
48  public class MetaDataUpdateSettingsService extends AbstractComponent implements ClusterStateListener {
49  
50      // the value we recognize in the "max" position to mean all the nodes
51      private static final String ALL_NODES_VALUE = "all";
52  
53      private final ClusterService clusterService;
54  
55      private final AllocationService allocationService;
56  
57      private final DynamicSettings dynamicSettings;
58  
59      @Inject
60      public MetaDataUpdateSettingsService(Settings settings, ClusterService clusterService, AllocationService allocationService, @IndexDynamicSettings DynamicSettings dynamicSettings) {
61          super(settings);
62          this.clusterService = clusterService;
63          this.clusterService.add(this);
64          this.allocationService = allocationService;
65          this.dynamicSettings = dynamicSettings;
66      }
67  
68      @Override
69      public void clusterChanged(ClusterChangedEvent event) {
70          // update an index with number of replicas based on data nodes if possible
71          if (!event.state().nodes().localNodeMaster()) {
72              return;
73          }
74          // we will want to know this for translating "all" to a number
75          final int dataNodeCount = event.state().nodes().dataNodes().size();
76  
77          Map<Integer, List<String>> nrReplicasChanged = new HashMap<>();
78  
79          // we need to do this each time in case it was changed by update settings
80          for (final IndexMetaData indexMetaData : event.state().metaData()) {
81              String autoExpandReplicas = indexMetaData.settings().get(IndexMetaData.SETTING_AUTO_EXPAND_REPLICAS);
82              if (autoExpandReplicas != null && Booleans.parseBoolean(autoExpandReplicas, true)) { // Booleans only work for false values, just as we want it here
83                  try {
84                      final int min;
85                      final int max;
86  
87                      final int dash = autoExpandReplicas.indexOf('-');
88                      if (-1 == dash) {
89                          logger.warn("Unexpected value [{}] for setting [{}]; it should be dash delimited",
90                                  autoExpandReplicas, IndexMetaData.SETTING_AUTO_EXPAND_REPLICAS);
91                          continue;
92                      }
93                      final String sMin = autoExpandReplicas.substring(0, dash);
94                      try {
95                          min = Integer.parseInt(sMin);
96                      } catch (NumberFormatException e) {
97                          logger.warn("failed to set [{}], minimum value is not a number [{}]",
98                                  e, IndexMetaData.SETTING_AUTO_EXPAND_REPLICAS, sMin);
99                          continue;
100                     }
101                     String sMax = autoExpandReplicas.substring(dash + 1);
102                     if (sMax.equals(ALL_NODES_VALUE)) {
103                         max = dataNodeCount - 1;
104                     } else {
105                         try {
106                             max = Integer.parseInt(sMax);
107                         } catch (NumberFormatException e) {
108                             logger.warn("failed to set [{}], maximum value is neither [{}] nor a number [{}]",
109                                     e, IndexMetaData.SETTING_AUTO_EXPAND_REPLICAS, ALL_NODES_VALUE, sMax);
110                             continue;
111                         }
112                     }
113 
114                     int numberOfReplicas = dataNodeCount - 1;
115                     if (numberOfReplicas < min) {
116                         numberOfReplicas = min;
117                     } else if (numberOfReplicas > max) {
118                         numberOfReplicas = max;
119                     }
120 
121                     // same value, nothing to do there
122                     if (numberOfReplicas == indexMetaData.numberOfReplicas()) {
123                         continue;
124                     }
125 
126                     if (numberOfReplicas >= min && numberOfReplicas <= max) {
127 
128                         if (!nrReplicasChanged.containsKey(numberOfReplicas)) {
129                             nrReplicasChanged.put(numberOfReplicas, new ArrayList<String>());
130                         }
131 
132                         nrReplicasChanged.get(numberOfReplicas).add(indexMetaData.index());
133                     }
134                 } catch (Exception e) {
135                     logger.warn("[{}] failed to parse auto expand replicas", e, indexMetaData.index());
136                 }
137             }
138         }
139 
140         if (nrReplicasChanged.size() > 0) {
141             for (final Integer fNumberOfReplicas : nrReplicasChanged.keySet()) {
142                 Settings settings = ImmutableSettings.settingsBuilder().put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, fNumberOfReplicas).build();
143                 final List<String> indices = nrReplicasChanged.get(fNumberOfReplicas);
144 
145                 UpdateSettingsClusterStateUpdateRequest updateRequest = new UpdateSettingsClusterStateUpdateRequest()
146                         .indices(indices.toArray(new String[indices.size()])).settings(settings)
147                         .ackTimeout(TimeValue.timeValueMillis(0)) //no need to wait for ack here
148                         .masterNodeTimeout(TimeValue.timeValueMinutes(10));
149 
150                 updateSettings(updateRequest, new ActionListener<ClusterStateUpdateResponse>() {
151                     @Override
152                     public void onResponse(ClusterStateUpdateResponse response) {
153                         for (String index : indices) {
154                             logger.info("[{}] auto expanded replicas to [{}]", index, fNumberOfReplicas);
155                         }
156                     }
157 
158                     @Override
159                     public void onFailure(Throwable t) {
160                         for (String index : indices) {
161                             logger.warn("[{}] fail to auto expand replicas to [{}]", index, fNumberOfReplicas);
162                         }
163                     }
164                 });
165             }
166         }
167     }

169     public void updateSettings(final UpdateSettingsClusterStateUpdateRequest request, final ActionListener<ClusterStateUpdateResponse> listener) {
170         ImmutableSettings.Builder updatedSettingsBuilder = ImmutableSettings.settingsBuilder();
171         for (Map.Entry<String, String> entry : request.settings().getAsMap().entrySet()) {
172             if (entry.getKey().equals("index")) {
173                 continue;
174             }
175             if (!entry.getKey().startsWith("index.")) {
176                 updatedSettingsBuilder.put("index." + entry.getKey(), entry.getValue());
177             } else {
178                 updatedSettingsBuilder.put(entry.getKey(), entry.getValue());
179             }
180         }
181         // never allow to change the number of shards
182         for (String key : updatedSettingsBuilder.internalMap().keySet()) {
183             if (key.equals(IndexMetaData.SETTING_NUMBER_OF_SHARDS)) {
184                 listener.onFailure(new ElasticsearchIllegalArgumentException("can't change the number of shards for an index"));
185                 return;
186             }
187         }
188 
189         final Settings closeSettings = updatedSettingsBuilder.build();
190 
191         final Set<String> removedSettings = Sets.newHashSet();
192         final Set<String> errors = Sets.newHashSet();
193         for (Map.Entry<String, String> setting : updatedSettingsBuilder.internalMap().entrySet()) {
194             if (!dynamicSettings.hasDynamicSetting(setting.getKey())) {
195                 removedSettings.add(setting.getKey());
196             } else {
197                 String error = dynamicSettings.validateDynamicSetting(setting.getKey(), setting.getValue());
198                 if (error != null) {
199                     errors.add("[" + setting.getKey() + "] - " + error);
200                 }
201             }
202         }
203 
204         if (!errors.isEmpty()) {
205             listener.onFailure(new ElasticsearchIllegalArgumentException("can't process the settings: " + errors.toString()));
206             return;
207         }
208 
209         if (!removedSettings.isEmpty()) {
210             for (String removedSetting : removedSettings) {
211                 updatedSettingsBuilder.remove(removedSetting);
212             }
213         }
214         final Settings openSettings = updatedSettingsBuilder.build();
215 
216         clusterService.submitStateUpdateTask("update-settings", Priority.URGENT, new AckedClusterStateUpdateTask<ClusterStateUpdateResponse>(request, listener) {
217 
218             @Override
219             protected ClusterStateUpdateResponse newResponse(boolean acknowledged) {
220                 return new ClusterStateUpdateResponse(acknowledged);
221             }
222 
223             @Override
224             public ClusterState execute(ClusterState currentState) {
225                 String[] actualIndices = currentState.metaData().concreteIndices(IndicesOptions.strictExpand(), request.indices());
226                 RoutingTable.Builder routingTableBuilder = RoutingTable.builder(currentState.routingTable());
227                 MetaData.Builder metaDataBuilder = MetaData.builder(currentState.metaData());
228 
229                 // allow to change any settings to a close index, and only allow dynamic settings to be changed
230                 // on an open index
231                 Set<String> openIndices = Sets.newHashSet();
232                 Set<String> closeIndices = Sets.newHashSet();
233                 for (String index : actualIndices) {
234                     if (currentState.metaData().index(index).state() == IndexMetaData.State.OPEN) {
235                         openIndices.add(index);
236                     } else {
237                         closeIndices.add(index);
238                     }
239                 }
240 
241                 if (!removedSettings.isEmpty() && !openIndices.isEmpty()) {
242                     throw new ElasticsearchIllegalArgumentException(String.format(Locale.ROOT,
243                             "Can't update non dynamic settings[%s] for open indices[%s]",
244                             removedSettings,
245                             openIndices
246                     ));
247                 }
248 
249                 int updatedNumberOfReplicas = openSettings.getAsInt(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, -1);
250                 if (updatedNumberOfReplicas != -1) {
251                     routingTableBuilder.updateNumberOfReplicas(updatedNumberOfReplicas, actualIndices);
252                     metaDataBuilder.updateNumberOfReplicas(updatedNumberOfReplicas, actualIndices);
253                     logger.info("updating number_of_replicas to [{}] for indices {}", updatedNumberOfReplicas, actualIndices);
254                 }
255 
256                 ClusterBlocks.Builder blocks = ClusterBlocks.builder().blocks(currentState.blocks());
257                 Boolean updatedReadOnly = openSettings.getAsBoolean(IndexMetaData.SETTING_READ_ONLY, null);
258                 if (updatedReadOnly != null) {
259                     for (String index : actualIndices) {
260                         if (updatedReadOnly) {
261                             blocks.addIndexBlock(index, IndexMetaData.INDEX_READ_ONLY_BLOCK);
262                         } else {
263                             blocks.removeIndexBlock(index, IndexMetaData.INDEX_READ_ONLY_BLOCK);
264                         }
265                     }
266                 }
267                 Boolean updateMetaDataBlock = openSettings.getAsBoolean(IndexMetaData.SETTING_BLOCKS_METADATA, null);
268                 if (updateMetaDataBlock != null) {
269                     for (String index : actualIndices) {
270                         if (updateMetaDataBlock) {
271                             blocks.addIndexBlock(index, IndexMetaData.INDEX_METADATA_BLOCK);
272                         } else {
273                             blocks.removeIndexBlock(index, IndexMetaData.INDEX_METADATA_BLOCK);
274                         }
275                     }
276                 }
277 
278                 Boolean updateWriteBlock = openSettings.getAsBoolean(IndexMetaData.SETTING_BLOCKS_WRITE, null);
279                 if (updateWriteBlock != null) {
280                     for (String index : actualIndices) {
281                         if (updateWriteBlock) {
282                             blocks.addIndexBlock(index, IndexMetaData.INDEX_WRITE_BLOCK);
283                         } else {
284                             blocks.removeIndexBlock(index, IndexMetaData.INDEX_WRITE_BLOCK);
285                         }
286                     }
287                 }
288 
289                 Boolean updateReadBlock = openSettings.getAsBoolean(IndexMetaData.SETTING_BLOCKS_READ, null);
290                 if (updateReadBlock != null) {
291                     for (String index : actualIndices) {
292                         if (updateReadBlock) {
293                             blocks.addIndexBlock(index, IndexMetaData.INDEX_READ_BLOCK);
294                         } else {
295                             blocks.removeIndexBlock(index, IndexMetaData.INDEX_READ_BLOCK);
296                         }
297                     }
298                 }
299 
300                 if (!openIndices.isEmpty()) {
301                     String[] indices = openIndices.toArray(new String[openIndices.size()]);
302                     metaDataBuilder.updateSettings(openSettings, indices);
303                 }
304 
305                 if (!closeIndices.isEmpty()) {
306                     String[] indices = closeIndices.toArray(new String[closeIndices.size()]);
307                     metaDataBuilder.updateSettings(closeSettings, indices);
308                 }
309 
310 
311                 ClusterState updatedState = ClusterState.builder(currentState).metaData(metaDataBuilder).routingTable(routingTableBuilder).blocks(blocks).build();
312 
313                 // now, reroute in case things change that require it (like number of replicas)
314                 RoutingAllocation.Result routingResult = allocationService.reroute(updatedState);
315                 updatedState = ClusterState.builder(updatedState).routingResult(routingResult).build();
316 
317                 return updatedState;
318             }
319         });
320     }
