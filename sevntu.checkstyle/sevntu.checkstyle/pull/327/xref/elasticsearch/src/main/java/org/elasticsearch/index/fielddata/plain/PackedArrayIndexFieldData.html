97      public PackedArrayIndexFieldData(Index index, @IndexSettings Settings indexSettings, FieldMapper.Names fieldNames,
98                                       FieldDataType fieldDataType, IndexFieldDataCache cache, NumericType numericType,
99                                       CircuitBreakerService breakerService) {
100         super(index, indexSettings, fieldNames, fieldDataType, cache);
101         Preconditions.checkNotNull(numericType);
102         Preconditions.checkArgument(EnumSet.of(NumericType.BYTE, NumericType.SHORT, NumericType.INT, NumericType.LONG).contains(numericType), getClass().getSimpleName() + " only supports integer types, not " + numericType);
103         this.numericType = numericType;
104         this.breakerService = breakerService;
105     }

112     @Override
113     public AtomicNumericFieldData loadDirect(LeafReaderContext context) throws Exception {
114         final LeafReader reader = context.reader();
115         Terms terms = reader.terms(getFieldNames().indexName());
116         AtomicNumericFieldData data = null;
117         PackedArrayEstimator estimator = new PackedArrayEstimator(breakerService.getBreaker(CircuitBreaker.FIELDDATA), getNumericType(), getFieldNames().fullName());
118         if (terms == null) {
119             data = AtomicLongFieldData.empty(reader.maxDoc());
120             estimator.adjustForNoTerms(data.ramBytesUsed());
121             return data;
122         }
123         // TODO: how can we guess the number of terms? numerics end up creating more terms per value...
124         // Lucene encodes numeric data so that the lexicographical (encoded) order matches the integer order so we know the sequence of
125         // longs is going to be monotonically increasing
126         final PackedLongValues.Builder valuesBuilder = PackedLongValues.monotonicBuilder(PackedInts.COMPACT);
127 
128         final float acceptableTransientOverheadRatio = fieldDataType.getSettings().getAsFloat("acceptable_transient_overhead_ratio", OrdinalsBuilder.DEFAULT_ACCEPTABLE_OVERHEAD_RATIO);
129         TermsEnum termsEnum = estimator.beforeLoad(terms);
130         boolean success = false;
131         try (OrdinalsBuilder builder = new OrdinalsBuilder(-1, reader.maxDoc(), acceptableTransientOverheadRatio)) {
132             BytesRefIterator iter = builder.buildFromTerms(termsEnum);
133             BytesRef term;
134             assert !getNumericType().isFloatingPoint();
135             final boolean indexedAsLong = getNumericType().requiredBits() > 32;
136             while ((term = iter.next()) != null) {
137                 final long value = indexedAsLong
138                         ? NumericUtils.prefixCodedToLong(term)
139                         : NumericUtils.prefixCodedToInt(term);
140                 valuesBuilder.add(value);
141             }
142             final PackedLongValues values = valuesBuilder.build();
143             final Ordinals build = builder.build(fieldDataType.getSettings());
144             CommonSettings.MemoryStorageFormat formatHint = CommonSettings.getMemoryStorageHint(fieldDataType);
145 
146             RandomAccessOrds ordinals = build.ordinals();
147             if (FieldData.isMultiValued(ordinals) || formatHint == CommonSettings.MemoryStorageFormat.ORDINALS) {
148                 final long ramBytesUsed = build.ramBytesUsed() + values.ramBytesUsed();
149                 data = new AtomicLongFieldData(ramBytesUsed) {
150 
151                     @Override
152                     public SortedNumericDocValues getLongValues() {
153                         return withOrdinals(build, values, reader.maxDoc());
154                     }
155 
156                     @Override
157                     public Collection<Accountable> getChildResources() {
158                         List<Accountable> resources = new ArrayList<>();
159                         resources.add(Accountables.namedAccountable("ordinals", build));
160                         resources.add(Accountables.namedAccountable("values", values));
161                         return Collections.unmodifiableList(resources);
162                     }
163                 };
164             } else {
165                 final BitSet docsWithValues = builder.buildDocsWithValuesSet();
166 
167                 long minV, maxV;
168                 minV = maxV = 0;
169                 if (values.size() > 0) {
170                     minV = values.get(0);
171                     maxV = values.get(values.size() - 1);
172                 }
173 
174 
175                 final float acceptableOverheadRatio = fieldDataType.getSettings().getAsFloat("acceptable_overhead_ratio", PackedInts.DEFAULT);
176                 final int pageSize = fieldDataType.getSettings().getAsInt("single_value_page_size", 1024);
177 
178                 if (formatHint == null) {
179                     formatHint = chooseStorageFormat(reader, values, build, ordinals, minV, maxV, acceptableOverheadRatio, pageSize);
180                 }
181 
182                 logger.trace("single value format for field [{}] set to [{}]", getFieldNames().fullName(), formatHint);
183 
184                 switch (formatHint) {
185                     case PACKED:
186                         // Encode document without a value with a special value
187                         long missingV = 0;
188                         if (docsWithValues != null) {
189                             if ((maxV - minV + 1) == values.size()) {
190                                 // values are dense
191                                 if (minV > Long.MIN_VALUE) {
192                                     missingV = --minV;
193                                 } else {
194                                     assert maxV != Long.MAX_VALUE;
195                                     missingV = ++maxV;
196                                 }
197                             } else {
198                                 for (long i = 1; i < values.size(); ++i) {
199                                     if (values.get(i) > values.get(i - 1) + 1) {
200                                         missingV = values.get(i - 1) + 1;
201                                         break;
202                                     }
203                                 }
204                             }
205                             missingV -= minV;
206                         }
207                         final long missingValue = missingV;
208                         final long minValue = minV;
209                         final long maxValue = maxV;
210 
211                         final long valuesDelta = maxValue - minValue;
212                         int bitsRequired = valuesDelta < 0 ? 64 : PackedInts.bitsRequired(valuesDelta);
213                         final PackedInts.Mutable sValues = PackedInts.getMutable(reader.maxDoc(), bitsRequired, acceptableOverheadRatio);
214 
215                         if (docsWithValues != null) {
216                             sValues.fill(0, sValues.size(), missingV);
217                         }
218 
219                         for (int i = 0; i < reader.maxDoc(); i++) {
220                             ordinals.setDocument(i);
221                             if (ordinals.cardinality() > 0) {
222                                 final long ord = ordinals.ordAt(0);
223                                 long value = values.get(ord);
224                                 sValues.set(i, value - minValue);
225                             }
226                         }
227                         long ramBytesUsed = values.ramBytesUsed() + (docsWithValues == null ? 0 : docsWithValues.ramBytesUsed());
228                         data = new AtomicLongFieldData(ramBytesUsed) {
229 
230                             @Override
231                             public SortedNumericDocValues getLongValues() {
232                                 if (docsWithValues == null) {
233                                     return singles(sValues, minValue);
234                                 } else {
235                                     return sparseSingles(sValues, minValue, missingValue, reader.maxDoc());
236                                 }
237                             }
238                             
239                             @Override
240                             public Collection<Accountable> getChildResources() {
241                                 List<Accountable> resources = new ArrayList<>();
242                                 resources.add(Accountables.namedAccountable("values", sValues));
243                                 if (docsWithValues != null) {
244                                     resources.add(Accountables.namedAccountable("missing bitset", docsWithValues));
245                                 }
246                                 return Collections.unmodifiableList(resources);
247                             }
248 
249                         };
250                         break;
251                     case PAGED:
252                         final PackedLongValues.Builder dpValues = PackedLongValues.deltaPackedBuilder(pageSize, acceptableOverheadRatio);
253 
254                         long lastValue = 0;
255                         for (int i = 0; i < reader.maxDoc(); i++) {
256                             ordinals.setDocument(i);
257                             if (ordinals.cardinality() > 0) {
258                                 final long ord = ordinals.ordAt(i);
259                                 lastValue = values.get(ord);
260                             }
261                             dpValues.add(lastValue);
262                         }
263                         final PackedLongValues pagedValues = dpValues.build();
264                         ramBytesUsed = pagedValues.ramBytesUsed();
265                         if (docsWithValues != null) {
266                             ramBytesUsed += docsWithValues.ramBytesUsed();
267                         }
268                         data = new AtomicLongFieldData(ramBytesUsed) {
269 
270                             @Override
271                             public SortedNumericDocValues getLongValues() {
272                                 return pagedSingles(pagedValues, docsWithValues);
273                             }
274 
275                             @Override
276                             public Collection<Accountable> getChildResources() {
277                                 List<Accountable> resources = new ArrayList<>();
278                                 resources.add(Accountables.namedAccountable("values", pagedValues));
279                                 if (docsWithValues != null) {
280                                     resources.add(Accountables.namedAccountable("missing bitset", docsWithValues));
281                                 }
282                                 return Collections.unmodifiableList(resources);
283                             }
284                             
285                         };
286                         break;
287                     case ORDINALS:
288                         ramBytesUsed = build.ramBytesUsed() + values.ramBytesUsed();
289                         data = new AtomicLongFieldData(ramBytesUsed) {
290 
291                             @Override
292                             public SortedNumericDocValues getLongValues() {
293                                 return withOrdinals(build, values, reader.maxDoc());
294                             }
295                             
296                             @Override
297                             public Collection<Accountable> getChildResources() {
298                                 List<Accountable> resources = new ArrayList<>();
299                                 resources.add(Accountables.namedAccountable("ordinals", build));
300                                 resources.add(Accountables.namedAccountable("values", values));
301                                 return Collections.unmodifiableList(resources);
302                             }
303 
304                         };
305                         break;
306                     default:
307                         throw new ElasticsearchException("unknown memory format: " + formatHint);
308                 }
309 
310             }
311 
312             success = true;
313             return data;
314         } finally {
315             if (!success) {
316                 // If something went wrong, unwind any current estimations we've made
317                 estimator.afterLoad(termsEnum, 0);
318             } else {
319                 // Adjust as usual, based on the actual size of the field data
320                 estimator.afterLoad(termsEnum, data.ramBytesUsed());
321             }
322 
323         }
324 
325     }

327     protected CommonSettings.MemoryStorageFormat chooseStorageFormat(LeafReader reader, PackedLongValues values, Ordinals build, RandomAccessOrds ordinals,
328                                                                      long minValue, long maxValue, float acceptableOverheadRatio, int pageSize) {
329 
330         CommonSettings.MemoryStorageFormat format;
331 
332         // estimate memory usage for a single packed array
333         long packedDelta = maxValue - minValue + 1; // allow for a missing value
334         // valuesDelta can be negative if the difference between max and min values overflows the positive side of longs.
335         int bitsRequired = packedDelta < 0 ? 64 : PackedInts.bitsRequired(packedDelta);
336         PackedInts.FormatAndBits formatAndBits = PackedInts.fastestFormatAndBits(reader.maxDoc(), bitsRequired, acceptableOverheadRatio);
337         final long singleValuesSize = formatAndBits.format.longCount(PackedInts.VERSION_CURRENT, reader.maxDoc(), formatAndBits.bitsPerValue) * 8L;
338 
339         // ordinal memory usage
340         final long ordinalsSize = build.ramBytesUsed() + values.ramBytesUsed();
341 
342         // estimate the memory signature of paged packing
343         long pagedSingleValuesSize = (reader.maxDoc() / pageSize + 1) * RamUsageEstimator.NUM_BYTES_OBJECT_REF; // array of pages
344         int pageIndex = 0;
345         long pageMinOrdinal = Long.MAX_VALUE;
346         long pageMaxOrdinal = Long.MIN_VALUE;
347         for (int i = 1; i < reader.maxDoc(); ++i, pageIndex = (pageIndex + 1) % pageSize) {
348             ordinals.setDocument(i);
349             if (ordinals.cardinality() > 0) {
350                 long ordinal = ordinals.ordAt(0);
351                 pageMaxOrdinal = Math.max(ordinal, pageMaxOrdinal);
352                 pageMinOrdinal = Math.min(ordinal, pageMinOrdinal);
353             }
354             if (pageIndex == pageSize - 1) {
355                 // end of page, we now know enough to estimate memory usage
356                 pagedSingleValuesSize += getPageMemoryUsage(values, acceptableOverheadRatio, pageSize, pageMinOrdinal, pageMaxOrdinal);
357 
358                 pageMinOrdinal = Long.MAX_VALUE;
359                 pageMaxOrdinal = Long.MIN_VALUE;
360             }
361         }
362 
363         if (pageIndex > 0) {
364             // last page estimation
365             pageIndex++;
366             pagedSingleValuesSize += getPageMemoryUsage(values, acceptableOverheadRatio, pageSize, pageMinOrdinal, pageMaxOrdinal);
367         }
368 
369         if (ordinalsSize < singleValuesSize) {
370             if (ordinalsSize < pagedSingleValuesSize) {
371                 format = CommonSettings.MemoryStorageFormat.ORDINALS;
372             } else {
373                 format = CommonSettings.MemoryStorageFormat.PAGED;
374             }
375         } else {
376             if (pagedSingleValuesSize < singleValuesSize) {
377                 format = CommonSettings.MemoryStorageFormat.PAGED;
378             } else {
379                 format = CommonSettings.MemoryStorageFormat.PACKED;
380             }
381         }
382         return format;
383     }

385     private long getPageMemoryUsage(PackedLongValues values, float acceptableOverheadRatio, int pageSize, long pageMinOrdinal, long pageMaxOrdinal) {
386         int bitsRequired;
387         long pageMemorySize = 0;
388         PackedInts.FormatAndBits formatAndBits;
389         if (pageMaxOrdinal == Long.MIN_VALUE) {
390             // empty page - will use the null reader which just stores size
391             pageMemorySize += RamUsageEstimator.alignObjectSize(RamUsageEstimator.NUM_BYTES_OBJECT_HEADER + RamUsageEstimator.NUM_BYTES_INT);
392 
393         } else {
394             long pageMinValue = values.get(pageMinOrdinal);
395             long pageMaxValue = values.get(pageMaxOrdinal);
396             long pageDelta = pageMaxValue - pageMinValue;
397             if (pageDelta != 0) {
398                 bitsRequired = pageDelta < 0 ? 64 : PackedInts.bitsRequired(pageDelta);
399                 formatAndBits = PackedInts.fastestFormatAndBits(pageSize, bitsRequired, acceptableOverheadRatio);
400                 pageMemorySize += formatAndBits.format.longCount(PackedInts.VERSION_CURRENT, pageSize, formatAndBits.bitsPerValue) * RamUsageEstimator.NUM_BYTES_LONG;
401                 pageMemorySize += RamUsageEstimator.NUM_BYTES_LONG; // min value per page storage
402             } else {
403                 // empty page
404                 pageMemorySize += RamUsageEstimator.alignObjectSize(RamUsageEstimator.NUM_BYTES_OBJECT_HEADER + RamUsageEstimator.NUM_BYTES_INT);
405             }
406         }
407         return pageMemorySize;
408     }

415     /**
416      * Estimator that wraps numeric field data loading in a
417      * RamAccountingTermsEnum, adjusting the breaker after data has been
418      * loaded
419      */
420     public class PackedArrayEstimator implements PerValueEstimator {
421 
422         private final CircuitBreaker breaker;
423         private final NumericType type;
424         private final String fieldName;
425 
426         public PackedArrayEstimator(CircuitBreaker breaker, NumericType type, String fieldName) {
427             this.breaker = breaker;
428             this.type = type;
429             this.fieldName = fieldName;
430         }
431 
432         /**
433          * @return number of bytes per term, based on the NumericValue.requiredBits()
434          */
435         @Override
436         public long bytesPerValue(BytesRef term) {
437             // Estimate about  about 0.8 (8 / 10) compression ratio for
438             // numbers, but at least 4 bytes
439             return Math.max(type.requiredBits() / 10, 4);
440         }
441 
442         /**
443          * @return A TermsEnum wrapped in a RamAccountingTermsEnum
444          * @throws IOException
445          */
446         @Override
447         public TermsEnum beforeLoad(Terms terms) throws IOException {
448             return new RamAccountingTermsEnum(type.wrapTermsEnum(terms.iterator(null)), breaker, this, this.fieldName);
449         }
450 
451         /**
452          * Adjusts the breaker based on the aggregated value from the RamAccountingTermsEnum
453          *
454          * @param termsEnum  terms that were wrapped and loaded
455          * @param actualUsed actual field data memory usage
456          */
457         @Override
458         public void afterLoad(TermsEnum termsEnum, long actualUsed) {
459             assert termsEnum instanceof RamAccountingTermsEnum;
460             long estimatedBytes = ((RamAccountingTermsEnum) termsEnum).getTotalBytes();
461             breaker.addWithoutBreaking(-(estimatedBytes - actualUsed));
462         }
463 
464         /**
465          * Adjust the breaker when no terms were actually loaded, but the field
466          * data takes up space regardless. For instance, when ordinals are
467          * used.
468          *
469          * @param actualUsed bytes actually used
470          */
471         public void adjustForNoTerms(long actualUsed) {
472             breaker.addWithoutBreaking(actualUsed);
473         }
474     }

476     private static SortedNumericDocValues withOrdinals(Ordinals ordinals, final LongValues values, int maxDoc) {
477         final RandomAccessOrds ords = ordinals.ordinals();
478         final SortedDocValues singleOrds = DocValues.unwrapSingleton(ords);
479         if (singleOrds != null) {
480             final NumericDocValues singleValues = new NumericDocValues() {
481                 @Override
482                 public long get(int docID) {
483                     final int ord = singleOrds.getOrd(docID);
484                     if (ord >= 0) {
485                         return values.get(singleOrds.getOrd(docID));
486                     } else {
487                         return 0;
488                     }
489                 }
490             };
491             return DocValues.singleton(singleValues, DocValues.docsWithValue(ords, maxDoc));
492         } else {
493             return new SortedNumericDocValues() {
494                 @Override
495                 public long valueAt(int index) {
496                     return values.get(ords.ordAt(index));
497                 }
498 
499                 @Override
500                 public void setDocument(int doc) {
501                     ords.setDocument(doc);
502                 }
503 
504                 @Override
505                 public int count() {
506                     return ords.cardinality();
507                 }
508             };
509         }
510     }

512     private static SortedNumericDocValues singles(final NumericDocValues deltas, final long minValue) {
513         final NumericDocValues values;
514         if (minValue == 0) {
515             values = deltas;
516         } else {
517             values = new NumericDocValues() {
518                 @Override
519                 public long get(int docID) {
520                     return minValue + deltas.get(docID);
521                 }
522             };
523         }
524         return DocValues.singleton(values, null);
525     }
526 
527     private static SortedNumericDocValues sparseSingles(final NumericDocValues deltas, final long minValue,  final long missingValue, final int maxDoc) {
528         final NumericDocValues values = new NumericDocValues() {
529             @Override
530             public long get(int docID) {
531                 final long delta = deltas.get(docID);
532                 if (delta == missingValue) {
533                     return 0;
534                 }
535                 return minValue + delta;
536             }
537         };
538         final Bits docsWithFields = new Bits() {
539             @Override
540             public boolean get(int index) {
541                 return deltas.get(index) != missingValue;
542             }
543             @Override
544             public int length() {
545                 return maxDoc;
546             }
547         };
548         return DocValues.singleton(values, docsWithFields);
549     }
550 
551     private static SortedNumericDocValues pagedSingles(final PackedLongValues values, final Bits docsWithValue) {
552         return DocValues.singleton(new NumericDocValues() {
553             // we need to wrap since NumericDocValues must return 0 when a doc has no value
554             @Override
555             public long get(int docID) {
556                 if (docsWithValue == null || docsWithValue.get(docID)) {
557                     return values.get(docID);
558                 } else {
559                     return 0;
560                 }
561             }
562         }, docsWithValue);
563     }
