142     public static class TypeParser implements Mapper.TypeParser {
143         @Override
144         public Mapper.Builder parse(String name, Map<String, Object> node, ParserContext parserContext) throws MapperParsingException {
145             TimestampFieldMapper.Builder builder = timestamp();
146             parseField(builder, builder.name, node, parserContext);
147             boolean defaultSet = false;
148             Boolean ignoreMissing = null;
149             for (Iterator<Map.Entry<String, Object>> iterator = node.entrySet().iterator(); iterator.hasNext();) {
150                 Map.Entry<String, Object> entry = iterator.next();
151                 String fieldName = Strings.toUnderscoreCase(entry.getKey());
152                 Object fieldNode = entry.getValue();
153                 if (fieldName.equals("enabled")) {
154                     EnabledAttributeMapper enabledState = nodeBooleanValue(fieldNode) ? EnabledAttributeMapper.ENABLED : EnabledAttributeMapper.DISABLED;
155                     builder.enabled(enabledState);
156                     iterator.remove();
157                 } else if (fieldName.equals("path")) {
158                     builder.path(fieldNode.toString());
159                     iterator.remove();
160                 } else if (fieldName.equals("format")) {
161                     builder.dateTimeFormatter(parseDateTimeFormatter(fieldNode.toString()));
162                     iterator.remove();
163                 } else if (fieldName.equals("default")) {
164                     if (fieldNode == null) {
165                         if (parserContext.indexVersionCreated().onOrAfter(Version.V_1_4_0_Beta1) &&
166                                 parserContext.indexVersionCreated().before(Version.V_1_5_0)) {
167                             // We are reading an index created in 1.4 with feature #7036
168                             // `default: null` was explicitly set. We need to change this index to
169                             // `ignore_missing: false`
170                             builder.ignoreMissing(false);
171                         } else {
172                             throw new TimestampParsingException("default timestamp can not be set to null");
173                         }
174                     } else {
175                         builder.defaultTimestamp(fieldNode.toString());
176                         defaultSet = true;
177                     }
178                     iterator.remove();
179                 } else if (fieldName.equals("ignore_missing")) {
180                     ignoreMissing = nodeBooleanValue(fieldNode);
181                     builder.ignoreMissing(ignoreMissing);
182                     iterator.remove();
183                 }
184             }
185 
186             // We can not accept a default value and rejecting null values at the same time
187             if (defaultSet && (ignoreMissing != null && ignoreMissing == false)) {
188                 throw new TimestampParsingException("default timestamp can not be set with ignore_missing set to false");
189             }
190 
191             return builder;
192         }
193     }

211     protected TimestampFieldMapper(FieldType fieldType, Boolean docValues, EnabledAttributeMapper enabledState, String path,
212                                    FormatDateTimeFormatter dateTimeFormatter, String defaultTimestamp,
213                                    Boolean ignoreMissing,
214                                    Explicit<Boolean> ignoreMalformed, Explicit<Boolean> coerce, Loading normsLoading,
215                                    @Nullable Settings fieldDataSettings, Settings indexSettings) {
216         super(new Names(Defaults.NAME, Defaults.NAME, Defaults.NAME, Defaults.NAME), dateTimeFormatter,
217                 Defaults.PRECISION_STEP_64_BIT, Defaults.BOOST, fieldType, docValues,
218                 Defaults.NULL_VALUE, TimeUnit.MILLISECONDS /*always milliseconds*/,
219                 ignoreMalformed, coerce, null, normsLoading, fieldDataSettings, 
220                 indexSettings, MultiFields.empty(), null);
221         this.enabledState = enabledState;
222         this.path = path;
223         this.defaultTimestamp = defaultTimestamp;
224         this.defaultFieldType = defaultFieldType(indexSettings);
225         this.ignoreMissing = ignoreMissing;
226     }

281     @Override
282     protected void innerParseCreateField(ParseContext context, List<Field> fields) throws IOException {
283         if (enabledState.enabled) {
284             long timestamp = context.sourceToParse().timestamp();
285             if (fieldType.indexOptions() == IndexOptions.NONE && !fieldType.stored() && !hasDocValues()) {
286                 context.ignoredValue(names.indexName(), String.valueOf(timestamp));
287             }
288             if (fieldType.indexOptions() != IndexOptions.NONE || fieldType.stored()) {
289                 fields.add(new LongFieldMapper.CustomLongNumericField(this, timestamp, fieldType));
290             }
291             if (hasDocValues()) {
292                 fields.add(new NumericDocValuesField(names.indexName(), timestamp));
293             }
294         }
295     }

302     @Override
303     public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
304         boolean includeDefaults = params.paramAsBoolean("include_defaults", false);
305         boolean indexed = fieldType.indexOptions() != IndexOptions.NONE;
306         boolean indexedDefault = Defaults.FIELD_TYPE.indexOptions() != IndexOptions.NONE;
307 
308         // if all are defaults, no sense to write it at all
309         if (!includeDefaults && indexed == indexedDefault && customFieldDataSettings == null &&
310                 fieldType.stored() == Defaults.FIELD_TYPE.stored() && enabledState == Defaults.ENABLED && path == Defaults.PATH
311                 && dateTimeFormatter.format().equals(Defaults.DATE_TIME_FORMATTER.format())
312                 && Defaults.DEFAULT_TIMESTAMP.equals(defaultTimestamp)
313                 && Defaults.DOC_VALUES == hasDocValues()) {
314             return builder;
315         }
316         builder.startObject(CONTENT_TYPE);
317         if (includeDefaults || enabledState != Defaults.ENABLED) {
318             builder.field("enabled", enabledState.enabled);
319         }
320         if (includeDefaults || (indexed != indexedDefault) || (fieldType.tokenized() != Defaults.FIELD_TYPE.tokenized())) {
321             builder.field("index", indexTokenizeOptionToString(indexed, fieldType.tokenized()));
322         }
323         if (includeDefaults || fieldType.stored() != Defaults.FIELD_TYPE.stored()) {
324             builder.field("store", fieldType.stored());
325         }
326         if (includeDefaults || hasDocValues() != Defaults.DOC_VALUES) {
327             builder.field(TypeParsers.DOC_VALUES, docValues);
328         }
329         if (includeDefaults || path != Defaults.PATH) {
330             builder.field("path", path);
331         }
332         if (includeDefaults || !dateTimeFormatter.format().equals(Defaults.DATE_TIME_FORMATTER.format())) {
333             builder.field("format", dateTimeFormatter.format());
334         }
335         if (includeDefaults || !Defaults.DEFAULT_TIMESTAMP.equals(defaultTimestamp)) {
336             builder.field("default", defaultTimestamp);
337         }
338         if (includeDefaults || ignoreMissing != null) {
339             builder.field("ignore_missing", ignoreMissing);
340         }
341         if (customFieldDataSettings != null) {
342             builder.field("fielddata", (Map) customFieldDataSettings.getAsMap());
343         } else if (includeDefaults) {
344             builder.field("fielddata", (Map) fieldDataType.getSettings().getAsMap());
345         }
346 
347         builder.endObject();
348         return builder;
349     }

351     @Override
352     public void merge(Mapper mergeWith, MergeContext mergeContext) throws MergeMappingException {
353         TimestampFieldMapper timestampFieldMapperMergeWith = (TimestampFieldMapper) mergeWith;
354         super.merge(mergeWith, mergeContext);
355         if (!mergeContext.mergeFlags().simulate()) {
356             if (timestampFieldMapperMergeWith.enabledState != enabledState && !timestampFieldMapperMergeWith.enabledState.unset()) {
357                 this.enabledState = timestampFieldMapperMergeWith.enabledState;
358             }
359         } else {
360             if (timestampFieldMapperMergeWith.defaultTimestamp() == null && defaultTimestamp == null) {
361                 return;
362             }
363             if (defaultTimestamp == null) {
364                 mergeContext.addConflict("Cannot update default in _timestamp value. Value is null now encountering " + timestampFieldMapperMergeWith.defaultTimestamp());
365             } else if (timestampFieldMapperMergeWith.defaultTimestamp() == null) {
366                 mergeContext.addConflict("Cannot update default in _timestamp value. Value is \" + defaultTimestamp.toString() + \" now encountering null");
367             } else if (!timestampFieldMapperMergeWith.defaultTimestamp().equals(defaultTimestamp)) {
368                 mergeContext.addConflict("Cannot update default in _timestamp value. Value is " + defaultTimestamp.toString() + " now encountering " + timestampFieldMapperMergeWith.defaultTimestamp());
369             }
370             if (this.path != null) {
371                 if (path.equals(timestampFieldMapperMergeWith.path()) == false) {
372                     mergeContext.addConflict("Cannot update path in _timestamp value. Value is " + path + " path in merged mapping is " + (timestampFieldMapperMergeWith.path() == null ? "missing" : timestampFieldMapperMergeWith.path()));
373                 }
374             } else if (timestampFieldMapperMergeWith.path() != null) {
375                 mergeContext.addConflict("Cannot update path in _timestamp value. Value is " + path + " path in merged mapping is missing");
376             }
377         }
