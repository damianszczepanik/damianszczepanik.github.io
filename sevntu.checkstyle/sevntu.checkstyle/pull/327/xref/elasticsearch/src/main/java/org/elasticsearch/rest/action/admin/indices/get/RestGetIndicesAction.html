50  import static org.elasticsearch.rest.RestRequest.Method.GET;
51  import static org.elasticsearch.rest.RestStatus.OK;

53  /**
54   *
55   */
56  public class RestGetIndicesAction extends BaseRestHandler {
57  
58      @Inject
59      public RestGetIndicesAction(Settings settings, RestController controller, Client client) {
60          super(settings, controller, client);
61          controller.registerHandler(GET, "/{index}", this);
62          controller.registerHandler(GET, "/{index}/{type}", this);
63      }
64  
65      @Override
66      public void handleRequest(final RestRequest request, final RestChannel channel, final Client client) {
67          String[] indices = Strings.splitStringByCommaToArray(request.param("index"));
68          String[] featureParams = request.paramAsStringArray("type", null);
69          // Work out if the indices is a list of features
70          if (featureParams == null && indices.length > 0 && indices[0] != null && indices[0].startsWith("_") && !"_all".equals(indices[0])) {
71              featureParams = indices;
72              indices = new String[] {"_all"};
73          }
74          final GetIndexRequest getIndexRequest = new GetIndexRequest();
75          getIndexRequest.indices(indices);
76          if (featureParams != null) {
77              Feature[] features = Feature.convertToFeatures(featureParams);
78              getIndexRequest.features(features);
79          }
80          getIndexRequest.indicesOptions(IndicesOptions.fromRequest(request, getIndexRequest.indicesOptions()));
81          getIndexRequest.local(request.paramAsBoolean("local", getIndexRequest.local()));
82          client.admin().indices().getIndex(getIndexRequest, new RestBuilderListener<GetIndexResponse>(channel) {
83  
84              @Override
85              public RestResponse buildResponse(GetIndexResponse response, XContentBuilder builder) throws Exception {
86  
87                  Feature[] features = getIndexRequest.featuresAsEnums();
88                  String[] indices = response.indices();
89  
90                  builder.startObject();
91                  for (String index : indices) {
92                      builder.startObject(index);
93                      for (Feature feature : features) {
94                          switch (feature) {
95                          case ALIASES:
96                              writeAliases(response.aliases().get(index), builder, request);
97                              break;
98                          case MAPPINGS:
99                              writeMappings(response.mappings().get(index), builder, request);
100                             break;
101                         case SETTINGS:
102                             writeSettings(response.settings().get(index), builder, request);
103                             break;
104                         case WARMERS:
105                             writeWarmers(response.warmers().get(index), builder, request);
106                             break;
107                         default:
108                             throw new ElasticsearchIllegalStateException("feature [" + feature + "] is not valid");
109                         }
110                     }
111                     builder.endObject();
112 
113                 }
114                 builder.endObject();
115 
116                 return new BytesRestResponse(OK, builder);
117             }
118 
119             private void writeAliases(ImmutableList<AliasMetaData> aliases, XContentBuilder builder, Params params) throws IOException {
120                 builder.startObject(Fields.ALIASES);
121                 if (aliases != null) {
122                     for (AliasMetaData alias : aliases) {
123                         AliasMetaData.Builder.toXContent(alias, builder, params);
124                     }
125                 }
126                 builder.endObject();
127             }
128 
129             private void writeMappings(ImmutableOpenMap<String, MappingMetaData> mappings, XContentBuilder builder, Params params) throws IOException {
130                 builder.startObject(Fields.MAPPINGS);
131                 if (mappings != null) {
132                     for (ObjectObjectCursor<String, MappingMetaData> typeEntry : mappings) {
133                         builder.field(typeEntry.key);
134                         builder.map(typeEntry.value.sourceAsMap());
135                     }
136                 }
137                 builder.endObject();
138             }
139 
140             private void writeSettings(Settings settings, XContentBuilder builder, Params params) throws IOException {
141                 builder.startObject(Fields.SETTINGS);
142                 settings.toXContent(builder, params);
143                 builder.endObject();
144             }
145 
146             private void writeWarmers(ImmutableList<IndexWarmersMetaData.Entry> warmers, XContentBuilder builder, Params params) throws IOException {
147                 builder.startObject(Fields.WARMERS);
148                 if (warmers != null) {
149                     for (IndexWarmersMetaData.Entry warmer : warmers) {
150                         IndexWarmersMetaData.FACTORY.toXContent(warmer, builder, params);
151                     }
152                 }
153                 builder.endObject();
154             }
155         });
156     }
157 
158     static class Fields {
159         static final XContentBuilderString ALIASES = new XContentBuilderString("aliases");
160         static final XContentBuilderString MAPPINGS = new XContentBuilderString("mappings");
161         static final XContentBuilderString SETTINGS = new XContentBuilderString("settings");
162         static final XContentBuilderString WARMERS = new XContentBuilderString("warmers");
163     }
