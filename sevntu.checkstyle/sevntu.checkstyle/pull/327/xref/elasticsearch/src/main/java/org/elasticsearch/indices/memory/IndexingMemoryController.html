51      private final ThreadPool threadPool;

55      private final ByteSizeValue minShardIndexBufferSize;
56      private final ByteSizeValue maxShardIndexBufferSize;
57  
58      private final ByteSizeValue translogBuffer;
59      private final ByteSizeValue minShardTranslogBufferSize;
60      private final ByteSizeValue maxShardTranslogBufferSize;

62      private final TimeValue inactiveTime;
63      private final TimeValue interval;

65      private volatile ScheduledFuture scheduler;

67      private static final EnumSet<IndexShardState> CAN_UPDATE_INDEX_BUFFER_STATES = EnumSet.of(IndexShardState.POST_RECOVERY, IndexShardState.STARTED, IndexShardState.RELOCATED);

69      @Inject
70      public IndexingMemoryController(Settings settings, ThreadPool threadPool, IndicesService indicesService) {
71          super(settings);
72          this.threadPool = threadPool;
73          this.indicesService = indicesService;
74  
75          ByteSizeValue indexingBuffer;
76          String indexingBufferSetting = this.settings.get("indices.memory.index_buffer_size", "10%");
77          if (indexingBufferSetting.endsWith("%")) {
78              double percent = Double.parseDouble(indexingBufferSetting.substring(0, indexingBufferSetting.length() - 1));
79              indexingBuffer = new ByteSizeValue((long) (((double) JvmInfo.jvmInfo().mem().heapMax().bytes()) * (percent / 100)));
80              ByteSizeValue minIndexingBuffer = this.settings.getAsBytesSize("indices.memory.min_index_buffer_size", new ByteSizeValue(48, ByteSizeUnit.MB));
81              ByteSizeValue maxIndexingBuffer = this.settings.getAsBytesSize("indices.memory.max_index_buffer_size", null);
82  
83              if (indexingBuffer.bytes() < minIndexingBuffer.bytes()) {
84                  indexingBuffer = minIndexingBuffer;
85              }
86              if (maxIndexingBuffer != null && indexingBuffer.bytes() > maxIndexingBuffer.bytes()) {
87                  indexingBuffer = maxIndexingBuffer;
88              }
89          } else {
90              indexingBuffer = ByteSizeValue.parseBytesSizeValue(indexingBufferSetting, null);
91          }
92          this.indexingBuffer = indexingBuffer;
93          this.minShardIndexBufferSize = this.settings.getAsBytesSize("indices.memory.min_shard_index_buffer_size", new ByteSizeValue(4, ByteSizeUnit.MB));
94          // LUCENE MONITOR: Based on this thread, currently (based on Mike), having a large buffer does not make a lot of sense: https://issues.apache.org/jira/browse/LUCENE-2324?focusedCommentId=13005155&page=com.atlassian.jira.plugin.system.issuetabpanels:comment-tabpanel#comment-13005155
95          this.maxShardIndexBufferSize = this.settings.getAsBytesSize("indices.memory.max_shard_index_buffer_size", new ByteSizeValue(512, ByteSizeUnit.MB));
96  
97          ByteSizeValue translogBuffer;
98          String translogBufferSetting = this.settings.get("indices.memory.translog_buffer_size", "1%");
99          if (translogBufferSetting.endsWith("%")) {
100             double percent = Double.parseDouble(translogBufferSetting.substring(0, translogBufferSetting.length() - 1));
101             translogBuffer = new ByteSizeValue((long) (((double) JvmInfo.jvmInfo().mem().heapMax().bytes()) * (percent / 100)));
102             ByteSizeValue minTranslogBuffer = this.settings.getAsBytesSize("indices.memory.min_translog_buffer_size", new ByteSizeValue(256, ByteSizeUnit.KB));
103             ByteSizeValue maxTranslogBuffer = this.settings.getAsBytesSize("indices.memory.max_translog_buffer_size", null);
104 
105             if (translogBuffer.bytes() < minTranslogBuffer.bytes()) {
106                 translogBuffer = minTranslogBuffer;
107             }
108             if (maxTranslogBuffer != null && translogBuffer.bytes() > maxTranslogBuffer.bytes()) {
109                 translogBuffer = maxTranslogBuffer;
110             }
111         } else {
112             translogBuffer = ByteSizeValue.parseBytesSizeValue(translogBufferSetting, null);
113         }
114         this.translogBuffer = translogBuffer;
115         this.minShardTranslogBufferSize = this.settings.getAsBytesSize("indices.memory.min_shard_translog_buffer_size", new ByteSizeValue(2, ByteSizeUnit.KB));
116         this.maxShardTranslogBufferSize = this.settings.getAsBytesSize("indices.memory.max_shard_translog_buffer_size", new ByteSizeValue(64, ByteSizeUnit.KB));
117 
118         this.inactiveTime = this.settings.getAsTime("indices.memory.shard_inactive_time", TimeValue.timeValueMinutes(30));
119         // we need to have this relatively small to move a shard from inactive to active fast (enough)
120         this.interval = this.settings.getAsTime("indices.memory.interval", TimeValue.timeValueSeconds(30));
121 
122         logger.debug("using index_buffer_size [{}], with min_shard_index_buffer_size [{}], max_shard_index_buffer_size [{}], shard_inactive_time [{}]", this.indexingBuffer, this.minShardIndexBufferSize, this.maxShardIndexBufferSize, this.inactiveTime);
123 
124     }

126     @Override
127     protected void doStart() throws ElasticsearchException {
128         // its fine to run it on the scheduler thread, no busy work
129         this.scheduler = threadPool.scheduleWithFixedDelay(new ShardsIndicesStatusChecker(), interval);
130     }
131 
132     @Override
133     protected void doStop() throws ElasticsearchException {
134         FutureUtils.cancel(scheduler);
135         scheduler = null;
136     }
137 
138     @Override
139     protected void doClose() throws ElasticsearchException {
140     }

150     class ShardsIndicesStatusChecker implements Runnable {
151 
152         private final Map<ShardId, ShardIndexingStatus> shardsIndicesStatus = new HashMap<>();
153 
154 
155         @Override
156         public void run() {
157             EnumSet<ShardStatusChangeType> changes = EnumSet.noneOf(ShardStatusChangeType.class);
158 
159             changes.addAll(purgeDeletedAndClosedShards());
160 
161             final List<IndexShard> activeToInactiveIndexingShards = Lists.newArrayList();
162             final int activeShards = updateShardStatuses(changes, activeToInactiveIndexingShards);
163             for (IndexShard indexShard : activeToInactiveIndexingShards) {
164                 // update inactive indexing buffer size
165                 try {
166                     indexShard.markAsInactive();
167                 } catch (EngineClosedException e) {
168                     // ignore
169                 } catch (FlushNotAllowedEngineException e) {
170                     // ignore
171                 }
172             }
173             if (!changes.isEmpty()) {
174                 calcAndSetShardBuffers(activeShards, "[" + changes + "]");
175             }
176         }
177 
178         /**
179          * goes through all existing shards and check whether the changes their active status
180          *
181          * @return the current count of active shards
182          */
183         private int updateShardStatuses(EnumSet<ShardStatusChangeType> changes, List<IndexShard> activeToInactiveIndexingShards) {
184             int activeShards = 0;
185             for (IndexService indexService : indicesService) {
186                 for (IndexShard indexShard : indexService) {
187 
188                     if (!CAN_UPDATE_INDEX_BUFFER_STATES.contains(indexShard.state())) {
189                         // not ready to be updated yet.
190                         continue;
191                     }
192 
193                     final long time = threadPool.estimatedTimeInMillis();
194 
195                     Translog translog = indexShard.translog();
196                     ShardIndexingStatus status = shardsIndicesStatus.get(indexShard.shardId());
197                     if (status == null) {
198                         status = new ShardIndexingStatus();
199                         shardsIndicesStatus.put(indexShard.shardId(), status);
200                         changes.add(ShardStatusChangeType.ADDED);
201                     }
202                     // check if it is deemed to be inactive (sam translogId and numberOfOperations over a long period of time)
203                     if (status.translogId == translog.currentId() && translog.estimatedNumberOfOperations() == 0) {
204                         if (status.time == -1) { // first time
205                             status.time = time;
206                         }
207                         // inactive?
208                         if (status.activeIndexing) {
209                             // mark it as inactive only if enough time has passed and there are no ongoing merges going on...
210                             if ((time - status.time) > inactiveTime.millis() && indexShard.mergeStats().getCurrent() == 0) {
211                                 // inactive for this amount of time, mark it
212                                 activeToInactiveIndexingShards.add(indexShard);
213                                 status.activeIndexing = false;
214                                 changes.add(ShardStatusChangeType.BECAME_INACTIVE);
215                                 logger.debug("marking shard [{}][{}] as inactive (inactive_time[{}]) indexing wise, setting size to [{}]", indexShard.shardId().index().name(), indexShard.shardId().id(), inactiveTime, EngineConfig.INACTIVE_SHARD_INDEXING_BUFFER);
216                             }
217                         }
218                     } else {
219                         if (!status.activeIndexing) {
220                             status.activeIndexing = true;
221                             changes.add(ShardStatusChangeType.BECAME_ACTIVE);
222                             logger.debug("marking shard [{}][{}] as active indexing wise", indexShard.shardId().index().name(), indexShard.shardId().id());
223                         }
224                         status.time = -1;
225                     }
226                     status.translogId = translog.currentId();
227                     status.translogNumberOfOperations = translog.estimatedNumberOfOperations();
228 
229                     if (status.activeIndexing) {
230                         activeShards++;
231                     }
232                 }
233             }
234             return activeShards;
235         }
236 
237         /**
238          * purge any existing statuses that are no longer updated
239          *
240          * @return true if any change
241          */
242         private EnumSet<ShardStatusChangeType> purgeDeletedAndClosedShards() {
243             EnumSet<ShardStatusChangeType> changes = EnumSet.noneOf(ShardStatusChangeType.class);
244 
245             Iterator<ShardId> statusShardIdIterator = shardsIndicesStatus.keySet().iterator();
246             while (statusShardIdIterator.hasNext()) {
247                 ShardId statusShardId = statusShardIdIterator.next();
248                 IndexService indexService = indicesService.indexService(statusShardId.getIndex());
249                 boolean remove = false;
250                 try {
251                     if (indexService == null) {
252                         remove = true;
253                         continue;
254                     }
255                     IndexShard indexShard = indexService.shard(statusShardId.id());
256                     if (indexShard == null) {
257                         remove = true;
258                         continue;
259                     }
260                     remove = !CAN_UPDATE_INDEX_BUFFER_STATES.contains(indexShard.state());
261 
262                 } finally {
263                     if (remove) {
264                         changes.add(ShardStatusChangeType.DELETED);
265                         statusShardIdIterator.remove();
266                     }
267                 }
268             }
269             return changes;
270         }
271 
272         private void calcAndSetShardBuffers(int activeShards, String reason) {
273             if (activeShards == 0) {
274                 return;
275             }
276             ByteSizeValue shardIndexingBufferSize = new ByteSizeValue(indexingBuffer.bytes() / activeShards);
277             if (shardIndexingBufferSize.bytes() < minShardIndexBufferSize.bytes()) {
278                 shardIndexingBufferSize = minShardIndexBufferSize;
279             }
280             if (shardIndexingBufferSize.bytes() > maxShardIndexBufferSize.bytes()) {
281                 shardIndexingBufferSize = maxShardIndexBufferSize;
282             }
283 
284             ByteSizeValue shardTranslogBufferSize = new ByteSizeValue(translogBuffer.bytes() / activeShards);
285             if (shardTranslogBufferSize.bytes() < minShardTranslogBufferSize.bytes()) {
286                 shardTranslogBufferSize = minShardTranslogBufferSize;
287             }
288             if (shardTranslogBufferSize.bytes() > maxShardTranslogBufferSize.bytes()) {
289                 shardTranslogBufferSize = maxShardTranslogBufferSize;
290             }
291 
292             logger.debug("recalculating shard indexing buffer (reason={}), total is [{}] with [{}] active shards, each shard set to indexing=[{}], translog=[{}]", reason, indexingBuffer, activeShards, shardIndexingBufferSize, shardTranslogBufferSize);
293             for (IndexService indexService : indicesService) {
294                 for (IndexShard indexShard : indexService) {
295                     IndexShardState state = indexShard.state();
296                     if (!CAN_UPDATE_INDEX_BUFFER_STATES.contains(state)) {
297                         logger.trace("shard [{}] is not yet ready for index buffer update. index shard state: [{}]", indexShard.shardId(), state);
298                         continue;
299                     }
300                     ShardIndexingStatus status = shardsIndicesStatus.get(indexShard.shardId());
301                     if (status == null || status.activeIndexing) {
302                         try {
303                             indexShard.updateBufferSize(shardIndexingBufferSize, shardTranslogBufferSize);
304                         } catch (EngineClosedException e) {
305                             // ignore
306                             continue;
307                         } catch (FlushNotAllowedEngineException e) {
308                             // ignore
309                             continue;
310                         } catch (Exception e) {
311                             logger.warn("failed to set shard {} index buffer to [{}]", indexShard.shardId(), shardIndexingBufferSize);
312                         }
313                     }
314                 }
315             }
316         }
317     }

319     private static enum ShardStatusChangeType {
320         ADDED, DELETED, BECAME_ACTIVE, BECAME_INACTIVE
321     }

324     static class ShardIndexingStatus {
