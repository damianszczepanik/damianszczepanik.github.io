20  package org.elasticsearch.cluster.routing.allocation.decider;
21  
22  import org.elasticsearch.cluster.routing.RoutingNode;
23  import org.elasticsearch.cluster.routing.ShardRouting;
24  import org.elasticsearch.cluster.routing.allocation.RoutingAllocation;
25  import org.elasticsearch.common.inject.Inject;
26  import org.elasticsearch.common.settings.Settings;
27  import org.elasticsearch.node.settings.NodeSettingsService;
28  
29  /**
30   * This {@link AllocationDecider} prevents cluster-wide shard allocations. The
31   * behavior of this {@link AllocationDecider} can be changed in real-time via
32   * the cluster settings API. It respects the following settings:
33   * <ul>
34   * <li><tt>cluster.routing.allocation.disable_new_allocation</tt> - if set to
35   * <code>true</code> no new shard-allocation are allowed. Note: this setting is
36   * only applied if the allocated shard is a primary and it has not been
37   * allocated before the this setting was applied.</li>
38   * <p/>
39   * <li><tt>cluster.routing.allocation.disable_allocation</tt> - if set to
40   * <code>true</code> cluster wide allocations are disabled</li>
41   * <p/>
42   * <li><tt>cluster.routing.allocation.disable_replica_allocation</tt> - if set
43   * to <code>true</code> cluster wide replica allocations are disabled while
44   * primary shards can still be allocated</li>
45   * </ul>
46   * <p/>
47   * <p>
48   * Note: all of the above settings might be ignored if the allocation happens on
49   * a shard that explicitly ignores disabled allocations via
50   * {@link RoutingAllocation#ignoreDisable()}. Which is set if allocation are
51   * explicit.
52   * </p>
53   *
54   * @deprecated In favour for {@link EnableAllocationDecider}.
55   */
56  @Deprecated
57  public class DisableAllocationDecider extends AllocationDecider {
58  
59      public static final String NAME = "disable";
60  
61      public static final String CLUSTER_ROUTING_ALLOCATION_DISABLE_NEW_ALLOCATION = "cluster.routing.allocation.disable_new_allocation";
62      public static final String CLUSTER_ROUTING_ALLOCATION_DISABLE_ALLOCATION = "cluster.routing.allocation.disable_allocation";
63      public static final String CLUSTER_ROUTING_ALLOCATION_DISABLE_REPLICA_ALLOCATION = "cluster.routing.allocation.disable_replica_allocation";
64  
65      public static final String INDEX_ROUTING_ALLOCATION_DISABLE_NEW_ALLOCATION = "index.routing.allocation.disable_new_allocation";
66      public static final String INDEX_ROUTING_ALLOCATION_DISABLE_ALLOCATION = "index.routing.allocation.disable_allocation";
67      public static final String INDEX_ROUTING_ALLOCATION_DISABLE_REPLICA_ALLOCATION = "index.routing.allocation.disable_replica_allocation";
68  
69      class ApplySettings implements NodeSettingsService.Listener {
70          @Override
71          public void onRefreshSettings(Settings settings) {
72              boolean disableNewAllocation = settings.getAsBoolean(CLUSTER_ROUTING_ALLOCATION_DISABLE_NEW_ALLOCATION, DisableAllocationDecider.this.disableNewAllocation);
73              if (disableNewAllocation != DisableAllocationDecider.this.disableNewAllocation) {
74                  logger.info("updating [cluster.routing.allocation.disable_new_allocation] from [{}] to [{}]", DisableAllocationDecider.this.disableNewAllocation, disableNewAllocation);
75                  DisableAllocationDecider.this.disableNewAllocation = disableNewAllocation;
76              }
77  
78              boolean disableAllocation = settings.getAsBoolean(CLUSTER_ROUTING_ALLOCATION_DISABLE_ALLOCATION, DisableAllocationDecider.this.disableAllocation);
79              if (disableAllocation != DisableAllocationDecider.this.disableAllocation) {
80                  logger.info("updating [cluster.routing.allocation.disable_allocation] from [{}] to [{}]", DisableAllocationDecider.this.disableAllocation, disableAllocation);
81                  DisableAllocationDecider.this.disableAllocation = disableAllocation;
82              }
83  
84              boolean disableReplicaAllocation = settings.getAsBoolean(CLUSTER_ROUTING_ALLOCATION_DISABLE_REPLICA_ALLOCATION, DisableAllocationDecider.this.disableReplicaAllocation);
85              if (disableReplicaAllocation != DisableAllocationDecider.this.disableReplicaAllocation) {
86                  logger.info("updating [cluster.routing.allocation.disable_replica_allocation] from [{}] to [{}]", DisableAllocationDecider.this.disableReplicaAllocation, disableReplicaAllocation);
87                  DisableAllocationDecider.this.disableReplicaAllocation = disableReplicaAllocation;
88              }
89          }
90      }
91  
92      private volatile boolean disableNewAllocation;
93      private volatile boolean disableAllocation;
94      private volatile boolean disableReplicaAllocation;
95  
96      @Inject
97      public DisableAllocationDecider(Settings settings, NodeSettingsService nodeSettingsService) {
98          super(settings);
99          this.disableNewAllocation = settings.getAsBoolean(CLUSTER_ROUTING_ALLOCATION_DISABLE_NEW_ALLOCATION, false);
100         this.disableAllocation = settings.getAsBoolean(CLUSTER_ROUTING_ALLOCATION_DISABLE_ALLOCATION, false);
101         this.disableReplicaAllocation = settings.getAsBoolean(CLUSTER_ROUTING_ALLOCATION_DISABLE_REPLICA_ALLOCATION, false);
102 
103         nodeSettingsService.addListener(new ApplySettings());
104     }
105 
106     @Override
107     public Decision canAllocate(ShardRouting shardRouting, RoutingNode node, RoutingAllocation allocation) {
108         if (allocation.ignoreDisable()) {
109             return allocation.decision(Decision.YES, NAME, "allocation disabling is ignored");
110         }
111         Settings indexSettings = allocation.routingNodes().metaData().index(shardRouting.index()).settings();
112         if (shardRouting.primary() && !allocation.routingNodes().routingTable().index(shardRouting.index()).shard(shardRouting.id()).primaryAllocatedPostApi()) {
113             // if its primary, and it hasn't been allocated post API (meaning its a "fresh newly created shard"), only disable allocation
114             // on a special disable allocation flag
115             if (indexSettings.getAsBoolean(INDEX_ROUTING_ALLOCATION_DISABLE_NEW_ALLOCATION, disableNewAllocation)) {
116                 return allocation.decision(Decision.NO, NAME, "new primary allocation is disabled");
117             } else {
118                 return allocation.decision(Decision.YES, NAME, "new primary allocation is enabled");
119             }
120         }
121         if (indexSettings.getAsBoolean(INDEX_ROUTING_ALLOCATION_DISABLE_ALLOCATION, disableAllocation)) {
122             return allocation.decision(Decision.NO, NAME, "all allocation is disabled");
123         }
124         if (indexSettings.getAsBoolean(INDEX_ROUTING_ALLOCATION_DISABLE_REPLICA_ALLOCATION, disableReplicaAllocation)) {
125             if (shardRouting.primary()) {
126                 return allocation.decision(Decision.YES, NAME, "primary allocation is enabled");
127             } else {
128                 return allocation.decision(Decision.NO, NAME, "replica allocation is disabled");
129             }
130         }
131         return allocation.decision(Decision.YES, NAME, "all allocation is enabled");
132     }
133 }
