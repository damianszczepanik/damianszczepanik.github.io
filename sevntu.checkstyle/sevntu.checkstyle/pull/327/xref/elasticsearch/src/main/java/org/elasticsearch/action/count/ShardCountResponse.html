20  package org.elasticsearch.action.count;
21  
22  import org.elasticsearch.Version;
23  import org.elasticsearch.action.support.broadcast.BroadcastShardOperationResponse;
24  import org.elasticsearch.common.io.stream.StreamInput;
25  import org.elasticsearch.common.io.stream.StreamOutput;
26  import org.elasticsearch.index.shard.ShardId;
27  
28  import java.io.IOException;
29  
30  /**
31   * Internal count response of a shard count request executed directly against a specific shard.
32   *
33   *
34   */
35  class ShardCountResponse extends BroadcastShardOperationResponse {
36  
37      private long count;
38      private boolean terminatedEarly;
39  
40      ShardCountResponse() {
41  
42      }
43  
44      ShardCountResponse(ShardId shardId, long count, boolean terminatedEarly) {
45          super(shardId);
46          this.count = count;
47          this.terminatedEarly = terminatedEarly;
48      }
49  
50      public long getCount() {
51          return this.count;
52      }
53  
54      public boolean terminatedEarly() {
55          return this.terminatedEarly;
56      }
57  
58      @Override
59      public void readFrom(StreamInput in) throws IOException {
60          super.readFrom(in);
61          count = in.readVLong();
62          terminatedEarly = in.readBoolean();
63      }
64  
65      @Override
66      public void writeTo(StreamOutput out) throws IOException {
67          super.writeTo(out);
68          out.writeVLong(count);
69          out.writeBoolean(terminatedEarly);
70      }
71  }
