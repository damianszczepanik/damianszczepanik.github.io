61  /**
62   * The recovery target handles recoveries of peer shards of the shard+node to recover to.
63   * <p/>
64   * <p>Note, it can be safely assumed that there will only be a single recovery per shard (index+id) and
65   * not several of them (since we don't allocate several shard replicas to the same node).
66   */
67  public class RecoveryTarget extends AbstractComponent {

87      @Inject
88      public RecoveryTarget(Settings settings, ThreadPool threadPool, TransportService transportService,
89                            IndicesLifecycle indicesLifecycle, RecoverySettings recoverySettings, ClusterService clusterService) {
90          super(settings);
91          this.threadPool = threadPool;
92          this.transportService = transportService;
93          this.recoverySettings = recoverySettings;
94          this.clusterService = clusterService;
95          this.onGoingRecoveries = new RecoveriesCollection(logger, threadPool);
96  
97          transportService.registerHandler(Actions.FILES_INFO, new FilesInfoRequestHandler());
98          transportService.registerHandler(Actions.FILE_CHUNK, new FileChunkTransportRequestHandler());
99          transportService.registerHandler(Actions.CLEAN_FILES, new CleanFilesRequestHandler());
100         transportService.registerHandler(Actions.PREPARE_TRANSLOG, new PrepareForTranslogOperationsRequestHandler());
101         transportService.registerHandler(Actions.TRANSLOG_OPS, new TranslogOperationsRequestHandler());
102         transportService.registerHandler(Actions.FINALIZE, new FinalizeRecoveryRequestHandler());
103 
104         indicesLifecycle.addListener(new IndicesLifecycle.Listener() {
105             @Override
106             public void beforeIndexShardClosed(ShardId shardId, @Nullable IndexShard indexShard,
107                                                @IndexSettings Settings indexSettings) {
108                 if (indexShard != null) {
109                     onGoingRecoveries.cancelRecoveriesForShard(shardId, "shard closed");
110                 }
111             }
112         });

115     public RecoveryState recoveryState(IndexShard indexShard) {
116         try (RecoveriesCollection.StatusRef statusRef = onGoingRecoveries.findRecoveryByShard(indexShard)) {
117             if (statusRef == null) {
118                 return null;
119             }
120             final RecoveryStatus recoveryStatus = statusRef.status();
121             return recoveryStatus.state();
122         } catch (Exception e) {
123             // shouldn't really happen, but have to be here due to auto close
124             throw new ElasticsearchException("error while getting recovery state", e);
125         }
126     }

128     public void startRecovery(final IndexShard indexShard, final RecoveryState.Type recoveryType, final DiscoveryNode sourceNode, final RecoveryListener listener) {
129         try {
130             indexShard.recovering("from " + sourceNode, recoveryType, sourceNode);
131         } catch (IllegalIndexShardStateException e) {
132             // that's fine, since we might be called concurrently, just ignore this, we are already recovering
133             logger.debug("{} ignore recovery. already in recovering process, {}", indexShard.shardId(), e.getMessage());
134             return;
135         }
136         // create a new recovery status, and process...
137         final long recoveryId = onGoingRecoveries.startRecovery(indexShard, sourceNode, listener, recoverySettings.activityTimeout());
138         threadPool.generic().execute(new RecoveryRunner(recoveryId));
139 
140     }

142     protected void retryRecovery(final RecoveryStatus recoveryStatus, final String reason, TimeValue retryAfter, final StartRecoveryRequest currentRequest) {
143         logger.trace("will retrying recovery with id [{}] in [{}] (reason [{}])", recoveryStatus.recoveryId(), retryAfter, reason);
144         try {
145             recoveryStatus.resetRecovery();
146         } catch (IOException e) {
147             onGoingRecoveries.failRecovery(recoveryStatus.recoveryId(), new RecoveryFailedException(currentRequest, e), true);
148         }
149         threadPool.schedule(retryAfter, ThreadPool.Names.GENERIC, new RecoveryRunner(recoveryStatus.recoveryId()));
150     }

152     private void doRecovery(final RecoveryStatus recoveryStatus) {
153         assert recoveryStatus.sourceNode() != null : "can't do a recovery without a source node";
154 
155         logger.trace("collecting local files for {}", recoveryStatus);
156         final Map<String, StoreFileMetaData> existingFiles;
157         try {
158             existingFiles = recoveryStatus.store().getMetadataOrEmpty().asMap();
159         } catch (Exception e) {
160             logger.debug("error while listing local files, recovery as if there are none", e);
161             onGoingRecoveries.failRecovery(recoveryStatus.recoveryId(),
162                     new RecoveryFailedException(recoveryStatus.state(), "failed to list local files", e), true);
163             return;
164         }
165         final StartRecoveryRequest request = new StartRecoveryRequest(recoveryStatus.shardId(), recoveryStatus.sourceNode(), clusterService.localNode(),
166                 false, existingFiles, recoveryStatus.state().getType(), recoveryStatus.recoveryId());
167 
168         final AtomicReference<RecoveryResponse> responseHolder = new AtomicReference<>();
169         try {
170             logger.trace("[{}][{}] starting recovery from {}", request.shardId().index().name(), request.shardId().id(), request.sourceNode());
171             recoveryStatus.indexShard().prepareForIndexRecovery();
172             recoveryStatus.CancellableThreads().execute(new CancellableThreads.Interruptable() {
173                 @Override
174                 public void run() throws InterruptedException {
175                     responseHolder.set(transportService.submitRequest(request.sourceNode(), RecoverySource.Actions.START_RECOVERY, request, new FutureTransportResponseHandler<RecoveryResponse>() {
176                         @Override
177                         public RecoveryResponse newInstance() {
178                             return new RecoveryResponse();
179                         }
180                     }).txGet());
181                 }
182             });
183             final RecoveryResponse recoveryResponse = responseHolder.get();
184             assert responseHolder != null;
185             final TimeValue recoveryTime = new TimeValue(recoveryStatus.state().getTimer().time());
186             // do this through ongoing recoveries to remove it from the collection
187             onGoingRecoveries.markRecoveryAsDone(recoveryStatus.recoveryId());
188             if (logger.isTraceEnabled()) {
189                 StringBuilder sb = new StringBuilder();
190                 sb.append('[').append(request.shardId().index().name()).append(']').append('[').append(request.shardId().id()).append("] ");
191                 sb.append("recovery completed from ").append(request.sourceNode()).append(", took[").append(recoveryTime).append("]\n");
192                 sb.append("   phase1: recovered_files [").append(recoveryResponse.phase1FileNames.size()).append("]").append(" with total_size of [").append(new ByteSizeValue(recoveryResponse.phase1TotalSize)).append("]")
193                         .append(", took [").append(timeValueMillis(recoveryResponse.phase1Time)).append("], throttling_wait [").append(timeValueMillis(recoveryResponse.phase1ThrottlingWaitTime)).append(']')
194                         .append("\n");
195                 sb.append("         : reusing_files   [").append(recoveryResponse.phase1ExistingFileNames.size()).append("] with total_size of [").append(new ByteSizeValue(recoveryResponse.phase1ExistingTotalSize)).append("]\n");
196                 sb.append("   phase2: start took [").append(timeValueMillis(recoveryResponse.startTime)).append("]\n");
197                 sb.append("         : recovered [").append(recoveryResponse.phase2Operations).append("]").append(" transaction log operations")
198                         .append(", took [").append(timeValueMillis(recoveryResponse.phase2Time)).append("]")
199                         .append("\n");
200                 sb.append("   phase3: recovered [").append(recoveryResponse.phase3Operations).append("]").append(" transaction log operations")
201                         .append(", took [").append(timeValueMillis(recoveryResponse.phase3Time)).append("]");
202                 logger.trace(sb.toString());
203             } else {
204                 logger.debug("{} recovery done from [{}], took [{}]", request.shardId(), recoveryStatus.sourceNode(), recoveryTime);
205             }
206         } catch (CancellableThreads.ExecutionCancelledException e) {
207             logger.trace("recovery cancelled", e);
208         } catch (Throwable e) {
209 
210             if (logger.isTraceEnabled()) {
211                 logger.trace("[{}][{}] Got exception on recovery", e, request.shardId().index().name(), request.shardId().id());
212             }
213             Throwable cause = ExceptionsHelper.unwrapCause(e);
214             if (cause instanceof RecoveryEngineException) {
215                 // unwrap an exception that was thrown as part of the recovery
216                 cause = cause.getCause();
217             }
218             // do it twice, in case we have double transport exception
219             cause = ExceptionsHelper.unwrapCause(cause);
220             if (cause instanceof RecoveryEngineException) {
221                 // unwrap an exception that was thrown as part of the recovery
222                 cause = cause.getCause();
223             }
224 
225             // here, we would add checks against exception that need to be retried (and not removeAndClean in this case)
226 
227             if (cause instanceof IndexShardNotStartedException || cause instanceof IndexMissingException || cause instanceof IndexShardMissingException) {
228                 // if the target is not ready yet, retry
229                 retryRecovery(recoveryStatus, "remote shard not ready", recoverySettings.retryDelayStateSync(), request);
230                 return;
231             }
232 
233             if (cause instanceof DelayRecoveryException) {
234                 retryRecovery(recoveryStatus, cause.getMessage(), recoverySettings.retryDelayStateSync(), request);
235                 return;
236             }
237 
238             if (cause instanceof ConnectTransportException) {
239                 logger.debug("delaying recovery of {} for [{}] due to networking error [{}]", recoveryStatus.shardId(), recoverySettings.retryDelayNetwork(), cause.getMessage());
240                 retryRecovery(recoveryStatus, cause.getMessage(), recoverySettings.retryDelayNetwork(), request);
241                 return;
242             }
243 
244             if (cause instanceof IndexShardClosedException) {
245                 onGoingRecoveries.failRecovery(recoveryStatus.recoveryId(), new RecoveryFailedException(request, "source shard is closed", cause), false);
246                 return;
247             }
248 
249             if (cause instanceof AlreadyClosedException) {
250                 onGoingRecoveries.failRecovery(recoveryStatus.recoveryId(), new RecoveryFailedException(request, "source shard is closed", cause), false);
251                 return;
252             }
253 
254             onGoingRecoveries.failRecovery(recoveryStatus.recoveryId(), new RecoveryFailedException(request, e), true);
255         }
256     }

324                 final RecoveryStatus recoveryStatus = statusRef.status();
325                 for (Translog.Operation operation : request.operations()) {

327                     recoveryStatus.state().getTranslog().incrementTranslogOperations();
328                 }
329             }
330             channel.sendResponse(TransportResponse.Empty.INSTANCE);

348         public void messageReceived(RecoveryFilesInfoRequest request, TransportChannel channel) throws Exception {
349             try (RecoveriesCollection.StatusRef statusRef = onGoingRecoveries.getStatusSafe(request.recoveryId(), request.shardId())) {
350                 final RecoveryStatus recoveryStatus = statusRef.status();
351                 final RecoveryState.Index index = recoveryStatus.state().getIndex();
352                 for (int i = 0; i < request.phase1ExistingFileNames.size(); i++) {
353                     index.addFileDetail(request.phase1ExistingFileNames.get(i), request.phase1ExistingFileSizes.get(i), true);
354                 }
355                 for (int i = 0; i < request.phase1FileNames.size(); i++) {
356                     index.addFileDetail(request.phase1FileNames.get(i), request.phase1FileSizes.get(i), false);
357                 }
358                 // recoveryBytesCount / recoveryFileCount will be set as we go...
359                 channel.sendResponse(TransportResponse.Empty.INSTANCE);
360             }
361         }

364     class CleanFilesRequestHandler extends BaseTransportRequestHandler<RecoveryCleanFilesRequest> {
365 
366         @Override
367         public RecoveryCleanFilesRequest newInstance() {
368             return new RecoveryCleanFilesRequest();
369         }
370 
371         @Override
372         public String executor() {
373             return ThreadPool.Names.GENERIC;
374         }
375 
376         @Override
377         public void messageReceived(RecoveryCleanFilesRequest request, TransportChannel channel) throws Exception {
378             try (RecoveriesCollection.StatusRef statusRef = onGoingRecoveries.getStatusSafe(request.recoveryId(), request.shardId())) {
379                 final RecoveryStatus recoveryStatus = statusRef.status();
380                 // first, we go and move files that were created with the recovery id suffix to
381                 // the actual names, its ok if we have a corrupted index here, since we have replicas
382                 // to recover from in case of a full cluster shutdown just when this code executes...
383                 recoveryStatus.renameAllTempFiles();
384                 final Store store = recoveryStatus.store();
385                 // now write checksums
386                 recoveryStatus.legacyChecksums().write(store);
387                 Store.MetadataSnapshot sourceMetaData = request.sourceMetaSnapshot();
388                 try {
389                     store.cleanupAndVerify("recovery CleanFilesRequestHandler", sourceMetaData);
390                 } catch (CorruptIndexException | IndexFormatTooNewException | IndexFormatTooOldException ex) {
391                     // this is a fatal exception at this stage.
392                     // this means we transferred files from the remote that have not be checksummed and they are
393                     // broken. We have to clean up this shard entirely, remove all files and bubble it up to the
394                     // source shard since this index might be broken there as well? The Source can handle this and checks
395                     // its content on disk if possible.
396                     try {
397                         Lucene.cleanLuceneIndex(store.directory()); // clean up and delete all files
398                     } catch (Throwable e) {
399                         logger.debug("Failed to clean lucene index", e);
400                         ex.addSuppressed(e);
401                     }
402                     throw new RecoveryFailedException(recoveryStatus.state(), "failed to clean after recovery", ex);
403                 } catch (Exception ex) {
404                     throw new RecoveryFailedException(recoveryStatus.state(), "failed to clean after recovery", ex);
405                 }
406                 channel.sendResponse(TransportResponse.Empty.INSTANCE);
407             }
408         }

411     class FileChunkTransportRequestHandler extends BaseTransportRequestHandler<RecoveryFileChunkRequest> {
412 
413         @Override
414         public RecoveryFileChunkRequest newInstance() {
415             return new RecoveryFileChunkRequest();
416         }
417 
418         @Override
419         public String executor() {
420             return ThreadPool.Names.GENERIC;
421         }
422 
423         @Override
424         public void messageReceived(final RecoveryFileChunkRequest request, TransportChannel channel) throws Exception {
425             try (RecoveriesCollection.StatusRef statusRef = onGoingRecoveries.getStatusSafe(request.recoveryId(), request.shardId())) {
426                 final RecoveryStatus recoveryStatus = statusRef.status();
427                 final Store store = recoveryStatus.store();
428                 IndexOutput indexOutput;
429                 if (request.position() == 0) {
430                     indexOutput = recoveryStatus.openAndPutIndexOutput(request.name(), request.metadata(), store);
431                 } else {
432                     indexOutput = recoveryStatus.getOpenIndexOutput(request.name());
433                 }
434                 if (recoverySettings.rateLimiter() != null) {
435                     recoverySettings.rateLimiter().pause(request.content().length());
436                 }
437                 BytesReference content = request.content();
438                 if (!content.hasArray()) {
439                     content = content.toBytesArray();
440                 }
441                 indexOutput.writeBytes(content.array(), content.arrayOffset(), content.length());
442                 recoveryStatus.state().getIndex().addRecoveredBytesToFile(request.name(), content.length());
443                 if (indexOutput.getFilePointer() >= request.length() || request.lastChunk()) {
444                     try {
445                         Store.verify(indexOutput);
446                     } finally {
447                         // we are done
448                         indexOutput.close();
449                     }
450                     // write the checksum
451                     recoveryStatus.legacyChecksums().add(request.metadata());
452                     final String temporaryFileName = recoveryStatus.getTempNameForFile(request.name());
453                     assert Arrays.asList(store.directory().listAll()).contains(temporaryFileName);
454                     store.directory().sync(Collections.singleton(temporaryFileName));
455                     IndexOutput remove = recoveryStatus.removeOpenIndexOutputs(request.name());
456                     assert remove == null || remove == indexOutput; // remove maybe null if we got finished
457                 }
458             }
459             channel.sendResponse(TransportResponse.Empty.INSTANCE);
460         }

463     class RecoveryRunner extends AbstractRunnable {
464 
465         final long recoveryId;
466 
467         RecoveryRunner(long recoveryId) {
468             this.recoveryId = recoveryId;
469         }
470 
471         @Override
472         public void onFailure(Throwable t) {
473             try (RecoveriesCollection.StatusRef statusRef = onGoingRecoveries.getStatus(recoveryId)) {
474                 if (statusRef != null) {
475                     logger.error("unexpected error during recovery [{}], failing shard", t, recoveryId);
476                     onGoingRecoveries.failRecovery(recoveryId,
477                             new RecoveryFailedException(statusRef.status().state(), "unexpected error", t),
478                             true // be safe
479                     );
480                 } else {
481                     logger.debug("unexpected error during recovery, but recovery id [{}] is finished", t, recoveryId);
482                 }
483             }
484         }
485 
486         @Override
487         public void doRun() {
488             RecoveriesCollection.StatusRef statusRef = onGoingRecoveries.getStatus(recoveryId);
489             if (statusRef == null) {
490                 logger.trace("not running recovery with id [{}] - can't find it (probably finished)", recoveryId);
491                 return;
492             }
493             try {
494                 doRecovery(statusRef.status());
495             } finally {
496                 statusRef.close();
497             }
498         }
499     }
