56      @Override
57      public void doRequest(final RestRequest request, final RestChannel channel, final Client client) {
58          final String[] indices = Strings.splitStringByCommaToArray(request.param("index"));
59          final ClusterStateRequest clusterStateRequest = new ClusterStateRequest();
60          clusterStateRequest.local(request.paramAsBoolean("local", clusterStateRequest.local()));
61          clusterStateRequest.masterNodeTimeout(request.paramAsTime("master_timeout", clusterStateRequest.masterNodeTimeout()));
62          clusterStateRequest.clear().nodes(true).metaData(true).routingTable(true).indices(indices);
63          client.admin().cluster().state(clusterStateRequest, new RestActionListener<ClusterStateResponse>(channel) {
64              @Override
65              public void processResponse(final ClusterStateResponse clusterStateResponse) {
66                  IndicesStatsRequest indicesStatsRequest = new IndicesStatsRequest();
67                  indicesStatsRequest.all();
68                  client.admin().indices().stats(indicesStatsRequest, new RestResponseListener<IndicesStatsResponse>(channel) {
69                      @Override
70                      public RestResponse buildResponse(IndicesStatsResponse indicesStatsResponse) throws Exception {
71                          return RestTable.buildResponse(buildTable(request, clusterStateResponse, indicesStatsResponse), channel);
72                      }
73                  });
74              }
75          });
76      }

159     private Table buildTable(RestRequest request, ClusterStateResponse state, IndicesStatsResponse stats) {
160         Table table = getTableWithHeader(request);
161 
162         for (ShardRouting shard : state.getState().routingTable().allShards()) {
163             CommonStats shardStats = stats.asMap().get(shard);
164 
165             table.startRow();
166 
167             table.addCell(shard.index());
168             table.addCell(shard.id());
169 
170             IndexMetaData indexMeta = state.getState().getMetaData().index(shard.index());
171             boolean usesShadowReplicas = false;
172             if (indexMeta != null) {
173                 usesShadowReplicas = IndexMetaData.isIndexUsingShadowReplicas(indexMeta.settings());
174             }
175             if (shard.primary()) {
176                 table.addCell("p");
177             } else {
178                 if (usesShadowReplicas) {
179                     table.addCell("s");
180                 } else {
181                     table.addCell("r");
182                 }
183             }
184             table.addCell(shard.state());
185             table.addCell(shardStats == null ? null : shardStats.getDocs().getCount());
186             table.addCell(shardStats == null ? null : shardStats.getStore().getSize());
187             if (shard.assignedToNode()) {
188                 String ip = state.getState().nodes().get(shard.currentNodeId()).getHostAddress();
189                 StringBuilder name = new StringBuilder();
190                 name.append(state.getState().nodes().get(shard.currentNodeId()).name());
191                 if (shard.relocating()) {
192                     String reloIp = state.getState().nodes().get(shard.relocatingNodeId()).getHostAddress();
193                     String reloNme = state.getState().nodes().get(shard.relocatingNodeId()).name();
194                     name.append(" -> ");
195                     name.append(reloIp);
196                     name.append(" ");
197                     name.append(reloNme);
198                 }
199                 table.addCell(ip);
200                 table.addCell(name);
201             } else {
202                 table.addCell(null);
203                 table.addCell(null);
204             }
205 
206             table.addCell(shardStats == null ? null : shardStats.getCompletion().getSize());
207 
208             table.addCell(shardStats == null ? null : shardStats.getFieldData().getMemorySize());
209             table.addCell(shardStats == null ? null : shardStats.getFieldData().getEvictions());
210 
211             table.addCell(shardStats == null ? null : shardStats.getFilterCache().getMemorySize());
212             table.addCell(shardStats == null ? null : shardStats.getFilterCache().getEvictions());
213 
214             table.addCell(shardStats == null ? null : shardStats.getFlush().getTotal());
215             table.addCell(shardStats == null ? null : shardStats.getFlush().getTotalTime());
216 
217             table.addCell(shardStats == null ? null : shardStats.getGet().current());
218             table.addCell(shardStats == null ? null : shardStats.getGet().getTime());
219             table.addCell(shardStats == null ? null : shardStats.getGet().getCount());
220             table.addCell(shardStats == null ? null : shardStats.getGet().getExistsTime());
221             table.addCell(shardStats == null ? null : shardStats.getGet().getExistsCount());
222             table.addCell(shardStats == null ? null : shardStats.getGet().getMissingTime());
223             table.addCell(shardStats == null ? null : shardStats.getGet().getMissingCount());
224 
225             table.addCell(shardStats == null ? null : shardStats.getIdCache().getMemorySize());
226 
227             table.addCell(shardStats == null ? null : shardStats.getIndexing().getTotal().getDeleteCurrent());
228             table.addCell(shardStats == null ? null : shardStats.getIndexing().getTotal().getDeleteTime());
229             table.addCell(shardStats == null ? null : shardStats.getIndexing().getTotal().getDeleteCount());
230             table.addCell(shardStats == null ? null : shardStats.getIndexing().getTotal().getIndexCurrent());
231             table.addCell(shardStats == null ? null : shardStats.getIndexing().getTotal().getIndexTime());
232             table.addCell(shardStats == null ? null : shardStats.getIndexing().getTotal().getIndexCount());
233 
234             table.addCell(shardStats == null ? null : shardStats.getMerge().getCurrent());
235             table.addCell(shardStats == null ? null : shardStats.getMerge().getCurrentNumDocs());
236             table.addCell(shardStats == null ? null : shardStats.getMerge().getCurrentSize());
237             table.addCell(shardStats == null ? null : shardStats.getMerge().getTotal());
238             table.addCell(shardStats == null ? null : shardStats.getMerge().getTotalNumDocs());
239             table.addCell(shardStats == null ? null : shardStats.getMerge().getTotalSize());
240             table.addCell(shardStats == null ? null : shardStats.getMerge().getTotalTime());
241 
242             table.addCell(shardStats == null ? null : shardStats.getPercolate().getCurrent());
243             table.addCell(shardStats == null ? null : shardStats.getPercolate().getMemorySize());
244             table.addCell(shardStats == null ? null : shardStats.getPercolate().getNumQueries());
245             table.addCell(shardStats == null ? null : shardStats.getPercolate().getTime());
246             table.addCell(shardStats == null ? null : shardStats.getPercolate().getCount());
247 
248             table.addCell(shardStats == null ? null : shardStats.getRefresh().getTotal());
249             table.addCell(shardStats == null ? null : shardStats.getRefresh().getTotalTime());
250 
251             table.addCell(shardStats == null ? null : shardStats.getSearch().getTotal().getFetchCurrent());
252             table.addCell(shardStats == null ? null : shardStats.getSearch().getTotal().getFetchTime());
253             table.addCell(shardStats == null ? null : shardStats.getSearch().getTotal().getFetchCount());
254             table.addCell(shardStats == null ? null : shardStats.getSearch().getOpenContexts());
255             table.addCell(shardStats == null ? null : shardStats.getSearch().getTotal().getQueryCurrent());
256             table.addCell(shardStats == null ? null : shardStats.getSearch().getTotal().getQueryTime());
257             table.addCell(shardStats == null ? null : shardStats.getSearch().getTotal().getQueryCount());
258 
259             table.addCell(shardStats == null ? null : shardStats.getSegments().getCount());
260             table.addCell(shardStats == null ? null : shardStats.getSegments().getMemory());
261             table.addCell(shardStats == null ? null : shardStats.getSegments().getIndexWriterMemory());
262             table.addCell(shardStats == null ? null : shardStats.getSegments().getIndexWriterMaxMemory());
263             table.addCell(shardStats == null ? null : shardStats.getSegments().getVersionMapMemory());
264             table.addCell(shardStats == null ? null : shardStats.getSegments().getBitsetMemory());
265 
266             table.addCell(shardStats == null ? null : shardStats.getWarmer().current());
267             table.addCell(shardStats == null ? null : shardStats.getWarmer().total());
268             table.addCell(shardStats == null ? null : shardStats.getWarmer().totalTime());
269 
270             table.endRow();
271         }
272 
273         return table;
274     }
