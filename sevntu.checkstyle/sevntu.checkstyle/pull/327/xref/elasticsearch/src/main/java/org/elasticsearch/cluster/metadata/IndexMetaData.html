125     public static enum State {
126         OPEN((byte) 0),
127         CLOSE((byte) 1);
128 
129         private final byte id;
130 
131         State(byte id) {
132             this.id = id;
133         }
134 
135         public byte id() {
136             return this.id;
137         }
138 
139         public static State fromId(byte id) {
140             if (id == 0) {
141                 return OPEN;
142             } else if (id == 1) {
143                 return CLOSE;
144             }
145             throw new ElasticsearchIllegalStateException("No state match for id [" + id + "]");
146         }
147 
148         public static State fromString(String state) {
149             if ("open".equals(state)) {
150                 return OPEN;
151             } else if ("close".equals(state)) {
152                 return CLOSE;
153             }
154             throw new ElasticsearchIllegalStateException("No state match for [" + state + "]");
155         }
156     }

202     private IndexMetaData(String index, long version, State state, Settings settings, ImmutableOpenMap<String, MappingMetaData> mappings, ImmutableOpenMap<String, AliasMetaData> aliases, ImmutableOpenMap<String, Custom> customs) {
203         Preconditions.checkArgument(settings.getAsInt(SETTING_NUMBER_OF_SHARDS, null) != null, "must specify numberOfShards for index [" + index + "]");
204         Preconditions.checkArgument(settings.getAsInt(SETTING_NUMBER_OF_REPLICAS, null) != null, "must specify numberOfReplicas for index [" + index + "]");
205         this.index = index;
206         this.version = version;
207         this.state = state;
208         this.settings = settings;
209         this.mappings = mappings;
210         this.customs = customs;
211         this.totalNumberOfShards = numberOfShards() * (numberOfReplicas() + 1);
212         this.aliases = aliases;
213 
214         ImmutableMap<String, String> requireMap = settings.getByPrefix("index.routing.allocation.require.").getAsMap();
215         if (requireMap.isEmpty()) {
216             requireFilters = null;
217         } else {
218             requireFilters = DiscoveryNodeFilters.buildFromKeyValue(AND, requireMap);
219         }
220         ImmutableMap<String, String> includeMap = settings.getByPrefix("index.routing.allocation.include.").getAsMap();
221         if (includeMap.isEmpty()) {
222             includeFilters = null;
223         } else {
224             includeFilters = DiscoveryNodeFilters.buildFromKeyValue(OR, includeMap);
225         }
226         ImmutableMap<String, String> excludeMap = settings.getByPrefix("index.routing.allocation.exclude.").getAsMap();
227         if (excludeMap.isEmpty()) {
228             excludeFilters = null;
229         } else {
230             excludeFilters = DiscoveryNodeFilters.buildFromKeyValue(OR, excludeMap);
231         }
232         indexCreatedVersion = Version.indexCreated(settings);
233         final Class<? extends HashFunction> hashFunctionClass = settings.getAsClass(SETTING_LEGACY_ROUTING_HASH_FUNCTION, null);
234         if (hashFunctionClass == null) {
235             routingHashFunction = MURMUR3_HASH_FUNCTION;
236         } else {
237             try {
238                 routingHashFunction = hashFunctionClass.newInstance();
239             } catch (InstantiationException | IllegalAccessException e) {
240                 throw new ElasticsearchIllegalStateException("Cannot instantiate hash function", e);
241             }
242         }
243         useTypeForRouting = settings.getAsBoolean(SETTING_LEGACY_ROUTING_USE_TYPE, false);
244     }

263      * Test whether the current index UUID is the same as the given one. Returns true if either are _na_

266         assert otherUUID != null;
267         assert uuid() != null;
268         if (INDEX_UUID_NA_VALUE.equals(otherUUID) || INDEX_UUID_NA_VALUE.equals(uuid())) {

270         }

430     @Override
431     public boolean equals(Object o) {
432         if (this == o) {
433             return true;
434         }
435         if (o == null || getClass() != o.getClass()) {
436             return false;
437         }
438 
439         IndexMetaData that = (IndexMetaData) o;
440 
441         if (!aliases.equals(that.aliases)) {
442             return false;
443         }
444         if (!index.equals(that.index)) {
445             return false;
446         }
447         if (!mappings.equals(that.mappings)) {
448             return false;
449         }
450         if (!settings.equals(that.settings)) {
451             return false;
452         }
453         if (state != that.state) {
454             return false;
455         }
456 
457         return true;
458     }

620         public IndexMetaData build() {
621             ImmutableOpenMap.Builder<String, AliasMetaData> tmpAliases = aliases;
622             Settings tmpSettings = settings;
623 
624             // update default mapping on the MappingMetaData
625             if (mappings.containsKey(MapperService.DEFAULT_MAPPING)) {
626                 MappingMetaData defaultMapping = mappings.get(MapperService.DEFAULT_MAPPING);
627                 for (ObjectCursor<MappingMetaData> cursor : mappings.values()) {
628                     cursor.value.updateDefaultMapping(defaultMapping);
629                 }
630             }
631 
632             return new IndexMetaData(index, version, state, tmpSettings, mappings.build(), tmpAliases.build(), customs.build());
633         }

635         public static void toXContent(IndexMetaData indexMetaData, XContentBuilder builder, ToXContent.Params params) throws IOException {
636             builder.startObject(indexMetaData.index(), XContentBuilder.FieldCaseConversion.NONE);
637 
638             builder.field("version", indexMetaData.version());
639             builder.field("state", indexMetaData.state().toString().toLowerCase(Locale.ENGLISH));
640 
641             boolean binary = params.paramAsBoolean("binary", false);
642 
643             builder.startObject("settings");
644             for (Map.Entry<String, String> entry : indexMetaData.settings().getAsMap().entrySet()) {
645                 builder.field(entry.getKey(), entry.getValue());
646             }
647             builder.endObject();
648 
649             builder.startArray("mappings");
650             for (ObjectObjectCursor<String, MappingMetaData> cursor : indexMetaData.mappings()) {
651                 if (binary) {
652                     builder.value(cursor.value.source().compressed());
653                 } else {
654                     byte[] data = cursor.value.source().uncompressed();
655                     XContentParser parser = XContentFactory.xContent(data).createParser(data);
656                     Map<String, Object> mapping = parser.mapOrdered();
657                     parser.close();
658                     builder.map(mapping);
659                 }
660             }
661             builder.endArray();
662 
663             for (ObjectObjectCursor<String, Custom> cursor : indexMetaData.customs()) {
664                 builder.startObject(cursor.key, XContentBuilder.FieldCaseConversion.NONE);
665                 lookupFactorySafe(cursor.key).toXContent(cursor.value, builder, params);
666                 builder.endObject();
667             }
668 
669             builder.startObject("aliases");
670             for (ObjectCursor<AliasMetaData> cursor : indexMetaData.aliases().values()) {
671                 AliasMetaData.Builder.toXContent(cursor.value, builder, params);
672             }
673             builder.endObject();
674 
675 
676             builder.endObject();
677         }

679         public static IndexMetaData fromXContent(XContentParser parser) throws IOException {
680             if (parser.currentToken() == null) { // fresh parser? move to the first token
681                 parser.nextToken();
682             }
683             if (parser.currentToken() == XContentParser.Token.START_OBJECT) {  // on a start object move to next token
684                 parser.nextToken();
685             }
686             Builder builder = new Builder(parser.currentName());
687 
688             String currentFieldName = null;
689             XContentParser.Token token = parser.nextToken();
690             while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
691                 if (token == XContentParser.Token.FIELD_NAME) {
692                     currentFieldName = parser.currentName();
693                 } else if (token == XContentParser.Token.START_OBJECT) {
694                     if ("settings".equals(currentFieldName)) {
695                         builder.settings(ImmutableSettings.settingsBuilder().put(SettingsLoader.Helper.loadNestedFromMap(parser.mapOrdered())));
696                     } else if ("mappings".equals(currentFieldName)) {
697                         while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
698                             if (token == XContentParser.Token.FIELD_NAME) {
699                                 currentFieldName = parser.currentName();
700                             } else if (token == XContentParser.Token.START_OBJECT) {
701                                 String mappingType = currentFieldName;
702                                 Map<String, Object> mappingSource = MapBuilder.<String, Object>newMapBuilder().put(mappingType, parser.mapOrdered()).map();
703                                 builder.putMapping(new MappingMetaData(mappingType, mappingSource));
704                             }
705                         }
706                     } else if ("aliases".equals(currentFieldName)) {
707                         while (parser.nextToken() != XContentParser.Token.END_OBJECT) {
708                             builder.putAlias(AliasMetaData.Builder.fromXContent(parser));
709                         }
710                     } else {
711                         // check if its a custom index metadata
712                         Custom.Factory<Custom> factory = lookupFactory(currentFieldName);
713                         if (factory == null) {
714                             //TODO warn
715                             parser.skipChildren();
716                         } else {
717                             builder.putCustom(factory.type(), factory.fromXContent(parser));
718                         }
719                     }
720                 } else if (token == XContentParser.Token.START_ARRAY) {
721                     if ("mappings".equals(currentFieldName)) {
722                         while ((token = parser.nextToken()) != XContentParser.Token.END_ARRAY) {
723                             if (token == XContentParser.Token.VALUE_EMBEDDED_OBJECT) {
724                                 builder.putMapping(new MappingMetaData(new CompressedString(parser.binaryValue())));
725                             } else {
726                                 Map<String, Object> mapping = parser.mapOrdered();
727                                 if (mapping.size() == 1) {
728                                     String mappingType = mapping.keySet().iterator().next();
729                                     builder.putMapping(new MappingMetaData(mappingType, mapping));
730                                 }
731                             }
732                         }
733                     }
734                 } else if (token.isValue()) {
735                     if ("state".equals(currentFieldName)) {
736                         builder.state(State.fromString(parser.text()));
737                     } else if ("version".equals(currentFieldName)) {
738                         builder.version(parser.longValue());
739                     }
740                 }
741             }
742             return builder.build();
743         }

745         public static IndexMetaData readFrom(StreamInput in) throws IOException {
746             Builder builder = new Builder(in.readString());
747             builder.version(in.readLong());
748             builder.state(State.fromId(in.readByte()));
749             builder.settings(readSettingsFromStream(in));
750             int mappingsSize = in.readVInt();
751             for (int i = 0; i < mappingsSize; i++) {
752                 MappingMetaData mappingMd = MappingMetaData.readFrom(in);
753                 builder.putMapping(mappingMd);
754             }
755             int aliasesSize = in.readVInt();
756             for (int i = 0; i < aliasesSize; i++) {
757                 AliasMetaData aliasMd = AliasMetaData.Builder.readFrom(in);
758                 builder.putAlias(aliasMd);
759             }
760             int customSize = in.readVInt();
761             for (int i = 0; i < customSize; i++) {
762                 String type = in.readString();
763                 Custom customIndexMetaData = lookupFactorySafe(type).readFrom(in);
764                 builder.putCustom(type, customIndexMetaData);
765             }
766             return builder.build();
767         }

769         public static void writeTo(IndexMetaData indexMetaData, StreamOutput out) throws IOException {
770             out.writeString(indexMetaData.index());
771             out.writeLong(indexMetaData.version());
772             out.writeByte(indexMetaData.state().id());
773             writeSettingsToStream(indexMetaData.settings(), out);
774             out.writeVInt(indexMetaData.mappings().size());
775             for (ObjectCursor<MappingMetaData> cursor : indexMetaData.mappings().values()) {
776                 MappingMetaData.writeTo(cursor.value, out);
777             }
778             out.writeVInt(indexMetaData.aliases().size());
779             for (ObjectCursor<AliasMetaData> cursor : indexMetaData.aliases().values()) {
780                 AliasMetaData.Builder.writeTo(cursor.value, out);
781             }
782             out.writeVInt(indexMetaData.customs().size());
783             for (ObjectObjectCursor<String, Custom> cursor : indexMetaData.customs()) {
784                 out.writeString(cursor.key);
785                 lookupFactorySafe(cursor.key).writeTo(cursor.value, out);
786             }
787         }
