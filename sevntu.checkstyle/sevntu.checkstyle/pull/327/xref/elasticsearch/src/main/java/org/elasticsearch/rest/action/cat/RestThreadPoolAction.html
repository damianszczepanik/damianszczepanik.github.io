52      private final static String[] SUPPORTED_NAMES = new String[]{
53              ThreadPool.Names.BULK,
54              ThreadPool.Names.FLUSH,
55              ThreadPool.Names.GENERIC,
56              ThreadPool.Names.GET,
57              ThreadPool.Names.INDEX,
58              ThreadPool.Names.MANAGEMENT,
59              ThreadPool.Names.OPTIMIZE,
60              ThreadPool.Names.PERCOLATE,
61              ThreadPool.Names.REFRESH,
62              ThreadPool.Names.SEARCH,
63              ThreadPool.Names.SNAPSHOT,
64              ThreadPool.Names.SUGGEST,
65              ThreadPool.Names.WARMER
66      };

68      private final static String[] SUPPORTED_ALIASES = new String[]{
69              "b",
70              "f",
71              "ge",
72              "g",
73              "i",
74              "ma",
75              "o",
76              "p",
77              "r",
78              "s",
79              "sn",
80              "su",
81              "w"
82      };
83  
84      static {
85          assert SUPPORTED_ALIASES.length == SUPPORTED_NAMES.length: "SUPPORTED_NAMES/ALIASES mismatch";
86      }
87  
88      private final static String[] DEFAULT_THREAD_POOLS = new String[]{
89              ThreadPool.Names.BULK,
90              ThreadPool.Names.INDEX,
91              ThreadPool.Names.SEARCH,
92      };
93  
94      private final static Map<String, String> ALIAS_TO_THREAD_POOL;
95      private final static Map<String, String> THREAD_POOL_TO_ALIAS;
96  
97      static {
98          ALIAS_TO_THREAD_POOL = Maps.newHashMapWithExpectedSize(SUPPORTED_NAMES.length);
99          for (String supportedThreadPool : SUPPORTED_NAMES) {
100             ALIAS_TO_THREAD_POOL.put(supportedThreadPool.substring(0, 3), supportedThreadPool);
101         }
102         THREAD_POOL_TO_ALIAS = Maps.newHashMapWithExpectedSize(SUPPORTED_NAMES.length);
103         for (int i = 0; i < SUPPORTED_NAMES.length; i++) {
104             THREAD_POOL_TO_ALIAS.put(SUPPORTED_NAMES[i], SUPPORTED_ALIASES[i]);
105         }
106     }

119     @Override
120     public void doRequest(final RestRequest request, final RestChannel channel, final Client client) {
121         final ClusterStateRequest clusterStateRequest = new ClusterStateRequest();
122         clusterStateRequest.clear().nodes(true);
123         clusterStateRequest.local(request.paramAsBoolean("local", clusterStateRequest.local()));
124         clusterStateRequest.masterNodeTimeout(request.paramAsTime("master_timeout", clusterStateRequest.masterNodeTimeout()));
125 
126         client.admin().cluster().state(clusterStateRequest, new RestActionListener<ClusterStateResponse>(channel) {
127             @Override
128             public void processResponse(final ClusterStateResponse clusterStateResponse) {
129                 NodesInfoRequest nodesInfoRequest = new NodesInfoRequest();
130                 nodesInfoRequest.clear().process(true).threadPool(true);
131                 client.admin().cluster().nodesInfo(nodesInfoRequest, new RestActionListener<NodesInfoResponse>(channel) {
132                     @Override
133                     public void processResponse(final NodesInfoResponse nodesInfoResponse) {
134                         NodesStatsRequest nodesStatsRequest = new NodesStatsRequest();
135                         nodesStatsRequest.clear().threadPool(true);
136                         client.admin().cluster().nodesStats(nodesStatsRequest, new RestResponseListener<NodesStatsResponse>(channel) {
137                             @Override
138                             public RestResponse buildResponse(NodesStatsResponse nodesStatsResponse) throws Exception {
139                                 return RestTable.buildResponse(buildTable(request, clusterStateResponse, nodesInfoResponse, nodesStatsResponse), channel);
140                             }
141                         });
142                     }
143                 });
144             }
145         });
146     }

148     @Override
149     Table getTableWithHeader(final RestRequest request) {
150         Table table = new Table();
151         table.startHeaders();
152         table.addCell("id", "default:false;alias:nodeId;desc:unique node id");
153         table.addCell("pid", "default:false;alias:p;desc:process id");
154         table.addCell("host", "alias:h;desc:host name");
155         table.addCell("ip", "alias:i;desc:ip address");
156         table.addCell("port", "default:false;alias:po;desc:bound transport port");
157 
158         final String[] requestedPools = fetchSortedPools(request, DEFAULT_THREAD_POOLS);
159         for (String pool : SUPPORTED_NAMES) {
160             String poolAlias = THREAD_POOL_TO_ALIAS.get(pool);
161             boolean display = false;
162             for (String requestedPool : requestedPools) {
163                 if (pool.equals(requestedPool)) {
164                     display = true;
165                     break;
166                 }
167             }
168 
169             String defaultDisplayVal = Boolean.toString(display);
170             table.addCell(
171                     pool + ".type",
172                     "alias:" + poolAlias + "t;default:false;desc:" + pool + " thread pool type"
173             );
174             table.addCell(
175                     pool + ".active",
176                     "alias:" + poolAlias + "a;default:" + defaultDisplayVal + ";text-align:right;desc:number of active " + pool + " threads"
177             );
178             table.addCell(
179                     pool + ".size",
180                     "alias:" + poolAlias + "s;default:false;text-align:right;desc:number of " + pool + " threads"
181             );
182             table.addCell(
183                     pool + ".queue",
184                     "alias:" + poolAlias + "q;default:" + defaultDisplayVal + ";text-align:right;desc:number of " + pool + " threads in queue"
185             );
186             table.addCell(
187                     pool + ".queueSize",
188                     "alias:" + poolAlias + "qs;default:false;text-align:right;desc:maximum number of " + pool + " threads in queue"
189             );
190             table.addCell(
191                     pool + ".rejected",
192                     "alias:" + poolAlias + "r;default:" + defaultDisplayVal + ";text-align:right;desc:number of rejected " + pool + " threads"
193             );
194             table.addCell(
195                     pool + ".largest",
196                     "alias:" + poolAlias + "l;default:false;text-align:right;desc:highest number of seen active " + pool + " threads"
197             );
198             table.addCell(
199                     pool + ".completed",
200                     "alias:" + poolAlias + "c;default:false;text-align:right;desc:number of completed " + pool + " threads"
201             );
202             table.addCell(
203                     pool + ".min",
204                     "alias:" + poolAlias + "mi;default:false;text-align:right;desc:minimum number of " + pool + " threads"
205             );
206             table.addCell(
207                     pool + ".max",
208                     "alias:" + poolAlias + "ma;default:false;text-align:right;desc:maximum number of " + pool + " threads"
209             );
210             table.addCell(
211                     pool + ".keepAlive",
212                     "alias:" + poolAlias + "k;default:false;text-align:right;desc:" + pool + " thread keep alive time"
213             );
214         }
215 
216         table.endHeaders();
217         return table;
218     }

221     private Table buildTable(RestRequest req, ClusterStateResponse state, NodesInfoResponse nodesInfo, NodesStatsResponse nodesStats) {
222         boolean fullId = req.paramAsBoolean("full_id", false);
223         DiscoveryNodes nodes = state.getState().nodes();
224         Table table = getTableWithHeader(req);
225 
226         for (DiscoveryNode node : nodes) {
227             NodeInfo info = nodesInfo.getNodesMap().get(node.id());
228             NodeStats stats = nodesStats.getNodesMap().get(node.id());
229             table.startRow();
230 
231             table.addCell(fullId ? node.id() : Strings.substring(node.getId(), 0, 4));
232             table.addCell(info == null ? null : info.getProcess().id());
233             table.addCell(node.getHostName());
234             table.addCell(node.getHostAddress());
235             if (node.address() instanceof InetSocketTransportAddress) {
236                 table.addCell(((InetSocketTransportAddress) node.address()).address().getPort());
237             } else {
238                 table.addCell("-");
239             }
240 
241             final Map<String, ThreadPoolStats.Stats> poolThreadStats;
242             final Map<String, ThreadPool.Info> poolThreadInfo;
243 
244             if (stats == null) {
245                 poolThreadStats = Collections.emptyMap();
246                 poolThreadInfo = Collections.emptyMap();
247             } else {
248                 poolThreadStats = new HashMap<>(14);
249                 poolThreadInfo = new HashMap<>(14);
250 
251                 ThreadPoolStats threadPoolStats = stats.getThreadPool();
252                 for (ThreadPoolStats.Stats threadPoolStat : threadPoolStats) {
253                     poolThreadStats.put(threadPoolStat.getName(), threadPoolStat);
254                 }
255                 if (info != null) {
256                     for (ThreadPool.Info threadPoolInfo : info.getThreadPool()) {
257                         poolThreadInfo.put(threadPoolInfo.getName(), threadPoolInfo);
258                     }
259                 }
260             }
261             for (String pool : SUPPORTED_NAMES) {
262                 ThreadPoolStats.Stats poolStats = poolThreadStats.get(pool);
263                 ThreadPool.Info poolInfo = poolThreadInfo.get(pool);
264 
265                 Long maxQueueSize = null;
266                 String keepAlive = null;
267                 Integer minThreads = null;
268                 Integer maxThreads = null;
269 
270                 if (poolInfo != null) {
271                     if (poolInfo.getQueueSize() != null) {
272                         maxQueueSize = poolInfo.getQueueSize().singles();
273                     }
274                     if (poolInfo.getKeepAlive() != null) {
275                         keepAlive = poolInfo.getKeepAlive().toString();
276                     }
277                     if (poolInfo.getMin() >= 0) {
278                         minThreads = poolInfo.getMin();
279                     }
280                     if (poolInfo.getMax() >= 0) {
281                         maxThreads = poolInfo.getMax();
282                     }
283                 }
284 
285                 table.addCell(poolInfo == null  ? null : poolInfo.getType());
286                 table.addCell(poolStats == null ? null : poolStats.getActive());
287                 table.addCell(poolStats == null ? null : poolStats.getThreads());
288                 table.addCell(poolStats == null ? null : poolStats.getQueue());
289                 table.addCell(maxQueueSize);
290                 table.addCell(poolStats == null ? null : poolStats.getRejected());
291                 table.addCell(poolStats == null ? null : poolStats.getLargest());
292                 table.addCell(poolStats == null ? null : poolStats.getCompleted());
293                 table.addCell(minThreads);
294                 table.addCell(maxThreads);
295                 table.addCell(keepAlive);
296             }
297 
298             table.endRow();
299         }
300 
301         return table;
302     }

304     // The thread pool columns should always be in the same order.
305     private String[] fetchSortedPools(RestRequest request, String[] defaults) {
306         String[] headers = request.paramAsStringArray("h", null);
307         if (headers == null) {
308             return defaults;
309         } else {
310             Set<String> requestedPools = new LinkedHashSet<>(headers.length);
311             for (String header : headers) {
312                 int dotIndex = header.indexOf('.');
313                 if (dotIndex != -1) {
314                     String headerPrefix = header.substring(0, dotIndex);
315                     if (THREAD_POOL_TO_ALIAS.containsKey(headerPrefix)) {
316                         requestedPools.add(headerPrefix);
317                     }
318                 } else if (ALIAS_TO_THREAD_POOL.containsKey(header)) {
319                     requestedPools.add(ALIAS_TO_THREAD_POOL.get(header));
320                 }
321 
322             }
323             return requestedPools.toArray(new String[requestedPools.size()]);
324         }
