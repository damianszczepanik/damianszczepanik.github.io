162     @SuppressWarnings("unchecked")
163     MetaData(String uuid, long version, Settings transientSettings, Settings persistentSettings, ImmutableOpenMap<String, IndexMetaData> indices, ImmutableOpenMap<String, IndexTemplateMetaData> templates, ImmutableOpenMap<String, Custom> customs) {
164         this.uuid = uuid;
165         this.version = version;
166         this.transientSettings = transientSettings;
167         this.persistentSettings = persistentSettings;
168         this.settings = ImmutableSettings.settingsBuilder().put(persistentSettings).put(transientSettings).build();
169         this.indices = indices;
170         this.customs = customs;
171         this.templates = templates;
172         int totalNumberOfShards = 0;
173         int numberOfShards = 0;
174         int numAliases = 0;
175         for (ObjectCursor<IndexMetaData> cursor : indices.values()) {
176             totalNumberOfShards += cursor.value.totalNumberOfShards();
177             numberOfShards += cursor.value.numberOfShards();
178             numAliases += cursor.value.aliases().size();
179         }
180         this.totalNumberOfShards = totalNumberOfShards;
181         this.numberOfShards = numberOfShards;
182 
183         // build all indices map
184         List<String> allIndicesLst = Lists.newArrayList();
185         for (ObjectCursor<IndexMetaData> cursor : indices.values()) {
186             allIndicesLst.add(cursor.value.index());
187         }
188         allIndices = allIndicesLst.toArray(new String[allIndicesLst.size()]);
189         int numIndices = allIndicesLst.size();
190 
191         List<String> allOpenIndices = Lists.newArrayList();
192         List<String> allClosedIndices = Lists.newArrayList();
193         for (ObjectCursor<IndexMetaData> cursor : indices.values()) {
194             IndexMetaData indexMetaData = cursor.value;
195             if (indexMetaData.state() == IndexMetaData.State.OPEN) {
196                 allOpenIndices.add(indexMetaData.index());
197             } else if (indexMetaData.state() == IndexMetaData.State.CLOSE) {
198                 allClosedIndices.add(indexMetaData.index());
199             }
200         }
201         this.allOpenIndices = allOpenIndices.toArray(new String[allOpenIndices.size()]);
202         this.allClosedIndices = allClosedIndices.toArray(new String[allClosedIndices.size()]);
203 
204         // build aliases map
205         ImmutableOpenMap.Builder<String, Object> tmpAliases = ImmutableOpenMap.builder(numAliases);
206         for (ObjectCursor<IndexMetaData> cursor : indices.values()) {
207             IndexMetaData indexMetaData = cursor.value;
208             String index = indexMetaData.index();
209             for (ObjectCursor<AliasMetaData> aliasCursor : indexMetaData.aliases().values()) {
210                 AliasMetaData aliasMd = aliasCursor.value;
211                 ImmutableOpenMap.Builder<String, AliasMetaData> indexAliasMap = (ImmutableOpenMap.Builder<String, AliasMetaData>) tmpAliases.get(aliasMd.alias());
212                 if (indexAliasMap == null) {
213                     indexAliasMap = ImmutableOpenMap.builder(1); // typically, there is 1 alias pointing to an index
214                     tmpAliases.put(aliasMd.alias(), indexAliasMap);
215                 }
216                 indexAliasMap.put(index, aliasMd);
217             }
218         }
219 
220         for (ObjectCursor<String> cursor : tmpAliases.keys()) {
221             String alias = cursor.value;
222             // if there is access to the raw values buffer of the map that the immutable maps wraps, then we don't need to use put, and just set array slots
223             ImmutableOpenMap<String, AliasMetaData> map = ((ImmutableOpenMap.Builder) tmpAliases.get(alias)).cast().build();
224             tmpAliases.put(alias, map);
225         }
226 
227         this.aliases = tmpAliases.<String, ImmutableOpenMap<String, AliasMetaData>>cast().build();
228         ImmutableOpenMap.Builder<String, Object> aliasAndIndexToIndexMap = ImmutableOpenMap.builder(numAliases + numIndices);
229         for (ObjectCursor<IndexMetaData> cursor : indices.values()) {
230             IndexMetaData indexMetaData = cursor.value;
231             ObjectArrayList<String> indicesLst = (ObjectArrayList<String>) aliasAndIndexToIndexMap.get(indexMetaData.index());
232             if (indicesLst == null) {
233                 indicesLst = new ObjectArrayList<>();
234                 aliasAndIndexToIndexMap.put(indexMetaData.index(), indicesLst);
235             }
236             indicesLst.add(indexMetaData.index());
237 
238             for (ObjectCursor<String> cursor1 : indexMetaData.aliases().keys()) {
239                 String alias = cursor1.value;
240                 indicesLst = (ObjectArrayList<String>) aliasAndIndexToIndexMap.get(alias);
241                 if (indicesLst == null) {
242                     indicesLst = new ObjectArrayList<>();
243                     aliasAndIndexToIndexMap.put(alias, indicesLst);
244                 }
245                 indicesLst.add(indexMetaData.index());
246             }
247         }
248 
249         for (ObjectObjectCursor<String, Object> cursor : aliasAndIndexToIndexMap) {
250             String[] indicesLst = ((ObjectArrayList<String>) cursor.value).toArray(String.class);
251             aliasAndIndexToIndexMap.put(cursor.key, indicesLst);
252         }
253 
254         this.aliasAndIndexToIndexMap = aliasAndIndexToIndexMap.<String, String[]>cast().build();
255     }

288     /**
289      * Finds the specific index aliases that match with the specified aliases directly or partially via wildcards and
290      * that point to the specified concrete indices or match partially with the indices via wildcards.
291      *
292      * @param aliases         The names of the index aliases to find
293      * @param concreteIndices The concrete indexes the index aliases must point to order to be returned.
294      * @return the found index aliases grouped by index
295      */
296     public ImmutableOpenMap<String, ImmutableList<AliasMetaData>> findAliases(final String[] aliases, String[] concreteIndices) {
297         assert aliases != null;
298         assert concreteIndices != null;
299         if (concreteIndices.length == 0) {
300             return ImmutableOpenMap.of();
301         }
302 
303         boolean matchAllAliases = matchAllAliases(aliases);
304         ImmutableOpenMap.Builder<String, ImmutableList<AliasMetaData>> mapBuilder = ImmutableOpenMap.builder();
305         Iterable<String> intersection = HppcMaps.intersection(ObjectOpenHashSet.from(concreteIndices), indices.keys());
306         for (String index : intersection) {
307             IndexMetaData indexMetaData = indices.get(index);
308             List<AliasMetaData> filteredValues = Lists.newArrayList();
309             for (ObjectCursor<AliasMetaData> cursor : indexMetaData.getAliases().values()) {
310                 AliasMetaData value = cursor.value;
311                 if (matchAllAliases || Regex.simpleMatch(aliases, value.alias())) {
312                     filteredValues.add(value);
313                 }
314             }
315 
316             if (!filteredValues.isEmpty()) {
317                 mapBuilder.put(index, ImmutableList.copyOf(filteredValues));
318             }
319         }
320         return mapBuilder.build();
321     }

323     private static boolean matchAllAliases(final String[] aliases) {
324         for (String alias : aliases) {
325             if (alias.equals(ALL)) {
326                 return true;
327             }
328         }
329         return aliases.length == 0;
330     }

332     /**
333      * Checks if at least one of the specified aliases exists in the specified concrete indices. Wildcards are supported in the
334      * alias names for partial matches.
335      *
336      * @param aliases         The names of the index aliases to find
337      * @param concreteIndices The concrete indexes the index aliases must point to order to be returned.
338      * @return whether at least one of the specified aliases exists in one of the specified concrete indices.
339      */
340     public boolean hasAliases(final String[] aliases, String[] concreteIndices) {
341         assert aliases != null;
342         assert concreteIndices != null;
343         if (concreteIndices.length == 0) {
344             return false;
345         }
346 
347         Iterable<String> intersection = HppcMaps.intersection(ObjectOpenHashSet.from(concreteIndices), indices.keys());
348         for (String index : intersection) {
349             IndexMetaData indexMetaData = indices.get(index);
350             List<AliasMetaData> filteredValues = Lists.newArrayList();
351             for (ObjectCursor<AliasMetaData> cursor : indexMetaData.getAliases().values()) {
352                 AliasMetaData value = cursor.value;
353                 if (Regex.simpleMatch(aliases, value.alias())) {
354                     filteredValues.add(value);
355                 }
356             }
357             if (!filteredValues.isEmpty()) {
358                 return true;
359             }
360         }
361         return false;
362     }

364     /*
365      * Finds all mappings for types and concrete indices. Types are expanded to
366      * include all types that match the glob patterns in the types array. Empty
367      * types array, null or {"_all"} will be expanded to all types available for
368      * the given indices.
369      */
370     public ImmutableOpenMap<String, ImmutableOpenMap<String, MappingMetaData>> findMappings(String[] concreteIndices, final String[] types) {
371         assert types != null;
372         assert concreteIndices != null;
373         if (concreteIndices.length == 0) {
374             return ImmutableOpenMap.of();
375         }
376 
377         ImmutableOpenMap.Builder<String, ImmutableOpenMap<String, MappingMetaData>> indexMapBuilder = ImmutableOpenMap.builder();
378         Iterable<String> intersection = HppcMaps.intersection(ObjectOpenHashSet.from(concreteIndices), indices.keys());
379         for (String index : intersection) {
380             IndexMetaData indexMetaData = indices.get(index);
381             ImmutableOpenMap.Builder<String, MappingMetaData> filteredMappings;
382             if (isAllTypes(types)) {
383                 indexMapBuilder.put(index, indexMetaData.getMappings()); // No types specified means get it all
384 
385             } else {
386                 filteredMappings = ImmutableOpenMap.builder();
387                 for (ObjectObjectCursor<String, MappingMetaData> cursor : indexMetaData.mappings()) {
388                     if (Regex.simpleMatch(types, cursor.key)) {
389                         filteredMappings.put(cursor.key, cursor.value);
390                     }
391                 }
392                 if (!filteredMappings.isEmpty()) {
393                     indexMapBuilder.put(index, filteredMappings.build());
394                 }
395             }
396         }
397         return indexMapBuilder.build();
398     }

400     public ImmutableOpenMap<String, ImmutableList<IndexWarmersMetaData.Entry>> findWarmers(String[] concreteIndices, final String[] types, final String[] uncheckedWarmers) {
401         assert uncheckedWarmers != null;
402         assert concreteIndices != null;
403         if (concreteIndices.length == 0) {
404             return ImmutableOpenMap.of();
405         }
406         // special _all check to behave the same like not specifying anything for the warmers (not for the indices)
407         final String[] warmers = Strings.isAllOrWildcard(uncheckedWarmers) ? Strings.EMPTY_ARRAY : uncheckedWarmers;
408 
409         ImmutableOpenMap.Builder<String, ImmutableList<IndexWarmersMetaData.Entry>> mapBuilder = ImmutableOpenMap.builder();
410         Iterable<String> intersection = HppcMaps.intersection(ObjectOpenHashSet.from(concreteIndices), indices.keys());
411         for (String index : intersection) {
412             IndexMetaData indexMetaData = indices.get(index);
413             IndexWarmersMetaData indexWarmersMetaData = indexMetaData.custom(IndexWarmersMetaData.TYPE);
414             if (indexWarmersMetaData == null || indexWarmersMetaData.entries().isEmpty()) {
415                 continue;
416             }
417 
418             Collection<IndexWarmersMetaData.Entry> filteredWarmers = Collections2.filter(indexWarmersMetaData.entries(), new Predicate<IndexWarmersMetaData.Entry>() {
419 
420                 @Override
421                 public boolean apply(IndexWarmersMetaData.Entry warmer) {
422                     if (warmers.length != 0 && types.length != 0) {
423                         return Regex.simpleMatch(warmers, warmer.name()) && Regex.simpleMatch(types, warmer.types());
424                     } else if (warmers.length != 0) {
425                         return Regex.simpleMatch(warmers, warmer.name());
426                     } else if (types.length != 0) {
427                         return Regex.simpleMatch(types, warmer.types());
428                     } else {
429                         return true;
430                     }
431                 }
432 
433             });
434             if (!filteredWarmers.isEmpty()) {
435                 mapBuilder.put(index, ImmutableList.copyOf(filteredWarmers));
436             }
437         }
438         return mapBuilder.build();
439     }

468     /**
469      * Returns indexing routing for the given index.
470      */
471     public String resolveIndexRouting(@Nullable String routing, String aliasOrIndex) {
472         // Check if index is specified by an alias
473         ImmutableOpenMap<String, AliasMetaData> indexAliases = aliases.get(aliasOrIndex);
474         if (indexAliases == null || indexAliases.isEmpty()) {
475             return routing;
476         }
477         if (indexAliases.size() > 1) {
478             throw new ElasticsearchIllegalArgumentException("Alias [" + aliasOrIndex + "] has more than one index associated with it [" + Arrays.toString(indexAliases.keys().toArray(String.class)) + "], can't execute a single index op");
479         }
480         AliasMetaData aliasMd = indexAliases.values().iterator().next().value;
481         if (aliasMd.indexRouting() != null) {
482             if (routing != null) {
483                 if (!routing.equals(aliasMd.indexRouting())) {
484                     throw new ElasticsearchIllegalArgumentException("Alias [" + aliasOrIndex + "] has index routing associated with it [" + aliasMd.indexRouting() + "], and was provided with routing value [" + routing + "], rejecting operation");
485                 }
486             }
487             routing = aliasMd.indexRouting();
488         }
489         if (routing != null) {
490             if (routing.indexOf(',') != -1) {
491                 throw new ElasticsearchIllegalArgumentException("index/alias [" + aliasOrIndex + "] provided with routing value [" + routing + "] that resolved to several routing values, rejecting operation");
492             }
493         }
494         return routing;
495     }

501     public Map<String, Set<String>> resolveSearchRouting(@Nullable String routing, String[] aliasesOrIndices) {
502         if (isAllIndices(aliasesOrIndices)) {
503             return resolveSearchRoutingAllIndices(routing);
504         }
505 
506         aliasesOrIndices = convertFromWildcards(aliasesOrIndices, IndicesOptions.lenientExpandOpen());
507 
508         if (aliasesOrIndices.length == 1) {
509             return resolveSearchRoutingSingleValue(routing, aliasesOrIndices[0]);
510         }
511 
512         Map<String, Set<String>> routings = null;
513         Set<String> paramRouting = null;
514         // List of indices that don't require any routing
515         Set<String> norouting = new HashSet<>();
516         if (routing != null) {
517             paramRouting = Strings.splitStringByCommaToSet(routing);
518         }
519 
520         for (String aliasOrIndex : aliasesOrIndices) {
521             ImmutableOpenMap<String, AliasMetaData> indexToRoutingMap = aliases.get(aliasOrIndex);
522             if (indexToRoutingMap != null && !indexToRoutingMap.isEmpty()) {
523                 for (ObjectObjectCursor<String, AliasMetaData> indexRouting : indexToRoutingMap) {
524                     if (!norouting.contains(indexRouting.key)) {
525                         if (!indexRouting.value.searchRoutingValues().isEmpty()) {
526                             // Routing alias
527                             if (routings == null) {
528                                 routings = newHashMap();
529                             }
530                             Set<String> r = routings.get(indexRouting.key);
531                             if (r == null) {
532                                 r = new HashSet<>();
533                                 routings.put(indexRouting.key, r);
534                             }
535                             r.addAll(indexRouting.value.searchRoutingValues());
536                             if (paramRouting != null) {
537                                 r.retainAll(paramRouting);
538                             }
539                             if (r.isEmpty()) {
540                                 routings.remove(indexRouting.key);
541                             }
542                         } else {
543                             // Non-routing alias
544                             if (!norouting.contains(indexRouting.key)) {
545                                 norouting.add(indexRouting.key);
546                                 if (paramRouting != null) {
547                                     Set<String> r = new HashSet<>(paramRouting);
548                                     if (routings == null) {
549                                         routings = newHashMap();
550                                     }
551                                     routings.put(indexRouting.key, r);
552                                 } else {
553                                     if (routings != null) {
554                                         routings.remove(indexRouting.key);
555                                     }
556                                 }
557                             }
558                         }
559                     }
560                 }
561             } else {
562                 // Index
563                 if (!norouting.contains(aliasOrIndex)) {
564                     norouting.add(aliasOrIndex);
565                     if (paramRouting != null) {
566                         Set<String> r = new HashSet<>(paramRouting);
567                         if (routings == null) {
568                             routings = newHashMap();
569                         }
570                         routings.put(aliasOrIndex, r);
571                     } else {
572                         if (routings != null) {
573                             routings.remove(aliasOrIndex);
574                         }
575                     }
576                 }
577             }
578 
579         }
580         if (routings == null || routings.isEmpty()) {
581             return null;
582         }
583         return routings;
584     }

586     private Map<String, Set<String>> resolveSearchRoutingSingleValue(@Nullable String routing, String aliasOrIndex) {
587         Map<String, Set<String>> routings = null;
588         Set<String> paramRouting = null;
589         if (routing != null) {
590             paramRouting = Strings.splitStringByCommaToSet(routing);
591         }
592 
593         ImmutableOpenMap<String, AliasMetaData> indexToRoutingMap = aliases.get(aliasOrIndex);
594         if (indexToRoutingMap != null && !indexToRoutingMap.isEmpty()) {
595             // It's an alias
596             for (ObjectObjectCursor<String, AliasMetaData> indexRouting : indexToRoutingMap) {
597                 if (!indexRouting.value.searchRoutingValues().isEmpty()) {
598                     // Routing alias
599                     Set<String> r = new HashSet<>(indexRouting.value.searchRoutingValues());
600                     if (paramRouting != null) {
601                         r.retainAll(paramRouting);
602                     }
603                     if (!r.isEmpty()) {
604                         if (routings == null) {
605                             routings = newHashMap();
606                         }
607                         routings.put(indexRouting.key, r);
608                     }
609                 } else {
610                     // Non-routing alias
611                     if (paramRouting != null) {
612                         Set<String> r = new HashSet<>(paramRouting);
613                         if (routings == null) {
614                             routings = newHashMap();
615                         }
616                         routings.put(indexRouting.key, r);
617                     }
618                 }
619             }
620         } else {
621             // It's an index
622             if (paramRouting != null) {
623                 routings = ImmutableMap.of(aliasOrIndex, paramRouting);
624             }
625         }
626         return routings;
627     }

646     /**
647      * Translates the provided indices or aliases, eventually containing wildcard expressions, into actual indices.
648      *
649      * @param indicesOptions how the aliases or indices need to be resolved to concrete indices
650      * @param aliasesOrIndices the aliases or indices to be resolved to concrete indices
651      * @return the obtained concrete indices
652      * @throws IndexMissingException if one of the aliases or indices is missing and the provided indices options
653      * don't allow such a case, or if the final result of the indices resolution is no indices and the indices options
654      * don't allow such a case.
655      * @throws ElasticsearchIllegalArgumentException if one of the aliases resolve to multiple indices and the provided
656      * indices options don't allow such a case.
657      */
658     public String[] concreteIndices(IndicesOptions indicesOptions, String... aliasesOrIndices) throws IndexMissingException, ElasticsearchIllegalArgumentException {
659         if (indicesOptions.expandWildcardsOpen() || indicesOptions.expandWildcardsClosed()) {
660             if (isAllIndices(aliasesOrIndices)) {
661                 String[] concreteIndices;
662                 if (indicesOptions.expandWildcardsOpen() && indicesOptions.expandWildcardsClosed()) {
663                     concreteIndices = concreteAllIndices();
664                 } else if (indicesOptions.expandWildcardsOpen()) {
665                     concreteIndices = concreteAllOpenIndices();
666                 } else {
667                     concreteIndices = concreteAllClosedIndices();
668                 }
669 
670                 if (!indicesOptions.allowNoIndices() && concreteIndices.length == 0) {
671                     throw new IndexMissingException(new Index("_all"));
672                 }
673                 return concreteIndices;
674             }
675 
676             aliasesOrIndices = convertFromWildcards(aliasesOrIndices, indicesOptions);
677         }
678         boolean failClosed = indicesOptions.forbidClosedIndices() && !indicesOptions.ignoreUnavailable();
679 
680         // optimize for single element index (common case)
681         if (aliasesOrIndices.length == 1) {
682             return concreteIndices(aliasesOrIndices[0], indicesOptions, !indicesOptions.allowNoIndices());
683         }
684 
685         // check if its a possible aliased index, if not, just return the passed array
686         boolean possiblyAliased = false;
687         boolean closedIndices = false;
688         for (String index : aliasesOrIndices) {
689             IndexMetaData indexMetaData = indices.get(index);
690             if (indexMetaData == null) {
691                 possiblyAliased = true;
692                 break;
693             } else {
694                 if (indicesOptions.forbidClosedIndices() && indexMetaData.getState() == IndexMetaData.State.CLOSE) {
695                     if (failClosed) {
696                         throw new IndexClosedException(new Index(index));
697                     } else {
698                         closedIndices = true;
699                     }
700                 }
701             }
702         }
703         if (!possiblyAliased) {
704             if (closedIndices) {
705                 Set<String> actualIndices = new HashSet<>(Arrays.asList(aliasesOrIndices));
706                 actualIndices.retainAll(new HashSet<Object>(Arrays.asList(allOpenIndices)));
707                 return actualIndices.toArray(new String[actualIndices.size()]);
708             } else {
709                 return aliasesOrIndices;
710             }
711         }
712 
713         Set<String> actualIndices = new HashSet<>();
714         for (String aliasOrIndex : aliasesOrIndices) {
715             String[] indices = concreteIndices(aliasOrIndex, indicesOptions, !indicesOptions.ignoreUnavailable());
716             Collections.addAll(actualIndices, indices);
717         }
718 
719         if (!indicesOptions.allowNoIndices() && actualIndices.isEmpty()) {
720             throw new IndexMissingException(new Index(Arrays.toString(aliasesOrIndices)));
721         }
722         return actualIndices.toArray(new String[actualIndices.size()]);
723     }

725     /**
726      * Utility method that allows to resolve an index or alias to its corresponding single concrete index.
727      * Callers should make sure they provide proper {@link org.elasticsearch.action.support.IndicesOptions}
728      * that require a single index as a result. The indices resolution must in fact return a single index when
729      * using this method, an {@link org.elasticsearch.ElasticsearchIllegalArgumentException} gets thrown otherwise.
730      *
731      * @param indexOrAlias   the index or alias to be resolved to concrete index
732      * @param indicesOptions the indices options to be used for the index resolution
733      * @return the concrete index obtained as a result of the index resolution
734      * @throws IndexMissingException                 if the index or alias provided doesn't exist
735      * @throws ElasticsearchIllegalArgumentException if the index resolution lead to more than one index
736      */
737     public String concreteSingleIndex(String indexOrAlias, IndicesOptions indicesOptions) throws IndexMissingException, ElasticsearchIllegalArgumentException {
738         String[] indices = concreteIndices(indicesOptions, indexOrAlias);
739         if (indices.length != 1) {
740             throw new ElasticsearchIllegalArgumentException("unable to return a single index as the index and options provided got resolved to multiple indices");
741         }
742         return indices[0];
743     }

745     private String[] concreteIndices(String aliasOrIndex, IndicesOptions options, boolean failNoIndices) throws IndexMissingException, ElasticsearchIllegalArgumentException {
746         boolean failClosed = options.forbidClosedIndices() && !options.ignoreUnavailable();
747 
748         // a quick check, if this is an actual index, if so, return it
749         IndexMetaData indexMetaData = indices.get(aliasOrIndex);
750         if (indexMetaData != null) {
751             if (indexMetaData.getState() == IndexMetaData.State.CLOSE) {
752                 if (failClosed) {
753                     throw new IndexClosedException(new Index(aliasOrIndex));
754                 } else {
755                     return options.forbidClosedIndices() ? Strings.EMPTY_ARRAY : new String[]{aliasOrIndex};
756                 }
757             } else {
758                 return new String[]{aliasOrIndex};
759             }
760         }
761         // not an actual index, fetch from an alias
762         String[] indices = aliasAndIndexToIndexMap.getOrDefault(aliasOrIndex, Strings.EMPTY_ARRAY);
763         if (indices.length == 0 && failNoIndices) {
764             throw new IndexMissingException(new Index(aliasOrIndex));
765         }
766         if (indices.length > 1 && !options.allowAliasesToMultipleIndices()) {
767             throw new ElasticsearchIllegalArgumentException("Alias [" + aliasOrIndex + "] has more than one indices associated with it [" + Arrays.toString(indices) + "], can't execute a single index op");
768         }
769 
770         // No need to check whether indices referred by aliases are closed, because there are no closed indices.
771         if (allClosedIndices.length == 0) {
772             return indices;
773         }
774 
775         switch (indices.length) {
776             case 0:
777                 return indices;
778             case 1:
779                 indexMetaData = this.indices.get(indices[0]);
780                 if (indexMetaData != null && indexMetaData.getState() == IndexMetaData.State.CLOSE) {
781                     if (failClosed) {
782                         throw new IndexClosedException(new Index(indexMetaData.getIndex()));
783                     } else {
784                         if (options.forbidClosedIndices()) {
785                             return Strings.EMPTY_ARRAY;
786                         }
787                     }
788                 }
789                 return indices;
790             default:
791                 ObjectArrayList<String> concreteIndices = new ObjectArrayList<>();
792                 for (String index : indices) {
793                     indexMetaData = this.indices.get(index);
794                     if (indexMetaData != null) {
795                         if (indexMetaData.getState() == IndexMetaData.State.CLOSE) {
796                             if (failClosed) {
797                                 throw new IndexClosedException(new Index(indexMetaData.getIndex()));
798                             } else if (!options.forbidClosedIndices()) {
799                                 concreteIndices.add(index);
800                             }
801                         } else if (indexMetaData.getState() == IndexMetaData.State.OPEN) {
802                             concreteIndices.add(index);
803                         } else {
804                             throw new IllegalStateException("index state [" + indexMetaData.getState() + "] not supported");
805                         }
806                     }
807                 }
808                 return concreteIndices.toArray(String.class);
809         }
810     }

812     /**
813      * Converts a list of indices or aliases wildcards, and special +/- signs, into their respective full matches. It
814      * won't convert only to indices, but also to aliases. For example, alias_* will expand to alias_1 and alias_2, not
815      * to the respective indices those aliases point to.
816      */
817     public String[] convertFromWildcards(String[] aliasesOrIndices, IndicesOptions indicesOptions) {
818         if (aliasesOrIndices == null) {
819             return null;
820         }
821         Set<String> result = null;
822         for (int i = 0; i < aliasesOrIndices.length; i++) {
823             String aliasOrIndex = aliasesOrIndices[i];
824             if (aliasAndIndexToIndexMap.containsKey(aliasOrIndex)) {
825                 if (result != null) {
826                     result.add(aliasOrIndex);
827                 }
828                 continue;
829             }
830             boolean add = true;
831             if (aliasOrIndex.charAt(0) == '+') {
832                 // if its the first, add empty result set
833                 if (i == 0) {
834                     result = new HashSet<>();
835                 }
836                 add = true;
837                 aliasOrIndex = aliasOrIndex.substring(1);
838             } else if (aliasOrIndex.charAt(0) == '-') {
839                 // if its the first, fill it with all the indices...
840                 if (i == 0) {
841                     String[] concreteIndices;
842                     if (indicesOptions.expandWildcardsOpen() && indicesOptions.expandWildcardsClosed()) {
843                         concreteIndices = concreteAllIndices();
844                     } else if (indicesOptions.expandWildcardsOpen()) {
845                         concreteIndices = concreteAllOpenIndices();
846                     } else if (indicesOptions.expandWildcardsClosed()) {
847                         concreteIndices = concreteAllClosedIndices();
848                     } else {
849                         assert false : "Shouldn't end up here";
850                         concreteIndices = Strings.EMPTY_ARRAY;
851                     }
852                     result = new HashSet<>(Arrays.asList(concreteIndices));
853                 }
854                 add = false;
855                 aliasOrIndex = aliasOrIndex.substring(1);
856             }
857             if (!Regex.isSimpleMatchPattern(aliasOrIndex)) {
858                 if (!indicesOptions.ignoreUnavailable() && !aliasAndIndexToIndexMap.containsKey(aliasOrIndex)) {
859                     throw new IndexMissingException(new Index(aliasOrIndex));
860                 }
861                 if (result != null) {
862                     if (add) {
863                         result.add(aliasOrIndex);
864                     } else {
865                         result.remove(aliasOrIndex);
866                     }
867                 }
868                 continue;
869             }
870             if (result == null) {
871                 // add all the previous ones...
872                 result = new HashSet<>();
873                 result.addAll(Arrays.asList(aliasesOrIndices).subList(0, i));
874             }
875             String[] indices;
876             if (indicesOptions.expandWildcardsOpen() && indicesOptions.expandWildcardsClosed()) {
877                 indices = concreteAllIndices();
878             } else if (indicesOptions.expandWildcardsOpen()) {
879                 indices = concreteAllOpenIndices();
880             } else if (indicesOptions.expandWildcardsClosed()) {
881                 indices = concreteAllClosedIndices();
882             } else {
883                 assert false : "convertFromWildcards shouldn't get called if wildcards expansion is disabled";
884                 indices = Strings.EMPTY_ARRAY;
885             }
886             boolean found = false;
887             // iterating over all concrete indices and see if there is a wildcard match
888             for (String index : indices) {
889                 if (Regex.simpleMatch(aliasOrIndex, index)) {
890                     found = true;
891                     if (add) {
892                         result.add(index);
893                     } else {
894                         result.remove(index);
895                     }
896                 }
897             }
898             // iterating over all aliases and see if there is a wildcard match
899             for (ObjectCursor<String> cursor : aliases.keys()) {
900                 String alias = cursor.value;
901                 if (Regex.simpleMatch(aliasOrIndex, alias)) {
902                     found = true;
903                     if (add) {
904                         result.add(alias);
905                     } else {
906                         result.remove(alias);
907                     }
908                 }
909             }
910             if (!found && !indicesOptions.allowNoIndices()) {
911                 throw new IndexMissingException(new Index(aliasOrIndex));
912             }
913         }
914         if (result == null) {
915             return aliasesOrIndices;
916         }
917         if (result.isEmpty() && !indicesOptions.allowNoIndices()) {
918             throw new IndexMissingException(new Index(Arrays.toString(aliasesOrIndices)));
919         }
920         return result.toArray(new String[result.size()]);
921     }

980     /**
981      * Iterates through the list of indices and selects the effective list of filtering aliases for the
982      * given index.
983      * <p/>
984      * <p>Only aliases with filters are returned. If the indices list contains a non-filtering reference to
985      * the index itself - null is returned. Returns <tt>null</tt> if no filtering is required.</p>
986      */
987     public String[] filteringAliases(String index, String... indicesOrAliases) {
988         // expand the aliases wildcard
989         indicesOrAliases = convertFromWildcards(indicesOrAliases, IndicesOptions.lenientExpandOpen());
990 
991         if (isAllIndices(indicesOrAliases)) {
992             return null;
993         }
994         // optimize for the most common single index/alias scenario
995         if (indicesOrAliases.length == 1) {
996             String alias = indicesOrAliases[0];
997             IndexMetaData indexMetaData = this.indices.get(index);
998             if (indexMetaData == null) {
999                 // Shouldn't happen
1000                 throw new IndexMissingException(new Index(index));
1001             }
1002             AliasMetaData aliasMetaData = indexMetaData.aliases().get(alias);
1003             boolean filteringRequired = aliasMetaData != null && aliasMetaData.filteringRequired();
1004             if (!filteringRequired) {
1005                 return null;
1006             }
1007             return new String[]{alias};
1008         }
1009         List<String> filteringAliases = null;
1010         for (String alias : indicesOrAliases) {
1011             if (alias.equals(index)) {
1012                 return null;
1013             }
1014 
1015             IndexMetaData indexMetaData = this.indices.get(index);
1016             if (indexMetaData == null) {
1017                 // Shouldn't happen
1018                 throw new IndexMissingException(new Index(index));
1019             }
1020 
1021             AliasMetaData aliasMetaData = indexMetaData.aliases().get(alias);
1022             // Check that this is an alias for the current index
1023             // Otherwise - skip it
1024             if (aliasMetaData != null) {
1025                 boolean filteringRequired = aliasMetaData.filteringRequired();
1026                 if (filteringRequired) {
1027                     // If filtering required - add it to the list of filters
1028                     if (filteringAliases == null) {
1029                         filteringAliases = newArrayList();
1030                     }
1031                     filteringAliases.add(alias);
1032                 } else {
1033                     // If not, we have a non filtering alias for this index - no filtering needed
1034                     return null;
1035                 }
1036             }
1037         }
1038         if (filteringAliases == null) {
1039             return null;
1040         }
1041         return filteringAliases.toArray(new String[filteringAliases.size()]);
1042     }

1126         if (!metaData1.persistentSettings.equals(metaData2.persistentSettings)) {
1127             return false;
1128         }
1129         if (!metaData1.templates.equals(metaData2.templates())) {
1130             return false;

1135             if (customFactories.get(cursor.key).context().contains(XContentContext.GATEWAY)) {

1138             }

1140         int customCount2 = 0;
1141         for (ObjectObjectCursor<String, Custom> cursor : metaData2.customs) {
1142             if (customFactories.get(cursor.key).context().contains(XContentContext.GATEWAY)) {
1143                 customCount2++;
1144             }
1145         }
1146         if (customCount1 != customCount2) return false;
1147         return true;

1320         public static void toXContent(MetaData metaData, XContentBuilder builder, ToXContent.Params params) throws IOException {
1321             XContentContext context = XContentContext.valueOf(params.param(CONTEXT_MODE_PARAM, "API"));
1322 
1323             builder.startObject("meta-data");
1324 
1325             builder.field("version", metaData.version());
1326             builder.field("uuid", metaData.uuid);
1327 
1328             if (!metaData.persistentSettings().getAsMap().isEmpty()) {
1329                 builder.startObject("settings");
1330                 for (Map.Entry<String, String> entry : metaData.persistentSettings().getAsMap().entrySet()) {
1331                     builder.field(entry.getKey(), entry.getValue());
1332                 }
1333                 builder.endObject();
1334             }
1335 
1336             if (context == XContentContext.API && !metaData.transientSettings().getAsMap().isEmpty()) {
1337                 builder.startObject("transient_settings");
1338                 for (Map.Entry<String, String> entry : metaData.transientSettings().getAsMap().entrySet()) {
1339                     builder.field(entry.getKey(), entry.getValue());
1340                 }
1341                 builder.endObject();
1342             }
1343 
1344             builder.startObject("templates");
1345             for (ObjectCursor<IndexTemplateMetaData> cursor : metaData.templates().values()) {
1346                 IndexTemplateMetaData.Builder.toXContent(cursor.value, builder, params);
1347             }
1348             builder.endObject();
1349 
1350             if (context == XContentContext.API && !metaData.indices().isEmpty()) {
1351                 builder.startObject("indices");
1352                 for (IndexMetaData indexMetaData : metaData) {
1353                     IndexMetaData.Builder.toXContent(indexMetaData, builder, params);
1354                 }
1355                 builder.endObject();
1356             }
1357 
1358             for (ObjectObjectCursor<String, Custom> cursor : metaData.customs()) {
1359                 Custom.Factory factory = lookupFactorySafe(cursor.key);
1360                 if (factory.context().contains(context)) {
1361                     builder.startObject(cursor.key);
1362                     factory.toXContent(cursor.value, builder, params);
1363                     builder.endObject();
1364                 }
1365             }
1366             builder.endObject();
1367         }

1369         public static MetaData fromXContent(XContentParser parser) throws IOException {
1370             Builder builder = new Builder();
1371 
1372             // we might get here after the meta-data element, or on a fresh parser
1373             XContentParser.Token token = parser.currentToken();
1374             String currentFieldName = parser.currentName();
1375             if (!"meta-data".equals(currentFieldName)) {
1376                 token = parser.nextToken();
1377                 if (token == XContentParser.Token.START_OBJECT) {
1378                     // move to the field name (meta-data)
1379                     token = parser.nextToken();
1380                     // move to the next object
1381                     token = parser.nextToken();
1382                 }
1383                 currentFieldName = parser.currentName();
1384                 if (token == null) {
1385                     // no data...
1386                     return builder.build();
1387                 }
1388             }
1389 
1390             while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
1391                 if (token == XContentParser.Token.FIELD_NAME) {
1392                     currentFieldName = parser.currentName();
1393                 } else if (token == XContentParser.Token.START_OBJECT) {
1394                     if ("settings".equals(currentFieldName)) {
1395                         builder.persistentSettings(ImmutableSettings.settingsBuilder().put(SettingsLoader.Helper.loadNestedFromMap(parser.mapOrdered())).build());
1396                     } else if ("indices".equals(currentFieldName)) {
1397                         while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
1398                             builder.put(IndexMetaData.Builder.fromXContent(parser), false);
1399                         }
1400                     } else if ("templates".equals(currentFieldName)) {
1401                         while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
1402                             builder.put(IndexTemplateMetaData.Builder.fromXContent(parser, parser.currentName()));
1403                         }
1404                     } else {
1405                         // check if its a custom index metadata
1406                         Custom.Factory<Custom> factory = lookupFactory(currentFieldName);
1407                         if (factory == null) {
1408                             //TODO warn
1409                             parser.skipChildren();
1410                         } else {
1411                             builder.putCustom(factory.type(), factory.fromXContent(parser));
1412                         }
1413                     }
1414                 } else if (token.isValue()) {
1415                     if ("version".equals(currentFieldName)) {
1416                         builder.version = parser.longValue();
1417                     } else if ("uuid".equals(currentFieldName)) {
1418                         builder.uuid = parser.text();
1419                     }
1420                 }
1421             }
1422             return builder.build();
1423         }

1425         public static MetaData readFrom(StreamInput in) throws IOException {
1426             Builder builder = new Builder();
1427             builder.version = in.readLong();
1428             builder.uuid = in.readString();
1429             builder.transientSettings(readSettingsFromStream(in));
1430             builder.persistentSettings(readSettingsFromStream(in));
1431             int size = in.readVInt();
1432             for (int i = 0; i < size; i++) {
1433                 builder.put(IndexMetaData.Builder.readFrom(in), false);
1434             }
1435             size = in.readVInt();
1436             for (int i = 0; i < size; i++) {
1437                 builder.put(IndexTemplateMetaData.Builder.readFrom(in));
1438             }
1439             int customSize = in.readVInt();
1440             for (int i = 0; i < customSize; i++) {
1441                 String type = in.readString();
1442                 Custom customIndexMetaData = lookupFactorySafe(type).readFrom(in);
1443                 builder.putCustom(type, customIndexMetaData);
1444             }
1445             return builder.build();
1446         }

1448         public static void writeTo(MetaData metaData, StreamOutput out) throws IOException {
1449             out.writeLong(metaData.version);
1450             out.writeString(metaData.uuid);
1451             writeSettingsToStream(metaData.transientSettings(), out);
1452             writeSettingsToStream(metaData.persistentSettings(), out);
1453             out.writeVInt(metaData.indices.size());
1454             for (IndexMetaData indexMetaData : metaData) {
1455                 IndexMetaData.Builder.writeTo(indexMetaData, out);
1456             }
1457             out.writeVInt(metaData.templates.size());
1458             for (ObjectCursor<IndexTemplateMetaData> cursor : metaData.templates.values()) {
1459                 IndexTemplateMetaData.Builder.writeTo(cursor.value, out);
1460             }
1461             out.writeVInt(metaData.customs().size());
1462             for (ObjectObjectCursor<String, Custom> cursor : metaData.customs()) {
1463                 out.writeString(cursor.key);
1464                 lookupFactorySafe(cursor.key).writeTo(cursor.value, out);
1465             }
1466         }
