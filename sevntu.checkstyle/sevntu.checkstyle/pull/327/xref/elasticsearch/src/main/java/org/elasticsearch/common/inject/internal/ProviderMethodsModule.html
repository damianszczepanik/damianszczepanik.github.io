17  package org.elasticsearch.common.inject.internal;
18  
19  import com.google.common.collect.ImmutableSet;
20  import com.google.common.collect.Lists;
21  import org.elasticsearch.common.inject.*;
22  import org.elasticsearch.common.inject.spi.Dependency;
23  import org.elasticsearch.common.inject.spi.Message;
24  import org.elasticsearch.common.inject.util.Modules;
25  
26  import java.lang.annotation.Annotation;
27  import java.lang.reflect.Member;
28  import java.lang.reflect.Method;
29  import java.util.List;
30  
31  import static com.google.common.base.Preconditions.checkNotNull;
32  
33  /**
34   * Creates bindings to methods annotated with {@literal @}{@link Provides}. Use the scope and
35   * binding annotations on the provider method to configure the binding.
36   *
37   * @author crazybob@google.com (Bob Lee)
38   * @author jessewilson@google.com (Jesse Wilson)
39   */
40  public final class ProviderMethodsModule implements Module {
41      private final Object delegate;
42      private final TypeLiteral<?> typeLiteral;
43  
44      private ProviderMethodsModule(Object delegate) {
45          this.delegate = checkNotNull(delegate, "delegate");
46          this.typeLiteral = TypeLiteral.get(this.delegate.getClass());
47      }
48  
49      /**
50       * Returns a module which creates bindings for provider methods from the given module.
51       */
52      public static Module forModule(Module module) {
53          return forObject(module);
54      }
55  
56      /**
57       * Returns a module which creates bindings for provider methods from the given object.
58       * This is useful notably for <a href="http://code.google.com/p/google-gin/">GIN</a>
59       */
60      public static Module forObject(Object object) {
61          // avoid infinite recursion, since installing a module always installs itself
62          if (object instanceof ProviderMethodsModule) {
63              return Modules.EMPTY_MODULE;
64          }
65  
66          return new ProviderMethodsModule(object);
67      }
68  
69      @Override
70      public synchronized void configure(Binder binder) {
71          for (ProviderMethod<?> providerMethod : getProviderMethods(binder)) {
72              providerMethod.configure(binder);
73          }
74      }
75  
76      public List<ProviderMethod<?>> getProviderMethods(Binder binder) {
77          List<ProviderMethod<?>> result = Lists.newArrayList();
78          for (Class<?> c = delegate.getClass(); c != Object.class; c = c.getSuperclass()) {
79              for (Method method : c.getDeclaredMethods()) {
80                  if (method.getAnnotation(Provides.class) != null) {
81                      result.add(createProviderMethod(binder, method));
82                  }
83              }
84          }
85          return result;
86      }
87  
88      <T> ProviderMethod<T> createProviderMethod(Binder binder, final Method method) {
89          binder = binder.withSource(method);
90          Errors errors = new Errors(method);
91  
92          // prepare the parameter providers
93          List<Dependency<?>> dependencies = Lists.newArrayList();
94          List<Provider<?>> parameterProviders = Lists.newArrayList();
95          List<TypeLiteral<?>> parameterTypes = typeLiteral.getParameterTypes(method);
96          Annotation[][] parameterAnnotations = method.getParameterAnnotations();
97          for (int i = 0; i < parameterTypes.size(); i++) {
98              Key<?> key = getKey(errors, parameterTypes.get(i), method, parameterAnnotations[i]);
99              dependencies.add(Dependency.get(key));
100             parameterProviders.add(binder.getProvider(key));
101         }
102 
103         @SuppressWarnings("unchecked") // Define T as the method's return type.
104                 TypeLiteral<T> returnType = (TypeLiteral<T>) typeLiteral.getReturnType(method);
105 
106         Key<T> key = getKey(errors, returnType, method, method.getAnnotations());
107         Class<? extends Annotation> scopeAnnotation
108                 = Annotations.findScopeAnnotation(errors, method.getAnnotations());
109 
110         for (Message message : errors.getMessages()) {
111             binder.addError(message);
112         }
113 
114         return new ProviderMethod<>(key, method, delegate, ImmutableSet.copyOf(dependencies),
115                 parameterProviders, scopeAnnotation);
116     }
117 
118     <T> Key<T> getKey(Errors errors, TypeLiteral<T> type, Member member, Annotation[] annotations) {
119         Annotation bindingAnnotation = Annotations.findBindingAnnotation(errors, member, annotations);
120         return bindingAnnotation == null ? Key.get(type) : Key.get(type, bindingAnnotation);
121     }
122 
123     @Override
124     public boolean equals(Object o) {
125         return o instanceof ProviderMethodsModule
126                 && ((ProviderMethodsModule) o).delegate == delegate;
127     }
128 
129     @Override
130     public int hashCode() {
131         return delegate.hashCode();
132     }
133 }
