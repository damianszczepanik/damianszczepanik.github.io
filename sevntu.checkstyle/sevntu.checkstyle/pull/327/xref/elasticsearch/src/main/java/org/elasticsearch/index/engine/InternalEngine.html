106     public InternalEngine(EngineConfig engineConfig) throws EngineException {
107         super(engineConfig);
108         this.versionMap = new LiveVersionMap();
109         store.incRef();
110         IndexWriter writer = null;
111         SearcherManager manager = null;
112         boolean success = false;
113         try {
114             this.onGoingRecoveries = new FlushingRecoveryCounter(this, store, logger);
115             this.lastDeleteVersionPruneTimeMSec = engineConfig.getThreadPool().estimatedTimeInMillis();
116             this.indexingService = engineConfig.getIndexingService();
117             this.warmer = engineConfig.getWarmer();
118             this.translog = engineConfig.getTranslog();
119             this.mergePolicyProvider = engineConfig.getMergePolicyProvider();
120             this.mergeScheduler = engineConfig.getMergeScheduler();
121             this.dirtyLocks = new Object[engineConfig.getIndexConcurrency() * 50]; // we multiply it to have enough...
122             for (int i = 0; i < dirtyLocks.length; i++) {
123                 dirtyLocks[i] = new Object();
124             }
125 
126             throttle = new IndexThrottle();
127             this.searcherFactory = new SearchFactory(engineConfig);
128             try {
129                 writer = createWriter();
130             } catch (IOException e) {
131                 throw new EngineCreationFailureException(shardId, "failed to create engine", e);
132             }
133             indexWriter = writer;
134             manager = createSearcherManager();
135             this.searcherManager = manager;
136             this.mergeSchedulerFailureListener = new FailEngineOnMergeFailure();
137             this.mergeSchedulerListener = new MergeSchedulerListener();
138             this.mergeScheduler.addListener(mergeSchedulerListener);
139             this.mergeScheduler.addFailureListener(mergeSchedulerFailureListener);
140             success = true;
141         } finally {
142             if (success == false) {
143                 IOUtils.closeWhileHandlingException(writer, manager);
144                 versionMap.clear();
145                 if (isClosed.get() == false) {
146                     // failure we need to dec the store reference
147                     store.decRef();
148                 }
149             }
150         }
151     }

153     private SearcherManager createSearcherManager() throws EngineException {
154         boolean success = false;
155         SearcherManager searcherManager = null;
156         try {
157             try {
158                 // commit on a just opened writer will commit even if there are no changes done to it
159                 // we rely on that for the commit data translog id key
160                 final long translogId = Math.max(0, translog.findLargestPresentTranslogId()) + 1;
161                 boolean mustCommitTranslogId = true;
162                 if (Lucene.indexExists(store.directory())) {
163                     final Map<String, String> commitUserData = Lucene.readSegmentInfos(store.directory()).getUserData();
164                     mustCommitTranslogId = !commitUserData.containsKey(Translog.TRANSLOG_ID_KEY);
165                 }
166                 if (mustCommitTranslogId) { // translog id is not in the metadata - fix this inconsistency some code relies on this and old indices might not have it.
167                     indexWriter.setCommitData(Collections.singletonMap(Translog.TRANSLOG_ID_KEY, Long.toString(translogId)));
168                     commitIndexWriter(indexWriter);
169                 }
170                 final DirectoryReader directoryReader = ElasticsearchDirectoryReader.wrap(DirectoryReader.open(indexWriter, true), shardId);
171                 searcherManager = new SearcherManager(directoryReader, searcherFactory);
172                 lastCommittedSegmentInfos = store.readLastCommittedSegmentsInfo();
173                 translog.newTranslog(translogId);
174                 versionMap.setManager(searcherManager);
175                 translogIdGenerator.set(translogId);
176                 success = true;
177                 return searcherManager;
178             } catch (IOException e) {
179                 maybeFailEngine("start", e);
180                 try {
181                     indexWriter.rollback();
182                 } catch (IOException e1) { // iw is closed below
183                     e.addSuppressed(e1);
184                 }
185                 throw new EngineCreationFailureException(shardId, "failed to open reader on writer", e);
186             }
187         } finally {
188             if (success == false) { // release everything we created on a failure
189                 IOUtils.closeWhileHandlingException(searcherManager, indexWriter);
190             }
191         }
192     }

204     @Override
205     public GetResult get(Get get) throws EngineException {
206         try (ReleasableLock lock = readLock.acquire()) {
207             ensureOpen();
208             if (get.realtime()) {
209                 VersionValue versionValue = versionMap.getUnderLock(get.uid().bytes());
210                 if (versionValue != null) {
211                     if (versionValue.delete()) {
212                         return GetResult.NOT_EXISTS;
213                     }
214                     if (get.versionType().isVersionConflictForReads(versionValue.version(), get.version())) {
215                         Uid uid = Uid.createUid(get.uid().text());
216                         throw new VersionConflictEngineException(shardId, uid.type(), uid.id(), versionValue.version(), get.version());
217                     }
218                     if (!get.loadSource()) {
219                         return new GetResult(true, versionValue.version(), null);
220                     }
221                     Translog.Operation op = translog.read(versionValue.translogLocation());
222                     if (op != null) {
223                         return new GetResult(true, versionValue.version(), op.getSource());
224                     }
225                 }
226             }
227 
228             // no version, get the version from the index, we know that we refresh on flush
229             return getFromSearcher(get);
230         }
231     }

233     @Override
234     public void create(Create create) throws EngineException {
235         try (ReleasableLock lock = readLock.acquire()) {
236             ensureOpen();
237             if (create.origin() == Operation.Origin.RECOVERY) {
238                 // Don't throttle recovery operations
239                 innerCreate(create);
240             } else {
241                 try (Releasable r = throttle.acquireThrottle()) {
242                     innerCreate(create);
243                 }
244             }
245             flushNeeded = true;
246         } catch (OutOfMemoryError | IllegalStateException | IOException t) {
247             maybeFailEngine("create", t);
248             throw new CreateFailedEngineException(shardId, create, t);
249         }
250         checkVersionMapRefresh();
251     }

253     private void innerCreate(Create create) throws IOException {
254         if (engineConfig.isOptimizeAutoGenerateId() && create.autoGeneratedId() && !create.canHaveDuplicates()) {
255             // We don't need to lock because this ID cannot be concurrently updated:
256             innerCreateNoLock(create, Versions.NOT_FOUND, null);
257         } else {
258             synchronized (dirtyLock(create.uid())) {
259                 final long currentVersion;
260                 final VersionValue versionValue;
261                 versionValue = versionMap.getUnderLock(create.uid().bytes());
262                 if (versionValue == null) {
263                     currentVersion = loadCurrentVersionFromIndex(create.uid());
264                 } else {
265                     if (engineConfig.isEnableGcDeletes() && versionValue.delete() && (engineConfig.getThreadPool().estimatedTimeInMillis() - versionValue.time()) > engineConfig.getGcDeletesInMillis()) {
266                         currentVersion = Versions.NOT_FOUND; // deleted, and GC
267                     } else {
268                         currentVersion = versionValue.version();
269                     }
270                 }
271                 innerCreateNoLock(create, currentVersion, versionValue);
272             }
273         }
274     }

276     private void innerCreateNoLock(Create create, long currentVersion, VersionValue versionValue) throws IOException {
277 
278         // same logic as index
279         long updatedVersion;
280         long expectedVersion = create.version();
281         if (create.versionType().isVersionConflictForWrites(currentVersion, expectedVersion)) {
282             if (create.origin() == Operation.Origin.RECOVERY) {
283                 return;
284             } else {
285                 throw new VersionConflictEngineException(shardId, create.type(), create.id(), currentVersion, expectedVersion);
286             }
287         }
288         updatedVersion = create.versionType().updateVersion(currentVersion, expectedVersion);
289 
290         // if the doc exists
291         boolean doUpdate = false;
292         if ((versionValue != null && versionValue.delete() == false) || (versionValue == null && currentVersion != Versions.NOT_FOUND)) {
293             if (create.origin() == Operation.Origin.RECOVERY) {
294                 return;
295             } else if (create.origin() == Operation.Origin.REPLICA) {
296                 // #7142: the primary already determined it's OK to index this document, and we confirmed above that the version doesn't
297                 // conflict, so we must also update here on the replica to remain consistent:
298                 doUpdate = true;
299             } else if (create.origin() == Operation.Origin.PRIMARY && create.autoGeneratedId() && create.canHaveDuplicates() && currentVersion == 1 && create.version() == Versions.MATCH_ANY) {
300                 /**
301                  * If bulk index request fails due to a disconnect, unavailable shard etc. then the request is
302                  * retried before it actually fails. However, the documents might already be indexed.
303                  * For autogenerated ids this means that a version conflict will be reported in the bulk request
304                  * although the document was indexed properly.
305                  * To avoid this we have to make sure that the index request is treated as an update and set updatedVersion to 1.
306                  * See also discussion on https://github.com/elasticsearch/elasticsearch/pull/9125
307                  */
308                 doUpdate = true;
309                 updatedVersion = 1;
310             } else {
311                 // On primary, we throw DAEE if the _uid is already in the index with an older version:
312                 assert create.origin() == Operation.Origin.PRIMARY;
313                 throw new DocumentAlreadyExistsException(shardId, create.type(), create.id());
314             }
315         }
316 
317         create.updateVersion(updatedVersion);
318 
319         if (doUpdate) {
320             if (create.docs().size() > 1) {
321                 indexWriter.updateDocuments(create.uid(), create.docs());
322             } else {
323                 indexWriter.updateDocument(create.uid(), create.docs().get(0));
324             }
325         } else {
326             if (create.docs().size() > 1) {
327                 indexWriter.addDocuments(create.docs());
328             } else {
329                 indexWriter.addDocument(create.docs().get(0));
330             }
331         }
332         Translog.Location translogLocation = translog.add(new Translog.Create(create));
333 
334         versionMap.putUnderLock(create.uid().bytes(), new VersionValue(updatedVersion, translogLocation));
335 
336         indexingService.postCreateUnderLock(create);
337     }

339     @Override
340     public void index(Index index) throws EngineException {
341         try (ReleasableLock lock = readLock.acquire()) {
342             ensureOpen();
343             if (index.origin() == Operation.Origin.RECOVERY) {
344                 // Don't throttle recovery operations
345                 innerIndex(index);
346             } else {
347                 try (Releasable r = throttle.acquireThrottle()) {
348                     innerIndex(index);
349                 }
350             }
351             flushNeeded = true;
352         } catch (OutOfMemoryError | IllegalStateException | IOException t) {
353             maybeFailEngine("index", t);
354             throw new IndexFailedEngineException(shardId, index, t);
355         }
356         checkVersionMapRefresh();
357     }

359     /**
360      * Forces a refresh if the versionMap is using too much RAM (currently > 25% of IndexWriter's RAM buffer).
361      */
362     private void checkVersionMapRefresh() {
363         // TODO: we force refresh when versionMap is using > 25% of IW's RAM buffer; should we make this separately configurable?
364         if (versionMap.ramBytesUsedForRefresh() > 0.25 * engineConfig.getIndexingBufferSize().bytes() && versionMapRefreshPending.getAndSet(true) == false) {
365             try {
366                 if (isClosed.get()) {
367                     // no point...
368                     return;
369                 }
370                 // Now refresh to clear versionMap:
371                 engineConfig.getThreadPool().executor(ThreadPool.Names.REFRESH).execute(new Runnable() {
372                     @Override
373                     public void run() {
374                         try {
375                             refresh("version_table_full");
376                         } catch (EngineClosedException ex) {
377                             // ignore
378                         }
379                     }
380                 });
381             } catch (EsRejectedExecutionException ex) {
382                 // that is fine too.. we might be shutting down
383             }
384         }
385     }

387     private void innerIndex(Index index) throws IOException {
388         synchronized (dirtyLock(index.uid())) {
389             final long currentVersion;
390             VersionValue versionValue = versionMap.getUnderLock(index.uid().bytes());
391             if (versionValue == null) {
392                 currentVersion = loadCurrentVersionFromIndex(index.uid());
393             } else {
394                 if (engineConfig.isEnableGcDeletes() && versionValue.delete() && (engineConfig.getThreadPool().estimatedTimeInMillis() - versionValue.time()) > engineConfig.getGcDeletesInMillis()) {
395                     currentVersion = Versions.NOT_FOUND; // deleted, and GC
396                 } else {
397                     currentVersion = versionValue.version();
398                 }
399             }
400 
401             long updatedVersion;
402             long expectedVersion = index.version();
403             if (index.versionType().isVersionConflictForWrites(currentVersion, expectedVersion)) {
404                 if (index.origin() == Operation.Origin.RECOVERY) {
405                     return;
406                 } else {
407                     throw new VersionConflictEngineException(shardId, index.type(), index.id(), currentVersion, expectedVersion);
408                 }
409             }
410             updatedVersion = index.versionType().updateVersion(currentVersion, expectedVersion);
411 
412             index.updateVersion(updatedVersion);
413             if (currentVersion == Versions.NOT_FOUND) {
414                 // document does not exists, we can optimize for create
415                 index.created(true);
416                 if (index.docs().size() > 1) {
417                     indexWriter.addDocuments(index.docs());
418                 } else {
419                     indexWriter.addDocument(index.docs().get(0));
420                 }
421             } else {
422                 if (versionValue != null) {
423                     index.created(versionValue.delete()); // we have a delete which is not GC'ed...
424                 }
425                 if (index.docs().size() > 1) {
426                     indexWriter.updateDocuments(index.uid(), index.docs());
427                 } else {
428                     indexWriter.updateDocument(index.uid(), index.docs().get(0));
429                 }
430             }
431             Translog.Location translogLocation = translog.add(new Translog.Index(index));
432 
433             versionMap.putUnderLock(index.uid().bytes(), new VersionValue(updatedVersion, translogLocation));
434 
435             indexingService.postIndexUnderLock(index);
436         }
437     }

439     @Override
440     public void delete(Delete delete) throws EngineException {
441         try (ReleasableLock lock = readLock.acquire()) {
442             ensureOpen();
443             // NOTE: we don't throttle this when merges fall behind because delete-by-id does not create new segments:
444             innerDelete(delete);
445             flushNeeded = true;
446         } catch (OutOfMemoryError | IllegalStateException | IOException t) {
447             maybeFailEngine("delete", t);
448             throw new DeleteFailedEngineException(shardId, delete, t);
449         }
450 
451         maybePruneDeletedTombstones();
452     }

454     private void maybePruneDeletedTombstones() {
455         // It's expensive to prune because we walk the deletes map acquiring dirtyLock for each uid so we only do it
456         // every 1/4 of gcDeletesInMillis:
457         if (engineConfig.isEnableGcDeletes() && engineConfig.getThreadPool().estimatedTimeInMillis() - lastDeleteVersionPruneTimeMSec > engineConfig.getGcDeletesInMillis() * 0.25) {
458             pruneDeletedTombstones();
459         }
460     }

462     private void innerDelete(Delete delete) throws IOException {
463         synchronized (dirtyLock(delete.uid())) {
464             final long currentVersion;
465             VersionValue versionValue = versionMap.getUnderLock(delete.uid().bytes());
466             if (versionValue == null) {
467                 currentVersion = loadCurrentVersionFromIndex(delete.uid());
468             } else {
469                 if (engineConfig.isEnableGcDeletes() && versionValue.delete() && (engineConfig.getThreadPool().estimatedTimeInMillis() - versionValue.time()) > engineConfig.getGcDeletesInMillis()) {
470                     currentVersion = Versions.NOT_FOUND; // deleted, and GC
471                 } else {
472                     currentVersion = versionValue.version();
473                 }
474             }
475 
476             long updatedVersion;
477             long expectedVersion = delete.version();
478             if (delete.versionType().isVersionConflictForWrites(currentVersion, expectedVersion)) {
479                 if (delete.origin() == Operation.Origin.RECOVERY) {
480                     return;
481                 } else {
482                     throw new VersionConflictEngineException(shardId, delete.type(), delete.id(), currentVersion, expectedVersion);
483                 }
484             }
485             updatedVersion = delete.versionType().updateVersion(currentVersion, expectedVersion);
486             final boolean found;
487             if (currentVersion == Versions.NOT_FOUND) {
488                 // doc does not exist and no prior deletes
489                 found = false;
490             } else if (versionValue != null && versionValue.delete()) {
491                 // a "delete on delete", in this case, we still increment the version, log it, and return that version
492                 found = false;
493             } else {
494                 // we deleted a currently existing document
495                 indexWriter.deleteDocuments(delete.uid());
496                 found = true;
497             }
498 
499             delete.updateVersion(updatedVersion, found);
500             Translog.Location translogLocation = translog.add(new Translog.Delete(delete));
501             versionMap.putUnderLock(delete.uid().bytes(), new DeleteVersionValue(updatedVersion, engineConfig.getThreadPool().estimatedTimeInMillis(), translogLocation));
502 
503             indexingService.postDeleteUnderLock(delete);
504         }
505     }

510             ensureOpen();
511             if (delete.origin() == Operation.Origin.RECOVERY) {
512                 // Don't throttle recovery operations
513                 innerDelete(delete);
514             } else {
515                 try (Releasable r = throttle.acquireThrottle()) {

518             }

522     private void innerDelete(DeleteByQuery delete) throws EngineException {
523         try {
524             Query query;
525             if (delete.nested() && delete.aliasFilter() != null) {
526                 query = new IncludeNestedDocsQuery(new FilteredQuery(delete.query(), delete.aliasFilter()), delete.parentFilter());
527             } else if (delete.nested()) {
528                 query = new IncludeNestedDocsQuery(delete.query(), delete.parentFilter());
529             } else if (delete.aliasFilter() != null) {
530                 query = new FilteredQuery(delete.query(), delete.aliasFilter());
531             } else {
532                 query = delete.query();
533             }
534 
535             indexWriter.deleteDocuments(query);
536             translog.add(new Translog.DeleteByQuery(delete));
537             flushNeeded = true;
538         } catch (Throwable t) {
539             maybeFailEngine("delete_by_query", t);
540             throw new DeleteByQueryFailedEngineException(shardId, delete, t);
541         }
542 
543         // TODO: This is heavy, since we refresh, but we must do this because we don't know which documents were in fact deleted (i.e., our
544         // versionMap isn't updated), so we must force a cutover to a new reader to "see" the deletions:
545         refresh("delete_by_query");
546     }

548     @Override
549     public void refresh(String source) throws EngineException {
550         // we obtain a read lock here, since we don't want a flush to happen while we are refreshing
551         // since it flushes the index as well (though, in terms of concurrency, we are allowed to do it)
552         try (ReleasableLock lock = readLock.acquire()) {
553             ensureOpen();
554             updateIndexWriterSettings();
555             searcherManager.maybeRefreshBlocking();
556         } catch (AlreadyClosedException e) {
557             ensureOpen();
558             maybeFailEngine("refresh", e);
559         } catch (EngineClosedException e) {
560             throw e;
561         } catch (Throwable t) {
562             failEngine("refresh failed", t);
563             throw new RefreshFailedEngineException(shardId, t);
564         }
565 
566         // TODO: maybe we should just put a scheduled job in threadPool?
567         // We check for pruning in each delete request, but we also prune here e.g. in case a delete burst comes in and then no more deletes
568         // for a long time:
569         maybePruneDeletedTombstones();
570         versionMapRefreshPending.set(false);
571     }

583     private void flush(boolean commitTranslog, boolean force, boolean waitIfOngoing) throws EngineException {
584         ensureOpen();
585         if (commitTranslog) {
586             // check outside the lock as well so we can check without blocking on the write lock
587             if (onGoingRecoveries.get() > 0) {
588                 throw new FlushNotAllowedEngineException(shardId, "recovery is in progress, flush with committing translog is not allowed");
589             }
590         }
591         /*
592          * Unfortunately the lock order is important here. We have to acquire the readlock fist otherwise
593          * if we are flushing at the end of the recovery while holding the write lock we can deadlock if:
594          *  Thread 1: flushes via API and gets the flush lock but blocks on the readlock since Thread 2 has the writeLock
595          *  Thread 2: flushes at the end of the recovery holding the writeLock and blocks on the flushLock owned by Thread 1
596          */
597         try (ReleasableLock lock = readLock.acquire()) {
598             ensureOpen();
599             updateIndexWriterSettings();
600             if (flushLock.tryLock() == false) {
601                 // if we can't get the lock right away we block if needed otherwise barf
602                 if (waitIfOngoing) {
603                     logger.trace("waiting fore in-flight flush to finish");
604                     flushLock.lock();
605                     logger.trace("acquired flush lock after blocking");
606                 } else {
607                     throw new FlushNotAllowedEngineException(shardId, "already flushing...");
608                 }
609             } else {
610                 logger.trace("acquired flush lock immediately");
611             }
612             try {
613                 if (commitTranslog) {
614                     if (onGoingRecoveries.get() > 0) {
615                         throw new FlushNotAllowedEngineException(shardId, "Recovery is in progress, flush is not allowed");
616                     }
617 
618                     if (flushNeeded || force) {
619                         flushNeeded = false;
620                         try {
621                             long translogId = translogIdGenerator.incrementAndGet();
622                             translog.newTransientTranslog(translogId);
623                             indexWriter.setCommitData(Collections.singletonMap(Translog.TRANSLOG_ID_KEY, Long.toString(translogId)));
624                             commitIndexWriter(indexWriter);
625                             // we need to refresh in order to clear older version values
626                             refresh("version_table_flush");
627                             // we need to move transient to current only after we refresh
628                             // so items added to current will still be around for realtime get
629                             // when tans overrides it
630                             translog.makeTransientCurrent();
631 
632                         } catch (Throwable e) {
633                             try {
634                                 translog.revertTransient();
635                             } catch (IOException ex) {
636                                 e.addSuppressed(ex);
637                             }
638                             throw new FlushFailedEngineException(shardId, e);
639                         }
640                     }
641                 } else {
642                     // note, its ok to just commit without cleaning the translog, its perfectly fine to replay a
643                     // translog on an index that was opened on a committed point in time that is "in the future"
644                     // of that translog
645                     // we allow to *just* commit if there is an ongoing recovery happening...
646                     // its ok to use this, only a flush will cause a new translogId, and we are locked here from
647                     // other flushes use flushLock
648                     try {
649                         long translogId = translog.currentId();
650                         indexWriter.setCommitData(Collections.singletonMap(Translog.TRANSLOG_ID_KEY, Long.toString(translogId)));
651                         commitIndexWriter(indexWriter);
652                     } catch (Throwable e) {
653                         throw new FlushFailedEngineException(shardId, e);
654                     }
655 
656                 }
657                 /*
658                  * we have to inc-ref the store here since if the engine is closed by a tragic event
659                  * we don't acquire the write lock and wait until we have exclusive access. This might also
660                  * dec the store reference which can essentially close the store and unless we can inc the reference
661                  * we can't use it.
662                  */
663                 store.incRef();
664                 try {
665                     // reread the last committed segment infos
666                     lastCommittedSegmentInfos = store.readLastCommittedSegmentsInfo();
667                 } catch (Throwable e) {
668                     if (isClosed.get() == false) {
669                         logger.warn("failed to read latest segment infos on flush", e);
670                         if (Lucene.isCorruptionException(e)) {
671                             throw new FlushFailedEngineException(shardId, e);
672                         }
673                     }
674                 } finally {
675                     store.decRef();
676                 }
677             } catch (FlushFailedEngineException ex) {
678                 maybeFailEngine("flush", ex);
679                 throw ex;
680             } finally {
681                 flushLock.unlock();
682             }
683         }
684         // We don't have to do this here; we do it defensively to make sure that even if wall clock time is misbehaving
685         // (e.g., moves backwards) we will at least still sometimes prune deleted tombstones:
686         if (engineConfig.isEnableGcDeletes()) {
687             pruneDeletedTombstones();
688         }
689     }
690 
691     private void pruneDeletedTombstones() {
692         long timeMSec = engineConfig.getThreadPool().estimatedTimeInMillis();
693 
694         // TODO: not good that we reach into LiveVersionMap here; can we move this inside VersionMap instead?  problem is the dirtyLock...
695 
696         // we only need to prune the deletes map; the current/old version maps are cleared on refresh:
697         for (Map.Entry<BytesRef, VersionValue> entry : versionMap.getAllTombstones()) {
698             BytesRef uid = entry.getKey();
699             synchronized (dirtyLock(uid)) { // can we do it without this lock on each value? maybe batch to a set and get the lock once per set?
700 
701                 // Must re-get it here, vs using entry.getValue(), in case the uid was indexed/deleted since we pulled the iterator:
702                 VersionValue versionValue = versionMap.getTombstoneUnderLock(uid);
703                 if (versionValue != null) {
704                     if (timeMSec - versionValue.time() > engineConfig.getGcDeletesInMillis()) {
705                         versionMap.removeTombstoneUnderLock(uid);
706                     }
707                 }
708             }
709         }
710 
711         lastDeleteVersionPruneTimeMSec = timeMSec;
712     }

731     @Override
732     public void forceMerge(final boolean flush, int maxNumSegments, boolean onlyExpungeDeletes, final boolean upgrade) throws EngineException {
733         if (optimizeMutex.compareAndSet(false, true)) {
734             try (ReleasableLock lock = readLock.acquire()) {
735                 ensureOpen();
736                 /*
737                  * The way we implement upgrades is a bit hackish in the sense that we set an instance
738                  * variable and that this setting will thus apply to the next forced merge that will be run.
739                  * This is ok because (1) this is the only place we call forceMerge, (2) we have a single
740                  * thread for optimize, and the 'optimizeMutex' guarding this code, and (3) ConcurrentMergeScheduler
741                  * syncs calls to findForcedMerges.
742                  */
743                 MergePolicy mp = indexWriter.getConfig().getMergePolicy();
744                 assert mp instanceof ElasticsearchMergePolicy : "MergePolicy is " + mp.getClass().getName();
745                 if (upgrade) {
746                     logger.info("Starting upgrade of " + shardId);
747                     ((ElasticsearchMergePolicy) mp).setUpgradeInProgress(true);
748                 }
749 
750                 if (onlyExpungeDeletes) {
751                     indexWriter.forceMergeDeletes(false);
752                 } else if (maxNumSegments <= 0) {
753                     indexWriter.maybeMerge();
754                 } else {
755                     indexWriter.forceMerge(maxNumSegments, false);
756                 }
757             } catch (Throwable t) {
758                 maybeFailEngine("optimize", t);
759                 throw new OptimizeFailedEngineException(shardId, t);
760             } finally {
761                 optimizeMutex.set(false);
762             }
763         }
764 
765         waitForMerges(flush, upgrade);
766     }

770         // we have to flush outside of the readlock otherwise we might have a problem upgrading
771         // the to a write lock when we fail the engine in this operation
772         flush(false, false, true);

781     @Override
782     public void recover(RecoveryHandler recoveryHandler) throws EngineException {
783         // take a write lock here so it won't happen while a flush is in progress
784         // this means that next commits will not be allowed once the lock is released
785         try (ReleasableLock lock = writeLock.acquire()) {
786             ensureOpen();
787             onGoingRecoveries.startRecovery();
788         }
789 
790         SnapshotIndexCommit phase1Snapshot;
791         try {
792             phase1Snapshot = deletionPolicy.snapshot();
793         } catch (Throwable e) {
794             maybeFailEngine("recovery", e);
795             Releasables.closeWhileHandlingException(onGoingRecoveries);
796             throw new RecoveryEngineException(shardId, 1, "Snapshot failed", e);
797         }
798 
799         try {
800             recoveryHandler.phase1(phase1Snapshot);
801         } catch (Throwable e) {
802             maybeFailEngine("recovery phase 1", e);
803             Releasables.closeWhileHandlingException(phase1Snapshot, onGoingRecoveries);
804             throw new RecoveryEngineException(shardId, 1, "Execution failed", wrapIfClosed(e));
805         }
806 
807         Translog.Snapshot phase2Snapshot;
808         try {
809             phase2Snapshot = translog.snapshot();
810         } catch (Throwable e) {
811             maybeFailEngine("snapshot recovery", e);
812             Releasables.closeWhileHandlingException(phase1Snapshot, onGoingRecoveries);
813             throw new RecoveryEngineException(shardId, 2, "Snapshot failed", wrapIfClosed(e));
814         }
815         try {
816             recoveryHandler.phase2(phase2Snapshot);
817         } catch (Throwable e) {
818             maybeFailEngine("recovery phase 2", e);
819             Releasables.closeWhileHandlingException(phase1Snapshot, phase2Snapshot, onGoingRecoveries);
820             throw new RecoveryEngineException(shardId, 2, "Execution failed", wrapIfClosed(e));
821         }
822 
823         writeLock.acquire();
824         Translog.Snapshot phase3Snapshot = null;
825         boolean success = false;
826         try {
827             ensureOpen();
828             phase3Snapshot = translog.snapshot(phase2Snapshot);
829             recoveryHandler.phase3(phase3Snapshot);
830             success = true;
831         } catch (Throwable e) {
832             maybeFailEngine("recovery phase 3", e);
833             throw new RecoveryEngineException(shardId, 3, "Execution failed", wrapIfClosed(e));
834         } finally {
835             Releasables.close(success, phase1Snapshot, phase2Snapshot, phase3Snapshot,
836                     onGoingRecoveries, writeLock); // hmm why can't we use try-with here?
837         }
838     }

840     @Override
841     protected boolean maybeFailEngine(String source, Throwable t) {
842         boolean shouldFail = super.maybeFailEngine(source, t);
843         if (shouldFail) {
844             return true;
845         }
846 
847         // Check for AlreadyClosedException
848         if (t instanceof AlreadyClosedException) {
849             // if we are already closed due to some tragic exception
850             // we need to fail the engine. it might have already been failed before
851             // but we are double-checking it's failed and closed
852             if (indexWriter.isOpen() == false && indexWriter.getTragicException() != null) {
853                 failEngine("already closed by tragic event", indexWriter.getTragicException());
854             }
855             return true;
856         } else if (t != null && indexWriter.isOpen() == false && indexWriter.getTragicException() == t) {
857             // this spot on - we are handling the tragic event exception here so we have to fail the engine
858             // right away
859             failEngine(source, t);
860             return true;
861         }
862         return false;
863     }

873     public List<Segment> segments(boolean verbose) {
874         try (ReleasableLock lock = readLock.acquire()) {
875             Segment[] segmentsArr = getSegmentInfo(lastCommittedSegmentInfos, verbose);
876 
877             // fill in the merges flag
878             Set<OnGoingMerge> onGoingMerges = mergeScheduler.onGoingMerges();
879             for (OnGoingMerge onGoingMerge : onGoingMerges) {
880                 for (SegmentCommitInfo segmentInfoPerCommit : onGoingMerge.getMergedSegments()) {
881                     for (Segment segment : segmentsArr) {

886                     }
887                 }

894     /**
895      * Closes the engine without acquiring the write lock. This should only be
896      * called while the write lock is hold or in a disaster condition ie. if the engine
897      * is failed.
898      */
899     @Override
900     protected final void closeNoLock(String reason) throws ElasticsearchException {
901         if (isClosed.compareAndSet(false, true)) {
902             assert rwl.isWriteLockedByCurrentThread() || failEngineLock.isHeldByCurrentThread() : "Either the write lock must be held or the engine must be currently be failing itself";
903             try {
904                 this.versionMap.clear();
905                 logger.trace("close searcherManager");
906                 try {
907                     IOUtils.close(searcherManager);
908                 } catch (Throwable t) {
909                     logger.warn("Failed to close SearcherManager", t);
910                 }
911                 // no need to commit in this case!, we snapshot before we close the shard, so translog and all sync'ed
912                 logger.trace("rollback indexWriter");
913                 try {
914                     indexWriter.rollback();
915                 } catch (AlreadyClosedException e) {
916                     // ignore
917                 }
918                 logger.trace("rollback indexWriter done");
919             } catch (Throwable e) {
920                 logger.warn("failed to rollback writer on close", e);
921             } finally {
922                 store.decRef();
923                 this.mergeScheduler.removeListener(mergeSchedulerListener);
924                 this.mergeScheduler.removeFailureListener(mergeSchedulerFailureListener);
925                 logger.debug("engine closed [{}]", reason);
926             }
927         }
928     }

950     private IndexWriter createWriter() throws IOException {
951         try {
952             boolean create = !Lucene.indexExists(store.directory());
953             final IndexWriterConfig iwc = new IndexWriterConfig(engineConfig.getAnalyzer());
954             iwc.setCommitOnClose(false); // we by default don't commit on close
955             iwc.setOpenMode(create ? IndexWriterConfig.OpenMode.CREATE : IndexWriterConfig.OpenMode.APPEND);
956             iwc.setIndexDeletionPolicy(deletionPolicy);
957             iwc.setInfoStream(new LoggerInfoStream(logger));
958             iwc.setMergeScheduler(mergeScheduler.newMergeScheduler());
959             MergePolicy mergePolicy = mergePolicyProvider.getMergePolicy();
960             // Give us the opportunity to upgrade old segments while performing
961             // background merges
962             mergePolicy = new ElasticsearchMergePolicy(mergePolicy);
963             iwc.setMergePolicy(mergePolicy);
964             iwc.setSimilarity(engineConfig.getSimilarity());
965             iwc.setRAMBufferSizeMB(engineConfig.getIndexingBufferSize().mbFrac());
966             iwc.setMaxThreadStates(engineConfig.getIndexConcurrency());
967             iwc.setCodec(engineConfig.getCodec());
968             /* We set this timeout to a highish value to work around
969              * the default poll interval in the Lucene lock that is
970              * 1000ms by default. We might need to poll multiple times
971              * here but with 1s poll this is only executed twice at most
972              * in combination with the default writelock timeout*/
973             iwc.setWriteLockTimeout(5000);
974             iwc.setUseCompoundFile(this.engineConfig.isCompoundOnFlush());
975             // Warm-up hook for newly-merged segments. Warming up segments here is better since it will be performed at the end
976             // of the merge operation and won't slow down _refresh
977             iwc.setMergedSegmentWarmer(new IndexReaderWarmer() {
978                 @Override
979                 public void warm(LeafReader reader) throws IOException {
980                     try {
981                         assert isMergedSegment(reader);
982                         if (warmer != null) {
983                             final Engine.Searcher searcher = new Searcher("warmer", new IndexSearcher(reader));
984                             final IndicesWarmer.WarmerContext context = new IndicesWarmer.WarmerContext(shardId, searcher);
985                             warmer.warmNewReaders(context);
986                         }
987                     } catch (Throwable t) {
988                         // Don't fail a merge if the warm-up failed
989                         if (isClosed.get() == false) {
990                             logger.warn("Warm-up failed", t);
991                         }
992                         if (t instanceof Error) {
993                             // assertion/out-of-memory error, don't ignore those
994                             throw (Error) t;
995                         }
996                     }
997                 }
998             });
999             return new IndexWriter(store.directory(), iwc);
1000         } catch (LockObtainFailedException ex) {
1001             boolean isLocked = IndexWriter.isLocked(store.directory());
1002             logger.warn("Could not lock IndexWriter isLocked [{}]", ex, isLocked);
1003             throw ex;
1004         }
1005     }

1007     /** Extended SearcherFactory that warms the segments if needed when acquiring a new searcher */
1008     class SearchFactory extends EngineSearcherFactory {
1009 
1010         SearchFactory(EngineConfig engineConfig) {
1011             super(engineConfig);
1012         }
1013 
1014         @Override
1015         public IndexSearcher newSearcher(IndexReader reader) throws IOException {
1016             IndexSearcher searcher = new IndexSearcher(reader);
1017             searcher.setSimilarity(engineConfig.getSimilarity());
1018             if (warmer != null) {
1019                 // we need to pass a custom searcher that does not release anything on Engine.Search Release,
1020                 // we will release explicitly
1021                 IndexSearcher newSearcher = null;
1022                 boolean closeNewSearcher = false;
1023                 try {
1024                     if (searcherManager == null) {
1025                         // we are starting up - no writer active so we can't acquire a searcher.
1026                         newSearcher = searcher;
1027                     } else {
1028                         try (final Searcher currentSearcher = acquireSearcher("search_factory")) {
1029                             // figure out the newSearcher, with only the new readers that are relevant for us
1030                             List<IndexReader> readers = Lists.newArrayList();
1031                             for (LeafReaderContext newReaderContext : searcher.getIndexReader().leaves()) {
1032                                 if (isMergedSegment(newReaderContext.reader())) {
1033                                     // merged segments are already handled by IndexWriterConfig.setMergedSegmentWarmer
1034                                     continue;
1035                                 }
1036                                 boolean found = false;
1037                                 for (LeafReaderContext currentReaderContext : currentSearcher.reader().leaves()) {
1038                                     if (currentReaderContext.reader().getCoreCacheKey().equals(newReaderContext.reader().getCoreCacheKey())) {
1039                                         found = true;
1040                                         break;
1041                                     }
1042                                 }
1043                                 if (!found) {
1044                                     readers.add(newReaderContext.reader());
1045                                 }
1046                             }
1047                             if (!readers.isEmpty()) {
1048                                 // we don't want to close the inner readers, just increase ref on them
1049                                 newSearcher = new IndexSearcher(new MultiReader(readers.toArray(new IndexReader[readers.size()]), false));
1050                                 closeNewSearcher = true;
1051                             }
1052                         }
1053                     }
1054 
1055                     if (newSearcher != null) {
1056                         IndicesWarmer.WarmerContext context = new IndicesWarmer.WarmerContext(shardId, new Searcher("warmer", newSearcher));
1057                         warmer.warmNewReaders(context);
1058                     }
1059                     warmer.warmTopReader(new IndicesWarmer.WarmerContext(shardId, new Searcher("warmer", searcher)));
1060                 } catch (Throwable e) {
1061                     if (isClosed.get() == false) {
1062                         logger.warn("failed to prepare/warm", e);
1063                     }
1064                 } finally {
1065                     // no need to release the fullSearcher, nothing really is done...
1066                     if (newSearcher != null && closeNewSearcher) {
1067                         IOUtils.closeWhileHandlingException(newSearcher.getIndexReader()); // ignore
1068                     }
1069                 }
1070             }
1071             return searcher;
1072         }
1073     }

1107     class MergeSchedulerListener implements MergeSchedulerProvider.Listener {
1108         private final AtomicInteger numMergesInFlight = new AtomicInteger(0);
1109         private final AtomicBoolean isThrottling = new AtomicBoolean();
1110 
1111         @Override
1112         public synchronized void beforeMerge(OnGoingMerge merge) {
1113             int maxNumMerges = mergeScheduler.getMaxMerges();
1114             if (numMergesInFlight.incrementAndGet() > maxNumMerges) {
1115                 if (isThrottling.getAndSet(true) == false) {
1116                     logger.info("now throttling indexing: numMergesInFlight={}, maxNumMerges={}", numMergesInFlight, maxNumMerges);
1117                     indexingService.throttlingActivated();
1118                     activateThrottling();
1119                 }
1120             }
1121         }
1122 
1123         @Override
1124         public synchronized void afterMerge(OnGoingMerge merge) {
1125             int maxNumMerges = mergeScheduler.getMaxMerges();
1126             if (numMergesInFlight.decrementAndGet() < maxNumMerges) {
1127                 if (isThrottling.getAndSet(false)) {
1128                     logger.info("stop throttling indexing: numMergesInFlight={}, maxNumMerges={}", numMergesInFlight, maxNumMerges);
1129                     indexingService.throttlingDeactivated();
1130                     deactivateThrottling();
1131                 }
1132             }
1133         }
1134     }
