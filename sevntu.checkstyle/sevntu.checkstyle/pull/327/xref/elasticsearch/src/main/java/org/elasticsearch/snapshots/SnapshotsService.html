71  /**
72   * Service responsible for creating snapshots
73   * <p/>
74   * A typical snapshot creating process looks like this:
75   * <ul>
76   * <li>On the master node the {@link #createSnapshot(SnapshotRequest, CreateSnapshotListener)} is called and makes sure that no snapshots is currently running
77   * and registers the new snapshot in cluster state</li>
78   * <li>When cluster state is updated the {@link #beginSnapshot(ClusterState, SnapshotMetaData.Entry, boolean, CreateSnapshotListener)} method
79   * kicks in and initializes the snapshot in the repository and then populates list of shards that needs to be snapshotted in cluster state</li>
80   * <li>Each data node is watching for these shards and when new shards scheduled for snapshotting appear in the cluster state, data nodes
81   * start processing them through {@link #processIndexShardSnapshots(SnapshotMetaData)} method</li>
82   * <li>Once shard snapshot is created data node updates state of the shard in the cluster state using the {@link #updateIndexShardSnapshotStatus(UpdateIndexShardSnapshotStatusRequest)} method</li>
83   * <li>When last shard is completed master node in {@link #innerUpdateSnapshotState} method marks the snapshot as completed</li>
84   * <li>After cluster state is updated, the {@link #endSnapshot(SnapshotMetaData.Entry)} finalizes snapshot in the repository,
85   * notifies all {@link #snapshotCompletionListeners} that snapshot is completed, and finally calls {@link #removeSnapshotFromClusterState(SnapshotId, SnapshotInfo, Throwable)} to remove snapshot from cluster state</li>
86   * </ul>
87   */
88  public class SnapshotsService extends AbstractLifecycleComponent<SnapshotsService> implements ClusterStateListener {

129      *
130      * @param snapshotId snapshot id
131      * @return snapshot
132      * @throws SnapshotMissingException if snapshot is not found
133      */
134     public Snapshot snapshot(SnapshotId snapshotId) {
135         ImmutableList<SnapshotMetaData.Entry> entries = currentSnapshots(snapshotId.getRepository(), new String[] {snapshotId.getSnapshot()});
136         if (!entries.isEmpty()) {
137             return inProgressSnapshot(entries.iterator().next());
138         }
139         return repositoriesService.repository(snapshotId.getRepository()).readSnapshot(snapshotId);

144      *
145      * @param repositoryName repository name
146      * @return list of snapshots

148     public ImmutableList<Snapshot> snapshots(String repositoryName) {
149         Set<Snapshot> snapshotSet = newHashSet();
150         ImmutableList<SnapshotMetaData.Entry> entries = currentSnapshots(repositoryName, null);
151         for (SnapshotMetaData.Entry entry : entries) {
152             snapshotSet.add(inProgressSnapshot(entry));
153         }
154         Repository repository = repositoriesService.repository(repositoryName);
155         ImmutableList<SnapshotId> snapshotIds = repository.snapshots();
156         for (SnapshotId snapshotId : snapshotIds) {
157             snapshotSet.add(repository.readSnapshot(snapshotId));
158         }
159         ArrayList<Snapshot> snapshotList = newArrayList(snapshotSet);
160         CollectionUtil.timSort(snapshotList);
161         return ImmutableList.copyOf(snapshotList);
162     }

180     /**
181      * Initializes the snapshotting process.
182      * <p/>
183      * This method is used by clients to start snapshot. It makes sure that there is no snapshots are currently running and
184      * creates a snapshot record in cluster state metadata.
185      *
186      * @param request  snapshot request
187      * @param listener snapshot creation listener
188      */
189     public void createSnapshot(final SnapshotRequest request, final CreateSnapshotListener listener) {
190         final SnapshotId snapshotId = new SnapshotId(request.repository(), request.name());
191         clusterService.submitStateUpdateTask(request.cause(), new TimeoutClusterStateUpdateTask() {
192 
193             private SnapshotMetaData.Entry newSnapshot = null;
194 
195             @Override
196             public ClusterState execute(ClusterState currentState) {
197                 validate(request, currentState);
198 
199                 MetaData metaData = currentState.metaData();
200                 MetaData.Builder mdBuilder = MetaData.builder(currentState.metaData());
201                 SnapshotMetaData snapshots = metaData.custom(SnapshotMetaData.TYPE);
202                 if (snapshots == null || snapshots.entries().isEmpty()) {
203                     // Store newSnapshot here to be processed in clusterStateProcessed
204                     ImmutableList<String> indices = ImmutableList.copyOf(metaData.concreteIndices(request.indicesOptions(), request.indices()));
205                     logger.trace("[{}][{}] creating snapshot for indices [{}]", request.repository(), request.name(), indices);
206                     newSnapshot = new SnapshotMetaData.Entry(snapshotId, request.includeGlobalState(), State.INIT, indices, System.currentTimeMillis(), null);
207                     snapshots = new SnapshotMetaData(newSnapshot);
208                 } else {
209                     // TODO: What should we do if a snapshot is already running?
210                     throw new ConcurrentSnapshotExecutionException(snapshotId, "a snapshot is already running");
211                 }
212                 mdBuilder.putCustom(SnapshotMetaData.TYPE, snapshots);
213                 return ClusterState.builder(currentState).metaData(mdBuilder).build();
214             }
215 
216             @Override
217             public void onFailure(String source, Throwable t) {
218                 logger.warn("[{}][{}] failed to create snapshot", t, request.repository(), request.name());
219                 newSnapshot = null;
220                 listener.onFailure(t);
221             }
222 
223             @Override
224             public void clusterStateProcessed(String source, ClusterState oldState, final ClusterState newState) {
225                 if (newSnapshot != null) {
226                     threadPool.executor(ThreadPool.Names.SNAPSHOT).execute(new Runnable() {
227                         @Override
228                         public void run() {
229                             beginSnapshot(newState, newSnapshot, request.partial, listener);
230                         }
231                     });
232                 }
233             }
234 
235             @Override
236             public TimeValue timeout() {
237                 return request.masterNodeTimeout();
238             }
239 
240         });
241     }

244      * Validates snapshot request
245      *
246      * @param request snapshot request
247      * @param state   current cluster state
248      * @throws org.elasticsearch.ElasticsearchException
249      */
250     private void validate(SnapshotRequest request, ClusterState state) throws ElasticsearchException {
251         RepositoriesMetaData repositoriesMetaData = state.getMetaData().custom(RepositoriesMetaData.TYPE);
252         if (repositoriesMetaData == null || repositoriesMetaData.repository(request.repository()) == null) {
253             throw new RepositoryMissingException(request.repository());
254         }
255         if (!Strings.hasLength(request.name())) {
256             throw new InvalidSnapshotNameException(new SnapshotId(request.repository(), request.name()), "cannot be empty");
257         }
258         if (request.name().contains(" ")) {
259             throw new InvalidSnapshotNameException(new SnapshotId(request.repository(), request.name()), "must not contain whitespace");
260         }
261         if (request.name().contains(",")) {
262             throw new InvalidSnapshotNameException(new SnapshotId(request.repository(), request.name()), "must not contain ','");
263         }
264         if (request.name().contains("#")) {
265             throw new InvalidSnapshotNameException(new SnapshotId(request.repository(), request.name()), "must not contain '#'");
266         }
267         if (request.name().charAt(0) == '_') {
268             throw new InvalidSnapshotNameException(new SnapshotId(request.repository(), request.name()), "must not start with '_'");
269         }
270         if (!request.name().toLowerCase(Locale.ROOT).equals(request.name())) {
271             throw new InvalidSnapshotNameException(new SnapshotId(request.repository(), request.name()), "must be lowercase");
272         }
273         if (!Strings.validFileName(request.name())) {
274             throw new InvalidSnapshotNameException(new SnapshotId(request.repository(), request.name()), "must not contain the following characters " + Strings.INVALID_FILENAME_CHARS);
275         }
276     }

278     /**
279      * Starts snapshot.
280      * <p/>
281      * Creates snapshot in repository and updates snapshot metadata record with list of shards that needs to be processed.
282      *
283      * @param clusterState               cluster state
284      * @param snapshot                   snapshot meta data
285      * @param partial                    allow partial snapshots
286      * @param userCreateSnapshotListener listener
287      */
288     private void beginSnapshot(ClusterState clusterState, final SnapshotMetaData.Entry snapshot, final boolean partial, final CreateSnapshotListener userCreateSnapshotListener) {
289         boolean snapshotCreated = false;
290         try {
291             Repository repository = repositoriesService.repository(snapshot.snapshotId().getRepository());
292 
293             MetaData metaData = clusterState.metaData();
294             if (!snapshot.includeGlobalState()) {
295                 // Remove global state from the cluster state
296                 MetaData.Builder builder = MetaData.builder();
297                 for (String index : snapshot.indices()) {
298                     builder.put(metaData.index(index), false);
299                 }
300                 metaData = builder.build();
301             }
302 
303             repository.initializeSnapshot(snapshot.snapshotId(), snapshot.indices(), metaData);
304             snapshotCreated = true;
305             if (snapshot.indices().isEmpty()) {
306                 // No indices in this snapshot - we are done
307                 userCreateSnapshotListener.onResponse();
308                 endSnapshot(snapshot);
309                 return;
310             }
311             clusterService.submitStateUpdateTask("update_snapshot [" + snapshot + "]", new ProcessedClusterStateUpdateTask() {
312                 boolean accepted = false;
313                 SnapshotMetaData.Entry updatedSnapshot;
314                 String failure = null;
315 
316                 @Override
317                 public ClusterState execute(ClusterState currentState) {
318                     MetaData metaData = currentState.metaData();
319                     MetaData.Builder mdBuilder = MetaData.builder(currentState.metaData());
320                     SnapshotMetaData snapshots = metaData.custom(SnapshotMetaData.TYPE);
321                     ImmutableList.Builder<SnapshotMetaData.Entry> entries = ImmutableList.builder();
322                     for (SnapshotMetaData.Entry entry : snapshots.entries()) {
323                         if (entry.snapshotId().equals(snapshot.snapshotId())) {
324                             // Replace the snapshot that was just created
325                             ImmutableMap<ShardId, SnapshotMetaData.ShardSnapshotStatus> shards = shards(currentState, entry.indices());
326                             if (!partial) {
327                                 Set<String> indicesWithMissingShards = indicesWithMissingShards(shards);
328                                 if (indicesWithMissingShards != null) {
329                                     updatedSnapshot = new SnapshotMetaData.Entry(entry, State.FAILED, shards);
330                                     entries.add(updatedSnapshot);
331                                     failure = "Indices don't have primary shards +[" + indicesWithMissingShards + "]";
332                                     continue;
333                                 }
334                             }
335                             updatedSnapshot = new SnapshotMetaData.Entry(entry, State.STARTED, shards);
336                             entries.add(updatedSnapshot);
337                             if (!completed(shards.values())) {
338                                 accepted = true;
339                             }
340                         } else {
341                             entries.add(entry);
342                         }
343                     }
344                     mdBuilder.putCustom(SnapshotMetaData.TYPE, new SnapshotMetaData(entries.build()));
345                     return ClusterState.builder(currentState).metaData(mdBuilder).build();
346                 }
347 
348                 @Override
349                 public void onFailure(String source, Throwable t) {
350                     logger.warn("[{}] failed to create snapshot", t, snapshot.snapshotId());
351                     removeSnapshotFromClusterState(snapshot.snapshotId(), null, t);
352                     try {
353                         repositoriesService.repository(snapshot.snapshotId().getRepository()).finalizeSnapshot(
354                                 snapshot.snapshotId(), snapshot.indices(), snapshot.startTime(), ExceptionsHelper.detailedMessage(t), 0, ImmutableList.<SnapshotShardFailure>of());
355                     } catch (Throwable t2) {
356                         logger.warn("[{}] failed to close snapshot in repository", snapshot.snapshotId());
357                     }
358                     userCreateSnapshotListener.onFailure(t);
359                 }
360 
361                 @Override
362                 public void clusterStateProcessed(String source, ClusterState oldState, ClusterState newState) {
363                     // The userCreateSnapshotListener.onResponse() notifies caller that the snapshot was accepted
364                     // for processing. If client wants to wait for the snapshot completion, it can register snapshot
365                     // completion listener in this method. For the snapshot completion to work properly, the snapshot
366                     // should still exist when listener is registered.
367                     userCreateSnapshotListener.onResponse();
368 
369                     // Now that snapshot completion listener is registered we can end the snapshot if needed
370                     // We should end snapshot only if 1) we didn't accept it for processing (which happens when there
371                     // is nothing to do) and 2) there was a snapshot in metadata that we should end. Otherwise we should
372                     // go ahead and continue working on this snapshot rather then end here.
373                     if (!accepted && updatedSnapshot != null) {
374                         endSnapshot(updatedSnapshot, failure);
375                     }
376                 }
377             });
378         } catch (Throwable t) {
379             logger.warn("failed to create snapshot [{}]", t, snapshot.snapshotId());
380             removeSnapshotFromClusterState(snapshot.snapshotId(), null, t);
381             if (snapshotCreated) {
382                 try {
383                     repositoriesService.repository(snapshot.snapshotId().getRepository()).finalizeSnapshot(snapshot.snapshotId(), snapshot.indices(), snapshot.startTime(),
384                             ExceptionsHelper.detailedMessage(t), 0, ImmutableList.<SnapshotShardFailure>of());
385                 } catch (Throwable t2) {
386                     logger.warn("[{}] failed to close snapshot in repository", snapshot.snapshotId());
387                 }
388             }
389             userCreateSnapshotListener.onFailure(t);
390         }
391     }

397     /**
398      * Returns status of the currently running snapshots
399      * <p>
400      * This method is executed on master node
401      * </p>
402      *
403      * @param repository repository id
404      * @param snapshots  optional list of snapshots that will be used as a filter
405      * @return list of metadata for currently running snapshots
406      */
407     public ImmutableList<SnapshotMetaData.Entry> currentSnapshots(String repository, String[] snapshots) {
408         MetaData metaData = clusterService.state().metaData();
409         SnapshotMetaData snapshotMetaData = metaData.custom(SnapshotMetaData.TYPE);
410         if (snapshotMetaData == null || snapshotMetaData.entries().isEmpty()) {
411             return ImmutableList.of();
412         }
413         if ("_all".equals(repository)) {
414             return snapshotMetaData.entries();
415         }
416         if (snapshotMetaData.entries().size() == 1) {
417             // Most likely scenario - one snapshot is currently running
418             // Check this snapshot against the query
419             SnapshotMetaData.Entry entry = snapshotMetaData.entries().get(0);
420             if (!entry.snapshotId().getRepository().equals(repository)) {
421                 return ImmutableList.of();
422             }
423             if (snapshots != null && snapshots.length > 0) {
424                 for (String snapshot : snapshots) {
425                     if (entry.snapshotId().getSnapshot().equals(snapshot)) {
426                         return snapshotMetaData.entries();
427                     }
428                 }
429                 return ImmutableList.of();
430             } else {
431                 return snapshotMetaData.entries();
432             }
433         }
434         ImmutableList.Builder<SnapshotMetaData.Entry> builder = ImmutableList.builder();
435         for (SnapshotMetaData.Entry entry : snapshotMetaData.entries()) {
436             if (!entry.snapshotId().getRepository().equals(repository)) {
437                 continue;
438             }
439             if (snapshots != null && snapshots.length > 0) {
440                 for (String snapshot : snapshots) {
441                     if (entry.snapshotId().getSnapshot().equals(snapshot)) {
442                         builder.add(entry);
443                         break;
444                     }
445                 }
446             } else {
447                 builder.add(entry);
448             }
449         }
450         return builder.build();
451     }

471     /**
472      * Returns status of shards  currently finished snapshots
473      * <p>
474      * This method is executed on master node and it's complimentary to the {@link #currentSnapshotShards(SnapshotId)} because it
475      * returns simliar information but for already finished snapshots.
476      * </p>
477      *
478      * @param snapshotId snapshot id
479      * @return map of shard id to snapshot status
480      */
481     public ImmutableMap<ShardId, IndexShardSnapshotStatus> snapshotShards(SnapshotId snapshotId) throws IOException {
482         ImmutableMap.Builder<ShardId, IndexShardSnapshotStatus> shardStatusBuilder = ImmutableMap.builder();
483         Repository repository = repositoriesService.repository(snapshotId.getRepository());
484         IndexShardRepository indexShardRepository = repositoriesService.indexShardRepository(snapshotId.getRepository());
485         Snapshot snapshot = repository.readSnapshot(snapshotId);
486         MetaData metaData = repository.readSnapshotMetaData(snapshotId, snapshot.indices());
487         for (String index : snapshot.indices()) {
488             IndexMetaData indexMetaData = metaData.indices().get(index);
489             if (indexMetaData != null) {
490                 int numberOfShards = indexMetaData.getNumberOfShards();
491                 for (int i = 0; i < numberOfShards; i++) {
492                     ShardId shardId = new ShardId(index, i);
493                     SnapshotShardFailure shardFailure = findShardFailure(snapshot.shardFailures(), shardId);
494                     if (shardFailure != null) {
495                         IndexShardSnapshotStatus shardSnapshotStatus = new IndexShardSnapshotStatus();
496                         shardSnapshotStatus.updateStage(IndexShardSnapshotStatus.Stage.FAILURE);
497                         shardSnapshotStatus.failure(shardFailure.reason());
498                         shardStatusBuilder.put(shardId, shardSnapshotStatus);
499                     } else {
500                         IndexShardSnapshotStatus shardSnapshotStatus = indexShardRepository.snapshotStatus(snapshotId, shardId);
501                         shardStatusBuilder.put(shardId, shardSnapshotStatus);
502                     }
503                 }
504             }
505         }
506         return shardStatusBuilder.build();
507     }

510     private SnapshotShardFailure findShardFailure(ImmutableList<SnapshotShardFailure> shardFailures, ShardId shardId) {
511         for (SnapshotShardFailure shardFailure : shardFailures) {
512             if (shardId.getIndex().equals(shardFailure.index()) && shardId.getId() == shardFailure.shardId()) {
513                 return shardFailure;
514             }
515         }
516         return null;
517     }

519     @Override
520     public void clusterChanged(ClusterChangedEvent event) {
521         try {
522             if (event.localNodeMaster()) {
523                 if (event.nodesRemoved()) {
524                     processSnapshotsOnRemovedNodes(event);
525                 }
526                 if (event.routingTableChanged()) {
527                     processStartedShards(event);
528                 }
529             }
530             SnapshotMetaData prev = event.previousState().metaData().custom(SnapshotMetaData.TYPE);
531             SnapshotMetaData curr = event.state().metaData().custom(SnapshotMetaData.TYPE);
532 
533             if (prev == null) {
534                 if (curr != null) {
535                     processIndexShardSnapshots(curr);
536                 }
537             } else {
538                 if (!prev.equals(curr)) {
539                     processIndexShardSnapshots(curr);
540                 }
541             }
542         } catch (Throwable t) {
543             logger.warn("Failed to update snapshot state ", t);
544         }
545     }

547     /**
548      * Cleans up shard snapshots that were running on removed nodes
549      *
550      * @param event cluster changed event
551      */
552     private void processSnapshotsOnRemovedNodes(ClusterChangedEvent event) {
553         if (removedNodesCleanupNeeded(event)) {
554             // Check if we just became the master
555             final boolean newMaster = !event.previousState().nodes().localNodeMaster();
556             clusterService.submitStateUpdateTask("update snapshot state after node removal", new ClusterStateUpdateTask() {
557                 @Override
558                 public ClusterState execute(ClusterState currentState) throws Exception {
559                     DiscoveryNodes nodes = currentState.nodes();
560                     MetaData metaData = currentState.metaData();
561                     MetaData.Builder mdBuilder = MetaData.builder(currentState.metaData());
562                     SnapshotMetaData snapshots = metaData.custom(SnapshotMetaData.TYPE);
563                     if (snapshots == null) {
564                         return currentState;
565                     }
566                     boolean changed = false;
567                     ArrayList<SnapshotMetaData.Entry> entries = newArrayList();
568                     for (final SnapshotMetaData.Entry snapshot : snapshots.entries()) {
569                         SnapshotMetaData.Entry updatedSnapshot = snapshot;
570                         boolean snapshotChanged = false;
571                         if (snapshot.state() == State.STARTED || snapshot.state() == State.ABORTED) {
572                             ImmutableMap.Builder<ShardId, ShardSnapshotStatus> shards = ImmutableMap.builder();
573                             for (ImmutableMap.Entry<ShardId, ShardSnapshotStatus> shardEntry : snapshot.shards().entrySet()) {
574                                 ShardSnapshotStatus shardStatus = shardEntry.getValue();
575                                 if (!shardStatus.state().completed() && shardStatus.nodeId() != null) {
576                                     if (nodes.nodeExists(shardStatus.nodeId())) {
577                                         shards.put(shardEntry);
578                                     } else {
579                                         // TODO: Restart snapshot on another node?
580                                         snapshotChanged = true;
581                                         logger.warn("failing snapshot of shard [{}] on closed node [{}]", shardEntry.getKey(), shardStatus.nodeId());
582                                         shards.put(shardEntry.getKey(), new ShardSnapshotStatus(shardStatus.nodeId(), State.FAILED, "node shutdown"));
583                                     }
584                                 }
585                             }
586                             if (snapshotChanged) {
587                                 changed = true;
588                                 ImmutableMap<ShardId, ShardSnapshotStatus> shardsMap = shards.build();
589                                 if (!snapshot.state().completed() && completed(shardsMap.values())) {
590                                     updatedSnapshot = new SnapshotMetaData.Entry(snapshot, State.SUCCESS, shardsMap);
591                                     endSnapshot(updatedSnapshot);
592                                 } else {
593                                     updatedSnapshot = new SnapshotMetaData.Entry(snapshot, snapshot.state(), shardsMap);
594                                 }
595                             }
596                             entries.add(updatedSnapshot);
597                         } else if (snapshot.state() == State.INIT && newMaster) {
598                             // Clean up the snapshot that failed to start from the old master
599                             deleteSnapshot(snapshot.snapshotId(), new DeleteSnapshotListener() {
600                                 @Override
601                                 public void onResponse() {
602                                     logger.debug("cleaned up abandoned snapshot {} in INIT state", snapshot.snapshotId());
603                                 }
604 
605                                 @Override
606                                 public void onFailure(Throwable t) {
607                                     logger.warn("failed to clean up abandoned snapshot {} in INIT state", snapshot.snapshotId());
608                                 }
609                             });
610                         } else if (snapshot.state() == State.SUCCESS && newMaster) {
611                             // Finalize the snapshot
612                             endSnapshot(snapshot);
613                         }
614                     }
615                     if (changed) {
616                         snapshots = new SnapshotMetaData(entries.toArray(new SnapshotMetaData.Entry[entries.size()]));
617                         mdBuilder.putCustom(SnapshotMetaData.TYPE, snapshots);
618                         return ClusterState.builder(currentState).metaData(mdBuilder).build();
619                     }
620                     return currentState;
621                 }
622 
623                 @Override
624                 public void onFailure(String source, Throwable t) {
625                     logger.warn("failed to update snapshot state after node removal");
626                 }
627             });
628         }

676     private ImmutableMap<ShardId, ShardSnapshotStatus> processWaitingShards(ImmutableMap<ShardId, ShardSnapshotStatus> snapshotShards, RoutingTable routingTable) {
677         boolean snapshotChanged = false;
678         ImmutableMap.Builder<ShardId, ShardSnapshotStatus> shards = ImmutableMap.builder();
679         for (ImmutableMap.Entry<ShardId, ShardSnapshotStatus> shardEntry : snapshotShards.entrySet()) {
680             ShardSnapshotStatus shardStatus = shardEntry.getValue();
681             if (shardStatus.state() == State.WAITING) {
682                 ShardId shardId = shardEntry.getKey();
683                 IndexRoutingTable indexShardRoutingTable = routingTable.index(shardId.getIndex());
684                 if (indexShardRoutingTable != null) {
685                     IndexShardRoutingTable shardRouting = indexShardRoutingTable.shard(shardId.id());
686                     if (shardRouting != null && shardRouting.primaryShard() != null) {
687                         if (shardRouting.primaryShard().started()) {
688                             // Shard that we were waiting for has started on a node, let's process it
689                             snapshotChanged = true;
690                             logger.trace("starting shard that we were waiting for [{}] on node [{}]", shardEntry.getKey(), shardStatus.nodeId());
691                             shards.put(shardEntry.getKey(), new ShardSnapshotStatus(shardRouting.primaryShard().currentNodeId()));
692                             continue;
693                         } else if (shardRouting.primaryShard().initializing() || shardRouting.primaryShard().relocating()) {
694                             // Shard that we were waiting for hasn't started yet or still relocating - will continue to wait
695                             shards.put(shardEntry);
696                             continue;
697                         }
698                     }
699                 }
700                 // Shard that we were waiting for went into unassigned state or disappeared - giving up
701                 snapshotChanged = true;
702                 logger.warn("failing snapshot of shard [{}] on unassigned shard [{}]", shardEntry.getKey(), shardStatus.nodeId());
703                 shards.put(shardEntry.getKey(), new ShardSnapshotStatus(shardStatus.nodeId(), State.FAILED, "shard is unassigned"));
704             } else {
705                 shards.put(shardEntry);
706             }
707         }
708         if (snapshotChanged) {
709             return shards.build();
710         } else {
711             return null;
712         }
713     }

754                         return true;
755                     }
756                 }
757             }
758         }

762     /**
763      * Checks if any new shards should be snapshotted on this node
764      *
765      * @param snapshotMetaData snapshot metadata to be processed
766      */
767     private void processIndexShardSnapshots(SnapshotMetaData snapshotMetaData) {
768         Map<SnapshotId, SnapshotShards> survivors = newHashMap();
769         // First, remove snapshots that are no longer there
770         for (Map.Entry<SnapshotId, SnapshotShards> entry : shardSnapshots.entrySet()) {
771             if (snapshotMetaData != null && snapshotMetaData.snapshot(entry.getKey()) != null) {
772                 survivors.put(entry.getKey(), entry.getValue());
773             }
774         }
775 
776         // For now we will be mostly dealing with a single snapshot at a time but might have multiple simultaneously running
777         // snapshots in the future
778         Map<SnapshotId, Map<ShardId, IndexShardSnapshotStatus>> newSnapshots = newHashMap();
779         // Now go through all snapshots and update existing or create missing
780         final String localNodeId = clusterService.localNode().id();
781         if (snapshotMetaData != null) {
782             for (SnapshotMetaData.Entry entry : snapshotMetaData.entries()) {
783                 if (entry.state() == State.STARTED) {
784                     Map<ShardId, IndexShardSnapshotStatus> startedShards = newHashMap();
785                     SnapshotShards snapshotShards = shardSnapshots.get(entry.snapshotId());
786                     for (Map.Entry<ShardId, SnapshotMetaData.ShardSnapshotStatus> shard : entry.shards().entrySet()) {
787                         // Add all new shards to start processing on
788                         if (localNodeId.equals(shard.getValue().nodeId())) {
789                             if (shard.getValue().state() == State.INIT && (snapshotShards == null || !snapshotShards.shards.containsKey(shard.getKey()))) {
790                                 logger.trace("[{}] - Adding shard to the queue", shard.getKey());
791                                 startedShards.put(shard.getKey(), new IndexShardSnapshotStatus());
792                             }
793                         }
794                     }
795                     if (!startedShards.isEmpty()) {
796                         newSnapshots.put(entry.snapshotId(), startedShards);
797                         if (snapshotShards != null) {
798                             // We already saw this snapshot but we need to add more started shards
799                             ImmutableMap.Builder<ShardId, IndexShardSnapshotStatus> shards = ImmutableMap.builder();
800                             // Put all shards that were already running on this node
801                             shards.putAll(snapshotShards.shards);
802                             // Put all newly started shards
803                             shards.putAll(startedShards);
804                             survivors.put(entry.snapshotId(), new SnapshotShards(shards.build()));
805                         } else {
806                             // Brand new snapshot that we haven't seen before
807                             survivors.put(entry.snapshotId(), new SnapshotShards(ImmutableMap.copyOf(startedShards)));
808                         }
809                     }
810                 } else if (entry.state() == State.ABORTED) {
811                     // Abort all running shards for this snapshot
812                     SnapshotShards snapshotShards = shardSnapshots.get(entry.snapshotId());
813                     if (snapshotShards != null) {
814                         for (Map.Entry<ShardId, SnapshotMetaData.ShardSnapshotStatus> shard : entry.shards().entrySet()) {
815                             IndexShardSnapshotStatus snapshotStatus = snapshotShards.shards.get(shard.getKey());
816                             if (snapshotStatus != null) {
817                                 snapshotStatus.abort();
818                             }
819                         }
820                     }
821                 }
822             }
823         }
824 
825         // Update the list of snapshots that we saw and tried to started
826         // If startup of these shards fails later, we don't want to try starting these shards again
827         shutdownLock.lock();
828         try {
829             shardSnapshots = ImmutableMap.copyOf(survivors);
830             if (shardSnapshots.isEmpty()) {
831                 // Notify all waiting threads that no more snapshots
832                 shutdownCondition.signalAll();
833             }
834         } finally {
835             shutdownLock.unlock();
836         }
837 
838         // We have new shards to starts
839         if (!newSnapshots.isEmpty()) {
840             for (final Map.Entry<SnapshotId, Map<ShardId, IndexShardSnapshotStatus>> entry : newSnapshots.entrySet()) {
841                 for (final Map.Entry<ShardId, IndexShardSnapshotStatus> shardEntry : entry.getValue().entrySet()) {
842                     try {
843                         final IndexShardSnapshotAndRestoreService shardSnapshotService = indicesService.indexServiceSafe(shardEntry.getKey().getIndex()).shardInjectorSafe(shardEntry.getKey().id())
844                                 .getInstance(IndexShardSnapshotAndRestoreService.class);
845                         threadPool.executor(ThreadPool.Names.SNAPSHOT).execute(new Runnable() {
846                             @Override
847                             public void run() {
848                                 try {
849                                     shardSnapshotService.snapshot(entry.getKey(), shardEntry.getValue());
850                                     updateIndexShardSnapshotStatus(new UpdateIndexShardSnapshotStatusRequest(entry.getKey(), shardEntry.getKey(), new ShardSnapshotStatus(localNodeId, SnapshotMetaData.State.SUCCESS)));
851                                 } catch (Throwable t) {
852                                     logger.warn("[{}] [{}] failed to create snapshot", t, shardEntry.getKey(), entry.getKey());
853                                     updateIndexShardSnapshotStatus(new UpdateIndexShardSnapshotStatusRequest(entry.getKey(), shardEntry.getKey(), new ShardSnapshotStatus(localNodeId, SnapshotMetaData.State.FAILED, ExceptionsHelper.detailedMessage(t))));
854                                 }
855                             }
856                         });
857                     } catch (Throwable t) {
858                         updateIndexShardSnapshotStatus(new UpdateIndexShardSnapshotStatusRequest(entry.getKey(), shardEntry.getKey(), new ShardSnapshotStatus(localNodeId, SnapshotMetaData.State.FAILED, ExceptionsHelper.detailedMessage(t))));
859                     }
860                 }
861             }
862         }
863     }

985      *
986      * @param entry   snapshot
987      * @param failure failure reason or null if snapshot was successful
988      */
989     private void endSnapshot(final SnapshotMetaData.Entry entry, final String failure) {
990         threadPool.executor(ThreadPool.Names.SNAPSHOT).execute(new Runnable() {
991             @Override
992             public void run() {

1074      * Deletes snapshot from repository.
1075      * <p/>
1076      * If the snapshot is still running cancels the snapshot first and then deletes it from the repository.
1077      *
1078      * @param snapshotId snapshot id
1079      * @param listener   listener
1080      */
1081     public void deleteSnapshot(final SnapshotId snapshotId, final DeleteSnapshotListener listener) {
1082         clusterService.submitStateUpdateTask("delete snapshot", new ProcessedClusterStateUpdateTask() {
1083 
1084             boolean waitForSnapshot = false;
1085 
1086             @Override
1087             public ClusterState execute(ClusterState currentState) throws Exception {
1088                 MetaData metaData = currentState.metaData();
1089                 MetaData.Builder mdBuilder = MetaData.builder(currentState.metaData());
1090                 SnapshotMetaData snapshots = metaData.custom(SnapshotMetaData.TYPE);
1091                 if (snapshots == null) {
1092                     // No snapshots running - we can continue
1093                     return currentState;
1094                 }
1095                 SnapshotMetaData.Entry snapshot = snapshots.snapshot(snapshotId);
1096                 if (snapshot == null) {
1097                     // This snapshot is not running - continue
1098                     if (!snapshots.entries().isEmpty()) {
1099                         // However other snapshots are running - cannot continue
1100                         throw new ConcurrentSnapshotExecutionException(snapshotId, "another snapshot is currently running cannot delete");
1101                     }
1102                     return currentState;
1103                 } else {
1104                     // This snapshot is currently running - stopping shards first
1105                     waitForSnapshot = true;
1106                     ImmutableMap<ShardId, ShardSnapshotStatus> shards;
1107                     if (snapshot.state() == State.STARTED && snapshot.shards() != null) {
1108                         // snapshot is currently running - stop started shards
1109                         ImmutableMap.Builder<ShardId, ShardSnapshotStatus> shardsBuilder = ImmutableMap.builder();
1110                         for (ImmutableMap.Entry<ShardId, ShardSnapshotStatus> shardEntry : snapshot.shards().entrySet()) {
1111                             ShardSnapshotStatus status = shardEntry.getValue();
1112                             if (!status.state().completed()) {
1113                                 shardsBuilder.put(shardEntry.getKey(), new ShardSnapshotStatus(status.nodeId(), State.ABORTED));
1114                             } else {
1115                                 shardsBuilder.put(shardEntry.getKey(), status);
1116                             }
1117                         }
1118                         shards = shardsBuilder.build();
1119                     } else if (snapshot.state() == State.INIT) {
1120                         // snapshot hasn't started yet - end it
1121                         shards = snapshot.shards();
1122                         endSnapshot(snapshot);
1123                     } else {
1124                         // snapshot is being finalized - wait for it
1125                         logger.trace("trying to delete completed snapshot - save to delete");
1126                         return currentState;
1127                     }
1128                     SnapshotMetaData.Entry newSnapshot = new SnapshotMetaData.Entry(snapshot, State.ABORTED, shards);
1129                     snapshots = new SnapshotMetaData(newSnapshot);
1130                     mdBuilder.putCustom(SnapshotMetaData.TYPE, snapshots);
1131                     return ClusterState.builder(currentState).metaData(mdBuilder).build();
1132                 }
1133             }
1134 
1135             @Override
1136             public void onFailure(String source, Throwable t) {
1137                 listener.onFailure(t);
1138             }
1139 
1140             @Override
1141             public void clusterStateProcessed(String source, ClusterState oldState, ClusterState newState) {
1142                 if (waitForSnapshot) {
1143                     logger.trace("adding snapshot completion listener to wait for deleted snapshot to finish");
1144                     addListener(new SnapshotCompletionListener() {
1145                         @Override
1146                         public void onSnapshotCompletion(SnapshotId completedSnapshotId, SnapshotInfo snapshot) {

1212     /**
1213      * Calculates the list of shards that should be included into the current snapshot
1214      *
1215      * @param clusterState cluster state
1216      * @param indices      list of indices to be snapshotted
1217      * @return list of shard to be included into current snapshot
1218      */
1219     private ImmutableMap<ShardId, SnapshotMetaData.ShardSnapshotStatus> shards(ClusterState clusterState, ImmutableList<String> indices) {
1220         ImmutableMap.Builder<ShardId, SnapshotMetaData.ShardSnapshotStatus> builder = ImmutableMap.builder();
1221         MetaData metaData = clusterState.metaData();
1222         for (String index : indices) {
1223             IndexMetaData indexMetaData = metaData.index(index);
1224             if (indexMetaData == null) {
1225                 // The index was deleted before we managed to start the snapshot - mark it as missing.
1226                 builder.put(new ShardId(index, 0), new SnapshotMetaData.ShardSnapshotStatus(null, State.MISSING, "missing index"));
1227             } else {
1228                 IndexRoutingTable indexRoutingTable = clusterState.getRoutingTable().index(index);
1229                 for (int i = 0; i < indexMetaData.numberOfShards(); i++) {
1230                     ShardId shardId = new ShardId(index, i);
1231                     if (indexRoutingTable != null) {
1232                         ShardRouting primary = indexRoutingTable.shard(i).primaryShard();
1233                         if (primary == null || !primary.assignedToNode()) {
1234                             builder.put(shardId, new SnapshotMetaData.ShardSnapshotStatus(null, State.MISSING, "primary shard is not allocated"));
1235                         } else if (primary.relocating() || primary.initializing()) {
1236                             // The WAITING state was introduced in V1.2.0 - don't use it if there are nodes with older version in the cluster
1237                             builder.put(shardId, new SnapshotMetaData.ShardSnapshotStatus(primary.currentNodeId(), State.WAITING));
1238                         } else if (!primary.started()) {
1239                             builder.put(shardId, new SnapshotMetaData.ShardSnapshotStatus(primary.currentNodeId(), State.MISSING, "primary shard hasn't been started yet"));
1240                         } else {
1241                             builder.put(shardId, new SnapshotMetaData.ShardSnapshotStatus(primary.currentNodeId()));
1242                         }
1243                     } else {
1244                         builder.put(shardId, new SnapshotMetaData.ShardSnapshotStatus(null, State.MISSING, "missing routing table"));
1245                     }
1246                 }
1247             }
1248         }
1249 
1250         return builder.build();
1251     }
