46  public class HistogramAggregator extends BucketsAggregator {
47  
48      private final ValuesSource.Numeric valuesSource;
49      private final @Nullable ValueFormatter formatter;
50      private final Rounding rounding;
51      private final InternalOrder order;
52      private final boolean keyed;
53  
54      private final long minDocCount;
55      private final ExtendedBounds extendedBounds;
56      private final InternalHistogram.Factory histogramFactory;
57  
58      private final LongHash bucketOrds;
59      private SortedNumericDocValues values;
60  
61      public HistogramAggregator(String name, AggregatorFactories factories, Rounding rounding, InternalOrder order,
62                                 boolean keyed, long minDocCount, @Nullable ExtendedBounds extendedBounds,
63                                 @Nullable ValuesSource.Numeric valuesSource, @Nullable ValueFormatter formatter,
64                                 InternalHistogram.Factory<?> histogramFactory,
65                                 AggregationContext aggregationContext, Aggregator parent, Map<String, Object> metaData) throws IOException {
66  
67          super(name, factories, aggregationContext, parent, metaData);
68          this.rounding = rounding;
69          this.order = order;
70          this.keyed = keyed;
71          this.minDocCount = minDocCount;
72          this.extendedBounds = extendedBounds;
73          this.valuesSource = valuesSource;
74          this.formatter = formatter;
75          this.histogramFactory = histogramFactory;
76  
77          bucketOrds = new LongHash(1, aggregationContext.bigArrays());
78      }
79  
80      @Override
81      public boolean needsScores() {
82          return (valuesSource != null && valuesSource.needsScores()) || super.needsScores();
83      }
84  
85      @Override
86      public LeafBucketCollector getLeafCollector(LeafReaderContext ctx,
87              final LeafBucketCollector sub) throws IOException {
88          if (valuesSource == null) {
89              return LeafBucketCollector.NO_OP_COLLECTOR;
90          }
91          final SortedNumericDocValues values = valuesSource.longValues(ctx);
92          return new LeafBucketCollectorBase(sub, values) {
93              @Override
94              public void collect(int doc, long bucket) throws IOException {
95                  assert bucket == 0;
96                  values.setDocument(doc);
97                  final int valuesCount = values.count();
98  
99                  long previousKey = Long.MIN_VALUE;
100                 for (int i = 0; i < valuesCount; ++i) {
101                     long value = values.valueAt(i);
102                     long key = rounding.roundKey(value);
103                     assert key >= previousKey;
104                     if (key == previousKey) {
105                         continue;
106                     }
107                     long bucketOrd = bucketOrds.add(key);
108                     if (bucketOrd < 0) { // already seen
109                         bucketOrd = -1 - bucketOrd;
110                         collectExistingBucket(sub, doc, bucketOrd);
111                     } else {
112                         collectBucket(sub, doc, bucketOrd);
113                     }
114                     previousKey = key;
115                 }
116             }
117         };
118     }
119 
120     @Override
121     public InternalAggregation buildAggregation(long owningBucketOrdinal) throws IOException {
122         assert owningBucketOrdinal == 0;
123         List<InternalHistogram.Bucket> buckets = new ArrayList<>((int) bucketOrds.size());
124         for (long i = 0; i < bucketOrds.size(); i++) {
125             buckets.add(histogramFactory.createBucket(rounding.valueForKey(bucketOrds.get(i)), bucketDocCount(i), bucketAggregations(i), keyed, formatter));
126         }
127 
128         // the contract of the histogram aggregation is that shards must return buckets ordered by key in ascending order
129         CollectionUtil.introSort(buckets, InternalOrder.KEY_ASC.comparator());
130 
131         // value source will be null for unmapped fields
132         InternalHistogram.EmptyBucketInfo emptyBucketInfo = minDocCount == 0 ? new InternalHistogram.EmptyBucketInfo(rounding, buildEmptySubAggregations(), extendedBounds) : null;
133         return histogramFactory.create(name, buckets, order, minDocCount, emptyBucketInfo, formatter, keyed, metaData());
134     }
135 
136     @Override
137     public InternalAggregation buildEmptyAggregation() {
138         InternalHistogram.EmptyBucketInfo emptyBucketInfo = minDocCount == 0 ? new InternalHistogram.EmptyBucketInfo(rounding, buildEmptySubAggregations(), extendedBounds) : null;
139         return histogramFactory.create(name, Collections.emptyList(), order, minDocCount, emptyBucketInfo, formatter, keyed, metaData());
140     }
141 
142     @Override
143     public void doClose() {
144         Releasables.close(bucketOrds);
145     }
146 
147     public static class Factory extends ValuesSourceAggregatorFactory<ValuesSource.Numeric> {
148 
149         private final Rounding rounding;
150         private final InternalOrder order;
151         private final boolean keyed;
152         private final long minDocCount;
153         private final ExtendedBounds extendedBounds;
154         private final InternalHistogram.Factory<?> histogramFactory;
155 
156         public Factory(String name, ValuesSourceConfig<ValuesSource.Numeric> config,
157                        Rounding rounding, InternalOrder order, boolean keyed, long minDocCount,
158                        ExtendedBounds extendedBounds, InternalHistogram.Factory<?> histogramFactory) {
159 
160             super(name, histogramFactory.type(), config);
161             this.rounding = rounding;
162             this.order = order;
163             this.keyed = keyed;
164             this.minDocCount = minDocCount;
165             this.extendedBounds = extendedBounds;
166             this.histogramFactory = histogramFactory;
167         }
168 
169         @Override
170         protected Aggregator createUnmapped(AggregationContext aggregationContext, Aggregator parent, Map<String, Object> metaData) throws IOException {
171             return new HistogramAggregator(name, factories, rounding, order, keyed, minDocCount, null, null, config.formatter(), histogramFactory, aggregationContext, parent, metaData);
172         }
173 
174         @Override
175         protected Aggregator doCreateInternal(ValuesSource.Numeric valuesSource, AggregationContext aggregationContext, Aggregator parent, boolean collectsFromSingleBucket, Map<String, Object> metaData) throws IOException {
176             if (collectsFromSingleBucket == false) {
177                 return asMultiBucketAggregator(this, aggregationContext, parent);
178             }
179             // we need to round the bounds given by the user and we have to do it for every aggregator we crate
180             // as the rounding is not necessarily an idempotent operation.
181             // todo we need to think of a better structure to the factory/agtor code so we won't need to do that
182             ExtendedBounds roundedBounds = null;
183             if (extendedBounds != null) {
184                 // we need to process & validate here using the parser
185                 extendedBounds.processAndValidate(name, aggregationContext.searchContext(), config.parser());
186                 roundedBounds = extendedBounds.round(rounding);
187             }
188             return new HistogramAggregator(name, factories, rounding, order, keyed, minDocCount, roundedBounds, valuesSource, config.formatter(), histogramFactory, aggregationContext, parent, metaData);
189         }
190 
191     }
