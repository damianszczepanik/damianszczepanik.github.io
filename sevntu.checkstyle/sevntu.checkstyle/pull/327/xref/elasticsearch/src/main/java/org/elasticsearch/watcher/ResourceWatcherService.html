19  package org.elasticsearch.watcher;
20  
21  import org.elasticsearch.ElasticsearchException;
22  import org.elasticsearch.ElasticsearchIllegalArgumentException;
23  import org.elasticsearch.common.component.AbstractLifecycleComponent;
24  import org.elasticsearch.common.inject.Inject;
25  import org.elasticsearch.common.settings.Settings;
26  import org.elasticsearch.common.unit.TimeValue;
27  import org.elasticsearch.common.util.concurrent.FutureUtils;
28  import org.elasticsearch.threadpool.ThreadPool;
29  
30  import java.io.IOException;
31  import java.util.Set;
32  import java.util.concurrent.CopyOnWriteArraySet;
33  import java.util.concurrent.ScheduledFuture;
34  
35  /**
36   * Generic resource watcher service
37   *
38   * Other elasticsearch services can register their resource watchers with this service using {@link #add(ResourceWatcher)}
39   * method. This service will call {@link org.elasticsearch.watcher.ResourceWatcher#checkAndNotify()} method of all
40   * registered watcher periodically. The frequency of checks can be specified using {@code watcher.interval} setting, which
41   * defaults to {@code 60s}. The service can be disabled by setting {@code watcher.enabled} setting to {@code false}.
42   */
43  public class ResourceWatcherService extends AbstractLifecycleComponent<ResourceWatcherService> {
44  
45      public static enum Frequency {
46  
47          /**
48           * Defaults to 5 seconds
49           */
50          HIGH(TimeValue.timeValueSeconds(5)),
51  
52          /**
53           * Defaults to 30 seconds
54           */
55          MEDIUM(TimeValue.timeValueSeconds(25)),
56  
57          /**
58           * Defaults to 60 seconds
59           */
60          LOW(TimeValue.timeValueSeconds(60));
61  
62          final TimeValue interval;
63  
64          private Frequency(TimeValue interval) {
65              this.interval = interval;
66          }
67      }
68  
69      private final boolean enabled;
70      private final ThreadPool threadPool;
71  
72      final ResourceMonitor lowMonitor;
73      final ResourceMonitor mediumMonitor;
74      final ResourceMonitor highMonitor;
75  
76      private volatile ScheduledFuture lowFuture;
77      private volatile ScheduledFuture mediumFuture;
78      private volatile ScheduledFuture highFuture;
79  
80      @Inject
81      public ResourceWatcherService(Settings settings, ThreadPool threadPool) {
82          super(settings);
83          this.enabled = settings.getAsBoolean("watcher.enabled", true);
84          this.threadPool = threadPool;
85  
86          TimeValue interval = settings.getAsTime("watcher.interval.low", Frequency.LOW.interval);
87          lowMonitor = new ResourceMonitor(interval, Frequency.LOW);
88          interval = settings.getAsTime("watcher.interval.medium", settings.getAsTime("watcher.interval", Frequency.MEDIUM.interval));
89          mediumMonitor = new ResourceMonitor(interval, Frequency.MEDIUM);
90          interval = settings.getAsTime("watcher.interval.high", Frequency.HIGH.interval);
91          highMonitor = new ResourceMonitor(interval, Frequency.HIGH);
92      }
93  
94      @Override
95      protected void doStart() throws ElasticsearchException {
96          if (!enabled) {
97              return;
98          }
99          lowFuture = threadPool.scheduleWithFixedDelay(lowMonitor, lowMonitor.interval);
100         mediumFuture = threadPool.scheduleWithFixedDelay(mediumMonitor, mediumMonitor.interval);
101         highFuture = threadPool.scheduleWithFixedDelay(highMonitor, highMonitor.interval);
102     }
103 
104     @Override
105     protected void doStop() throws ElasticsearchException {
106         if (!enabled) {
107             return;
108         }
109         FutureUtils.cancel(lowFuture);
110         FutureUtils.cancel(mediumFuture);
111         FutureUtils.cancel(highFuture);
112     }
113 
114     @Override
115     protected void doClose() throws ElasticsearchException {
116     }
117 
118     /**
119      * Register new resource watcher that will be checked in default {@link Frequency#MEDIUM MEDIUM} frequency
120      */
121     public <W extends ResourceWatcher> WatcherHandle<W> add(W watcher) throws IOException {
122         return add(watcher, Frequency.MEDIUM);
123     }
124 
125     /**
126      * Register new resource watcher that will be checked in the given frequency
127      */
128     public <W extends ResourceWatcher> WatcherHandle<W> add(W watcher, Frequency frequency) throws IOException {
129         watcher.init();
130         switch (frequency) {
131             case LOW:
132                 return lowMonitor.add(watcher);
133             case MEDIUM:
134                 return mediumMonitor.add(watcher);
135             case HIGH:
136                 return highMonitor.add(watcher);
137             default:
138                 throw new ElasticsearchIllegalArgumentException("Unknown frequency [" + frequency + "]");
139         }
140     }
141 
142     public void notifyNow() {
143         notifyNow(Frequency.MEDIUM);
144     }
145 
146     public void notifyNow(Frequency frequency) {
147         switch (frequency) {
148             case LOW:
149                 lowMonitor.run();
150                 break;
151             case MEDIUM:
152                 mediumMonitor.run();
153                 break;
154             case HIGH:
155                 highMonitor.run();
156                 break;
157             default:
158                 throw new ElasticsearchIllegalArgumentException("Unknown frequency [" + frequency + "]");
159         }
160     }
161 
162     class ResourceMonitor implements Runnable {
163 
164         final TimeValue interval;
165         final Frequency frequency;
166 
167         final Set<ResourceWatcher> watchers = new CopyOnWriteArraySet<>();
168 
169         private ResourceMonitor(TimeValue interval, Frequency frequency) {
170             this.interval = interval;
171             this.frequency = frequency;
172         }
173 
174         private <W extends ResourceWatcher> WatcherHandle<W> add(W watcher) {
175             watchers.add(watcher);
176             return new WatcherHandle<>(this, watcher);
177         }
178 
179         @Override
180         public synchronized void run() {
181             for(ResourceWatcher watcher : watchers) {
182                 try {
183                     watcher.checkAndNotify();
184                 } catch (IOException e) {
185                     logger.trace("failed to check resource watcher", e);
186                 }
187             }
188         }
189     }
190 }
