81      public DoubleArrayIndexFieldData(Index index, @IndexSettings Settings indexSettings, FieldMapper.Names fieldNames,
82                                       FieldDataType fieldDataType, IndexFieldDataCache cache, CircuitBreakerService breakerService) {
83          super(index, indexSettings, fieldNames, fieldDataType, cache);
84          this.breakerService = breakerService;
85      }
86  
87      @Override
88      public NumericType getNumericType() {
89          return NumericType.DOUBLE;
90      }
91  
92      @Override
93      public AtomicNumericFieldData loadDirect(LeafReaderContext context) throws Exception {
94  
95          final LeafReader reader = context.reader();
96          Terms terms = reader.terms(getFieldNames().indexName());
97          AtomicNumericFieldData data = null;
98          // TODO: Use an actual estimator to estimate before loading.
99          NonEstimatingEstimator estimator = new NonEstimatingEstimator(breakerService.getBreaker(CircuitBreaker.FIELDDATA));
100         if (terms == null) {
101             data = AtomicDoubleFieldData.empty(reader.maxDoc());
102             estimator.afterLoad(null, data.ramBytesUsed());
103             return data;
104         }
105         // TODO: how can we guess the number of terms? numerics end up creating more terms per value...
106         DoubleArray values = BigArrays.NON_RECYCLING_INSTANCE.newDoubleArray(128);
107 
108         final float acceptableTransientOverheadRatio = fieldDataType.getSettings().getAsFloat("acceptable_transient_overhead_ratio", OrdinalsBuilder.DEFAULT_ACCEPTABLE_OVERHEAD_RATIO);
109         boolean success = false;
110         try (OrdinalsBuilder builder = new OrdinalsBuilder(reader.maxDoc(), acceptableTransientOverheadRatio)) {
111             final BytesRefIterator iter = builder.buildFromTerms(getNumericType().wrapTermsEnum(terms.iterator(null)));
112             BytesRef term;
113             long numTerms = 0;
114             while ((term = iter.next()) != null) {
115                 values = BigArrays.NON_RECYCLING_INSTANCE.grow(values, numTerms + 1);
116                 values.set(numTerms++, NumericUtils.sortableLongToDouble(NumericUtils.prefixCodedToLong(term)));
117             }
118             values = BigArrays.NON_RECYCLING_INSTANCE.resize(values, numTerms);
119             final DoubleArray finalValues = values;
120             final Ordinals build = builder.build(fieldDataType.getSettings());
121             RandomAccessOrds ordinals = build.ordinals();
122             if (FieldData.isMultiValued(ordinals) || CommonSettings.getMemoryStorageHint(fieldDataType) == CommonSettings.MemoryStorageFormat.ORDINALS) {
123                 final long ramBytesUsed = build.ramBytesUsed() + values.ramBytesUsed();
124                 data = new AtomicDoubleFieldData(ramBytesUsed) {
125 
126                     @Override
127                     public SortedNumericDoubleValues getDoubleValues() {
128                         return withOrdinals(build, finalValues, reader.maxDoc());
129                     }
130                     
131                     @Override
132                     public Collection<Accountable> getChildResources() {
133                         List<Accountable> resources = new ArrayList<>();
134                         resources.add(Accountables.namedAccountable("ordinals", build));
135                         resources.add(Accountables.namedAccountable("values", finalValues));
136                         return Collections.unmodifiableList(resources);
137                     }
138 
139                 };
140             } else {
141                 final BitSet set = builder.buildDocsWithValuesSet();
142 
143                 // there's sweet spot where due to low unique value count, using ordinals will consume less memory
144                 long singleValuesArraySize = reader.maxDoc() * RamUsageEstimator.NUM_BYTES_DOUBLE + (set == null ? 0 : set.ramBytesUsed());
145                 long uniqueValuesArraySize = values.ramBytesUsed();
146                 long ordinalsSize = build.ramBytesUsed();
147                 if (uniqueValuesArraySize + ordinalsSize < singleValuesArraySize) {
148                     final long ramBytesUsed = build.ramBytesUsed() + values.ramBytesUsed();
149                     success = true;
150                     return data = new AtomicDoubleFieldData(ramBytesUsed) {
151 
152                         @Override
153                         public SortedNumericDoubleValues getDoubleValues() {
154                             return withOrdinals(build, finalValues, reader.maxDoc());
155                         }
156                         
157                         @Override
158                         public Collection<Accountable> getChildResources() {
159                             List<Accountable> resources = new ArrayList<>();
160                             resources.add(Accountables.namedAccountable("ordinals", build));
161                             resources.add(Accountables.namedAccountable("values", finalValues));
162                             return Collections.unmodifiableList(resources);
163                         }
164 
165                     };
166                 }
167 
168                 int maxDoc = reader.maxDoc();
169                 final DoubleArray sValues = BigArrays.NON_RECYCLING_INSTANCE.newDoubleArray(maxDoc);
170                 for (int i = 0; i < maxDoc; i++) {
171                     ordinals.setDocument(i);
172                     final long ordinal = ordinals.nextOrd();
173                     if (ordinal != SortedSetDocValues.NO_MORE_ORDS) {
174                         sValues.set(i, values.get(ordinal));
175                     }
176                 }
177                 assert sValues.size() == maxDoc;
178                 final long ramBytesUsed = sValues.ramBytesUsed() + (set == null ? 0 : set.ramBytesUsed());
179                 data = new AtomicDoubleFieldData(ramBytesUsed) {
180 
181                     @Override
182                     public SortedNumericDoubleValues getDoubleValues() {
183                         return singles(sValues, set);
184                     }
185                     
186                     @Override
187                     public Collection<Accountable> getChildResources() {
188                         List<Accountable> resources = new ArrayList<>();
189                         resources.add(Accountables.namedAccountable("values", sValues));
190                         resources.add(Accountables.namedAccountable("missing bitset", set));
191                         return Collections.unmodifiableList(resources);
192                     }
193 
194                 };
195                 success = true;
196             }
197             success = true;
198             return data;
199         } finally {
200             if (success) {
201                 estimator.afterLoad(null, data.ramBytesUsed());
202             }
203 
204         }
205 
206     }

210         return new DoubleValuesComparatorSource(this, missingValue, sortMode, nested);

213     private static SortedNumericDoubleValues withOrdinals(Ordinals ordinals, final DoubleArray values, int maxDoc) {
214         final RandomAccessOrds ords = ordinals.ordinals();
215         final SortedDocValues singleOrds = DocValues.unwrapSingleton(ords);
216         if (singleOrds != null) {
217             final NumericDoubleValues singleValues = new NumericDoubleValues() {
218                 @Override
219                 public double get(int docID) {
220                     final int ord = singleOrds.getOrd(docID);
221                     if (ord >= 0) {
222                         return values.get(singleOrds.getOrd(docID));
223                     } else {
224                         return 0;
225                     }
226                 }
227             };
228             return FieldData.singleton(singleValues, DocValues.docsWithValue(ords, maxDoc));
229         } else {
230             return new SortedNumericDoubleValues() {
231                 @Override
232                 public double valueAt(int index) {
233                     return values.get(ords.ordAt(index));
234                 }
235 
236                 @Override
237                 public void setDocument(int doc) {
238                     ords.setDocument(doc);
239                 }
240 
241                 @Override
242                 public int count() {
243                     return ords.cardinality();
244                 }
245             };
246         }
247     }
248 
249     private static SortedNumericDoubleValues singles(final DoubleArray values, Bits set) {
250         final NumericDoubleValues numValues = new NumericDoubleValues() {
251             @Override
252             public double get(int docID) {
253                 return values.get(docID);
254             }
255         };
256         return FieldData.singleton(numValues, set);
257     }
