100     /**
101      * Lists actively running benchmarks on the cluster
102      *
103      * @param request  Status request
104      * @param listener Response listener
105      */
106     public void listBenchmarks(final BenchmarkStatusRequest request, final ActionListener<BenchmarkStatusResponse> listener) {
107 
108         final List<DiscoveryNode> nodes = availableBenchmarkNodes();
109         if (nodes.size() == 0) {
110             listener.onResponse(new BenchmarkStatusResponse());
111         } else {
112             BenchmarkStatusAsyncHandler async = new BenchmarkStatusAsyncHandler(nodes.size(), request, listener);
113             for (DiscoveryNode node : nodes) {
114                 assert isBenchmarkNode(node);
115                 transportService.sendRequest(node, STATUS_ACTION_NAME, new NodeStatusRequest(request), async);
116             }
117         }
118     }

120     /**
121      * Aborts actively running benchmarks on the cluster
122      *
123      * @param benchmarkNames Benchmark name(s) to abort
124      * @param listener       Response listener
125      */
126     public void abortBenchmark(final String[] benchmarkNames, final ActionListener<AbortBenchmarkResponse> listener) {
127 
128         final List<DiscoveryNode> nodes = availableBenchmarkNodes();
129         if (nodes.size() == 0) {
130             listener.onFailure(new BenchmarkNodeMissingException("No available nodes for executing benchmarks"));
131         } else {
132             BenchmarkStateListener benchmarkStateListener = new BenchmarkStateListener() {
133                 @Override
134                 public void onResponse(final ClusterState newState, final List<BenchmarkMetaData.Entry> changed) {
135                     if (!changed.isEmpty()) {
136                         threadPool.executor(ThreadPool.Names.GENERIC).execute(new Runnable() {
137                             @Override
138                             public void run() {
139                                 Set<String> names = new HashSet<>();
140                                 Set<String> nodeNames = new HashSet<>();
141                                 final ImmutableOpenMap<String, DiscoveryNode> nodes = newState.nodes().nodes();
142 
143                                 for (BenchmarkMetaData.Entry e : changed) {
144                                     names.add(e.benchmarkId());
145                                     nodeNames.addAll(Arrays.asList(e.nodes()));
146                                 }
147                                 BenchmarkAbortAsyncHandler asyncHandler = new BenchmarkAbortAsyncHandler(nodeNames.size(), listener);
148                                 String[] benchmarkNames = names.toArray(new String[names.size()]);
149                                 for (String nodeId : nodeNames) {
150                                     final DiscoveryNode node = nodes.get(nodeId);
151                                     if (node != null) {
152                                         transportService.sendRequest(node, ABORT_ACTION_NAME, new NodeAbortRequest(benchmarkNames), asyncHandler);
153                                     } else {
154                                         asyncHandler.countDown.countDown();
155                                         logger.debug("Node for ID [" + nodeId + "] not found in cluster state - skipping");
156                                     }
157                                 }
158                             }
159                         });
160                     } else {
161                         listener.onFailure(new BenchmarkMissingException("No benchmarks found for " + Arrays.toString(benchmarkNames)));
162                     }
163                 }
164 
165                 @Override
166                 public void onFailure(Throwable t) {
167                     listener.onFailure(t);
168                 }
169             };
170             clusterService.submitStateUpdateTask("abort_benchmark", new AbortBenchmarkTask(benchmarkNames, benchmarkStateListener));
171         }
172     }

174     /**
175      * Executes benchmarks on the cluster
176      *
177      * @param request  Benchmark request
178      * @param listener Response listener
179      */
180     public void startBenchmark(final BenchmarkRequest request, final ActionListener<BenchmarkResponse> listener) {
181 
182         final List<DiscoveryNode> nodes = availableBenchmarkNodes();
183         if (nodes.size() == 0) {
184             listener.onFailure(new BenchmarkNodeMissingException("No available nodes for executing benchmark [" +
185                     request.benchmarkName() + "]"));
186         } else {
187             final BenchmarkStateListener benchListener = new BenchmarkStateListener() {
188                 @Override
189                 public void onResponse(final ClusterState newState, final List<BenchmarkMetaData.Entry> entries) {
190                     threadPool.executor(ThreadPool.Names.GENERIC).execute(new Runnable() {
191                         @Override
192                         public void run() {
193                             assert entries.size() == 1;
194                             BenchmarkMetaData.Entry entry = entries.get(0);
195                             final ImmutableOpenMap<String, DiscoveryNode> nodes = newState.nodes().nodes();
196                             final BenchmarkSearchAsyncHandler async = new BenchmarkSearchAsyncHandler(entry.nodes().length, request, listener);
197                             for (String nodeId : entry.nodes()) {
198                                 final DiscoveryNode node = nodes.get(nodeId);
199                                 if (node == null) {
200                                     async.handleExceptionInternal(
201                                             new ElasticsearchIllegalStateException("Node for ID [" + nodeId + "] not found in cluster state - skipping"));
202                                 } else {
203                                     logger.debug("Starting benchmark [{}] node [{}]", request.benchmarkName(), node.name());
204                                     transportService.sendRequest(node, START_ACTION_NAME, new NodeBenchRequest(request), async);
205                                 }
206                             }
207                         }
208                     });
209                 }
210 
211                 @Override
212                 public void onFailure(Throwable t) {
213                     listener.onFailure(t);
214                 }
215             };
216 
217             clusterService.submitStateUpdateTask("start_benchmark", new StartBenchmarkTask(request, benchListener));
218         }
219     }

221     private void finishBenchmark(final BenchmarkResponse benchmarkResponse, final String benchmarkId, final ActionListener<BenchmarkResponse> listener) {
222 
223         clusterService.submitStateUpdateTask("finish_benchmark", new FinishBenchmarkTask("finish_benchmark", benchmarkId, new BenchmarkStateListener() {
224             @Override
225             public void onResponse(ClusterState newClusterState, List<BenchmarkMetaData.Entry> changed) {
226                 listener.onResponse(benchmarkResponse);
227             }
228 
229             @Override
230             public void onFailure(Throwable t) {
231                 listener.onFailure(t);
232             }
233         }, (benchmarkResponse.state() != BenchmarkResponse.State.ABORTED) &&
234                 (benchmarkResponse.state() != BenchmarkResponse.State.FAILED)));
235     }

237     private final boolean isBenchmarkNode(DiscoveryNode node) {
238         ImmutableMap<String, String> attributes = node.getAttributes();
239         if (attributes.containsKey("bench")) {
240             String bench = attributes.get("bench");
241             return Boolean.parseBoolean(bench);
242         }
243         return false;
244     }

246     private List<DiscoveryNode> findNodes(BenchmarkRequest request) {
247         final int numNodes = request.numExecutorNodes();
248         final DiscoveryNodes nodes = clusterService.state().nodes();
249         DiscoveryNode localNode = nodes.localNode();
250         List<DiscoveryNode> benchmarkNodes = new ArrayList<DiscoveryNode>();
251         if (isBenchmarkNode(localNode)) {
252             benchmarkNodes.add(localNode);
253         }
254         for (DiscoveryNode node : nodes) {
255             if (benchmarkNodes.size() >= numNodes) {
256                 return benchmarkNodes;
257             }
258             if (node != localNode && isBenchmarkNode(node)) {
259                 benchmarkNodes.add(node);
260             }
261         }
262         return benchmarkNodes;
263     }

398     private abstract class CountDownAsyncHandler<T extends TransportResponse> implements TransportResponseHandler<T> {
399 
400         protected final CountDown countDown;
401         protected final CopyOnWriteArrayList<T> responses = new CopyOnWriteArrayList<T>();
402         protected final CopyOnWriteArrayList<Throwable> failures = new CopyOnWriteArrayList<Throwable>();
403 
404         protected CountDownAsyncHandler(int size) {
405             countDown = new CountDown(size);
406         }
407 
408         @Override
409         public abstract T newInstance();
410 
411         protected abstract void sendResponse();
412 
413         @Override
414         public void handleResponse(T t) {
415             responses.add(t);
416             if (countDown.countDown()) {
417                 sendResponse();
418             }
419         }
420 
421         @Override
422         public void handleException(TransportException t) {
423             failures.add(t);
424             logger.error(t.getMessage(), t);
425             if (countDown.countDown()) {
426                 sendResponse();
427             }
428         }
429 
430         @Override
431         public String executor() {
432             return ThreadPool.Names.SAME;
433         }
434     }

445         @Override

447             return new AbortBenchmarkResponse();
448         }
449 
450         @Override
451         protected void sendResponse() {
452             boolean acked = true;
453             for (AbortBenchmarkResponse nodeResponse : responses) {
454                 if (!nodeResponse.isAcknowledged()) {
455                     acked = false;
456                     break;
457                 }
458             }
459             listener.onResponse(new AbortBenchmarkResponse(acked));
460         }

463     private class BenchmarkStatusAsyncHandler extends CountDownAsyncHandler<BenchmarkStatusNodeResponse> {
464 
465         private final BenchmarkStatusRequest request;
466         private final ActionListener<BenchmarkStatusResponse> listener;
467 
468         public BenchmarkStatusAsyncHandler(int nodeCount, final BenchmarkStatusRequest request, ActionListener<BenchmarkStatusResponse> listener) {
469             super(nodeCount);
470             this.request = request;
471             this.listener = listener;
472         }
473 
474         @Override
475         public BenchmarkStatusNodeResponse newInstance() {
476             return new BenchmarkStatusNodeResponse();
477         }
478 
479         @Override
480         protected void sendResponse() {
481             int activeBenchmarks = 0;
482             BenchmarkStatusResponse consolidatedResponse = new BenchmarkStatusResponse();
483             Map<String, List<BenchmarkResponse>> nameNodeResponseMap = new HashMap<>();
484 
485             // Group node responses by benchmark name
486             for (BenchmarkStatusNodeResponse nodeResponse : responses) {
487                 for (BenchmarkResponse benchmarkResponse : nodeResponse.benchResponses()) {
488                     List<BenchmarkResponse> benchmarkResponses = nameNodeResponseMap.get(benchmarkResponse.benchmarkName());
489                     if (benchmarkResponses == null) {
490                         benchmarkResponses = new ArrayList<>();
491                         nameNodeResponseMap.put(benchmarkResponse.benchmarkName(), benchmarkResponses);
492                     }
493                     benchmarkResponses.add(benchmarkResponse);
494                 }
495                 activeBenchmarks += nodeResponse.activeBenchmarks();
496             }
497 
498             for (Map.Entry<String, List<BenchmarkResponse>> entry : nameNodeResponseMap.entrySet()) {
499                 BenchmarkResponse consolidated = consolidateBenchmarkResponses(entry.getValue());
500                 consolidatedResponse.addBenchResponse(consolidated);
501             }
502 
503             consolidatedResponse.totalActiveBenchmarks(activeBenchmarks);
504             listener.onResponse(consolidatedResponse);
505         }
506     }

508     private BenchmarkResponse consolidateBenchmarkResponses(List<BenchmarkResponse> responses) {
509         BenchmarkResponse response = new BenchmarkResponse();
510 
511         // Merge node responses into a single consolidated response
512         List<String> errors = new ArrayList<>();
513         for (BenchmarkResponse r : responses) {
514             for (Map.Entry<String, CompetitionResult> entry : r.competitionResults.entrySet()) {
515                 if (!response.competitionResults.containsKey(entry.getKey())) {
516                     response.competitionResults.put(entry.getKey(),
517                             new CompetitionResult(
518                                     entry.getKey(), entry.getValue().concurrency(), entry.getValue().multiplier(),
519                                     false, entry.getValue().percentiles()));
520                 }
521                 CompetitionResult cr = response.competitionResults.get(entry.getKey());
522                 cr.nodeResults().addAll(entry.getValue().nodeResults());
523             }
524             if (r.hasErrors()) {
525                 for (String error : r.errors()) {
526                     errors.add(error);
527                 }
528             }
529 
530             if (response.benchmarkName() == null) {
531                 response.benchmarkName(r.benchmarkName());
532             }
533             assert response.benchmarkName().equals(r.benchmarkName());
534             if (!errors.isEmpty()) {
535                 response.errors(errors.toArray(new String[errors.size()]));
536             }
537             response.mergeState(r.state());
538             assert errors.isEmpty() || response.state() != BenchmarkResponse.State.COMPLETE : "Response can't be complete since it has errors";
539         }
540 
541         return response;
542     }

544     private class BenchmarkSearchAsyncHandler extends CountDownAsyncHandler<BenchmarkResponse> {
545 
546         private final ActionListener<BenchmarkResponse> listener;
547         private final BenchmarkRequest request;
548 
549         public BenchmarkSearchAsyncHandler(int size, BenchmarkRequest request, ActionListener<BenchmarkResponse> listener) {
550             super(size);
551             this.listener = listener;
552             this.request = request;

583     public final class StartBenchmarkTask extends BenchmarkStateChangeAction<BenchmarkRequest> {
584 
585         private final BenchmarkStateListener stateListener;
586         private List<BenchmarkMetaData.Entry> newBenchmark = new ArrayList<>();
587 
588         public StartBenchmarkTask(BenchmarkRequest request, BenchmarkStateListener stateListener) {
589             super(request);
590             this.stateListener = stateListener;
591         }
592 
593         @Override
594         public ClusterState execute(ClusterState currentState) {
595             MetaData metaData = currentState.getMetaData();
596             BenchmarkMetaData bmd = metaData.custom(BenchmarkMetaData.TYPE);
597             MetaData.Builder mdBuilder = MetaData.builder(metaData);
598             ImmutableList.Builder<BenchmarkMetaData.Entry> builder = ImmutableList.builder();
599 
600             if (bmd != null) {
601                 for (BenchmarkMetaData.Entry entry : bmd.entries()) {
602                     if (request.benchmarkName().equals(entry.benchmarkId())) {
603                         if (entry.state() != BenchmarkMetaData.State.SUCCESS && entry.state() != BenchmarkMetaData.State.FAILED) {
604                             throw new ElasticsearchException("A benchmark with ID [" + request.benchmarkName() + "] is already running in state [" + entry.state() + "]");
605                         }
606                         // just drop the entry it it has finished successfully or it failed!
607                     } else {
608                         builder.add(entry);
609                     }
610                 }
611             }
612             List<DiscoveryNode> nodes = findNodes(request);
613             String[] nodeIds = new String[nodes.size()];
614             int i = 0;
615             for (DiscoveryNode node : nodes) {
616                 nodeIds[i++] = node.getId();
617             }
618             BenchmarkMetaData.Entry entry = new BenchmarkMetaData.Entry(request.benchmarkName(), BenchmarkMetaData.State.STARTED, nodeIds);
619             newBenchmark.add(entry);
620             bmd = new BenchmarkMetaData(builder.add(entry).build());
621             mdBuilder.putCustom(BenchmarkMetaData.TYPE, bmd);
622             return ClusterState.builder(currentState).metaData(mdBuilder).build();
623         }
624 
625         @Override
626         public void onFailure(String source, Throwable t) {
627             logger.warn("Failed to start benchmark: [{}]", t, request.benchmarkName());
628             newBenchmark = null;
629             stateListener.onFailure(t);
630         }
631 
632         @Override
633         public void clusterStateProcessed(String source, ClusterState oldState, final ClusterState newState) {
634             if (newBenchmark != null) {
635                 stateListener.onResponse(newState, newBenchmark);
636             }
637         }
638 
639         @Override
640         public TimeValue timeout() {
641             return request.masterNodeTimeout();
642         }
643     }

645     public final class FinishBenchmarkTask extends UpdateBenchmarkStateTask {
646 
647         private final boolean success;
648 
649         public FinishBenchmarkTask(String reason, String benchmarkId, BenchmarkStateListener listener, boolean success) {
650             super(reason, benchmarkId, listener);
651             this.success = success;
652         }
653 
654         @Override
655         protected BenchmarkMetaData.Entry process(BenchmarkMetaData.Entry entry) {
656             BenchmarkMetaData.State state = entry.state();
657             assert state == BenchmarkMetaData.State.STARTED || state == BenchmarkMetaData.State.ABORTED : "Expected state: STARTED or ABORTED but was: " + entry.state();
658             if (success) {
659                 return new BenchmarkMetaData.Entry(entry, BenchmarkMetaData.State.SUCCESS);
660             } else {
661                 return new BenchmarkMetaData.Entry(entry, BenchmarkMetaData.State.FAILED);
662             }
663         }
664     }

667         private final String[] patterns;
668 
669         public AbortBenchmarkTask(String[] patterns, BenchmarkStateListener listener) {
670             super("abort_benchmark", null, listener);
671             this.patterns = patterns;
672         }
673 
674         @Override
675         protected boolean match(BenchmarkMetaData.Entry entry) {

677         }
678 
679         @Override
680         protected BenchmarkMetaData.Entry process(BenchmarkMetaData.Entry entry) {
681             return new BenchmarkMetaData.Entry(entry, BenchmarkMetaData.State.ABORTED);

685     public abstract class UpdateBenchmarkStateTask extends ProcessedClusterStateUpdateTask {
686 
687         private final String reason;
688         protected final String benchmarkId;
689         private final BenchmarkStateListener listener;
690         private final List<BenchmarkMetaData.Entry> instances = new ArrayList<>();
691 
692 
693         protected UpdateBenchmarkStateTask(String reason, String benchmarkId, BenchmarkStateListener listener) {
694             this.reason = reason;
695             this.listener = listener;
696             this.benchmarkId = benchmarkId;
697         }
698 
699         protected boolean match(BenchmarkMetaData.Entry entry) {
700             return entry.benchmarkId().equals(this.benchmarkId);
701         }
702 
703         @Override
704         public ClusterState execute(ClusterState currentState) {
705             MetaData metaData = currentState.getMetaData();
706             BenchmarkMetaData bmd = metaData.custom(BenchmarkMetaData.TYPE);
707             MetaData.Builder mdBuilder = MetaData.builder(metaData);
708             if (bmd != null && !bmd.entries().isEmpty()) {
709                 ImmutableList.Builder<BenchmarkMetaData.Entry> builder = new ImmutableList.Builder<BenchmarkMetaData.Entry>();
710                 for (BenchmarkMetaData.Entry e : bmd.entries()) {
711                     if (benchmarkId == null || match(e)) {
712                         e = process(e);
713                         instances.add(e);
714                     }
715                     // Don't keep finished benchmarks around in cluster state
716                     if (e != null && (e.state() != BenchmarkMetaData.State.SUCCESS &&
717                             e.state() != BenchmarkMetaData.State.ABORTED &&
718                             e.state() != BenchmarkMetaData.State.FAILED)) {
719                         builder.add(e);
720                     }
721                 }
722                 if (instances.isEmpty()) {
723                     throw new ElasticsearchException("No Benchmark found for id: [" + benchmarkId + "]");
724                 }
725                 bmd = new BenchmarkMetaData(builder.build());
726             }
727             if (bmd != null) {
728                 mdBuilder.putCustom(BenchmarkMetaData.TYPE, bmd);
729             }
730             return ClusterState.builder(currentState).metaData(mdBuilder).build();
731         }
732 
733         protected abstract BenchmarkMetaData.Entry process(BenchmarkMetaData.Entry entry);
734 
735         @Override
736         public void onFailure(String source, Throwable t) {
737             logger.warn("Failed updating benchmark state for ID [{}] triggered by: [{}]", t, benchmarkId, reason);
738             listener.onFailure(t);
739         }
740 
741         @Override
742         public void clusterStateProcessed(String source, ClusterState oldState, final ClusterState newState) {
743             listener.onResponse(newState, instances);
744         }
745 
746         public String reason() {
747             return reason;
748         }
749     }

764     private List<DiscoveryNode> availableBenchmarkNodes() {
765         DiscoveryNodes nodes = clusterService.state().nodes();
766         List<DiscoveryNode> benchmarkNodes = new ArrayList<>(nodes.size());
767         for (DiscoveryNode node : nodes) {
768             if (isBenchmarkNode(node)) {
769                 benchmarkNodes.add(node);
770             }
771         }
772         return benchmarkNodes;
