19  package org.elasticsearch.action.percolate;
20  
21  import org.elasticsearch.action.ActionListener;
22  import org.elasticsearch.action.ActionRequestBuilder;
23  import org.elasticsearch.action.support.IndicesOptions;
24  import org.elasticsearch.client.Client;
25  
26  /**
27   * A builder for to ease the use of defining a {@link MultiPercolateRequest} instance.
28   */
29  public class MultiPercolateRequestBuilder extends ActionRequestBuilder<MultiPercolateRequest, MultiPercolateResponse, MultiPercolateRequestBuilder, Client> {
30  
31      public MultiPercolateRequestBuilder(Client client) {
32          super(client, new MultiPercolateRequest());
33      }
34  
35      /**
36       * Bundles the specified percolate request to the multi percolate request.
37       */
38      public MultiPercolateRequestBuilder add(PercolateRequest percolateRequest) {
39          request.add(percolateRequest);
40          return this;
41      }
42  
43      /**
44       * Bundles the specified percolate request build to the multi percolate request.
45       */
46      public MultiPercolateRequestBuilder add(PercolateRequestBuilder percolateRequestBuilder) {
47          request.add(percolateRequestBuilder);
48          return this;
49      }
50  
51      /**
52       * Specifies how to globally ignore indices that are not available and how to deal with wildcard indices expressions.
53       *
54       * Invoke this method before invoking {@link #add(PercolateRequestBuilder)}.
55       */
56      public MultiPercolateRequestBuilder setIndicesOptions(IndicesOptions indicesOptions) {
57          request.indicesOptions(indicesOptions);
58          return this;
59      }
60  
61      @Override
62      protected void doExecute(ActionListener<MultiPercolateResponse> listener) {
63          client.multiPercolate(request, listener);
64      }
65  }
