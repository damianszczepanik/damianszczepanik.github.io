45  /**
46   * The source recovery accepts recovery requests from other peer shards and start the recovery process from this
47   * source shard to the target shard.
48   */
49  public class RecoverySource extends AbstractComponent {
50  
51      public static class Actions {
52          public static final String START_RECOVERY = "internal:index/shard/recovery/start_recovery";
53      }
54  
55      private final TransportService transportService;
56      private final IndicesService indicesService;
57      private final RecoverySettings recoverySettings;
58      private final MappingUpdatedAction mappingUpdatedAction;
59  
60      private final ClusterService clusterService;
61  
62      private final OngoingRecoveres ongoingRecoveries = new OngoingRecoveres();
63  
64  
65      @Inject
66      public RecoverySource(Settings settings, TransportService transportService, IndicesService indicesService,
67                            RecoverySettings recoverySettings, MappingUpdatedAction mappingUpdatedAction, ClusterService clusterService) {
68          super(settings);
69          this.transportService = transportService;
70          this.indicesService = indicesService;
71          this.mappingUpdatedAction = mappingUpdatedAction;
72          this.clusterService = clusterService;
73          this.indicesService.indicesLifecycle().addListener(new IndicesLifecycle.Listener() {
74              @Override
75              public void beforeIndexShardClosed(ShardId shardId, @Nullable IndexShard indexShard,
76                                                 @IndexSettings Settings indexSettings) {
77                  if (indexShard != null) {
78                      ongoingRecoveries.cancel(indexShard, "shard is closed");
79                  }
80              }
81          });
82  
83          this.recoverySettings = recoverySettings;
84  
85          transportService.registerHandler(Actions.START_RECOVERY, new StartRecoveryTransportRequestHandler());
86      }
87  
88      private RecoveryResponse recover(final StartRecoveryRequest request) {
89          final IndexService indexService = indicesService.indexServiceSafe(request.shardId().index().name());
90          final IndexShard shard = indexService.shardSafe(request.shardId().id());
91  
92          // starting recovery from that our (the source) shard state is marking the shard to be in recovery mode as well, otherwise
93          // the index operations will not be routed to it properly
94          RoutingNode node = clusterService.state().readOnlyRoutingNodes().node(request.targetNode().id());
95          if (node == null) {
96              logger.debug("delaying recovery of {} as source node {} is unknown", request.shardId(), request.targetNode());
97              throw new DelayRecoveryException("source node does not have the node [" + request.targetNode() + "] in its state yet..");
98          }
99          ShardRouting targetShardRouting = null;
100         for (ShardRouting shardRouting : node) {
101             if (shardRouting.shardId().equals(request.shardId())) {
102                 targetShardRouting = shardRouting;
103                 break;
104             }
105         }
106         if (targetShardRouting == null) {
107             logger.debug("delaying recovery of {} as it is not listed as assigned to target node {}", request.shardId(), request.targetNode());
108             throw new DelayRecoveryException("source node does not have the shard listed in its state as allocated on the node");
109         }
110         if (!targetShardRouting.initializing()) {
111             logger.debug("delaying recovery of {} as it is not listed as initializing on the target node {}. known shards state is [{}]",
112                     request.shardId(), request.targetNode(), targetShardRouting.state());
113             throw new DelayRecoveryException("source node has the state of the target shard to be [" + targetShardRouting.state() + "], expecting to be [initializing]");
114         }
115 
116         logger.trace("[{}][{}] starting recovery to {}, mark_as_relocated {}", request.shardId().index().name(), request.shardId().id(), request.targetNode(), request.markAsRelocated());
117         final RecoverySourceHandler handler;
118         if (IndexMetaData.isOnSharedFilesystem(shard.indexSettings())) {
119             handler = new SharedFSRecoverySourceHandler(shard, request, recoverySettings, transportService, clusterService, indicesService, mappingUpdatedAction, logger);
120         } else {
121             handler = new RecoverySourceHandler(shard, request, recoverySettings, transportService, clusterService, indicesService, mappingUpdatedAction, logger);
122         }
123         ongoingRecoveries.add(shard, handler);
124         try {
125             shard.recover(handler);
126         } finally {
127             ongoingRecoveries.remove(shard, handler);
128         }
129         return handler.getResponse();
130     }
131 
132     class StartRecoveryTransportRequestHandler extends BaseTransportRequestHandler<StartRecoveryRequest> {
133 
134         @Override
135         public StartRecoveryRequest newInstance() {
136             return new StartRecoveryRequest();
137         }
138 
139         @Override
140         public String executor() {
141             return ThreadPool.Names.GENERIC;
142         }
143 
144         @Override
145         public void messageReceived(final StartRecoveryRequest request, final TransportChannel channel) throws Exception {
146             RecoveryResponse response = recover(request);
147             channel.sendResponse(response);
148         }
149     }
150 
151 
152     private static final class OngoingRecoveres {
153         private final Map<IndexShard, Set<RecoverySourceHandler>> ongoingRecoveries = new HashMap<>();
154 
155         synchronized void add(IndexShard shard, RecoverySourceHandler handler) {
156             Set<RecoverySourceHandler> shardRecoveryHandlers = ongoingRecoveries.get(shard);
157             if (shardRecoveryHandlers == null) {
158                 shardRecoveryHandlers = new HashSet<>();
159                 ongoingRecoveries.put(shard, shardRecoveryHandlers);
160             }
161             assert shardRecoveryHandlers.contains(handler) == false : "Handler was already registered [" + handler + "]";
162             shardRecoveryHandlers.add(handler);
163         }
164 
165         synchronized void remove(IndexShard shard, RecoverySourceHandler handler) {
166             final Set<RecoverySourceHandler> shardRecoveryHandlers = ongoingRecoveries.get(shard);
167             assert shardRecoveryHandlers != null : "Shard was not registered [" + shard + "]";
168             boolean remove = shardRecoveryHandlers.remove(handler);
169             assert remove : "Handler was not registered [" + handler + "]";
170             if (shardRecoveryHandlers.isEmpty()) {
171                 ongoingRecoveries.remove(shard);
172             }
173         }
174 
175         synchronized void cancel(IndexShard shard, String reason) {
176             final Set<RecoverySourceHandler> shardRecoveryHandlers = ongoingRecoveries.get(shard);
177             if (shardRecoveryHandlers != null) {
178                 final List<Exception> failures = new ArrayList<>();
179                 for (RecoverySourceHandler handlers : shardRecoveryHandlers) {
180                     try {
181                         handlers.cancel(reason);
182                     } catch (Exception ex) {
183                         failures.add(ex);
184                     }
185                 }
186                 ExceptionsHelper.maybeThrowRuntimeAndSuppress(failures);
187             }
188         }
189     }
190 }
