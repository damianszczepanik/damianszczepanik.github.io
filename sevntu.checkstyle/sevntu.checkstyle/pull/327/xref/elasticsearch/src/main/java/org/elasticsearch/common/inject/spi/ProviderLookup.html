17  package org.elasticsearch.common.inject.spi;
18  
19  import org.elasticsearch.common.inject.Binder;
20  import org.elasticsearch.common.inject.Key;
21  import org.elasticsearch.common.inject.Provider;
22  
23  import static com.google.common.base.Preconditions.checkNotNull;
24  import static com.google.common.base.Preconditions.checkState;
25  
26  /**
27   * A lookup of the provider for a type. Lookups are created explicitly in a module using
28   * {@link org.elasticsearch.common.inject.Binder#getProvider(Class) getProvider()} statements:
29   * <pre>
30   *     Provider&lt;PaymentService&gt; paymentServiceProvider
31   *         = getProvider(PaymentService.class);</pre>
32   *
33   * @author jessewilson@google.com (Jesse Wilson)
34   * @since 2.0
35   */
36  public final class ProviderLookup<T> implements Element {
37      private final Object source;
38      private final Key<T> key;
39      private Provider<T> delegate;
40  
41      public ProviderLookup(Object source, Key<T> key) {
42          this.source = checkNotNull(source, "source");
43          this.key = checkNotNull(key, "key");
44      }
45  
46      @Override
47      public Object getSource() {
48          return source;
49      }
50  
51      public Key<T> getKey() {
52          return key;
53      }
54  
55      @Override
56      public <T> T acceptVisitor(ElementVisitor<T> visitor) {
57          return visitor.visit(this);
58      }
59  
60      /**
61       * Sets the actual provider.
62       *
63       * @throws IllegalStateException if the delegate is already set
64       */
65      public void initializeDelegate(Provider<T> delegate) {
66          checkState(this.delegate == null, "delegate already initialized");
67          this.delegate = checkNotNull(delegate, "delegate");
68      }
69  
70      @Override
71      public void applyTo(Binder binder) {
72          initializeDelegate(binder.withSource(getSource()).getProvider(key));
73      }
74  
75      /**
76       * Returns the delegate provider, or {@code null} if it has not yet been initialized. The delegate
77       * will be initialized when this element is processed, or otherwise used to create an injector.
78       */
79      public Provider<T> getDelegate() {
80          return delegate;
81      }
82  
83      /**
84       * Returns the looked up provider. The result is not valid until this lookup has been initialized,
85       * which usually happens when the injector is created. The provider will throw an {@code
86       * IllegalStateException} if you try to use it beforehand.
87       */
88      public Provider<T> getProvider() {
89          return new Provider<T>() {
90              @Override
91              public T get() {
92                  checkState(delegate != null,
93                          "This Provider cannot be used until the Injector has been created.");
94                  return delegate.get();
95              }
96  
97              @Override
98              public String toString() {
99                  return "Provider<" + key.getTypeLiteral() + ">";
100             }
101         };
102     }
103 }
