50  public class GatewayMetaState extends AbstractComponent implements ClusterStateListener {
51  
52      private static final String DEPRECATED_SETTING_ROUTING_HASH_FUNCTION = "cluster.routing.operation.hash.type";
53      private static final String DEPRECATED_SETTING_ROUTING_USE_TYPE = "cluster.routing.operation.use_type";
54  
55      private final NodeEnvironment nodeEnv;
56      private final MetaStateService metaStateService;
57      private final DanglingIndicesState danglingIndicesState;
58  
59      @Nullable
60      private volatile MetaData currentMetaData;
61  
62      @Inject
63      public GatewayMetaState(Settings settings, NodeEnvironment nodeEnv, MetaStateService metaStateService,
64                              DanglingIndicesState danglingIndicesState, TransportNodesListGatewayMetaState nodesListGatewayMetaState) throws Exception {
65          super(settings);
66          this.nodeEnv = nodeEnv;
67          this.metaStateService = metaStateService;
68          this.danglingIndicesState = danglingIndicesState;
69          nodesListGatewayMetaState.init(this);
70  
71  
72          if (DiscoveryNode.masterNode(settings) || DiscoveryNode.dataNode(settings)) {
73              nodeEnv.ensureAtomicMoveSupported();
74          }
75          if (DiscoveryNode.masterNode(settings)) {
76              try {
77                  ensureNoPre019State();
78                  pre20Upgrade();
79                  long start = System.currentTimeMillis();
80                  metaStateService.loadFullState();
81                  logger.debug("took {} to load state", TimeValue.timeValueMillis(System.currentTimeMillis() - start));
82              } catch (Exception e) {
83                  logger.error("failed to read local state, exiting...", e);
84                  throw e;
85              }
86          }
87      }
88  
89      public MetaData loadMetaState() throws Exception {
90          return metaStateService.loadFullState();
91      }
92  
93      @Override
94      public void clusterChanged(ClusterChangedEvent event) {
95          final ClusterState state = event.state();
96          if (state.blocks().disableStatePersistence()) {
97              // reset the current metadata, we need to start fresh...
98              this.currentMetaData = null;
99              return;
100         }
101 
102         MetaData newMetaData = state.metaData();
103         // we don't check if metaData changed, since we might be called several times and we need to check dangling...
104 
105         boolean success = true;
106         // only applied to master node, writing the global and index level states
107         if (state.nodes().localNode().masterNode()) {
108             // check if the global state changed?
109             if (currentMetaData == null || !MetaData.isGlobalStateEquals(currentMetaData, newMetaData)) {
110                 try {
111                     metaStateService.writeGlobalState("changed", newMetaData);
112                 } catch (Throwable e) {
113                     success = false;
114                 }
115             }
116 
117             // check and write changes in indices
118             for (IndexMetaData indexMetaData : newMetaData) {
119                 String writeReason = null;
120                 IndexMetaData currentIndexMetaData;
121                 if (currentMetaData == null) {
122                     // a new event..., check from the state stored
123                     try {
124                         currentIndexMetaData = metaStateService.loadIndexState(indexMetaData.index());
125                     } catch (IOException ex) {
126                         throw new ElasticsearchException("failed to load index state", ex);
127                     }
128                 } else {
129                     currentIndexMetaData = currentMetaData.index(indexMetaData.index());
130                 }
131                 if (currentIndexMetaData == null) {
132                     writeReason = "freshly created";
133                 } else if (currentIndexMetaData.version() != indexMetaData.version()) {
134                     writeReason = "version changed from [" + currentIndexMetaData.version() + "] to [" + indexMetaData.version() + "]";
135                 }
136 
137                 // we update the writeReason only if we really need to write it
138                 if (writeReason == null) {
139                     continue;
140                 }
141 
142                 try {
143                     metaStateService.writeIndex(writeReason, indexMetaData, currentIndexMetaData);
144                 } catch (Throwable e) {
145                     success = false;
146                 }
147             }
148         }
149 
150         danglingIndicesState.processDanglingIndices(newMetaData);
151 
152         if (success) {
153             currentMetaData = newMetaData;
154         }
155     }

157     /**
158      * Throws an IAE if a pre 0.19 state is detected
159      */
160     private void ensureNoPre019State() throws Exception {
161         for (Path dataLocation : nodeEnv.nodeDataPaths()) {
162             final Path stateLocation = dataLocation.resolve(MetaDataStateFormat.STATE_DIR_NAME);
163             if (!Files.exists(stateLocation)) {
164                 continue;
165             }
166             try (DirectoryStream<Path> stream = Files.newDirectoryStream(stateLocation)) {
167                 for (Path stateFile : stream) {
168                     if (logger.isTraceEnabled()) {
169                         logger.trace("[upgrade]: processing [" + stateFile.getFileName() + "]");
170                     }
171                     final String name = stateFile.getFileName().toString();
172                     if (name.startsWith("metadata-")) {
173                         throw new ElasticsearchIllegalStateException("Detected pre 0.19 metadata file please upgrade to a version before "
174                                 + Version.CURRENT.minimumCompatibilityVersion()
175                                 + " first to upgrade state structures - metadata found: [" + stateFile.getParent().toAbsolutePath());
176                     }
177                 }
178             }
179         }
180     }

182     /**
183      * Elasticsearch 2.0 deprecated custom routing hash functions. So what we do here is that for old indices, we
184      * move this old & deprecated node setting to an index setting so that we can keep things backward compatible.
185      */
186     private void pre20Upgrade() throws Exception {
187         final Class<? extends HashFunction> pre20HashFunction;
188         final String pre20HashFunctionName = settings.get(DEPRECATED_SETTING_ROUTING_HASH_FUNCTION, null);
189         final boolean hasCustomPre20HashFunction = pre20HashFunctionName != null;
190         // the hash function package has changed we replace the two hash functions if their fully qualified name is used.
191         if (hasCustomPre20HashFunction) {
192             switch (pre20HashFunctionName) {
193                 case "org.elasticsearch.cluster.routing.operation.hash.simple.SimpleHashFunction":
194                     pre20HashFunction = SimpleHashFunction.class;
195                     break;
196                 case "org.elasticsearch.cluster.routing.operation.hash.djb.DjbHashFunction":
197                     pre20HashFunction = DjbHashFunction.class;
198                     break;
199                 default:
200                     pre20HashFunction = settings.getAsClass(DEPRECATED_SETTING_ROUTING_HASH_FUNCTION, DjbHashFunction.class, "org.elasticsearch.cluster.routing.", "HashFunction");
201             }
202         } else {
203             pre20HashFunction = DjbHashFunction.class;
204         }
205         final Boolean pre20UseType = settings.getAsBoolean(DEPRECATED_SETTING_ROUTING_USE_TYPE, null);
206         MetaData metaData = loadMetaState();
207         for (IndexMetaData indexMetaData : metaData) {
208             if (indexMetaData.settings().get(IndexMetaData.SETTING_LEGACY_ROUTING_HASH_FUNCTION) == null
209                     && indexMetaData.getCreationVersion().before(Version.V_2_0_0)) {
210                 // these settings need an upgrade
211                 Settings indexSettings = ImmutableSettings.builder().put(indexMetaData.settings())
212                         .put(IndexMetaData.SETTING_LEGACY_ROUTING_HASH_FUNCTION, pre20HashFunction)
213                         .put(IndexMetaData.SETTING_LEGACY_ROUTING_USE_TYPE, pre20UseType == null ? false : pre20UseType)
214                         .build();
215                 IndexMetaData newMetaData = IndexMetaData.builder(indexMetaData)
216                         .version(indexMetaData.version())
217                         .settings(indexSettings)
218                         .build();
219                 metaStateService.writeIndex("upgrade", newMetaData, null);
220             } else if (indexMetaData.getCreationVersion().onOrAfter(Version.V_2_0_0)) {
221                 if (indexMetaData.getSettings().get(IndexMetaData.SETTING_LEGACY_ROUTING_HASH_FUNCTION) != null
222                         || indexMetaData.getSettings().get(IndexMetaData.SETTING_LEGACY_ROUTING_USE_TYPE) != null) {
223                     throw new ElasticsearchIllegalStateException("Indices created on or after 2.0 should NOT contain [" + IndexMetaData.SETTING_LEGACY_ROUTING_HASH_FUNCTION
224                             + "] + or [" + IndexMetaData.SETTING_LEGACY_ROUTING_USE_TYPE + "] in their index settings");
225                 }
226             }
227         }
228         if (hasCustomPre20HashFunction|| pre20UseType != null) {
229             logger.warn("Settings [{}] and [{}] are deprecated. Index settings from your old indices have been updated to record the fact that they "
230                     + "used some custom routing logic, you can now remove these settings from your `elasticsearch.yml` file", DEPRECATED_SETTING_ROUTING_HASH_FUNCTION, DEPRECATED_SETTING_ROUTING_USE_TYPE);
231         }
232     }
