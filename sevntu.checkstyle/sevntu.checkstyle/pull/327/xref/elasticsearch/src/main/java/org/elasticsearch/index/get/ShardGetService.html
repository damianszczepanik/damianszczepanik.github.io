99      public GetResult get(String type, String id, String[] gFields, boolean realtime, long version, VersionType versionType, FetchSourceContext fetchSourceContext, boolean ignoreErrorsOnGeneratedFields)
100             throws ElasticsearchException {
101         currentMetric.inc();
102         try {
103             long now = System.nanoTime();
104             GetResult getResult = innerGet(type, id, gFields, realtime, version, versionType, fetchSourceContext, ignoreErrorsOnGeneratedFields);
105 
106             if (getResult.isExists()) {
107                 existsMetric.inc(System.nanoTime() - now);
108             } else {
109                 missingMetric.inc(System.nanoTime() - now);
110             }
111             return getResult;
112         } finally {
113             currentMetric.dec();
114         }
115     }

117     /**
118      * Returns {@link GetResult} based on the specified {@link Engine.GetResult} argument.
119      * This method basically loads specified fields for the associated document in the engineGetResult.
120      * This method load the fields from the Lucene index and not from transaction log and therefore isn't realtime.
121      * <p/>
122      * Note: Call <b>must</b> release engine searcher associated with engineGetResult!
123      */
124     public GetResult get(Engine.GetResult engineGetResult, String id, String type, String[] fields, FetchSourceContext fetchSourceContext, boolean ignoreErrorsOnGeneratedFields) {
125         if (!engineGetResult.exists()) {
126             return new GetResult(shardId.index().name(), type, id, -1, false, null, null);
127         }
128 
129         currentMetric.inc();
130         try {
131             long now = System.nanoTime();
132             DocumentMapper docMapper = mapperService.documentMapper(type);
133             if (docMapper == null) {
134                 missingMetric.inc(System.nanoTime() - now);
135                 return new GetResult(shardId.index().name(), type, id, -1, false, null, null);
136             }
137             fetchSourceContext = normalizeFetchSourceContent(fetchSourceContext, fields);
138             GetResult getResult = innerGetLoadFromStoredFields(type, id, fields, fetchSourceContext, engineGetResult, docMapper, ignoreErrorsOnGeneratedFields);
139             if (getResult.isExists()) {
140                 existsMetric.inc(System.nanoTime() - now);
141             } else {
142                 missingMetric.inc(System.nanoTime() - now); // This shouldn't happen...
143             }
144             return getResult;
145         } finally {
146             currentMetric.dec();
147         }
148     }

150     /**
151      * decides what needs to be done based on the request input and always returns a valid non-null FetchSourceContext
152      */
153     protected FetchSourceContext normalizeFetchSourceContent(@Nullable FetchSourceContext context, @Nullable String[] gFields) {
154         if (context != null) {
155             return context;
156         }
157         if (gFields == null) {
158             return FetchSourceContext.FETCH_SOURCE;
159         }
160         for (String field : gFields) {
161             if (SourceFieldMapper.NAME.equals(field)) {
162                 return FetchSourceContext.FETCH_SOURCE;
163             }
164         }
165         return FetchSourceContext.DO_NOT_FETCH_SOURCE;
166     }

168     public GetResult innerGet(String type, String id, String[] gFields, boolean realtime, long version, VersionType versionType, FetchSourceContext fetchSourceContext, boolean ignoreErrorsOnGeneratedFields) throws ElasticsearchException {
169         fetchSourceContext = normalizeFetchSourceContent(fetchSourceContext, gFields);
170 
171         boolean loadSource = (gFields != null && gFields.length > 0) || fetchSourceContext.fetchSource();
172 
173         Engine.GetResult get = null;
174         if (type == null || type.equals("_all")) {
175             for (String typeX : mapperService.types()) {
176                 get = indexShard.get(new Engine.Get(realtime, new Term(UidFieldMapper.NAME, Uid.createUidAsBytes(typeX, id)))
177                         .loadSource(loadSource).version(version).versionType(versionType));
178                 if (get.exists()) {
179                     type = typeX;
180                     break;
181                 } else {
182                     get.release();
183                 }
184             }
185             if (get == null) {
186                 return new GetResult(shardId.index().name(), type, id, -1, false, null, null);
187             }
188             if (!get.exists()) {
189                 // no need to release here as well..., we release in the for loop for non exists
190                 return new GetResult(shardId.index().name(), type, id, -1, false, null, null);
191             }
192         } else {
193             get = indexShard.get(new Engine.Get(realtime, new Term(UidFieldMapper.NAME, Uid.createUidAsBytes(type, id)))
194                     .loadSource(loadSource).version(version).versionType(versionType));
195             if (!get.exists()) {
196                 get.release();
197                 return new GetResult(shardId.index().name(), type, id, -1, false, null, null);
198             }
199         }
200 
201         DocumentMapper docMapper = mapperService.documentMapper(type);
202         if (docMapper == null) {
203             get.release();
204             return new GetResult(shardId.index().name(), type, id, -1, false, null, null);
205         }
206 
207         try {
208             // break between having loaded it from translog (so we only have _source), and having a document to load
209             if (get.docIdAndVersion() != null) {
210                 return innerGetLoadFromStoredFields(type, id, gFields, fetchSourceContext, get, docMapper, ignoreErrorsOnGeneratedFields);
211             } else {
212                 Translog.Source source = get.source();
213 
214                 Map<String, GetField> fields = null;
215                 SearchLookup searchLookup = null;
216 
217                 // we can only load scripts that can run against the source
218                 if (gFields != null && gFields.length > 0) {
219                     for (String field : gFields) {
220                         if (SourceFieldMapper.NAME.equals(field)) {
221                             // dealt with when normalizing fetchSourceContext.
222                             continue;
223                         }
224                         Object value = null;
225                         if (field.equals(RoutingFieldMapper.NAME) && docMapper.routingFieldMapper().fieldType().stored()) {
226                             value = source.routing;
227                         } else if (field.equals(ParentFieldMapper.NAME) && docMapper.parentFieldMapper().active() && docMapper.parentFieldMapper().fieldType().stored()) {
228                             value = source.parent;
229                         } else if (field.equals(TimestampFieldMapper.NAME) && docMapper.timestampFieldMapper().fieldType().stored()) {
230                             value = source.timestamp;
231                         } else if (field.equals(TTLFieldMapper.NAME) && docMapper.TTLFieldMapper().fieldType().stored()) {
232                             // Call value for search with timestamp + ttl here to display the live remaining ttl value and be consistent with the search result display
233                             if (source.ttl > 0) {
234                                 value = docMapper.TTLFieldMapper().valueForSearch(source.timestamp + source.ttl);
235                             }
236                         } else if (field.equals(SizeFieldMapper.NAME) && docMapper.rootMapper(SizeFieldMapper.class).fieldType().stored()) {
237                             value = source.source.length();
238                         } else {
239                             if (searchLookup == null) {
240                                 searchLookup = new SearchLookup(mapperService, fieldDataService, new String[]{type});
241                                 searchLookup.source().setNextSource(source.source);
242                             }
243 
244                             FieldMapper<?> fieldMapper = docMapper.mappers().smartNameFieldMapper(field);
245                             if (fieldMapper == null) {
246                                 if (docMapper.objectMappers().get(field) != null) {
247                                     // Only fail if we know it is a object field, missing paths / fields shouldn't fail.
248                                     throw new ElasticsearchIllegalArgumentException("field [" + field + "] isn't a leaf field");
249                                 }
250                             } else if (shouldGetFromSource(ignoreErrorsOnGeneratedFields, docMapper, fieldMapper)) {
251                                 List<Object> values = searchLookup.source().extractRawValues(field);
252                                 if (!values.isEmpty()) {
253                                     for (int i = 0; i < values.size(); i++) {
254                                         values.set(i, fieldMapper.valueForSearch(values.get(i)));
255                                     }
256                                     value = values;
257                                 }
258 
259                             }
260                         }
261                         if (value != null) {
262                             if (fields == null) {
263                                 fields = newHashMapWithExpectedSize(2);
264                             }
265                             if (value instanceof List) {
266                                 fields.put(field, new GetField(field, (List) value));
267                             } else {
268                                 fields.put(field, new GetField(field, ImmutableList.of(value)));
269                             }
270                         }
271                     }
272                 }
273 
274                 // deal with source, but only if it's enabled (we always have it from the translog)
275                 BytesReference sourceToBeReturned = null;
276                 SourceFieldMapper sourceFieldMapper = docMapper.sourceMapper();
277                 if (fetchSourceContext.fetchSource() && sourceFieldMapper.enabled()) {
278 
279                     sourceToBeReturned = source.source;
280 
281                     // Cater for source excludes/includes at the cost of performance
282                     // We must first apply the field mapper filtering to make sure we get correct results
283                     // in the case that the fetchSourceContext white lists something that's not included by the field mapper
284 
285                     boolean sourceFieldFiltering = sourceFieldMapper.includes().length > 0 || sourceFieldMapper.excludes().length > 0;
286                     boolean sourceFetchFiltering = fetchSourceContext.includes().length > 0 || fetchSourceContext.excludes().length > 0;
287                     if (fetchSourceContext.transformSource() || sourceFieldFiltering || sourceFetchFiltering) {
288                         // TODO: The source might parsed and available in the sourceLookup but that one uses unordered maps so different. Do we care?
289                         Tuple<XContentType, Map<String, Object>> typeMapTuple = XContentHelper.convertToMap(source.source, true);
290                         XContentType sourceContentType = typeMapTuple.v1();
291                         Map<String, Object> sourceAsMap = typeMapTuple.v2();
292                         if (fetchSourceContext.transformSource()) {
293                             sourceAsMap = docMapper.transformSourceAsMap(sourceAsMap);
294                         }
295                         if (sourceFieldFiltering) {
296                             sourceAsMap = XContentMapValues.filter(sourceAsMap, sourceFieldMapper.includes(), sourceFieldMapper.excludes());
297                         }
298                         if (sourceFetchFiltering) {
299                             sourceAsMap = XContentMapValues.filter(sourceAsMap, fetchSourceContext.includes(), fetchSourceContext.excludes());
300                         }
301                         try {
302                             sourceToBeReturned = XContentFactory.contentBuilder(sourceContentType).map(sourceAsMap).bytes();
303                         } catch (IOException e) {
304                             throw new ElasticsearchException("Failed to get type [" + type + "] and id [" + id + "] with includes/excludes set", e);
305                         }
306                     }
307                 }
308 
309                 return new GetResult(shardId.index().name(), type, id, get.version(), get.exists(), sourceToBeReturned, fields);
310             }
311         } finally {
312             get.release();
313         }
314     }

316     protected boolean shouldGetFromSource(boolean ignoreErrorsOnGeneratedFields, DocumentMapper docMapper, FieldMapper<?> fieldMapper) {
317         if (!fieldMapper.isGenerated()) {
318             //if the field is always there we check if either source mapper is enabled, in which case we get the field
319             // from source, or, if the field is stored, in which case we have to get if from source here also (we are in the translog phase, doc not indexed yet, we annot access the stored fields)
320             return docMapper.sourceMapper().enabled() || fieldMapper.fieldType().stored();
321         } else {
322             if (!fieldMapper.fieldType().stored()) {
323                 //if it is not stored, user will not get the generated field back
324                 return false;
325             } else {
326                 if (ignoreErrorsOnGeneratedFields) {
327                     return false;
328                 } else {
329                     throw new ElasticsearchException("Cannot access field " + fieldMapper.name() + " from transaction log. You can only get this field after refresh() has been called.");
330                 }
331             }
332 
333         }
334     }

336     private GetResult innerGetLoadFromStoredFields(String type, String id, String[] gFields, FetchSourceContext fetchSourceContext, Engine.GetResult get, DocumentMapper docMapper, boolean ignoreErrorsOnGeneratedFields) {
337         Map<String, GetField> fields = null;
338         BytesReference source = null;
339         Versions.DocIdAndVersion docIdAndVersion = get.docIdAndVersion();
340         FieldsVisitor fieldVisitor = buildFieldsVisitors(gFields, fetchSourceContext);
341         if (fieldVisitor != null) {
342             try {
343                 docIdAndVersion.context.reader().document(docIdAndVersion.docId, fieldVisitor);
344             } catch (IOException e) {
345                 throw new ElasticsearchException("Failed to get type [" + type + "] and id [" + id + "]", e);
346             }
347             source = fieldVisitor.source();
348 
349             if (!fieldVisitor.fields().isEmpty()) {
350                 fieldVisitor.postProcess(docMapper);
351                 fields = new HashMap<>(fieldVisitor.fields().size());
352                 for (Map.Entry<String, List<Object>> entry : fieldVisitor.fields().entrySet()) {
353                     fields.put(entry.getKey(), new GetField(entry.getKey(), entry.getValue()));
354                 }
355             }
356         }
357 
358         // now, go and do the script thingy if needed
359 
360         if (gFields != null && gFields.length > 0) {
361             SearchLookup searchLookup = null;
362             for (String field : gFields) {
363                 Object value = null;
364                 FieldMappers fieldMapper = docMapper.mappers().smartName(field);
365                 if (fieldMapper == null) {
366                     if (docMapper.objectMappers().get(field) != null) {
367                         // Only fail if we know it is a object field, missing paths / fields shouldn't fail.
368                         throw new ElasticsearchIllegalArgumentException("field [" + field + "] isn't a leaf field");
369                     }
370                 } else if (!fieldMapper.mapper().fieldType().stored() && !fieldMapper.mapper().isGenerated()) {
371                     if (searchLookup == null) {
372                         searchLookup = new SearchLookup(mapperService, fieldDataService, new String[]{type});
373                         searchLookup.setNextReader(docIdAndVersion.context);
374                         searchLookup.source().setNextSource(source);
375                         searchLookup.setNextDocId(docIdAndVersion.docId);
376                     }
377 
378                     List<Object> values = searchLookup.source().extractRawValues(field);
379                     if (!values.isEmpty()) {
380                         for (int i = 0; i < values.size(); i++) {
381                             values.set(i, fieldMapper.mapper().valueForSearch(values.get(i)));
382                         }
383                         value = values;
384                     }
385                 }
386 
387                 if (value != null) {
388                     if (fields == null) {
389                         fields = newHashMapWithExpectedSize(2);
390                     }
391                     if (value instanceof List) {
392                         fields.put(field, new GetField(field, (List) value));
393                     } else {
394                         fields.put(field, new GetField(field, ImmutableList.of(value)));
395                     }
396                 }
397             }
398         }
399 
400         if (!fetchSourceContext.fetchSource()) {
401             source = null;
402         } else if (fetchSourceContext.transformSource() || fetchSourceContext.includes().length > 0 || fetchSourceContext.excludes().length > 0) {
403             Map<String, Object> sourceAsMap;
404             XContentType sourceContentType = null;
405             // TODO: The source might parsed and available in the sourceLookup but that one uses unordered maps so different. Do we care?
406             Tuple<XContentType, Map<String, Object>> typeMapTuple = XContentHelper.convertToMap(source, true);
407             sourceContentType = typeMapTuple.v1();
408             sourceAsMap = typeMapTuple.v2();
409             if (fetchSourceContext.transformSource()) {
410                 sourceAsMap = docMapper.transformSourceAsMap(sourceAsMap);
411             }
412             sourceAsMap = XContentMapValues.filter(sourceAsMap, fetchSourceContext.includes(), fetchSourceContext.excludes());
413             try {
414                 source = XContentFactory.contentBuilder(sourceContentType).map(sourceAsMap).bytes();
415             } catch (IOException e) {
416                 throw new ElasticsearchException("Failed to get type [" + type + "] and id [" + id + "] with includes/excludes set", e);
417             }
418         }
419 
420         return new GetResult(shardId.index().name(), type, id, get.version(), get.exists(), source, fields);
421     }

423     private static FieldsVisitor buildFieldsVisitors(String[] fields, FetchSourceContext fetchSourceContext) {
424         if (fields == null || fields.length == 0) {
425             return fetchSourceContext.fetchSource() ? new JustSourceFieldsVisitor() : null;
426         }
427 
428         return new CustomFieldsVisitor(Sets.newHashSet(fields), fetchSourceContext.fetchSource());
429     }
