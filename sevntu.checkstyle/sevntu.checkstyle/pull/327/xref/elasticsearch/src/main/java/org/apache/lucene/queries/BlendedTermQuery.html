32  /**
33   * BlendedTermQuery can be used to unify term statistics across
34   * one or more fields in the index. A common problem with structured
35   * documents is that a term that is significant in on field might not be
36   * significant in other fields like in a scenario where documents represent
37   * users with a "first_name" and a "second_name". When someone searches
38   * for "simon" it will very likely get "paul simon" first since "simon" is a
39   * an uncommon last name ie. has a low document frequency. This query
40   * tries to "lie" about the global statistics like document frequency as well
41   * total term frequency to rank based on the estimated statistics.
42   * <p>
43   * While aggregating the total term frequency is trivial since it
44   * can be summed up not every {@link org.apache.lucene.search.similarities.Similarity}
45   * makes use of this statistic. The document frequency which is used in the
46   * {@link org.apache.lucene.search.similarities.DefaultSimilarity}
47   * can only be estimated as an lower-bound since it is a document based statistic. For
48   * the document frequency the maximum frequency across all fields per term is used
49   * which is the minimum number of documents the terms occurs in.
50   * </p>
51   */
52  // TODO maybe contribute to Lucene
53  public abstract class BlendedTermQuery extends Query {
54  
55      private final Term[] terms;
56  
57  
58      public BlendedTermQuery(Term[] terms) {
59          if (terms == null || terms.length == 0) {
60              throw new IllegalArgumentException("terms must not be null or empty");
61          }
62          this.terms = terms;
63      }
64  
65      @Override
66      public Query rewrite(IndexReader reader) throws IOException {
67          IndexReaderContext context = reader.getContext();
68          TermContext[] ctx = new TermContext[terms.length];
69          int[] docFreqs = new int[ctx.length];
70          for (int i = 0; i < terms.length; i++) {
71              ctx[i] = TermContext.build(context, terms[i]);
72              docFreqs[i] = ctx[i].docFreq();
73          }
74  
75          final int maxDoc = reader.maxDoc();
76          blend(ctx, maxDoc, reader);
77          Query query = topLevelQuery(terms, ctx, docFreqs, maxDoc);
78          query.setBoost(getBoost());
79          return query;
80      }
81  
82      protected abstract Query topLevelQuery(Term[] terms, TermContext[] ctx, int[] docFreqs, int maxDoc);
83  
84      protected void blend(TermContext[] contexts, int maxDoc, IndexReader reader) throws IOException {
85          if (contexts.length <= 1) {
86              return;
87          }
88          int max = 0;
89          long minSumTTF = Long.MAX_VALUE;
90          for (int i = 0; i < contexts.length; i++) {
91              TermContext ctx = contexts[i];
92              int df = ctx.docFreq();
93              // we use the max here since it's the only "true" estimation we can make here
94              // at least max(df) documents have that term. Sum or Averages don't seem
95              // to have a significant meaning here.
96              // TODO: Maybe it could also make sense to assume independent distributions of documents and eg. have:
97              //   df = df1 + df2 - (df1 * df2 / maxDoc)?
98              max = Math.max(df, max);
99              if (minSumTTF != -1 && ctx.totalTermFreq() != -1) {
100                 // we need to find out the minimum sumTTF to adjust the statistics
101                 // otherwise the statistics don't match
102                 minSumTTF = Math.min(minSumTTF, reader.getSumTotalTermFreq(terms[i].field()));
103             } else {
104                 minSumTTF = -1;
105             }
106 
107         }
108         if (minSumTTF != -1 && maxDoc > minSumTTF) {
109             maxDoc = (int)minSumTTF;
110         }
111 
112         if (max == 0) {
113             return; // we are done that term doesn't exist at all
114         }
115         long sumTTF = minSumTTF == -1 ? -1 : 0;
116         final TermContext[] tieBreak = new TermContext[contexts.length];
117         System.arraycopy(contexts, 0, tieBreak, 0, contexts.length);
118         ArrayUtil.timSort(tieBreak, new Comparator<TermContext>() {
119             @Override
120             public int compare(TermContext o1, TermContext o2) {
121                 return Ints.compare(o2.docFreq(), o1.docFreq());
122             }
123         });
124         int prev = tieBreak[0].docFreq();
125         int actualDf = Math.min(maxDoc, max);
126         assert actualDf >=0 : "DF must be >= 0";
127 
128 
129         // here we try to add a little bias towards
130         // the more popular (more frequent) fields
131         // that acts as a tie breaker
132         for (TermContext ctx : tieBreak) {
133             if (ctx.docFreq() == 0) {
134                 break;
135             }
136             final int current = ctx.docFreq();
137             if (prev > current) {
138                 actualDf++;
139             }
140             ctx.setDocFreq(Math.min(maxDoc, actualDf));
141             prev = current;
142             if (sumTTF >= 0 && ctx.totalTermFreq() >= 0) {
143                 sumTTF += ctx.totalTermFreq();
144             } else {
145                 sumTTF = -1;  // omit once TF is omitted anywhere!
146             }
147         }
148         sumTTF = Math.min(sumTTF, minSumTTF);
149         for (int i = 0; i < contexts.length; i++) {
150             int df = contexts[i].docFreq();
151             if (df == 0) {
152                 continue;
153             }
154             // the blended sumTTF can't be greater than the sumTTTF on the field
155             final long fixedTTF = sumTTF == -1 ? -1 : sumTTF;
156             contexts[i] = adjustTTF(contexts[i], fixedTTF);
157         }
158     }
159 
160     private TermContext adjustTTF(TermContext termContext, long sumTTF) {
161         if (sumTTF == -1 && termContext.totalTermFreq() == -1) {
162             return termContext;
163         }
164         TermContext newTermContext = new TermContext(termContext.topReaderContext);
165         List<LeafReaderContext> leaves = termContext.topReaderContext.leaves();
166         final int len;
167         if (leaves == null) {
168             len = 1;
169         } else {
170             len = leaves.size();
171         }
172         int df = termContext.docFreq();
173         long ttf = sumTTF;
174         for (int i = 0; i < len; i++) {
175             TermState termState = termContext.get(i);
176             if (termState == null) {
177                 continue;
178             }
179             newTermContext.register(termState, i, df, ttf);
180             df = 0;
181             ttf = 0;
182         }
183         return newTermContext;
184     }
185 
186     @Override
187     public String toString(String field) {
188         return "blended(terms: " + Arrays.toString(terms) + ")";
189 
190     }
191 
192     @Override
193     public void extractTerms(Set<Term> terms) {
194         for (Term term : this.terms) {
195             terms.add(term);
196         }
197     }
198 
199     private volatile Term[] equalTerms = null;
200 
201     private Term[] equalsTerms() {
202         if (terms.length == 1) {
203             return terms;
204         }
205         if (equalTerms == null) {
206             // sort the terms to make sure equals and hashCode are consistent
207             // this should be a very small cost and equivalent to a HashSet but less object creation
208             final Term[] t = new Term[terms.length];
209             System.arraycopy(terms, 0, t, 0, terms.length);
210             ArrayUtil.timSort(t);
211             equalTerms = t;
212         }
213         return equalTerms;
214 
215     }
216 
217     @Override
218     public boolean equals(Object o) {
219         if (this == o) return true;
220         if (o == null || getClass() != o.getClass()) return false;
221         if (!super.equals(o)) return false;
222 
223         BlendedTermQuery that = (BlendedTermQuery) o;
224         if (!Arrays.equals(equalsTerms(), that.equalsTerms())) return false;
225 
226         return true;
227     }
228 
229     @Override
230     public int hashCode() {
231         int result = super.hashCode();
232         result = 31 * result + Arrays.hashCode(equalsTerms());
233         return result;
234     }

236     public static BlendedTermQuery booleanBlendedQuery(Term[] terms, final boolean disableCoord) {
237         return booleanBlendedQuery(terms, null, disableCoord);
238     }

240     public static BlendedTermQuery booleanBlendedQuery(Term[] terms, final float[] boosts, final boolean disableCoord) {
241         return new BlendedTermQuery(terms) {
242             @Override
243             protected Query topLevelQuery(Term[] terms, TermContext[] ctx, int[] docFreqs, int maxDoc) {
244                 BooleanQuery query = new BooleanQuery(disableCoord);
245                 for (int i = 0; i < terms.length; i++) {
246                     TermQuery termQuery = new TermQuery(terms[i], ctx[i]);
247                     if (boosts != null) {
248                         termQuery.setBoost(boosts[i]);
249                     }
250                     query.add(termQuery, BooleanClause.Occur.SHOULD);
251                 }
252                 return query;
253             }
254         };
255     }

257     public static BlendedTermQuery commonTermsBlendedQuery(Term[] terms, final float[] boosts, final boolean disableCoord, final float maxTermFrequency) {
258         return new BlendedTermQuery(terms) {
259             @Override
260             protected Query topLevelQuery(Term[] terms, TermContext[] ctx, int[] docFreqs, int maxDoc) {
261                 BooleanQuery query = new BooleanQuery(true);
262                 BooleanQuery high = new BooleanQuery(disableCoord);
263                 BooleanQuery low = new BooleanQuery(disableCoord);
264                 for (int i = 0; i < terms.length; i++) {
265                     TermQuery termQuery = new TermQuery(terms[i], ctx[i]);
266                     if (boosts != null) {
267                         termQuery.setBoost(boosts[i]);
268                     }
269                     if ((maxTermFrequency >= 1f && docFreqs[i] > maxTermFrequency)
270                             || (docFreqs[i] > (int) Math.ceil(maxTermFrequency
271                             * (float) maxDoc))) {
272                         high.add(termQuery, BooleanClause.Occur.SHOULD);
273                     } else {
274                         low.add(termQuery, BooleanClause.Occur.SHOULD);
275                     }
276                 }
277                 if (low.clauses().isEmpty()) {
278                     for (BooleanClause booleanClause : high) {
279                         booleanClause.setOccur(BooleanClause.Occur.MUST);
280                     }
281                     return high;
282                 } else if (high.clauses().isEmpty()) {
283                     return low;
284                 } else {
285                     query.add(high, BooleanClause.Occur.SHOULD);
286                     query.add(low, BooleanClause.Occur.MUST);
287                     return query;
288                 }
289             }
290         };
291     }

297     public static BlendedTermQuery dismaxBlendedQuery(Term[] terms, final float[] boosts, final float tieBreakerMultiplier) {
298         return new BlendedTermQuery(terms) {
299             @Override
300             protected Query topLevelQuery(Term[] terms, TermContext[] ctx, int[] docFreqs, int maxDoc) {
301                 DisjunctionMaxQuery query = new DisjunctionMaxQuery(tieBreakerMultiplier);
302                 for (int i = 0; i < terms.length; i++) {
303                     TermQuery termQuery = new TermQuery(terms[i], ctx[i]);
304                     if (boosts != null) {
305                         termQuery.setBoost(boosts[i]);
306                     }
307                     query.add(termQuery);
308                 }
309                 return query;
310             }
311         };
