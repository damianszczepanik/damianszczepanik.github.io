142     @Override
143     protected void doStop() throws ElasticsearchException {
144         ImmutableSet<String> indices = ImmutableSet.copyOf(this.indices.keySet());
145         final CountDownLatch latch = new CountDownLatch(indices.size());
146 
147         final ExecutorService indicesStopExecutor = Executors.newFixedThreadPool(5, EsExecutors.daemonThreadFactory("indices_shutdown"));
148 
149         for (final String index : indices) {
150             indicesStopExecutor.execute(new Runnable() {
151                 @Override
152                 public void run() {
153                     try {
154                         removeIndex(index, "shutdown", false);
155                     } catch (Throwable e) {
156                         logger.warn("failed to remove index on stop [" + index + "]", e);
157                     } finally {
158                         latch.countDown();
159                     }
160                 }
161             });
162         }
163         try {
164             if (latch.await(30, TimeUnit.SECONDS) == false) {
165               logger.warn("Not all shards are closed yet, waited 30sec - stopping service");
166             }
167         } catch (InterruptedException e) {
168             // ignore
169         } finally {
170             indicesStopExecutor.shutdown();
171         }
172     }

193     public NodeIndicesStats stats(boolean includePrevious, CommonStatsFlags flags) {
194         CommonStats oldStats = new CommonStats(flags);
195 
196         if (includePrevious) {
197             Flag[] setFlags = flags.getFlags();
198             for (Flag flag : setFlags) {
199                 switch (flag) {
200                     case Get:
201                         oldStats.get.add(oldShardsStats.getStats);
202                         break;
203                     case Indexing:
204                         oldStats.indexing.add(oldShardsStats.indexingStats);
205                         break;
206                     case Search:
207                         oldStats.search.add(oldShardsStats.searchStats);
208                         break;
209                     case Merge:
210                         oldStats.merge.add(oldShardsStats.mergeStats);
211                         break;
212                     case Refresh:
213                         oldStats.refresh.add(oldShardsStats.refreshStats);
214                         break;
215                     case Flush:
216                         oldStats.flush.add(oldShardsStats.flushStats);
217                         break;
218                 }
219             }
220         }
221 
222         Map<Index, List<IndexShardStats>> statsByShard = Maps.newHashMap();
223         for (Tuple<IndexService, Injector> value : indices.values()) {
224             IndexService indexService = value.v1();
225             for (IndexShard indexShard : indexService) {
226                 try {
227                     if (indexShard.routingEntry() == null) {
228                         continue;
229                     }
230                     IndexShardStats indexShardStats = new IndexShardStats(indexShard.shardId(), new ShardStats[] { new ShardStats(indexShard, indexShard.routingEntry(), flags) });
231                     if (!statsByShard.containsKey(indexService.index())) {
232                         statsByShard.put(indexService.index(), Lists.<IndexShardStats>newArrayList(indexShardStats));
233                     } else {
234                         statsByShard.get(indexService.index()).add(indexShardStats);
235                     }
236                 } catch (IllegalIndexShardStateException e) {
237                     // we can safely ignore illegal state on ones that are closing for example
238                 }
239             }
240         }
241         return new NodeIndicesStats(oldStats, statsByShard);
242     }

291     public synchronized IndexService createIndex(String sIndexName, @IndexSettings Settings settings, String localNodeId) throws ElasticsearchException {
292         if (!lifecycle.started()) {
293             throw new ElasticsearchIllegalStateException("Can't create an index [" + sIndexName + "], node is closed");
294         }
295         Index index = new Index(sIndexName);
296         if (indices.containsKey(index.name())) {
297             throw new IndexAlreadyExistsException(index);
298         }
299 
300         indicesLifecycle.beforeIndexCreated(index, settings);
301 
302         logger.debug("creating Index [{}], shards [{}]/[{}]", sIndexName, settings.get(SETTING_NUMBER_OF_SHARDS), settings.get(SETTING_NUMBER_OF_REPLICAS));
303 
304         Settings indexSettings = settingsBuilder()
305                 .put(this.settings)
306                 .put(settings)
307                 .classLoader(settings.getClassLoader())
308                 .build();
309 
310         ModulesBuilder modules = new ModulesBuilder();
311         modules.add(new IndexNameModule(index));
312         modules.add(new LocalNodeIdModule(localNodeId));
313         modules.add(new IndexSettingsModule(index, indexSettings));
314         modules.add(new IndexPluginsModule(indexSettings, pluginsService));
315         modules.add(new IndexStoreModule(indexSettings));
316         modules.add(new AnalysisModule(indexSettings, indicesAnalysisService));
317         modules.add(new SimilarityModule(indexSettings));
318         modules.add(new IndexCacheModule(indexSettings));
319         modules.add(new IndexFieldDataModule(indexSettings));
320         modules.add(new MapperServiceModule());
321         modules.add(new IndexQueryParserModule(indexSettings));
322         modules.add(new IndexAliasesServiceModule());
323         modules.add(new IndexModule(indexSettings));
324 
325         Injector indexInjector;
326         try {
327             indexInjector = modules.createChildInjector(injector);
328         } catch (CreationException e) {
329             throw new IndexCreationException(index, Injectors.getFirstErrorFailure(e));
330         } catch (Throwable e) {
331             throw new IndexCreationException(index, e);
332         }
333 
334         IndexService indexService = indexInjector.getInstance(IndexService.class);
335 
336         indicesLifecycle.afterIndexCreated(indexService);
337 
338         indices = newMapBuilder(indices).put(index.name(), new Tuple<>(indexService, indexInjector)).immutableMap();
339 
340         return indexService;
341     }

353     private void removeIndex(String index, String reason, boolean delete) throws ElasticsearchException {
354         try {
355             final IndexService indexService;
356             final Injector indexInjector;
357             synchronized (this) {
358                 if (indices.containsKey(index) == false) {
359                     return;
360                 }
361 
362                 logger.debug("[{}] closing ... (reason [{}])", index, reason);
363                 Map<String, Tuple<IndexService, Injector>> tmpMap = newHashMap(indices);
364                 Tuple<IndexService, Injector> remove = tmpMap.remove(index);
365                 indexService = remove.v1();
366                 indexInjector = remove.v2();
367                 indices = ImmutableMap.copyOf(tmpMap);
368             }
369 
370             indicesLifecycle.beforeIndexClosed(indexService);
371             if (delete) {
372                 indicesLifecycle.beforeIndexDeleted(indexService);
373             }
374             IOUtils.close(Iterables.transform(pluginsService.indexServices(), new Function<Class<? extends Closeable>, Closeable>() {
375                 @Override
376                 public Closeable apply(Class<? extends Closeable> input) {
377                     return indexInjector.getInstance(input);
378                 }
379             }));
380 
381             logger.debug("[{}] closing index service (reason [{}])", index, reason);
382             indexService.close(reason, delete);
383 
384             logger.debug("[{}] closing index cache (reason [{}])", index, reason);
385             indexInjector.getInstance(IndexCache.class).close();
386             logger.debug("[{}] clearing index field data (reason [{}])", index, reason);
387             indexInjector.getInstance(IndexFieldDataService.class).clear();
388             logger.debug("[{}] closing analysis service (reason [{}])", index, reason);
389             indexInjector.getInstance(AnalysisService.class).close();
390 
391             logger.debug("[{}] closing mapper service (reason [{}])", index, reason);
392             indexInjector.getInstance(MapperService.class).close();
393             logger.debug("[{}] closing index query parser service (reason [{}])", index, reason);
394             indexInjector.getInstance(IndexQueryParserService.class).close();
395 
396             logger.debug("[{}] closing index service (reason [{}])", index, reason);
397             indexInjector.getInstance(IndexStore.class).close();
398 
399             logger.debug("[{}] closed... (reason [{}])", index, reason);
400             indicesLifecycle.afterIndexClosed(indexService.index(), indexService.settingsService().getSettings());
401             if (delete) {
402                 final Settings indexSettings = indexService.getIndexSettings();
403                 indicesLifecycle.afterIndexDeleted(indexService.index(), indexSettings);
404                 // now we are done - try to wipe data on disk if possible
405                 deleteIndexStore(reason, indexService.index(), indexSettings);
406             }
407         } catch (IOException ex) {
408             throw new ElasticsearchException("failed to remove index " + index, ex);
409         }
410     }

448     public void deleteClosedIndex(String reason, IndexMetaData metaData, ClusterState clusterState) {
449         if (nodeEnv.hasNodeFile()) {
450             String indexName = metaData.getIndex();
451             try {
452                 if (clusterState.metaData().hasIndex(indexName)) {
453                     final IndexMetaData index = clusterState.metaData().index(indexName);
454                     throw new ElasticsearchIllegalStateException("Can't delete closed index store for [" + indexName + "] - it's still part of the cluster state [" + index.getUUID() + "] [" + metaData.getUUID() + "]");
455                 }
456                 deleteIndexStore(reason, metaData, clusterState);
457             } catch (IOException e) {
458                 logger.warn("[{}] failed to delete closed index", e, metaData.index());
459             }
460         }
461     }

463     /**
464      * Deletes the index store trying to acquire all shards locks for this index.
465      * This method will delete the metadata for the index even if the actual shards can't be locked.
466      */
467     public void deleteIndexStore(String reason, IndexMetaData metaData, ClusterState clusterState) throws IOException {
468         if (nodeEnv.hasNodeFile()) {
469             synchronized (this) {
470                 String indexName = metaData.index();
471                 if (indices.containsKey(indexName)) {
472                     String localUUid = indices.get(indexName).v1().indexUUID();
473                     throw new ElasticsearchIllegalStateException("Can't delete index store for [" + indexName + "] - it's still part of the indices service [" + localUUid+ "] [" + metaData.getUUID() + "]");
474                 }
475                 if (clusterState.metaData().hasIndex(indexName) && (clusterState.nodes().localNode().masterNode() == true)) {
476                     // we do not delete the store if it is a master eligible node and the index is still in the cluster state
477                     // because we want to keep the meta data for indices around even if no shards are left here
478                     final IndexMetaData index = clusterState.metaData().index(indexName);
479                     throw new ElasticsearchIllegalStateException("Can't delete closed index store for [" + indexName + "] - it's still part of the cluster state [" + index.getUUID() + "] [" + metaData.getUUID() + "]");
480                 }
481             }
482             Index index = new Index(metaData.index());
483             final Settings indexSettings = buildIndexSettings(metaData);
484             deleteIndexStore(reason, index, indexSettings);
485         }
486     }

488     private void deleteIndexStore(String reason, Index index, Settings indexSettings) throws IOException {
489         boolean success = false;
490         try {
491             // we are trying to delete the index store here - not a big deal if the lock can't be obtained
492             // the store metadata gets wiped anyway even without the lock this is just best effort since
493             // every shards deletes its content under the shard lock it owns.
494             logger.debug("{} deleting index store reason [{}]", index, reason);
495             if (canDeleteIndexContents(index, indexSettings)) {
496                 nodeEnv.deleteIndexDirectorySafe(index, 0, indexSettings);
497             }
498             success = true;
499         } catch (LockObtainFailedException ex) {
500             logger.debug("{} failed to delete index store - at least one shards is still locked", ex, index);
501         } catch (Exception ex) {
502             logger.warn("{} failed to delete index", ex, index);
503         } finally {
504             if (success == false) {
505                 addPendingDelete(index, indexSettings);
506             }
507             // this is a pure protection to make sure this index doesn't get re-imported as a dangeling index.
508             // we should in the future rather write a tombstone rather than wiping the metadata.
509             MetaDataStateFormat.deleteMetaState(nodeEnv.indexPaths(index));
510         }
511     }

526     /**
527      * This method deletes the shard contents on disk for the given shard ID. This method will fail if the shard deleting
528      * is prevented by {@link #canDeleteShardContent(org.elasticsearch.index.shard.ShardId, org.elasticsearch.cluster.metadata.IndexMetaData)}
529      * of if the shards lock can not be acquired.
530      * @param reason the reason for the shard deletion
531      * @param shardId the shards ID to delete
532      * @param metaData the shards index metadata. This is required to access the indexes settings etc.
533      * @throws IOException if an IOException occurs
534      */
535     public void deleteShardStore(String reason, ShardId shardId, IndexMetaData metaData) throws IOException {
536         final Settings indexSettings = buildIndexSettings(metaData);
537         if (canDeleteShardContent(shardId, indexSettings) == false) {
538             throw new ElasticsearchIllegalStateException("Can't delete shard " + shardId);
539         }
540         nodeEnv.deleteShardDirectorySafe(shardId, indexSettings);
541         logger.trace("{} deleting shard reason [{}]", shardId, reason);
542     }

544     /**
545      * This method returns true if the current node is allowed to delete the
546      * given index. If the index uses a shared filesystem this method always
547      * returns false.
548      * @param index {@code Index} to check whether deletion is allowed
549      * @param indexSettings {@code Settings} for the given index
550      * @return true if the index can be deleted on this node
551      */
552     public boolean canDeleteIndexContents(Index index, Settings indexSettings) {
553         final Tuple<IndexService, Injector> indexServiceInjectorTuple = this.indices.get(index.name());
554         if (IndexMetaData.isOnSharedFilesystem(indexSettings) == false) {
555             if (indexServiceInjectorTuple == null && nodeEnv.hasNodeFile()) {
556                 return true;
557             }
558         } else {
559             logger.trace("{} skipping index directory deletion due to shadow replicas", index);
560         }
561         return false;
562     }

564     /**
565      * Returns <code>true</code> iff the shards content for the given shard can be deleted.
566      * This method will return <code>false</code> if:
567      * <ul>
568      *     <li>if the shard is still allocated / active on this node</li>
569      *     <li>if for instance if the shard is located on shared and should not be deleted</li>
570      *     <li>if the shards data locations do not exists</li>
571      * </ul>
572      *
573      * @param shardId the shard to delete.
574      * @param metaData the shards index metadata. This is required to access the indexes settings etc.
575      */
576     public boolean canDeleteShardContent(ShardId shardId, IndexMetaData metaData) {
577         // we need the metadata here since we have to build the complete settings
578         // to decide where the shard content lives. In the future we might even need more info here ie. for shadow replicas
579         // The plan was to make it harder to miss-use and ask for metadata instead of simple settings
580         assert shardId.getIndex().equals(metaData.getIndex());
581         final Settings indexSettings = buildIndexSettings(metaData);
582         return canDeleteShardContent(shardId, indexSettings);
583     }

585     private boolean canDeleteShardContent(ShardId shardId, @IndexSettings Settings indexSettings) {
586         final Tuple<IndexService, Injector> indexServiceInjectorTuple = this.indices.get(shardId.getIndex());
587         if (IndexMetaData.isOnSharedFilesystem(indexSettings) == false) {
588             if (indexServiceInjectorTuple != null && nodeEnv.hasNodeFile()) {
589                 final IndexService indexService = indexServiceInjectorTuple.v1();
590                 return indexService.hasShard(shardId.id()) == false;
591             } else if (nodeEnv.hasNodeFile()) {
592                 final Path[] shardLocations = nodeEnv.shardDataPaths(shardId, indexSettings);
593                 return FileSystemUtils.exists(shardLocations);
594             }
595         } else {
596             logger.trace("{} skipping shard directory deletion due to shadow replicas", shardId);
597         }
598         return false;
599     }

601     private Settings buildIndexSettings(IndexMetaData metaData) {
602         // play safe here and make sure that we take node level settings into account.
603         // we might run on nodes where we use shard FS and then in the future don't delete
604         // actual content.
605         ImmutableSettings.Builder builder = settingsBuilder();
606         builder.put(settings);
607         builder.put(metaData.getSettings());
608         return builder.build();

625     private void addPendingDelete(Index index, PendingDelete pendingDelete) {
626         synchronized (pendingDeletes) {
627             List<PendingDelete> list = pendingDeletes.get(index);
628             if (list == null) {
629                 list = new ArrayList<>();
630                 pendingDeletes.put(index, list);
631             }
632             list.add(pendingDelete);
633         }
634     }

644     private static final class PendingDelete implements Comparable<PendingDelete> {
645         final ShardId shardId;
646         final Settings settings;
647         final boolean deleteIndex;
648 
649         public PendingDelete(ShardId shardId, Settings settings, boolean deleteIndex) {
650             this.shardId = shardId;
651             this.settings = settings;
652             this.deleteIndex = deleteIndex;
653             assert deleteIndex || shardId != null;
654         }
655 
656         @Override
657         public String toString() {
658             return shardId.toString();
659         }
660 
661         @Override
662         public int compareTo(PendingDelete o) {
663             int left = deleteIndex ? -1 : shardId.id();
664             int right = o.deleteIndex ? -1 : o.shardId.id();
665             return Integer.compare(left, right);
666         }
667     }

669     /**
670      * Processes all pending deletes for the given index. This method will acquire all locks for the given index and will
671      * process all pending deletes for this index. Pending deletes might occur if the OS doesn't allow deletion of files because
672      * they are used by a different process ie. on Windows where files might still be open by a virus scanner. On a shared
673      * filesystem a replica might not have been closed when the primary is deleted causing problems on delete calls so we
674      * schedule there deletes later.
675      * @param index the index to process the pending deletes for
676      * @param timeout the timeout used for processing pending deletes
677      */
678     public void processPendingDeletes(Index index, @IndexSettings Settings indexSettings, TimeValue timeout) throws IOException {
679         logger.debug("{} processing pending deletes", index);
680         final long startTime = System.currentTimeMillis();
681         final List<ShardLock> shardLocks = nodeEnv.lockAllForIndex(index, indexSettings, timeout.millis());
682         try {
683             Map<ShardId, ShardLock> locks = new HashMap<>();
684             for (ShardLock lock : shardLocks) {
685                 locks.put(lock.getShardId(), lock);
686             }
687             final List<PendingDelete> remove;
688             synchronized (pendingDeletes) {
689                  remove = pendingDeletes.remove(index);
690             }
691             if (remove != null && remove.isEmpty() == false) {
692                 CollectionUtil.timSort(remove); // make sure we delete indices first
693                 final long maxSleepTimeMs = 10 * 1000; // ensure we retry after 10 sec
694                 long sleepTime = 10;
695                 do {
696                     if (remove.isEmpty()) {
697                         break;
698                     }
699                     Iterator<PendingDelete> iterator = remove.iterator();
700                     while (iterator.hasNext()) {
701                         PendingDelete delete = iterator.next();
702 
703                         if (delete.deleteIndex) {
704                             logger.debug("{} deleting index store reason [{}]", index, "pending delete");
705                             try {
706                                 nodeEnv.deleteIndexDirectoryUnderLock(index, indexSettings);
707                                 iterator.remove();
708                             } catch (IOException ex) {
709                                 logger.debug("{} retry pending delete", ex, index);
710                             }
711                         } else {
712                             ShardLock shardLock = locks.get(delete.shardId);
713                             if (shardLock != null) {
714                                 try {
715                                     deleteShardStore("pending delete", shardLock, delete.settings);
716                                     iterator.remove();
717                                 } catch (IOException ex) {
718                                     logger.debug("{} retry pending delete", ex, shardLock.getShardId());
719                                 }
720                             } else {
721                                 logger.warn("{} no shard lock for pending delete", delete.shardId);
722                                 iterator.remove();
723                             }
724                         }
725                     }
726                     if (remove.isEmpty() == false) {
727                         logger.warn("{} still pending deletes present for shards {} - retrying", index, remove.toString());
728                         try {
729                             Thread.sleep(sleepTime);
730                             sleepTime = Math.min(maxSleepTimeMs, sleepTime * 2); // increase the sleep time gradually
731                             logger.debug("{} schedule pending delete retry after {} ms", index, sleepTime);
732                         } catch (InterruptedException e) {
733                             Thread.interrupted();
734                             return;
735                         }
736                     }
737                 } while ((System.currentTimeMillis() - startTime) < timeout.millis());
738             }
739         } finally {
740             IOUtils.close(shardLocks);
741         }
742     }

744     int numPendingDeletes(Index index) {
745         synchronized (pendingDeletes) {
746             List<PendingDelete> deleteList = pendingDeletes.get(index);
747             if (deleteList == null) {
748                 return 0;
749             }
750             return deleteList.size();
751         }
