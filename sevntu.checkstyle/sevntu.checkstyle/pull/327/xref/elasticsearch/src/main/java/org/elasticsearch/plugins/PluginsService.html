95      /**
96       * Constructs a new PluginService
97       * @param settings The settings of the system
98       * @param environment The environment of the system
99       */
100     public PluginsService(Settings settings, Environment environment) {
101         super(settings);
102         this.environment = environment;
103         this.checkLucene = settings.getAsBoolean(PLUGINS_CHECK_LUCENE_KEY, true);
104         this.esPluginPropertiesFile = settings.get(ES_PLUGIN_PROPERTIES_FILE_KEY, ES_PLUGIN_PROPERTIES);
105         this.loadClasspathPlugins = settings.getAsBoolean(LOAD_PLUGIN_FROM_CLASSPATH, true);
106 
107         ImmutableList.Builder<Tuple<PluginInfo, Plugin>> tupleBuilder = ImmutableList.builder();
108 
109         // first we load all the default plugins from the settings
110         String[] defaultPluginsClasses = settings.getAsArray("plugin.types");
111         for (String pluginClass : defaultPluginsClasses) {
112             Plugin plugin = loadPlugin(pluginClass, settings);
113             PluginInfo pluginInfo = new PluginInfo(plugin.name(), plugin.description(), hasSite(plugin.name()), true, PluginInfo.VERSION_NOT_AVAILABLE);
114             if (logger.isTraceEnabled()) {
115                 logger.trace("plugin loaded from settings [{}]", pluginInfo);
116             }
117             tupleBuilder.add(new Tuple<>(pluginInfo, plugin));
118         }
119 
120         // now, find all the ones that are in the classpath
121         try {
122             loadPluginsIntoClassLoader();
123         } catch (IOException ex) {
124             throw new ElasticsearchIllegalStateException("Can't load plugins into classloader", ex);
125         }
126         if (loadClasspathPlugins) {
127             tupleBuilder.addAll(loadPluginsFromClasspath(settings));
128         }
129         this.plugins = tupleBuilder.build();
130 
131         // We need to build a List of jvm and site plugins for checking mandatory plugins
132         Map<String, Plugin> jvmPlugins = Maps.newHashMap();
133         List<String> sitePlugins = Lists.newArrayList();
134 
135         for (Tuple<PluginInfo, Plugin> tuple : this.plugins) {
136             jvmPlugins.put(tuple.v2().name(), tuple.v2());
137             if (tuple.v1().isSite()) {
138                 sitePlugins.add(tuple.v1().getName());
139             }
140         }
141         try {
142             // we load site plugins
143             ImmutableList<Tuple<PluginInfo, Plugin>> tuples = loadSitePlugins();
144             for (Tuple<PluginInfo, Plugin> tuple : tuples) {
145                 sitePlugins.add(tuple.v1().getName());
146             }
147         } catch (IOException ex) {
148             throw new ElasticsearchIllegalStateException("Can't load site  plugins", ex);
149         }
150 
151         // Checking expected plugins
152         String[] mandatoryPlugins = settings.getAsArray("plugin.mandatory", null);
153         if (mandatoryPlugins != null) {
154             Set<String> missingPlugins = Sets.newHashSet();
155             for (String mandatoryPlugin : mandatoryPlugins) {
156                 if (!jvmPlugins.containsKey(mandatoryPlugin) && !sitePlugins.contains(mandatoryPlugin) && !missingPlugins.contains(mandatoryPlugin)) {
157                     missingPlugins.add(mandatoryPlugin);
158                 }
159             }
160             if (!missingPlugins.isEmpty()) {
161                 throw new ElasticsearchException("Missing mandatory plugins [" + Strings.collectionToDelimitedString(missingPlugins, ", ") + "]");
162             }
163         }
164 
165         logger.info("loaded {}, sites {}", jvmPlugins.keySet(), sitePlugins);
166 
167         MapBuilder<Plugin, List<OnModuleReference>> onModuleReferences = MapBuilder.newMapBuilder();
168         for (Plugin plugin : jvmPlugins.values()) {
169             List<OnModuleReference> list = Lists.newArrayList();
170             for (Method method : plugin.getClass().getDeclaredMethods()) {
171                 if (!method.getName().equals("onModule")) {
172                     continue;
173                 }
174                 if (method.getParameterTypes().length == 0 || method.getParameterTypes().length > 1) {
175                     logger.warn("Plugin: {} implementing onModule with no parameters or more than one parameter", plugin.name());
176                     continue;
177                 }
178                 Class moduleClass = method.getParameterTypes()[0];
179                 if (!Module.class.isAssignableFrom(moduleClass)) {
180                     logger.warn("Plugin: {} implementing onModule by the type is not of Module type {}", plugin.name(), moduleClass);
181                     continue;
182                 }
183                 method.setAccessible(true);
184                 list.add(new OnModuleReference(moduleClass, method));
185             }
186             if (!list.isEmpty()) {
187                 onModuleReferences.put(plugin, list);
188             }
189         }
190         this.onModuleReferences = onModuleReferences.immutableMap();
191 
192         this.refreshInterval = settings.getAsTime(PLUGINS_INFO_REFRESH_INTERVAL_KEY, TimeValue.timeValueSeconds(10));
193     }

205     public void processModule(Module module) {
206         for (Tuple<PluginInfo, Plugin> plugin : plugins()) {
207             plugin.v2().processModule(module);
208             // see if there are onModule references
209             List<OnModuleReference> references = onModuleReferences.get(plugin.v2());
210             if (references != null) {
211                 for (OnModuleReference reference : references) {
212                     if (reference.moduleClass.isAssignableFrom(module.getClass())) {
213                         try {
214                             reference.onModuleMethod.invoke(plugin.v2(), module);
215                         } catch (Exception e) {
216                             logger.warn("plugin {}, failed to invoke custom onModule method", e, plugin.v2().name());
217                         }
218                     }
219                 }
220             }
221         }
222     }

305     /**
306      * Get information about plugins (jvm and site plugins).
307      * Information are cached for 10 seconds by default. Modify `plugins.info_refresh_interval` property if needed.
308      * Setting `plugins.info_refresh_interval` to `-1` will cause infinite caching.
309      * Setting `plugins.info_refresh_interval` to `0` will disable caching.
310      * @return List of plugins information
311      */
312     synchronized public PluginsInfo info() {
313         if (refreshInterval.millis() != 0) {
314             if (cachedPluginsInfo != null &&
315                     (refreshInterval.millis() < 0 || (System.currentTimeMillis() - lastRefresh) < refreshInterval.millis())) {
316                 if (logger.isTraceEnabled()) {
317                     logger.trace("using cache to retrieve plugins info");
318                 }
319                 return cachedPluginsInfo;
320             }
321             lastRefresh = System.currentTimeMillis();
322         }
323 
324         if (logger.isTraceEnabled()) {
325             logger.trace("starting to fetch info on plugins");
326         }
327         cachedPluginsInfo = new PluginsInfo();
328 
329         // We first add all JvmPlugins
330         for (Tuple<PluginInfo, Plugin> plugin : this.plugins) {
331             if (logger.isTraceEnabled()) {
332                 logger.trace("adding jvm plugin [{}]", plugin.v1());
333             }
334             cachedPluginsInfo.add(plugin.v1());
335         }
336 
337         try {
338             // We reload site plugins (in case of some changes)
339             for (Tuple<PluginInfo, Plugin> plugin : loadSitePlugins()) {
340                 if (logger.isTraceEnabled()) {
341                     logger.trace("adding site plugin [{}]", plugin.v1());
342                 }
343                 cachedPluginsInfo.add(plugin.v1());
344             }
345         } catch (IOException ex) {
346             logger.warn("can load site plugins info", ex);
347         }
348 
349         return cachedPluginsInfo;
350     }

352     private void loadPluginsIntoClassLoader() throws IOException {
353         Path pluginsDirectory = environment.pluginsFile();
354         if (!isAccessibleDirectory(pluginsDirectory, logger)) {
355             return;
356         }
357 
358         ClassLoader classLoader = settings.getClassLoader();
359         Class classLoaderClass = classLoader.getClass();
360         Method addURL = null;
361         while (!classLoaderClass.equals(Object.class)) {
362             try {
363                 addURL = classLoaderClass.getDeclaredMethod("addURL", URL.class);
364                 addURL.setAccessible(true);
365                 break;
366             } catch (NoSuchMethodException e) {
367                 // no method, try the parent
368                 classLoaderClass = classLoaderClass.getSuperclass();
369             }
370         }
371         if (addURL == null) {
372             logger.debug("failed to find addURL method on classLoader [" + classLoader + "] to add methods");
373             return;
374         }
375         try (DirectoryStream<Path> stream = Files.newDirectoryStream(pluginsDirectory)) {
376 
377             for (Path plugin : stream) {
378                 // We check that subdirs are directories and readable
379                 if (!isAccessibleDirectory(plugin, logger)) {
380                     continue;
381                 }
382 
383                 logger.trace("--- adding plugin [{}]", plugin.toAbsolutePath());
384 
385                 try {
386                     // add the root
387                     addURL.invoke(classLoader, plugin.toUri().toURL());
388                     // gather files to add
389                     List<Path> libFiles = Lists.newArrayList();
390                     libFiles.addAll(Arrays.asList(files(plugin)));
391                     Path libLocation = plugin.resolve("lib");
392                     if (Files.isDirectory(libLocation)) {
393                         libFiles.addAll(Arrays.asList(files(libLocation)));
394                     }
395 
396                     // if there are jars in it, add it as well
397                     for (Path libFile : libFiles) {
398                         if (!hasLibExtension(libFile)) {
399                             continue;
400                         }
401                         addURL.invoke(classLoader, libFile.toUri().toURL());
402                     }
403                 } catch (Throwable e) {
404                     logger.warn("failed to add plugin [" + plugin + "]", e);
405                 }
406             }
407         }
408     }
409 
410     protected static boolean hasLibExtension(Path lib) {
411         return PLUGIN_LIB_MATCHER.matches(lib);
412     }

414     private Path[] files(Path from) throws IOException {
415         try (DirectoryStream<Path> stream = Files.newDirectoryStream(from)) {
416             return Iterators.toArray(stream.iterator(), Path.class);
417         }

420     private ImmutableList<Tuple<PluginInfo,Plugin>> loadPluginsFromClasspath(Settings settings) {
421         ImmutableList.Builder<Tuple<PluginInfo, Plugin>> plugins = ImmutableList.builder();
422 
423         // Trying JVM plugins: looking for es-plugin.properties files
424         try {
425             Enumeration<URL> pluginUrls = settings.getClassLoader().getResources(esPluginPropertiesFile);
426             while (pluginUrls.hasMoreElements()) {
427                 URL pluginUrl = pluginUrls.nextElement();
428                 Properties pluginProps = new Properties();
429                 InputStream is = null;
430                 try {
431                     is = pluginUrl.openStream();
432                     pluginProps.load(is);
433                     String pluginClassName = pluginProps.getProperty("plugin");
434                     String pluginVersion = pluginProps.getProperty("version", PluginInfo.VERSION_NOT_AVAILABLE);
435                     Plugin plugin = loadPlugin(pluginClassName, settings);
436 
437                     // Is it a site plugin as well? Does it have also an embedded _site structure
438                     Path siteFile = environment.pluginsFile().resolve(plugin.name()).resolve("_site");
439                     boolean isSite = isAccessibleDirectory(siteFile, logger);
440                     if (logger.isTraceEnabled()) {
441                         logger.trace("found a jvm plugin [{}], [{}]{}",
442                                 plugin.name(), plugin.description(), isSite ? ": with _site structure" : "");
443                     }
444 
445                     PluginInfo pluginInfo = new PluginInfo(plugin.name(), plugin.description(), isSite, true, pluginVersion);
446 
447                     plugins.add(new Tuple<>(pluginInfo, plugin));
448                 } catch (Throwable e) {
449                     logger.warn("failed to load plugin from [" + pluginUrl + "]", e);
450                 } finally {
451                     IOUtils.closeWhileHandlingException(is);
452                 }
453             }
454         } catch (IOException e) {
455             logger.warn("failed to find jvm plugins from classpath", e);
456         }
457 
458         return plugins.build();
459     }

461     private ImmutableList<Tuple<PluginInfo,Plugin>> loadSitePlugins() throws IOException {
462         ImmutableList.Builder<Tuple<PluginInfo, Plugin>> sitePlugins = ImmutableList.builder();
463         List<String> loadedJvmPlugins = new ArrayList<>();
464 
465         // Already known jvm plugins are ignored
466         for(Tuple<PluginInfo, Plugin> tuple : plugins) {
467             if (tuple.v1().isSite()) {
468                 loadedJvmPlugins.add(tuple.v1().getName());
469             }
470         }
471 
472         // Let's try to find all _site plugins we did not already found
473         Path pluginsFile = environment.pluginsFile();
474 
475         if (FileSystemUtils.isAccessibleDirectory(pluginsFile, logger) == false) {
476             return sitePlugins.build();
477         }
478 
479         try (DirectoryStream<Path> stream = Files.newDirectoryStream(pluginsFile)) {
480             for (Path pluginFile : stream) {
481                 if (!loadedJvmPlugins.contains(pluginFile.getFileName().toString())) {
482                     Path sitePluginDir = pluginFile.resolve("_site");
483                     if (isAccessibleDirectory(sitePluginDir, logger)) {
484                         // We have a _site plugin. Let's try to get more information on it
485                         String name = pluginFile.getFileName().toString();
486                         String version = PluginInfo.VERSION_NOT_AVAILABLE;
487                         String description = PluginInfo.DESCRIPTION_NOT_AVAILABLE;
488 
489                         // We check if es-plugin.properties exists in plugin/_site dir
490                         final Path pluginPropFile = sitePluginDir.resolve(esPluginPropertiesFile);
491                         if (Files.exists(pluginPropFile)) {
492 
493                             final Properties pluginProps = new Properties();
494                             try (final BufferedReader reader = Files.newBufferedReader(pluginPropFile, Charsets.UTF_8)) {
495                                 pluginProps.load(reader);
496                                 description = pluginProps.getProperty("description", PluginInfo.DESCRIPTION_NOT_AVAILABLE);
497                                 version = pluginProps.getProperty("version", PluginInfo.VERSION_NOT_AVAILABLE);
498                             } catch (Exception e) {
499                                 // Can not load properties for this site plugin. Ignoring.
500                                 logger.debug("can not load {} file.", e, esPluginPropertiesFile);
501                             }
502                         }
503 
504                         if (logger.isTraceEnabled()) {
505                             logger.trace("found a site plugin name [{}], version [{}], description [{}]",
506                                     name, version, description);
507                         }
508                         sitePlugins.add(new Tuple<PluginInfo, Plugin>(new PluginInfo(name, description, true, false, version), null));
509                     }
510                 }
511             }
512         }
513         return sitePlugins.build();
514     }

516     /**
517      * @param name plugin name
518      * @return if this jvm plugin has also a _site structure
519      */
520     private boolean hasSite(String name) {
521         // Let's try to find all _site plugins we did not already found
522         Path pluginsFile = environment.pluginsFile();
523 
524         if (!Files.isDirectory(pluginsFile)) {
525             return false;
526         }
527 
528         Path sitePluginDir = pluginsFile.resolve(name).resolve("_site");
529         return isAccessibleDirectory(sitePluginDir, logger);
530     }

532     private Plugin loadPlugin(String className, Settings settings) {
533         try {
534             Class<? extends Plugin> pluginClass = (Class<? extends Plugin>) settings.getClassLoader().loadClass(className);
535             Plugin plugin;
536 
537             if (!checkLucene || checkLuceneCompatibility(pluginClass, settings, logger, esPluginPropertiesFile)) {
538                 try {
539                     plugin = pluginClass.getConstructor(Settings.class).newInstance(settings);
540                 } catch (NoSuchMethodException e) {
541                     try {
542                         plugin = pluginClass.getConstructor().newInstance();
543                     } catch (NoSuchMethodException e1) {
544                         throw new ElasticsearchException("No constructor for [" + pluginClass + "]. A plugin class must " +
545                                 "have either an empty default constructor or a single argument constructor accepting a " +
546                                 "Settings instance");
547                     }
548                 }
549             } else {
550                 throw new ElasticsearchException("Plugin is incompatible with the current node");
551             }
552 
553 
554             return plugin;
555 
556         } catch (Throwable e) {
557             throw new ElasticsearchException("Failed to load plugin class [" + className + "]", e);
558         }
559     }

561     /**
562      * <p>Check that a plugin is Lucene compatible with the current running node using `lucene` property
563      * in `es-plugin.properties` file.</p>
564      * <p>If plugin does not provide `lucene` property, we consider that the plugin is compatible.</p>
565      * <p>If plugin provides `lucene` property, we try to load related Enum org.apache.lucene.util.Version. If this
566      * fails, it means that the node is too "old" comparing to the Lucene version the plugin was built for.</p>
567      * <p>We compare then two first digits of current node lucene version against two first digits of plugin Lucene
568      * version. If not equal, it means that the plugin is too "old" for the current node.</p>
569      *
570      * @param pluginClass Plugin class we are checking
571      * @return true if the plugin is Lucene compatible
572      */
573     public static boolean checkLuceneCompatibility(Class<? extends Plugin> pluginClass, Settings settings, ESLogger logger, String propertiesFile) {
574         String luceneVersion = null;
575         try {
576             // We try to read the es-plugin.properties file
577             // But as we can have several plugins in the same classloader,
578             // we have to find the right es-plugin.properties file
579             Enumeration<URL> pluginUrls = settings.getClassLoader().getResources(propertiesFile);
580 
581             while (pluginUrls.hasMoreElements()) {
582                 URL pluginUrl = pluginUrls.nextElement();
583                 try (InputStream is = pluginUrl.openStream()) {
584                     Properties pluginProps = new Properties();
585                     pluginProps.load(is);
586                     String plugin = pluginProps.getProperty("plugin");
587                     // If we don't have the expected plugin, let's continue to the next one
588                     if (pluginClass.getName().equals(plugin)) {
589                         luceneVersion = pluginProps.getProperty("lucene");
590                         break;
591                     }
592                     logger.debug("skipping [{}]", pluginUrl);
593                 }
594             }
595 
596             if (luceneVersion != null) {
597                 // Should fail if the running node is too old!
598                 org.apache.lucene.util.Version luceneExpectedVersion = Lucene.parseVersionLenient(luceneVersion, null);
599                 if (Version.CURRENT.luceneVersion.equals(luceneExpectedVersion)) {
600                     logger.debug("starting analysis plugin for Lucene [{}].", luceneExpectedVersion);
601                     return true;
602                 }
603             } else {
604                 logger.debug("lucene property is not set in plugin {} file. Skipping test.", propertiesFile);
605                 return true;
606             }
607         } catch (Throwable t) {
608             // We don't have the expected version... Let's fail after.
609             logger.debug("exception raised while checking plugin Lucene version.", t);
610         }
611         logger.error("cannot start plugin due to incorrect Lucene version: plugin [{}], node [{}].",
612                 luceneVersion, Constants.LUCENE_MAIN_VERSION);
613         return false;
614     }
