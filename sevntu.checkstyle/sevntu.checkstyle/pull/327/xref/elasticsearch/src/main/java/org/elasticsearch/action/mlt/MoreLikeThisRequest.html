124     @Override
125     public List<? extends IndicesRequest> subRequests() {
126         //we create two fake indices subrequests as we don't have the actual ones yet
127         //since they get created later on in TransportMoreLikeThisAction
128         List<IndicesRequest> requests = Lists.newArrayList();
129         requests.add(new IndicesRequest() {
130             @Override
131             public String[] indices() {
132                 return new String[]{index};
133             }
134 
135             @Override
136             public IndicesOptions indicesOptions() {
137                 return MoreLikeThisRequest.this.indicesOptions();
138             }
139         });
140         requests.add(new IndicesRequest.Replaceable() {
141             @Override
142             public String[] indices() {
143                 if (searchIndices != null) {
144                     return searchIndices;
145                 }
146                 return new String[]{index};
147             }
148

150             public IndicesRequest indices(String[] indices) {
151                 searchIndices = indices;
152                 return this;
153             }
154 
155             @Override
156             public IndicesOptions indicesOptions() {
157                 return SearchRequest.DEFAULT_INDICES_OPTIONS;
158             }
159         });
160         return requests;
161     }

582     @Override
583     public ActionRequestValidationException validate() {
584         ActionRequestValidationException validationException = null;
585         if (index == null) {
586             validationException = ValidateActions.addValidationError("index is missing", validationException);
587         }
588         if (type == null) {
589             validationException = ValidateActions.addValidationError("type is missing", validationException);
590         }
591         if (id == null) {
592             validationException = ValidateActions.addValidationError("id is missing", validationException);
593         }
594         return validationException;
595     }

597     @Override
598     public void readFrom(StreamInput in) throws IOException {
599         super.readFrom(in);
600         index = in.readString();
601         type = in.readString();
602         id = in.readString();
603         // no need to pass threading over the network, they are always false when coming throw a thread pool
604         int size = in.readVInt();
605         if (size == 0) {
606             fields = Strings.EMPTY_ARRAY;
607         } else {
608             fields = new String[size];
609             for (int i = 0; i < size; i++) {
610                 fields[i] = in.readString();
611             }
612         }
613 
614         minimumShouldMatch(in.readString());
615 
616         minTermFreq = in.readVInt();
617         maxQueryTerms = in.readVInt();
618         size = in.readVInt();
619         if (size > 0) {
620             stopWords = new String[size];
621             for (int i = 0; i < size; i++) {
622                 stopWords[i] = in.readString();
623             }
624         }
625         minDocFreq = in.readVInt();
626         maxDocFreq = in.readVInt();
627         minWordLength = in.readVInt();
628         maxWordLength = in.readVInt();
629         boostTerms = in.readFloat();
630         include = in.readBoolean();
631 
632         searchType = SearchType.fromId(in.readByte());
633         size = in.readVInt();
634         if (size == 0) {
635             searchIndices = null;
636         } else if (size == 1) {
637             searchIndices = Strings.EMPTY_ARRAY;
638         } else {
639             searchIndices = new String[size - 1];
640             for (int i = 0; i < searchIndices.length; i++) {
641                 searchIndices[i] = in.readString();
642             }
643         }
644         size = in.readVInt();
645         if (size == 0) {
646             searchTypes = null;
647         } else if (size == 1) {
648             searchTypes = Strings.EMPTY_ARRAY;
649         } else {
650             searchTypes = new String[size - 1];
651             for (int i = 0; i < searchTypes.length; i++) {
652                 searchTypes[i] = in.readString();
653             }
654         }
655         if (in.readBoolean()) {
656             searchScroll = readScroll(in);
657         }
658 
659         searchSourceUnsafe = false;
660         searchSource = in.readBytesReference();
661 
662         searchSize = in.readVInt();
663         searchFrom = in.readVInt();
664         routing = in.readOptionalString();
665     }

667     @Override
668     public void writeTo(StreamOutput out) throws IOException {
669         super.writeTo(out);
670         out.writeString(index);
671         out.writeString(type);
672         out.writeString(id);
673         if (fields == null) {
674             out.writeVInt(0);
675         } else {
676             out.writeVInt(fields.length);
677             for (String field : fields) {
678                 out.writeString(field);
679             }
680         }
681 
682         out.writeString(minimumShouldMatch);
683 
684         out.writeVInt(minTermFreq);
685         out.writeVInt(maxQueryTerms);
686         if (stopWords == null) {
687             out.writeVInt(0);
688         } else {
689             out.writeVInt(stopWords.length);
690             for (String stopWord : stopWords) {
691                 out.writeString(stopWord);
692             }
693         }
694         out.writeVInt(minDocFreq);
695         out.writeVInt(maxDocFreq);
696         out.writeVInt(minWordLength);
697         out.writeVInt(maxWordLength);
698         out.writeFloat(boostTerms);
699         out.writeBoolean(include);
700 
701         out.writeByte(searchType.id());
702         if (searchIndices == null) {
703             out.writeVInt(0);
704         } else {
705             out.writeVInt(searchIndices.length + 1);
706             for (String index : searchIndices) {
707                 out.writeString(index);
708             }
709         }
710         if (searchTypes == null) {
711             out.writeVInt(0);
712         } else {
713             out.writeVInt(searchTypes.length + 1);
714             for (String type : searchTypes) {
715                 out.writeString(type);
716             }
717         }
718         if (searchScroll == null) {
719             out.writeBoolean(false);
720         } else {
721             out.writeBoolean(true);
722             searchScroll.writeTo(out);
723         }
724         out.writeBytesReference(searchSource);
725 
726         out.writeVInt(searchSize);
727         out.writeVInt(searchFrom);
728         out.writeOptionalString(routing);
729     }
