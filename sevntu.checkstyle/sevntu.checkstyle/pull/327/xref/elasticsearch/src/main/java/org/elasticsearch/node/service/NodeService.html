20  package org.elasticsearch.node.service;
21  
22  import com.google.common.collect.ImmutableMap;
23  import org.elasticsearch.Build;
24  import org.elasticsearch.Version;
25  import org.elasticsearch.action.admin.cluster.node.info.NodeInfo;
26  import org.elasticsearch.action.admin.cluster.node.stats.NodeStats;
27  import org.elasticsearch.action.admin.indices.stats.CommonStatsFlags;
28  import org.elasticsearch.common.Nullable;
29  import org.elasticsearch.common.collect.MapBuilder;
30  import org.elasticsearch.common.component.AbstractComponent;
31  import org.elasticsearch.common.inject.Inject;
32  import org.elasticsearch.common.settings.Settings;
33  import org.elasticsearch.discovery.Discovery;
34  import org.elasticsearch.http.HttpServer;
35  import org.elasticsearch.indices.IndicesService;
36  import org.elasticsearch.indices.breaker.CircuitBreakerService;
37  import org.elasticsearch.monitor.MonitorService;
38  import org.elasticsearch.plugins.PluginsService;
39  import org.elasticsearch.threadpool.ThreadPool;
40  import org.elasticsearch.transport.TransportService;
41  
42  import java.io.IOException;
43  
44  /**
45   */
46  public class NodeService extends AbstractComponent {
47  
48      private final ThreadPool threadPool;
49      private final MonitorService monitorService;
50      private final TransportService transportService;
51      private final IndicesService indicesService;
52      private final PluginsService pluginService;
53      private final CircuitBreakerService circuitBreakerService;
54      @Nullable
55      private HttpServer httpServer;
56  
57      private volatile ImmutableMap<String, String> serviceAttributes = ImmutableMap.of();
58  
59      private final Version version;
60  
61      private final Discovery discovery;
62  
63      @Inject
64      public NodeService(Settings settings, ThreadPool threadPool, MonitorService monitorService, Discovery discovery,
65                         TransportService transportService, IndicesService indicesService,
66                         PluginsService pluginService, CircuitBreakerService circuitBreakerService, Version version) {
67          super(settings);
68          this.threadPool = threadPool;
69          this.monitorService = monitorService;
70          this.transportService = transportService;
71          this.indicesService = indicesService;
72          this.discovery = discovery;
73          discovery.setNodeService(this);
74          this.version = version;
75          this.pluginService = pluginService;
76          this.circuitBreakerService = circuitBreakerService;
77      }
78  
79      public void setHttpServer(@Nullable HttpServer httpServer) {
80          this.httpServer = httpServer;
81      }
82  
83      @Deprecated
84      public void putNodeAttribute(String key, String value) {
85          putAttribute(key, value);
86      }
87  
88      @Deprecated
89      public void removeNodeAttribute(String key) {
90          removeAttribute(key);
91      }
92  
93      public synchronized void putAttribute(String key, String value) {
94          serviceAttributes = new MapBuilder<>(serviceAttributes).put(key, value).immutableMap();
95      }
96  
97      public synchronized void removeAttribute(String key) {
98          serviceAttributes = new MapBuilder<>(serviceAttributes).remove(key).immutableMap();
99      }
100 
101     /**
102      * Attributes different services in the node can add to be reported as part of the node info (for example).
103      */
104     public ImmutableMap<String, String> attributes() {
105         return this.serviceAttributes;
106     }
107 
108     public NodeInfo info() {
109         return new NodeInfo(version, Build.CURRENT, discovery.localNode(), serviceAttributes,
110                 settings,
111                 monitorService.osService().info(),
112                 monitorService.processService().info(),
113                 monitorService.jvmService().info(),
114                 threadPool.info(),
115                 monitorService.networkService().info(),
116                 transportService.info(),
117                 httpServer == null ? null : httpServer.info(),
118                 pluginService == null ? null : pluginService.info()
119         );
120     }
121 
122     public NodeInfo info(boolean settings, boolean os, boolean process, boolean jvm, boolean threadPool,
123                          boolean network, boolean transport, boolean http, boolean plugin) {
124         return new NodeInfo(version, Build.CURRENT, discovery.localNode(), serviceAttributes,
125                 settings ? this.settings : null,
126                 os ? monitorService.osService().info() : null,
127                 process ? monitorService.processService().info() : null,
128                 jvm ? monitorService.jvmService().info() : null,
129                 threadPool ? this.threadPool.info() : null,
130                 network ? monitorService.networkService().info() : null,
131                 transport ? transportService.info() : null,
132                 http ? (httpServer == null ? null : httpServer.info()) : null,
133                 plugin ? (pluginService == null ? null : pluginService.info()) : null
134         );
135     }
136 
137     public NodeStats stats() throws IOException {
138         // for indices stats we want to include previous allocated shards stats as well (it will
139         // only be applied to the sensible ones to use, like refresh/merge/flush/indexing stats)
140         return new NodeStats(discovery.localNode(), System.currentTimeMillis(),
141                 indicesService.stats(true),
142                 monitorService.osService().stats(),
143                 monitorService.processService().stats(),
144                 monitorService.jvmService().stats(),
145                 threadPool.stats(),
146                 monitorService.networkService().stats(),
147                 monitorService.fsService().stats(),
148                 transportService.stats(),
149                 httpServer == null ? null : httpServer.stats(),
150                 circuitBreakerService.stats()
151         );
152     }
153 
154     public NodeStats stats(CommonStatsFlags indices, boolean os, boolean process, boolean jvm, boolean threadPool, boolean network,
155                            boolean fs, boolean transport, boolean http, boolean circuitBreaker) {
156         // for indices stats we want to include previous allocated shards stats as well (it will
157         // only be applied to the sensible ones to use, like refresh/merge/flush/indexing stats)
158         return new NodeStats(discovery.localNode(), System.currentTimeMillis(),
159                 indices.anySet() ? indicesService.stats(true, indices) : null,
160                 os ? monitorService.osService().stats() : null,
161                 process ? monitorService.processService().stats() : null,
162                 jvm ? monitorService.jvmService().stats() : null,
163                 threadPool ? this.threadPool.stats() : null,
164                 network ? monitorService.networkService().stats() : null,
165                 fs ? monitorService.fsService().stats() : null,
166                 transport ? transportService.stats() : null,
167                 http ? (httpServer == null ? null : httpServer.stats()) : null,
168                 circuitBreaker ? circuitBreakerService.stats() : null
169         );
170     }
171 }
