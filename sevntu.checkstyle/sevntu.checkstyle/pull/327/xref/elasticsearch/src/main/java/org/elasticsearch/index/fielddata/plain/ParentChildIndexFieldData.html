87  /**
88   * ParentChildIndexFieldData is responsible for loading the id cache mapping
89   * needed for has_child and has_parent queries into memory.
90   */
91  public class ParentChildIndexFieldData extends AbstractIndexFieldData<AtomicParentChildFieldData> implements IndexParentChildFieldData, DocumentTypeListener {
92  
93      private final NavigableSet<BytesRef> parentTypes;
94      private final CircuitBreakerService breakerService;
95  
96      // If child type (a type with _parent field) is added or removed, we want to make sure modifications don't happen
97      // while loading.
98      private final Object lock = new Object();
99  
100     public ParentChildIndexFieldData(Index index, @IndexSettings Settings indexSettings, FieldMapper.Names fieldNames,
101                                      FieldDataType fieldDataType, IndexFieldDataCache cache, MapperService mapperService,
102                                      CircuitBreakerService breakerService) {
103         super(index, indexSettings, fieldNames, fieldDataType, cache);
104         parentTypes = new TreeSet<>(BytesRef.getUTF8SortedAsUnicodeComparator());
105         this.breakerService = breakerService;
106         for (DocumentMapper documentMapper : mapperService.docMappers(false)) {
107             beforeCreate(documentMapper);
108         }
109         mapperService.addTypeListener(this);
110     }

112     @Override
113     public XFieldComparatorSource comparatorSource(@Nullable Object missingValue, MultiValueMode sortMode, Nested nested) {
114         return new BytesRefFieldComparatorSource(this, missingValue, sortMode, nested);
115     }

117     @Override
118     public ParentChildAtomicFieldData loadDirect(LeafReaderContext context) throws Exception {
119         LeafReader reader = context.reader();
120         final float acceptableTransientOverheadRatio = fieldDataType.getSettings().getAsFloat(
121                 "acceptable_transient_overhead_ratio", OrdinalsBuilder.DEFAULT_ACCEPTABLE_OVERHEAD_RATIO
122         );
123 
124         final NavigableSet<BytesRef> parentTypes;
125         synchronized (lock) {
126             parentTypes = ImmutableSortedSet.copyOf(BytesRef.getUTF8SortedAsUnicodeComparator(), this.parentTypes);
127         }
128         boolean success = false;
129         ParentChildAtomicFieldData data = null;
130         ParentChildFilteredTermsEnum termsEnum = new ParentChildFilteredTermsEnum(
131                 new ParentChildIntersectTermsEnum(reader, UidFieldMapper.NAME, ParentFieldMapper.NAME),
132                 parentTypes
133         );
134         ParentChildEstimator estimator = new ParentChildEstimator(breakerService.getBreaker(CircuitBreaker.FIELDDATA), termsEnum);
135         TermsEnum estimatedTermsEnum = estimator.beforeLoad(null);
136         ObjectObjectOpenHashMap<String, TypeBuilder> typeBuilders = ObjectObjectOpenHashMap.newInstance();
137         try {
138             try {
139                 PostingsEnum docsEnum = null;
140                 for (BytesRef term = estimatedTermsEnum.next(); term != null; term = estimatedTermsEnum.next()) {
141                     // Usually this would be estimatedTermsEnum, but the
142                     // abstract TermsEnum class does not support the .type()
143                     // and .id() methods, so we skip using the wrapped
144                     // TermsEnum and delegate directly to the
145                     // ParentChildFilteredTermsEnum that was originally wrapped
146                     String type = termsEnum.type();
147                     TypeBuilder typeBuilder = typeBuilders.get(type);
148                     if (typeBuilder == null) {
149                         typeBuilders.put(type, typeBuilder = new TypeBuilder(acceptableTransientOverheadRatio, reader));
150                     }
151 
152                     BytesRef id = termsEnum.id();
153                     final long termOrd = typeBuilder.builder.nextOrdinal();
154                     assert termOrd == typeBuilder.termOrdToBytesOffset.size();
155                     typeBuilder.termOrdToBytesOffset.add(typeBuilder.bytes.copyUsingLengthPrefix(id));
156                     docsEnum = estimatedTermsEnum.postings(null, docsEnum, PostingsEnum.NONE);
157                     for (int docId = docsEnum.nextDoc(); docId != DocIdSetIterator.NO_MORE_DOCS; docId = docsEnum.nextDoc()) {
158                         typeBuilder.builder.addDoc(docId);
159                     }
160                 }
161 
162                 ImmutableOpenMap.Builder<String, AtomicOrdinalsFieldData> typeToAtomicFieldData = ImmutableOpenMap.builder(typeBuilders.size());
163                 for (ObjectObjectCursor<String, TypeBuilder> cursor : typeBuilders) {
164                     PagedBytes.Reader bytesReader = cursor.value.bytes.freeze(true);
165                     final Ordinals ordinals = cursor.value.builder.build(fieldDataType.getSettings());
166 
167                     typeToAtomicFieldData.put(
168                             cursor.key,
169                             new PagedBytesAtomicFieldData(bytesReader, cursor.value.termOrdToBytesOffset.build(), ordinals)
170                     );
171                 }
172                 data = new ParentChildAtomicFieldData(typeToAtomicFieldData.build());
173             } finally {
174                 for (ObjectObjectCursor<String, TypeBuilder> cursor : typeBuilders) {
175                     cursor.value.builder.close();
176                 }
177             }
178             success = true;
179             return data;
180         } finally {
181             if (success) {
182                 estimator.afterLoad(estimatedTermsEnum, data.ramBytesUsed());
183             } else {
184                 estimator.afterLoad(estimatedTermsEnum, 0);
185             }
186         }
187     }

189     @Override
190     public void beforeCreate(DocumentMapper mapper) {
191         synchronized (lock) {
192             ParentFieldMapper parentFieldMapper = mapper.parentFieldMapper();
193             if (parentFieldMapper.active()) {
194                 // A _parent field can never be added to an existing mapping, so a _parent field either exists on
195                 // a new created or doesn't exists. This is why we can update the known parent types via DocumentTypeListener
196                 if (parentTypes.add(new BytesRef(parentFieldMapper.type()))) {
197                     clear();
198                 }
199             }
200         }
201     }
202 
203     @Override
204     public void afterRemove(DocumentMapper mapper) {
205         synchronized (lock) {
206             ParentFieldMapper parentFieldMapper = mapper.parentFieldMapper();
207             if (parentFieldMapper.active()) {
208                 parentTypes.remove(new BytesRef(parentFieldMapper.type()));
209             }
210         }
211     }

213     class TypeBuilder {
214 
215         final PagedBytes bytes;
216         final PackedLongValues.Builder termOrdToBytesOffset;
217         final OrdinalsBuilder builder;
218 
219         TypeBuilder(float acceptableTransientOverheadRatio, LeafReader reader) throws IOException {
220             bytes = new PagedBytes(15);
221             termOrdToBytesOffset = PackedLongValues.monotonicBuilder(PackedInts.COMPACT);
222             builder = new OrdinalsBuilder(-1, reader.maxDoc(), acceptableTransientOverheadRatio);
223         }
224     }

237     /**
238      * Estimator that wraps parent/child id field data by wrapping the data
239      * in a RamAccountingTermsEnum.
240      */
241     public class ParentChildEstimator implements PerValueEstimator {
242 
243         private final CircuitBreaker breaker;
244         private final TermsEnum filteredEnum;
245 
246         // The TermsEnum is passed in here instead of being generated in the
247         // beforeLoad() function since it's filtered inside the previous
248         // TermsEnum wrappers
249         public ParentChildEstimator(CircuitBreaker breaker, TermsEnum filteredEnum) {
250             this.breaker = breaker;
251             this.filteredEnum = filteredEnum;
252         }
253 
254         /**
255          * General overhead for ids is 2 times the length of the ID
256          */
257         @Override
258         public long bytesPerValue(BytesRef term) {
259             if (term == null) {
260                 return 0;
261             }
262             return 2 * term.length;
263         }
264 
265         /**
266          * Wraps the already filtered {@link TermsEnum} in a
267          * {@link RamAccountingTermsEnum} and returns it
268          */
269         @Override
270         public TermsEnum beforeLoad(Terms terms) throws IOException {
271             return new RamAccountingTermsEnum(filteredEnum, breaker, this, "parent/child id cache");
272         }
273 
274         /**
275          * Adjusts the breaker based on the difference between the actual usage
276          * and the aggregated estimations.
277          */
278         @Override
279         public void afterLoad(TermsEnum termsEnum, long actualUsed) {
280             assert termsEnum instanceof RamAccountingTermsEnum;
281             long estimatedBytes = ((RamAccountingTermsEnum) termsEnum).getTotalBytes();
282             breaker.addWithoutBreaking(-(estimatedBytes - actualUsed));
283         }
284     }

286     @Override
287     public IndexParentChildFieldData loadGlobal(IndexReader indexReader) {
288         if (indexReader.leaves().size() <= 1) {
289             // ordinals are already global
290             return this;
291         }
292         try {
293             return cache.load(indexReader, this);
294         } catch (Throwable e) {
295             if (e instanceof ElasticsearchException) {
296                 throw (ElasticsearchException) e;
297             } else {
298                 throw new ElasticsearchException(e.getMessage(), e);
299             }
300         }
301     }

303     private static OrdinalMap buildOrdinalMap(AtomicParentChildFieldData[] atomicFD, String parentType) throws IOException {
304         final SortedDocValues[] ordinals = new SortedDocValues[atomicFD.length];
305         for (int i = 0; i < ordinals.length; ++i) {
306             ordinals[i] = atomicFD[i].getOrdinalsValues(parentType);
307         }
308         return OrdinalMap.build(null, ordinals, PackedInts.DEFAULT);
309     }

311     private static class OrdinalMapAndAtomicFieldData {
312         final OrdinalMap ordMap;
313         final AtomicParentChildFieldData[] fieldData;
314 
315         public OrdinalMapAndAtomicFieldData(OrdinalMap ordMap, AtomicParentChildFieldData[] fieldData) {
316             this.ordMap = ordMap;
317             this.fieldData = fieldData;
318         }
319     }

321     @Override
322     public IndexParentChildFieldData localGlobalDirect(IndexReader indexReader) throws Exception {
323         final long startTime = System.nanoTime();
324         final Set<String> parentTypes = new HashSet<>();
325         synchronized (lock) {
326             for (BytesRef type : this.parentTypes) {
327                 parentTypes.add(type.utf8ToString());
328             }
329         }
330 
331         long ramBytesUsed = 0;
332         final Map<String, OrdinalMapAndAtomicFieldData> perType = new HashMap<>();
333         for (String type : parentTypes) {
334             final AtomicParentChildFieldData[] fieldData = new AtomicParentChildFieldData[indexReader.leaves().size()];
335             for (LeafReaderContext context : indexReader.leaves()) {
336                 fieldData[context.ord] = load(context);
337             }
338             final OrdinalMap ordMap = buildOrdinalMap(fieldData, type);
339             ramBytesUsed += ordMap.ramBytesUsed();
340             perType.put(type, new OrdinalMapAndAtomicFieldData(ordMap, fieldData));
341         }
342 
343         final AtomicParentChildFieldData[] fielddata = new AtomicParentChildFieldData[indexReader.leaves().size()];
344         for (int i = 0; i < fielddata.length; ++i) {
345             fielddata[i] = new GlobalAtomicFieldData(parentTypes, perType, i);
346         }
347 
348         breakerService.getBreaker(CircuitBreaker.FIELDDATA).addWithoutBreaking(ramBytesUsed);
349         if (logger.isDebugEnabled()) {
350             logger.debug(
351                     "Global-ordinals[_parent] took {}",
352                     new TimeValue(System.nanoTime() - startTime, TimeUnit.NANOSECONDS)
353             );
354         }
355 
356         return new GlobalFieldData(indexReader, fielddata, ramBytesUsed);
357     }

359     private static class GlobalAtomicFieldData extends AbstractAtomicParentChildFieldData {
360 
361         private final Set<String> types;
362         private final Map<String, OrdinalMapAndAtomicFieldData> atomicFD;
363         private final int segmentIndex;
364 
365         public GlobalAtomicFieldData(Set<String> types, Map<String, OrdinalMapAndAtomicFieldData> atomicFD, int segmentIndex) {
366             this.types = types;
367             this.atomicFD = atomicFD;
368             this.segmentIndex = segmentIndex;
369         }
370 
371         @Override
372         public Set<String> types() {
373             return types;
374         }
375 
376         @Override
377         public SortedDocValues getOrdinalsValues(String type) {
378             final OrdinalMapAndAtomicFieldData atomicFD = this.atomicFD.get(type);
379             final OrdinalMap ordMap = atomicFD.ordMap;
380             final SortedDocValues[] allSegmentValues = new SortedDocValues[atomicFD.fieldData.length];
381             for (int i = 0; i < allSegmentValues.length; ++i) {
382                 allSegmentValues[i] = atomicFD.fieldData[i].getOrdinalsValues(type);
383             }
384             final SortedDocValues segmentValues = allSegmentValues[segmentIndex];
385             if (segmentValues.getValueCount() == ordMap.getValueCount()) {
386                 // ords are already global
387                 return segmentValues;
388             }
389             final LongValues globalOrds = ordMap.getGlobalOrds(segmentIndex);
390             return new SortedDocValues() {
391 
392                 @Override
393                 public BytesRef lookupOrd(int ord) {
394                     final int segmentIndex = ordMap.getFirstSegmentNumber(ord);
395                     final int segmentOrd = (int) ordMap.getFirstSegmentOrd(ord);
396                     return allSegmentValues[segmentIndex].lookupOrd(segmentOrd);
397                 }
398 
399                 @Override
400                 public int getValueCount() {
401                     return (int) ordMap.getValueCount();
402                 }
403 
404                 @Override
405                 public int getOrd(int docID) {
406                     final int segmentOrd = segmentValues.getOrd(docID);
407                     // TODO: is there a way we can get rid of this branch?
408                     if (segmentOrd >= 0) {
409                         return (int) globalOrds.get(segmentOrd);
410                     } else {
411                         return segmentOrd;
412                     }
413                 }
414             };
415         }
416 
417         @Override
418         public long ramBytesUsed() {
419             // this class does not take memory on its own, the index-level field data does
420             // it through the use of ordinal maps
421             return 0;
422         }
423 
424         @Override
425         public Collection<Accountable> getChildResources() {
426             return Collections.emptyList();
427         }
428 
429         @Override
430         public void close() throws ElasticsearchException {
431             List<Releasable> closeables = new ArrayList<>();
432             for (OrdinalMapAndAtomicFieldData fds : atomicFD.values()) {
433                 closeables.addAll(Arrays.asList(fds.fieldData));
434             }
435             Releasables.close(closeables);
436         }
437 
438     }

440     private class GlobalFieldData implements IndexParentChildFieldData, Accountable {
441 
442         private final AtomicParentChildFieldData[] fielddata;
443         private final IndexReader reader;
444         private final long ramBytesUsed;
445 
446         GlobalFieldData(IndexReader reader, AtomicParentChildFieldData[] fielddata, long ramBytesUsed) {
447             this.reader = reader;
448             this.ramBytesUsed = ramBytesUsed;
449             this.fielddata = fielddata;
450         }
451 
452         @Override
453         public Names getFieldNames() {
454             return ParentChildIndexFieldData.this.getFieldNames();
455         }
456 
457         @Override
458         public FieldDataType getFieldDataType() {
459             return ParentChildIndexFieldData.this.getFieldDataType();
460         }
461 
462         @Override
463         public AtomicParentChildFieldData load(LeafReaderContext context) {
464             assert context.reader().getCoreCacheKey() == reader.leaves().get(context.ord).reader().getCoreCacheKey();
465             return fielddata[context.ord];
466         }
467 
468         @Override
469         public AtomicParentChildFieldData loadDirect(LeafReaderContext context) throws Exception {
470             return load(context);
471         }
472 
473         @Override
474         public XFieldComparatorSource comparatorSource(Object missingValue, MultiValueMode sortMode, Nested nested) {
475             throw new UnsupportedOperationException("No sorting on global ords");
476         }
477 
478         @Override
479         public void clear() {
480             ParentChildIndexFieldData.this.clear();
481         }
482 
483         @Override
484         public void clear(IndexReader reader) {
485             ParentChildIndexFieldData.this.clear(reader);
486         }
487 
488         @Override
489         public Index index() {
490             return ParentChildIndexFieldData.this.index();
491         }
492 
493         @Override
494         public long ramBytesUsed() {
495             return ramBytesUsed;
496         }
497 
498         @Override
499         public Collection<Accountable> getChildResources() {
500             return Collections.emptyList();
501         }
502 
503         @Override
504         public IndexParentChildFieldData loadGlobal(IndexReader indexReader) {
505             if (indexReader.getCoreCacheKey() == reader.getCoreCacheKey()) {
506                 return this;
507             }
508             throw new ElasticsearchIllegalStateException();
509         }
510 
511         @Override
512         public IndexParentChildFieldData localGlobalDirect(IndexReader indexReader) throws Exception {
513             return loadGlobal(indexReader);
514         }
