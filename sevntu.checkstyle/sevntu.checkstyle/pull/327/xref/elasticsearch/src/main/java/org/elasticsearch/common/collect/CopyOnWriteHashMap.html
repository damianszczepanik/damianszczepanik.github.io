30  /**
31   * An immutable map whose writes result in a new copy of the map to be created.
32   *
33   * This is essentially a hash array mapped trie: inner nodes use a bitmap in
34   * order to map hashes to slots by counting ones. In case of a collision (two
35   * values having the same 32-bits hash), a leaf node is created which stores
36   * and searches for values sequentially.
37   *
38   * Reads and writes both perform in logarithmic time. Null keys and values are
39   * not supported.
40   *
41   * This structure might need to perform several object creations per write so
42   * it is better suited for work-loads that are not too write-intensive.
43   *
44   * @see <a href="http://en.wikipedia.org/wiki/Hash_array_mapped_trie">the wikipedia page</a>
45   */
46  public final class CopyOnWriteHashMap<K, V> extends AbstractMap<K, V> {
47  
48      private static final int TOTAL_HASH_BITS = 32;
49      private static final Object[] EMPTY_ARRAY = new Object[0];
50  
51      private static final int HASH_BITS = 6;
52      private static final int HASH_MASK = 0x3F;
53  
54      /**
55       * Return a copy of the provided map.
56       */
57      public static <K, V> CopyOnWriteHashMap<K, V> copyOf(Map<? extends K, ? extends V> map) {
58          if (map instanceof CopyOnWriteHashMap) {
59              // no need to copy in that case
60              @SuppressWarnings("unchecked")
61              final CopyOnWriteHashMap<K, V> cowMap = (CopyOnWriteHashMap<K, V>) map;
62              return cowMap;
63          } else {
64              return new CopyOnWriteHashMap<K, V>().copyAndPutAll(map);
65          }
66      }
67  
68      /**
69       * Abstraction of a node, implemented by both inner and leaf nodes.
70       */
71      private static abstract class Node<K, V> {
72  
73          /**
74           * Recursively get the key with the given hash.
75           */
76          abstract V get(Object key, int hash);
77  
78          /**
79           * Recursively add a new entry to this node. <code>hashBits</code> is
80           * the number of bits that are still set in the hash. When this value
81           * reaches a number that is less than or equal to <tt>0</tt>, a leaf
82           * node needs to be created since it means that a collision occurred
83           * on the 32 bits of the hash.
84           */
85          abstract Node<K, V> put(K key, int hash, int hashBits, V value, MutableValueInt newValue);
86  
87          /**
88           * Recursively remove an entry from this node.
89           */
90          abstract Node<K, V> remove(Object key, int hash);
91  
92          /**
93           * For the current node only, append entries that are stored on this
94           * node to <code>entries</code> and sub nodes to <code>nodes</code>.
95           */
96          abstract void visit(Deque<Map.Entry<K, V>> entries, Deque<Node<K, V>> nodes);
97  
98          /**
99           * Whether this node stores nothing under it.
100          */
101         abstract boolean isEmpty();
102 
103     }
104 
105     /**
106      * A leaf of the tree where all hashes are equal. Values are added and retrieved in linear time.
107      */
108     private static class Leaf<K, V> extends Node<K, V> {
109 
110         private final K[] keys;
111         private final V[] values;
112 
113         Leaf(K[] keys, V[] values) {
114             this.keys = keys;
115             this.values = values;
116         }
117 
118         @SuppressWarnings("unchecked")
119         Leaf() {
120             this((K[]) EMPTY_ARRAY, (V[]) EMPTY_ARRAY);
121         }
122 
123         @Override
124         boolean isEmpty() {
125             return keys.length == 0;
126         }
127 
128         @Override
129         void visit(Deque<Map.Entry<K, V>> entries, Deque<Node<K, V>> nodes) {
130             for (int i = 0; i < keys.length; ++i) {
131                 entries.add(Maps.immutableEntry(keys[i], values[i]));
132             }
133         }
134 
135         @Override
136         V get(Object key, int hash) {
137             final int slot = ArrayUtils.indexOf(keys, key);
138             if (slot < 0) {
139                 return null;
140             } else {
141                 return values[slot];
142             }
143         }
144 
145         private static <T> T[] replace(T[] array, int index, T value) {
146             final T[] copy = Arrays.copyOf(array, array.length);
147             copy[index] = value;
148             return copy;
149         }
150 
151         @Override
152         Leaf<K, V> put(K key, int hash, int hashBits, V value, MutableValueInt newValue) {
153             assert hashBits <= 0 : hashBits;
154             final int slot = ArrayUtils.indexOf(keys, key);
155 
156             final K[] keys2;
157             final V[] values2;
158 
159             if (slot < 0) {
160                 keys2 = ArrayUtils.add(keys, key);
161                 values2 = ArrayUtils.add(values, value);
162                 newValue.value = 1;
163             } else {
164                 keys2 = replace(keys, slot, key);
165                 values2 = replace(values, slot, value);
166             }
167 
168             return new Leaf<>(keys2, values2);
169         }
170 
171         @Override
172         Leaf<K, V> remove(Object key, int hash) {
173             final int slot = ArrayUtils.indexOf(keys, key);
174             if (slot < 0) {
175                 return this;
176             }
177             final K[] keys2 = ArrayUtils.remove(keys, slot);
178             final V[] values2 = ArrayUtils.remove(values, slot);
179             return new Leaf<>(keys2, values2);
180         }
181     }

183     /**
184      * An inner node in this trie. Inner nodes store up to 64 key-value pairs
185      * and use a bitmap in order to associate hashes to them. For example, if
186      * an inner node contains 5 values, then 5 bits will be set in the bitmap
187      * and the ordinal of the bit set in this bit map will be the slot number.
188      *
189      * As a consequence, the number of slots in an inner node is equal to the
190      * number of one bits in the bitmap.
191      */
192     private static class InnerNode<K, V> extends Node<K, V> {
193 
194         private final long mask; // the bitmap
195         private final K[] keys;
196         final Object[] subNodes; // subNodes[slot] is either a value or a sub node in case of a hash collision
197 
198         InnerNode(long mask, K[] keys, Object[] subNodes) {
199             this.mask = mask;
200             this.keys = keys;
201             this.subNodes = subNodes;
202             assert consistent();
203         }
204 
205         // only used in assert
206         private boolean consistent() {
207             assert Long.bitCount(mask) == keys.length;
208             assert Long.bitCount(mask) == subNodes.length;
209             for (int i = 0; i < keys.length; ++i) {
210                 if (subNodes[i] instanceof Node) {
211                     assert keys[i] == null;
212                 } else {
213                     assert keys[i] != null;
214                 }
215             }
216             return true;
217         }
218 
219         @Override
220         boolean isEmpty() {
221             return mask == 0;
222         }
223 
224         @SuppressWarnings("unchecked")
225         InnerNode() {
226             this(0, (K[]) EMPTY_ARRAY, EMPTY_ARRAY);
227         }
228 
229         @Override
230         void visit(Deque<Map.Entry<K, V>> entries, Deque<Node<K, V>> nodes) {
231             for (int i = 0; i < keys.length; ++i) {
232                 final Object sub = subNodes[i];
233                 if (sub instanceof Node) {
234                     @SuppressWarnings("unchecked")
235                     final Node<K, V> subNode = (Node<K, V>) sub;
236                     assert keys[i] == null;
237                     nodes.add(subNode);
238                 } else {
239                     @SuppressWarnings("unchecked")
240                     final V value = (V) sub;
241                     entries.add(Maps.immutableEntry(keys[i], value));
242                 }
243             }
244         }
245 
246         /**
247          * For a given hash on 6 bits, its value is set if the bitmap has a one
248          * at the corresponding index.
249          */
250         private boolean exists(int hash6) {
251             return (mask & (1L << hash6)) != 0;
252         }
253 
254         /**
255          * For a given hash on 6 bits, the slot number is the number of one
256          * bits on the right of the <code>hash6</code>-th bit.
257          */
258         private int slot(int hash6) {
259             return Long.bitCount(mask & ((1L << hash6) - 1));
260         }
261 
262         @Override
263         V get(Object key, int hash) {
264             final int hash6 = hash & HASH_MASK;
265             if (!exists(hash6)) {
266                 return null;
267             }
268             final int slot = slot(hash6);
269             final Object sub = subNodes[slot];
270             assert sub != null;
271             if (sub instanceof Node) {
272                 assert keys[slot] == null; // keys don't make sense on inner nodes
273                 @SuppressWarnings("unchecked")
274                 final Node<K, V> subNode = (Node<K, V>) sub;
275                 return subNode.get(key, hash >>> HASH_BITS);
276             } else {
277                 if (keys[slot].equals(key)) {
278                     @SuppressWarnings("unchecked")
279                     final V v = (V) sub;
280                     return v;
281                 } else {
282                     // we have an entry for this hash, but the value is different
283                     return null;
284                 }
285             }
286         }
287 
288         private Node<K, V> newSubNode(int hashBits) {
289             if (hashBits <= 0) {
290                 return new Leaf<K, V>();
291             } else {
292                 return new InnerNode<K, V>();
293             }
294         }
295 
296         private InnerNode<K, V> putExisting(K key, int hash, int hashBits, int slot, V value, MutableValueInt newValue) {
297             final K[] keys2 = Arrays.copyOf(keys, keys.length);
298             final Object[] subNodes2 = Arrays.copyOf(subNodes, subNodes.length);
299 
300             final Object previousValue = subNodes2[slot];
301             if (previousValue instanceof Node) {
302                 // insert recursively
303                 assert keys[slot] == null;
304                 subNodes2[slot] = ((Node<K, V>) previousValue).put(key, hash, hashBits, value, newValue);
305             } else if (keys[slot].equals(key)) {
306                 // replace the existing entry
307                 subNodes2[slot] = value;
308             } else {
309                 // hash collision
310                 final K previousKey = keys[slot];
311                 final int previousHash = previousKey.hashCode() >>> (TOTAL_HASH_BITS - hashBits);
312                 Node<K, V> subNode = newSubNode(hashBits);
313                 subNode = subNode.put(previousKey, previousHash, hashBits, (V) previousValue, newValue);
314                 subNode = subNode.put(key, hash, hashBits, value, newValue);
315                 keys2[slot] = null;
316                 subNodes2[slot] = subNode;
317             }
318             return new InnerNode<>(mask, keys2, subNodes2);
319         }
320 
321         private InnerNode<K, V> putNew(K key, int hash6, int slot, V value) {
322             final long mask2 = mask | (1L << hash6);
323             final K[] keys2 = ArrayUtils.add(keys, slot, key);
324             final Object[] subNodes2 = ArrayUtils.add(subNodes, slot, value);
325             return new InnerNode<>(mask2, keys2, subNodes2);
326         }
327 
328         @Override
329         InnerNode<K, V> put(K key, int hash, int hashBits, V value, MutableValueInt newValue) {
330             final int hash6 = hash & HASH_MASK;
331             final int slot = slot(hash6);
332 
333             if (exists(hash6)) {
334                 hash >>>= HASH_BITS;
335                 hashBits -= HASH_BITS;
336                 return putExisting(key, hash, hashBits, slot, value, newValue);
337             } else {
338                 newValue.value = 1;
339                 return putNew(key, hash6, slot, value);
340             }
341         }
342 
343         private InnerNode<K, V> removeSlot(int hash6, int slot) {
344             final long mask2 = mask  & ~(1L << hash6);
345             final K[] keys2 = ArrayUtils.remove(keys, slot);
346             final Object[] subNodes2 = ArrayUtils.remove(subNodes, slot);
347             return new InnerNode<>(mask2, keys2, subNodes2);
348         }
349 
350         @Override
351         InnerNode<K, V> remove(Object key, int hash) {
352             final int hash6 = hash & HASH_MASK;
353             if (!exists(hash6)) {
354                 return this;
355             }
356             final int slot = slot(hash6);
357             final Object previousValue = subNodes[slot];
358             if (previousValue instanceof Node) {
359                 @SuppressWarnings("unchecked")
360                 final Node<K, V> subNode = (Node<K, V>) previousValue;
361                 final Node<K, V> removed = subNode.remove(key, hash >>> HASH_BITS);
362                 if (removed == subNode) {
363                     // not in sub-nodes
364                     return this;
365                 }
366                 if (removed.isEmpty()) {
367                     return removeSlot(hash6, slot);
368                 }
369                 final K[] keys2 = Arrays.copyOf(keys, keys.length);
370                 final Object[] subNodes2 = Arrays.copyOf(subNodes, subNodes.length);
371                 subNodes2[slot] = removed;
372                 return new InnerNode<>(mask, keys2, subNodes2);
373             } else if (keys[slot].equals(key)) {
374                 // remove entry
375                 return removeSlot(hash6, slot);
376             } else {
377                 // hash collision, nothing to remove
378                 return this;
379             }
380         }

384     private static class EntryIterator<K, V> extends UnmodifiableIterator<Map.Entry<K, V>> {
385 
386         private final Deque<Map.Entry<K, V>> entries;
387         private final Deque<Node<K, V>> nodes;
388 
389         public EntryIterator(Node<K, V> node) {
390             entries = new ArrayDeque<>();
391             nodes = new ArrayDeque<>();
392             node.visit(entries, nodes);
393         }
394 
395         @Override
396         public boolean hasNext() {
397             return !entries.isEmpty() || !nodes.isEmpty();
398         }
399 
400         @Override
401         public Map.Entry<K, V> next() {
402             while (entries.isEmpty()) {
403                 if (nodes.isEmpty()) {
404                     throw new NoSuchElementException();
405                 }
406                 final Node<K, V> nextNode = nodes.pop();
407                 nextNode.visit(entries, nodes);
408             }
409             return entries.pop();
410         }
411 
412     }

414     private final InnerNode<K, V> root;
415     private final int size;
416 
417     /**
418      * Create a new empty map.
419      */
420     public CopyOnWriteHashMap() {
421         this(new InnerNode<K, V>(), 0);
422     }
423 
424     private CopyOnWriteHashMap(InnerNode<K, V> root, int size) {
425         this.root = root;
426         this.size = size;
427     }
428 
429     @Override
430     public boolean containsKey(Object key) {
431         // works fine since null values are not supported
432         return get(key) != null;
433     }
434 
435     @Override
436     public V get(Object key) {
437         Preconditions.checkArgument(key != null, "Null keys are not supported");
438         final int hash = key.hashCode();
439         return root.get(key, hash);
440     }
441 
442     @Override
443     public int size() {
444         assert size != 0 || root.isEmpty();
445         return size;
446     }

448     /**
449      * Associate <code>key</code> with <code>value</code> and return a new copy
450      * of the hash table. The current hash table is not modified.
451      */
452     public CopyOnWriteHashMap<K, V> copyAndPut(K key, V value) {
453         Preconditions.checkArgument(key != null, "null keys are not supported");
454         Preconditions.checkArgument(value != null, "null values are not supported");
455         final int hash = key.hashCode();
456         final MutableValueInt newValue = new MutableValueInt();
457         final InnerNode<K, V> newRoot = root.put(key, hash, TOTAL_HASH_BITS, value, newValue);
458         final int newSize = size + newValue.value;
459         return new CopyOnWriteHashMap<>(newRoot, newSize);
460     }

462     /**
463      * Same as {@link #copyAndPut(Object, Object)} but for an arbitrary number of entries.
464      */
465     public CopyOnWriteHashMap<K, V> copyAndPutAll(Map<? extends K, ? extends V> other) {
466         return copyAndPutAll(other.entrySet());
467     }

469     <K1 extends K, V1 extends V> CopyOnWriteHashMap<K, V> copyAndPutAll(Collection<Map.Entry<K1, V1>> entries) {
470         CopyOnWriteHashMap<K, V> result = this;
471         for (Map.Entry<K1, V1> entry : entries) {
472             result = result.copyAndPut(entry.getKey(), entry.getValue());
473         }
474         return result;
475     }
476 
477     /**
478      * Remove the given key from this map. The current hash table is not modified.
479      */
480     public CopyOnWriteHashMap<K, V> copyAndRemove(Object key) {
481         Preconditions.checkArgument(key != null, "Null keys are not supported");
482         final int hash = key.hashCode();
483         final InnerNode<K, V> newRoot = root.remove(key, hash);
484         if (root == newRoot) {
485             return this;
486         } else {
487             return new CopyOnWriteHashMap<>(newRoot, size - 1);
488         }
489     }
490 
491     /**
492      * Same as {@link #copyAndRemove(Object)} but for an arbitrary number of entries.
493      */
494     public CopyOnWriteHashMap<K, V> copyAndRemoveAll(Collection<?> keys) {
495         CopyOnWriteHashMap<K, V> result = this;
496         for (Object key : keys) {
497             result = result.copyAndRemove(key);
498         }
499         return result;
500     }

502     @Override
503     public Set<Map.Entry<K, V>> entrySet() {
504         return new AbstractSet<Map.Entry<K, V>>() {
505 
506             @Override
507             public Iterator<java.util.Map.Entry<K, V>> iterator() {
508                 return new EntryIterator<>(root);
509             }
510 
511             @Override
512             public boolean contains(Object o) {
513                 if (o == null || !(o instanceof Map.Entry)) {
514                     return false;
515                 }
516                 Map.Entry<?, ?> entry = (java.util.Map.Entry<?, ?>) o;
517                 return entry.getValue().equals(CopyOnWriteHashMap.this.get(entry.getKey()));
518             }
519 
520             @Override
521             public int size() {
522                 return CopyOnWriteHashMap.this.size();
523             }
524         };
