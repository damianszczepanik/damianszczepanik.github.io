19  package org.elasticsearch.search.aggregations.bucket.global;
20  
21  import org.apache.lucene.index.LeafReaderContext;
22  import org.elasticsearch.ElasticsearchIllegalStateException;
23  import org.elasticsearch.search.aggregations.AggregationExecutionException;
24  import org.elasticsearch.search.aggregations.Aggregator;
25  import org.elasticsearch.search.aggregations.AggregatorFactories;
26  import org.elasticsearch.search.aggregations.AggregatorFactory;
27  import org.elasticsearch.search.aggregations.InternalAggregation;
28  import org.elasticsearch.search.aggregations.LeafBucketCollector;
29  import org.elasticsearch.search.aggregations.LeafBucketCollectorBase;
30  import org.elasticsearch.search.aggregations.bucket.SingleBucketAggregator;
31  import org.elasticsearch.search.aggregations.support.AggregationContext;
32  
33  import java.io.IOException;
34  import java.util.Map;
35  
36  /**
37   *
38   */
39  public class GlobalAggregator extends SingleBucketAggregator {
40  
41      public GlobalAggregator(String name, AggregatorFactories subFactories, AggregationContext aggregationContext, Map<String, Object> metaData) throws IOException {
42          super(name, subFactories, aggregationContext, null, metaData);
43      }
44  
45      @Override
46      public LeafBucketCollector getLeafCollector(LeafReaderContext ctx,
47              final LeafBucketCollector sub) throws IOException {
48          return new LeafBucketCollectorBase(sub, null) {
49              @Override
50              public void collect(int doc, long bucket) throws IOException {
51                  assert bucket == 0 : "global aggregator can only be a top level aggregator";
52                  collectBucket(sub, doc, bucket);
53              }
54          };
55      }
56  
57      @Override
58      public InternalAggregation buildAggregation(long owningBucketOrdinal) throws IOException {
59          assert owningBucketOrdinal == 0 : "global aggregator can only be a top level aggregator";
60          return new InternalGlobal(name, bucketDocCount(owningBucketOrdinal), bucketAggregations(owningBucketOrdinal), metaData());
61      }
62  
63      @Override
64      public InternalAggregation buildEmptyAggregation() {
65          throw new UnsupportedOperationException("global aggregations cannot serve as sub-aggregations, hence should never be called on #buildEmptyAggregations");
66      }
67  
68      public static class Factory extends AggregatorFactory {
69  
70          public Factory(String name) {
71              super(name, InternalGlobal.TYPE.name());
72          }
73  
74          @Override
75          public Aggregator createInternal(AggregationContext context, Aggregator parent, boolean collectsFromSingleBucket, Map<String, Object> metaData) throws IOException {
76              if (parent != null) {
77                  throw new AggregationExecutionException("Aggregation [" + parent.name() + "] cannot have a global " +
78                          "sub-aggregation [" + name + "]. Global aggregations can only be defined as top level aggregations");
79              }
80              if (collectsFromSingleBucket == false) {
81                  throw new ElasticsearchIllegalStateException();
82              }
83              return new GlobalAggregator(name, factories, context, metaData);
84          }
85  
86      }
87  }
