20  package org.elasticsearch.indices.recovery;
21  
22  import org.elasticsearch.ElasticsearchException;
23  import org.elasticsearch.ElasticsearchIllegalStateException;
24  import org.elasticsearch.cluster.ClusterService;
25  import org.elasticsearch.cluster.action.index.MappingUpdatedAction;
26  import org.elasticsearch.common.logging.ESLogger;
27  import org.elasticsearch.index.deletionpolicy.SnapshotIndexCommit;
28  import org.elasticsearch.index.shard.IndexShard;
29  import org.elasticsearch.index.translog.Translog;
30  import org.elasticsearch.indices.IndicesService;
31  import org.elasticsearch.transport.TransportService;
32  
33  /**
34   * A recovery handler that skips phase 1 as well as sending the snapshot. During phase 3 the shard is marked
35   * as relocated an closed to ensure that the engine is closed and the target can acquire the IW write lock.
36   */
37  public class SharedFSRecoverySourceHandler extends RecoverySourceHandler {
38  
39      private final IndexShard shard;
40      private final StartRecoveryRequest request;
41  
42      public SharedFSRecoverySourceHandler(IndexShard shard, StartRecoveryRequest request, RecoverySettings recoverySettings, TransportService transportService, ClusterService clusterService, IndicesService indicesService, MappingUpdatedAction mappingUpdatedAction, ESLogger logger) {
43          super(shard, request, recoverySettings, transportService, clusterService, indicesService, mappingUpdatedAction, logger);
44          this.shard = shard;
45          this.request = request;
46      }
47  
48      @Override
49      public void phase1(SnapshotIndexCommit snapshot) throws ElasticsearchException {
50          if (request.recoveryType() == RecoveryState.Type.RELOCATION && shard.routingEntry().primary()) {
51              // here we simply fail the primary shard since we can't move them (have 2 writers open at the same time)
52              // by failing the shard we play safe and just go through the entire reallocation procedure of the primary
53              // it would be ideal to make sure we flushed the translog here but that is not possible in the current design.
54              ElasticsearchIllegalStateException exception = new ElasticsearchIllegalStateException("Can't relocate primary - failing");
55              shard.failShard("primary_relocation", exception);
56              throw exception;
57          }
58          logger.trace("{} recovery [phase2] to {}: skipping phase 1 for shared filesystem", request.shardId(), request.targetNode());
59      }
60  
61  
62      @Override
63      protected int sendSnapshot(Translog.Snapshot snapshot) throws ElasticsearchException {
64          logger.trace("{} recovery [phase3] to {}: skipping transaction log operations for file sync", shard.shardId(), request.targetNode());
65          return 0;
66      }
67  
68  }
