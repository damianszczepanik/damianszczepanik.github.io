38  import static org.elasticsearch.common.Strings.isAllOrWildcard;
39  import static org.elasticsearch.rest.RestRequest.Method.GET;
40  import static org.elasticsearch.rest.RestStatus.OK;

42  /**
43   */
44  @Deprecated
45  public class RestGetIndicesAliasesAction extends BaseRestHandler {
46  
47      @Inject
48      public RestGetIndicesAliasesAction(Settings settings, RestController controller, Client client) {
49          super(settings, controller, client);
50          controller.registerHandler(GET, "/{index}/_aliases/{name}", this);
51          controller.registerHandler(GET, "/_aliases/{name}", this);
52      }
53  
54      @Override
55      public void handleRequest(final RestRequest request, final RestChannel channel, final Client client) {
56          final String[] indices = Strings.splitStringByCommaToArray(request.param("index"));
57          final String[] aliases = Strings.splitStringByCommaToArray(request.param("name"));
58  
59          ClusterStateRequest clusterStateRequest = Requests.clusterStateRequest()
60                  .routingTable(false)
61                  .nodes(false)
62                  .indices(indices);
63          clusterStateRequest.local(request.paramAsBoolean("local", clusterStateRequest.local()));
64          clusterStateRequest.listenerThreaded(false);
65  
66          client.admin().cluster().state(clusterStateRequest, new RestBuilderListener<ClusterStateResponse>(channel) {
67              @Override
68              public RestResponse buildResponse(ClusterStateResponse response, XContentBuilder builder) throws Exception {
69                  MetaData metaData = response.getState().metaData();
70                  builder.startObject();
71  
72                  final boolean isAllAliasesRequested = isAllOrWildcard(aliases);
73                  for (IndexMetaData indexMetaData : metaData) {
74                      builder.startObject(indexMetaData.index(), XContentBuilder.FieldCaseConversion.NONE);
75                      builder.startObject("aliases");
76  
77                      for (ObjectCursor<AliasMetaData> cursor : indexMetaData.aliases().values()) {
78                          if (isAllAliasesRequested || Regex.simpleMatch(aliases, cursor.value.alias())) {
79                              AliasMetaData.Builder.toXContent(cursor.value, builder, ToXContent.EMPTY_PARAMS);
80                          }
81                      }
82  
83                      builder.endObject();
84                      builder.endObject();
85                  }
86  
87                  builder.endObject();
88                  return new BytesRestResponse(OK, builder);
89              }
90          });
91      }
