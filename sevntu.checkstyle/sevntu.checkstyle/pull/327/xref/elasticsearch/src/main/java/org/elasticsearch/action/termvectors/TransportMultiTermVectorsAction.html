20  package org.elasticsearch.action.termvectors;
21  
22  import org.elasticsearch.ExceptionsHelper;
23  import org.elasticsearch.action.ActionListener;
24  import org.elasticsearch.action.support.ActionFilters;
25  import org.elasticsearch.action.support.HandledTransportAction;
26  import org.elasticsearch.cluster.ClusterService;
27  import org.elasticsearch.cluster.ClusterState;
28  import org.elasticsearch.cluster.block.ClusterBlockLevel;
29  import org.elasticsearch.common.inject.Inject;
30  import org.elasticsearch.common.settings.Settings;
31  import org.elasticsearch.common.util.concurrent.AtomicArray;
32  import org.elasticsearch.index.shard.ShardId;
33  import org.elasticsearch.threadpool.ThreadPool;
34  import org.elasticsearch.transport.TransportService;
35  
36  import java.util.HashMap;
37  import java.util.Map;
38  import java.util.concurrent.atomic.AtomicInteger;
39  
40  public class TransportMultiTermVectorsAction extends HandledTransportAction<MultiTermVectorsRequest, MultiTermVectorsResponse> {
41  
42      private final ClusterService clusterService;
43  
44      private final TransportShardMultiTermsVectorAction shardAction;
45  
46      @Inject
47      public TransportMultiTermVectorsAction(Settings settings, ThreadPool threadPool, TransportService transportService,
48                                             ClusterService clusterService, TransportShardMultiTermsVectorAction shardAction, ActionFilters actionFilters) {
49          super(settings, MultiTermVectorsAction.NAME, threadPool, transportService, actionFilters);
50          this.clusterService = clusterService;
51          this.shardAction = shardAction;
52      }
53  
54      @Override
55      protected void doExecute(final MultiTermVectorsRequest request, final ActionListener<MultiTermVectorsResponse> listener) {
56          ClusterState clusterState = clusterService.state();
57  
58          clusterState.blocks().globalBlockedRaiseException(ClusterBlockLevel.READ);
59  
60          final AtomicArray<MultiTermVectorsItemResponse> responses = new AtomicArray<>(request.requests.size());
61  
62          Map<ShardId, MultiTermVectorsShardRequest> shardRequests = new HashMap<>();
63          for (int i = 0; i < request.requests.size(); i++) {
64              TermVectorsRequest termVectorsRequest = request.requests.get(i);
65              termVectorsRequest.startTime = System.currentTimeMillis();
66              termVectorsRequest.routing(clusterState.metaData().resolveIndexRouting(termVectorsRequest.routing(), termVectorsRequest.index()));
67              if (!clusterState.metaData().hasConcreteIndex(termVectorsRequest.index())) {
68                  responses.set(i, new MultiTermVectorsItemResponse(null, new MultiTermVectorsResponse.Failure(termVectorsRequest.index(),
69                          termVectorsRequest.type(), termVectorsRequest.id(), "[" + termVectorsRequest.index() + "] missing")));
70                  continue;
71              }
72              String concreteSingleIndex = clusterState.metaData().concreteSingleIndex(termVectorsRequest.index(), termVectorsRequest.indicesOptions());
73              if (termVectorsRequest.routing() == null && clusterState.getMetaData().routingRequired(concreteSingleIndex, termVectorsRequest.type())) {
74                  responses.set(i, new MultiTermVectorsItemResponse(null, new MultiTermVectorsResponse.Failure(concreteSingleIndex, termVectorsRequest.type(), termVectorsRequest.id(),
75                          "routing is required for [" + concreteSingleIndex + "]/[" + termVectorsRequest.type() + "]/[" + termVectorsRequest.id() + "]")));
76                  continue;
77              }
78              ShardId shardId = clusterService.operationRouting().getShards(clusterState, concreteSingleIndex,
79                      termVectorsRequest.type(), termVectorsRequest.id(), termVectorsRequest.routing(), null).shardId();
80              MultiTermVectorsShardRequest shardRequest = shardRequests.get(shardId);
81              if (shardRequest == null) {
82                  shardRequest = new MultiTermVectorsShardRequest(request, shardId.index().name(), shardId.id());
83                  shardRequest.preference(request.preference);
84                  shardRequests.put(shardId, shardRequest);
85              }
86              shardRequest.add(i, termVectorsRequest);
87          }
88          
89          if (shardRequests.size() == 0) {
90              // only failures..
91              listener.onResponse(new MultiTermVectorsResponse(responses.toArray(new MultiTermVectorsItemResponse[responses.length()])));
92          }
93          
94          final AtomicInteger counter = new AtomicInteger(shardRequests.size());
95          for (final MultiTermVectorsShardRequest shardRequest : shardRequests.values()) {
96              shardAction.execute(shardRequest, new ActionListener<MultiTermVectorsShardResponse>() {
97                  @Override
98                  public void onResponse(MultiTermVectorsShardResponse response) {
99                      for (int i = 0; i < response.locations.size(); i++) {
100                         responses.set(response.locations.get(i), new MultiTermVectorsItemResponse(response.responses.get(i),
101                                 response.failures.get(i)));
102                     }
103                     if (counter.decrementAndGet() == 0) {
104                         finishHim();
105                     }
106                 }
107 
108                 @Override
109                 public void onFailure(Throwable e) {
110                     // create failures for all relevant requests
111                     String message = ExceptionsHelper.detailedMessage(e);
112                     for (int i = 0; i < shardRequest.locations.size(); i++) {
113                         TermVectorsRequest termVectorsRequest = shardRequest.requests.get(i);
114                         responses.set(shardRequest.locations.get(i), new MultiTermVectorsItemResponse(null,
115                                 new MultiTermVectorsResponse.Failure(shardRequest.index(), termVectorsRequest.type(),
116                                         termVectorsRequest.id(), message)));
117                     }
118                     if (counter.decrementAndGet() == 0) {
119                         finishHim();
120                     }
121                 }
122 
123                 private void finishHim() {
124                     listener.onResponse(new MultiTermVectorsResponse(
125                             responses.toArray(new MultiTermVectorsItemResponse[responses.length()])));
126                 }
127             });
128         }
129     }
130 
131     @Override
132     public MultiTermVectorsRequest newRequestInstance() {
133         return new MultiTermVectorsRequest();
134     }
135 }
