20  package org.elasticsearch.index.fielddata.plain;
21  
22  import org.apache.lucene.util.BytesRef;
23  import org.apache.lucene.util.BytesRefIterator;
24  import org.apache.lucene.util.CharsRefBuilder;
25  import org.elasticsearch.ElasticsearchIllegalArgumentException;
26  import org.elasticsearch.common.Nullable;
27  import org.elasticsearch.common.geo.GeoPoint;
28  import org.elasticsearch.common.settings.Settings;
29  import org.elasticsearch.index.Index;
30  import org.elasticsearch.index.fielddata.*;
31  import org.elasticsearch.index.fielddata.IndexFieldData.XFieldComparatorSource.Nested;
32  import org.elasticsearch.index.mapper.FieldMapper.Names;
33  import org.elasticsearch.search.MultiValueMode;
34  
35  import java.io.IOException;
36  
37  abstract class AbstractIndexGeoPointFieldData extends AbstractIndexFieldData<AtomicGeoPointFieldData> implements IndexGeoPointFieldData {
38  
39      protected static class GeoPointEnum {
40  
41          private final BytesRefIterator termsEnum;
42          private final GeoPoint next;
43          private final CharsRefBuilder spare;
44  
45          protected GeoPointEnum(BytesRefIterator termsEnum) {
46              this.termsEnum = termsEnum;
47              next = new GeoPoint();
48              spare = new CharsRefBuilder();
49          }
50  
51          public GeoPoint next() throws IOException {
52              final BytesRef term = termsEnum.next();
53              if (term == null) {
54                  return null;
55              }
56              spare.copyUTF8Bytes(term);
57              int commaIndex = -1;
58              for (int i = 0; i < spare.length(); i++) {
59                  if (spare.charAt(i) == ',') { // saves a string creation
60                      commaIndex = i;
61                      break;
62                  }
63              }
64              if (commaIndex == -1) {
65                  assert false;
66                  return next.reset(0, 0);
67              }
68              final double lat = Double.parseDouble(new String(spare.chars(), 0, commaIndex));
69              final double lon = Double.parseDouble(new String(spare.chars(), commaIndex + 1, spare.length() - (commaIndex + 1)));
70              return next.reset(lat, lon);
71          }
72  
73      }
74  
75      public AbstractIndexGeoPointFieldData(Index index, Settings indexSettings, Names fieldNames, FieldDataType fieldDataType, IndexFieldDataCache cache) {
76          super(index, indexSettings, fieldNames, fieldDataType, cache);
77      }
78  
79      @Override
80      public final XFieldComparatorSource comparatorSource(@Nullable Object missingValue, MultiValueMode sortMode, Nested nested) {
81          throw new ElasticsearchIllegalArgumentException("can't sort on geo_point field without using specific sorting feature, like geo_distance");
82      }
83  
84  }
