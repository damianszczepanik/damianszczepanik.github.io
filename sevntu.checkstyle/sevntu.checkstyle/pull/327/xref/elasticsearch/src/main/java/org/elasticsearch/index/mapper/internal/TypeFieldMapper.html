62  public class TypeFieldMapper extends AbstractFieldMapper<String> implements InternalMapper, RootMapper {
63  
64      public static final String NAME = "_type";
65  
66      public static final String CONTENT_TYPE = "_type";
67  
68      public static class Defaults extends AbstractFieldMapper.Defaults {
69          public static final String NAME = TypeFieldMapper.NAME;
70  
71          public static final FieldType FIELD_TYPE = new FieldType(AbstractFieldMapper.Defaults.FIELD_TYPE);
72  
73          static {
74              FIELD_TYPE.setIndexOptions(IndexOptions.DOCS);
75              FIELD_TYPE.setTokenized(false);
76              FIELD_TYPE.setStored(false);
77              FIELD_TYPE.setOmitNorms(true);
78              FIELD_TYPE.freeze();
79          }
80      }
81  
82      public static class Builder extends AbstractFieldMapper.Builder<Builder, TypeFieldMapper> {
83  
84          public Builder() {
85              super(Defaults.NAME, new FieldType(Defaults.FIELD_TYPE));
86              indexName = Defaults.NAME;
87          }
88  
89          @Override
90          public TypeFieldMapper build(BuilderContext context) {
91              return new TypeFieldMapper(name, indexName, boost, fieldType, fieldDataSettings, context.indexSettings());
92          }
93      }
94  
95      public static class TypeParser implements Mapper.TypeParser {
96          @Override
97          public Mapper.Builder parse(String name, Map<String, Object> node, ParserContext parserContext) throws MapperParsingException {
98              if (parserContext.indexVersionCreated().onOrAfter(Version.V_2_0_0)) {
99                  throw new MapperParsingException(NAME + " is not configurable");
100             }
101             TypeFieldMapper.Builder builder = type();
102             parseField(builder, builder.name, node, parserContext);
103             return builder;
104         }
105     }
106 
107     public TypeFieldMapper(Settings indexSettings) {
108         this(Defaults.NAME, Defaults.NAME, Defaults.BOOST, new FieldType(Defaults.FIELD_TYPE), null, indexSettings);
109     }
110 
111     public TypeFieldMapper(String name, String indexName, float boost, FieldType fieldType, @Nullable Settings fieldDataSettings, Settings indexSettings) {
112         super(new Names(name, indexName, indexName, name), boost, fieldType, null, Lucene.KEYWORD_ANALYZER,
113                 Lucene.KEYWORD_ANALYZER, null, null, fieldDataSettings, indexSettings);
114     }
115 
116     @Override
117     public FieldType defaultFieldType() {
118         return Defaults.FIELD_TYPE;
119     }
120 
121     @Override
122     public FieldDataType defaultFieldDataType() {
123         return new FieldDataType("string");
124     }
125 
126     @Override
127     public boolean hasDocValues() {
128         return false;
129     }
130 
131     @Override
132     public String value(Object value) {
133         if (value == null) {
134             return null;
135         }
136         return value.toString();
137     }
138 
139     @Override
140     public Query termQuery(Object value, @Nullable QueryParseContext context) {
141         return new ConstantScoreQuery(context.cacheFilter(termFilter(value, context), null, context.autoFilterCachePolicy()));
142     }
143 
144     @Override
145     public Filter termFilter(Object value, @Nullable QueryParseContext context) {
146         if (fieldType.indexOptions() == IndexOptions.NONE) {
147             return new PrefixFilter(new Term(UidFieldMapper.NAME, Uid.typePrefixAsBytes(BytesRefs.toBytesRef(value))));
148         }
149         return new TermFilter(names().createIndexNameTerm(BytesRefs.toBytesRef(value)));
150     }
151 
152     @Override
153     public boolean useTermQueryWithQueryString() {
154         return true;
155     }
156 
157     @Override
158     public void preParse(ParseContext context) throws IOException {
159         super.parse(context);
160     }
161 
162     @Override
163     public void postParse(ParseContext context) throws IOException {
164     }
165 
166     @Override
167     public void parse(ParseContext context) throws IOException {
168         // we parse in pre parse
169     }
170 
171     @Override
172     public boolean includeInObject() {
173         return false;
174     }
175 
176     @Override
177     protected void parseCreateField(ParseContext context, List<Field> fields) throws IOException {
178         if (fieldType.indexOptions() == IndexOptions.NONE && !fieldType.stored()) {
179             return;
180         }
181         fields.add(new Field(names.indexName(), context.type(), fieldType));
182         if (hasDocValues()) {
183             fields.add(new SortedSetDocValuesField(names.indexName(), new BytesRef(context.type())));
184         }
185     }
186 
187     @Override
188     protected String contentType() {
189         return CONTENT_TYPE;
190     }
191 
192     @Override
193     public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
194         if (writePre2xSettings == false) {
195             return builder;
196         }
197         boolean includeDefaults = params.paramAsBoolean("include_defaults", false);
198 
199         // if all are defaults, no sense to write it at all
200         boolean indexed = fieldType.indexOptions() != IndexOptions.NONE;
201         boolean defaultIndexed = Defaults.FIELD_TYPE.indexOptions() != IndexOptions.NONE;
202         if (!includeDefaults && fieldType.stored() == Defaults.FIELD_TYPE.stored() && indexed == defaultIndexed) {
203             return builder;
204         }
205         builder.startObject(CONTENT_TYPE);
206         if (includeDefaults || fieldType.stored() != Defaults.FIELD_TYPE.stored()) {
207             builder.field("store", fieldType.stored());
208         }
209         if (includeDefaults || indexed != defaultIndexed) {
210             builder.field("index", indexTokenizeOptionToString(indexed, fieldType.tokenized()));
211         }
212         builder.endObject();
213         return builder;
214     }
215 
216     @Override
217     public void merge(Mapper mergeWith, MergeContext mergeContext) throws MergeMappingException {
218         // do nothing here, no merging, but also no exception
219     }
