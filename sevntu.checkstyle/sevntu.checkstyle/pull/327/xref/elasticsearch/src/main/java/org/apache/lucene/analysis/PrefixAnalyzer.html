20  package org.apache.lucene.analysis;
21  
22  import org.apache.lucene.analysis.tokenattributes.CharTermAttribute;
23  import org.apache.lucene.analysis.tokenattributes.PositionIncrementAttribute;
24  import org.elasticsearch.ElasticsearchIllegalArgumentException;
25  
26  import java.io.IOException;
27  import java.util.Collections;
28  import java.util.Iterator;
29  
30  /**
31   * This {@link Analyzer} wraps another analyzer and adds a set of prefixes to the
32   * underlying TokenStream. While these prefixes are iterated the position attribute
33   * will not be incremented. Also each prefix will be separated from the other tokens
34   * by a separator character.
35   * NOTE: The sequence of prefixes needs to be not empty 
36   */
37  public class PrefixAnalyzer extends Analyzer {
38  
39      private final char separator;
40      private final Iterable<? extends CharSequence> prefix;
41      private final Analyzer analyzer;
42  
43      /**
44       * Create a new {@link PrefixAnalyzer}. The separator will be set to the DEFAULT_SEPARATOR.
45       * 
46       * @param analyzer {@link Analyzer} to wrap
47       * @param prefix Single prefix 
48       */
49      public PrefixAnalyzer(Analyzer analyzer, char separator, CharSequence prefix) {
50          this(analyzer, separator, Collections.singleton(prefix));
51      }
52  
53      /**
54       * Create a new {@link PrefixAnalyzer}. The separator will be set to the DEFAULT_SEPARATOR.
55       * 
56       * @param analyzer {@link Analyzer} to wrap
57       * @param prefix {@link Iterable} of {@link CharSequence} which keeps all prefixes 
58       */
59      public PrefixAnalyzer(Analyzer analyzer, char separator, Iterable<? extends CharSequence> prefix) {
60          super();
61          this.analyzer = analyzer;
62          this.prefix = prefix;
63          this.separator = separator;
64      }
65  
66      @Override
67      protected TokenStreamComponents createComponents(String fieldName) {
68          TokenStreamComponents createComponents = analyzer.createComponents(fieldName);
69          TokenStream stream = new PrefixTokenFilter(createComponents.getTokenStream(), separator, prefix);
70          TokenStreamComponents tsc = new TokenStreamComponents(createComponents.getTokenizer(), stream);
71          return tsc;
72      }
73  
74      /**
75       * The {@link PrefixTokenFilter} wraps a {@link TokenStream} and adds a set
76       * prefixes ahead. The position attribute will not be incremented for the prefixes.
77       */
78      public static final class PrefixTokenFilter extends TokenFilter {
79  
80          private final char separator;
81          private final CharTermAttribute termAttr = addAttribute(CharTermAttribute.class);
82          private final PositionIncrementAttribute posAttr = addAttribute(PositionIncrementAttribute.class);
83          private final Iterable<? extends CharSequence> prefixes;
84  
85          private Iterator<? extends CharSequence> currentPrefix;
86  
87          /**
88           * Create a new {@link PrefixTokenFilter}. The separator will be set to the DEFAULT_SEPARATOR.
89           * 
90           * @param input {@link TokenStream} to wrap
91           * @param separator Character used separate prefixes from other tokens
92           * @param prefixes {@link Iterable} of {@link CharSequence} which keeps all prefixes 
93           */
94          public PrefixTokenFilter(TokenStream input, char separator, Iterable<? extends CharSequence> prefixes) {
95              super(input);
96              this.prefixes = prefixes;
97              this.currentPrefix = null;
98              this.separator = separator;
99              if (prefixes == null || !prefixes.iterator().hasNext()) {
100                 throw new ElasticsearchIllegalArgumentException("one or more prefixes needed");
101             }
102         }
103 
104         @Override
105         public boolean incrementToken() throws IOException {
106             if (currentPrefix != null) {
107                 if (!currentPrefix.hasNext()) {
108                     return input.incrementToken();
109                 } else {
110                     posAttr.setPositionIncrement(0);
111                 }
112             } else {
113                 currentPrefix = prefixes.iterator();
114                 termAttr.setEmpty();
115                 posAttr.setPositionIncrement(1);
116                 assert (currentPrefix.hasNext()) : "one or more prefixes needed";
117             }
118             termAttr.setEmpty();
119             termAttr.append(currentPrefix.next());
120             termAttr.append(separator);
121             return true;
122         }
123 
124         @Override
125         public void reset() throws IOException {
126             super.reset();
127             currentPrefix = null;
128         }
129     }
130 }
