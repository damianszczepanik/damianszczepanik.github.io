42  public final class PhraseSuggestParser implements SuggestContextParser {
43  
44      private PhraseSuggester suggester;
45  
46      public PhraseSuggestParser(PhraseSuggester suggester) {
47          this.suggester = suggester;
48      }
49  
50      @Override
51      public SuggestionSearchContext.SuggestionContext parse(XContentParser parser, MapperService mapperService) throws IOException {
52          PhraseSuggestionContext suggestion = new PhraseSuggestionContext(suggester);
53          XContentParser.Token token;
54          String fieldName = null;
55          boolean gramSizeSet = false; 
56          while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
57              if (token == XContentParser.Token.FIELD_NAME) {
58                  fieldName = parser.currentName();
59              } else if (token.isValue()) {
60                  if (!SuggestUtils.parseSuggestContext(parser, mapperService, fieldName, suggestion)) {
61                      if ("real_word_error_likelihood".equals(fieldName) || "realWorldErrorLikelihood".equals(fieldName)) {
62                          suggestion.setRealWordErrorLikelihood(parser.floatValue());
63                          if (suggestion.realworldErrorLikelyhood() <= 0.0) {
64                              throw new ElasticsearchIllegalArgumentException("real_word_error_likelihood must be > 0.0");
65                          }
66                      } else if ("confidence".equals(fieldName)) {
67                          suggestion.setConfidence(parser.floatValue());
68                          if (suggestion.confidence() < 0.0) {
69                              throw new ElasticsearchIllegalArgumentException("confidence must be >= 0.0");
70                          }
71                      } else if ("separator".equals(fieldName)) {
72                          suggestion.setSeparator(new BytesRef(parser.text()));
73                      } else if ("max_errors".equals(fieldName) || "maxErrors".equals(fieldName)) {
74                          suggestion.setMaxErrors(parser.floatValue());
75                          if (suggestion.maxErrors() <= 0.0) {
76                              throw new ElasticsearchIllegalArgumentException("max_error must be > 0.0");
77                          }
78                      } else if ("gram_size".equals(fieldName) || "gramSize".equals(fieldName)) {
79                          suggestion.setGramSize(parser.intValue());
80                          if (suggestion.gramSize() < 1) {
81                              throw new ElasticsearchIllegalArgumentException("gram_size must be >= 1");
82                          }
83                          gramSizeSet = true;
84                      } else if ("force_unigrams".equals(fieldName) || "forceUnigrams".equals(fieldName)) {
85                          suggestion.setRequireUnigram(parser.booleanValue());
86                      } else if ("token_limit".equals(fieldName) || "tokenLimit".equals(fieldName)) {
87                          int tokenLimit = parser.intValue();
88                          if (tokenLimit <= 0) {
89                              throw new ElasticsearchIllegalArgumentException("token_limit must be >= 1");
90                          }
91                          suggestion.setTokenLimit(tokenLimit);
92                      } else {
93                          throw new ElasticsearchIllegalArgumentException("suggester[phrase] doesn't support field [" + fieldName + "]");
94                      }
95                  }
96              } else if (token == Token.START_ARRAY) {
97                  if ("direct_generator".equals(fieldName) || "directGenerator".equals(fieldName)) {
98                      // for now we only have a single type of generators
99                      while ((token = parser.nextToken()) == Token.START_OBJECT) {
100                         PhraseSuggestionContext.DirectCandidateGenerator generator = new PhraseSuggestionContext.DirectCandidateGenerator();
101                         while ((token = parser.nextToken()) != Token.END_OBJECT) {
102                             if (token == XContentParser.Token.FIELD_NAME) {
103                                 fieldName = parser.currentName();
104                             }
105                             if (token.isValue()) {
106                                 parseCandidateGenerator(parser, mapperService, fieldName, generator);
107                             }
108                         }
109                         verifyGenerator(generator);
110                         suggestion.addGenerator(generator);
111                     }
112                 } else {
113                     throw new ElasticsearchIllegalArgumentException("suggester[phrase]  doesn't support array field [" + fieldName + "]");
114                 }
115             } else if (token == Token.START_OBJECT) {
116                 if ("smoothing".equals(fieldName)) {
117                     parseSmoothingModel(parser, suggestion, fieldName);
118                 } else if ("highlight".equals(fieldName)) {
119                     while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
120                         if (token == XContentParser.Token.FIELD_NAME) {
121                             fieldName = parser.currentName();
122                         } else if (token.isValue()) {
123                             if ("pre_tag".equals(fieldName) || "preTag".equals(fieldName)) {
124                                 suggestion.setPreTag(parser.utf8Bytes());
125                             } else if ("post_tag".equals(fieldName) || "postTag".equals(fieldName)) {
126                                 suggestion.setPostTag(parser.utf8Bytes());
127                             } else {
128                                 throw new ElasticsearchIllegalArgumentException(
129                                     "suggester[phrase][highlight] doesn't support field [" + fieldName + "]");
130                             }
131                         }
132                     }
133                 } else if ("collate".equals(fieldName)) {
134                     while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
135                         if (token == XContentParser.Token.FIELD_NAME) {
136                             fieldName = parser.currentName();
137                         } else if ("query".equals(fieldName) || "filter".equals(fieldName)) {
138                             String templateNameOrTemplateContent;
139                             if (token == XContentParser.Token.START_OBJECT) {
140                                 XContentBuilder builder = XContentBuilder.builder(parser.contentType().xContent());
141                                 builder.copyCurrentStructure(parser);
142                                 templateNameOrTemplateContent = builder.string();
143                             } else {
144                                 templateNameOrTemplateContent = parser.text();
145                             }
146                             if (templateNameOrTemplateContent == null) {
147                                 throw new ElasticsearchIllegalArgumentException("suggester[phrase][collate] no query/filter found in collate object");
148                             }
149                             if (suggestion.getCollateFilterScript() != null) {
150                                 throw new ElasticsearchIllegalArgumentException("suggester[phrase][collate] filter already set, doesn't support additional [" + fieldName + "]");
151                             }
152                             if (suggestion.getCollateQueryScript() != null) {
153                                 throw new ElasticsearchIllegalArgumentException("suggester[phrase][collate] query already set, doesn't support additional [" + fieldName + "]");
154                             }
155                             CompiledScript compiledScript = suggester.scriptService().compile(MustacheScriptEngineService.NAME, templateNameOrTemplateContent, ScriptService.ScriptType.INLINE);
156                             if ("query".equals(fieldName)) {
157                                 suggestion.setCollateQueryScript(compiledScript);
158                             } else {
159                                 suggestion.setCollateFilterScript(compiledScript);
160                             }
161                         } else if ("preference".equals(fieldName)) {
162                             suggestion.setPreference(parser.text());
163                         } else if ("params".equals(fieldName)) {
164                             suggestion.setCollateScriptParams(parser.map());
165                         } else if ("prune".equals(fieldName)) {
166                             if (parser.isBooleanValue()) {
167                                 suggestion.setCollatePrune(parser.booleanValue());
168                             } else {
169                                 throw new ElasticsearchIllegalArgumentException("suggester[phrase][collate] prune must be either 'true' or 'false'");
170                             }
171                         } else {
172                             throw new ElasticsearchIllegalArgumentException(
173                                     "suggester[phrase][collate] doesn't support field [" + fieldName + "]");
174                         }
175                     }
176                 } else {
177                     throw new ElasticsearchIllegalArgumentException("suggester[phrase]  doesn't support array field [" + fieldName + "]");
178                 }
179             } else {
180                 throw new ElasticsearchIllegalArgumentException("suggester[phrase] doesn't support field [" + fieldName + "]");
181             }
182         }
183         
184         if (suggestion.getField() == null) {
185             throw new ElasticsearchIllegalArgumentException("The required field option is missing");
186         }
187 
188         FieldMapper fieldMapper = mapperService.smartNameFieldMapper(suggestion.getField());
189         if (fieldMapper == null) {
190             throw new ElasticsearchIllegalArgumentException("No mapping found for field [" + suggestion.getField() + "]");
191         } else if (suggestion.getAnalyzer() == null) {
192             // no analyzer name passed in, so try the field's analyzer, or the default analyzer
193             if (fieldMapper.searchAnalyzer() == null) {
194                 suggestion.setAnalyzer(mapperService.searchAnalyzer());
195             } else {
196                 suggestion.setAnalyzer(fieldMapper.searchAnalyzer());
197             }
198         }
199         
200         if (suggestion.model() == null) {
201             suggestion.setModel(StupidBackoffScorer.FACTORY);
202         }
203         
204         if (!gramSizeSet || suggestion.generators().isEmpty()) {
205             final ShingleTokenFilterFactory.Factory shingleFilterFactory = SuggestUtils.getShingleFilterFactory(suggestion.getAnalyzer());
206             if (!gramSizeSet) {
207                 // try to detect the shingle size
208                 if (shingleFilterFactory != null) {
209                     suggestion.setGramSize(shingleFilterFactory.getMaxShingleSize());
210                     if (suggestion.getAnalyzer() == null && shingleFilterFactory.getMinShingleSize() > 1 && !shingleFilterFactory.getOutputUnigrams()) {
211                         throw new ElasticsearchIllegalArgumentException("The default analyzer for field: [" + suggestion.getField() + "] doesn't emit unigrams. If this is intentional try to set the analyzer explicitly");
212                     }
213                 }
214             }
215             if (suggestion.generators().isEmpty()) {
216                 if (shingleFilterFactory != null && shingleFilterFactory.getMinShingleSize() > 1 && !shingleFilterFactory.getOutputUnigrams() && suggestion.getRequireUnigram()) {
217                     throw new ElasticsearchIllegalArgumentException("The default candidate generator for phrase suggest can't operate on field: [" + suggestion.getField() + "] since it doesn't emit unigrams. If this is intentional try to set the candidate generator field explicitly");
218                 }
219                 // use a default generator on the same field
220                 DirectCandidateGenerator generator = new DirectCandidateGenerator();
221                 generator.setField(suggestion.getField());
222                 suggestion.addGenerator(generator);
223             }
224         }
225         
226         
227         
228         return suggestion;
229     }

231     public void parseSmoothingModel(XContentParser parser, PhraseSuggestionContext suggestion, String fieldName) throws IOException {
232         XContentParser.Token token;
233         while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
234             if (token == XContentParser.Token.FIELD_NAME) {
235                 fieldName = parser.currentName();
236                 if ("linear".equals(fieldName)) {
237                     ensureNoSmoothing(suggestion);
238                     final double[] lambdas = new double[3];
239                     while ((token = parser.nextToken()) != Token.END_OBJECT) {
240                         if (token == XContentParser.Token.FIELD_NAME) {
241                             fieldName = parser.currentName();
242                         }
243                         if (token.isValue()) {
244                             if ("trigram_lambda".equals(fieldName) || "trigramLambda".equals(fieldName)) {
245                                 lambdas[0] = parser.doubleValue();
246                                 if (lambdas[0] < 0) {
247                                     throw new ElasticsearchIllegalArgumentException("trigram_lambda must be positive");
248                                 }
249                             } else if ("bigram_lambda".equals(fieldName) || "bigramLambda".equals(fieldName)) {
250                                 lambdas[1] = parser.doubleValue();
251                                 if (lambdas[1] < 0) {
252                                     throw new ElasticsearchIllegalArgumentException("bigram_lambda must be positive");
253                                 }
254                             } else if ("unigram_lambda".equals(fieldName) || "unigramLambda".equals(fieldName)) {
255                                 lambdas[2] = parser.doubleValue();
256                                 if (lambdas[2] < 0) {
257                                     throw new ElasticsearchIllegalArgumentException("unigram_lambda must be positive");
258                                 }
259                             } else {
260                                 throw new ElasticsearchIllegalArgumentException(
261                                         "suggester[phrase][smoothing][linear] doesn't support field [" + fieldName + "]");
262                             }
263                         }
264                     }
265                     double sum = 0.0d;
266                     for (int i = 0; i < lambdas.length; i++) {
267                         sum += lambdas[i];
268                     }
269                     if (Math.abs(sum - 1.0) > 0.001) {
270                         throw new ElasticsearchIllegalArgumentException("linear smoothing lambdas must sum to 1");
271                     }
272                     suggestion.setModel(new WordScorer.WordScorerFactory() {
273                         @Override
274                         public WordScorer newScorer(IndexReader reader, Terms terms, String field, double realWordLikelyhood, BytesRef separator)
275                                 throws IOException {
276                             return new LinearInterpoatingScorer(reader, terms, field, realWordLikelyhood, separator, lambdas[0], lambdas[1],
277                                     lambdas[2]);
278                         }
279                     });
280                 } else if ("laplace".equals(fieldName)) {
281                     ensureNoSmoothing(suggestion);
282                     double theAlpha = 0.5;
283 
284                     while ((token = parser.nextToken()) != Token.END_OBJECT) {
285                         if (token == XContentParser.Token.FIELD_NAME) {
286                             fieldName = parser.currentName();
287                         }
288                         if (token.isValue() && "alpha".equals(fieldName)) {
289                             theAlpha = parser.doubleValue();
290                         }
291                     }
292                     final double alpha = theAlpha;
293                     suggestion.setModel(new WordScorer.WordScorerFactory() {
294                         @Override
295                         public WordScorer newScorer(IndexReader reader, Terms terms, String field, double realWordLikelyhood, BytesRef separator)
296                                 throws IOException {
297                             return new LaplaceScorer(reader, terms,  field, realWordLikelyhood, separator, alpha);
298                         }
299                     });
300 
301                 } else if ("stupid_backoff".equals(fieldName) || "stupidBackoff".equals(fieldName)) {
302                     ensureNoSmoothing(suggestion);
303                     double theDiscount = 0.4;
304                     while ((token = parser.nextToken()) != Token.END_OBJECT) {
305                         if (token == XContentParser.Token.FIELD_NAME) {
306                             fieldName = parser.currentName();
307                         }
308                         if (token.isValue() && "discount".equals(fieldName)) {
309                             theDiscount = parser.doubleValue();
310                         }
311                     }
312                     final double discount = theDiscount;
313                     suggestion.setModel(new WordScorer.WordScorerFactory() {
314                         @Override
315                         public WordScorer newScorer(IndexReader reader, Terms terms, String field, double realWordLikelyhood, BytesRef separator)
316                                 throws IOException {
317                             return new StupidBackoffScorer(reader, terms, field, realWordLikelyhood, separator, discount);
318                         }
319                     });
320 
321                 } else {
322                     throw new ElasticsearchIllegalArgumentException("suggester[phrase] doesn't support object field [" + fieldName + "]");
323                 }
324             }
325         }
326     }

328     private void ensureNoSmoothing(PhraseSuggestionContext suggestion) {
329         if (suggestion.model() != null) {
330             throw new ElasticsearchIllegalArgumentException("only one smoothing model supported");
331         }
332     }

334     private void verifyGenerator(PhraseSuggestionContext.DirectCandidateGenerator suggestion) {
335         // Verify options and set defaults
336         if (suggestion.field() == null) {
337             throw new ElasticsearchIllegalArgumentException("The required field option is missing");
338         }
339     }

341     private void parseCandidateGenerator(XContentParser parser, MapperService mapperService, String fieldName,
342             PhraseSuggestionContext.DirectCandidateGenerator generator) throws IOException {
343         if (!SuggestUtils.parseDirectSpellcheckerSettings(parser, fieldName, generator)) {
344             if ("field".equals(fieldName)) {
345                 generator.setField(parser.text());
346                 if (mapperService.smartNameFieldMapper(generator.field()) == null) {
347                     throw new ElasticsearchIllegalArgumentException("No mapping found for field [" + generator.field() + "]");
348                 }
349             } else if ("size".equals(fieldName)) {
350                 generator.size(parser.intValue());
351             } else if ("pre_filter".equals(fieldName) || "preFilter".equals(fieldName)) {
352                 String analyzerName = parser.text();
353                 Analyzer analyzer = mapperService.analysisService().analyzer(analyzerName);
354                 if (analyzer == null) {
355                     throw new ElasticsearchIllegalArgumentException("Analyzer [" + analyzerName + "] doesn't exists");
356                 }
357                 generator.preFilter(analyzer);
358             } else if ("post_filter".equals(fieldName) || "postFilter".equals(fieldName)) {
359                 String analyzerName = parser.text();
360                 Analyzer analyzer = mapperService.analysisService().analyzer(analyzerName);
361                 if (analyzer == null) {
362                     throw new ElasticsearchIllegalArgumentException("Analyzer [" + analyzerName + "] doesn't exists");
363                 }
364                 generator.postFilter(analyzer);
365             } else {
366                 throw new ElasticsearchIllegalArgumentException("CandidateGenerator doesn't support [" + fieldName + "]");
367             }
368         }
369     }
