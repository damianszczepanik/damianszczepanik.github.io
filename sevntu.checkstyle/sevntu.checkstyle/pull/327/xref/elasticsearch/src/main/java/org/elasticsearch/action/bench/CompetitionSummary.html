32  /**
33   * Summary statistics for a benchmark search competition.
34   *
35   * Statistics are calculated over all iteration results for all nodes
36   * that executed the competition.
37   *
38   * Since values are calculated lazily on first access, users of this class
39   * should first call computeSummaryStatistics() prior to accessing individual
40   * measurements.
41   */
42  public class CompetitionSummary implements ToXContent {
43  
44      private List<CompetitionNodeResult> nodeResults;
45      private boolean computed = false;
46  
47      private long min = 0;
48      private long max = 0;
49      private long totalTime = 0;
50      private long sumTotalHits = 0;
51      private long totalIterations = 0;
52      private long completedIterations = 0;
53      private long totalQueries = 0;
54      private double avgWarmupTime = 0;
55      private int concurrency = 0;
56      private int multiplier = 0;
57      private double mean = 0;
58      private double millisPerHit = 0.0;
59      private double stdDeviation = 0.0;
60      private double queriesPerSecond = 0.0;
61      private double[] percentiles;
62      Map<Double, Double> percentileValues = new TreeMap<>();
63  
64      private List<Tuple<String, CompetitionIteration.SlowRequest>> slowest = new ArrayList<>();
65  
66      public CompetitionSummary() { }
67  
68      public CompetitionSummary(List<CompetitionNodeResult> nodeResults, int concurrency, int multiplier, double[] percentiles) {
69          this.nodeResults = nodeResults;
70          this.concurrency = concurrency;
71          this.multiplier = multiplier;
72          this.percentiles = percentiles;
73      }
74  
75      /**
76       * Gets node-level competition results
77       * @return  A list of node-level competition results
78       */
79      public List<CompetitionNodeResult> nodeResults() {
80          return nodeResults;
81      }

83      /**
84       * Calculates statistical measures from raw measurements. Should be called prior to accessing
85       * individual measurements.
86       */
87      public void computeSummaryStatistics() {
88  
89          if (computed) {
90              return;
91          }
92  
93          long totalWarmupTime = 0;
94          final SinglePassStatistics single = new SinglePassStatistics();
95  
96          for (CompetitionNodeResult nodeResult : nodeResults) {
97  
98              totalWarmupTime += nodeResult.warmUpTime();
99              totalIterations += nodeResult.totalIterations();
100             completedIterations += nodeResult.completedIterations();
101 
102             // only calculate statistics for iterations completed thus far
103             for (int i = 0; i < nodeResult.completedIterations(); i++) {
104 
105                 CompetitionIteration competitionIteration = nodeResult.iterations().get(i);
106                 CompetitionIterationData iterationData = competitionIteration.competitionIterationData();
107                 long[] data = iterationData.data();
108 
109                 for (long datum : data) {
110                     if (datum > -1) {   // ignore unset values in the underlying array
111                         single.push(datum);
112                     }
113                 }
114 
115                 totalQueries += competitionIteration.numQueries();
116                 totalTime += competitionIteration.totalTime();
117                 sumTotalHits += competitionIteration.sumTotalHits();
118 
119                 // keep track of slowest requests
120                 if (competitionIteration.slowRequests() != null) {
121                     for (CompetitionIteration.SlowRequest slow : competitionIteration.slowRequests()) {
122                         slowest.add(new Tuple<>(nodeResult.nodeName(), slow));
123                     }
124                 }
125             }
126         }
127 
128         min = single.min();
129         max = single.max();
130         mean = single.mean();
131         stdDeviation = single.stddev();
132         avgWarmupTime = (nodeResults.size() > 0) ? totalWarmupTime / nodeResults.size() : 0.0;
133         queriesPerSecond = (single.sum() > 0) ? (totalQueries * (1000.0 / (double) single.sum())) : 0.0;
134         millisPerHit = (sumTotalHits > 0) ? (totalTime / (double) sumTotalHits) : 0.0;
135 
136         for (double percentile : percentiles) {
137             percentileValues.put(percentile, single.percentile(percentile / 100.0d));
138         }
139 
140         CollectionUtil.timSort(slowest, new Comparator<Tuple<String, CompetitionIteration.SlowRequest>>() {
141             @Override
142             public int compare(Tuple<String, CompetitionIteration.SlowRequest> o1, Tuple<String, CompetitionIteration.SlowRequest> o2) {
143                 return Long.compare(o2.v2().maxTimeTaken(), o1.v2().maxTimeTaken());
144             }
145         });
146         computed = true;
147     }
148 
149     @Override
150     public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
151 
152         computeSummaryStatistics();
153 
154         builder.startObject(Fields.SUMMARY);
155         builder.startArray(Fields.NODES);
156         for (CompetitionNodeResult nodeResult : nodeResults) {
157             builder.field(nodeResult.nodeName());
158         }
159         builder.endArray();
160 
161         builder.field(Fields.TOTAL_ITERATIONS, totalIterations);
162         builder.field(Fields.COMPLETED_ITERATIONS, completedIterations);
163         builder.field(Fields.TOTAL_QUERIES, totalQueries);
164         builder.field(Fields.CONCURRENCY, concurrency);
165         builder.field(Fields.MULTIPLIER, multiplier);
166         builder.field(Fields.AVG_WARMUP_TIME, avgWarmupTime);
167 
168         builder.startObject(Fields.STATISTICS);
169         builder.field(Fields.MIN, min == Long.MAX_VALUE ? 0 : min);
170         builder.field(Fields.MAX, max == Long.MIN_VALUE ? 0 : max);
171         builder.field(Fields.MEAN, mean);
172         builder.field(Fields.QPS, queriesPerSecond);
173         builder.field(Fields.STD_DEV, stdDeviation);
174         builder.field(Fields.MILLIS_PER_HIT, millisPerHit);
175 
176         for (Map.Entry<Double, Double> entry : percentileValues.entrySet()) {
177             // Change back to integral value for display purposes
178             builder.field(new XContentBuilderString("percentile_" + entry.getKey().longValue()),
179                     (entry.getValue().isNaN()) ? 0.0 : entry.getValue());
180         }
181 
182         builder.endObject();
183 
184         if (totalIterations > 0 && slowest.size() > 0) {
185             builder.startArray(Fields.SLOWEST);
186             int n = (int) (slowest.size() / totalIterations);
187             for (int i = 0; i < n; i++) {
188                 builder.startObject();
189                 builder.field(Fields.NODE, slowest.get(i).v1());
190                 slowest.get(i).v2().toXContent(builder, params);
191                 builder.endObject();
192             }
193             builder.endArray();
194         }
195 
196         builder.endObject();
197         return builder;
198     }
