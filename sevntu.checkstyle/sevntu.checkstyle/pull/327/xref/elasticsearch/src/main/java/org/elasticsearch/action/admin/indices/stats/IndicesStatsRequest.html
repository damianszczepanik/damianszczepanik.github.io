20  package org.elasticsearch.action.admin.indices.stats;
21  
22  import org.elasticsearch.action.admin.indices.stats.CommonStatsFlags.Flag;
23  import org.elasticsearch.action.support.broadcast.BroadcastOperationRequest;
24  import org.elasticsearch.common.io.stream.StreamInput;
25  import org.elasticsearch.common.io.stream.StreamOutput;
26  
27  import java.io.IOException;
28  
29  /**
30   * A request to get indices level stats. Allow to enable different stats to be returned.
31   * <p/>
32   * <p>By default, all statistics are enabled.
33   * <p/>
34   * <p>All the stats to be returned can be cleared using {@link #clear()}, at which point, specific
35   * stats can be enabled.
36   */
37  public class IndicesStatsRequest extends BroadcastOperationRequest<IndicesStatsRequest> {
38  
39      private CommonStatsFlags flags = new CommonStatsFlags();
40  
41      /**
42       * Sets all flags to return all stats.
43       */
44      public IndicesStatsRequest all() {
45          flags.all();
46          return this;
47      }
48  
49      /**
50       * Clears all stats.
51       */
52      public IndicesStatsRequest clear() {
53          flags.clear();
54          return this;
55      }
56  
57      /**
58       * Document types to return stats for. Mainly affects {@link #indexing(boolean)} when
59       * enabled, returning specific indexing stats for those types.
60       */
61      public IndicesStatsRequest types(String... types) {
62          flags.types(types);
63          return this;
64      }
65  
66      /**
67       * Document types to return stats for. Mainly affects {@link #indexing(boolean)} when
68       * enabled, returning specific indexing stats for those types.
69       */
70      public String[] types() {
71          return this.flags.types();
72      }
73  
74      /**
75       * Sets specific search group stats to retrieve the stats for. Mainly affects search
76       * when enabled.
77       */
78      public IndicesStatsRequest groups(String... groups) {
79          flags.groups(groups);
80          return this;
81      }
82  
83      public String[] groups() {
84          return this.flags.groups();
85      }
86  
87      public IndicesStatsRequest docs(boolean docs) {
88          flags.set(Flag.Docs, docs);
89          return this;
90      }
91  
92      public boolean docs() {
93          return flags.isSet(Flag.Docs);
94      }
95  
96      public IndicesStatsRequest store(boolean store) {
97          flags.set(Flag.Store, store);
98          return this;
99      }
100 
101     public boolean store() {
102         return flags.isSet(Flag.Store);
103     }
104 
105     public IndicesStatsRequest indexing(boolean indexing) {
106         flags.set(Flag.Indexing, indexing);
107 
108         return this;
109     }
110 
111     public boolean indexing() {
112         return flags.isSet(Flag.Indexing);
113     }
114 
115     public IndicesStatsRequest get(boolean get) {
116         flags.set(Flag.Get, get);
117         return this;
118     }
119 
120     public boolean get() {
121         return flags.isSet(Flag.Get);
122     }
123 
124     public IndicesStatsRequest search(boolean search) {
125         flags.set(Flag.Search, search);
126         return this;
127     }
128 
129     public boolean search() {
130         return flags.isSet(Flag.Search);
131     }
132 
133     public IndicesStatsRequest merge(boolean merge) {
134         flags.set(Flag.Merge, merge);
135         return this;
136     }
137 
138     public boolean merge() {
139         return flags.isSet(Flag.Merge);
140     }
141

143         flags.set(Flag.Refresh, refresh);
144         return this;
145     }
146 
147     public boolean refresh() {
148         return flags.isSet(Flag.Refresh);
149     }
150

152         flags.set(Flag.Flush, flush);

154     }
155 
156     public boolean flush() {
157         return flags.isSet(Flag.Flush);
158     }
159 
160     public IndicesStatsRequest warmer(boolean warmer) {
161         flags.set(Flag.Warmer, warmer);
162         return this;
163     }
164

166         return flags.isSet(Flag.Warmer);
167     }
168

170         flags.set(Flag.FilterCache, filterCache);
171         return this;
172     }
173

179         flags.set(Flag.IdCache, idCache);
180         return this;
181     }
182 
183     public boolean idCache() {
184         return flags.isSet(Flag.IdCache);
185     }

190     }
191

244     }

248     }

252         return this;
253     }

257     }

260         flags.set(Flag.QueryCache, queryCache);
261         return this;
262     }
263 
264     public boolean queryCache() {
265         return flags.isSet(Flag.QueryCache);
266     }
267 
268     @Override
269     public void writeTo(StreamOutput out) throws IOException {
270         super.writeTo(out);
271         flags.writeTo(out);
272     }
273 
274     @Override
275     public void readFrom(StreamInput in) throws IOException {
276         super.readFrom(in);
277         flags = CommonStatsFlags.readCommonStatsFlags(in);
278     }
279 }
