80      public GroupShardsIterator deleteByQueryShards(ClusterState clusterState, String index, @Nullable Set<String> routing) throws IndexMissingException {
81          if (routing == null || routing.isEmpty()) {
82              return indexRoutingTable(clusterState, index).groupByShardsIt();
83          }
84  
85          // we use set here and not identity set since we might get duplicates
86          HashSet<ShardIterator> set = new HashSet<>();
87          IndexRoutingTable indexRouting = indexRoutingTable(clusterState, index);
88          for (String r : routing) {
89              int shardId = shardId(clusterState, index, null, null, r);
90              IndexShardRoutingTable indexShard = indexRouting.shard(shardId);
91              if (indexShard == null) {
92                  throw new IndexShardMissingException(new ShardId(index, shardId));
93              }
94              set.add(indexShard.shardsRandomIt());
95          }
96          return new GroupShardsIterator(Lists.newArrayList(set));
97      }

99      public int searchShardsCount(ClusterState clusterState, String[] indices, String[] concreteIndices, @Nullable Map<String, Set<String>> routing, @Nullable String preference) throws IndexMissingException {
100         final Set<IndexShardRoutingTable> shards = computeTargetedShards(clusterState, concreteIndices, routing);
101         return shards.size();
102     }

104     public GroupShardsIterator searchShards(ClusterState clusterState, String[] indices, String[] concreteIndices, @Nullable Map<String, Set<String>> routing, @Nullable String preference) throws IndexMissingException {
105         final Set<IndexShardRoutingTable> shards = computeTargetedShards(clusterState, concreteIndices, routing);
106         final Set<ShardIterator> set = new HashSet<>(shards.size());
107         for (IndexShardRoutingTable shard : shards) {
108             ShardIterator iterator = preferenceActiveShardIterator(shard, clusterState.nodes().localNodeId(), clusterState.nodes(), preference);
109             if (iterator != null) {
110                 set.add(iterator);
111             }
112         }
113         return new GroupShardsIterator(Lists.newArrayList(set));
114     }

116     private static final Map<String, Set<String>> EMPTY_ROUTING = Collections.emptyMap();
117 
118     private Set<IndexShardRoutingTable> computeTargetedShards(ClusterState clusterState, String[] concreteIndices, @Nullable Map<String, Set<String>> routing) throws IndexMissingException {
119         routing = routing == null ? EMPTY_ROUTING : routing; // just use an empty map
120         final Set<IndexShardRoutingTable> set = new HashSet<>();
121         // we use set here and not list since we might get duplicates
122         for (String index : concreteIndices) {
123             final IndexRoutingTable indexRouting = indexRoutingTable(clusterState, index);
124             final Set<String> effectiveRouting = routing.get(index);
125             if (effectiveRouting != null) {
126                 for (String r : effectiveRouting) {
127                     int shardId = shardId(clusterState, index, null, null, r);
128                     IndexShardRoutingTable indexShard = indexRouting.shard(shardId);
129                     if (indexShard == null) {
130                         throw new IndexShardMissingException(new ShardId(index, shardId));
131                     }
132                     // we might get duplicates, but that's ok, they will override one another
133                     set.add(indexShard);
134                 }
135             } else {
136                 for (IndexShardRoutingTable indexShard : indexRouting) {
137                     set.add(indexShard);
138                 }
139             }
140         }
141         return set;
142     }

144     private ShardIterator preferenceActiveShardIterator(IndexShardRoutingTable indexShard, String localNodeId, DiscoveryNodes nodes, @Nullable String preference) {
145         if (preference == null || preference.isEmpty()) {
146             String[] awarenessAttributes = awarenessAllocationDecider.awarenessAttributes();
147             if (awarenessAttributes.length == 0) {
148                 return indexShard.activeInitializingShardsRandomIt();
149             } else {
150                 return indexShard.preferAttributesActiveInitializingShardsIt(awarenessAttributes, nodes);
151             }
152         }
153         if (preference.charAt(0) == '_') {
154             Preference preferenceType = Preference.parse(preference);
155             if (preferenceType == Preference.SHARDS) {
156                 // starts with _shards, so execute on specific ones
157                 int index = preference.indexOf(';');
158 
159                 String shards;
160                 if (index == -1) {
161                     shards = preference.substring(Preference.SHARDS.type().length() + 1);
162                 } else {
163                     shards = preference.substring(Preference.SHARDS.type().length() + 1, index);
164                 }
165                 String[] ids = Strings.splitStringByCommaToArray(shards);
166                 boolean found = false;
167                 for (String id : ids) {
168                     if (Integer.parseInt(id) == indexShard.shardId().id()) {
169                         found = true;
170                         break;
171                     }
172                 }
173                 if (!found) {
174                     return null;
175                 }
176                 // no more preference
177                 if (index == -1 || index == preference.length() - 1) {
178                     String[] awarenessAttributes = awarenessAllocationDecider.awarenessAttributes();
179                     if (awarenessAttributes.length == 0) {
180                         return indexShard.activeInitializingShardsRandomIt();
181                     } else {
182                         return indexShard.preferAttributesActiveInitializingShardsIt(awarenessAttributes, nodes);
183                     }
184                 } else {
185                     // update the preference and continue
186                     preference = preference.substring(index + 1);
187                 }
188             }
189             preferenceType = Preference.parse(preference);
190             switch (preferenceType) {
191                 case PREFER_NODE:
192                     return indexShard.preferNodeActiveInitializingShardsIt(preference.substring(Preference.PREFER_NODE.type().length() + 1));
193                 case LOCAL:
194                     return indexShard.preferNodeActiveInitializingShardsIt(localNodeId);
195                 case PRIMARY:
196                     return indexShard.primaryActiveInitializingShardIt();
197                 case PRIMARY_FIRST:
198                     return indexShard.primaryFirstActiveInitializingShardsIt();
199                 case ONLY_LOCAL:
200                     return indexShard.onlyNodeActiveInitializingShardsIt(localNodeId);
201                 case ONLY_NODE:
202                     String nodeId = preference.substring(Preference.ONLY_NODE.type().length() + 1);
203                     ensureNodeIdExists(nodes, nodeId);
204                     return indexShard.onlyNodeActiveInitializingShardsIt(nodeId);
205                 default:
206                     throw new ElasticsearchIllegalArgumentException("unknown preference [" + preferenceType + "]");
207             }
208         }
209         // if not, then use it as the index
210         String[] awarenessAttributes = awarenessAllocationDecider.awarenessAttributes();
211         if (awarenessAttributes.length == 0) {
212             return indexShard.activeInitializingShardsIt(DjbHashFunction.DJB_HASH(preference));
213         } else {
214             return indexShard.preferAttributesActiveInitializingShardsIt(awarenessAttributes, nodes, DjbHashFunction.DJB_HASH(preference));
215         }
216     }

237     protected IndexShardRoutingTable shards(ClusterState clusterState, String index, String type, String id, String routing) {
238         int shardId = shardId(clusterState, index, type, id, routing);

242     protected IndexShardRoutingTable shards(ClusterState clusterState, String index, int shardId) {
243         IndexShardRoutingTable indexShard = indexRoutingTable(clusterState, index).shard(shardId);
244         if (indexShard == null) {
245             throw new IndexShardMissingException(new ShardId(index, shardId));
246         }
247         return indexShard;
248     }

250     private int shardId(ClusterState clusterState, String index, String type, String id, @Nullable String routing) {
251         final IndexMetaData indexMetaData = indexMetaData(clusterState, index);
252         final Version createdVersion = indexMetaData.getCreationVersion();
253         final HashFunction hashFunction = indexMetaData.getRoutingHashFunction();
254         final boolean useType = indexMetaData.getRoutingUseType();
255 
256         final int hash;
257         if (routing == null) {
258             if (!useType) {
259                 hash = hash(hashFunction, id);
260             } else {
261                 hash = hash(hashFunction, type, id);
262             }
263         } else {
264             hash = hash(hashFunction, routing);
265         }
266         if (createdVersion.onOrAfter(Version.V_2_0_0)) {
267             return MathUtils.mod(hash, indexMetaData.numberOfShards());
268         } else {
269             return Math.abs(hash % indexMetaData.numberOfShards());
270         }
271     }

273     protected int hash(HashFunction hashFunction, String routing) {
274         return hashFunction.hash(routing);
275     }

277     @Deprecated
278     protected int hash(HashFunction hashFunction, String type, String id) {
279         if (type == null || "_all".equals(type)) {
280             throw new ElasticsearchIllegalArgumentException("Can't route an operation with no type and having type part of the routing (for backward comp)");
281         }
282         return hashFunction.hash(type, id);

285     private void ensureNodeIdExists(DiscoveryNodes nodes, String nodeId) {
286         if (!nodes.dataNodes().keys().contains(nodeId)) {
287             throw new ElasticsearchIllegalArgumentException("No data node with id[" + nodeId + "] found");
288         }
289     }
