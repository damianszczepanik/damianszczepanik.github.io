20  package org.elasticsearch.search.aggregations.bucket.histogram;
21  
22  import org.elasticsearch.common.unit.TimeValue;
23  import org.elasticsearch.common.xcontent.XContentBuilder;
24  import org.elasticsearch.search.aggregations.ValuesSourceAggregationBuilder;
25  import org.elasticsearch.search.builder.SearchSourceBuilderException;
26  import org.joda.time.DateTime;
27  
28  import java.io.IOException;
29  
30  /**
31   * Builder for the {@link DateHistogram} aggregation.
32   */
33  public class DateHistogramBuilder extends ValuesSourceAggregationBuilder<DateHistogramBuilder> {
34  
35      private Object interval;
36      private Histogram.Order order;
37      private Long minDocCount;
38      private Object extendedBoundsMin;
39      private Object extendedBoundsMax;
40      private String timeZone;
41      private String format;
42      private String offset;
43      private float factor = 1.0f;
44  
45      /**
46       * Sole constructor.
47       */
48      public DateHistogramBuilder(String name) {
49          super(name, InternalDateHistogram.TYPE.name());
50      }
51  
52      /**
53       * Set the interval in milliseconds.
54       */
55      public DateHistogramBuilder interval(long interval) {
56          this.interval = interval;
57          return this;
58      }
59  
60      /**
61       * Set the interval.
62       */
63      public DateHistogramBuilder interval(DateHistogramInterval interval) {
64          this.interval = interval;
65          return this;
66      }
67  
68      /**
69       * Set the order by which the buckets will be returned.
70       */
71      public DateHistogramBuilder order(Histogram.Order order) {
72          this.order = order;
73          return this;
74      }
75  
76      /**
77       * Set the minimum document count per bucket. Buckets with less documents
78       * than this min value will not be returned.
79       */
80      public DateHistogramBuilder minDocCount(long minDocCount) {
81          this.minDocCount = minDocCount;
82          return this;
83      }
84  
85      /**
86       * Set the timezone in which to translate dates before computing buckets.
87       */
88      public DateHistogramBuilder timeZone(String timeZone) {
89          this.timeZone = timeZone;
90          return this;
91      }
92  
93      /**
94       * @param offset sets the offset of time intervals in this histogram
95       * @return the current builder
96       */
97      public DateHistogramBuilder offset(String offset) {
98         this.offset = offset;
99         return this;
100     }
101 
102     /**
103      * Set a factor to apply to values of the field, typically used if times
104      * are stored in seconds instead of milliseconds.
105      */
106     public DateHistogramBuilder factor(float factor) {
107         this.factor = factor;
108         return this;
109     }
110 
111     /**
112      * Set the format to use for dates.
113      */
114     public DateHistogramBuilder format(String format) {
115         this.format = format;
116         return this;
117     }
118 
119     /**
120      * Set extended bounds for the histogram. In case the lower value in the
121      * histogram would be greater than <code>min</code> or the upper value would
122      * be less than <code>max</code>, empty buckets will be generated.
123      */
124     public DateHistogramBuilder extendedBounds(Long min, Long max) {
125         extendedBoundsMin = min;
126         extendedBoundsMax = max;
127         return this;
128     }
129 
130     /**
131      * Set extended bounds for the histogram. In case the lower value in the
132      * histogram would be greater than <code>min</code> or the upper value would
133      * be less than <code>max</code>, empty buckets will be generated.
134      */
135     public DateHistogramBuilder extendedBounds(String min, String max) {
136         extendedBoundsMin = min;
137         extendedBoundsMax = max;
138         return this;
139     }
140 
141     /**
142      * Set extended bounds for the histogram. In case the lower value in the
143      * histogram would be greater than <code>min</code> or the upper value would
144      * be less than <code>max</code>, empty buckets will be generated.
145      */
146     public DateHistogramBuilder extendedBounds(DateTime min, DateTime max) {
147         extendedBoundsMin = min;
148         extendedBoundsMax = max;
149         return this;
150     }
151 
152     @Override
153     protected XContentBuilder doInternalXContent(XContentBuilder builder, Params params) throws IOException {
154         if (interval == null) {
155             throw new SearchSourceBuilderException("[interval] must be defined for histogram aggregation [" + getName() + "]");
156         }
157         if (interval instanceof Number) {
158             interval = TimeValue.timeValueMillis(((Number) interval).longValue()).toString();
159         }
160         builder.field("interval", interval);
161 
162         if (minDocCount != null) {
163             builder.field("min_doc_count", minDocCount);
164         }
165 
166         if (order != null) {
167             builder.field("order");
168             order.toXContent(builder, params);
169         }
170 
171         if (timeZone != null) {
172             builder.field("time_zone", timeZone);
173         }
174 
175         if (offset != null) {
176             builder.field("offset", offset);
177         }
178 
179         if (factor != 1.0f) {
180             builder.field("factor", factor);
181         }
182 
183         if (format != null) {
184             builder.field("format", format);
185         }
186 
187         if (extendedBoundsMin != null || extendedBoundsMax != null) {
188             builder.startObject(DateHistogramParser.EXTENDED_BOUNDS.getPreferredName());
189             if (extendedBoundsMin != null) {
190                 builder.field("min", extendedBoundsMin);
191             }
192             if (extendedBoundsMax != null) {
193                 builder.field("max", extendedBoundsMax);
194             }
195             builder.endObject();
196         }
197 
198         return builder;
199     }
200 }
