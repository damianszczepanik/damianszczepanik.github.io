44  public class ScriptedMetricAggregator extends MetricsAggregator {
45  
46      private final String scriptLang;
47      private final SearchScript mapScript;
48      private final ExecutableScript combineScript;
49      private final String reduceScript;
50      // initial parameters for same shard scripts {init, map, combine}
51      // state can be passed in params between them too
52      private final Map<String, Object> params;
53      // initial parameters for {reduce}
54      private final Map<String, Object> reduceParams;
55      private final ScriptType reduceScriptType;
56  
57      protected ScriptedMetricAggregator(String name, String scriptLang, ScriptType initScriptType, String initScript,
58              ScriptType mapScriptType, String mapScript, ScriptType combineScriptType, String combineScript, ScriptType reduceScriptType,
59              String reduceScript, Map<String, Object> params, Map<String, Object> reduceParams, AggregationContext context, Aggregator parent, Map<String, Object> metaData) throws IOException {
60          super(name, context, parent, metaData);
61          this.scriptLang = scriptLang;
62          this.reduceScriptType = reduceScriptType;
63          if (params == null) {
64              this.params = new HashMap<>();
65              this.params.put("_agg", new HashMap<>());
66          } else {
67              this.params = new HashMap<>(params);
68          }
69          if (reduceParams == null) {
70              this.reduceParams = new HashMap<>();
71          } else {
72              this.reduceParams = reduceParams;
73          }
74          ScriptService scriptService = context.searchContext().scriptService();
75          if (initScript != null) {
76              scriptService.executable(scriptLang, initScript, initScriptType, this.params).run();
77          }
78          this.mapScript = scriptService.search(context.searchContext().lookup(), scriptLang, mapScript, mapScriptType, this.params);
79          if (combineScript != null) {
80              this.combineScript = scriptService.executable(scriptLang, combineScript, combineScriptType, this.params);
81          } else {
82              this.combineScript = null;
83          }
84          this.reduceScript = reduceScript;
85      }
86  
87      @Override
88      public boolean needsScores() {
89          return true; // TODO: how can we know if the script relies on scores?
90      }
91  
92      @Override
93      public LeafBucketCollector getLeafCollector(LeafReaderContext ctx,
94              final LeafBucketCollector sub) throws IOException {
95          mapScript.setNextReader(ctx);
96          return new LeafBucketCollectorBase(sub, mapScript) {
97              @Override
98              public void collect(int doc, long bucket) throws IOException {
99                  assert bucket == 0 : bucket;
100                 mapScript.setNextDocId(doc);
101                 mapScript.run();
102             }
103         };
104     }
105 
106     @Override
107     public InternalAggregation buildAggregation(long owningBucketOrdinal) {
108         Object aggregation;
109         if (combineScript != null) {
110             aggregation = combineScript.run();
111         } else {
112             aggregation = params.get("_agg");
113         }
114         return new InternalScriptedMetric(name, aggregation, scriptLang, reduceScriptType, reduceScript, reduceParams, metaData());
115     }
116 
117     @Override
118     public InternalAggregation buildEmptyAggregation() {
119         return new InternalScriptedMetric(name, null, scriptLang, reduceScriptType, reduceScript, reduceParams, metaData());
120     }

122     public static class Factory extends AggregatorFactory {
123 
124         private String scriptLang;
125         private ScriptType initScriptType;
126         private ScriptType mapScriptType;
127         private ScriptType combineScriptType;
128         private ScriptType reduceScriptType;
129         private String initScript;
130         private String mapScript;
131         private String combineScript;
132         private String reduceScript;
133         private Map<String, Object> params;
134         private Map<String, Object> reduceParams;
135 
136         public Factory(String name, String scriptLang, ScriptType initScriptType, String initScript, ScriptType mapScriptType, String mapScript, ScriptType combineScriptType, String combineScript, ScriptType reduceScriptType, String reduceScript,
137                 Map<String, Object> params, Map<String, Object> reduceParams) {
138             super(name, InternalScriptedMetric.TYPE.name());
139             this.scriptLang = scriptLang;
140             this.initScriptType = initScriptType;
141             this.mapScriptType = mapScriptType;
142             this.combineScriptType = combineScriptType;
143             this.reduceScriptType = reduceScriptType;
144             this.initScript = initScript;
145             this.mapScript = mapScript;
146             this.combineScript = combineScript;
147             this.reduceScript = reduceScript;
148             this.params = params;
149             this.reduceParams = reduceParams;
150         }
151 
152         @Override
153         public Aggregator createInternal(AggregationContext context, Aggregator parent, boolean collectsFromSingleBucket, Map<String, Object> metaData) throws IOException {
154             if (collectsFromSingleBucket == false) {
155                 return asMultiBucketAggregator(this, context, parent);
156             }
157             Map<String, Object> params = null;
158             if (this.params != null) {
159                 params = deepCopyParams(this.params, context.searchContext());
160             }
161             Map<String, Object> reduceParams = null;
162             if (this.reduceParams != null) {
163                 reduceParams = deepCopyParams(this.reduceParams, context.searchContext());
164             }
165             return new ScriptedMetricAggregator(name, scriptLang, initScriptType, initScript, mapScriptType, mapScript, combineScriptType,
166                     combineScript, reduceScriptType, reduceScript, params, reduceParams, context, parent, metaData);
167         }
168         
169         @SuppressWarnings({ "unchecked" })
170         private static <T> T deepCopyParams(T original, SearchContext context) {
171             T clone;
172             if (original instanceof Map) {
173                 Map<?, ?> originalMap = (Map<?, ?>) original;
174                 Map<Object, Object> clonedMap = new HashMap<>();
175                 for (Entry<?, ?> e : originalMap.entrySet()) {
176                     clonedMap.put(deepCopyParams(e.getKey(), context), deepCopyParams(e.getValue(), context));
177                 }
178                 clone = (T) clonedMap;
179             } else if (original instanceof List) {
180                 List<?> originalList = (List<?>) original;
181                 List<Object> clonedList = new ArrayList<Object>();
182                 for (Object o : originalList) {
183                     clonedList.add(deepCopyParams(o, context));
184                 }
185                 clone = (T) clonedList;
186             } else if (original instanceof String || original instanceof Integer || original instanceof Long || original instanceof Short
187                     || original instanceof Byte || original instanceof Float || original instanceof Double || original instanceof Character
188                     || original instanceof Boolean) {
189                 clone = original;
190             } else {
191                 throw new SearchParseException(context, "Can only clone primitives, String, ArrayList, and HashMap. Found: "
192                         + original.getClass().getCanonicalName());
193             }
194             return clone;
195         }
196 
197     }
