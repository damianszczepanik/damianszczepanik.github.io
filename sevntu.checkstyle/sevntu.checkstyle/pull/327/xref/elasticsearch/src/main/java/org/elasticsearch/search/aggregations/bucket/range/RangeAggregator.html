55      public static class Range {
56  
57          public String key;
58          public double from = Double.NEGATIVE_INFINITY;
59          String fromAsStr;
60          public double to = Double.POSITIVE_INFINITY;
61          String toAsStr;
62  
63          public Range(String key, double from, String fromAsStr, double to, String toAsStr) {
64              this.key = key;
65              this.from = from;
66              this.fromAsStr = fromAsStr;
67              this.to = to;
68              this.toAsStr = toAsStr;
69          }
70  
71          boolean matches(double value) {
72              return value >= from && value < to;
73          }
74  
75          @Override
76          public String toString() {
77              return "[" + from + " to " + to + ")";
78          }
79  
80          public void process(ValueParser parser, SearchContext context) {
81              assert parser != null;
82              if (fromAsStr != null) {
83                  from = parser.parseDouble(fromAsStr, context);
84              }
85              if (toAsStr != null) {
86                  to = parser.parseDouble(toAsStr, context);
87              }
88          }
89      }
90  
91      final ValuesSource.Numeric valuesSource;
92      final @Nullable ValueFormatter formatter;
93      final Range[] ranges;
94      final boolean keyed;
95      final InternalRange.Factory rangeFactory;
96  
97      final double[] maxTo;
98  
99      public RangeAggregator(String name,
100                            AggregatorFactories factories,
101                            ValuesSource.Numeric valuesSource,
102                            @Nullable ValueFormat format,
103                            InternalRange.Factory rangeFactory,
104                            List<Range> ranges,
105                            boolean keyed,
106                            AggregationContext aggregationContext,
107                            Aggregator parent,
108                            Map<String, Object> metaData) throws IOException {
109 
110         super(name, factories, aggregationContext, parent, metaData);
111         assert valuesSource != null;
112         this.valuesSource = valuesSource;
113         this.formatter = format != null ? format.formatter() : null;
114         this.keyed = keyed;
115         this.rangeFactory = rangeFactory;
116         this.ranges = ranges.toArray(new Range[ranges.size()]);
117 
118         ValueParser parser = format != null ? format.parser() : ValueParser.RAW;
119         for (int i = 0; i < this.ranges.length; i++) {
120             this.ranges[i].process(parser, context.searchContext());
121         }
122         sortRanges(this.ranges);
123 
124         maxTo = new double[this.ranges.length];
125         maxTo[0] = this.ranges[0].to;
126         for (int i = 1; i < this.ranges.length; ++i) {
127             maxTo[i] = Math.max(this.ranges[i].to,maxTo[i-1]);
128         }
129 
130     }

137     @Override
138     public LeafBucketCollector getLeafCollector(LeafReaderContext ctx,
139             final LeafBucketCollector sub) throws IOException {
140         final SortedNumericDoubleValues values = valuesSource.doubleValues(ctx);
141         return new LeafBucketCollectorBase(sub, values) {
142             @Override
143             public void collect(int doc, long bucket) throws IOException {
144                 values.setDocument(doc);
145                 final int valuesCount = values.count();
146                 for (int i = 0, lo = 0; i < valuesCount; ++i) {
147                     final double value = values.valueAt(i);
148                     lo = collect(doc, value, bucket, lo);
149                 }
150             }
151 
152             private int collect(int doc, double value, long owningBucketOrdinal, int lowBound) throws IOException {
153                 int lo = lowBound, hi = ranges.length - 1; // all candidates are between these indexes
154                 int mid = (lo + hi) >>> 1;
155                 while (lo <= hi) {
156                     if (value < ranges[mid].from) {
157                         hi = mid - 1;
158                     } else if (value >= maxTo[mid]) {
159                         lo = mid + 1;
160                     } else {
161                         break;
162                     }
163                     mid = (lo + hi) >>> 1;
164                 }
165                 if (lo > hi) return lo; // no potential candidate
166 
167                 // binary search the lower bound
168                 int startLo = lo, startHi = mid;
169                 while (startLo <= startHi) {
170                     final int startMid = (startLo + startHi) >>> 1;
171                     if (value >= maxTo[startMid]) {
172                         startLo = startMid + 1;
173                     } else {
174                         startHi = startMid - 1;
175                     }
176                 }
177 
178                 // binary search the upper bound
179                 int endLo = mid, endHi = hi;
180                 while (endLo <= endHi) {
181                     final int endMid = (endLo + endHi) >>> 1;
182                     if (value < ranges[endMid].from) {
183                         endHi = endMid - 1;
184                     } else {
185                         endLo = endMid + 1;
186                     }
187                 }
188 
189                 assert startLo == lowBound || value >= maxTo[startLo - 1];
190                 assert endHi == ranges.length - 1 || value < ranges[endHi + 1].from;
191 
192                 for (int i = startLo; i <= endHi; ++i) {
193                     if (ranges[i].matches(value)) {
194                         collectBucket(sub, doc, subBucketOrdinal(owningBucketOrdinal, i));
195                     }
196                 }
197 
198                 return endHi + 1;
199             }
200         };
201     }
202 
203     private final long subBucketOrdinal(long owningBucketOrdinal, int rangeOrd) {
204         return owningBucketOrdinal * ranges.length + rangeOrd;
205     }
206 
207     @Override
208     public InternalAggregation buildAggregation(long owningBucketOrdinal) throws IOException {
209         List<org.elasticsearch.search.aggregations.bucket.range.Range.Bucket> buckets = Lists.newArrayListWithCapacity(ranges.length);
210         for (int i = 0; i < ranges.length; i++) {
211             Range range = ranges[i];
212             final long bucketOrd = subBucketOrdinal(owningBucketOrdinal, i);
213             org.elasticsearch.search.aggregations.bucket.range.Range.Bucket bucket =
214                     rangeFactory.createBucket(range.key, range.from, range.to, bucketDocCount(bucketOrd), bucketAggregations(bucketOrd), keyed, formatter);
215             buckets.add(bucket);
216         }
217         // value source can be null in the case of unmapped fields
218         return rangeFactory.create(name, buckets, formatter, keyed, metaData());
219     }
220 
221     @Override
222     public InternalAggregation buildEmptyAggregation() {
223         InternalAggregations subAggs = buildEmptySubAggregations();
224         List<org.elasticsearch.search.aggregations.bucket.range.Range.Bucket> buckets = Lists.newArrayListWithCapacity(ranges.length);
225         for (int i = 0; i < ranges.length; i++) {
226             Range range = ranges[i];
227             org.elasticsearch.search.aggregations.bucket.range.Range.Bucket bucket =
228                     rangeFactory.createBucket(range.key, range.from, range.to, 0, subAggs, keyed, formatter);
229             buckets.add(bucket);
230         }
231         // value source can be null in the case of unmapped fields
232         return rangeFactory.create(name, buckets, formatter, keyed, metaData());
233     }

235     private static final void sortRanges(final Range[] ranges) {
236         new InPlaceMergeSorter() {
237 
238             @Override
239             protected void swap(int i, int j) {
240                 final Range tmp = ranges[i];
241                 ranges[i] = ranges[j];
242                 ranges[j] = tmp;
243             }
244 
245             @Override
246             protected int compare(int i, int j) {
247                 int cmp = Double.compare(ranges[i].from, ranges[j].from);
248                 if (cmp == 0) {
249                     cmp = Double.compare(ranges[i].to, ranges[j].to);
250                 }
251                 return cmp;
252             }
253         }.sort(0, ranges.length);
254     }

256     public static class Unmapped extends NonCollectingAggregator {
257 
258         private final List<RangeAggregator.Range> ranges;
259         private final boolean keyed;
260         private final InternalRange.Factory factory;
261         private final ValueFormatter formatter;
262 
263         public Unmapped(String name,
264                         List<RangeAggregator.Range> ranges,
265                         boolean keyed,
266                         ValueFormat format,
267                         AggregationContext context,
268                         Aggregator parent,
269                         InternalRange.Factory factory,
270                         Map<String, Object> metaData) throws IOException {
271 
272             super(name, context, parent, metaData);
273             this.ranges = ranges;
274             ValueParser parser = format != null ? format.parser() : ValueParser.RAW;
275             for (Range range : this.ranges) {
276                 range.process(parser, context.searchContext());
277             }
278             this.keyed = keyed;
279             this.formatter = format != null ? format.formatter() : null;
280             this.factory = factory;
281         }
282 
283         @Override
284         public InternalAggregation buildEmptyAggregation() {
285             InternalAggregations subAggs = buildEmptySubAggregations();
286             List<org.elasticsearch.search.aggregations.bucket.range.Range.Bucket> buckets = new ArrayList<>(ranges.size());
287             for (RangeAggregator.Range range : ranges) {
288                 buckets.add(factory.createBucket(range.key, range.from, range.to, 0, subAggs, keyed, formatter));
289             }
290             return factory.create(name, buckets, formatter, keyed, metaData());
291         }
292     }

294     public static class Factory extends ValuesSourceAggregatorFactory<ValuesSource.Numeric> {
295 
296         private final InternalRange.Factory rangeFactory;
297         private final List<Range> ranges;
298         private final boolean keyed;
299 
300         public Factory(String name, ValuesSourceConfig<ValuesSource.Numeric> valueSourceConfig, InternalRange.Factory rangeFactory, List<Range> ranges, boolean keyed) {
301             super(name, rangeFactory.type(), valueSourceConfig);
302             this.rangeFactory = rangeFactory;
303             this.ranges = ranges;
304             this.keyed = keyed;
305         }
306 
307         @Override
308         protected Aggregator createUnmapped(AggregationContext aggregationContext, Aggregator parent, Map<String, Object> metaData) throws IOException {
309             return new Unmapped(name, ranges, keyed, config.format(), aggregationContext, parent, rangeFactory, metaData);
310         }
311 
312         @Override
313         protected Aggregator doCreateInternal(ValuesSource.Numeric valuesSource, AggregationContext aggregationContext, Aggregator parent, boolean collectsFromSingleBucket, Map<String, Object> metaData) throws IOException {
314             return new RangeAggregator(name, factories, valuesSource, config.format(), rangeFactory, ranges, keyed, aggregationContext, parent, metaData);
315         }
