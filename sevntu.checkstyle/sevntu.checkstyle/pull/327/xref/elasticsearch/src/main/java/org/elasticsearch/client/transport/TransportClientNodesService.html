97      @Inject
98      public TransportClientNodesService(Settings settings, ClusterName clusterName, TransportService transportService,
99                                         ThreadPool threadPool, Headers headers, Version version) {
100         super(settings);
101         this.clusterName = clusterName;
102         this.transportService = transportService;
103         this.threadPool = threadPool;
104         this.minCompatibilityVersion = version.minimumCompatibilityVersion();
105         this.headers = headers;
106 
107         this.nodesSamplerInterval = this.settings.getAsTime("client.transport.nodes_sampler_interval", timeValueSeconds(5));
108         this.pingTimeout = this.settings.getAsTime("client.transport.ping_timeout", timeValueSeconds(5)).millis();
109         this.ignoreClusterName = this.settings.getAsBoolean("client.transport.ignore_cluster_name", false);
110 
111         if (logger.isDebugEnabled()) {
112             logger.debug("node_sampler_interval[" + nodesSamplerInterval + "]");
113         }
114 
115         if (this.settings.getAsBoolean("client.transport.sniff", false)) {
116             this.nodesSampler = new SniffNodesSampler();
117         } else {
118             this.nodesSampler = new SimpleNodeSampler();
119         }
120         this.nodesSamplerFuture = threadPool.schedule(nodesSamplerInterval, ThreadPool.Names.GENERIC, new ScheduledNodeSampler());
121     }

143     public TransportClientNodesService addTransportAddresses(TransportAddress... transportAddresses) {
144         synchronized (mutex) {
145             if (closed) {
146                 throw new ElasticsearchIllegalStateException("transport client is closed, can't add an address");
147             }
148             List<TransportAddress> filtered = Lists.newArrayListWithExpectedSize(transportAddresses.length);
149             for (TransportAddress transportAddress : transportAddresses) {
150                 boolean found = false;
151                 for (DiscoveryNode otherNode : listedNodes) {
152                     if (otherNode.address().equals(transportAddress)) {
153                         found = true;
154                         logger.debug("address [{}] already exists with [{}], ignoring...", transportAddress, otherNode);
155                         break;
156                     }
157                 }
158                 if (!found) {
159                     filtered.add(transportAddress);
160                 }
161             }
162             if (filtered.isEmpty()) {
163                 return this;
164             }
165             ImmutableList.Builder<DiscoveryNode> builder = ImmutableList.builder();
166             builder.addAll(listedNodes());
167             for (TransportAddress transportAddress : filtered) {
168                 DiscoveryNode node = new DiscoveryNode("#transport#-" + tempNodeIdGenerator.incrementAndGet(), transportAddress, minCompatibilityVersion);
169                 logger.debug("adding address [{}]", node);
170                 builder.add(node);
171             }
172             listedNodes = builder.build();
173             nodesSampler.sample();
174         }
175         return this;
176     }
177 
178     public TransportClientNodesService removeTransportAddress(TransportAddress transportAddress) {
179         synchronized (mutex) {
180             if (closed) {
181                 throw new ElasticsearchIllegalStateException("transport client is closed, can't remove an address");
182             }
183             ImmutableList.Builder<DiscoveryNode> builder = ImmutableList.builder();
184             for (DiscoveryNode otherNode : listedNodes) {
185                 if (!otherNode.address().equals(transportAddress)) {
186                     builder.add(otherNode);
187                 } else {
188                     logger.debug("removing address [{}]", otherNode);
189                 }
190             }
191             listedNodes = builder.build();
192             nodesSampler.sample();
193         }
194         return this;
195     }

197     public <Response> void execute(NodeListenerCallback<Response> callback, ActionListener<Response> listener) throws ElasticsearchException {
198         ImmutableList<DiscoveryNode> nodes = this.nodes;
199         ensureNodesAreAvailable(nodes);
200         int index = getNodeNumber();
201         RetryListener<Response> retryListener = new RetryListener<>(callback, listener, nodes, index);
202         DiscoveryNode node = nodes.get((index) % nodes.size());
203         try {
204             callback.doWithNode(node, retryListener);
205         } catch (Throwable t) {
206             //this exception can't come from the TransportService as it doesn't throw exception at all
207             listener.onFailure(t);
208         }
209     }

211     public static class RetryListener<Response> implements ActionListener<Response> {
212         private final NodeListenerCallback<Response> callback;
213         private final ActionListener<Response> listener;
214         private final ImmutableList<DiscoveryNode> nodes;
215         private final int index;
216 
217         private volatile int i;
218 
219         public RetryListener(NodeListenerCallback<Response> callback, ActionListener<Response> listener, ImmutableList<DiscoveryNode> nodes, int index) {
220             this.callback = callback;
221             this.listener = listener;
222             this.nodes = nodes;
223             this.index = index;
224         }
225 
226         @Override
227         public void onResponse(Response response) {
228             listener.onResponse(response);
229         }
230 
231         @Override
232         public void onFailure(Throwable e) {
233             if (ExceptionsHelper.unwrapCause(e) instanceof ConnectTransportException) {
234                 int i = ++this.i;
235                 if (i >= nodes.size()) {
236                     listener.onFailure(new NoNodeAvailableException("None of the configured nodes were available: " + nodes, e));
237                 } else {
238                     try {
239                         callback.doWithNode(nodes.get((index + i) % nodes.size()), this);
240                     } catch(Throwable t) {
241                         //this exception can't come from the TransportService as it doesn't throw exceptions at all
242                         listener.onFailure(t);
243                     }
244                 }
245             } else {
246                 listener.onFailure(e);
247             }
248         }
249     }
250 
251     public void close() {
252         synchronized (mutex) {
253             if (closed) {
254                 return;
255             }
256             closed = true;
257             FutureUtils.cancel(nodesSamplerFuture);
258             for (DiscoveryNode node : nodes) {
259                 transportService.disconnectFromNode(node);
260             }
261             for (DiscoveryNode listedNode : listedNodes) {
262                 transportService.disconnectFromNode(listedNode);
263             }
264             nodes = ImmutableList.of();
265         }
266     }
267 
268     private int getNodeNumber() {
269         int index = randomNodeGenerator.incrementAndGet();
270         if (index < 0) {
271             index = 0;
272             randomNodeGenerator.set(0);
273         }
274         return index;
275     }
276 
277     private void ensureNodesAreAvailable(ImmutableList<DiscoveryNode> nodes) {
278         if (nodes.isEmpty()) {
279             String message = String.format(Locale.ROOT, "None of the configured nodes are available: %s", nodes);
280             throw new NoNodeAvailableException(message);
281         }
282     }

284     abstract class NodeSampler {
285         public void sample() {
286             synchronized (mutex) {
287                 if (closed) {
288                     return;
289                 }
290                 doSample();
291             }
292         }
293 
294         protected abstract void doSample();
295 
296         /**
297          * validates a set of potentially newly discovered nodes and returns an immutable
298          * list of the nodes that has passed.
299          */
300         protected ImmutableList<DiscoveryNode> validateNewNodes(Set<DiscoveryNode> nodes) {
301             for (Iterator<DiscoveryNode> it = nodes.iterator(); it.hasNext(); ) {
302                 DiscoveryNode node = it.next();
303                 if (!transportService.nodeConnected(node)) {
304                     try {
305                         logger.trace("connecting to node [{}]", node);
306                         transportService.connectToNode(node);
307                     } catch (Throwable e) {
308                         it.remove();
309                         logger.debug("failed to connect to discovered node [" + node + "]", e);
310                     }
311                 }
312             }
313 
314             return new ImmutableList.Builder<DiscoveryNode>().addAll(nodes).build();
315         }
316 
317     }

319     class ScheduledNodeSampler implements Runnable {
320         @Override
321         public void run() {
322             try {
323                 nodesSampler.sample();
324                 if (!closed) {
325                     nodesSamplerFuture = threadPool.schedule(nodesSamplerInterval, ThreadPool.Names.GENERIC, this);
326                 }
327             } catch (Exception e) {
328                 logger.warn("failed to sample", e);
329             }
330         }
331     }

333     class SimpleNodeSampler extends NodeSampler {
334 
335         @Override
336         protected void doSample() {
337             HashSet<DiscoveryNode> newNodes = new HashSet<>();
338             HashSet<DiscoveryNode> newFilteredNodes = new HashSet<>();
339             for (DiscoveryNode listedNode : listedNodes) {
340                 if (!transportService.nodeConnected(listedNode)) {
341                     try {
342                         // its a listed node, light connect to it...
343                         logger.trace("connecting to listed node (light) [{}]", listedNode);
344                         transportService.connectToNodeLight(listedNode);
345                     } catch (Throwable e) {
346                         logger.debug("failed to connect to node [{}], removed from nodes list", e, listedNode);
347                         continue;
348                     }
349                 }
350                 try {
351                     LivenessResponse livenessResponse = transportService.submitRequest(listedNode, TransportLivenessAction.NAME,
352                             headers.applyTo(new LivenessRequest()),
353                             TransportRequestOptions.options().withType(TransportRequestOptions.Type.STATE).withTimeout(pingTimeout),
354                             new FutureTransportResponseHandler<LivenessResponse>() {
355                                 @Override
356                                 public LivenessResponse newInstance() {
357                                     return new LivenessResponse();
358                                 }
359                             }).txGet();
360                     if (!ignoreClusterName && !clusterName.equals(livenessResponse.getClusterName())) {
361                         logger.warn("node {} not part of the cluster {}, ignoring...", listedNode, clusterName);
362                         newFilteredNodes.add(listedNode);
363                     } else if (livenessResponse.getDiscoveryNode() != null) {
364                         // use discovered information but do keep the original transport address, so people can control which address is exactly used.
365                         DiscoveryNode nodeWithInfo = livenessResponse.getDiscoveryNode();
366                         newNodes.add(new DiscoveryNode(nodeWithInfo.name(), nodeWithInfo.id(), nodeWithInfo.getHostName(), nodeWithInfo.getHostAddress(), listedNode.address(), nodeWithInfo.attributes(), nodeWithInfo.version()));
367                     } else {
368                         // although we asked for one node, our target may not have completed initialization yet and doesn't have cluster nodes
369                         logger.debug("node {} didn't return any discovery info, temporarily using transport discovery node", listedNode);
370                         newNodes.add(listedNode);
371                     }
372                 } catch (Throwable e) {
373                     logger.info("failed to get node info for {}, disconnecting...", e, listedNode);
374                     transportService.disconnectFromNode(listedNode);
375                 }
376             }
377 
378             nodes = validateNewNodes(newNodes);
379             filteredNodes = ImmutableList.copyOf(newFilteredNodes);
380         }
381     }

383     class SniffNodesSampler extends NodeSampler {
384 
385         @Override
386         protected void doSample() {
387             // the nodes we are going to ping include the core listed nodes that were added
388             // and the last round of discovered nodes
389             Set<DiscoveryNode> nodesToPing = Sets.newHashSet();
390             for (DiscoveryNode node : listedNodes) {
391                 nodesToPing.add(node);
392             }
393             for (DiscoveryNode node : nodes) {
394                 nodesToPing.add(node);
395             }
396 
397             final CountDownLatch latch = new CountDownLatch(nodesToPing.size());
398             final ConcurrentMap<DiscoveryNode, ClusterStateResponse> clusterStateResponses = ConcurrentCollections.newConcurrentMap();
399             for (final DiscoveryNode listedNode : nodesToPing) {
400                 threadPool.executor(ThreadPool.Names.MANAGEMENT).execute(new Runnable() {
401                     @Override
402                     public void run() {
403                         try {
404                             if (!transportService.nodeConnected(listedNode)) {
405                                 try {
406 
407                                     // if its one of the actual nodes we will talk to, not to listed nodes, fully connect
408                                     if (nodes.contains(listedNode)) {
409                                         logger.trace("connecting to cluster node [{}]", listedNode);
410                                         transportService.connectToNode(listedNode);
411                                     } else {
412                                         // its a listed node, light connect to it...
413                                         logger.trace("connecting to listed node (light) [{}]", listedNode);
414                                         transportService.connectToNodeLight(listedNode);
415                                     }
416                                 } catch (Exception e) {
417                                     logger.debug("failed to connect to node [{}], ignoring...", e, listedNode);
418                                     latch.countDown();
419                                     return;
420                                 }
421                             }
422                             transportService.sendRequest(listedNode, ClusterStateAction.NAME,
423                                     headers.applyTo(Requests.clusterStateRequest().clear().nodes(true).local(true)),
424                                     TransportRequestOptions.options().withType(TransportRequestOptions.Type.STATE).withTimeout(pingTimeout),
425                                     new BaseTransportResponseHandler<ClusterStateResponse>() {
426 
427                                         @Override
428                                         public ClusterStateResponse newInstance() {
429                                             return new ClusterStateResponse();
430                                         }
431 
432                                         @Override
433                                         public String executor() {
434                                             return ThreadPool.Names.SAME;
435                                         }
436 
437                                         @Override
438                                         public void handleResponse(ClusterStateResponse response) {
439                                             clusterStateResponses.put(listedNode, response);
440                                             latch.countDown();
441                                         }
442 
443                                         @Override
444                                         public void handleException(TransportException e) {
445                                             logger.info("failed to get local cluster state for {}, disconnecting...", e, listedNode);
446                                             transportService.disconnectFromNode(listedNode);
447                                             latch.countDown();
448                                         }
449                                     });
450                         } catch (Throwable e) {
451                             logger.info("failed to get local cluster state info for {}, disconnecting...", e, listedNode);
452                             transportService.disconnectFromNode(listedNode);
453                             latch.countDown();
454                         }
455                     }
456                 });
457             }
458 
459             try {
460                 latch.await();
461             } catch (InterruptedException e) {
462                 return;
463             }
464 
465             HashSet<DiscoveryNode> newNodes = new HashSet<>();
466             HashSet<DiscoveryNode> newFilteredNodes = new HashSet<>();
467             for (Map.Entry<DiscoveryNode, ClusterStateResponse> entry : clusterStateResponses.entrySet()) {
468                 if (!ignoreClusterName && !clusterName.equals(entry.getValue().getClusterName())) {
469                     logger.warn("node {} not part of the cluster {}, ignoring...", entry.getValue().getState().nodes().localNode(), clusterName);
470                     newFilteredNodes.add(entry.getKey());
471                     continue;
472                 }
473                 for (ObjectCursor<DiscoveryNode> cursor : entry.getValue().getState().nodes().dataNodes().values()) {
474                     newNodes.add(cursor.value);
475                 }
476             }
477 
478             nodes = validateNewNodes(newNodes);
479             filteredNodes = ImmutableList.copyOf(newFilteredNodes);
480         }
481     }
