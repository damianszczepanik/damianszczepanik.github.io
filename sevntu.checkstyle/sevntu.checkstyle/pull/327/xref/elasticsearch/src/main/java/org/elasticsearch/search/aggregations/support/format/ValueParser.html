19  package org.elasticsearch.search.aggregations.support.format;
20  
21  import org.elasticsearch.common.joda.DateMathParser;
22  import org.elasticsearch.common.joda.FormatDateTimeFormatter;
23  import org.elasticsearch.common.joda.Joda;
24  import org.elasticsearch.index.mapper.core.DateFieldMapper;
25  import org.elasticsearch.index.mapper.ip.IpFieldMapper;
26  import org.elasticsearch.search.aggregations.AggregationExecutionException;
27  import org.elasticsearch.search.internal.SearchContext;
28  
29  import java.text.DecimalFormat;
30  import java.text.DecimalFormatSymbols;
31  import java.text.NumberFormat;
32  import java.text.ParseException;
33  import java.util.Locale;
34  import java.util.concurrent.Callable;
35  import java.util.concurrent.TimeUnit;
36  
37  /**
38   *
39   */
40  public interface ValueParser {
41  
42      static final ValueParser IPv4 = new IPv4();
43      static final ValueParser RAW = new Raw();
44  
45      long parseLong(String value, SearchContext searchContext);
46  
47      double parseDouble(String value, SearchContext searchContext);
48  
49  
50      /**
51       * Knows how to parse datatime values based on date/time format
52       */
53      static class DateTime implements ValueParser {
54  
55          public static final DateTime DEFAULT = new DateTime(DateFieldMapper.Defaults.DATE_TIME_FORMATTER);
56  
57          private FormatDateTimeFormatter formatter;
58  
59          public DateTime(String format) {
60              this(Joda.forPattern(format));
61          }
62  
63          public DateTime(FormatDateTimeFormatter formatter) {
64              this.formatter = formatter;
65          }
66  
67          @Override
68          public long parseLong(String value, SearchContext searchContext) {
69              return formatter.parser().parseMillis(value);
70          }
71  
72          @Override
73          public double parseDouble(String value, SearchContext searchContext) {
74              return parseLong(value, searchContext);
75          }
76      }
77  
78      /**
79       * Knows how to parse datatime values based on elasticsearch's date math expression
80       */
81      static class DateMath implements ValueParser {
82  
83          public static final DateMath DEFAULT = new ValueParser.DateMath(new DateMathParser(DateFieldMapper.Defaults.DATE_TIME_FORMATTER, DateFieldMapper.Defaults.TIME_UNIT));
84  
85          private DateMathParser parser;
86  
87          public DateMath(String format, TimeUnit timeUnit) {
88              this(new DateMathParser(Joda.forPattern(format), timeUnit));
89          }
90  
91          public DateMath(DateMathParser parser) {
92              this.parser = parser;
93          }
94  
95          @Override
96          public long parseLong(String value, final SearchContext searchContext) {
97              final Callable<Long> now = new Callable<Long>() {
98                  @Override
99                  public Long call() throws Exception {
100                     return searchContext.nowInMillis();
101                 }
102             };
103             return parser.parse(value, now);
104         }
105 
106         @Override
107         public double parseDouble(String value, SearchContext searchContext) {
108             return parseLong(value, searchContext);
109         }
110 
111         public static DateMath mapper(DateFieldMapper mapper) {
112             return new DateMath(new DateMathParser(mapper.dateTimeFormatter(), DateFieldMapper.Defaults.TIME_UNIT));
113         }
114     }
115 
116     /**
117      * Knows how to parse IPv4 formats
118      */
119     static class IPv4 implements ValueParser {
120 
121         private IPv4() {
122         }
123 
124         @Override
125         public long parseLong(String value, SearchContext searchContext) {
126             return IpFieldMapper.ipToLong(value);
127         }
128 
129         @Override
130         public double parseDouble(String value, SearchContext searchContext) {
131             return parseLong(value, searchContext);
132         }
133     }
134 
135     static class Raw implements ValueParser {
136 
137         private Raw() {
138         }
139 
140         @Override
141         public long parseLong(String value, SearchContext searchContext) {
142             return Long.parseLong(value);
143         }
144 
145         @Override
146         public double parseDouble(String value, SearchContext searchContext) {
147             return Double.parseDouble(value);
148         }
149     }
150 
151     public static abstract class Number implements ValueParser {
152 
153         NumberFormat format;
154 
155         Number(NumberFormat format) {
156            this.format = format;
157         }
158 
159         public static class Pattern extends Number {
160 
161             private static final DecimalFormatSymbols SYMBOLS = new DecimalFormatSymbols(Locale.ROOT);
162 
163             public Pattern(String pattern) {
164                 super(new DecimalFormat(pattern, SYMBOLS));
165             }
166 
167             @Override
168             public long parseLong(String value, SearchContext searchContext) {
169                 try {
170                     return format.parse(value).longValue();
171                 } catch (ParseException nfe) {
172                     throw new AggregationExecutionException("Invalid number format [" + ((DecimalFormat) format).toPattern() + "]");
173                 }
174             }
175 
176             @Override
177             public double parseDouble(String value, SearchContext searchContext) {
178                 try {
179                     return format.parse(value).doubleValue();
180                 } catch (ParseException nfe) {
181                     throw new AggregationExecutionException("Invalid number format [" + ((DecimalFormat) format).toPattern() + "]");
182                 }
183             }
184         }
185     }
186 
187 }
