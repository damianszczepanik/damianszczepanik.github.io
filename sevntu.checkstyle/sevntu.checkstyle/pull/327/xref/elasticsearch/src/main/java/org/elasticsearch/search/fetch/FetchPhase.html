81      private final FetchSubPhase[] fetchSubPhases;

83      @Inject
84      public FetchPhase(HighlightPhase highlightPhase, ScriptFieldsFetchSubPhase scriptFieldsPhase,
85                        MatchedQueriesFetchSubPhase matchedQueriesPhase, ExplainFetchSubPhase explainPhase, VersionFetchSubPhase versionPhase,
86                        FetchSourceSubPhase fetchSourceSubPhase, FieldDataFieldsFetchSubPhase fieldDataFieldsFetchSubPhase, 
87                        InnerHitsFetchSubPhase innerHitsFetchSubPhase) {
88          innerHitsFetchSubPhase.setFetchPhase(this);
89          this.fetchSubPhases = new FetchSubPhase[]{scriptFieldsPhase, matchedQueriesPhase, explainPhase, highlightPhase,
90                  fetchSourceSubPhase, versionPhase, fieldDataFieldsFetchSubPhase, innerHitsFetchSubPhase};
91      }

93      @Override
94      public Map<String, ? extends SearchParseElement> parseElements() {
95          ImmutableMap.Builder<String, SearchParseElement> parseElements = ImmutableMap.builder();
96          parseElements.put("fields", new FieldsParseElement());
97          for (FetchSubPhase fetchSubPhase : fetchSubPhases) {
98              parseElements.putAll(fetchSubPhase.parseElements());
99          }
100         return parseElements.build();
101     }

107     @Override
108     public void execute(SearchContext context) {
109         FieldsVisitor fieldsVisitor;
110         Set<String> fieldNames = null;
111         List<String> extractFieldNames = null;
112 
113         boolean loadAllStored = false;
114         if (!context.hasFieldNames()) {
115             // no fields specified, default to return source if no explicit indication
116             if (!context.hasScriptFields() && !context.hasFetchSourceContext()) {
117                 context.fetchSourceContext(new FetchSourceContext(true));
118             }
119             fieldsVisitor = context.sourceRequested() ? new UidAndSourceFieldsVisitor() : new JustUidFieldsVisitor();
120         } else if (context.fieldNames().isEmpty()) {
121             if (context.sourceRequested()) {
122                 fieldsVisitor = new UidAndSourceFieldsVisitor();
123             } else {
124                 fieldsVisitor = new JustUidFieldsVisitor();
125             }
126         } else {
127             for (String fieldName : context.fieldNames()) {
128                 if (fieldName.equals("*")) {
129                     loadAllStored = true;
130                     continue;
131                 }
132                 if (fieldName.equals(SourceFieldMapper.NAME)) {
133                     if (context.hasFetchSourceContext()) {
134                         context.fetchSourceContext().fetchSource(true);
135                     } else {
136                         context.fetchSourceContext(new FetchSourceContext(true));
137                     }
138                     continue;
139                 }
140                 FieldMappers x = context.smartNameFieldMappers(fieldName);
141                 if (x == null) {
142                     // Only fail if we know it is a object field, missing paths / fields shouldn't fail.
143                     if (context.smartNameObjectMapper(fieldName) != null) {
144                         throw new ElasticsearchIllegalArgumentException("field [" + fieldName + "] isn't a leaf field");
145                     }
146                 } else if (x.mapper().fieldType().stored()) {
147                     if (fieldNames == null) {
148                         fieldNames = new HashSet<>();
149                     }
150                     fieldNames.add(x.mapper().names().indexName());
151                 } else {
152                     if (extractFieldNames == null) {
153                         extractFieldNames = newArrayList();
154                     }
155                     extractFieldNames.add(fieldName);
156                 }
157             }
158             if (loadAllStored) {
159                 fieldsVisitor = new AllFieldsVisitor(); // load everything, including _source
160             } else if (fieldNames != null) {
161                 boolean loadSource = extractFieldNames != null || context.sourceRequested();
162                 fieldsVisitor = new CustomFieldsVisitor(fieldNames, loadSource);
163             } else if (extractFieldNames != null || context.sourceRequested()) {
164                 fieldsVisitor = new UidAndSourceFieldsVisitor();
165             } else {
166                 fieldsVisitor = new JustUidFieldsVisitor();
167             }
168         }
169 
170         InternalSearchHit[] hits = new InternalSearchHit[context.docIdsToLoadSize()];
171         FetchSubPhase.HitContext hitContext = new FetchSubPhase.HitContext();
172         for (int index = 0; index < context.docIdsToLoadSize(); index++) {
173             int docId = context.docIdsToLoad()[context.docIdsToLoadFrom() + index];
174             int readerIndex = ReaderUtil.subIndex(docId, context.searcher().getIndexReader().leaves());
175             LeafReaderContext subReaderContext = context.searcher().getIndexReader().leaves().get(readerIndex);
176             int subDocId = docId - subReaderContext.docBase;
177 
178             final InternalSearchHit searchHit;
179             try {
180                 int rootDocId = findRootDocumentIfNested(context, subReaderContext, subDocId);
181                 if (rootDocId != -1) {
182                     searchHit = createNestedSearchHit(context, docId, subDocId, rootDocId, extractFieldNames, loadAllStored, fieldNames, subReaderContext);
183                 } else {
184                     searchHit = createSearchHit(context, fieldsVisitor, docId, subDocId, extractFieldNames, subReaderContext);
185                 }
186             } catch (IOException e) {
187                 throw ExceptionsHelper.convertToElastic(e);
188             }
189 
190             hits[index] = searchHit;
191             hitContext.reset(searchHit, subReaderContext, subDocId, context.searcher().getIndexReader());
192             for (FetchSubPhase fetchSubPhase : fetchSubPhases) {
193                 if (fetchSubPhase.hitExecutionNeeded(context)) {
194                     fetchSubPhase.hitExecute(context, hitContext);
195                 }
196             }
197         }
198 
199         for (FetchSubPhase fetchSubPhase : fetchSubPhases) {
200             if (fetchSubPhase.hitsExecutionNeeded(context)) {
201                 fetchSubPhase.hitsExecute(context, hits);
202             }
203         }
204 
205         context.fetchResult().hits(new InternalSearchHits(hits, context.queryResult().topDocs().totalHits, context.queryResult().topDocs().getMaxScore()));
206     }

208     private int findRootDocumentIfNested(SearchContext context, LeafReaderContext subReaderContext, int subDocId) throws IOException {
209         if (context.mapperService().hasNested()) {
210             BitDocIdSet nonNested = context.bitsetFilterCache().getBitDocIdSetFilter(NonNestedDocsFilter.INSTANCE).getDocIdSet(subReaderContext);
211             BitSet bits = nonNested.bits();
212             if (!bits.get(subDocId)) {
213                 return bits.nextSetBit(subDocId);
214             }
215         }
216         return -1;
217     }

219     private InternalSearchHit createSearchHit(SearchContext context, FieldsVisitor fieldsVisitor, int docId, int subDocId, List<String> extractFieldNames, LeafReaderContext subReaderContext) {
220         loadStoredFields(context, subReaderContext, fieldsVisitor, subDocId);
221         fieldsVisitor.postProcess(context.mapperService());
222 
223         Map<String, SearchHitField> searchFields = null;
224         if (!fieldsVisitor.fields().isEmpty()) {
225             searchFields = new HashMap<>(fieldsVisitor.fields().size());
226             for (Map.Entry<String, List<Object>> entry : fieldsVisitor.fields().entrySet()) {
227                 searchFields.put(entry.getKey(), new InternalSearchHitField(entry.getKey(), entry.getValue()));
228             }
229         }
230 
231         DocumentMapper documentMapper = context.mapperService().documentMapper(fieldsVisitor.uid().type());
232         Text typeText;
233         if (documentMapper == null) {
234             typeText = new StringAndBytesText(fieldsVisitor.uid().type());
235         } else {
236             typeText = documentMapper.typeText();
237         }
238         InternalSearchHit searchHit = new InternalSearchHit(docId, fieldsVisitor.uid().id(), typeText, searchFields);
239 
240         // go over and extract fields that are not mapped / stored
241         context.lookup().setNextReader(subReaderContext);
242         context.lookup().setNextDocId(subDocId);
243         if (fieldsVisitor.source() != null) {
244             context.lookup().source().setNextSource(fieldsVisitor.source());
245         }
246         if (extractFieldNames != null) {
247             for (String extractFieldName : extractFieldNames) {
248                 List<Object> values = context.lookup().source().extractRawValues(extractFieldName);
249                 if (!values.isEmpty()) {
250                     if (searchHit.fieldsOrNull() == null) {
251                         searchHit.fields(new HashMap<String, SearchHitField>(2));
252                     }
253 
254                     SearchHitField hitField = searchHit.fields().get(extractFieldName);
255                     if (hitField == null) {
256                         hitField = new InternalSearchHitField(extractFieldName, new ArrayList<>(2));
257                         searchHit.fields().put(extractFieldName, hitField);
258                     }
259                     for (Object value : values) {
260                         hitField.values().add(value);
261                     }
262                 }
263             }
264         }
265 
266         return searchHit;
267     }

269     private InternalSearchHit createNestedSearchHit(SearchContext context, int nestedTopDocId, int nestedSubDocId, int rootSubDocId, List<String> extractFieldNames, boolean loadAllStored, Set<String> fieldNames, LeafReaderContext subReaderContext) throws IOException {
270         final FieldsVisitor rootFieldsVisitor;
271         if (context.sourceRequested() || extractFieldNames != null || context.highlight() != null) {
272             // Also if highlighting is requested on nested documents we need to fetch the _source from the root document,
273             // otherwise highlighting will attempt to fetch the _source from the nested doc, which will fail,
274             // because the entire _source is only stored with the root document.
275             rootFieldsVisitor = new UidAndSourceFieldsVisitor();
276         } else {
277             rootFieldsVisitor = new JustUidFieldsVisitor();
278         }
279         loadStoredFields(context, subReaderContext, rootFieldsVisitor, rootSubDocId);
280         rootFieldsVisitor.postProcess(context.mapperService());
281 
282         Map<String, SearchHitField> searchFields = getSearchFields(context, nestedSubDocId, loadAllStored, fieldNames, subReaderContext);
283         DocumentMapper documentMapper = context.mapperService().documentMapper(rootFieldsVisitor.uid().type());
284         context.lookup().setNextReader(subReaderContext);
285         context.lookup().setNextDocId(nestedSubDocId);
286 
287         ObjectMapper nestedObjectMapper = documentMapper.findNestedObjectMapper(nestedSubDocId, context.bitsetFilterCache(), subReaderContext);
288         assert nestedObjectMapper != null;
289         InternalSearchHit.InternalNestedIdentity nestedIdentity = getInternalNestedIdentity(context, nestedSubDocId, subReaderContext, documentMapper, nestedObjectMapper);
290 
291         BytesReference source = rootFieldsVisitor.source();
292         if (source != null) {
293             Tuple<XContentType, Map<String, Object>> tuple = XContentHelper.convertToMap(source, true);
294             Map<String, Object> sourceAsMap = tuple.v2();
295 
296             List<Map<String, Object>> nestedParsedSource;
297             SearchHit.NestedIdentity nested = nestedIdentity;
298             do {
299                 Object extractedValue = XContentMapValues.extractValue(nested.getField().string(), sourceAsMap);
300                 if (extractedValue instanceof List) {
301                     // nested field has an array value in the _source
302                     nestedParsedSource = (List<Map<String, Object>>) extractedValue;
303                 } else if (extractedValue instanceof Map) {
304                     // nested field has an object value in the _source. This just means the nested field has just one inner object, which is valid, but uncommon.
305                     nestedParsedSource = ImmutableList.of((Map < String, Object >) extractedValue);
306                 } else {
307                     throw new ElasticsearchIllegalStateException("extracted source isn't an object or an array");
308                 }
309                 sourceAsMap = nestedParsedSource.get(nested.getOffset());
310                 nested = nested.getChild();
311             } while (nested != null);
312 
313             context.lookup().source().setNextSource(sourceAsMap);
314             XContentType contentType = tuple.v1();
315             BytesReference nestedSource = contentBuilder(contentType).map(sourceAsMap).bytes();
316             context.lookup().source().setNextSource(nestedSource);
317             context.lookup().source().setNextSourceContentType(contentType);
318         }
319 
320         InternalSearchHit searchHit = new InternalSearchHit(nestedTopDocId, rootFieldsVisitor.uid().id(), documentMapper.typeText(), nestedIdentity, searchFields);
321         if (extractFieldNames != null) {
322             for (String extractFieldName : extractFieldNames) {
323                 List<Object> values = context.lookup().source().extractRawValues(extractFieldName);
324                 if (!values.isEmpty()) {
325                     if (searchHit.fieldsOrNull() == null) {
326                         searchHit.fields(new HashMap<String, SearchHitField>(2));
327                     }
328 
329                     SearchHitField hitField = searchHit.fields().get(extractFieldName);
330                     if (hitField == null) {
331                         hitField = new InternalSearchHitField(extractFieldName, new ArrayList<>(2));
332                         searchHit.fields().put(extractFieldName, hitField);
333                     }
334                     for (Object value : values) {
335                         hitField.values().add(value);
336                     }
337                 }
338             }
339         }
340 
341         return searchHit;
342     }

344     private Map<String, SearchHitField> getSearchFields(SearchContext context, int nestedSubDocId, boolean loadAllStored, Set<String> fieldNames, LeafReaderContext subReaderContext) {
345         Map<String, SearchHitField> searchFields = null;
346         if (context.hasFieldNames() && !context.fieldNames().isEmpty()) {
347             FieldsVisitor nestedFieldsVisitor = null;
348             if (loadAllStored) {
349                 nestedFieldsVisitor = new AllFieldsVisitor();
350             } else if (fieldNames != null) {
351                 nestedFieldsVisitor = new CustomFieldsVisitor(fieldNames, false);
352             }
353 
354             if (nestedFieldsVisitor != null) {
355                 loadStoredFields(context, subReaderContext, nestedFieldsVisitor, nestedSubDocId);
356                 nestedFieldsVisitor.postProcess(context.mapperService());
357                 if (!nestedFieldsVisitor.fields().isEmpty()) {
358                     searchFields = new HashMap<>(nestedFieldsVisitor.fields().size());
359                     for (Map.Entry<String, List<Object>> entry : nestedFieldsVisitor.fields().entrySet()) {
360                         searchFields.put(entry.getKey(), new InternalSearchHitField(entry.getKey(), entry.getValue()));
361                     }
362                 }
363             }
364         }
365         return searchFields;
366     }

368     private InternalSearchHit.InternalNestedIdentity getInternalNestedIdentity(SearchContext context, int nestedSubDocId, LeafReaderContext subReaderContext, DocumentMapper documentMapper, ObjectMapper nestedObjectMapper) throws IOException {
369         int currentParent = nestedSubDocId;
370         ObjectMapper nestedParentObjectMapper;
371         InternalSearchHit.InternalNestedIdentity nestedIdentity = null;
372         do {
373             String field;
374             Filter parentFilter;
375             nestedParentObjectMapper = documentMapper.findParentObjectMapper(nestedObjectMapper);
376             if (nestedParentObjectMapper != null && nestedObjectMapper.nested().isNested()) {
377                 field = nestedObjectMapper.name();
378                 parentFilter = nestedParentObjectMapper.nestedTypeFilter();
379             } else {
380                 field = nestedObjectMapper.fullPath();
381                 parentFilter = NonNestedDocsFilter.INSTANCE;
382             }
383 
384             BitDocIdSet parentBitSet = context.bitsetFilterCache().getBitDocIdSetFilter(parentFilter).getDocIdSet(subReaderContext);
385             BitSet parentBits = parentBitSet.bits();
386             int offset = 0;
387             BitDocIdSet nestedDocsBitSet = context.bitsetFilterCache().getBitDocIdSetFilter(nestedObjectMapper.nestedTypeFilter()).getDocIdSet(subReaderContext);
388             BitSet nestedBits = nestedDocsBitSet.bits();
389             int nextParent = parentBits.nextSetBit(currentParent);
390             for (int docId = nestedBits.nextSetBit(currentParent + 1); docId < nextParent && docId != DocIdSetIterator.NO_MORE_DOCS; docId = nestedBits.nextSetBit(docId + 1)) {
391                 offset++;
392             }
393             currentParent = nextParent;
394             nestedObjectMapper = nestedParentObjectMapper;
395             nestedIdentity = new InternalSearchHit.InternalNestedIdentity(field, offset, nestedIdentity);
396         } while (nestedParentObjectMapper != null);
397         return nestedIdentity;
398     }

400     private void loadStoredFields(SearchContext searchContext, LeafReaderContext readerContext, FieldsVisitor fieldVisitor, int docId) {
401         fieldVisitor.reset();
402         try {
403             readerContext.reader().document(docId, fieldVisitor);
404         } catch (IOException e) {
405             throw new FetchPhaseExecutionException(searchContext, "Failed to fetch doc id [" + docId + "]", e);
406         }
407     }
