20  package org.elasticsearch.common.geo.builders;
21  
22  import java.io.IOException;
23  import java.util.ArrayList;
24  import java.util.Arrays;
25  import java.util.Collection;
26  
27  import org.elasticsearch.common.xcontent.XContentBuilder;
28  
29  import com.vividsolutions.jts.geom.Coordinate;
30  
31  /**
32   * The {@link PointCollection} is an abstract base implementation for all GeoShapes. It simply handles a set of points. 
33   */
34  public abstract class PointCollection<E extends PointCollection<E>> extends ShapeBuilder {
35  
36      protected final ArrayList<Coordinate> points;
37      protected boolean translated = false;
38  
39      protected PointCollection() {
40          this(new ArrayList<Coordinate>());
41      }
42  
43      protected PointCollection(ArrayList<Coordinate> points) {
44          this.points = points;
45      }
46      
47      @SuppressWarnings("unchecked")
48      private E thisRef() {
49          return (E)this;
50      }
51  
52      /**
53       * Add a new point to the collection
54       * @param longitude longitude of the coordinate
55       * @param latitude latitude of the coordinate
56       * @return this
57       */
58      public E point(double longitude, double latitude) {
59          return this.point(coordinate(longitude, latitude));
60      } 
61  
62      /**
63       * Add a new point to the collection
64       * @param coordinate coordinate of the point
65       * @return this
66       */
67      public E point(Coordinate coordinate) {
68          this.points.add(coordinate);
69          return thisRef();
70      }
71  
72      /**
73       * Add a array of points to the collection
74       * 
75       * @param coordinates array of {@link Coordinate}s to add
76       * @return this
77       */
78      public E points(Coordinate...coordinates) {
79          return this.points(Arrays.asList(coordinates));
80      }
81  
82      /**
83       * Add a collection of points to the collection
84       * 
85       * @param coordinates array of {@link Coordinate}s to add
86       * @return this
87       */
88      public E points(Collection<? extends Coordinate> coordinates) {
89          this.points.addAll(coordinates);
90          return thisRef();
91      }
92  
93      /**
94       * Copy all points to a new Array
95       * 
96       * @param closed if set to true the first point of the array is repeated as last element
97       * @return Array of coordinates
98       */
99      protected Coordinate[] coordinates(boolean closed) {
100         Coordinate[] result = points.toArray(new Coordinate[points.size() + (closed?1:0)]);
101         if(closed) {
102             result[result.length-1] = result[0];
103         }
104         return result;
105     }
106 
107     /**
108      * builds an array of coordinates to a {@link XContentBuilder}
109      * 
110      * @param builder builder to use 
111      * @param closed repeat the first point at the end of the array if it's not already defines as last element of the array  
112      * @return the builder
113      * @throws IOException
114      */
115     protected XContentBuilder coordinatesToXcontent(XContentBuilder builder, boolean closed) throws IOException {
116         builder.startArray();
117         for(Coordinate point : points) {
118             toXContent(builder, point);
119         }
120         if(closed) {
121             Coordinate start = points.get(0);
122             Coordinate end = points.get(points.size()-1);
123             if(start.x != end.x || start.y != end.y) {
124                 toXContent(builder, points.get(0));
125             }
126         }
127         builder.endArray();
128         return builder;
129     }
130 }
