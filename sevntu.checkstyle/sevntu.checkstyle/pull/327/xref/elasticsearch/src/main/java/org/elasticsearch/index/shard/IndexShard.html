314     public IndexShard routingEntry(ShardRouting newRouting) {
315         ShardRouting currentRouting = this.shardRouting;
316         if (!newRouting.shardId().equals(shardId())) {
317             throw new ElasticsearchIllegalArgumentException("Trying to set a routing entry with shardId [" + newRouting.shardId() + "] on a shard with shardId [" + shardId() + "]");
318         }
319         if (currentRouting != null) {
320             if (!newRouting.primary() && currentRouting.primary()) {
321                 logger.warn("suspect illegal state: trying to move shard from primary mode to replica mode");
322             }
323             // if its the same routing, return
324             if (currentRouting.equals(newRouting)) {
325                 return this;
326             }
327         }
328 
329         if (state == IndexShardState.POST_RECOVERY) {
330             // if the state is started or relocating (cause it might move right away from started to relocating)
331             // then move to STARTED
332             if (newRouting.state() == ShardRoutingState.STARTED || newRouting.state() == ShardRoutingState.RELOCATING) {
333                 // we want to refresh *before* we move to internal STARTED state
334                 try {
335                     engine().refresh("cluster_state_started");
336                 } catch (Throwable t) {
337                     logger.debug("failed to refresh due to move to cluster wide started", t);
338                 }
339 
340                 boolean movedToStarted = false;
341                 synchronized (mutex) {
342                     // do the check under a mutex, so we make sure to only change to STARTED if in POST_RECOVERY
343                     if (state == IndexShardState.POST_RECOVERY) {
344                         changeState(IndexShardState.STARTED, "global state is [" + newRouting.state() + "]");
345                         movedToStarted = true;
346                     } else {
347                         logger.debug("state [{}] not changed, not in POST_RECOVERY, global state is [{}]", state, newRouting.state());
348                     }
349                 }
350                 if (movedToStarted) {
351                     indicesLifecycle.afterIndexShardStarted(this);
352                 }
353             }
354         }
355 
356         this.shardRouting = newRouting;
357         indicesLifecycle.shardRoutingChanged(this, currentRouting, newRouting);
358 
359         return this;
360     }

378     private IndexShardState recovering(String reason, RecoveryState recoveryState) throws IndexShardStartedException,
379             IndexShardRelocatedException, IndexShardRecoveringException, IndexShardClosedException {
380         synchronized (mutex) {
381             if (state == IndexShardState.CLOSED) {
382                 throw new IndexShardClosedException(shardId);
383             }
384             if (state == IndexShardState.STARTED) {
385                 throw new IndexShardStartedException(shardId);
386             }
387             if (state == IndexShardState.RELOCATED) {
388                 throw new IndexShardRelocatedException(shardId);
389             }
390             if (state == IndexShardState.RECOVERING) {
391                 throw new IndexShardRecoveringException(shardId);
392             }
393             if (state == IndexShardState.POST_RECOVERY) {
394                 throw new IndexShardRecoveringException(shardId);
395             }
396             this.recoveryState = recoveryState;
397             return changeState(IndexShardState.RECOVERING, reason);
398         }
399     }
400 
401     public IndexShard relocated(String reason) throws IndexShardNotStartedException {
402         synchronized (mutex) {
403             if (state != IndexShardState.STARTED) {
404                 throw new IndexShardNotStartedException(shardId, state);
405             }
406             changeState(IndexShardState.RELOCATED, reason);
407         }
408         return this;
409     }

415     /**
416      * Changes the state of the current shard
417      *
418      * @param newState the new shard state
419      * @param reason   the reason for the state change
420      * @return the previous shard state
421      */
422     private IndexShardState changeState(IndexShardState newState, String reason) {
423         logger.debug("state: [{}]->[{}], reason [{}]", state, newState, reason);
424         IndexShardState previousState = state;
425         state = newState;
426         this.indicesLifecycle.indexShardStateChanged(this, previousState, reason);
427         return previousState;
428     }

430     public Engine.Create prepareCreate(SourceToParse source, long version, VersionType versionType, Engine.Operation.Origin origin, boolean canHaveDuplicates, boolean autoGeneratedId) throws ElasticsearchException {
431         long startTime = System.nanoTime();
432         Tuple<DocumentMapper, Boolean> docMapper = mapperService.documentMapperWithAutoCreate(source.type());
433         try {
434             ParsedDocument doc = docMapper.v1().parse(source).setMappingsModified(docMapper);
435             return new Engine.Create(docMapper.v1(), docMapper.v1().uidMapper().term(doc.uid().stringValue()), doc, version, versionType, origin, startTime, state != IndexShardState.STARTED || canHaveDuplicates, autoGeneratedId);
436         } catch (Throwable t) {
437             if (docMapper.v2()) {
438                 throw new WriteFailureException(t, docMapper.v1().type());
439             } else {
440                 throw t;
441             }
442         }
443     }

518     public Engine.DeleteByQuery prepareDeleteByQuery(BytesReference source, @Nullable String[] filteringAliases, Engine.Operation.Origin origin, String... types) throws ElasticsearchException {
519         long startTime = System.nanoTime();
520         if (types == null) {
521             types = Strings.EMPTY_ARRAY;
522         }
523         Query query = queryParserService.parseQuery(source).query();
524         query = filterQueryIfNeeded(query, types);
525 
526         Filter aliasFilter = indexAliasesService.aliasFilter(filteringAliases);
527         BitDocIdSetFilter parentFilter = mapperService.hasNested() ? indexCache.bitsetFilterCache().getBitDocIdSetFilter(NonNestedDocsFilter.INSTANCE) : null;
528         return new Engine.DeleteByQuery(query, source, filteringAliases, aliasFilter, parentFilter, origin, startTime, types);
529     }

704     public void close(String reason, boolean flushEngine) throws IOException {
705         synchronized (mutex) {
706             try {
707                 indexSettingsService.removeListener(applyRefreshSettings);
708                 if (state != IndexShardState.CLOSED) {
709                     FutureUtils.cancel(refreshScheduledFuture);
710                     refreshScheduledFuture = null;
711                     FutureUtils.cancel(mergeScheduleFuture);
712                     mergeScheduleFuture = null;
713                 }
714                 changeState(IndexShardState.CLOSED, reason);
715             } finally {
716                 final Engine engine = this.currentEngineReference.getAndSet(null);
717                 try {
718                     if (flushEngine && this.flushOnClose) {
719                         engine.flushAndClose();
720                     }
721                 } finally { // playing safe here and close the engine even if the above succeeds - close can be called multiple times
722                     IOUtils.close(engine);
723                 }
724             }
725         }
726     }

728     public IndexShard postRecovery(String reason) throws IndexShardStartedException, IndexShardRelocatedException, IndexShardClosedException {
729         synchronized (mutex) {
730             if (state == IndexShardState.CLOSED) {
731                 throw new IndexShardClosedException(shardId);
732             }
733             if (state == IndexShardState.STARTED) {
734                 throw new IndexShardStartedException(shardId);
735             }
736             if (state == IndexShardState.RELOCATED) {
737                 throw new IndexShardRelocatedException(shardId);
738             }
739             recoveryState.setStage(RecoveryState.Stage.DONE);
740             changeState(IndexShardState.POST_RECOVERY, reason);
741         }
742         indicesLifecycle.afterIndexShardPostRecovery(this);
743         return this;
744     }

755     /**
756      * After the store has been recovered, we need to start the engine in order to apply operations
757      */
758     public void prepareForTranslogRecovery() throws ElasticsearchException {
759         if (state != IndexShardState.RECOVERING) {
760             throw new IndexShardNotRecoveringException(shardId, state);
761         }
762         recoveryState.setStage(RecoveryState.Stage.START);
763         // also check here, before we apply the translog
764         if (Booleans.parseBoolean(checkIndexOnStartup, false)) {
765             checkIndex();
766         }
767         // we disable deletes since we allow for operations to be executed against the shard while recovering
768         // but we need to make sure we don't loose deletes until we are done recovering
769         config.setEnableGcDeletes(false);
770         createNewEngine();
771         recoveryState.setStage(RecoveryState.Stage.TRANSLOG);
772     }

807     /**
808      * Performs a single recovery operation, and returns the indexing operation (or null if its not an indexing operation)
809      * that can then be used for mapping updates (for example) if needed.
810      */
811     public Engine.IndexingOperation performRecoveryOperation(Translog.Operation operation) throws ElasticsearchException {
812         if (state != IndexShardState.RECOVERING) {
813             throw new IndexShardNotRecoveringException(shardId, state);
814         }
815         Engine.IndexingOperation indexOperation = null;
816         try {
817             switch (operation.opType()) {
818                 case CREATE:
819                     Translog.Create create = (Translog.Create) operation;
820                     Engine.Create engineCreate = prepareCreate(
821                             source(create.source()).type(create.type()).id(create.id())
822                                     .routing(create.routing()).parent(create.parent()).timestamp(create.timestamp()).ttl(create.ttl()),
823                             create.version(), create.versionType().versionTypeForReplicationAndRecovery(), Engine.Operation.Origin.RECOVERY, true, false);
824                     mapperAnalyzer.setType(create.type());
825                     engine().create(engineCreate);
826                     indexOperation = engineCreate;
827                     break;
828                 case SAVE:
829                     Translog.Index index = (Translog.Index) operation;
830                     Engine.Index engineIndex = prepareIndex(source(index.source()).type(index.type()).id(index.id())
831                                     .routing(index.routing()).parent(index.parent()).timestamp(index.timestamp()).ttl(index.ttl()),
832                             index.version(), index.versionType().versionTypeForReplicationAndRecovery(), Engine.Operation.Origin.RECOVERY, true);
833                     mapperAnalyzer.setType(index.type());
834                     engine().index(engineIndex);
835                     indexOperation = engineIndex;
836                     break;
837                 case DELETE:
838                     Translog.Delete delete = (Translog.Delete) operation;
839                     Uid uid = Uid.createUid(delete.uid().text());
840                     engine().delete(new Engine.Delete(uid.type(), uid.id(), delete.uid(), delete.version(),
841                             delete.versionType().versionTypeForReplicationAndRecovery(), Engine.Operation.Origin.RECOVERY, System.nanoTime(), false));
842                     break;
843                 case DELETE_BY_QUERY:
844                     Translog.DeleteByQuery deleteByQuery = (Translog.DeleteByQuery) operation;
845                     engine().delete(prepareDeleteByQuery(deleteByQuery.source(), deleteByQuery.filteringAliases(), Engine.Operation.Origin.RECOVERY, deleteByQuery.types()));
846                     break;
847                 default:
848                     throw new ElasticsearchIllegalStateException("No operation defined for [" + operation + "]");
849             }
850         } catch (ElasticsearchException e) {
851             boolean hasIgnoreOnRecoveryException = false;
852             ElasticsearchException current = e;
853             while (true) {
854                 if (current instanceof IgnoreOnRecoveryEngineException) {
855                     hasIgnoreOnRecoveryException = true;
856                     break;
857                 }
858                 if (current.getCause() instanceof ElasticsearchException) {
859                     current = (ElasticsearchException) current.getCause();
860                 } else {
861                     break;
862                 }
863             }
864             if (!hasIgnoreOnRecoveryException) {
865                 throw e;
866             }
867         }
868         return indexOperation;
869     }

885     private void readAllowed(boolean writeOperation) throws IllegalIndexShardStateException {
886         IndexShardState state = this.state; // one time volatile read
887         if (writeOperation) {
888             if (state != IndexShardState.STARTED && state != IndexShardState.RELOCATED && state != IndexShardState.RECOVERING && state != IndexShardState.POST_RECOVERY) {
889                 throw new IllegalIndexShardStateException(shardId, state, "operations only allowed when started/relocated");
890             }
891         } else {
892             if (state != IndexShardState.STARTED && state != IndexShardState.RELOCATED) {
893                 throw new IllegalIndexShardStateException(shardId, state, "operations only allowed when started/relocated");
894             }
895         }
896     }

898     private void writeAllowed(Engine.Operation.Origin origin) throws IllegalIndexShardStateException {
899         IndexShardState state = this.state; // one time volatile read
900 
901         if (origin == Engine.Operation.Origin.PRIMARY) {
902             // for primaries, we only allow to write when actually started (so the cluster has decided we started)
903             // otherwise, we need to retry, we also want to still allow to index if we are relocated in case it fails
904             if (state != IndexShardState.STARTED && state != IndexShardState.RELOCATED) {
905                 throw new IllegalIndexShardStateException(shardId, state, "operation only allowed when started/recovering, origin [" + origin + "]");
906             }
907         } else {
908             // for replicas, we allow to write also while recovering, since we index also during recovery to replicas
909             // and rely on version checks to make sure its consistent
910             if (state != IndexShardState.STARTED && state != IndexShardState.RELOCATED && state != IndexShardState.RECOVERING && state != IndexShardState.POST_RECOVERY) {
911                 throw new IllegalIndexShardStateException(shardId, state, "operation only allowed when started/recovering, origin [" + origin + "]");
912             }
913         }
914     }

960     public void updateBufferSize(ByteSizeValue shardIndexingBufferSize, ByteSizeValue shardTranslogBufferSize) {
961         ByteSizeValue preValue = config.getIndexingBufferSize();
962         config.setIndexingBufferSize(shardIndexingBufferSize);
963         if (preValue.bytes() != shardIndexingBufferSize.bytes()) {
964             // its inactive, make sure we do a refresh / full IW flush in this case, since the memory
965             // changes only after a "data" change has happened to the writer
966             // the index writer lazily allocates memory and a refresh will clean it all up.
967             if (shardIndexingBufferSize == EngineConfig.INACTIVE_SHARD_INDEXING_BUFFER && preValue != EngineConfig.INACTIVE_SHARD_INDEXING_BUFFER) {
968                 logger.debug("updating index_buffer_size from [{}] to (inactive) [{}]", preValue, shardIndexingBufferSize);
969                 try {
970                     refresh("update index buffer");
971                 } catch (Throwable e) {
972                     logger.warn("failed to refresh after setting shard to inactive", e);
973                 }
974             } else {
975                 logger.debug("updating index_buffer_size from [{}] to [{}]", preValue, shardIndexingBufferSize);
976             }
977         }
978         translog().updateBuffer(shardTranslogBufferSize);
979     }

989     private class ApplyRefreshSettings implements IndexSettingsService.Listener {
990         @Override
991         public void onRefreshSettings(Settings settings) {
992             boolean change = false;
993             synchronized (mutex) {
994                 if (state == IndexShardState.CLOSED) {
995                     return;
996                 }
997                 final boolean flushOnClose = settings.getAsBoolean(INDEX_FLUSH_ON_CLOSE, IndexShard.this.flushOnClose);
998                 if (flushOnClose != IndexShard.this.flushOnClose) {
999                     logger.info("updating {} from [{}] to [{}]", INDEX_FLUSH_ON_CLOSE, IndexShard.this.flushOnClose, flushOnClose);
1000                     IndexShard.this.flushOnClose = flushOnClose;
1001                 }
1002 
1003                 TimeValue refreshInterval = settings.getAsTime(INDEX_REFRESH_INTERVAL, IndexShard.this.refreshInterval);
1004                 if (!refreshInterval.equals(IndexShard.this.refreshInterval)) {
1005                     logger.info("updating refresh_interval from [{}] to [{}]", IndexShard.this.refreshInterval, refreshInterval);
1006                     if (refreshScheduledFuture != null) {
1007                         // NOTE: we pass false here so we do NOT attempt Thread.interrupt if EngineRefresher.run is currently running.  This is
1008                         // very important, because doing so can cause files to suddenly be closed if they were doing IO when the interrupt
1009                         // hit.  See https://issues.apache.org/jira/browse/LUCENE-2239
1010                         FutureUtils.cancel(refreshScheduledFuture);
1011                         refreshScheduledFuture = null;
1012                     }
1013                     IndexShard.this.refreshInterval = refreshInterval;
1014                     if (refreshInterval.millis() > 0) {
1015                         refreshScheduledFuture = threadPool.schedule(refreshInterval, ThreadPool.Names.SAME, new EngineRefresher());
1016                     }
1017                 }
1018 
1019                 long gcDeletesInMillis = settings.getAsTime(EngineConfig.INDEX_GC_DELETES_SETTING, TimeValue.timeValueMillis(config.getGcDeletesInMillis())).millis();
1020                 if (gcDeletesInMillis != config.getGcDeletesInMillis()) {
1021                     logger.info("updating {} from [{}] to [{}]", EngineConfig.INDEX_GC_DELETES_SETTING, TimeValue.timeValueMillis(config.getGcDeletesInMillis()), TimeValue.timeValueMillis(gcDeletesInMillis));
1022                     config.setGcDeletesInMillis(gcDeletesInMillis);
1023                     change = true;
1024                 }
1025 
1026                 final boolean compoundOnFlush = settings.getAsBoolean(EngineConfig.INDEX_COMPOUND_ON_FLUSH, config.isCompoundOnFlush());
1027                 if (compoundOnFlush != config.isCompoundOnFlush()) {
1028                     logger.info("updating {} from [{}] to [{}]", EngineConfig.INDEX_COMPOUND_ON_FLUSH, config.isCompoundOnFlush(), compoundOnFlush);
1029                     config.setCompoundOnFlush(compoundOnFlush);
1030                     change = true;
1031                 }
1032 
1033                 final boolean failEngineOnCorruption = settings.getAsBoolean(EngineConfig.INDEX_FAIL_ON_CORRUPTION_SETTING, config.isFailEngineOnCorruption());
1034                 if (failEngineOnCorruption != config.isFailEngineOnCorruption()) {
1035                     logger.info("updating {} from [{}] to [{}]", EngineConfig.INDEX_FAIL_ON_CORRUPTION_SETTING, config.isFailEngineOnCorruption(), failEngineOnCorruption);
1036                     config.setFailEngineOnCorruption(failEngineOnCorruption);
1037                     change = true;
1038                 }
1039                 final boolean failOnMergeFailure = settings.getAsBoolean(EngineConfig.INDEX_FAIL_ON_MERGE_FAILURE_SETTING, config.isFailOnMergeFailure());
1040                 if (failOnMergeFailure != config.isFailOnMergeFailure()) {
1041                     logger.info("updating {} from [{}] to [{}]", EngineConfig.INDEX_FAIL_ON_MERGE_FAILURE_SETTING, config.isFailOnMergeFailure(), failOnMergeFailure);
1042                     config.setFailOnMergeFailure(failOnMergeFailure);
1043                     change = true;
1044                 }
1045             }
1046             if (change) {
1047                 refresh("apply settings");
1048             }
1049         }
1050     }

1052     class EngineRefresher implements Runnable {
1053         @Override
1054         public void run() {
1055             // we check before if a refresh is needed, if not, we reschedule, otherwise, we fork, refresh, and then reschedule
1056             if (!engine().refreshNeeded()) {
1057                 reschedule();
1058                 return;
1059             }
1060             threadPool.executor(ThreadPool.Names.REFRESH).execute(new Runnable() {
1061                 @Override
1062                 public void run() {
1063                     try {
1064                         if (engine().refreshNeeded()) {
1065                             refresh("schedule");
1066                         }
1067                     } catch (EngineClosedException e) {
1068                         // we are being closed, ignore
1069                     } catch (RefreshFailedEngineException e) {
1070                         if (e.getCause() instanceof InterruptedException) {
1071                             // ignore, we are being shutdown
1072                         } else if (e.getCause() instanceof ClosedByInterruptException) {
1073                             // ignore, we are being shutdown
1074                         } else if (e.getCause() instanceof ThreadInterruptedException) {
1075                             // ignore, we are being shutdown
1076                         } else {
1077                             if (state != IndexShardState.CLOSED) {
1078                                 logger.warn("Failed to perform scheduled engine refresh", e);
1079                             }
1080                         }
1081                     } catch (Exception e) {
1082                         if (state != IndexShardState.CLOSED) {
1083                             logger.warn("Failed to perform scheduled engine refresh", e);
1084                         }
1085                     }
1086 
1087                     reschedule();
1088                 }
1089             });
1090         }
1091 
1092         /** Schedules another (future) refresh, if refresh_interval is still enabled. */
1093         private void reschedule() {
1094             synchronized (mutex) {
1095                 if (state != IndexShardState.CLOSED && refreshInterval.millis() > 0) {
1096                     refreshScheduledFuture = threadPool.schedule(refreshInterval, ThreadPool.Names.SAME, this);
1097                 }
1098             }
1099         }
1100     }

1102     private void checkIndex() throws IndexShardException {
1103         if (store.tryIncRef()) {
1104             try {
1105                 doCheckIndex();
1106             } catch (IOException e) {
1107                 throw new IndexShardException(shardId, "exception during checkindex", e);
1108             } finally {
1109                 store.decRef();
1110             }
1111         }
1112     }

1114     private void doCheckIndex() throws IndexShardException, IOException {
1115         long time = System.currentTimeMillis();
1116         if (!Lucene.indexExists(store.directory())) {
1117             return;
1118         }
1119         BytesStreamOutput os = new BytesStreamOutput();
1120         PrintStream out = new PrintStream(os, false, Charsets.UTF_8.name());
1121         
1122         if ("checksum".equalsIgnoreCase(checkIndexOnStartup)) {
1123             // physical verification only: verify all checksums for the latest commit
1124             boolean corrupt = false;
1125             MetadataSnapshot metadata = store.getMetadata();
1126             for (Map.Entry<String,StoreFileMetaData> entry : metadata.asMap().entrySet()) {
1127                 try {
1128                     Store.checkIntegrity(entry.getValue(), store.directory());
1129                     out.println("checksum passed: " + entry.getKey());
1130                 } catch (IOException exc) {
1131                     out.println("checksum failed: " + entry.getKey());
1132                     exc.printStackTrace(out);
1133                     corrupt = true;
1134                 }
1135             }
1136             out.flush();
1137             if (corrupt) {
1138                 logger.warn("check index [failure]\n{}", new String(os.bytes().toBytes(), Charsets.UTF_8));
1139                 throw new IndexShardException(shardId, "index check failure");
1140             }
1141         } else {
1142             // full checkindex
1143             try (CheckIndex checkIndex = new CheckIndex(store.directory())) {
1144                 checkIndex.setInfoStream(out);
1145                 CheckIndex.Status status = checkIndex.checkIndex();
1146                 out.flush();
1147                 
1148                 if (!status.clean) {
1149                     if (state == IndexShardState.CLOSED) {
1150                         // ignore if closed....
1151                         return;
1152                     }
1153                     logger.warn("check index [failure]\n{}", new String(os.bytes().toBytes(), Charsets.UTF_8));
1154                     if ("fix".equalsIgnoreCase(checkIndexOnStartup)) {
1155                         if (logger.isDebugEnabled()) {
1156                             logger.debug("fixing index, writing new segments file ...");
1157                         }
1158                         checkIndex.exorciseIndex(status);
1159                         if (logger.isDebugEnabled()) {
1160                             logger.debug("index fixed, wrote new segments file \"{}\"", status.segmentsFileName);
1161                         }
1162                     } else {
1163                         // only throw a failure if we are not going to fix the index
1164                         throw new IndexShardException(shardId, "index check failure");
1165                     }
1166                 }
1167             }
1168         }
1169         
1170         if (logger.isDebugEnabled()) {
1171             logger.debug("check index [success]\n{}", new String(os.bytes().toBytes(), Charsets.UTF_8));
1172         }
1173 
1174         recoveryState.getStart().checkIndexTime(Math.max(0, System.currentTimeMillis() - time));
1175     }

1177     public Engine engine() {
1178         Engine engine = this.currentEngineReference.get();
1179         if (engine == null) {
1180             throw new EngineClosedException(shardId);
1181         }
1182         return engine;
1183     }
