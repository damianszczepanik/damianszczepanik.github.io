58  /**
59   * A query implementation that executes the wrapped child query and connects all the matching child docs to the related
60   * parent documents using {@link ParentChildIndexFieldData}.
61   * <p/>
62   * This query is executed in two rounds. The first round resolves all the matching child documents and groups these
63   * documents by parent uid value. Also the child scores are aggregated per parent uid value. During the second round
64   * all parent documents having the same uid value that is collected in the first phase are emitted as hit including
65   * a score based on the aggregated child scores and score type.
66   */
67  public class ChildrenQuery extends Query {

78      protected final BitDocIdSetFilter nonNestedDocsFilter;

83      public ChildrenQuery(ParentChildIndexFieldData ifd, String parentType, String childType, Filter parentFilter, Query childQuery, ScoreType scoreType, int minChildren, int maxChildren, int shortCircuitParentDocSet, BitDocIdSetFilter nonNestedDocsFilter) {
84          this.ifd = ifd;
85          this.parentType = parentType;
86          this.childType = childType;
87          this.parentFilter = parentFilter;
88          this.originalChildQuery = childQuery;
89          this.scoreType = scoreType;
90          this.shortCircuitParentDocSet = shortCircuitParentDocSet;
91          this.nonNestedDocsFilter = nonNestedDocsFilter;
92          assert maxChildren == 0 || minChildren <= maxChildren;
93          this.minChildren = minChildren > 1 ? minChildren : 0;
94          this.maxChildren = maxChildren;
95      }

99          if (this == obj) {
100             return true;
101         }
102         if (obj == null || obj.getClass() != this.getClass()) {
103             return false;
104         }
105 
106         ChildrenQuery that = (ChildrenQuery) obj;
107         if (!originalChildQuery.equals(that.originalChildQuery)) {
108             return false;
109         }
110         if (!childType.equals(that.childType)) {
111             return false;
112         }
113         if (getBoost() != that.getBoost()) {
114             return false;
115         }
116         if (minChildren != that.minChildren) {
117             return false;
118         }
119         if (maxChildren != that.maxChildren) {
120             return false;
121         }
122         return true;
123     }

167     @Override
168     public Weight createWeight(IndexSearcher searcher, boolean needsScores) throws IOException {
169         SearchContext sc = SearchContext.current();
170         assert rewrittenChildQuery != null;
171         assert rewriteIndexReader == searcher.getIndexReader() : "not equal, rewriteIndexReader=" + rewriteIndexReader
172                 + " searcher.getIndexReader()=" + searcher.getIndexReader();
173         final Query childQuery = rewrittenChildQuery;
174 
175         IndexParentChildFieldData globalIfd = ifd.loadGlobal(searcher.getIndexReader());
176         if (globalIfd == null) {
177             // No docs of the specified type exist on this shard
178             return Queries.newMatchNoDocsQuery().createWeight(searcher, needsScores);
179         }
180         IndexSearcher indexSearcher = new IndexSearcher(searcher.getIndexReader());
181         indexSearcher.setSimilarity(searcher.getSimilarity());
182 
183         boolean abort = true;
184         long numFoundParents;
185         ParentCollector collector = null;
186         try {
187             if (minChildren == 0 && maxChildren == 0 && scoreType != ScoreType.NONE) {
188                 switch (scoreType) {
189                 case MIN:
190                     collector = new MinCollector(globalIfd, sc, parentType);
191                     break;
192                 case MAX:
193                     collector = new MaxCollector(globalIfd, sc, parentType);
194                     break;
195                 case SUM:
196                     collector = new SumCollector(globalIfd, sc, parentType);
197                     break;
198                 }
199             }
200             if (collector == null) {
201                 switch (scoreType) {
202                 case MIN:
203                     collector = new MinCountCollector(globalIfd, sc, parentType);
204                     break;
205                 case MAX:
206                     collector = new MaxCountCollector(globalIfd, sc, parentType);
207                     break;
208                 case SUM:
209                 case AVG:
210                     collector = new SumCountAndAvgCollector(globalIfd, sc, parentType);
211                     break;
212                 case NONE:
213                     collector = new CountCollector(globalIfd, sc, parentType);
214                     break;
215                 default:
216                     throw new RuntimeException("Are we missing a score type here? -- " + scoreType);
217                 }
218             }
219 
220             indexSearcher.search(childQuery, collector);
221             numFoundParents = collector.foundParents();
222             if (numFoundParents == 0) {
223                 return Queries.newMatchNoDocsQuery().createWeight(searcher, needsScores);
224             }
225             abort = false;
226         } finally {
227             if (abort) {
228                 Releasables.close(collector);
229             }
230         }
231         sc.addReleasable(collector, Lifetime.COLLECTION);
232         final Filter parentFilter;
233         if (numFoundParents <= shortCircuitParentDocSet) {
234             parentFilter = ParentIdsFilter.createShortCircuitFilter(nonNestedDocsFilter, sc, parentType, collector.values,
235                     collector.parentIdxs, numFoundParents);
236         } else {
237             parentFilter = this.parentFilter;
238         }
239         return new ParentWeight(this, rewrittenChildQuery.createWeight(searcher, needsScores), parentFilter, numFoundParents, collector, minChildren,
240                 maxChildren);
241     }

243     protected class ParentWeight extends Weight {
244 
245         protected final Weight childWeight;
246         protected final Filter parentFilter;
247         protected final ParentCollector collector;
248         protected final int minChildren;
249         protected final int maxChildren;
250 
251         protected long remaining;
252         protected float queryNorm;
253         protected float queryWeight;
254 
255         protected ParentWeight(Query query, Weight childWeight, Filter parentFilter, long remaining, ParentCollector collector, int minChildren, int maxChildren) {
256             super(query);
257             this.childWeight = childWeight;
258             this.parentFilter = parentFilter;
259             this.remaining = remaining;
260             this.collector = collector;
261             this.minChildren = minChildren;
262             this.maxChildren = maxChildren;
263         }
264 
265         @Override
266         public Explanation explain(LeafReaderContext context, int doc) throws IOException {
267             return new Explanation(getBoost(), "not implemented yet...");
268         }
269 
270         @Override
271         public void normalize(float norm, float topLevelBoost) {
272             this.queryNorm = norm * topLevelBoost;
273             queryWeight *= this.queryNorm;
274         }
275 
276         @Override
277         public float getValueForNormalization() throws IOException {
278             queryWeight = getBoost();
279             if (scoreType == ScoreType.NONE) {
280                 return queryWeight * queryWeight;
281             }
282             float sum = childWeight.getValueForNormalization();
283             sum *= queryWeight * queryWeight;
284             return sum;
285         }
286 
287         @Override
288         public Scorer scorer(LeafReaderContext context, Bits acceptDocs) throws IOException {
289             DocIdSet parentsSet = parentFilter.getDocIdSet(context, acceptDocs);
290             if (DocIdSets.isEmpty(parentsSet) || remaining == 0) {
291                 return null;
292             }
293 
294             // We can't be sure of the fact that liveDocs have been applied, so we apply it here. The "remaining"
295             // count down (short circuit) logic will then work as expected.
296             DocIdSetIterator parents = BitsFilteredDocIdSet.wrap(parentsSet, context.reader().getLiveDocs()).iterator();
297 
298             if (parents != null) {
299                 SortedDocValues bytesValues = collector.globalIfd.load(context).getOrdinalsValues(parentType);
300                 if (bytesValues == null) {
301                     return null;
302                 }
303 
304                 if (minChildren > 0 || maxChildren != 0 || scoreType == ScoreType.NONE) {
305                     switch (scoreType) {
306                     case NONE:
307                         DocIdSetIterator parentIdIterator = new CountParentOrdIterator(this, parents, collector, bytesValues,
308                                 minChildren, maxChildren);
309                         return ConstantScorer.create(parentIdIterator, this, queryWeight);
310                     case AVG:
311                         return new AvgParentCountScorer(this, parents, collector, bytesValues, minChildren, maxChildren);
312                     default:
313                         return new ParentCountScorer(this, parents, collector, bytesValues, minChildren, maxChildren);
314                     }
315                 }
316                 switch (scoreType) {
317                 case AVG:
318                     return new AvgParentScorer(this, parents, collector, bytesValues);
319                 default:
320                     return new ParentScorer(this, parents, collector, bytesValues);
321                 }
322             }
323             return null;
324         }
325     }

327     protected abstract static class ParentCollector extends NoopCollector implements Releasable {
328 
329         protected final IndexParentChildFieldData globalIfd;
330         protected final LongHash parentIdxs;
331         protected final BigArrays bigArrays;
332         protected final SearchContext searchContext;
333         protected final String parentType;
334 
335         protected SortedDocValues values;
336         protected Scorer scorer;
337 
338         protected ParentCollector(IndexParentChildFieldData globalIfd, SearchContext searchContext, String parentType) {
339             this.globalIfd = globalIfd;
340             this.searchContext = searchContext;
341             this.bigArrays = searchContext.bigArrays();
342             this.parentIdxs = new LongHash(512, bigArrays);
343             this.parentType = parentType;
344         }
345 
346         @Override
347         public final void collect(int doc) throws IOException {
348             if (values != null) {
349                 final long globalOrdinal = values.getOrd(doc);
350                 if (globalOrdinal >= 0) {
351                     long parentIdx = parentIdxs.add(globalOrdinal);
352                     if (parentIdx >= 0) {
353                         newParent(parentIdx);
354                     } else {
355                         parentIdx = -1 - parentIdx;
356                         existingParent(parentIdx);
357                     }
358                 }
359             }
360         }
361 
362         protected void newParent(long parentIdx) throws IOException {
363         }
364 
365         protected void existingParent(long parentIdx) throws IOException {
366         }
367 
368         public long foundParents() {
369             return parentIdxs.size();
370         }
371 
372         @Override
373         protected void doSetNextReader(LeafReaderContext context) throws IOException {
374             values = globalIfd.load(context).getOrdinalsValues(parentType);
375         }
376 
377         @Override
378         public void setScorer(Scorer scorer) throws IOException {
379             this.scorer = scorer;
380         }
381 
382         @Override
383         public void close() throws ElasticsearchException {
384             Releasables.close(parentIdxs);
385         }
386     }

550     private static class ParentScorer extends Scorer {
551 
552         final ParentWeight parentWeight;
553         final LongHash parentIds;
554         final FloatArray scores;
555 
556         final SortedDocValues globalOrdinals;
557         final DocIdSetIterator parentsIterator;
558 
559         int currentDocId = -1;
560         float currentScore;
561 
562         ParentScorer(ParentWeight parentWeight, DocIdSetIterator parentsIterator, ParentCollector collector, SortedDocValues globalOrdinals) {
563             super(parentWeight);
564             this.parentWeight = parentWeight;
565             this.globalOrdinals = globalOrdinals;
566             this.parentsIterator = parentsIterator;
567             this.parentIds = collector.parentIdxs;
568             this.scores = ((ParentScoreCollector) collector).scores;
569         }
570 
571         @Override
572         public float score() throws IOException {
573             return currentScore;
574         }
575 
576         protected boolean acceptAndScore(long parentIdx) {
577             currentScore = scores.get(parentIdx);
578             return true;
579         }
580 
581         @Override
582         public int freq() throws IOException {
583             // We don't have the original child query hit info here...
584             // But the freq of the children could be collector and returned here, but makes this Scorer more expensive.
585             return 1;
586         }
587 
588         @Override
589         public int docID() {
590             return currentDocId;
591         }
592 
593         @Override
594         public int nextDoc() throws IOException {
595             if (parentWeight.remaining == 0) {
596                 return currentDocId = NO_MORE_DOCS;
597             }
598 
599             while (true) {
600                 currentDocId = parentsIterator.nextDoc();
601                 if (currentDocId == DocIdSetIterator.NO_MORE_DOCS) {
602                     return currentDocId;
603                 }
604 
605                 final int globalOrdinal = globalOrdinals.getOrd(currentDocId);
606                 if (globalOrdinal < 0) {
607                     continue;
608                 }
609 
610                 final long parentIdx = parentIds.find(globalOrdinal);
611                 if (parentIdx != -1) {
612                     parentWeight.remaining--;
613                     if (acceptAndScore(parentIdx)) {
614                         return currentDocId;
615                     }
616                 }
617             }
618         }
619 
620         @Override
621         public int advance(int target) throws IOException {
622             if (parentWeight.remaining == 0) {
623                 return currentDocId = NO_MORE_DOCS;
624             }
625 
626             currentDocId = parentsIterator.advance(target);
627             if (currentDocId == DocIdSetIterator.NO_MORE_DOCS) {
628                 return currentDocId;
629             }
630 
631             final long globalOrdinal = globalOrdinals.getOrd(currentDocId);
632             if (globalOrdinal < 0) {
633                 return nextDoc();
634             }
635 
636             final long parentIdx = parentIds.find(globalOrdinal);
637             if (parentIdx != -1) {
638                 parentWeight.remaining--;
639                 if (acceptAndScore(parentIdx)) {
640                     return currentDocId;
641                 }
642             }
643             return nextDoc();
644         }
645 
646         @Override
647         public long cost() {
648             return parentsIterator.cost();
649         }
650     }

708     private final static class CountParentOrdIterator extends XFilteredDocIdSetIterator {
709 
710         private final LongHash parentIds;
711         protected final IntArray occurrences;
712         private final int minChildren;
713         private final int maxChildren;
714         private final SortedDocValues ordinals;
715         private final ParentWeight parentWeight;
716 
717         private CountParentOrdIterator(ParentWeight parentWeight, DocIdSetIterator innerIterator, ParentCollector collector, SortedDocValues ordinals, int minChildren, int maxChildren) {
718             super(innerIterator);
719             this.parentIds = ((CountCollector) collector).parentIdxs;
720             this.occurrences = ((CountCollector) collector).occurrences;
721             this.ordinals = ordinals;
722             this.parentWeight = parentWeight;
723             this.minChildren = minChildren;
724             this.maxChildren = maxChildren == 0 ? Integer.MAX_VALUE : maxChildren;
725         }
726 
727         @Override
728         protected boolean match(int doc) {
729             if (parentWeight.remaining == 0) {
730                 throw new CollectionTerminatedException();
731             }
732 
733             final long parentOrd = ordinals.getOrd(doc);
734             if (parentOrd >= 0) {
735                 final long parentIdx = parentIds.find(parentOrd);
736                 if (parentIdx != -1) {
737                     parentWeight.remaining--;
738                     int count = occurrences.get(parentIdx);
739                     if (count >= minChildren && count <= maxChildren) {
740                         return true;
741                     }
742                 }
743             }
744             return false;
745         }
746     }
