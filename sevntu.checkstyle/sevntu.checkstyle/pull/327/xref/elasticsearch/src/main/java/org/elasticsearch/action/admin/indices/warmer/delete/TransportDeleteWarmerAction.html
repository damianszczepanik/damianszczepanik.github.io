38  import org.elasticsearch.search.warmer.IndexWarmerMissingException;

43  import java.util.Arrays;

46  /**
47   * Internal Actions executed on the master deleting the warmer from the cluster state metadata.
48   *
49   * Note: this is an internal API and should not be used / called by any client code.
50   */
51  public class TransportDeleteWarmerAction extends TransportMasterNodeOperationAction<DeleteWarmerRequest, DeleteWarmerResponse> {
52  
53      @Inject
54      public TransportDeleteWarmerAction(Settings settings, TransportService transportService, ClusterService clusterService, ThreadPool threadPool, ActionFilters actionFilters) {
55          super(settings, DeleteWarmerAction.NAME, transportService, clusterService, threadPool, actionFilters);
56      }
57  
58      @Override
59      protected String executor() {
60          // we go async right away
61          return ThreadPool.Names.SAME;
62      }
63  
64      @Override
65      protected DeleteWarmerRequest newRequest() {
66          return new DeleteWarmerRequest();
67      }
68  
69      @Override
70      protected DeleteWarmerResponse newResponse() {
71          return new DeleteWarmerResponse();
72      }
73  
74      @Override
75      protected ClusterBlockException checkBlock(DeleteWarmerRequest request, ClusterState state) {
76          return state.blocks().indicesBlockedException(ClusterBlockLevel.METADATA, clusterService.state().metaData().concreteIndices(request.indicesOptions(), request.indices()));
77      }
78  
79      @Override
80      protected void masterOperation(final DeleteWarmerRequest request, final ClusterState state, final ActionListener<DeleteWarmerResponse> listener) throws ElasticsearchException {
81          final String[] concreteIndices = clusterService.state().metaData().concreteIndices(request.indicesOptions(), request.indices());
82          clusterService.submitStateUpdateTask("delete_warmer [" + Arrays.toString(request.names()) + "]", new AckedClusterStateUpdateTask<DeleteWarmerResponse>(request, listener) {
83  
84              @Override
85              protected DeleteWarmerResponse newResponse(boolean acknowledged) {
86                  return new DeleteWarmerResponse(acknowledged);
87              }
88  
89              @Override
90              public void onFailure(String source, Throwable t) {
91                  logger.debug("failed to delete warmer [{}] on indices [{}]", t, Arrays.toString(request.names()), concreteIndices);
92                  super.onFailure(source, t);
93              }
94  
95              @Override
96              public ClusterState execute(ClusterState currentState) {
97                  MetaData.Builder mdBuilder = MetaData.builder(currentState.metaData());
98  
99                  boolean globalFoundAtLeastOne = false;
100                 for (String index : concreteIndices) {
101                     IndexMetaData indexMetaData = currentState.metaData().index(index);
102                     if (indexMetaData == null) {
103                         throw new IndexMissingException(new Index(index));
104                     }
105                     IndexWarmersMetaData warmers = indexMetaData.custom(IndexWarmersMetaData.TYPE);
106                     if (warmers != null) {
107                         List<IndexWarmersMetaData.Entry> entries = Lists.newArrayList();
108                         for (IndexWarmersMetaData.Entry entry : warmers.entries()) {
109                             boolean keepWarmer = true;
110                             for (String warmer : request.names()) {
111                                 if (Regex.simpleMatch(warmer, entry.name()) || warmer.equals("_all")) {
112                                     globalFoundAtLeastOne = true;
113                                     keepWarmer =  false;
114                                     // don't add it...
115                                     break;
116                                 } 
117                             }
118                             if (keepWarmer) {
119                                 entries.add(entry);
120                             }
121                         }
122                         // a change, update it...
123                         if (entries.size() != warmers.entries().size()) {
124                             warmers = new IndexWarmersMetaData(entries.toArray(new IndexWarmersMetaData.Entry[entries.size()]));
125                             IndexMetaData.Builder indexBuilder = IndexMetaData.builder(indexMetaData).putCustom(IndexWarmersMetaData.TYPE, warmers);
126                             mdBuilder.put(indexBuilder);
127                         }
128                     }
129                 }
130 
131                 if (!globalFoundAtLeastOne) {
132                     throw new IndexWarmerMissingException(request.names());
133                 }
134 
135                 if (logger.isInfoEnabled()) {
136                     for (String index : concreteIndices) {
137                         IndexMetaData indexMetaData = currentState.metaData().index(index);
138                         if (indexMetaData == null) {
139                             throw new IndexMissingException(new Index(index));
140                         }
141                         IndexWarmersMetaData warmers = indexMetaData.custom(IndexWarmersMetaData.TYPE);
142                         if (warmers != null) {
143                             for (IndexWarmersMetaData.Entry entry : warmers.entries()) {
144                                 for (String warmer : request.names()) {
145                                     if (Regex.simpleMatch(warmer, entry.name()) || warmer.equals("_all")) {
146                                         logger.info("[{}] delete warmer [{}]", index, entry.name());
147                                     }
148                                 }
149                             }
150                         }
151                     }
152                 }
153 
154                 return ClusterState.builder(currentState).metaData(mdBuilder).build();
155             }
156         });
157     }
158 }
