20  package org.elasticsearch.index.fielddata;
21  
22  import com.carrotsearch.hppc.ObjectLongOpenHashMap;
23  import org.elasticsearch.common.Nullable;
24  import org.elasticsearch.common.io.stream.StreamInput;
25  import org.elasticsearch.common.io.stream.StreamOutput;
26  import org.elasticsearch.common.io.stream.Streamable;
27  import org.elasticsearch.common.unit.ByteSizeValue;
28  import org.elasticsearch.common.xcontent.ToXContent;
29  import org.elasticsearch.common.xcontent.XContentBuilder;
30  import org.elasticsearch.common.xcontent.XContentBuilderString;
31  
32  import java.io.IOException;
33  
34  /**
35   */
36  public class FieldDataStats implements Streamable, ToXContent {
37  
38      long memorySize;
39      long evictions;
40      @Nullable
41      ObjectLongOpenHashMap<String> fields;
42  
43      public FieldDataStats() {
44  
45      }
46  
47      public FieldDataStats(long memorySize, long evictions, @Nullable ObjectLongOpenHashMap<String> fields) {
48          this.memorySize = memorySize;
49          this.evictions = evictions;
50          this.fields = fields;
51      }
52  
53      public void add(FieldDataStats stats) {
54          this.memorySize += stats.memorySize;
55          this.evictions += stats.evictions;
56          if (stats.fields != null) {
57              if (fields == null) fields = new ObjectLongOpenHashMap<>();
58              final boolean[] states = stats.fields.allocated;
59              final Object[] keys = stats.fields.keys;
60              final long[] values = stats.fields.values;
61              for (int i = 0; i < states.length; i++) {
62                  if (states[i]) {
63                      fields.addTo((String) keys[i], values[i]);
64                  }
65              }
66          }
67      }
68  
69      public long getMemorySizeInBytes() {
70          return this.memorySize;
71      }
72  
73      public ByteSizeValue getMemorySize() {
74          return new ByteSizeValue(memorySize);
75      }
76  
77      public long getEvictions() {
78          return this.evictions;
79      }
80  
81      @Nullable
82      public ObjectLongOpenHashMap<String> getFields() {
83          return fields;
84      }
85  
86      public static FieldDataStats readFieldDataStats(StreamInput in) throws IOException {
87          FieldDataStats stats = new FieldDataStats();
88          stats.readFrom(in);
89          return stats;
90      }
91  
92      @Override
93      public void readFrom(StreamInput in) throws IOException {
94          memorySize = in.readVLong();
95          evictions = in.readVLong();
96          if (in.readBoolean()) {
97              int size = in.readVInt();
98              fields = new ObjectLongOpenHashMap<>(size);
99              for (int i = 0; i < size; i++) {
100                 fields.put(in.readString(), in.readVLong());
101             }
102         }
103     }
104 
105     @Override
106     public void writeTo(StreamOutput out) throws IOException {
107         out.writeVLong(memorySize);
108         out.writeVLong(evictions);
109         if (fields == null) {
110             out.writeBoolean(false);
111         } else {
112             out.writeBoolean(true);
113             out.writeVInt(fields.size());
114             final boolean[] states = fields.allocated;
115             final Object[] keys = fields.keys;
116             final long[] values = fields.values;
117             for (int i = 0; i < states.length; i++) {
118                 if (states[i]) {
119                     out.writeString((String) keys[i]);
120                     out.writeVLong(values[i]);
121                 }
122             }
123         }
124     }
125 
126     @Override
127     public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
128         builder.startObject(Fields.FIELDDATA);
129         builder.byteSizeField(Fields.MEMORY_SIZE_IN_BYTES, Fields.MEMORY_SIZE, memorySize);
130         builder.field(Fields.EVICTIONS, getEvictions());
131         if (fields != null) {
132             builder.startObject(Fields.FIELDS);
133             final boolean[] states = fields.allocated;
134             final Object[] keys = fields.keys;
135             final long[] values = fields.values;
136             for (int i = 0; i < states.length; i++) {
137                 if (states[i]) {
138                     builder.startObject((String) keys[i], XContentBuilder.FieldCaseConversion.NONE);
139                     builder.byteSizeField(Fields.MEMORY_SIZE_IN_BYTES, Fields.MEMORY_SIZE, values[i]);
140                     builder.endObject();
141                 }
142             }
143             builder.endObject();
144         }
145         builder.endObject();
146         return builder;
147     }
148 
149     static final class Fields {
150         static final XContentBuilderString FIELDDATA = new XContentBuilderString("fielddata");
151         static final XContentBuilderString MEMORY_SIZE = new XContentBuilderString("memory_size");
152         static final XContentBuilderString MEMORY_SIZE_IN_BYTES = new XContentBuilderString("memory_size_in_bytes");
153         static final XContentBuilderString EVICTIONS = new XContentBuilderString("evictions");
154         static final XContentBuilderString FIELDS = new XContentBuilderString("fields");
155     }
156 }
