19  package org.elasticsearch.search.aggregations.bucket.significant;
20  
21  import org.elasticsearch.common.inject.Inject;
22  import org.elasticsearch.common.xcontent.XContentParser;
23  import org.elasticsearch.search.aggregations.Aggregator;
24  import org.elasticsearch.search.aggregations.AggregatorFactory;
25  import org.elasticsearch.search.aggregations.bucket.BucketUtils;
26  import org.elasticsearch.search.aggregations.bucket.significant.heuristics.JLHScore;
27  import org.elasticsearch.search.aggregations.bucket.significant.heuristics.SignificanceHeuristic;
28  import org.elasticsearch.search.aggregations.bucket.significant.heuristics.SignificanceHeuristicParserMapper;
29  import org.elasticsearch.search.aggregations.bucket.terms.TermsAggregator;
30  import org.elasticsearch.search.aggregations.bucket.terms.support.IncludeExclude;
31  import org.elasticsearch.search.aggregations.support.ValuesSourceParser;
32  import org.elasticsearch.search.internal.SearchContext;
33  
34  import java.io.IOException;
35  
36  /**
37   *
38   */
39  public class SignificantTermsParser implements Aggregator.Parser {
40  
41      private final SignificanceHeuristicParserMapper significanceHeuristicParserMapper;
42  
43      @Inject
44      public SignificantTermsParser(SignificanceHeuristicParserMapper significanceHeuristicParserMapper) {
45          this.significanceHeuristicParserMapper = significanceHeuristicParserMapper;
46      }
47  
48      @Override
49      public String type() {
50          return SignificantStringTerms.TYPE.name();
51      }
52  
53      @Override
54      public AggregatorFactory parse(String aggregationName, XContentParser parser, SearchContext context) throws IOException {
55          SignificantTermsParametersParser aggParser = new SignificantTermsParametersParser(significanceHeuristicParserMapper);
56          ValuesSourceParser vsParser = ValuesSourceParser.any(aggregationName, SignificantStringTerms.TYPE, context)
57                  .scriptable(false)
58                  .formattable(true)
59                  .build();
60          IncludeExclude.Parser incExcParser = new IncludeExclude.Parser(aggregationName, SignificantStringTerms.TYPE, context);
61          aggParser.parse(aggregationName, parser, context, vsParser, incExcParser);
62  
63          TermsAggregator.BucketCountThresholds bucketCountThresholds = aggParser.getBucketCountThresholds();
64          if (bucketCountThresholds.getShardSize() == aggParser.getDefaultBucketCountThresholds().getShardSize()) {
65              //The user has not made a shardSize selection .
66              //Use default heuristic to avoid any wrong-ranking caused by distributed counting
67              //but request double the usual amount.
68              //We typically need more than the number of "top" terms requested by other aggregations
69              //as the significance algorithm is in less of a position to down-select at shard-level -
70              //some of the things we want to find have only one occurrence on each shard and as
71              // such are impossible to differentiate from non-significant terms at that early stage.
72              bucketCountThresholds.setShardSize(2 * BucketUtils.suggestShardSideQueueSize(bucketCountThresholds.getRequiredSize(), context.numberOfShards()));
73          }
74  
75          bucketCountThresholds.ensureValidity();
76          SignificanceHeuristic significanceHeuristic = aggParser.getSignificanceHeuristic();
77          if (significanceHeuristic == null) {
78              significanceHeuristic = JLHScore.INSTANCE;
79          }
80          return new SignificantTermsAggregatorFactory(aggregationName, vsParser.config(), bucketCountThresholds, aggParser.getIncludeExclude(), aggParser.getExecutionHint(), aggParser.getFilter(), significanceHeuristic);
81      }
82  }
