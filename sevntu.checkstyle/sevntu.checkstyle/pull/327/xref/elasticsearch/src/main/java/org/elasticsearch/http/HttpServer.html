45  public class HttpServer extends AbstractLifecycleComponent<HttpServer> {
46  
47      private final Environment environment;
48  
49      private final HttpServerTransport transport;
50  
51      private final RestController restController;
52  
53      private final NodeService nodeService;
54  
55      private final boolean disableSites;
56  
57      private final PluginSiteFilter pluginSiteFilter = new PluginSiteFilter();
58  
59      @Inject
60      public HttpServer(Settings settings, Environment environment, HttpServerTransport transport,
61                        RestController restController,
62                        NodeService nodeService) {
63          super(settings);
64          this.environment = environment;
65          this.transport = transport;
66          this.restController = restController;
67          this.nodeService = nodeService;
68          nodeService.setHttpServer(this);
69  
70          this.disableSites = this.settings.getAsBoolean("http.disable_sites", false);
71  
72          transport.httpServerAdapter(new Dispatcher(this));
73      }
74  
75      static class Dispatcher implements HttpServerAdapter {
76  
77          private final HttpServer server;
78  
79          Dispatcher(HttpServer server) {
80              this.server = server;
81          }
82  
83          @Override
84          public void dispatchRequest(HttpRequest request, HttpChannel channel) {
85              server.internalDispatchRequest(request, channel);
86          }
87      }
88  
89      @Override
90      protected void doStart() throws ElasticsearchException {
91          transport.start();
92          if (logger.isInfoEnabled()) {
93              logger.info("{}", transport.boundAddress());
94          }
95          nodeService.putAttribute("http_address", transport.boundAddress().publishAddress().toString());
96      }
97  
98      @Override
99      protected void doStop() throws ElasticsearchException {
100         nodeService.removeAttribute("http_address");
101         transport.stop();
102     }
103 
104     @Override
105     protected void doClose() throws ElasticsearchException {
106         transport.close();
107     }
108 
109     public HttpInfo info() {
110         return transport.info();
111     }
112 
113     public HttpStats stats() {
114         return transport.stats();
115     }

117     public void internalDispatchRequest(final HttpRequest request, final HttpChannel channel) {
118         if (request.rawPath().startsWith("/_plugin/")) {
119             RestFilterChain filterChain = restController.filterChain(pluginSiteFilter);
120             filterChain.continueProcessing(request, channel);
121             return;
122         }
123         restController.dispatchRequest(request, channel);
124     }

127     class PluginSiteFilter extends RestFilter {
128 
129         @Override
130         public void process(RestRequest request, RestChannel channel, RestFilterChain filterChain) throws IOException {
131             handlePluginSite((HttpRequest) request, (HttpChannel) channel);
132         }
133     }
134 
135     void handlePluginSite(HttpRequest request, HttpChannel channel) throws IOException {
136         if (disableSites) {
137             channel.sendResponse(new BytesRestResponse(FORBIDDEN));
138             return;
139         }
140         if (request.method() == RestRequest.Method.OPTIONS) {
141             // when we have OPTIONS request, simply send OK by default (with the Access Control Origin header which gets automatically added)
142             channel.sendResponse(new BytesRestResponse(OK));
143             return;
144         }
145         if (request.method() != RestRequest.Method.GET) {
146             channel.sendResponse(new BytesRestResponse(FORBIDDEN));
147             return;
148         }
149         // TODO for a "/_plugin" endpoint, we should have a page that lists all the plugins?
150 
151         String path = request.rawPath().substring("/_plugin/".length());
152         int i1 = path.indexOf('/');
153         String pluginName;
154         String sitePath;
155         if (i1 == -1) {
156             pluginName = path;
157             sitePath = null;
158             // If a trailing / is missing, we redirect to the right page #2654
159             String redirectUrl = request.rawPath() + "/";
160             BytesRestResponse restResponse = new BytesRestResponse(RestStatus.MOVED_PERMANENTLY, "text/html", "<head><meta http-equiv=\"refresh\" content=\"0; URL=" + redirectUrl + "></head>");
161             restResponse.addHeader("Location", redirectUrl);
162             channel.sendResponse(restResponse);
163             return;
164         } else {
165             pluginName = path.substring(0, i1);
166             sitePath = path.substring(i1 + 1);
167         }
168 
169         if (sitePath.length() == 0) {
170             sitePath = "/index.html";
171         }
172         final Path siteFile = environment.pluginsFile().resolve(pluginName).resolve("_site");
173 
174         final String separator = siteFile.getFileSystem().getSeparator();
175         // Convert file separators.
176         sitePath = sitePath.replace("/", separator);
177         // this is a plugin provided site, serve it as static files from the plugin location
178         Path file = FileSystemUtils.append(siteFile, Paths.get(sitePath), 0);
179         if (!Files.exists(file) || Files.isHidden(file)) {
180             channel.sendResponse(new BytesRestResponse(NOT_FOUND));
181             return;
182         }
183         
184         BasicFileAttributes attributes = Files.readAttributes(file, BasicFileAttributes.class);
185         if (!attributes.isRegularFile()) {
186             // If it's not a dir, we send a 403
187             if (!attributes.isDirectory()) {
188                 channel.sendResponse(new BytesRestResponse(FORBIDDEN));
189                 return;
190             }
191             // We don't serve dir but if index.html exists in dir we should serve it
192             file = file.resolve("index.html");
193             if (!Files.exists(file) || Files.isHidden(file) || !Files.isRegularFile(file)) {
194                 channel.sendResponse(new BytesRestResponse(FORBIDDEN));
195                 return;
196             }
197         }
198         if (!file.toAbsolutePath().startsWith(siteFile)) {
199             channel.sendResponse(new BytesRestResponse(FORBIDDEN));
200             return;
201         }
202         try {
203             byte[] data = Files.readAllBytes(file);
204             channel.sendResponse(new BytesRestResponse(OK, guessMimeType(sitePath), data));
205         } catch (IOException e) {
206             channel.sendResponse(new BytesRestResponse(INTERNAL_SERVER_ERROR));
207         }
208     }

211     // TODO: Don't respond with a mime type that violates the request's Accept header
212     private String guessMimeType(String path) {
213         int lastDot = path.lastIndexOf('.');
214         if (lastDot == -1) {
215             return "";
216         }
217         String extension = path.substring(lastDot + 1).toLowerCase(Locale.ROOT);
218         String mimeType = DEFAULT_MIME_TYPES.get(extension);
219         if (mimeType == null) {
220             return "";
221         }
222         return mimeType;
223     }

250     }
