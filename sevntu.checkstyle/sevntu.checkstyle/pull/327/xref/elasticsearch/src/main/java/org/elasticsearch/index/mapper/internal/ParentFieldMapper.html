122     public static class TypeParser implements Mapper.TypeParser {
123         @Override
124         public Mapper.Builder parse(String name, Map<String, Object> node, ParserContext parserContext) throws MapperParsingException {
125             ParentFieldMapper.Builder builder = parent();
126             for (Iterator<Map.Entry<String, Object>> iterator = node.entrySet().iterator(); iterator.hasNext();) {
127                 Map.Entry<String, Object> entry = iterator.next();
128                 String fieldName = Strings.toUnderscoreCase(entry.getKey());
129                 Object fieldNode = entry.getValue();
130                 if (fieldName.equals("type")) {
131                     builder.type(fieldNode.toString());
132                     iterator.remove();
133                 } else if (fieldName.equals("postings_format") && parserContext.indexVersionCreated().before(Version.V_2_0_0)) {
134                     // ignore before 2.0, reject on and after 2.0
135                     iterator.remove();
136                 } else if (fieldName.equals("fielddata")) {
137                     // Only take over `loading`, since that is the only option now that is configurable:
138                     Map<String, String> fieldDataSettings = SettingsLoader.Helper.loadNestedFromMap(nodeMapValue(fieldNode, "fielddata"));
139                     if (fieldDataSettings.containsKey(Loading.KEY)) {
140                         Settings settings = settingsBuilder().put(Loading.KEY, fieldDataSettings.get(Loading.KEY)).build();
141                         builder.fieldDataSettings(settings);
142                     }
143                     iterator.remove();
144                 }
145             }
146             return builder;
147         }

198     @Override
199     protected void parseCreateField(ParseContext context, List<Field> fields) throws IOException {
200         if (!active()) {
201             return;
202         }
203 
204         if (context.parser().currentName() != null && context.parser().currentName().equals(Defaults.NAME)) {
205             // we are in the parsing of _parent phase
206             String parentId = context.parser().text();
207             context.sourceToParse().parent(parentId);
208             fields.add(new Field(names.indexName(), Uid.createUid(context.stringBuilder(), type, parentId), fieldType));
209         } else {
210             // otherwise, we are running it post processing of the xcontent
211             String parsedParentId = context.doc().get(Defaults.NAME);
212             if (context.sourceToParse().parent() != null) {
213                 String parentId = context.sourceToParse().parent();
214                 if (parsedParentId == null) {
215                     if (parentId == null) {
216                         throw new MapperParsingException("No parent id provided, not within the document, and not externally");
217                     }
218                     // we did not add it in the parsing phase, add it now
219                     fields.add(new Field(names.indexName(), Uid.createUid(context.stringBuilder(), type, parentId), fieldType));
220                 } else if (parentId != null && !parsedParentId.equals(Uid.createUid(context.stringBuilder(), type, parentId))) {
221                     throw new MapperParsingException("Parent id mismatch, document value is [" + Uid.createUid(parsedParentId).id() + "], while external value is [" + parentId + "]");
222                 }
223             }
224         }
225         // we have parent mapping, yet no value was set, ignore it...
226     }

236     @Override
237     public Object valueForSearch(Object value) {
238         if (value == null) {
239             return null;
240         }
241         String sValue = value.toString();
242         if (sValue == null) {
243             return null;
244         }
245         int index = sValue.indexOf(Uid.DELIMITER);
246         if (index == -1) {
247             return sValue;
248         }
249         return sValue.substring(index + 1);
250     }

252     @Override
253     public BytesRef indexedValueForSearch(Object value) {
254         if (value instanceof BytesRef) {
255             BytesRef bytesRef = (BytesRef) value;
256             if (Uid.hasDelimiter(bytesRef)) {
257                 return bytesRef;
258             }
259             return Uid.createUidAsBytes(typeAsBytes, bytesRef);
260         }
261         String sValue = value.toString();
262         if (sValue.indexOf(Uid.DELIMITER) == -1) {
263             return Uid.createUidAsBytes(type, sValue);
264         }
265         return super.indexedValueForSearch(value);
266     }

276     @Override
277     public Filter termFilter(Object value, @Nullable QueryParseContext context) {
278         if (context == null) {
279             return super.termFilter(value, context);
280         }
281         BytesRef bValue = BytesRefs.toBytesRef(value);
282         if (Uid.hasDelimiter(bValue)) {
283             return new TermFilter(new Term(names.indexName(), bValue));
284         }
285 
286         List<String> types = new ArrayList<>(context.mapperService().types().size());
287         for (DocumentMapper documentMapper : context.mapperService().docMappers(false)) {
288             if (!documentMapper.parentFieldMapper().active()) {
289                 types.add(documentMapper.type());
290             }
291         }
292 
293         if (types.isEmpty()) {
294             return Queries.MATCH_NO_FILTER;
295         } else if (types.size() == 1) {
296             return new TermFilter(new Term(names.indexName(), Uid.createUidAsBytes(types.get(0), bValue)));
297         } else {
298             // we use all non child types, cause we don't know if its exact or not...
299             List<BytesRef> typesValues = new ArrayList<>(types.size());
300             for (String type : context.mapperService().types()) {
301                 typesValues.add(Uid.createUidAsBytes(type, bValue));
302             }
303             return new TermsFilter(names.indexName(), typesValues);
304         }
305     }
306 
307     @Override
308     public Filter termsFilter(List values, @Nullable QueryParseContext context) {
309         if (context == null) {
310             return super.termsFilter(values, context);
311         }
312         // This will not be invoked if values is empty, so don't check for empty
313         if (values.size() == 1) {
314             return termFilter(values.get(0), context);
315         }
316 
317         List<String> types = new ArrayList<>(context.mapperService().types().size());
318         for (DocumentMapper documentMapper : context.mapperService().docMappers(false)) {
319             if (!documentMapper.parentFieldMapper().active()) {
320                 types.add(documentMapper.type());
321             }
322         }
323 
324         List<BytesRef> bValues = new ArrayList<>(values.size());
325         for (Object value : values) {
326             BytesRef bValue = BytesRefs.toBytesRef(value);
327             if (Uid.hasDelimiter(bValue)) {
328                 bValues.add(bValue);
329             } else {
330                 // we use all non child types, cause we don't know if its exact or not...
331                 for (String type : types) {
332                     bValues.add(Uid.createUidAsBytes(type, bValue));
333                 }
334             }
335         }
336         return new TermsFilter(names.indexName(), bValues);
337     }

352     @Override
353     public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
354         if (!active()) {
355             return builder;
356         }
357         boolean includeDefaults = params.paramAsBoolean("include_defaults", false);
358 
359         builder.startObject(CONTENT_TYPE);
360         builder.field("type", type);
361         if (customFieldDataSettings != null) {
362             builder.field("fielddata", (Map) customFieldDataSettings.getAsMap());
363         } else if (includeDefaults) {
364             builder.field("fielddata", (Map) fieldDataType.getSettings().getAsMap());
365         }
366         builder.endObject();
367         return builder;
368     }

370     @Override
371     public void merge(Mapper mergeWith, MergeContext mergeContext) throws MergeMappingException {
372         ParentFieldMapper other = (ParentFieldMapper) mergeWith;
373         if (!Objects.equal(type, other.type)) {
374             mergeContext.addConflict("The _parent field's type option can't be changed");
375         }
376 
377         if (!mergeContext.mergeFlags().simulate()) {
378             ParentFieldMapper fieldMergeWith = (ParentFieldMapper) mergeWith;
379             if (fieldMergeWith.customFieldDataSettings != null) {
380                 if (!Objects.equal(fieldMergeWith.customFieldDataSettings, this.customFieldDataSettings)) {
381                     this.customFieldDataSettings = fieldMergeWith.customFieldDataSettings;
382                     this.fieldDataType = new FieldDataType(defaultFieldDataType().getType(),
383                             builder().put(defaultFieldDataType().getSettings()).put(this.customFieldDataSettings)
384                     );
385                 }
386             }
387         }
388     }
