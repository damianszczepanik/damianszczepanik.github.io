134      * {@inheritDoc}
135      */
136     @Override
137     public void snapshot(SnapshotId snapshotId, ShardId shardId, SnapshotIndexCommit snapshotIndexCommit, IndexShardSnapshotStatus snapshotStatus) {
138         SnapshotContext snapshotContext = new SnapshotContext(snapshotId, shardId, snapshotStatus);
139         snapshotStatus.startTime(System.currentTimeMillis());
140 
141         try {
142             snapshotContext.snapshot(snapshotIndexCommit);
143             snapshotStatus.time(System.currentTimeMillis() - snapshotStatus.startTime());
144             snapshotStatus.updateStage(IndexShardSnapshotStatus.Stage.DONE);
145         } catch (Throwable e) {
146             snapshotStatus.time(System.currentTimeMillis() - snapshotStatus.startTime());
147             snapshotStatus.updateStage(IndexShardSnapshotStatus.Stage.FAILURE);
148             snapshotStatus.failure(ExceptionsHelper.detailedMessage(e));
149             if (e instanceof IndexShardSnapshotFailedException) {
150                 throw (IndexShardSnapshotFailedException) e;
151             } else {
152                 throw new IndexShardSnapshotFailedException(shardId, e.getMessage(), e);
153             }
154         }
155     }

171      * {@inheritDoc}

173     @Override
174     public IndexShardSnapshotStatus snapshotStatus(SnapshotId snapshotId, ShardId shardId) {
175         Context context = new Context(snapshotId, shardId);
176         BlobStoreIndexShardSnapshot snapshot = context.loadSnapshot();
177         IndexShardSnapshotStatus status = new IndexShardSnapshotStatus();
178         status.updateStage(IndexShardSnapshotStatus.Stage.DONE);
179         status.startTime(snapshot.startTime());
180         status.files(snapshot.numberOfFiles(), snapshot.totalSize());
181         // The snapshot is done which means the number of processed files is the same as total
182         status.processedFiles(snapshot.numberOfFiles(), snapshot.totalSize());
183         status.time(snapshot.time());
184         return status;
185     }

187     @Override
188     public void verify(String seed) {
189         BlobContainer testBlobContainer = blobStore.blobContainer(basePath);;
190         DiscoveryNode localNode = clusterService.localNode();
191         if (testBlobContainer.blobExists(testBlobPrefix(seed) + "-master")) {
192             try (OutputStream outputStream = testBlobContainer.createOutput(testBlobPrefix(seed) + "-" + localNode.getId())) {
193                 outputStream.write(Strings.toUTF8Bytes(seed));
194             } catch (IOException exp) {
195                 throw new RepositoryVerificationException(repositoryName, "store location [" + blobStore + "] is not accessible on the node [" + localNode + "]", exp);
196             }
197         } else {
198             throw new RepositoryVerificationException(repositoryName, "store location [" + blobStore + "] is not shared between node [" + localNode + "] and the master node");
199         }

279         /**
280          * Delete shard snapshot
281          */
282         public void delete() {
283             final ImmutableMap<String, BlobMetaData> blobs;
284             try {
285                 blobs = blobContainer.listBlobs();
286             } catch (IOException e) {
287                 throw new IndexShardSnapshotException(shardId, "Failed to list content of gateway", e);
288             }
289 
290             BlobStoreIndexShardSnapshots snapshots = buildBlobStoreIndexShardSnapshots(blobs);
291 
292             String commitPointName = snapshotBlobName(snapshotId);
293 
294             try {
295                 blobContainer.deleteBlob(commitPointName);
296             } catch (IOException e) {
297                 logger.debug("[{}] [{}] failed to delete shard snapshot file", shardId, snapshotId);
298             }
299 
300             // delete all files that are not referenced by any commit point
301             // build a new BlobStoreIndexShardSnapshot, that includes this one and all the saved ones
302             List<BlobStoreIndexShardSnapshot> newSnapshotsList = Lists.newArrayList();
303             for (BlobStoreIndexShardSnapshot point : snapshots) {
304                 if (!point.snapshot().equals(snapshotId.getSnapshot())) {
305                     newSnapshotsList.add(point);
306                 }
307             }
308             cleanup(newSnapshotsList, blobs);
309         }
310 
311         /**
312          * Loads information about shard snapshot
313          */
314         public BlobStoreIndexShardSnapshot loadSnapshot() {
315             BlobStoreIndexShardSnapshot snapshot;
316             try (InputStream stream = blobContainer.openInput(snapshotBlobName(snapshotId))) {
317                     snapshot = readSnapshot(stream);
318             } catch (IOException ex) {
319                 throw new IndexShardRestoreFailedException(shardId, "failed to read shard snapshot file", ex);
320             }
321             return snapshot;
322         }
323 
324         /**
325          * Removes all unreferenced files from the repository
326          *
327          * @param snapshots list of active snapshots in the container
328          * @param blobs     list of blobs in the container
329          */
330         protected void cleanup(List<BlobStoreIndexShardSnapshot> snapshots, ImmutableMap<String, BlobMetaData> blobs) {
331             BlobStoreIndexShardSnapshots newSnapshots = new BlobStoreIndexShardSnapshots(snapshots);
332             // now go over all the blobs, and if they don't exists in a snapshot, delete them
333             for (String blobName : blobs.keySet()) {
334                 if (!blobName.startsWith("__")) {
335                     continue;
336                 }
337                 if (newSnapshots.findNameFile(FileInfo.canonicalName(blobName)) == null) {
338                     try {
339                         blobContainer.deleteBlob(blobName);
340                     } catch (IOException e) {
341                         logger.debug("[{}] [{}] error deleting blob [{}] during cleanup", e, snapshotId, shardId, blobName);
342                     }
343                 }
344             }
345         }

347         /**
348          * Generates blob name
349          *
350          * @param generation the blob number
351          * @return the blob name
352          */
353         protected String fileNameFromGeneration(long generation) {
354             return "__" + Long.toString(generation, Character.MAX_RADIX);
355         }
356 
357         /**
358          * Finds the next available blob number
359          *
360          * @param blobs list of blobs in the repository
361          * @return next available blob number
362          */
363         protected long findLatestFileNameGeneration(ImmutableMap<String, BlobMetaData> blobs) {
364             long generation = -1;
365             for (String name : blobs.keySet()) {
366                 if (!name.startsWith("__")) {
367                     continue;
368                 }
369                 name = FileInfo.canonicalName(name);
370                 try {
371                     long currentGen = Long.parseLong(name.substring(2) /*__*/, Character.MAX_RADIX);
372                     if (currentGen > generation) {
373                         generation = currentGen;
374                     }
375                 } catch (NumberFormatException e) {
376                     logger.warn("file [{}] does not conform to the '__' schema");
377                 }
378             }
379             return generation;
380         }
381 
382         /**
383          * Loads all available snapshots in the repository
384          *
385          * @param blobs list of blobs in repository
386          * @return BlobStoreIndexShardSnapshots
387          */
388         protected BlobStoreIndexShardSnapshots buildBlobStoreIndexShardSnapshots(ImmutableMap<String, BlobMetaData> blobs) {
389             List<BlobStoreIndexShardSnapshot> snapshots = Lists.newArrayList();
390             for (String name : blobs.keySet()) {
391                 if (name.startsWith(SNAPSHOT_PREFIX)) {
392                     try (InputStream stream = blobContainer.openInput(name)) {
393                         snapshots.add(readSnapshot(stream));
394                     } catch (IOException e) {
395                         logger.warn("failed to read commit point [{}]", e, name);
396                     }
397                 }
398             }
399             return new BlobStoreIndexShardSnapshots(snapshots);
400         }

415          *
416          * @param snapshotId     snapshot id

418          * @param snapshotStatus snapshot status to report progress
419          */
420         public SnapshotContext(SnapshotId snapshotId, ShardId shardId, IndexShardSnapshotStatus snapshotStatus) {
421             super(snapshotId, shardId);
422             IndexService indexService = indicesService.indexServiceSafe(shardId.getIndex());
423             store = indexService.shardInjectorSafe(shardId.id()).getInstance(Store.class);
424             this.snapshotStatus = snapshotStatus;

428         /**
429          * Create snapshot from index commit point
430          *
431          * @param snapshotIndexCommit snapshot commit point
432          */
433         public void snapshot(SnapshotIndexCommit snapshotIndexCommit) {
434             logger.debug("[{}] [{}] snapshot to [{}] ...", shardId, snapshotId, repositoryName);
435             store.incRef();
436             try {
437                 final ImmutableMap<String, BlobMetaData> blobs;
438                 try {
439                     blobs = blobContainer.listBlobs();
440                 } catch (IOException e) {
441                     throw new IndexShardSnapshotFailedException(shardId, "failed to list blobs", e);
442                 }
443 
444                 long generation = findLatestFileNameGeneration(blobs);
445                 BlobStoreIndexShardSnapshots snapshots = buildBlobStoreIndexShardSnapshots(blobs);
446 
447                 final CopyOnWriteArrayList<Throwable> failures = new CopyOnWriteArrayList<>();
448                 final List<BlobStoreIndexShardSnapshot.FileInfo> indexCommitPointFiles = newArrayList();
449 
450                 int indexNumberOfFiles = 0;
451                 long indexTotalFilesSize = 0;
452                 ArrayList<FileInfo> filesToSnapshot = newArrayList();
453                 final Store.MetadataSnapshot metadata;
454                 // TODO apparently we don't use the MetadataSnapshot#.recoveryDiff(...) here but we should
455                 try {
456                     metadata = store.getMetadata(snapshotIndexCommit);
457                 } catch (IOException e) {
458                     throw new IndexShardSnapshotFailedException(shardId, "Failed to get store file metadata", e);
459                 }
460                 for (String fileName : snapshotIndexCommit.getFiles()) {
461                     if (snapshotStatus.aborted()) {
462                         logger.debug("[{}] [{}] Aborted on the file [{}], exiting", shardId, snapshotId, fileName);
463                         throw new IndexShardSnapshotFailedException(shardId, "Aborted");
464                     }
465                     logger.trace("[{}] [{}] Processing [{}]", shardId, snapshotId, fileName);
466                     final StoreFileMetaData md = metadata.get(fileName);
467                     boolean snapshotRequired = false;
468                     BlobStoreIndexShardSnapshot.FileInfo fileInfo = snapshots.findPhysicalIndexFile(fileName);
469                     try {
470                         // in 1.3.3 we added additional hashes for .si / segments_N files
471                         // to ensure we don't double the space in the repo since old snapshots
472                         // don't have this hash we try to read that hash from the blob store
473                         // in a bwc compatible way.
474                         maybeRecalculateMetadataHash(blobContainer, fileInfo, metadata);
475                     }  catch (Throwable e) {
476                         logger.warn("{} Can't calculate hash from blob for file [{}] [{}]", e, shardId, fileInfo.physicalName(), fileInfo.metadata());
477                     }
478                     if (fileInfo == null || !fileInfo.isSame(md) || !snapshotFileExistsInBlobs(fileInfo, blobs)) {
479                         // commit point file does not exists in any commit point, or has different length, or does not fully exists in the listed blobs
480                         snapshotRequired = true;
481                     }
482 
483                     if (snapshotRequired) {
484                         indexNumberOfFiles++;
485                         indexTotalFilesSize += md.length();
486                         // create a new FileInfo
487                         BlobStoreIndexShardSnapshot.FileInfo snapshotFileInfo = new BlobStoreIndexShardSnapshot.FileInfo(fileNameFromGeneration(++generation), md, chunkSize);
488                         indexCommitPointFiles.add(snapshotFileInfo);
489                         filesToSnapshot.add(snapshotFileInfo);
490                     } else {
491                         indexCommitPointFiles.add(fileInfo);
492                     }
493                 }
494 
495                 snapshotStatus.files(indexNumberOfFiles, indexTotalFilesSize);
496 
497                 snapshotStatus.updateStage(IndexShardSnapshotStatus.Stage.STARTED);
498 
499                 for (FileInfo snapshotFileInfo : filesToSnapshot) {
500                     try {
501                         snapshotFile(snapshotFileInfo);
502                     } catch (IOException e) {
503                         throw new IndexShardSnapshotFailedException(shardId, "Failed to perform snapshot (index files)", e);
504                     }
505                 }
506 
507                 snapshotStatus.indexVersion(snapshotIndexCommit.getGeneration());
508                 // now create and write the commit point
509                 snapshotStatus.updateStage(IndexShardSnapshotStatus.Stage.FINALIZE);
510 
511                 String snapshotBlobName = snapshotBlobName(snapshotId);
512                 BlobStoreIndexShardSnapshot snapshot = new BlobStoreIndexShardSnapshot(snapshotId.getSnapshot(),
513                         snapshotIndexCommit.getGeneration(), indexCommitPointFiles, snapshotStatus.startTime(),
514                         // snapshotStatus.startTime() is assigned on the same machine, so it's safe to use with VLong
515                         System.currentTimeMillis() - snapshotStatus.startTime(), indexNumberOfFiles, indexTotalFilesSize);
516                 //TODO: The time stored in snapshot doesn't include cleanup time.
517                 logger.trace("[{}] [{}] writing shard snapshot file", shardId, snapshotId);
518                 try (OutputStream output = blobContainer.createOutput(snapshotBlobName)) {
519                     writeSnapshot(snapshot, output);
520                 } catch (IOException e) {
521                     throw new IndexShardSnapshotFailedException(shardId, "Failed to write commit point", e);
522                 }
523 
524                 // delete all files that are not referenced by any commit point
525                 // build a new BlobStoreIndexShardSnapshot, that includes this one and all the saved ones
526                 List<BlobStoreIndexShardSnapshot> newSnapshotsList = Lists.newArrayList();
527                 newSnapshotsList.add(snapshot);
528                 for (BlobStoreIndexShardSnapshot point : snapshots) {
529                     newSnapshotsList.add(point);
530                 }
531                 cleanup(newSnapshotsList, blobs);
532                 snapshotStatus.updateStage(IndexShardSnapshotStatus.Stage.DONE);
533             } finally {
534                 store.decRef();
535             }
536         }

538         /**
539          * Snapshot individual file
540          * <p/>
541          * This is asynchronous method. Upon completion of the operation latch is getting counted down and any failures are
542          * added to the {@code failures} list
543          *
544          * @param fileInfo file to be snapshotted
545          * @throws IOException
546          */
547         private void snapshotFile(final BlobStoreIndexShardSnapshot.FileInfo fileInfo) throws IOException {
548             final String file = fileInfo.physicalName();
549             final byte[] buffer = new byte[BUFFER_SIZE];
550             try (IndexInput indexInput = store.openVerifyingInput(file, IOContext.READONCE, fileInfo.metadata())) {
551                 for (int i = 0; i < fileInfo.numberOfParts(); i++) {
552                     final InputStreamIndexInput inputStreamIndexInput = new InputStreamIndexInput(indexInput, fileInfo.partBytes());
553                     InputStream inputStream = snapshotRateLimiter == null ? inputStreamIndexInput :  new RateLimitingInputStream(inputStreamIndexInput, snapshotRateLimiter, snapshotThrottleListener);
554                     inputStream = new AbortableInputStream(inputStream, fileInfo.physicalName());
555                     try (OutputStream output = blobContainer.createOutput(fileInfo.partName(i))) {
556                         int len;
557                         while ((len = inputStream.read(buffer)) > 0) {
558                             output.write(buffer, 0, len);
559                         }
560                     }
561                 }
562                 Store.verify(indexInput);
563                 snapshotStatus.addProcessedFile(fileInfo.length());
564             } catch (Throwable t) {
565                 failStoreIfCorrupted(t);
566                 snapshotStatus.addProcessedFile(0);
567                 throw t;
568             }
569         }
570 
571         private void failStoreIfCorrupted(Throwable t) {
572             if (t instanceof CorruptIndexException || t instanceof IndexFormatTooOldException || t instanceof IndexFormatTooNewException) {
573                 try {
574                     store.markStoreCorrupted((IOException) t);
575                 } catch (IOException e) {
576                     logger.warn("store cannot be marked as corrupted", e);
577                 }
578             }
579         }

581         /**
582          * Checks if snapshot file already exists in the list of blobs
583          *
584          * @param fileInfo file to check
585          * @param blobs    list of blobs
586          * @return true if file exists in the list of blobs
587          */
588         private boolean snapshotFileExistsInBlobs(BlobStoreIndexShardSnapshot.FileInfo fileInfo, ImmutableMap<String, BlobMetaData> blobs) {
589             BlobMetaData blobMetaData = blobs.get(fileInfo.name());
590             if (blobMetaData != null) {
591                 return blobMetaData.length() == fileInfo.length();
592             } else if (blobs.containsKey(fileInfo.partName(0))) {
593                 // multi part file sum up the size and check
594                 int part = 0;
595                 long totalSize = 0;
596                 while (true) {
597                     blobMetaData = blobs.get(fileInfo.partName(part++));
598                     if (blobMetaData == null) {
599                         break;
600                     }
601                     totalSize += blobMetaData.length();
602                 }
603                 return totalSize == fileInfo.length();
604             }
605             // no file, not exact and not multipart
606             return false;
607         }

609         private class AbortableInputStream extends FilterInputStream {
610             private final String fileName;
611 
612             public AbortableInputStream(InputStream delegate, String fileName) {
613                 super(delegate);
614                 this.fileName = fileName;
615             }
616 
617             @Override
618             public int read() throws IOException {
619                 checkAborted();
620                 return in.read();
621             }
622 
623             @Override
624             public int read(byte[] b, int off, int len) throws IOException {
625                 checkAborted();
626                 return in.read(b, off, len);
627             }
628 
629             private void checkAborted() {
630                 if (snapshotStatus.aborted()) {
631                     logger.debug("[{}] [{}] Aborted on the file [{}], exiting", shardId, snapshotId, fileName);
632                     throw new IndexShardSnapshotFailedException(shardId, "Aborted");
633                 }
634             }
635         }

638     /**
639      * This is a BWC layer to ensure we update the snapshots metdata with the corresponding hashes before we compare them.
640      * The new logic for StoreFileMetaData reads the entire <tt>.si</tt> and <tt>segments.n</tt> files to strengthen the
641      * comparison of the files on a per-segment / per-commit level.
642      */
643     private static final void maybeRecalculateMetadataHash(final BlobContainer blobContainer, final FileInfo fileInfo, Store.MetadataSnapshot snapshot) throws Throwable {
644         final StoreFileMetaData metadata;
645         if (fileInfo != null && (metadata = snapshot.get(fileInfo.physicalName())) != null) {
646             if (metadata.hash().length > 0 && fileInfo.metadata().hash().length == 0) {
647                 // we have a hash - check if our repo has a hash too otherwise we have
648                 // to calculate it.
649                 // we might have multiple parts even though the file is small... make sure we read all of it.
650                 try (final InputStream stream = new PartSliceStream(blobContainer, fileInfo)) {
651                     BytesRefBuilder builder = new BytesRefBuilder();
652                     Store.MetadataSnapshot.hashFile(builder, stream, fileInfo.length());
653                     BytesRef hash = fileInfo.metadata().hash(); // reset the file infos metadata hash
654                     assert hash.length == 0;
655                     hash.bytes = builder.bytes();
656                     hash.offset = 0;
657                     hash.length = builder.length();
658                 }
659             }
660         }
661     }
662 
663     private static final class PartSliceStream extends SlicedInputStream {
664 
665         private final BlobContainer container;
666         private final FileInfo info;
667 
668         public PartSliceStream(BlobContainer container, FileInfo info) {
669             super(info.numberOfParts());
670             this.info = info;
671             this.container = container;
672         }
673 
674         @Override
675         protected InputStream openSlice(long slice) throws IOException {
676             return container.openInput(info.partName(slice));
677 
678         }
679     }

681     /**
682      * Context for restore operations
683      */
684     private class RestoreContext extends Context {
685 
686         private final Store store;
687 
688         private final RecoveryState recoveryState;
689 
690         /**
691          * Constructs new restore context
692          *
693          * @param snapshotId      snapshot id
694          * @param shardId         shard to be restored
695          * @param snapshotShardId shard in the snapshot that data should be restored from
696          * @param recoveryState   recovery state to report progress
697          */
698         public RestoreContext(SnapshotId snapshotId, ShardId shardId, ShardId snapshotShardId, RecoveryState recoveryState) {
699             super(snapshotId, shardId, snapshotShardId);
700             store = indicesService.indexServiceSafe(shardId.getIndex()).shardInjectorSafe(shardId.id()).getInstance(Store.class);
701             this.recoveryState = recoveryState;
702         }
703 
704         /**
705          * Performs restore operation
706          */
707         public void restore() throws IOException {
708             store.incRef();
709             try {
710                 logger.debug("[{}] [{}] restoring to [{}] ...", snapshotId, repositoryName, shardId);
711                 BlobStoreIndexShardSnapshot snapshot = loadSnapshot();
712                 final Store.MetadataSnapshot recoveryTargetMetadata;
713                 try {
714                     recoveryTargetMetadata = store.getMetadataOrEmpty();
715                 } catch (CorruptIndexException | IndexFormatTooOldException | IndexFormatTooNewException e) {
716                     logger.warn("{} Can't read metadata from store", e, shardId);
717                     throw new IndexShardRestoreFailedException(shardId, "Can't restore corrupted shard", e);
718                 }
719 
720                 final List<FileInfo> filesToRecover = Lists.newArrayList();
721                 final Map<String, StoreFileMetaData> snapshotMetaData = new HashMap<>();
722                 final Map<String, FileInfo> fileInfos = new HashMap<>();
723                 for (final FileInfo fileInfo : snapshot.indexFiles()) {
724                     try {
725                         // in 1.3.3 we added additional hashes for .si / segments_N files
726                         // to ensure we don't double the space in the repo since old snapshots
727                         // don't have this hash we try to read that hash from the blob store
728                         // in a bwc compatible way.
729                         maybeRecalculateMetadataHash(blobContainer, fileInfo, recoveryTargetMetadata);
730                     }  catch (Throwable e) {
731                         // if the index is broken we might not be able to read it
732                         logger.warn("{} Can't calculate hash from blog for file [{}] [{}]", e, shardId, fileInfo.physicalName(), fileInfo.metadata());
733                     }
734                     snapshotMetaData.put(fileInfo.metadata().name(), fileInfo.metadata());
735                     fileInfos.put(fileInfo.metadata().name(), fileInfo);
736                 }
737                 final Store.MetadataSnapshot sourceMetaData = new Store.MetadataSnapshot(snapshotMetaData);
738                 final Store.RecoveryDiff diff = sourceMetaData.recoveryDiff(recoveryTargetMetadata);
739                 for (StoreFileMetaData md : diff.identical) {
740                     FileInfo fileInfo = fileInfos.get(md.name());
741                     recoveryState.getIndex().addFileDetail(fileInfo.name(), fileInfo.length(), true);
742                     if (logger.isTraceEnabled()) {
743                         logger.trace("[{}] [{}] not_recovering [{}] from [{}], exists in local store and is same", shardId, snapshotId, fileInfo.physicalName(), fileInfo.name());
744                     }
745                 }
746 
747                 for (StoreFileMetaData md : Iterables.concat(diff.different, diff.missing)) {
748                     FileInfo fileInfo = fileInfos.get(md.name());
749                     filesToRecover.add(fileInfo);
750                     recoveryState.getIndex().addFileDetail(fileInfo.name(), fileInfo.length(), false);
751                     if (logger.isTraceEnabled()) {
752                         if (md == null) {
753                             logger.trace("[{}] [{}] recovering [{}] from [{}], does not exists in local store", shardId, snapshotId, fileInfo.physicalName(), fileInfo.name());
754                         } else {
755                             logger.trace("[{}] [{}] recovering [{}] from [{}], exists in local store but is different", shardId, snapshotId, fileInfo.physicalName(), fileInfo.name());
756                         }
757                     }
758                 }
759                 final RecoveryState.Index index = recoveryState.getIndex();
760                 if (filesToRecover.isEmpty()) {
761                     logger.trace("no files to recover, all exists within the local store");
762                 }
763 
764                 if (logger.isTraceEnabled()) {
765                     logger.trace("[{}] [{}] recovering_files [{}] with total_size [{}], reusing_files [{}] with reused_size [{}]", shardId, snapshotId,
766                             index.totalRecoverFiles(), new ByteSizeValue(index.totalRecoverBytes()), index.reusedFileCount(), new ByteSizeValue(index.reusedFileCount()));
767                 }
768                 try {
769                     for (final FileInfo fileToRecover : filesToRecover) {
770                         logger.trace("[{}] [{}] restoring file [{}]", shardId, snapshotId, fileToRecover.name());
771                         restoreFile(fileToRecover);
772                     }
773                 } catch (IOException ex) {
774                     throw new IndexShardRestoreFailedException(shardId, "Failed to recover index", ex);
775                 }
776                 final StoreFileMetaData restoredSegmentsFile = sourceMetaData.getSegmentsFile();
777                 if (recoveryTargetMetadata == null) {
778                     throw new IndexShardRestoreFailedException(shardId, "Snapshot has no segments file");
779                 }
780                 assert restoredSegmentsFile != null;
781                 // read the snapshot data persisted
782                 final SegmentInfos segmentCommitInfos;
783                 try {
784                     segmentCommitInfos = Lucene.pruneUnreferencedFiles(restoredSegmentsFile.name(), store.directory());
785                 } catch (IOException e) {
786                     throw new IndexShardRestoreFailedException(shardId, "Failed to fetch index version after copying it over", e);
787                 }
788                 recoveryState.getIndex().updateVersion(segmentCommitInfos.getVersion());
789             } finally {
790                 store.decRef();
791             }
792         }
793 
794         /**
795          * Restores a file
796          * This is asynchronous method. Upon completion of the operation latch is getting counted down and any failures are
797          * added to the {@code failures} list
798          *
799          * @param fileInfo file to be restored
800          */
801         private void restoreFile(final FileInfo fileInfo) throws IOException {
802             boolean success = false;
803             try (InputStream stream = new PartSliceStream(blobContainer, fileInfo)) {
804                 try (final IndexOutput indexOutput = store.createVerifyingOutput(fileInfo.physicalName(), fileInfo.metadata(), IOContext.DEFAULT)) {
805                     final byte[] buffer = new byte[BUFFER_SIZE];
806                     int length;
807                     while((length=stream.read(buffer))>0){
808                         indexOutput.writeBytes(buffer,0,length);
809                         recoveryState.getIndex().addRecoveredBytesToFile(fileInfo.name(), length);
810                         if (restoreRateLimiter != null) {
811                             rateLimiterListener.onRestorePause(restoreRateLimiter.pause(length));
812                         }
813                     }
814                     Store.verify(indexOutput);
815                     indexOutput.close();
816                     // write the checksum
817                     if (fileInfo.metadata().hasLegacyChecksum()) {
818                         Store.LegacyChecksums legacyChecksums = new Store.LegacyChecksums();
819                         legacyChecksums.add(fileInfo.metadata());
820                         legacyChecksums.write(store);
821 
822                     }
823                     store.directory().sync(Collections.singleton(fileInfo.physicalName()));
824                     success = true;
825                 } catch (CorruptIndexException | IndexFormatTooOldException | IndexFormatTooNewException ex) {
826                     try {
827                         store.markStoreCorrupted(ex);
828                     } catch (IOException e) {
829                         logger.warn("store cannot be marked as corrupted", e);
830                     }
831                     throw ex;
832                 } finally {
833                     if (success == false) {
834                         store.deleteQuiet(fileInfo.physicalName());
835                     }
836                 }
837             }
838         }
839 
840     }
