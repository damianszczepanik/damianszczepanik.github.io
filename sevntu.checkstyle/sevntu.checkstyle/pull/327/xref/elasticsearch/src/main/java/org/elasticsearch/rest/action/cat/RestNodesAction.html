60      @Override
61      public void doRequest(final RestRequest request, final RestChannel channel, final Client client) {
62          final ClusterStateRequest clusterStateRequest = new ClusterStateRequest();
63          clusterStateRequest.clear().nodes(true);
64          clusterStateRequest.local(request.paramAsBoolean("local", clusterStateRequest.local()));
65          clusterStateRequest.masterNodeTimeout(request.paramAsTime("master_timeout", clusterStateRequest.masterNodeTimeout()));
66  
67          client.admin().cluster().state(clusterStateRequest, new RestActionListener<ClusterStateResponse>(channel) {
68              @Override
69              public void processResponse(final ClusterStateResponse clusterStateResponse) {
70                  NodesInfoRequest nodesInfoRequest = new NodesInfoRequest();
71                  nodesInfoRequest.clear().jvm(true).os(true).process(true);
72                  client.admin().cluster().nodesInfo(nodesInfoRequest, new RestActionListener<NodesInfoResponse>(channel) {
73                      @Override
74                      public void processResponse(final NodesInfoResponse nodesInfoResponse) {
75                          NodesStatsRequest nodesStatsRequest = new NodesStatsRequest();
76                          nodesStatsRequest.clear().jvm(true).os(true).fs(true).indices(true).process(true);
77                          client.admin().cluster().nodesStats(nodesStatsRequest, new RestResponseListener<NodesStatsResponse>(channel) {
78                              @Override
79                              public RestResponse buildResponse(NodesStatsResponse nodesStatsResponse) throws Exception {
80                                  return RestTable.buildResponse(buildTable(request, clusterStateResponse, nodesInfoResponse, nodesStatsResponse), channel);
81                              }
82                          });
83                      }
84                  });
85              }
86          });
87      }

192     private Table buildTable(RestRequest req, ClusterStateResponse state, NodesInfoResponse nodesInfo, NodesStatsResponse nodesStats) {
193         boolean fullId = req.paramAsBoolean("full_id", false);
194 
195         DiscoveryNodes nodes = state.getState().nodes();
196         String masterId = nodes.masterNodeId();
197         Table table = getTableWithHeader(req);
198 
199         for (DiscoveryNode node : nodes) {
200             NodeInfo info = nodesInfo.getNodesMap().get(node.id());
201             NodeStats stats = nodesStats.getNodesMap().get(node.id());
202 
203             table.startRow();
204 
205             table.addCell(fullId ? node.id() : Strings.substring(node.getId(), 0, 4));
206             table.addCell(info == null ? null : info.getProcess().id());
207             table.addCell(node.getHostName());
208             table.addCell(node.getHostAddress());
209             if (node.address() instanceof InetSocketTransportAddress) {
210                 table.addCell(((InetSocketTransportAddress) node.address()).address().getPort());
211             } else {
212                 table.addCell("-");
213             }
214 
215             table.addCell(node.getVersion().number());
216             table.addCell(info == null ? null : info.getBuild().hashShort());
217             table.addCell(info == null ? null : info.getJvm().version());
218             table.addCell(stats == null ? null : stats.getFs() == null ? null : stats.getFs().total().getAvailable());
219             table.addCell(stats == null ? null : stats.getJvm().getMem().getHeapUsed());
220             table.addCell(stats == null ? null : stats.getJvm().getMem().getHeapUsedPrecent());
221             table.addCell(info == null ? null : info.getJvm().getMem().getHeapMax());
222             table.addCell(stats == null ? null : stats.getOs().mem() == null ? null : stats.getOs().mem().used());
223             table.addCell(stats == null ? null : stats.getOs().mem() == null ? null : stats.getOs().mem().usedPercent());
224             table.addCell(info == null ? null : info.getOs().mem() == null ? null : info.getOs().mem().total()); // sigar fails to load in IntelliJ
225             table.addCell(stats == null ? null : stats.getProcess().getOpenFileDescriptors());
226             table.addCell(stats == null || info == null ? null :
227                           calculatePercentage(stats.getProcess().getOpenFileDescriptors(), info.getProcess().getMaxFileDescriptors()));
228             table.addCell(info == null ? null : info.getProcess().getMaxFileDescriptors());
229 
230             table.addCell(stats == null ? null : stats.getOs() == null ? null : stats.getOs().getLoadAverage().length < 1 ? null : String.format(Locale.ROOT, "%.2f", stats.getOs().getLoadAverage()[0]));
231             table.addCell(stats == null ? null : stats.getJvm().uptime());
232             table.addCell(node.clientNode() ? "c" : node.dataNode() ? "d" : "-");
233             table.addCell(masterId == null ? "x" : masterId.equals(node.id()) ? "*" : node.masterNode() ? "m" : "-");
234             table.addCell(node.name());
235 
236             table.addCell(stats == null ? null : stats.getIndices().getCompletion().getSize());
237 
238             table.addCell(stats == null ? null : stats.getIndices().getFieldData().getMemorySize());
239             table.addCell(stats == null ? null : stats.getIndices().getFieldData().getEvictions());
240 
241             table.addCell(stats == null ? null : stats.getIndices().getFilterCache().getMemorySize());
242             table.addCell(stats == null ? null : stats.getIndices().getFilterCache().getEvictions());
243 
244             table.addCell(stats == null ? null : stats.getIndices().getQueryCache().getMemorySize());
245             table.addCell(stats == null ? null : stats.getIndices().getQueryCache().getEvictions());
246             table.addCell(stats == null ? null : stats.getIndices().getQueryCache().getHitCount());
247             table.addCell(stats == null ? null : stats.getIndices().getQueryCache().getMissCount());
248 
249             table.addCell(stats == null ? null : stats.getIndices().getFlush().getTotal());
250             table.addCell(stats == null ? null : stats.getIndices().getFlush().getTotalTime());
251 
252             table.addCell(stats == null ? null : stats.getIndices().getGet().current());
253             table.addCell(stats == null ? null : stats.getIndices().getGet().getTime());
254             table.addCell(stats == null ? null : stats.getIndices().getGet().getCount());
255             table.addCell(stats == null ? null : stats.getIndices().getGet().getExistsTime());
256             table.addCell(stats == null ? null : stats.getIndices().getGet().getExistsCount());
257             table.addCell(stats == null ? null : stats.getIndices().getGet().getMissingTime());
258             table.addCell(stats == null ? null : stats.getIndices().getGet().getMissingCount());
259 
260             table.addCell(stats == null ? null : stats.getIndices().getIdCache().getMemorySize());

311     /**
312      * Calculate the percentage of {@code used} from the {@code max} number.
313      * @param used The currently used number.
314      * @param max The maximum number.
315      * @return 0 if {@code max} is 0. Otherwise 100 * {@code used} / {@code max}.
316      */
317     private short calculatePercentage(long used, long max) {
318         return max == 0 ? 0 : (short)((100d * used) / max);
319     }
