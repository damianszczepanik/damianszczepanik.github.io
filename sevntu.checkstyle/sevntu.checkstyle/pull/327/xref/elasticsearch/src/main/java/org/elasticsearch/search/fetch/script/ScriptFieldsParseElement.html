20  package org.elasticsearch.search.fetch.script;
21  
22  import org.elasticsearch.common.xcontent.XContentParser;
23  import org.elasticsearch.script.ScriptParameterParser;
24  import org.elasticsearch.script.ScriptParameterParser.ScriptParameterValue;
25  import org.elasticsearch.script.ScriptService;
26  import org.elasticsearch.script.SearchScript;
27  import org.elasticsearch.search.SearchParseElement;
28  import org.elasticsearch.search.internal.SearchContext;
29  
30  import java.util.Map;
31  
32  /**
33   * <pre>
34   * "script_fields" : {
35   *  "test1" : {
36   *      "script" : "doc['field_name'].value"
37   *  },
38   *  "test2" : {
39   *      "script" : "..."
40   *  }
41   * }
42   * </pre>
43   *
44   *
45   */
46  public class ScriptFieldsParseElement implements SearchParseElement {
47  
48      @Override
49      public void parse(XContentParser parser, SearchContext context) throws Exception {
50          XContentParser.Token token;
51          String currentFieldName = null;
52          while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
53              if (token == XContentParser.Token.FIELD_NAME) {
54                  currentFieldName = parser.currentName();
55              } else if (token == XContentParser.Token.START_OBJECT) {
56                  String fieldName = currentFieldName;
57                  ScriptParameterParser scriptParameterParser = new ScriptParameterParser();
58                  String script = null;
59                  ScriptService.ScriptType scriptType = null;
60                  Map<String, Object> params = null;
61                  boolean ignoreException = false;
62                  while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
63                      if (token == XContentParser.Token.FIELD_NAME) {
64                          currentFieldName = parser.currentName();
65                      } else if (token == XContentParser.Token.START_OBJECT) {
66                          params = parser.map();
67                      } else if (token.isValue()) {
68                          if ("ignore_failure".equals(currentFieldName)) {
69                              ignoreException = parser.booleanValue();
70                          } else {
71                              scriptParameterParser.token(currentFieldName, token, parser);
72                          }
73                      }
74                  }
75  
76                  ScriptParameterValue scriptValue = scriptParameterParser.getDefaultScriptParameterValue();
77                  if (scriptValue != null) {
78                      script = scriptValue.script();
79                      scriptType = scriptValue.scriptType();
80                  }
81                  SearchScript searchScript = context.scriptService().search(context.lookup(), scriptParameterParser.lang(), script, scriptType, params);
82                  context.scriptFields().add(new ScriptFieldsContext.ScriptField(fieldName, searchScript, ignoreException));
83              }
84          }
85      }
86  }
