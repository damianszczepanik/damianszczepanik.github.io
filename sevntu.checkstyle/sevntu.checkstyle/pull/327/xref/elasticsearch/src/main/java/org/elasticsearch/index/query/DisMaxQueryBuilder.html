20  package org.elasticsearch.index.query;
21  
22  import org.elasticsearch.common.xcontent.XContentBuilder;
23  
24  import java.io.IOException;
25  import java.util.ArrayList;
26  
27  import static com.google.common.collect.Lists.newArrayList;
28  
29  /**
30   * A query that generates the union of documents produced by its sub-queries, and that scores each document
31   * with the maximum score for that document as produced by any sub-query, plus a tie breaking increment for any
32   * additional matching sub-queries.
33   *
34   *
35   */
36  public class DisMaxQueryBuilder extends BaseQueryBuilder implements BoostableQueryBuilder<DisMaxQueryBuilder> {
37  
38      private ArrayList<QueryBuilder> queries = newArrayList();
39  
40      private float boost = -1;
41  
42      private float tieBreaker = -1;
43  
44      private String queryName;
45  
46      /**
47       * Add a sub-query to this disjunction.
48       */
49      public DisMaxQueryBuilder add(QueryBuilder queryBuilder) {
50          queries.add(queryBuilder);
51          return this;
52      }
53  
54      /**
55       * Sets the boost for this query.  Documents matching this query will (in addition to the normal
56       * weightings) have their score multiplied by the boost provided.
57       */
58      @Override
59      public DisMaxQueryBuilder boost(float boost) {
60          this.boost = boost;
61          return this;
62      }
63  
64      /**
65       * The score of each non-maximum disjunct for a document is multiplied by this weight
66       * and added into the final score.  If non-zero, the value should be small, on the order of 0.1, which says that
67       * 10 occurrences of word in a lower-scored field that is also in a higher scored field is just as good as a unique
68       * word in the lower scored field (i.e., one that is not in any higher scored field.
69       */
70      public DisMaxQueryBuilder tieBreaker(float tieBreaker) {
71          this.tieBreaker = tieBreaker;
72          return this;
73      }
74  
75      /**
76       * Sets the query name for the filter that can be used when searching for matched_filters per hit.
77       */
78      public DisMaxQueryBuilder queryName(String queryName) {
79          this.queryName = queryName;
80          return this;
81      }
82  
83      @Override
84      protected void doXContent(XContentBuilder builder, Params params) throws IOException {
85          builder.startObject(DisMaxQueryParser.NAME);
86          if (tieBreaker != -1) {
87              builder.field("tie_breaker", tieBreaker);
88          }
89          if (boost != -1) {
90              builder.field("boost", boost);
91          }
92          if (queryName != null) {
93              builder.field("_name", queryName);
94          }
95          builder.startArray("queries");
96          for (QueryBuilder queryBuilder : queries) {
97              queryBuilder.toXContent(builder, params);
98          }
99          builder.endArray();
100         builder.endObject();
101     }
102 }
