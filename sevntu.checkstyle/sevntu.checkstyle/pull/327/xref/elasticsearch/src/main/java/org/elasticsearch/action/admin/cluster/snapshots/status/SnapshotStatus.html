20  package org.elasticsearch.action.admin.cluster.snapshots.status;
21  
22  import com.google.common.collect.ImmutableList;
23  import com.google.common.collect.ImmutableMap;
24  import org.elasticsearch.cluster.metadata.SnapshotId;
25  import org.elasticsearch.cluster.metadata.SnapshotMetaData.State;
26  import org.elasticsearch.common.io.stream.StreamInput;
27  import org.elasticsearch.common.io.stream.StreamOutput;
28  import org.elasticsearch.common.io.stream.Streamable;
29  import org.elasticsearch.common.xcontent.ToXContent;
30  import org.elasticsearch.common.xcontent.XContentBuilder;
31  import org.elasticsearch.common.xcontent.XContentBuilderString;
32  import org.elasticsearch.common.xcontent.XContentFactory;
33  
34  import java.io.IOException;
35  import java.util.List;
36  import java.util.Map;
37  import java.util.Set;
38  
39  import static com.google.common.collect.Lists.newArrayList;
40  import static com.google.common.collect.Sets.newHashSet;
41  
42  /**
43   * Status of a snapshot
44   */
45  public class SnapshotStatus implements ToXContent, Streamable {
46  
47      private SnapshotId snapshotId;
48  
49      private State state;
50  
51      private ImmutableList<SnapshotIndexShardStatus> shards;
52  
53      private ImmutableMap<String, SnapshotIndexStatus> indicesStatus;
54  
55      private SnapshotShardsStats shardsStats;
56  
57      private SnapshotStats stats;
58  
59  
60      SnapshotStatus(SnapshotId snapshotId, State state, ImmutableList<SnapshotIndexShardStatus> shards) {
61          this.snapshotId = snapshotId;
62          this.state = state;
63          this.shards = shards;
64          shardsStats = new SnapshotShardsStats(shards);
65          updateShardStats();
66      }
67  
68      SnapshotStatus() {
69      }
70  
71      /**
72       * Returns snapshot id
73       */
74      public SnapshotId getSnapshotId() {
75          return snapshotId;
76      }
77  
78      /**
79       * Returns snapshot state
80       */
81      public State getState() {
82          return state;
83      }
84  
85      /**
86       * Returns list of snapshot shards
87       */
88      public List<SnapshotIndexShardStatus> getShards() {
89          return shards;
90      }
91  
92      public SnapshotShardsStats getShardsStats() {
93          return shardsStats;
94      }
95  
96      /**
97       * Returns list of snapshot indices
98       */
99      public Map<String, SnapshotIndexStatus> getIndices() {
100         if (this.indicesStatus != null) {
101             return this.indicesStatus;
102         }
103 
104         ImmutableMap.Builder<String, SnapshotIndexStatus> indicesStatus = ImmutableMap.builder();
105 
106         Set<String> indices = newHashSet();
107         for (SnapshotIndexShardStatus shard : shards) {
108             indices.add(shard.getIndex());
109         }
110 
111         for (String index : indices) {
112             List<SnapshotIndexShardStatus> shards = newArrayList();
113             for (SnapshotIndexShardStatus shard : this.shards) {
114                 if (shard.getIndex().equals(index)) {
115                     shards.add(shard);
116                 }
117             }
118             indicesStatus.put(index, new SnapshotIndexStatus(index, shards));
119         }
120         this.indicesStatus = indicesStatus.build();
121         return this.indicesStatus;
122 
123     }
124 
125     @Override
126     public void readFrom(StreamInput in) throws IOException {
127         snapshotId = SnapshotId.readSnapshotId(in);
128         state = State.fromValue(in.readByte());
129         int size = in.readVInt();
130         ImmutableList.Builder<SnapshotIndexShardStatus> builder = ImmutableList.builder();
131         for (int i = 0; i < size; i++) {
132             builder.add(SnapshotIndexShardStatus.readShardSnapshotStatus(in));
133         }
134         shards = builder.build();
135         updateShardStats();
136     }
137 
138     @Override
139     public void writeTo(StreamOutput out) throws IOException {
140         snapshotId.writeTo(out);
141         out.writeByte(state.value());
142         out.writeVInt(shards.size());
143         for (SnapshotIndexShardStatus shard : shards) {
144             shard.writeTo(out);
145         }
146     }
147 
148     /**
149      * Reads snapshot status from stream input
150      *
151      * @param in stream input
152      * @return deserialized snapshot status
153      * @throws IOException
154      */
155     public static SnapshotStatus readSnapshotStatus(StreamInput in) throws IOException {
156         SnapshotStatus snapshotInfo = new SnapshotStatus();
157         snapshotInfo.readFrom(in);
158         return snapshotInfo;
159     }
160 
161     @Override
162     public String toString() {
163         try {
164             XContentBuilder builder = XContentFactory.jsonBuilder().prettyPrint();
165             builder.startObject();
166             toXContent(builder, EMPTY_PARAMS);
167             builder.endObject();
168             return builder.string();
169         } catch (IOException e) {
170             return "{ \"error\" : \"" + e.getMessage() + "\"}";
171         }
172     }
173 
174 
175     /**
176      * Returns number of files in the snapshot
177      */
178     public SnapshotStats getStats() {
179         return stats;
180     }
181 
182     static final class Fields {
183         static final XContentBuilderString SNAPSHOT = new XContentBuilderString("snapshot");
184         static final XContentBuilderString REPOSITORY = new XContentBuilderString("repository");
185         static final XContentBuilderString STATE = new XContentBuilderString("state");
186         static final XContentBuilderString INDICES = new XContentBuilderString("indices");
187     }
188 
189     @Override
190     public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
191         builder.startObject();
192         builder.field(Fields.SNAPSHOT, snapshotId.getSnapshot());
193         builder.field(Fields.REPOSITORY, snapshotId.getRepository());
194         builder.field(Fields.STATE, state.name());
195         shardsStats.toXContent(builder, params);
196         stats.toXContent(builder, params);
197         builder.startObject(Fields.INDICES);
198         for (SnapshotIndexStatus indexStatus : getIndices().values()) {
199             indexStatus.toXContent(builder, params);
200         }
201         builder.endObject();
202         builder.endObject();
203         return builder;
204     }
205 
206     private void updateShardStats() {
207         stats = new SnapshotStats();
208         shardsStats = new SnapshotShardsStats(shards);
209         for (SnapshotIndexShardStatus shard : shards) {
210             stats.add(shard.getStats());
211         }
212     }
213 }
