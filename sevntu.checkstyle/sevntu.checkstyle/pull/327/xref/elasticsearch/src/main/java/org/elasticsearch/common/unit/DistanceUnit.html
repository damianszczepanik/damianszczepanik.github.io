21  
22  import org.elasticsearch.ElasticsearchIllegalArgumentException;
23  import org.elasticsearch.common.geo.GeoUtils;
24  import org.elasticsearch.common.io.stream.StreamInput;
25  import org.elasticsearch.common.io.stream.StreamOutput;

27  import java.io.IOException;

29  /**
30   * The DistanceUnit enumerates several units for measuring distances. These units 
31   * provide methods for converting strings and methods to convert units among each
32   * others. Some methods like {@link DistanceUnit#getEarthCircumference} refer to
33   * the earth ellipsoid defined in {@link GeoUtils}. The default unit used within
34   * this project is <code>METERS</code> which is defined by <code>DEFAULT</code>
35   */
36  public enum DistanceUnit {
37      INCH(0.0254, "in", "inch"),
38      YARD(0.9144, "yd", "yards"),
39      FEET(0.3048, "ft", "feet"),
40      KILOMETERS(1000.0, "km", "kilometers"),
41      NAUTICALMILES(1852.0, "NM", "nmi", "nauticalmiles"),
42      MILLIMETERS(0.001, "mm", "millimeters"),
43      CENTIMETERS(0.01, "cm", "centimeters"),
44  
45      // 'm' is a suffix of 'nmi' so it must follow 'nmi'
46      MILES(1609.344, "mi", "miles"),
47  
48      // since 'm' is suffix of other unit
49      // it must be the last entry of unit
50      // names ending with 'm'. otherwise
51      // parsing would fail
52      METERS(1, "m", "meters");
53  
54      public static final DistanceUnit DEFAULT = METERS;
55  
56      private double meters; 
57      private final String[] names;
58  
59      DistanceUnit(double meters, String...names) {
60          this.meters = meters;
61          this.names = names;
62      }
63  
64      /**
65       * Measures the circumference of earth in this unit
66       * 
67       * @return length of earth circumference in this unit
68       */
69      public double getEarthCircumference() {
70          return GeoUtils.EARTH_EQUATOR / meters;
71      }
72  
73      /**
74       * Measures the radius of earth in this unit
75       * 
76       * @return length of earth radius in this unit
77       */
78      public double getEarthRadius() {
79          return GeoUtils.EARTH_SEMI_MAJOR_AXIS / meters;
80      }
81  
82      /**
83       * Measures a longitude in this unit
84       * 
85       * @return length of a longitude degree in this unit
86       */
87      public double getDistancePerDegree() {
88          return GeoUtils.EARTH_EQUATOR / (360.0 * meters);
89      }
90  
91      /**
92       * Convert a value into meters
93       * 
94       * @param distance distance in this unit
95       * @return value in meters
96       */
97      public double toMeters(double distance) {
98          return convert(distance, this, DistanceUnit.METERS);
99      }
100 
101     /**
102      * Convert a value given in meters to a value of this unit
103      * 
104      * @param distance distance in meters
105      * @return value in this unit
106      */
107     public double fromMeters(double distance) {
108         return convert(distance, DistanceUnit.METERS, this);
109     }
110 
111     /** 
112      * Convert a given value into another unit
113      * 
114      * @param distance value in this unit
115      * @param unit source unit
116      * @return value in this unit
117      */
118     public double convert(double distance, DistanceUnit unit) {
119         return convert(distance, unit, this);
120     }
121 
122     /**
123      * Convert a value to a distance string
124      * 
125      * @param distance value to convert
126      * @return String representation of the distance 
127      */
128     public String toString(double distance) {
129         return distance + toString();
130     }
131 
132     @Override
133     public String toString() {
134         return names[0];
135     }
136 
137     /**
138      * Converts the given distance from the given DistanceUnit, to the given DistanceUnit
139      *
140      * @param distance Distance to convert
141      * @param from     Unit to convert the distance from
142      * @param to       Unit of distance to convert to
143      * @return Given distance converted to the distance in the given unit
144      */
145     public static double convert(double distance, DistanceUnit from, DistanceUnit to) {
146         if (from == to) {
147             return distance;
148         } else {
149             return distance * from.meters / to.meters;
150         }
151     }
152 
153     /**
154      * Parses a given distance and converts it to the specified unit.
155      * 
156      * @param distance String defining a distance (value and unit)
157      * @param defaultUnit unit assumed if none is defined
158      * @param to unit of result
159      * @return parsed distance
160      */
161     public static double parse(String distance, DistanceUnit defaultUnit, DistanceUnit to) {
162         Distance dist = Distance.parseDistance(distance, defaultUnit);
163         return convert(dist.value, dist.unit, to);
164     }
165 
166     /**
167      * Parses a given distance and converts it to this unit.
168      * 
169      * @param distance String defining a distance (value and unit)
170      * @param defaultUnit unit to expect if none if provided
171      * @return parsed distance
172      */
173     public double parse(String distance, DistanceUnit defaultUnit) {
174         return parse(distance, defaultUnit, this);
175     }
176 
177     /**
178      * Convert a String to a {@link DistanceUnit}
179      * 
180      * @param unit name of the unit
181      * @return unit matching the given name
182      * @throws org.elasticsearch.ElasticsearchIllegalArgumentException if no unit matches the given name
183      */
184     public static DistanceUnit fromString(String unit) {
185         for (DistanceUnit dunit : values()) {
186             for (String name : dunit.names) {
187                 if(name.equals(unit)) {
188                     return dunit;
189                 }
190             }
191         }
192         throw new ElasticsearchIllegalArgumentException("No distance unit match [" + unit + "]");
193     }
194 
195     /**
196      * Parses the suffix of a given distance string and return the corresponding {@link DistanceUnit}
197      * 
198      * @param distance string representing a distance
199      * @param defaultUnit default unit to use, if no unit is provided by the string
200      * @return unit of the given distance
201      */
202     public static DistanceUnit parseUnit(String distance, DistanceUnit defaultUnit) {
203         for (DistanceUnit unit : values()) {
204             for (String name : unit.names) {
205                 if(distance.endsWith(name)) {
206                     return unit;
207                 }
208             }
209         }
210         return defaultUnit;
211     }
212 
213     /**
214      * Write a {@link DistanceUnit} to a {@link StreamOutput}
215      * 
216      * @param out {@link StreamOutput} to write to
217      * @param unit {@link DistanceUnit} to write 
218      * @throws IOException
219      */
220     public static void writeDistanceUnit(StreamOutput out, DistanceUnit unit) throws IOException {
221         out.writeByte((byte) unit.ordinal());
222     }
223 
224     /**
225      * Read a {@link DistanceUnit} from a {@link StreamInput} 
226      * 
227      * @param in {@link StreamInput} to read the {@link DistanceUnit} from
228      * @return {@link DistanceUnit} read from the {@link StreamInput}
229      * @throws IOException if no unit can be read from the {@link StreamInput}
230      * @thrown ElasticsearchIllegalArgumentException if no matching {@link DistanceUnit} can be found
231      */
232     public static DistanceUnit readDistanceUnit(StreamInput in) throws IOException {
233         byte b = in.readByte();
234 
235         if(b<0 || b>=values().length) {
236             throw new ElasticsearchIllegalArgumentException("No type for distance unit matching [" + b + "]");
237         } else {
238             return values()[b];
239         }
240     }

242     /**
243      * This class implements a value+unit tuple.
244      */
245     public static class Distance implements Comparable<Distance> {
246         public final double value;
247         public final DistanceUnit unit;
248 
249         public Distance(double value, DistanceUnit unit) {
250             super();
251             this.value = value;
252             this.unit = unit;
253         }
254 
255         /**
256          * Converts a {@link Distance} value given in a specific {@link DistanceUnit} into
257          * a value equal to the specified value but in a other {@link DistanceUnit}.
258          * 
259          * @param unit unit of the result
260          * @return converted distance
261          */
262         public Distance convert(DistanceUnit unit) {
263             if(this.unit == unit) {
264                 return this;
265             } else {
266                 return new Distance(DistanceUnit.convert(value, this.unit, unit), unit);
267             }
268         }
269 
270         @Override
271         public boolean equals(Object obj) {
272             if(obj == null) {
273                 return false;
274             } else if (obj instanceof Distance) {
275                 Distance other = (Distance) obj;
276                 return DistanceUnit.convert(value, unit, other.unit) == other.value;
277             } else {
278                 return false;
279             }
280         }
281 
282         @Override
283         public int hashCode() {
284             return Double.valueOf(value * unit.meters).hashCode();
285         }
286 
287         @Override
288         public int compareTo(Distance o) {
289             return Double.compare(value, DistanceUnit.convert(o.value, o.unit, unit));
290         }
291 
292         @Override
293         public String toString() {
294             return unit.toString(value);
295         }
296 
297         /**
298          * Parse a {@link Distance} from a given String. If no unit is given
299          * <code>DistanceUnit.DEFAULT</code> will be used 
300          * 
301          * @param distance String defining a {@link Distance} 
302          * @return parsed {@link Distance}
303          */
304         public static Distance parseDistance(String distance) {
305             return parseDistance(distance, DEFAULT);
306         }
307 
308         /**
309          * Parse a {@link Distance} from a given String
310          * 
311          * @param distance String defining a {@link Distance} 
312          * @param defaultUnit {@link DistanceUnit} to be assumed
313          *          if not unit is provided in the first argument  
314          * @return parsed {@link Distance}
315          */
316         private static Distance parseDistance(String distance, DistanceUnit defaultUnit) {
317             for (DistanceUnit unit : values()) {
318                 for (String name : unit.names) {
319                     if(distance.endsWith(name)) {
320                         return new Distance(Double.parseDouble(distance.substring(0, distance.length() - name.length())), unit);
321                     }
322                 }
323             }
324             return new Distance(Double.parseDouble(distance), defaultUnit);
325         }
326     }
