17  package org.elasticsearch.common.inject;
18  
19  /**
20   * Injects dependencies into the fields and methods on instances of type {@code T}. Ignores the
21   * presence or absence of an injectable constructor.
22   *
23   * @param <T> type to inject members of
24   * @author crazybob@google.com (Bob Lee)
25   * @author jessewilson@google.com (Jesse Wilson)
26   * @since 2.0
27   */
28  public interface MembersInjector<T> {
29  
30      /**
31       * Injects dependencies into the fields and methods of {@code instance}. Ignores the presence or
32       * absence of an injectable constructor.
33       * <p/>
34       * <p>Whenever Guice creates an instance, it performs this injection automatically (after first
35       * performing constructor injection), so if you're able to let Guice create all your objects for
36       * you, you'll never need to use this method.
37       *
38       * @param instance to inject members on. May be {@code null}.
39       */
40      void injectMembers(T instance);
41  }
