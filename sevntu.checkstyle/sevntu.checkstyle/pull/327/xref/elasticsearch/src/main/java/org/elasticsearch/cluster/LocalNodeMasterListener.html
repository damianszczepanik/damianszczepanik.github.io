19  package org.elasticsearch.cluster;
20  
21  /**
22   * Enables listening to master changes events of the local node (when the local node becomes the master, and when the local
23   * node cease being a master).
24   */
25  public interface LocalNodeMasterListener {
26  
27      /**
28       * Called when local node is elected to be the master
29       */
30      void onMaster();
31  
32      /**
33       * Called when the local node used to be the master, a new master was elected and it's no longer the local node.
34       */
35      void offMaster();
36  
37      /**
38       * The name of the executor that the implementation of the callbacks of this lister should be executed on. The thread
39       * that is responsible for managing instances of this lister is the same thread handling the cluster state events. If
40       * the work done is the callbacks above is inexpensive, this value may be {@link org.elasticsearch.threadpool.ThreadPool.Names#SAME SAME}
41       * (indicating that the callbaks will run on the same thread as the cluster state events are fired with). On the other hand,
42       * if the logic in the callbacks are heavier and take longer to process (or perhaps involve blocking due to IO operations),
43       * prefer to execute them on a separte more appropriate executor (eg. {@link org.elasticsearch.threadpool.ThreadPool.Names#GENERIC GENERIC}
44       * or {@link org.elasticsearch.threadpool.ThreadPool.Names#MANAGEMENT MANAGEMENT}).
45       *
46       * @return The name of the executor that will run the callbacks of this listener.
47       */
48      String executorName();
49  
50  }
