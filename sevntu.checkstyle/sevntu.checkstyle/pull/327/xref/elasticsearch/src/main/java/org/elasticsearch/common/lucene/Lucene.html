143     /**
144      * This method removes all files from the given directory that are not referenced by the given segments file.
145      * This method will open an IndexWriter and relies on index file deleter to remove all unreferenced files. Segment files
146      * that are newer than the given segments file are removed forcefully to prevent problems with IndexWriter opening a potentially
147      * broken commit point / leftover.
148      * <b>Note:</b> this method will fail if there is another IndexWriter open on the given directory. This method will also acquire
149      * a write lock from the directory while pruning unused files. This method expects an existing index in the given directory that has
150      * the given segments file.
151      */
152     public static SegmentInfos pruneUnreferencedFiles(String segmentsFileName, Directory directory) throws IOException {
153         final SegmentInfos si = readSegmentInfos(segmentsFileName, directory);
154         try (Lock writeLock = directory.makeLock(IndexWriter.WRITE_LOCK_NAME)) {
155             if (!writeLock.obtain(IndexWriterConfig.getDefaultWriteLockTimeout())) { // obtain write lock
156                 throw new LockObtainFailedException("Index locked for write: " + writeLock);
157             }
158             int foundSegmentFiles = 0;
159             for (final String file : directory.listAll()) {
160                 /**
161                  * we could also use a deletion policy here but in the case of snapshot and restore
162                  * sometimes we restore an index and override files that were referenced by a "future"
163                  * commit. If such a commit is opened by the IW it would likely throw a corrupted index exception
164                  * since checksums don's match anymore. that's why we prune the name here directly.
165                  * We also want the caller to know if we were not able to remove a segments_N file.
166                  */
167                 if (file.startsWith(IndexFileNames.SEGMENTS) || file.equals(IndexFileNames.OLD_SEGMENTS_GEN)) {
168                     foundSegmentFiles++;
169                     if (file.equals(si.getSegmentsFileName()) == false) {
170                         directory.deleteFile(file); // remove all segment_N files except of the one we wanna keep
171                     }
172                 }
173             }
174             assert SegmentInfos.getLastCommitSegmentsFileName(directory).equals(segmentsFileName);
175             if (foundSegmentFiles == 0) {
176                 throw new IllegalStateException("no commit found in the directory");
177             }
178         }
179         final CommitPoint cp = new CommitPoint(si, directory);
180         try (IndexWriter writer = new IndexWriter(directory, new IndexWriterConfig(Lucene.STANDARD_ANALYZER)
181                 .setIndexCommit(cp)
182                 .setCommitOnClose(false)
183                 .setMergePolicy(NoMergePolicy.INSTANCE)
184                 .setOpenMode(IndexWriterConfig.OpenMode.APPEND))) {
185             // do nothing and close this will kick of IndexFileDeleter which will remove all pending files
186         }
187         return si;
188     }

190     /**
191      * This method removes all lucene files from the given directory. It will first try to delete all commit points / segments
192      * files to ensure broken commits or corrupted indices will not be opened in the future. If any of the segment files can't be deleted
193      * this operation fails.
194      */
195     public static void cleanLuceneIndex(Directory directory) throws IOException {
196         try (Lock writeLock = directory.makeLock(IndexWriter.WRITE_LOCK_NAME)) {
197             if (!writeLock.obtain(IndexWriterConfig.getDefaultWriteLockTimeout())) { // obtain write lock
198                 throw new LockObtainFailedException("Index locked for write: " + writeLock);
199             }
200             for (final String file : directory.listAll()) {
201                 if (file.startsWith(IndexFileNames.SEGMENTS) || file.equals(IndexFileNames.OLD_SEGMENTS_GEN)) {
202                     directory.deleteFile(file); // remove all segment_N files
203                 }
204             }
205         }
206         try (IndexWriter writer = new IndexWriter(directory, new IndexWriterConfig(Lucene.STANDARD_ANALYZER)
207                 .setMergePolicy(NoMergePolicy.INSTANCE) // no merges
208                 .setCommitOnClose(false) // no commits
209                 .setOpenMode(IndexWriterConfig.OpenMode.CREATE))) // force creation - don't append...
210         {
211             // do nothing and close this will kick of IndexFileDeleter which will remove all pending files
212         }
213     }

215     public static void checkSegmentInfoIntegrity(final Directory directory) throws IOException {
216         new SegmentInfos.FindSegmentsFile(directory) {
217 
218             @Override
219             protected Object doBody(String segmentFileName) throws IOException {
220                 try (IndexInput input = directory.openInput(segmentFileName, IOContext.READ)) {
221                     final int format = input.readInt();
222                     final int actualFormat;
223                     if (format == CodecUtil.CODEC_MAGIC) {
224                         // 4.0+
225                         actualFormat = CodecUtil.checkHeaderNoMagic(input, "segments", SegmentInfos.VERSION_40, Integer.MAX_VALUE);
226                         if (actualFormat >= SegmentInfos.VERSION_48) {
227                             CodecUtil.checksumEntireFile(input);
228                         }
229                     }
230                     // legacy....
231                 }
232                 return null;
233             }
234         }.run();
235     }

309     /**
310      * Performs a count on <code>query</code> and <code>filter</code> with early termination using <code>searcher</code>.
311      * The early termination threshold is specified by the provided <code>collector</code>
312      */
313     public static boolean countWithEarlyTermination(IndexSearcher searcher, Filter filter, Query query,
314                                                         EarlyTerminatingCollector collector) throws IOException {
315         query = wrapCountQuery(query);
316         try {
317             if (filter == null) {
318                 searcher.search(query, collector);
319             } else {
320                 searcher.search(query, filter, collector);
321             }
322         } catch (EarlyTerminationException e) {
323             // early termination
324             return true;
325         }
326         return false;
327     }

360     public static TopDocs readTopDocs(StreamInput in) throws IOException {
361         if (in.readBoolean()) {
362             int totalHits = in.readVInt();
363             float maxScore = in.readFloat();
364 
365             SortField[] fields = new SortField[in.readVInt()];
366             for (int i = 0; i < fields.length; i++) {
367                 String field = null;
368                 if (in.readBoolean()) {
369                     field = in.readString();
370                 }
371                 fields[i] = new SortField(field, readSortType(in), in.readBoolean());
372             }
373 
374             FieldDoc[] fieldDocs = new FieldDoc[in.readVInt()];
375             for (int i = 0; i < fieldDocs.length; i++) {
376                 fieldDocs[i] = readFieldDoc(in);
377             }
378             return new TopFieldDocs(totalHits, fieldDocs, fields, maxScore);
379         } else {
380             int totalHits = in.readVInt();
381             float maxScore = in.readFloat();
382 
383             ScoreDoc[] scoreDocs = new ScoreDoc[in.readVInt()];
384             for (int i = 0; i < scoreDocs.length; i++) {
385                 scoreDocs[i] = new ScoreDoc(in.readVInt(), in.readFloat());
386             }
387             return new TopDocs(totalHits, scoreDocs, maxScore);
388         }
389     }

391     public static FieldDoc readFieldDoc(StreamInput in) throws IOException {
392         Comparable[] cFields = new Comparable[in.readVInt()];
393         for (int j = 0; j < cFields.length; j++) {
394             byte type = in.readByte();
395             if (type == 0) {
396                 cFields[j] = null;
397             } else if (type == 1) {
398                 cFields[j] = in.readString();
399             } else if (type == 2) {
400                 cFields[j] = in.readInt();
401             } else if (type == 3) {
402                 cFields[j] = in.readLong();
403             } else if (type == 4) {
404                 cFields[j] = in.readFloat();
405             } else if (type == 5) {
406                 cFields[j] = in.readDouble();
407             } else if (type == 6) {
408                 cFields[j] = in.readByte();
409             } else if (type == 7) {
410                 cFields[j] = in.readShort();
411             } else if (type == 8) {
412                 cFields[j] = in.readBoolean();
413             } else if (type == 9) {
414                 cFields[j] = in.readBytesRef();
415             } else {
416                 throw new IOException("Can't match type [" + type + "]");
417             }
418         }
419         return new FieldDoc(in.readVInt(), in.readFloat(), cFields);
420     }

426     public static void writeTopDocs(StreamOutput out, TopDocs topDocs, int from) throws IOException {
427         from = Math.min(from, topDocs.scoreDocs.length);
428         if (topDocs instanceof TopFieldDocs) {
429             out.writeBoolean(true);
430             TopFieldDocs topFieldDocs = (TopFieldDocs) topDocs;
431 
432             out.writeVInt(topDocs.totalHits);
433             out.writeFloat(topDocs.getMaxScore());
434 
435             out.writeVInt(topFieldDocs.fields.length);
436             for (SortField sortField : topFieldDocs.fields) {
437                 if (sortField.getField() == null) {
438                     out.writeBoolean(false);
439                 } else {
440                     out.writeBoolean(true);
441                     out.writeString(sortField.getField());
442                 }
443                 if (sortField.getComparatorSource() != null) {
444                     writeSortType(out, ((IndexFieldData.XFieldComparatorSource) sortField.getComparatorSource()).reducedType());
445                 } else {
446                     writeSortType(out, sortField.getType());
447                 }
448                 out.writeBoolean(sortField.getReverse());
449             }
450 
451             out.writeVInt(topDocs.scoreDocs.length - from);
452             for (int i = from; i < topFieldDocs.scoreDocs.length; ++i) {
453                 ScoreDoc doc = topFieldDocs.scoreDocs[i];
454                 writeFieldDoc(out, (FieldDoc) doc);
455             }
456         } else {
457             out.writeBoolean(false);
458             out.writeVInt(topDocs.totalHits);
459             out.writeFloat(topDocs.getMaxScore());
460 
461             out.writeVInt(topDocs.scoreDocs.length - from);
462             for (int i = from; i < topDocs.scoreDocs.length; ++i) {
463                 ScoreDoc doc = topDocs.scoreDocs[i];
464                 writeScoreDoc(out, doc);
465             }
466         }
467     }

469     public static void writeFieldDoc(StreamOutput out, FieldDoc fieldDoc) throws IOException {
470         out.writeVInt(fieldDoc.fields.length);
471         for (Object field : fieldDoc.fields) {
472             if (field == null) {
473                 out.writeByte((byte) 0);
474             } else {
475                 Class type = field.getClass();
476                 if (type == String.class) {
477                     out.writeByte((byte) 1);
478                     out.writeString((String) field);
479                 } else if (type == Integer.class) {
480                     out.writeByte((byte) 2);
481                     out.writeInt((Integer) field);
482                 } else if (type == Long.class) {
483                     out.writeByte((byte) 3);
484                     out.writeLong((Long) field);
485                 } else if (type == Float.class) {
486                     out.writeByte((byte) 4);
487                     out.writeFloat((Float) field);
488                 } else if (type == Double.class) {
489                     out.writeByte((byte) 5);
490                     out.writeDouble((Double) field);
491                 } else if (type == Byte.class) {
492                     out.writeByte((byte) 6);
493                     out.writeByte((Byte) field);
494                 } else if (type == Short.class) {
495                     out.writeByte((byte) 7);
496                     out.writeShort((Short) field);
497                 } else if (type == Boolean.class) {
498                     out.writeByte((byte) 8);
499                     out.writeBoolean((Boolean) field);
500                 } else if (type == BytesRef.class) {
501                     out.writeByte((byte) 9);
502                     out.writeBytesRef((BytesRef) field);
503                 } else {
504                     throw new IOException("Can't handle sort field value of type [" + type + "]");
505                 }
506             }
507         }
508         out.writeVInt(fieldDoc.doc);
509         out.writeFloat(fieldDoc.score);
510     }

529     public static Explanation readExplanation(StreamInput in) throws IOException {
530         Explanation explanation;
531         if (in.readBoolean()) {
532             Boolean match = in.readOptionalBoolean();
533             explanation = new ComplexExplanation();
534             ((ComplexExplanation) explanation).setMatch(match);
535 
536         } else {
537             explanation = new Explanation();
538         }
539         explanation.setValue(in.readFloat());
540         explanation.setDescription(in.readString());
541         if (in.readBoolean()) {
542             int size = in.readVInt();
543             for (int i = 0; i < size; i++) {
544                 explanation.addDetail(readExplanation(in));
545             }
546         }
547         return explanation;
548     }

550     public static void writeExplanation(StreamOutput out, Explanation explanation) throws IOException {
551 
552         if (explanation instanceof ComplexExplanation) {
553             out.writeBoolean(true);
554             out.writeOptionalBoolean(((ComplexExplanation) explanation).getMatch());
555         } else {
556             out.writeBoolean(false);
557         }
558         out.writeFloat(explanation.getValue());
559         out.writeString(explanation.getDescription());
560         Explanation[] subExplanations = explanation.getDetails();
561         if (subExplanations == null) {
562             out.writeBoolean(false);
563         } else {
564             out.writeBoolean(true);
565             out.writeVInt(subExplanations.length);
566             for (Explanation subExp : subExplanations) {
567                 writeExplanation(out, subExp);
568             }
569         }
570     }

583     /**
584      * A collector that terminates early by throwing {@link org.elasticsearch.common.lucene.Lucene.EarlyTerminationException}
585      * when count of matched documents has reached <code>maxCountHits</code>
586      */
587     public final static class EarlyTerminatingCollector extends SimpleCollector {
588 
589         private final int maxCountHits;
590         private final Collector delegate;
591 
592         private int count = 0;
593         private LeafCollector leafCollector;
594 
595         EarlyTerminatingCollector(int maxCountHits) {
596             this.maxCountHits = maxCountHits;
597             this.delegate = NOOP_COLLECTOR;
598         }
599 
600         EarlyTerminatingCollector(final Collector delegate, int maxCountHits) {
601             this.maxCountHits = maxCountHits;
602             this.delegate = (delegate == null) ? NOOP_COLLECTOR : delegate;
603         }
604 
605         public void reset() {
606             count = 0;
