63  /**
64   * Each shard will have a percolator registry even if there isn't a {@link PercolatorService#TYPE_NAME} document type in the index.
65   * For shards with indices that have no {@link PercolatorService#TYPE_NAME} document type, this will hold no percolate queries.
66   * <p/>
67   * Once a document type has been created, the real-time percolator will start to listen to write events and update the
68   * this registry with queries in real time.
69   */
70  public class PercolatorQueriesRegistry extends AbstractIndexShardComponent implements Closeable{

99      @Inject
100     public PercolatorQueriesRegistry(ShardId shardId, @IndexSettings Settings indexSettings, IndexQueryParserService queryParserService,
101                                      ShardIndexingService indexingService, IndicesLifecycle indicesLifecycle, MapperService mapperService,
102                                      IndexCache indexCache, IndexFieldDataService indexFieldDataService, ShardPercolateService shardPercolateService) {
103         super(shardId, indexSettings);
104         this.queryParserService = queryParserService;
105         this.mapperService = mapperService;
106         this.indicesLifecycle = indicesLifecycle;
107         this.indexingService = indexingService;
108         this.indexCache = indexCache;
109         this.indexFieldDataService = indexFieldDataService;
110         this.shardPercolateService = shardPercolateService;
111         this.mapUnmappedFieldsAsString = indexSettings.getAsBoolean(MAP_UNMAPPED_FIELDS_AS_STRING, false);
112 
113         indicesLifecycle.addListener(shardLifecycleListener);
114         mapperService.addTypeListener(percolateTypeListener);
115     }

133     void enableRealTimePercolator() {
134         if (realTimePercolatorEnabled.compareAndSet(false, true)) {
135             indexingService.addListener(realTimePercolatorOperationListener);
136         }
137     }
138 
139     void disableRealTimePercolator() {
140         if (realTimePercolatorEnabled.compareAndSet(true, false)) {
141             indexingService.removeListener(realTimePercolatorOperationListener);
142         }
143     }

145     public void addPercolateQuery(String idAsString, BytesReference source) {
146         Query newquery = parsePercolatorDocument(idAsString, source);
147         BytesRef id = new BytesRef(idAsString);
148         Query previousQuery = percolateQueries.put(id, newquery);
149         shardPercolateService.addedQuery(id, previousQuery, newquery);
150     }
151 
152     public void removePercolateQuery(String idAsString) {
153         BytesRef id = new BytesRef(idAsString);
154         Query query = percolateQueries.remove(id);
155         if (query != null) {
156             shardPercolateService.removedQuery(id, query);
157         }
158     }

160     Query parsePercolatorDocument(String id, BytesReference source) {
161         String type = null;
162         BytesReference querySource = null;
163         try (XContentParser sourceParser = XContentHelper.createParser(source)) {
164             String currentFieldName = null;
165             XContentParser.Token token = sourceParser.nextToken(); // move the START_OBJECT
166             if (token != XContentParser.Token.START_OBJECT) {
167                 throw new ElasticsearchException("failed to parse query [" + id + "], not starting with OBJECT");
168             }
169             while ((token = sourceParser.nextToken()) != XContentParser.Token.END_OBJECT) {
170                 if (token == XContentParser.Token.FIELD_NAME) {
171                     currentFieldName = sourceParser.currentName();
172                 } else if (token == XContentParser.Token.START_OBJECT) {
173                     if ("query".equals(currentFieldName)) {
174                         if (type != null) {
175                             return parseQuery(type, sourceParser);
176                         } else {
177                             XContentBuilder builder = XContentFactory.contentBuilder(sourceParser.contentType());
178                             builder.copyCurrentStructure(sourceParser);
179                             querySource = builder.bytes();
180                             builder.close();
181                         }
182                     } else {
183                         sourceParser.skipChildren();
184                     }
185                 } else if (token == XContentParser.Token.START_ARRAY) {
186                     sourceParser.skipChildren();
187                 } else if (token.isValue()) {
188                     if ("type".equals(currentFieldName)) {
189                         type = sourceParser.text();
190                     }
191                 }
192             }
193             try (XContentParser queryParser = XContentHelper.createParser(querySource)) {
194                 return parseQuery(type, queryParser);
195             }
196         } catch (Exception e) {
197             throw new PercolatorException(shardId().index(), "failed to parse query [" + id + "]", e);
198         }
199     }
200 
201     private Query parseQuery(String type, XContentParser parser) {
202         String[] previousTypes = null;
203         if (type != null) {
204             QueryParseContext.setTypesWithPrevious(new String[]{type});
205         }
206         QueryParseContext context = cache.get();
207         try {
208             context.reset(parser);
209             // This means that fields in the query need to exist in the mapping prior to registering this query
210             // The reason that this is required, is that if a field doesn't exist then the query assumes defaults, which may be undesired.
211             //
212             // Even worse when fields mentioned in percolator queries do go added to map after the queries have been registered
213             // then the percolator queries don't work as expected any more.
214             //
215             // Query parsing can't introduce new fields in mappings (which happens when registering a percolator query),
216             // because field type can't be inferred from queries (like document do) so the best option here is to disallow
217             // the usage of unmapped fields in percolator queries to avoid unexpected behaviour
218             //
219             // if index.percolator.map_unmapped_fields_as_string is set to true, query can contain unmapped fields which will be mapped
220             // as an analyzed string.
221             context.setAllowUnmappedFields(false);
222             context.setMapUnmappedFieldAsString(mapUnmappedFieldsAsString ? true : false);
223             return queryParserService.parseInnerQuery(context);
224         } catch (IOException e) {
225             throw new QueryParsingException(queryParserService.index(), "Failed to parse", e);
226         } finally {
227             if (type != null) {
228                 QueryParseContext.setTypes(previousTypes);
229             }
230             context.reset(null);
231         }
232     }

234     private class PercolateTypeListener implements DocumentTypeListener {
235 
236         @Override
237         public void beforeCreate(DocumentMapper mapper) {
238             if (PercolatorService.TYPE_NAME.equals(mapper.type())) {
239                 enableRealTimePercolator();
240             }
241         }
242 
243         @Override
244         public void afterRemove(DocumentMapper mapper) {
245             if (PercolatorService.TYPE_NAME.equals(mapper.type())) {
246                 disableRealTimePercolator();
247                 clear();
248             }
249         }

253     private class ShardLifecycleListener extends IndicesLifecycle.Listener {
254 
255         @Override
256         public void afterIndexShardCreated(IndexShard indexShard) {
257             if (hasPercolatorType(indexShard)) {
258                 enableRealTimePercolator();
259             }
260         }
261 
262         @Override
263         public void afterIndexShardPostRecovery(IndexShard indexShard) {
264             if (hasPercolatorType(indexShard)) {
265                 // percolator index has started, fetch what we can from it and initialize the indices
266                 // we have
267                 logger.trace("loading percolator queries for [{}]...", shardId);
268                 int loadedQueries = loadQueries(indexShard);
269                 logger.debug("done loading [{}] percolator queries for [{}]", loadedQueries, shardId);
270             }
271         }
272 
273         private boolean hasPercolatorType(IndexShard indexShard) {
274             ShardId otherShardId = indexShard.shardId();
275             return shardId.equals(otherShardId) && mapperService.hasMapping(PercolatorService.TYPE_NAME);
276         }
277 
278         private int loadQueries(IndexShard shard) {
279             shard.refresh("percolator_load_queries");
280             // Maybe add a mode load? This isn't really a write. We need write b/c state=post_recovery
281             try (Engine.Searcher searcher = shard.acquireSearcher("percolator_load_queries", true)) {
282                 Query query = new ConstantScoreQuery(
283                         indexCache.filter().cache(
284                                 new TermFilter(new Term(TypeFieldMapper.NAME, PercolatorService.TYPE_NAME)),
285                                 null,
286                                 queryParserService.autoFilterCachePolicy()
287                         )
288                 );
289                 QueriesLoaderCollector queryCollector = new QueriesLoaderCollector(PercolatorQueriesRegistry.this, logger, mapperService, indexFieldDataService);
290                 searcher.searcher().search(query, queryCollector);
291                 Map<BytesRef, Query> queries = queryCollector.queries();
292                 for (Map.Entry<BytesRef, Query> entry : queries.entrySet()) {
293                     Query previousQuery = percolateQueries.put(entry.getKey(), entry.getValue());
294                     shardPercolateService.addedQuery(entry.getKey(), previousQuery, entry.getValue());
295                 }
296                 return queries.size();
297             } catch (Exception e) {
298                 throw new PercolatorException(shardId.index(), "failed to load queries from percolator index", e);
299             }
300         }

304     private class RealTimePercolatorOperationListener extends IndexingOperationListener {
305 
306         @Override
307         public Engine.Create preCreate(Engine.Create create) {
308             // validate the query here, before we index
309             if (PercolatorService.TYPE_NAME.equals(create.type())) {
310                 parsePercolatorDocument(create.id(), create.source());
311             }
312             return create;
313         }
314 
315         @Override
316         public void postCreateUnderLock(Engine.Create create) {
317             // add the query under a doc lock
318             if (PercolatorService.TYPE_NAME.equals(create.type())) {
319                 addPercolateQuery(create.id(), create.source());
320             }
321         }
322 
323         @Override
324         public Engine.Index preIndex(Engine.Index index) {
325             // validate the query here, before we index
326             if (PercolatorService.TYPE_NAME.equals(index.type())) {
327                 parsePercolatorDocument(index.id(), index.source());
328             }
329             return index;
330         }
331 
332         @Override
333         public void postIndexUnderLock(Engine.Index index) {
334             // add the query under a doc lock
335             if (PercolatorService.TYPE_NAME.equals(index.type())) {
336                 addPercolateQuery(index.id(), index.source());
337             }
338         }
339 
340         @Override
341         public void postDeleteUnderLock(Engine.Delete delete) {
342             // remove the query under a lock
343             if (PercolatorService.TYPE_NAME.equals(delete.type())) {
344                 removePercolateQuery(delete.id());
345             }
346         }
