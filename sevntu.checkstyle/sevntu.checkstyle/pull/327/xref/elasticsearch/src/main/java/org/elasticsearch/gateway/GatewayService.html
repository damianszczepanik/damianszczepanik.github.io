48  public class GatewayService extends AbstractLifecycleComponent<GatewayService> implements ClusterStateListener {

50      public static final ClusterBlock STATE_NOT_RECOVERED_BLOCK = new ClusterBlock(1, "state not recovered / initialized", true, true, RestStatus.SERVICE_UNAVAILABLE, ClusterBlockLevel.ALL);
51  
52      public static final TimeValue DEFAULT_RECOVER_AFTER_TIME_IF_EXPECTED_NODES_IS_SET = TimeValue.timeValueMinutes(5);
53  
54      private final Gateway gateway;
55  
56      private final ThreadPool threadPool;
57  
58      private final AllocationService allocationService;
59  
60      private final ClusterService clusterService;
61  
62      private final DiscoveryService discoveryService;
63  
64      private final TimeValue recoverAfterTime;
65      private final int recoverAfterNodes;
66      private final int expectedNodes;
67      private final int recoverAfterDataNodes;
68      private final int expectedDataNodes;
69      private final int recoverAfterMasterNodes;
70      private final int expectedMasterNodes;
71  
72  
73      private final AtomicBoolean recovered = new AtomicBoolean();
74      private final AtomicBoolean scheduledRecovery = new AtomicBoolean();
75  
76      @Inject
77      public GatewayService(Settings settings, Gateway gateway, AllocationService allocationService, ClusterService clusterService, DiscoveryService discoveryService, ThreadPool threadPool) {
78          super(settings);
79          this.gateway = gateway;
80          this.allocationService = allocationService;
81          this.clusterService = clusterService;
82          this.discoveryService = discoveryService;
83          this.threadPool = threadPool;
84          // allow to control a delay of when indices will get created
85          this.expectedNodes = this.settings.getAsInt("gateway.expected_nodes", -1);
86          this.expectedDataNodes = this.settings.getAsInt("gateway.expected_data_nodes", -1);
87          this.expectedMasterNodes = this.settings.getAsInt("gateway.expected_master_nodes", -1);
88  
89          TimeValue defaultRecoverAfterTime = null;
90          if (expectedNodes >= 0 || expectedDataNodes >= 0 || expectedMasterNodes >= 0) {
91              defaultRecoverAfterTime = DEFAULT_RECOVER_AFTER_TIME_IF_EXPECTED_NODES_IS_SET;
92          }
93  
94          this.recoverAfterTime = this.settings.getAsTime("gateway.recover_after_time", defaultRecoverAfterTime);
95          this.recoverAfterNodes = this.settings.getAsInt("gateway.recover_after_nodes", -1);
96          this.recoverAfterDataNodes = this.settings.getAsInt("gateway.recover_after_data_nodes", -1);
97          // default the recover after master nodes to the minimum master nodes in the discovery
98          this.recoverAfterMasterNodes = this.settings.getAsInt("gateway.recover_after_master_nodes", settings.getAsInt("discovery.zen.minimum_master_nodes", -1));
99  
100         // Add the not recovered as initial state block, we don't allow anything until
101         this.clusterService.addInitialStateBlock(STATE_NOT_RECOVERED_BLOCK);
102     }
103 
104     @Override
105     protected void doStart() throws ElasticsearchException {
106         clusterService.addLast(this);
107         // if we received initial state, see if we can recover within the start phase, so we hold the
108         // node from starting until we recovered properly
109         if (discoveryService.initialStateReceived()) {
110             ClusterState clusterState = clusterService.state();
111             if (clusterState.nodes().localNodeMaster() && clusterState.blocks().hasGlobalBlock(STATE_NOT_RECOVERED_BLOCK)) {
112                 checkStateMeetsSettingsAndMaybeRecover(clusterState, false);
113             }
114         } else {
115             logger.debug("can't wait on start for (possibly) reading state from gateway, will do it asynchronously");
116         }
117     }
118 
119     @Override
120     protected void doStop() throws ElasticsearchException {
121         clusterService.remove(this);
122     }
123 
124     @Override
125     protected void doClose() throws ElasticsearchException {
126     }
127 
128     @Override
129     public void clusterChanged(final ClusterChangedEvent event) {
130         if (lifecycle.stoppedOrClosed()) {
131             return;
132         }
133         if (event.localNodeMaster() && event.state().blocks().hasGlobalBlock(STATE_NOT_RECOVERED_BLOCK)) {
134             checkStateMeetsSettingsAndMaybeRecover(event.state(), true);
135         }
136     }
137 
138     protected void checkStateMeetsSettingsAndMaybeRecover(ClusterState state, boolean asyncRecovery) {
139         DiscoveryNodes nodes = state.nodes();
140         if (state.blocks().hasGlobalBlock(discoveryService.getNoMasterBlock())) {
141             logger.debug("not recovering from gateway, no master elected yet");
142         } else if (recoverAfterNodes != -1 && (nodes.masterAndDataNodes().size()) < recoverAfterNodes) {
143             logger.debug("not recovering from gateway, nodes_size (data+master) [" + nodes.masterAndDataNodes().size() + "] < recover_after_nodes [" + recoverAfterNodes + "]");
144         } else if (recoverAfterDataNodes != -1 && nodes.dataNodes().size() < recoverAfterDataNodes) {
145             logger.debug("not recovering from gateway, nodes_size (data) [" + nodes.dataNodes().size() + "] < recover_after_data_nodes [" + recoverAfterDataNodes + "]");
146         } else if (recoverAfterMasterNodes != -1 && nodes.masterNodes().size() < recoverAfterMasterNodes) {
147             logger.debug("not recovering from gateway, nodes_size (master) [" + nodes.masterNodes().size() + "] < recover_after_master_nodes [" + recoverAfterMasterNodes + "]");
148         } else {
149             boolean enforceRecoverAfterTime;
150             String reason;
151             if (expectedNodes == -1 && expectedMasterNodes == -1 && expectedDataNodes == -1) {
152                 // no expected is set, honor the setting if they are there
153                 enforceRecoverAfterTime = true;
154                 reason = "recovery_after_time was set to [" + recoverAfterTime + "]";
155             } else {
156                 // one of the expected is set, see if all of them meet the need, and ignore the timeout in this case
157                 enforceRecoverAfterTime = false;
158                 reason = "";
159                 if (expectedNodes != -1 && (nodes.masterAndDataNodes().size() < expectedNodes)) { // does not meet the expected...
160                     enforceRecoverAfterTime = true;
161                     reason = "expecting [" + expectedNodes + "] nodes, but only have [" + nodes.masterAndDataNodes().size() + "]";
162                 } else if (expectedDataNodes != -1 && (nodes.dataNodes().size() < expectedDataNodes)) { // does not meet the expected...
163                     enforceRecoverAfterTime = true;
164                     reason = "expecting [" + expectedDataNodes + "] data nodes, but only have [" + nodes.dataNodes().size() + "]";
165                 } else if (expectedMasterNodes != -1 && (nodes.masterNodes().size() < expectedMasterNodes)) { // does not meet the expected...
166                     enforceRecoverAfterTime = true;
167                     reason = "expecting [" + expectedMasterNodes + "] master nodes, but only have [" + nodes.masterNodes().size() + "]";
168                 }
169             }
170             performStateRecovery(asyncRecovery, enforceRecoverAfterTime, reason);
171         }
172     }
173 
174     private void performStateRecovery(boolean asyncRecovery, boolean enforceRecoverAfterTime, String reason) {
175         final Gateway.GatewayStateRecoveredListener recoveryListener = new GatewayRecoveryListener(new CountDownLatch(1));
176 
177         if (enforceRecoverAfterTime && recoverAfterTime != null) {
178             if (scheduledRecovery.compareAndSet(false, true)) {
179                 logger.info("delaying initial state recovery for [{}]. {}", recoverAfterTime, reason);
180                 threadPool.schedule(recoverAfterTime, ThreadPool.Names.GENERIC, new Runnable() {
181                     @Override
182                     public void run() {
183                         if (recovered.compareAndSet(false, true)) {
184                             logger.info("recovery_after_time [{}] elapsed. performing state recovery...", recoverAfterTime);
185                             gateway.performStateRecovery(recoveryListener);
186                         }
187                     }
188                 });
189             }
190         } else {
191             if (recovered.compareAndSet(false, true)) {
192                 if (asyncRecovery) {
193                     threadPool.generic().execute(new Runnable() {
194                         @Override
195                         public void run() {
196                             gateway.performStateRecovery(recoveryListener);
197                         }
198                     });
199                 } else {
200                     logger.trace("performing state recovery...");
201                     gateway.performStateRecovery(recoveryListener);
202                 }
203             }
204         }
205     }
206 
207     class GatewayRecoveryListener implements Gateway.GatewayStateRecoveredListener {
208 
209         private final CountDownLatch latch;
210 
211         GatewayRecoveryListener(CountDownLatch latch) {
212             this.latch = latch;
213         }
214 
215         @Override
216         public void onSuccess(final ClusterState recoveredState) {
217             logger.trace("successful state recovery, importing cluster state...");
218             clusterService.submitStateUpdateTask("local-gateway-elected-state", new ProcessedClusterStateUpdateTask() {
219                 @Override
220                 public ClusterState execute(ClusterState currentState) {
221                     assert currentState.metaData().indices().isEmpty();
222 
223                     // remove the block, since we recovered from gateway
224                     ClusterBlocks.Builder blocks = ClusterBlocks.builder()
225                             .blocks(currentState.blocks())
226                             .blocks(recoveredState.blocks())
227                             .removeGlobalBlock(STATE_NOT_RECOVERED_BLOCK);
228 
229                     MetaData.Builder metaDataBuilder = MetaData.builder(recoveredState.metaData());
230                     // automatically generate a UID for the metadata if we need to
231                     metaDataBuilder.generateUuidIfNeeded();
232 
233                     if (recoveredState.metaData().settings().getAsBoolean(MetaData.SETTING_READ_ONLY, false) || currentState.metaData().settings().getAsBoolean(MetaData.SETTING_READ_ONLY, false)) {
234                         blocks.addGlobalBlock(MetaData.CLUSTER_READ_ONLY_BLOCK);
235                     }
236 
237                     for (IndexMetaData indexMetaData : recoveredState.metaData()) {
238                         metaDataBuilder.put(indexMetaData, false);
239                         blocks.addBlocks(indexMetaData);
240                     }
241 
242                     // update the state to reflect the new metadata and routing
243                     ClusterState updatedState = ClusterState.builder(currentState)
244                             .blocks(blocks)
245                             .metaData(metaDataBuilder)
246                             .build();
247 
248                     // initialize all index routing tables as empty
249                     RoutingTable.Builder routingTableBuilder = RoutingTable.builder(updatedState.routingTable());
250                     for (ObjectCursor<IndexMetaData> cursor : updatedState.metaData().indices().values()) {
251                         routingTableBuilder.addAsRecovery(cursor.value);
252                     }
253                     // start with 0 based versions for routing table
254                     routingTableBuilder.version(0);
255 
256                     // now, reroute
257                     RoutingAllocation.Result routingResult = allocationService.reroute(ClusterState.builder(updatedState).routingTable(routingTableBuilder).build());
258 
259                     return ClusterState.builder(updatedState).routingResult(routingResult).build();
260                 }
261 
262                 @Override
263                 public void onFailure(String source, Throwable t) {
264                     logger.error("unexpected failure during [{}]", t, source);
265                 }
266 
267                 @Override
268                 public void clusterStateProcessed(String source, ClusterState oldState, ClusterState newState) {
269                     logger.info("recovered [{}] indices into cluster_state", newState.metaData().indices().size());
270                     latch.countDown();
271                 }
272             });
273         }
274 
275         @Override
276         public void onFailure(String message) {
277             recovered.set(false);
278             scheduledRecovery.set(false);
279             // don't remove the block here, we don't want to allow anything in such a case
280             logger.info("metadata state not restored, reason: {}", message);
281         }
282     }
283 
284     // used for testing
285     public TimeValue recoverAfterTime() {
286         return recoverAfterTime;
287     }
