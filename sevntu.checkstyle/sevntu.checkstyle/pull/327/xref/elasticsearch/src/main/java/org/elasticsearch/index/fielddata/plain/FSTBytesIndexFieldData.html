40  /**
41   */
42  public class FSTBytesIndexFieldData extends AbstractIndexOrdinalsFieldData {
43  
44      private final CircuitBreakerService breakerService;
45  
46      public static class Builder implements IndexFieldData.Builder {
47  
48          @Override
49          public IndexOrdinalsFieldData build(Index index, @IndexSettings Settings indexSettings, FieldMapper<?> mapper,
50                                                               IndexFieldDataCache cache, CircuitBreakerService breakerService, MapperService mapperService) {
51              return new FSTBytesIndexFieldData(index, indexSettings, mapper.names(), mapper.fieldDataType(), cache, breakerService);
52          }
53      }
54  
55      FSTBytesIndexFieldData(Index index, @IndexSettings Settings indexSettings, FieldMapper.Names fieldNames, FieldDataType fieldDataType,
56                             IndexFieldDataCache cache, CircuitBreakerService breakerService) {
57          super(index, indexSettings, fieldNames, fieldDataType, cache, breakerService);
58          this.breakerService = breakerService;
59      }
60  
61      @Override
62      public AtomicOrdinalsFieldData loadDirect(LeafReaderContext context) throws Exception {
63          LeafReader reader = context.reader();
64  
65          Terms terms = reader.terms(getFieldNames().indexName());
66          AtomicOrdinalsFieldData data = null;
67          // TODO: Use an actual estimator to estimate before loading.
68          NonEstimatingEstimator estimator = new NonEstimatingEstimator(breakerService.getBreaker(CircuitBreaker.FIELDDATA));
69          if (terms == null) {
70              data = AbstractAtomicOrdinalsFieldData.empty();
71              estimator.afterLoad(null, data.ramBytesUsed());
72              return data;
73          }
74          PositiveIntOutputs outputs = PositiveIntOutputs.getSingleton();
75          org.apache.lucene.util.fst.Builder<Long> fstBuilder = new org.apache.lucene.util.fst.Builder<>(INPUT_TYPE.BYTE1, outputs);
76          final IntsRefBuilder scratch = new IntsRefBuilder();
77  
78          final long numTerms;
79          if (regex == null && frequency == null) {
80              numTerms = terms.size();
81          } else {
82              numTerms = -1;
83          }
84          final float acceptableTransientOverheadRatio = fieldDataType.getSettings().getAsFloat("acceptable_transient_overhead_ratio", OrdinalsBuilder.DEFAULT_ACCEPTABLE_OVERHEAD_RATIO);
85          boolean success = false;
86          try (OrdinalsBuilder builder = new OrdinalsBuilder(numTerms, reader.maxDoc(), acceptableTransientOverheadRatio)) {
87  
88              // we don't store an ord 0 in the FST since we could have an empty string in there and FST don't support
89              // empty strings twice. ie. them merge fails for long output.
90              TermsEnum termsEnum = filter(terms, reader);
91              PostingsEnum docsEnum = null;
92              for (BytesRef term = termsEnum.next(); term != null; term = termsEnum.next()) {
93                  final long termOrd = builder.nextOrdinal();
94                  fstBuilder.add(Util.toIntsRef(term, scratch), (long) termOrd);
95                  docsEnum = termsEnum.postings(null, docsEnum, PostingsEnum.NONE);
96                  for (int docId = docsEnum.nextDoc(); docId != DocIdSetIterator.NO_MORE_DOCS; docId = docsEnum.nextDoc()) {
97                      builder.addDoc(docId);
98                  }
99              }
100 
101             FST<Long> fst = fstBuilder.finish();
102 
103             final Ordinals ordinals = builder.build(fieldDataType.getSettings());
104 
105             data = new FSTBytesAtomicFieldData(fst, ordinals);
106             success = true;
107             return data;
108         } finally {
109             if (success) {
110                 estimator.afterLoad(null, data.ramBytesUsed());
111             }
112 
113         }
114     }
115 }
