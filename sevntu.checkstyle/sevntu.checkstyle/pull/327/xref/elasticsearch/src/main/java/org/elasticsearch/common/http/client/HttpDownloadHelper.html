40      private boolean useTimestamp = false;
41      private boolean skipExisting = false;
42  
43      public boolean download(URL source, Path dest, @Nullable DownloadProgress progress, TimeValue timeout) throws Exception {
44          if (Files.exists(dest) && skipExisting) {
45              return true;
46          }
47  
48          //don't do any progress, unless asked
49          if (progress == null) {
50              progress = new NullProgress();
51          }
52  
53          //set the timestamp to the file date.
54          long timestamp = 0;
55  
56          boolean hasTimestamp = false;
57          if (useTimestamp && Files.exists(dest) ) {
58              timestamp = Files.getLastModifiedTime(dest).toMillis();
59              hasTimestamp = true;
60          }
61  
62          GetThread getThread = new GetThread(source, dest, hasTimestamp, timestamp, progress);
63  
64          try {
65              getThread.setDaemon(true);
66              getThread.start();
67              getThread.join(timeout.millis());
68  
69              if (getThread.isAlive()) {
70                  throw new ElasticsearchTimeoutException("The GET operation took longer than " + timeout + ", stopping it.");
71              }
72          }
73          catch (InterruptedException ie) {
74              return false;
75          } finally {
76              getThread.closeStreams();
77          }
78  
79          return getThread.wasSuccessful();
80      }

133     /**
134      * verbose progress system prints to some output stream
135      */
136     public static class VerboseProgress implements DownloadProgress {
137         private int dots = 0;
138         // CheckStyle:VisibilityModifier OFF - bc
139         PrintWriter writer;
140         // CheckStyle:VisibilityModifier ON
141 
142         /**
143          * Construct a verbose progress reporter.
144          *
145          * @param out the output stream.
146          */
147         public VerboseProgress(PrintStream out) {
148             this.writer = new PrintWriter(out);
149         }
150 
151         /**
152          * Construct a verbose progress reporter.
153          *
154          * @param writer the output stream.
155          */
156         public VerboseProgress(PrintWriter writer) {
157             this.writer = writer;
158         }
159 
160         /**
161          * begin a download
162          */
163         @Override
164         public void beginDownload() {
165             writer.print("Downloading ");
166             dots = 0;
167         }
168 
169         /**
170          * tick handler
171          */
172         @Override
173         public void onTick() {
174             writer.print(".");
175             if (dots++ > 50) {
176                 writer.flush();
177                 dots = 0;
178             }
179         }
180 
181         /**
182          * end a download
183          */
184         @Override
185         public void endDownload() {
186             writer.println("DONE");
187             writer.flush();
188         }

191     private class GetThread extends Thread {
192 
193         private final URL source;
194         private final Path dest;
195         private final boolean hasTimestamp;
196         private final long timestamp;
197         private final DownloadProgress progress;
198 
199         private boolean success = false;
200         private IOException ioexception = null;
201         private InputStream is = null;
202         private OutputStream os = null;
203         private URLConnection connection;
204         private int redirections = 0;
205 
206         GetThread(URL source, Path dest, boolean h, long t, DownloadProgress p) {
207             this.source = source;
208             this.dest = dest;
209             hasTimestamp = h;
210             timestamp = t;
211             progress = p;
212         }
213 
214         @Override
215         public void run() {
216             try {
217                 success = get();
218             } catch (IOException ioex) {
219                 ioexception = ioex;
220             }
221         }
222 
223         private boolean get() throws IOException {
224 
225             connection = openConnection(source);
226 
227             if (connection == null) {
228                 return false;
229             }
230 
231             boolean downloadSucceeded = downloadFile();
232 
233             //if (and only if) the use file time option is set, then
234             //the saved file now has its timestamp set to that of the
235             //downloaded file
236             if (downloadSucceeded && useTimestamp) {
237                 updateTimeStamp();
238             }
239 
240             return downloadSucceeded;
241         }

258                 throw new IOException(message);

265         private URLConnection openConnection(URL aSource) throws IOException {
266 
267             // set up the URL connection
268             URLConnection connection = aSource.openConnection();
269             // modify the headers
270             // NB: things like user authentication could go in here too.
271             if (hasTimestamp) {
272                 connection.setIfModifiedSince(timestamp);
273             }
274 
275             if (connection instanceof HttpURLConnection) {
276                 ((HttpURLConnection) connection).setInstanceFollowRedirects(false);
277                 ((HttpURLConnection) connection).setUseCaches(true);
278                 ((HttpURLConnection) connection).setConnectTimeout(5000);
279             }
280             // connect to the remote site (may take some time)
281             connection.connect();
282 
283             // First check on a 301 / 302 (moved) response (HTTP only)
284             if (connection instanceof HttpURLConnection) {
285                 HttpURLConnection httpConnection = (HttpURLConnection) connection;
286                 int responseCode = httpConnection.getResponseCode();
287                 if (responseCode == HttpURLConnection.HTTP_MOVED_PERM ||
288                         responseCode == HttpURLConnection.HTTP_MOVED_TEMP ||
289                         responseCode == HttpURLConnection.HTTP_SEE_OTHER) {
290                     String newLocation = httpConnection.getHeaderField("Location");
291                     String message = aSource
292                             + (responseCode == HttpURLConnection.HTTP_MOVED_PERM ? " permanently"
293                             : "") + " moved to " + newLocation;
294                     URL newURL = new URL(newLocation);
295                     if (!redirectionAllowed(aSource, newURL)) {
296                         return null;
297                     }
298                     return openConnection(newURL);
299                 }
300                 // next test for a 304 result (HTTP only)
301                 long lastModified = httpConnection.getLastModified();
302                 if (responseCode == HttpURLConnection.HTTP_NOT_MODIFIED
303                         || (lastModified != 0 && hasTimestamp && timestamp >= lastModified)) {
304                     // not modified so no file download. just return
305                     // instead and trace out something so the user
306                     // doesn't think that the download happened when it
307                     // didn't
308                     return null;
309                 }
310                 // test for 401 result (HTTP only)
311                 if (responseCode == HttpURLConnection.HTTP_UNAUTHORIZED) {
312                     String message = "HTTP Authorization failure";
313                     throw new IOException(message);
314                 }
315             }
316 
317             //REVISIT: at this point even non HTTP connections may
318             //support the if-modified-since behaviour -we just check
319             //the date of the content and skip the write if it is not
320             //newer. Some protocols (FTP) don't include dates, of
321             //course.
322             return connection;
323         }

325         private boolean downloadFile() throws FileNotFoundException, IOException {
326             IOException lastEx = null;
327             for (int i = 0; i < 3; i++) {
328                 // this three attempt trick is to get round quirks in different
329                 // Java implementations. Some of them take a few goes to bind
330                 // property; we ignore the first couple of such failures.
331                 try {
332                     is = connection.getInputStream();
333                     break;
334                 } catch (IOException ex) {
335                     lastEx = ex;
336                 }
337             }
338             if (is == null) {
339                 throw new IOException("Can't get " + source + " to " + dest, lastEx);
340             }
341 
342             os = Files.newOutputStream(dest);
343             progress.beginDownload();
344             boolean finished = false;
345             try {
346                 byte[] buffer = new byte[1024 * 100];
347                 int length;
348                 while (!isInterrupted() && (length = is.read(buffer)) >= 0) {
349                     os.write(buffer, 0, length);
350                     progress.onTick();
351                 }
352                 finished = !isInterrupted();
353             } finally {
354                 if (!finished) {
355                     // we have started to (over)write dest, but failed.
356                     // Try to delete the garbage we'd otherwise leave
357                     // behind.
358                     IOUtils.closeWhileHandlingException(os, is);
359                     IOUtils.deleteFilesIgnoringExceptions(dest);
360                 } else {
361                     IOUtils.close(os, is);
362                 }
363             }
364             progress.endDownload();
365             return true;
366         }
367 
368         private void updateTimeStamp() throws IOException {
369             long remoteTimestamp = connection.getLastModified();
370             if (remoteTimestamp != 0) {
371                 Files.setLastModifiedTime(dest, FileTime.fromMillis(remoteTimestamp));
372             }
373         }
374 
375         /**
376          * Has the download completed successfully?
377          * <p/>
378          * <p>Re-throws any exception caught during executaion.</p>
379          */
380         boolean wasSuccessful() throws IOException {
381             if (ioexception != null) {
382                 throw ioexception;
383             }
384             return success;
385         }
386 
387         /**
388          * Closes streams, interrupts the download, may delete the
389          * output file.
390          */
391         void closeStreams() throws IOException {
392             interrupt();
393             if (success) {
394                 IOUtils.close(is, os);
395             } else {
396                 IOUtils.closeWhileHandlingException(is, os);
397                 if (dest != null && Files.exists(dest)) {
398                     IOUtils.deleteFilesIgnoringExceptions(dest);
399                 }
400             }
401         }
