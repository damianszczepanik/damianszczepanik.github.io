161         @Override
162         public Mapper.Builder<?, ?> parse(String name, Map<String, Object> node, ParserContext parserContext) throws MapperParsingException {
163             CompletionFieldMapper.Builder builder = completionField(name);
164             NamedAnalyzer indexAnalyzer = null;
165             NamedAnalyzer searchAnalyzer = null;
166             for (Iterator<Map.Entry<String, Object>> iterator = node.entrySet().iterator(); iterator.hasNext();) {
167                 Map.Entry<String, Object> entry = iterator.next();
168                 String fieldName = entry.getKey();
169                 Object fieldNode = entry.getValue();
170                 if (fieldName.equals("type")) {
171                     continue;
172                 }
173                 if (Fields.ANALYZER.equals(fieldName) || // index_analyzer is for backcompat, remove for v3.0
174                     fieldName.equals("index_analyzer") && parserContext.indexVersionCreated().before(Version.V_2_0_0)) {
175                     
176                     indexAnalyzer = getNamedAnalyzer(parserContext, fieldNode.toString());
177                     iterator.remove();
178                 } else if (Fields.SEARCH_ANALYZER.match(fieldName)) {
179                     searchAnalyzer = getNamedAnalyzer(parserContext, fieldNode.toString());
180                     iterator.remove();
181                 } else if (fieldName.equals(Fields.PAYLOADS)) {
182                     builder.payloads(Boolean.parseBoolean(fieldNode.toString()));
183                     iterator.remove();
184                 } else if (Fields.PRESERVE_SEPARATORS.match(fieldName)) {
185                     builder.preserveSeparators(Boolean.parseBoolean(fieldNode.toString()));
186                     iterator.remove();
187                 } else if (Fields.PRESERVE_POSITION_INCREMENTS.match(fieldName)) {
188                     builder.preservePositionIncrements(Boolean.parseBoolean(fieldNode.toString()));
189                     iterator.remove();
190                 } else if (Fields.MAX_INPUT_LENGTH.match(fieldName)) {
191                     builder.maxInputLength(Integer.parseInt(fieldNode.toString()));
192                     iterator.remove();
193                 } else if (parseMultiField(builder, name, parserContext, fieldName, fieldNode)) {
194                     iterator.remove();
195                 } else if (fieldName.equals(Fields.CONTEXT)) {
196                     builder.contextMapping(ContextBuilder.loadMappings(fieldNode, parserContext.indexVersionCreated()));
197                     iterator.remove();
198                 }
199             }
200             
201             if (indexAnalyzer == null) {
202                 if (searchAnalyzer != null) {
203                     throw new MapperParsingException("analyzer on completion field [" + name + "] must be set when search_analyzer is set");
204                 }
205                 indexAnalyzer = searchAnalyzer = parserContext.analysisService().analyzer("simple");
206             } else if (searchAnalyzer == null) {
207                 searchAnalyzer = indexAnalyzer;
208             }
209             builder.indexAnalyzer(indexAnalyzer);
210             builder.searchAnalyzer(searchAnalyzer);
211             
212             return builder;
213         }
214 
215         private NamedAnalyzer getNamedAnalyzer(ParserContext parserContext, String name) {
216             NamedAnalyzer analyzer = parserContext.analysisService().analyzer(name);
217             if (analyzer == null) {
218                 throw new ElasticsearchIllegalArgumentException("Can't find default or mapped analyzer with name [" + name + "]");
219             }
220             return analyzer;
221         }

235      * 
236      * @param contextMappings Configuration of context type. If none should be used set {@link ContextMapping.EMPTY_MAPPING}
237      * @param wrappedPostingsFormat the postings format to wrap, or {@code null} to wrap the codec's default postings format

239     // Custom postings formats are deprecated but we still accept a postings format here to be able to test backward compatibility
240     // with older postings formats such as Elasticsearch090
241     public CompletionFieldMapper(Names names, NamedAnalyzer indexAnalyzer, NamedAnalyzer searchAnalyzer, PostingsFormat wrappedPostingsFormat, SimilarityProvider similarity, boolean payloads,
242                                  boolean preserveSeparators, boolean preservePositionIncrements, int maxInputLength, Settings indexSettings, MultiFields multiFields, CopyTo copyTo, SortedMap<String, ContextMapping> contextMappings) {
243         super(names, 1.0f, Defaults.FIELD_TYPE, null, indexAnalyzer, searchAnalyzer, similarity, null, null, indexSettings, multiFields, copyTo);
244         analyzingSuggestLookupProvider = new AnalyzingCompletionLookupProvider(preserveSeparators, false, preservePositionIncrements, payloads);
245         if (wrappedPostingsFormat == null) {
246             // delayed until postingsFormat() is called
247             this.postingsFormat = null;
248         } else {
249             this.postingsFormat = new Completion090PostingsFormat(wrappedPostingsFormat, analyzingSuggestLookupProvider);
250         }
251         this.preserveSeparators = preserveSeparators;
252         this.payloads = payloads;
253         this.preservePositionIncrements = preservePositionIncrements;
254         this.maxInputLength = maxInputLength;
255         this.contextMapping = contextMappings;
256     }

258     public synchronized PostingsFormat postingsFormat(PostingsFormat in) {
259         if (in instanceof Completion090PostingsFormat) {
260             throw new ElasticsearchIllegalStateException("Double wrapping of " + Completion090PostingsFormat.class);
261         }
262         if (postingsFormat == null) {
263             postingsFormat = new Completion090PostingsFormat(in, analyzingSuggestLookupProvider);
264         }
265         return postingsFormat;
266     }

268     @Override
269     public void parse(ParseContext context) throws IOException {
270         XContentParser parser = context.parser();
271         XContentParser.Token token = parser.currentToken();
272 
273         String surfaceForm = null;
274         BytesRef payload = null;
275         long weight = -1;
276         List<String> inputs = Lists.newArrayListWithExpectedSize(4);
277 
278         SortedMap<String, ContextConfig> contextConfig = null;
279 
280         if (token == XContentParser.Token.VALUE_STRING) {
281             inputs.add(parser.text());
282             multiFields.parse(this, context);
283         } else {
284             String currentFieldName = null;
285             while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
286                 if (token == XContentParser.Token.FIELD_NAME) {
287                     currentFieldName = parser.currentName();
288                     if (!ALLOWED_CONTENT_FIELD_NAMES.contains(currentFieldName)) {
289                         throw new ElasticsearchIllegalArgumentException("Unknown field name[" + currentFieldName + "], must be one of " + ALLOWED_CONTENT_FIELD_NAMES);
290                     }
291                 } else if (Fields.CONTEXT.equals(currentFieldName)) {
292                     SortedMap<String, ContextConfig> configs = Maps.newTreeMap(); 
293                     
294                     if (token == Token.START_OBJECT) {
295                         while ((token = parser.nextToken()) != Token.END_OBJECT) {
296                             String name = parser.text();
297                             ContextMapping mapping = contextMapping.get(name);
298                             if (mapping == null) {
299                                 throw new ElasticsearchParseException("context [" + name + "] is not defined");
300                             } else {
301                                 token = parser.nextToken();
302                                 configs.put(name, mapping.parseContext(context, parser));
303                             }
304                         }
305                         contextConfig = Maps.newTreeMap();
306                         for (ContextMapping mapping : contextMapping.values()) {
307                             ContextConfig config = configs.get(mapping.name());
308                             contextConfig.put(mapping.name(), config==null ? mapping.defaultConfig() : config);
309                         }
310                     } else {
311                         throw new ElasticsearchParseException("context must be an object");
312                     }
313                 } else if (Fields.CONTENT_FIELD_NAME_PAYLOAD.equals(currentFieldName)) {
314                     if (!isStoringPayloads()) {
315                         throw new MapperException("Payloads disabled in mapping");
316                     }
317                     if (token == XContentParser.Token.START_OBJECT) {
318                         XContentBuilder payloadBuilder = XContentFactory.contentBuilder(parser.contentType()).copyCurrentStructure(parser);
319                         payload = payloadBuilder.bytes().toBytesRef();
320                         payloadBuilder.close();
321                     } else if (token.isValue()) {
322                         payload = parser.utf8BytesOrNull();
323                     } else {
324                         throw new MapperException("payload doesn't support type " + token);
325                     }
326                 } else if (token == XContentParser.Token.VALUE_STRING) {
327                     if (Fields.CONTENT_FIELD_NAME_OUTPUT.equals(currentFieldName)) {
328                         surfaceForm = parser.text();
329                     }
330                     if (Fields.CONTENT_FIELD_NAME_INPUT.equals(currentFieldName)) {
331                         inputs.add(parser.text());
332                     }
333                     if (Fields.CONTENT_FIELD_NAME_WEIGHT.equals(currentFieldName)) {
334                         Number weightValue;
335                         try {
336                             weightValue = Long.parseLong(parser.text());
337                         } catch (NumberFormatException e) {
338                             throw new ElasticsearchIllegalArgumentException("Weight must be a string representing a numeric value, but was [" + parser.text() + "]");
339                         }
340                         weight = weightValue.longValue(); // always parse a long to make sure we don't get overflow
341                         checkWeight(weight);
342                     }
343                 } else if (token == XContentParser.Token.VALUE_NUMBER) {
344                     if (Fields.CONTENT_FIELD_NAME_WEIGHT.equals(currentFieldName)) {
345                         NumberType numberType = parser.numberType();
346                         if (NumberType.LONG != numberType && NumberType.INT != numberType) {
347                             throw new ElasticsearchIllegalArgumentException("Weight must be an integer, but was [" + parser.numberValue() + "]");
348                         }
349                         weight = parser.longValue(); // always parse a long to make sure we don't get overflow
350                         checkWeight(weight);
351                     }
352                 } else if (token == XContentParser.Token.START_ARRAY) {
353                     if (Fields.CONTENT_FIELD_NAME_INPUT.equals(currentFieldName)) {
354                         while ((token = parser.nextToken()) != XContentParser.Token.END_ARRAY) {
355                             inputs.add(parser.text());
356                         }
357                     }
358                 }
359             }
360         }
361 
362         if(contextConfig == null) {
363             contextConfig = Maps.newTreeMap();
364             for (ContextMapping mapping : contextMapping.values()) {
365                 contextConfig.put(mapping.name(), mapping.defaultConfig());
366             }
367         }
368 
369         final ContextMapping.Context ctx = new ContextMapping.Context(contextConfig, context.doc());
370 
371         payload = payload == null ? EMPTY : payload;
372         if (surfaceForm == null) { // no surface form use the input
373             for (String input : inputs) {
374                 BytesRef suggestPayload = analyzingSuggestLookupProvider.buildPayload(new BytesRef(
375                         input), weight, payload);
376                 context.doc().add(getCompletionField(ctx, input, suggestPayload));
377             }
378         } else {
379             BytesRef suggestPayload = analyzingSuggestLookupProvider.buildPayload(new BytesRef(
380                     surfaceForm), weight, payload);
381             for (String input : inputs) {
382                 context.doc().add(getCompletionField(ctx, input, suggestPayload));
383             }
384         }
385     }

387     private void checkWeight(long weight) {
388         if (weight < 0 || weight > Integer.MAX_VALUE) {
389             throw new ElasticsearchIllegalArgumentException("Weight must be in the interval [0..2147483647], but was [" + weight + "]");
390         }
391     }

409     public Field getCompletionField(ContextMapping.Context ctx, String input, BytesRef payload) {
410         final String originalInput = input;
411         if (input.length() > maxInputLength) {
412             final int len = correctSubStringLen(input, Math.min(maxInputLength, input.length()));
413             input = input.substring(0, len);
414         }
415         for (int i = 0; i < input.length(); i++) {
416             if (isReservedChar(input.charAt(i))) {
417                 throw new ElasticsearchIllegalArgumentException("Illegal input [" + originalInput + "] UTF-16 codepoint  [0x"
418                         + Integer.toHexString((int) input.charAt(i)).toUpperCase(Locale.ROOT)
419                         + "] at position " + i + " is a reserved character");
420             }
421         }
422         return new SuggestField(names.indexName(), ctx, input, this.fieldType, payload, analyzingSuggestLookupProvider);
423     }

425     public static int correctSubStringLen(String input, int len) {
426         if (Character.isHighSurrogate(input.charAt(len - 1))) {
427             assert input.length() >= len + 1 && Character.isLowSurrogate(input.charAt(len));
428             return len + 1;
429         }
430         return len;
431     }

438     private static final class SuggestField extends Field {
439         private final BytesRef payload;
440         private final CompletionTokenStream.ToFiniteStrings toFiniteStrings;
441         private final ContextMapping.Context ctx;
442 
443         public SuggestField(String name, ContextMapping.Context ctx, String value, FieldType type, BytesRef payload, CompletionTokenStream.ToFiniteStrings toFiniteStrings) {
444             super(name, value, type);
445             this.payload = payload;
446             this.toFiniteStrings = toFiniteStrings;
447             this.ctx = ctx;
448         }
449 
450         @Override
451         public TokenStream tokenStream(Analyzer analyzer, TokenStream previous) throws IOException {
452             TokenStream ts = ctx.wrapTokenStream(super.tokenStream(analyzer, previous));
453             return new CompletionTokenStream(ts, payload, toFiniteStrings);
454         }
455     }

457     @Override
458     public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
459         builder.startObject(name())
460                 .field(Fields.TYPE, CONTENT_TYPE);
461         
462         builder.field(Fields.ANALYZER, indexAnalyzer.name());
463         if (indexAnalyzer.name().equals(searchAnalyzer.name()) == false) {
464             builder.field(Fields.SEARCH_ANALYZER.getPreferredName(), searchAnalyzer.name());
465         }
466         builder.field(Fields.PAYLOADS, this.payloads);
467         builder.field(Fields.PRESERVE_SEPARATORS.getPreferredName(), this.preserveSeparators);
468         builder.field(Fields.PRESERVE_POSITION_INCREMENTS.getPreferredName(), this.preservePositionIncrements);
469         builder.field(Fields.MAX_INPUT_LENGTH.getPreferredName(), this.maxInputLength);
470         multiFields.toXContent(builder, params);
471 
472         if(!contextMapping.isEmpty()) {
473             builder.startObject(Fields.CONTEXT);
474             for (ContextMapping mapping : contextMapping.values()) {
475                 builder.value(mapping);
476             }
477             builder.endObject();
478         }
479 
480         return builder.endObject();
481     }

529     @Override
530     public void merge(Mapper mergeWith, MergeContext mergeContext) throws MergeMappingException {
531         super.merge(mergeWith, mergeContext);
532         CompletionFieldMapper fieldMergeWith = (CompletionFieldMapper) mergeWith;
533         if (payloads != fieldMergeWith.payloads) {
534             mergeContext.addConflict("mapper [" + names.fullName() + "] has different payload values");
535         }
536         if (preservePositionIncrements != fieldMergeWith.preservePositionIncrements) {
537             mergeContext.addConflict("mapper [" + names.fullName() + "] has different 'preserve_position_increments' values");
538         }
539         if (preserveSeparators != fieldMergeWith.preserveSeparators) {
540             mergeContext.addConflict("mapper [" + names.fullName() + "] has different 'preserve_separators' values");
541         }
542         if(!ContextMapping.mappingsAreEqual(getContextMapping(), fieldMergeWith.getContextMapping())) {
543             mergeContext.addConflict("mapper [" + names.fullName() + "] has different 'context_mapping' values");
544         }
545         if (!mergeContext.mergeFlags().simulate()) {
546             this.maxInputLength = fieldMergeWith.maxInputLength;
547         }

550     // this should be package private but our tests don't allow it.
551     public static boolean isReservedChar(char character) {
552         /*  we use 0x001F as a SEP_LABEL in the suggester but we can use the UTF-16 representation since they
553          *  are equivalent. We also don't need to convert the input character to UTF-8 here to check for
554          *  the 0x00 end label since all multi-byte  UTF-8 chars start with 0x10 binary so if the UTF-16 CP is == 0x00
555          *  it's the single byte UTF-8 CP */
556         assert XAnalyzingSuggester.PAYLOAD_SEP == XAnalyzingSuggester.SEP_LABEL; // ensure they are the same!
557         switch(character) {
558              case  XAnalyzingSuggester.END_BYTE:
559              case  XAnalyzingSuggester.SEP_LABEL:
560              case  XAnalyzingSuggester.HOLE_CHARACTER:
561              case  ContextMapping.SEPARATOR:
562                 return true;
563             default:
564                 return false;
565         }
