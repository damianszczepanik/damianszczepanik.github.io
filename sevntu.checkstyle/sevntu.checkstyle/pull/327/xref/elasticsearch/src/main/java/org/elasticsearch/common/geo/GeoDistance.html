35  /**
36   * Geo distance calculation.
37   */
38  public enum GeoDistance {
39      /**
40       * Calculates distance as points on a plane. Faster, but less accurate than {@link #ARC}.
41       */
42      PLANE() {
43          @Override
44          public double calculate(double sourceLatitude, double sourceLongitude, double targetLatitude, double targetLongitude, DistanceUnit unit) {
45              double px = targetLongitude - sourceLongitude;
46              double py = targetLatitude - sourceLatitude;
47              return Math.sqrt(px * px + py * py) * unit.getDistancePerDegree();
48          }
49  
50          @Override
51          public double normalize(double distance, DistanceUnit unit) {
52              return distance;
53          }
54  
55          @Override
56          public FixedSourceDistance fixedSourceDistance(double sourceLatitude, double sourceLongitude, DistanceUnit unit) {
57              return new PlaneFixedSourceDistance(sourceLatitude, sourceLongitude, unit);
58          }
59      },
60  
61      /**
62       * Calculates distance factor.
63       */
64      FACTOR() {
65          @Override
66          public double calculate(double sourceLatitude, double sourceLongitude, double targetLatitude, double targetLongitude, DistanceUnit unit) {
67              double longitudeDifference = targetLongitude - sourceLongitude;
68              double a = Math.toRadians(90D - sourceLatitude);
69              double c = Math.toRadians(90D - targetLatitude);
70              return (Math.cos(a) * Math.cos(c)) + (Math.sin(a) * Math.sin(c) * Math.cos(Math.toRadians(longitudeDifference)));
71          }
72  
73          @Override
74          public double normalize(double distance, DistanceUnit unit) {
75              return Math.cos(distance / unit.getEarthRadius());
76          }
77  
78          @Override
79          public FixedSourceDistance fixedSourceDistance(double sourceLatitude, double sourceLongitude, DistanceUnit unit) {
80              return new FactorFixedSourceDistance(sourceLatitude, sourceLongitude, unit);
81          }
82      },
83      /**
84       * Calculates distance as points on a globe.
85       */
86      ARC() {
87          @Override
88          public double calculate(double sourceLatitude, double sourceLongitude, double targetLatitude, double targetLongitude, DistanceUnit unit) {
89              double x1 = sourceLatitude * Math.PI / 180D;
90              double x2 = targetLatitude * Math.PI / 180D;
91              double h1 = 1D - Math.cos(x1 - x2);
92              double h2 = 1D - Math.cos((sourceLongitude - targetLongitude) * Math.PI / 180D);
93              double h = (h1 + Math.cos(x1) * Math.cos(x2) * h2) / 2;
94              double averageLatitude = (x1 + x2) / 2;
95              double diameter = GeoUtils.earthDiameter(averageLatitude);
96              return unit.fromMeters(diameter * Math.asin(Math.min(1, Math.sqrt(h))));
97          }
98  
99          @Override
100         public double normalize(double distance, DistanceUnit unit) {
101             return distance;
102         }
103 
104         @Override
105         public FixedSourceDistance fixedSourceDistance(double sourceLatitude, double sourceLongitude, DistanceUnit unit) {
106             return new ArcFixedSourceDistance(sourceLatitude, sourceLongitude, unit);
107         }
108     },
109     /**
110      * Calculates distance as points on a globe in a sloppy way. Close to the pole areas the accuracy
111      * of this function decreases.
112      */
113     SLOPPY_ARC() {
114 
115         @Override
116         public double normalize(double distance, DistanceUnit unit) {
117             return distance;
118         }
119 
120         @Override
121         public double calculate(double sourceLatitude, double sourceLongitude, double targetLatitude, double targetLongitude, DistanceUnit unit) {
122             return unit.fromMeters(SloppyMath.haversin(sourceLatitude, sourceLongitude, targetLatitude, targetLongitude) * 1000.0);
123         }
124 
125         @Override
126         public FixedSourceDistance fixedSourceDistance(double sourceLatitude, double sourceLongitude, DistanceUnit unit) {
127             return new SloppyArcFixedSourceDistance(sourceLatitude, sourceLongitude, unit);
128         }
129     };

143     private static final double MIN_LAT = Math.toRadians(-90d);  // -PI/2
144     private static final double MAX_LAT = Math.toRadians(90d);   //  PI/2
145     private static final double MIN_LON = Math.toRadians(-180d); // -PI
146     private static final double MAX_LON = Math.toRadians(180d);  //  PI
147 
148     public static DistanceBoundingCheck distanceBoundingCheck(double sourceLatitude, double sourceLongitude, double distance, DistanceUnit unit) {
149         // angular distance in radians on a great circle
150         // assume worst-case: use the minor axis
151         double radDist = unit.toMeters(distance) / GeoUtils.EARTH_SEMI_MINOR_AXIS;
152 
153         double radLat = Math.toRadians(sourceLatitude);
154         double radLon = Math.toRadians(sourceLongitude);
155 
156         double minLat = radLat - radDist;
157         double maxLat = radLat + radDist;
158 
159         double minLon, maxLon;
160         if (minLat > MIN_LAT && maxLat < MAX_LAT) {
161             double deltaLon = Math.asin(Math.sin(radDist) / Math.cos(radLat));
162             minLon = radLon - deltaLon;
163             if (minLon < MIN_LON) minLon += 2d * Math.PI;
164             maxLon = radLon + deltaLon;
165             if (maxLon > MAX_LON) maxLon -= 2d * Math.PI;
166         } else {
167             // a pole is within the distance
168             minLat = Math.max(minLat, MIN_LAT);
169             maxLat = Math.min(maxLat, MAX_LAT);
170             minLon = MIN_LON;
171             maxLon = MAX_LON;
172         }
173 
174         GeoPoint topLeft = new GeoPoint(Math.toDegrees(maxLat), Math.toDegrees(minLon));
175         GeoPoint bottomRight = new GeoPoint(Math.toDegrees(minLat), Math.toDegrees(maxLon));
176         if (minLon > maxLon) {
177             return new Meridian180DistanceBoundingCheck(topLeft, bottomRight);
178         }
179         return new SimpleDistanceBoundingCheck(topLeft, bottomRight);
180     }

182     /**
183      * Get a {@link GeoDistance} according to a given name. Valid values are
184      * 
185      * <ul>
186      *     <li><b>plane</b> for <code>GeoDistance.PLANE</code></li>
187      *     <li><b>sloppy_arc</b> for <code>GeoDistance.SLOPPY_ARC</code></li>
188      *     <li><b>factor</b> for <code>GeoDistance.FACTOR</code></li>
189      *     <li><b>arc</b> for <code>GeoDistance.ARC</code></li>
190      * </ul>
191      * 
192      * @param name name of the {@link GeoDistance}
193      * @return a {@link GeoDistance}
194      */
195     public static GeoDistance fromString(String name) {
196         name = name.toLowerCase(Locale.ROOT);
197         if ("plane".equals(name)) {
198             return PLANE;
199         } else if ("arc".equals(name)) {
200             return ARC;
201         } else if ("sloppy_arc".equals(name)) {
202             return SLOPPY_ARC;
203         } else if ("factor".equals(name)) {
204             return FACTOR;
205         }
206         throw new ElasticsearchIllegalArgumentException("No geo distance for [" + name + "]");
207     }

242     public static class Meridian180DistanceBoundingCheck implements DistanceBoundingCheck {
243 
244         private final GeoPoint topLeft;
245         private final GeoPoint bottomRight;
246 
247         public Meridian180DistanceBoundingCheck(GeoPoint topLeft, GeoPoint bottomRight) {
248             this.topLeft = topLeft;
249             this.bottomRight = bottomRight;
250         }
251 
252         @Override
253         public boolean isWithin(double targetLatitude, double targetLongitude) {
254             return (targetLatitude >= bottomRight.lat() && targetLatitude <= topLeft.lat()) &&
255                     (targetLongitude >= topLeft.lon() || targetLongitude <= bottomRight.lon());
256         }
257 
258         @Override
259         public GeoPoint topLeft() {
260             return topLeft;
261         }
262 
263         @Override
264         public GeoPoint bottomRight() {
265             return bottomRight;
266         }
267     }

295     public static class PlaneFixedSourceDistance implements FixedSourceDistance {
296 
297         private final double sourceLatitude;
298         private final double sourceLongitude;
299         private final double distancePerDegree;
300 
301         public PlaneFixedSourceDistance(double sourceLatitude, double sourceLongitude, DistanceUnit unit) {
302             this.sourceLatitude = sourceLatitude;
303             this.sourceLongitude = sourceLongitude;
304             this.distancePerDegree = unit.getDistancePerDegree();
305         }
306 
307         @Override
308         public double calculate(double targetLatitude, double targetLongitude) {
309             double px = targetLongitude - sourceLongitude;
310             double py = targetLatitude - sourceLatitude;
311             return Math.sqrt(px * px + py * py) * distancePerDegree;
312         }
313     }

315     public static class FactorFixedSourceDistance implements FixedSourceDistance {
316 
317         private final double sourceLongitude;
318 
319         private final double a;
320         private final double sinA;
321         private final double cosA;
322 
323         public FactorFixedSourceDistance(double sourceLatitude, double sourceLongitude, DistanceUnit unit) {
324             this.sourceLongitude = sourceLongitude;
325             this.a = Math.toRadians(90D - sourceLatitude);
326             this.sinA = Math.sin(a);
327             this.cosA = Math.cos(a);
328         }
329 
330         @Override
331         public double calculate(double targetLatitude, double targetLongitude) {
332             double longitudeDifference = targetLongitude - sourceLongitude;
333             double c = Math.toRadians(90D - targetLatitude);
334             return (cosA * Math.cos(c)) + (sinA * Math.sin(c) * Math.cos(Math.toRadians(longitudeDifference)));
335         }
336     }

380     /**
381      * Return a {@link SortedNumericDoubleValues} instance that returns the distances to a list of geo-points for each document.
382      */
383     public static SortedNumericDoubleValues distanceValues(final MultiGeoPointValues geoPointValues, final FixedSourceDistance... distances) {
384         final GeoPointValues singleValues = FieldData.unwrapSingleton(geoPointValues);
385         if (singleValues != null && distances.length == 1) {
386             final Bits docsWithField = FieldData.unwrapSingletonBits(geoPointValues);
387             return FieldData.singleton(new NumericDoubleValues() {
388 
389                 @Override
390                 public double get(int docID) {
391                     if (docsWithField != null && !docsWithField.get(docID)) {
392                         return 0d;
393                     }
394                     final GeoPoint point = singleValues.get(docID);
395                     return distances[0].calculate(point.lat(), point.lon());
396                 }
397 
398             }, docsWithField);
399         } else {
400             return new SortingNumericDoubleValues() {
401 
402                 @Override
403                 public void setDocument(int doc) {
404                     geoPointValues.setDocument(doc);
405                     resize(geoPointValues.count() * distances.length);
406                     int valueCounter = 0;
407                     for (FixedSourceDistance distance : distances) {
408                         for (int i = 0; i < geoPointValues.count(); ++i) {
409                             final GeoPoint point = geoPointValues.valueAt(i);
410                             values[valueCounter] = distance.calculate(point.lat(), point.lon());
411                             valueCounter++;
412                         }
413                     }
414                     sort();
415                 }
416             };
417         }
