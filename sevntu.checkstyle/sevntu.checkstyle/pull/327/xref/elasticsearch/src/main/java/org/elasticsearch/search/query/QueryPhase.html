20  package org.elasticsearch.search.query;
21  
22  import com.google.common.collect.ImmutableMap;
23  import org.apache.lucene.search.Query;
24  import org.apache.lucene.search.ScoreDoc;
25  import org.apache.lucene.search.TopDocs;
26  import org.apache.lucene.search.TotalHitCountCollector;
27  import org.elasticsearch.action.search.SearchType;
28  import org.elasticsearch.common.inject.Inject;
29  import org.elasticsearch.common.lucene.Lucene;
30  import org.elasticsearch.search.SearchParseElement;
31  import org.elasticsearch.search.SearchPhase;
32  import org.elasticsearch.search.aggregations.AggregationPhase;
33  import org.elasticsearch.search.internal.ContextIndexSearcher;
34  import org.elasticsearch.search.internal.SearchContext;
35  import org.elasticsearch.search.rescore.RescorePhase;
36  import org.elasticsearch.search.rescore.RescoreSearchContext;
37  import org.elasticsearch.search.sort.SortParseElement;
38  import org.elasticsearch.search.sort.TrackScoresParseElement;
39  import org.elasticsearch.search.suggest.SuggestPhase;
40  
41  import java.util.Map;
42  
43  /**
44   *
45   */
46  public class QueryPhase implements SearchPhase {
47  
48      private final AggregationPhase aggregationPhase;
49      private final SuggestPhase suggestPhase;
50      private RescorePhase rescorePhase;
51  
52      @Inject
53      public QueryPhase(AggregationPhase aggregationPhase, SuggestPhase suggestPhase, RescorePhase rescorePhase) {
54          this.aggregationPhase = aggregationPhase;
55          this.suggestPhase = suggestPhase;
56          this.rescorePhase = rescorePhase;
57      }
58  
59      @Override
60      public Map<String, ? extends SearchParseElement> parseElements() {
61          ImmutableMap.Builder<String, SearchParseElement> parseElements = ImmutableMap.builder();
62          parseElements.put("from", new FromParseElement()).put("size", new SizeParseElement())
63                  .put("indices_boost", new IndicesBoostParseElement())
64                  .put("indicesBoost", new IndicesBoostParseElement())
65                  .put("query", new QueryParseElement())
66                  .put("queryBinary", new QueryBinaryParseElement())
67                  .put("query_binary", new QueryBinaryParseElement())
68                  .put("filter", new PostFilterParseElement()) // For bw comp reason, should be removed in version 1.1
69                  .put("post_filter", new PostFilterParseElement())
70                  .put("postFilter", new PostFilterParseElement())
71                  .put("filterBinary", new FilterBinaryParseElement())
72                  .put("filter_binary", new FilterBinaryParseElement())
73                  .put("sort", new SortParseElement())
74                  .put("trackScores", new TrackScoresParseElement())
75                  .put("track_scores", new TrackScoresParseElement())
76                  .put("min_score", new MinScoreParseElement())
77                  .put("minScore", new MinScoreParseElement())
78                  .put("timeout", new TimeoutParseElement())
79                  .put("terminate_after", new TerminateAfterParseElement())
80                  .putAll(aggregationPhase.parseElements())
81                  .putAll(suggestPhase.parseElements())
82                  .putAll(rescorePhase.parseElements());
83          return parseElements.build();
84      }
85  
86      @Override
87      public void preProcess(SearchContext context) {
88          context.preProcess();
89      }
90  
91      @Override
92      public void execute(SearchContext searchContext) throws QueryPhaseExecutionException {
93          // Pre-process aggregations as late as possible. In the case of a DFS_Q_T_F
94          // request, preProcess is called on the DFS phase phase, this is why we pre-process them
95          // here to make sure it happens during the QUERY phase
96          aggregationPhase.preProcess(searchContext);
97  
98          searchContext.queryResult().searchTimedOut(false);
99  
100         searchContext.searcher().inStage(ContextIndexSearcher.Stage.MAIN_QUERY);
101         boolean rescore = false;
102         try {
103             searchContext.queryResult().from(searchContext.from());
104             searchContext.queryResult().size(searchContext.size());
105 
106             Query query = searchContext.query();
107 
108             TopDocs topDocs;
109             int numDocs = searchContext.from() + searchContext.size();
110 
111             if (searchContext.searchType() == SearchType.COUNT || numDocs == 0) {
112                 TotalHitCountCollector collector = new TotalHitCountCollector();
113                 searchContext.searcher().search(query, collector);
114                 topDocs = new TopDocs(collector.getTotalHits(), Lucene.EMPTY_SCORE_DOCS, 0);
115             } else if (searchContext.searchType() == SearchType.SCAN) {
116                 topDocs = searchContext.scanContext().execute(searchContext);
117             } else {
118                 // Perhaps have a dedicated scroll phase?
119                 if (searchContext.request().scroll() != null) {
120                     numDocs = searchContext.size();
121                     ScoreDoc lastEmittedDoc = searchContext.lastEmittedDoc();
122                     if (searchContext.sort() != null) {
123                         topDocs = searchContext.searcher().searchAfter(
124                                 lastEmittedDoc, query, null, numDocs, searchContext.sort(),
125                                 searchContext.trackScores(), searchContext.trackScores()
126                         );
127                     } else {
128                         rescore = !searchContext.rescore().isEmpty();
129                         for (RescoreSearchContext rescoreContext : searchContext.rescore()) {
130                             numDocs = Math.max(rescoreContext.window(), numDocs);
131                         }
132                         topDocs = searchContext.searcher().searchAfter(lastEmittedDoc, query, numDocs);
133                     }
134 
135                     int size = topDocs.scoreDocs.length;
136                     if (size > 0) {
137                         // In the case of *QUERY_AND_FETCH we don't get back to shards telling them which least
138                         // relevant docs got emitted as hit, we can simply mark the last doc as last emitted
139                         if (searchContext.searchType() == SearchType.QUERY_AND_FETCH ||
140                                 searchContext.searchType() == SearchType.DFS_QUERY_AND_FETCH) {
141                             searchContext.lastEmittedDoc(topDocs.scoreDocs[size - 1]);
142                         }
143                     }
144                 } else {
145                     if (searchContext.sort() != null) {
146                         topDocs = searchContext.searcher().search(query, null, numDocs, searchContext.sort(),
147                                 searchContext.trackScores(), searchContext.trackScores());
148                     } else {
149                         rescore = !searchContext.rescore().isEmpty();
150                         for (RescoreSearchContext rescoreContext : searchContext.rescore()) {
151                             numDocs = Math.max(rescoreContext.window(), numDocs);
152                         }
153                         topDocs = searchContext.searcher().search(query, numDocs);
154                     }
155                 }
156             }
157             searchContext.queryResult().topDocs(topDocs);
158         } catch (Throwable e) {
159             throw new QueryPhaseExecutionException(searchContext, "Failed to execute main query", e);
160         } finally {
161             searchContext.searcher().finishStage(ContextIndexSearcher.Stage.MAIN_QUERY);
162         }
163         if (rescore) { // only if we do a regular search
164             rescorePhase.execute(searchContext);
165         }
166         suggestPhase.execute(searchContext);
167         aggregationPhase.execute(searchContext);
168     }
169 }
