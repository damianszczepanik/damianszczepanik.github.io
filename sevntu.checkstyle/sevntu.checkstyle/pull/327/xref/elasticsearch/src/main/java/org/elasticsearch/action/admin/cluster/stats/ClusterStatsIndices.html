58      public ClusterStatsIndices(ClusterStatsNodeResponse[] nodeResponses) {
59          ObjectObjectOpenHashMap<String, ShardStats> countsPerIndex = new ObjectObjectOpenHashMap<>();
60  
61          this.docs = new DocsStats();
62          this.store = new StoreStats();
63          this.fieldData = new FieldDataStats();
64          this.filterCache = new FilterCacheStats();
65          this.idCache = new IdCacheStats();
66          this.completion = new CompletionStats();
67          this.segments = new SegmentsStats();
68          this.percolate = new PercolateStats();
69  
70          for (ClusterStatsNodeResponse r : nodeResponses) {
71              for (org.elasticsearch.action.admin.indices.stats.ShardStats shardStats : r.shardsStats()) {
72                  ShardStats indexShardStats = countsPerIndex.get(shardStats.getIndex());
73                  if (indexShardStats == null) {
74                      indexShardStats = new ShardStats();
75                      countsPerIndex.put(shardStats.getIndex(), indexShardStats);
76                  }
77  
78                  indexShardStats.total++;
79  
80                  CommonStats shardCommonStats = shardStats.getStats();
81  
82                  if (shardStats.getShardRouting().primary()) {
83                      indexShardStats.primaries++;
84                      docs.add(shardCommonStats.docs);
85                  }
86                  store.add(shardCommonStats.store);
87                  fieldData.add(shardCommonStats.fieldData);
88                  filterCache.add(shardCommonStats.filterCache);
89                  idCache.add(shardCommonStats.idCache);
90                  completion.add(shardCommonStats.completion);
91                  segments.add(shardCommonStats.segments);
92                  percolate.add(shardCommonStats.percolate);
93              }
94          }
95  
96          shards = new ShardStats();
97          indexCount = countsPerIndex.size();
98          for (ObjectObjectCursor<String, ShardStats> indexCountsCursor : countsPerIndex) {
99              shards.addIndexShardCount(indexCountsCursor.value);
100         }
101     }

196     public static class ShardStats implements ToXContent, Streamable {
197 
198         int indices;
199         int total;
200         int primaries;
201 
202         // min/max
203         int minIndexShards = -1;
204         int maxIndexShards = -1;
205         int minIndexPrimaryShards = -1;
206         int maxIndexPrimaryShards = -1;
207         double minIndexReplication = -1;
208         double totalIndexReplication = 0;
209         double maxIndexReplication = -1;
210 
211         public ShardStats() {
212         }
213 
214         /**
215          * number of indices in the cluster
216          */
217         public int getIndices() {
218             return this.indices;
219         }
220 
221         /**
222          * total number of shards in the cluster
223          */
224         public int getTotal() {
225             return this.total;
226         }
227 
228         /**
229          * total number of primary shards in the cluster
230          */
231         public int getPrimaries() {
232             return this.primaries;
233         }
234 
235         /**
236          * returns how many *redundant* copies of the data the cluster holds - running with no replicas will return 0
237          */
238         public double getReplication() {
239             if (primaries == 0) {
240                 return 0;
241             }
242             return (((double) (total - primaries)) / primaries);
243         }
244 
245         /**
246          * the maximum number of shards (primary+replicas) an index has
247          */
248         public int getMaxIndexShards() {
249             return this.maxIndexShards;
250         }
251 
252         /**
253          * the minimum number of shards (primary+replicas) an index has
254          */
255         public int getMinIndexShards() {
256             return this.minIndexShards;
257         }
258 
259         /**
260          * average number of shards (primary+replicas) across the indices
261          */
262         public double getAvgIndexShards() {
263             if (this.indices == 0) {
264                 return -1;
265             }
266             return ((double) this.total) / this.indices;
267         }
268 
269         /**
270          * the maximum number of primary shards an index has
271          */
272         public int getMaxIndexPrimaryShards() {
273             return this.maxIndexPrimaryShards;
274         }
275 
276         /**
277          * the minimum number of primary shards an index has
278          */
279         public int getMinIndexPrimaryShards() {
280             return this.minIndexPrimaryShards;
281         }
282 
283         /**
284          * the average number primary shards across the indices
285          */
286         public double getAvgIndexPrimaryShards() {
287             if (this.indices == 0) {
288                 return -1;
289             }
290             return ((double) this.primaries) / this.indices;
291         }
292 
293         /**
294          * minimum replication factor across the indices. See {@link #getReplication}
295          */
296         public double getMinIndexReplication() {
297             return this.minIndexReplication;
298         }
299 
300         /**
301          * average replication factor across the indices. See {@link #getReplication}
302          */
303         public double getAvgIndexReplication() {
304             if (indices == 0) {
305                 return -1;
306             }
307             return this.totalIndexReplication / this.indices;
308         }
309 
310         /**
311          * maximum replication factor across the indices. See {@link #getReplication
312          */
313         public double getMaxIndexReplication() {
314             return this.maxIndexReplication;
315         }
316 
317         public void addIndexShardCount(ShardStats indexShardCount) {
318             this.indices++;
319             this.primaries += indexShardCount.primaries;
320             this.total += indexShardCount.total;
321             this.totalIndexReplication += indexShardCount.getReplication();
322             if (this.indices == 1) {
323                 // first index, uninitialized.
324                 minIndexPrimaryShards = indexShardCount.primaries;
325                 maxIndexPrimaryShards = indexShardCount.primaries;
326                 minIndexShards = indexShardCount.total;
327                 maxIndexShards = indexShardCount.total;
328                 minIndexReplication = indexShardCount.getReplication();
329                 maxIndexReplication = minIndexReplication;
330             } else {
331                 minIndexShards = Math.min(minIndexShards, indexShardCount.total);
332                 minIndexPrimaryShards = Math.min(minIndexPrimaryShards, indexShardCount.primaries);
333                 minIndexReplication = Math.min(minIndexReplication, indexShardCount.getReplication());
334 
335                 maxIndexShards = Math.max(maxIndexShards, indexShardCount.total);
336                 maxIndexPrimaryShards = Math.max(maxIndexPrimaryShards, indexShardCount.primaries);
337                 maxIndexReplication = Math.max(maxIndexReplication, indexShardCount.getReplication());
338             }
339         }
340 
341         public static ShardStats readShardStats(StreamInput in) throws IOException {
342             ShardStats c = new ShardStats();
343             c.readFrom(in);
344             return c;
345         }

347         @Override
348         public void readFrom(StreamInput in) throws IOException {
349             indices = in.readVInt();
350             total = in.readVInt();
351             primaries = in.readVInt();
352             minIndexShards = in.readVInt();
353             maxIndexShards = in.readVInt();
354             minIndexPrimaryShards = in.readVInt();
355             maxIndexPrimaryShards = in.readVInt();
356             minIndexReplication = in.readDouble();
357             totalIndexReplication = in.readDouble();
358             maxIndexReplication = in.readDouble();
359         }
360 
361         @Override
362         public void writeTo(StreamOutput out) throws IOException {
363             out.writeVInt(indices);
364             out.writeVInt(total);
365             out.writeVInt(primaries);
366             out.writeVInt(minIndexShards);
367             out.writeVInt(maxIndexShards);
368             out.writeVInt(minIndexPrimaryShards);
369             out.writeVInt(maxIndexPrimaryShards);
370             out.writeDouble(minIndexReplication);
371             out.writeDouble(totalIndexReplication);
372             out.writeDouble(maxIndexReplication);
373         }

375         static final class Fields {
376             static final XContentBuilderString SHARDS = new XContentBuilderString("shards");
377             static final XContentBuilderString TOTAL = new XContentBuilderString("total");
378             static final XContentBuilderString PRIMARIES = new XContentBuilderString("primaries");
379             static final XContentBuilderString REPLICATION = new XContentBuilderString("replication");
380             static final XContentBuilderString MIN = new XContentBuilderString("min");
381             static final XContentBuilderString MAX = new XContentBuilderString("max");
382             static final XContentBuilderString AVG = new XContentBuilderString("avg");
383             static final XContentBuilderString INDEX = new XContentBuilderString("index");
384         }
385 
386         private void addIntMinMax(XContentBuilderString field, int min, int max, double avg, XContentBuilder builder) throws IOException {
387             builder.startObject(field);
388             builder.field(Fields.MIN, min);
389             builder.field(Fields.MAX, max);
390             builder.field(Fields.AVG, avg);
391             builder.endObject();
392         }
393 
394         private void addDoubleMinMax(XContentBuilderString field, double min, double max, double avg, XContentBuilder builder) throws IOException {
395             builder.startObject(field);
396             builder.field(Fields.MIN, min);
397             builder.field(Fields.MAX, max);
398             builder.field(Fields.AVG, avg);
399             builder.endObject();
400         }
401 
402         @Override
403         public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
404             builder.startObject(Fields.SHARDS);
405             if (indices > 0) {
406 
407                 builder.field(Fields.TOTAL, total);
408                 builder.field(Fields.PRIMARIES, primaries);
409                 builder.field(Fields.REPLICATION, getReplication());
410 
411                 builder.startObject(Fields.INDEX);
412                 addIntMinMax(Fields.SHARDS, minIndexShards, maxIndexShards, getAvgIndexShards(), builder);
413                 addIntMinMax(Fields.PRIMARIES, minIndexPrimaryShards, maxIndexPrimaryShards, getAvgIndexPrimaryShards(), builder);
414                 addDoubleMinMax(Fields.REPLICATION, minIndexReplication, maxIndexReplication, getAvgIndexReplication(), builder);
415                 builder.endObject();
416             }
417             builder.endObject();
418             return builder;
419         }
420 
421         @Override
422         public String toString() {
423             return "total [" + total + "] primaries [" + primaries + "]";
424         }
425     }
