49      public static enum Stage {
50          INIT((byte) 0),
51  
52          /** recovery of lucene files, either reusing local ones are copying new ones */
53          INDEX((byte) 1),
54  
55          /** starting up the engine, potentially running checks */
56          START((byte) 2),
57  
58          /** replaying the translog */
59          TRANSLOG((byte) 3),
60  
61          /** performing final task after all translog ops have been done */
62          FINALIZE((byte) 4),
63  
64          DONE((byte) 5);
65  
66          private static final Stage[] STAGES = new Stage[Stage.values().length];
67  
68          static {
69              for (Stage stage : Stage.values()) {
70                  assert stage.id() < STAGES.length && stage.id() >= 0;
71                  STAGES[stage.id] = stage;
72              }
73          }
74  
75          private final byte id;
76  
77          Stage(byte id) {
78              this.id = id;
79          }
80  
81          public byte id() {
82              return id;
83          }
84  
85          public static Stage fromId(byte id) throws ElasticsearchIllegalArgumentException {
86              if (id < 0 || id >= STAGES.length) {
87                  throw new ElasticsearchIllegalArgumentException("No mapping for id [" + id + "]");
88              }
89              return STAGES[id];
90          }
91      }

171     private void validateAndSetStage(Stage expected, Stage next) {
172         if (stage != expected) {
173             throw new ElasticsearchIllegalStateException("can't move recovery to stage [" + next + "]. current stage: ["
174                     + stage + "] (expected [" + expected + "])");
175         }
176         stage = next;
177     }

179     // synchronized is strictly speaking not needed (this is called by a single thread), but just to be safe
180     public synchronized RecoveryState setStage(Stage stage) {
181         switch (stage) {
182             case INIT:
183                 // reinitializing stop remove all state except for start time
184                 this.stage = Stage.INIT;
185                 getIndex().reset();
186                 getStart().reset();
187                 getTranslog().reset();
188                 break;
189             case INDEX:
190                 validateAndSetStage(Stage.INIT, stage);
191                 getIndex().start();
192                 break;
193             case START:
194                 validateAndSetStage(Stage.INDEX, stage);
195                 getIndex().stop();
196                 getStart().start();
197                 break;
198             case TRANSLOG:
199                 validateAndSetStage(Stage.START, stage);
200                 getStart().stop();
201                 getTranslog().start();
202                 break;
203             case FINALIZE:
204                 validateAndSetStage(Stage.TRANSLOG, stage);
205                 getTranslog().stop();
206                 break;
207             case DONE:
208                 validateAndSetStage(Stage.FINALIZE, stage);
209                 getTimer().stop();
210                 break;
211             default:
212                 throw new ElasticsearchIllegalArgumentException("unknown RecoveryState.Stage [" + stage + "]");
213         }
214         return this;
215     }

259     @Override
260     public synchronized void readFrom(StreamInput in) throws IOException {
261         timer.readFrom(in);
262         type = Type.fromId(in.readByte());
263         stage = Stage.fromId(in.readByte());
264         shardId = ShardId.readShardId(in);
265         restoreSource = RestoreSource.readOptionalRestoreSource(in);
266         targetNode = DiscoveryNode.readNode(in);
267         if (in.readBoolean()) {
268             sourceNode = DiscoveryNode.readNode(in);
269         }
270         index.readFrom(in);
271         translog.readFrom(in);
272         start.readFrom(in);
273         primary = in.readBoolean();
274     }

276     @Override
277     public void writeTo(StreamOutput out) throws IOException {
278         timer.writeTo(out);
279         out.writeByte(type.id());
280         out.writeByte(stage.id());
281         shardId.writeTo(out);
282         out.writeOptionalStreamable(restoreSource);
283         targetNode.writeTo(out);
284         out.writeBoolean(sourceNode != null);
285         if (sourceNode != null) {
286             sourceNode.writeTo(out);
287         }
288         index.writeTo(out);
289         translog.writeTo(out);
290         start.writeTo(out);
291         out.writeBoolean(primary);
292     }

294     @Override
295     public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
296 
297         builder.field(Fields.ID, shardId.id());
298         builder.field(Fields.TYPE, type.toString());
299         builder.field(Fields.STAGE, stage.toString());
300         builder.field(Fields.PRIMARY, primary);
301         builder.dateValueField(Fields.START_TIME_IN_MILLIS, Fields.START_TIME, timer.startTime);
302         if (timer.stopTime > 0) {
303             builder.dateValueField(Fields.STOP_TIME_IN_MILLIS, Fields.STOP_TIME, timer.stopTime);
304         }
305         builder.timeValueField(Fields.TOTAL_TIME_IN_MILLIS, Fields.TOTAL_TIME, timer.time());
306 
307         if (restoreSource != null) {
308             builder.field(Fields.SOURCE);
309             restoreSource.toXContent(builder, params);
310         } else {
311             builder.startObject(Fields.SOURCE);
312             builder.field(Fields.ID, sourceNode.id());
313             builder.field(Fields.HOST, sourceNode.getHostName());
314             builder.field(Fields.TRANSPORT_ADDRESS, sourceNode.address().toString());
315             builder.field(Fields.IP, sourceNode.getHostAddress());
316             builder.field(Fields.NAME, sourceNode.name());
317             builder.endObject();
318         }
319 
320         builder.startObject(Fields.TARGET);
321         builder.field(Fields.ID, targetNode.id());
322         builder.field(Fields.HOST, targetNode.getHostName());
323         builder.field(Fields.TRANSPORT_ADDRESS, targetNode.address().toString());
324         builder.field(Fields.IP, targetNode.getHostAddress());
325         builder.field(Fields.NAME, targetNode.name());
326         builder.endObject();
327 
328         builder.startObject(Fields.INDEX);
329         index.toXContent(builder, params);
330         builder.endObject();
331 
332         builder.startObject(Fields.TRANSLOG);
333         translog.toXContent(builder, params);
334         builder.endObject();
335 
336         builder.startObject(Fields.START);
337         start.toXContent(builder, params);
338         builder.endObject();
339 
340         return builder;
341     }

379     public static class Timer implements Streamable {
380         protected long startTime = 0;
381         protected long time = -1;
382         protected long stopTime = 0;
383 
384         public synchronized void start() {
385             assert startTime == 0 : "already started";
386             startTime = System.currentTimeMillis();
387         }
388 
389         public synchronized long startTime() {
390             return startTime;
391         }
392 
393         public synchronized long time() {
394             if (startTime == 0) {
395                 return 0;
396             }
397             if (time >= 0) {
398                 return time;
399             }
400             return Math.max(0, System.currentTimeMillis() - startTime);
401         }
402 
403         public synchronized long stopTime() {
404             return stopTime;
405         }
406 
407         public synchronized void stop() {
408             assert stopTime == 0 : "already stopped";
409             stopTime = Math.max(System.currentTimeMillis(), startTime);
410             time = stopTime - startTime;
411             assert time >= 0;
412         }
413 
414         public synchronized void reset() {
415             startTime = 0;
416             time = -1;
417             stopTime = 0;
418         }
419 
420 
421         @Override
422         public synchronized void readFrom(StreamInput in) throws IOException {
423             startTime = in.readVLong();
424             stopTime = in.readVLong();
425             time = in.readVLong();
426         }
427 
428         @Override
429         public synchronized void writeTo(StreamOutput out) throws IOException {
430             out.writeVLong(startTime);
431             out.writeVLong(stopTime);
432             // write a snapshot of current time, which is not per se the time field
433             out.writeVLong(time());
434         }
435 
436     }

515     public static class File implements ToXContent, Streamable {
516         private String name;
517         private long length;
518         private long recovered;
519         private boolean reused;
520 
521         public File() {
522         }
523 
524         public File(String name, long length, boolean reused) {
525             assert name != null;
526             this.name = name;
527             this.length = length;
528             this.reused = reused;
529         }
530 
531         void addRecoveredBytes(long bytes) {
532             assert reused == false : "file is marked as reused, can't update recovered bytes";
533             assert bytes >= 0 : "can't recovered negative bytes. got [" + bytes + "]";
534             recovered += bytes;
535         }
536 
537         /** file name * */
538         public String name() {
539             return name;
540         }
541 
542         /** file length * */
543         public long length() {
544             return length;
545         }
546 
547         /** number of bytes recovered for this file (so far). 0 if the file is reused * */
548         public long recovered() {
549             return recovered;
550         }
551 
552         /** returns true if the file is reused from a local copy */
553         public boolean reused() {
554             return reused;
555         }
556 
557         boolean fullyRecovered() {
558             return reused == false && length == recovered;
559         }
560 
561         public static File readFile(StreamInput in) throws IOException {
562             File file = new File();
563             file.readFrom(in);
564             return file;
565         }
566 
567         @Override
568         public void readFrom(StreamInput in) throws IOException {
569             name = in.readString();
570             length = in.readVLong();
571             recovered = in.readVLong();
572             reused = in.readBoolean();
573         }
574 
575         @Override
576         public void writeTo(StreamOutput out) throws IOException {
577             out.writeString(name);
578             out.writeVLong(length);
579             out.writeVLong(recovered);
580             out.writeBoolean(reused);
581         }
582 
583         @Override
584         public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
585             builder.startObject();
586             builder.field(Fields.NAME, name);
587             builder.byteSizeField(Fields.LENGTH_IN_BYTES, Fields.LENGTH, length);
588             builder.field(Fields.REUSED, reused);
589             builder.byteSizeField(Fields.RECOVERED_IN_BYTES, Fields.RECOVERED, length);
590             builder.endObject();
591             return builder;
592         }
593 
594         @Override
595         public boolean equals(Object obj) {
596             if (obj instanceof File) {
597                 File other = (File) obj;
598                 return name.equals(other.name) && length == other.length() && reused == other.reused() && recovered == other.recovered();
599             }
600             return false;
601         }
602 
603         @Override
604         public String toString() {
605             return "file (name [" + name + "], reused [" + reused + "], length [" + length + "], recovered [" + recovered + "])";
606         }
607     }

629             assert existing == null : "file [" + name + "] is already reported";
630         }

632         public void addRecoveredBytesToFile(String name, long bytes) {
633             File file = fileDetails.get(name);
634             file.addRecoveredBytes(bytes);
635         }
636 
637         public long version() {
638             return this.version;
639         }
640 
641         /** total number of files that are part of this recovery, both re-used and recovered */
642         public int totalFileCount() {
643             return fileDetails.size();
644         }
645 
646         /** total number of files to be recovered (potentially not yet done) */
647         public int totalRecoverFiles() {
648             int total = 0;
649             for (File file : fileDetails.values()) {
650                 if (file.reused() == false) {
651                     total++;
652                 }
653             }
654             return total;
655         }
656 
657 
658         /** number of file that were recovered (excluding on ongoing files) */
659         public int recoveredFileCount() {
660             int count = 0;
661             for (File file : fileDetails.values()) {
662                 if (file.fullyRecovered()) {
663                     count++;
664                 }
665             }
666             return count;
667         }
668 
669         /** percent of recovered (i.e., not reused) files out of the total files to be recovered */
670         public float recoveredFilesPercent() {
671             int total = 0;
672             int recovered = 0;
673             for (File file : fileDetails.values()) {
674                 if (file.reused() == false) {
675                     total++;
676                     if (file.fullyRecovered()) {
677                         recovered++;
678                     }
679                 }
680             }
681             if (total == 0 && fileDetails.size() == 0) {      // indicates we are still in init phase
682                 return 0.0f;
683             }
684             if (total == recovered) {
685                 return 100.0f;
686             } else {
687                 float result = 100.0f * (recovered / (float) total);
688                 return result;
689             }
690         }
691 
692         /** total number of bytes in th shard */
693         public long totalBytes() {
694             long total = 0;
695             for (File file : fileDetails.values()) {
696                 total += file.length();
697             }
698             return total;
699         }
700 
701         /** total number of bytes recovered so far, including both existing and reused */
702         public long recoveredBytes() {
703             long recovered = 0;
704             for (File file : fileDetails.values()) {
705                 recovered += file.recovered();
706             }
707             return recovered;
708         }
709 
710         /** total bytes of files to be recovered (potentially not yet done) */
711         public long totalRecoverBytes() {
712             long total = 0;
713             for (File file : fileDetails.values()) {
714                 if (file.reused() == false) {
715                     total += file.length();
716                 }
717             }
718             return total;
719         }
720 
721         public long totalReuseBytes() {
722             long total = 0;
723             for (File file : fileDetails.values()) {
724                 if (file.reused()) {
725                     total += file.length();
726                 }
727             }
728             return total;
729         }
730 
731         /** percent of bytes recovered out of total files bytes *to be* recovered */
732         public float recoveredBytesPercent() {
733             long total = 0;
734             long recovered = 0;
735             for (File file : fileDetails.values()) {
736                 if (file.reused() == false) {
737                     total += file.length();
738                     recovered += file.recovered();
739                 }
740             }
741             if (total == 0 && fileDetails.size() == 0) {
742                 // indicates we are still in init phase
743                 return 0.0f;
744             }
745             if (total == recovered) {
746                 return 100.0f;
747             } else {
748                 return 100.0f * recovered / total;
749             }
750         }
751 
752         public int reusedFileCount() {
753             int reused = 0;
754             for (File file : fileDetails.values()) {
755                 if (file.reused()) {
756                     reused++;
757                 }
758             }
759             return reused;
760         }

763             long reused = 0;
764             for (File file : fileDetails.values()) {

766                     reused += file.length();
767                 }
768             }

796         @Override
797         public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
798             // stream size first, as it matters more and the files section can be long
799             builder.startObject(Fields.SIZE);
800             builder.byteSizeField(Fields.TOTAL_IN_BYTES, Fields.TOTAL, totalBytes());
801             builder.byteSizeField(Fields.REUSED_IN_BYTES, Fields.REUSED, totalBytes());
802             builder.byteSizeField(Fields.RECOVERED_IN_BYTES, Fields.RECOVERED, recoveredBytes());
803             builder.field(Fields.PERCENT, String.format(Locale.ROOT, "%1.1f%%", recoveredBytesPercent()));
804             builder.endObject();
805 
806             builder.startObject(Fields.FILES);
807             builder.field(Fields.TOTAL, totalFileCount());
808             builder.field(Fields.REUSED, reusedFileCount());
809             builder.field(Fields.RECOVERED, recoveredFileCount());
810             builder.field(Fields.PERCENT, String.format(Locale.ROOT, "%1.1f%%", recoveredFilesPercent()));
811             if (params.paramAsBoolean("details", false)) {
812                 builder.startArray(Fields.DETAILS);
813                 for (File file : fileDetails.values()) {
814                     file.toXContent(builder, params);
815                 }
816                 builder.endArray();
817             }
818             builder.endObject();
819             builder.timeValueField(Fields.TOTAL_TIME_IN_MILLIS, Fields.TOTAL_TIME, time());
820             return builder;
821         }
