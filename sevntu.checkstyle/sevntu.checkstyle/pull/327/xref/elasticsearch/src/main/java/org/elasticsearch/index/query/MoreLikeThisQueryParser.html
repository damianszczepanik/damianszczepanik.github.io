58      public static final String NAME = "mlt";
59      private MoreLikeThisFetchService fetchService = null;

61      public static class Fields {
62          public static final ParseField LIKE_TEXT = new ParseField("like_text").withAllDeprecated("like");
63          public static final ParseField MIN_TERM_FREQ = new ParseField("min_term_freq");
64          public static final ParseField MAX_QUERY_TERMS = new ParseField("max_query_terms");
65          public static final ParseField MIN_WORD_LENGTH = new ParseField("min_word_length", "min_word_len");
66          public static final ParseField MAX_WORD_LENGTH = new ParseField("max_word_length", "max_word_len");
67          public static final ParseField MIN_DOC_FREQ = new ParseField("min_doc_freq");
68          public static final ParseField MAX_DOC_FREQ = new ParseField("max_doc_freq");
69          public static final ParseField BOOST_TERMS = new ParseField("boost_terms");
70          public static final ParseField MINIMUM_SHOULD_MATCH = new ParseField("minimum_should_match");
71          public static final ParseField PERCENT_TERMS_TO_MATCH = new ParseField("percent_terms_to_match").withAllDeprecated("minimum_should_match");
72          public static final ParseField FAIL_ON_UNSUPPORTED_FIELD = new ParseField("fail_on_unsupported_field");
73          public static final ParseField STOP_WORDS = new ParseField("stop_words");
74          public static final ParseField DOCUMENT_IDS = new ParseField("ids").withAllDeprecated("like");
75          public static final ParseField DOCUMENTS = new ParseField("docs").withAllDeprecated("like");
76          public static final ParseField LIKE = new ParseField("like");
77          public static final ParseField IGNORE_LIKE = new ParseField("ignore_like");
78          public static final ParseField INCLUDE = new ParseField("include");
79      }

95      @Override
96      public Query parse(QueryParseContext parseContext) throws IOException, QueryParsingException {
97          XContentParser parser = parseContext.parser();
98  
99          MoreLikeThisQuery mltQuery = new MoreLikeThisQuery();
100         mltQuery.setSimilarity(parseContext.searchSimilarity());
101         Analyzer analyzer = null;
102         List<String> moreLikeFields = null;
103         boolean failOnUnsupportedField = true;
104         String queryName = null;
105         boolean include = false;
106 
107         XContentParser.Token token;
108         String currentFieldName = null;
109 
110         List<String> likeTexts = new ArrayList<>();
111         MultiTermVectorsRequest likeItems = new MultiTermVectorsRequest();
112 
113         List<String> ignoreTexts = new ArrayList<>();
114         MultiTermVectorsRequest ignoreItems = new MultiTermVectorsRequest();
115 
116         while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
117             if (token == XContentParser.Token.FIELD_NAME) {
118                 currentFieldName = parser.currentName();
119             } else if (token.isValue()) {
120                 if (Fields.LIKE_TEXT.match(currentFieldName, parseContext.parseFlags())) {
121                     likeTexts.add(parser.text());
122                 } else if (Fields.LIKE.match(currentFieldName, parseContext.parseFlags())) {
123                     parseLikeField(parser, likeTexts, likeItems);
124                 } else if (Fields.IGNORE_LIKE.match(currentFieldName, parseContext.parseFlags())) {
125                     parseLikeField(parser, ignoreTexts, ignoreItems);
126                 } else if (Fields.MIN_TERM_FREQ.match(currentFieldName, parseContext.parseFlags())) {
127                     mltQuery.setMinTermFrequency(parser.intValue());
128                 } else if (Fields.MAX_QUERY_TERMS.match(currentFieldName, parseContext.parseFlags())) {
129                     mltQuery.setMaxQueryTerms(parser.intValue());
130                 } else if (Fields.MIN_DOC_FREQ.match(currentFieldName, parseContext.parseFlags())) {
131                     mltQuery.setMinDocFreq(parser.intValue());
132                 } else if (Fields.MAX_DOC_FREQ.match(currentFieldName, parseContext.parseFlags())) {
133                     mltQuery.setMaxDocFreq(parser.intValue());
134                 } else if (Fields.MIN_WORD_LENGTH.match(currentFieldName, parseContext.parseFlags())) {
135                     mltQuery.setMinWordLen(parser.intValue());
136                 } else if (Fields.MAX_WORD_LENGTH.match(currentFieldName, parseContext.parseFlags())) {
137                     mltQuery.setMaxWordLen(parser.intValue());
138                 } else if (Fields.BOOST_TERMS.match(currentFieldName, parseContext.parseFlags())) {
139                     float boostFactor = parser.floatValue();
140                     if (boostFactor != 0) {
141                         mltQuery.setBoostTerms(true);
142                         mltQuery.setBoostTermsFactor(boostFactor);
143                     }
144                 } else if (Fields.MINIMUM_SHOULD_MATCH.match(currentFieldName, parseContext.parseFlags())) {
145                     mltQuery.setMinimumShouldMatch(parser.text());
146                 } else if (Fields.PERCENT_TERMS_TO_MATCH.match(currentFieldName, parseContext.parseFlags())) {
147                     mltQuery.setMinimumShouldMatch(Math.round(parser.floatValue() * 100) + "%");
148                 } else if ("analyzer".equals(currentFieldName)) {
149                     analyzer = parseContext.analysisService().analyzer(parser.text());
150                 } else if ("boost".equals(currentFieldName)) {
151                     mltQuery.setBoost(parser.floatValue());
152                 } else if (Fields.FAIL_ON_UNSUPPORTED_FIELD.match(currentFieldName, parseContext.parseFlags())) {
153                     failOnUnsupportedField = parser.booleanValue();
154                 } else if ("_name".equals(currentFieldName)) {
155                     queryName = parser.text();
156                 } else if (Fields.INCLUDE.match(currentFieldName, parseContext.parseFlags())) {
157                     include = parser.booleanValue();
158                 } else {
159                     throw new QueryParsingException(parseContext.index(), "[mlt] query does not support [" + currentFieldName + "]");
160                 }
161             } else if (token == XContentParser.Token.START_ARRAY) {
162                 if (Fields.STOP_WORDS.match(currentFieldName, parseContext.parseFlags())) {
163                     Set<String> stopWords = Sets.newHashSet();
164                     while ((token = parser.nextToken()) != XContentParser.Token.END_ARRAY) {
165                         stopWords.add(parser.text());
166                     }
167                     mltQuery.setStopWords(stopWords);
168                 } else if ("fields".equals(currentFieldName)) {
169                     moreLikeFields = Lists.newLinkedList();
170                     while ((token = parser.nextToken()) != XContentParser.Token.END_ARRAY) {
171                         moreLikeFields.add(parseContext.indexName(parser.text()));
172                     }
173                 } else if (Fields.DOCUMENT_IDS.match(currentFieldName, parseContext.parseFlags())) {
174                     while ((token = parser.nextToken()) != XContentParser.Token.END_ARRAY) {
175                         if (!token.isValue()) {
176                             throw new ElasticsearchIllegalArgumentException("ids array element should only contain ids");
177                         }
178                         likeItems.add(newTermVectorsRequest().id(parser.text()));
179                     }
180                 } else if (Fields.DOCUMENTS.match(currentFieldName, parseContext.parseFlags())) {
181                     while ((token = parser.nextToken()) != XContentParser.Token.END_ARRAY) {
182                         if (token != XContentParser.Token.START_OBJECT) {
183                             throw new ElasticsearchIllegalArgumentException("docs array element should include an object");
184                         }
185                         likeItems.add(parseDocument(parser));
186                     }
187                 } else if (Fields.LIKE.match(currentFieldName, parseContext.parseFlags())) {
188                     while ((token = parser.nextToken()) != XContentParser.Token.END_ARRAY) {
189                         parseLikeField(parser, likeTexts, likeItems);
190                     }
191                 } else if (Fields.IGNORE_LIKE.match(currentFieldName, parseContext.parseFlags())) {
192                     while ((token = parser.nextToken()) != XContentParser.Token.END_ARRAY) {
193                         parseLikeField(parser, ignoreTexts, ignoreItems);
194                     }
195                 } else {
196                     throw new QueryParsingException(parseContext.index(), "[mlt] query does not support [" + currentFieldName + "]");
197                 }
198             } else if (token == XContentParser.Token.START_OBJECT) {
199                 if (Fields.LIKE.match(currentFieldName, parseContext.parseFlags())) {
200                     parseLikeField(parser, likeTexts, likeItems);
201                 }
202                 else if (Fields.IGNORE_LIKE.match(currentFieldName, parseContext.parseFlags())) {
203                     parseLikeField(parser, ignoreTexts, ignoreItems);
204                 } else {
205                     throw new QueryParsingException(parseContext.index(), "[mlt] query does not support [" + currentFieldName + "]");
206                 }
207             }
208         }
209 
210         if (likeTexts.isEmpty() && likeItems.isEmpty()) {
211             throw new QueryParsingException(parseContext.index(), "more_like_this requires 'like' to be specified");
212         }
213         if (moreLikeFields != null && moreLikeFields.isEmpty()) {
214             throw new QueryParsingException(parseContext.index(), "more_like_this requires 'fields' to be non-empty");
215         }
216 
217         // set analyzer
218         if (analyzer == null) {
219             analyzer = parseContext.mapperService().searchAnalyzer();
220         }
221         mltQuery.setAnalyzer(analyzer);
222 
223         // set like text fields
224         boolean useDefaultField = (moreLikeFields == null);
225         if (useDefaultField) {
226             moreLikeFields = Lists.newArrayList(parseContext.defaultField());
227         }
228         // possibly remove unsupported fields
229         removeUnsupportedFields(moreLikeFields, analyzer, failOnUnsupportedField);
230         if (moreLikeFields.isEmpty()) {
231             return null;
232         }
233         mltQuery.setMoreLikeFields(moreLikeFields.toArray(Strings.EMPTY_ARRAY));
234 
235         // support for named query
236         if (queryName != null) {
237             parseContext.addNamedQuery(queryName, mltQuery);
238         }
239 
240         // handle like texts
241         if (!likeTexts.isEmpty()) {
242             mltQuery.setLikeText(likeTexts);
243         }
244         if (!ignoreTexts.isEmpty()) {
245             mltQuery.setIgnoreText(ignoreTexts);
246         }
247 
248         // handle items
249         if (!likeItems.isEmpty()) {
250             // set default index, type and fields if not specified
251             MultiTermVectorsRequest items = likeItems;
252             for (TermVectorsRequest item : ignoreItems) {
253                 items.add(item);
254             }
255 
256             for (TermVectorsRequest item : items) {
257                 if (item.index() == null) {
258                     item.index(parseContext.index().name());
259                 }
260                 if (item.type() == null) {
261                     if (parseContext.queryTypes().size() > 1) {
262                         throw new QueryParsingException(parseContext.index(),
263                                 "ambiguous type for item with id: " + item.id() + " and index: " + item.index());
264                     } else {
265                         item.type(parseContext.queryTypes().iterator().next());
266                     }
267                 }
268                 // default fields if not present but don't override for artificial docs
269                 if (item.selectedFields() == null && item.doc() == null) {
270                     if (useDefaultField) {
271                         item.selectedFields("*");
272                     } else {
273                         item.selectedFields(moreLikeFields.toArray(new String[moreLikeFields.size()]));
274                     }
275                 }
276             }
277             // fetching the items with multi-termvectors API
278             BooleanQuery boolQuery = new BooleanQuery();
279             MultiTermVectorsResponse responses = fetchService.fetchResponse(items);
280 
281             // getting the Fields for liked items
282             mltQuery.setLikeText(MoreLikeThisFetchService.getFields(responses, likeItems));
283 
284             // getting the Fields for ignored items
285             if (!ignoreItems.isEmpty()) {
286                 org.apache.lucene.index.Fields[] ignoreFields = MoreLikeThisFetchService.getFields(responses, ignoreItems);
287                 if (ignoreFields.length > 0) {
288                     mltQuery.setIgnoreText(ignoreFields);
289                 }
290             }
291 
292             boolQuery.add(mltQuery, BooleanClause.Occur.SHOULD);
293 
294             // exclude the items from the search
295             if (!include) {
296                 handleExclude(boolQuery, likeItems);
297             }
298             return boolQuery;
299         }
300 
301         return mltQuery;
302     }
303 
304     private TermVectorsRequest parseDocument(XContentParser parser) throws IOException {
305         TermVectorsRequest termVectorsRequest = newTermVectorsRequest();
306         TermVectorsRequest.parseRequest(termVectorsRequest, parser);
307         return termVectorsRequest;
308     }
309 
310     private void parseLikeField(XContentParser parser, List<String> likeTexts, MultiTermVectorsRequest items) throws IOException {
311         if (parser.currentToken().isValue()) {
312             likeTexts.add(parser.text());
313         } else if (parser.currentToken() == XContentParser.Token.START_OBJECT) {
314             items.add(parseDocument(parser));
315         } else {
316             throw new ElasticsearchIllegalArgumentException("Content of 'like' parameter should either be a string or an object");
317         }
318     }
319 
320     private TermVectorsRequest newTermVectorsRequest() {
321         return new TermVectorsRequest()
322                 .positions(false)
323                 .offsets(false)
324                 .payloads(false)
325                 .fieldStatistics(false)
326                 .termStatistics(false);
327     }
328 
329     private List<String> removeUnsupportedFields(List<String> moreLikeFields, Analyzer analyzer, boolean failOnUnsupportedField) throws IOException {
330         for (Iterator<String> it = moreLikeFields.iterator(); it.hasNext(); ) {
331             final String fieldName = it.next();
332             if (!Analysis.generatesCharacterTokenStream(analyzer, fieldName)) {
333                 if (failOnUnsupportedField) {
334                     throw new ElasticsearchIllegalArgumentException("more_like_this doesn't support binary/numeric fields: [" + fieldName + "]");
335                 } else {
336                     it.remove();
337                 }
338             }
339         }
340         return moreLikeFields;
341     }
342 
343     private void handleExclude(BooleanQuery boolQuery, MultiTermVectorsRequest likeItems) {
344         // artificial docs get assigned a random id and should be disregarded
345         List<BytesRef> uids = new ArrayList<>();
346         for (TermVectorsRequest item : likeItems) {
347             if (item.doc() != null) {
348                 continue;
349             }
350             uids.add(createUidAsBytes(item.type(), item.id()));
351         }
352         if (!uids.isEmpty()) {
353             TermsFilter filter = new TermsFilter(UidFieldMapper.NAME, uids.toArray(new BytesRef[0]));
354             ConstantScoreQuery query = new ConstantScoreQuery(filter);
355             boolQuery.add(query, BooleanClause.Occur.MUST_NOT);
356         }
357     }
