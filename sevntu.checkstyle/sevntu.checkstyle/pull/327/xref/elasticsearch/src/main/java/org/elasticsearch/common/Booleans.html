20  package org.elasticsearch.common;

22  import org.elasticsearch.ElasticsearchIllegalArgumentException;
23  /**
24   *
25   */
26  public class Booleans {
27  
28      /**
29       * Returns <code>true</code> iff the sequence is neither of the following:
30       *   <tt>false</tt>, <tt>0</tt>, <tt>off</tt>, <tt>no</tt>,
31       *   otherwise <code>false</code>
32       */
33      public static boolean parseBoolean(char[] text, int offset, int length, boolean defaultValue) {
34          if (text == null || length == 0) {
35              return defaultValue;
36          }
37          if (length == 1) {
38              return text[offset] != '0';
39          }
40          if (length == 2) {
41              return !(text[offset] == 'n' && text[offset + 1] == 'o');
42          }
43          if (length == 3) {
44              return !(text[offset] == 'o' && text[offset + 1] == 'f' && text[offset + 2] == 'f');
45          }
46          if (length == 5) {
47              return !(text[offset] == 'f' && text[offset + 1] == 'a' && text[offset + 2] == 'l' && text[offset + 3] == 's' && text[offset + 4] == 'e');
48          }
49          return true;
50      }
51  
52      /**
53       * returns true if the a sequence of chars is one of "true","false","on","off","yes","no","0","1"
54       *
55       * @param text   sequence to check
56       * @param offset offset to start
57       * @param length length to check
58       */
59      public static boolean isBoolean(char[] text, int offset, int length) {
60          if (text == null || length == 0) {
61              return false;
62          }
63          if (length == 1) {
64              return text[offset] == '0' || text[offset] == '1';
65          }
66          if (length == 2) {
67              return (text[offset] == 'n' && text[offset + 1] == 'o') || (text[offset] == 'o' && text[offset + 1] == 'n');
68          }
69          if (length == 3) {
70              return (text[offset] == 'o' && text[offset + 1] == 'f' && text[offset + 2] == 'f') ||
71                      (text[offset] == 'y' && text[offset + 1] == 'e' && text[offset + 2] == 's');
72          }
73          if (length == 4) {
74              return (text[offset] == 't' && text[offset + 1] == 'r' && text[offset + 2] == 'u' && text[offset + 3] == 'e');
75          }
76          if (length == 5) {
77              return (text[offset] == 'f' && text[offset + 1] == 'a' && text[offset + 2] == 'l' && text[offset + 3] == 's' && text[offset + 4] == 'e');
78          }
79          return false;
80      }
81  
82      /***
83       *
84       * @param value
85       * @return true/false
86       * throws exception if string cannot be parsed to boolean
87       */
88      public static Boolean parseBooleanExact(String value){
89  
90          boolean isFalse = isExplicitFalse(value);
91          if (isFalse) {
92              return false;
93          }
94          boolean isTrue = isExplicitTrue(value);
95          if (isTrue) {
96              return true;
97          }
98  
99          throw new ElasticsearchIllegalArgumentException("value cannot be parsed to boolean [ true/1/on/yes OR false/0/off/no ]  ");
100     }
101 
102     public static Boolean parseBoolean(String value, Boolean defaultValue) {
103         if (value == null) { // only for the null case we do that here!
104             return defaultValue;
105         }
106         return parseBoolean(value, false);
107     }
108     /**
109      * Returns <code>true</code> iff the value is neither of the following:
110      *   <tt>false</tt>, <tt>0</tt>, <tt>off</tt>, <tt>no</tt>
111      *   otherwise <code>false</code>
112      */
113     public static boolean parseBoolean(String value, boolean defaultValue) {
114         if (value == null) {
115             return defaultValue;
116         }
117         return !(value.equals("false") || value.equals("0") || value.equals("off") || value.equals("no"));
118     }
119 
120     /**
121      * Returns <code>true</code> iff the value is either of the following:
122      *   <tt>false</tt>, <tt>0</tt>, <tt>off</tt>, <tt>no</tt>
123      *   otherwise <code>false</code>
124      */
125     public static boolean isExplicitFalse(String value) {
126         return value != null && (value.equals("false") || value.equals("0") || value.equals("off") || value.equals("no"));
127     }
128 
129     /**
130      * Returns <code>true</code> iff the value is either of the following:
131      *   <tt>true</tt>, <tt>1</tt>, <tt>on</tt>, <tt>yes</tt>
132      *   otherwise <code>false</code>
133      */
134     public static boolean isExplicitTrue(String value) {
135         return value != null && (value.equals("true") || value.equals("1") || value.equals("on") || value.equals("yes"));
136     }

138 }
