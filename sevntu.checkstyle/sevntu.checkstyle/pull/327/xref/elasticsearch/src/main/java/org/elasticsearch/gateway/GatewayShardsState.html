57      private static final String SHARD_STATE_FILE_PREFIX = "state-";
58      private static final Pattern SHARD_STATE_FILE_PATTERN = Pattern.compile(SHARD_STATE_FILE_PREFIX + "(\\d+)(" + MetaDataStateFormat.STATE_FILE_EXTENSION + ")?");
59      private static final String PRIMARY_KEY = "primary";
60      private static final String VERSION_KEY = "version";

62      private final NodeEnvironment nodeEnv;

64      private volatile Map<ShardId, ShardStateInfo> currentState = Maps.newHashMap();

66      @Inject
67      public GatewayShardsState(Settings settings, NodeEnvironment nodeEnv,
68                                TransportNodesListGatewayStartedShards listGatewayStartedShards) throws Exception {
69          super(settings);
70          this.nodeEnv = nodeEnv;
71          if (listGatewayStartedShards != null) { // for testing
72              listGatewayStartedShards.initGateway(this);
73          }
74          if (DiscoveryNode.dataNode(settings)) {
75              try {
76                  ensureNoPre019State();
77                  long start = System.currentTimeMillis();
78                  currentState = loadShardsStateInfo();
79                  logger.debug("took {} to load started shards state", TimeValue.timeValueMillis(System.currentTimeMillis() - start));
80              } catch (Exception e) {
81                  logger.error("failed to read local state (started shards), exiting...", e);
82                  throw e;
83              }
84          }
85      }
86  
87      public ShardStateInfo loadShardInfo(ShardId shardId) throws Exception {
88          return loadShardStateInfo(shardId);
89      }

91      @Override
92      public void clusterChanged(ClusterChangedEvent event) {
93          final ClusterState state = event.state();
94          if (state.blocks().disableStatePersistence() == false
95              && state.nodes().localNode().dataNode()
96              && event.routingTableChanged()) {
97              // now, add all the ones that are active and on this node
98              RoutingNode routingNode = state.readOnlyRoutingNodes().node(state.nodes().localNodeId());
99              final Map<ShardId, ShardStateInfo> newState;
100             if (routingNode != null) {
101                newState = persistRoutingNodeState(routingNode);
102             } else {
103                newState = Maps.newHashMap();
104             }
105 
106             // preserve all shards that:
107             //   * are not already in the new map AND
108             //   * belong to an active index AND
109             //   * used to be on this node but are not yet completely stated on any other node
110             // since these shards are NOT active on this node the won't need to be written above - we just preserve these
111             // in this map until they are fully started anywhere else or are re-assigned and we need to update the state
112             final RoutingTable indexRoutingTables = state.routingTable();
113             for (Map.Entry<ShardId, ShardStateInfo> entry : this.currentState.entrySet()) {
114                 ShardId shardId = entry.getKey();
115                 if (newState.containsKey(shardId) == false) { // this shard used to be here
116                     String indexName = shardId.index().getName();
117                     if (state.metaData().hasIndex(indexName)) { // it's index is not deleted
118                         IndexRoutingTable index = indexRoutingTables.index(indexName);
119                         if (index != null && index.shard(shardId.id()).allShardsStarted() == false) {
120                            // not all shards are active on another node so we put it back until they are active
121                            newState.put(shardId, entry.getValue());
122                         }
123                     }
124                 }
125             }
126             this.currentState = newState;
127         }
128     }

130     Map<ShardId, ShardStateInfo> persistRoutingNodeState(RoutingNode routingNode) {
131         final Map<ShardId, ShardStateInfo> newState = Maps.newHashMap();
132         for (MutableShardRouting shardRouting : routingNode) {
133             if (shardRouting.active()) {
134                 ShardId shardId = shardRouting.shardId();
135                 ShardStateInfo shardStateInfo = new ShardStateInfo(shardRouting.version(), shardRouting.primary());
136                 final ShardStateInfo previous = currentState.get(shardId);
137                 if (maybeWriteShardState(shardId, shardStateInfo, previous)) {
138                     newState.put(shardId, shardStateInfo);
139                 } else if (previous != null) {
140                     currentState.put(shardId, previous);
141                 }
142             }
143         }
144         return newState;
145     }
146 
147     Map<ShardId, ShardStateInfo> getCurrentState() {
148         return currentState;
149     }

151     boolean maybeWriteShardState(ShardId shardId, ShardStateInfo shardStateInfo, ShardStateInfo previousState) {
152         final String writeReason;
153         if (previousState == null) {
154             writeReason = "freshly started, version [" + shardStateInfo.version + "]";
155         } else if (previousState.version < shardStateInfo.version) {
156             writeReason = "version changed from [" + previousState.version + "] to [" + shardStateInfo.version + "]";
157         } else {
158             logger.trace("skip writing shard state - has been written before shardID: " + shardId + " previous version:  [" +
159                     previousState.version + "] current version [" + shardStateInfo.version + "]");
160             assert previousState.version <= shardStateInfo.version : "version should not go backwards for shardID: " + shardId +
161                     " previous version:  [" + previousState.version + "] current version [" + shardStateInfo.version + "]";
162             return previousState.version == shardStateInfo.version;
163         }
164 
165         try {
166             writeShardState(writeReason, shardId, shardStateInfo, previousState);
167         } catch (Exception e) {
168             logger.warn("failed to write shard state for shard " + shardId, e);
169             // we failed to write the shard state, we will try and write
170             // it next time...
171         }
172         return true;
173     }

176     private Map<ShardId, ShardStateInfo> loadShardsStateInfo() throws Exception {
177         Set<ShardId> shardIds = nodeEnv.findAllShardIds();
178         long highestVersion = -1;
179         Map<ShardId, ShardStateInfo> shardsState = Maps.newHashMap();
180         for (ShardId shardId : shardIds) {
181             ShardStateInfo shardStateInfo = loadShardStateInfo(shardId);
182             if (shardStateInfo == null) {
183                 continue;
184             }
185             shardsState.put(shardId, shardStateInfo);
186 
187             // update the global version
188             if (shardStateInfo.version > highestVersion) {
189                 highestVersion = shardStateInfo.version;
190             }
191         }
192         return shardsState;
193     }
194 
195     private ShardStateInfo loadShardStateInfo(ShardId shardId) throws IOException {
196         return MetaDataStateFormat.loadLatestState(logger, newShardStateInfoFormat(false), SHARD_STATE_FILE_PATTERN,
197                 shardId.toString(), nodeEnv.shardPaths(shardId));
198     }
199 
200     private void writeShardState(String reason, ShardId shardId, ShardStateInfo shardStateInfo,
201                                  @Nullable ShardStateInfo previousStateInfo) throws Exception {
202         logger.trace("{} writing shard state, reason [{}]", shardId, reason);
203         final boolean deleteOldFiles = previousStateInfo != null && previousStateInfo.version != shardStateInfo.version;
204         MetaDataStateFormat<ShardStateInfo> stateFormat = newShardStateInfoFormat(deleteOldFiles);
205         stateFormat.write(shardStateInfo, SHARD_STATE_FILE_PREFIX, shardStateInfo.version, nodeEnv.shardPaths(shardId));
206     }

208     private MetaDataStateFormat<ShardStateInfo> newShardStateInfoFormat(boolean deleteOldFiles) {
209         return new MetaDataStateFormat<ShardStateInfo>(XContentType.JSON, deleteOldFiles) {
210 
211             @Override
212             protected XContentBuilder newXContentBuilder(XContentType type, OutputStream stream) throws IOException {
213                 XContentBuilder xContentBuilder = super.newXContentBuilder(type, stream);
214                 xContentBuilder.prettyPrint();
215                 return xContentBuilder;
216             }
217 
218             @Override
219             public void toXContent(XContentBuilder builder, ShardStateInfo shardStateInfo) throws IOException {
220                 builder.field(VERSION_KEY, shardStateInfo.version);
221                 if (shardStateInfo.primary != null) {
222                     builder.field(PRIMARY_KEY, shardStateInfo.primary);
223                 }
224             }
225 
226             @Override
227             public ShardStateInfo fromXContent(XContentParser parser) throws IOException {
228                 XContentParser.Token token = parser.nextToken();
229                 if (token == null) {
230                     return null;
231                 }
232                 long version = -1;
233                 Boolean primary = null;
234                 String currentFieldName = null;
235                 while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
236                     if (token == XContentParser.Token.FIELD_NAME) {
237                         currentFieldName = parser.currentName();
238                     } else if (token.isValue()) {
239                         if (VERSION_KEY.equals(currentFieldName)) {
240                             version = parser.longValue();
241                         } else if (PRIMARY_KEY.equals(currentFieldName)) {
242                             primary = parser.booleanValue();
243                         } else {
244                             throw new CorruptStateException("unexpected field in shard state [" + currentFieldName + "]");
245                         }
246                     } else {
247                         throw new CorruptStateException("unexpected token in shard state [" + token.name() + "]");
248                     }
249                 }
250                 if (primary == null) {
251                     throw new CorruptStateException("missing value for [primary] in shard state");
252                 }
253                 if (version == -1) {
254                     throw new CorruptStateException("missing value for [version] in shard state");
255                 }
256                 return new ShardStateInfo(version, primary);
257             }
258         };
259     }

261     private void ensureNoPre019State() throws Exception {
262         for (Path dataLocation : nodeEnv.nodeDataPaths()) {
263             final Path stateLocation = dataLocation.resolve(MetaDataStateFormat.STATE_DIR_NAME);
264             if (Files.exists(stateLocation)) {
265                 try (DirectoryStream<Path> stream = Files.newDirectoryStream(stateLocation, "shards-*")) {
266                     for (Path stateFile : stream) {
267                         throw new ElasticsearchIllegalStateException("Detected pre 0.19 shard state file please upgrade to a version before "
268                                 + Version.CURRENT.minimumCompatibilityVersion()
269                                 + " first to upgrade state structures - shard state found: [" + stateFile.getParent().toAbsolutePath());
270                     }
271                 }
272             }
273         }
