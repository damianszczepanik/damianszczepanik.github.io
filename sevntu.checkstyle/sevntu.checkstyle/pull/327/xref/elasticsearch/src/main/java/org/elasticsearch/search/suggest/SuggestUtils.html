55      public static DirectSpellChecker getDirectSpellChecker(DirectSpellcheckerSettings suggestion) {
56          
57          DirectSpellChecker directSpellChecker = new DirectSpellChecker();
58          directSpellChecker.setAccuracy(suggestion.accuracy());
59          Comparator<SuggestWord> comparator;
60          switch (suggestion.sort()) {
61              case SCORE:
62                  comparator = SCORE_COMPARATOR;
63                  break;
64              case FREQUENCY:
65                  comparator = LUCENE_FREQUENCY;
66                  break;
67              default:
68                  throw new ElasticsearchIllegalArgumentException("Illegal suggest sort: " + suggestion.sort());
69          }
70          directSpellChecker.setComparator(comparator);
71          directSpellChecker.setDistance(suggestion.stringDistance());
72          directSpellChecker.setMaxEdits(suggestion.maxEdits());
73          directSpellChecker.setMaxInspections(suggestion.maxInspections());
74          directSpellChecker.setMaxQueryFrequency(suggestion.maxTermFreq());
75          directSpellChecker.setMinPrefix(suggestion.prefixLength());
76          directSpellChecker.setMinQueryLength(suggestion.minWordLength());
77          directSpellChecker.setThresholdFrequency(suggestion.minDocFreq());
78          directSpellChecker.setLowerCaseTerms(false);
79          return directSpellChecker;
80      }

82      public static BytesRef join(BytesRef separator, BytesRefBuilder result, BytesRef... toJoin) {
83          result.clear();
84          for (int i = 0; i < toJoin.length - 1; i++) {
85              result.append(toJoin[i]);
86              result.append(separator);
87          }
88          result.append(toJoin[toJoin.length-1]);
89          return result.get();
90      }

92      public static abstract class TokenConsumer {
93          protected CharTermAttribute charTermAttr;
94          protected PositionIncrementAttribute posIncAttr;
95          protected OffsetAttribute offsetAttr;
96          
97          public void reset(TokenStream stream) {
98              charTermAttr = stream.addAttribute(CharTermAttribute.class);
99              posIncAttr = stream.addAttribute(PositionIncrementAttribute.class);
100             offsetAttr = stream.addAttribute(OffsetAttribute.class);
101         }
102         
103         protected BytesRef fillBytesRef(BytesRefBuilder spare) {
104             spare.copyChars(charTermAttr);
105             return spare.get();
106         }
107         
108         public abstract void nextToken() throws IOException;
109 
110         public void end() {}
111     }

113     public static int analyze(Analyzer analyzer, BytesRef toAnalyze, String field, TokenConsumer consumer, CharsRefBuilder spare) throws IOException {
114         spare.copyUTF8Bytes(toAnalyze);
115         return analyze(analyzer, spare.get(), field, consumer);
116     }
117     
118     public static int analyze(Analyzer analyzer, CharsRef toAnalyze, String field, TokenConsumer consumer) throws IOException {
119         TokenStream ts = analyzer.tokenStream(
120                 field, new FastCharArrayReader(toAnalyze.chars, toAnalyze.offset, toAnalyze.length)
121         );
122         return analyze(ts, consumer);
123     }
124     
125     public static int analyze(TokenStream stream, TokenConsumer consumer) throws IOException {
126         stream.reset();
127         consumer.reset(stream);
128         int numTokens = 0;
129         while (stream.incrementToken()) {
130             consumer.nextToken();
131             numTokens++;
132         }
133         consumer.end();
134         stream.close();
135         return numTokens;
136     }

138     public static SuggestMode resolveSuggestMode(String suggestMode) {
139         suggestMode = suggestMode.toLowerCase(Locale.US);
140         if ("missing".equals(suggestMode)) {
141             return SuggestMode.SUGGEST_WHEN_NOT_IN_INDEX;
142         } else if ("popular".equals(suggestMode)) {
143             return SuggestMode.SUGGEST_MORE_POPULAR;
144         } else if ("always".equals(suggestMode)) {
145             return SuggestMode.SUGGEST_ALWAYS;
146         } else {
147             throw new ElasticsearchIllegalArgumentException("Illegal suggest mode " + suggestMode);
148         }
149     }
150 
151     public static Suggest.Suggestion.Sort resolveSort(String sortVal) {
152         if ("score".equals(sortVal)) {
153             return Suggest.Suggestion.Sort.SCORE;
154         } else if ("frequency".equals(sortVal)) {
155             return Suggest.Suggestion.Sort.FREQUENCY;
156         } else {
157             throw new ElasticsearchIllegalArgumentException("Illegal suggest sort " + sortVal);
158         }
159     }

161     public static StringDistance resolveDistance(String distanceVal) {
162         if ("internal".equals(distanceVal)) {
163             return DirectSpellChecker.INTERNAL_LEVENSHTEIN;
164         } else if ("damerau_levenshtein".equals(distanceVal) || "damerauLevenshtein".equals(distanceVal)) {
165             return new LuceneLevenshteinDistance();
166         } else if ("levenstein".equals(distanceVal)) {
167             return new LevensteinDistance();
168           //TODO Jaro and Winkler are 2 people - so apply same naming logic as damerau_levenshtein  
169         } else if ("jarowinkler".equals(distanceVal)) {
170             return new JaroWinklerDistance();
171         } else if ("ngram".equals(distanceVal)) {
172             return new NGramDistance();
173         } else {
174             throw new ElasticsearchIllegalArgumentException("Illegal distance option " + distanceVal);
175         }
176     }

193     public static boolean parseDirectSpellcheckerSettings(XContentParser parser, String fieldName,
194                 DirectSpellcheckerSettings suggestion) throws IOException {
195             if ("accuracy".equals(fieldName)) {
196                 suggestion.accuracy(parser.floatValue());
197             } else if (Fields.SUGGEST_MODE.match(fieldName)) {
198                 suggestion.suggestMode(SuggestUtils.resolveSuggestMode(parser.text()));
199             } else if ("sort".equals(fieldName)) {
200                 suggestion.sort(SuggestUtils.resolveSort(parser.text()));
201             } else if (Fields.STRING_DISTANCE.match(fieldName)) {
202                 suggestion.stringDistance(SuggestUtils.resolveDistance(parser.text()));
203             } else if (Fields.MAX_EDITS.match(fieldName)) {
204                 suggestion.maxEdits(parser.intValue());
205                 if (suggestion.maxEdits() < 1 || suggestion.maxEdits() > LevenshteinAutomata.MAXIMUM_SUPPORTED_DISTANCE) {
206                     throw new ElasticsearchIllegalArgumentException("Illegal max_edits value " + suggestion.maxEdits());
207                 }
208             } else if (Fields.MAX_INSPECTIONS.match(fieldName)) {
209                 suggestion.maxInspections(parser.intValue());
210             } else if (Fields.MAX_TERM_FREQ.match(fieldName)) {
211                 suggestion.maxTermFreq(parser.floatValue());
212             } else if (Fields.PREFIX_LENGTH.match(fieldName)) {
213                 suggestion.prefixLength(parser.intValue());
214             } else if (Fields.MIN_WORD_LENGTH.match(fieldName)) {
215                 suggestion.minQueryLength(parser.intValue());
216             } else if (Fields.MIN_DOC_FREQ.match(fieldName)) {
217                 suggestion.minDocFreq(parser.floatValue());
218             } else {
219                 return false;
220             }
221             return true;
222     }

224     public static boolean parseSuggestContext(XContentParser parser, MapperService mapperService, String fieldName,
225             SuggestionSearchContext.SuggestionContext suggestion) throws IOException {
226         
227         if ("analyzer".equals(fieldName)) {
228             String analyzerName = parser.text();
229             Analyzer analyzer = mapperService.analysisService().analyzer(analyzerName);
230             if (analyzer == null) {
231                 throw new ElasticsearchIllegalArgumentException("Analyzer [" + analyzerName + "] doesn't exists");
232             }
233             suggestion.setAnalyzer(analyzer);
234         } else if ("field".equals(fieldName)) {
235             suggestion.setField(parser.text());
236         } else if ("size".equals(fieldName)) {
237             suggestion.setSize(parser.intValue());
238         } else if (Fields.SHARD_SIZE.match(fieldName)) {
239             suggestion.setShardSize(parser.intValue());
240         } else {
241            return false;
242         }
243         return true;
244         
245     }

248     public static void verifySuggestion(MapperService mapperService, BytesRef globalText, SuggestionContext suggestion) {
249         // Verify options and set defaults
250         if (suggestion.getField() == null) {
251             throw new ElasticsearchIllegalArgumentException("The required field option is missing");
252         }
253         if (suggestion.getText() == null) {
254             if (globalText == null) {
255                 throw new ElasticsearchIllegalArgumentException("The required text option is missing");
256             }
257             suggestion.setText(globalText);
258         }
259         if (suggestion.getAnalyzer() == null) {
260             suggestion.setAnalyzer(mapperService.searchAnalyzer());
261         }
262         if (suggestion.getShardSize() == -1) {
263             suggestion.setShardSize(Math.max(suggestion.getSize(), 5));
264         }
265     }

268     public static ShingleTokenFilterFactory.Factory getShingleFilterFactory(Analyzer analyzer) {
269         if (analyzer instanceof NamedAnalyzer) {
270             analyzer = ((NamedAnalyzer)analyzer).analyzer();
271         }
272         if (analyzer instanceof CustomAnalyzer) {
273             final CustomAnalyzer a = (CustomAnalyzer) analyzer;
274             final TokenFilterFactory[] tokenFilters = a.tokenFilters();
275             for (TokenFilterFactory tokenFilterFactory : tokenFilters) {
276                 if (tokenFilterFactory instanceof ShingleTokenFilterFactory) {
277                     return ((ShingleTokenFilterFactory)tokenFilterFactory).getInnerFactory();
278                 } else if (tokenFilterFactory instanceof ShingleTokenFilterFactory.Factory) {
279                     return (ShingleTokenFilterFactory.Factory) tokenFilterFactory;
280                 }
281             }
282         }
283         return null;
284     }
