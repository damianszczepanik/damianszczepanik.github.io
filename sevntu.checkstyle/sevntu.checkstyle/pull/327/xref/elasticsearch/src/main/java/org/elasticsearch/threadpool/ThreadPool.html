101     @Inject
102     public ThreadPool(Settings settings, @Nullable NodeSettingsService nodeSettingsService) {
103         super(settings);
104 
105         assert settings.get("name") != null : "ThreadPool's settings should contain a name";
106 
107         Map<String, Settings> groupSettings = settings.getGroups(THREADPOOL_GROUP);
108 
109         int availableProcessors = EsExecutors.boundedNumberOfProcessors(settings);
110         int halfProcMaxAt5 = Math.min(((availableProcessors + 1) / 2), 5);
111         int halfProcMaxAt10 = Math.min(((availableProcessors + 1) / 2), 10);
112         defaultExecutorTypeSettings = ImmutableMap.<String, Settings>builder()
113                 .put(Names.GENERIC, settingsBuilder().put("type", "cached").put("keep_alive", "30s").build())
114                 .put(Names.INDEX, settingsBuilder().put("type", "fixed").put("size", availableProcessors).put("queue_size", 200).build())
115                 .put(Names.BULK, settingsBuilder().put("type", "fixed").put("size", availableProcessors).put("queue_size", 50).build())
116                 .put(Names.GET, settingsBuilder().put("type", "fixed").put("size", availableProcessors).put("queue_size", 1000).build())
117                 .put(Names.SEARCH, settingsBuilder().put("type", "fixed").put("size", availableProcessors * 3).put("queue_size", 1000).build())
118                 .put(Names.SUGGEST, settingsBuilder().put("type", "fixed").put("size", availableProcessors).put("queue_size", 1000).build())
119                 .put(Names.PERCOLATE, settingsBuilder().put("type", "fixed").put("size", availableProcessors).put("queue_size", 1000).build())
120                 .put(Names.MANAGEMENT, settingsBuilder().put("type", "scaling").put("keep_alive", "5m").put("size", 5).build())
121                 // no queue as this means clients will need to handle rejections on listener queue even if the operation succeeded
122                 // the assumption here is that the listeners should be very lightweight on the listeners side
123                 .put(Names.LISTENER, settingsBuilder().put("type", "fixed").put("size", halfProcMaxAt10).build())
124                 .put(Names.FLUSH, settingsBuilder().put("type", "scaling").put("keep_alive", "5m").put("size", halfProcMaxAt5).build())
125                 .put(Names.REFRESH, settingsBuilder().put("type", "scaling").put("keep_alive", "5m").put("size", halfProcMaxAt10).build())
126                 .put(Names.WARMER, settingsBuilder().put("type", "scaling").put("keep_alive", "5m").put("size", halfProcMaxAt5).build())
127                 .put(Names.SNAPSHOT, settingsBuilder().put("type", "scaling").put("keep_alive", "5m").put("size", halfProcMaxAt5).build())
128                 .put(Names.OPTIMIZE, settingsBuilder().put("type", "fixed").put("size", 1).build())
129                 .put(Names.BENCH, settingsBuilder().put("type", "scaling").put("keep_alive", "5m").put("size", halfProcMaxAt5).build())
130                 .build();
131 
132         Map<String, ExecutorHolder> executors = Maps.newHashMap();
133         for (Map.Entry<String, Settings> executor : defaultExecutorTypeSettings.entrySet()) {
134             executors.put(executor.getKey(), build(executor.getKey(), groupSettings.get(executor.getKey()), executor.getValue()));
135         }
136 
137         // Building custom thread pools
138         for (Map.Entry<String, Settings> entry : groupSettings.entrySet()) {
139             if (executors.containsKey(entry.getKey())) {
140                 continue;
141             }
142             executors.put(entry.getKey(), build(entry.getKey(), entry.getValue(), ImmutableSettings.EMPTY));
143         }
144 
145         executors.put(Names.SAME, new ExecutorHolder(MoreExecutors.directExecutor(), new Info(Names.SAME, "same")));
146         if (!executors.get(Names.GENERIC).info.getType().equals("cached")) {
147             throw new ElasticsearchIllegalArgumentException("generic thread pool must be of type cached");
148         }
149         this.executors = ImmutableMap.copyOf(executors);
150         this.scheduler = new ScheduledThreadPoolExecutor(1, EsExecutors.daemonThreadFactory(settings, "scheduler"), new EsAbortPolicy());
151         this.scheduler.setExecuteExistingDelayedTasksAfterShutdownPolicy(false);
152         this.scheduler.setContinueExistingPeriodicTasksAfterShutdownPolicy(false);
153         this.scheduler.setRemoveOnCancelPolicy(true);
154         if (nodeSettingsService != null) {
155             nodeSettingsService.addListener(new ApplySettings());
156         }
157 
158         TimeValue estimatedTimeInterval = settings.getAsTime("threadpool.estimated_time_interval", TimeValue.timeValueMillis(200));
159         this.estimatedTimeThread = new EstimatedTimeThread(EsExecutors.threadName(settings, "[timer]"), estimatedTimeInterval.millis());
160         this.estimatedTimeThread.start();
161     }

192     public ThreadPoolStats stats() {
193         List<ThreadPoolStats.Stats> stats = new ArrayList<>();
194         for (ExecutorHolder holder : executors.values()) {
195             String name = holder.info.getName();
196             // no need to have info on "same" thread pool
197             if ("same".equals(name)) {
198                 continue;
199             }
200             int threads = -1;
201             int queue = -1;
202             int active = -1;
203             long rejected = -1;
204             int largest = -1;
205             long completed = -1;
206             if (holder.executor() instanceof ThreadPoolExecutor) {
207                 ThreadPoolExecutor threadPoolExecutor = (ThreadPoolExecutor) holder.executor();
208                 threads = threadPoolExecutor.getPoolSize();
209                 queue = threadPoolExecutor.getQueue().size();
210                 active = threadPoolExecutor.getActiveCount();
211                 largest = threadPoolExecutor.getLargestPoolSize();
212                 completed = threadPoolExecutor.getCompletedTaskCount();
213                 RejectedExecutionHandler rejectedExecutionHandler = threadPoolExecutor.getRejectedExecutionHandler();
214                 if (rejectedExecutionHandler instanceof XRejectedExecutionHandler) {
215                     rejected = ((XRejectedExecutionHandler) rejectedExecutionHandler).rejected();
216                 }
217             }
218             stats.add(new ThreadPoolStats.Stats(name, threads, queue, active, rejected, largest, completed));
219         }
220         return new ThreadPoolStats(stats);
221     }

250     public void shutdown() {
251         estimatedTimeThread.running = false;
252         estimatedTimeThread.interrupt();
253         scheduler.shutdown();
254         for (ExecutorHolder executor : executors.values()) {
255             if (executor.executor() instanceof ThreadPoolExecutor) {
256                 ((ThreadPoolExecutor) executor.executor()).shutdown();
257             }
258         }
259     }

261     public void shutdownNow() {
262         estimatedTimeThread.running = false;
263         estimatedTimeThread.interrupt();
264         scheduler.shutdownNow();
265         for (ExecutorHolder executor : executors.values()) {
266             if (executor.executor() instanceof ThreadPoolExecutor) {
267                 ((ThreadPoolExecutor) executor.executor()).shutdownNow();
268             }
269         }
270         while (!retiredExecutors.isEmpty()) {
271             ((ThreadPoolExecutor) retiredExecutors.remove().executor()).shutdownNow();
272         }
273     }

275     public boolean awaitTermination(long timeout, TimeUnit unit) throws InterruptedException {
276         boolean result = scheduler.awaitTermination(timeout, unit);
277         for (ExecutorHolder executor : executors.values()) {
278             if (executor.executor() instanceof ThreadPoolExecutor) {
279                 result &= ((ThreadPoolExecutor) executor.executor()).awaitTermination(timeout, unit);
280             }
281         }
282         while (!retiredExecutors.isEmpty()) {
283             ThreadPoolExecutor executor = (ThreadPoolExecutor) retiredExecutors.remove().executor();
284             result &= executor.awaitTermination(timeout, unit);
285         }
286         estimatedTimeThread.join(unit.toMillis(timeout));
287         return result;
288     }

294     private ExecutorHolder rebuild(String name, ExecutorHolder previousExecutorHolder, @Nullable Settings settings, Settings defaultSettings) {
295         if (Names.SAME.equals(name)) {
296             // Don't allow to change the "same" thread executor
297             return previousExecutorHolder;
298         }
299         if (settings == null) {
300             settings = ImmutableSettings.Builder.EMPTY_SETTINGS;
301         }
302         Info previousInfo = previousExecutorHolder != null ? previousExecutorHolder.info : null;
303         String type = settings.get("type", previousInfo != null ? previousInfo.getType() : defaultSettings.get("type"));
304         ThreadFactory threadFactory = EsExecutors.daemonThreadFactory(this.settings, name);
305         if ("same".equals(type)) {
306             if (previousExecutorHolder != null) {
307                 logger.debug("updating thread_pool [{}], type [{}]", name, type);
308             } else {
309                 logger.debug("creating thread_pool [{}], type [{}]", name, type);
310             }
311             return new ExecutorHolder(MoreExecutors.directExecutor(), new Info(name, type));
312         } else if ("cached".equals(type)) {
313             TimeValue defaultKeepAlive = defaultSettings.getAsTime("keep_alive", timeValueMinutes(5));
314             if (previousExecutorHolder != null) {
315                 if ("cached".equals(previousInfo.getType())) {
316                     TimeValue updatedKeepAlive = settings.getAsTime("keep_alive", previousInfo.getKeepAlive());
317                     if (!previousInfo.getKeepAlive().equals(updatedKeepAlive)) {
318                         logger.debug("updating thread_pool [{}], type [{}], keep_alive [{}]", name, type, updatedKeepAlive);
319                         ((EsThreadPoolExecutor) previousExecutorHolder.executor()).setKeepAliveTime(updatedKeepAlive.millis(), TimeUnit.MILLISECONDS);
320                         return new ExecutorHolder(previousExecutorHolder.executor(), new Info(name, type, -1, -1, updatedKeepAlive, null));
321                     }
322                     return previousExecutorHolder;
323                 }
324                 if (previousInfo.getKeepAlive() != null) {
325                     defaultKeepAlive = previousInfo.getKeepAlive();
326                 }
327             }
328             TimeValue keepAlive = settings.getAsTime("keep_alive", defaultKeepAlive);
329             if (previousExecutorHolder != null) {
330                 logger.debug("updating thread_pool [{}], type [{}], keep_alive [{}]", name, type, keepAlive);
331             } else {
332                 logger.debug("creating thread_pool [{}], type [{}], keep_alive [{}]", name, type, keepAlive);
333             }
334             Executor executor = EsExecutors.newCached(keepAlive.millis(), TimeUnit.MILLISECONDS, threadFactory);
335             return new ExecutorHolder(executor, new Info(name, type, -1, -1, keepAlive, null));
336         } else if ("fixed".equals(type)) {
337             int defaultSize = defaultSettings.getAsInt("size", EsExecutors.boundedNumberOfProcessors(settings));
338             SizeValue defaultQueueSize = getAsSizeOrUnbounded(defaultSettings, "queue", getAsSizeOrUnbounded(defaultSettings, "queue_size", null));
339 
340             if (previousExecutorHolder != null) {
341                 if ("fixed".equals(previousInfo.getType())) {
342                     SizeValue updatedQueueSize = getAsSizeOrUnbounded(settings, "capacity", getAsSizeOrUnbounded(settings, "queue", getAsSizeOrUnbounded(settings, "queue_size", previousInfo.getQueueSize())));
343                     if (Objects.equal(previousInfo.getQueueSize(), updatedQueueSize)) {
344                         int updatedSize = settings.getAsInt("size", previousInfo.getMax());
345                         if (previousInfo.getMax() != updatedSize) {
346                             logger.debug("updating thread_pool [{}], type [{}], size [{}], queue_size [{}]", name, type, updatedSize, updatedQueueSize);
347                             ((EsThreadPoolExecutor) previousExecutorHolder.executor()).setCorePoolSize(updatedSize);
348                             ((EsThreadPoolExecutor) previousExecutorHolder.executor()).setMaximumPoolSize(updatedSize);
349                             return new ExecutorHolder(previousExecutorHolder.executor(), new Info(name, type, updatedSize, updatedSize, null, updatedQueueSize));
350                         }
351                         return previousExecutorHolder;
352                     }
353                 }
354                 if (previousInfo.getMax() >= 0) {
355                     defaultSize = previousInfo.getMax();
356                 }
357                 defaultQueueSize = previousInfo.getQueueSize();
358             }
359 
360             int size = settings.getAsInt("size", defaultSize);
361             SizeValue queueSize = getAsSizeOrUnbounded(settings, "capacity", getAsSizeOrUnbounded(settings, "queue", getAsSizeOrUnbounded(settings, "queue_size", defaultQueueSize)));
362             logger.debug("creating thread_pool [{}], type [{}], size [{}], queue_size [{}]", name, type, size, queueSize);
363             Executor executor = EsExecutors.newFixed(size, queueSize == null ? -1 : (int) queueSize.singles(), threadFactory);
364             return new ExecutorHolder(executor, new Info(name, type, size, size, null, queueSize));
365         } else if ("scaling".equals(type)) {
366             TimeValue defaultKeepAlive = defaultSettings.getAsTime("keep_alive", timeValueMinutes(5));
367             int defaultMin = defaultSettings.getAsInt("min", 1);
368             int defaultSize = defaultSettings.getAsInt("size", EsExecutors.boundedNumberOfProcessors(settings));
369             if (previousExecutorHolder != null) {
370                 if ("scaling".equals(previousInfo.getType())) {
371                     TimeValue updatedKeepAlive = settings.getAsTime("keep_alive", previousInfo.getKeepAlive());
372                     int updatedMin = settings.getAsInt("min", previousInfo.getMin());
373                     int updatedSize = settings.getAsInt("max", settings.getAsInt("size", previousInfo.getMax()));
374                     if (!previousInfo.getKeepAlive().equals(updatedKeepAlive) || previousInfo.getMin() != updatedMin || previousInfo.getMax() != updatedSize) {
375                         logger.debug("updating thread_pool [{}], type [{}], keep_alive [{}]", name, type, updatedKeepAlive);
376                         if (!previousInfo.getKeepAlive().equals(updatedKeepAlive)) {
377                             ((EsThreadPoolExecutor) previousExecutorHolder.executor()).setKeepAliveTime(updatedKeepAlive.millis(), TimeUnit.MILLISECONDS);
378                         }
379                         if (previousInfo.getMin() != updatedMin) {
380                             ((EsThreadPoolExecutor) previousExecutorHolder.executor()).setCorePoolSize(updatedMin);
381                         }
382                         if (previousInfo.getMax() != updatedSize) {
383                             ((EsThreadPoolExecutor) previousExecutorHolder.executor()).setMaximumPoolSize(updatedSize);
384                         }
385                         return new ExecutorHolder(previousExecutorHolder.executor(), new Info(name, type, updatedMin, updatedSize, updatedKeepAlive, null));
386                     }
387                     return previousExecutorHolder;
388                 }
389                 if (previousInfo.getKeepAlive() != null) {
390                     defaultKeepAlive = previousInfo.getKeepAlive();
391                 }
392                 if (previousInfo.getMin() >= 0) {
393                     defaultMin = previousInfo.getMin();
394                 }
395                 if (previousInfo.getMax() >= 0) {
396                     defaultSize = previousInfo.getMax();
397                 }
398             }
399             TimeValue keepAlive = settings.getAsTime("keep_alive", defaultKeepAlive);
400             int min = settings.getAsInt("min", defaultMin);
401             int size = settings.getAsInt("max", settings.getAsInt("size", defaultSize));
402             if (previousExecutorHolder != null) {
403                 logger.debug("updating thread_pool [{}], type [{}], min [{}], size [{}], keep_alive [{}]", name, type, min, size, keepAlive);
404             } else {
405                 logger.debug("creating thread_pool [{}], type [{}], min [{}], size [{}], keep_alive [{}]", name, type, min, size, keepAlive);
406             }
407             Executor executor = EsExecutors.newScaling(min, size, keepAlive.millis(), TimeUnit.MILLISECONDS, threadFactory);
408             return new ExecutorHolder(executor, new Info(name, type, min, size, keepAlive, null));
409         }
410         throw new ElasticsearchIllegalArgumentException("No type found [" + type + "], for [" + name + "]");
411     }

413     public void updateSettings(Settings settings) {
414         Map<String, Settings> groupSettings = settings.getGroups("threadpool");
415         if (groupSettings.isEmpty()) {
416             return;
417         }
418 
419         for (Map.Entry<String, Settings> executor : defaultExecutorTypeSettings.entrySet()) {
420             Settings updatedSettings = groupSettings.get(executor.getKey());
421             if (updatedSettings == null) {
422                 continue;
423             }
424 
425             ExecutorHolder oldExecutorHolder = executors.get(executor.getKey());
426             ExecutorHolder newExecutorHolder = rebuild(executor.getKey(), oldExecutorHolder, updatedSettings, executor.getValue());
427             if (!oldExecutorHolder.equals(newExecutorHolder)) {
428                 executors = newMapBuilder(executors).put(executor.getKey(), newExecutorHolder).immutableMap();
429                 if (!oldExecutorHolder.executor().equals(newExecutorHolder.executor()) && oldExecutorHolder.executor() instanceof EsThreadPoolExecutor) {
430                     retiredExecutors.add(oldExecutorHolder);
431                     ((EsThreadPoolExecutor) oldExecutorHolder.executor()).shutdown(new ExecutorShutdownListener(oldExecutorHolder));
432                 }
433             }
434         }
435 
436         // Building custom thread pools
437         for (Map.Entry<String, Settings> entry : groupSettings.entrySet()) {
438             if (defaultExecutorTypeSettings.containsKey(entry.getKey())) {
439                 continue;
440             }
441 
442             ExecutorHolder oldExecutorHolder = executors.get(entry.getKey());
443             ExecutorHolder newExecutorHolder = rebuild(entry.getKey(), oldExecutorHolder, entry.getValue(), ImmutableSettings.EMPTY);
444             // Can't introduce new thread pools at runtime, because The oldExecutorHolder variable will be null in the
445             // case the settings contains a thread pool not defined in the initial settings in the constructor. The if
446             // statement will then fail and so this prevents the addition of new thread groups at runtime, which is desired.
447             if (!newExecutorHolder.equals(oldExecutorHolder)) {
448                 executors = newMapBuilder(executors).put(entry.getKey(), newExecutorHolder).immutableMap();
449                 if (!oldExecutorHolder.executor().equals(newExecutorHolder.executor()) && oldExecutorHolder.executor() instanceof EsThreadPoolExecutor) {
450                     retiredExecutors.add(oldExecutorHolder);
451                     ((EsThreadPoolExecutor) oldExecutorHolder.executor()).shutdown(new ExecutorShutdownListener(oldExecutorHolder));
452                 }
453             }
454         }
455     }

457     /**
458      * A thread pool size can also be unbounded and is represented by -1, which is not supported by SizeValue (which only supports positive numbers)
459      */
460     private SizeValue getAsSizeOrUnbounded(Settings settings, String setting, SizeValue defaultValue) throws SettingsException {
461         if ("-1".equals(settings.get(setting))) {
462             return null;
463         }
464         return parseSizeValue(settings.get(setting), defaultValue);
465     }

546     static class EstimatedTimeThread extends Thread {
547 
548         final long interval;
549         final TimeCounter counter;
550         volatile boolean running = true;
551         volatile long estimatedTimeInMillis;
552 
553         EstimatedTimeThread(String name, long interval) {
554             super(name);
555             this.interval = interval;
556             this.estimatedTimeInMillis = System.currentTimeMillis();
557             this.counter = new TimeCounter();
558             setDaemon(true);
559         }
560 
561         public long estimatedTimeInMillis() {

742     /**
743      * Returns <code>true</code> if the given service was terminated successfully. If the termination timed out,
744      * the service is <code>null</code> this method will return <code>false</code>.
745      */
746     public static boolean terminate(ExecutorService service, long timeout, TimeUnit timeUnit) {
747         if (service != null) {
748             service.shutdown();
749             try {
750                 if (service.awaitTermination(timeout, timeUnit)) {
751                     return true;
752                 }
753             } catch (InterruptedException e) {
754                 Thread.currentThread().interrupt();
755             }
756             service.shutdownNow();
757         }
758         return false;
759     }

762      * Returns <code>true</code> if the given pool was terminated successfully. If the termination timed out,
763      * the service is <code>null</code> this method will return <code>false</code>.
764      */
765     public static boolean terminate(ThreadPool pool, long timeout, TimeUnit timeUnit) {

770                     return true;
771                 }
772             } catch (InterruptedException e) {
773                 Thread.currentThread().interrupt();
774             }
775             // last resort
776             pool.shutdownNow();
777         }
