86      public NodeStats(DiscoveryNode node, long timestamp, @Nullable NodeIndicesStats indices,
87                       @Nullable OsStats os, @Nullable ProcessStats process, @Nullable JvmStats jvm, @Nullable ThreadPoolStats threadPool,
88                       @Nullable NetworkStats network, @Nullable FsStats fs, @Nullable TransportStats transport, @Nullable HttpStats http,
89                       @Nullable AllCircuitBreakerStats breaker) {
90          super(node);
91          this.timestamp = timestamp;
92          this.indices = indices;
93          this.os = os;
94          this.process = process;
95          this.jvm = jvm;
96          this.threadPool = threadPool;
97          this.network = network;
98          this.fs = fs;
99          this.transport = transport;
100         this.http = http;
101         this.breaker = breaker;
102     }

184     public static NodeStats readNodeStats(StreamInput in) throws IOException {
185         NodeStats nodeInfo = new NodeStats();
186         nodeInfo.readFrom(in);
187         return nodeInfo;
188     }

190     @Override
191     public void readFrom(StreamInput in) throws IOException {
192         super.readFrom(in);
193         timestamp = in.readVLong();
194         if (in.readBoolean()) {
195             indices = NodeIndicesStats.readIndicesStats(in);
196         }
197         if (in.readBoolean()) {
198             os = OsStats.readOsStats(in);
199         }
200         if (in.readBoolean()) {
201             process = ProcessStats.readProcessStats(in);
202         }
203         if (in.readBoolean()) {
204             jvm = JvmStats.readJvmStats(in);
205         }
206         if (in.readBoolean()) {
207             threadPool = ThreadPoolStats.readThreadPoolStats(in);
208         }
209         if (in.readBoolean()) {
210             network = NetworkStats.readNetworkStats(in);
211         }
212         if (in.readBoolean()) {
213             fs = FsStats.readFsStats(in);
214         }
215         if (in.readBoolean()) {
216             transport = TransportStats.readTransportStats(in);
217         }
218         if (in.readBoolean()) {
219             http = HttpStats.readHttpStats(in);
220         }
221         breaker = AllCircuitBreakerStats.readOptionalAllCircuitBreakerStats(in);
222 
223     }
224 
225     @Override
226     public void writeTo(StreamOutput out) throws IOException {
227         super.writeTo(out);
228         out.writeVLong(timestamp);
229         if (indices == null) {
230             out.writeBoolean(false);
231         } else {
232             out.writeBoolean(true);
233             indices.writeTo(out);
234         }
235         if (os == null) {
236             out.writeBoolean(false);
237         } else {
238             out.writeBoolean(true);
239             os.writeTo(out);
240         }
241         if (process == null) {
242             out.writeBoolean(false);
243         } else {
244             out.writeBoolean(true);
245             process.writeTo(out);
246         }
247         if (jvm == null) {
248             out.writeBoolean(false);
249         } else {
250             out.writeBoolean(true);
251             jvm.writeTo(out);
252         }
253         if (threadPool == null) {
254             out.writeBoolean(false);
255         } else {
256             out.writeBoolean(true);
257             threadPool.writeTo(out);
258         }
259         if (network == null) {
260             out.writeBoolean(false);
261         } else {
262             out.writeBoolean(true);
263             network.writeTo(out);
264         }
265         if (fs == null) {
266             out.writeBoolean(false);
267         } else {
268             out.writeBoolean(true);
269             fs.writeTo(out);
270         }
271         if (transport == null) {
272             out.writeBoolean(false);
273         } else {
274             out.writeBoolean(true);
275             transport.writeTo(out);
276         }
277         if (http == null) {
278             out.writeBoolean(false);
279         } else {
280             out.writeBoolean(true);
281             http.writeTo(out);
282         }
283         out.writeOptionalStreamable(breaker);
284     }

286     @Override
287     public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
288         if (!params.param("node_info_format", "default").equals("none")) {
289             builder.field("name", getNode().name(), XContentBuilder.FieldCaseConversion.NONE);
290             builder.field("transport_address", getNode().address().toString(), XContentBuilder.FieldCaseConversion.NONE);
291             builder.field("host", getNode().getHostName(), XContentBuilder.FieldCaseConversion.NONE);
292             builder.field("ip", getNode().getAddress(), XContentBuilder.FieldCaseConversion.NONE);
293 
294             if (!getNode().attributes().isEmpty()) {
295                 builder.startObject("attributes");
296                 for (Map.Entry<String, String> attr : getNode().attributes().entrySet()) {
297                     builder.field(attr.getKey(), attr.getValue(), XContentBuilder.FieldCaseConversion.NONE);
298                 }
299                 builder.endObject();
300             }
301         }
302 
303         if (getIndices() != null) {
304             getIndices().toXContent(builder, params);
305         }
306 
307         if (getOs() != null) {
308             getOs().toXContent(builder, params);
309         }
310         if (getProcess() != null) {
311             getProcess().toXContent(builder, params);
312         }
313         if (getJvm() != null) {
314             getJvm().toXContent(builder, params);
315         }
316         if (getThreadPool() != null) {
317             getThreadPool().toXContent(builder, params);
318         }
319         if (getNetwork() != null) {
320             getNetwork().toXContent(builder, params);
321         }
322         if (getFs() != null) {
323             getFs().toXContent(builder, params);
324         }
325         if (getTransport() != null) {
326             getTransport().toXContent(builder, params);
327         }
328         if (getHttp() != null) {
329             getHttp().toXContent(builder, params);
330         }
331         if (getBreaker() != null) {
332             getBreaker().toXContent(builder, params);
333         }
334 
335         return builder;
336     }
