55  /**
56   * FieldMapper for indexing {@link com.spatial4j.core.shape.Shape}s.
57   * <p/>
58   * Currently Shapes can only be indexed and can only be queried using
59   * {@link org.elasticsearch.index.query.GeoShapeFilterParser}, consequently
60   * a lot of behavior in this Mapper is disabled.
61   * <p/>
62   * Format supported:
63   * <p/>
64   * "field" : {
65   * "type" : "polygon",
66   * "coordinates" : [
67   * [ [100.0, 0.0], [101.0, 0.0], [101.0, 1.0], [100.0, 1.0], [100.0, 0.0] ]
68   * ]
69   * }
70   */
71  public class GeoShapeFieldMapper extends AbstractFieldMapper<String> {
72  
73      public static final String CONTENT_TYPE = "geo_shape";
74  
75      public static class Names {
76          public static final String TREE = "tree";
77          public static final String TREE_GEOHASH = "geohash";
78          public static final String TREE_QUADTREE = "quadtree";
79          public static final String TREE_LEVELS = "tree_levels";
80          public static final String TREE_PRESISION = "precision";
81          public static final String DISTANCE_ERROR_PCT = "distance_error_pct";
82          public static final String ORIENTATION = "orientation";
83          public static final String STRATEGY = "strategy";
84      }
85  
86      public static class Defaults {
87          public static final String TREE = Names.TREE_GEOHASH;
88          public static final String STRATEGY = SpatialStrategy.RECURSIVE.getStrategyName();
89          public static final int GEOHASH_LEVELS = GeoUtils.geoHashLevelsForPrecision("50m");
90          public static final int QUADTREE_LEVELS = GeoUtils.quadTreeLevelsForPrecision("50m");
91          public static final double DISTANCE_ERROR_PCT = 0.025d;
92          public static final Orientation ORIENTATION = Orientation.RIGHT;
93  
94          public static final FieldType FIELD_TYPE = new FieldType();
95  
96          static {
97              FIELD_TYPE.setIndexOptions(IndexOptions.DOCS);
98              FIELD_TYPE.setTokenized(false);
99              FIELD_TYPE.setStored(false);
100             FIELD_TYPE.setStoreTermVectors(false);
101             FIELD_TYPE.setOmitNorms(true);
102             FIELD_TYPE.freeze();
103         }
104     }

106     public static class Builder extends AbstractFieldMapper.Builder<Builder, GeoShapeFieldMapper> {
107 
108         private String tree = Defaults.TREE;
109         private String strategyName = Defaults.STRATEGY;
110         private int treeLevels = 0;
111         private double precisionInMeters = -1;
112         private double distanceErrorPct = Defaults.DISTANCE_ERROR_PCT;
113         private Orientation orientation = Defaults.ORIENTATION;
114 
115         private SpatialPrefixTree prefixTree;
116 
117         public Builder(String name) {
118             super(name, new FieldType(Defaults.FIELD_TYPE));
119         }
120 
121         public Builder tree(String tree) {
122             this.tree = tree;
123             return this;
124         }

127             this.strategyName = strategy;
128             return this;
129         }

138             return this;
139         }

142             this.distanceErrorPct = distanceErrorPct;
143             return this;
144         }
145 
146         public Builder orientation(Orientation orientation) {
147             this.orientation = orientation;
148             return this;
149         }

151         @Override
152         public GeoShapeFieldMapper build(BuilderContext context) {
153 
154             final FieldMapper.Names names = buildNames(context);
155             if (Names.TREE_GEOHASH.equals(tree)) {
156                 prefixTree = new GeohashPrefixTree(ShapeBuilder.SPATIAL_CONTEXT, getLevels(treeLevels, precisionInMeters, Defaults.GEOHASH_LEVELS, true));
157             } else if (Names.TREE_QUADTREE.equals(tree)) {
158                 prefixTree = new QuadPrefixTree(ShapeBuilder.SPATIAL_CONTEXT, getLevels(treeLevels, precisionInMeters, Defaults.QUADTREE_LEVELS, false));
159             } else {
160                 throw new ElasticsearchIllegalArgumentException("Unknown prefix tree type [" + tree + "]");
161             }
162 
163             return new GeoShapeFieldMapper(names, prefixTree, strategyName, distanceErrorPct, orientation, fieldType,
164                     context.indexSettings(), multiFieldsBuilder.build(this, context), copyTo);
165         }
166     }
167 
168     private static final int getLevels(int treeLevels, double precisionInMeters, int defaultLevels, boolean geoHash) {
169         if (treeLevels > 0 || precisionInMeters >= 0) {
170             return Math.max(treeLevels, precisionInMeters >= 0 ? (geoHash ? GeoUtils.geoHashLevelsForPrecision(precisionInMeters)
171                     : GeoUtils.quadTreeLevelsForPrecision(precisionInMeters)) : 0);
172         }
173         return defaultLevels;
174     }

177     public static class TypeParser implements Mapper.TypeParser {
178 
179         @Override
180         public Mapper.Builder parse(String name, Map<String, Object> node, ParserContext parserContext) throws MapperParsingException {
181             Builder builder = geoShapeField(name);
182 
183             for (Iterator<Map.Entry<String, Object>> iterator = node.entrySet().iterator(); iterator.hasNext();) {
184                 Map.Entry<String, Object> entry = iterator.next();
185                 String fieldName = Strings.toUnderscoreCase(entry.getKey());
186                 Object fieldNode = entry.getValue();
187                 if (Names.TREE.equals(fieldName)) {
188                     builder.tree(fieldNode.toString());
189                     iterator.remove();
190                 } else if (Names.TREE_LEVELS.equals(fieldName)) {
191                     builder.treeLevels(Integer.parseInt(fieldNode.toString()));
192                     iterator.remove();
193                 } else if (Names.TREE_PRESISION.equals(fieldName)) {
194                     builder.treeLevelsByDistance(DistanceUnit.parse(fieldNode.toString(), DistanceUnit.DEFAULT, DistanceUnit.DEFAULT));
195                     iterator.remove();
196                 } else if (Names.DISTANCE_ERROR_PCT.equals(fieldName)) {
197                     builder.distanceErrorPct(Double.parseDouble(fieldNode.toString()));
198                     iterator.remove();
199                 } else if (Names.ORIENTATION.equals(fieldName)) {
200                     builder.orientation(ShapeBuilder.orientationFromString(fieldNode.toString()));
201                     iterator.remove();
202                 } else if (Names.STRATEGY.equals(fieldName)) {
203                     builder.strategy(fieldNode.toString());
204                     iterator.remove();
205                 }
206             }
207             return builder;
208         }
209     }

211     private final PrefixTreeStrategy defaultStrategy;
212     private final RecursivePrefixTreeStrategy recursiveStrategy;
213     private final TermQueryPrefixTreeStrategy termStrategy;
214     private Orientation shapeOrientation;
215 
216     public GeoShapeFieldMapper(FieldMapper.Names names, SpatialPrefixTree tree, String defaultStrategyName, double distanceErrorPct,
217                                Orientation shapeOrientation, FieldType fieldType, Settings indexSettings, MultiFields multiFields, CopyTo copyTo) {
218         super(names, 1, fieldType, null, null, null, null, null, null, indexSettings, multiFields, copyTo);
219         this.recursiveStrategy = new RecursivePrefixTreeStrategy(tree, names.indexName());
220         this.recursiveStrategy.setDistErrPct(distanceErrorPct);
221         this.termStrategy = new TermQueryPrefixTreeStrategy(tree, names.indexName());
222         this.termStrategy.setDistErrPct(distanceErrorPct);
223         this.defaultStrategy = resolveStrategy(defaultStrategyName);
224         this.shapeOrientation = shapeOrientation;
225     }

242     @Override
243     public void parse(ParseContext context) throws IOException {
244         try {
245             Shape shape = context.parseExternalValue(Shape.class);
246             if (shape == null) {
247                 ShapeBuilder shapeBuilder = ShapeBuilder.parse(context.parser(), this);
248                 if (shapeBuilder == null) {
249                     return;
250                 }
251                 shape = shapeBuilder.build();
252             }
253             Field[] fields = defaultStrategy.createIndexableFields(shape);
254             if (fields == null || fields.length == 0) {
255                 return;
256             }
257             for (Field field : fields) {
258                 if (!customBoost()) {
259                     field.setBoost(boost);
260                 }
261                 if (context.listener().beforeFieldAdded(this, field, context)) {
262                     context.doc().add(field);
263                 }
264             }
265         } catch (Exception e) {
266             throw new MapperParsingException("failed to parse [" + names.fullName() + "]", e);
267         }
268     }

274     @Override
275     protected void doXContentBody(XContentBuilder builder, boolean includeDefaults, Params params) throws IOException {
276         builder.field("type", contentType());
277 
278         // TODO: Come up with a better way to get the name, maybe pass it from builder
279         if (defaultStrategy.getGrid() instanceof GeohashPrefixTree) {
280             // Don't emit the tree name since GeohashPrefixTree is the default
281             // Only emit the tree levels if it isn't the default value
282             if (includeDefaults || defaultStrategy.getGrid().getMaxLevels() != Defaults.GEOHASH_LEVELS) {
283                 builder.field(Names.TREE_LEVELS, defaultStrategy.getGrid().getMaxLevels());
284             }
285         } else {
286             builder.field(Names.TREE, Names.TREE_QUADTREE);
287             if (includeDefaults || defaultStrategy.getGrid().getMaxLevels() != Defaults.QUADTREE_LEVELS) {
288                 builder.field(Names.TREE_LEVELS, defaultStrategy.getGrid().getMaxLevels());
289             }
290         }
291 
292         if (includeDefaults || defaultStrategy.getDistErrPct() != Defaults.DISTANCE_ERROR_PCT) {
293             builder.field(Names.DISTANCE_ERROR_PCT, defaultStrategy.getDistErrPct());
294         }
295     }

321     public PrefixTreeStrategy resolveStrategy(String strategyName) {
322         if (SpatialStrategy.RECURSIVE.getStrategyName().equals(strategyName)) {
323             return recursiveStrategy;
324         }
325         if (SpatialStrategy.TERM.getStrategyName().equals(strategyName)) {
326             return termStrategy;
327         }
328         throw new ElasticsearchIllegalArgumentException("Unknown prefix tree strategy [" + strategyName + "]");
329     }
