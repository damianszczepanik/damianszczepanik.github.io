17  package org.elasticsearch.common.inject;
18  
19  import com.google.common.collect.ImmutableSet;
20  import com.google.common.collect.Lists;
21  import com.google.common.collect.Maps;
22  import org.elasticsearch.common.inject.internal.*;
23  import org.elasticsearch.common.inject.spi.InjectionPoint;
24  import org.elasticsearch.common.inject.spi.TypeListenerBinding;
25  
26  import java.lang.annotation.Annotation;
27  import java.util.ArrayList;
28  import java.util.Collections;
29  import java.util.List;
30  import java.util.Map;
31  
32  import static com.google.common.base.Preconditions.checkNotNull;
33  
34  /**
35   * @author jessewilson@google.com (Jesse Wilson)
36   */
37  class InheritingState implements State {
38  
39      private final State parent;
40  
41      // Must be a linked hashmap in order to preserve order of bindings in Modules.
42      private final Map<Key<?>, Binding<?>> explicitBindingsMutable = Maps.newLinkedHashMap();
43      private final Map<Key<?>, Binding<?>> explicitBindings
44              = Collections.unmodifiableMap(explicitBindingsMutable);
45      private final Map<Class<? extends Annotation>, Scope> scopes = Maps.newHashMap();
46      private final List<MatcherAndConverter> converters = Lists.newArrayList();
47      private final List<TypeListenerBinding> listenerBindings = Lists.newArrayList();
48      private WeakKeySet blacklistedKeys = new WeakKeySet();
49      private final Object lock;
50  
51      InheritingState(State parent) {
52          this.parent = checkNotNull(parent, "parent");
53          this.lock = (parent == State.NONE) ? this : parent.lock();
54      }
55  
56      @Override
57      public State parent() {
58          return parent;
59      }
60  
61      @Override
62      @SuppressWarnings("unchecked") // we only put in BindingImpls that match their key types
63      public <T> BindingImpl<T> getExplicitBinding(Key<T> key) {
64          Binding<?> binding = explicitBindings.get(key);
65          return binding != null ? (BindingImpl<T>) binding : parent.getExplicitBinding(key);
66      }
67  
68      @Override
69      public Map<Key<?>, Binding<?>> getExplicitBindingsThisLevel() {
70          return explicitBindings;
71      }
72  
73      @Override
74      public void putBinding(Key<?> key, BindingImpl<?> binding) {
75          explicitBindingsMutable.put(key, binding);
76      }
77  
78      @Override
79      public Scope getScope(Class<? extends Annotation> annotationType) {
80          Scope scope = scopes.get(annotationType);
81          return scope != null ? scope : parent.getScope(annotationType);
82      }
83  
84      @Override
85      public void putAnnotation(Class<? extends Annotation> annotationType, Scope scope) {
86          scopes.put(annotationType, scope);
87      }
88  
89      @Override
90      public Iterable<MatcherAndConverter> getConvertersThisLevel() {
91          return converters;
92      }
93  
94      @Override
95      public void addConverter(MatcherAndConverter matcherAndConverter) {
96          converters.add(matcherAndConverter);
97      }
98  
99      @Override
100     public MatcherAndConverter getConverter(
101             String stringValue, TypeLiteral<?> type, Errors errors, Object source) {
102         MatcherAndConverter matchingConverter = null;
103         for (State s = this; s != State.NONE; s = s.parent()) {
104             for (MatcherAndConverter converter : s.getConvertersThisLevel()) {
105                 if (converter.getTypeMatcher().matches(type)) {
106                     if (matchingConverter != null) {
107                         errors.ambiguousTypeConversion(stringValue, source, type, matchingConverter, converter);
108                     }
109                     matchingConverter = converter;
110                 }
111             }
112         }
113         return matchingConverter;
114     }
115 
116     @Override
117     public void addTypeListener(TypeListenerBinding listenerBinding) {
118         listenerBindings.add(listenerBinding);
119     }
120 
121     @Override
122     public List<TypeListenerBinding> getTypeListenerBindings() {
123         List<TypeListenerBinding> parentBindings = parent.getTypeListenerBindings();
124         List<TypeListenerBinding> result
125                 = new ArrayList<>(parentBindings.size() + 1);
126         result.addAll(parentBindings);
127         result.addAll(listenerBindings);
128         return result;
129     }
130 
131     @Override
132     public void blacklist(Key<?> key) {
133         parent.blacklist(key);
134         blacklistedKeys.add(key);
135     }
136 
137     @Override
138     public boolean isBlacklisted(Key<?> key) {
139         return blacklistedKeys.contains(key);
140     }
141 
142     @Override
143     public void clearBlacklisted() {
144         blacklistedKeys = new WeakKeySet();
145     }
146 
147     @Override
148     public void makeAllBindingsToEagerSingletons(Injector injector) {
149         Map<Key<?>, Binding<?>> x = Maps.newLinkedHashMap();
150         for (Map.Entry<Key<?>, Binding<?>> entry : this.explicitBindingsMutable.entrySet()) {
151             Key key = entry.getKey();
152             BindingImpl<?> binding = (BindingImpl<?>) entry.getValue();
153             Object value = binding.getProvider().get();
154             x.put(key, new InstanceBindingImpl<Object>(injector, key, SourceProvider.UNKNOWN_SOURCE, new InternalFactory.Instance(value), ImmutableSet.<InjectionPoint>of(), value));
155         }
156         this.explicitBindingsMutable.clear();
157         this.explicitBindingsMutable.putAll(x);
158     }
159 
160     @Override
161     public Object lock() {
162         return lock;
163     }
164 }
