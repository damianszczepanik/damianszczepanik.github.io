20  package org.elasticsearch.index.query;
21  
22  import com.google.common.collect.Lists;
23  import com.google.common.collect.Maps;
24  import org.elasticsearch.common.inject.AbstractModule;
25  import org.elasticsearch.common.inject.Scopes;
26  import org.elasticsearch.common.inject.assistedinject.FactoryProvider;
27  import org.elasticsearch.common.inject.multibindings.MapBinder;
28  import org.elasticsearch.common.settings.Settings;
29  import org.elasticsearch.index.query.support.InnerHitsQueryParserHelper;
30  
31  import java.util.LinkedList;
32  import java.util.Map;
33  
34  /**
35   *
36   */
37  public class IndexQueryParserModule extends AbstractModule {
38  
39      /**
40       * A custom processor that can be extended to process and bind custom implementations of
41       * {@link QueryParserFactory}, and {@link FilterParser}.
42       */
43      public static class QueryParsersProcessor {
44  
45          /**
46           * Extension point to bind a custom {@link QueryParserFactory}.
47           */
48          public void processXContentQueryParsers(XContentQueryParsersBindings bindings) {
49  
50          }
51  
52          public static class XContentQueryParsersBindings {
53  
54              private final MapBinder<String, QueryParserFactory> binder;
55              private final Map<String, Settings> groupSettings;
56  
57              public XContentQueryParsersBindings(MapBinder<String, QueryParserFactory> binder, Map<String, Settings> groupSettings) {
58                  this.binder = binder;
59                  this.groupSettings = groupSettings;
60              }
61  
62              public MapBinder<String, QueryParserFactory> binder() {
63                  return binder;
64              }
65  
66              public Map<String, Settings> groupSettings() {
67                  return groupSettings;
68              }
69  
70              public void processXContentQueryParser(String name, Class<? extends QueryParser> xcontentQueryParser) {
71                  if (!groupSettings.containsKey(name)) {
72                      binder.addBinding(name).toProvider(FactoryProvider.newFactory(QueryParserFactory.class, xcontentQueryParser)).in(Scopes.SINGLETON);
73                  }
74              }
75          }
76  
77          /**
78           * Extension point to bind a custom {@link FilterParserFactory}.
79           */
80          public void processXContentFilterParsers(XContentFilterParsersBindings bindings) {
81  
82          }
83  
84          public static class XContentFilterParsersBindings {
85  
86              private final MapBinder<String, FilterParserFactory> binder;
87              private final Map<String, Settings> groupSettings;
88  
89              public XContentFilterParsersBindings(MapBinder<String, FilterParserFactory> binder, Map<String, Settings> groupSettings) {
90                  this.binder = binder;
91                  this.groupSettings = groupSettings;
92              }
93  
94              public MapBinder<String, FilterParserFactory> binder() {
95                  return binder;
96              }
97  
98              public Map<String, Settings> groupSettings() {
99                  return groupSettings;
100             }
101 
102             public void processXContentQueryFilter(String name, Class<? extends FilterParser> xcontentFilterParser) {
103                 if (!groupSettings.containsKey(name)) {
104                     binder.addBinding(name).toProvider(FactoryProvider.newFactory(FilterParserFactory.class, xcontentFilterParser)).in(Scopes.SINGLETON);
105                 }
106             }
107         }
108     }
109 
110     private final Settings settings;
111 
112     private final LinkedList<QueryParsersProcessor> processors = Lists.newLinkedList();
113 
114     private final Map<String, Class<? extends QueryParser>> queries = Maps.newHashMap();
115     private final Map<String, Class<? extends FilterParser>> filters = Maps.newHashMap();
116 
117     public IndexQueryParserModule(Settings settings) {
118         this.settings = settings;
119     }
120 
121     /**
122      * Adds a custom query parser.
123      *
124      * @param name        The name of the query parser
125      * @param queryParser the class of the query parser
126      */
127     public void addQueryParser(String name, Class<? extends QueryParser> queryParser) {
128         queries.put(name, queryParser);
129     }
130 
131     /**
132      * Adds a custom filter parser.
133      *
134      * @param name         The name of the filter parser
135      * @param filterParser the class of the filter parser
136      */
137     public void addFilterParser(String name, Class<? extends FilterParser> filterParser) {
138         filters.put(name, filterParser);
139     }
140 
141     public IndexQueryParserModule addProcessor(QueryParsersProcessor processor) {
142         processors.addFirst(processor);
143         return this;
144     }
145 
146     @Override
147     protected void configure() {
148 
149         bind(IndexQueryParserService.class).asEagerSingleton();
150         bind(InnerHitsQueryParserHelper.class).asEagerSingleton();
151 
152         // handle XContenQueryParsers
153         MapBinder<String, QueryParserFactory> queryBinder
154                 = MapBinder.newMapBinder(binder(), String.class, QueryParserFactory.class);
155         Map<String, Settings> xContentQueryParserGroups = settings.getGroups(IndexQueryParserService.Defaults.QUERY_PREFIX);
156         for (Map.Entry<String, Settings> entry : xContentQueryParserGroups.entrySet()) {
157             String qName = entry.getKey();
158             Settings qSettings = entry.getValue();
159             Class<? extends QueryParser> type = qSettings.getAsClass("type", null);
160             if (type == null) {
161                 throw new IllegalArgumentException("Query Parser [" + qName + "] must be provided with a type");
162             }
163             queryBinder.addBinding(qName).toProvider(FactoryProvider.newFactory(QueryParserFactory.class,
164                     qSettings.getAsClass("type", null))).in(Scopes.SINGLETON);
165         }
166 
167         QueryParsersProcessor.XContentQueryParsersBindings xContentQueryParsersBindings = new QueryParsersProcessor.XContentQueryParsersBindings(queryBinder, xContentQueryParserGroups);
168         for (QueryParsersProcessor processor : processors) {
169             processor.processXContentQueryParsers(xContentQueryParsersBindings);
170         }
171 
172         for (Map.Entry<String, Class<? extends QueryParser>> entry : queries.entrySet()) {
173             queryBinder.addBinding(entry.getKey()).toProvider(FactoryProvider.newFactory(QueryParserFactory.class, entry.getValue())).in(Scopes.SINGLETON);
174         }
175 
176         // handle XContentFilterParsers
177         MapBinder<String, FilterParserFactory> filterBinder
178                 = MapBinder.newMapBinder(binder(), String.class, FilterParserFactory.class);
179         Map<String, Settings> xContentFilterParserGroups = settings.getGroups(IndexQueryParserService.Defaults.FILTER_PREFIX);
180         for (Map.Entry<String, Settings> entry : xContentFilterParserGroups.entrySet()) {
181             String fName = entry.getKey();
182             Settings fSettings = entry.getValue();
183             Class<? extends FilterParser> type = fSettings.getAsClass("type", null);
184             if (type == null) {
185                 throw new IllegalArgumentException("Filter Parser [" + fName + "] must be provided with a type");
186             }
187             filterBinder.addBinding(fName).toProvider(FactoryProvider.newFactory(FilterParserFactory.class,
188                     fSettings.getAsClass("type", null))).in(Scopes.SINGLETON);
189         }
190 
191         QueryParsersProcessor.XContentFilterParsersBindings xContentFilterParsersBindings = new QueryParsersProcessor.XContentFilterParsersBindings(filterBinder, xContentFilterParserGroups);
192         for (QueryParsersProcessor processor : processors) {
193             processor.processXContentFilterParsers(xContentFilterParsersBindings);
194         }
195 
196         for (Map.Entry<String, Class<? extends FilterParser>> entry : filters.entrySet()) {
197             filterBinder.addBinding(entry.getKey()).toProvider(FactoryProvider.newFactory(FilterParserFactory.class, entry.getValue())).in(Scopes.SINGLETON);
198         }
199     }
