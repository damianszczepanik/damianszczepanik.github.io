20  package org.elasticsearch.discovery;
21  
22  import org.elasticsearch.ElasticsearchIllegalArgumentException;
23  import org.elasticsearch.cluster.block.ClusterBlock;
24  import org.elasticsearch.cluster.block.ClusterBlockLevel;
25  import org.elasticsearch.common.component.AbstractComponent;
26  import org.elasticsearch.common.inject.Inject;
27  import org.elasticsearch.common.settings.Settings;
28  import org.elasticsearch.common.unit.TimeValue;
29  import org.elasticsearch.node.settings.NodeSettingsService;
30  import org.elasticsearch.rest.RestStatus;
31  
32  import java.util.EnumSet;
33  
34  /**
35   * Exposes common discovery settings that may be supported by all the different discovery implementations
36   */
37  public class DiscoverySettings extends AbstractComponent {
38  
39      public static final String PUBLISH_TIMEOUT = "discovery.zen.publish_timeout";
40      public static final String NO_MASTER_BLOCK = "discovery.zen.no_master_block";
41  
42      public static final TimeValue DEFAULT_PUBLISH_TIMEOUT = TimeValue.timeValueSeconds(30);
43      public static final String DEFAULT_NO_MASTER_BLOCK = "write";
44      public final static int NO_MASTER_BLOCK_ID = 2;
45  
46      public final static ClusterBlock NO_MASTER_BLOCK_ALL = new ClusterBlock(NO_MASTER_BLOCK_ID, "no master", true, true, RestStatus.SERVICE_UNAVAILABLE, ClusterBlockLevel.ALL);
47      public final static ClusterBlock NO_MASTER_BLOCK_WRITES = new ClusterBlock(NO_MASTER_BLOCK_ID, "no master", true, false, RestStatus.SERVICE_UNAVAILABLE, EnumSet.of(ClusterBlockLevel.WRITE, ClusterBlockLevel.METADATA));
48  
49      private volatile ClusterBlock noMasterBlock;
50      private volatile TimeValue publishTimeout = DEFAULT_PUBLISH_TIMEOUT;
51  
52      @Inject
53      public DiscoverySettings(Settings settings, NodeSettingsService nodeSettingsService) {
54          super(settings);
55          nodeSettingsService.addListener(new ApplySettings());
56          this.noMasterBlock = parseNoMasterBlock(settings.get(NO_MASTER_BLOCK, DEFAULT_NO_MASTER_BLOCK));
57          this.publishTimeout = settings.getAsTime(PUBLISH_TIMEOUT, publishTimeout);
58      }
59  
60      /**
61       * Returns the current publish timeout
62       */
63      public TimeValue getPublishTimeout() {
64          return publishTimeout;
65      }
66  
67      public ClusterBlock getNoMasterBlock() {
68          return noMasterBlock;
69      }
70  
71      private class ApplySettings implements NodeSettingsService.Listener {
72          @Override
73          public void onRefreshSettings(Settings settings) {
74              TimeValue newPublishTimeout = settings.getAsTime(PUBLISH_TIMEOUT, null);
75              if (newPublishTimeout != null) {
76                  if (newPublishTimeout.millis() != publishTimeout.millis()) {
77                      logger.info("updating [{}] from [{}] to [{}]", PUBLISH_TIMEOUT, publishTimeout, newPublishTimeout);
78                      publishTimeout = newPublishTimeout;
79                  }
80              }
81              String newNoMasterBlockValue = settings.get(NO_MASTER_BLOCK);
82              if (newNoMasterBlockValue != null) {
83                  ClusterBlock newNoMasterBlock = parseNoMasterBlock(newNoMasterBlockValue);
84                  if (newNoMasterBlock != noMasterBlock) {
85                      noMasterBlock = newNoMasterBlock;
86                  }
87              }
88          }
89      }
90  
91      private ClusterBlock parseNoMasterBlock(String value) {
92          switch (value) {
93              case "all":
94                  return NO_MASTER_BLOCK_ALL;
95              case "write":
96                  return NO_MASTER_BLOCK_WRITES;
97              default:
98                  throw new ElasticsearchIllegalArgumentException("invalid master block [" + value + "]");
99          }
100     }
101 }
