98       * from the provided index. Use {@link #type(String)} and
99       * {@link #id(String)} to specify the document to load.

101     public TermVectorsRequest(TermVectorsRequest other) {
102         super(other.index());
103         this.id = other.id();
104         this.type = other.type();
105         if (this.doc != null) {
106             this.doc = other.doc().copyBytesArray();
107         }
108         this.flagsEnum = other.getFlags().clone();
109         this.preference = other.preference();
110         this.routing = other.routing();
111         if (other.selectedFields != null) {
112             this.selectedFields = new HashSet<>(other.selectedFields);
113         }
114         if (other.perFieldAnalyzer != null) {
115             this.perFieldAnalyzer = new HashMap<>(other.perFieldAnalyzer);
116         }
117         this.realtime = other.realtime();
118         this.version = other.version();
119         this.versionType = VersionType.fromValue(other.versionType().getValue());
120         this.startTime = other.startTime();
121     }

396     private void setFlag(Flag flag, boolean set) {
397         if (set && !flagsEnum.contains(flag)) {
398             flagsEnum.add(flag);
399         } else if (!set) {
400             flagsEnum.remove(flag);
401             assert (!flagsEnum.contains(flag));
402         }
403     }

428     @Override
429     public void readFrom(StreamInput in) throws IOException {
430         super.readFrom(in);
431         type = in.readString();
432         id = in.readString();
433 
434         if (in.readBoolean()) {
435             doc = in.readBytesReference();
436         }
437         routing = in.readOptionalString();
438         preference = in.readOptionalString();
439         long flags = in.readVLong();
440 
441         flagsEnum.clear();
442         for (Flag flag : Flag.values()) {
443             if ((flags & (1 << flag.ordinal())) != 0) {
444                 flagsEnum.add(flag);
445             }
446         }
447         int numSelectedFields = in.readVInt();
448         if (numSelectedFields > 0) {
449             selectedFields = new HashSet<>();
450             for (int i = 0; i < numSelectedFields; i++) {
451                 selectedFields.add(in.readString());
452             }
453         }
454         if (in.readBoolean()) {
455             perFieldAnalyzer = readPerFieldAnalyzer(in.readMap());
456         }
457         realtime = in.readBoolean();
458         versionType = VersionType.fromValue(in.readByte());
459         version = in.readLong();
460     }
461 
462     @Override
463     public void writeTo(StreamOutput out) throws IOException {
464         super.writeTo(out);
465         out.writeString(type);
466         out.writeString(id);
467 
468         out.writeBoolean(doc != null);
469         if (doc != null) {
470             out.writeBytesReference(doc);
471         }
472         out.writeOptionalString(routing);
473         out.writeOptionalString(preference);
474         long longFlags = 0;
475         for (Flag flag : flagsEnum) {
476             longFlags |= (1 << flag.ordinal());
477         }
478         out.writeVLong(longFlags);
479         if (selectedFields != null) {
480             out.writeVInt(selectedFields.size());
481             for (String selectedField : selectedFields) {
482                 out.writeString(selectedField);
483             }
484         } else {
485             out.writeVInt(0);
486         }
487         out.writeBoolean(perFieldAnalyzer != null);
488         if (perFieldAnalyzer != null) {
489             out.writeGenericValue(perFieldAnalyzer);
490         }
491         out.writeBoolean(realtime());
492         out.writeByte(versionType.getValue());
493         out.writeLong(version);
494     }

502     /**
503      * populates a request object (pre-populated with defaults) based on a parser.
504      */
505     public static void parseRequest(TermVectorsRequest termVectorsRequest, XContentParser parser) throws IOException {
506         XContentParser.Token token;
507         String currentFieldName = null;
508         List<String> fields = new ArrayList<>();
509         while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
510             if (token == XContentParser.Token.FIELD_NAME) {
511                 currentFieldName = parser.currentName();
512             } else if (currentFieldName != null) {
513                 if (currentFieldName.equals("fields")) {
514                     if (token == XContentParser.Token.START_ARRAY) {
515                         while (parser.nextToken() != XContentParser.Token.END_ARRAY) {
516                             fields.add(parser.text());
517                         }
518                     } else {
519                         throw new ElasticsearchParseException(
520                                 "The parameter fields must be given as an array! Use syntax : \"fields\" : [\"field1\", \"field2\",...]");
521                     }
522                 } else if (currentFieldName.equals("offsets")) {
523                     termVectorsRequest.offsets(parser.booleanValue());
524                 } else if (currentFieldName.equals("positions")) {
525                     termVectorsRequest.positions(parser.booleanValue());
526                 } else if (currentFieldName.equals("payloads")) {
527                     termVectorsRequest.payloads(parser.booleanValue());
528                 } else if (currentFieldName.equals("term_statistics") || currentFieldName.equals("termStatistics")) {
529                     termVectorsRequest.termStatistics(parser.booleanValue());
530                 } else if (currentFieldName.equals("field_statistics") || currentFieldName.equals("fieldStatistics")) {
531                     termVectorsRequest.fieldStatistics(parser.booleanValue());
532                 } else if (currentFieldName.equals("dfs")) {
533                     termVectorsRequest.dfs(parser.booleanValue());
534                 } else if (currentFieldName.equals("per_field_analyzer") || currentFieldName.equals("perFieldAnalyzer")) {
535                     termVectorsRequest.perFieldAnalyzer(readPerFieldAnalyzer(parser.map()));
536                 } else if ("_index".equals(currentFieldName)) { // the following is important for multi request parsing.
537                     termVectorsRequest.index = parser.text();
538                 } else if ("_type".equals(currentFieldName)) {
539                     termVectorsRequest.type = parser.text();
540                 } else if ("_id".equals(currentFieldName)) {
541                     if (termVectorsRequest.doc != null) {
542                         throw new ElasticsearchParseException("Either \"id\" or \"doc\" can be specified, but not both!");
543                     }
544                     termVectorsRequest.id = parser.text();
545                 } else if ("doc".equals(currentFieldName)) {
546                     if (termVectorsRequest.id != null) {
547                         throw new ElasticsearchParseException("Either \"id\" or \"doc\" can be specified, but not both!");
548                     }
549                     termVectorsRequest.doc(jsonBuilder().copyCurrentStructure(parser));
550                 } else if ("_routing".equals(currentFieldName) || "routing".equals(currentFieldName)) {
551                     termVectorsRequest.routing = parser.text();
552                 } else if ("_version".equals(currentFieldName) || "version".equals(currentFieldName)) {
553                     termVectorsRequest.version = parser.longValue();
554                 } else if ("_version_type".equals(currentFieldName) || "_versionType".equals(currentFieldName) || "version_type".equals(currentFieldName) || "versionType".equals(currentFieldName)) {
555                     termVectorsRequest.versionType = VersionType.fromString(parser.text());
556                 } else {
557                     throw new ElasticsearchParseException("The parameter " + currentFieldName
558                             + " is not valid for term vector request!");
559                 }
560             }
561         }
562         if (fields.size() > 0) {
563             String[] fieldsAsArray = new String[fields.size()];
564             termVectorsRequest.selectedFields(fields.toArray(fieldsAsArray));
565         }
566     }

568     private static Map<String, String> readPerFieldAnalyzer(Map<String, Object> map) {
569         Map<String, String> mapStrStr = new HashMap<>();
570         for (Map.Entry<String, Object> e : map.entrySet()) {
571             if (e.getValue() instanceof String) {
572                 mapStrStr.put(e.getKey(), (String) e.getValue());
573             } else {
574                 throw new ElasticsearchException(
575                         "The analyzer at " + e.getKey() + " should be of type String, but got a " + e.getValue().getClass() + "!");
576             }
577         }
578         return mapStrStr;
579     }
