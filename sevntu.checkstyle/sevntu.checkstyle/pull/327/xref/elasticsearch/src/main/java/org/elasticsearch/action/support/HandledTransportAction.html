19  package org.elasticsearch.action.support;
20  
21  import org.elasticsearch.action.ActionListener;
22  import org.elasticsearch.action.ActionRequest;
23  import org.elasticsearch.action.ActionResponse;
24  import org.elasticsearch.common.settings.Settings;
25  import org.elasticsearch.threadpool.ThreadPool;
26  import org.elasticsearch.transport.BaseTransportRequestHandler;
27  import org.elasticsearch.transport.TransportChannel;
28  import org.elasticsearch.transport.TransportService;
29  
30  /**
31   * A TransportAction that self registers a handler into the transport service
32   */
33  public abstract class HandledTransportAction<Request extends ActionRequest, Response extends ActionResponse> extends TransportAction<Request,Response>{
34  
35      /**
36       * Sub classes implement this call to get new instance of a Request object
37       * @return Request
38       */
39      public abstract Request newRequestInstance();
40  
41      protected HandledTransportAction(Settings settings, String actionName, ThreadPool threadPool, TransportService transportService, ActionFilters actionFilters){
42          super(settings, actionName, threadPool, actionFilters);
43          transportService.registerHandler(actionName, new TransportHandler() {
44              @Override
45              public Request newInstance(){
46                  return newRequestInstance();
47              }
48          });
49      }
50  
51      private abstract class TransportHandler extends BaseTransportRequestHandler<Request>{
52  
53          /**
54           * Call to get an instance of type Request
55           * @return Request
56           */
57          @Override
58          public abstract Request newInstance();
59  
60          @Override
61          public final void messageReceived(final Request request, final TransportChannel channel) throws Exception {
62              // no need to use threaded listener, since we just send a response
63              request.listenerThreaded(false);
64              execute(request, new ActionListener<Response>() {
65                  @Override
66                  public void onResponse(Response response) {
67                      try {
68                          channel.sendResponse(response);
69                      } catch (Throwable e) {
70                          onFailure(e);
71                      }
72                  }
73  
74                  @Override
75                  public void onFailure(Throwable e) {
76                      try {
77                          channel.sendResponse(e);
78                      } catch (Exception e1) {
79                          logger.warn("Failed to send error response for action [{}] and request [{}]",
80                                  actionName, request, e1);
81                      }
82                  }
83              });
84          }
85  
86          @Override
87          public String executor() {
88              return ThreadPool.Names.SAME;
89          }
90  
91      }
92  
93  }
