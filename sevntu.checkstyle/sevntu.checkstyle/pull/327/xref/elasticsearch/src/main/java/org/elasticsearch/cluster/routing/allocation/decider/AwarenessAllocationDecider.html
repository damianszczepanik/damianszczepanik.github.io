38  /**
39   * This {@link AllocationDecider} controls shard allocation based on
40   * <tt>awareness</tt> key-value pairs defined in the node configuration.
41   * Awareness explicitly controls where replicas should be allocated based on
42   * attributes like node or physical rack locations. Awareness attributes accept
43   * arbitrary configuration keys like a rack data-center identifier. For example
44   * the setting:
45   * <p/>
46   * <pre>
47   * cluster.routing.allocation.awareness.attributes: rack_id
48   * </pre>
49   * <p/>
50   * will cause allocations to be distributed over different racks such that
51   * ideally at least one replicas of the all shard is available on the same rack.
52   * To enable allocation awareness in this example nodes should contain a value
53   * for the <tt>rack_id</tt> key like:
54   * <p/>
55   * <pre>
56   * node.rack_id:1
57   * </pre>
58   * <p/>
59   * Awareness can also be used to prevent over-allocation in the case of node or
60   * even "zone" failure. For example in cloud-computing infrastructures like
61   * Amazone AWS a cluster might span over multiple "zones". Awareness can be used
62   * to distribute replicas to individual zones by setting:
63   * <p/>
64   * <pre>
65   * cluster.routing.allocation.awareness.attributes: zone
66   * </pre>
67   * <p/>
68   * and forcing allocation to be aware of the following zone the data resides in:
69   * <p/>
70   * <pre>
71   * cluster.routing.allocation.awareness.force.zone.values: zone1,zone2
72   * </pre>
73   * <p/>
74   * In contrast to regular awareness this setting will prevent over-allocation on
75   * <tt>zone1</tt> even if <tt>zone2</tt> fails partially or becomes entirely
76   * unavailable. Nodes that belong to a certain zone / group should be started
77   * with the zone id configured on the node-level settings like:
78   * <p/>
79   * <pre>
80   * node.zone: zone1
81   * </pre>
82   */
83  public class AwarenessAllocationDecider extends AllocationDecider {
84  
85      public static final String NAME = "awareness";
86  
87      public static final String CLUSTER_ROUTING_ALLOCATION_AWARENESS_ATTRIBUTES = "cluster.routing.allocation.awareness.attributes";
88      public static final String CLUSTER_ROUTING_ALLOCATION_AWARENESS_FORCE_GROUP = "cluster.routing.allocation.awareness.force.";
89  
90      class ApplySettings implements NodeSettingsService.Listener {
91          @Override
92          public void onRefreshSettings(Settings settings) {
93              String[] awarenessAttributes = settings.getAsArray(CLUSTER_ROUTING_ALLOCATION_AWARENESS_ATTRIBUTES, null);
94              if (awarenessAttributes == null && "".equals(settings.get(CLUSTER_ROUTING_ALLOCATION_AWARENESS_ATTRIBUTES, null))) {
95                  awarenessAttributes = Strings.EMPTY_ARRAY; // the empty string resets this
96              }
97              if (awarenessAttributes != null) {
98                  logger.info("updating [cluster.routing.allocation.awareness.attributes] from [{}] to [{}]", AwarenessAllocationDecider.this.awarenessAttributes, awarenessAttributes);
99                  AwarenessAllocationDecider.this.awarenessAttributes = awarenessAttributes;
100             }
101             Map<String, String[]> forcedAwarenessAttributes = new HashMap<>(AwarenessAllocationDecider.this.forcedAwarenessAttributes);
102             Map<String, Settings> forceGroups = settings.getGroups(CLUSTER_ROUTING_ALLOCATION_AWARENESS_FORCE_GROUP);
103             if (!forceGroups.isEmpty()) {
104                 for (Map.Entry<String, Settings> entry : forceGroups.entrySet()) {
105                     String[] aValues = entry.getValue().getAsArray("values");
106                     if (aValues.length > 0) {
107                         forcedAwarenessAttributes.put(entry.getKey(), aValues);
108                     }
109                 }
110             }
111             AwarenessAllocationDecider.this.forcedAwarenessAttributes = forcedAwarenessAttributes;
112         }
113     }
114 
115     private String[] awarenessAttributes;
116 
117     private Map<String, String[]> forcedAwarenessAttributes;
118 
119     /**
120      * Creates a new {@link AwarenessAllocationDecider} instance
121      */
122     public AwarenessAllocationDecider() {
123         this(ImmutableSettings.Builder.EMPTY_SETTINGS);
124     }
125 
126     /**
127      * Creates a new {@link AwarenessAllocationDecider} instance from given settings
128      *
129      * @param settings {@link Settings} to use
130      */
131     public AwarenessAllocationDecider(Settings settings) {
132         this(settings, new NodeSettingsService(settings));
133     }
134 
135     @Inject
136     public AwarenessAllocationDecider(Settings settings, NodeSettingsService nodeSettingsService) {
137         super(settings);
138         this.awarenessAttributes = settings.getAsArray(CLUSTER_ROUTING_ALLOCATION_AWARENESS_ATTRIBUTES);
139 
140         forcedAwarenessAttributes = Maps.newHashMap();
141         Map<String, Settings> forceGroups = settings.getGroups(CLUSTER_ROUTING_ALLOCATION_AWARENESS_FORCE_GROUP);
142         for (Map.Entry<String, Settings> entry : forceGroups.entrySet()) {
143             String[] aValues = entry.getValue().getAsArray("values");
144             if (aValues.length > 0) {
145                 forcedAwarenessAttributes.put(entry.getKey(), aValues);
146             }
147         }
148 
149         nodeSettingsService.addListener(new ApplySettings());
150     }
151 
152     /**
153      * Get the attributes defined by this instance
154      *
155      * @return attributes defined by this instance
156      */
157     public String[] awarenessAttributes() {
158         return this.awarenessAttributes;
159     }
160 
161     @Override
162     public Decision canAllocate(ShardRouting shardRouting, RoutingNode node, RoutingAllocation allocation) {
163         return underCapacity(shardRouting, node, allocation, true);
164     }
165 
166     @Override
167     public Decision canRemain(ShardRouting shardRouting, RoutingNode node, RoutingAllocation allocation) {
168         return underCapacity(shardRouting, node, allocation, false);
169     }
170 
171     private Decision underCapacity(ShardRouting shardRouting, RoutingNode node, RoutingAllocation allocation, boolean moveToNode) {
172         if (awarenessAttributes.length == 0) {
173             return allocation.decision(Decision.YES, NAME, "no allocation awareness enabled");
174         }
175 
176         IndexMetaData indexMetaData = allocation.metaData().index(shardRouting.index());
177         int shardCount = indexMetaData.numberOfReplicas() + 1; // 1 for primary
178         for (String awarenessAttribute : awarenessAttributes) {
179             // the node the shard exists on must be associated with an awareness attribute
180             if (!node.node().attributes().containsKey(awarenessAttribute)) {
181                 return allocation.decision(Decision.NO, NAME, "node does not contain awareness attribute: [%s]", awarenessAttribute);
182             }
183 
184             // build attr_value -> nodes map
185             ObjectIntOpenHashMap<String> nodesPerAttribute = allocation.routingNodes().nodesPerAttributesCounts(awarenessAttribute);
186 
187             // build the count of shards per attribute value
188             ObjectIntOpenHashMap<String> shardPerAttribute = new ObjectIntOpenHashMap<>();
189             for (MutableShardRouting assignedShard : allocation.routingNodes().assignedShards(shardRouting)) {
190                 // if the shard is relocating, then make sure we count it as part of the node it is relocating to
191                 if (assignedShard.relocating()) {
192                     RoutingNode relocationNode = allocation.routingNodes().node(assignedShard.relocatingNodeId());
193                     shardPerAttribute.addTo(relocationNode.node().attributes().get(awarenessAttribute), 1);
194                 } else if (assignedShard.started()) {
195                     RoutingNode routingNode = allocation.routingNodes().node(assignedShard.currentNodeId());
196                     shardPerAttribute.addTo(routingNode.node().attributes().get(awarenessAttribute), 1);
197                 }
198             }
199 
200             if (moveToNode) {
201                 if (shardRouting.assignedToNode()) {
202                     String nodeId = shardRouting.relocating() ? shardRouting.relocatingNodeId() : shardRouting.currentNodeId();
203                     if (!node.nodeId().equals(nodeId)) {
204                         // we work on different nodes, move counts around
205                         shardPerAttribute.putOrAdd(allocation.routingNodes().node(nodeId).node().attributes().get(awarenessAttribute), 0, -1);
206                         shardPerAttribute.addTo(node.node().attributes().get(awarenessAttribute), 1);
207                     }
208                 } else {
209                     shardPerAttribute.addTo(node.node().attributes().get(awarenessAttribute), 1);
210                 }
211             }
212 
213             int numberOfAttributes = nodesPerAttribute.size();
214             String[] fullValues = forcedAwarenessAttributes.get(awarenessAttribute);
215             if (fullValues != null) {
216                 for (String fullValue : fullValues) {
217                     if (!shardPerAttribute.containsKey(fullValue)) {
218                         numberOfAttributes++;
219                     }
220                 }
221             }
222             // TODO should we remove ones that are not part of full list?
223 
224             int averagePerAttribute = shardCount / numberOfAttributes;
225             int totalLeftover = shardCount % numberOfAttributes;
226             int requiredCountPerAttribute;
227             if (averagePerAttribute == 0) {
228                 // if we have more attributes values than shard count, no leftover
229                 totalLeftover = 0;
230                 requiredCountPerAttribute = 1;
231             } else {
232                 requiredCountPerAttribute = averagePerAttribute;
233             }
234             int leftoverPerAttribute = totalLeftover == 0 ? 0 : 1;
235 
236             int currentNodeCount = shardPerAttribute.get(node.node().attributes().get(awarenessAttribute));
237             // if we are above with leftover, then we know we are not good, even with mod
238             if (currentNodeCount > (requiredCountPerAttribute + leftoverPerAttribute)) {
239                 return allocation.decision(Decision.NO, NAME, "too many shards on nodes for attribute: [%s]", awarenessAttribute);
240             }
241             // all is well, we are below or same as average
242             if (currentNodeCount <= requiredCountPerAttribute) {
243                 continue;
244             }
245         }
246 
247         return allocation.decision(Decision.YES, NAME, "node meets awareness requirements");
248     }
