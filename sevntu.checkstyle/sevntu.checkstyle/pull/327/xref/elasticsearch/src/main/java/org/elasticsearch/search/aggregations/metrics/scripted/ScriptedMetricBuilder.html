20  package org.elasticsearch.search.aggregations.metrics.scripted;
21  
22  import org.elasticsearch.common.xcontent.XContentBuilder;
23  import org.elasticsearch.script.ScriptParameterParser;
24  import org.elasticsearch.search.aggregations.metrics.MetricsAggregationBuilder;
25  
26  import java.io.IOException;
27  import java.util.Map;
28  
29  /**
30   * Builder for the {@link ScriptedMetric} aggregation.
31   */
32  public class ScriptedMetricBuilder extends MetricsAggregationBuilder {
33  
34      private Map<String, Object> params = null;
35      private Map<String, Object> reduceParams = null;
36      private String initScript = null;
37      private String mapScript = null;
38      private String combineScript = null;
39      private String reduceScript = null;
40      private String initScriptFile = null;
41      private String mapScriptFile = null;
42      private String combineScriptFile = null;
43      private String reduceScriptFile = null;
44      private String initScriptId = null;
45      private String mapScriptId = null;
46      private String combineScriptId = null;
47      private String reduceScriptId = null;
48      private String lang = null;
49  
50      /**
51       * Sole constructor.
52       */
53      public ScriptedMetricBuilder(String name) {
54          super(name, InternalScriptedMetric.TYPE.name());
55      }
56  
57      /**
58       * Set parameters that will be available in the <tt>init</tt>, <tt>map</tt>
59       * and <tt>combine</tt> phases.
60       */
61      public ScriptedMetricBuilder params(Map<String, Object> params) {
62          this.params = params;
63          return this;
64      }
65  
66      /**
67       * Set parameters that will be available in the <tt>reduce</tt> phase.
68       */
69      public ScriptedMetricBuilder reduceParams(Map<String, Object> reduceParams) {
70          this.reduceParams = reduceParams;
71          return this;
72      }
73  
74      /**
75       * Set the <tt>init</tt> script.
76       */
77      public ScriptedMetricBuilder initScript(String initScript) {
78          this.initScript = initScript;
79          return this;
80      }
81  
82      /**
83       * Set the <tt>map</tt> script.
84       */
85      public ScriptedMetricBuilder mapScript(String mapScript) {
86          this.mapScript = mapScript;
87          return this;
88      }
89  
90      /**
91       * Set the <tt>combine</tt> script.
92       */
93      public ScriptedMetricBuilder combineScript(String combineScript) {
94          this.combineScript = combineScript;
95          return this;
96      }
97  
98      /**
99       * Set the <tt>reduce</tt> script.
100      */
101     public ScriptedMetricBuilder reduceScript(String reduceScript) {
102         this.reduceScript = reduceScript;
103         return this;
104     }
105 
106     /**
107      * Set the <tt>init</tt> script file.
108      */
109     public ScriptedMetricBuilder initScriptFile(String initScriptFile) {
110         this.initScriptFile = initScriptFile;
111         return this;
112     }
113 
114     /**
115      * Set the <tt>map</tt> script file.
116      */
117     public ScriptedMetricBuilder mapScriptFile(String mapScriptFile) {
118         this.mapScriptFile = mapScriptFile;
119         return this;
120     }
121 
122     /**
123      * Set the <tt>combine</tt> script file.
124      */
125     public ScriptedMetricBuilder combineScriptFile(String combineScriptFile) {
126         this.combineScriptFile = combineScriptFile;
127         return this;
128     }
129 
130     /**
131      * Set the <tt>reduce</tt> script file.
132      */
133     public ScriptedMetricBuilder reduceScriptFile(String reduceScriptFile) {
134         this.reduceScriptFile = reduceScriptFile;
135         return this;
136     }
137 
138     /**
139      * Set the indexed <tt>init</tt> script id.
140      */
141     public ScriptedMetricBuilder initScriptId(String initScriptId) {
142         this.initScriptId = initScriptId;
143         return this;
144     }
145 
146     /**
147      * Set the indexed <tt>map</tt> script id.
148      */
149     public ScriptedMetricBuilder mapScriptId(String mapScriptId) {
150         this.mapScriptId = mapScriptId;
151         return this;
152     }
153 
154     /**
155      * Set the indexed <tt>combine</tt> script id.
156      */
157     public ScriptedMetricBuilder combineScriptId(String combineScriptId) {
158         this.combineScriptId = combineScriptId;
159         return this;
160     }
161 
162     /**
163      * Set the indexed <tt>reduce</tt> script id.
164      */
165     public ScriptedMetricBuilder reduceScriptId(String reduceScriptId) {
166         this.reduceScriptId = reduceScriptId;
167         return this;
168     }
169 
170     /**
171      * Set the script language.
172      */
173     public ScriptedMetricBuilder lang(String lang) {
174         this.lang = lang;
175         return this;
176     }
177 
178     @Override
179     protected void internalXContent(XContentBuilder builder, Params builderParams) throws IOException {
180         if (params != null) {
181             builder.field(ScriptedMetricParser.PARAMS_FIELD.getPreferredName());
182             builder.map(params);
183         }
184         
185         if (reduceParams != null) {
186             builder.field(ScriptedMetricParser.REDUCE_PARAMS_FIELD.getPreferredName());
187             builder.map(reduceParams);
188         }
189         
190         if (initScript != null) {
191             builder.field(ScriptedMetricParser.INIT_SCRIPT, initScript);
192         }
193         
194         if (mapScript != null) {
195             builder.field(ScriptedMetricParser.MAP_SCRIPT, mapScript);
196         }
197         
198         if (combineScript != null) {
199             builder.field(ScriptedMetricParser.COMBINE_SCRIPT, combineScript);
200         }
201         
202         if (reduceScript != null) {
203             builder.field(ScriptedMetricParser.REDUCE_SCRIPT, reduceScript);
204         }
205         
206         if (initScriptFile != null) {
207             builder.field(ScriptedMetricParser.INIT_SCRIPT + ScriptParameterParser.FILE_SUFFIX, initScriptFile);
208         }
209         
210         if (mapScriptFile != null) {
211             builder.field(ScriptedMetricParser.MAP_SCRIPT + ScriptParameterParser.FILE_SUFFIX, mapScriptFile);
212         }
213         
214         if (combineScriptFile != null) {
215             builder.field(ScriptedMetricParser.COMBINE_SCRIPT + ScriptParameterParser.FILE_SUFFIX, combineScriptFile);
216         }
217         
218         if (reduceScriptFile != null) {
219             builder.field(ScriptedMetricParser.REDUCE_SCRIPT + ScriptParameterParser.FILE_SUFFIX, reduceScriptFile);
220         }
221         
222         if (initScriptId != null) {
223             builder.field(ScriptedMetricParser.INIT_SCRIPT + ScriptParameterParser.INDEXED_SUFFIX, initScriptId);
224         }
225         
226         if (mapScriptId != null) {
227             builder.field(ScriptedMetricParser.MAP_SCRIPT + ScriptParameterParser.INDEXED_SUFFIX, mapScriptId);
228         }
229         
230         if (combineScriptId != null) {
231             builder.field(ScriptedMetricParser.COMBINE_SCRIPT + ScriptParameterParser.INDEXED_SUFFIX, combineScriptId);
232         }
233         
234         if (reduceScriptId != null) {
235             builder.field(ScriptedMetricParser.REDUCE_SCRIPT + ScriptParameterParser.INDEXED_SUFFIX, reduceScriptId);
236         }
237         
238         if (lang != null) {
239             builder.field(ScriptedMetricParser.LANG_FIELD.getPreferredName(), lang);
240         }
241     }
