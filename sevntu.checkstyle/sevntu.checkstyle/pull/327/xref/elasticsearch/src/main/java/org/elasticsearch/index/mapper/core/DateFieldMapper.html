146     public static class TypeParser implements Mapper.TypeParser {
147         @Override
148         public Mapper.Builder<?, ?> parse(String name, Map<String, Object> node, ParserContext parserContext) throws MapperParsingException {
149             DateFieldMapper.Builder builder = dateField(name);
150             parseNumberField(builder, name, node, parserContext);
151             for (Iterator<Map.Entry<String, Object>> iterator = node.entrySet().iterator(); iterator.hasNext();) {
152                 Map.Entry<String, Object> entry = iterator.next();
153                 String propName = Strings.toUnderscoreCase(entry.getKey());
154                 Object propNode = entry.getValue();
155                 if (propName.equals("null_value")) {
156                     if (propNode == null) {
157                         throw new MapperParsingException("Property [null_value] cannot be null.");
158                     }
159                     builder.nullValue(propNode.toString());
160                     iterator.remove();
161                 } else if (propName.equals("format")) {
162                     builder.dateTimeFormatter(parseDateTimeFormatter(propNode));
163                     iterator.remove();
164                 } else if (propName.equals("numeric_resolution")) {
165                     builder.timeUnit(TimeUnit.valueOf(propNode.toString().toUpperCase(Locale.ROOT)));
166                     iterator.remove();
167                 } else if (propName.equals("locale")) {
168                     builder.locale(LocaleUtils.parse(propNode.toString()));
169                     iterator.remove();
170                 }
171             }
172             return builder;
173         }

220     @Override
221     public Long value(Object value) {
222         if (value == null) {
223             return null;
224         }
225         if (value instanceof Number) {
226             return ((Number) value).longValue();
227         }
228         if (value instanceof BytesRef) {
229             return Numbers.bytesToLong((BytesRef) value);
230         }
231         return parseStringValue(value.toString());
232     }

235     @Override
236     public Object valueForSearch(Object value) {
237         if (value instanceof String) {
238             // assume its the string that was indexed, just return it... (for example, with get)
239             return value;
240         }
241         Long val = value(value);
242         if (val == null) {
243             return null;
244         }
245         return dateTimeFormatter.printer().print(val);
246     }

255     private long parseValue(Object value) {
256         if (value instanceof Number) {
257             return ((Number) value).longValue();
258         }
259         if (value instanceof BytesRef) {
260             return dateTimeFormatter.parser().parseMillis(((BytesRef) value).utf8ToString());
261         }
262         return dateTimeFormatter.parser().parseMillis(value.toString());
263     }
264 
265     private String convertToString(Object value) {
266         if (value instanceof BytesRef) {
267             return ((BytesRef) value).utf8ToString();
268         }
269         return value.toString();
270     }

272     private static Callable<Long> now() {
273         return new Callable<Long>() {
274             @Override
275             public Long call() {
276                 final SearchContext context = SearchContext.current();
277                 return context != null
278                     ? context.nowInMillis()
279                     : System.currentTimeMillis();
280             }
281         };

284     @Override
285     public Query fuzzyQuery(String value, Fuzziness fuzziness, int prefixLength, int maxExpansions, boolean transpositions) {
286         long iValue = dateMathParser.parse(value, now());
287         long iSim;
288         try {
289             iSim = fuzziness.asTimeValue().millis();
290         } catch (Exception e) {
291             // not a time format
292             iSim =  fuzziness.asLong();
293         }
294         return NumericRangeQuery.newLongRange(names.indexName(), precisionStep,
295                 iValue - iSim,
296                 iValue + iSim,
297                 true, true);
298     }

338     public Query rangeQuery(Object lowerTerm, Object upperTerm, boolean includeLower, boolean includeUpper, @Nullable DateTimeZone timeZone, @Nullable DateMathParser forcedDateParser, @Nullable QueryParseContext context) {
339         // If the current search context is null we're parsing percolator query or a index alias filter.

342         } else {

348         return NumericRangeQuery.newLongRange(names.indexName(), precisionStep,

368     /*
369      * `timeZone` parameter is only applied when:
370      * - not null
371      * - the object to parse is a String (does not apply to ms since epoch which are UTC based time values)
372      * - the String to parse does not have already a timezone defined (ie. `2014-01-01T00:00:00+03:00`)
373      */
374     public Filter rangeFilter(QueryParseContext parseContext, Object lowerTerm, Object upperTerm, boolean includeLower, boolean includeUpper, @Nullable DateTimeZone timeZone, @Nullable DateMathParser forcedDateParser, @Nullable QueryParseContext context) {
375         IndexNumericFieldData fieldData = parseContext != null ? (IndexNumericFieldData) parseContext.getForField(this) : null;
376         // If the current search context is null we're parsing percolator query or a index alias filter.
377         if (SearchContext.current() == null) {
378             return new LateParsingFilter(fieldData, lowerTerm, upperTerm, includeLower, includeUpper, timeZone, forcedDateParser);
379         } else {
380             return innerRangeFilter(fieldData, lowerTerm, upperTerm, includeLower, includeUpper, timeZone, forcedDateParser);
381         }
382     }

384     private Filter innerRangeFilter(IndexNumericFieldData fieldData, Object lowerTerm, Object upperTerm, boolean includeLower, boolean includeUpper, @Nullable DateTimeZone timeZone, @Nullable DateMathParser forcedDateParser) {
385         Long lowerVal = null;
386         Long upperVal = null;
387         if (lowerTerm != null) {
388             if (lowerTerm instanceof Number) {
389                 lowerVal = ((Number) lowerTerm).longValue();
390             } else {
391                 String value = convertToString(lowerTerm);
392                 lowerVal = parseToMilliseconds(value, false, timeZone, forcedDateParser);
393             }
394         }
395         if (upperTerm != null) {
396             if (upperTerm instanceof Number) {
397                 upperVal = ((Number) upperTerm).longValue();
398             } else {
399                 String value = convertToString(upperTerm);
400                 upperVal = parseToMilliseconds(value, includeUpper, timeZone, forcedDateParser);
401             }
402         }
403 
404         Filter filter;
405         if (fieldData != null) {
406             filter = NumericRangeFieldDataFilter.newLongRange(fieldData, lowerVal,upperVal, includeLower, includeUpper);
407         } else {
408             filter = NumericRangeFilter.newLongRange(
409                     names.indexName(), precisionStep, lowerVal, upperVal, includeLower, includeUpper
410             );
411         }
412 
413         return filter;
414     }

434     @Override
435     protected void innerParseCreateField(ParseContext context, List<Field> fields) throws IOException {
436         String dateAsString = null;
437         Long value = null;
438         float boost = this.boost;
439         if (context.externalValueSet()) {
440             Object externalValue = context.externalValue();
441             if (externalValue instanceof Number) {
442                 value = ((Number) externalValue).longValue();
443             } else {
444                 dateAsString = (String) externalValue;
445                 if (dateAsString == null) {
446                     dateAsString = nullValue;
447                 }
448             }
449         } else {
450             XContentParser parser = context.parser();
451             XContentParser.Token token = parser.currentToken();
452             if (token == XContentParser.Token.VALUE_NULL) {
453                 dateAsString = nullValue;
454             } else if (token == XContentParser.Token.VALUE_NUMBER) {
455                 value = parser.longValue(coerce.value());
456             } else if (token == XContentParser.Token.START_OBJECT) {
457                 String currentFieldName = null;
458                 while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
459                     if (token == XContentParser.Token.FIELD_NAME) {
460                         currentFieldName = parser.currentName();
461                     } else {
462                         if ("value".equals(currentFieldName) || "_value".equals(currentFieldName)) {
463                             if (token == XContentParser.Token.VALUE_NULL) {
464                                 dateAsString = nullValue;
465                             } else if (token == XContentParser.Token.VALUE_NUMBER) {
466                                 value = parser.longValue(coerce.value());
467                             } else {
468                                 dateAsString = parser.text();
469                             }
470                         } else if ("boost".equals(currentFieldName) || "_boost".equals(currentFieldName)) {
471                             boost = parser.floatValue();
472                         } else {
473                             throw new ElasticsearchIllegalArgumentException("unknown property [" + currentFieldName + "]");
474                         }
475                     }
476                 }
477             } else {
478                 dateAsString = parser.text();
479             }
480         }
481 
482         if (dateAsString != null) {
483             assert value == null;
484             if (context.includeInAll(includeInAll, this)) {
485                 context.allEntries().addText(names.fullName(), dateAsString, boost);
486             }
487             value = parseStringValue(dateAsString);
488         }
489 
490         if (value != null) {
491             if (fieldType.indexOptions() != IndexOptions.NONE || fieldType.stored()) {
492                 CustomLongNumericField field = new CustomLongNumericField(this, value, fieldType);
493                 field.setBoost(boost);
494                 fields.add(field);
495             }
496             if (hasDocValues()) {
497                 addDocValue(context, fields, value);
498             }
499         }
500     }

507     @Override
508     public void merge(Mapper mergeWith, MergeContext mergeContext) throws MergeMappingException {
509         super.merge(mergeWith, mergeContext);
510         if (!this.getClass().equals(mergeWith.getClass())) {
511             return;
512         }
513         if (!mergeContext.mergeFlags().simulate()) {
514             this.nullValue = ((DateFieldMapper) mergeWith).nullValue;
515             this.dateTimeFormatter = ((DateFieldMapper) mergeWith).dateTimeFormatter;
516         }
517     }

519     @Override
520     protected void doXContentBody(XContentBuilder builder, boolean includeDefaults, Params params) throws IOException {
521         super.doXContentBody(builder, includeDefaults, params);
522 
523         if (includeDefaults || precisionStep != Defaults.PRECISION_STEP_64_BIT) {
524             builder.field("precision_step", precisionStep);
525         }
526         builder.field("format", dateTimeFormatter.format());
527         if (includeDefaults || nullValue != null) {
528             builder.field("null_value", nullValue);
529         }
530         if (includeInAll != null) {
531             builder.field("include_in_all", includeInAll);
532         } else if (includeDefaults) {
533             builder.field("include_in_all", false);
534         }
535 
536         if (includeDefaults || timeUnit != Defaults.TIME_UNIT) {
537             builder.field("numeric_resolution", timeUnit.name().toLowerCase(Locale.ROOT));
538         }
539         // only serialize locale if needed, ROOT is the default, so no need to serialize that case as well...
540         if (dateTimeFormatter.locale() != null && dateTimeFormatter.locale() != Locale.ROOT) {
541             builder.field("locale", dateTimeFormatter.locale());
542         } else if (includeDefaults) {
543             if (dateTimeFormatter.locale() == null) {
544                 builder.field("locale", Locale.ROOT);
545             } else {
546                 builder.field("locale", dateTimeFormatter.locale());
547             }
548         }
549     }

551     private long parseStringValue(String value) {
552         try {
553             return dateTimeFormatter.parser().parseMillis(value);
554         } catch (RuntimeException e) {
555             try {
556                 long time = Long.parseLong(value);
557                 return timeUnit.toMillis(time);
558             } catch (NumberFormatException e1) {
559                 throw new MapperParsingException("failed to parse date field [" + value + "], tried both date format [" + dateTimeFormatter.format() + "], and timestamp number with locale [" + dateTimeFormatter.locale() + "]", e);
560             }
561         }

595     public final class LateParsingQuery extends NoCacheQuery {
596 
597         final Object lowerTerm;
598         final Object upperTerm;
599         final boolean includeLower;
600         final boolean includeUpper;
601         final DateTimeZone timeZone;
602         final DateMathParser forcedDateParser;
603 
604         public LateParsingQuery(Object lowerTerm, Object upperTerm, boolean includeLower, boolean includeUpper, DateTimeZone timeZone, DateMathParser forcedDateParser) {
605             this.lowerTerm = lowerTerm;
606             this.upperTerm = upperTerm;
607             this.includeLower = includeLower;
608             this.includeUpper = includeUpper;
609             this.timeZone = timeZone;
610             this.forcedDateParser = forcedDateParser;
611         }

613         @Override

615             Query query = innerRangeQuery(lowerTerm, upperTerm, includeLower, includeUpper, timeZone, forcedDateParser);
616             return query.rewrite(reader);
617         }

619         @Override
620         public String innerToString(String s) {
621             final StringBuilder sb = new StringBuilder();
622             return sb.append(names.indexName()).append(':')
623                     .append(includeLower ? '[' : '{')
624                     .append((lowerTerm == null) ? "*" : lowerTerm.toString())
625                     .append(" TO ")
626                     .append((upperTerm == null) ? "*" : upperTerm.toString())
627                     .append(includeUpper ? ']' : '}')
628                     .append(ToStringUtils.boost(getBoost()))
629                     .toString();
630         }
