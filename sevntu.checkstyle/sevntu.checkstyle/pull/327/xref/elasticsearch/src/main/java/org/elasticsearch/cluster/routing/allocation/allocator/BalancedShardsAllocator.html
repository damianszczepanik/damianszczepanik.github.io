49  /**
50   * The {@link BalancedShardsAllocator} re-balances the nodes allocations
51   * within an cluster based on a {@link WeightFunction}. The clusters balance is defined by four parameters which can be set
52   * in the cluster update API that allows changes in real-time:
53   * <p/>
54   * <ul><li><code>cluster.routing.allocation.balance.shard</code> - The <b>shard balance</b> defines the weight factor
55   * for shards allocated on a {@link RoutingNode}</li>
56   * <li><code>cluster.routing.allocation.balance.index</code> - The <b>index balance</b> defines a factor to the number
57   * of {@link org.elasticsearch.cluster.routing.ShardRouting}s per index allocated on a specific node</li>
58   * <li><code>cluster.routing.allocation.balance.threshold</code> - A <b>threshold</b> to set the minimal optimization
59   * value of operations that should be performed</li>
60   * </ul>
61   * <p/>
62   * These parameters are combined in a {@link WeightFunction} that allows calculation of node weights which
63   * are used to re-balance shards based on global as well as per-index factors.
64   */
65  public class BalancedShardsAllocator extends AbstractComponent implements ShardsAllocator {
66

74      class ApplySettings implements NodeSettingsService.Listener {
75          @Override
76          public void onRefreshSettings(Settings settings) {
77              final float indexBalance = settings.getAsFloat(SETTING_INDEX_BALANCE_FACTOR, weightFunction.indexBalance);
78              final float shardBalance = settings.getAsFloat(SETTING_SHARD_BALANCE_FACTOR, weightFunction.shardBalance);
79              float threshold = settings.getAsFloat(SETTING_THRESHOLD, BalancedShardsAllocator.this.threshold);
80              if (threshold <= 0.0f) {
81                  throw new ElasticsearchIllegalArgumentException("threshold must be greater than 0.0f but was: " + threshold);
82              }
83              BalancedShardsAllocator.this.threshold = threshold;
84              BalancedShardsAllocator.this.weightFunction = new WeightFunction(indexBalance, shardBalance);
85          }
86      }

150     /**
151      * This class is the primary weight function used to create balanced over nodes and shards in the cluster.
152      * Currently this function has 3 properties:
153      * <ul>
154      * <li><code>index balance</code> - balance property over shards per index</li>
155      * <li><code>shard balance</code> - balance property over shards per cluster</li>
156      * </ul>
157      * <p>
158      * Each of these properties are expressed as factor such that the properties factor defines the relative importance of the property for the
159      * weight function. For example if the weight function should calculate the weights only based on a global (shard) balance the index balance
160      * can be set to <tt>0.0</tt> and will in turn have no effect on the distribution.
161      * </p>
162      * The weight per index is calculated based on the following formula:
163      * <ul>
164      * <li>
165      * <code>weight<sub>index</sub>(node, index) = indexBalance * (node.numShards(index) - avgShardsPerNode(index))</code>
166      * </li>
167      * <li>
168      * <code>weight<sub>node</sub>(node, index) = shardBalance * (node.numShards() - avgShardsPerNode)</code>
169      * </li>
170      * </ul>
171      * <code>weight(node, index) = weight<sub>index</sub>(node, index) + weight<sub>node</sub>(node, index)</code>
172      */
173     public static class WeightFunction {
174 
175         private final float indexBalance;
176         private final float shardBalance;
177         private final float[] theta;
178 
179 
180         public WeightFunction(float indexBalance, float shardBalance) {
181             float sum = indexBalance + shardBalance;
182             if (sum <= 0.0f) {
183                 throw new ElasticsearchIllegalArgumentException("Balance factors must sum to a value > 0 but was: " + sum);
184             }
185             theta = new float[]{shardBalance / sum, indexBalance / sum};
186             this.indexBalance = indexBalance;
187             this.shardBalance = shardBalance;
188         }
189 
190         public float weight(Operation operation, Balancer balancer, ModelNode node, String index) {
191             final float weightShard = (node.numShards() - balancer.avgShardsPerNode());
192             final float weightIndex = (node.numShards(index) - balancer.avgShardsPerNode(index));
193             assert theta != null;
194             return theta[0] * weightShard + theta[1] * weightIndex;
195         }
196 
197     }

291         private boolean initialize(RoutingNodes routing, RoutingNodes.UnassignedShards unassigned) {
292             if (logger.isTraceEnabled()) {
293                 logger.trace("Start distributing Shards");
294             }
295             indices.addAll(allocation.routingTable().indicesRouting().keySet());
296             buildModelFromAssigned(routing.shards(assignedFilter));
297             return allocateUnassigned(unassigned, routing.ignoredUnassigned());

300         private static float absDelta(float lower, float higher) {
301             assert higher >= lower : higher + " lt " + lower +" but was expected to be gte";
302             return Math.abs(higher - lower);
303         }
304 
305         private static boolean lessThan(float delta, float threshold) {
306             /* deltas close to the threshold are "rounded" to the threshold manually
307                to prevent floating point problems if the delta is very close to the
308                threshold ie. 1.000000002 which can trigger unnecessary balance actions*/
309             return delta <= (threshold + 0.001f);
310         }

312         /**
313          * Balances the nodes on the cluster model according to the weight
314          * function. The configured threshold is the minimum delta between the
315          * weight of the maximum node and the minimum node according to the
316          * {@link WeightFunction}. This weight is calculated per index to
317          * distribute shards evenly per index. The balancer tries to relocate
318          * shards only if the delta exceeds the threshold. If the default case
319          * the threshold is set to <tt>1.0</tt> to enforce gaining relocation
320          * only, or in other words relocations that move the weight delta closer
321          * to <tt>0.0</tt>
322          *
323          * @return <code>true</code> if the current configuration has been
324          *         changed, otherwise <code>false</code>
325          */
326         public boolean balance() {
327             if (this.nodes.isEmpty()) {
328                 /* with no nodes this is pointless */
329                 return false;
330             }
331             if (logger.isTraceEnabled()) {
332                 logger.trace("Start balancing cluster");
333             }
334             final RoutingNodes.UnassignedShards unassigned = routingNodes.unassigned().transactionBegin();
335             boolean changed = initialize(routingNodes, unassigned);
336             if (!changed) {
337                 NodeSorter sorter = newNodeSorter();
338                 if (nodes.size() > 1) { /* skip if we only have one node */
339                     for (String index : buildWeightOrderedIndidces(Operation.BALANCE, sorter)) {
340                         sorter.reset(Operation.BALANCE, index);
341                         final float[] weights = sorter.weights;
342                         final ModelNode[] modelNodes = sorter.modelNodes;
343                         int lowIdx = 0;
344                         int highIdx = weights.length - 1;
345                         while (true) {
346                             final ModelNode minNode = modelNodes[lowIdx];
347                             final ModelNode maxNode = modelNodes[highIdx];
348                             advance_range:
349                             if (maxNode.numShards(index) > 0) {
350                                 final float delta = absDelta(weights[lowIdx], weights[highIdx]);
351                                 if (lessThan(delta, threshold)) {
352                                     if (lowIdx > 0 && highIdx-1 > 0 // is there a chance for a higher delta?
353                                         && (absDelta(weights[0], weights[highIdx-1]) > threshold) // check if we need to break at all
354                                         ) {
355                                         /* This is a special case if allocations from the "heaviest" to the "lighter" nodes is not possible
356                                          * due to some allocation decider restrictions like zone awareness. if one zone has for instance
357                                          * less nodes than another zone. so one zone is horribly overloaded from a balanced perspective but we
358                                          * can't move to the "lighter" shards since otherwise the zone would go over capacity.
359                                          *
360                                          * This break jumps straight to the condition below were we start moving from the high index towards
361                                          * the low index to shrink the window we are considering for balance from the other direction.
362                                          * (check shrinking the window from MAX to MIN)
363                                          * See #3580
364                                          */
365                                         break advance_range;
366                                     }
367                                     if (logger.isTraceEnabled()) {
368                                         logger.trace("Stop balancing index [{}]  min_node [{}] weight: [{}]  max_node [{}] weight: [{}]  delta: [{}]",
369                                                 index, maxNode.getNodeId(), weights[highIdx], minNode.getNodeId(), weights[lowIdx], delta);
370                                     }
371                                     break;
372                                 }
373                                 if (logger.isTraceEnabled()) {
374                                     logger.trace("Balancing from node [{}] weight: [{}] to node [{}] weight: [{}]  delta: [{}]",
375                                             maxNode.getNodeId(), weights[highIdx], minNode.getNodeId(), weights[lowIdx], delta);
376                                 }
377                                 /* pass the delta to the replication function to prevent relocations that only swap the weights of the two nodes.
378                                  * a relocation must bring us closer to the balance if we only achieve the same delta the relocation is useless */
379                                 if (tryRelocateShard(Operation.BALANCE, minNode, maxNode, index, delta)) {
380                                     /*
381                                      * TODO we could be a bit smarter here, we don't need to fully sort necessarily
382                                      * we could just find the place to insert linearly but the win might be minor
383                                      * compared to the added complexity
384                                      */
385                                     weights[lowIdx] = sorter.weight(Operation.BALANCE, modelNodes[lowIdx]);
386                                     weights[highIdx] = sorter.weight(Operation.BALANCE, modelNodes[highIdx]);
387                                     sorter.sort(0, weights.length);
388                                     lowIdx = 0;
389                                     highIdx = weights.length - 1;
390                                     changed = true;
391                                     continue;
392                                 }
393                             }
394                             if (lowIdx < highIdx - 1) {
395                                 /* Shrinking the window from MIN to MAX
396                                  * we can't move from any shard from the min node lets move on to the next node
397                                  * and see if the threshold still holds. We either don't have any shard of this
398                                  * index on this node of allocation deciders prevent any relocation.*/
399                                 lowIdx++;
400                             } else if (lowIdx > 0) {
401                                 /* Shrinking the window from MAX to MIN
402                                  * now we go max to min since obviously we can't move anything to the max node
403                                  * lets pick the next highest */
404                                 lowIdx = 0;
405                                 highIdx--;
406                             } else {
407                                 /* we are done here, we either can't relocate anymore or we are balanced */
408                                 break;
409                             }
410                         }
411                     }
412                 }
413             }
414             routingNodes.unassigned().transactionEnd(unassigned);
415             return changed;
416         }

418         /**
419          * This builds a initial index ordering where the indices are returned
420          * in most unbalanced first. We need this in order to prevent over
421          * allocations on added nodes from one index when the weight parameters
422          * for global balance overrule the index balance at an intermediate
423          * state. For example this can happen if we have 3 nodes and 3 indices
424          * with 3 shards and 1 shard. At the first stage all three nodes hold
425          * 2 shard for each index. now we add another node and the first index
426          * is balanced moving 3 two of the nodes over to the new node since it
427          * has no shards yet and global balance for the node is way below
428          * average. To re-balance we need to move shards back eventually likely
429          * to the nodes we relocated them from.
430          */
431         private String[] buildWeightOrderedIndidces(Operation operation, NodeSorter sorter) {
432             final String[] indices = this.indices.toArray(new String[this.indices.size()]);
433             final float[] deltas = new float[indices.length];
434             for (int i = 0; i < deltas.length; i++) {
435                 sorter.reset(operation, indices[i]);
436                 deltas[i] = sorter.delta();
437             }
438             new IntroSorter() {
439                 
440                 float pivotWeight;
441 
442                 @Override
443                 protected void swap(int i, int j) {
444                     final String tmpIdx = indices[i];
445                     indices[i] = indices[j];
446                     indices[j] = tmpIdx;
447                     final float tmpDelta = deltas[i];
448                     deltas[i] = deltas[j];
449                     deltas[j] = tmpDelta;
450                 }
451 
452                 @Override
453                 protected int compare(int i, int j) {
454                     return Float.compare(deltas[j], deltas[i]);
455                 }
456 
457                 @Override
458                 protected void setPivot(int i) {
459                     pivotWeight = deltas[i];
460                 }
461 
462                 @Override
463                 protected int comparePivot(int j) {
464                     return Float.compare(deltas[j], pivotWeight);
465                 }
466             }.sort(0, deltas.length);
467 
468             return indices;
469         }

471         /**
472          * This function executes a move operation moving the given shard from
473          * the given node to the minimal eligible node with respect to the
474          * weight function. Iff the shard is moved the shard will be set to
475          * {@link ShardRoutingState#RELOCATING} and a shadow instance of this
476          * shard is created with an incremented version in the state
477          * {@link ShardRoutingState#INITIALIZING}.
478          *
479          * @return <code>true</code> iff the shard has successfully been moved.
480          */
481         public boolean move(MutableShardRouting shard, RoutingNode node ) {
482             if (nodes.isEmpty() || !shard.started()) {
483                 /* with no nodes or a not started shard this is pointless */
484                 return false;
485             }
486             if (logger.isTraceEnabled()) {
487                 logger.trace("Try moving shard [{}] from [{}]", shard, node);
488             }
489             final RoutingNodes.UnassignedShards unassigned = routingNodes.unassigned().transactionBegin();
490             boolean changed = initialize(routingNodes, unassigned);
491             if (!changed) {
492                 final ModelNode sourceNode = nodes.get(node.nodeId());
493                 assert sourceNode != null;
494                 final NodeSorter sorter = newNodeSorter();
495                 sorter.reset(Operation.MOVE, shard.getIndex());
496                 final ModelNode[] nodes = sorter.modelNodes;
497                 assert sourceNode.containsShard(shard);
498                 /*
499                  * the sorter holds the minimum weight node first for the shards index.
500                  * We now walk through the nodes until we find a node to allocate the shard.
501                  * This is not guaranteed to be balanced after this operation we still try best effort to
502                  * allocate on the minimal eligible node.
503                  */
504 
505                 for (ModelNode currentNode : nodes) {
506                     if (currentNode.getNodeId().equals(node.nodeId())) {
507                         continue;
508                     }
509                     RoutingNode target = routingNodes.node(currentNode.getNodeId());
510                     Decision decision = allocation.deciders().canAllocate(shard, target, allocation);
511                     if (decision.type() == Type.YES) { // TODO maybe we can respect throttling here too?
512                         sourceNode.removeShard(shard);
513                         final MutableShardRouting initializingShard = new MutableShardRouting(shard.index(), shard.id(), currentNode.getNodeId(),
514                                 shard.currentNodeId(), shard.restoreSource(), shard.primary(), INITIALIZING, shard.version() + 1);
515                         currentNode.addShard(initializingShard, decision);
516                         routingNodes.assign(initializingShard, target.nodeId());
517                         routingNodes.relocate(shard, target.nodeId()); // set the node to relocate after we added the initializing shard
518                         if (logger.isTraceEnabled()) {
519                             logger.trace("Moved shard [{}] to node [{}]", shard, currentNode.getNodeId());
520                         }
521                         changed = true;
522                         break;
523                     }
524                 }
525             }
526             routingNodes.unassigned().transactionEnd(unassigned);
527             return changed;
528         }

530         /**
531          * Builds the internal model from all shards in the given
532          * {@link Iterable}. All shards in the {@link Iterable} must be assigned
533          * to a node. This method will skip shards in the state
534          * {@link ShardRoutingState#RELOCATING} since each relocating shard has
535          * a shadow shard in the state {@link ShardRoutingState#INITIALIZING}
536          * on the target node which we respect during the allocation / balancing
537          * process. In short, this method recreates the status-quo in the cluster.
538          */
539         private void buildModelFromAssigned(Iterable<MutableShardRouting> shards) {
540             for (MutableShardRouting shard : shards) {
541                 assert shard.assignedToNode();
542                 /* we skip relocating shards here since we expect an initializing shard with the same id coming in */
543                 if (shard.state() == RELOCATING) {
544                     continue;
545                 }
546                 ModelNode node = nodes.get(shard.currentNodeId());
547                 assert node != null;
548                 node.addShard(shard, Decision.single(Type.YES, "Already allocated on node", node.getNodeId()));
549                 if (logger.isTraceEnabled()) {
550                     logger.trace("Assigned shard [{}] to node [{}]", shard, node.getNodeId());
551                 }
552             }
553         }

555         /**
556          * Allocates all given shards on the minimal eligable node for the shards index
557          * with respect to the weight function. All given shards must be unassigned.
558          */
559         private boolean allocateUnassigned(RoutingNodes.UnassignedShards unassigned, List<MutableShardRouting> ignoredUnassigned) {
560             assert !nodes.isEmpty();
561             if (logger.isTraceEnabled()) {
562                 logger.trace("Start allocating unassigned shards");
563             }
564             if (unassigned.isEmpty()) {
565                 return false;
566             }
567             boolean changed = false;
568           
569             /*
570              * TODO: We could be smarter here and group the shards by index and then
571              * use the sorter to save some iterations. 
572              */
573             final AllocationDeciders deciders = allocation.deciders();
574             final Comparator<MutableShardRouting> comparator = new Comparator<MutableShardRouting>() {
575                 @Override
576                 public int compare(MutableShardRouting o1,
577                                    MutableShardRouting o2) {
578                     if (o1.primary() ^ o2.primary()) {
579                         return o1.primary() ? -1 : o2.primary() ? 1 : 0;
580                     }
581                     final int indexCmp;
582                     if ((indexCmp = o1.index().compareTo(o2.index())) == 0) {
583                         return o1.getId() - o2.getId();
584                     }
585                     return indexCmp;
586                 }
587             };
588             /*
589              * we use 2 arrays and move replicas to the second array once we allocated an identical
590              * replica in the current iteration to make sure all indices get allocated in the same manner.
591              * The arrays are sorted by primaries first and then by index and shard ID so a 2 indices with 2 replica and 1 shard would look like:
592              * [(0,P,IDX1), (0,P,IDX2), (0,R,IDX1), (0,R,IDX1), (0,R,IDX2), (0,R,IDX2)]
593              * if we allocate for instance (0, R, IDX1) we move the second replica to the secondary array and proceed with
594              * the next replica. If we could not find a node to allocate (0,R,IDX1) we move all it's replicas to ingoreUnassigned.
595              */
596             MutableShardRouting[] primary = unassigned.drain();
597             MutableShardRouting[] secondary = new MutableShardRouting[primary.length];
598             int secondaryLength = 0;
599             int primaryLength = primary.length;
600             ArrayUtil.timSort(primary, comparator);
601             final Set<ModelNode> throttledNodes = new IdentityHashSet<>();
602             do {
603                 for (int i = 0; i < primaryLength; i++) {
604                     MutableShardRouting shard = primary[i];
605                     if (!shard.primary()) {
606                         boolean drop = deciders.canAllocate(shard, allocation).type() == Type.NO;
607                         if (drop) {
608                             ignoredUnassigned.add(shard);
609                             while(i < primaryLength-1 && comparator.compare(primary[i], primary[i+1]) == 0) {
610                                 ignoredUnassigned.add(primary[++i]);
611                             }
612                             continue;
613                         } else {
614                             while(i < primaryLength-1 && comparator.compare(primary[i], primary[i+1]) == 0) {
615                                 secondary[secondaryLength++] = primary[++i];
616                             }
617                         }
618                     }
619                     assert !shard.assignedToNode() : shard;
620                     /* find an node with minimal weight we can allocate on*/
621                     float minWeight = Float.POSITIVE_INFINITY;
622                     ModelNode minNode = null;
623                     Decision decision = null;
624                     if (throttledNodes.size() < nodes.size()) {
625                         /* Don't iterate over an identity hashset here the
626                          * iteration order is different for each run and makes testing hard */
627                         for (ModelNode node : nodes.values()) {
628                             if (throttledNodes.contains(node)) {
629                                 continue;
630                             }
631                             /*
632                              * The shard we add is removed below to simulate the
633                              * addition for weight calculation we use Decision.ALWAYS to
634                              * not violate the not null condition.
635                              */
636                             if (!node.containsShard(shard)) {
637                                 node.addShard(shard, Decision.ALWAYS);
638                                 float currentWeight = weight.weight(Operation.ALLOCATE, this, node, shard.index());
639                                 /*
640                                  * Remove the shard from the node again this is only a
641                                  * simulation
642                                  */
643                                 Decision removed = node.removeShard(shard);
644                                 assert removed != null;
645                                 /*
646                                  * Unless the operation is not providing any gains we
647                                  * don't check deciders
648                                  */
649                                 if (currentWeight <= minWeight) {
650                                     Decision currentDecision = deciders.canAllocate(shard, routingNodes.node(node.getNodeId()), allocation);
651                                     NOUPDATE:
652                                     if (currentDecision.type() == Type.YES || currentDecision.type() == Type.THROTTLE) {
653                                         if (currentWeight == minWeight) {
654                                             /*  we have an equal weight tie breaking:
655                                              *  1. if one decision is YES prefer it
656                                              *  2. prefer the node that holds the primary for this index with the next id in the ring ie.
657                                              *  for the 3 shards 2 replica case we try to build up:
658                                              *    1 2 0
659                                              *    2 0 1
660                                              *    0 1 2
661                                              *  such that if we need to tie-break we try to prefer the node holding a shard with the minimal id greater
662                                              *  than the id of the shard we need to assign. This works find when new indices are created since
663                                              *  primaries are added first and we only add one shard set a time in this algorithm.
664                                              */
665                                             if (currentDecision.type() == decision.type()) {
666                                                 final int repId = shard.id();
667                                                 final int nodeHigh = node.highestPrimary(shard.index());
668                                                 final int minNodeHigh = minNode.highestPrimary(shard.index());
669                                                 if ((((nodeHigh > repId && minNodeHigh > repId) || (nodeHigh < repId && minNodeHigh < repId)) && (nodeHigh < minNodeHigh))
670                                                         || (nodeHigh > minNodeHigh && nodeHigh > repId && minNodeHigh < repId)) {
671                                                     minNode = node;
672                                                     minWeight = currentWeight;
673                                                     decision = currentDecision;
674                                                 } else {
675                                                     break NOUPDATE;
676                                                 }
677                                             } else if (currentDecision.type() != Type.YES) {
678                                                 break NOUPDATE;
679                                             }
680                                         }
681                                         minNode = node;
682                                         minWeight = currentWeight;
683                                         decision = currentDecision;
684                                     }
685                                 }
686                             }
687                         }
688                     }
689                     assert decision != null && minNode != null || decision == null && minNode == null;
690                     if (minNode != null) {
691                         minNode.addShard(shard, decision);
692                         if (decision.type() == Type.YES) {
693                             if (logger.isTraceEnabled()) {
694                                 logger.trace("Assigned shard [{}] to [{}]", shard, minNode.getNodeId());
695                             }
696                             routingNodes.assign(shard, routingNodes.node(minNode.getNodeId()).nodeId());
697                             changed = true;
698                             continue; // don't add to ignoreUnassigned
699                         } else {
700                             final RoutingNode node = routingNodes.node(minNode.getNodeId());
701                             if (deciders.canAllocate(node, allocation).type() != Type.YES) {
702                                 if (logger.isTraceEnabled()) {
703                                     logger.trace("Can not allocate on node [{}] remove from round decision [{}]", node, decision.type());
704                                 }
705                                 throttledNodes.add(minNode);
706                             }
707                         }
708                         if (logger.isTraceEnabled()) {
709                             logger.trace("No eligable node found to assign shard [{}] decision [{}]", shard, decision.type());
710                         }
711                     } else if (logger.isTraceEnabled()) {
712                         logger.trace("No Node found to assign shard [{}]", shard);
713                     }
714                     ignoredUnassigned.add(shard);
715                     if (!shard.primary()) { // we could not allocate it and we are a replica - check if we can ignore the other replicas
716                         while(secondaryLength > 0 && comparator.compare(shard, secondary[secondaryLength-1]) == 0) {
717                             ignoredUnassigned.add(secondary[--secondaryLength]);
718                         }
719                     }
720                 }
721                 primaryLength = secondaryLength;
722                 MutableShardRouting[] tmp = primary;
723                 primary = secondary;
724                 secondary = tmp;
725                 secondaryLength = 0;
726             } while (primaryLength > 0);
727             // clear everything we have either added it or moved to ingoreUnassigned
728             return changed;
729         }
730 
731         /**
732          * Tries to find a relocation from the max node to the minimal node for an arbitrary shard of the given index on the
733          * balance model. Iff this method returns a <code>true</code> the relocation has already been executed on the
734          * simulation model as well as on the cluster.
735          */
736         private boolean tryRelocateShard(Operation operation, ModelNode minNode, ModelNode maxNode, String idx, float minCost) {
737             final ModelIndex index = maxNode.getIndex(idx);
738             Decision decision = null;
739             if (index != null) {
740                 if (logger.isTraceEnabled()) {
741                     logger.trace("Try relocating shard for index index [{}] from node [{}] to node [{}]", idx, maxNode.getNodeId(),
742                             minNode.getNodeId());
743                 }
744                 final RoutingNode node = routingNodes.node(minNode.getNodeId());
745                 MutableShardRouting candidate = null;
746                 final AllocationDeciders deciders = allocation.deciders();
747                 /* make a copy since we modify this list in the loop */
748                 final ArrayList<MutableShardRouting> shards = new ArrayList<>(index.getAllShards());
749                 for (MutableShardRouting shard : shards) {
750                     if (shard.started()) {
751                         // skip initializing, unassigned and relocating shards we can't relocate them anyway
752                         Decision allocationDecision = deciders.canAllocate(shard, node, allocation);
753                         Decision rebalanceDecision = deciders.canRebalance(shard, allocation);
754                         if (((allocationDecision.type() == Type.YES) || (allocationDecision.type() == Type.THROTTLE))
755                                 && ((rebalanceDecision.type() == Type.YES) || (rebalanceDecision.type() == Type.THROTTLE))) {
756                             Decision srcDecision;
757                             if ((srcDecision = maxNode.removeShard(shard)) != null) {
758                                 minNode.addShard(shard, srcDecision);
759                                 final float delta = weight.weight(operation, this, minNode, idx) - weight.weight(operation, this, maxNode, idx);
760                                 if (delta < minCost ||
761                                         (candidate != null && delta == minCost && candidate.id() > shard.id())) {
762                                     /* this last line is a tie-breaker to make the shard allocation alg deterministic
763                                      * otherwise we rely on the iteration order of the index.getAllShards() which is a set.*/
764                                     minCost = delta;
765                                     candidate = shard;
766                                     decision = new Decision.Multi().add(allocationDecision).add(rebalanceDecision);
767                                 }
768                                 minNode.removeShard(shard);
769                                 maxNode.addShard(shard, srcDecision);
770                             }
771                         }
772                     }
773                 }
774 
775                 if (candidate != null) {
776                   
777                     /* allocate on the model even if not throttled */
778                     maxNode.removeShard(candidate);
779                     minNode.addShard(candidate, decision);
780                     if (decision.type() == Type.YES) { /* only allocate on the cluster if we are not throttled */
781                         if (logger.isTraceEnabled()) {
782                             logger.trace("Relocate shard [{}] from node [{}] to node [{}]", candidate, maxNode.getNodeId(),
783                                     minNode.getNodeId());
784                         }
785                         /* now allocate on the cluster - if we are started we need to relocate the shard */
786                         if (candidate.started()) {
787                             RoutingNode lowRoutingNode = routingNodes.node(minNode.getNodeId());
788                             routingNodes.assign(new MutableShardRouting(candidate.index(), candidate.id(), lowRoutingNode.nodeId(), candidate
789                                     .currentNodeId(), candidate.restoreSource(), candidate.primary(), INITIALIZING, candidate.version() + 1), lowRoutingNode.nodeId());
790                             routingNodes.relocate(candidate, lowRoutingNode.nodeId());
791 
792                         } else {
793                             assert candidate.unassigned();
794                             routingNodes.assign(candidate, routingNodes.node(minNode.getNodeId()).nodeId());
795                         }
796                         return true;
797 
798                     }
799                 }
800             }
801             if (logger.isTraceEnabled()) {
802                 logger.trace("Couldn't find shard to relocate from node [{}] to node [{}] allocation decision [{}]", maxNode.getNodeId(),
803                         minNode.getNodeId(), decision == null ? "NO" : decision.type().name());
804             }
805             return false;
806         }

810     static class ModelNode implements Iterable<ModelIndex> {
811         private final String id;
812         private final Map<String, ModelIndex> indices = new HashMap<>();
813         /* cached stats - invalidated on add/remove and lazily calculated */
814         private int numShards = -1;
815 
816         public ModelNode(String id) {
817             this.id = id;
818         }
819 
820         public ModelIndex getIndex(String indexId) {
821             return indices.get(indexId);
822         }
823 
824         public String getNodeId() {
825             return id;
826         }
827 
828         public int numShards() {
829             if (numShards == -1) {
830                 int sum = 0;
831                 for (ModelIndex index : indices.values()) {
832                     sum += index.numShards();
833                 }
834                 numShards = sum;
835             }
836             return numShards;
837         }
838 
839         public int numShards(String idx) {
840             ModelIndex index = indices.get(idx);
841             return index == null ? 0 : index.numShards();
842         }
843 
844         public Collection<MutableShardRouting> shards() {
845             Collection<MutableShardRouting> result = new ArrayList<>();
846             for (ModelIndex index : indices.values()) {
847                 result.addAll(index.getAllShards());
848             }
849             return result;
850         }
851 
852         public int highestPrimary(String index) {
853             ModelIndex idx = indices.get(index);
854             if (idx != null) {
855                 return idx.highestPrimary();
856             }
857             return -1;
858         }
859 
860         public void addShard(MutableShardRouting shard, Decision decision) {
861             numShards = -1;
862             ModelIndex index = indices.get(shard.index());
863             if (index == null) {
864                 index = new ModelIndex(shard.index());
865                 indices.put(index.getIndexId(), index);
866             }
867             index.addShard(shard, decision);
868         }
869 
870         public Decision removeShard(MutableShardRouting shard) {
871             numShards = -1;
872             ModelIndex index = indices.get(shard.index());
873             Decision removed = null;
874             if (index != null) {
875                 removed = index.removeShard(shard);
876                 if (removed != null && index.numShards() == 0) {
877                     indices.remove(shard.index());
878                 }
879             }
880             return removed;
881         }
882 
883         @Override
884         public String toString() {
885             StringBuilder sb = new StringBuilder();
886             sb.append("Node(").append(id).append(")");
887             return sb.toString();
888         }
889 
890         @Override
891         public Iterator<ModelIndex> iterator() {
892             return indices.values().iterator();
893         }
894 
895         public boolean containsShard(MutableShardRouting shard) {
896             ModelIndex index = getIndex(shard.getIndex());
897             return index == null ? false : index.containsShard(shard);
898         }

902     static final class ModelIndex {
903         private final String id;
904         private final Map<MutableShardRouting, Decision> shards = new HashMap<>();
905         private int numPrimaries = -1;
906         private int highestPrimary = -1;
907 
908         public ModelIndex(String id) {
909             this.id = id;
910         }
911 
912         public int highestPrimary() {
913             if (highestPrimary == -1) {
914                 int maxId = -1;
915                 for (MutableShardRouting shard : shards.keySet()) {
916                     if (shard.primary()) {
917                         maxId = Math.max(maxId, shard.id());
918                     }
919                 }
920                 return highestPrimary = maxId;
921             }
922             return highestPrimary;
923         }
924 
925         public String getIndexId() {
926             return id;
927         }
928 
929         public Decision getDecicion(MutableShardRouting shard) {
930             return shards.get(shard);
931         }
932 
933         public int numShards() {
934             return shards.size();
935         }
936 
937         public Collection<MutableShardRouting> getAllShards() {
938             return shards.keySet();
939         }
940 
941         public int numPrimaries() {
942             if (numPrimaries == -1) {
943                 int num = 0;
944                 for (MutableShardRouting shard : shards.keySet()) {
945                     if (shard.primary()) {
946                         num++;
947                     }
948                 }
949                 return numPrimaries = num;
950             }
951             return numPrimaries;
952         }
953 
954         public Decision removeShard(MutableShardRouting shard) {
955             highestPrimary = numPrimaries = -1;
956             return shards.remove(shard);
957         }
958 
959         public void addShard(MutableShardRouting shard, Decision decision) {
960             highestPrimary = numPrimaries = -1;
961             assert decision != null;
962             assert !shards.containsKey(shard) : "Shard already allocated on current node: " + shards.get(shard) + " " + shard;
963             shards.put(shard, decision);
964         }
965 
966         public boolean containsShard(MutableShardRouting shard) {
967             return shards.containsKey(shard);
968         }
969     }

971     static final class NodeSorter extends IntroSorter {
972 
973         final ModelNode[] modelNodes;
974         /* the nodes weights with respect to the current weight function / index */
975         final float[] weights;
976         private final WeightFunction function;
977         private String index;
978         private final Balancer balancer;
979         private float pivotWeight;
980 
981         public NodeSorter(ModelNode[] modelNodes, WeightFunction function, Balancer balancer) {
982             this.function = function;
983             this.balancer = balancer;
984             this.modelNodes = modelNodes;
985             weights = new float[modelNodes.length];
986         }
987 
988         /**
989          * Resets the sorter, recalculates the weights per node and sorts the
990          * nodes by weight, with minimal weight first.
991          */
992         public void reset(Operation operation, String index) {
993             this.index = index;
994             for (int i = 0; i < weights.length; i++) {
995                 weights[i] = weight(operation, modelNodes[i]);
996             }
997             sort(0, modelNodes.length);
998         }
999 
1000         public float weight(Operation operation, ModelNode node) {
1001             return function.weight(operation, balancer, node, index);
1002         }
1003 
1004         @Override
1005         protected void swap(int i, int j) {
1006             final ModelNode tmpNode = modelNodes[i];
1007             modelNodes[i] = modelNodes[j];
1008             modelNodes[j] = tmpNode;
1009             final float tmpWeight = weights[i];
1010             weights[i] = weights[j];
1011             weights[j] = tmpWeight;
1012         }
1013 
1014         @Override
1015         protected int compare(int i, int j) {
1016             return Float.compare(weights[i], weights[j]);
1017         }
1018 
1019         @Override
1020         protected void setPivot(int i) {
1021             pivotWeight = weights[i];
1022         }
1023 
1024         @Override
1025         protected int comparePivot(int j) {
1026             return Float.compare(pivotWeight, weights[j]);
1027         }
1028 
1029         public float delta() {
1030             return weights[weights.length - 1] - weights[0];
1031         }
