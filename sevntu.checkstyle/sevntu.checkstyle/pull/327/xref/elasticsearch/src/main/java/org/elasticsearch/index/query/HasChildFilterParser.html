19  package org.elasticsearch.index.query;
20  
21  import org.apache.lucene.search.Filter;
22  import org.apache.lucene.search.FilteredQuery;
23  import org.apache.lucene.search.Query;
24  import org.apache.lucene.search.join.BitDocIdSetFilter;
25  import org.elasticsearch.common.Strings;
26  import org.elasticsearch.common.collect.Tuple;
27  import org.elasticsearch.common.inject.Inject;
28  import org.elasticsearch.common.xcontent.XContentParser;
29  import org.elasticsearch.index.fielddata.plain.ParentChildIndexFieldData;
30  import org.elasticsearch.index.mapper.DocumentMapper;
31  import org.elasticsearch.index.mapper.internal.ParentFieldMapper;
32  import org.elasticsearch.index.query.support.InnerHitsQueryParserHelper;
33  import org.elasticsearch.index.query.support.XContentStructure;
34  import org.elasticsearch.index.search.child.ChildrenConstantScoreQuery;
35  import org.elasticsearch.index.search.child.ChildrenQuery;
36  import org.elasticsearch.index.search.child.CustomQueryWrappingFilter;
37  import org.elasticsearch.index.search.child.ScoreType;
38  import org.elasticsearch.index.search.nested.NonNestedDocsFilter;
39  import org.elasticsearch.search.fetch.innerhits.InnerHitsContext;
40  import org.elasticsearch.search.internal.SubSearchContext;
41  
42  import java.io.IOException;
43  
44  import static org.elasticsearch.index.query.QueryParserUtils.ensureNotDeleteByQuery;
45  
46  /**
47   *
48   */
49  public class HasChildFilterParser implements FilterParser {
50  
51      public static final String NAME = "has_child";
52  
53      private final InnerHitsQueryParserHelper innerHitsQueryParserHelper;
54  
55      @Inject
56      public HasChildFilterParser(InnerHitsQueryParserHelper innerHitsQueryParserHelper) {
57          this.innerHitsQueryParserHelper = innerHitsQueryParserHelper;
58      }
59  
60      @Override
61      public String[] names() {
62          return new String[]{NAME, Strings.toCamelCase(NAME)};
63      }
64  
65      @Override
66      public Filter parse(QueryParseContext parseContext) throws IOException, QueryParsingException {
67          ensureNotDeleteByQuery(NAME, parseContext);
68          XContentParser parser = parseContext.parser();
69  
70          boolean queryFound = false;
71          boolean filterFound = false;
72          String childType = null;
73          int shortCircuitParentDocSet = 8192; // Tests show a cut of point between 8192 and 16384.
74          int minChildren = 0;
75          int maxChildren = 0;
76          Tuple<String, SubSearchContext> innerHits = null;
77  
78          String filterName = null;
79          String currentFieldName = null;
80          XContentParser.Token token;
81          XContentStructure.InnerQuery innerQuery = null;
82          XContentStructure.InnerFilter innerFilter = null;
83          while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
84              if (token == XContentParser.Token.FIELD_NAME) {
85                  currentFieldName = parser.currentName();
86              } else if (token == XContentParser.Token.START_OBJECT) {
87                  // Usually, the query would be parsed here, but the child
88                  // type may not have been extracted yet, so use the
89                  // XContentStructure.<type> facade to parse if available,
90                  // or delay parsing if not.
91                  if ("query".equals(currentFieldName)) {
92                      innerQuery = new XContentStructure.InnerQuery(parseContext, childType == null ? null : new String[] {childType});
93                      queryFound = true;
94                  } else if ("filter".equals(currentFieldName)) {
95                      innerFilter = new XContentStructure.InnerFilter(parseContext, childType == null ? null : new String[] {childType});
96                      filterFound = true;
97                  } else if ("inner_hits".equals(currentFieldName)) {
98                      innerHits = innerHitsQueryParserHelper.parse(parseContext);
99                  } else {
100                     throw new QueryParsingException(parseContext.index(), "[has_child] filter does not support [" + currentFieldName + "]");
101                 }
102             } else if (token.isValue()) {
103                 if ("type".equals(currentFieldName) || "child_type".equals(currentFieldName) || "childType".equals(currentFieldName)) {
104                     childType = parser.text();
105                 } else if ("_name".equals(currentFieldName)) {
106                     filterName = parser.text();
107                 } else if ("_cache".equals(currentFieldName)) {
108                     // noop to be backwards compatible
109                 } else if ("_cache_key".equals(currentFieldName) || "_cacheKey".equals(currentFieldName)) {
110                     // noop to be backwards compatible
111                 } else if ("short_circuit_cutoff".equals(currentFieldName)) {
112                     shortCircuitParentDocSet = parser.intValue();
113                 } else if ("min_children".equals(currentFieldName) || "minChildren".equals(currentFieldName)) {
114                     minChildren = parser.intValue(true);
115                 } else if ("max_children".equals(currentFieldName) || "maxChildren".equals(currentFieldName)) {
116                     maxChildren = parser.intValue(true);
117                 } else {
118                     throw new QueryParsingException(parseContext.index(), "[has_child] filter does not support [" + currentFieldName + "]");
119                 }
120             }
121         }
122         if (!queryFound && !filterFound) {
123             throw new QueryParsingException(parseContext.index(), "[has_child] filter requires 'query' or 'filter' field");
124         }
125         if (childType == null) {
126             throw new QueryParsingException(parseContext.index(), "[has_child] filter requires 'type' field");
127         }
128 
129         Query query;
130         if (queryFound) {
131             query = innerQuery.asQuery(childType);
132         } else {
133             query = innerFilter.asFilter(childType);
134         }
135 
136         if (query == null) {
137             return null;
138         }
139 
140         DocumentMapper childDocMapper = parseContext.mapperService().documentMapper(childType);
141         if (childDocMapper == null) {
142             throw new QueryParsingException(parseContext.index(), "No mapping for for type [" + childType + "]");
143         }
144         if (innerHits != null) {
145             InnerHitsContext.ParentChildInnerHits parentChildInnerHits = new InnerHitsContext.ParentChildInnerHits(innerHits.v2(), query, null, childDocMapper);
146             String name = innerHits.v1() != null ? innerHits.v1() : childType;
147             parseContext.addInnerHits(name, parentChildInnerHits);
148         }
149         ParentFieldMapper parentFieldMapper = childDocMapper.parentFieldMapper();
150         if (!parentFieldMapper.active()) {
151             throw new QueryParsingException(parseContext.index(), "Type [" + childType + "] does not have parent mapping");
152         }
153         String parentType = parentFieldMapper.type();
154 
155         // wrap the query with type query
156         query = new FilteredQuery(query, parseContext.cacheFilter(childDocMapper.typeFilter(), null, parseContext.autoFilterCachePolicy()));
157 
158         DocumentMapper parentDocMapper = parseContext.mapperService().documentMapper(parentType);
159         if (parentDocMapper == null) {
160             throw new QueryParsingException(parseContext.index(), "[has_child]  Type [" + childType + "] points to a non existent parent type [" + parentType + "]");
161         }
162 
163         if (maxChildren > 0 && maxChildren < minChildren) {
164             throw new QueryParsingException(parseContext.index(), "[has_child] 'max_children' is less than 'min_children'");
165         }
166 
167         BitDocIdSetFilter nonNestedDocsFilter = null;
168         if (parentDocMapper.hasNestedObjects()) {
169             nonNestedDocsFilter = parseContext.bitsetFilter(NonNestedDocsFilter.INSTANCE);
170         }
171 
172         Filter parentFilter = parseContext.cacheFilter(parentDocMapper.typeFilter(), null, parseContext.autoFilterCachePolicy());
173         ParentChildIndexFieldData parentChildIndexFieldData = parseContext.getForField(parentFieldMapper);
174 
175         Query childrenQuery;
176         if (minChildren > 1 || maxChildren > 0) {
177             childrenQuery = new ChildrenQuery(parentChildIndexFieldData,  parentType, childType, parentFilter,query,ScoreType.NONE,minChildren, maxChildren, shortCircuitParentDocSet, nonNestedDocsFilter);
178         } else {
179             childrenQuery = new ChildrenConstantScoreQuery(parentChildIndexFieldData, query, parentType, childType, parentFilter,
180                     shortCircuitParentDocSet, nonNestedDocsFilter);
181         }
182         if (filterName != null) {
183             parseContext.addNamedFilter(filterName, new CustomQueryWrappingFilter(childrenQuery));
184         }
185         return new CustomQueryWrappingFilter(childrenQuery);
186     }
187 
188 }
