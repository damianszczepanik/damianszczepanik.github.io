181     @SuppressWarnings({"unchecked"})
182     public DocumentMapper parse(@Nullable String type, String source, String defaultSource) throws MapperParsingException {
183         Map<String, Object> mapping = null;
184         if (source != null) {
185             Tuple<String, Map<String, Object>> t = extractMapping(type, source);
186             type = t.v1();
187             mapping = t.v2();
188         }
189         if (mapping == null) {
190             mapping = Maps.newHashMap();
191         }
192         return parse(type, mapping, defaultSource);
193     }

199     @SuppressWarnings({"unchecked"})
200     public DocumentMapper parseCompressed(@Nullable String type, CompressedString source, String defaultSource) throws MapperParsingException {
201         Map<String, Object> mapping = null;
202         if (source != null) {
203             Map<String, Object> root = XContentHelper.convertToMap(source.compressed(), true).v2();
204             Tuple<String, Map<String, Object>> t = extractMapping(type, root);
205             type = t.v1();
206             mapping = t.v2();
207         }
208         if (mapping == null) {
209             mapping = Maps.newHashMap();
210         }
211         return parse(type, mapping, defaultSource);
212     }

214     @SuppressWarnings({"unchecked"})
215     private DocumentMapper parse(String type, Map<String, Object> mapping, String defaultSource) throws MapperParsingException {
216         if (type == null) {
217             throw new MapperParsingException("Failed to derive type");
218         }
219 
220         if (defaultSource != null) {
221             Tuple<String, Map<String, Object>> t = extractMapping(MapperService.DEFAULT_MAPPING, defaultSource);
222             if (t.v2() != null) {
223                 XContentHelper.mergeDefaults(mapping, t.v2());
224             }
225         }
226 
227 
228         Mapper.TypeParser.ParserContext parserContext = parserContext();
229         // parse RootObjectMapper
230         DocumentMapper.Builder docBuilder = doc(index.name(), indexSettings, (RootObjectMapper.Builder) rootObjectTypeParser.parse(type, mapping, parserContext));
231         Iterator<Map.Entry<String, Object>> iterator = mapping.entrySet().iterator();
232         // parse DocumentMapper
233         while(iterator.hasNext()) {
234             Map.Entry<String, Object> entry = iterator.next();
235             String fieldName = Strings.toUnderscoreCase(entry.getKey());
236             Object fieldNode = entry.getValue();
237 
238             if ("transform".equals(fieldName)) {
239                 iterator.remove();
240                 if (fieldNode instanceof Map) {
241                     parseTransform(docBuilder, (Map<String, Object>) fieldNode, parserContext.indexVersionCreated());
242                 } else if (fieldNode instanceof List) {
243                     for (Object transformItem: (List)fieldNode) {
244                         if (!(transformItem instanceof Map)) {
245                             throw new MapperParsingException("Elements of transform list must be objects but one was:  " + fieldNode);
246                         }
247                         parseTransform(docBuilder, (Map<String, Object>) transformItem, parserContext.indexVersionCreated());
248                     }
249                 } else {
250                     throw new MapperParsingException("Transform must be an object or an array but was:  " + fieldNode);
251                 }
252             } else {
253                 Mapper.TypeParser typeParser = rootTypeParsers.get(fieldName);
254                 if (typeParser != null) {
255                     iterator.remove();
256                     Map<String, Object> fieldNodeMap = (Map<String, Object>) fieldNode;
257                     docBuilder.put(typeParser.parse(fieldName, fieldNodeMap, parserContext));
258                     fieldNodeMap.remove("type");
259                     checkNoRemainingFields(fieldName, fieldNodeMap, parserContext.indexVersionCreated());
260                 }
261             }
262         }
263 
264         ImmutableMap<String, Object> attributes = ImmutableMap.of();
265         if (mapping.containsKey("_meta")) {
266             attributes = ImmutableMap.copyOf((Map<String, Object>) mapping.remove("_meta"));
267         }
268         docBuilder.meta(attributes);
269 
270         checkNoRemainingFields(mapping, parserContext.indexVersionCreated(), "Root mapping definition has unsupported parameters: ");
271 
272         DocumentMapper documentMapper = docBuilder.build(this);
273         // update the source with the generated one
274         documentMapper.refreshSource();
275         return documentMapper;
276     }

278     public static void checkNoRemainingFields(String fieldName, Map<String, Object> fieldNodeMap, Version indexVersionCreated) {
279         checkNoRemainingFields(fieldNodeMap, indexVersionCreated, "Mapping definition for [" + fieldName + "] has unsupported parameters: ");
280     }
281 
282     public static void checkNoRemainingFields(Map<String, Object> fieldNodeMap, Version indexVersionCreated, String message) {
283         if (!fieldNodeMap.isEmpty()) {
284             if (indexVersionCreated.onOrAfter(Version.V_2_0_0)) {
285                 throw new MapperParsingException(message + getRemainingFields(fieldNodeMap));
286             } else {
287                 logger.debug(message + "{}", getRemainingFields(fieldNodeMap));
288             }
289         }
290     }

292     private static String getRemainingFields(Map<String, ?> map) {
293         StringBuilder remainingFields = new StringBuilder();
294         for (String key : map.keySet()) {
295             remainingFields.append(" [").append(key).append(" : ").append(map.get(key)).append("]");
296         }
297         return remainingFields.toString();
298     }

300     @SuppressWarnings("unchecked")
301     private void parseTransform(DocumentMapper.Builder docBuilder, Map<String, Object> transformConfig, Version indexVersionCreated) {
302         ScriptParameterParser scriptParameterParser = new ScriptParameterParser();
303         scriptParameterParser.parseConfig(transformConfig, true);
304         
305         String script = null;
306         ScriptType scriptType = null;
307         ScriptParameterValue scriptValue = scriptParameterParser.getDefaultScriptParameterValue();
308         if (scriptValue != null) {
309             script = scriptValue.script();
310             scriptType = scriptValue.scriptType();
311         }
312         
313         if (script != null) {
314             String scriptLang = scriptParameterParser.lang();
315             Map<String, Object> params = (Map<String, Object>)transformConfig.remove("params");
316             docBuilder.transform(scriptService, script, scriptType, scriptLang, params);
317         }
318         checkNoRemainingFields(transformConfig, indexVersionCreated, "Transform config has unsupported parameters: ");
319     }

321     private Tuple<String, Map<String, Object>> extractMapping(String type, String source) throws MapperParsingException {
322         Map<String, Object> root;
323         try {
324             root = XContentFactory.xContent(source).createParser(source).mapOrderedAndClose();
325         } catch (Exception e) {
326             throw new MapperParsingException("failed to parse mapping definition", e);
327         }
328         return extractMapping(type, root);
329     }
330 
331     @SuppressWarnings({"unchecked"})
332     private Tuple<String, Map<String, Object>> extractMapping(String type, Map<String, Object> root) throws MapperParsingException {
333         if (root.size() == 0) {
334             // if we don't have any keys throw an exception
335             throw new MapperParsingException("malformed mapping no root object found");
336         }
337         String rootName = root.keySet().iterator().next();
338         Tuple<String, Map<String, Object>> mapping;
339         if (type == null || type.equals(rootName)) {
340             mapping = new Tuple<>(rootName, (Map<String, Object>) root.get(rootName));
341         } else {
342             mapping = new Tuple<>(type, root);
343         }
344         return mapping;
345     }
