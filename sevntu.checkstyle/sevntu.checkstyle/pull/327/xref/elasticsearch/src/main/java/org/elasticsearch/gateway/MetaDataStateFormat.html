47  /**
48   * MetaDataStateFormat is a base class to write checksummed
49   * XContent based files to one or more directories in a standardized directory structure.
50   * @param <T> the type of the XContent base data-structure
51   */
52  public abstract class MetaDataStateFormat<T> {
53      public static final String STATE_DIR_NAME = "_state";
54      public static final String STATE_FILE_EXTENSION = ".st";
55      private static final String STATE_FILE_CODEC = "state";
56      private static final int STATE_FILE_VERSION = 0;
57      private static final int BUFFER_SIZE = 4096;
58      private final XContentType format;
59      private final boolean deleteOldFiles;
60  
61      /**
62       * Creates a new {@link MetaDataStateFormat} instance
63       * @param format the format of the x-content
64       * @param deleteOldFiles if <code>true</code> write operations will
65       *                       clean up old files written with this format.
66       */
67      protected MetaDataStateFormat(XContentType format, boolean deleteOldFiles) {
68          this.format = format;
69          this.deleteOldFiles = deleteOldFiles;
70      }
71  
72      /**
73       * Returns the {@link XContentType} used to serialize xcontent on write.
74       */
75      public XContentType format() {
76          return format;
77      }

79      /**
80       * Writes the given state to the given directories. The state is written to a
81       * state directory ({@value #STATE_DIR_NAME}) underneath each of the given file locations and is created if it
82       * doesn't exist. The state is serialized to a temporary file in that directory and is then atomically moved to
83       * it's target filename of the pattern <tt>{prefix}{version}.st</tt>.
84       *
85       * @param state the state object to write
86       * @param prefix the state names prefix used to compose the file name.
87       * @param version the version of the state
88       * @param locations the locations where the state should be written to.
89       * @throws IOException if an IOException occurs
90       */
91      public final void write(final T state, final String prefix, final long version, final Path... locations) throws IOException {
92          Preconditions.checkArgument(locations != null, "Locations must not be null");
93          Preconditions.checkArgument(locations.length > 0, "One or more locations required");
94          String fileName = prefix + version + STATE_FILE_EXTENSION;
95          Path stateLocation = locations[0].resolve(STATE_DIR_NAME);
96          Files.createDirectories(stateLocation);
97          final Path tmpStatePath = stateLocation.resolve(fileName + ".tmp");
98          final Path finalStatePath = stateLocation.resolve(fileName);
99          try {
100             final String resourceDesc = "MetaDataStateFormat.write(path=\"" + tmpStatePath + "\")";
101             try (OutputStreamIndexOutput out = new OutputStreamIndexOutput(resourceDesc, Files.newOutputStream(tmpStatePath), BUFFER_SIZE)) {
102                 CodecUtil.writeHeader(out, STATE_FILE_CODEC, STATE_FILE_VERSION);
103                 out.writeInt(format.index());
104                 out.writeLong(version);
105                 try (XContentBuilder builder = newXContentBuilder(format, new org.elasticsearch.common.lucene.store.OutputStreamIndexOutput(out) {
106                     @Override
107                     public void close() throws IOException {
108                         // this is important since some of the XContentBuilders write bytes on close.
109                         // in order to write the footer we need to prevent closing the actual index input.
110                     } })) {
111 
112                     builder.startObject();
113                     {
114                         toXContent(builder, state);
115                     }
116                     builder.endObject();
117                 }
118                 CodecUtil.writeFooter(out);
119             }
120             IOUtils.fsync(tmpStatePath, false); // fsync the state file
121             Files.move(tmpStatePath, finalStatePath, StandardCopyOption.ATOMIC_MOVE);
122             IOUtils.fsync(stateLocation, true);
123             for (int i = 1; i < locations.length; i++) {
124                 stateLocation = locations[i].resolve(STATE_DIR_NAME);
125                 Files.createDirectories(stateLocation);
126                 Path tmpPath = stateLocation.resolve(fileName + ".tmp");
127                 Path finalPath = stateLocation.resolve(fileName);
128                 try {
129                     Files.copy(finalStatePath, tmpPath);
130                     Files.move(tmpPath, finalPath, StandardCopyOption.ATOMIC_MOVE); // we are on the same FileSystem / Partition here we can do an atomic move
131                     IOUtils.fsync(stateLocation, true); // we just fsync the dir here..
132                 } finally {
133                     Files.deleteIfExists(tmpPath);
134                 }
135             }
136         } finally {
137             Files.deleteIfExists(tmpStatePath);
138         }
139         if (deleteOldFiles) {
140             cleanupOldFiles(prefix, fileName, locations);
141         }
142     }

160     /**
161      * Reads the state from a given file and compares the expected version against the actual version of
162      * the state.
163      */
164     public final T read(Path file, long expectedVersion) throws IOException {
165         try (Directory dir = newDirectory(file.getParent())) {
166             try (final IndexInput indexInput = dir.openInput(file.getFileName().toString(), IOContext.DEFAULT)) {
167                  // We checksum the entire file before we even go and parse it. If it's corrupted we barf right here.
168                 CodecUtil.checksumEntireFile(indexInput);
169                 CodecUtil.checkHeader(indexInput, STATE_FILE_CODEC, STATE_FILE_VERSION, STATE_FILE_VERSION);
170                 final XContentType xContentType = XContentType.values()[indexInput.readInt()];
171                 final long version = indexInput.readLong();
172                 if (version != expectedVersion) {
173                     throw new CorruptStateException("State version mismatch expected: " + expectedVersion + " but was: " + version);
174                 }
175                 long filePointer = indexInput.getFilePointer();
176                 long contentSize = indexInput.length() - CodecUtil.footerLength() - filePointer;
177                 try (IndexInput slice = indexInput.slice("state_xcontent", filePointer, contentSize)) {
178                     try (XContentParser parser = XContentFactory.xContent(xContentType).createParser(new InputStreamIndexInput(slice, contentSize))) {
179                         return fromXContent(parser);
180                     }
181                 }
182             } catch(CorruptIndexException | IndexFormatTooOldException | IndexFormatTooNewException ex) {
183                 // we trick this into a dedicated exception with the original stacktrace
184                 throw new CorruptStateException(ex);
185             }
186         }
187     }
188 
189     protected Directory newDirectory(Path dir) throws IOException {
190         return new SimpleFSDirectory(dir);
191     }

193     private void cleanupOldFiles(final String prefix, final String currentStateFile, Path[] locations) throws IOException {
194         final DirectoryStream.Filter<Path> filter = new DirectoryStream.Filter<Path>() {
195             @Override
196             public boolean accept(Path entry) throws IOException {
197                 final String entryFileName = entry.getFileName().toString();
198                 return Files.isRegularFile(entry)
199                         && entryFileName.startsWith(prefix) // only state files
200                         && currentStateFile.equals(entryFileName) == false; // keep the current state file around
201             }
202         };
203         // now clean up the old files
204         for (Path dataLocation : locations) {
205             try (DirectoryStream<Path> stream = Files.newDirectoryStream(dataLocation.resolve(STATE_DIR_NAME), filter)) {
206                 for (Path stateFile : stream) {
207                     Files.deleteIfExists(stateFile);
208                 }
209             }
210         }
211     }

213     /**
214      * Tries to load the latest state from the given data-locations. It tries to load the latest state determined by
215      * the states version from one or more data directories and if none of the latest states can be loaded an exception
216      * is thrown to prevent accidentally loading a previous state and silently omitting the latest state.
217      *
218      * @param logger an elasticsearch logger instance
219      * @param format the actual metastate format to use
220      * @param pattern the file name pattern to identify files belonging to this pattern and to read the version from.
221      *                The first capture group should return the version of the file. If the second capture group is has a
222      *                null value the files is considered a legacy file and will be treated as if the file contains a plain
223      *                x-content payload.
224      * @param stateType the state type we are loading. used for logging contenxt only.
225      * @param dataLocations the data-locations to try.
226      * @return the latest state or <code>null</code> if no state was found.
227      */
228     public static <T> T loadLatestState(ESLogger logger, MetaDataStateFormat<T> format, Pattern pattern, String stateType, Path... dataLocations) throws IOException {
229         List<PathAndVersion> files = new ArrayList<>();
230         long maxVersion = -1;
231         boolean maxVersionIsLegacy = true;
232         if (dataLocations != null) { // select all eligable files first
233             for (Path dataLocation : dataLocations) {
234                 final Path stateDir = dataLocation.resolve(STATE_DIR_NAME);
235                 if (!Files.isDirectory(stateDir)) {
236                     continue;
237                 }
238                 // now, iterate over the current versions, and find latest one
239                 try (DirectoryStream<Path> paths = Files.newDirectoryStream(stateDir)) { // we don't pass a glob since we need the group part for parsing
240                     for (Path stateFile : paths) {
241                         final Matcher matcher = pattern.matcher(stateFile.getFileName().toString());
242                         if (matcher.matches()) {
243                             final long version = Long.parseLong(matcher.group(1));
244                             maxVersion = Math.max(maxVersion, version);
245                             final boolean legacy = MetaDataStateFormat.STATE_FILE_EXTENSION.equals(matcher.group(2)) == false;
246                             maxVersionIsLegacy &= legacy; // on purpose, see NOTE below
247                             PathAndVersion pav = new PathAndVersion(stateFile, version, legacy);
248                             logger.trace("found state file: {}", pav);
249                             files.add(pav);
250                         }
251                     }
252                 }
253             }
254         }
255         final List<Throwable> exceptions = new ArrayList<>();
256         T state = null;
257         // NOTE: we might have multiple version of the latest state if there are multiple data dirs.. for this case
258         //       we iterate only over the ones with the max version. If we have at least one state file that uses the
259         //       new format (ie. legacy == false) then we know that the latest version state ought to use this new format.
260         //       In case the state file with the latest version does not use the new format while older state files do,
261         //       the list below will be empty and loading the state will fail
262         for (PathAndVersion pathAndVersion : Collections2.filter(files, new VersionAndLegacyPredicate(maxVersion, maxVersionIsLegacy))) {
263             try {
264                 final Path stateFile = pathAndVersion.file;
265                 final long version = pathAndVersion.version;
266                 final XContentParser parser;
267                 if (pathAndVersion.legacy) { // read the legacy format -- plain XContent
268                     final byte[] data = Files.readAllBytes(stateFile);
269                     if (data.length == 0) {
270                         logger.debug("{}: no data for [{}], ignoring...", stateType, stateFile.toAbsolutePath());
271                         continue;
272                     }
273                     parser = XContentHelper.createParser(data, 0, data.length);
274                     state = format.fromXContent(parser);
275                     if (state == null) {
276                         logger.debug("{}: no data for [{}], ignoring...", stateType, stateFile.toAbsolutePath());
277                     }
278                 } else {
279                     state = format.read(stateFile, version);
280                     logger.trace("state version [{}] read from [{}]", version, stateFile.getFileName());
281                 }
282                 return state;
283             } catch (Throwable e) {
284                 exceptions.add(e);
285                 logger.debug("{}: failed to read [{}], ignoring...", e, pathAndVersion.file.toAbsolutePath(), stateType);
286             }
287         }
288         // if we reach this something went wrong
289         ExceptionsHelper.maybeThrowRuntimeAndSuppress(exceptions);
290         if (files.size() > 0) {
291             // We have some state files but none of them gave us a usable state
292             throw new ElasticsearchIllegalStateException("Could not find a state file to recover from among " + files);
293         }
294         return state;
295     }

297     /**
298      * Filters out all {@link MetaDataStateFormat.PathAndVersion} instances with a different version than
299      * the given one.
300      */
301     private static final class VersionAndLegacyPredicate implements Predicate<PathAndVersion> {
302         private final long version;
303         private final boolean legacy;
304 
305         VersionAndLegacyPredicate(long version, boolean legacy) {
306             this.version = version;
307             this.legacy = legacy;
308         }
309 
310         @Override
311         public boolean apply(PathAndVersion input) {
312             return input.version == version && input.legacy == legacy;
313         }
314     }

316     /**
317      * Internal struct-like class that holds the parsed state version, the file
318      * and a flag if the file is a legacy state ie. pre 1.5
319      */
320     private static class PathAndVersion {
321         final Path file;
322         final long version;
323         final boolean legacy;
324 
325         private PathAndVersion(Path file, long version, boolean legacy) {
326             this.file = file;
327             this.version = version;
328             this.legacy = legacy;
329         }
330 
331         @Override
332         public String toString() {
333             return "[version:" + version + ", legacy:" + legacy + ", file:" + file.toAbsolutePath() + "]";
334         }
335     }

337     /**
338      * Deletes all meta state directories recursively for the given data locations
339      * @param dataLocations the data location to delete
340      */
341     public static void deleteMetaState(Path... dataLocations) throws IOException {
342         Path[] stateDirectories = new Path[dataLocations.length];
343         for (int i = 0; i < dataLocations.length; i++) {
344             stateDirectories[i] = dataLocations[i].resolve(STATE_DIR_NAME);
345         }
346         IOUtils.rm(stateDirectories);
347     }
