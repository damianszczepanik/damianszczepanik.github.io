20  package org.elasticsearch.index.gateway;
21  
22  import com.google.common.collect.ImmutableList;
23  import com.google.common.collect.Lists;
24  import org.apache.lucene.util.CollectionUtil;
25  import org.elasticsearch.common.xcontent.XContentBuilder;
26  import org.elasticsearch.common.xcontent.XContentFactory;
27  import org.elasticsearch.common.xcontent.XContentParser;
28  import org.elasticsearch.common.xcontent.XContentType;
29  
30  import java.io.IOException;
31  import java.util.Comparator;
32  import java.util.Iterator;
33  import java.util.List;

35  /**
36   *
37   */
38  public class CommitPoints implements Iterable<CommitPoint> {
39  
40      private final ImmutableList<CommitPoint> commitPoints;
41  
42      public CommitPoints(List<CommitPoint> commitPoints) {
43          CollectionUtil.introSort(commitPoints, new Comparator<CommitPoint>() {
44              @Override
45              public int compare(CommitPoint o1, CommitPoint o2) {
46                  return (o2.version() < o1.version() ? -1 : (o2.version() == o1.version() ? 0 : 1));
47              }
48          });
49          this.commitPoints = ImmutableList.copyOf(commitPoints);
50      }
51  
52      public ImmutableList<CommitPoint> commits() {
53          return this.commitPoints;
54      }
55  
56      public boolean hasVersion(long version) {
57          for (CommitPoint commitPoint : commitPoints) {
58              if (commitPoint.version() == version) {
59                  return true;
60              }
61          }
62          return false;
63      }
64  
65      public CommitPoint.FileInfo findPhysicalIndexFile(String physicalName) {
66          for (CommitPoint commitPoint : commitPoints) {
67              CommitPoint.FileInfo fileInfo = commitPoint.findPhysicalIndexFile(physicalName);
68              if (fileInfo != null) {
69                  return fileInfo;
70              }
71          }
72          return null;
73      }
74  
75      public CommitPoint.FileInfo findNameFile(String name) {
76          for (CommitPoint commitPoint : commitPoints) {
77              CommitPoint.FileInfo fileInfo = commitPoint.findNameFile(name);
78              if (fileInfo != null) {
79                  return fileInfo;
80              }
81          }
82          return null;
83      }
84  
85      @Override
86      public Iterator<CommitPoint> iterator() {
87          return commitPoints.iterator();
88      }
89  
90      public static byte[] toXContent(CommitPoint commitPoint) throws Exception {
91          XContentBuilder builder = XContentFactory.contentBuilder(XContentType.JSON).prettyPrint();
92          builder.startObject();
93          builder.field("version", commitPoint.version());
94          builder.field("name", commitPoint.name());
95          builder.field("type", commitPoint.type().toString());
96  
97          builder.startObject("index_files");
98          for (CommitPoint.FileInfo fileInfo : commitPoint.indexFiles()) {
99              builder.startObject(fileInfo.name());
100             builder.field("physical_name", fileInfo.physicalName());
101             builder.field("length", fileInfo.length());
102             if (fileInfo.checksum() != null) {
103                 builder.field("checksum", fileInfo.checksum());
104             }
105             builder.endObject();
106         }
107         builder.endObject();
108 
109         builder.startObject("translog_files");
110         for (CommitPoint.FileInfo fileInfo : commitPoint.translogFiles()) {
111             builder.startObject(fileInfo.name());
112             builder.field("physical_name", fileInfo.physicalName());
113             builder.field("length", fileInfo.length());
114             builder.endObject();
115         }
116         builder.endObject();
117 
118         builder.endObject();
119         return builder.bytes().toBytes();
120     }
121 
122     public static CommitPoint fromXContent(byte[] data) throws Exception {
123         try (XContentParser parser = XContentFactory.xContent(XContentType.JSON).createParser(data)) {
124             String currentFieldName = null;
125             XContentParser.Token token = parser.nextToken();
126             if (token == null) {
127                 // no data...
128                 throw new IOException("No commit point data");
129             }
130             long version = -1;
131             String name = null;
132             CommitPoint.Type type = null;
133             List<CommitPoint.FileInfo> indexFiles = Lists.newArrayList();
134             List<CommitPoint.FileInfo> translogFiles = Lists.newArrayList();
135             while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
136                 if (token == XContentParser.Token.FIELD_NAME) {
137                     currentFieldName = parser.currentName();
138                 } else if (token == XContentParser.Token.START_OBJECT) {
139                     List<CommitPoint.FileInfo> files = null;
140                     if ("index_files".equals(currentFieldName) || "indexFiles".equals(currentFieldName)) {
141                         files = indexFiles;
142                     } else if ("translog_files".equals(currentFieldName) || "translogFiles".equals(currentFieldName)) {
143                         files = translogFiles;
144                     } else {
145                         throw new IOException("Can't handle object with name [" + currentFieldName + "]");
146                     }
147                     while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
148                         if (token == XContentParser.Token.FIELD_NAME) {
149                             currentFieldName = parser.currentName();
150                         } else if (token == XContentParser.Token.START_OBJECT) {
151                             String fileName = currentFieldName;
152                             String physicalName = null;
153                             long size = -1;
154                             String checksum = null;
155                             while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
156                                 if (token == XContentParser.Token.FIELD_NAME) {
157                                     currentFieldName = parser.currentName();
158                                 } else if (token.isValue()) {
159                                     if ("physical_name".equals(currentFieldName) || "physicalName".equals(currentFieldName)) {
160                                         physicalName = parser.text();
161                                     } else if ("length".equals(currentFieldName)) {
162                                         size = parser.longValue();
163                                     } else if ("checksum".equals(currentFieldName)) {
164                                         checksum = parser.text();
165                                     }
166                                 }
167                             }
168                             if (physicalName == null) {
169                                 throw new IOException("Malformed commit, missing physical_name for [" + fileName + "]");
170                             }
171                             if (size == -1) {
172                                 throw new IOException("Malformed commit, missing length for [" + fileName + "]");
173                             }
174                             files.add(new CommitPoint.FileInfo(fileName, physicalName, size, checksum));
175                         }
176                     }
177                 } else if (token.isValue()) {
178                     if ("version".equals(currentFieldName)) {
179                         version = parser.longValue();
180                     } else if ("name".equals(currentFieldName)) {
181                         name = parser.text();
182                     } else if ("type".equals(currentFieldName)) {
183                         type = CommitPoint.Type.valueOf(parser.text());
184                     }
185                 }
186             }
187 
188             if (version == -1) {
189                 throw new IOException("Malformed commit, missing version");
190             }
191             if (name == null) {
192                 throw new IOException("Malformed commit, missing name");
193             }
194             if (type == null) {
195                 throw new IOException("Malformed commit, missing type");
196             }
197 
198             return new CommitPoint(version, name, type, indexFiles, translogFiles);
199         }
200     }
