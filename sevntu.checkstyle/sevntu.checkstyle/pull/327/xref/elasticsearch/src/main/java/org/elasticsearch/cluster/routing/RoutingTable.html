119     /**
120      * All the shards (replicas) for the provided indices.
121      *
122      * @param indices The indices to return all the shards (replicas), can be <tt>null</tt> or empty array to indicate all indices
123      * @return All the shards matching the specific index
124      * @throws IndexMissingException If an index passed does not exists
125      */
126     public List<ShardRouting> allShards(String... indices) throws IndexMissingException {
127         List<ShardRouting> shards = Lists.newArrayList();
128         if (indices == null || indices.length == 0) {
129             indices = indicesRouting.keySet().toArray(new String[indicesRouting.keySet().size()]);
130         }
131         for (String index : indices) {
132             IndexRoutingTable indexRoutingTable = index(index);
133             if (indexRoutingTable == null) {
134                 throw new IndexMissingException(new Index(index));
135             }
136             for (IndexShardRoutingTable indexShardRoutingTable : indexRoutingTable) {
137                 for (ShardRouting shardRouting : indexShardRoutingTable) {
138                     shards.add(shardRouting);
139                 }
140             }
141         }
142         return shards;
143     }

145     /**
146      * All the shards (primary + replicas) for the provided indices grouped (each group is a single element, consisting
147      * of the shard). This is handy for components that expect to get group iterators, but still want in some
148      * cases to iterate over all the shards (and not just one shard in replication group).
149      *
150      * @param indices The indices to return all the shards (replicas), can be <tt>null</tt> or empty array to indicate all indices
151      * @return All the shards grouped into a single shard element group each
152      * @throws IndexMissingException If an index passed does not exists
153      * @see IndexRoutingTable#groupByAllIt()
154      */
155     public GroupShardsIterator allShardsGrouped(String... indices) throws IndexMissingException {
156         // use list here since we need to maintain identity across shards
157         ArrayList<ShardIterator> set = new ArrayList<>();
158         if (indices == null || indices.length == 0) {
159             indices = indicesRouting.keySet().toArray(new String[indicesRouting.keySet().size()]);
160         }
161         for (String index : indices) {
162             IndexRoutingTable indexRoutingTable = index(index);
163             if (indexRoutingTable == null) {
164                 continue;
165                 // we simply ignore indices that don't exists (make sense for operations that use it currently)
166 //                throw new IndexMissingException(new Index(index));
167             }
168             for (IndexShardRoutingTable indexShardRoutingTable : indexRoutingTable) {
169                 for (ShardRouting shardRouting : indexShardRoutingTable) {
170                     set.add(shardRouting.shardsIt());
171                 }
172             }
173         }
174         return new GroupShardsIterator(set);
175     }

181     /**
182      * Return GroupShardsIterator where each active shard routing has it's own shard iterator.
183      *
184      * @param includeEmpty             if true, a shard iterator will be added for non-assigned shards as well
185      * @param includeRelocationTargets if true, an <b>extra</b> shard iterator will be added for relocating shards. The extra
186      *                                 iterator contains a single ShardRouting pointing at the relocating target
187      */
188     public GroupShardsIterator allActiveShardsGrouped(String[] indices, boolean includeEmpty, boolean includeRelocationTargets) throws IndexMissingException {
189         // use list here since we need to maintain identity across shards
190         ArrayList<ShardIterator> set = new ArrayList<>();
191         if (indices == null || indices.length == 0) {
192             indices = indicesRouting.keySet().toArray(new String[indicesRouting.keySet().size()]);
193         }
194         for (String index : indices) {
195             IndexRoutingTable indexRoutingTable = index(index);
196             if (indexRoutingTable == null) {
197                 continue;
198                 // we simply ignore indices that don't exists (make sense for operations that use it currently)
199 //                throw new IndexMissingException(new Index(index));
200             }
201             for (IndexShardRoutingTable indexShardRoutingTable : indexRoutingTable) {
202                 for (ShardRouting shardRouting : indexShardRoutingTable) {
203                     if (shardRouting.active()) {
204                         set.add(shardRouting.shardsIt());
205                         if (includeRelocationTargets && shardRouting.relocating()) {
206                             set.add(new PlainShardIterator(shardRouting.shardId(), ImmutableList.of(shardRouting.targetRoutingIfRelocating())));
207                         }
208                     } else if (includeEmpty) { // we need this for counting properly, just make it an empty one
209                         set.add(new PlainShardIterator(shardRouting.shardId(), ImmutableList.<ShardRouting>of()));
210                     }
211                 }
212             }
213         }
214         return new GroupShardsIterator(set);
215     }

257     /**
258      * All the *active* primary shards for the provided indices grouped (each group is a single element, consisting
259      * of the primary shard). This is handy for components that expect to get group iterators, but still want in some
260      * cases to iterate over all primary shards (and not just one shard in replication group).
261      *
262      * @param indices The indices to return all the shards (replicas), can be <tt>null</tt> or empty array to indicate all indices
263      * @return All the primary shards grouped into a single shard element group each
264      * @throws IndexMissingException If an index passed does not exists
265      * @see IndexRoutingTable#groupByAllIt()
266      */
267     public GroupShardsIterator activePrimaryShardsGrouped(String[] indices, boolean includeEmpty) throws IndexMissingException {
268         // use list here since we need to maintain identity across shards
269         ArrayList<ShardIterator> set = new ArrayList<>();
270         if (indices == null || indices.length == 0) {
271             indices = indicesRouting.keySet().toArray(new String[indicesRouting.keySet().size()]);
272         }
273         for (String index : indices) {
274             IndexRoutingTable indexRoutingTable = index(index);
275             if (indexRoutingTable == null) {
276                 throw new IndexMissingException(new Index(index));
277             }
278             for (IndexShardRoutingTable indexShardRoutingTable : indexRoutingTable) {
279                 ShardRouting primary = indexShardRoutingTable.primaryShard();
280                 if (primary.active()) {
281                     set.add(primary.shardsIt());
282                 } else if (includeEmpty) { // we need this for counting properly, just make it an empty one
283                     set.add(new PlainShardIterator(primary.shardId(), ImmutableList.<ShardRouting>of()));
284                 }
285             }
286         }
287         return new GroupShardsIterator(set);
288     }

307         public Builder(RoutingTable routingTable) {
308             version = routingTable.version;
309             for (IndexRoutingTable indexRoutingTable : routingTable) {
310                 indicesRouting.put(indexRoutingTable.index(), indexRoutingTable);
311             }

314         public Builder updateNodes(RoutingNodes routingNodes) {
315             // this is being called without pre initializing the routing table, so we must copy over the version as well
316             this.version = routingNodes.routingTable().version();
317 
318             Map<String, IndexRoutingTable.Builder> indexRoutingTableBuilders = newHashMap();
319             for (RoutingNode routingNode : routingNodes) {
320                 for (MutableShardRouting shardRoutingEntry : routingNode) {
321                     // every relocating shard has a double entry, ignore the target one.
322                     if (shardRoutingEntry.state() == ShardRoutingState.INITIALIZING && shardRoutingEntry.relocatingNodeId() != null)
323                         continue;
324 
325                     String index = shardRoutingEntry.index();
326                     IndexRoutingTable.Builder indexBuilder = indexRoutingTableBuilders.get(index);
327                     if (indexBuilder == null) {
328                         indexBuilder = new IndexRoutingTable.Builder(index);
329                         indexRoutingTableBuilders.put(index, indexBuilder);
330                     }
331 
332                     IndexShardRoutingTable refData = routingNodes.routingTable().index(shardRoutingEntry.index()).shard(shardRoutingEntry.id());
333                     indexBuilder.addShard(refData, shardRoutingEntry);
334                 }
335             }
336             for (MutableShardRouting shardRoutingEntry : Iterables.concat(routingNodes.unassigned(), routingNodes.ignoredUnassigned())) {
337                 String index = shardRoutingEntry.index();
338                 IndexRoutingTable.Builder indexBuilder = indexRoutingTableBuilders.get(index);
339                 if (indexBuilder == null) {
340                     indexBuilder = new IndexRoutingTable.Builder(index);
341                     indexRoutingTableBuilders.put(index, indexBuilder);
342                 }
343                 IndexShardRoutingTable refData = routingNodes.routingTable().index(shardRoutingEntry.index()).shard(shardRoutingEntry.id());
344                 indexBuilder.addShard(refData, shardRoutingEntry);
345             }
346 
347             for (ShardId shardId : routingNodes.getShardsToClearPostAllocationFlag()) {
348                 IndexRoutingTable.Builder indexRoutingBuilder = indexRoutingTableBuilders.get(shardId.index().name());
349                 if (indexRoutingBuilder != null) {
350                     indexRoutingBuilder.clearPostAllocationFlag(shardId);
351                 }
352             }
353 
354             for (IndexRoutingTable.Builder indexBuilder : indexRoutingTableBuilders.values()) {
355                 add(indexBuilder);
356             }
357             return this;
358         }

360         public Builder updateNumberOfReplicas(int numberOfReplicas, String... indices) throws IndexMissingException {
361             if (indices == null || indices.length == 0) {
362                 indices = indicesRouting.keySet().toArray(new String[indicesRouting.keySet().size()]);
363             }
364             for (String index : indices) {
365                 IndexRoutingTable indexRoutingTable = indicesRouting.get(index);
366                 if (indexRoutingTable == null) {
367                     // ignore index missing failure, its closed...
368                     continue;
369                 }
370                 int currentNumberOfReplicas = indexRoutingTable.shards().get(0).size() - 1; // remove the required primary
371                 IndexRoutingTable.Builder builder = new IndexRoutingTable.Builder(index);
372                 // re-add all the shards
373                 for (IndexShardRoutingTable indexShardRoutingTable : indexRoutingTable) {
374                     builder.addIndexShard(indexShardRoutingTable);
375                 }
376                 if (currentNumberOfReplicas < numberOfReplicas) {
377                     // now, add "empty" ones
378                     for (int i = 0; i < (numberOfReplicas - currentNumberOfReplicas); i++) {
379                         builder.addReplica();
380                     }
381                 } else if (currentNumberOfReplicas > numberOfReplicas) {
382                     int delta = currentNumberOfReplicas - numberOfReplicas;
383                     if (delta <= 0) {
384                         // ignore, can't remove below the current one...
385                     } else {
386                         for (int i = 0; i < delta; i++) {
387                             builder.removeReplica();
388                         }
389                     }
390                 }
391                 indicesRouting.put(index, builder.build());
392             }
393             return this;
394         }

396         public Builder addAsNew(IndexMetaData indexMetaData) {
397             if (indexMetaData.state() == IndexMetaData.State.OPEN) {
398                 IndexRoutingTable.Builder indexRoutingBuilder = new IndexRoutingTable.Builder(indexMetaData.index())
399                         .initializeAsNew(indexMetaData);
400                 add(indexRoutingBuilder);
401             }
402             return this;
403         }

449         public RoutingTable build() {
450             // normalize the versions right before we build it...
451             for (IndexRoutingTable indexRoutingTable : indicesRouting.values()) {
452                 indicesRouting.put(indexRoutingTable.index(), indexRoutingTable.normalizeVersions());
453             }
454             return new RoutingTable(version, indicesRouting);
455         }

457         public static RoutingTable readFrom(StreamInput in) throws IOException {
458             Builder builder = new Builder();
459             builder.version = in.readLong();
460             int size = in.readVInt();
461             for (int i = 0; i < size; i++) {
462                 IndexRoutingTable index = IndexRoutingTable.Builder.readFrom(in);
463                 builder.add(index);
464             }
465 
466             return builder.build();
467         }

469         public static void writeTo(RoutingTable table, StreamOutput out) throws IOException {
470             out.writeLong(table.version);
471             out.writeVInt(table.indicesRouting.size());
472             for (IndexRoutingTable index : table.indicesRouting.values()) {
473                 IndexRoutingTable.Builder.writeTo(index, out);
474             }
475         }

478     public String prettyPrint() {
479         StringBuilder sb = new StringBuilder("routing_table (version ").append(version).append("):\n");
480         for (Map.Entry<String, IndexRoutingTable> entry : indicesRouting.entrySet()) {
481             sb.append(entry.getValue().prettyPrint()).append('\n');
482         }
