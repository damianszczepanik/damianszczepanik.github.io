20  package org.elasticsearch.index.deletionpolicy;
21  
22  import org.apache.lucene.index.IndexCommit;
23  import org.elasticsearch.common.lease.Releasable;
24  import org.elasticsearch.common.lucene.IndexCommitDelegate;
25  
26  import java.io.IOException;
27  import java.util.ArrayList;
28  
29  /**
30   * A snapshot index commit point. While this is held and {@link #close()}
31   * was not called, no files will be deleted that relates to this commit point
32   * ({@link #getFileNames()}).
33   *
34   *
35   */
36  public class SnapshotIndexCommit extends IndexCommitDelegate implements Releasable {
37  
38      private final SnapshotDeletionPolicy deletionPolicy;
39  
40      private final String[] files;
41  
42      SnapshotIndexCommit(SnapshotDeletionPolicy deletionPolicy, IndexCommit cp) throws IOException {
43          super(cp);
44          this.deletionPolicy = deletionPolicy;
45          ArrayList<String> tmpFiles = new ArrayList<>();
46          for (String o : cp.getFileNames()) {
47              tmpFiles.add(o);
48          }
49          files = tmpFiles.toArray(new String[tmpFiles.size()]);
50      }
51  
52      public String[] getFiles() {
53          return files;
54      }
55  
56      /**
57       * Releases the current snapshot.
58       */
59      @Override
60      public void close() {
61          deletionPolicy.close(getGeneration());
62      }
63  
64      /**
65       * Override the delete operation, and only actually delete it if it
66       * is not held by the {@link SnapshotDeletionPolicy}.
67       */
68      @Override
69      public void delete() {
70          if (!deletionPolicy.isHeld(getGeneration())) {
71              delegate.delete();
72          }
73      }
74  }
