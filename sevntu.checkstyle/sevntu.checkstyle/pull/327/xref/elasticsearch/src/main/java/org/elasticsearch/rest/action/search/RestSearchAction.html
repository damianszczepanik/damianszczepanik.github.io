51      @Inject
52      public RestSearchAction(Settings settings, RestController controller, Client client) {
53          super(settings, controller, client);
54          controller.registerHandler(GET, "/_search", this);
55          controller.registerHandler(POST, "/_search", this);
56          controller.registerHandler(GET, "/{index}/_search", this);
57          controller.registerHandler(POST, "/{index}/_search", this);
58          controller.registerHandler(GET, "/{index}/{type}/_search", this);
59          controller.registerHandler(POST, "/{index}/{type}/_search", this);
60          controller.registerHandler(GET, "/_search/template", this);
61          controller.registerHandler(POST, "/_search/template", this);
62          controller.registerHandler(GET, "/{index}/_search/template", this);
63          controller.registerHandler(POST, "/{index}/_search/template", this);
64          controller.registerHandler(GET, "/{index}/{type}/_search/template", this);
65          controller.registerHandler(POST, "/{index}/{type}/_search/template", this);
66  
67          RestExistsAction restExistsAction = new RestExistsAction(settings, controller, client);
68          controller.registerHandler(GET, "/_search/exists", restExistsAction);
69          controller.registerHandler(POST, "/_search/exists", restExistsAction);
70          controller.registerHandler(GET, "/{index}/_search/exists", restExistsAction);
71          controller.registerHandler(POST, "/{index}/_search/exists", restExistsAction);
72          controller.registerHandler(GET, "/{index}/{type}/_search/exists", restExistsAction);
73          controller.registerHandler(POST, "/{index}/{type}/_search/exists", restExistsAction);
74      }

76      @Override
77      public void handleRequest(final RestRequest request, final RestChannel channel, final Client client) {
78          SearchRequest searchRequest;
79          searchRequest = RestSearchAction.parseSearchRequest(request);
80          searchRequest.listenerThreaded(false);
81          client.search(searchRequest, new RestStatusToXContentListener<SearchResponse>(channel));
82      }
83  
84      public static SearchRequest parseSearchRequest(RestRequest request) {
85          String[] indices = Strings.splitStringByCommaToArray(request.param("index"));
86          SearchRequest searchRequest = new SearchRequest(indices);
87          // get the content, and put it in the body
88          // add content/source as template if template flag is set
89          boolean isTemplateRequest = request.path().endsWith("/template");
90          if (request.hasContent()) {
91              if (isTemplateRequest) {
92                  searchRequest.templateSource(request.content(), request.contentUnsafe());
93              } else {
94                  searchRequest.source(request.content(), request.contentUnsafe());
95              }
96          } else {
97              String source = request.param("source");
98              if (source != null) {
99                  if (isTemplateRequest) {
100                     searchRequest.templateSource(source);
101                 } else {
102                     searchRequest.source(source);
103                 }
104             }
105         }
106 
107         searchRequest.extraSource(parseSearchSource(request));
108         searchRequest.searchType(request.param("search_type"));
109         searchRequest.queryCache(request.paramAsBoolean("query_cache", null));
110 
111         String scroll = request.param("scroll");
112         if (scroll != null) {
113             searchRequest.scroll(new Scroll(parseTimeValue(scroll, null)));
114         }
115 
116         searchRequest.types(Strings.splitStringByCommaToArray(request.param("type")));
117         searchRequest.routing(request.param("routing"));
118         searchRequest.preference(request.param("preference"));
119         searchRequest.indicesOptions(IndicesOptions.fromRequest(request, searchRequest.indicesOptions()));
120 
121         return searchRequest;
122     }

124     public static SearchSourceBuilder parseSearchSource(RestRequest request) {
125         SearchSourceBuilder searchSourceBuilder = null;
126         String queryString = request.param("q");
127         if (queryString != null) {
128             QueryStringQueryBuilder queryBuilder = QueryBuilders.queryStringQuery(queryString);
129             queryBuilder.defaultField(request.param("df"));
130             queryBuilder.analyzer(request.param("analyzer"));
131             queryBuilder.analyzeWildcard(request.paramAsBoolean("analyze_wildcard", false));
132             queryBuilder.lowercaseExpandedTerms(request.paramAsBoolean("lowercase_expanded_terms", true));
133             queryBuilder.lenient(request.paramAsBoolean("lenient", null));
134             String defaultOperator = request.param("default_operator");
135             if (defaultOperator != null) {
136                 if ("OR".equals(defaultOperator)) {
137                     queryBuilder.defaultOperator(QueryStringQueryBuilder.Operator.OR);
138                 } else if ("AND".equals(defaultOperator)) {
139                     queryBuilder.defaultOperator(QueryStringQueryBuilder.Operator.AND);
140                 } else {
141                     throw new ElasticsearchIllegalArgumentException("Unsupported defaultOperator [" + defaultOperator + "], can either be [OR] or [AND]");
142                 }
143             }
144             if (searchSourceBuilder == null) {
145                 searchSourceBuilder = new SearchSourceBuilder();
146             }
147             searchSourceBuilder.query(queryBuilder);
148         }
149 
150         int from = request.paramAsInt("from", -1);
151         if (from != -1) {
152             if (searchSourceBuilder == null) {
153                 searchSourceBuilder = new SearchSourceBuilder();
154             }
155             searchSourceBuilder.from(from);
156         }
157         int size = request.paramAsInt("size", -1);
158         if (size != -1) {
159             if (searchSourceBuilder == null) {
160                 searchSourceBuilder = new SearchSourceBuilder();
161             }
162             searchSourceBuilder.size(size);
163         }
164 
165         if (request.hasParam("explain")) {
166             if (searchSourceBuilder == null) {
167                 searchSourceBuilder = new SearchSourceBuilder();
168             }
169             searchSourceBuilder.explain(request.paramAsBoolean("explain", null));
170         }
171         if (request.hasParam("version")) {
172             if (searchSourceBuilder == null) {
173                 searchSourceBuilder = new SearchSourceBuilder();
174             }
175             searchSourceBuilder.version(request.paramAsBoolean("version", null));
176         }
177         if (request.hasParam("timeout")) {
178             if (searchSourceBuilder == null) {
179                 searchSourceBuilder = new SearchSourceBuilder();
180             }
181             searchSourceBuilder.timeout(request.paramAsTime("timeout", null));
182         }
183         if (request.hasParam("terminate_after")) {
184             if (searchSourceBuilder == null) {
185                 searchSourceBuilder = new SearchSourceBuilder();
186             }
187             int terminateAfter = request.paramAsInt("terminate_after",
188                     SearchContext.DEFAULT_TERMINATE_AFTER);
189             if (terminateAfter < 0) {
190                 throw new ElasticsearchIllegalArgumentException("terminateAfter must be > 0");
191             } else if (terminateAfter > 0) {
192                 searchSourceBuilder.terminateAfter(terminateAfter);
193             }
194         }
195 
196         String sField = request.param("fields");
197         if (sField != null) {
198             if (searchSourceBuilder == null) {
199                 searchSourceBuilder = new SearchSourceBuilder();
200             }
201             if (!Strings.hasText(sField)) {
202                 searchSourceBuilder.noFields();
203             } else {
204                 String[] sFields = Strings.splitStringByCommaToArray(sField);
205                 if (sFields != null) {
206                     for (String field : sFields) {
207                         searchSourceBuilder.field(field);
208                     }
209                 }
210             }
211         }
212         FetchSourceContext fetchSourceContext = FetchSourceContext.parseFromRestRequest(request);
213         if (fetchSourceContext != null) {
214             if (searchSourceBuilder == null) {
215                 searchSourceBuilder = new SearchSourceBuilder();
216             }
217             searchSourceBuilder.fetchSource(fetchSourceContext);
218         }
219 
220         if (request.hasParam("track_scores")) {
221             if (searchSourceBuilder == null) {
222                 searchSourceBuilder = new SearchSourceBuilder();
223             }
224             searchSourceBuilder.trackScores(request.paramAsBoolean("track_scores", false));
225         }
226 
227         String sSorts = request.param("sort");
228         if (sSorts != null) {
229             if (searchSourceBuilder == null) {
230                 searchSourceBuilder = new SearchSourceBuilder();
231             }
232             String[] sorts = Strings.splitStringByCommaToArray(sSorts);
233             for (String sort : sorts) {
234                 int delimiter = sort.lastIndexOf(":");
235                 if (delimiter != -1) {
236                     String sortField = sort.substring(0, delimiter);
237                     String reverse = sort.substring(delimiter + 1);
238                     if ("asc".equals(reverse)) {
239                         searchSourceBuilder.sort(sortField, SortOrder.ASC);
240                     } else if ("desc".equals(reverse)) {
241                         searchSourceBuilder.sort(sortField, SortOrder.DESC);
242                     }
243                 } else {
244                     searchSourceBuilder.sort(sort);
245                 }
246             }
247         }
248 
249         String sIndicesBoost = request.param("indices_boost");
250         if (sIndicesBoost != null) {
251             if (searchSourceBuilder == null) {
252                 searchSourceBuilder = new SearchSourceBuilder();
253             }
254             String[] indicesBoost = Strings.splitStringByCommaToArray(sIndicesBoost);
255             for (String indexBoost : indicesBoost) {
256                 int divisor = indexBoost.indexOf(',');
257                 if (divisor == -1) {
258                     throw new ElasticsearchIllegalArgumentException("Illegal index boost [" + indexBoost + "], no ','");
259                 }
260                 String indexName = indexBoost.substring(0, divisor);
261                 String sBoost = indexBoost.substring(divisor + 1);
262                 try {
263                     searchSourceBuilder.indexBoost(indexName, Float.parseFloat(sBoost));
264                 } catch (NumberFormatException e) {
265                     throw new ElasticsearchIllegalArgumentException("Illegal index boost [" + indexBoost + "], boost not a float number");
266                 }
267             }
268         }
269 
270         String sStats = request.param("stats");
271         if (sStats != null) {
272             if (searchSourceBuilder == null) {
273                 searchSourceBuilder = new SearchSourceBuilder();
274             }
275             searchSourceBuilder.stats(Strings.splitStringByCommaToArray(sStats));
276         }
277 
278         String suggestField = request.param("suggest_field");
279         if (suggestField != null) {
280             String suggestText = request.param("suggest_text", queryString);
281             int suggestSize = request.paramAsInt("suggest_size", 5);
282             if (searchSourceBuilder == null) {
283                 searchSourceBuilder = new SearchSourceBuilder();
284             }
285             String suggestMode = request.param("suggest_mode");
286             searchSourceBuilder.suggest().addSuggestion(
287                     termSuggestion(suggestField).field(suggestField).text(suggestText).size(suggestSize)
288                             .suggestMode(suggestMode)
289             );
290         }
291 
292         return searchSourceBuilder;
293     }
