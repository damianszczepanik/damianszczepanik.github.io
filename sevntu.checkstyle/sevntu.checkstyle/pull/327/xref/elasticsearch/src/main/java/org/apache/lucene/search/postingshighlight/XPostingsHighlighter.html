94      /**
95       * Creates a new highlighter, specifying maximum content length.
96       * @param maxLength maximum content size to process.
97       * @throws IllegalArgumentException if <code>maxLength</code> is negative or <code>Integer.MAX_VALUE</code>
98       */
99      public XPostingsHighlighter(int maxLength) {
100         if (maxLength < 0 || maxLength == Integer.MAX_VALUE) {
101             // two reasons: no overflow problems in BreakIterator.preceding(offset+1),
102             // our sentinel in the offsets queue uses this value to terminate.
103             throw new IllegalArgumentException("maxLength must be < Integer.MAX_VALUE");
104         }
105         this.maxLength = maxLength;
106     }

269     public Map<String,String[]> highlightFields(String fieldsIn[], Query query, IndexSearcher searcher, int[] docidsIn, int maxPassagesIn[]) throws IOException {
270         Map<String,String[]> snippets = new HashMap<>();
271         for(Map.Entry<String,Object[]> ent : highlightFieldsAsObjects(fieldsIn, query, searcher, docidsIn, maxPassagesIn).entrySet()) {
272             Object[] snippetObjects = ent.getValue();
273             String[] snippetStrings = new String[snippetObjects.length];
274             snippets.put(ent.getKey(), snippetStrings);
275             for(int i=0;i<snippetObjects.length;i++) {
276                 Object snippet = snippetObjects[i];
277                 if (snippet != null) {
278                     snippetStrings[i] = snippet.toString();
279                 }
280             }
281         }
282 
283         return snippets;
284     }

286     public Map<String,Object[]> highlightFieldsAsObjects(String fieldsIn[], Query query, IndexSearcher searcher, int[] docidsIn, int maxPassagesIn[]) throws IOException {
287         if (fieldsIn.length < 1) {
288             throw new IllegalArgumentException("fieldsIn must not be empty");
289         }
290         if (fieldsIn.length != maxPassagesIn.length) {
291             throw new IllegalArgumentException("invalid number of maxPassagesIn");
292         }
293         final IndexReader reader = searcher.getIndexReader();
294         query = rewrite(query);
295         SortedSet<Term> queryTerms = new TreeSet<>();
296         query.extractTerms(queryTerms);
297 
298         IndexReaderContext readerContext = reader.getContext();
299         List<LeafReaderContext> leaves = readerContext.leaves();
300 
301         // Make our own copies because we sort in-place:
302         int[] docids = new int[docidsIn.length];
303         System.arraycopy(docidsIn, 0, docids, 0, docidsIn.length);
304         final String fields[] = new String[fieldsIn.length];
305         System.arraycopy(fieldsIn, 0, fields, 0, fieldsIn.length);
306         final int maxPassages[] = new int[maxPassagesIn.length];
307         System.arraycopy(maxPassagesIn, 0, maxPassages, 0, maxPassagesIn.length);
308 
309         // sort for sequential io
310         Arrays.sort(docids);
311         new InPlaceMergeSorter() {
312 
313             @Override
314             protected void swap(int i, int j) {
315                 String tmp = fields[i];
316                 fields[i] = fields[j];
317                 fields[j] = tmp;
318                 int tmp2 = maxPassages[i];
319                 maxPassages[i] = maxPassages[j];
320                 maxPassages[j] = tmp2;
321             }
322 
323             @Override
324             protected int compare(int i, int j) {
325                 return fields[i].compareTo(fields[j]);
326             }
327 
328         }.sort(0, fields.length);
329 
330         // pull stored data:
331         String[][] contents = loadFieldValues(searcher, fields, docids, maxLength);
332 
333         Map<String,Object[]> highlights = new HashMap<>();
334         for (int i = 0; i < fields.length; i++) {
335             String field = fields[i];
336             int numPassages = maxPassages[i];
337 
338             Term floor = new Term(field, "");
339             Term ceiling = new Term(field, UnicodeUtil.BIG_TERM);
340             SortedSet<Term> fieldTerms = queryTerms.subSet(floor, ceiling);
341             // TODO: should we have some reasonable defaults for term pruning? (e.g. stopwords)
342 
343             // Strip off the redundant field:
344             BytesRef terms[] = new BytesRef[fieldTerms.size()];
345             int termUpto = 0;
346             for(Term term : fieldTerms) {
347                 terms[termUpto++] = term.bytes();
348             }
349             Map<Integer,Object> fieldHighlights = highlightField(field, contents[i], getBreakIterator(field), terms, docids, leaves, numPassages);
350 
351             Object[] result = new Object[docids.length];
352             for (int j = 0; j < docidsIn.length; j++) {
353                 result[j] = fieldHighlights.get(docidsIn[j]);
354             }
355             highlights.put(field, result);
356         }
357         return highlights;
358     }

360     /** Loads the String values for each field X docID to be
361      *  highlighted.  By default this loads from stored
362      *  fields, but a subclass can change the source.  This
363      *  method should allocate the String[fields.length][docids.length]
364      *  and fill all values.  The returned Strings must be
365      *  identical to what was indexed. */
366     protected String[][] loadFieldValues(IndexSearcher searcher, String[] fields, int[] docids, int maxLength) throws IOException {
367         String contents[][] = new String[fields.length][docids.length];
368         char valueSeparators[] = new char[fields.length];
369         for (int i = 0; i < fields.length; i++) {
370             valueSeparators[i] = getMultiValuedSeparator(fields[i]);
371         }
372         LimitedStoredFieldVisitor visitor = new LimitedStoredFieldVisitor(fields, valueSeparators, maxLength);
373         for (int i = 0; i < docids.length; i++) {
374             searcher.doc(docids[i], visitor);
375             for (int j = 0; j < fields.length; j++) {
376                 contents[j][i] = visitor.getValue(j);
377             }
378             visitor.reset();
379         }
380         return contents;
381     }

393     //BEGIN EDIT: made protected so that we can call from our subclass and pass in the terms by ourselves
394     protected Map<Integer,Object> highlightField(String field, String contents[], BreakIterator bi, BytesRef terms[], int[] docids, List<LeafReaderContext> leaves, int maxPassages) throws IOException {
395     //private Map<Integer,Object> highlightField(String field, String contents[], BreakIterator bi, BytesRef terms[], int[] docids, List<LeafReaderContext > leaves, int maxPassages) throws IOException {
396     //END EDIT
397 
398         Map<Integer,Object> highlights = new HashMap<>();
399 
400         // reuse in the real sense... for docs in same segment we just advance our old enum
401         PostingsEnum postings[] = null;
402         TermsEnum termsEnum = null;
403         int lastLeaf = -1;
404 
405         PassageFormatter fieldFormatter = getFormatter(field);
406         if (fieldFormatter == null) {
407             throw new NullPointerException("PassageFormatter cannot be null");
408         }
409 
410         for (int i = 0; i < docids.length; i++) {
411             String content = contents[i];
412             if (content.length() == 0) {
413                 continue; // nothing to do
414             }
415             bi.setText(content);
416             int doc = docids[i];
417             int leaf = ReaderUtil.subIndex(doc, leaves);
418             LeafReaderContext subContext = leaves.get(leaf);
419             LeafReader r = subContext.reader();
420             Terms t = r.terms(field);
421             if (t == null) {
422                 continue; // nothing to do
423             }
424             if (leaf != lastLeaf) {
425                 termsEnum = t.iterator(null);
426                 postings = new PostingsEnum[terms.length];
427             }
428             Passage passages[] = highlightDoc(field, terms, content.length(), bi, doc - subContext.docBase, termsEnum, postings, maxPassages);
429             if (passages.length == 0) {
430                 passages = getEmptyHighlight(field, bi, maxPassages);
431             }
432             if (passages.length > 0) {
433                 // otherwise a null snippet (eg if field is missing
434                 // entirely from the doc)
435                   highlights.put(doc, fieldFormatter.format(passages, content));
436             }
437             lastLeaf = leaf;
438         }
439 
440         return highlights;
441     }

443     // algorithm: treat sentence snippets as miniature documents
444     // we can intersect these with the postings lists via BreakIterator.preceding(offset),s
445     // score each sentence as norm(sentenceStartOffset) * sum(weight * tf(freq))
446     private Passage[] highlightDoc(String field, BytesRef terms[], int contentLength, BreakIterator bi, int doc,
447                                    TermsEnum termsEnum, PostingsEnum[] postings, int n) throws IOException {
448 
449         //BEGIN EDIT added call to method that returns the offset for the current value (discrete highlighting)
450         int valueOffset = getOffsetForCurrentValue(field, doc);
451         //END EDIT
452 
453         PassageScorer scorer = getScorer(field);
454         if (scorer == null) {
455             throw new NullPointerException("PassageScorer cannot be null");
456         }
457 
458 
459         //BEGIN EDIT discrete highlighting
460         // the scoring needs to be based on the length of the whole field (all values rather than only the current one)
461         int totalContentLength = getContentLength(field, doc);
462         if (totalContentLength == -1) {
463             totalContentLength = contentLength;
464         }
465         //END EDIT
466 
467 
468         PriorityQueue<OffsetsEnum> pq = new PriorityQueue<>();
469         float weights[] = new float[terms.length];
470         // initialize postings
471         for (int i = 0; i < terms.length; i++) {
472             PostingsEnum de = postings[i];
473             int pDoc;
474             if (de == EMPTY) {
475                 continue;
476             } else if (de == null) {
477                 postings[i] = EMPTY; // initially
478                 if (!termsEnum.seekExact(terms[i])) {
479                     continue; // term not found
480                 }
481                 de = postings[i] = termsEnum.postings(null, null, PostingsEnum.OFFSETS);
482                 if (de == null) {
483                     // no positions available
484                     throw new IllegalArgumentException("field '" + field + "' was indexed without offsets, cannot highlight");
485                 }
486                 pDoc = de.advance(doc);
487             } else {
488                 pDoc = de.docID();
489                 if (pDoc < doc) {
490                     pDoc = de.advance(doc);
491                 }
492             }
493 
494             if (doc == pDoc) {
495                 //BEGIN EDIT we take into account the length of the whole field (all values) to properly score the snippets
496                 weights[i] = scorer.weight(totalContentLength, de.freq());
497                 //weights[i] = scorer.weight(contentLength, de.freq());
498                 //END EDIT
499                 de.nextPosition();
500                 pq.add(new OffsetsEnum(de, i));
501             }
502         }
503 
504         pq.add(new OffsetsEnum(EMPTY, Integer.MAX_VALUE)); // a sentinel for termination
505 
506         PriorityQueue<Passage> passageQueue = new PriorityQueue<>(n, new Comparator<Passage>() {
507             @Override
508             public int compare(Passage left, Passage right) {
509                 if (left.score < right.score) {
510                     return -1;
511                 } else if (left.score > right.score) {
512                     return 1;
513                 } else {
514                     return left.startOffset - right.startOffset;
515                 }
516             }
517         });
518         Passage current = new Passage();
519 
520         OffsetsEnum off;
521         while ((off = pq.poll()) != null) {
522             final PostingsEnum dp = off.dp;
523 
524             int start = dp.startOffset();
525             if (start == -1) {
526                 throw new IllegalArgumentException("field '" + field + "' was indexed without offsets, cannot highlight");
527             }
528             int end = dp.endOffset();
529             // LUCENE-5166: this hit would span the content limit... however more valid
530             // hits may exist (they are sorted by start). so we pretend like we never
531             // saw this term, it won't cause a passage to be added to passageQueue or anything.
532             assert EMPTY.startOffset() == Integer.MAX_VALUE;
533             if (start < contentLength && end > contentLength) {
534                 continue;
535             }
536 
537 
538             //BEGIN EDIT support for discrete highlighting (added block code)
539             //switch to the first match in the current value if there is one
540             boolean seenEnough = false;
541             while (start < valueOffset) {
542                 if (off.pos == dp.freq()) {
543                     seenEnough = true;
544                     break;
545                 } else {
546                     off.pos++;
547                     dp.nextPosition();
548                     start = dp.startOffset();
549                     end = dp.endOffset();
550                 }
551             }
552 
553             //continue with next term if we've already seen the current one all the times it appears
554             //that means that the current value doesn't hold matches for the current term
555             if (seenEnough) {
556                 continue;
557             }
558 
559             //we now subtract the offset of the current value to both start and end
560             start -= valueOffset;
561             end -= valueOffset;
562             //END EDIT
563 
564 
565             if (start >= current.endOffset) {
566                 if (current.startOffset >= 0) {
567                     // finalize current
568                     //BEGIN EDIT we take into account the value offset when scoring the snippet based on its position
569                     current.score *= scorer.norm(current.startOffset + valueOffset);
570                     //current.score *= scorer.norm(current.startOffset);
571                     //END EDIT
572                     // new sentence: first add 'current' to queue
573                     if (passageQueue.size() == n && current.score < passageQueue.peek().score) {
574                         current.reset(); // can't compete, just reset it
575                     } else {
576                         passageQueue.offer(current);
577                         if (passageQueue.size() > n) {
578                             current = passageQueue.poll();
579                             current.reset();
580                         } else {
581                             current = new Passage();
582                         }
583                     }
584                 }
585                 // if we exceed limit, we are done
586                 if (start >= contentLength) {
587                     Passage passages[] = new Passage[passageQueue.size()];
588                     passageQueue.toArray(passages);
589                     for (Passage p : passages) {
590                         p.sort();
591                     }
592                     // sort in ascending order
593                     Arrays.sort(passages, new Comparator<Passage>() {
594                         @Override
595                         public int compare(Passage left, Passage right) {
596                             return left.startOffset - right.startOffset;
597                         }
598                     });
599                     return passages;
600                 }
601                 // advance breakiterator
602                 assert BreakIterator.DONE < 0;
603                 current.startOffset = Math.max(bi.preceding(start+1), 0);
604                 current.endOffset = Math.min(bi.next(), contentLength);
605             }
606             int tf = 0;
607             while (true) {
608                 tf++;
609                 current.addMatch(start, end, terms[off.id]);
610                 if (off.pos == dp.freq()) {
611                     break; // removed from pq
612                 } else {
613                     off.pos++;
614                     dp.nextPosition();
615                     //BEGIN EDIT support for discrete highlighting
616                     start = dp.startOffset() - valueOffset;
617                     end = dp.endOffset() - valueOffset;
618                     //start = dp.startOffset();
619                     //end = dp.endOffset();
620                     //END EDIT
621                 }
622                 if (start >= current.endOffset || end > contentLength) {
623                     pq.offer(off);
624                     break;
625                 }
626             }
627             current.score += weights[off.id] * scorer.tf(tf, current.endOffset - current.startOffset);
628         }
629 
630         // Dead code but compiler disagrees:
631         assert false;
632         return null;
633     }

635     /** Called to summarize a document when no hits were
636      *  found.  By default this just returns the first
637      *  {@code maxPassages} sentences; subclasses can override
638      *  to customize. */
639     protected Passage[] getEmptyHighlight(String fieldName, BreakIterator bi, int maxPassages) {
640         // BreakIterator should be un-next'd:
641         List<Passage> passages = new ArrayList<>();
642         int pos = bi.current();
643         assert pos == 0;
644         while (passages.size() < maxPassages) {
645             int next = bi.next();
646             if (next == BreakIterator.DONE) {
647                 break;
648             }
649             Passage passage = new Passage();
650             passage.score = Float.NaN;
651             passage.startOffset = pos;
652             passage.endOffset = next;
653             passages.add(passage);
654             pos = next;
655         }
656 
657         return passages.toArray(new Passage[passages.size()]);
658     }

660     private static class OffsetsEnum implements Comparable<OffsetsEnum> {
661         PostingsEnum dp;
662         int pos;
663         int id;
664 
665         OffsetsEnum(PostingsEnum dp, int id) throws IOException {
666             this.dp = dp;
667             this.id = id;
668             this.pos = 1;
669         }
670 
671         @Override
672         public int compareTo(OffsetsEnum other) {
673             try {
674                 int off = dp.startOffset();
675                 int otherOff = other.dp.startOffset();
676                 if (off == otherOff) {
677                     return id - other.id;
678                 } else {
679                     return Long.signum(((long)off) - otherOff);
680                 }
681             } catch (IOException e) {
682                 throw new RuntimeException(e);
683             }
684         }
685     }

718      * we rewrite against an empty indexreader: as we don't want things like
719      * rangeQueries that don't summarize the document
720      */
721     private static Query rewrite(Query original) throws IOException {
722         Query query = original;
723         for (Query rewrittenQuery = query.rewrite(EMPTY_INDEXREADER); rewrittenQuery != query;
724              rewrittenQuery = query.rewrite(EMPTY_INDEXREADER)) {
725             query = rewrittenQuery;
726         }
727         return query;

730     private static class LimitedStoredFieldVisitor extends StoredFieldVisitor {
731         private final String fields[];
732         private final char valueSeparators[];
733         private final int maxLength;
734         private final StringBuilder builders[];
735         private int currentField = -1;
736 
737         public LimitedStoredFieldVisitor(String fields[], char valueSeparators[], int maxLength) {
738             assert fields.length == valueSeparators.length;
739             this.fields = fields;
740             this.valueSeparators = valueSeparators;
741             this.maxLength = maxLength;
742             builders = new StringBuilder[fields.length];
743             for (int i = 0; i < builders.length; i++) {
744                 builders[i] = new StringBuilder();
745             }
746         }
747 
748         @Override
749         public void stringField(FieldInfo fieldInfo, String value) throws IOException {
750             assert currentField >= 0;
751             StringBuilder builder = builders[currentField];
752             if (builder.length() > 0 && builder.length() < maxLength) {
753                 builder.append(valueSeparators[currentField]);
754             }
755             if (builder.length() + value.length() > maxLength) {
756                 builder.append(value, 0, maxLength - builder.length());
757             } else {
758                 builder.append(value);
759             }
760         }
761 
762         @Override
763         public Status needsField(FieldInfo fieldInfo) throws IOException {
764             currentField = Arrays.binarySearch(fields, fieldInfo.name);
765             if (currentField < 0) {
766                 return Status.NO;
767             } else if (builders[currentField].length() > maxLength) {
768                 return fields.length == 1 ? Status.STOP : Status.NO;
769             }
770             return Status.YES;
771         }
772 
773         String getValue(int i) {
774             return builders[i].toString();
775         }
776 
777         void reset() {
778             currentField = -1;
779             for (int i = 0; i < fields.length; i++) {
780                 builders[i].setLength(0);
781             }
782         }
783     }
