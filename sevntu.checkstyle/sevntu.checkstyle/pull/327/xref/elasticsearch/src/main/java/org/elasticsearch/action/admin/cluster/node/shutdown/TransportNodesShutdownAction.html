50  public class TransportNodesShutdownAction extends TransportMasterNodeOperationAction<NodesShutdownRequest, NodesShutdownResponse> {
51  
52      public static final String SHUTDOWN_NODE_ACTION_NAME = NodesShutdownAction.NAME + "[n]";
53  
54      private final Node node;
55      private final ClusterName clusterName;
56      private final boolean disabled;
57      private final TimeValue delay;
58  
59      @Inject
60      public TransportNodesShutdownAction(Settings settings, TransportService transportService, ClusterService clusterService, ThreadPool threadPool,
61                                          Node node, ClusterName clusterName, ActionFilters actionFilters) {
62          super(settings, NodesShutdownAction.NAME, transportService, clusterService, threadPool, actionFilters);
63          this.node = node;
64          this.clusterName = clusterName;
65          this.disabled = settings.getAsBoolean("action.disable_shutdown", this.settings.getAsBoolean("action.admin.cluster.node.shutdown.disabled", false));
66          this.delay = this.settings.getAsTime("action.admin.cluster.node.shutdown.delay", TimeValue.timeValueMillis(200));
67  
68          this.transportService.registerHandler(SHUTDOWN_NODE_ACTION_NAME, new NodeShutdownRequestHandler());
69      }
70  
71      @Override
72      protected String executor() {
73          return ThreadPool.Names.GENERIC;
74      }
75  
76      @Override
77      protected ClusterBlockException checkBlock(NodesShutdownRequest request, ClusterState state) {
78          return state.blocks().globalBlockedException(ClusterBlockLevel.METADATA);
79      }
80  
81      @Override
82      protected NodesShutdownRequest newRequest() {
83          return new NodesShutdownRequest();
84      }
85  
86      @Override
87      protected NodesShutdownResponse newResponse() {
88          return new NodesShutdownResponse();
89      }
90  
91      @Override
92      protected void processBeforeDelegationToMaster(NodesShutdownRequest request, ClusterState state) {
93          String[] nodesIds = request.nodesIds;
94          if (nodesIds != null) {
95              for (int i = 0; i < nodesIds.length; i++) {
96                  // replace the _local one, since it looses its meaning when going over to the master...
97                  if ("_local".equals(nodesIds[i])) {
98                      nodesIds[i] = state.nodes().localNodeId();
99                  }
100             }
101         }
102     }
103 
104     @Override
105     protected void masterOperation(final NodesShutdownRequest request, final ClusterState state, final ActionListener<NodesShutdownResponse> listener) throws ElasticsearchException {
106         if (disabled) {
107             throw new ElasticsearchIllegalStateException("Shutdown is disabled");
108         }
109         final ObjectOpenHashSet<DiscoveryNode> nodes = new ObjectOpenHashSet<>();
110         if (state.nodes().isAllNodes(request.nodesIds)) {
111             logger.info("[cluster_shutdown]: requested, shutting down in [{}]", request.delay);
112             nodes.addAll(state.nodes().dataNodes().values());
113             nodes.addAll(state.nodes().masterNodes().values());
114             Thread t = new Thread(new Runnable() {
115                 @Override
116                 public void run() {
117                     try {
118                         Thread.sleep(request.delay.millis());
119                     } catch (InterruptedException e) {
120                         // ignore
121                     }
122                     // first, stop the cluster service
123                     logger.trace("[cluster_shutdown]: stopping the cluster service so no re-routing will occur");
124                     clusterService.stop();
125 
126                     final CountDownLatch latch = new CountDownLatch(nodes.size());
127                     for (ObjectCursor<DiscoveryNode> cursor : nodes) {
128                         final DiscoveryNode node = cursor.value;
129                         if (node.id().equals(state.nodes().masterNodeId())) {
130                             // don't shutdown the master yet...
131                             latch.countDown();
132                         } else {
133                             logger.trace("[cluster_shutdown]: sending shutdown request to [{}]", node);
134                             transportService.sendRequest(node, SHUTDOWN_NODE_ACTION_NAME, new NodeShutdownRequest(request), new EmptyTransportResponseHandler(ThreadPool.Names.SAME) {
135                                 @Override
136                                 public void handleResponse(TransportResponse.Empty response) {
137                                     logger.trace("[cluster_shutdown]: received shutdown response from [{}]", node);
138                                     latch.countDown();
139                                 }
140 
141                                 @Override
142                                 public void handleException(TransportException exp) {
143                                     logger.warn("[cluster_shutdown]: received failed shutdown response from [{}]", exp, node);
144                                     latch.countDown();
145                                 }
146                             });
147                         }
148                     }
149                     try {
150                         latch.await();
151                     } catch (InterruptedException e) {
152                         // ignore
153                     }
154                     logger.info("[cluster_shutdown]: done shutting down all nodes except master, proceeding to master");
155 
156                     // now, kill the master
157                     logger.trace("[cluster_shutdown]: shutting down the master [{}]", state.nodes().masterNode());
158                     transportService.sendRequest(state.nodes().masterNode(), SHUTDOWN_NODE_ACTION_NAME, new NodeShutdownRequest(request), new EmptyTransportResponseHandler(ThreadPool.Names.SAME) {
159                         @Override
160                         public void handleResponse(TransportResponse.Empty response) {
161                             logger.trace("[cluster_shutdown]: received shutdown response from master");
162                         }
163 
164                         @Override
165                         public void handleException(TransportException exp) {
166                             logger.warn("[cluster_shutdown]: received failed shutdown response master", exp);
167                         }
168                     });
169                 }
170             });
171             t.start();
172         } else {
173             final String[] nodesIds = state.nodes().resolveNodesIds(request.nodesIds);
174             logger.info("[partial_cluster_shutdown]: requested, shutting down [{}] in [{}]", nodesIds, request.delay);
175 
176             for (String nodeId : nodesIds) {
177                 final DiscoveryNode node = state.nodes().get(nodeId);
178                 if (node != null) {
179                     nodes.add(node);
180                 }
181             }
182 
183             Thread t = new Thread(new Runnable() {
184                 @Override
185                 public void run() {
186                     try {
187                         Thread.sleep(request.delay.millis());
188                     } catch (InterruptedException e) {
189                         // ignore
190                     }
191 
192                     final CountDownLatch latch = new CountDownLatch(nodesIds.length);
193                     for (String nodeId : nodesIds) {
194                         final DiscoveryNode node = state.nodes().get(nodeId);
195                         if (node == null) {
196                             logger.warn("[partial_cluster_shutdown]: no node to shutdown for node_id [{}]", nodeId);
197                             latch.countDown();
198                             continue;
199                         }
200 
201                         logger.trace("[partial_cluster_shutdown]: sending shutdown request to [{}]", node);
202                         transportService.sendRequest(node, SHUTDOWN_NODE_ACTION_NAME, new NodeShutdownRequest(request), new EmptyTransportResponseHandler(ThreadPool.Names.SAME) {
203                             @Override
204                             public void handleResponse(TransportResponse.Empty response) {
205                                 logger.trace("[partial_cluster_shutdown]: received shutdown response from [{}]", node);
206                                 latch.countDown();
207                             }
208 
209                             @Override
210                             public void handleException(TransportException exp) {
211                                 logger.warn("[partial_cluster_shutdown]: received failed shutdown response from [{}]", exp, node);
212                                 latch.countDown();
213                             }
214                         });
215                     }
216 
217                     try {
218                         latch.await();
219                     } catch (InterruptedException e) {
220                         // ignore
221                     }
222 
223                     logger.info("[partial_cluster_shutdown]: done shutting down [{}]", ((Object) nodesIds));
224                 }
225             });
226             t.start();
227         }
228         listener.onResponse(new NodesShutdownResponse(clusterName, nodes.toArray(DiscoveryNode.class)));
229     }
230 
231     private class NodeShutdownRequestHandler extends BaseTransportRequestHandler<NodeShutdownRequest> {
232 
233         @Override
234         public NodeShutdownRequest newInstance() {
235             return new NodeShutdownRequest();
236         }
237 
238         @Override
239         public String executor() {
240             return ThreadPool.Names.SAME;
241         }
242 
243         @Override
244         public void messageReceived(final NodeShutdownRequest request, TransportChannel channel) throws Exception {
245             if (disabled) {
246                 throw new ElasticsearchIllegalStateException("Shutdown is disabled");
247             }
248             logger.info("shutting down in [{}]", delay);
249             Thread t = new Thread(new Runnable() {
250                 @Override
251                 public void run() {
252                     try {
253                         Thread.sleep(delay.millis());
254                     } catch (InterruptedException e) {
255                         // ignore
256                     }
257                     if (!request.exit) {
258                         logger.info("initiating requested shutdown (no exit)...");
259                         try {
260                             node.close();
261                         } catch (Exception e) {
262                             logger.warn("Failed to shutdown", e);
263                         }
264                         return;
265                     }
266                     boolean shutdownWithWrapper = false;
267                     if (System.getProperty("elasticsearch-service") != null) {
268                         try {
269                             Class wrapperManager = settings.getClassLoader().loadClass("org.tanukisoftware.wrapper.WrapperManager");
270                             logger.info("initiating requested shutdown (using service)");
271                             wrapperManager.getMethod("stopAndReturn", int.class).invoke(null, 0);
272                             shutdownWithWrapper = true;
273                         } catch (Throwable e) {
274                             logger.error("failed to initial shutdown on service wrapper", e);
275                         }
276                     }
277                     if (!shutdownWithWrapper) {
278                         logger.info("initiating requested shutdown...");
279                         try {
280                             node.close();
281                         } catch (Exception e) {
282                             logger.warn("Failed to shutdown", e);
283                         } finally {
284                             // make sure we initiate the shutdown hooks, so the Bootstrap#main thread will exit
285                             System.exit(0);
286                         }
287                     }
288                 }
289             });
290             t.start();
291 
292             channel.sendResponse(TransportResponse.Empty.INSTANCE);
293         }
294     }
295 
296     static class NodeShutdownRequest extends TransportRequest {
297 
298         boolean exit;
299 
300         NodeShutdownRequest() {
301         }
302 
303         NodeShutdownRequest(NodesShutdownRequest request) {
304             super(request);
305             this.exit = request.exit();
306         }
307 
308         @Override
309         public void readFrom(StreamInput in) throws IOException {
310             super.readFrom(in);
311             exit = in.readBoolean();
312         }
313 
314         @Override
315         public void writeTo(StreamOutput out) throws IOException {
316             super.writeTo(out);
317             out.writeBoolean(exit);
318         }
319     }
