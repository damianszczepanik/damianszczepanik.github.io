62      public static final String INDEX_RECOVERY_INITIAL_SHARDS = "index.recovery.initial_shards";
63  
64      private final TransportNodesListGatewayStartedShards listGatewayStartedShards;
65  
66      private final TransportNodesListShardStoreMetaData listShardStoreMetaData;
67  
68      private final ConcurrentMap<ShardId, Map<DiscoveryNode, TransportNodesListShardStoreMetaData.StoreFilesMetaData>> cachedStores = ConcurrentCollections.newConcurrentMap();
69  
70      private final ConcurrentMap<ShardId, ObjectLongOpenHashMap<DiscoveryNode>> cachedShardsState = ConcurrentCollections.newConcurrentMap();
71  
72      private final TimeValue listTimeout;
73  
74      private final String initialShards;

76      @Inject
77      public GatewayAllocator(Settings settings,
78                              TransportNodesListGatewayStartedShards listGatewayStartedShards, TransportNodesListShardStoreMetaData listShardStoreMetaData) {
79          super(settings);
80          this.listGatewayStartedShards = listGatewayStartedShards;
81          this.listShardStoreMetaData = listShardStoreMetaData;
82  
83          this.listTimeout = settings.getAsTime("gateway.list_timeout", settings.getAsTime("gateway.local.list_timeout", TimeValue.timeValueSeconds(30)));
84          this.initialShards = settings.get("gateway.initial_shards", settings.get("gateway.local.initial_shards", "quorum"));
85  
86          logger.debug("using initial_shards [{}], list_timeout [{}]", initialShards, listTimeout);
87      }

89      public void applyStartedShards(StartedRerouteAllocation allocation) {
90          for (ShardRouting shardRouting : allocation.startedShards()) {
91              cachedStores.remove(shardRouting.shardId());
92              cachedShardsState.remove(shardRouting.shardId());
93          }
94      }
95  
96      public void applyFailedShards(FailedRerouteAllocation allocation) {
97          for (ShardRouting failedShard : allocation.failedShards()) {
98              cachedStores.remove(failedShard.shardId());
99              cachedShardsState.remove(failedShard.shardId());
100         }
101     }

103     public boolean allocateUnassigned(RoutingAllocation allocation) {
104         boolean changed = false;
105         DiscoveryNodes nodes = allocation.nodes();
106         RoutingNodes routingNodes = allocation.routingNodes();
107 
108         // First, handle primaries, they must find a place to be allocated on here
109         Iterator<MutableShardRouting> unassignedIterator = routingNodes.unassigned().iterator();
110         while (unassignedIterator.hasNext()) {
111             MutableShardRouting shard = unassignedIterator.next();
112 
113             if (!shard.primary()) {
114                 continue;
115             }
116 
117             // this is an API allocation, ignore since we know there is no data...
118             if (!routingNodes.routingTable().index(shard.index()).shard(shard.id()).primaryAllocatedPostApi()) {
119                 continue;
120             }
121 
122             ObjectLongOpenHashMap<DiscoveryNode> nodesState = buildShardStates(nodes, shard);
123 
124             int numberOfAllocationsFound = 0;
125             long highestVersion = -1;
126             Set<DiscoveryNode> nodesWithHighestVersion = Sets.newHashSet();
127             final boolean[] states = nodesState.allocated;
128             final Object[] keys = nodesState.keys;
129             final long[] values = nodesState.values;
130             for (int i = 0; i < states.length; i++) {
131                 if (!states[i]) {
132                     continue;
133                 }
134 
135                 DiscoveryNode node = (DiscoveryNode) keys[i];
136                 long version = values[i];
137                 // since we don't check in NO allocation, we need to double check here
138                 if (allocation.shouldIgnoreShardForNode(shard.shardId(), node.id())) {
139                     continue;
140                 }
141                 if (version != -1) {
142                     numberOfAllocationsFound++;
143                     if (highestVersion == -1) {
144                         nodesWithHighestVersion.add(node);
145                         highestVersion = version;
146                     } else {
147                         if (version > highestVersion) {
148                             nodesWithHighestVersion.clear();
149                             nodesWithHighestVersion.add(node);
150                             highestVersion = version;
151                         } else if (version == highestVersion) {
152                             nodesWithHighestVersion.add(node);
153                         }
154                     }
155                 }
156             }
157 
158             // check if the counts meets the minimum set
159             int requiredAllocation = 1;
160             // if we restore from a repository one copy is more then enough
161             if (shard.restoreSource() == null) {
162                 try {
163                     IndexMetaData indexMetaData = routingNodes.metaData().index(shard.index());
164                     String initialShards = indexMetaData.settings().get(INDEX_RECOVERY_INITIAL_SHARDS, settings.get(INDEX_RECOVERY_INITIAL_SHARDS, this.initialShards));
165                     if ("quorum".equals(initialShards)) {
166                         if (indexMetaData.numberOfReplicas() > 1) {
167                             requiredAllocation = ((1 + indexMetaData.numberOfReplicas()) / 2) + 1;
168                         }
169                     } else if ("quorum-1".equals(initialShards) || "half".equals(initialShards)) {
170                         if (indexMetaData.numberOfReplicas() > 2) {
171                             requiredAllocation = ((1 + indexMetaData.numberOfReplicas()) / 2);
172                         }
173                     } else if ("one".equals(initialShards)) {
174                         requiredAllocation = 1;
175                     } else if ("full".equals(initialShards) || "all".equals(initialShards)) {
176                         requiredAllocation = indexMetaData.numberOfReplicas() + 1;
177                     } else if ("full-1".equals(initialShards) || "all-1".equals(initialShards)) {
178                         if (indexMetaData.numberOfReplicas() > 1) {
179                             requiredAllocation = indexMetaData.numberOfReplicas();
180                         }
181                     } else {
182                         requiredAllocation = Integer.parseInt(initialShards);
183                     }
184                 } catch (Exception e) {
185                     logger.warn("[{}][{}] failed to derived initial_shards from value {}, ignore allocation for {}", shard.index(), shard.id(), initialShards, shard);
186                 }
187             }
188 
189             // not enough found for this shard, continue...
190             if (numberOfAllocationsFound < requiredAllocation) {
191                 // if we are restoring this shard we still can allocate
192                 if (shard.restoreSource() == null) {
193                     // we can't really allocate, so ignore it and continue
194                     unassignedIterator.remove();
195                     routingNodes.ignoredUnassigned().add(shard);
196                     if (logger.isDebugEnabled()) {
197                         logger.debug("[{}][{}]: not allocating, number_of_allocated_shards_found [{}], required_number [{}]", shard.index(), shard.id(), numberOfAllocationsFound, requiredAllocation);
198                     }
199                 } else if (logger.isDebugEnabled()) {
200                     logger.debug("[{}][{}]: missing local data, will restore from [{}]", shard.index(), shard.id(), shard.restoreSource());
201                 }
202                 continue;
203             }
204 
205             Set<DiscoveryNode> throttledNodes = Sets.newHashSet();
206             Set<DiscoveryNode> noNodes = Sets.newHashSet();
207             for (DiscoveryNode discoNode : nodesWithHighestVersion) {
208                 RoutingNode node = routingNodes.node(discoNode.id());
209                 if (node == null) {
210                     continue;
211                 }
212 
213                 Decision decision = allocation.deciders().canAllocate(shard, node, allocation);
214                 if (decision.type() == Decision.Type.THROTTLE) {
215                     throttledNodes.add(discoNode);
216                 } else if (decision.type() == Decision.Type.NO) {
217                     noNodes.add(discoNode);
218                 } else {
219                     if (logger.isDebugEnabled()) {
220                         logger.debug("[{}][{}]: allocating [{}] to [{}] on primary allocation", shard.index(), shard.id(), shard, discoNode);
221                     }
222                     // we found a match
223                     changed = true;
224                     // make sure we create one with the version from the recovered state
225                     allocation.routingNodes().assign(new MutableShardRouting(shard, highestVersion), node.nodeId());
226                     unassignedIterator.remove();
227 
228                     // found a node, so no throttling, no "no", and break out of the loop
229                     throttledNodes.clear();
230                     noNodes.clear();
231                     break;
232                 }
233             }
234             if (throttledNodes.isEmpty()) {
235                 // if we have a node that we "can't" allocate to, force allocation, since this is our master data!
236                 if (!noNodes.isEmpty()) {
237                     DiscoveryNode discoNode = noNodes.iterator().next();
238                     RoutingNode node = routingNodes.node(discoNode.id());
239                     if (logger.isDebugEnabled()) {
240                         logger.debug("[{}][{}]: forcing allocating [{}] to [{}] on primary allocation", shard.index(), shard.id(), shard, discoNode);
241                     }
242                     // we found a match
243                     changed = true;
244                     // make sure we create one with the version from the recovered state
245                     allocation.routingNodes().assign(new MutableShardRouting(shard, highestVersion), node.nodeId());
246                     unassignedIterator.remove();
247                 }
248             } else {
249                 if (logger.isDebugEnabled()) {
250                     logger.debug("[{}][{}]: throttling allocation [{}] to [{}] on primary allocation", shard.index(), shard.id(), shard, throttledNodes);
251                 }
252                 // we are throttling this, but we have enough to allocate to this node, ignore it for now
253                 unassignedIterator.remove();
254                 routingNodes.ignoredUnassigned().add(shard);
255             }
256         }
257 
258         if (!routingNodes.hasUnassigned()) {
259             return changed;
260         }
261 
262         // Now, handle replicas, try to assign them to nodes that are similar to the one the primary was allocated on
263         unassignedIterator = routingNodes.unassigned().iterator();
264         while (unassignedIterator.hasNext()) {
265             MutableShardRouting shard = unassignedIterator.next();
266 
267             // pre-check if it can be allocated to any node that currently exists, so we won't list the store for it for nothing
268             boolean canBeAllocatedToAtLeastOneNode = false;
269             for (ObjectCursor<DiscoveryNode> cursor : nodes.dataNodes().values()) {
270                 RoutingNode node = routingNodes.node(cursor.value.id());
271                 if (node == null) {
272                     continue;
273                 }
274                 // if we can't allocate it on a node, ignore it, for example, this handles
275                 // cases for only allocating a replica after a primary
276                 Decision decision = allocation.deciders().canAllocate(shard, node, allocation);
277                 if (decision.type() == Decision.Type.YES) {
278                     canBeAllocatedToAtLeastOneNode = true;
279                     break;
280                 }
281             }
282 
283             if (!canBeAllocatedToAtLeastOneNode) {
284                 continue;
285             }
286 
287             Map<DiscoveryNode, TransportNodesListShardStoreMetaData.StoreFilesMetaData> shardStores = buildShardStores(nodes, shard);
288 
289             long lastSizeMatched = 0;
290             DiscoveryNode lastDiscoNodeMatched = null;
291             RoutingNode lastNodeMatched = null;
292 
293             for (Map.Entry<DiscoveryNode, TransportNodesListShardStoreMetaData.StoreFilesMetaData> nodeStoreEntry : shardStores.entrySet()) {
294                 DiscoveryNode discoNode = nodeStoreEntry.getKey();
295                 TransportNodesListShardStoreMetaData.StoreFilesMetaData storeFilesMetaData = nodeStoreEntry.getValue();
296                 logger.trace("{}: checking node [{}]", shard, discoNode);
297 
298                 if (storeFilesMetaData == null) {
299                     // already allocated on that node...
300                     continue;
301                 }
302 
303                 RoutingNode node = routingNodes.node(discoNode.id());
304                 if (node == null) {
305                     continue;
306                 }
307 
308                 // check if we can allocate on that node...
309                 // we only check for NO, since if this node is THROTTLING and it has enough "same data"
310                 // then we will try and assign it next time
311                 Decision decision = allocation.deciders().canAllocate(shard, node, allocation);
312                 if (decision.type() == Decision.Type.NO) {
313                     continue;
314                 }
315 
316                 // if it is already allocated, we can't assign to it...
317                 if (storeFilesMetaData.allocated()) {
318                     continue;
319                 }
320 
321                 if (!shard.primary()) {
322                     MutableShardRouting primaryShard = routingNodes.activePrimary(shard);
323                     if (primaryShard != null) {
324                         assert primaryShard.active();
325                         DiscoveryNode primaryNode = nodes.get(primaryShard.currentNodeId());
326                         if (primaryNode != null) {
327                             TransportNodesListShardStoreMetaData.StoreFilesMetaData primaryNodeStore = shardStores.get(primaryNode);
328                             if (primaryNodeStore != null && primaryNodeStore.allocated()) {
329                                 long sizeMatched = 0;
330 
331                                 for (StoreFileMetaData storeFileMetaData : storeFilesMetaData) {
332                                     if (primaryNodeStore.fileExists(storeFileMetaData.name()) && primaryNodeStore.file(storeFileMetaData.name()).isSame(storeFileMetaData)) {
333                                         sizeMatched += storeFileMetaData.length();
334                                     }
335                                 }
336                                 logger.trace("{}: node [{}] has [{}/{}] bytes of re-usable data",
337                                         shard, discoNode.name(), new ByteSizeValue(sizeMatched), sizeMatched);
338                                 if (sizeMatched > lastSizeMatched) {
339                                     lastSizeMatched = sizeMatched;
340                                     lastDiscoNodeMatched = discoNode;
341                                     lastNodeMatched = node;
342                                 }
343                             }
344                         }
345                     }
346                 }
347             }
348 
349             if (lastNodeMatched != null) {
350                 // we only check on THROTTLE since we checked before before on NO
351                 Decision decision = allocation.deciders().canAllocate(shard, lastNodeMatched, allocation);
352                 if (decision.type() == Decision.Type.THROTTLE) {
353                     if (logger.isDebugEnabled()) {
354                         logger.debug("[{}][{}]: throttling allocation [{}] to [{}] in order to reuse its unallocated persistent store with total_size [{}]", shard.index(), shard.id(), shard, lastDiscoNodeMatched, new ByteSizeValue(lastSizeMatched));
355                     }
356                     // we are throttling this, but we have enough to allocate to this node, ignore it for now
357                     unassignedIterator.remove();
358                     routingNodes.ignoredUnassigned().add(shard);
359                 } else {
360                     if (logger.isDebugEnabled()) {
361                         logger.debug("[{}][{}]: allocating [{}] to [{}] in order to reuse its unallocated persistent store with total_size [{}]", shard.index(), shard.id(), shard, lastDiscoNodeMatched, new ByteSizeValue(lastSizeMatched));
362                     }
363                     // we found a match
364                     changed = true;
365                     allocation.routingNodes().assign(shard, lastNodeMatched.nodeId());
366                     unassignedIterator.remove();
367                 }
368             }
369         }
370         return changed;
371     }

373     private ObjectLongOpenHashMap<DiscoveryNode> buildShardStates(final DiscoveryNodes nodes, MutableShardRouting shard) {
374         ObjectLongOpenHashMap<DiscoveryNode> shardStates = cachedShardsState.get(shard.shardId());
375         ObjectOpenHashSet<String> nodeIds;
376         if (shardStates == null) {
377             shardStates = new ObjectLongOpenHashMap<>();
378             cachedShardsState.put(shard.shardId(), shardStates);
379             nodeIds = ObjectOpenHashSet.from(nodes.dataNodes().keys());
380         } else {
381             // clean nodes that have failed
382             shardStates.keys().removeAll(new ObjectPredicate<DiscoveryNode>() {
383                 @Override
384                 public boolean apply(DiscoveryNode node) {
385                     return !nodes.nodeExists(node.id());
386                 }
387             });
388             nodeIds = ObjectOpenHashSet.newInstance();
389             // we have stored cached from before, see if the nodes changed, if they have, go fetch again
390             for (ObjectCursor<DiscoveryNode> cursor : nodes.dataNodes().values()) {
391                 DiscoveryNode node = cursor.value;
392                 if (!shardStates.containsKey(node)) {
393                     nodeIds.add(node.id());
394                 }
395             }
396         }
397         if (nodeIds.isEmpty()) {
398             return shardStates;
399         }
400 
401         String[] nodesIdsArray = nodeIds.toArray(String.class);
402         TransportNodesListGatewayStartedShards.NodesGatewayStartedShards response = listGatewayStartedShards.list(shard.shardId(), nodesIdsArray, listTimeout).actionGet();
403         logListActionFailures(shard, "state", response.failures());
404 
405         for (TransportNodesListGatewayStartedShards.NodeGatewayStartedShards nodeShardState : response) {
406             // -1 version means it does not exists, which is what the API returns, and what we expect to
407             logger.trace("[{}] on node [{}] has version [{}] of shard",
408                     shard, nodeShardState.getNode(), nodeShardState.version());
409             shardStates.put(nodeShardState.getNode(), nodeShardState.version());
410         }
411         return shardStates;
412     }

414     private void logListActionFailures(MutableShardRouting shard, String actionType, FailedNodeException[] failures) {
415         for (final FailedNodeException failure : failures) {
416             Throwable cause = ExceptionsHelper.unwrapCause(failure);
417             if (cause instanceof ConnectTransportException) {
418                 continue;
419             }
420             // we log warn here. debug logs with full stack traces will be logged if debug logging is turned on for TransportNodeListGatewayStartedShards
421             logger.warn("{}: failed to list shard {} on node [{}]", failure, shard.shardId(), actionType, failure.nodeId());
422         }
423     }

425     private Map<DiscoveryNode, TransportNodesListShardStoreMetaData.StoreFilesMetaData> buildShardStores(DiscoveryNodes nodes, MutableShardRouting shard) {
426         Map<DiscoveryNode, TransportNodesListShardStoreMetaData.StoreFilesMetaData> shardStores = cachedStores.get(shard.shardId());
427         ObjectOpenHashSet<String> nodesIds;
428         if (shardStores == null) {
429             shardStores = Maps.newHashMap();
430             cachedStores.put(shard.shardId(), shardStores);
431             nodesIds = ObjectOpenHashSet.from(nodes.dataNodes().keys());
432         } else {
433             nodesIds = ObjectOpenHashSet.newInstance();
434             // clean nodes that have failed
435             for (Iterator<DiscoveryNode> it = shardStores.keySet().iterator(); it.hasNext(); ) {
436                 DiscoveryNode node = it.next();
437                 if (!nodes.nodeExists(node.id())) {
438                     it.remove();
439                 }
440             }
441 
442             for (ObjectCursor<DiscoveryNode> cursor : nodes.dataNodes().values()) {
443                 DiscoveryNode node = cursor.value;
444                 if (!shardStores.containsKey(node)) {
445                     nodesIds.add(node.id());
446                 }
447             }
448         }
449 
450         if (!nodesIds.isEmpty()) {
451             String[] nodesIdsArray = nodesIds.toArray(String.class);
452             TransportNodesListShardStoreMetaData.NodesStoreFilesMetaData nodesStoreFilesMetaData = listShardStoreMetaData.list(shard.shardId(), false, nodesIdsArray, listTimeout).actionGet();
453             logListActionFailures(shard, "stores", nodesStoreFilesMetaData.failures());
454 
455             for (TransportNodesListShardStoreMetaData.NodeStoreFilesMetaData nodeStoreFilesMetaData : nodesStoreFilesMetaData) {
456                 if (nodeStoreFilesMetaData.storeFilesMetaData() != null) {
457                     shardStores.put(nodeStoreFilesMetaData.getNode(), nodeStoreFilesMetaData.storeFilesMetaData());
458                 }
459             }
460         }
461 
462         return shardStores;
463     }
