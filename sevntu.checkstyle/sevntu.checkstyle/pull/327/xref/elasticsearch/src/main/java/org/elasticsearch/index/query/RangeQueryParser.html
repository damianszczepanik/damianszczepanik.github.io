20  package org.elasticsearch.index.query;
21  
22  import org.apache.lucene.search.Query;
23  import org.apache.lucene.search.TermRangeQuery;
24  import org.elasticsearch.common.inject.Inject;
25  import org.elasticsearch.common.joda.DateMathParser;
26  import org.elasticsearch.common.joda.Joda;
27  import org.elasticsearch.common.lucene.BytesRefs;
28  import org.elasticsearch.common.xcontent.XContentParser;
29  import org.elasticsearch.index.mapper.FieldMapper;
30  import org.elasticsearch.index.mapper.MapperService;
31  import org.elasticsearch.index.mapper.core.DateFieldMapper;
32  import org.joda.time.DateTimeZone;
33  
34  import java.io.IOException;
35  
36  /**
37   *
38   */
39  public class RangeQueryParser implements QueryParser {
40  
41      public static final String NAME = "range";
42  
43      @Inject
44      public RangeQueryParser() {
45      }
46  
47      @Override
48      public String[] names() {
49          return new String[]{NAME};
50      }
51  
52      @Override
53      public Query parse(QueryParseContext parseContext) throws IOException, QueryParsingException {
54          XContentParser parser = parseContext.parser();
55  
56          XContentParser.Token token = parser.nextToken();
57          if (token != XContentParser.Token.FIELD_NAME) {
58              throw new QueryParsingException(parseContext.index(), "[range] query malformed, no field to indicate field name");
59          }
60          String fieldName = parser.currentName();
61          token = parser.nextToken();
62          if (token != XContentParser.Token.START_OBJECT) {
63              throw new QueryParsingException(parseContext.index(), "[range] query malformed, after field missing start object");
64          }
65  
66          Object from = null;
67          Object to = null;
68          boolean includeLower = true;
69          boolean includeUpper = true;
70          DateTimeZone timeZone = null;
71          DateMathParser forcedDateParser = null;
72          float boost = 1.0f;
73          String queryName = null;
74  
75          String currentFieldName = null;
76          while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
77              if (token == XContentParser.Token.FIELD_NAME) {
78                  currentFieldName = parser.currentName();
79              } else {
80                  if ("from".equals(currentFieldName)) {
81                      from = parser.objectBytes();
82                  } else if ("to".equals(currentFieldName)) {
83                      to = parser.objectBytes();
84                  } else if ("include_lower".equals(currentFieldName) || "includeLower".equals(currentFieldName)) {
85                      includeLower = parser.booleanValue();
86                  } else if ("include_upper".equals(currentFieldName) || "includeUpper".equals(currentFieldName)) {
87                      includeUpper = parser.booleanValue();
88                  } else if ("boost".equals(currentFieldName)) {
89                      boost = parser.floatValue();
90                  } else if ("gt".equals(currentFieldName)) {
91                      from = parser.objectBytes();
92                      includeLower = false;
93                  } else if ("gte".equals(currentFieldName) || "ge".equals(currentFieldName)) {
94                      from = parser.objectBytes();
95                      includeLower = true;
96                  } else if ("lt".equals(currentFieldName)) {
97                      to = parser.objectBytes();
98                      includeUpper = false;
99                  } else if ("lte".equals(currentFieldName) || "le".equals(currentFieldName)) {
100                     to = parser.objectBytes();
101                     includeUpper = true;
102                 } else if ("time_zone".equals(currentFieldName) || "timeZone".equals(currentFieldName)) {
103                     timeZone = DateTimeZone.forID(parser.text());
104                 } else if ("_name".equals(currentFieldName)) {
105                     queryName = parser.text();
106                 } else if ("format".equals(currentFieldName)) {
107                     forcedDateParser = new DateMathParser(Joda.forPattern(parser.text()), DateFieldMapper.Defaults.TIME_UNIT);
108                 } else {
109                     throw new QueryParsingException(parseContext.index(), "[range] query does not support [" + currentFieldName + "]");
110                 }
111             }
112         }
113 
114         // move to the next end object, to close the field name
115         token = parser.nextToken();
116         if (token != XContentParser.Token.END_OBJECT) {
117             throw new QueryParsingException(parseContext.index(), "[range] query malformed, does not end with an object");
118         }
119 
120         Query query = null;
121         MapperService.SmartNameFieldMappers smartNameFieldMappers = parseContext.smartFieldMappers(fieldName);
122         if (smartNameFieldMappers != null) {
123             if (smartNameFieldMappers.hasMapper()) {
124                 FieldMapper mapper = smartNameFieldMappers.mapper();
125                 if (mapper instanceof DateFieldMapper) {
126                     if ((from instanceof Number || to instanceof Number) && timeZone != null) {
127                         throw new QueryParsingException(parseContext.index(), "[range] time_zone when using ms since epoch format as it's UTC based can not be applied to [" + fieldName + "]");
128                     }
129                     query = ((DateFieldMapper) mapper).rangeQuery(from, to, includeLower, includeUpper, timeZone, forcedDateParser, parseContext);
130                 } else  {
131                     if (timeZone != null) {
132                         throw new QueryParsingException(parseContext.index(), "[range] time_zone can not be applied to non date field [" + fieldName + "]");
133                     }
134                     //LUCENE 4 UPGRADE Mapper#rangeQuery should use bytesref as well?
135                     query = mapper.rangeQuery(from, to, includeLower, includeUpper, parseContext);
136                 }
137 
138             }
139         }
140         if (query == null) {
141             query = new TermRangeQuery(fieldName, BytesRefs.toBytesRef(from), BytesRefs.toBytesRef(to), includeLower, includeUpper);
142         }
143         query.setBoost(boost);
144         if (queryName != null) {
145             parseContext.addNamedQuery(queryName, query);
146         }
147         return query;
148     }
149 }
