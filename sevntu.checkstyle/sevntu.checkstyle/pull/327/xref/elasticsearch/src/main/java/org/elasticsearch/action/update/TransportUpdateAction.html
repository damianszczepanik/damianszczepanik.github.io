66  public class TransportUpdateAction extends TransportInstanceSingleOperationAction<UpdateRequest, UpdateResponse> {
67  
68      private final TransportDeleteAction deleteAction;
69      private final TransportIndexAction indexAction;
70      private final AutoCreateIndex autoCreateIndex;
71      private final TransportCreateIndexAction createIndexAction;
72      private final UpdateHelper updateHelper;
73      private final IndicesService indicesService;
74  
75      @Inject
76      public TransportUpdateAction(Settings settings, ThreadPool threadPool, ClusterService clusterService, TransportService transportService,
77                                   TransportIndexAction indexAction, TransportDeleteAction deleteAction, TransportCreateIndexAction createIndexAction,
78                                   UpdateHelper updateHelper, ActionFilters actionFilters, IndicesService indicesService) {
79          super(settings, UpdateAction.NAME, threadPool, clusterService, transportService, actionFilters);
80          this.indexAction = indexAction;
81          this.deleteAction = deleteAction;
82          this.createIndexAction = createIndexAction;
83          this.updateHelper = updateHelper;
84          this.indicesService = indicesService;
85          this.autoCreateIndex = new AutoCreateIndex(settings);
86      }
87  
88      @Override
89      protected String executor() {
90          return ThreadPool.Names.INDEX;
91      }
92  
93      @Override
94      protected UpdateRequest newRequest() {
95          return new UpdateRequest();
96      }
97  
98      @Override
99      protected UpdateResponse newResponse() {
100         return new UpdateResponse();
101     }
102 
103     @Override
104     protected boolean retryOnFailure(Throwable e) {
105         return TransportActions.isShardNotAvailableException(e);
106     }
107 
108     @Override
109     protected boolean resolveRequest(ClusterState state, InternalRequest request, ActionListener<UpdateResponse> listener) {
110         request.request().routing((state.metaData().resolveIndexRouting(request.request().routing(), request.request().index())));
111         // Fail fast on the node that received the request, rather than failing when translating on the index or delete request.
112         if (request.request().routing() == null && state.getMetaData().routingRequired(request.concreteIndex(), request.request().type())) {
113             throw new RoutingMissingException(request.concreteIndex(), request.request().type(), request.request().id());
114         }
115         return true;
116     }
117 
118     @Override
119     protected void doExecute(final UpdateRequest request, final ActionListener<UpdateResponse> listener) {
120         // if we don't have a master, we don't have metadata, that's fine, let it find a master using create index API
121         if (autoCreateIndex.shouldAutoCreate(request.index(), clusterService.state())) {
122             request.beforeLocalFork(); // we fork on another thread...
123             createIndexAction.execute(new CreateIndexRequest(request).index(request.index()).cause("auto(update api)").masterNodeTimeout(request.timeout()), new ActionListener<CreateIndexResponse>() {
124                 @Override
125                 public void onResponse(CreateIndexResponse result) {
126                     innerExecute(request, listener);
127                 }
128 
129                 @Override
130                 public void onFailure(Throwable e) {
131                     if (ExceptionsHelper.unwrapCause(e) instanceof IndexAlreadyExistsException) {
132                         // we have the index, do it
133                         try {
134                             innerExecute(request, listener);
135                         } catch (Throwable e1) {
136                             listener.onFailure(e1);
137                         }
138                     } else {
139                         listener.onFailure(e);
140                     }
141                 }
142             });
143         } else {
144             innerExecute(request, listener);
145         }
146     }
147 
148     private void innerExecute(final UpdateRequest request, final ActionListener<UpdateResponse> listener) {
149         super.doExecute(request, listener);
150     }
151 
152     @Override
153     protected ShardIterator shards(ClusterState clusterState, InternalRequest request) throws ElasticsearchException {
154         if (request.request().shardId() != -1) {
155             return clusterState.routingTable().index(request.concreteIndex()).shard(request.request().shardId()).primaryShardIt();
156         }
157         ShardIterator shardIterator = clusterService.operationRouting()
158                 .indexShards(clusterState, request.concreteIndex(), request.request().type(), request.request().id(), request.request().routing());
159         ShardRouting shard;
160         while ((shard = shardIterator.nextOrNull()) != null) {
161             if (shard.primary()) {
162                 return new PlainShardIterator(shardIterator.shardId(), ImmutableList.of(shard));
163             }
164         }
165         return new PlainShardIterator(shardIterator.shardId(), ImmutableList.<ShardRouting>of());
166     }
167 
168     @Override
169     protected void shardOperation(final InternalRequest request, final ActionListener<UpdateResponse> listener) throws ElasticsearchException {
170         shardOperation(request, listener, 0);
171     }
172 
173     protected void shardOperation(final InternalRequest request, final ActionListener<UpdateResponse> listener, final int retryCount) throws ElasticsearchException {
174         IndexService indexService = indicesService.indexServiceSafe(request.concreteIndex());
175         IndexShard indexShard = indexService.shardSafe(request.request().shardId());
176         final UpdateHelper.Result result = updateHelper.prepare(request.request(), indexShard);
177         switch (result.operation()) {
178             case UPSERT:
179                 IndexRequest upsertRequest = new IndexRequest((IndexRequest)result.action(), request.request());
180                 // we fetch it from the index request so we don't generate the bytes twice, its already done in the index request
181                 final BytesReference upsertSourceBytes = upsertRequest.source();
182                 indexAction.execute(upsertRequest, new ActionListener<IndexResponse>() {
183                     @Override
184                     public void onResponse(IndexResponse response) {
185                         UpdateResponse update = new UpdateResponse(response.getShardInfo(), response.getIndex(), response.getType(), response.getId(), response.getVersion(), response.isCreated());
186                         if (request.request().fields() != null && request.request().fields().length > 0) {
187                             Tuple<XContentType, Map<String, Object>> sourceAndContent = XContentHelper.convertToMap(upsertSourceBytes, true);
188                             update.setGetResult(updateHelper.extractGetResult(request.request(), request.concreteIndex(), response.getVersion(), sourceAndContent.v2(), sourceAndContent.v1(), upsertSourceBytes));
189                         } else {
190                             update.setGetResult(null);
191                         }
192                         listener.onResponse(update);
193                     }
194 
195                     @Override
196                     public void onFailure(Throwable e) {
197                         e = ExceptionsHelper.unwrapCause(e);
198                         if (e instanceof VersionConflictEngineException || e instanceof DocumentAlreadyExistsException) {
199                             if (retryCount < request.request().retryOnConflict()) {
200                                 threadPool.executor(executor()).execute(new ActionRunnable<UpdateResponse>(listener) {
201                                     @Override
202                                     protected void doRun() {
203                                         shardOperation(request, listener, retryCount + 1);
204                                     }
205                                 });
206                                 return;
207                             }
208                         }
209                         listener.onFailure(e);
210                     }
211                 });
212                 break;
213             case INDEX:
214                 IndexRequest indexRequest = new IndexRequest((IndexRequest)result.action(), request.request());
215                 // we fetch it from the index request so we don't generate the bytes twice, its already done in the index request
216                 final BytesReference indexSourceBytes = indexRequest.source();
217                 indexAction.execute(indexRequest, new ActionListener<IndexResponse>() {
218                     @Override
219                     public void onResponse(IndexResponse response) {
220                         UpdateResponse update = new UpdateResponse(response.getShardInfo(), response.getIndex(), response.getType(), response.getId(), response.getVersion(), response.isCreated());
221                         update.setGetResult(updateHelper.extractGetResult(request.request(), request.concreteIndex(), response.getVersion(), result.updatedSourceAsMap(), result.updateSourceContentType(), indexSourceBytes));
222                         listener.onResponse(update);
223                     }
224 
225                     @Override
226                     public void onFailure(Throwable e) {
227                         e = ExceptionsHelper.unwrapCause(e);
228                         if (e instanceof VersionConflictEngineException) {
229                             if (retryCount < request.request().retryOnConflict()) {
230                                 threadPool.executor(executor()).execute(new ActionRunnable<UpdateResponse>(listener) {
231                                     @Override
232                                     protected void doRun() {
233                                         shardOperation(request, listener, retryCount + 1);
234                                     }
235                                 });
236                                 return;
237                             }
238                         }
239                         listener.onFailure(e);
240                     }
241                 });
242                 break;
243             case DELETE:
244                 DeleteRequest deleteRequest = new DeleteRequest((DeleteRequest)result.action(), request.request());
245                 deleteAction.execute(deleteRequest, new ActionListener<DeleteResponse>() {
246                     @Override
247                     public void onResponse(DeleteResponse response) {
248                         UpdateResponse update = new UpdateResponse(response.getShardInfo(), response.getIndex(), response.getType(), response.getId(), response.getVersion(), false);
249                         update.setGetResult(updateHelper.extractGetResult(request.request(), request.concreteIndex(), response.getVersion(), result.updatedSourceAsMap(), result.updateSourceContentType(), null));
250                         listener.onResponse(update);
251                     }
252 
253                     @Override
254                     public void onFailure(Throwable e) {
255                         e = ExceptionsHelper.unwrapCause(e);
256                         if (e instanceof VersionConflictEngineException) {
257                             if (retryCount < request.request().retryOnConflict()) {
258                                 threadPool.executor(executor()).execute(new ActionRunnable<UpdateResponse>(listener) {
259                                     @Override
260                                     protected void doRun() {
261                                         shardOperation(request, listener, retryCount + 1);
262                                     }
263                                 });
264                                 return;
265                             }
266                         }
267                         listener.onFailure(e);
268                     }
269                 });
270                 break;
271             case NONE:
272                 UpdateResponse update = result.action();
273                 IndexService indexServiceOrNull = indicesService.indexService(request.concreteIndex());
274                 if (indexServiceOrNull !=  null) {
275                     IndexShard shard = indexService.shard(request.request().shardId());
276                     if (shard != null) {
277                         shard.indexingService().noopUpdate(request.request().type());
278                     }
279                 }
280                 listener.onResponse(update);
281                 break;
282             default:
283                 throw new ElasticsearchIllegalStateException("Illegal operation " + result.operation());
284         }
285     }
