109 /**
110  * A node represent a node within a cluster (<tt>cluster.name</tt>). The {@link #client()} can be used
111  * in order to use a {@link Client} to perform actions/operations against the cluster.
112  * <p/>
113  * <p>In order to create a node, the {@link NodeBuilder} can be used. When done with it, make sure to
114  * call {@link #close()} on it.
115  */
116 public class Node implements Releasable {
117 
118     private static final String CLIENT_TYPE = "node";
119     public static final String HTTP_ENABLED = "http.enabled";
120     private final Lifecycle lifecycle = new Lifecycle();
121     private final Injector injector;
122     private final Settings settings;
123     private final Environment environment;
124     private final PluginsService pluginsService;
125     private final Client client;
126 
127     public Node() throws ElasticsearchException {
128         this(ImmutableSettings.Builder.EMPTY_SETTINGS, true);
129     }
130 
131     public Node(Settings preparedSettings, boolean loadConfigSettings) throws ElasticsearchException {
132         final Settings pSettings = settingsBuilder().put(preparedSettings)
133                 .put(Client.CLIENT_TYPE_SETTING, CLIENT_TYPE).build();
134         Tuple<Settings, Environment> tuple = InternalSettingsPreparer.prepareSettings(pSettings, loadConfigSettings);
135         tuple = new Tuple<>(TribeService.processSettings(tuple.v1()), tuple.v2());
136 
137         // The only place we can actually fake the version a node is running on:
138         Version version = pSettings.getAsVersion("tests.mock.version", Version.CURRENT);
139 
140         ESLogger logger = Loggers.getLogger(Node.class, tuple.v1().get("name"));
141         logger.info("version[{}], pid[{}], build[{}/{}]", version, JvmInfo.jvmInfo().pid(), Build.CURRENT.hashShort(), Build.CURRENT.timestamp());
142 
143         logger.info("initializing ...");
144 
145         if (logger.isDebugEnabled()) {
146             Environment env = tuple.v2();
147             logger.debug("using home [{}], config [{}], data [{}], logs [{}], work [{}], plugins [{}]",
148                     env.homeFile(), env.configFile(), Arrays.toString(env.dataFiles()), env.logsFile(),
149                     env.workFile(), env.pluginsFile());
150         }
151 
152         this.pluginsService = new PluginsService(tuple.v1(), tuple.v2());
153         this.settings = pluginsService.updatedSettings();
154         // create the environment based on the finalized (processed) view of the settings
155         this.environment = new Environment(this.settings());
156 
157         CompressorFactory.configure(settings);
158         final NodeEnvironment nodeEnvironment;
159         try {
160             nodeEnvironment = new NodeEnvironment(this.settings, this.environment);
161         } catch (IOException ex) {
162             throw new ElasticsearchIllegalStateException("Failed to created node environment", ex);
163         }
164 
165         boolean success = false;
166         try {
167             ModulesBuilder modules = new ModulesBuilder();
168             modules.add(new Version.Module(version));
169             modules.add(new PageCacheRecyclerModule(settings));
170             modules.add(new CircuitBreakerModule(settings));
171             modules.add(new BigArraysModule(settings));
172             modules.add(new PluginsModule(settings, pluginsService));
173             modules.add(new SettingsModule(settings));
174             modules.add(new NodeModule(this));
175             modules.add(new NetworkModule());
176             modules.add(new ScriptModule(settings));
177             modules.add(new EnvironmentModule(environment));
178             modules.add(new NodeEnvironmentModule(nodeEnvironment));
179             modules.add(new ClusterNameModule(settings));
180             modules.add(new ThreadPoolModule(settings));
181             modules.add(new DiscoveryModule(settings));
182             modules.add(new ClusterModule(settings));
183             modules.add(new RestModule(settings));
184             modules.add(new TransportModule(settings));
185             if (settings.getAsBoolean(HTTP_ENABLED, true)) {
186                 modules.add(new HttpServerModule(settings));
187             }
188             modules.add(new RiversModule(settings));
189             modules.add(new IndicesModule(settings));
190             modules.add(new SearchModule());
191             modules.add(new ActionModule(false));
192             modules.add(new MonitorModule(settings));
193             modules.add(new GatewayModule());
194             modules.add(new NodeClientModule());
195             modules.add(new ShapeModule());
196             modules.add(new PercolatorModule());
197             modules.add(new ResourceWatcherModule());
198             modules.add(new RepositoriesModule());
199             modules.add(new TribeModule());
200             modules.add(new BenchmarkModule(settings));
201 
202             injector = modules.createInjector();
203 
204             client = injector.getInstance(Client.class);
205             success = true;
206         } finally {
207             if (!success) {
208                 nodeEnvironment.close();
209             }
210         }
211 
212         logger.info("initialized");
213     }

229     /**
230      * Start the node. If the node is already started, this method is no-op.
231      */
232     public Node start() {
233         if (!lifecycle.moveToStarted()) {
234             return this;
235         }
236 
237         ESLogger logger = Loggers.getLogger(Node.class, settings.get("name"));
238         logger.info("starting ...");
239 
240         // hack around dependency injection problem (for now...)
241         injector.getInstance(Discovery.class).setAllocationService(injector.getInstance(AllocationService.class));
242 
243         for (Class<? extends LifecycleComponent> plugin : pluginsService.services()) {
244             injector.getInstance(plugin).start();
245         }
246 
247         injector.getInstance(MappingUpdatedAction.class).start();
248         injector.getInstance(IndicesService.class).start();
249         injector.getInstance(IndexingMemoryController.class).start();
250         injector.getInstance(IndicesClusterStateService.class).start();
251         injector.getInstance(IndicesTTLService.class).start();
252         injector.getInstance(RiversManager.class).start();
253         injector.getInstance(SnapshotsService.class).start();
254         injector.getInstance(TransportService.class).start();
255         injector.getInstance(ClusterService.class).start();
256         injector.getInstance(RoutingService.class).start();
257         injector.getInstance(SearchService.class).start();
258         injector.getInstance(MonitorService.class).start();
259         injector.getInstance(RestController.class).start();
260         DiscoveryService discoService = injector.getInstance(DiscoveryService.class).start();
261         discoService.waitForInitialState();
262 
263         // gateway should start after disco, so it can try and recovery from gateway on "start"
264         injector.getInstance(GatewayService.class).start();
265 
266         if (settings.getAsBoolean("http.enabled", true)) {
267             injector.getInstance(HttpServer.class).start();
268         }
269         injector.getInstance(ResourceWatcherService.class).start();
270         injector.getInstance(TribeService.class).start();
271 
272         logger.info("started");
273 
274         return this;
275     }
276 
277     private Node stop() {
278         if (!lifecycle.moveToStopped()) {
279             return this;
280         }
281         ESLogger logger = Loggers.getLogger(Node.class, settings.get("name"));
282         logger.info("stopping ...");
283 
284         injector.getInstance(TribeService.class).stop();
285         injector.getInstance(ResourceWatcherService.class).stop();
286         if (settings.getAsBoolean("http.enabled", true)) {
287             injector.getInstance(HttpServer.class).stop();
288         }
289 
290         injector.getInstance(MappingUpdatedAction.class).stop();
291         injector.getInstance(RiversManager.class).stop();
292 
293         injector.getInstance(SnapshotsService.class).stop();
294         // stop any changes happening as a result of cluster state changes
295         injector.getInstance(IndicesClusterStateService.class).stop();
296         // we close indices first, so operations won't be allowed on it
297         injector.getInstance(IndexingMemoryController.class).stop();
298         injector.getInstance(IndicesTTLService.class).stop();
299         injector.getInstance(RoutingService.class).stop();
300         injector.getInstance(ClusterService.class).stop();
301         injector.getInstance(DiscoveryService.class).stop();
302         injector.getInstance(MonitorService.class).stop();
303         injector.getInstance(GatewayService.class).stop();
304         injector.getInstance(SearchService.class).stop();
305         injector.getInstance(RestController.class).stop();
306         injector.getInstance(TransportService.class).stop();
307 
308         for (Class<? extends LifecycleComponent> plugin : pluginsService.services()) {
309             injector.getInstance(plugin).stop();
310         }
311         // we should stop this last since it waits for resources to get released
312         // if we had scroll searchers etc or recovery going on we wait for to finish.
313         injector.getInstance(IndicesService.class).stop();
314         logger.info("stopped");
315 
316         return this;
317     }
318 
319     // During concurrent close() calls we want to make sure that all of them return after the node has completed it's shutdown cycle.
320     // If not, the hook that is added in Bootstrap#setup() will be useless: close() might not be executed, in case another (for example api) call
321     // to close() has already set some lifecycles to stopped. In this case the process will be terminated even if the first call to close() has not finished yet.
322     @Override
323     public synchronized void close() {
324         if (lifecycle.started()) {
325             stop();
326         }
327         if (!lifecycle.moveToClosed()) {
328             return;
329         }
330 
331         ESLogger logger = Loggers.getLogger(Node.class, settings.get("name"));
332         logger.info("closing ...");
333 
334         StopWatch stopWatch = new StopWatch("node_close");
335         stopWatch.start("tribe");
336         injector.getInstance(TribeService.class).close();
337         stopWatch.stop().start("http");
338         if (settings.getAsBoolean("http.enabled", true)) {
339             injector.getInstance(HttpServer.class).close();
340         }
341 
342         stopWatch.stop().start("rivers");
343         injector.getInstance(RiversManager.class).close();
344 
345         stopWatch.stop().start("snapshot_service");
346         injector.getInstance(SnapshotsService.class).close();
347         stopWatch.stop().start("client");
348         Releasables.close(injector.getInstance(Client.class));
349         stopWatch.stop().start("indices_cluster");
350         injector.getInstance(IndicesClusterStateService.class).close();
351         stopWatch.stop().start("indices");
352         injector.getInstance(IndicesFilterCache.class).close();
353         injector.getInstance(IndicesFieldDataCache.class).close();
354         injector.getInstance(IndexingMemoryController.class).close();
355         injector.getInstance(IndicesTTLService.class).close();
356         injector.getInstance(IndicesService.class).close();
357         injector.getInstance(IndicesStore.class).close();
358         stopWatch.stop().start("routing");
359         injector.getInstance(RoutingService.class).close();
360         stopWatch.stop().start("cluster");
361         injector.getInstance(ClusterService.class).close();
362         stopWatch.stop().start("discovery");
363         injector.getInstance(DiscoveryService.class).close();
364         stopWatch.stop().start("monitor");
365         injector.getInstance(MonitorService.class).close();
366         stopWatch.stop().start("gateway");
367         injector.getInstance(GatewayService.class).close();
368         stopWatch.stop().start("search");
369         injector.getInstance(SearchService.class).close();
370         stopWatch.stop().start("rest");
371         injector.getInstance(RestController.class).close();
372         stopWatch.stop().start("transport");
373         injector.getInstance(TransportService.class).close();
374         stopWatch.stop().start("percolator_service");
375         injector.getInstance(PercolatorService.class).close();
376 
377         for (Class<? extends LifecycleComponent> plugin : pluginsService.services()) {
378             stopWatch.stop().start("plugin(" + plugin.getName() + ")");
379             injector.getInstance(plugin).close();
380         }
381 
382         stopWatch.stop().start("script");
383         injector.getInstance(ScriptService.class).close();
384 
385         stopWatch.stop().start("thread_pool");
386         // TODO this should really use ThreadPool.terminate()
387         injector.getInstance(ThreadPool.class).shutdown();
388         try {
389             injector.getInstance(ThreadPool.class).awaitTermination(10, TimeUnit.SECONDS);
390         } catch (InterruptedException e) {
391             // ignore
392         }
393         stopWatch.stop().start("thread_pool_force_shutdown");
394         try {
395             injector.getInstance(ThreadPool.class).shutdownNow();
396         } catch (Exception e) {
397             // ignore
398         }
399         stopWatch.stop();
400 
401         if (logger.isTraceEnabled()) {
402             logger.trace("Close times for each service:\n{}", stopWatch.prettyPrint());
403         }
404 
405         injector.getInstance(NodeEnvironment.class).close();
406         injector.getInstance(PageCacheRecycler.class).close();
407 
408         logger.info("closed");
409     }

412     /**
413      * Returns <tt>true</tt> if the node is closed.
414      */
415     public boolean isClosed() {
416         return lifecycle.closed();
417     }

420         return this.injector;

423     public static void main(String[] args) throws Exception {
424         final Node node = new Node();
425         node.start();
426         Runtime.getRuntime().addShutdownHook(new Thread() {
427             @Override
428             public void run() {
429                 node.close();
430             }
431         });
432     }
