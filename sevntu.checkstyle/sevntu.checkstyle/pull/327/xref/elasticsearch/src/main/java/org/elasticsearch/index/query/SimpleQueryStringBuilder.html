20  package org.elasticsearch.index.query;
21  
22  import org.elasticsearch.common.xcontent.XContentBuilder;
23  
24  import java.io.IOException;
25  import java.util.HashMap;
26  import java.util.Locale;
27  import java.util.Map;
28  
29  /**
30   * SimpleQuery is a query parser that acts similar to a query_string
31   * query, but won't throw exceptions for any weird string syntax.
32   */
33  public class SimpleQueryStringBuilder extends BaseQueryBuilder {
34      private Map<String, Float> fields = new HashMap<>();
35      private String analyzer;
36      private Operator operator;
37      private final String queryText;
38      private String queryName;
39      private String minimumShouldMatch;
40      private int flags = -1;
41      private Boolean lowercaseExpandedTerms;
42      private Boolean lenient;
43      private Boolean analyzeWildcard;
44      private Locale locale;
45  
46      /**
47       * Operators for the default_operator
48       */
49      public static enum Operator {
50          AND,
51          OR
52      }
53  
54      /**
55       * Construct a new simple query with the given text
56       */
57      public SimpleQueryStringBuilder(String text) {
58          this.queryText = text;
59      }
60  
61      /**
62       * Add a field to run the query against
63       */
64      public SimpleQueryStringBuilder field(String field) {
65          this.fields.put(field, null);
66          return this;
67      }
68  
69      /**
70       * Add a field to run the query against with a specific boost
71       */
72      public SimpleQueryStringBuilder field(String field, float boost) {
73          this.fields.put(field, boost);
74          return this;
75      }
76  
77      /**
78       * Specify a name for the query
79       */
80      public SimpleQueryStringBuilder queryName(String name) {
81          this.queryName = name;
82          return this;
83      }
84  
85      /**
86       * Specify an analyzer to use for the query
87       */
88      public SimpleQueryStringBuilder analyzer(String analyzer) {
89          this.analyzer = analyzer;
90          return this;
91      }
92  
93      /**
94       * Specify the default operator for the query. Defaults to "OR" if no
95       * operator is specified
96       */
97      public SimpleQueryStringBuilder defaultOperator(Operator defaultOperator) {
98          this.operator = defaultOperator;
99          return this;
100     }
101 
102     /**
103      * Specify the enabled features of the SimpleQueryString.
104      */
105     public SimpleQueryStringBuilder flags(SimpleQueryStringFlag... flags) {
106         int value = 0;
107         if (flags.length == 0) {
108             value = SimpleQueryStringFlag.ALL.value;
109         } else {
110             for (SimpleQueryStringFlag flag : flags) {
111                 value |= flag.value;
112             }
113         }
114         this.flags = value;
115         return this;
116     }
117 
118     public SimpleQueryStringBuilder lowercaseExpandedTerms(boolean lowercaseExpandedTerms) {
119         this.lowercaseExpandedTerms = lowercaseExpandedTerms;
120         return this;
121     }
122 
123     public SimpleQueryStringBuilder locale(Locale locale) {
124         this.locale = locale;
125         return this;
126     }
127 
128     public SimpleQueryStringBuilder lenient(boolean lenient) {
129         this.lenient = lenient;
130         return this;
131     }
132 
133     public SimpleQueryStringBuilder analyzeWildcard(boolean analyzeWildcard) {
134         this.analyzeWildcard = analyzeWildcard;
135         return this;
136     }
137 
138     public SimpleQueryStringBuilder minimumShouldMatch(String minimumShouldMatch) {
139         this.minimumShouldMatch = minimumShouldMatch;
140         return this;
141     }
142 
143     @Override
144     public void doXContent(XContentBuilder builder, Params params) throws IOException {
145         builder.startObject(SimpleQueryStringParser.NAME);
146 
147         builder.field("query", queryText);
148 
149         if (fields.size() > 0) {
150             builder.startArray("fields");
151             for (Map.Entry<String, Float> entry : fields.entrySet()) {
152                 String field = entry.getKey();
153                 Float boost = entry.getValue();
154                 if (boost != null) {
155                     builder.value(field + "^" + boost);
156                 } else {
157                     builder.value(field);
158                 }
159             }
160             builder.endArray();
161         }
162 
163         if (flags != -1) {
164             builder.field("flags", flags);
165         }
166 
167         if (analyzer != null) {
168             builder.field("analyzer", analyzer);
169         }
170 
171         if (operator != null) {
172             builder.field("default_operator", operator.name().toLowerCase(Locale.ROOT));
173         }
174 
175         if (lowercaseExpandedTerms != null) {
176             builder.field("lowercase_expanded_terms", lowercaseExpandedTerms);
177         }
178 
179         if (lenient != null) {
180             builder.field("lenient", lenient);
181         }
182 
183         if (analyzeWildcard != null) {
184             builder.field("analyze_wildcard", analyzeWildcard);
185         }
186 
187         if (locale != null) {
188             builder.field("locale", locale.toString());
189         }
190 
191         if (queryName != null) {
192             builder.field("_name", queryName);
193         }
194 
195         if (minimumShouldMatch != null) {
196             builder.field("minimum_should_match", minimumShouldMatch);
197         }
198 
199         builder.endObject();
200     }
201 }
