20  package org.elasticsearch.gateway;
21  
22  import com.google.common.collect.Maps;
23  import org.elasticsearch.cluster.metadata.IndexMetaData;
24  import org.elasticsearch.cluster.metadata.MetaData;
25  import org.elasticsearch.common.Nullable;
26  import org.elasticsearch.common.component.AbstractComponent;
27  import org.elasticsearch.common.inject.Inject;
28  import org.elasticsearch.common.settings.Settings;
29  import org.elasticsearch.common.xcontent.ToXContent;
30  import org.elasticsearch.common.xcontent.XContentBuilder;
31  import org.elasticsearch.common.xcontent.XContentParser;
32  import org.elasticsearch.common.xcontent.XContentType;
33  import org.elasticsearch.env.NodeEnvironment;
34  import org.elasticsearch.index.Index;
35  
36  import java.io.IOException;
37  import java.util.Map;
38  import java.util.Set;
39  import java.util.regex.Pattern;
40  
41  /**
42   * Handles writing and loading both {@link MetaData} and {@link IndexMetaData}
43   */
44  public class MetaStateService extends AbstractComponent {
45  
46      static final String FORMAT_SETTING = "gateway.format";
47  
48      static final String GLOBAL_STATE_FILE_PREFIX = "global-";
49      private static final String INDEX_STATE_FILE_PREFIX = "state-";
50      static final Pattern GLOBAL_STATE_FILE_PATTERN = Pattern.compile(GLOBAL_STATE_FILE_PREFIX + "(\\d+)(" + MetaDataStateFormat.STATE_FILE_EXTENSION + ")?");
51      static final Pattern INDEX_STATE_FILE_PATTERN = Pattern.compile(INDEX_STATE_FILE_PREFIX + "(\\d+)(" + MetaDataStateFormat.STATE_FILE_EXTENSION + ")?");
52      private static final String GLOBAL_STATE_LOG_TYPE = "[_global]";
53  
54      private final NodeEnvironment nodeEnv;
55  
56      private final XContentType format;
57      private final ToXContent.Params formatParams;
58      private final ToXContent.Params gatewayModeFormatParams;
59  
60      @Inject
61      public MetaStateService(Settings settings, NodeEnvironment nodeEnv) {
62          super(settings);
63          this.nodeEnv = nodeEnv;
64          this.format = XContentType.fromRestContentType(settings.get(FORMAT_SETTING, "smile"));
65  
66          if (this.format == XContentType.SMILE) {
67              Map<String, String> params = Maps.newHashMap();
68              params.put("binary", "true");
69              formatParams = new ToXContent.MapParams(params);
70              Map<String, String> gatewayModeParams = Maps.newHashMap();
71              gatewayModeParams.put("binary", "true");
72              gatewayModeParams.put(MetaData.CONTEXT_MODE_PARAM, MetaData.CONTEXT_MODE_GATEWAY);
73              gatewayModeFormatParams = new ToXContent.MapParams(gatewayModeParams);
74          } else {
75              formatParams = ToXContent.EMPTY_PARAMS;
76              Map<String, String> gatewayModeParams = Maps.newHashMap();
77              gatewayModeParams.put(MetaData.CONTEXT_MODE_PARAM, MetaData.CONTEXT_MODE_GATEWAY);
78              gatewayModeFormatParams = new ToXContent.MapParams(gatewayModeParams);
79          }
80      }
81  
82      /**
83       * Loads the full state, which includes both the global state and all the indices
84       * meta state.
85       */
86      MetaData loadFullState() throws Exception {
87          MetaData globalMetaData = loadGlobalState();
88          MetaData.Builder metaDataBuilder;
89          if (globalMetaData != null) {
90              metaDataBuilder = MetaData.builder(globalMetaData);
91          } else {
92              metaDataBuilder = MetaData.builder();
93          }
94  
95          final Set<String> indices = nodeEnv.findAllIndices();
96          for (String index : indices) {
97              IndexMetaData indexMetaData = loadIndexState(index);
98              if (indexMetaData == null) {
99                  logger.debug("[{}] failed to find metadata for existing index location", index);
100             } else {
101                 metaDataBuilder.put(indexMetaData, false);
102             }
103         }
104         return metaDataBuilder.build();
105     }
106 
107     /**
108      * Loads the index state for the provided index name, returning null if doesn't exists.
109      */
110     @Nullable
111     IndexMetaData loadIndexState(String index) throws IOException {
112         return MetaDataStateFormat.loadLatestState(logger, indexStateFormat(format, formatParams, true),
113                 INDEX_STATE_FILE_PATTERN, "[" + index + "]", nodeEnv.indexPaths(new Index(index)));
114     }
115 
116     /**
117      * Loads the global state, *without* index state, see {@link #loadFullState()} for that.
118      */
119     MetaData loadGlobalState() throws IOException {
120         return MetaDataStateFormat.loadLatestState(logger, globalStateFormat(format, gatewayModeFormatParams, true), GLOBAL_STATE_FILE_PATTERN, GLOBAL_STATE_LOG_TYPE, nodeEnv.nodeDataPaths());
121     }
122 
123     /**
124      * Writes the index state.
125      */
126     void writeIndex(String reason, IndexMetaData indexMetaData, @Nullable IndexMetaData previousIndexMetaData) throws Exception {
127         logger.trace("[{}] writing state, reason [{}]", indexMetaData.index(), reason);
128         final boolean deleteOldFiles = previousIndexMetaData != null && previousIndexMetaData.version() != indexMetaData.version();
129         final MetaDataStateFormat<IndexMetaData> writer = indexStateFormat(format, formatParams, deleteOldFiles);
130         try {
131             writer.write(indexMetaData, INDEX_STATE_FILE_PREFIX, indexMetaData.version(),
132                     nodeEnv.indexPaths(new Index(indexMetaData.index())));
133         } catch (Throwable ex) {
134             logger.warn("[{}]: failed to write index state", ex, indexMetaData.index());
135             throw new IOException("failed to write state for [" + indexMetaData.index() + "]", ex);
136         }
137     }
138 
139     /**
140      * Writes the global state, *without* the indices states.
141      */
142     void writeGlobalState(String reason, MetaData metaData) throws Exception {
143         logger.trace("{} writing state, reason [{}]", GLOBAL_STATE_LOG_TYPE, reason);
144         final MetaDataStateFormat<MetaData> writer = globalStateFormat(format, gatewayModeFormatParams, true);
145         try {
146             writer.write(metaData, GLOBAL_STATE_FILE_PREFIX, metaData.version(), nodeEnv.nodeDataPaths());
147         } catch (Throwable ex) {
148             logger.warn("{}: failed to write global state", ex, GLOBAL_STATE_LOG_TYPE);
149             throw new IOException("failed to write global state", ex);
150         }
151     }
152 
153     /**
154      * Returns a StateFormat that can read and write {@link MetaData}
155      */
156     static MetaDataStateFormat<MetaData> globalStateFormat(XContentType format, final ToXContent.Params formatParams, final boolean deleteOldFiles) {
157         return new MetaDataStateFormat<MetaData>(format, deleteOldFiles) {
158 
159             @Override
160             public void toXContent(XContentBuilder builder, MetaData state) throws IOException {
161                 MetaData.Builder.toXContent(state, builder, formatParams);
162             }
163 
164             @Override
165             public MetaData fromXContent(XContentParser parser) throws IOException {
166                 return MetaData.Builder.fromXContent(parser);
167             }
168         };
169     }
170 
171     /**
172      * Returns a StateFormat that can read and write {@link IndexMetaData}
173      */
174     static MetaDataStateFormat<IndexMetaData> indexStateFormat(XContentType format, final ToXContent.Params formatParams, boolean deleteOldFiles) {
175         return new MetaDataStateFormat<IndexMetaData>(format, deleteOldFiles) {
176 
177             @Override
178             public void toXContent(XContentBuilder builder, IndexMetaData state) throws IOException {
179                 IndexMetaData.Builder.toXContent(state, builder, formatParams);            }
180 
181             @Override
182             public IndexMetaData fromXContent(XContentParser parser) throws IOException {
183                 return IndexMetaData.Builder.fromXContent(parser);
184             }
185         };
186     }
