67  public class IndexShardGateway extends AbstractIndexShardComponent implements Closeable {

69      private final ThreadPool threadPool;
70      private final MappingUpdatedAction mappingUpdatedAction;
71      private final IndexService indexService;
72      private final IndexShard indexShard;
73      private final TimeValue waitForMappingUpdatePostRecovery;
74      private final TimeValue syncInterval;
75  
76      private volatile ScheduledFuture flushScheduler;
77      private final CancellableThreads cancellableThreads = new CancellableThreads();
78  
79  
80      @Inject
81      public IndexShardGateway(ShardId shardId, @IndexSettings Settings indexSettings, ThreadPool threadPool, MappingUpdatedAction mappingUpdatedAction,
82                               IndexService indexService, IndexShard indexShard) {
83          super(shardId, indexSettings);
84          this.threadPool = threadPool;
85          this.mappingUpdatedAction = mappingUpdatedAction;
86          this.indexService = indexService;
87          this.indexShard = indexShard;
88  
89          this.waitForMappingUpdatePostRecovery = indexSettings.getAsTime("index.gateway.wait_for_mapping_update_post_recovery", TimeValue.timeValueSeconds(30));
90          syncInterval = indexSettings.getAsTime("index.gateway.sync", TimeValue.timeValueSeconds(5));
91          if (syncInterval.millis() > 0) {
92              this.indexShard.translog().syncOnEachOperation(false);
93              flushScheduler = threadPool.schedule(syncInterval, ThreadPool.Names.SAME, new Sync());
94          } else if (syncInterval.millis() == 0) {
95              flushScheduler = null;
96              this.indexShard.translog().syncOnEachOperation(true);
97          } else {
98              flushScheduler = null;
99          }
100     }

102     /**
103      * Recovers the state of the shard from the gateway.
104      */
105     public void recover(boolean indexShouldExists, RecoveryState recoveryState) throws IndexShardGatewayRecoveryException {
106         indexShard.prepareForIndexRecovery();
107         long version = -1;
108         long translogId = -1;
109         final Set<String> typesToUpdate = Sets.newHashSet();
110         SegmentInfos si = null;
111         indexShard.store().incRef();
112         try {
113             try {
114                 indexShard.store().failIfCorrupted();
115                 try {
116                     si = Lucene.readSegmentInfos(indexShard.store().directory());
117                 } catch (Throwable e) {
118                     String files = "_unknown_";
119                     try {
120                         files = Arrays.toString(indexShard.store().directory().listAll());
121                     } catch (Throwable e1) {
122                         files += " (failure=" + ExceptionsHelper.detailedMessage(e1) + ")";
123                     }
124                     if (indexShouldExists) {
125                         throw new IndexShardGatewayRecoveryException(shardId(), "shard allocated for local recovery (post api), should exist, but doesn't, current files: " + files, e);
126                     }
127                 }
128                 if (si != null) {
129                     if (indexShouldExists) {
130                         version = si.getVersion();
131                         /**
132                          * We generate the translog ID before each lucene commit to ensure that
133                          * we can read the current translog ID safely when we recover. The commits metadata
134                          * therefor contains always the current / active translog ID.
135                          */
136                         if (si.getUserData().containsKey(Translog.TRANSLOG_ID_KEY)) {
137                             translogId = Long.parseLong(si.getUserData().get(Translog.TRANSLOG_ID_KEY));
138                         } else {
139                             translogId = version;
140                         }
141                         logger.trace("using existing shard data, translog id [{}]", translogId);
142                     } else {
143                         // it exists on the directory, but shouldn't exist on the FS, its a leftover (possibly dangling)
144                         // its a "new index create" API, we have to do something, so better to clean it than use same data
145                         logger.trace("cleaning existing shard, shouldn't exists");
146                         IndexWriter writer = new IndexWriter(indexShard.store().directory(), new IndexWriterConfig(Lucene.STANDARD_ANALYZER).setOpenMode(IndexWriterConfig.OpenMode.CREATE));
147                         writer.close();
148                     }
149                 }
150             } catch (Throwable e) {
151                 throw new IndexShardGatewayRecoveryException(shardId(), "failed to fetch index version after copying it over", e);
152             }
153             recoveryState.getIndex().updateVersion(version);
154 
155             // since we recover from local, just fill the files and size
156             try {
157                 final RecoveryState.Index index = recoveryState.getIndex();
158                 if (si != null) {
159                     final Directory directory = indexShard.store().directory();
160                     for (String name : Lucene.files(si)) {
161                         long length = directory.fileLength(name);
162                         index.addFileDetail(name, length, true);
163                     }
164                 }
165             } catch (IOException e) {
166                 logger.debug("failed to list file details", e);
167             }
168 
169             Path recoveringTranslogFile = null;
170             if (translogId == -1) {
171                 logger.trace("no translog id set (indexShouldExist [{}])", indexShouldExists);
172             } else {
173                 final Translog translog = indexShard.translog();
174                 final Path translogName = translog.getPath(translogId);
175                 logger.trace("try recover from translog file {} locations: {}", translogName, Arrays.toString(translog.locations()));
176                 OUTER:
177                 for (Path translogLocation : translog.locations()) {
178                     // we have to support .recovering since it's a leftover from previous version but might still be on the filesystem
179                     // we used to rename the foo into foo.recovering since foo was reused / overwritten but we fixed that in 1.5
180                     for (Path recoveryFiles : FileSystemUtils.files(translogLocation, translogName.getFileName() + "{.recovering,}")) {
181                         logger.trace("Translog file found in {}", recoveryFiles);
182                         recoveringTranslogFile = recoveryFiles;
183                         break OUTER;
184                     }
185                     logger.trace("Translog file NOT found in {} - continue", translogLocation);
186                 }
187             }
188             // we must do this *after* we capture translog name so the engine creation will not make a new one.
189             // also we have to do this regardless of whether we have a translog, to follow the recovery stages.
190             indexShard.prepareForTranslogRecovery();
191 
192             if (recoveringTranslogFile == null || Files.exists(recoveringTranslogFile) == false) {
193                 // no translog files, bail
194                 indexShard.finalizeRecovery();
195                 indexShard.postRecovery("post recovery from gateway, no translog");
196                 // no index, just start the shard and bail
197                 return;
198             }
199 
200             StreamInput in = null;
201 
202             try {
203                 if (logger.isTraceEnabled()) {
204                     logger.trace("recovering translog file: {} length: {}", recoveringTranslogFile, Files.size(recoveringTranslogFile));
205                 }
206                 TranslogStream stream = TranslogStreams.translogStreamFor(recoveringTranslogFile);
207                 try {
208                     in = stream.openInput(recoveringTranslogFile);
209                 } catch (TruncatedTranslogException e) {
210                     // file is empty or header has been half-written and should be ignored
211                     logger.trace("ignoring truncation exception, the translog is either empty or half-written", e);
212                 }
213                 while (true) {
214                     if (in == null) {
215                         break;
216                     }
217                     Translog.Operation operation;
218                     try {
219                         if (stream instanceof LegacyTranslogStream) {
220                             in.readInt(); // ignored opSize
221                         }
222                         operation = stream.read(in);
223                     } catch (TruncatedTranslogException|EOFException e) {
224                         // ignore, not properly written the last op
225                         logger.trace("ignoring translog EOF exception, the last operation was not properly written", e);
226                         break;
227                     } catch (IOException e) {
228                         // ignore, not properly written last op
229                         logger.trace("ignoring translog IO exception, the last operation was not properly written", e);
230                         break;
231                     }
232                     try {
233                         Engine.IndexingOperation potentialIndexOperation = indexShard.performRecoveryOperation(operation);
234                         if (potentialIndexOperation != null && potentialIndexOperation.parsedDoc().mappingsModified()) {
235                             if (!typesToUpdate.contains(potentialIndexOperation.docMapper().type())) {
236                                 typesToUpdate.add(potentialIndexOperation.docMapper().type());
237                             }
238                         }
239                         recoveryState.getTranslog().addTranslogOperations(1);
240                     } catch (ElasticsearchException e) {
241                         if (e.status() == RestStatus.BAD_REQUEST) {
242                             // mainly for MapperParsingException and Failure to detect xcontent
243                             logger.info("ignoring recovery of a corrupt translog entry", e);
244                         } else {
245                             throw e;
246                         }
247                     }
248                 }
249             } catch (Throwable e) {
250                 IOUtils.closeWhileHandlingException(indexShard.translog());
251                 throw new IndexShardGatewayRecoveryException(shardId, "failed to recover shard", e);
252             } finally {
253                 IOUtils.closeWhileHandlingException(in);
254             }
255             // we flush to trim the translog, in case it was big.
256             try {
257                 FlushRequest flushRequest = new FlushRequest();
258                 flushRequest.force(false);
259                 flushRequest.waitIfOngoing(false);
260                 indexShard.flush(flushRequest);
261             } catch (FlushNotAllowedEngineException e) {
262                 // to be safe we catch the FNAEX , at this point no one can recover
263                 logger.debug("skipping flush at end of recovery (not allowed)", e);
264             }
265             indexShard.finalizeRecovery();
266             indexShard.postRecovery("post recovery from gateway");
267 
268             try {
269                 Files.deleteIfExists(recoveringTranslogFile);
270             } catch (Exception ex) {
271                 logger.debug("Failed to delete recovering translog file {}", ex, recoveringTranslogFile);
272             }
273         } catch (IOException e) {
274             throw new IndexShardGatewayRecoveryException(shardId, "failed to recovery from gateway", e);
275         } finally {
276             indexShard.store().decRef();
277         }
278         for (final String type : typesToUpdate) {
279             final CountDownLatch latch = new CountDownLatch(1);
280             mappingUpdatedAction.updateMappingOnMaster(indexService.index().name(), indexService.mapperService().documentMapper(type), indexService.indexUUID(), new MappingUpdatedAction.MappingUpdateListener() {
281                 @Override
282                 public void onMappingUpdate() {
283                     latch.countDown();
284                 }
285 
286                 @Override
287                 public void onFailure(Throwable t) {
288                     latch.countDown();
289                     logger.debug("failed to send mapping update post recovery to master for [{}]", t, type);
290                 }
291             });
292             cancellableThreads.execute(new CancellableThreads.Interruptable() {
293                 @Override
294                 public void run() throws InterruptedException {
295                     try {
296                         if (latch.await(waitForMappingUpdatePostRecovery.millis(), TimeUnit.MILLISECONDS) == false) {
297                             logger.debug("waited for mapping update on master for [{}], yet timed out", type);
298                         }
299                     } catch (InterruptedException e) {
300                         logger.debug("interrupted while waiting for mapping update");
301                         throw e;
302                     }
303                 }
304             });
305 
306         }
307     }
308 
309     @Override
310     public void close() {
311         FutureUtils.cancel(flushScheduler);
312         cancellableThreads.cancel("closed");
313     }
314 
315     class Sync implements Runnable {
316         @Override
317         public void run() {
318             // don't re-schedule  if its closed..., we are done
319             if (indexShard.state() == IndexShardState.CLOSED) {
320                 return;
321             }
322             if (indexShard.state() == IndexShardState.STARTED && indexShard.translog().syncNeeded()) {
323                 threadPool.executor(ThreadPool.Names.FLUSH).execute(new Runnable() {
324                     @Override
325                     public void run() {
326                         try {
327                             indexShard.translog().sync();
328                         } catch (Exception e) {
329                             if (indexShard.state() == IndexShardState.STARTED) {
330                                 logger.warn("failed to sync translog", e);
331                             }
332                         }
333                         if (indexShard.state() != IndexShardState.CLOSED) {
334                             flushScheduler = threadPool.schedule(syncInterval, ThreadPool.Names.SAME, Sync.this);
335                         }
336                     }
337                 });
338             } else {
339                 flushScheduler = threadPool.schedule(syncInterval, ThreadPool.Names.SAME, Sync.this);
340             }
341         }
342     }
