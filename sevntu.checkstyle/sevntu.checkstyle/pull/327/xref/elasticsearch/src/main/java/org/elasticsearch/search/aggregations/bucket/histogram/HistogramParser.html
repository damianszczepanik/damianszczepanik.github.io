19  package org.elasticsearch.search.aggregations.bucket.histogram;
20  
21  import org.elasticsearch.common.ParseField;
22  import org.elasticsearch.common.rounding.Rounding;
23  import org.elasticsearch.common.xcontent.XContentParser;
24  import org.elasticsearch.search.SearchParseException;
25  import org.elasticsearch.search.aggregations.Aggregator;
26  import org.elasticsearch.search.aggregations.AggregatorFactory;
27  import org.elasticsearch.search.aggregations.support.ValueType;
28  import org.elasticsearch.search.aggregations.support.ValuesSourceParser;
29  import org.elasticsearch.search.aggregations.support.format.ValueParser;
30  import org.elasticsearch.search.internal.SearchContext;
31  
32  import java.io.IOException;
33  
34  /**
35   * Parses the histogram request
36   */
37  public class HistogramParser implements Aggregator.Parser {
38  
39      static final ParseField EXTENDED_BOUNDS = new ParseField("extended_bounds");
40  
41      @Override
42      public String type() {
43          return InternalHistogram.TYPE.name();
44      }
45  
46      @Override
47      public AggregatorFactory parse(String aggregationName, XContentParser parser, SearchContext context) throws IOException {
48  
49          ValuesSourceParser vsParser = ValuesSourceParser.numeric(aggregationName, InternalHistogram.TYPE, context)
50                  .targetValueType(ValueType.NUMERIC)
51                  .formattable(true)
52                  .build();
53  
54          boolean keyed = false;
55          long minDocCount = 1;
56          InternalOrder order = (InternalOrder) InternalOrder.KEY_ASC;
57          long interval = -1;
58          ExtendedBounds extendedBounds = null;
59          long offset = 0;
60  
61          XContentParser.Token token;
62          String currentFieldName = null;
63          while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
64              if (token == XContentParser.Token.FIELD_NAME) {
65                  currentFieldName = parser.currentName();
66              } else if (vsParser.token(currentFieldName, token, parser)) {
67                  continue;
68              } else if (token.isValue()) {
69                  if ("interval".equals(currentFieldName)) {
70                      interval = parser.longValue();
71                  } else if ("min_doc_count".equals(currentFieldName) || "minDocCount".equals(currentFieldName)) {
72                      minDocCount = parser.longValue();
73                  } else if ("keyed".equals(currentFieldName)) {
74                      keyed = parser.booleanValue();
75                  } else if ("offset".equals(currentFieldName)) {
76                      offset = parser.longValue();
77                  } else {
78                      throw new SearchParseException(context, "Unknown key for a " + token + " in aggregation [" + aggregationName + "]: [" + currentFieldName + "].");
79                  }
80              } else if (token == XContentParser.Token.START_OBJECT) {
81                  if ("order".equals(currentFieldName)) {
82                      while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
83                          if (token == XContentParser.Token.FIELD_NAME) {
84                              currentFieldName = parser.currentName();
85                          } else if (token == XContentParser.Token.VALUE_STRING) {
86                              String dir = parser.text();
87                              boolean asc = "asc".equals(dir);
88                              if (!asc && !"desc".equals(dir)) {
89                                  throw new SearchParseException(context, "Unknown order direction [" + dir + "] in aggregation [" + aggregationName + "]. Should be either [asc] or [desc]");
90                              }
91                              order = resolveOrder(currentFieldName, asc);
92                          }
93                      }
94                  } else if (EXTENDED_BOUNDS.match(currentFieldName)) {
95                      extendedBounds = new ExtendedBounds();
96                      while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
97                          if (token == XContentParser.Token.FIELD_NAME) {
98                              currentFieldName = parser.currentName();
99                          } else if (token.isValue()) {
100                             if ("min".equals(currentFieldName)) {
101                                 extendedBounds.min = parser.longValue(true);
102                             } else if ("max".equals(currentFieldName)) {
103                                 extendedBounds.max = parser.longValue(true);
104                             } else {
105                                 throw new SearchParseException(context, "Unknown extended_bounds key for a " + token + " in aggregation [" + aggregationName + "]: [" + currentFieldName + "].");
106                             }
107                         }
108                     }
109 
110                 } else {
111                     throw new SearchParseException(context, "Unknown key for a " + token + " in aggregation [" + aggregationName + "]: [" + currentFieldName + "].");
112                 }
113             } else {
114                 throw new SearchParseException(context, "Unexpected token " + token + " in aggregation [" + aggregationName + "].");
115             }
116         }
117 
118         if (interval < 1) {
119             throw new SearchParseException(context, "Missing required field [interval] for histogram aggregation [" + aggregationName + "]");
120         }
121 
122         Rounding rounding = new Rounding.Interval(interval);
123         if (offset != 0) {
124             rounding = new Rounding.OffsetRounding((Rounding.Interval) rounding, offset);
125         }
126 
127         if (extendedBounds != null) {
128             // with numeric histogram, we can process here and fail fast if the bounds are invalid
129             extendedBounds.processAndValidate(aggregationName, context, ValueParser.RAW);
130         }
131 
132         return new HistogramAggregator.Factory(aggregationName, vsParser.config(), rounding, order, keyed, minDocCount, extendedBounds,
133                 new InternalHistogram.Factory());
134 
135     }
136 
137     static InternalOrder resolveOrder(String key, boolean asc) {
138         if ("_key".equals(key)) {
139             return (InternalOrder) (asc ? InternalOrder.KEY_ASC : InternalOrder.KEY_DESC);
140         }
141         if ("_count".equals(key)) {
142             return (InternalOrder) (asc ? InternalOrder.COUNT_ASC : InternalOrder.COUNT_DESC);
143         }
144         return new InternalOrder.Aggregation(key, asc);
145     }
146 }
