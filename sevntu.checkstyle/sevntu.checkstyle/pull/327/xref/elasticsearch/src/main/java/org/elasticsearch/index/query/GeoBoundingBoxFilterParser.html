20  package org.elasticsearch.index.query;
21  
22  import org.apache.lucene.search.Filter;
23  import org.apache.lucene.search.FilterCachingPolicy;
24  import org.elasticsearch.ElasticsearchParseException;
25  import org.elasticsearch.common.geo.GeoPoint;
26  import org.elasticsearch.common.geo.GeoUtils;
27  import org.elasticsearch.common.inject.Inject;
28  import org.elasticsearch.common.lucene.HashedBytesRef;
29  import org.elasticsearch.common.xcontent.XContentParser;
30  import org.elasticsearch.index.fielddata.IndexGeoPointFieldData;
31  import org.elasticsearch.index.mapper.FieldMapper;
32  import org.elasticsearch.index.mapper.MapperService;
33  import org.elasticsearch.index.mapper.geo.GeoPointFieldMapper;
34  import org.elasticsearch.index.search.geo.InMemoryGeoBoundingBoxFilter;
35  import org.elasticsearch.index.search.geo.IndexedGeoBoundingBoxFilter;
36  
37  import java.io.IOException;
38  
39  /**
40   *
41   */
42  public class GeoBoundingBoxFilterParser implements FilterParser {
43  
44      public static final String TOP = "top";
45      public static final String LEFT = "left";
46      public static final String RIGHT = "right";
47      public static final String BOTTOM = "bottom";
48  
49      public static final String TOP_LEFT = TOP + "_" + LEFT;
50      public static final String TOP_RIGHT = TOP + "_" + RIGHT;
51      public static final String BOTTOM_LEFT = BOTTOM + "_" + LEFT;
52      public static final String BOTTOM_RIGHT = BOTTOM + "_" + RIGHT;
53  
54      public static final String TOPLEFT = "topLeft";
55      public static final String TOPRIGHT = "topRight";
56      public static final String BOTTOMLEFT = "bottomLeft";
57      public static final String BOTTOMRIGHT = "bottomRight";
58  
59      public static final String NAME = "geo_bbox";
60      public static final String FIELD = "field";
61  
62      @Inject
63      public GeoBoundingBoxFilterParser() {
64      }
65  
66      @Override
67      public String[] names() {
68          return new String[]{NAME, "geoBbox", "geo_bounding_box", "geoBoundingBox"};
69      }
70  
71      @Override
72      public Filter parse(QueryParseContext parseContext) throws IOException, QueryParsingException {
73          XContentParser parser = parseContext.parser();
74  
75          FilterCachingPolicy cache = parseContext.autoFilterCachePolicy();
76          HashedBytesRef cacheKey = null;
77          String fieldName = null;
78  
79          double top = Double.NaN;
80          double bottom = Double.NaN;
81          double left = Double.NaN;
82          double right = Double.NaN;
83          
84          String filterName = null;
85          String currentFieldName = null;
86          XContentParser.Token token;
87          boolean normalize = true;
88  
89          GeoPoint sparse = new GeoPoint();
90          
91          String type = "memory";
92  
93          while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
94              if (token == XContentParser.Token.FIELD_NAME) {
95                  currentFieldName = parser.currentName();
96              } else if (token == XContentParser.Token.START_OBJECT) {
97                  fieldName = currentFieldName;
98  
99                  while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
100                     if (token == XContentParser.Token.FIELD_NAME) {
101                         currentFieldName = parser.currentName();
102                         token = parser.nextToken();
103                         if (FIELD.equals(currentFieldName)) {
104                             fieldName = parser.text();
105                         } else if (TOP.equals(currentFieldName)) {
106                             top = parser.doubleValue();
107                         } else if (BOTTOM.equals(currentFieldName)) {
108                             bottom = parser.doubleValue();
109                         } else if (LEFT.equals(currentFieldName)) {
110                             left = parser.doubleValue();
111                         } else if (RIGHT.equals(currentFieldName)) {
112                             right = parser.doubleValue();
113                         } else {
114                             if (TOP_LEFT.equals(currentFieldName) || TOPLEFT.equals(currentFieldName)) {
115                                 GeoUtils.parseGeoPoint(parser, sparse);
116                                 top = sparse.getLat();
117                                 left = sparse.getLon();
118                             } else if (BOTTOM_RIGHT.equals(currentFieldName) || BOTTOMRIGHT.equals(currentFieldName)) {
119                                 GeoUtils.parseGeoPoint(parser, sparse);
120                                 bottom = sparse.getLat();
121                                 right = sparse.getLon();
122                             } else if (TOP_RIGHT.equals(currentFieldName) || TOPRIGHT.equals(currentFieldName)) {
123                                 GeoUtils.parseGeoPoint(parser, sparse);
124                                 top = sparse.getLat();
125                                 right = sparse.getLon();
126                             } else if (BOTTOM_LEFT.equals(currentFieldName) || BOTTOMLEFT.equals(currentFieldName)) {
127                                 GeoUtils.parseGeoPoint(parser, sparse);
128                                 bottom = sparse.getLat();
129                                 left = sparse.getLon();
130                             } else {
131                                 throw new ElasticsearchParseException("Unexpected field [" + currentFieldName + "]");
132                             }
133                         }
134                     } else {
135                         throw new ElasticsearchParseException("fieldname expected but [" + token + "] found");
136                     }
137                 }
138             } else if (token.isValue()) {
139                 if ("_name".equals(currentFieldName)) {
140                     filterName = parser.text();
141                 } else if ("_cache".equals(currentFieldName)) {
142                     cache = parseContext.parseFilterCachePolicy();
143                 } else if ("_cache_key".equals(currentFieldName) || "_cacheKey".equals(currentFieldName)) {
144                     cacheKey = new HashedBytesRef(parser.text());
145                 } else if ("normalize".equals(currentFieldName)) {
146                     normalize = parser.booleanValue();
147                 } else if ("type".equals(currentFieldName)) {
148                     type = parser.text();
149                 } else {
150                     throw new QueryParsingException(parseContext.index(), "[geo_bbox] filter does not support [" + currentFieldName + "]");
151                 }
152             }
153         }
154 
155         final GeoPoint topLeft = sparse.reset(top, left);  //just keep the object
156         final GeoPoint bottomRight = new GeoPoint(bottom, right);
157 
158         if (normalize) {
159             // Special case: if the difference bettween the left and right is 360 and the right is greater than the left, we are asking for 
160             // the complete longitude range so need to set longitude to the complete longditude range
161             boolean completeLonRange = ((right - left) % 360 == 0 && right > left);
162             GeoUtils.normalizePoint(topLeft, true, !completeLonRange);
163             GeoUtils.normalizePoint(bottomRight, true, !completeLonRange);
164             if (completeLonRange) {
165                 topLeft.resetLon(-180);
166                 bottomRight.resetLon(180);
167             }
168         }
169 
170         MapperService.SmartNameFieldMappers smartMappers = parseContext.smartFieldMappers(fieldName);
171         if (smartMappers == null || !smartMappers.hasMapper()) {
172             throw new QueryParsingException(parseContext.index(), "failed to find geo_point field [" + fieldName + "]");
173         }
174         FieldMapper<?> mapper = smartMappers.mapper();
175         if (!(mapper instanceof GeoPointFieldMapper)) {
176             throw new QueryParsingException(parseContext.index(), "field [" + fieldName + "] is not a geo_point field");
177         }
178         GeoPointFieldMapper geoMapper = ((GeoPointFieldMapper) mapper);
179 
180         Filter filter;
181         if ("indexed".equals(type)) {
182             filter = IndexedGeoBoundingBoxFilter.create(topLeft, bottomRight, geoMapper);
183         } else if ("memory".equals(type)) {
184             IndexGeoPointFieldData indexFieldData = parseContext.getForField(mapper);
185             filter = new InMemoryGeoBoundingBoxFilter(topLeft, bottomRight, indexFieldData);
186         } else {
187             throw new QueryParsingException(parseContext.index(), "geo bounding box type [" + type + "] not supported, either 'indexed' or 'memory' are allowed");
188         }
189 
190         if (cache != null) {
191             filter = parseContext.cacheFilter(filter, cacheKey, cache);
192         }
193         if (filterName != null) {
194             parseContext.addNamedFilter(filterName, filter);
195         }
196         return filter;
197     }    
198 }
